### 1) tools/snapshot_manifest.py

```python
#!/usr/bin/env python3
"""
Generate file_manifest.json for one date partition.

Usage:
  HF_TOKEN=<token> python tools/snapshot_manifest.py \
    --repo axentx/surrogate-1-training-pairs \
    --date 2026-05-03 \
    --out file_manifest.json

Produces:
{
  "repo": "axentx/surrogate-1-training-pairs",
  "date": "2026-05-03",
  "generated_at": "2026-05-03T14:03:00Z",
  "files": [
    {
      "path": "batches/public-merged/2026-05-03/shard0-140300.jsonl",
      "size": 12345678,
      "cdn_url": "https://huggingface.co/datasets/axentx/surrogate-1-training-pairs/resolve/main/batches/public-merged/2026-05-03/shard0-140300.jsonl",
      "etag": "\"abc123\""
    },
    ...
  ]
}
"""

from __future__ import annotations

import argparse
import json
import os
import sys
from datetime import datetime, timezone
from pathlib import Path
from typing import Any

from huggingface_hub import HfApi, list_repo_tree

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def build_manifest(repo: str, date: str, out_path: Path) -> None:
    api = HfApi(token=os.getenv("HF_TOKEN"))
    prefix = f"batches/public-merged/{date}"

    # Single API call: non-recursive at date folder level.
    tree = list_repo_tree(
        repo=repo,
        path=prefix,
        recursive=False,
        repo_type="dataset",
    )

    files = []
    for entry in tree:
        if entry.type != "file":
            continue
        path = f"{prefix}/{entry.path.split('/')[-1]}"
        files.append(
            {
                "path": path,
                "size": entry.size or 0,
                "cdn_url": CDN_TEMPLATE.format(repo=repo, path=path),
                "etag": entry.etag or "",
            }
        )

    # Deterministic ordering.
    files.sort(key=lambda f: f["path"])

    manifest = {
        "repo": repo,
        "date": date,
        "generated_at": datetime.now(timezone.utc).isoformat(),
        "files": files,
    }

    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"Wrote {len(files)} files to {out_path}")

def main() -> None:
    parser = argparse.ArgumentParser(description="Generate CDN manifest for a date partition.")
    parser.add_argument("--repo", default="axentx/surrogate-1-training-pairs")
    parser.add_argument("--date", required=True, help="Date folder, e.g. 2026-05-03")
    parser.add_argument("--out", default="file_manifest.json", help="Output JSON path")
    args = parser.parse_args()

    if not os.getenv("HF_TOKEN"):
        print("Error: HF_TOKEN environment variable is required.", file=sys.stderr)
        sys.exit(1)

    build_manifest(repo=args.repo, date=args.date, out_path=Path(args.out))

if __name__ == "__main__":
    main()
```

#!/usr/bin/env python3
"""
Lightning training entrypoint that consumes file_manifest.json
and performs CDN-only fetches (zero HF API calls during training).
"""

from __future__ import annotations

import argparse
import json
import time
from pathlib import Path
from typing import Iterator, Tuple

import numpy as np
import pyarrow as pa
import pyarrow.compute as pc
import requests
from datasets import Features, Sequence, Value

CDN_RETRY_BACKOFF = [1, 2, 4, 8, 16]
CDN_TIMEOUT = 30

def robust_cdn_get(url: str, max_retries: int = 5) -> bytes:
    for attempt in range(max_retries):
        try:
            resp = requests.get(url, timeout=CDN_TIMEOUT, stream=True)
            resp.raise_for_status()
            # Stream into memory; for large files consider temp file + memory-map.
            chunks = []
            for chunk in resp.iter_content(chunk_size=8192):
                chunks.append(chunk)
            return b"".join(chunks)
        except Exception as exc:
            if attempt == max_retries - 1:
                raise
            sleep = CDN_RETRY_BACKOFF[min(attempt, len(CDN_RETRY_BACKOFF) - 1)]
            print(f"CDN fetch failed ({exc}), retry {attempt + 1}/{max_retries} in {sleep}s")
            time.sleep(sleep)
    raise RuntimeError("unreachable")
