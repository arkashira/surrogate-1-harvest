#### 2) Knowledge graph service — `app/services/knowledge_graph.py`
```python
import time
from datetime import datetime, timedelta
from typing import Optional, Dict, Any, List
from app.core.config import settings

# Simple in-process cache for top-hub (fast, deterministic for ops)
_TOP_HUB_CACHE: Dict[str, Any] = {"value": None, "expires_at": 0}


def _now() -> float:
    return time.time()


def get_top_hub(*, graph_client, min_degree: int = 1, lookback_days: int = 1) -> Dict[str, Any]:
    """
    Return top hub for today (highest-degree or highest-cost node).
    Cached for TOP_HUB_CACHE_TTL_SECONDS to avoid repeated heavy queries.
    """
    ttl = getattr(settings, "TOP_HUB_CACHE_TTL_SECONDS", 300)
    if _TOP_HUB_CACHE["value"] and _now() < _TOP_HUB_CACHE["expires_at"]:
        return _TOP_HUB_CACHE["value"]

    # Deterministic selection: highest weighted degree in last lookback_days
    # Replace with your graph client query (Neo4j / NetworkX / etc.)
    # Example Cypher-like pseudocode:
    #   MATCH (h:Hub)
    #   WHERE h.last_seen >= datetime() - duration({days: lookback_days})
    #   RETURN h.id AS hub_id, h.name AS hub_name, h.type AS hub_type,
    #          size((h)--()) AS degree, h.total_cost_1d AS cost
    #   ORDER BY degree DESC, cost DESC
    #   LIMIT 1
    result = graph_client.run_query(
        """
        MATCH (h:Hub)
        WHERE h.last_seen >= datetime() - duration({days: $lookback_days})
        RETURN h.id AS hub_id, h.name AS hub_name, h.type AS hub_type,
               size((h)--()) AS degree, h.total_cost_1d AS cost
        ORDER BY degree DESC, cost DESC
        LIMIT 1
        """,
        {"lookback_days": lookback_days},
    )
    top = result[0] if result else {"hub_id": "unknown", "hub_name": "unknown", "hub_type": "unknown", "degree": 0, "cost": 0}

    _TOP_HUB_CACHE["value"] = top
    _TOP_HUB_CACHE["expires_at"] = _now() + ttl
    return top


def get_strongest_signal_for_hub(*, graph_client, hub_id: str, lookback_days: int = 1) -> Optional[Dict[str, Any]]:
    """
    Return strongest cost-anomaly signal for a hub.
    Signal nodes labeled :Signal with category='cost_anomaly' and severity ordering.
    """
    severity_order = {"critical": 4, "high": 3, "medium": 2, "low": 1}
    # Example query:
    #   MATCH (h:Hub {id: $hub_id})<-[:RELATED_TO]-(s:Signal {category: 'cost_anomaly'})
    #   WHERE s.timestamp >= datetime() - duration({days: $lookback_days})
    #   RETURN s
    #   ORDER BY s.severity_weight DESC, s.timestamp DESC
    #   LIMIT 1
    results = graph_client.run_query(
        """
        MATCH (h:Hub {id: $hub_id})<-[:RELATED_TO]-(s:Signal {category: 'cost_anomaly'})
        WHERE s.timestamp >= datetime() - duration({days: $lookback_days})
        RETURN s.id AS signal_id, s.severity AS severity, s.metric AS metric,
               s.value AS value, s.baseline AS baseline, s.deviation_pct AS deviation_pct,
               s.timestamp AS timestamp, s.window_start AS window_start,
               s.window_end AS window_end, s.recommendation AS recommendation,
               s.context AS context, s.source AS source
        ORDER BY
          CASE s.severity
            WHEN 'critical' THEN 4
            WHEN 'high' THEN 3
            WHEN 'medium' THEN 2
            WHEN 'low' THEN 1
            ELSE 0
          END DESC,
          s.timestamp DESC
        LIMIT 1
        """,
        {"hub_id": hub_id, "lookback_days": lookback_days},
    )
    if not results:
        return None
    return results[0]
```
