# workio / backend

## Final Synthesis — Idempotent LINE Webhook with Real-Time Push and Audit

**Core problem**: LINE retries on 5xx/timeouts + no deduplication + inline DB mutations → duplicate/mis-ordered clock/leave/OT events, partial side-effects, and frontend staleness.

**Chosen solution** (merged & hardened):
- **Idempotent webhook processor** with strict de-dup (external delivery ID + tenant) and transactional unit-of-work.
- **Transactional outbox** to guarantee side-effects (notifications, real-time push, audit) atomically with domain changes.
- **Lightweight WebSocket broadcaster** scoped by tenant for immediate frontend updates (no polling).
- **Structured activity/audit log** for compliance, debugging, and traceability.
- **Background outbox processor** with `SKIP LOCKED` + retries and bounded concurrency to avoid webhook timeouts.
- **Fast 200 OK path**: validate, dedupe, persist event + outbox in one transaction, return quickly; heavy work happens async.

### 1. DB schema additions

```sql
-- Idempotency for LINE deliveries (one processed outcome per external delivery)
CREATE TABLE IF NOT EXISTS webhook_events (
  id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  external_id   TEXT NOT NULL,               -- LINE deliveryId / messageId
  tenant_id     UUID NOT NULL,
  event_type    TEXT NOT NULL,               -- clock, leave, ot, etc.
  processed_at  TIMESTAMPTZ,
  created_at    TIMESTAMPTZ NOT NULL DEFAULT now(),
  CONSTRAINT uniq_external_tenant UNIQUE (external_id, tenant_id)
);

-- Transactional outbox for side-effects and real-time push
CREATE TABLE IF NOT EXISTS outbox_events (
  id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id     UUID NOT NULL,
  event_type    TEXT NOT NULL,               -- clock_update, leave_update, ot_update, etc.
  payload       JSONB NOT NULL,
  broadcast     BOOLEAN NOT NULL DEFAULT true,
  created_at    TIMESTAMPTZ NOT NULL DEFAULT now(),
  processed_at  TIMESTAMPTZ
);

CREATE INDEX IF NOT EXISTS idx_outbox_unprocessed ON outbox_events (processed_at)
  WHERE processed_at IS NULL;

-- Structured activity/audit log (compliance + debugging)
CREATE TABLE IF NOT EXISTS activity_log (
  id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id     UUID NOT NULL,
  actor_type    TEXT NOT NULL,               -- system | user | line_bot
  actor_id      TEXT NOT NULL,
  action        TEXT NOT NULL,               -- clock_in, clock_out, leave_request, etc.
  target_type   TEXT,                        -- clock | leave | ot
  target_id     UUID,                        -- FK to domain row if applicable
  details       JSONB,
  ip_address    INET,
  user_agent    TEXT,
  occurred_at   TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE INDEX IF NOT EXISTS idx_activity_tenant ON activity_log (tenant_id, occurred_at DESC);
```

### 2. WebSocket broadcaster (lightweight, tenant-scoped)

`server/src/lib/ws.ts`
```ts
import { WebSocket } from 'ws';

type Client = {
  tenantId: string;
  ws: WebSocket;
};

const clients = new Set<Client>();

export const wsServer = {
  attach(server: any) {
    const wss = new WebSocket.Server({ server, path: '/ws' });
    wss.on('connection', (ws, req) => {
      try {
        const url = new URL(req.url || '', `http://${req.headers.host}`);
        const tenantId = url.searchParams.get('tenantId');
        if (!tenantId) { ws.close(); return; }

        const client: Client = { tenantId, ws };
        clients.add(client);

        ws.on('close', () => clients.delete(client));
        ws.on('error', () => clients.delete(client));
      } catch {
        ws.close();
      }
    });
  },

  broadcast(tenantId: string, event: { type: string; payload: any }) {
    const msg = JSON.stringify(event);
    for (const c of clients) {
      if (c.tenantId === tenantId && c.ws.readyState === WebSocket.OPEN) {
        try {
          c.ws.send(msg);
        } catch {
          // best-effort; broken sockets cleaned on close/error
        }
      }
    }
  }
};
```

### 3. Outbox service (enqueue + background processor)

`server/src/services/outboxService.ts`
```ts
import { pool } from '../db';
import { wsServer } from '../lib/ws';

export async function enqueueOutbox(
  tenantId: string,
  eventType: string,
  payload: any,
  options?: { broadcast?: boolean }
) {
  const { rows } = await pool.query(
    `INSERT INTO outbox_events (tenant_id, event_type, payload, broadcast)
     VALUES ($1, $2, $3, $4) RETURNING id`,
    [tenantId, eventType, JSON.stringify(payload), options?.broadcast ?? true]
  );
  return rows[0].id;
}

// Background processor (run via cron/worker/interval)
export async function processOutbox(limit = 100) {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');
    const { rows } = await client.query(
      `SELECT id, tenant_id, event_type, payload, broadcast
       FROM outbox_events
       WHERE processed_at IS NULL
       ORDER BY created_at
       FOR UPDATE SKIP LOCKED
       LIMIT $1`,
      [limit]
    );

    for (const row of rows) {
      try {
        if (row.broadcast !== false) {
          wsServer.broadcast(row.tenant_id, { type: row.event_type, payload: row.payload });
        }
        // mark processed atomically
        await client.query(
          `UPDATE outbox_events SET processed_at = now() WHERE id = $1`,
          [row.id]
        );
      } catch (err) {
        // continue processing others; failed item can be retried next cycle
        // optionally implement retry counter + dead-letter after N attempts
      }
    }
    await client.query('COMMIT');
  } catch (e) {
    await client.query('ROLLBACK');
    throw e;
  } finally {
    client.release();
  }
}
```
