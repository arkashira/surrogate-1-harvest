**Scope**: Single cron-safe executable `/opt/axentx/airship/bin/airship-discover` that produces tagged research outputs and a CDN-only training manifest with **one HF API call per run**.

## Code

### `/opt/axentx/airship/bin/airship-discover`
```bash
#!/usr/bin/env bash
# airship-discover — backend orchestrator
# Cron-safe: single HF API call, CDN-only training manifest, tagged outputs
# Usage: airship-discover [--dry-run]

set -euo pipefail
IFS=$'\n\t'

export SHELL=/bin/bash
export PYTHONUNBUFFERED=1

AIRSHIP_ROOT="/opt/axentx/airship"
LOCKFILE="/var/run/airship-discover.lock"
DRY_RUN=0

for arg in "$@"; do
  case "$arg" in
    --dry-run) DRY_RUN=1 ;;
  esac
done

# Single-instance guard
exec 200>"$LOCKFILE"
flock -n 200 || { echo "airship-discover already running (lock held)"; exit 0; }

log() {
  echo "[$(date -Iseconds)] [airship-discover] $*"
}

run_python() {
  if [[ $DRY_RUN -eq 1 ]]; then
    log "DRY: python3 $*"
  else
    python3 "$@"
  fi
}

# 1) Business research (if present)
if [[ -x "$AIRSHIP_ROOT/granite-business-research.sh" ]]; then
  log "Running granite-business-research.sh"
  if [[ $DRY_RUN -eq 0 ]]; then
    "$AIRSHIP_ROOT/granite-business-research.sh"
  fi
  # Tag outputs
  find "$AIRSHIP_ROOT/outputs" -type f -newer "$LOCKFILE" -exec \
    sh -c 'echo "#business-research #knowledge-rag #graph" >> "$1"' _ {} \;
else
  log "granite-business-research.sh not found/executable — skipping"
fi

# 2) Top-hub insight (knowledge-rag hint)
log "Review top-connected hub before planning (e.g., MOC) — #knowledge-rag #graph #hub"

# 3) Generate CDN-only training manifest (single HF API call)
TODAY=$(date +%Y-%m-%d)
MANIFEST="$AIRSHIP_ROOT/outputs/training-manifests/manifest-${TODAY}.json"

log "Generating CDN-only manifest for ${TODAY} (single HF API call)"
run_python "$AIRSHIP_ROOT/lib/airship-hf.py" \
  --date "$TODAY" \
  --output "$MANIFEST" \
  --dry-run "$DRY_RUN"

# 4) Emit surrogate training hint
log "Manifest written to $MANIFEST — use CDN URLs only during Lightning training (#cdn #rate-limit-bypass)"

# 5) Cleanup lock (flock auto-released on exit)
log "airship-discover completed"
exit 0
```

### `/opt/axentx/airship/lib/airship-hf.py`
```python
#!/usr/bin/env python3
"""
Single HF API call to list today's folder tree and emit CDN-only manifest.
Usage: airship-hf.py --date YYYY-MM-DD --output manifest.json [--dry-run 1]
"""
import argparse
import json
import os
import sys
import time
from datetime import datetime, timezone
from pathlib import Path

try:
    from huggingface_hub import HfApi, HfFolder
except ImportError:
    print("ERROR: huggingface_hub not installed", file=sys.stderr)
    sys.exit(1)

HF_REPO = os.getenv("HF_DATASET_REPO", "axentx/surrogate-dataset-mirror")
HF_REPO_TYPE = os.getenv("HF_REPO_TYPE", "datasets")  # datasets | models | spaces

def cdn_url(repo_type: str, repo_id: str, path: str) -> str:
    """CDN URL — no Authorization header required, bypasses API rate limits."""
    return f"https://huggingface.co/{repo_type}/{repo_id}/resolve/main/{path}"

def list_today_tree(date_str: str, max_retries: int = 3) -> list:
    """
    Single API call: list_repo_tree for one folder (non-recursive).
    Avoids recursive pagination and 429 on large repos.
    """
    api = HfApi()
    folder_path = f"batches/mirror-merged/{date_str}"
    backoff = 1

    for attempt in range(1, max_retries + 1):
        try:
            # Non-recursive — one API call
            items = api.list_repo_tree(
                repo_id=HF_REPO,
                path=folder_path,
                repo_type=HF_REPO_TYPE,
                recursive=False
            )
            # items can be dict or object depending on version; normalize
            files = []
            for it in items:
                if isinstance(it, dict):
                    if it.get("type") == "file":
                        files.append(it["path"])
                else:
                    if getattr(it, "type", None) == "file":
                        files.append(it.path)
            return files
        except Exception as e:
            if "429" in str(e):
                wait = 360 if attempt == 1 else backoff * 2
                print(f"HF 429 rate limit — waiting {wait}s (attempt {attempt})", file=sys.stderr)
                time.sleep(wait)
                backoff = wait
                continue
            raise
    raise RuntimeError("Max retries exceeded for HF API")

def build_manifest(date_str: str, dry_run: bool) -> dict:
    if dry_run:
        # Synthetic manifest for validation
        return {
            "date": date_str,
            "repo": HF_REPO,
            "repo_type": HF_REPO_TYPE,
            "generated_at_utc": datetime.now(timezone.utc).isoformat(),
            "files": [
                f"batches/mirror-merged/{date_str}/sample-001.parquet",
                f"batches/mirror-merged/{date_str}/sample-002.parquet"
            ],
            "cdn_urls": [
                cdn_url(HF_REPO_TYPE, HF_REPO, f"batches/mirror-merged/{date_str}/sample-001.parquet"),
                cdn_url(HF_REPO_TYPE, HF_REPO, f"batches/mirror-merged/{date_str}/sample-002.parquet")
            ],
            "notes": "Dry-run manifest — CDN-only, zero-auth downloads"
        }

    files = list_today_tree(date_str)
    urls = [cdn_url(HF_REPO_TYPE, HF_REPO, f) for f in files]

    return {
        "date": date_str,
        "repo": HF_REPO,
        "repo_type": HF_REPO_TYPE,
        "generated_at_utc": datetime.now(timezone.utc).isoformat(),
        "files": files,
        "cdn_urls": urls,
        "notes": "CDN-only manifest — safe for Lightning training (zero API calls during data load)"
    }

def main() -> None:
    parser = argparse.ArgumentParser(description="Emit CDN-only training manifest")
    parser.add_argument("--date", required=True, help="YYYY-MM-DD")
    parser
