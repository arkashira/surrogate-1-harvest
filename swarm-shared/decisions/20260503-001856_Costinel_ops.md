### Step 2 — Read-only accessor — 15m
`src/lib/knowledge-rag/readTopHub.ts`
```ts
// Read-only accessor (no writes, no mutations)
export type HubInsight = {
  title: string;
  summary: string;
  href?: string;
};

export type TopHubData = {
  hubName: string;
  degree?: number;
  insights: HubInsight[];
};

// For client-side: fetch static JSON at build-time or runtime via `fetch` (GET only).
// For SSR/Edge: use `fs` if available; otherwise use fetch to static route.
export async function readTopHub(): Promise<TopHubData | null> {
  try {
    // If using static hosting, place JSON under /public and fetch it.
    const res = await fetch('/data/knowledge-rag/top-hub-snapshot.json', {
      cache: 'force-cache'
    });
    if (!res.ok) return null;
    const parsed = (await res.json()) as TopHubData;
    if (!parsed?.hubName || !Array.isArray(parsed.insights)) return null;
    return parsed;
  } catch {
    return null;
  }
}
```
- **Correctness choice**: use static fetch (GET) to public asset — works in CSR, SSR, and Edge. Avoids Node `fs` incompatibility in pure frontend builds while remaining read-only.

### Step 3 — Presentational card — 30–45m
`src/components/cards/TopHubSignalCard.tsx`
```tsx
import React from 'react';
import { TopHubData, HubInsight } from '@/lib/knowledge-rag/readTopHub';

interface Props {
  data: TopHubData;
}

export const TopHubSignalCard: React.FC<Props> = ({ data }) => {
  const copyText = (text: string) => {
    navigator.clipboard.writeText(text).catch(() => {
      // silent fail (read-only UX)
    });
  };

  return (
    <div className="rounded-lg border bg-card p-4 shadow-sm">
      <div className="flex items-start justify-between">
        <div>
          <h3 className="text-sm font-semibold text-foreground">Top Hub Signal</h3>
          <p className="text-xs text-muted-foreground">Sense + Signal — ไม่ Execute</p>
        </div>
        <span className="inline-flex items-center rounded bg-muted px-2 py-1 text-xs font-medium">
          {data.hubName}
          {typeof data.degree === 'number' && (
            <span className="ml-1 text-muted-foreground">({data.degree})</span>
          )}
        </span>
      </div>

      <div className="mt-3 space-y-2">
        {data.insights.slice(0, 5).map((item: HubInsight, idx: number) => (
          <div key={idx} className="rounded border p-2 text-xs">
            <div className="flex items-start justify-between gap-2">
              <div className="flex-1">
                <p className="font-medium text-foreground">{item.title}</p>
                <p className="text-muted-foreground">{item.summary}</p>
                {item.href && (
                  <a
                    href={item.href}
                    target="_blank"
                    rel="noopener noreferrer"
                    className="mt-1 inline-block text-xs text-primary hover:underline"
                  >
                    View
                  </a>
                )}
              </div>
              <button
                type="button"
                onClick={() => copyText(`${item.title}\n${item.summary}`)}
                className="ml-2 rounded p-1 hover:bg-muted"
                aria-label="Copy insight"
              >
                <svg className="h-3 w-3" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2">
                  <rect x="9" y="9" width="13" height="13" rx="2" ry="2" />
                  <path d="M5 15H4a2 2 0 0 1-2-2V4a2 2 0 0 1 2-2h9a2 2 0 0 1 2 2v1" />
                </svg>
              </button>
            </div>
          </div>
        ))}
      </div>

      <p className="mt-3 text-[10px] text-muted-foreground/60">
        Tags: #knowledge-rag #graph #hub
      </p>
    </div>
  );
};
```
