### 1) Strategy (merged, no contradictions)

- **Never list files via HF API during training.**  
  Generate a CDN-only manifest once (Mac/CI) after any rate-limit window. Training reads only from the manifest and fetches parquet files directly from `https://huggingface.co/datasets/.../resolve/main/...` (no auth, no rate limits).

- **Lightning Studio must survive idle shutdowns and reuse running instances.**  
  Before launch: reuse a running studio if present; if stopped, start it. During training: poll studio status every 5–10 min; if stopped, restart and resume from checkpoint. Checkpoints use deterministic shard names and are saved to HF dataset repos (or artifact bucket) to respect commit caps.

- **Training code is streaming and minimal.**  
  Use an `IterableDataset` that streams CDN parquet files, keeps only `prompt`/`response`, and never materializes full datasets locally.

### `scripts/build_hf_manifest.py`

```python
#!/usr/bin/env python3
"""
Build HF CDN manifest for a dataset repo/date folder.
Run from Mac/CI after rate-limit window clears.
"""
import json
import os
import sys
from datetime import datetime
from pathlib import Path

from huggingface_hub import HfApi

HF_REPO = os.getenv("HF_DATASET_REPO", "datasets/surrogate-mirror")
DATE_FOLDER = os.getenv("DATE_FOLDER", datetime.utcnow().strftime("%Y-%m-%d"))
OUT_DIR = Path(os.getenv("OUT_DIR", "manifests")) / HF_REPO / DATE_FOLDER

api = HfApi()

def build_manifest() -> Path:
    OUT_DIR.mkdir(parents=True, exist_ok=True)
    entries = api.list_repo_tree(repo_id=HF_REPO, path=DATE_FOLDER, recursive=False)

    files = []
    for entry in entries:
        if entry.type != "file":
            continue
        cdn_url = f"https://huggingface.co/datasets/{HF_REPO}/resolve/main/{DATE_FOLDER}/{entry.path}"
        files.append({
            "path": entry.path,
            "cdn_url": cdn_url,
            "size": getattr(entry, "size", None)
        })

    manifest = {
        "repo": HF_REPO,
        "date": DATE_FOLDER,
        "generated_at": datetime.utcnow().isoformat() + "Z",
        "files": files
    }

    out_path = OUT_DIR / "file_manifest.json"
    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"Manifest written to {out_path} ({len(files)} files)")
    return out_path

if __name__ == "__main__":
    build_manifest()
```

### `surrogate/training/data.py`

```python
import io
import json
import logging
import time
from typing import Dict, Iterator

import pyarrow.parquet as pq
import requests
from torch.utils.data import IterableDataset

logger = logging.getLogger(__name__)

class CDNParquetIterable(IterableDataset):
    def __init__(self, manifest_path: str, max_retries: int = 5):
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.max_retries = max_retries

    def _fetch_parquet(self, item: Dict) -> bytes:
        url = item["cdn_url"]
        for attempt in range(1, self.max_retries + 1):
            try:
                resp = requests.get(url, timeout=30, stream=True)
                resp.raise_for_status()
                return resp.content
            except Exception as exc:
                wait = 2 ** attempt
                logger.warning("Fetch failed %s (attempt %s), retry in %ss: %s", url, attempt, wait, exc)
                if attempt == self.max_retries:
                    raise
                time.sleep(wait)
        raise RuntimeError(f"Failed to fetch {url} after {self.max_retries} attempts")

    def __iter__(self) -> Iterator[Dict]:
        for item in self.manifest["files"]:
            try:
                raw = self._fetch_parquet(item)
                table = pq.read_table(io.BytesIO(raw))
                if "prompt" in table.column_names and "response" in table.column_names:
                    prompts = table["prompt"].to_pylist()
                    responses = table["response"].to_pylist()
                    for p, r in zip(prompts, responses):
                        if p is not None and r is not None:
                            yield {"prompt": p, "response": r}
                else:
                    logger.warning("Missing prompt/response in %s, skipping", item["path"])
            except Exception as exc:
                logger.error("Failed to process %s: %s", item["path"], exc, exc_info=True)
                continue
```
