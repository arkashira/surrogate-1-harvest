# surrogate-1 / design

## Implementation Plan (≤2h)

**Highest-value change**: Add a Mac-side `tools/snapshot_manifest.py` that lists one date-partition via a **single** HF API call, emits `file_manifest.json` with CDN URLs and a training script that uses **CDN-only** fetches (zero HF API calls during training). This implements the CDN bypass pattern and avoids HF API rate limits while enabling Lightning Studio reuse.

### Steps (1h 30m total)

1. **Create tools/snapshot_manifest.py** (20m)  
   - Single API call: `list_repo_tree(path=date_partition, recursive=True)`  
   - Filter to parquet/jsonl files  
   - Emit `file_manifest.json`: `{"cdn_url": "...", "local_path": "...", "size": ...}`  
   - Deterministic ordering for reproducible sharding

2. **Create training/train_cdn.py** (30m)  
   - Load `file_manifest.json`  
   - Use `urllib.request`/`requests` to stream from CDN URLs directly  
   - Parse bytes → `{prompt, response}` projection (no schema mixing)  
   - Yield torch Dataset with zero HF API calls during training

3. **Create launcher/launch_lightning.py** (20m)  
   - Reuse running Studio: check `Teamspace.studios`, attach if running  
   - Fallback: create with `Machine.L40S` (free tier)  
   - Upload manifest + train script to Studio  
   - Start training run with idle monitoring

4. **Update README** (10m)  
   - Add Mac orchestration instructions  
   - Note: “Mac runs only orchestration; training → Lightning Studio”

5. **Test locally** (10m)  
   - Dry-run manifest generation  
   - Validate CDN URLs resolve without auth

### tools/snapshot_manifest.py
```python
#!/usr/bin/env python3
"""
Generate CDN-only file manifest for a date partition.
Usage:
    python tools/snapshot_manifest.py --repo axentx/surrogate-1-training-pairs \
        --date 2026-04-29 --out file_manifest.json
"""
import argparse
import json
import os
import sys
from datetime import datetime

import huggingface_hub

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def build_manifest(repo: str, date_partition: str, out_path: str):
    """
    Single HF API call to list files, then emit CDN manifest.
    """
    api = huggingface_hub.HfApi()
    # Single API call: non-recursive list per folder is safer, but we need full tree.
    # Use recursive=True but limit to one date partition to keep response small.
    try:
        tree = api.list_repo_tree(
            repo_id=repo,
            path=date_partition,
            recursive=True,
            repo_type="dataset",
        )
    except Exception as e:
        print(f"HF API error: {e}", file=sys.stderr)
        # If rate-limited, wait and retry (caller should handle)
        raise

    files = []
    for item in tree:
        if item.type != "file":
            continue
        if not item.path.lower().endswith((".parquet", ".jsonl")):
            continue

        cdn_url = CDN_TEMPLATE.format(repo=repo, path=item.path)
        files.append({
            "cdn_url": cdn_url,
            "hf_path": item.path,
            "size": getattr(item, "size", None),
            "lfs": getattr(item, "lfs", None),
        })

    # Deterministic ordering
    files.sort(key=lambda x: x["hf_path"])

    manifest = {
        "repo": repo,
        "date_partition": date_partition,
        "generated_at": datetime.utcnow().isoformat() + "Z",
        "total_files": len(files),
        "files": files,
    }

    os.makedirs(os.path.dirname(os.path.abspath(out_path)), exist_ok=True)
    with open(out_path, "w") as f:
        json.dump(manifest, f, indent=2)

    print(f"Wrote {len(files)} files to {out_path}")
    return manifest

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Generate CDN manifest")
    parser.add_argument("--repo", default="axentx/surrogate-1-training-pairs")
    parser.add_argument("--date", required=True, help="Date partition, e.g. 2026-04-29")
    parser.add_argument("--out", default="file_manifest.json")
    args = parser.parse_args()

    build_manifest(args.repo, args.date, args.out)
```

### training/train_cdn.py
```python
#!/usr/bin/env python3
"""
CDN-only dataset loader for surrogate-1 training.
Zero HF API calls during training.
"""
import json
import tempfile
from typing import Dict, Iterator, Optional

import numpy as np
import pyarrow as pa
import pyarrow.parquet as pq
import requests
import torch
from torch.utils.data import Dataset, IterableDataset

class CDNParquetIterable(IterableDataset):
    """
    Stream parquet files from CDN URLs with schema projection to {prompt, response}.
    """
    def __init__(self, manifest_path: str, columns=("prompt", "response")):
        with open(manifest_path) as f:
            manifest = json.load(f)
        self.file_entries = manifest["files"]
        self.columns = columns

    def _stream_file(self, entry: Dict):
        url = entry["cdn_url"]
        # Stream download to avoid large memory spikes
        with requests.get(url, stream=True, timeout=60) as r:
            r.raise_for_status()
            with tempfile.NamedTemporaryFile(suffix=".parquet", delete=False) as tmp:
                for chunk in r.iter_content(chunk_size=8192):
                    tmp.write(chunk)
                tmp_path = tmp.name

        try:
            # Read with column projection to avoid mixed-schema issues
            table = pq.read_table(tmp_path, columns=self.columns)
            for batch in table.to_batches():
                df = batch.to_pydict()
                for i in range(len(df[self.columns[0]])):
                    yield {k: df[k][i] for k in self.columns}
        finally:
            import os
            os.unlink(tmp_path)

    def __iter__(self) -> Iterator[Dict]:
        for entry in self.file_entries:
            yield from self._stream_file(entry)

class SimpleMapDataset(Dataset):
    """
    Materialized version for small debug runs.
    """
    def __init__(self, manifest_path: str, limit: Optional[int] = None):
        with open(manifest_path) as f:
            manifest = json.load(f)
        self.samples = []
        count = 0
        for entry in manifest["files"]:
            url = entry["cdn_url"]
            try:
                table = pq.read_table(url, columns=["prompt", "response"])
                df = table.to_pydict()
                for i in range(len(df["prompt"])):
                    self.samples.append({
                        "prompt": df["prompt"][i],
                        "response": df["response"][i],
                    })
                    count += 1
                    if limit and count >= limit:
                        return
            except Exception as e:
                print(f"Skip {url}: {e}")

    def __len__(self):
        return len(self.samples)

    def __getitem__(self, idx):
        return self.samples[idx]

if __name__ == "__main__":
    # Quick test
    import sys
    manifest = sys.argv[1] if len(sys.argv) > 1 else "file_manifest.json"
    ds = SimpleMapDataset(manifest, limit=100)
    print(f"Loaded {len(ds)} samples")
```

#!/usr/bin/env python3
"""
Reuse running Lightning Studio or start new L40S job.
"""
import lightning as L
from lightning.app import LightningStudio

def get_or_create_studio(name: str = "surrogate-1-train"):
    # Reuse running studio to save quota (80hr/mo)
    for studio in L.Teamspace().studios:
        if studio.name == name and studio.status == "running":
            print(f"Reusing running studio: {studio.name}")
            return studio
