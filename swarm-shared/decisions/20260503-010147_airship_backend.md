   - Runs on Mac (or CI) after rate-limit window clears  
   - Uses single `list_repo_tree` per date folder in `dataset-mirror`  
   - Outputs `manifests/{date}/filelist.json` with CDN URLs and byte ranges  
   - Deterministic repo selection via hash-slug → sibling repo (0..5) for commit-cap scaling

2. **Add CDN-only dataset loader** (`surrogate/data/cdn_dataset.py`)  
   - Accepts manifest path at init  
   - Downloads via `https://huggingface.co/datasets/{repo}/resolve/main/{path}` (no auth)  
   - Streams parquet → projects `{prompt, response}` only (no mixed-schema load)  
   - Uses `pyarrow`/`pandas` chunked reads to keep memory bounded  
   - Implements retries with 360s backoff on 429 (CDN tier rarely hits this)

### 1. Manifest Generator

```python
# scripts/build_dataset_manifest.py
#!/usr/bin/env python3
"""
Build a CDN-only manifest for dataset-mirror/YYYY-MM-DD folders.
Usage:
  python build_dataset_manifest.py --repo org/surrogate-dataset-mirror \
    --date 2026-04-29 --out manifests/2026-04-29/filelist.json
"""
import argparse
import hashlib
import json
import os
from pathlib import Path
from typing import List, Dict

from huggingface_hub import HfApi

HF_API = HfApi()

def pick_sibling_repo(repo: str, date: str, slug: str, n_siblings: int = 5) -> str:
    """Deterministic sibling repo selection for commit-cap scaling."""
    key = f"{repo}::{date}::{slug}"
    idx = int(hashlib.sha256(key.encode()).hexdigest(), 16) % n_siblings
    return f"{repo}-sibling-{idx}"

def build_manifest(repo: str, date: str, out_path: Path) -> Dict:
    root = f"batches/mirror-merged/{date}"
    entries: List[Dict] = []

    # Single non-recursive tree call per folder
    tree = HF_API.list_repo_tree(repo=repo, path=root, recursive=False)
    for item in tree:
        if not item.path.endswith(".parquet"):
            continue
        slug = Path(item.path).stem
        target_repo = pick_sibling_repo(repo, date, slug)
        cdn_url = (
            f"https://huggingface.co/datasets/{target_repo}/resolve/main/{item.path}"
        )
        entries.append({
            "path": item.path,
            "cdn_url": cdn_url,
            "repo": target_repo,
            "slug": slug,
            "size": item.size if hasattr(item, "size") else None,
        })

    manifest = {
        "date": date,
        "source_repo": repo,
        "root": root,
        "n_files": len(entries),
        "entries": entries,
    }

    out_path.parent.mkdir(parents=True, exist_ok=True)
    with open(out_path, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Manifest written: {out_path} ({len(entries)} files)")
    return manifest

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Build CDN dataset manifest")
    parser.add_argument("--repo", default="org/surrogate-dataset-mirror")
    parser.add_argument("--date", required=True, help="YYYY-MM-DD")
    parser.add_argument("--out", default=lambda p: Path(f"manifests/{p.date}/filelist.json"))
    args = parser.parse_args()

    out_path = Path(args.out) if isinstance(args.out, str) else args.out
    build_manifest(args.repo, args.date, out_path)
```

### 2. CDN-Only Dataset Loader

```python
# surrogate/data/cdn_dataset.py
import json
import logging
from pathlib import Path
from typing import Iterator, Dict, Any

import pandas as pd
import pyarrow.parquet as pq
import requests
from torch.utils.data import IterableDataset

logger = logging.getLogger(__name__)

class CdnDataset(IterableDataset):
    """
    CDN-only dataset loader for surrogate-1 training.
    Avoids HF API rate limits and mixed-schema pyarrow errors by:
      - Using public CDN URLs (no auth)
      - Streaming parquet and projecting {prompt, response} only
    """
    def __init__(self, manifest_path: str | Path, max_retries: int = 5, backoff: int = 360):
        super().__init__()
        self.manifest_path = Path(manifest_path)
        self.max_retries = max_retries
        self.backoff = backoff
        with open(self.manifest_path) as f:
            self.manifest = json.load(f)
        self.entries = self.manifest["entries"]
        if not self.entries:
            raise ValueError(f"No entries in manifest: {self.manifest_path}")

    def _stream_parquet(self, cdn_url: str) -> Iterator[Dict[str, Any]]:
        for attempt in range(1, self.max_retries + 1):
            try:
                with requests.get(cdn_url, stream=True, timeout=60) as r:
                    r.raise_for_status()
                    # Stream to temp file-like buffer in chunks
                    content = b""
                    for chunk in r.iter_content(chunk_size=8192):
                        content += chunk
                    table = pq.read_table(pq.ParquetFile(pq.ParquetReader(content)))
                    # Project only required fields; ignore mixed schema extras
                    cols = set(table.column_names)
                    prompt_col = next((c for c in ["prompt", "instruction", "input"] if c in cols), None)
                    response_col = next((c for c in ["response", "output", "completion"] if c in cols), None)
                    if not prompt_col or not response_col:
                        logger.warning(f"Missing prompt/response in {cdn_url}, skipping")
                        return
                    df = table.select([prompt_col, response_col]).to_pandas()
                    df.columns = ["prompt", "response"]
                    for _, row in df.iterrows():
                        yield {"prompt": str(row.prompt), "response": str(row.response)}
                return
            except Exception as exc:
                wait = self.backoff if attempt == self.max_retries else 2 ** attempt
                logger.warning(f"Attempt {attempt}/{self.max_retries} failed for {cdn_url}: {exc}. Retry in {wait}s")
                if attempt == self.max_retries:
                    raise
                import time
                time.sleep(wait)

    def __iter__(self) -> Iterator[Dict[str, Any]]:
        for entry in self.entries:
            yield from self._stream_parquet(entry["cdn_url"])
```
