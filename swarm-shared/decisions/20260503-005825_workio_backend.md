# workio / backend

## Final Synthesized Solution

**Core principle**: enforce invariants at the database layer, make retries idempotent, and keep application logic simple and safe under concurrency.

---

### 1. Database schema (migrations)

```sql
-- workio/server/src/db/migrations/0001_punches_and_idempotency.sql
-- At most one open punch per user (any timezone-safe date on created_at)
CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_one_open_per_user
ON punches (user_id, DATE(created_at))
WHERE clock_out_at IS NULL;

-- Idempotency for LINE webhook events (and any other retry-prone source)
CREATE TABLE IF NOT EXISTS line_webhook_idempotency (
  idempotency_key VARCHAR(255) PRIMARY KEY,
  event_type      VARCHAR(50)  NOT NULL,
  user_id         INT          NOT NULL,
  punch_id        INT,
  processed_at    TIMESTAMPTZ  NOT NULL DEFAULT NOW(),
  created_at      TIMESTAMPTZ  NOT NULL DEFAULT NOW()
);

CREATE INDEX IF NOT EXISTS idx_line_idempotency_processed_at
ON line_webhookIdempotency (processed_at);
```

**Why this index**:  
- `DATE(created_at)` is safe and stable (unlike `DATE(clock_in_at)` which could be NULL on inserts).  
- Partial index enforces “one open punch per user per calendar day” at the DB level; duplicates are rejected with a unique violation, not app logic.

### 2. Idempotent punch service (atomic, locked, retry-safe)

```ts
// workio/server/src/services/punchService.ts
import { db } from '../db';
import { punches, lineWebhookIdempotency } from '../db/schema';
import { and, eq, isNull } from 'drizzle-orm';

export type PunchType = 'in' | 'out';

export interface ProcessPunchInput {
  userId: number;
  lineEventId: string;   // stable idempotency key from LINE
  type: PunchType;
  location?: string | null;
  notes?: string | null;
}

export async function processPunch(input: ProcessPunchInput) {
  const { userId, lineEventId, type, location, notes } = input;

  return await db.transaction(async (tx) => {
    // 1) Idempotency check
    const prior = await tx
      .select()
      .from(lineWebhookIdempotency)
      .where(eq(lineWebhookIdempotency.idempotency_key, lineEventId))
      .limit(1);

    if (prior.length > 0) {
      if (!prior[0].punch_id) {
        // Previously recorded as no-op (e.g., clock-out with no open punch)
        return { punch: null, idempotent: true, status: 200 };
      }
      const punch = await tx
        .select()
        .from(punches)
        .where(eq(punches.id, prior[0].punch_id))
        .limit(1);
      return { punch: punch[0] ?? null, idempotent: true, status: 200 };
    }

    // 2) Branch by type
    if (type === 'in') {
      // Let DB reject second open punch via partial unique index
      const [punch] = await tx
        .insert(punches)
        .values({
          user_id: userId,
          clock_in_at: new Date(),
          clock_out_at: null,
          location,
          notes,
        })
        .returning();

      await tx.insert(lineWebhookIdempotency).values({
        idempotency_key: lineEventId,
        event_type: 'clock_in',
        user_id,
        punch_id: punch.id,
      });

      return { punch, idempotent: false, status: 201 };
    }

    // type === 'out'
    // Lock the open row to prevent concurrent updates/races
    const open = await tx
      .select()
      .from(punches)
      .where(and(eq(punches.user_id, userId), isNull(punches.clock_out_at)))
      .limit(1)
      .for('update', { skipLocked: true });

    if (!open.length) {
      // Record idempotent no-op so retries are safe
      await tx.insert(lineWebhookIdempotency).values({
        idempotency_key: lineEventId,
        event_type: 'clock_out_no_open_punch',
        user_id: userId,
      });
      return { punch: null, idempotent: false, status: 409, error: 'No open punch to clock out' };
    }

    const [updated] = await tx
      .update(punches)
      .set({
        clock_out_at: new Date(),
        location_out: location,
        updated_at: new Date(),
      })
      .where(eq(punches.id, open[0].id))
      .returning();

    await tx.insert(lineWebhookIdempotency).values({
      idempotency_key: lineEventId,
      event_type: 'clock_out',
      user_id,
      punch_id: updated.id,
    });

    return { punch: updated, idempotent: false, status: 200 };
  });
}
```

**Key improvements vs candidates**:  
- Single atomic transaction with row-level locking (`for update`) for clock-out.  
- Partial unique index guarantees no duplicate open punches even under concurrent clock-ins (DB rejects, not app).  
- Idempotency table + key prevents duplicate processing on LINE retries.  
- Structured return values let HTTP layer map cleanly to status codes.

### 3. HTTP routes (clear, retry-safe responses)

```ts
// workio/server/src/routes/punchRoutes.ts
import { Router } from 'express';
import { processPunch } from '../services/punchService';

const router = Router();

router.post('/punch', async (req, res) => {
  const { userId, lineEventId, type, location, notes } = req.body;

  if (!lineEventId) {
    return res.status(400).json({ error: 'Missing lineEventId for idempotency' });
  }

  const result = await processPunch({ userId, lineEventId, type, location, notes });

  if (result.error) {
    return res.status(result.status).json({ error: result.error });
  }

  return res.status(result.status).json({
    punch: result.punch,
    idempotent: result.idempotent,
  });
});

export default router;
```

**Error semantics**:  
- `400` for bad requests (missing idempotency key).  
- `409` for domain conflicts (no open punch to clock out).  
- `500` only for unexpected failures (DB constraint violations should be handled and mapped to clear responses where possible).

### 4. LINE webhook adapter (extract stable idempotency key)

```ts
// workio/server/src/routes/lineWebhook.ts
import { Router } from 'express';
import { processPunch } from '../services/punchService';

const router = Router();

router.post('/webhook/line', async (req, res) => {
  const events = req.body.events || [];
  const results = [];

  for (const ev of events) {
    // Build a stable idempotency key from LINE payload
    const lineEventId = `${ev.source?.userId || ''}:${ev.timestamp || ''}:${ev.type || ''}`;
    const type = ev.type === 'message' && ev.message?.text?.toLowerCase().includes('ออก') ? 'out' : 'in';

    const result = await processPunch({
      userId: Number(ev.source?.userId),
      lineEventId,
      type,
      location: ev.message?.location || null,
      notes: ev.message?.text || null,
    });

    results.push({ lineEventId, status: result.status, punch: result.punch, idempotent: result.idempotent });
  }

  return res.json({ results });
});

export default router;
```
