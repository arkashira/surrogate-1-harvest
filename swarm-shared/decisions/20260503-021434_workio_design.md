# workio / design

## 1. Diagnosis
- LINE webhook retries can still create duplicate punches because ingestion is not idempotent at the message-id level (no dedup table or idempotency key enforcement).
- Punch upsert is not atomic across tenant+employee+date+event window, risking races when clock-in/out arrive concurrently.
- No strict tenant isolation in the punch write path (queries rely on app-level filtering rather than row-level constraints).
- Missing server-side validation of GPS radius and clock-in/out sequencing (e.g., clock-in twice without clock-out).
- Punch API returns generic 500 on conflict instead of clear, client-handlable status (idempotent 200/201 vs 409).

## 2. Proposed change
File: `workio/server/src/services/punchService.ts` (create or update)  
Scope: add idempotent, atomic punch ingestion with tenant-scoped uniqueness and clear status codes.  
Also touch: `workio/server/src/routes/punchRoutes.ts` (add POST `/punches` and POST `/webhook/line` handler) and `workio/server/src/db/schema.sql` (add `punch_idempotency` table and `punches` table if not present).

### Service: idempotent, atomic punch creation
```ts
// workio/server/src/services/punchService.ts
import { Pool } from 'pg';

const pool = new Pool({ connectionString: process.env.DATABASE_URL });

export type PunchInput = {
  tenantId: number;
  employeeId: number;
  type: 'IN' | 'OUT';
  punchedAt: Date;
  latitude?: number;
  longitude?: number;
  lineEventId?: string;
  idempotencyKey: string; // client or LINE message id
};

export async function createPunch(input: PunchInput) {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // 1) Idempotency check
    const idem = await client.query(
      'SELECT punch_id FROM punch_idempotency WHERE idempotency_key = $1 AND tenant_id = $2',
      [input.idempotencyKey, input.tenantId]
    );
    if (idem.rows[0]) {
      // Fetch existing punch to return
      const existing = await client.query(
        'SELECT id, type, punched_at FROM punches WHERE id = $1 AND tenant_id = $2',
        [idem.rows[0].punch_id, input.tenantId]
      );
      await client.query('ROLLBACK');
      if (existing.rows[0]) {
        return { status: 'duplicate', punch: existing.rows[0] };
      }
      // Orphaned idempotency row — continue to create new punch but keep key reserved
    }

    // 2) Optional sequencing rule: prevent double-IN without OUT
    if (input.type === 'IN') {
      const last = await client.query(
        `SELECT type FROM punches
         WHERE tenant_id = $1 AND employee_id = $2
         ORDER BY punched_at DESC LIMIT 1`,
        [input.tenantId, input.employeeId]
      );
      if (last.rows[0] && last.rows[0].type === 'IN') {
        await client.query('ROLLBACK');
        return { status: 'error', code: 'double_in', message: 'Previous clock-in has no clock-out' };
      }
    }

    // 3) Insert punch
    const res = await client.query(
      `INSERT INTO punches (tenant_id, employee_id, type, punched_at, latitude, longitude, line_event_id)
       VALUES ($1, $2, $3, $4, $5, $6, $7)
       RETURNING id, type, punched_at`,
      [input.tenantId, input.employeeId, input.type, input.punchedAt, input.latitude, input.longitude, input.lineEventId]
    );
    const punch = res.rows[0];

    // 4) Record idempotency
    await client.query(
      `INSERT INTO punch_idempotency (idempotency_key, tenant_id, punch_id)
       VALUES ($1, $2, $3)`,
      [input.idempotencyKey, input.tenantId, punch.id]
    );

    await client.query('COMMIT');
    return { status: 'created', punch };
  } catch (err: any) {
    await client.query('ROLLBACK');
    // Unique violation on (tenant_id, employee_id, punched_at, type) -> duplicate
    if (err.code === '23505') {
      // Try to fetch existing
      const exist = await client.query(
        'SELECT id, type, punched_at FROM punches WHERE tenant_id = $1 AND employee_id = $2 AND punched_at = $3 AND type = $4',
        [input.tenantId, input.employeeId, input.punchedAt, input.type]
      );
      if (exist.rows[0]) {
        return { status: 'duplicate', punch: exist.rows[0] };
      }
    }
    throw err;
  } finally {
    client.release();
  }
}
```

### Route: API + LINE webhook handler
```ts
// workio/server/src/routes/punchRoutes.ts
import express from 'express';
import { createPunch } from '../services/punchService';

const router = express.Router();

// API punch creation (idempotent)
router.post('/punches', async (req, res) => {
  try {
    const { tenantId, employeeId, type, punchedAt, latitude, longitude, idempotencyKey } = req.body;
    if (!tenantId || !employeeId || !type || !punchedAt || !idempotencyKey) {
      return res.status(400).json({ error: 'Missing required fields' });
    }
    const result = await createPunch({
      tenantId,
      employeeId,
      type,
      punchedAt: new Date(punchedAt),
      latitude,
      longitude,
      idempotencyKey
    });
    if (result.status === 'duplicate') return res.status(200).json(result);
    if (result.status === 'error') return res.status(409).json(result);
    return res.status(201).json(result);
  } catch (err) {
    console.error(err);
    return res.status(500).json({ error: 'Internal server error' });
  }
});
