#### ops/heartbeat.sh  (Lightning Studio watchdog)
```bash
#!/usr/bin/env bash
# Cron-safe heartbeat + restart for Lightning Studio.
# Crontab:
#   SHELL=/bin/bash
#   */5 * * * * cd /opt/axentx/vanguard/ops && ./heartbeat.sh >> cron.log 2>&1

set -euo pipefail
cd "$(dirname "$0")"

TEAMSPACE="${LIGHTNING_TEAMSPACE:-vanguard-team}"
STUDIO_NAME="${LIGHTNING_STUDIO:-vanguard-training}"
MACHINE="${LIGHTNING_MACHINE:-L40S}"
HEARTBEAT_FILE="heartbeat.json"
TIMESTAMP=$(date -u +"%Y-%m-%dT%H:%M:%SZ")

STATUS="unknown"

if python3 -c "import lightning; print('ok')" 2>/dev/null; then
  STATUS_SCRIPT=$(python3 - <<PY
import json, sys, traceback
try:
    from lightning import Teamspace, Studio, Machine
    ts = Teamspace(name="${TEAMSPACE}")
    running = [s for s in ts.studios if s.name == "${STUDIO_NAME}" and s.status == "Running"]
    if running:
        s = running[0]
        out = {"status": "running", "machine": str(s.machine or "${MACHINE}"), "last_ping": "${TIMESTAMP}"}
    else:
        stopped = [s for s in ts.studios if s.name == "${STUDIO_NAME}"]
        if stopped:
            s = stopped[0]
            s.start(machine=Machine("${MACHINE}"))
            out = {"status": "restarted", "machine": "${MACHINE}", "last_ping": "${TIMESTAMP}"}
        else:
            Studio.create(name="${STUDIO_NAME}", teamspace="${TEAMSPACE}", machine=Machine("${MACHINE}"))
            out = {"status": "created", "machine": "${MACHINE}", "last_ping": "${TIMESTAMP}"}
    print(json.dumps(out))
except Exception as e:
    print(json.dumps({"status": "error", "error": str(e), "traceback": traceback.format_exc(), "last_ping": "${TIMESTAMP}"}))
PY
  )
  echo "$STATUS_SCRIPT" > "$HEARTBEAT_FILE"
  STATUS=$(echo "$STATUS_SCRIPT" | python3 -c "import sys,json; print(json.load(sys.stdin).get('status','unknown'))")
else
  echo "{\"status\":\"no_sdk\",\"last_ping\":\"$TIMESTAMP\"}" > "$HEARTBEAT_FILE"
  STATUS="no_sdk"
fi

echo "[$TIMESTAMP] heartbeat: $STATUS"
```

#### ops/manifest.py  (CDN-only manifest generator)
```python
#!/usr/bin/env python3
"""
Generate CDN-only manifest for a HuggingFace dataset date folder.
Run once per ingestion to avoid HF API 429s during training.

Usage:
  python3 manifest.py --repo datasets/vanguard-ingest --date 2026-05-02 --out manifest.json
"""
import argparse
import json
import os
import sys
from datetime import datetime, timezone

try:
    from huggingface_hub import list_repo_tree
except ImportError:
    print("Install huggingface_hub: pip install huggingface_hub", file=sys.stderr)
    sys.exit(1)

def main() -> None:
    parser = argparse.ArgumentParser(description="Generate CDN manifest for HF dataset folder")
    parser.add_argument("--repo", required=True, help="HF repo (e.g., datasets/vanguard-ingest)")
    parser.add_argument("--date", required=True, help="Date folder (YYYY-MM-DD)")
    parser.add_argument("--out", default="manifest.json", help="Output JSON file")
    parser.add_argument("--token", default=os.getenv("HF_TOKEN", ""), help="HF token (optional for public repos)")
    args = parser.parse_args()

    folder = f"{args.date}/"
    print(f"Listing {args.repo} tree at {folder} (non-recursive)...")
    try:
        tree = list_repo_tree(repo_id=args.repo, path=folder, recursive=False, token=args.token or None)
    except Exception as e:
        print(f"Error listing repo tree: {e}", file=sys.stderr)
        sys.exit(1)

    files = [item.rfilename for item in tree if item.rfilename.endswith((".parquet", ".jsonl", ".json"))]
    manifest = {
        "repo": args.repo,
        "folder": folder.rstrip("/"),
        "generated_at_utc": datetime.now(timezone.utc).isoformat(),
        "cdn_base": f"https://huggingface.co/datasets/{args.repo}/resolve/main",
        "files": sorted(files),
        "file_urls": [f"https://huggingface.co/datasets/{args.repo}/resolve/main/{f}" for f in sorted(files)],
        "note": "CDN-only URLs — safe for Lightning training with zero API calls during data loading"
    }

    out_path = os.path.join(os.path.dirname(__file__), args.out)
    with open(out_path, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Wrote {len(files)} files to {out_path}")

if __name__ == "__main__":
    main()
```
