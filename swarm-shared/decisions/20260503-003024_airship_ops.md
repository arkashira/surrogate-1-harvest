**Goal**: Eliminate HF API rate limits and mixed-schema ingestion failures by switching surrogate training to a manifest-driven, CDN-only data loader that is deterministic, retry-safe, and memory-efficient.

### Core actions (do in order)
1. Add `scripts/build_file_manifest.py` — runs on Mac orchestrator, lists one date folder via `list_repo_tree`, writes `file_manifest.json` with CDN URLs and optional ETag/sha256.
2. Add `surrogate/data/cdn_dataset.py` — `IterableDataset` that reads `file_manifest.json`, streams files via `requests` (no auth), yields `{prompt, response}` with per-file retries, integrity checks, and graceful skip on malformed files.
3. Update `surrogate/train.py` — replace `load_dataset(...)` with `CdnDataset(manifest_path=...)`.
4. Validate locally with a small manifest; confirm zero HF API calls during epoch and stable training step.

### 1) `scripts/build_file_manifest.py`

```python
#!/usr/bin/env python3
"""
Generate file_manifest.json for a single date folder in a HF dataset repo.
Run on Mac orchestrator (once per snapshot). Avoids HF API list_repo_files recursion.
"""
import hashlib
import json
import os
import sys
from datetime import datetime
from pathlib import Path

try:
    from huggingface_hub import HfApi
except Exception:  # pragma: no cover
    HfApi = None

HF_REPO = os.getenv("HF_DATASET_REPO", "datasets/your-org/surrogate-ingest")
DATE_FOLDER = os.getenv("HF_DATE_FOLDER", datetime.utcnow().strftime("%Y-%m-%d"))
OUTPUT_PATH = Path(os.getenv("MANIFEST_OUT", "file_manifest.json"))
CDN_PREFIX = f"https://huggingface.co/datasets/{HF_REPO}/resolve/main"
INCLUDE_SHA256 = os.getenv("INCLUDE_SHA256", "1").strip().lower() in ("1", "true", "yes")

def build_via_hf_api() -> list[dict]:
    api = HfApi()
    entries = api.list_repo_tree(repo_id=HF_REPO, path=DATE_FOLDER, recursive=True)
    files = []
    for e in entries:
        if e.type != "file":
            continue
        if not e.path.lower().endswith((".parquet", ".jsonl")):
            continue
        item = {
            "repo": HF_REPO,
            "path": e.path,
            "cdn_url": f"{CDN_PREFIX}/{e.path}",
            "size": getattr(e, "size", None),
        }
        if INCLUDE_SHA256 and hasattr(e, "lfs") and isinstance(getattr(e, "lfs", None), dict):
            oid = e.lfs.get("oid", "") if isinstance(e.lfs, dict) else ""
            if oid.startswith("sha256:"):
                item["sha256"] = oid.split(":", 1)[1]
        files.append(item)
    return files

def build_via_cdn_list_fallback() -> list[dict]:
    raise RuntimeError(
        "HF API unavailable and no fallback listing implemented. "
        "Set HF_DATASET_REPO/HF_DATE_FOLDER and ensure huggingface_hub is installed, "
        "or provide MANIFEST_OUT manually."
    )

def main() -> None:
    if HfApi is not None:
        files = build_via_hf_api()
    else:
        files = build_via_cdn_list_fallback()

    manifest = {
        "generated_at_utc": datetime.utcnow().isoformat() + "Z",
        "repo": HF_REPO,
        "date_folder": DATE_FOLDER,
        "files": files,
    }
    OUTPUT_PATH.parent.mkdir(parents=True, exist_ok=True)
    with open(OUTPUT_PATH, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Wrote {len(files)} files to {OUTPUT_PATH}")

if __name__ == "__main__":
    main()
```

Make executable:
```bash
chmod +x scripts/build_file_manifest.py
```

### 2) `surrogate/data/cdn_dataset.py`

```python
import hashlib
import json
import logging
import os
import tempfile
from typing import Dict, Iterator, Optional

import requests
import torch
from torch.utils.data import IterableDataset

try:
    import pyarrow.parquet as pq
    import pyarrow as pa
except Exception:  # pragma: no cover
    pq = None
    pa = None

logger = logging.getLogger(__name__)


class CdnDataset(IterableDataset):
    """
    CDN-only dataset loader.
    Reads file_manifest.json and streams files directly via HTTP (no HF auth).
    Yields { "prompt": str, "response": str } samples.
    """

    def __init__(
        self,
        manifest_path: str,
        prompt_field: str = "prompt",
        response_field: str = "response",
        max_retries: int = 3,
        timeout: int = 30,
        verify_checksum: bool = True,
        worker_chunk_shuffle: bool = True,
    ):
        super().__init__()
        self.manifest_path = manifest_path
        self.prompt_field = prompt_field
        self.response_field = response_field
        self.max_retries = max_retries
        self.timeout = timeout
        self.verify_checksum = verify_checksum
        self.worker_chunk_shuffle = worker_chunk_shuffle
        self._load_manifest()

    def _load_manifest(self) -> None:
        with open(self.manifest_path) as f:
            manifest = json.load(f)
        self.files = manifest.get("files", [])
        if not self.files:
            logger.warning("No files found in manifest: %s", self.manifest_path)

    def _worker_init(self) -> Iterator[Dict[str, str]]:
        worker_info = torch.utils.data.get_worker_info()
        files = [f["cdn_url"] for f in self.files]
        if worker_info is not None:
            # deterministic per-worker shard
            per_worker = len(files) // worker_info.num_workers
            wid = worker_info.id
            files = files[wid * per_worker : (wid + 1) * per_worker]
        if self.worker_chunk_shuffle:
            import random
            rng = random.Random(worker_info.seed if worker_info else 42)
            rng.shuffle(files)
        for url in files:
            yield from self._iter_file(url)

    def __iter__(self) -> Iterator[Dict[str, str]]:
        return self._worker_init()

    # ---- download & integrity ----

    def _sha256_of_bytes(self, data: bytes) -> str:
        return hashlib.sha256(data).hexdigest()

    def _download_to_bytes(self, url: str, expected_sha256: Optional[str] = None) -> bytes:
        for attempt in range(1, self.max_retries + 1):
            try:
                with requests.get(url, timeout=self.timeout, stream=True) as r:
                    r.raise_for_status()
                    chunks = []
                    for chunk in r.iter_content(chunk_size=1024 * 1024):
                        chunks.append(chunk)
                    data = b"".join(chunks)
                    if self.verify_checksum and expected_sha256:
                        got = self._sha256_of_bytes(data)
                        if got != expected_sha256:
                            raise ValueError(f"Checksum mismatch for {url}: expected {expected_sha256}, got {got}")
                    return data
            except Exception as exc:
                logger.warning("Attempt %s/%s failed for %s: %s", attempt, self.max_retries, url, exc)
                if attempt == self.max_retries:
                    raise
        raise RuntimeError(f"Unreachable: failed to download {url}")

    def _parse_jsonl_lines(self, url: str, data: bytes) -> Iterator[Dict[str, str]]:
