# workio / ops

## 1. Diagnosis

- **No production-grade process manager** — backend runs via `npm run dev` (nodemon) which is not resilient to crashes, reboots, or log rotation in prod.
- **Missing health/liveness endpoint** — no `/health` or `/ready` for container/orchestration checks and automated restart decisions.
- **No log structure or rotation** — console logs go to stdout unstructured; on crash/restart context is lost and disk can fill.
- **Environment drift risk** — `.env.example` exists but no validation that required vars are present and correctly typed at startup.
- **No graceful shutdown** — in-flight LINE webhook requests can be cut mid-processing on restart, increasing duplicate punch risk.

## 2. Proposed change

Add a small, zero-dependency ops layer to `workio/server/src/ops/` and wire it into the main entrypoint:

- `workio/server/src/ops/health.ts` — `/health` and `/ready` endpoints.
- `workio/server/src/ops/env.ts` — validate required env vars and cast types at boot.
- `workio/server/src/ops/logging.ts` — structured JSON logger with level and correlation id.
- `workio/server/src/index.ts` — integrate ops, add graceful shutdown, and export for PM2/start script.
- `workio/server/ecosystem.config.js` — PM2 process file for resilient prod runtime (or systemd unit if preferred).

## 3. Implementation

### 3.1 Environment validation

`workio/server/src/ops/env.ts`
```ts
import { config } from 'dotenv';
config();

const required = {
  NODE_ENV: process.env.NODE_ENV || 'development',
  PORT: process.env.PORT || '3000',
  DATABASE_URL: process.env.DATABASE_URL,
  LINE_CHANNEL_ACCESS_TOKEN: process.env.LINE_CHANNEL_ACCESS_TOKEN,
  LINE_CHANNEL_SECRET: process.env.LINE_CHANNEL_SECRET,
  BASE_URL: process.env.BASE_URL,
};

const missing = Object.entries(required).filter(([, v]) => !v || String(v).trim() === '');
if (missing.length > 0) {
  console.error('Missing required environment variables:', missing.map(([k]) => k).join(', '));
  process.exit(1);
}

export const env = {
  NODE_ENV: required.NODE_ENV,
  PORT: Number(required.PORT),
  DATABASE_URL: required.DATABASE_URL,
  LINE_CHANNEL_ACCESS_TOKEN: required.LINE_CHANNEL_ACCESS_TOKEN,
  LINE_CHANNEL_SECRET: required.LINE_CHANNEL_SECRET,
  BASE_URL: required.BASE_URL,
  LOG_LEVEL: (process.env.LOG_LEVEL || 'info') as 'debug' | 'info' | 'warn' | 'error',
};
```

### 3.2 Structured logger

`workio/server/src/ops/logging.ts`
```ts
import { env } from './env';

type Level = 'debug' | 'info' | 'warn' | 'error';

function serialize(level: Level, message: string, meta?: Record<string, unknown>) {
  const entry = {
    ts: new Date().toISOString(),
    level,
    msg: message,
    pid: process.pid,
    ...(meta && Object.keys(meta).length > 0 && { meta }),
  };
  if (level === 'error') console.error(JSON.stringify(entry));
  else console.log(JSON.stringify(entry));
}

export const logger = {
  debug: (msg: string, meta?: Record<string, unknown>) => env.LOG_LEVEL === 'debug' && serialize('debug', msg, meta),
  info: (msg: string, meta?: Record<string, unknown>) => serialize('info', msg, meta),
  warn: (msg: string, meta?: Record<string, unknown>) => serialize('warn', msg, meta),
  error: (msg: string, meta?: Record<string, unknown>) => serialize('error', msg, meta),
};
```

### 3.3 Health endpoints

`workio/server/src/ops/health.ts`
```ts
import { Router } from 'express';
import { logger } from './logging';
import { pool } from '../db'; // adjust import path to your db client

export const healthRouter = Router();

let isReady = false;

export function setReady(v: boolean) {
  isReady = v;
}

// Liveness: process is alive
healthRouter.get('/health', (_req, res) => {
  res.json({ status: 'ok', uptime: process.uptime() });
});

// Readiness: dependencies available
healthRouter.get('/ready', async (_req, res) => {
  if (!isReady) return res.status(503).json({ status: 'not_ready' });
  try {
    await pool.query('SELECT 1');
    res.json({ status: 'ready' });
  } catch (err) {
    logger.error('Readiness check failed', { error: err instanceof Error ? err.message : String(err) });
    res.status(503).json({ status: 'db_unavailable' });
  }
});
```

### 3.4 Main index with graceful shutdown

`workio/server/src/index.ts`
```ts
import express from 'express';
import { env } from './ops/env';
import { logger } from './ops/logging';
import { healthRouter, setReady } from './ops/health';
import app from './app'; // your existing express app
import { pool } from './db';

const server = express();

// Ops endpoints mounted first
server.use(healthRouter);

// Main app
server.use(app);

// Start
const start = async () => {
  try {
    // quick db ping
    await pool.query('SELECT 1');
    setReady(true);
    logger.info('Database ready');
  } catch (err) {
    logger.error('Database unavailable at startup', { error: err instanceof Error ? err.message : String(err) });
    // continue — readiness will fail until db recovers
  }

  const http = server.listen(env.PORT, () => {
    logger.info(`Server listening`, { port: env.PORT, env: env.NODE_ENV });
  });

  const shutdown = async (signal: string) => {
    logger.info(`Received ${signal}, shutting down gracefully`);
    setReady(false);

    // Stop accepting new connections
    server.close(async () => {
      logger.info('HTTP server closed');
      try {
        await pool.end();
        logger.info('Database pool closed');
      } catch (err) {
        logger.error('Error closing database pool', { error: err instanceof Error ? err.message : String(err) });
      }
      http.close(() => {
        logger.info('Process exiting');
        process.exit(0);
      });

      // Force exit after timeout
      setTimeout(() => {
        logger.error('Forced shutdown after timeout');
        process.exit(1);
      }, 30_000).unref();
    });
  };

  process.on('SIGTERM', () => shutdown('SIGTERM'));
  process.on('SIGINT', () => shutdown('SIGINT'));
};

start().catch((err) => {
  logger.error('Fatal startup error', { error: err instanceof Error ? err.message : String(err) });
  process.exit(1);
});
```

### 3.5 PM2 ecosystem (prod)

`workio/server/ecosystem.config.js`
```js
module.exports = {
  apps: [
    {
      name: 'workio-server',
      script: 'dist/index.js',
      cwd: '/opt/axentx/workio/server',
      instances: 1,
      exec_mode: 'fork',
      autorestart: true,
      watch: false,
      max_memory_restart: '500M',
      env: {
        NODE_ENV: 'production',
        LOG_LEVEL: 'info',
      },
      error_file: '/var/log/workio/server-error.log',
      out_file: '/var/log/workio/server-out.log',
      log_date_format: 'YYYY-MM-DD HH:mm:ss',
      merge_logs: true,
    },
  ],
};
```

Add npm scripts in `workio/server/package.json`:
```json
{
  "scripts": {
    "build": "tsc",
    "start": "node dist/index.js",
    "dev": "nodemon src/index.ts",
    "start:prod": "pm2 start ecosystem.config.js",
    "stop:prod": "pm2 stop ecosystem.config.js",
    "logs": "pm2 logs workio-server"
  }
}
```
