# workio / ops

### Final synthesized implementation (correct + actionable)

**Diagnosis (merged, resolved)**
- Missing `X-Line-Signature` verification on `/webhook/line` allows spoofed/replayed events.
- No idempotency/replay protection — LINE retries on 5xx/timeouts can create duplicate `attendance_punches`, `leave_requests`, `ot_requests`.
- No durable, DB-level deduplication key (e.g., `line_event_id`) to detect and reject duplicates.
- Handler returns 200 too early or inconsistently; side-effects may not be committed before response, increasing duplicate risk.
- Signature logic (if present) is likely inline and untested; no defense-in-depth middleware.

**Proposed change (merged)**
- Add a `line_webhook_events` table with a unique constraint on `line_event_id` to enforce idempotency at the DB level.
- Add middleware `verifyLineSignature()` that validates `X-Line-Signature` using HMAC-SHA256 and `timingSafeEqual` before route handlers.
- Wrap webhook handler in idempotency check: upsert `line_webhook_events` with `line_event_id` first; if conflict, return 200 immediately (already processed).
- Return 200 only after all side-effects (DB commits) succeed; on failure, return non-2xx so LINE retries.
- Keep signature verification strict: canonical JSON body, constant-time comparison, and reject missing/invalid signatures.

#### 2) Middleware: verifyLineSignature (defense-in-depth)
```ts
// workio/server/src/middleware/lineAuth.ts
import crypto from 'crypto';

export function verifyLineSignature(channelSecret: string) {
  return (req: any, res: any, next: any) => {
    const signature = req.headers['x-line-signature'] as string | undefined;
    if (!signature) {
      return res.status(401).json({ error: 'Missing X-Line-Signature' });
    }

    // Use raw body if available (recommended), otherwise canonical JSON.stringify
    const body = typeof req.rawBody === 'string' ? req.rawBody : JSON.stringify(req.body);
    const expected = crypto
      .createHmac('sha256', channelSecret)
      .update(body, 'utf8')
      .digest('base64');

    // Constant-time comparison
    const sigBuf = Buffer.from(signature);
    const expBuf = Buffer.from(expected);
    if (sigBuf.length !== expBuf.length || !crypto.timingSafeEqual(sigBuf, expBuf)) {
      return res.status(401).json({ error: 'Invalid signature' });
    }

    // Attach validated payload for handlers
    req.validatedBody = req.body;
    next();
  };
}
```

```ts
// workio/server/src/routes/webhook/line.ts
import { Router } from 'express';
import { Pool } from 'pg';
import { verifyLineSignature } from '../../middleware/lineAuth';
import { pool } from '../../db';

const router = Router();
router.use(verifyLineSignature(process.env.LINE_CHANNEL_SECRET || ''));

async function recordEventIfNew(
  client: any,
  lineEventId: string,
  eventType: string,
  sourceType?: string,
  sourceId?: string
) {
  const res = await client.query(
    `INSERT INTO line_webhook_events (line_event_id, event_type, source_type, source_id)
     VALUES ($1, $2, $3, $4)
     ON CONFLICT (line_event_id) DO NOTHING
     RETURNING id`,
    [lineEventId, eventType, sourceType || null, sourceId || null]
  );
  return res.rowCount > 0; // true = newly inserted (not duplicate)
}

router.post('/line', async (req, res) => {
  const client = await pool.connect();
  try {
    const events = req.validatedBody?.events;
    if (!Array.isArray(events) || events.length === 0) {
      return res.status(400).json({ error: 'No events' });
    }

    await client.query('BEGIN');

    // Process each event sequentially within the same transaction
    for (const ev of events) {
      const lineEventId = ev.eventId;
      if (!lineEventId) continue;

      const isNew = await recordEventIfNew(client, lineEventId, ev.type, ev.source?.type, ev.source?.userId);
      if (!isNew) {
        // Already processed — skip side-effects but continue transaction (safe to commit)
        continue;
      }

      // Handle attendance punches
      if (ev.type === 'message' || ev.type === 'postback') {
        const payload = ev.message || ev.postback;
        const data = payload?.data || (payload?.text ? JSON.parse(payload.text) : null);
        const action = data?.action || ev.postback?.data?.action;
        const userId = ev.source?.userId;
        const timestamp = ev.timestamp ? new Date(ev.timestamp) : new Date();

        if (!userId) continue;

        if (action === 'clock_in' || action === 'clock_out') {
          await client.query(
            `INSERT INTO attendance_punches (user_id, punch_type, punched_at, line_event_id, created_at)
             VALUES ($1, $2, $3, $4, NOW())`,
            [userId, action, timestamp, lineEventId]
          );
        }

        if (action === 'request_leave') {
          const leaveType = data?.leaveType || 'other';
          const startAt = data?.startAt ? new Date(data.startAt) : timestamp;
          const endAt = data?.endAt ? new Date(data.endAt) : startAt;
          await client.query(
            `INSERT INTO leave_requests (user_id, leave_type, start_at, end_at, status, line_event_id, created_at)
             VALUES ($1, $2, $3, $4, 'pending', $5, NOW())`,
            [userId, leaveType, startAt, endAt, lineEventId]
          );
        }

        if (action === 'request_ot') {
          const startAt = data?.startAt ? new Date(data.startAt) : timestamp;
          const endAt = data?.endAt ? new Date(data.endAt) : startAt;

            `INSERT INTO ot_requests (user_id, start_at, end_at, status, line_event_id, created_at)
             VALUES ($1, $2, $3, 'pending', $4, NOW())`,
            [userId, startAt, endAt, lineEventId]
          );
        }
      }
