# workio / quality

## 1. Diagnosis

- LINE webhook endpoint lacks idempotency handling: retries and at-least-once delivery can create duplicate punches/leave/OT rows because no idempotency key or dedupe window exists.
- Punch state transitions are not race-safe: concurrent webhook deliveries can produce multiple open punches for the same user/tenant (read-then-insert without row-level locking or atomic upsert).
- No transactional boundary across punch write + audit/notification side-effects: partial failures can leave DB inconsistent vs LINE state.
- Missing server-side validation of clock-in/out rules (geofence, schedule, max daily hours) before write — leads to noisy bad data that downstream reports must reconcile.
- No structured logging/trace-id propagation on webhook requests: hard to debug duplicates or ordering issues in production.

## 2. Proposed change

File: `/opt/axentx/workio/server/src/routes/webhook/line.ts` (or equivalent route handling LINE Messaging API webhooks).  
Scope: add idempotency + atomic punch state transition + transactional write + trace logging. Target: single endpoint handler for clock-in/out events (and leave/OT if handled here).

import crypto from 'crypto';

// Idempotency window: 24h (tunable)
const IDEMPOTENCY_TTL_SECONDS = 24 * 60 * 60;

// Helper: derive stable idempotency key from LINE event
function makeIdempotencyKey(event: any): string {
  // Prefer event.id (LINE message ID) when present; fallback to hash of payload+type+timestamp
  const base = event.id
    ? `line:${event.id}`
    : `line:${event.type}:${event.source.userId}:${event.timestamp}:${JSON.stringify(event.message || event.postback || {})}`;
  return crypto.createHash('sha256').update(base).digest('hex');
}

// Acquire idempotency lock via advisory lock (per-key) to prevent races across workers
async function withIdempotencyLock<T>(
  client: any,
  key: string,
  fn: () => Promise<T>
): Promise<{ executed: boolean; result?: T }> {
  // Map key to a stable 64-bit integer for pg advisory lock
  const buffer = crypto.createHash('md5').update(key).digest();
  const lockId = buffer.readUInt32BE(0) * 4294967296 + buffer.readUInt32BE(4); // 64-bit

  // Try to take lock with short timeout; if taken, check idempotency table
  await client.query('BEGIN');
  try {
    await client.query('SELECT pg_advisory_xact_lock($1)', [lockId]);

    // Check if already processed (idempotency table)
    const { rows } = await client.query(
      `SELECT response_status, response_body
       FROM webhook_idempotency
       WHERE idempotency_key = $1
         AND created_at >= NOW() - ($2 || ' seconds')::interval
       ORDER BY created_at DESC
       LIMIT 1`,
      [key, IDEMPOTENCY_TTL_SECONDS]
    );

    if (rows.length > 0) {
      await client.query('ROLLBACK');
      return { executed: false, result: rows[0].response_body };
    }

    const result = await fn();

    // Record successful processing
    await client.query(
      `INSERT INTO webhook_idempotency (idempotency_key, response_status, response_body)
       VALUES ($1, $2, $3)`,
      [key, 200, JSON.stringify(result)]
    );

    await client.query('COMMIT');
    return { executed: true, result };
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  }
}

// Atomic punch transition (clock-in/out)
async function upsertPunch(
  client: any,
  { tenantId, userId, timestamp, latitude, longitude, lineSource }: any
) {
  // Use single atomic statement to avoid race:
  // - If there is an open punch (clocked_in_at NOT NULL AND clocked_out_at IS NULL), close it.
  // - Otherwise create new open punch.
  // This keeps at most one open punch per user per tenant.
  const { rows } = await client.query(
    `WITH closed AS (
       UPDATE punches
       SET clocked_out_at = $4, updated_at = NOW()
       WHERE tenant_id = $1
         AND user_id = $2
         AND clocked_in_at IS NOT NULL
         AND clocked_out_at IS NULL
       RETURNING *
     )
     INSERT INTO punches (tenant_id, user_id, clocked_in_at, clocked_out_at, latitude, longitude, line_source, created_at, updated_at)
     SELECT $1, $2, $4, NULL, $5, $6, $7, NOW(), NOW()
     WHERE NOT EXISTS (SELECT 1 FROM closed)
     RETURNING *`,
    [tenantId, userId, timestamp, timestamp, latitude, longitude, lineSource]
  );

  return rows[0];
}
