# airship / backend

### Final Consolidated Plan  
**Title**: Airship Backend — HF CDN Bypass, Schema-Safe Loader, Lightning Quota Guard  
**Goal**: Eliminate HF API 429s and `pyarrow.CastError`, reduce Lightning Studio quota burn, and make training resilient to idle-stop.  
**Scope**: `/opt/axentx/airship/surrogate/` (backend only)  

---

### Implementation Steps (≤2h)

1. **HF CDN bypass via manifest**  
   - Add `scripts/build_hf_manifest.py` that performs **one** `list_repo_tree` call per date folder and emits `manifest-<date>.json` containing `{path, cdn_url, size}`.  
   - Training config references this manifest; workers fetch exclusively via CDN (zero API calls during training).

2. **Schema-safe dataset loader**  
   - Replace `load_dataset(streaming=True)` with per-file fetch (CDN-first, fallback to `hf_hub_download`).  
   - Parse only `{prompt, response}` with best-effort column mapping; skip malformed files with warning instead of crashing.  
   - Guard against `pyarrow.CastError` by validating schema before casting and using `pyarrow.BufferReader`.

3. **Lightning Studio reuse + idle-stop guard**  
   - Before `.run()`, list running studios; reuse if name matches and status is `Running`.  
   - If stopped, restart with `target.start(machine=Machine.L40S)`.  
   - Add exponential backoff/retry for transient failures.

4. **Training launcher**  
   - Add `scripts/run_surrogate_training.py` accepting a date-folder arg, building/using manifest, and logging CDN-only fetch + studio reuse confirmation.

```python
# /opt/axentx/airship/surrogate/scripts/build_hf_manifest.py
#!/usr/bin/env python3
import json
import os
import sys
from datetime import datetime
from pathlib import Path

from huggingface_hub import HfApi

HF_REPO = os.getenv("HF_DATASET_REPO", "axentx/surrogate-training-data")
OUT_DIR = Path(__file__).parent.parent / "training" / "manifests"
OUT_DIR.mkdir(parents=True, exist_ok=True)

def build_manifest(date_folder: str) -> dict:
    api = HfApi()
    items = api.list_repo_tree(
        repo_id=HF_REPO,
        path=date_folder,
        repo_type="dataset",
        recursive=False,
    )

    files = []
    for item in items:
        if item.type != "file":
            continue
        cdn_url = (
            f"https://huggingface.co/datasets/{HF_REPO}/resolve/main/"
            f"{date_folder}/{item.path}"
        )
        files.append({
            "path": item.path,
            "cdn_url": cdn_url,
            "size": getattr(item, "size", None),
        })

    manifest = {
        "repo": HF_REPO,
        "date_folder": date_folder,
        "created_at": datetime.utcnow().isoformat() + "Z",
        "files": files,
        "total_files": len(files),
    }

    out_path = OUT_DIR / f"manifest-{date_folder}.json"
    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"Manifest written: {out_path} ({len(files)} files)")
    return manifest

if __name__ == "__main__":
    if len(sys.argv) < 2:
        print("Usage: build_hf_manifest.py <date-folder>")
        sys.exit(1)
    build_manifest(sys.argv[1])
```

#### 2. Schema-Safe Loader (CDN-first)
```python
# /opt/axentx/airship/surrogate/training/data.py
import json
import logging
from pathlib import Path
from typing import Dict, Iterator

import pyarrow as pa
import pyarrow.parquet as pq
import requests
from huggingface_hub import hf_hub_download

logger = logging.getLogger(__name__)
HF_REPO = "axentx/surrogate-training-data"

def load_via_cdn(file_path: str, cdn_url: str) -> bytes:
    resp = requests.get(cdn_url, timeout=30)
    resp.raise_for_status()
    return resp.content

def project_to_prompt_response(content: bytes, file_path: str) -> Iterator[Dict[str, str]]:
    suffix = Path(file_path).suffix.lower()
    try:
        if suffix == ".parquet":
            table = pq.read_table(pa.BufferReader(content))
            cols = set(table.column_names)
            prompt_col = next((c for c in ("prompt", "input", "question") if c in cols), None)
            response_col = next((c for c in ("response", "output", "answer") if c in cols), None)
            if not prompt_col or not response_col:
                logger.warning("No prompt/response cols in %s (cols: %s)", file_path, cols)
                return
            prompts = table.column(prompt_col).to_pylist()
            responses = table.column(response_col).to_pylist()
            for p, r in zip(prompts, responses):
                if p is not None and r is not None:
                    yield {"prompt": str(p), "response": str(r)}

        elif suffix == ".jsonl":
            for line in content.decode("utf-8").splitlines():
                line = line.strip()
                if not line:
                    continue
                obj = json.loads(line)
                p = obj.get("prompt") or obj.get("input") or obj.get("question")
                r = obj.get("response") or obj.get("output") or obj.get("answer")
                if p is not None and r is not None:
                    yield {"prompt": str(p), "response": str(r)}

        elif suffix == ".json":
            obj = json.loads(content.decode("utf-8"))
            items = obj if isinstance(obj, list) else [obj]
            for item in items:
                p = item.get("prompt") or item.get("input") or item.get("question")
                r = item.get("response") or item.get("output") or item.get("answer")
                if p is not None and r is not None:
                    yield {"prompt": str(p), "response": str(r)}
        else:
            logger.warning("Unsupported file type: %s", file_path)
    except Exception as exc:
        logger.warning("Failed to project %s: %s", file_path, exc)

def iter_dataset(manifest_path: str):
    manifest = json.loads(Path(manifest_path).read_text())
    for entry in manifest["files"]:
        try:
            content = load_via_cdn(entry["path"], entry["cdn_url"])
            yield from project_to_prompt_response(content, entry["path"])
        except Exception as exc:
            logger.warning("Skipping %s: %s", entry["path"], exc)
```

#### 3. Lightning Studio Reuse + Idle-Stop Guard
```python
# /opt/axentx/airship/surrogate/training/lightning_utils.py
import time
import lightning as L
from typing import Optional

def get_or_create_studio(name: str, machine_type=L.Machine.L40S) -> L.Studio:
    teamspace = L.Teamspace()
    for studio in teamspace.studios:
        if studio.name == name:
            if studio.status == "Running":
                print(f"Reusing running studio: {name}")
                return studio
            else:
                print(f"Restarting stopped studio: {name}")
                studio.start(machine=machine_type)
                studio.wait_for_ready()
                return studio

    print(f"Creating new studio: {name}")
    return teamspace.create_studio(name=name, machine=machine_type)

def run_with_retry(fn, max_retries=3, base_delay=1.0):
    for attempt in range(1, max_retries + 1):
        try:
            return fn()
        except Exception as exc:
            if attempt == max_retries:
                raise
            delay = base_delay * (2 ** (attempt - 1))
            print(f"Attempt {attempt} failed: {exc}. Retrying in {delay}s...")
            time.sleep(delay)
```
