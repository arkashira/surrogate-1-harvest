## 1) Goal (≤2 h)
Implement a **manifest-driven, CDN-only dataset loader** for Surrogate-1 training that:
- Pre-lists the target date folder once (Mac orchestration) → saves `files_YYYYMMDD.json`  
- Embeds the manifest in `train.py` so Lightning training does **zero HF API calls** during data loading  
- Projects heterogeneous repo files to `{prompt, response}` only at parse time (avoids `pyarrow.CastError`)  
- Uses HF CDN URLs (no auth, bypasses `/api/` 429 limits)  
- Reuses a running Lightning Studio and auto-restarts if idle-stopped

### 3.1 `scripts/build_manifest.py`
```python
#!/usr/bin/env python3
"""
Generate a manifest of parquet/json/jsonl files for a single date folder
in the surrogate-1 repo. Run from Mac (or any orchestrator).

Usage:
  python scripts/build_manifest.py --repo surrogate-1 --date 2026-04-29 --out manifests/files_20260429.json
"""
import argparse
import json
import os
import time
from typing import List

from huggingface_hub import HfApi, login

CDN_BASE = "https://huggingface.co/datasets"

def build_manifest(repo_id: str, date_folder: str, out_path: str, max_retries: int = 3) -> List[dict]:
    api = HfApi()
    if token := os.getenv("HF_TOKEN"):
        login(token, add_to_git_credential=False)

    prefix = f"{date_folder}/"
    for attempt in range(1, max_retries + 1):
        try:
            files = api.list_repo_tree(
                repo_id=repo_id,
                repo_type="dataset",
                path=prefix,
                recursive=False
            )
            file_paths = [
                f.rfilename for f in files
                if f.rfilename.endswith((".parquet", ".json", ".jsonl"))
            ]
            break
        except Exception as exc:
            if attempt == max_retries:
                raise
            wait = 30 * attempt
            print(f"[attempt {attempt}] list_repo_tree failed: {exc}. Retrying in {wait}s")
            time.sleep(wait)
    else:
        file_paths = []

    items = [
        {
            "repo_id": repo_id,
            "file": fp,
            "cdn_url": f"{CDN_BASE}/{repo_id}/resolve/main/{fp}",
        }
        for fp in sorted(file_paths)
    ]

    os.makedirs(os.path.dirname(out_path), exist_ok=True)
    with open(out_path, "w") as f:
        json.dump(items, f, indent=2)

    print(f"Wrote {len(items)} files to {out_path}")
    return items

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Build CDN manifest for a date folder.")
    parser.add_argument("--repo", default="surrogate-1", help="HF dataset repo id")
    parser.add_argument("--date", required=True, help="Date folder (e.g. 2026-04-29)")
    parser.add_argument("--out", required=True, help="Output JSON path")
    args = parser.parse_args()
    build_manifest(repo_id=args.repo, date_folder=args.date, out_path=args.out)
```

### 3.2 `surrogate/data.py`
```python
import json
import time
from io import BytesIO
from pathlib import Path
from typing import Dict, Generator, List, Optional

import pyarrow as pa
import pyarrow.parquet as pq
import requests
from datasets import Features, Value

CDN_TIMEOUT = 30
MAX_RETRIES = 5
RETRY_BACKOFF = 5

TRAIN_FEATURES = Features({"prompt": Value("string"), "response": Value("string")})

def _download_with_retry(url: str) -> bytes:
    for attempt in range(1, MAX_RETRIES + 1):
        try:
            resp = requests.get(url, timeout=CDN_TIMEOUT)
            resp.raise_for_status()
            return resp.content
        except Exception as exc:
            if attempt == MAX_RETRIES:
                raise
            wait = RETRY_BACKOFF * attempt
            print(f"[cdn] {url} failed ({exc}), retry {attempt}/{MAX_RETRIES} in {wait}s")
            time.sleep(wait)
    raise RuntimeError(f"All retries exhausted for {url}")

def project_to_prompt_response(table: pa.Table) -> List[Dict[str, str]]:
    """
    Project heterogeneous HF parquet/json to {prompt, response}.
    Handles common field aliases and missing columns gracefully.
    """
    cols = set(table.column_names)

    prompt_col = next((c for c in ("prompt", "instruction", "input", "question") if c in cols), None)
    response_col = next((c for c in ("response", "output", "answer", "completion") if c in cols), None)

    out = []
    for i in range(table.num_rows):
        row = {k: table[k][i].as_py() if hasattr(table[k][i], "as_py") else table[k][i] for k in cols}
        prompt = row.get(prompt_col) if prompt_col else ""
        response = row.get(response_col) if response_col else ""
        # Coerce to string and skip if both empty
        prompt = str(prompt) if prompt is not None else ""
        response = str(response) if response is not None else ""
        if not prompt and not response:
            continue
        out.append({"prompt": prompt, "response": response})
    return out

class CdnParquetLoader:
    """
    Loads parquet/json/jsonl from HF CDN using a pre-built manifest.
    Yields {prompt, response} dicts.
    """
    def __init__(self, manifest_path: str):
        manifest_path = Path(manifest_path)
        with open(manifest_path) as f:
            self.items = json.load(f)

    def __iter__(self) -> Generator[Dict[str, str], None, None]:
        for item in self.items:
            url = item["cdn_url"]
            try:
                data = _download_with_retry(url)
            except Exception as exc:
                print(f"Skipping {url} due to {exc}")
                continue

            try:
                table = pq.read_table(BytesIO(data))
            except Exception:
                # fallback: try as JSON/JSONL
                try:
                    import pyarrow.json as pj
                    table = pj.read_json(BytesIO(data))
                except Exception:
                    print(f"Cannot parse {url} as parquet or json; skipping")
                    continue

            for record in project_to_prompt_response(table):
                yield record
```
