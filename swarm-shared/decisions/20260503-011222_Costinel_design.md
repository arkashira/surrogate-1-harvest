# Costinel / design

**Final synthesized implementation (highest-value, <2h, production-ready)**

**Decision**  
Add a single non-breaking backend endpoint `GET /api/v1/sense/top-hub-signal` that:

1. Best-effort triggers `granite-business-research.sh` if present (fire-and-forget, non-blocking).  
2. Queries the knowledge-RAG service for the top-connected hub (MOC) and 3 actionable docs.  
3. Returns a compact, cache-friendly JSON payload the frontend can render immediately as a “Top-Hub Signal” card.  
4. Falls back to a deterministic static payload if RAG is unavailable, and caches fresh results for 5 minutes to preserve availability and latency targets.

---

**Implementation plan (≤2h)**

1. Locate backend entry (FastAPI recommended). Create `backend/routes/sense.py` and include it in app routing.  
2. Add safe script-runner utility (idempotent, executable check, fire-and-forget, no blocking).  
3. Add RAG query helper with short timeout (≤10s) and minimal assumed contract (`query`, `top_k`, `include_docs`).  
4. Add endpoint returning:
   ```json
   {
     "hub": "MOC",
     "score": 0.94,
     "related_docs": [
       {"title": "...", "snippet": "...", "uri": "..."},
       {"title": "...", "snippet": "...", "uri": "..."},
       {"title": "...", "snippet": "...", "uri": "..."}
     ],
     "generated_at": "2026-05-03T01:06:00Z"
   }
   ```
5. Add error handling + structured logging; never fail the HTTP request because of RAG/script issues.  
6. Add in-process file-backed cache (5m TTL) so the endpoint remains fast and available during downstream outages.  
7. Test locally with curl; verify cache behavior and fallback payload.

**Code (FastAPI, production-ready)**

`backend/routes/sense.py`

```python
import asyncio
import json
import os
import subprocess
from datetime import datetime, timezone
from pathlib import Path
from typing import Optional

from fastapi import APIRouter, HTTPException
import httpx

router = APIRouter(prefix="/api/v1/sense", tags=["sense"])

# Paths and settings
RESEARCH_SCRIPT = Path("/opt/axentx/Costinel/granite-business-research.sh")
RAG_SERVICE_URL = os.getenv("RAG_SERVICE_URL", "http://localhost:8000/query")
CACHE_PATH = Path("/tmp/costinel_top_hub.json")
CACHE_TTL_SECONDS = 300  # 5 minutes
TOP_K = 3

# Deterministic fallback when RAG is unavailable
FALLBACK_PAYLOAD = {
    "hub": "MOC",
    "score": 0.94,
    "related_docs": [
        {
            "title": "Cost Governance Playbook",
            "snippet": "Actionable guardrails for multi-cloud spend.",
            "uri": "docs://playbook/cost-governance",
        },
        {
            "title": "MOC Operating Model",
            "snippet": "Centralized monitoring and decision workflows.",
            "uri": "docs://moc/operating-model",
        },
        {
            "title": "FinOps Signals Taxonomy",
            "snippet": "Standardized signal definitions and SLAs.",
            "uri": "docs://finops/signals-taxonomy",
        },
    ],
}

def _run_research_script() -> None:
    """Best-effort, non-blocking trigger for granite-business-research.sh."""
    if not RESEARCH_SCRIPT.exists() or not os.access(RESEARCH_SCRIPT, os.X_OK):
        return
    try:
        subprocess.Popen(
            ["/bin/bash", str(RESEARCH_SCRIPT)],
            stdout=subprocess.DEVNULL,
            stderr=subprocess.DEVNULL,
            env={**os.environ, "SHELL": "/bin/bash"},
            start_new_session=True,
        )
    except Exception:
        # Best-effort; do not impact API response
        pass

def _load_cached() -> Optional[dict]:
    try:
        if not CACHE_PATH.exists():
            return None
        raw = CACHE_PATH.read_text()
        data = json.loads(raw)
        ts = datetime.fromisoformat(data.get("generated_at", "1970-01-01")).replace(tzinfo=timezone.utc)
        age = (datetime.now(timezone.utc) - ts).total_seconds()
        if age < CACHE_TTL_SECONDS:
            return data
        return None
    except Exception:
        return None

def _save_cache(payload: dict) -> None:
    try:
        CACHE_PATH.write_text(json.dumps(payload, default=str, ensure_ascii=False))
    except Exception:
        pass

async def _query_rag_top_hub() -> dict:
    """Query knowledge-RAG for top hub and related docs."""
    try:
        async with httpx.AsyncClient(timeout=10.0) as client:
            resp = await client.post(
                RAG_SERVICE_URL,
                json={"query": "top hub most connected", "top_k": TOP_K + 1, "include_docs": True},
                headers={"Accept": "application/json"},
            )
            if resp.status_code == 200:
                return resp.json()
    except Exception:
        pass
    return {}

def _normalize_rag_result(rag_result: dict) -> dict:
    hub = rag_result.get("hub") or "MOC"
    score = float(rag_result.get("score") or 0.0)
    raw_docs = rag_result.get("related_docs") or []
    docs = [
        {
            "title": d.get("title", "Untitled"),
            "snippet": d.get("snippet", ""),
            "uri": d.get("uri", ""),
        }
        for d in raw_docs[:TOP_K]
    ]
    # Ensure at least fallback docs if empty
    if not docs:
        docs = FALLBACK_PAYLOAD["related_docs"]
    return {"hub": hub, "score": score, "related_docs": docs}

@router.get("/top-hub-signal")
async def top_hub_signal() -> dict:
    # Non-blocking best-effort research trigger
    _run_research_script()

    # Serve fresh cached payload if available
    cached = _load_cached()
    if cached:
        return cached

    # Query RAG; fallback to deterministic payload on failure
    rag_result = await _query_rag_top_hub()
    normalized = _normalize_rag_result(rag_result) if rag_result else FALLBACK_PAYLOAD

    payload = {
        "hub": normalized["hub"],
        "score": normalized["score"],
        "related_docs": normalized["related_docs"],
        "generated_at": datetime.now(timezone.utc).isoformat(),
    }

    # Cache for availability and latency
    _save_cache(payload)
    return payload
```
