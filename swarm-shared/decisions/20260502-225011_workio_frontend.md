# workio / frontend

## Final Synthesis — One Correct, Actionable Plan

**Core problem**: attendance events are stored ephemerally and submitted unreliably, causing duplicates, lost data, and poor offline UX.  
**Goal**: durable, idempotent, cross-tab-safe queue with automatic retry/backoff and background flush.

---

### 1. Architecture (single source of truth)

- **Queue storage**: IndexedDB (durable, transactional, queryable).  
- **Idempotency**: client-generated idempotency key per event; server must dedupe on this key.  
- **Retry policy**: exponential backoff with jitter, max 5 retries, persisted retry count.  
- **Cross-tab safety**: single-writer via `BroadcastChannel` + storage events; flush triggered by any tab.  
- **Flush triggers**:
  1. Immediately after enqueue (best-effort).
  2. `online` event.
  3. Periodic interval (default 30s).
  4. Optional: Background Sync (if Service Worker available — add later).

```ts
// workio/src/lib/queue/attendanceQueue.ts
import { openDB, DBSchema, IDBPDatabase } from 'idb';

interface AttendanceDB extends DBSchema {
  pending: {
    key: string; // idempotencyKey
    value: {
      idempotencyKey: string;
      action: 'clock-in' | 'clock-out';
      timestamp: number;
      location?: { lat: number; lng: number };
      retries: number;
      lastError: string | null;
      createdAt: number;
    };
  };
}

const DB_NAME = 'workio-attendance';
const STORE = 'pending';
const MAX_RETRIES = 5;
const BASE_DELAY = 1000;
const FLUSH_INTERVAL = 30_000;

async function getDB(): Promise<IDBPDatabase<AttendanceDB>> {
  return openDB<AttendanceDB>(DB_NAME, 1, {
    upgrade(db) {
      db.createObjectStore(STORE, { keyPath: 'key' });
    },
  });
}

function jitter(ms: number): number {
  return ms * 0.5 + Math.random() * ms * 0.5;
}

function wait(ms: number): Promise<void> {
  return new Promise((res) => setTimeout(res, ms));
}

// Cross-tab coordination
let broadcastChannel: BroadcastChannel | null = null;

function getChannel(): BroadcastChannel | null {
  try {
    if (!broadcastChannel) broadcastChannel = new BroadcastChannel('workio-attendance');
    return broadcastChannel;
  } catch {
    return null;
  }
}

function broadcast(msg: { type: 'enqueue' | 'flush' }) {
  try {
    getChannel()?.postMessage(msg);
  } catch {
    // ignore
  }
}

export async function enqueueAttendance(
  action: 'clock-in' | 'clock-out',
  location?: { lat: number; lng: number }
): Promise<string> {
  const idempotencyKey = `${action}:${Date.now()}:${crypto.randomUUID()}`;
  const db = await getDB();
  await db.add(STORE, {
    idempotencyKey,
    action,
    timestamp: Date.now(),
    location,
    retries: 0,
    lastError: null,
    createdAt: Date.now(),
  });

  // Notify other tabs and attempt immediate flush
  broadcast({ type: 'enqueue' });
  // Best-effort fire-and-forget flush; errors handled per-item in flushQueue
  void flushQueue();
  return idempotencyKey;
}

export async function flushQueue(): Promise<void> {
  // Don't hammer when offline
  if (!navigator.onLine) return;

  const db = await getDB();
  const tx = db.transaction(STORE, 'readwrite');
  const store = tx.objectStore(STORE);
  let cursor = await store.openCursor();

  while (cursor) {
    const record = cursor.value;

    // Skip items that exceeded retry limit (keep for manual recovery)
    if (record.retries >= MAX_RETRIES) {
      cursor = await cursor.continue();
      continue;
    }

    // Apply exponential backoff per record to avoid thundering herd
    const delayMs = jitter(BASE_DELAY * 2 ** record.retries);
    await wait(delayMs);

    try {
      const res = await fetch('/api/attendance/clock', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          action: record.action,
          timestamp: record.timestamp,
          location: record.location,
          idempotencyKey: record.idempotencyKey,
        }),
      });

      if (!res.ok) throw new Error(`HTTP ${res.status}`);

      // Remove on success
      await store.delete(record.idempotencyKey);
    } catch (err) {
      // Update retry metadata in-place
      await store.put({
        ...record,
        retries: record.retries + 1,
        lastError: err instanceof Error ? err.message : String(err),
      });
    }

    cursor = await cursor.continue();
  }

  await tx.done;
}

export function startQueueAutoFlush(intervalMs = FLUSH_INTERVAL): () => void {
  // Flush when browser comes online
  window.addEventListener('online', flushQueue);

  // Periodic flush
  const interval = setInterval(flushQueue, intervalMs);

  // Cross-tab flush coordination
  try {
    const ch = getChannel();
    if (ch) {
      ch.onmessage = (ev) => {
        if (ev.data?.type === 'enqueue' || ev.data?.type === 'flush') {
          void flushQueue();
        }
      };
    }
  } catch {
    // ignore environments without BroadcastChannel
  }

  return () => {
    window.removeEventListener('online', flushQueue);
    clearInterval(interval);
    try {
      broadcastChannel?.close();
      broadcastChannel = null;
    } catch {
      // ignore
    }
  };
}
```

Hook:

```ts
// workio/src/hooks/useAttendanceQueue.ts
import { useEffect } from 'react';
import { startQueueAutoFlush, enqueueAttendance } from '../lib/queue/attendanceQueue';

export function useAttendanceQueue() {
  useEffect(() => {
    const stop = startQueueAutoFlush();
    return stop;
  }, []);

  return { enqueueAttendance };
}
```
