# airship / backend

## Final Hardened Implementation (Best-of Synthesis)

**Goal**: Harden `airship discover` into a deterministic, CDN-only orchestrator that eliminates HF API rate limits and PyArrow schema errors while producing reproducible file manifests for Surrogate-1 training.

### Core Synthesis Decisions
- **CDN-first strategy** (C1 + C2 + C3): Use `list_repo_tree` once per date folder, then construct all URLs via `https://huggingface.co/datasets/{repo}/resolve/main/{path}` to bypass auth rate limits.
- **Reproducibility via hashing** (C1 + C3): SHA-256 manifest hash stored alongside manifest; verification on load ensures bitwise reproducibility.
- **Schema hardening** (C2 + C3): Never use `load_dataset(streaming=True)` on heterogeneous repos; download via CDN/`hf_hub_download`, project to `{prompt, response}` only at parse time.
- **Actionable 2-hour scope** (C3): Single-file orchestrator, deterministic manifest, CDN fetch strategy, Lightning Studio integration.
- **Cron/bash hygiene** (C3): Shebang, executable, `SHELL=/bin/bash` in crontab, restart studio on idle timeout.

```python
#!/usr/bin/env python3
"""
CDN-only orchestrator for airship discover.
Eliminates HF API rate limits and PyArrow schema errors.
"""
import json
import hashlib
import os
from datetime import datetime
from pathlib import Path
from typing import List, Dict

from huggingface_hub import HfApi, hf_hub_download
import requests

# Config
REPO_ID = "axentx/surrogate-mirror"  # adjust as needed
DATE_FOLDER = datetime.utcnow().strftime("%Y-%m-%d")
OUTPUT_DIR = Path("/opt/axentx/airship/batches/mirror-merged") / DATE_FOLDER
MANIFEST_PATH = OUTPUT_DIR / "manifest.json"

api = HfApi()

def list_date_folder_cdn() -> List[Dict]:
    """Single API call to list files in date folder (no recursive)."""
    try:
        tree = api.list_repo_tree(
            repo_id=REPO_ID,
            path=DATE_FOLDER,
            recursive=False
        )
        files = [f for f in tree if f.type == "file"]
        return [{"path": f.path, "size": f.size} for f in files]
    except Exception as e:
        raise RuntimeError(f"Failed to list repo tree for {REPO_ID}/{DATE_FOLDER}: {e}")

def build_cdn_url(repo_id: str, path: str) -> str:
    return f"https://huggingface.co/datasets/{repo_id}/resolve/main/{path}"

def hash_manifest(manifest: Dict) -> str:
    return hashlib.sha256(json.dumps(manifest, sort_keys=True).encode()).hexdigest()

def generate_manifest() -> Dict:
    OUTPUT_DIR.mkdir(parents=True, exist_ok=True)

    entries = list_date_folder_cdn()
    files = []
    for e in entries:
        cdn_url = build_cdn_url(REPO_ID, e["path"])
        files.append({
            "path": e["path"],
            "size": e["size"],
            "cdn_url": cdn_url
        })

    manifest = {
        "repo_id": REPO_ID,
        "date": DATE_FOLDER,
        "generated_at_utc": datetime.utcnow().isoformat() + "Z",
        "files": files
    }
    manifest["sha256"] = hash_manifest(manifest)
    return manifest

def save_manifest(manifest: Dict) -> Path:
    with open(MANIFEST_PATH, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2, sort_keys=True)
    return MANIFEST_PATH

def verify_manifest(manifest_path: Path) -> bool:
    with open(manifest_path, "r", encoding="utf-8") as f:
        manifest = json.load(f)
    expected = manifest.get("sha256")
    actual = hash_manifest({k: v for k, v in manifest.items() if k != "sha256"})
    return expected == actual

def airship_discover(repo_id: str, date: str) -> Dict:
    """Deterministic CDN-only discovery for one date folder."""
    global REPO_ID, DATE_FOLDER, OUTPUT_DIR, MANIFEST_PATH
    REPO_ID = repo_id
    DATE_FOLDER = date
    OUTPUT_DIR = Path("/opt/axentx/airship/batches/mirror-merged") / DATE_FOLDER
    MANIFEST_PATH = OUTPUT_DIR / "manifest.json"

    # If valid cached manifest exists, verify and return
    if MANIFEST_PATH.exists() and verify_manifest(MANIFEST_PATH):
        with open(MANIFEST_PATH, "r", encoding="utf-8") as f:
            return json.load(f)

    # Generate and save new manifest
    manifest = generate_manifest()
    save_manifest(manifest)
    return manifest

if __name__ == "__main__":
    m = airship_discover(REPO_ID, DATE_FOLDER)
    print(json.dumps(m, indent=2))
```

#### `train.py` (CDN-only ingestion + schema projection)
```python
import json
from pathlib import Path
import pyarrow.parquet as pq
import requests
from typing import List, Dict

def load_manifest(date: str) -> Dict:
    manifest_path = Path(f"/opt/axentx/airship/batches/mirror-merged/{date}/manifest.json")
    with open(manifest_path, "r", encoding="utf-8") as f:
        manifest = json.load(f)
    return manifest

def cdn_fetch_to_parquet(cdn_url: str, local_path: Path) -> None:
    local_path.parent.mkdir(parents=True, exist_ok=True)
    with requests.get(cdn_url, stream=True) as r:
        r.raise_for_status()
        with open(local_path, "wb") as f:
            for chunk in r.iter_content(chunk_size=8192):
                f.write(chunk)

def project_prompt_response(parquet_path: Path) -> List[Dict]:
    # Project only {prompt, response} at parse time to avoid PyArrow schema errors
    table = pq.read_table(parquet_path, columns=["prompt", "response"])
    return table.to_pylist()

def train(date: str):
    manifest = load_manifest(date)
    records = []
    for f in manifest["files"]:
        if not f["path"].lower().endswith(".parquet"):
            continue
        local_path = Path("/tmp/airship_cdn_cache") / f["path"]
        if not local_path.exists():
            cdn_fetch_to_parquet(f["cdn_url"], local_path)
        records.extend(project_prompt_response(local_path))
    # Continue training with `records`
    return records
```
