### pipeline/manifest.py
```python
#!/usr/bin/env python3
"""
Generate and persist (repo, dateFolder) file manifests to avoid HF API pagination/rate-limits.

Usage:
    python -m pipeline.manifest --repo datasets/owner/repo --date 2026-05-01 --out manifests/
"""
import argparse
import json
import sys
from datetime import datetime, timezone
from pathlib import Path

from huggingface_hub import HfApi

HF_API = HfApi()

def list_date_files(repo_id: str, date_folder: str) -> list[str]:
    """
    Non-recursive top-level list for a date folder. Returns repo-relative paths.
    Avoids recursive/paginated enumeration.
    """
    tree = HF_API.list_repo_tree(repo_id, path=date_folder, recursive=False)
    files = [item["path"] for item in tree if item.get("type") == "file"]
    return sorted(files)

def build_manifest(repo_id: str, date_folder: str, output_path: Path) -> Path:
    files = list_date_files(repo_id, date_folder)
    manifest = {
        "repo_id": repo_id,
        "date_folder": date_folder,
        "generated_at": datetime.now(timezone.utc).isoformat(),
        "files": files,
        "count": len(files),
    }
    output_path.parent.mkdir(parents=True, exist_ok=True)
    output_path.write_text(json.dumps(manifest, indent=2))
    return output_path

def main() -> None:
    parser = argparse.ArgumentParser(description="Generate HF dataset file manifest for a date folder.")
    parser.add_argument("--repo", required=True, help="HF dataset repo id (e.g., datasets/owner/repo)")
    parser.add_argument("--date", required=True, help="Date folder (e.g., 2026-05-01 or batches/mirror-merged/2026-05-01)")
    parser.add_argument("--out", default="manifests", help="Output directory (default: manifests)")
    args = parser.parse_args()

    safe_repo = args.repo.replace("/", "_")
    out_file = Path(args.out) / f"{safe_repo}__{args.date}.json"
    manifest_path = build_manifest(args.repo, args.date, out_file)
    print(f"Manifest written: {manifest_path}")
    print(f"Files: {manifest_path.read_text().count('\"path\"')}")  # quick sanity

if __name__ == "__main__":
    main()
```

#!/usr/bin/env python3
"""
Surrogate-1 training entry (Lightning Studio). Uses CDN-only fetches via pre-built manifest.

Key behaviors:
- Reuse running Studio when possible to save quota.
- Load manifest and fetch files via HF CDN (no authenticated API pagination during training).
- Project each record to {prompt, response} only before tokenization.
"""
import json
import os
import sys
from pathlib import Path
from typing import Dict, List, Optional

import torch
from torch.utils.data import Dataset, DataLoader
from transformers import AutoTokenizer, AutoModelForCausalLM, Trainer, TrainingArguments

try:
    from huggingface_hub import hf_hub_download, HfApi
except ImportError as e:
    print("Install: pip install transformers torch huggingface-hub")
    raise e

# ---- Manifest loader ----
def load_manifest(manifest_path: Path) -> dict:
    if not manifest_path.is_file():
        raise FileNotFoundError(f"Manifest not found: {manifest_path}")
    return json.loads(manifest_path.read_text())

# ---- CDN dataset (zero authenticated API calls during training) ----
class HFCDNDataset(Dataset):
    """
    Downloads files via HF CDN (resolve/main) and yields {prompt, response} pairs.
    Assumes each file is JSONL with at least 'prompt' and 'response' fields.
    """
    def __init__(
        self,
        repo_id: str,
        file_paths: List[str],
        tokenizer_name: str,
        max_length: int = 2048,
        cache_dir: Optional[Path] = None,
    ):
        self.repo_id = repo_id
        self.file_paths = file_paths
        self.tokenizer = AutoTokenizer.from_pretrained(tokenizer_name)
        if self.tokenizer.pad_token is None:
            self.tokenizer.pad_token = self.tokenizer.eos_token
        self.max_length = max_length
        self._cache_dir = Path(cache_dir or ".cache/cdn")
        self._cache_dir.mkdir(parents=True, exist_ok=True)

    def _ensure_cached(self, repo_path: str) -> Path:
        local = hf_hub_download(
            repo_id=self.repo_id,
            filename=repo_path,
            cache_dir=str(self._cache_dir),
            force_download=False,
            resume_download=True,
        )
        return Path(local)

    def __len__(self) -> int:
        return len(self.file_paths)

    def _load_jsonl_records(self, path: Path):
        with path.open("r", encoding="utf-8") as f:
            for line in f:
                line = line.strip()
                if not line:
                    continue
                try:
                    obj = json.loads(line)
                    # Surrogate-1 schema projection: keep only prompt/response
                    prompt = obj.get("prompt")
                    response = obj.get("response")
                    if prompt is None or response is None:
                        continue
                    yield {"prompt": prompt, "response": response}
                except json.JSONDecodeError:
                    continue

    def __getitem__(self, idx: int) -> Dict[str, torch.Tensor]:
        repo_path = self.file_paths[idx]
        local_path = self._ensure_cached(repo_path)

        for record in self._load_jsonl_records(local_path):
            text = f"{record['prompt']}\n\n{record['response']}"
            enc = self.tokenizer(
                text,
                max_length=self.max_length,
                truncation=True,
                padding="max_length",
                return_tensors="pt",
            )
            return {k: v.squeeze(0) for k, v in enc.items()}

        enc = self.tokenizer(
            "",
            max_length=self.max_length,
            truncation=True,
            padding="max_length",
            return_tensors="pt",
        )
        return {k: v.squeeze(0) for k, v in enc.items()}

# ---- Lightning Studio reuse helper ----
def reuse_or_create_studio(target_name: str = "vanguard-train"):
    """
    Reuse a running Studio if present to save quota. Requires lightning-ai installed.
    If not available, returns None (train locally or via script runner).
    """
    try:
        from lightning.app import LightningFlow
        from lightning.app.utilities.cloud import _get_project
        from lightning.app.utilities.network import LightningClient
    except ImportError:
        return None

    try:
        client = LightningClient()
        project = _get_project()
        apps = client.lightningapp_instance_list(project_name=project.project_name)
        for app in apps:
            if app.name == target_name and app.phase == "RUNNING":
                print(f"Reusing running Studio: {app.name} ({app.id})")
                return app
        print("No running Studio found; will create or run locally.")
        return None
    except Exception:
        return None
