# Costinel / ops

## Final Implementation Plan — Top-Hub Signal Panel (Costinel Dashboard)

**Scope & value**  
- Frontend-only, read-only panel that surfaces the most-connected hub (default “MOC”) and its actionable proposals from the knowledge graph.  
- Resilient to missing backend: falls back to local sample.  
- Ships in <2h with zero backend changes and minimal blast radius.

**Chosen stack**: Vue 3 + TypeScript + fetch-first resilient data layer.

### 2) Types (`src/types/knowledge-graph.d.ts`)

```ts
export interface Proposal {
  id: string;
  title: string;
  summary: string;
  impact: 'high' | 'medium' | 'low';
  tags: string[];
  href?: string;
  createdAt: string; // ISO
}

export interface HubInsight {
  hub: string;
  description: string;
  degree: number;
  proposals: Proposal[];
  lastUpdated: string;
}

export interface TopHubSignal {
  ok: boolean;
  data?: HubInsight;
  source: 'api' | 'cache';
  error?: string;
}
```

### 4) Composable (`src/composables/useTopHub.ts`)

```ts
import { ref } from 'vue';
import type { TopHubSignal } from '@/types/knowledge-graph';

const FALLBACK = () => import('@/local-data/top-hub-sample.json').then(m => ({
  ok: true,
  data: m.default as any,
  source: 'cache' as const
}) as TopHubSignal);

export function useTopHub() {
  const signal = ref<TopHubSignal | null>(null);
  const loading = ref(false);
  const error = ref<string | null>(null);

  async function fetchTopHub() {
    loading.value = true;
    error.value = null;

    try {
      const res = await fetch('/api/knowledge-graph/top-hub', {
        method: 'GET',
        headers: { Accept: 'application/json' },
        cache: 'no-store'
      }).catch(() => null);

      if (res && res.ok) {
        const json = (await res.json()) as TopHubSignal;
        if (json?.ok && json.data) {
          signal.value = { ...json, source: 'api' };
          loading.value = false;
          return;
        }
      }
    } catch {
      // noop
    }

    // Fallback to local sample (resilient UX)
    try {
      const cached = await FALLBACK();
      signal.value = { ...cached, source: 'cache' };
    } catch (e: any) {
      error.value = e?.message || 'Failed to load top-hub insights';
      signal.value = null;
    } finally {
      loading.value = false;
    }
  }

  return { signal, loading, error, fetchTopHub };
}
```

### 5) Component (`src/components/TopHubSignalPanel.vue`)

```vue
<template>
  <section class="top-hub-panel">
    <header class="panel-header">
      <h3 class="title">Top-Hub Insight</h3>
      <button
        v-if="!loading"
        class="refresh-btn"
        @click="fetchTopHub"
        title="Refresh insights"
        :disabled="loading"
      >
        ↻
      </button>
    </header>

    <div v-if="loading" class="loading" aria-busy="true">
      Loading insights…
    </div>

    <div v-else-if="error" class="error" role="alert">
      {{ error }}
    </div>

    <div v-else-if="signal?.data" class="content">
      <div class="hub-meta">
        <span class="hub-name">{{ signal.data.hub }}</span>
        <span class="degree">Degree: {{ signal.data.degree }}</span>
        <span class="source-badge" :class="signal.source">
          {{ signal.source === 'api' ? 'Live' : 'Cached' }}
        </span>
      </div>

      <p class="description">{{ signal.data.description }}</p>

      <ul class="proposals" role="list">
        <li
          v-for="p in signal.data.proposals"
          :key="p.id"
          class="proposal"
          :class="`impact-${p.impact}`"
        >
          <div class="proposal-title">{{ p.title }}</div>
          <div class="proposal-summary">{{ p.summary }}</div>
          <div class="proposal-meta">
            <span class="tags">
              <span
                v-for="t in p.tags"
                :key="t"
                class="tag"
              >
                {{ t }}
              </span>
            </span>
            <a
              v-if="p.href"
              :href="p.href"
              class="link"
            >
              View
            </a>
          </div>
        </li>
      </ul>

      <div class="last-updated">
        Updated {{ formatDate(signal.data.lastUpdated) }}
      </div>
    </div>

    <div v-else class="empty">No hub insights available.</div>
  </section>
</template>

<script setup lang="ts">
import { useTopHub } from '@/composables/useTopHub';
import { onMounted } from 'vue';

const { signal, loading, error, fetchTopHub } = useTopHub();

function formatDate(iso: string) {
  return new Intl.DateTimeFormat(undefined, {
    dateStyle: 'medium',
    timeStyle: 'short'
  }).format(new Date(iso));
}

onMounted(() => {
  fetchTopHub();
});
</script>
