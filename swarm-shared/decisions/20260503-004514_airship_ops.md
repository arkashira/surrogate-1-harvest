## Final Integrated Implementation (Best-of-Both)

**Core Value**: Eliminate HF API rate-limit and `pyarrow.CastError` risks during surrogate training by replacing `load_dataset(streaming=True)` with a **manifest-driven, CDN-only data loader** that is reproducible, cache-friendly, and production-ready.

## 1) Manifest Builder (single API call per date folder)

```python
# scripts/build_file_manifest.py
#!/usr/bin/env python3
"""
Build a CDN manifest for a specific date folder in a HF dataset repo.
Usage:
  python build_file_manifest.py \
    --repo datasets/your-org/your-repo \
    --date 2026-04-29 \
    --out data/manifests/manifest-2026-04-29.json
"""

import argparse
import json
import sys
from datetime import datetime, timezone
from pathlib import Path

import requests

HF_API_BASE = "https://huggingface.co/api"


def list_repo_tree(repo: str, path: str = "", recursive: bool = False):
    url = f"{HF_API_BASE}/datasets/{repo}/tree"
    params = {"path": path, "recursive": int(recursive)}
    resp = requests.get(url, params=params, timeout=30)
    if resp.status_code == 429:
        print("HF API 429 — wait 360s before retry", file=sys.stderr)
        sys.exit(1)
    resp.raise_for_status()
    return resp.json()


def build_manifest(repo: str, date_folder: str, out_path: Path):
    tree = list_repo_tree(repo, path=date_folder, recursive=False)
    manifest = []
    for item in tree:
        if item.get("type") != "file":
            continue
        if not item["path"].lower().endswith(".parquet"):
            continue
        repo_slug = repo.replace("datasets/", "")
        manifest.append(
            {
                "repo": repo_slug,
                "path": item["path"],
                "sha": item.get("sha", ""),
                "size": item.get("size", 0),
                "url": f"https://huggingface.co/datasets/{repo_slug}/resolve/main/{item['path']}",
            }
        )

    out_path.parent.mkdir(parents=True, exist_ok=True)
    with out_path.open("w") as f:
        json.dump(
            {
                "generated_at": datetime.now(timezone.utc).isoformat(),
                "repo": repo,
                "date_folder": date_folder,
                "files": manifest,
            },
            f,
            indent=2,
        )
    print(f"Manifest written to {out_path} ({len(manifest)} files)")


if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Build CDN manifest for HF dataset folder.")
    parser.add_argument("--repo", required=True, help="HF dataset repo, e.g. datasets/org/name")
    parser.add_argument("--date", required=True, help="Date folder, e.g. 2026-04-29")
    parser.add_argument("--out", required=True, help="Output JSON path")
    args = parser.parse_args()
    build_manifest(args.repo, args.date, Path(args.out))
```

## 2) CDN-only Dataset Iterable (schema-safe, robust)

```python
# surrogate/data/cdn_parquet_dataset.py
import json
import logging
from typing import Dict, Iterator, Optional

import pyarrow as pa
import pyarrow.parquet as pq
import requests
from tqdm import tqdm

logger = logging.getLogger(__name__)


class CdnParquetDataset:
    """
    Iterate rows from parquet files listed in a manifest using CDN URLs.
    Projects only `prompt` and `response` fields at parse time to avoid CastError.
    """

    def __init__(self, manifest_path: str, max_files: Optional[int] = None):
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.files = self.manifest["files"]
        if max_files:
            self.files = self.files[:max_files]

    def _download_parquet(self, url: str) -> bytes:
        resp = requests.get(url, timeout=60)
        resp.raise_for_status()
        return resp.content

    def _project_row(self, row: Dict) -> Dict:
        return {
            "prompt": row.get("prompt", ""),
            "response": row.get("response", ""),
        }

    def __iter__(self) -> Iterator[Dict]:
        for item in tqdm(self.files, desc="CDN parquet files"):
            try:
                data = self._download_parquet(item["url"])
                pf = pq.ParquetFile(pa.BufferReader(data))
                table = pf.read()

                available = set(table.column_names)
                cols = [c for c in ["prompt", "response"] if c in available]
                if not cols:
                    logger.warning(f"No prompt/response in {item['path']}, skipping")
                    continue

                subtable = table.select(cols)
                # Ensure consistent dtypes to avoid downstream CastError
                if "prompt" in cols:
                    subtable = subtable.set_column(
                        subtable.schema.get_field_index("prompt"),
                        "prompt",
                        pa.array(subtable["prompt"].to_pylist(), type=pa.string()),
                    )
                if "response" in cols:
                    subtable = subtable.set_column(
                        subtable.schema.get_field_index("response"),
                        "response",
                        pa.array(subtable["response"].to_pylist(), type=pa.string()),
                    )

                df = subtable.to_pandas()
                for _, row in df.iterrows():
                    yield self._project_row(row.to_dict())
            except Exception as exc:
                logger.warning(f"Failed to process {item['path']}: {exc}", exc_info=True)
                continue
```

#!/usr/bin/env bash
# scripts/run_training_with_manifest.sh
set -euo pipefail

REPO="datasets/your-org/your-repo"
DATE="2026-04-29"
MANIFEST="data/manifests/manifest-${DATE}.json"

cd "$(dirname "$0")/.."

echo "Building manifest..."
python scripts/build_file_manifest.py --repo "$REPO" --date "$DATE" --out "$MANIFEST"

echo "Starting surrogate training..."
python -m surrogate.train --manifest "$MANIFEST" "$@"

## Key Decisions & Rationale

1. **Manifest-driven, CDN-only**: Removes HF API during training (no 429s) and avoids schema collisions from `load_dataset(streaming=True)`.
2. **Explicit schema projection + dtype enforcement**: Prevents `pyarrow.CastError` by selecting only `prompt`/`response` and casting to `pa.string()`.
