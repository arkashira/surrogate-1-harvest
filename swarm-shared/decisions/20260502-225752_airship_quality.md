## Code Snippets

### 1) Bash-first CLI: `scripts/airship-discover.sh`

```bash
#!/usr/bin/env bash
# airship-discover.sh
# Deterministic CDN-safe manifest generator for HF repos/datasets.
# Usage:
#   ./airship-discover.sh <repo> <ref> <out_dir>
#
# Outputs:
#   ${out_dir}/status.json  - run metadata + top-level tree
#   ${out_dir}/manifest.json - per-file CDN URLs + sibling repo assignment
#
# Notes:
# - Uses HF REST tree API (non-recursive) to avoid 429 on big repos.
# - CDN URLs bypass /api/ auth and rate limits.
# - Deterministic sibling assignment via hash-slug for commit-cap scaling.

set -euo pipefail

REPO="${1:-}"
REF="${2:-main}"
OUT_DIR="${3:-./airship-manifest}"

if [[ -z "${REPO}" ]]; then
  echo "Usage: $0 <repo> [ref] [out_dir]" >&2
  echo "  repo: huggingface repo, e.g. 'datasets/mycorp/myrepo'" >&2
  exit 1
fi

mkdir -p "${OUT_DIR}"

# Normalize repo (strip prefixes)
REPO_CLEAN="${REPO#https://huggingface.co/}"
REPO_CLEAN="${REPO_CLEAN%/}"

# Split org/repo (supports datasets/ or models/ prefix)
IFS='/' read -r -a PARTS <<< "${REPO_CLEAN}"
if [[ "${#PARTS[@]}" -lt 2 ]]; then
  echo "Invalid repo format. Expected org/repo or type/org/repo" >&2
  exit 1
fi

# If first part is datasets|models, shift
if [[ "${PARTS[0]}" == "datasets" || "${PARTS[0]}" == "models" ]]; then
  HF_TYPE="${PARTS[0]}"
  ORG="${PARTS[1]}"
  NAME_PARTS=("${PARTS[@]:2}")
else
  HF_TYPE="datasets"
  ORG="${PARTS[0]}"
  NAME_PARTS=("${PARTS[@]:1}")
fi
REPO_NAME=$(IFS='/'; echo "${NAME_PARTS[*]}")

# Deterministic sibling assignment (0..4) for commit-cap scaling
# Use repo slug to pick sibling repo index deterministically.
SLUG="${ORG}/${REPO_NAME}"
HASH=$(echo -n "${SLUG}" | sha256sum | head -c 8)
SIBLING_IDX=$(( 0x${HASH} % 5 ))  # 5 siblings => 0..4
SIBLING_REPO="${REPO_CLEAN}-sib${SIBLING_IDX}"

# HF API tree (non-recursive) - one call per folder.
# We'll do a shallow listing of the ref root. If deeper traversal is needed,
# embed a path list in CI beforehand (per pattern).
API_TREE="https://huggingface.co/api/${HF_TYPE}/${ORG}/${REPO_NAME}/tree/${REF}?recursive=false"

echo "Fetching tree: ${API_TREE}" >&2
TREE_JSON=$(curl -fsSL "${API_JSON:-${API_TREE}}" || echo '[]')

TIMESTAMP=$(date -u +"%Y-%m-%dT%H:%M:%SZ")
RUN_ID=$(echo -n "${SLUG}:${REF}:${TIMESTAMP}" | sha256sum | head -c 16)

# Build file entries
MANIFEST_ENTRIES=()
PATH_LIST=()

while IFS= read -r row; do
  path=$(echo "$row" | jq -r '.path // empty')
  type=$(echo "$row" | jq -r '.type // empty')
  size=$(echo "$row" | jq -r '.size // 0')
  lfs=$(echo "$row" | jq -r '.lfs // empty')

  if [[ -z "${path}" || "${type}" != "file" ]]; then
    continue
  fi

  # CDN URL (bypasses API auth/rate-limit)
  CDN_URL="https://huggingface.co/${HF_TYPE}/${ORG}/${REPO_NAME}/resolve/${REF}/${path}"

  # For sibling write strategy: if pushing enriched outputs, use sibling repo.
  # This manifest is read-only for training; sibling info is for coordinating writes.
  entry=$(jq -n \
    --arg p "${path}" \
    --arg url "${CDN_URL}" \
    --arg t "${type}" \
    --arg l "${lfs}" \
    --argjson sz "${size}" \
    '{path: $p, cdn_url: $url, type: $t, size: $sz, lfs: $l}')
  MANIFEST_ENTRIES+=("${entry}")
  PATH_LIST+=("${path}")
done < <(echo "${TREE_JSON}" | jq -c '.[]')

MANIFEST_JSON=$(jq -n --argjson items "$(printf '%s\n' "${MANIFEST_ENTRIES[@]}" | jq -s '.')" \
  '{repo: $repo, ref: $ref, generated_at: $ts, sibling_idx: $si, sibling_repo: $sr, files: $items}' \
  --arg repo "${REPO_CLEAN}" \
  --arg ref "${REF}" \
  --arg ts "${TIMESTAMP}" \
  --argjson si "${SIBLING_IDX}" \
  --arg sr "${SIBLING_REPO}")

STATUS_JSON=$(jq -n \
  --arg run_id "${RUN_ID}" \
  --arg repo "${REPO_CLEAN}" \
  --arg ref "${REF}" \
  --arg ts "${TIMESTAMP}" \
  --arg out "${OUT_DIR}" \
  --arg api_tree "${API_TREE}" \
  '{run_id: $run_id, repo: $repo, ref: $ref, generated_at: $ts, out_dir: $out, api_tree: $api_tree, status: "ok", file_count: ($files_count | tonumber)}' \
  --argjson files_count "${#PATH_LIST[@]}")

echo "${MANIFEST_JSON}" > "${OUT_DIR}/manifest.json"
echo "${STATUS_JSON}" > "${OUT_DIR}/status.json"

echo "OK: ${#PATH_LIST[@]} files -> ${OUT_DIR}/"
echo "  manifest.json + status.json written"
```

Make executable:

```bash
chmod +x /opt/axentx/airship/scripts/airship-discover.sh
```
