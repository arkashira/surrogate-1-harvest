- Bypass HF API auth during training: use `https://huggingface.co/datasets/{repo}/resolve/main/{path}` with no Authorization header

#!/usr/bin/env python3
"""
Manifest-driven, CDN-bypass ingestion worker for surrogate-1.

Usage:
  SHARD_ID=0 SHARD_TOTAL=16 DATE=2026-05-03 \
  HF_TOKEN=hf_xxx \
  python bin/dataset-enrich.py \
    --repo axentx/surrogate-1-training-pairs \
    --file-list file-list.json \
    --out-dir batches/public-merged
"""

import argparse
import hashlib
import json
import os
import sys
import time
from datetime import datetime
from pathlib import Path
from typing import Dict, Any

import pyarrow as pa
import pyarrow.parquet as pq
import requests
from datasets import load_dataset

# Local dedup store
sys.path.insert(0, str(Path(__file__).parent / "lib"))
from dedup import DedupStore  # type: ignore

HF_API_BASE = "https://huggingface.co/api"
HF_CDN_BASE = "https://huggingface.co/datasets"

HEADERS = {"Authorization": f"Bearer {os.environ.get('HF_TOKEN', '')}"}


def deterministic_shard(slug: str, total: int) -> int:
    return int(hashlib.md5(slug.encode()).hexdigest()[0:8], 16) % total


def list_date_folder(repo: str, date: str) -> list[str]:
    """Single API call to list files in date folder (non-recursive)."""
    url = f"{HF_API_BASE}/datasets/{repo}/tree/main/batches/public-merged/{date}"
    r = requests.get(url, headers=HEADERS, timeout=30)
    if r.status_code == 429:
        wait = 360
        print(f"Rate limited 429 — waiting {wait}s", file=sys.stderr)
        time.sleep(wait)
        return list_date_folder(repo, date)
    r.raise_for_status()
    items = r.json()
    return [item["path"] for item in items if item["type"] == "file"]


def cdn_download(repo: str, path: str) -> bytes:
    """CDN bypass: no Authorization header, separate rate limits."""
    url = f"{HF_CDN_BASE}/{repo}/resolve/main/{path}"
    r = requests.get(url, timeout=60)
    r.raise_for_status()
    return r.content


def normalize_record(raw: Dict[str, Any]) -> Dict[str, str]:
    """Project heterogeneous schemas to {prompt, response} only."""
    prompt = raw.get("prompt") or raw.get("input") or raw.get("question") or ""
    response = raw.get("response") or raw.get("output") or raw.get("answer") or ""
    return {"prompt": str(prompt).strip(), "response": str(response).strip()}


def process_shard(
    file_paths: list[str],
    shard_id: int,
    shard_total: int,
    dedup: DedupStore,
) -> list[Dict[str, str]]:
    accepted = []
    for path in file_paths:
        try:
            content = cdn_download(REPO, path)
            # Try parquet first, fallback to jsonl
            if path.endswith(".parquet"):
                table = pq.read_table(pa.BufferReader(content))
                rows = table.to_pylist()
            else:
                rows = [json.loads(l) for l in content.decode().splitlines() if l.strip()]

            for row in rows:
                normalized = normalize_record(row)
                if not normalized["prompt"] or not normalized["response"]:
                    continue

                # Deterministic shard assignment by slug (use path as slug)
                if deterministic_shard(path, shard_total) != shard_id:
                    continue

                # Dedup by content hash
                payload = f"{normalized['prompt']}\n{normalized['response']}".encode()
                md5 = hashlib.md5(payload).hexdigest()
                if dedup.seen(md5):
                    continue
                dedup.add(md5)

                accepted.append(normalized)
        except Exception as exc:
            print(f"Skip {path}: {exc}", file=sys.stderr)
    return accepted


def upload_batch(repo: str, date: str, shard_id: int, records: list[Dict[str, str]]) -> None:
    if not records:
        print("No records to upload", file=sys.stderr)
        return

    timestamp = datetime.utcnow().strftime("%H%M%S")
    filename = f"shard{shard_id}-{timestamp}.jsonl"
    out_path = f"batches/public-merged/{date}/{filename}"

    # Write local temp then push via huggingface_hub
    from huggingface_hub import HfApi

    api = HfApi(token=os.environ["HF_TOKEN"])
    lines = "\n".join(json.dumps(r, ensure_ascii=False) for r in records)
    api.upload_file(
        path_or_fileobj=lines.encode(),
        path_in_repo=out_path,
        repo_id=repo,
        repo_type="dataset",
    )
    print(f"Uploaded {len(records)} records to {out_path}")


def main() -> None:
    parser = argparse.ArgumentParser(description="CDN-bypass ingestion worker")
    parser.add_argument("--repo", default="axentx/surrogate-1-training-pairs")
    parser.add_argument("--file-list", help="Path to file-list.json (date folder)")
    parser.add_argument("--out-dir", default="batches/public-merged")
    args = parser.parse_args()

    shard_id = int(os.environ.get("SHARD_ID", "0"))
    shard_total = int(os.environ.get("SHARD_TOTAL", "16"))
    date = os.environ.get("DATE", datetime.utcnow().strftime("%Y-%m-%d"))

    global REPO
    REPO = args.repo

    # Dedup store (central SQLite)
    dedup = DedupStore()

    # Load or generate file list
    if args.file_list and Path(args.file_list).exists():
        with open(args.file_list) as f:
            file_paths = json.load(f)
    else:
        print("No file-list provided — listing date folder via API", file=sys.stderr)
        file_paths = list_date_folder(REPO, date)

    records = process_shard(file_paths, shard_id, shard_total, dedup)
    upload_batch(REPO, date, shard_id, records)
    dedup.close()
