# surrogate-1 / design

## Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16`, `DATE`, `HF_TOKEN`, `REPO_ID` (default: `axentx/surrogate-1-training-pairs`)
- Single API call on the Mac orchestrator (or at worker start if no manifest) to `list_repo_tree(path=f"batches/public-merged/{DATE}", recursive=False)` → save `manifest.json`
- Worker deterministically hashes each filename → `hash(filename) % SHARD_TOTAL`; processes only its shard
- Downloads via **HF CDN bypass** (`https://huggingface.co/datasets/{repo}/resolve/main/{path}`) with no Authorization header → avoids 429 during training data load
- Projects heterogeneous schemas to `{prompt, response}` only at parse time (avoids pyarrow CastError)
- Deduplicates via central md5 store (`lib/dedup.py`)
- Writes `batches/public-merged/{DATE}/shard{SHARD_ID}-{HHMMSS}.jsonl` with deterministic filename (no collisions)
- Optional: reuse existing Lightning Studio for downstream training (not in this worker)

#!/usr/bin/env python3
"""
CDN-bypass, manifest-driven ingestion worker for surrogate-1.
Usage:
  SHARD_ID=3 SHARD_TOTAL=16 DATE=2026-05-03 \
  HF_TOKEN=hf_xxx \
  python bin/dataset-enrich.py
"""

import os
import sys
import json
import hashlib
import datetime
import requests
from pathlib import Path
from typing import List, Dict, Any

import pyarrow as pa
import pyarrow.parquet as pq
import numpy as np

# ---------- config ----------
REPO_ID = os.getenv("REPO_ID", "axentx/surrogate-1-training-pairs")
HF_TOKEN = os.getenv("HF_TOKEN", "")
SHARD_ID = int(os.getenv("SHARD_ID", "0"))
SHARD_TOTAL = int(os.getenv("SHARD_TOTAL", "16"))
DATE = os.getenv("DATE", datetime.date.today().isoformat())
MANIFEST_PATH = Path(os.getenv("MANIFEST_PATH", f"manifest-{DATE}.json"))
OUT_DIR = Path("output")
OUT_DIR.mkdir(exist_ok=True)

# ---------- helpers ----------
def hf_api_get(path: str, params: dict = None, use_auth: bool = True) -> Any:
    url = f"https://huggingface.co/api/{path}"
    headers = {"Authorization": f"Bearer {HF_TOKEN}"} if use_auth and HF_TOKEN else {}
    r = requests.get(url, headers=headers, params=params, timeout=30)
    r.raise_for_status()
    return r.json()

def list_date_folder(date: str) -> List[str]:
    """List files in batches/public-merged/{date} (non-recursive)."""
    folder = f"datasets/{REPO_ID}/tree/main/batches/public-merged/{date}"
    items = hf_api_get(folder, use_auth=True)
    # items are dicts with "path", "type"
    return [it["path"] for it in items if it["type"] == "file"]

def build_manifest(date: str) -> List[str]:
    paths = list_date_folder(date)
    MANIFEST_PATH.write_text(json.dumps(paths, indent=2))
    return paths

def load_manifest(date: str) -> List[str]:
    if MANIFEST_PATH.exists():
        return json.loads(MANIFEST_PATH.read_text())
    return build_manifest(date)

def deterministic_shard(path: str) -> int:
    return int(hashlib.sha256(path.encode()).hexdigest(), 16) % SHARD_TOTAL

def cdn_download_url(path: str) -> str:
    # path is like "batches/public-merged/2026-05-03/file.parquet"
    return f"https://huggingface.co/datasets/{REPO_ID}/resolve/main/{path}"

def safe_project_to_pair(raw: Dict[str, Any]) -> Dict[str, str]:
    """
    Project heterogeneous schemas to {prompt, response}.
    Heuristic fallback keys; adapt per known schema if needed.
    """
    prompt = None
    response = None

    # Common key variants
    for pkey in ("prompt", "instruction", "input", "question", "query"):
        if pkey in raw and isinstance(raw[pkey], str) and raw[pkey].strip():
            prompt = raw[pkey].strip()
            break
    for rkey in ("response", "output", "answer", "completion", "text"):
        if rkey in raw and isinstance(raw[rkey], str) and raw[rkey].strip():
            response = raw[rkey].strip()
            break

    # If both missing, try to salvage from known nested shapes
    if prompt is None or response is None:
        # fallback: join string fields
        str_fields = [v.strip() for v in raw.values() if isinstance(v, str) and v.strip()]
        if len(str_fields) >= 2:
            prompt, response = str_fields[0], str_fields[1]
        elif len(str_fields) == 1:
            prompt, response = str_fields[0], ""
        else:
            prompt, response = "", ""

    return {"prompt": prompt, "response": response}

def process_parquet_file(path: str) -> List[Dict[str, str]]:
    url = cdn_download_url(path)
    # stream download to temp file to avoid loading full parquet into memory twice
    import tempfile
    with tempfile.NamedTemporaryFile(suffix=".parquet", delete=False) as tmp:
        r = requests.get(url, stream=True, timeout=60)
        r.raise_for_status()
        for chunk in r.iter_content(chunk_size=8192):
            tmp.write(chunk)
        tmp_path = tmp.name

    try:
        table = pq.read_table(tmp_path, use_threads=False)
        rows = []
        for i in range(table.num_rows):
            raw = {col: table[col][i].as_py() for col in table.column_names}
            pair = safe_project_to_pair(raw)
            if pair["prompt"] or pair["response"]:
                rows.append(pair)
        return rows
    finally:
        try:
            os.unlink(tmp_path)
        except Exception:
            pass

def process_jsonl_file(path: str) -> List[Dict[str, str]]:
    url = cdn_download_url(path)
    rows = []
    r = requests.get(url, stream=True, timeout=60)
    r.raise_for_status()
    for line in r.iter_lines(decode_unicode=True):
        if not line or not line.strip():
            continue
        try:
            raw = json.loads(line)
        except Exception:
            continue
        pair = safe_project_to_pair(raw)
        if pair["prompt"] or pair["response"]:
            rows.append(pair)
    return rows

def process_file(path: str) -> List[Dict[str, str]]:
    if path.endswith(".parquet"):
        return process_parquet_file(path)
    elif path.endswith(".jsonl"):
        return process_jsonl_file(path)
    else:
        # skip unknown
        return []
