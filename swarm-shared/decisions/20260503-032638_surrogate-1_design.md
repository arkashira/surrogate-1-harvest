# surrogate-1 / design

## Final Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16`, `DATE`, `HF_TOKEN` via env
- Single `list_repo_tree(path, recursive=False)` per date folder → deterministic shard assignment via `hash(slug) % SHARD_TOTAL`
- Downloads assigned files via **HF CDN bypass** (`resolve/main/...`) with no Authorization header during data fetch
- Projects each file to `{prompt, response}` only at parse time (avoids pyarrow CastError on mixed schemas)
- Deduplicates via central `lib/dedup.py` md5 store (same as Space)
- Writes `batches/public-merged/{DATE}/shard{N}-{HHMMSS}.jsonl` with deterministic naming
- Includes retry/backoff for 429 (wait 360s) and commit-cap spreading across sibling repos if needed
- Keeps orchestration on Mac (or Actions) only; no local model loading

#!/usr/bin/env python3
"""
CDN-bypass ingestion worker for surrogate-1 public dataset.
Usage (env):
  SHARD_ID=0 SHARD_TOTAL=16 DATE=2026-04-29 HF_TOKEN=hf_xxx python bin/dataset-enrich.py
"""
import os
import sys
import json
import hashlib
import time
from datetime import datetime, timezone
from pathlib import Path

import requests
from huggingface_hub import HfApi, list_repo_tree

REPO_DATASET = "axentx/surrogate-1-training-pairs"
HF_API = HfApi(token=os.getenv("HF_TOKEN"))

SHARD_ID = int(os.getenv("SHARD_ID", "0"))
SHARD_TOTAL = int(os.getenv("SHARD_TOTAL", "16"))
DATE = os.getenv("DATE", datetime.now(timezone.utc).strftime("%Y-%m-%d"))
OUT_DIR = Path("batches/public-merged") / DATE
OUT_DIR.mkdir(parents=True, exist_ok=True)

# dedup store
sys.path.insert(0, str(Path(__file__).parent))
from lib.dedup import is_duplicate, mark_seen  # noqa: E402

HEADERS_NO_AUTH = {}
TIMEOUT = 30
MAX_RETRIES = 3
RETRY_WAIT_429 = 360

def slug_for_path(path: str) -> str:
    """Deterministic slug for shard assignment."""
    return path.strip("/").replace("/", "--")

def assign_shard(path: str) -> int:
    h = int(hashlib.sha256(path.encode()).hexdigest(), 16)
    return h % SHARD_TOTAL

def list_date_files(date: str) -> list[str]:
    """Single API call: list top-level files for date folder."""
    try:
        tree = list_repo_tree(
            repo_id=REPO_DATASET,
            path=date,
            recursive=False,
            token=os.getenv("HF_TOKEN"),
        )
        return [t.rfilename for t in tree if t.type == "file"]
    except Exception as e:
        print(f"[error] list_repo_tree failed: {e}", file=sys.stderr)
        raise

def cdn_url(path: str) -> str:
    return f"https://huggingface.co/datasets/{REPO_DATASET}/resolve/main/{path}"

def safe_get(url: str) -> bytes:
    for attempt in range(1, MAX_RETRIES + 1):
        try:
            resp = requests.get(url, headers=HEADERS_NO_AUTH, timeout=TIMEOUT)
            if resp.status_code == 429:
                print(f"[warn] 429 rate-limited, waiting {RETRY_WAIT_429}s")
                time.sleep(RETRY_WAIT_429)
                continue
            resp.raise_for_status()
            return resp.content
        except Exception as e:
            if attempt == MAX_RETRIES:
                raise
            wait = 2 ** attempt
            print(f"[warn] fetch failed ({e}), retry {attempt}/{MAX_RETRIES} in {wait}s")
            time.sleep(wait)
    raise RuntimeError("unreachable")

def project_to_pair(raw_bytes: bytes, filename: str) -> dict | None:
    """
    Minimal projection to {prompt,response}. Avoids pyarrow schema issues.
    Supports .jsonl and .parquet via hf_hub_download fallback for complex types.
    """
    name = filename.lower()
    if name.endswith(".jsonl"):
        import json as jsonlib
        lines = raw_bytes.decode().strip().splitlines()
        if not lines:
            return None
        # take first valid record as representative; caller should stream all lines in prod
        for line in lines:
            try:
                obj = jsonlib.loads(line)
                prompt = obj.get("prompt") or obj.get("input") or obj.get("text")
                response = obj.get("response") or obj.get("output") or obj.get("completion")
                if prompt is not None and response is not None:
                    return {"prompt": str(prompt), "response": str(response)}
            except Exception:
                continue
        return None

    elif name.endswith(".parquet"):
        # Avoid loading full mixed-schema parquet in worker; delegate to hf_hub_download + streaming reader
        import pyarrow.parquet as pq
        import io
        table = pq.read_table(io.BytesIO(raw_bytes))
        df = table.to_pandas()
        for col in ("prompt", "input", "text"):
            if col in df.columns:
                prompt = df[col].astype(str).iloc[0]
                break
        else:
            return None
        for col in ("response", "output", "completion"):
            if col in df.columns:
                response = df[col].astype(str).iloc[0]
                break
        else:
            return None
        return {"prompt": prompt, "response": response}

    else:
        # fallback: try to parse as utf-8 text pair heuristics
        try:
            text = raw_bytes.decode()
            # naive split; customize per corpus
            parts = text.split("\n\n", 1)
            if len(parts) == 2:
                return {"prompt": parts[0].strip(), "response": parts[1].strip()}
        except Exception:
            pass
    return None

def upload_shard(entries: list[dict]) -> None:
    ts = datetime.now(timezone.utc).strftime("%H%M%S")
    out_file = OUT_DIR / f"shard{SHARD_ID}-{ts}.jsonl"
    with out_file.open("w") as f:
        for e in entries:
            f.write(json.dumps(e, ensure_ascii=False) + "\n")
    print(f"[info] wrote {len(entries)} pairs to {out_file}")

    # push to HF dataset repo
    try:
        HF_API.upload_file(
            path_or_fileobj=str(out_file),
            path_in_repo=str(out_file.relative_to(Path.cwd())),
            repo_id=REPO_DATASET,
            repo_type="dataset",
            commit_message=f"shard{SHARD_ID} {DATE} {ts}",
        )
        print("[info] upload complete")
    except Exception as e:
        print(f"[error] upload failed: {e}", file=sys.stderr)
        raise

def main() -> None:
    print(f"[info] shard {SHARD_ID}/{SHARD_TOTAL} | date {DATE}")
    files = list_date_files(DATE)
    assigned = [f for f in files if assign_shard(f) == SHARD_ID]
    print(f"[info] assigned {len(assigned)} files")

    pairs = []
    for path in assigned:
        try:
            raw = safe_get(cdn_url(path))
            pair = project_to_pair(raw, path)
            if not pair:
                continue
            # dedup by content hash
            h = hashlib.md5(json.dumps(pair, sort_keys=True).encode()).hexdigest()
            if is_duplicate(h):
                continue
            mark_seen(h)
            pairs.append(pair)
        except Exception as e:
            print(f"[error] processing {path}: {e}", file=sys.stderr)
            continue

    if pairs:
        upload_shard(pairs)
    else:
        print("[info] no new pairs to upload")
