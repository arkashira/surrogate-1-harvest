### 1) Add `bin/list_files.py` (Mac orchestrator)
Single API call to snapshot one date folder → JSON. Run after rate-limit window clears.

```python
#!/usr/bin/env python3
"""
Snapshot public repo file list for CDN-only ingestion.
Usage:
  python bin/list_files.py --repo axentx/surrogate-1-training-pairs \
                           --path batches/public-merged/2026-05-02 \
                           --out filelist.json
"""
import argparse
import json
import os
import sys
from huggingface_hub import HfApi

def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True)
    parser.add_argument("--path", required=True, help="Folder path in repo")
    parser.add_argument("--out", required=True, help="Output JSON file")
    args = parser.parse_args()

    api = HfApi(token=os.getenv("HF_TOKEN"))
    # recursive=False to avoid 100x pagination; we snapshot one date folder
    tree = api.list_repo_tree(repo_id=args.repo, path=args.path, recursive=False)
    files = [f.path for f in tree if f.type == "file"]

    payload = {
        "repo": args.repo,
        "path": args.path,
        "files": sorted(files),
        "snapshot": True
    }

    with open(args.out, "w") as f:
        json.dump(payload, f, indent=2)
    print(f"Wrote {len(files)} files to {args.out}")

if __name__ == "__main__":
    main()
```

```bash
chmod +x bin/list_files.py
```

### 2) Update `bin/dataset-enrich.sh` to use CDN-only fetches
Replace `load_dataset(streaming=True)` with per-file CDN downloads via `hf_hub_download` (or raw CDN URL) and project to `{prompt,response}` only at parse time.

```bash
#!/usr/bin/env bash
# bin/dataset-enrich.sh
# Deterministic shard worker: uses CDN URLs only (zero HF API calls during ingest).
set -euo pipefail

REPO="${HF_REPO:-axentx/surrogate-1-training-pairs}"
DATE="${INGEST_DATE:-$(date +%Y-%m-%d)}"
SHARD_ID="${SHARD_ID:-0}"
TOTAL_SHARDS="${TOTAL_SHARDS:-16}"
OUT_DIR="batches/public-merged/${DATE}"
mkdir -p "${OUT_DIR}"

# Optional: use pre-computed filelist if present (avoids any API usage)
FILELIST="${FILELIST:-}"
if [[ -n "${FILELIST}" && -f "${FILELIST}" ]]; then
  mapfile -t ALL_FILES < <(jq -r '.files[]' "${FILELIST}")
else
  # Fallback: single non-recursive API call (one date folder only)
  mapfile -t ALL_FILES < <(python3 -c "
import os, json
from huggingface_hub import HfApi
api = HfApi(token=os.getenv('HF_TOKEN', None))
tree = api.list_repo_tree('${REPO}', 'batches/public-merged/${DATE}', recursive=False)
for f in tree:
    if f.type == 'file':
        print(f.path)
")
fi

# Deterministic shard assignment: hash(slug) % TOTAL_SHARDS
shard_files=()
for f in "${ALL_FILES[@]}"; do
  slug=$(basename "${f%.*}")
  hash=$(echo -n "${slug}" | md5sum | cut -c1-8)
  bucket=$((0x${hash} % TOTAL_SHARDS))
  if [[ "${bucket}" -eq "${SHARD_ID}" ]]; then
    shard_files+=("${f}")
  fi
done

echo "Shard ${SHARD_ID}/${TOTAL_SHARDS} processing ${#shard_files[@]} files"

# Process each file via CDN URL (no Authorization header -> bypasses /api/ rate limits)
OUTPUT="${OUT_DIR}/shard${SHARD_ID}-$(date +%H%M%S).jsonl"
> "${OUTPUT}"

for rel_path in "${shard_files[@]}"; do
  # CDN URL (public, no auth)
  CDN_URL="https://huggingface.co/datasets/${REPO}/resolve/main/${rel_path}"
  TMP_FILE=$(mktemp)
  curl -fsSL --retry 3 --retry-delay 5 -o "${TMP_FILE}" "${CDN_URL}"

  # Project to {prompt,response} only at parse time; ignore mixed schemas
  python3 -c "
import json, sys, os, pyarrow.parquet as pq, tempfile, itertools
tmp = '${TMP_FILE}'
out = '${OUTPUT}'
try:
    table = pq.read_table(tmp, columns=['prompt', 'response'])
except Exception:
    # fallback: try common column names
    try:
        table = pq.read_table(tmp)
        cols = [c for c in table.column_names if 'prompt' in c.lower() or 'instruction' in c.lower()]
        resp = [c for c in table.column_names if 'response' in c.lower() or 'output' in c.lower()]
        if not cols or not resp:
            raise
        table = table.select([cols[0], resp[0]]).rename_columns(['prompt', 'response'])
    except Exception:
        # last resort: skip file
        os.unlink(tmp)
        sys.exit(0)

df = table.to_pandas()
df = df.dropna(subset=['prompt', 'response'])
with open(out, 'a') as f:
    for _, row in df.iterrows():
        f.write(json.dumps({'prompt': str(row['prompt']), 'response': str(row['response'])}, ensure_ascii=False) + '\n')
os.unlink(tmp)
"
done

echo "Shard ${SHARD_ID} wrote ${OUTPUT}"
```
