# workio / frontend

## Final synthesized implementation (best of both proposals)

**Chosen approach**
- Keep the punch logic colocated in `PunchButton` (Candidate 1) for simplicity and direct UI coupling, but extract a small, reusable `useOptimisticPunch` hook (inspired by Candidate 2) to encapsulate idempotency, retry, and state. This gives concrete actionability without overcomplicating the component.
- Accept an `Idempotency-Key` header (or include in body) and rely on server deduplication by that key + `userId`/`tenantId`/date window.
- Use optimistic UI, an in-flight lock, and a persisted (IndexedDB) retry queue so punches survive page reloads and offline periods.
- Add clear, accessible feedback (toast + button spinner + disabled state) and lightweight exponential backoff with jitter.

### 1) Hook: `useOptimisticPunch`

File: `workio/src/hooks/useOptimisticPunch.ts`

```tsx
// workio/src/hooks/useOptimisticPunch.ts
import { useCallback, useRef, useEffect } from 'react';
import { toast } from '@/hooks/use-toast';
import { generateId } from '@/lib/id';
import { queueDB } from '@/lib/queueDB';

type PunchType = 'CLOCK_IN' | 'CLOCK_OUT';

interface PunchPayload {
  userId: string;
  tenantId: string;
  type: PunchType;
  lat?: number;
  lng?: number;
  note?: string;
  timestamp?: string; // ISO
}

interface Options {
  onPunchRequest: (payload: PunchPayload, idempotencyKey: string) => Promise<unknown>;
  onSuccess?: (result: unknown) => void;
  onError?: (err: unknown) => void;
}

export function useOptimisticPunch({ onPunchRequest, onSuccess, onError }: Options) {
  const inflightRef = useRef(false);

  // Exponential backoff with jitter
  const sleep = (ms: number) => new Promise((r) => setTimeout(r, ms));
  const attemptWithBackoff = useCallback(
    async (payload: PunchPayload, idempotencyKey: string, signal?: AbortSignal) => {
      const maxRetries = 3;
      let attempt = 0;
      const baseDelay = 1000;

      while (attempt <= maxRetries) {
        if (signal?.aborted) throw new Error('aborted');
        try {
          const result = await onPunchRequest(payload, idempotencyKey);
          return { result, ok: true };
        } catch (err: any) {
          attempt++;
          if (attempt > maxRetries) throw err;
          const jitter = Math.random() * 500;
          await sleep(baseDelay * 2 ** (attempt - 1) + jitter);
        }
      }
      throw new Error('max retries exceeded');
    },
    [onPunchRequest]
  );

  const punch = useCallback(
    async (payload: PunchPayload) => {
      if (inflightRef.current) return;
      const idempotencyKey = generateId();
      inflightRef.current = true;

      try {
        // Optimistic local DB write for immediate UI consistency (optional)
        await queueDB.add({ payload, idempotencyKey, status: 'pending', createdAt: Date.now() });

        const { result } = await attemptWithBackoff(payload, idempotencyKey);

        await queueDB.updateStatus(idempotencyKey, 'success');
        onSuccess?.(result);

        toast({
          title: payload.type === 'CLOCK_IN' ? 'เข้างานแล้ว' : 'ออกงานแล้ว',
          description: new Date().toLocaleTimeString(),
          variant: 'default',
        });
        return result;
      } catch (err: any) {
        await queueDB.updateStatus(idempotencyKey, 'failed');
        onError?.(err);

        toast({
          title: 'ไม่สามารถบันทึกได้',
          description: 'ระบบจะลองใหม่โดยอัตโนมัติ',
          variant: 'destructive',
        });
        throw err;
      } finally {
        inflightRef.current = false;
      }
    },
    [attemptWithBackoff, onSuccess, onError, toast]
  );

  // Background retry processor for failed queue items
  useEffect(() => {
    let stopped = false;
    const process = async () => {
      if (stopped) return;
      const items = await queueDB.getFailedOrPending();
      for (const item of items) {
        if (stopped) return;
        if (inflightRef.current) break; // serialize with active punches
        try {
          await attemptWithBackoff(item.payload, item.idempotencyKey);
          await queueDB.updateStatus(item.idempotencyKey, 'success');
        } catch {
          // remain in queue; next interval will retry
        }
      }
    };

    const interval = setInterval(process, 8000);
    process(); // immediate attempt on mount
    return () => {
      stopped = true;
      clearInterval(interval);
    };
  }, [attemptWithBackoff]);

  return { punch, inflight: inflightRef.current };
}
```

### 2) IndexedDB queue helper (lightweight)

File: `workio/src/lib/queueDB.ts`

```ts
// workio/src/lib/queueDB.ts
const DB_NAME = 'PunchQueueDB';
const STORE_NAME = 'punches';
const DB_VERSION = 1;

function openDB(): Promise<IDBDatabase> {
  return new Promise((resolve, reject) => {
    const req = indexedDB.open(DB_NAME, DB_VERSION);
    req.onupgradeneeded = () => {
      const db = req.result;
      if (!db.objectStoreNames.contains(STORE_NAME)) {
        const store = db.createObjectStore(STORE_NAME, { keyPath: 'idempotencyKey' });
        store.createIndex('status', 'status', { unique: false });
        store.createIndex('createdAt', 'createdAt', { unique: false });
      }
    };
    req.onsuccess = () => resolve(req.result);
    req.onerror = () => reject(req.error);
  });
}

function withStore<T>(mode: IDBTransactionMode, fn: (store: IDBObjectStore) => Promise<T>): Promise<T> {
  return openDB().then((db) => {
    const tx = db.transaction(STORE_NAME, mode);
    const store = tx.objectStore(STORE_NAME);
    return fn(store).then((res) => {
      return new Promise<T>((resolve, reject) => {
        tx.oncomplete = () => resolve(res);
        tx.onerror = () => reject(tx.error);
      });
    });
  });
}

export const queueDB = {
  async add(item: { payload: any; idempotencyKey: string; status: 'pending' | 'success' | 'failed'; createdAt: number }) {
    return withStore('readwrite', (store) => store.put(item));
  },
  async updateStatus(idempotencyKey: string, status: 'success' | 'failed') {
    return withStore('readwrite', async (store) => {
      const existing = await new Promise<any>((resolve) => {
        const req = store.get(idempotencyKey);
        req.onsuccess = () => resolve(req.result);
      });
      if (existing) {
        existing.status = status;
        store.put(existing);
      }
    });
  },
  async getFailedOrPending(): Promise<Array<{ payload: any; idempotencyKey: string; status: string; createdAt: number }>> {
    return withStore('readonly', (store) => {
      return new Promise((resolve) => {
        const index = store.index('status');
        const req = index.getAll(IDBKeyRange.bound('pending', 'failed'));
        req.onsuccess = () => resolve(req.result || []);
      });
    });
  },
};
```
