# surrogate-1 / design

## Final Implementation (merged & hardened)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16`, `DATE`, `HF_TOKEN`, `REPO_ID` (default: `axentx/surrogate-1-training-pairs`)
- Single API call (or cached manifest) to list one date folder → deterministic shard assignment by `hash(slug) % SHARD_TOTAL`
- Downloads only assigned files via HF CDN (`resolve/main/...`) **without Authorization** → bypasses `/api/` rate limits
- Projects each file to `{prompt, response}` at parse time (avoids pyarrow `CastError` on mixed schemas)
- Dedups via central md5 store (`lib/dedup.py`) and streams output to `batches/public-merged/{DATE}/shard{N}-{HHMMSS}.jsonl`
- Commits via HF API (respects 128/hr/repo cap; one commit per run)
- Exits 0 on success, non-zero on fatal error; logs JSON for Actions

### Why this ships in <2h
- Reuses existing patterns: HF CDN bypass, schema projection, sharding, dedup
- One small, focused Python script + minimal workflow tweak
- No infra changes, no new deps beyond current `requirements.txt`

#!/usr/bin/env python3
"""
surrogate-1 CDN-bypass ingestion worker.

Usage (local/test):
  SHARD_ID=0 SHARD_TOTAL=16 DATE=2026-05-03 \
    HF_TOKEN=hf_xxx REPO_ID=axentx/surrogate-1-training-pairs \
    python bin/dataset-enrich.py

GitHub Actions matrix:
  strategy:
    matrix:
      shard: [0,1,2,...,15]
  env:
    SHARD_ID: ${{ matrix.shard }}
    SHARD_TOTAL: 16
    DATE: ${{ github.run_number }}-$(date +%Y-%m-%d)
    HF_TOKEN: ${{ secrets.HF_TOKEN }}
    REPO_ID: axentx/surrogate-1-training-pairs
"""

import json
import logging
import os
import sys
from datetime import datetime
from pathlib import Path
from typing import Dict, List, Optional

import pandas as pd
import requests
from huggingface_hub import HfApi, list_repo_tree

# ---- config ----
REPO_ID = os.getenv("REPO_ID", "axentx/surrogate-1-training-pairs")
HF_TOKEN = os.getenv("HF_TOKEN")
SHARD_ID = int(os.getenv("SHARD_ID", "0"))
SHARD_TOTAL = int(os.getenv("SHARD_TOTAL", "16"))
DATE = os.getenv("DATE", datetime.utcnow().strftime("%Y-%m-%d"))
WORKDIR = Path(os.getenv("WORKDIR", "."))
API = HfApi(token=HF_TOKEN)

# ---- logging ----
logging.basicConfig(
    level=logging.INFO,
    format='{"time":"%(asctime)s","level":"%(levelname)s","msg":"%(message)s"}',
    handlers=[logging.StreamHandler(sys.stdout)],
)
log = logging.getLogger("surrogate-ingest")

# ---- dedup ----
sys.path.insert(0, str(WORKDIR / "lib"))
try:
    from dedup import DedupStore
except Exception as e:
    log.warning(f"dedup import failed ({e}); falling back to no-op dedup")

    class DedupStore:
        def __init__(self, *a, **kw):
            pass

        def exists(self, h: str) -> bool:
            return False

        def mark(self, h: str) -> bool:
            return True


# ---- helpers ----
def deterministic_shard(slug: str) -> int:
    return int(hashlib.sha256(slug.encode()).hexdigest(), 16) % SHARD_TOTAL


def list_date_files() -> List[str]:
    """
    Single API call: list files in one date folder (non-recursive).
    Returns relative paths.
    """
    folder = f"batches/public-raw/{DATE}"
    try:
        tree = list_repo_tree(repo_id=REPO_ID, path=folder, recursive=False, token=HF_TOKEN)
    except Exception as e:
        log.warning(f"list_repo_tree failed for {folder}: {e}")
        return []
    items = [t.r_path for t in tree if not t.type == "tree"]
    log.info(f"listed {len(items)} files in {folder}")
    return items


def cdn_url(path: str) -> str:
    return f"https://huggingface.co/datasets/{REPO_ID}/resolve/main/{path}"


def safe_download(path: str, dest: Path) -> bool:
    url = cdn_url(path)
    try:
        r = requests.get(url, timeout=60, stream=True)
        r.raise_for_status()
        with open(dest, "wb") as f:
            for chunk in r.iter_content(chunk_size=8192):
                f.write(chunk)
        return True
    except Exception as e:
        log.error(f"download failed {url}: {e}")
        return False


def project_pairs(local_path: Path) -> List[Dict[str, str]]:
    """
    Project heterogeneous files to [{prompt, response}] only.
    Supports .jsonl and .parquet.
    """
    suffix = local_path.suffix.lower()
    try:
        if suffix == ".parquet":
            df = pd.read_parquet(local_path)
        elif suffix == ".jsonl":
            df = pd.read_json(local_path, lines=True)
        else:
            log.warning(f"unsupported suffix {suffix} for {local_path}")
            return []
    except Exception as e:
        log.error(f"parse failed {local_path}: {e}")
        return []

    col_map = {}
    for c in df.columns:
        cl = c.strip().lower()
        if cl in ("prompt", "question", "input", "instruction"):
            col_map[c] = "prompt"
        elif cl in ("response", "answer", "output", "completion"):
            col_map[c] = "response"

    if "prompt" not in col_map.values() or "response" not in col_map.values():
        log.warning(f"missing prompt/response cols in {local_path}")
        return []

    rename = {old: new for old, new in col_map.items() if old != new}
    if rename:
        df = df.rename(columns=rename)

    out = []
    for _, row in df[["prompt", "response"]].iterrows():
        prompt = str(row["prompt"]).strip()
        response = str(row["response"]).strip()
        if prompt and response:
            out.append({"prompt": prompt, "response": response})
    return out


def upload_to_hf(local_file: Path, repo_path: str) -> bool:
    try:
        API.upload_file(
            path_or_fileobj=str(local_file),
            path_in_repo=repo_path,
            repo_id=REPO_ID,
            repo_type="dataset",
            token=HF_TOKEN,
        )
        return True
    except Exception as e:
        log.error(f"HF upload failed {repo_path}: {e}")
        return False


# ---- main ----
def main() -> int:
    if not HF_TOKEN:
        log.error("HF_TOKEN is required")
        return 1

    if SHARD_ID < 0 or SHARD_ID >= SHARD_TOTAL:
        log.error(f"invalid SHARD_ID={SHARD_ID} (SHARD_TOTAL={SHARD_TOTAL})")
        return 1

    raw_files = list_date_files()
    if not raw_files:
        log.warning("no files to process")
        return 0

    my_files = [p for p in raw_files if deterministic_shard(p) == SHARD_ID]
    log.info(f"shard {SHARD_ID}/{SHARD_TOTAL} assigned {len(my_files)} files")

    ts = datetime.utcnow().strftime("%Y%m%d-%H%M%S")
    out_dir = WORKDIR / "batches" / "public-merged" / DATE
    out_dir.mkdir(parents=True, exist_ok=True)
    out_file = out_dir
