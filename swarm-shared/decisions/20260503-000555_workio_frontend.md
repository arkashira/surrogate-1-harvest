# workio / frontend

## 1. Diagnosis
- No client-side idempotency key on punch submissions → double-taps / LINE webhook replays create duplicate punches.
- Clock-in/out button has no optimistic state or debounce → users can spam and queue multiple requests.
- No local validation of valid state transitions (e.g., allow clock-in when already clocked-in) → UI can propose invalid actions and confuse users.
- Missing visual feedback for in-flight requests (loading, success, error) → poor UX and retries.
- No persisted last-punch cache to detect duplicates locally before hitting API (reduces noise and latency).

## 2. Proposed change
**Scope:** `workio/src/features/punch/PunchButton.tsx` (new) + `workio/src/features/punch/usePunch.ts` (new) + minor updates to `workio/src/pages/Dashboard.tsx` (or wherever clock-in/out UI lives).  
**Goal:** Add idempotent punch submission with client-side state-machine guard, optimistic UI, and duplicate detection — <2h incremental frontend-only change.

## 3. Implementation

### 3.1 Add punch state-machine + idempotency hook
`workio/src/features/punch/usePunch.ts`
```ts
import { useState, useCallback, useRef } from 'react';

export type PunchState = 'IDLE' | 'CLOCKED_IN' | 'CLOCKED_OUT';

type UsePunchOpts = {
  tenantId: string;
  userId: string;
  onSuccess?: (punch: any) => void;
  onError?: (err: Error) => void;
};

export function usePunch({ tenantId, userId, onSuccess, onError }: UsePunchOpts) {
  const [state, setState] = useState<PunchState>('IDLE');
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<Error | null>(null);
  const inflightKeyRef = useRef<string | null>(null);
  const lastPunchRef = useRef<{ ts: number; type: 'IN' | 'OUT' } | null>(null);

  const fetchState = useCallback(async () => {
    try {
      const res = await fetch(`/api/punch/latest?tenantId=${tenantId}&userId=${userId}`);
      if (!res.ok) return;
      const data = await res.json();
      if (data?.type === 'IN') setState('CLOCKED_IN');
      else if (data?.type === 'OUT') setState('CLOCKED_OUT');
      lastPunchRef.current = data ? { ts: new Date(data.ts).getTime(), type: data.type } : null;
    } catch {
      // best-effort
    }
  }, [tenantId, userId]);

  const submitPunch = useCallback(
    async (type: 'IN' | 'OUT') => {
      // Guard invalid transitions
      if (type === 'IN' && state === 'CLOCKED_IN') {
        setError(new Error('Already clocked in'));
        return;
      }
      if (type === 'OUT' && state === 'CLOCKED_OUT') {
        setError(new Error('Already clocked out'));
        return;
      }

      // Idempotency: prevent double-submit within same minute for same user/tenant
      const now = Date.now();
      const minuteKey = `${tenantId}:${userId}:${new Date().toISOString().slice(0, 16)}`; // YYYY-MM-DDTHH:MM
      if (inflightKeyRef.current === minuteKey) {
        setError(new Error('Request already in flight'));
        return;
      }
      // Local duplicate detection: if last punch same type within N minutes, warn (optional block)
      const DUPE_WINDOW_MS = 5 * 60 * 1000;
      if (
        lastPunchRef.current &&
        lastPunchRef.current.type === type &&
        now - lastPunchRef.current.ts < DUPE_WINDOW_MS
      ) {
        setError(new Error('Recent duplicate detected'));
        return;
      }

      setLoading(true);
      setError(null);
      inflightKeyRef.current = minuteKey;

      const idempotencyKey = `${minuteKey}:${Math.random().toString(36).slice(2, 10)}`;

      try {
        const res = await fetch('/api/punch', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            tenantId,
            userId,
            type,
            idempotencyKey,
            ts: new Date().toISOString(),
          }),
        });

        if (!res.ok) throw new Error(`Server error: ${res.status}`);
        const data = await res.json();
        setState(type === 'IN' ? 'CLOCKED_IN' : 'CLOCKED_OUT');
        lastPunchRef.current = { ts: now, type };
        onSuccess?.(data);
      } catch (err: any) {
        setError(err);
        onError?.(err);
      } finally {
        setLoading(false);
        inflightKeyRef.current = null;
      }
    },
    [state, tenantId, userId, onSuccess, onError]
  );

  return {
    state,
    loading,
    error,
    submitPunch,
    fetchState,
  };
}
```

### 3.2 Punch button UI with optimistic state
`workio/src/features/punch/PunchButton.tsx`
```tsx
import React from 'react';
import { usePunch, PunchState } from './usePunch';

interface Props {
  tenantId: string;
  userId: string;
}

export function PunchButton({ tenantId, userId }: Props) {
  const { state, loading, error, submitPunch } = usePunch({
    tenantId,
    userId,
    onSuccess: () => {
      // optional toast/notification
    },
    onError: () => {
      // optional toast
    },
  });

  const isClocked = state === 'CLOCKED_IN';
  const nextType = isClocked ? 'OUT' : 'IN';
  const label = isClocked ? 'Clock Out' : 'Clock In';
  const sublabel = isClocked ? 'Tap to end shift' : 'Tap to start shift';

  return (
    <div className="flex flex-col items-center gap-2">
      <button
        onClick={() => submitPunch(nextType)}
        disabled={loading}
        className={`
          px-8 py-4 rounded-xl font-semibold text-white shadow-lg transition-all
          ${loading ? 'opacity-60 cursor-wait' : ''}
          ${
            isClocked
              ? 'bg-red-500 hover:bg-red-600'
              : 'bg-green-600 hover:bg-green-700'
          }
        `}
      >
        {loading ? (
          <span className="flex items-center gap-2">
            <svg className="animate-spin h-5 w-5 text-white" xmlns="http://www.w3.org/2000/svg" fill="none" viewBox="0 0 24 24">
              <circle className="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" strokeWidth="4"></circle>
              <path className="opacity-75" fill="currentColor" d="M4 12a8 8 0 018-8v4a4 4 0 00-4 4H4z"></path>
            </svg>
            Processing...
          </span>
        ) : (
          <>
            <div className="text-xl">{label}</div>
            <div className="text-sm opacity-90">{sublabel}</div>
          </>
        )}
      </button>

      {error && (
        <div className="text-sm text-red-600 mt-1">{error.message}</div>
      )}

      <div className="text-xs text-gray-500 mt-1">
        State: {state}
      </div>
    </div>
  );
}
```
