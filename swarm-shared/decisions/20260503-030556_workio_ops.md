### 1. Diagnosis (consolidated)
- **Replay/forgery**: missing `X-Line-Signature` verification allows spoofed or retried LINE webhooks.  
- **Duplicate processing**: no idempotency layer + LINE retries on 5xx/timeouts create duplicate punches.  
- **Race condition**: concurrent requests for same `(tenant_id, employee_id)` can create multiple open punches (`clock_out_at IS NULL`).  
- **Missing guardrails**: no unique constraint/index for open punches; no audit trail for debugging.

### 3. Signature verification middleware

`server/src/middleware/lineWebhook.ts`

```ts
import crypto from 'crypto';

const LINE_CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';

export function verifyLineSignature(rawBody: string, signature: string): boolean {
  if (!LINE_CHANNEL_SECRET) {
    // Fail closed in prod; allow override only via explicit env
    if (process.env.NODE_ENV === 'production') return false;
    return true;
  }
  const expected = crypto
    .createHmac('sha256', LINE_CHANNEL_SECRET)
    .update(rawBody)
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

export function lineWebhookMiddleware(req, res, next) {
  const signature = req.headers['x-line-signature'] as string;
  const rawBody = req.rawBody || JSON.stringify(req.body);

  if (!signature || !verifyLineSignature(rawBody, signature)) {
    return res.status(401).json({ error: 'Invalid LINE signature' });
  }
  next();
}
```

**Note**: ensure your framework preserves raw body (e.g., `express.raw` for this route) before JSON parsing.

### 4. Idempotency + race-free punch mutation

`server/src/services/punchService.ts`

```ts
import { db } from '../db';

export async function processPunchEvent(
  idempotencyKey: string,
  tenantId: string,
  employeeId: string,
  eventType: 'clock_in' | 'clock_out'
) {
  return await db.transaction(async (tx) => {
    // Idempotency check
    const existing = await tx`line_webhook_idempotency`
      .where({ idempotency_key: idempotencyKey })
      .first();
    if (existing) {
      const punch = await tx`punches`
        .where({ id: existing.punch_id })
        .first();
      return { punch, fromCache: true };
    }

    // Serialize concurrent operations for this employee
    // Use SELECT FOR UPDATE to lock existing rows
    const openPunch = await tx`punches`
      .where({ tenant_id: tenantId, employee_id: employeeId, clock_out_at: null })
      .forUpdate()
      .first();

    let punch;
    if (eventType === 'clock_in') {
      if (openPunch) {
        // Already clocked in — treat as no-op or update depending on policy
        punch = openPunch;
      } else {
        [punch] = await tx`punches`
          .insert({
            tenant_id: tenantId,
            employee_id: employeeId,
            clock_in_at: new Date(),
            clock_out_at: null,
          })
          .returning('*');
      }
    } else {
      // clock_out
      if (!openPunch) {
        throw new Error('No open punch to clock out');
      }
      [punch] = await tx`punches`
        .where({ id: openPunch.id })
        .update({ clock_out_at: new Date() })
        .returning('*');
    }

    // Record idempotency
    await tx`line_webhook_idempotency`
      .insert({
        idempotency_key: idempotencyKey,
        tenant_id: tenantId,
        employee_id: employeeId,
        event_type: eventType,
        punch_id: punch.id,
      });

    // Optional audit log
    await tx`line_webhook_logs`
      .insert({
        tenant_id: tenantId,
        employee_id: employeeId,
        event_type: eventType,
        line_signature: '', // filled upstream if available
        body_hash: '', // optional
        body: JSON.stringify({ eventType, employeeId, tenantId }),
      });

    return { punch, fromCache: false };
  });
}
```

### 5. Webhook handler

`server/src/routes/webhook/line.ts`

```ts
import express from 'express';
import { lineWebhookMiddleware } from '../middleware/lineWebhook';
import { processPunchEvent } from '../services/punchService';

const router = express.Router();

// Preserve raw body for signature verification
router.use('/webhook/line', express.raw({ type: 'application/json' }));

router.post('/webhook/line', lineWebhookMiddleware, async (req, res) => {
  try {
    const body = JSON.parse(req.body.toString());
    const events = body.events || [];
    if (!events.length) return res.status(400).json({ error: 'No events' });

    // Use deterministic idempotency key:
    // Prefer client-provided X-Idempotency-Key, else derive from LINE signature + timestamp window
    const clientKey = req.headers['x-idempotency-key'] as string;
    const signature = req.headers['x-line-signature'] as string;
    const idempotencyKey = clientKey || `${signature}:${Math.floor(Date.now() / 1000 / 60)}`; // 1-min window fallback

    const results = [];
    for (const ev of events) {
      const source = ev.source || {};
      const tenantId = source.tenantId;
      const employeeId = source.userId || source.employeeId;
      const eventType = ev.type; // e.g., 'clock_in', 'clock_out'

      if (!tenantId || !employeeId || !['clock_in', 'clock_out'].includes(eventType)) {
        continue; // or return 400
      }

      const result = await processPunchEvent(idempotencyKey, tenantId, employeeId, eventType);
      results.push(result);
    }

    res.json({ ok: true, results });
  } catch (err: any) {
    if (err.message?.includes('No open punch')) {
      return res.status(409).json({ error: err.message });
    }
    console.error(err);
    res.status(500).json({ error: 'Internal server error' });
  }
});

export default router;
```
