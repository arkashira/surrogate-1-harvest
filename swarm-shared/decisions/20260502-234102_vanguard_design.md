# vanguard / design

## Final Synthesis (Best Parts + Contradictions Resolved)

**Core diagnosis (merged):**
- No persistent file-list cache → repeated HF API calls → 429 (1000 req/5 min).
- Lightning Studio lifecycle not reused → quota burned on recreation; idle-stop kills state and forces full restart.
- Training script re-fetches file lists/data each run instead of embedding a deterministic CDN-only fetch plan.
- Schema leakage (attribution cols, timestamps) risks CastError in surrogate training.
- No deterministic shard-to-repo mapping for HF commit cap (128/hr/repo) → ingestion stalls when sibling repos saturate.
- Design artifacts (file lists, studio configs, shard maps) are ephemeral → no handoff between orchestration (Mac) and remote compute (Lightning/HF).

**Chosen approach:**
- Build a minimal, deterministic cache layer (CDN-only URLs) + deterministic repo hashing + Lightning Studio reuse with idle-safe `.run()`.
- Enforce schema hygiene at cache time (drop non-feature columns) and embed the manifest so training is zero-HF-API.
- Keep orchestration Mac-side; remote training uses the embedded manifest (no `list_repo_tree`, no `load_dataset` discovery calls).

# /opt/axentx/vanguard/design/file_list_cache.py
import json
import hashlib
import os
from datetime import datetime, timezone
from pathlib import Path
from typing import Dict, List

import requests

HF_DATASET_OWNER = "axentx"
HF_DATASET_NAME = "vanguard-mirror"
CACHE_DIR = Path(__file__).parent / ".cache"
CACHE_DIR.mkdir(exist_ok=True)

# Columns allowed through to training (prevent CastError from attribution/timestamp leakage)
ALLOWED_COLS = {"path", "cdn_url", "size", "split", "modality"}

def _cache_path(date_slug: str) -> Path:
    return CACHE_DIR / f"filelist_{date_slug}.json"

def _cdn_url_for(path: str) -> str:
    return (
        f"https://huggingface.co/datasets/{HF_DATASET_OWNER}/{HF_DATASET_NAME}"
        f"/resolve/main/{path}"
    )

def list_date_folder(date_slug: str, folder_path: str = "") -> List[Dict]:
    """
    Single non-recursive HF tree call for one date folder, cached to disk.
    Returns list of dicts with at minimum: {"path": "...", "cdn_url": "...", "size": ...}
    """
    cache = _cache_path(date_slug)
    if cache.exists():
        with open(cache) as f:
            return json.load(f)

    token = os.getenv("HF_TOKEN")
    headers = {"Authorization": f"Bearer {token}"} if token else {}
    url = f"https://huggingface.co/api/datasets/{HF_DATASET_OWNER}/{HF_DATASET_NAME}/tree/{date_slug}"
    params = {"recursive": "false"}
    if folder_path:
        params["path"] = folder_path

    resp = requests.get(url, headers=headers, params=params, timeout=30)
    if resp.status_code == 429:
        raise RuntimeError("HF API rate-limited; wait 360s")
    resp.raise_for_status()
    items = resp.json()

    out = []
    for item in items:
        if item.get("type") != "file":
            continue
        record = {
            "path": item["path"],
            "cdn_url": _cdn_url_for(item["path"]),
            "size": item.get("size", 0),
            "split": item.get("split", "unknown"),
            "modality": item.get("modality", "unknown"),
        }
        # Enforce schema hygiene: drop unexpected fields that could cause CastError downstream
        record = {k: v for k, v in record.items() if k in ALLOWED_COLS}
        out.append(record)

    with open(cache, "w") as f:
        json.dump(out, f, indent=2)
    return out

def pick_shard_repo(slug: str, n_siblings: int = 5) -> str:
    """
    Deterministic repo assignment to spread HF commit cap (128/hr/repo).
    Returns repo string like:
      axentx/vanguard-mirror
      axentx/vanguard-mirror-shard1
      ...
    """
    digest = hashlib.sha256(slug.encode()).digest()
    idx = int.from_bytes(digest[:2], "big") % n_siblings
    if idx == 0:
        return f"{HF_DATASET_OWNER}/{HF_DATASET_NAME}"
    return f"{HF_DATASET_OWNER}/{HF_DATASET_NAME}-shard{idx}"

def build_cdn_manifest(date_slug: str, folder_path: str = "") -> Dict:
    files = list_date_folder(date_slug, folder_path)
    return {
        "date_slug": date_slug,
        "generated_at_utc": datetime.now(timezone.utc).isoformat(),
        "shard_repo": pick_shard_repo(date_slug),
        "files": files,
        "total_files": len(files),
    }

# /opt/axentx/vanguard/design/studio_lifecycle.py
import time
from pathlib import Path
from typing import Optional

from lightning import LightningApp, LightningWork, Machine

L40S = Machine(name="L40S", count=1, disk_size=500)

class StudioReuser:
    """
    Reuse a running Lightning Studio instead of recreating.
    Idle-safe: checks status before .run() and restarts if stopped.
    """
    def __init__(self, studio_name: str):
        self.studio_name = studio_name

    def get_running(self) -> Optional[LightningWork]:
        # Teamspace.studios available in Lightning runtime contexts
        for s in getattr(self, "_studios", []):
            if s.name == self.studio_name and getattr(s, "status", None) == "running":
                return s
        return None

    def _list_studios(self):
        # Best-effort: in many Lightning environments Teamspace.studios exists.
        # If not, subclasses or runtime injection can populate self._studios.
        try:
            from lightning.app import Teamspace
            self._studios = list(Teamspace.studios)
        except Exception:
            self._studios = []

    def ensure_running(self, target: LightningWork) -> LightningWork:
        self._list_studios()
        existing = self.get_running()
        if existing:
            return existing

        for s in self._studios:
            if s.name == self.studio_name:
                if getattr(s, "status", None) == "stopped":
                    s.start(machine=L40S)
                    return s
                # terminal states -> create fresh
                break

        # Create new
        target.name = self.studio_name
        # Note: LightningApp registration typically happens via Lightning runtime.
        # We return target so caller can invoke run() in the active context.
        return target

    def run_with_retry(self, target: LightningWork, max_retries: int = 3, delay: int = 10):
        for attempt in range(1, max_retries + 1):
            try:
                studio = self.ensure_running(target)
                # studio.run() may fail if idle-stop killed the process; retry.
                studio.run()
                return
            except Exception as exc:
                if attempt == max_retries:
                    raise
                time.sleep(delay * attempt)
