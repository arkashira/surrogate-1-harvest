#### 3.2 Idempotency middleware (lightweight, safe)

```ts
// workio/server/src/middleware/idempotency.ts
import { Request, Response, NextFunction } from 'express';
import { db } from '../db';

export function lineWebhookIdempotency(
  getIdempotencyKey: (req: Request) => string | undefined
) {
  return async (req: Request, res: Response, next: NextFunction) => {
    const key = getIdempotencyKey(req);
    const tenantId = req.body.tenantId || (req as any).tenantId;
    if (!key || !tenantId) return next();

    // Fast DB check (indexed). Could add in-memory LRU for hot keys if needed.
    const exists = await db.oneOrNone(
      `SELECT 1 FROM punches WHERE webhook_event_id = $1 AND tenant_id = $2`,
      [key, tenantId]
    );
    if (exists) {
      (req as any).idempotentReplay = true;
      return res.status(200).json({ ok: true, reason: 'duplicate' });
    }
    (req as any).idempotencyKey = key;
    next();
  };
}
```

#### 3.3 LINE webhook handler (atomic upsert + fast ack)

```ts
// workio/server/src/routes/line/webhook.ts
import { Router, Request, Response } from 'express';
import { db } from '../../db';
import { lineWebhookIdempotency } from '../../middleware/idempotency';

const router = Router();

function extractWebhookEventId(req: Request): string | undefined {
  const events = req.body.events || req.body;
  if (Array.isArray(events) && events[0]?.webhookEventId) return events[0].webhookEventId;
  if (events?.webhookEventId) return events.webhookEventId;
  return undefined;
}

router.post(
  '/line/webhook',
  lineWebhookIdempotency(extractWebhookEventId),
  async (req: Request, res: Response) => {
    if ((req as any).idempotentReplay) return;

    const events = req.body.events || req.body;
    const event = Array.isArray(events) ? events[0] : events;
    if (!event) return res.status(400).json({ error: 'no_event' });

    const tenantId = req.body.tenantId || (req as any).tenantId;
    const userId = event.source?.userId;
    const type = event.type; // 'clock_in' | 'clock_out'
    const webhookEventId = event.webhookEventId;
    const ts = event.timestamp ? new Date(event.timestamp) : new Date();
    const location = event.location; // { latitude, longitude }

    try {
      // Atomic upsert using webhook_event_id uniqueness
      await db.none(
        `INSERT INTO punches (tenant_id, user_id, type, punched_at, latitude, longitude, webhook_event_id, created_at)
         VALUES ($1, $2, $3, $4, $5, $6, $7, NOW())
         ON CONFLICT (tenant_id, webhook_event_id) DO NOTHING`,
        [tenantId, userId, type, ts, location?.latitude, location?.longitude, webhookEventId]
      );

      // Fast ack to prevent LINE retries
      res.status(200).json({ ok: true });
    } catch (err) {
      console.error('Webhook processing error', err);
      // Return 200 to prevent retry storms; log for investigation
      res.status(200).json({ ok: true, warning: 'processing_error_but_ack_to_prevent_retry' });
    }
  }
);

export default router;
```
