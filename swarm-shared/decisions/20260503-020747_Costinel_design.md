# Costinel / design

## Implementation Plan — Top-Hub Signal Panel (Costinel Dashboard)

**Scope & Value**  
- Frontend-only, read-only React panel that surfaces the highest-signal / most-connected hub (default “MOC”) and its actionable proposals from the knowledge graph.  
- CDN-first data: a static `top-hub.json` (published by knowledge-rag) is fetched from CDN (`/data/top-hub.json`) to avoid backend/API rate limits.  
- Zero-runtime dependencies beyond existing React stack; ships in <2h.

### 2) Component: `TopHubSignalPanel`

Add to `src/components/dashboard/TopHubSignalPanel.tsx`:

```tsx
import React, { useEffect, useState } from "react";
import { Card, CardHeader, CardTitle, CardContent } from "@/components/ui/card";
import { Badge } from "@/components/ui/badge";
import { Button } from "@/components/ui/button";
import { ExternalLink, TrendingUp, Tag } from "lucide-react";

interface Impact {
  annualSavingsUSD?: number;
  monthlySavingsUSD?: number;
  confidence: number;
}

interface Proposal {
  id: string;
  title: string;
  type: string;
  impact: Impact;
  actions: Array<{ label: string; href: string }>;
}

interface TopHubPayload {
  hub: string;
  title: string;
  score: number;
  updatedAt: string;
  proposals: Proposal[];
}

const formatUSD = (n: number) =>
  new Intl.NumberFormat("en-US", { style: "currency", currency: "USD" }).format(n);

export const TopHubSignalPanel: React.FC = () => {
  const [data, setData] = useState<TopHubPayload | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    // CDN-first fetch (no auth, bypasses API rate limits)
    fetch("/data/top-hub.json", { cache: "no-store" })
      .then((r) => {
        if (!r.ok) throw new Error(`HTTP ${r.status}`);
        return r.json();
      })
      .then((json) => {
        setData(json);
        setLoading(false);
      })
      .catch((err) => {
        console.error("TopHub fetch failed:", err);
        setError("Unable to load top-hub signals");
        setLoading(false);
      });
  }, []);

  if (loading) {
    return (
      <Card>
        <CardContent className="p-6">
          <div className="animate-pulse space-y-3">
            <div className="h-5 w-32 bg-muted rounded" />
            <div className="h-4 w-full bg-muted rounded" />
            <div className="h-4 w-5/6 bg-muted rounded" />
          </div>
        </CardContent>
      </Card>
    );
  }

  if (error || !data) {
    return (
      <Card>
        <CardContent className="p-6 text-sm text-muted-foreground">
          {error || "No signal data available"}
        </CardContent>
      </Card>
    );
  }

  const topProposal = data.proposals?.[0];
  const savings =
    topProposal?.impact.annualSavingsUSD ?? topProposal?.impact.monthlySavingsUSD ?? null;
  const isAnnual = topProposal?.impact.annualSavingsUSD !== undefined;

  return (
    <Card className="relative overflow-hidden">
      <div className="absolute right-4 top-4">
        <Badge variant="secondary" className="gap-1">
          <TrendingUp className="h-3 w-3" />
          {data.hub}
        </Badge>
      </div>

      <CardHeader className="pb-3">
        <CardTitle className="text-base font-semibold flex items-center gap-2">
          <Tag className="h-4 w-4 text-muted-foreground" />
          Top-hub signal: {data.title}
        </CardTitle>
        <p className="text-xs text-muted-foreground">
          Updated {new Date(data.updatedAt).toLocaleDateString(undefined, {
            month: "short",
            day: "numeric",
            hour: "2-digit",
            minute: "2-digit",
          })}
        </p>
      </CardHeader>

      <CardContent className="space-y-4">
        {topProposal && (
          <div className="border rounded-lg p-3 bg-muted/30">
            <div className="flex items-start justify-between gap-2">
              <div className="min-w-0 flex-1">
                <p className="font-medium text-sm truncate">{topProposal.title}</p>
                <p className="text-xs text-muted-foreground capitalize">{topProposal.type}</p>
                {savings !== null && (
                  <p className="text-xs font-semibold text-emerald-600 dark:text-emerald-400 mt-1">
                    {isAnnual ? "Annual" : "Monthly"} savings: {formatUSD(savings)}
                  </p>
                )}
              </div>
              <Button size="sm" variant="ghost" asChild>
                <a href={topProposal.actions[0]?.href} target="_blank" rel="noopener noreferrer">
                  Review
                  <ExternalLink className="ml-1 h-3 w-3" />
                </a>
              </Button>
            </div>
          </div>
        )}

        <div className="space-y-2">
          {data.proposals.slice(1, 3).map((p) => (
            <div key={p.id} className="flex items-center justify-between text-sm">
              <span className="truncate text-muted-foreground">{p.title}</span>
              <Button variant="link" size="sm" asChild>
                <a href={p.actions[0]?.href} target="_blank" rel="noopener noreferrer">
                  View
                </a>
              </Button>
            </div>
          ))}
        </div>

        <div className="pt-2">
          <Button variant="outline" size="sm" className="w-full" asChild>
            <a href="/knowledge-rag/hubs">Explore all hubs</a>
          </Button>
        </div>
      </CardContent>
    </Card>
  );
};
```
