# Costinel / design

## Implementation Plan: CDN-First Top-Hub Signal Panel (<2h)

**Highest-value improvement**: Add a resilient “Top Hub” panel to Costinel’s dashboard that surfaces the most-connected hub (e.g., “MOC”) and related docs using **CDN-only fetches**, zero model compute, offline-first caching, and zero API-rate-limit risk.

### Why this ships fast
- No backend changes; pure frontend + static JSON.
- Uses CDN bypass pattern (`/resolve/main/`) — no HF auth/rate limits.
- Reuses existing design tokens and dashboard grid.
- <100 lines of code; deployable via existing CI/CD.

### 3. TopHubPanel component

`src/components/TopHubPanel.tsx`

```tsx
import { useEffect, useState } from "react";
import "./TopHubPanel.css";

interface RelatedDoc {
  slug: string;
  title: string;
  href: string;
  tags: string[];
}

interface HubInfo {
  slug: string;
  title: string;
  description: string;
  tags: string[];
  priority: number;
}

interface TopHubData {
  hub: HubInfo;
  related: RelatedDoc[];
  updatedAt: string;
}

const CDN_JSON =
  "https://huggingface.co/datasets/axentx/knowledge/resolve/main/top-hub.json";

export default function TopHubPanel() {
  const [data, setData] = useState<TopHubData | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    // CDN-only fetch; no Authorization header required.
    fetch(CDN_JSON, { cache: "force-cache" })
      .then((res) => {
        if (!res.ok) throw new Error(`CDN fetch ${res.status}`);
        return res.json();
      })
      .then((json) => {
        setData(json);
        setLoading(false);
      })
      .catch((err) => {
        setError(err.message);
        setLoading(false);
      });
  }, []);

  if (loading) return <div className="top-hub-panel loading">Loading insights…</div>;
  if (error) return <div className="top-hub-panel error">Insights unavailable</div>;
  if (!data) return null;

  return (
    <div className="top-hub-panel">
      <div className="hub-header">
        <span className="hub-badge">{data.hub.slug}</span>
        <h3>{data.hub.title}</h3>
        <p className="hub-desc">{data.hub.description}</p>
        <div className="hub-tags">
          {data.hub.tags.map((t) => (
            <span key={t} className="tag">
              {t}
            </span>
          ))}
        </div>
      </div>

      <div className="related-list">
        {data.related.map((doc) => (
          <a
            key={doc.slug}
            className="related-item"
            href={doc.href}
            target="_blank"
            rel="noopener noreferrer"
          >
            <span className="doc-title">{doc.title}</span>
            <div className="doc-tags">
              {doc.tags.map((t) => (
                <span key={t} className="tag small">
                  {t}
                </span>
              ))}
            </div>
          </a>
        ))}
      </div>

      <div className="panel-footer">
        Updated {new Date(data.updatedAt).toLocaleDateString()}
      </div>
    </div>
  );
}
```
