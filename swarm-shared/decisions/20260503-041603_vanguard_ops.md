# vanguard / ops

## Final synthesized solution

**Core problem**: training and UI hit the HF API at runtime (429s, non-reproducible epochs, no shareable snapshots), mixed-schema `enriched/` files break surrogate-1 training, and Lightning Studio churn wastes quota and risks idle-stop death.  

**Single change to implement**: add a content-addressed snapshot (manifest) for one date folder and wire training to use CDN-only fetches with strict `{prompt, response}` projection; reuse or safely restart a Lightning Studio.

### 1. Manifest generator (single file)

`/opt/axentx/vanguard/ops/make_snapshot.py`

```python
#!/usr/bin/env python3
"""
Create a content-addressed snapshot (manifest) for one date folder
so training can fetch exclusively via CDN (no HF API calls during training).

Usage:
  HF_REPO=user/ds DATE_DIR=2026-05-03 python3 make_snapshot.py > snapshot.json
"""
import json, hashlib, os, sys, datetime, requests
from typing import List, Dict

HF_REPO = os.getenv("HF_REPO")
DATE_DIR = os.getenv("DATE_DIR")

if not HF_REPO or not DATE_DIR:
    sys.stderr.write("HF_REPO and DATE_DIR are required.\n")
    sys.exit(1)

API_ROOT = f"https://huggingface.co/api/datasets/{HF_REPO}"

def list_folder(path: str) -> List[Dict]:
    r = requests.get(f"{API_ROOT}/tree", params={"path": path, "recursive": "false"}, timeout=30)
    r.raise_for_status()
    return r.json()

def sha256_hex(s: str) -> str:
    return hashlib.sha256(s.encode()).hexdigest()

def build_manifest() -> Dict:
    entries = list_folder(DATE_DIR)
    files = [e for e in entries if e["type"] == "file" and e["path"].lower().endswith((".parquet",))]

    if not files:
        sys.stderr.write(f"No parquet files found in {DATE_DIR}.\n")
        sys.exit(1)

    manifest = {
        "repo": HF_REPO,
        "date_dir": DATE_DIR,
        "generated_at": datetime.datetime.utcnow().isoformat() + "Z",
        "tree_hash": sha256_hex(json.dumps([f["path"] for f in sorted(files, key=lambda x: x["path"])], sort_keys=True)),
        "files": [
            {
                "path": f["path"],
                "size": f["size"],
                "cdn_url": f"https://huggingface.co/datasets/{HF_REPO}/resolve/main/{f['path']}"
            }
            for f in files
        ]
    }
    return manifest

if __name__ == "__main__":
    manifest = build_manifest()
    json.dump(manifest, sys.stdout, indent=2)
```

Make executable:

```bash
chmod +x /opt/axentx/vanguard/ops/make_snapshot.py
```

### 2. Training entrypoint (single file)

`/opt/axentx/vanguard/train.py`

```python
"""
Surrogate-1 training entry (Lightning).
- Uses content-addressed snapshot for CDN-only fetches (no HF API during training).
- Projects mixed-schema parquet to {prompt, response} at parse time.
- Reuses or restarts a Lightning Studio safely.
"""
import json, os, sys, io
from pathlib import Path

import pyarrow.parquet as pq
import requests
import torch
from torch.utils.data import Dataset, DataLoader
from lightning.pytorch import Trainer, LightningModule

MANIFEST_PATH = Path(os.getenv("MANIFEST_PATH", "snapshot.json"))
BATCH_SIZE = int(os.getenv("BATCH_SIZE", "4"))
MAX_EPOCHS = int(os.getenv("MAX_EPOCHS", "1"))

class CDNParquetDataset(Dataset):
    """Read parquet files from CDN and project to {prompt, response}."""
    def __init__(self, manifest_path: Path):
        if not manifest_path.exists():
            raise FileNotFoundError(f"Manifest not found: {manifest_path}")
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.file_urls = [f["cdn_url"] for f in self.manifest["files"]]

    def __len__(self):
        return len(self.file_urls)

    @staticmethod
    def _project_table(table) -> list[dict]:
        cols = set(table.column_names)
        prompt_col = next((c for c in ("prompt", "instruction", "input") if c in cols), None)
        response_col = next((c for c in ("response", "output", "completion") if c in cols), None)
        if prompt_col is None or response_col is None:
            raise ValueError(f"Missing prompt/response in columns: {cols}")
        prompts = table.column(prompt_col).to_pylist()
        responses = table.column(response_col).to_pylist()
        return [{"prompt": str(p), "response": str(r)} for p, r in zip(prompts, responses)]

    def __getitem__(self, idx):
        url = self.file_urls[idx]
        r = requests.get(url, timeout=30)
        r.raise_for_status()
        table = pq.read_table(io.BytesIO(r.content))
        rows = self._project_table(table)
        if not rows:
            return {"prompt": "", "response": ""}
        # Return first row; DataLoader batches across files.
        return rows[0]

class SurrogateDataModule(LightningModule):
    def __init__(self, manifest_path: Path, batch_size: int = BATCH_SIZE):
        super().__init__()
        self.ds = CDNParquetDataset(manifest_path)
        self.batch_size = batch_size

    def train_dataloader(self):
        return DataLoader(self.ds, batch_size=self.batch_size, shuffle=True, num_workers=0)

class SurrogateModel(LightningModule):
    def __init__(self):
        super().__init__()
        # Placeholder; replace with real model.
        self.net = torch.nn.Linear(1024, 1024)

    def training_step(self, batch, batch_idx):
        # Minimal step to prove pipeline runs; replace with real forward/loss.
        loss = torch.tensor(0.0, requires_grad=True)
        self.log("train_loss", loss)
        return loss

    def configure_optimizers(self):
        return torch.optim.Adam(self.parameters(), lr=1e-4)

def reuse_or_start_studio(name: str = "vanguard-training"):
    """
    Reuse a running Studio or create one.
    Prefer H200-capable machine when available; fallback to L40S.
    """
    from lightning.pytorch.studio import Studio, Machine
    from lightning.pytorch import Teamspace

    ts = Teamspace()
    for s in ts.studios:
        if s.name == name and s.status == "Running":
            return s

    # Use L40S by default; switch to H200 if account supports it.
    machine = Machine.L40S
    return Studio(name=name, machine=machine, create_ok=True)

if __name__ == "__main__":
    if not MANIFEST_PATH.exists():
        print(f"Missing {MANIFEST_PATH}. Run ops/make_snapshot.py first.", file=sys.stderr)
        sys.exit(1)

    # Reuse or start Studio
    studio = reuse_or_start_studio()

    dm = SurrogateDataModule(MANIFEST_PATH)
    model = SurrogateModel()
    trainer = Trainer(max_epochs=MAX_EPOCHS, accelerator="gpu", devices=1)
    trainer.fit(model, dm)
```
