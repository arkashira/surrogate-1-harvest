# workio / design

## Final synthesized implementation

**Core problem**: LINE retries webhooks on 5xx/timeouts and notifications can silently fail; without idempotency and retry/backoff the system can double-apply clock/leave events and lose user-facing messages.

**Chosen approach** (favoring correctness + deployability):
- Use **DB-backed idempotency** (not Redis) so retries are safe even if the cache layer is unavailable.
- Enforce **exact-once semantics for state changes** by making handlers idempotent at the domain level (unique constraint on `(tenant_id, user_id, date, period)` for clock/leave events) and using the webhook event log as a replay-suppression layer.
- Add **exponential backoff + jitter** for outbound LINE with short, clear timeouts and non-retryable client error fast-fail.
- Add **structured request-scoped logging/tracing** so webhook→action→notification chains are observable.

### 1) DB schema (`server/src/db/schema.sql`)
```sql
-- Durable webhook event log for idempotency and replay protection
CREATE TABLE IF NOT EXISTS webhook_events (
  id              BIGSERIAL PRIMARY KEY,
  tenant_id       BIGINT NOT NULL,
  event_id        TEXT   NOT NULL,       -- stable LINE event identifier
  event_type      TEXT   NOT NULL,       -- clock-in, clock-out, leave-request, etc.
  payload_hash    TEXT   NOT NULL,       -- sha256 of normalized body
  processed_at    TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  UNIQUE(tenant_id, event_id)
);

CREATE INDEX IF NOT EXISTS idx_webhook_events_tenant_event ON webhook_events(tenant_id, event_id);

-- Idempotent clock/leave events (prevent double-apply even if handler runs twice)
CREATE TABLE IF NOT EXISTS attendance_records (
  id              BIGSERIAL PRIMARY KEY,
  tenant_id       BIGINT NOT NULL,
  user_id         TEXT   NOT NULL,
  event_date      DATE   NOT NULL,
  period          TEXT   NOT NULL,      -- 'morning', 'afternoon', etc.
  event_type      TEXT   NOT NULL,      -- clock-in, clock-out, leave
  recorded_at     TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  UNIQUE(tenant_id, user_id, event_date, period)
);
```

### 2) Idempotency middleware (`server/src/middleware/idempotency.ts`)
```ts
import { Request, Response, NextFunction } from 'express';
import { Pool } from 'pg';
import crypto from 'crypto';
import { v4 as uuidv4 } from 'uuid';

const pool = new Pool({ connectionString: process.env.DATABASE_URL });

export type IdempotencyRecord = {
  tenantId: number;
  eventId: string;
  eventType: string;
  payloadHash: string;
};

function stableEventId(body: any): string {
  // Prefer a stable LINE identifier; fallback to deterministic hash
  const events = body?.events;
  if (Array.isArray(events) && events[0]) {
    const e = events[0];
    const type = e.type || 'unknown';
    const mode = e.mode || 'unknown';
    const sourceId = e.source?.userId || e.source?.groupId || e.source?.roomId;
    if (sourceId) return `${sourceId}:${type}:${mode}`;
  }
  // Deterministic fallback for malformed/unknown shapes
  return crypto.createHash('sha256').update(JSON.stringify(body)).digest('hex');
}

export async function ensureIdempotent(
  req: Request,
  _res: Response,
  next: NextFunction
) {
  const tenantId = (req as any).tenant?.id;
  if (!tenantId) {
    return next(new Error('tenant missing'));
  }

  const eventId = stableEventId(req.body);
  const payloadHash = crypto
    .createHash('sha256')
    .update(JSON.stringify(req.body))
    .digest('hex');

  const client = await pool.connect();
  try {
    await client.query('BEGIN');
    const dup = await client.query(
      `SELECT event_type, payload_hash FROM webhook_events
       WHERE tenant_id = $1 AND event_id = $2 FOR UPDATE`,
      [tenantId, eventId]
    );

    if (dup.rows.length > 0) {
      const existing = dup.rows[0];
      // Same payload: safe to skip processing.
      if (existing.payload_hash === payloadHash) {
        await client.query('ROLLBACK');
        (req as any).idempotency = { duplicate: true, tenantId, eventId };
        return next();
      }
      // Different payload on same event_id indicates conflict/replay; reject.
      console.warn('webhook conflict', { tenantId, eventId, existing, payloadHash });
      await client.query('ROLLBACK');
      const err: any = new Error('conflicting replay');
      err.status = 409;
      return next(err);
    }

    await client.query(
      `INSERT INTO webhook_events(tenant_id, event_id, event_type, payload_hash)
       VALUES ($1, $2, $3, $4)`,
      [tenantId, eventId, req.body?.events?.[0]?.type || 'unknown', payloadHash]
    );
    await client.query('COMMIT');

    (req as any).idempotency = { tenantId, eventId, eventType: req.body?.events?.[0]?.type, payloadHash } as IdempotencyRecord;
    next();
  } catch (err) {
    await client.query('ROLLBACK').catch(() => {});
    console.error('idempotency error', { tenantId, eventId, err });
    next(err);
  } finally {
    client.release();
  }
}
```

### 3) Resilient LINE notifier (`server/src/services/lineNotify.ts`)
```ts
import axios from 'axios';

const LINE_API = 'https://api.line.me/v2/bot/message/push';
const MAX_RETRIES = 5;
const BASE_DELAY_MS = 1000;
const TIMEOUT_MS = 10000;

function isRetryableStatus(status: number | undefined): boolean {
  if (status === undefined) return true;
  if (status >= 500) return true;
  if (status === 429) return true;
  return false;
}

export async function notifyLine(
  userId: string,
  message: any,
  accessToken: string,
  requestId?: string
): Promise<{ ok: boolean; error?: any }> {
  let attempt = 0;
  while (attempt < MAX_RETRIES) {
    try {
      const res = await axios.post(
        LINE_API,
        { to: userId, messages: Array.isArray(message) ? message : [message] },
        {
          headers: {
            Authorization: `Bearer ${accessToken}`,
            'Content-Type': 'application/json',
            ...(requestId && { 'X-Request-ID': requestId }),
          },
          timeout: TIMEOUT_MS,
        }
      );
      if (res.status === 200) return { ok: true };

      if (!isRetryableStatus(res.status)) {
        console.warn('line notify client error', { userId, status: res.status, body: res.data, requestId });
        return { ok: false, error: res.data };
      }
    } catch (err: any) {
      const status = err.response?.status;
      if (!isRetryableStatus(status)) {
        console.warn('line notify client error', { userId, status, err: err.message, requestId });
        return { ok: false, error: err.response?.data };
      }
    }

    attempt++;
    const delay = BASE_DELAY_MS * Math.pow(2, attempt) + Math.random() * 500;
    console.warn(`line notify retry ${attempt}/${MAX_RETRIES}`, { userId, delay, requestId });
    await new Promise((r) => setTimeout(r, delay));
  }
  console.error('line notify exhausted', { userId, requestId });
  return { ok: false, error: 'exhausted' };
}
```
