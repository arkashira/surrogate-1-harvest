# vanguard / design

## 1. Diagnosis
- No content-addressed manifest per date folder → training repeatedly calls `list_repo_tree`/`load_dataset` at runtime, causing HF API 429s and non-reproducible epochs.
- Data loader uses Hugging Face `datasets` API during training (streaming/list calls) instead of CDN-only fetches, violating the HF CDN bypass pattern.
- Missing deterministic shard selection and ordering → epoch-to-epoch shard order drifts, breaking reproducibility and resumption.
- Lightning Studio reuse not enforced → each run risks quota waste by creating new studios instead of reusing running ones.
- Idle-stop resilience absent → Lightning idle timeout kills training; no auto-restart on stopped studio.

## 2. Proposed change
Add a single design artifact and minimal training shim:
- `vanguard/design/cdn_manifest.py` — generates a content-addressed manifest (JSON) for one date folder via a single `list_repo_tree` call and embeds it in training.
- `vanguard/train/cdn_train_shim.py` — Lightning-compatible training loop that uses only CDN URLs (`https://huggingface.co/datasets/{repo}/resolve/main/...`) with zero HF API calls during data load, deterministic shard order, studio reuse, and idle-stop resilience.

### design/cdn_manifest.py
```python
#!/usr/bin/env python3
"""
Generate a content-addressed manifest for one date folder.
Run once (on Mac) after rate-limit window clears. Embed output in train.py.
"""
import json
import hashlib
import os
import sys
from datetime import datetime

try:
    from huggingface_hub import HfApi
except ImportError:
    print("pip install huggingface_hub")
    sys.exit(1)

REPO = os.getenv("HF_DATASET_REPO", "axentx/surrogate-1")
DATE_FOLDER = os.getenv("DATE_FOLDER", datetime.utcnow().strftime("%Y-%m-%d"))
OUT = os.getenv("MANIFEST_OUT", f"manifest_{DATE_FOLDER}.json")

def build_manifest(repo: str, date_folder: str) -> dict:
    api = HfApi()
    # Single API call: non-recursive per folder to minimize pagination/rate-limit risk
    entries = api.list_repo_tree(repo=repo, path=date_folder, recursive=False)
    files = []
    for e in entries:
        if e.type != "file":
            continue
        # CDN URL (no auth, bypasses API rate limits)
        cdn_url = f"https://huggingface.co/datasets/{repo}/resolve/main/{date_folder}/{e.path}"
        # content-addressed slug by filename only (deterministic)
        slug = e.path
        files.append({
            "slug": slug,
            "cdn_url": cdn_url,
            "size": e.size or 0,
        })
    # Deterministic ordering
    files.sort(key=lambda f: f["slug"])
    manifest = {
        "repo": repo,
        "date_folder": date_folder,
        "generated_at_utc": datetime.utcnow().isoformat() + "Z",
        "count": len(files),
        "files": files,
    }
    # content hash of the canonical file list
    listing = json.dumps([f["slug"] for f in files], separators=(",", ":"), sort_keys=True)
    manifest["content_hash"] = hashlib.sha256(listing.encode()).hexdigest()[:16]
    return manifest

if __name__ == "__main__":
    m = build_manifest(REPO, DATE_FOLDER)
    with open(OUT, "w") as f:
        json.dump(m, f, indent=2)
    print(f"Wrote {OUT} ({m['count']} files) content_hash={m['content_hash']}")
```

### train/cdn_train_shim.py

#!/usr/bin/env python3
"""
Lightning CDN-only training shim.
- Zero HF API calls during data load (CDN-only)
- Deterministic shard order from embedded manifest
- Studio reuse + idle-stop resilience
"""
import json
import os
import sys
import time
from pathlib import Path

import lightning as L
import torch
from torch.utils.data import Dataset, DataLoader

try:
    from huggingface_hub import HfApi, Teamspace
except ImportError:
    print("pip install lightning huggingface_hub")
    sys.exit(1)

# ---- Manifest (embed at build time or copy into container) ----
MANIFEST_PATH = os.getenv("MANIFEST_PATH", "manifest_latest.json")
with open(MANIFEST_PATH) as f:
    MANIFEST = json.load(f)

# ---- CDN Dataset ----
class CDNParquetDataset(Dataset):
    def __init__(self, manifest, transform=None):
        self.entries = manifest["files"]
        self.transform = transform

    def __len__(self):
        return len(self.entries)

    def __getitem__(self, idx):
        entry = self.entries[idx]
        url = entry["cdn_url"]
        # Lightweight: stream parquet row or download on demand.
        # For speed, consider prefetching batches to local disk/ram.
        # Here we return url + idx; replace with actual parquet row reading.
        sample = {"url": url, "idx": idx, "slug": entry["slug"]}
        if self.transform:
            sample = self.transform(sample)
        return sample

# ---- Lightning Module (stub) ----
class SurrogateTrainer(L.LightningModule):
    def __init__(self, lr=1e-4):
        super().__init__()
        self.lr = lr
        self.model = torch.nn.Linear(1, 1)  # replace with real model

    def training_step(self, batch, batch_idx):
        # dummy loss
        x = torch.tensor([float(batch_idx % 100)], dtype=torch.float32)
        y = self.model(x)
        loss = (y - 1.0).pow(2).mean()
        self.log("train_loss", loss)
        return loss

    def configure_optimizers(self):
        return torch.optim.Adam(self.model.parameters(), lr=self.lr)

# ---- Studio reuse + idle-stop resilience ----
def get_or_create_studio(name: str, machine=L.Machine.L40S) -> L.Studio:
    # Reuse running studio if exists
    teamspace = Teamspace.current()
    for s in teamspace.studios:
        if s.name == name and s.status == "running":
            print(f"Reusing running studio: {s.name}")
            return s
    # Create or start stopped one
    studio = L.Studio(
        name=name,
        machine=machine,
        create_ok=True,
    )
    if studio.status != "running":
        print(f"Starting studio {studio.name} on {machine}")
        studio.start(machine=machine)
    return studio

def run_training_with_resilience(manifest_path=MANIFEST_PATH, max_retries=3):
    retries = 0
    while retries < max_retries:
        try:
            studio = get_or_create_studio("vanguard-cdn-train")
            # Check status before run
            if studio.status != "running":
                print(f"Studio not running (status={studio.status}). Restarting...")
                studio.start(machine=L.Machine.L40S)
                time.sleep(10)

            dataset = CDNParquetDataset(MANIFEST)
            loader = DataLoader(dataset, batch_size=8, shuffle=False)  # deterministic order
            model = SurrogateTrainer()

            trainer = L.Trainer(
                max_epochs=1,
                accelerator="gpu",
                devices=1,
                logger=False,
                enable_checkpointing=False,
            )
            trainer.fit(model, loader)
            print("Training completed.")
            return 0
        except Exception as exc:
            retries += 1
            print(f"Training failed (attempt {retries}/{max_retries}): {exc}")
            if retries >= max_retries:
                print("Max retries exceeded.")
                return 1
            time.sleep(30)
    return 1

if __name__ == "__main__":
    sys.exit(run_training_with_resilience())
