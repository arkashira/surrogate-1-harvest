# Costinel / ops

## Final Implementation — Top-Hub Signal Panel (Costinel Dashboard)

**Scope & value**  
- Frontend-only, read-only panel that surfaces the most-connected hub (default **MOC**) and its actionable proposals from the knowledge graph.  
- Resilient by design: tries live API first, falls back to static CDN cache, then to in-code fixture.  
- Zero backend changes; ships in <2h with Vue 3 + TypeScript.

---

### 1) Types (`src/types/knowledge-rag.ts`)

```ts
export interface HubNode {
  id: string;
  label: string;
  type: 'hub' | 'proposal' | 'metric';
  score: number;
  edges: number;
}

export interface ActionableProposal {
  id: string;
  title: string;
  summary: string;
  impact: 'high' | 'medium' | 'low';
  effort: 'low' | 'medium' | 'high';
  roi: string;
  tags: string[];
  href?: string;
}

export interface TopHubPayload {
  hub: HubNode | null;
  proposals: ActionableProposal[];
  generatedAt: string;
  source: 'api' | 'cache' | 'fixture';
}
```

### 2) Resilient composable (`src/composables/useTopHub.ts`)

```ts
// src/composables/useTopHub.ts
import { ref, onMounted } from 'vue';
import type { TopHubPayload } from '@/types/knowledge-rag';

const API_ENDPOINT = '/api/knowledge-rag/top-hub';
const CACHE_ENDPOINT = '/knowledge-cache/top-hub-moc.json';

const FIXTURE: TopHubPayload = {
  hub: {
    id: 'hub-moc',
    label: 'MOC',
    type: 'hub',
    score: 96,
    edges: 128,
  },
  proposals: [
    {
      id: 'moc-ri-2026-05',
      title: 'Increase 1-yr No Upfront RI coverage to 65% for prod workloads',
      summary: 'Covers steady-state EC2/RDS; projected 22% savings vs on-demand.',
      impact: 'high',
      effort: 'medium',
      roi: '22%',
      tags: ['RI', 'AWS', 'savings'],
    },
    {
      id: 'moc-orphan-2026-05',
      title: 'Orphaned EBS & ENI cleanup candidates',
      summary: '37 unattached volumes + 12 idle ENIs across non-prod accounts.',
      impact: 'medium',
      effort: 'low',
      roi: '4%',
      tags: ['cleanup', 'AWS'],
    },
  ],
  generatedAt: new Date().toISOString(),
  source: 'fixture',
};

export function useTopHub() {
  const payload = ref<TopHubPayload | null>(null);
  const loading = ref(false);
  const error = ref<Error | null>(null);

  async function fetchTopHub() {
    loading.value = true;
    error.value = null;

    // 1) Try live API
    try {
      const res = await fetch(API_ENDPOINT, {
        method: 'GET',
        headers: { Accept: 'application/json' },
        // CDN-friendly: if API is behind CDN, auth/rate limits are minimized.
        // Single API call from orchestration layer; UI uses CDN-only fetches.
      });

      if (!res.ok) throw new Error(`API HTTP ${res.status}`);

      const json = (await res.json()) as TopHubPayload;
      payload.value = {
        ...json,
        source: 'api',
      };
      return;
    } catch (err: any) {
      console.warn('[useTopHub] API failed, trying cache', err);
    }

    // 2) Try static cache (CDN)
    try {
      const res = await fetch(CACHE_ENDPOINT, {
        method: 'GET',
        headers: { Accept: 'application/json' },
      });

      if (!res.ok) throw new Error(`Cache HTTP ${res.status}`);

      const json = (await res.json()) as TopHubPayload;
      payload.value = {
        ...json,
        source: 'cache',
      };
      return;
    } catch (err: any) {
      console.warn('[useTopHub] Cache failed, using fixture', err);
    }

    // 3) Fallback to fixture
    payload.value = FIXTURE;
    loading.value = false;
  }

  onMounted(fetchTopHub);

  return {
    payload,
    loading,
    error,
    refetch: fetchTopHub,
  };
}
```

### 3) Panel component (`src/components/TopHubSignalPanel.vue`)

```vue
<!-- src/components/TopHubSignalPanel.vue -->
<template>
  <section class="top-hub-panel card" aria-label="Top hub signals">
    <header class="panel-header">
      <h2 class="title">Top Hub: {{ hubLabel }}</h2>
      <span v-if="generatedAt" class="muted">Updated {{ formatDate(generatedAt) }}</span>
    </header>

    <div v-if="loading" class="loading" role="status">Loading signals…</div>

    <div v-else-if="proposals?.length" class="proposals">
      <article v-for="p in proposals" :key="p.id" class="proposal">
        <div class="proposal-title">{{ p.title }}</div>
        <p class="proposal-summary">{{ p.summary }}</p>
        <div class="proposal-meta">
          <span :class="['impact', p.impact]">{{ p.impact }}</span>
          <span :class="['effort', p.effort]">{{ p.effort }}</span>
          <span v-if="p.roi" class="roi">ROI {{ p.roi }}</span>
          <span v-for="tag in p.tags" :key="tag" class="tag">{{ tag }}</span>
        </div>
      </article>
    </div>

    <div v-else class="empty">No actionable proposals available.</div>

    <div v-if="error || payload?.source !== 'api'" class="error">
      <template v-if="error">
        Using fallback data. <button @click="refetch">Retry</button>
      </template>
      <template v-else>
        Source: {{ payload?.source }}
      </template>
    </div>
  </section>
</template>

<script setup lang="ts">
import { computed } from 'vue';
import { useTopHub } from '@/composables/useTopHub';

const { payload, loading, error, refetch } = useTopHub();

const hubLabel = computed(() => payload.value?.hub?.label ?? '—');
const proposals = computed(() => payload.value?.proposals ?? []);
const generatedAt = computed(() => payload.value?.generatedAt ?? '');

function formatDate(iso: string) {
  try {
    return new Intl.DateTimeFormat(undefined, {
      dateStyle: 'short',
      timeStyle: 'short',
    }).format(new Date(iso));
  } catch {
    return iso;
  }
}
</script>
