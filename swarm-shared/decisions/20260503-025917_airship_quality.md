### 1) One-time manifest creation on Mac (after rate-limit window clears)

`scripts/list_hf_date_folder.py`

```python
#!/usr/bin/env python3
"""
Run from Mac after rate-limit window clears.
Produces manifest with CDN-only paths for Lightning training.

Usage:
  python scripts/list_hf_date_folder.py \
    --repo datasets/axentx/surrogate-mirror \
    --folder batches/mirror-merged/2026-05-03 \
    --out manifest/2026-05-03_files.json
"""
import argparse
import json
import os
import time
from huggingface_hub import HfApi

HF_API_RATE_LIMIT_WAIT = 360  # seconds after 429

def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True)
    parser.add_argument("--folder", required=True, help="folder inside repo")
    parser.add_argument("--out", required=True)
    args = parser.parse_args()

    api = HfApi()
    retries = 3
    for attempt in range(1, retries + 1):
        try:
            # recursive=False keeps API calls minimal (one page per folder)
            items = api.list_repo_tree(
                repo_id=args.repo,
                path=args.folder,
                repo_type="dataset",
                recursive=False,
            )
            break
        except Exception as exc:
            if attempt == retries:
                raise
            print(f"[{attempt}/{retries}] list_repo_tree failed: {exc}. Waiting {HF_API_RATE_LIMIT_WAIT}s")
            time.sleep(HF_API_RATE_LIMIT_WAIT)

    # Keep only files (ignore subfolders)
    files = sorted([p.path for p in items if p.type == "file"])
    os.makedirs(os.path.dirname(os.path.abspath(args.out)), exist_ok=True)
    with open(args.out, "w") as f:
        json.dump(files, f, indent=2)
    print(f"Wrote {len(files)} files -> {args.out}")

if __name__ == "__main__":
    main()
```

#!/usr/bin/env python3
"""
Lightning Studio training job.
- Uses CDN URLs only (no HF API auth/rate-limit during data loading).
- Reuses running Studio if present.
- Restarts if Studio was stopped (idle timeout).
"""
import json
import os
import sys
from pathlib import Path

import lightning as L
import torch
from torch.utils.data import DataLoader, IterableDataset

# --
# 1) Dataset: stream parquet via CDN URLs (no HF API)
# --
class CDNParquetIterable(IterableDataset):
    def __init__(self, manifest_path: str, repo: str, split: str = "train"):
        super().__init__()
        self.repo = repo
        with open(manifest_path) as f:
            self.files = json.load(f)
        self.split = split

    def _cdn_url(self, path: str) -> str:
        # Public CDN URL — no Authorization header required
        return f"https://huggingface.co/datasets/{self.repo}/resolve/main/{path}"

    def __iter__(self):
        worker_info = torch.utils.data.get_worker_info()
        if worker_info is None:
            files = self.files
        else:
            per_worker = len(self.files) // worker_info.num_workers
            files = self.files[worker_info.id * per_worker : (worker_info.id + 1) * per_worker]

        for path in files:
            url = self._cdn_url(path)
            try:
                import fsspec
                import pyarrow.parquet as pq

                with fsspec.open(url, "rb") as f:
                    table = pq.read_table(f, columns=["prompt", "response"])
                    for row in table.to_pylist():
                        yield {
                            "input": row.get("prompt", ""),
                            "target": row.get("response", ""),
                        }
            except Exception as exc:
                print(f"[WARN] failed to read {url}: {exc}", file=sys.stderr)
                continue

class SurrogateModule(L.LightningModule):
    def __init__(self, lr: float = 1e-4):
        super().__init__()
        self.lr = lr
        self.net = torch.nn.Linear(1024, 1024)

    def training_step(self, batch, batch_idx):
        loss = torch.tensor(0.0, requires_grad=True)
        self.log("train_loss", loss)
        return loss

    def configure_optimizers(self):
        return torch.optim.AdamW(self.net.parameters(), lr=self.lr)

# --
# 3) Studio reuse + idle resilience
# --
def get_or_create_studio(name: str, machine: L.Machine = L.Machine.L40S) -> L.Studio:
    teamspace = L.Teamspace()
    for s in teamspace.studios:
        if s.name == name and s.status == "running":
            print(f"Reusing running Studio: {s.name}")
            return s
    print(f"Creating Studio: {name}")
    return L.Studio(name=name, machine=machine, create_ok=True)

def ensure_running(studio: L.Studio, machine: L.Machine = L.Machine.L40S) -> L.Studio:
    if studio.status != "running":
        print(f"Studio stopped (idle). Restarting on {machine}...")
        studio.start(machine=machine)
        studio.wait_until_ready()
    return studio

# --
# 4) Train entrypoint used by Lightning
# --
def train(manifest_path: str, repo: str, max_epochs: int = 1):
    dataset = CDNParquetIterable(manifest_path=manifest_path, repo=repo, split="train")
    loader = DataLoader(dataset, batch_size=8, num_workers=2)

    model = SurrogateModule()
    trainer = L.Trainer(max_epochs=max_epochs, devices=1)
    trainer.fit(model, loader)

# --
# 5) CLI wrapper for Studio launch (Mac orchestration)
# --
def main() -> None:
    import argparse
    parser = argparse.ArgumentParser()
    parser.add_argument("--manifest", required=True)
    parser.add_argument("--repo", default="datasets/axentx/surrogate-mirror")
    parser.add_argument("--max-epochs", type=int, default=1)
    parser.add_argument("--studio-name", default="surrogate-train")
    parser.add_argument("--machine", default="L40S", choices=["L40S", "L4", "CPU"])
    args = parser.parse_args()

    machine_map = {
        "L40S": L.Machine.L40S,
        "L4": L.Machine.L4,
        "CPU": L.Machine.CPU,
    }
    machine = machine_map[args.machine]

    # If running inside Lightning Studio, just train
    if os.getenv("LIGHTNING_STUDIO_NAME"):
        print("Running inside Lightning Studio. Starting training.")
        train(manifest_path=args.manifest, repo=args.repo, max_epochs=args.max_epochs)
        return

    # Mac orchestration path: reuse/create studio and run script remotely
    studio = get_or_create_studio(name=args.studio_name, machine=machine)
    studio = ensure_running(studio, machine=machine)

    # Copy repo code + manifest to Studio and run training
    studio.upload_folder(".", ".")
    studio.run(
        command=(
            f"python train/train_surrogate.py "
            f"--manifest {args.manifest} "
            f"--repo {args.repo} "
            f"--max-epochs {args.max_epochs}"
        ),
        environment={"PYTHONPATH": "."},
    )
