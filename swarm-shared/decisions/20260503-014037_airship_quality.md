## 1. Mac Orchestrator: Generate file manifest (single API call, outside rate-limit window)

**Script** (corrected path + robust Python):  
`/opt/axentx/airship/scripts/generate_hf_manifest.py`

```python
#!/usr/bin/env python3
"""
Generate CDN file manifest for a HF dataset repo.
Run from Mac orchestrator. Embed output in training repo.
"""
import json
import os
import sys
from datetime import datetime, timezone

try:
    from huggingface_hub import HfApi
except ImportError:
    print("pip install huggingface_hub", file=sys.stderr)
    sys.exit(1)

HF_REPO = os.getenv("HF_DATASET_REPO", "axentx/surrogate-datasets")
DATE_FOLDER = os.getenv("DATE_FOLDER", datetime.now(timezone.utc).strftime("%Y-%m-%d"))
OUT_PATH = os.getenv("MANIFEST_OUT", "/opt/axentx/airship/manifests/hf_files_latest.json")

def main() -> None:
    api = HfApi()
    entries = api.list_repo_tree(
        repo_id=HF_REPO,
        path=DATE_FOLDER,
        repo_type="dataset",
        recursive=False,
    )
    files = sorted(
        e.path for e in entries
        if e.type == "file" and e.path.lower().endswith(".parquet")
    )
    if not files:
        print(f"No parquet files found in {HF_REPO}/{DATE_FOLDER}", file=sys.stderr)
        sys.exit(1)

    manifest = {
        "repo": HF_REPO,
        "date_folder": DATE_FOLDER,
        "generated_utc": datetime.now(timezone.utc).isoformat(),
        "files": files,
    }
    os.makedirs(os.path.dirname(OUT_PATH), exist_ok=True)
    tmp = OUT_PATH + ".tmp"
    with open(tmp, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2)
    os.replace(tmp, OUT_PATH)
    print(f"Wrote {len(files)} files to {OUT_PATH}")

if __name__ == "__main__":
    main()
```

## 2. CDN-only DataLoader for Lightning Studio  
*Zero HF API calls during data loading; retry/backoff for CDN flakes; worker-aware sharding.*

`/opt/axentx/airship/surrogate/training/data.py`

```python
import json
import os
import time
from typing import Dict, Any, Optional

import pyarrow.parquet as pq
import requests
from torch.utils.data import IterableDataset, get_worker_info


class CDNParquetIterable(IterableDataset):
    """
    Stream parquet shards directly from HF CDN (no Authorization header).
    Manifest must contain {"repo": "...", "files": [...]}.
    """

    CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

    def __init__(
        self,
        manifest_path: str,
        max_shards: int = -1,
        max_retries: int = 5,
        backoff_factor: float = 1.0,
    ):
        with open(manifest_path, "r", encoding="utf-8") as f:
            cfg = json.load(f)
        self.repo = cfg["repo"]
        self.files = cfg["files"]
        if max_shards > 0:
            self.files = self.files[:max_shards]
        self.max_retries = max_retries
        self.backoff_factor = backoff_factor
        self.worker_id: Optional[int] = None
        self.num_workers: int = 1

    def _cdn_url(self, path: str) -> str:
        return self.CDN_TEMPLATE.format(repo=self.repo, path=path)

    def _stream_shard(self, path: str):
        url = self._cdn_url(path)
        for attempt in range(1, self.max_retries + 1):
            try:
                with requests.get(url, timeout=60, stream=True) as r:
                    r.raise_for_status()
                    content = r.content
                table = pq.read_table(pq.ParquetFile(pq.BufferReader(content)))
                for batch in table.to_batches(columns=["prompt", "response"]):
                    for i in range(batch.num_rows):
                        yield {
                            "prompt": batch["prompt"][i].as_py(),
                            "response": batch["response"][i].as_py(),
                        }
                return
            except Exception as exc:
                if attempt == self.max_retries:
                    raise
                sleep_sec = self.backoff_factor * (2 ** (attempt - 1))
                print(f"[{path}] attempt {attempt}/{self.max_retries} failed: {exc}. Retry in {sleep_sec}s")
                time.sleep(sleep_sec)

    def __iter__(self):
        worker_info = get_worker_info()
        if worker_info is not None:
            files = [f for i, f in enumerate(self.files) if i % worker_info.num_workers == worker_info.id]
        else:
            files = self.files

        for path in files:
            yield from self._stream_shard(path)
```
