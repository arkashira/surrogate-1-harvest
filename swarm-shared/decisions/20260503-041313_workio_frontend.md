# workio / frontend

## Final Synthesis (Best of Both Candidates)

**Diagnosis (merged, corrected)**
- Double-tap and rapid toggles create overlapping requests before idempotency settles; client-side deduplication is required, not just backend keys.
- No optimistic/pending UI → perceived latency and retries.
- GPS failures surface as opaque errors with no recovery path.
- Clock state derived only from network → sluggish/offline UX.
- Race between toggles can overlap requests even with backend idempotency.

**Proposed change (merged, corrected)**
- `workio/src/features/clock/ClockButton.tsx` — optimistic UI, locked interaction while pending, GPS pre-check, clear actionable errors.
- `workio/src/lib/api.ts` — lightweight in-flight dedupe + abort + short timeout; stable idempotency key per logical action (not per retry) so retries reuse same key.
- `workio/src/features/clock/useClock.ts` — encapsulate optimistic state, GPS guard, deduped call, and recovery fallback.

**Implementation (merged, corrected, production-ready)**

### `workio/src/lib/api.ts`
```ts
const inFlight = new Map<string, { controller: AbortController; promise: Promise<any> }>();

export function dedupedFetch<T>(
  key: string,
  fn: (signal: AbortSignal) => Promise<T>,
  { timeoutMs = 10000 }: { timeoutMs?: number } = {}
): Promise<T> {
  if (inFlight.has(key)) {
    return inFlight.get(key)!.promise;
  }

  const controller = new AbortController();
  const timeout = setTimeout(() => controller.abort(), timeoutMs);

  const promise = fn(controller.signal)
    .finally(() => {
      clearTimeout(timeout);
      inFlight.delete(key);
    })
    .catch((err) => {
      // Normalize abort to a recognizable error for UI
      if (err.name === 'AbortError') {
        throw new Error('Request cancelled');
      }
      throw err;
    });

  inFlight.set(key, { controller, promise });
  return promise;
}

export function cancelDeduped(key: string) {
  inFlight.get(key)?.controller.abort();
}
```

### `workio/src/features/clock/useClock.ts`
```ts
import { useState, useCallback, useRef } from 'react';
import { dedupedFetch, cancelDeduped } from '@/lib/api';

type ClockState = 'idle' | 'pending' | 'success' | 'error';

export function useClock() {
  const [state, setState] = useState<ClockState>('idle');
  const [lastAction, setLastAction] = useState<'in' | 'out' | null>(null);
  const actionKeyRef = useRef<string | null>(null);

  const checkGPS = useCallback(async (): Promise<GeolocationPosition> => {
    return new Promise((resolve, reject) => {
      if (!navigator.geolocation) {
        reject(new Error('Geolocation not supported'));
        return;
      }
      navigator.geolocation.getCurrentPosition(resolve, reject, {
        enableHighAccuracy: true,
        timeout: 10000,
        maximumAge: 30000,
      });
    });
  }, []);

  const clock = useCallback(
    async (type: 'in' | 'out', opts?: { skipGPS?: boolean }) => {
      if (state === 'pending') return;

      // Stable idempotency key per logical action+date so retries reuse same key
      const dayKey = new Date().toDateString();
      const key = `clock:${type}:${dayKey}`;
      actionKeyRef.current = key;

      setState('pending');
      try {
        let position: GeolocationPosition | undefined;
        if (!opts?.skipGPS) {
          try {
            position = await checkGPS();
          } catch (err) {
            // GPS failure: surface as actionable error but allow optional skip via recovery
            setState('error');
            const e = err as Error;
            throw new Error(
              e.message.includes('not supported')
                ? 'Geolocation not supported on this device.'
                : 'Location access required. Enable location or use recovery flow to proceed without GPS.'
            );
          }
        }

        const payload = {
          type,
          timestamp: new Date().toISOString(),
          coords: position
            ? {
                lat: position.coords.latitude,
                lng: position.coords.longitude,
                accuracy: position.coords.accuracy,
              }
            : null,
        };

        const result = await dedupedFetch(key, (signal) =>
          fetch('/api/clock', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(payload),
            signal,
          }).then((r) => {
            if (!r.ok) throw new Error('Clock request failed');
            return r.json();
          })
        );

        setState('success');
        setLastAction(type);
        return result;
      } catch (err) {
        setState('error');
        throw err;
      }
    },
    [state, checkGPS]
  );

  const reset = useCallback(() => setState('idle'), []);
  const cancel = useCallback(() => {
    if (actionKeyRef.current) cancelDeduped(actionKeyRef.current);
    setState('idle');
  }, []);

  return { state, lastAction, clock, reset, cancel };
}
```

### `workio/src/features/clock/ClockButton.tsx`
```tsx
import { useClock } from './useClock';
import { Button } from '@/components/ui/button';
import { Loader2, Clock, MapPin, AlertCircle } from 'lucide-react';

export function ClockButton() {
  const { state, lastAction, clock, reset, cancel } = useClock();
  const isPending = state === 'pending';

  const handleClockIn = () => clock('in').catch(() => {});
  const handleClockOut = () => clock('out').catch(() => {});

  return (
    <div className="flex flex-col items-center gap-3">
      <div className="flex gap-2">
        <Button
          onClick={handleClockIn}
          disabled={isPending}
          variant={lastAction === 'in' && !isPending ? 'default' : 'outline'}
          className="min-w-[100px]"
        >
          {isPending ? (
            <>
              <Loader2 className="mr-2 h-4 w-4 animate-spin" />
              Clocking in
            </>
          ) : (
            <>
              <Clock className="mr-2 h-4 w-4" />
              Clock In
            </>
          )}
        </Button>

        <Button
          onClick={handleClockOut}
          disabled={isPending}
          variant={lastAction === 'out' && !isPending ? 'default' : 'outline'}
          className="min-w-[100px]"
        >
          {isPending ? (
            <>
              <Loader2 className="mr-2 h-4 w-4 animate-spin" />
              Clocking out
            </>
          ) : (
            <>
              <Clock className="mr-2 h-4 w-4" />
              Clock Out
            </>
          )}
        </Button>
      </div>

      {state === 'error' && (
        <div className="flex flex-col items-center gap-2 text-sm text-destructive">
          <div className="flex items-center gap-1">
            <AlertCircle className="h-3 w-3" />
            <span>Location required or clock failed.</span>
          </div>
          <div className="flex gap-2">
            <Button size="sm" variant="ghost" onClick={reset}>
              Retry
            </Button>
            <Button
              size="sm"
              variant="outline"
              onClick={() => {
                // Recovery: skip GPS and retry last action if available
                if (lastAction) {
                  clock(lastAction, { skipGPS: true }).catch(() => {});
                } else {
                  reset();
                }
              }}
            >
              Skip location
            </Button>
            <Button size="sm" variant="ghost" onClick={cancel}>
              Cancel
            </Button>
          </div>
        </div>
