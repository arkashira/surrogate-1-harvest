# workio / frontend

## Final synthesized implementation (best of both proposals)

**Goal:** Add compile-time type safety and runtime validation for route params, search queries, and location state — with concrete, incremental adoption and zero breaking URL changes.

---

### 1. Central route contract (`src/lib/routes.ts`)

Combines Candidate 1’s explicit route templates with Candidate 2’s discriminated state union.

```ts
// src/lib/routes.ts
import { z } from 'zod';

// -------------------------
// Discriminated route state
// -------------------------
export type RouteState =
  | { page: 'login'; redirectTo?: string }
  | { page: 'employees'; employeeId?: string }
  | { page: 'leave-request'; prefill?: { date?: string; type?: string } }
  | { page: 'ot-request'; prefill?: { date?: string; hours?: number } }
  | { page: 'clock'; tenantId?: string; prefilledDate?: string }
  | { page: '*' };

// Builders (type-safe state creators)
export const routeState = {
  login: (redirectTo?: string): RouteState => ({ page: 'login', redirectTo }),
  employees: (employeeId?: string): RouteState => ({ page: 'employees', employeeId }),
  leaveRequest: (prefill?: RouteState extends { page: 'leave-request' } ? RouteState['prefill'] : never): RouteState => ({
    page: 'leave-request',
    prefill,
  }),
  otRequest: (prefill?: RouteState extends { page: 'ot-request' } ? RouteState['prefill'] : never): RouteState => ({
    page: 'ot-request',
    prefill,
  }),
  clock: (tenantId?: string, prefilledDate?: string): RouteState => ({
    page: 'clock',
    tenantId,
    prefilledDate,
  }),
  wildcard: (): RouteState => ({ page: '*' }),
} as const;

// -------------------------
// Route path templates
// -------------------------
export const ROUTES = {
  clock: '/clock',
  clockIn: '/clock/in',
  clockOut: '/clock/out',
  leaveRequest: '/leave/request',
  leaveDetail: '/leave/:id',
  otRequest: '/ot/request',
} as const;

// -------------------------
// Query schemas + parsing
// -------------------------
const LeaveListQuerySchema = z.object({
  status: z.enum(['pending', 'approved', 'rejected']).optional(),
  from: z.string().optional(),
  to: z.string().optional(),
  page: z.string().optional(),
});
export type LeaveListQuery = z.infer<typeof LeaveListQuerySchema>;

const OtListQuerySchema = z.object({
  status: z.enum(['pending', 'approved', 'rejected']).optional(),
  from: z.string().optional(),
  to: z.string().optional(),
});
export type OtListQuery = z.infer<typeof OtListQuerySchema>;

export function parseLeaveQuery(search: string | URLSearchParams): LeaveListQuery {
  const params = new URLSearchParams(search);
  const raw = Object.fromEntries(params.entries());
  return LeaveListQuerySchema.parse(raw);
}

export function parseOtQuery(search: string | URLSearchParams): OtListQuery {
  const params = new URLSearchParams(search);
  const raw = Object.fromEntries(params.entries());
  return OtListQuerySchema.parse(raw);
}

export function toQueryString(obj: Record<string, string | undefined>): string {
  const params = new URLSearchParams();
  for (const [k, v] of Object.entries(obj)) {
    if (v !== undefined) params.set(k, v);
  }
  return params.toString();
}
```

### 2. Typed location hook (`src/hooks/useTypedLocation.ts`)

Merges Candidate 1’s query parsing with Candidate 2’s narrowed state.

```ts
// src/hooks/useTypedLocation.ts
import { useLocation, useSearchParams } from 'react-router-dom';
import type { RouteState } from '../lib/routes';
import { parseLeaveQuery, parseOtQuery } from '../lib/routes';

export function useTypedLocation() {
  const location = useLocation();
  const [searchParams] = useSearchParams();

  const state = (location.state as RouteState | null) ?? null;

  return {
    pathname: location.pathname,
    state,
    search: location.search,
    leaveQuery: parseLeaveQuery(searchParams),
    otQuery: parseOtQuery(searchParams),
    rawSearchParams: searchParams,
  };
}
```

### 3. Typed navigation hook (`src/hooks/useTypedNavigate.ts`)

Keeps Candidate 1’s simple, practical wrapper and adds type-safe state.

```ts
// src/hooks/useTypedNavigate.ts
import { useNavigate } from 'react-router-dom';
import type { RouteState } from '../lib/routes';

export function useTypedNavigate() {
  const navigate = useNavigate();

  return {
    to: (path: string, options?: { state?: RouteState; replace?: boolean; search?: string }) => {
      navigate(options?.search ? `${path}${options.search}` : path, {
        state: options?.state ?? null,
        replace: options?.replace ?? false,
      });
    },
    goBack: () => navigate(-1),
  };
}
```

### 4. Example usage (clock-in with typed state + query)

```tsx
import { ROUTES } from '../lib/routes';
import { useTypedNavigate, useTypedLocation } from '../hooks';

export function ClockInButton() {
  const { to } = useTypedNavigate();
  const { state } = useTypedLocation();

  const handleClockIn = () => {
    to(ROUTES.clockIn, {
      state: {
        page: 'clock',
        tenantId: state?.tenantId,
        prefilledDate: new Date().toISOString(),
      },
    });
  };

  return <button onClick={handleClockIn}>Clock In</button>;
}
```

### Key decisions (why this synthesis)

- **Zod schemas for queries** (Candidate 1 + validation): ensures parsed query shapes are correct at runtime and gives single source of truth.
- **Discriminated `RouteState` union** (Candidate 2): makes state shapes explicit per logical page and enables narrowing.
- **Simple typed navigate wrapper** (Candidate 1): practical, low-friction adoption without forcing full route definition rewrites.
- **No URL changes**: preserves backwards compatibility; only adds types and validation.
