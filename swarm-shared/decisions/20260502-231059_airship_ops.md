# airship / ops

## Highest-Value Incremental Improvement (≤2h)

**Goal**: Harden `airship discover` into a **deterministic, CDN-only, zero-HF-API-runtime orchestrator** that eliminates HF API rate limits and PyArrow schema errors while producing reproducible file lists and safe ingestion payloads.

**Why this wins**
- Removes runtime HF API calls during training/ingest → bypasses 429/1000-per-5min and 128-commit caps.
- Eliminates `load_dataset(streaming=True)` schema hazards by projecting `{prompt,response}` only at parse time.
- Fits <2h: single CLI change + small orchestration wrapper + one training-script patch.
- Reuses known patterns: CDN bypass, pre-list + embed file list, surrogate-1 schema hygiene, Mac=CLI/remote compute.

---

## Implementation Plan

1. **Add `airship discover` CLI command**
   - Single API call (after rate-limit window) to `list_repo_tree(path, recursive=False)` per date folder.
   - Save canonical file list to `file_list.json` (repo-root relative paths).
   - Output: `batches/mirror-merged/{date}/file_list.json`.

2. **Embed file list in training script**
   - `train.py` loads `file_list.json` and uses only CDN URLs (`https://huggingface.co/datasets/{repo}/resolve/main/{path}`) with `requests`/`urllib`; zero HF API calls during data loading.
   - Stream parse each file and project to `{prompt, response}`; drop all other fields.

3. **Schema-safe ingestion writer**
   - Produce `batches/mirror-merged/{date}/{slug}.parquet` with exactly `{prompt, response}`.
   - Attribution via filename pattern only (no `source`/`ts` columns).

4. **Lightning Studio reuse guard**
   - Launcher checks running studios and reuses; avoids quota waste.

5. **Cron/env hygiene**
   - Ensure wrapper scripts have `#!/usr/bin/env bash`, are `chmod +x`, and crontab sets `SHELL=/bin/bash`.

### 1) CLI: `airship/cli/discover.py`

```python
#!/usr/bin/env python3
import json
import os
import sys
from datetime import datetime
from pathlib import Path

try:
    from huggingface_hub import HfApi
except ImportError:
    print("ERROR: huggingface_hub required", file=sys.stderr)
    sys.exit(1)

API = HfApi()
REPO = "your-org/your-dataset-repo"  # parameterize via env/arg later
OUT_ROOT = Path("batches/mirror-merged")

def list_date_folder(date_folder: str):
    """Single API call per date folder; recursive=False avoids pagination explosion."""
    try:
        items = API.list_repo_tree(repo_id=REPO, path=date_folder, recursive=False)
    except Exception as exc:
        print(f"ERROR listing {REPO}/{date_folder}: {exc}", file=sys.stderr)
        return []
    # Keep only files (tree entries have .path)
    return [it.path for it in items if getattr(it, "type", None) == "file" or "." in Path(it.path).name]

def main():
    # Usage: airship discover <date_folder|today>
    arg = sys.argv[2] if len(sys.argv) > 2 else datetime.utcnow().strftime("%Y-%m-%d")
    date_folder = arg.strip("/")
    files = list_date_folder(date_folder)
    if not files:
        print(f"No files found in {REPO}/{date_folder}", file=sys.stderr)
        sys.exit(1)

    out_dir = OUT_ROOT / date_folder
    out_dir.mkdir(parents=True, exist_ok=True)
    list_path = out_dir / "file_list.json"
    with open(list_path, "w", encoding="utf-8") as f:
        json.dump({
            "repo": REPO,
            "date_folder": date_folder,
            "generated_utc": datetime.utcnow().isoformat() + "Z",
            "files": sorted(files)
        }, f, indent=2)
    print(f"OK {list_path}")
    print(f"CDN-only fetch example: https://huggingface.co/datasets/{REPO}/resolve/main/{files[0]}")

if __name__ == "__main__":
    main()
```

### 2) Training loader: `surrogate/train.py` (minimal patch)

```python
import json
from pathlib import Path
import requests
import pyarrow as pa
import pyarrow.parquet as pq
from datasets import Dataset

REPO = "your-org/your-dataset-repo"
CDN_ROOT = f"https://huggingface.co/datasets/{REPO}/resolve/main"

def load_file_list(list_path: Path):
    with open(list_path, encoding="utf-8") as f:
        data = json.load(f)
    return data["files"], data["repo"]

def cdn_url(repo: str, filepath: str):
    return f"https://huggingface.co/datasets/{repo}/resolve/main/{filepath}"

def project_to_prompt_response(raw):
    # Surrogate-1 schema hygiene: keep only {prompt, response}
    # Adapt to your actual field names (e.g., prompt/response or instruction/output)
    if isinstance(raw, dict):
        return {
            "prompt": raw.get("prompt") or raw.get("instruction") or raw.get("input") or "",
            "response": raw.get("response") or raw.get("output") or raw.get("completion") or ""
        }
    return {"prompt": "", "response": ""}

def stream_parse_and_write(file_list_path, out_parquet):
    files, repo = load_file_list(file_list_path)
    rows = []
    for rel in files:
        url = cdn_url(repo, rel)
        try:
            r = requests.get(url, timeout=30)
            r.raise_for_status()
        except Exception as exc:
            print(f"WARN skip {rel}: {exc}")
            continue

        # Lightweight projection: handle JSONL/JSON/parquet minimally
        # For HF datasets-like repos, many files are JSONL
        fname = Path(rel).name.lower()
        content = r.content
        if fname.endswith(".jsonl"):
            for line in content.decode("utf-8").splitlines():
                line = line.strip()
                if not line:
                    continue
                try:
                    obj = json.loads(line)
                    rows.append(project_to_prompt_response(obj))
                except Exception:
                    continue
        elif fname.endswith(".json"):
            try:
                arr = json.loads(content.decode("utf-8"))
                if isinstance(arr, list):
                    for obj in arr:
                        rows.append(project_to_prompt_response(obj))
                else:
                    rows.append(project_to_prompt_response(arr))
            except Exception:
                continue
        else:
            # skip unknown binary; or add parquet handling if needed
            continue

    if not rows:
        print("WARN: no rows extracted")
        return

    table = pa.Table.from_pylist(rows, schema=pa.schema([
        pa.field("prompt", pa.string()),
        pa.field("response", pa.string())
    ]))
    pq.write_table(table, out_parquet)
    print(f"Wrote {len(rows)} rows -> {out_parquet}")

if __name__ == "__main__":
    # Example: python train.py 2026-05-02
    import sys
    date_folder = sys.argv[1] if len(sys.argv) > 1 else "2026-05-02"
    list_p = Path("batches/mirror-merged") / date_folder / "file_list.json"
    out_p = Path("batches/mirror-merged") / date_folder / "merged.parquet"
    stream_parse_and_write(list_p, out_p)
```
