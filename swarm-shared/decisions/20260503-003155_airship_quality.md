## Implementation Plan — Manifest-Driven CDN-Only Dataset Loader

**Scope**: Replace `load_dataset`/`list_repo_files` usage in surrogate training with a manifest-driven, CDN-only iterable loader to eliminate HF API rate limits and mixed-schema errors.

**Estimated effort**: <2h (including tests)

### 2) Add manifest generator (run once on Mac)

Create `surrogate/scripts/build_manifest.py`:

```python
#!/usr/bin/env python3
"""
Generate a manifest of public dataset files for CDN-only loading.
Run on Mac (or any machine with HF token) once per date-folder.
Output: manifest.json used by train.py (zero API calls during training).
"""
import json
import os
import sys
from datetime import datetime
from pathlib import Path

try:
    from huggingface_hub import HfApi
except ImportError:
    print("pip install huggingface_hub")
    sys.exit(1)

HF_REPO = os.getenv("HF_DATASET_REPO", "axentx/surrogate-mirror")
DATE_FOLDER = os.getenv("DATE_FOLDER", datetime.utcnow().strftime("%Y-%m-%d"))
OUTPUT = Path(os.getenv("MANIFEST_OUT", "surrogate/manifest.json"))

def build_manifest() -> None:
    api = HfApi()
    # Single non-recursive call per date folder (avoids 100x pagination)
    entries = api.list_repo_tree(
        repo_id=HF_REPO,
        path=DATE_FOLDER,
        repo_type="dataset",
        recursive=False,
    )
    files = [e.path for e in entries if e.type == "file" and e.path.endswith(".parquet")]
    manifest = {
        "repo": HF_REPO,
        "date": DATE_FOLDER,
        "generated_at": datetime.utcnow().isoformat() + "Z",
        "files": sorted(files),
        "total_files": len(files),
    }
    OUTPUT.parent.mkdir(parents=True, exist_ok=True)
    OUTPUT.write_text(json.dumps(manifest, indent=2))
    print(f"Manifest written to {OUTPUT} with {len(files)} files")

if __name__ == "__main__":
    build_manifest()
```

Make executable:

```bash
chmod +x surrogate/scripts/build_manifest.py
```

### 3) Add CDN-only iterable dataset loader

Create `surrogate/data/cdn_dataset.py`:

```python
from __future__ import annotations

import json
import logging
from dataclasses import dataclass
from pathlib import Path
from typing import Iterator, Dict, Any

import pyarrow.parquet as pq
import requests
from tqdm import tqdm

logger = logging.getLogger(__name__)

CDN_ROOT = "https://huggingface.co/datasets"

@dataclass
class CDNManifest:
    repo: str
    date: str
    files: list[str]

    @classmethod
    def load(cls, manifest_path: Path | str) -> "CDNManifest":
        data = json.loads(Path(manifest_path).read_text())
        return cls(repo=data["repo"], date=data["date"], files=data["files"])

    def resolve_url(self, file_path: str) -> str:
        # CDN download; no Authorization header required
        return f"{CDN_ROOT}/{self.repo}/resolve/main/{file_path}"

class CDNParquetIterable:
    """
    Zero HF API calls during training.
    Downloads parquet shards via CDN and yields {prompt, response} rows.
    """

    def __init__(self, manifest: CDNManifest, columns=("prompt", "response")):
        self.manifest = manifest
        self.columns = columns

    def _download_shard(self, file_path: str, tmp_path: Path) -> Path:
        url = self.manifest.resolve_url(file_path)
        resp = requests.get(url, timeout=60, stream=True)
        resp.raise_for_status()
        tmp_path.parent.mkdir(parents=True, exist_ok=True)
        with open(tmp_path, "wb") as f:
            for chunk in resp.iter_content(chunk_size=8192):
                f.write(chunk)
        return tmp_path

    def __iter__(self) -> Iterator[Dict[str, Any]]:
        cache_dir = Path(".cache/cdn_parquet") / self.manifest.repo.replace("/", "_") / self.manifest.date
        for file_path in tqdm(self.manifest.files, desc="CDN shards"):
            tmp_path = cache_dir / Path(file_path).name
            if not tmp_path.exists():
                self._download_shard(file_path, tmp_path)
            try:
                table = pq.read_table(tmp_path, columns=self.columns)
                for batch in table.to_batches(max_chunksize=1024):
                    cols = {c: batch.column(c).to_pylist() for c in self.columns}
                    for i in range(len(cols[self.columns[0]])):
                        yield {c: cols[c][i] for c in self.columns}
            except Exception as exc:
                logger.warning("Failed to read %s: %s", file_path, exc)
                continue
```

### 4) Update training script to use manifest + CDN loader

In `surrogate/train.py` (or existing entrypoint), replace:

```python
# OLD (remove)
# from datasets import load_dataset
# dataset = load_dataset("axentx/surrogate-mirror", split="train", streaming=True)
```

with:

```python
# NEW
import logging
from pathlib import Path

from surrogate.data.cdn_dataset import CDNManifest, CDNParquetIterable

logging.basicConfig(level=logging.INFO)

MANIFEST_PATH = Path("surrogate/manifest.json")
if not MANIFEST_PATH.exists():
    raise RuntimeError(
        "Manifest not found. Run on Mac first:\n"
        "  HF_DATASET_REPO=axentx/surrogate-mirror "
        "DATE_FOLDER=YYYY-MM-DD "
        "MANIFEST_OUT=surrogate/manifest.json "
        "python surrogate/scripts/build_manifest.py"
    )

manifest = CDNManifest.load(MANIFEST_PATH)
train_iter = CDNParquetIterable(manifest, columns=("prompt", "response"))

# Example usage in training loop
for i, row in enumerate(train_iter):
    if i >= 10:
        break
    print(row)
```
