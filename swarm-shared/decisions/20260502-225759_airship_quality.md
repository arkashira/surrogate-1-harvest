# airship / quality

## Implementation Plan — `airship discover` quality hardening (<2h)

### Scope
Ship a zero-runtime-dependency, CDN-safe `airship discover` orchestrator (Bash-first) that deterministically emits `status.json` + `manifest.json` for local/CI use and CDN upload. Frontends fetch via CDN URIs.

### Highest-value incremental improvement
Make `airship discover` safe for CI/CD and CDN publishing by:
- enforcing deterministic output (sorted keys, stable hashes)
- using CDN URLs for public datasets (bypass HF API rate limits)
- emitting machine-readable `status.json` + `manifest.json` with content-addressable filenames
- validating schema and permissions before upload
- idempotent runs (skip unchanged outputs)

### Implementation steps (≤2h)
1. Create `bin/airship-discover` (Bash, shebang, executable) — orchestrator entrypoint.
2. Implement deterministic JSON normalization via `jq` (sorted keys).
3. Implement file discovery + CDN URL projection (HF CDN bypass).
4. Emit `status.json` (run metadata + per-file status) and `manifest.json` (content-addressable mapping).
5. Add idempotency check via content hash (skip upload if file exists with same hash).
6. Validate permissions and CDN reachability before upload.
7. Minimal smoke test in CI path.

#!/usr/bin/env bash
# airship discover — zero-runtime-dependency orchestrator
# Usage: bin/airship-discover [--upload] [--date YYYY-MM-DD]
# Emits: status.json, manifest.json
set -euo pipefail
IFS=$'\n\t'

# ---- config ----
REPO_ROOT="$(cd "$(dirname "${BASH_SOURCE[0]}")/.." && pwd)"
OUT_DIR="${REPO_ROOT}/dist/discover"
DATE="${2:-$(date +%Y-%m-%d)}"
UPLOAD="${1:-}"
HF_REPO="datasets/axentx/airship-mirror"
CDN_BASE="https://huggingface.co/datasets/${HF_REPO}/resolve/main"

mkdir -p "${OUT_DIR}/${DATE}"

# ---- helpers ----
normalize_json() {
  # deterministic: sorted keys, no whitespace variance
  jq -S '.' "$1" > "$1.tmp" && mv "$1.tmp" "$1"
}

content_hash() {
  sha256sum "$1" | awk '{print $1}'
}

cdn_url() {
  # path relative to dataset root
  local rel_path="$1"
  echo "${CDN_BASE}/${rel_path}"
}

# ---- discovery ----
# Single API call from Mac (run when rate-limit window clear) to list date folder.
# Save list to JSON; embed in training script for CDN-only fetches during training.
# This script expects a pre-generated file list for the date (or will create one via HF API if missing).
FILE_LIST="${OUT_DIR}/${DATE}/file-list.json"
if [[ ! -f "${FILE_LIST}" ]]; then
  echo "INFO: file-list missing; generating via HF API (may be rate-limited)."
  # Requires HF_TOKEN in env for API; fallback to CDN-only mode if unavailable.
  if [[ -n "${HF_TOKEN:-}" ]]; then
    curl -s -H "Authorization: Bearer ${HF_TOKEN}" \
      "https://huggingface.co/api/datasets/${HF_REPO}/tree/${DATE}?recursive=false" \
      | jq '[.[] | select(.type=="file") | .path]' > "${FILE_LIST}"
  else
    echo "WARN: HF_TOKEN not set — creating empty file-list (CDN-only mode)."
    echo '[]' > "${FILE_LIST}"
  fi
fi

# ---- build status + manifest ----
STATUS_FILE="${OUT_DIR}/${DATE}/status.json"
MANIFEST_FILE="${OUT_DIR}/${DATE}/manifest.json"

# Start status
cat > "${STATUS_FILE}" <<EOF
{
  "run_id": "$(date -u +%Y%m%dT%H%M%SZ)-$(head -c 8 /dev/urandom | xxd -p)",
  "date": "${DATE}",
  "created_at": "$(date -u +%Y-%m-%dT%H:%M:%SZ)",
  "repo": "${HF_REPO}",
  "cdn_base": "${CDN_BASE}",
  "status": "running",
  "files": []
}
EOF

# Manifest accumulator
echo '{"files":[]}' > "${MANIFEST_FILE}"

mapfile -t PATHS < <(jq -r '.[]' "${FILE_LIST}")
for rel_path in "${PATHS[@]}"; do
  filename="$(basename "${rel_path}")"
  cdn_href="$(cdn_url "${rel_path}")"

  # Try to fetch via CDN (no auth) to validate availability
  tmp_dl="${OUT_DIR}/${DATE}/.tmp_${filename}"
  if curl -fsSL --max-time 10 "${cdn_href}" -o "${tmp_dl}" 2>/dev/null; then
    hash=$(content_hash "${tmp_dl}")
    size=$(wc -c < "${tmp_dl}" | tr -d ' ')
    mime=$(file --mime-type -b "${tmp_dl}" 2>/dev/null || echo "application/octet-stream")

    # Idempotency: skip upload if file already present with same hash
    target_path="batches/mirror-merged/${DATE}/${hash}.parquet"
    target_cdn="$(cdn_url "${target_path}")"

    upload_needed="false"
    if [[ "${UPLOAD}" == "--upload" ]]; then
      # Check remote existence via CDN HEAD (best-effort)
      if curl -fsSL -I --max-time 5 "${target_cdn}" >/dev/null 2>&1; then
        remote_hash=$(curl -s -H "Authorization: Bearer ${HF_TOKEN:-}" \
          "https://huggingface.co/api/datasets/${HF_REPO}/blobs/${hash}" 2>/dev/null || echo "")
        if [[ "${remote_hash}" == "${hash}" ]]; then
          upload_needed="false"
        else
          upload_needed="true"
        fi
      else
        upload_needed="true"
      fi

      if [[ "${upload_needed}" == "true" ]]; then
        if [[ -n "${HF_TOKEN:-}" ]]; then
          echo "INFO: uploading ${filename} -> ${target_path}"
          # Use HF API upload (lfs) — keep within commit cap by batching or sibling repos if needed.
          # For simplicity, we use huggingface_hub CLI if available; otherwise skip.
          if command -v huggingface_hub >/dev/null 2>&1; then
            huggingface_hub upload \
              --repo-type dataset \
              --repo-id "${HF_REPO}" \
              "${tmp_dl}" "${target_path}" || true
          else
            echo "WARN: huggingface_hub CLI not installed — skipping upload."
          fi
        else
          echo "WARN: HF_TOKEN not set — skipping upload."
        fi
      else
        echo "INFO: ${target_path} already present (hash match) — skipping upload."
      fi
    fi

    # Record file entry
    entry=$(jq -n \
      --arg path "${rel_path}" \
      --arg filename "${filename}" \
      --arg cdn "${cdn_href}" \
      --arg target "${target_path}" \
      --arg hash "${hash}" \
      --arg mime "${mime}" \
      --argjson size "${size}" \
      '{
        path: $path,
        filename: $filename,
        cdn_url: $cdn,
        target_path: $target,
        hash: $hash,
        mime: $mime,
        size: $size,
        uploaded: ($target != null and $target != ""),
        status: "ok"
      }')
