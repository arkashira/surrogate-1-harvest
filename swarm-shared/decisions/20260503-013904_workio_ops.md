# workio / ops

## Final consolidated solution

**Core problem**: LINE webhook retries + concurrent deliveries create duplicate punches and inconsistent state because ingestion is not idempotent, not transactional, and not race-safe.

**Guiding principles**
- **Idempotency first**: reject duplicates deterministically before side effects.
- **Transactional + atomic**: punch + audit + notification state must commit or roll back together.
- **Race-safe state machine**: enforce at most one open punch per user/tenant with DB-level exclusion (application lock + unique constraint).
- **Observable + replayable**: request-scoped IDs, structured logs, and durable idempotency keys so ops can replay safely.
- **Fail-safe outbound**: retry transient LINE API failures without blocking webhook commit.

### 2. Service layer: idempotent, transactional, race-safe ingestion

File: `/opt/axentx/workio/server/src/services/punchService.ts`

```typescript
import { pool } from '../db';
import { Logger } from '../utils/logger';

const log = new Logger('punchService');

export interface PunchIngest {
  tenantId: number;
  userId: number;
  type: 'clock_in' | 'clock_out';
  lineEventId?: string | null;
  idempotencyKey: string;
  latitude?: number | null;
  longitude?: number | null;
  locationLabel?: string | null;
  deviceInfo?: string | null;
  timestamp?: Date;
}

export async function ingestPunch(payload: PunchIngest) {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // Idempotency: fast unique-key check
    const idem = await client.query(
      `SELECT id, type, status FROM punches WHERE idempotency_key = $1`,
      [payload.idempotencyKey]
    );
    if (idem.rowCount > 0) {
      log.info({ idempotencyKey: payload.idempotencyKey }, 'duplicate punch ingestion');
      await client.query('ROLLBACK');
      return { id: idem.rows[0].id, duplicate: true, type: idem.rows[0].type };
    }

    // Optional: also dedupe by line_event_id when present
    if (payload.lineEventId) {
      const lineDup = await client.query(
        `SELECT id, type, status FROM punches WHERE line_event_id = $1 AND tenant_id = $2`,
        [payload.lineEventId, payload.tenantId]
      );
      if (lineDup.rowCount > 0) {
        log.info({ lineEventId: payload.lineEventId }, 'duplicate LINE event ingestion');
        await client.query('ROLLBACK');
        return { id: lineDup.rows[0].id, duplicate: true, type: lineDup.rows[0].type };
      }
    }

    // State-machine safety: ensure business rules hold under concurrency
    if (payload.type === 'clock_in') {
      // Must not have an open punch
      const open = await client.query(
        `SELECT id FROM punches
         WHERE tenant_id = $1 AND user_id = $2 AND status = 'open'
         LIMIT 1`,
        [payload.tenantId, payload.userId]
      );
      if (open.rowCount > 0) {
        await client.query('ROLLBACK');
        throw new Error('OPEN_PUNCH_EXISTS');
      }
    } else {
      // clock_out: must have exactly one open punch to close
      const open = await client.query(
        `SELECT id FROM punches
         WHERE tenant_id = $1 AND user_id = $2 AND status = 'open'
         ORDER BY created_at DESC
         LIMIT 1`,
        [payload.tenantId, payload.userId]
      );
      if (open.rowCount === 0) {
        await client.query('ROLLBACK');
        throw new Error('NO_OPEN_PUNCH_TO_CLOSE');
      }
    }

    const now = payload.timestamp || new Date();
    const insert = await client.query(
      `INSERT INTO punches (
         tenant_id, user_id, type, status,
         latitude, longitude, location_label, device_info,
         line_event_id, idempotency_key,
         created_at, updated_at
       ) VALUES ($1,$2,$3,'open',$4,$5,$6,$7,$8,$9,$10,$10)
       RETURNING id`,
      [
        payload.tenantId,
        payload.userId,
        payload.type,
        payload.latitude ?? null,
        payload.longitude ?? null,
        payload.locationLabel ?? null,
        payload.deviceInfo ?? null,
        payload.lineEventId ?? null,
        payload.idempotencyKey,
        now
      ]
    );

    // Audit trail (same transaction)
    await client.query(
      `INSERT INTO punch_audits (punch_id, tenant_id, user_id, action, metadata, created_at)
       VALUES ($1,$2,$3,'ingest',$4,$5)`,
      [insert.rows[0].id, payload.tenantId, payload.userId, payload, now]
    );

    await client.query('COMMIT');
    return { id: insert.rows[0].id, duplicate: false };
  } catch (err) {
    await client.query('ROLLBACK');
    log.error({ err, payload }, 'ingestPunch failed');
    throw err;
  } finally {
    client.release();
  }
}
```
