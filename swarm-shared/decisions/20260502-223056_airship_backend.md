## Implementation Plan — `airship discover` (backend)

**Highest-value incremental improvement (<2h):**  
Ship a single, cron-safe orchestrator (`bin/airship-discover`) that:
1. Runs `granite-business-research.sh` (if present) and tags outputs `#business-research #knowledge-rag #graph`.
2. Executes `knowledge-rag` to query the top hub (e.g., "MOC") and emit tagged insights.
3. Applies all past lessons: Bash shebang, executable bit, `SHELL=/bin/bash` in crontab, no Mac-side heavy compute, and safe HF CDN bypass patterns for any dataset fetches.

### 1) Library: `/opt/axentx/airship/lib/airship-discover.sh`

```bash
#!/usr/bin/env bash
# airship-discover library
# Usage: source this file; call airship_discover_main "$@"

set -euo pipefail

readonly AIRSHIP_ROOT="$(cd "$(dirname "${BASH_SOURCE[0]}")/.." && pwd)"
readonly LOG_TAG="airship-discover"

log() {
  echo "[$(date -u +"%Y-%m-%dT%H:%M:%SZ")] [$LOG_TAG] $*" >&2
}

tag_output() {
  local tags="$1"
  local file="$2"
  if [[ -f "$file" ]]; then
    # Append tags as YAML frontmatter-style block if file looks text-y
    if head -1 "$file" | grep -qE '^[[:space:]]*#'; then
      sed -i.bak "1s/^/# Tags: ${tags}\n/" "$file"
    else
      echo "# Tags: ${tags}" | cat - "$file" > "${file}.tmp" && mv "${file}.tmp" "$file"
    fi
    log "Tagged $file with: $tags"
  fi
}

run_granite_business_research() {
  local script="${AIRSHIP_ROOT}/granite-business-research.sh"
  if [[ -x "$script" ]]; then
    log "Running granite-business-research.sh"
    local out_dir="${AIRSHIP_ROOT}/outputs/business-research"
    mkdir -p "$out_dir"
    local stamp
    stamp="$(date -u +"%Y%m%dT%H%M%SZ")"
    local outfile="${out_dir}/research-${stamp}.md"
    if "$script" > "$outfile" 2>&1; then
      tag_output "#business-research #knowledge-rag #graph" "$outfile"
      log "Research output saved to $outfile"
    else
      log "WARNING: granite-business-research.sh exited non-zero (see $outfile)"
    fi
  else
    log "No executable granite-business-research.sh found at $script — skipping"
  fi
}

run_knowledge_rag_top_hub() {
  local cmd="${AIRSHIP_ROOT}/knowledge-rag"
  if [[ -x "$cmd" ]]; then
    log "Running knowledge-rag to query top hub (e.g., MOC)"
    local out_dir="${AIRSHIP_ROOT}/outputs/knowledge-rag"
    mkdir -p "$out_dir"
    local stamp
    stamp="$(date -u +"%Y%m%dT%H%M%SZ")"
    local outfile="${out_dir}/top-hub-${stamp}.md"

    # Try common query patterns; prefer non-interactive/JSON flags if available.
    # Fallback to simple query string.
    if "$cmd" --query "top hub" --format markdown > "$outfile" 2>/dev/null \
       || "$cmd" "top hub" > "$outfile" 2>/dev/null \
       || "$cmd" --top-hub > "$outfile" 2>/dev/null; then
      tag_output "#knowledge-rag #graph #hub" "$outfile"
      log "Top-hub insight saved to $outfile"
    else
      log "WARNING: knowledge-rag failed to produce output"
      rm -f "$outfile"
    fi
  else
    log "No executable knowledge-rag found at $cmd — skipping"
  fi
}

safe_hf_file_list() {
  # Optional helper: pre-list one date folder to avoid HF API rate limits during training.
  # Usage: safe_hf_file_list <repo> <date-folder> > file-list.json
  # Example: safe_hf_file_list "datasets/myorg/mirror-merged" "2026-04-29"
  local repo="$1"
  local folder="$2"
  local out="${3:-/dev/stdout}"

  if command -v huggingface-cli >/dev/null 2>&1; then
    # Use non-recursive tree to avoid pagination explosion.
    huggingface-cli repo tree "$repo" --path "$folder" --recursive false --json > "$out" 2>/dev/null || true
  else
    log "huggingface-cli not available — skipping HF file-list generation"
  fi
}

airship_discover_main() {
  log "Starting airship-discover"

  # Core tasks
  run_granite_business_research
  run_knowledge_rag_top_hub

  # Optional HF pre-listing (best-effort) — useful for downstream surrogate training scripts.
  # Uncomment and adjust repo/folder as needed:
  # local hf_out="${AIRSHIP_ROOT}/outputs/hf-file-list/$(date -u +"%Y-%m-%d").json"
  # mkdir -p "$(dirname "$hf_out")"
  # safe_hf_file_list "datasets/myorg/mirror-merged" "$(date -u +"%Y-%m-%d")" "$hf_out"

  log "Finished airship-discover"
}
```

### 2) Orchestrator: `/opt/axentx/airship/bin/airship-discover`

```bash
#!/usr/bin/env bash
# airship-discover — cron-safe orchestrator
# Intended for crontab:
#   SHELL=/bin/bash
#   0 6 * * * /opt/axentx/airship/bin/airship-discover >> /var/log/airship-discover.log 2>&1

set -euo pipefail

readonly AIRSHIP_ROOT="$(cd "$(dirname "${BASH_SOURCE[0]}")/.." && pwd)"

# Ensure library is available
if [[ -f "${AIRSHIP_ROOT}/lib/airship-discover.sh" ]]; then
  # shellcheck source=../lib/airship-discover.sh
  source "${AIRSHIP_ROOT}/lib/airship-discover.sh"
else
  echo "FATAL: airship-discover.sh not found" >&2
  exit 1
fi

# Run main
airship_discover_main "$@"
```

Make executable:
```bash
chmod +x /opt/axentx/airship/bin/airship-discover
chmod +x /opt/axentx/airship/lib/airship-discover.sh
```
