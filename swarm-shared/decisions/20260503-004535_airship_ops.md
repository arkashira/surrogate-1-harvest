## Implementation Plan: CDN-Only Surrogate-1 Training Loader (Airship Ops)

**Highest-value improvement**: Eliminate HF API rate-limit failures and `pyarrow.CastError` during Surrogate-1 training by replacing `load_dataset(streaming=True)` with a manifest-driven, CDN-only data loader. This is a <2h ops win that unblocks training and removes recurring 429/cast errors.

**Scope**: Add one manifest generator (Mac/Linux orchestration) + one Lightning-compatible dataset wrapper (training). No schema changes, no new dependencies.

### 1) Manifest generator (run on Mac after rate-limit window clears)

File: `/opt/axentx/airship/scripts/build_hf_cdn_manifest.py`

```python
#!/usr/bin/env python3
"""
Generate file manifest for CDN-only training.
Usage:
  bash scripts/build_hf_cdn_manifest.py \
    --repo datasets/your-org/surrogate-ingest \
    --date 2026-04-29 \
    --out surrogate/file_manifest_2026-04-29.json
"""
import argparse
import json
import os
import sys
from datetime import datetime

try:
    from huggingface_hub import list_repo_tree
except ImportError:
    print("ERROR: huggingface_hub not installed. Run: pip install huggingface_hub")
    sys.exit(1)

CDN_BASE = "https://huggingface.co/datasets"

def build_manifest(repo: str, date_folder: str, output_path: str):
    """
    List single date folder (non-recursive) and emit CDN URLs.
    Avoids recursive list_repo_files (paginated) and API calls during training.
    """
    print(f"[{datetime.utcnow().isoformat()}] Listing {repo}/{date_folder} ...")

    # list_repo_tree(path, recursive=False) => one page, cheap
    entries = list_repo_tree(repo=repo, path=date_folder, recursive=False)

    files = []
    for entry in entries:
        if entry.type != "file":
            continue
        if not entry.path.lower().endswith(".parquet"):
            continue

        cdn_url = f"{CDN_BASE}/{repo}/resolve/main/{entry.path}"
        files.append({
            "repo": repo,
            "path": entry.path,
            "cdn_url": cdn_url,
            "size": getattr(entry, "size", None)
        })

    manifest = {
        "repo": repo,
        "date_folder": date_folder,
        "generated_at": datetime.utcnow().isoformat() + "Z",
        "count": len(files),
        "files": files
    }

    os.makedirs(os.path.dirname(os.path.abspath(output_path)), exist_ok=True)
    with open(output_path, "w") as f:
        json.dump(manifest, f, indent=2)

    print(f"[{datetime.utcnow().isoformat()}] Wrote {len(files)} files -> {output_path}")
    return manifest

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Build HF CDN manifest for training")
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g. datasets/org/name)")
    parser.add_argument("--date", required=True, help="Date folder (e.g. 2026-04-29)")
    parser.add_argument("--out", required=True, help="Output JSON path")
    args = parser.parse_args()

    build_manifest(args.repo, args.date, args.out)
```

Make executable:

```bash
chmod +x /opt/axentx/airship/scripts/build_hf_cdn_manifest.py
```

### 2) Lightning-compatible dataset (CDN-only, no HF API during training)

File: `/opt/axentx/airship/surrogate/data/cdn_parquet_dataset.py`

```python
import json
import pyarrow as pa
import pyarrow.parquet as pq
import requests
import torch
from torch.utils.data import Dataset, IterableDataset
from typing import List, Dict, Optional

class CDNParquetIterable(IterableDataset):
    """
    CDN-only Parquet loader for Surrogate-1 training.
    - Uses manifest produced by build_hf_cdn_manifest.py
    - Zero HF API calls during training (bypasses 429)
    - Projects to {prompt, response} on parse to avoid pyarrow.CastError
      from mixed-schema repos
    """

    def __init__(self, manifest_path: str, columns=("prompt", "response"), max_files: Optional[int] = None):
        super().__init__()
        with open(manifest_path) as f:
            manifest = json.load(f)

        self.files = [f["cdn_url"] for f in manifest["files"]]
        if max_files:
            self.files = self.files[:max_files]
        self.columns = columns

    def _stream_parquet(self, url: str):
        # CDN download (no auth) with retry/backoff handled by requests
        resp = requests.get(url, timeout=60)
        resp.raise_for_status()

        with pa.BufferReader(resp.content) as buf:
            table = pq.read_table(buf, columns=self.columns)

        # Normalize to prompt/response (best-effort; drop rows missing either)
        if "prompt" not in table.column_names or "response" not in table.column_names:
            # If schema differs, try common aliases once (then skip)
            return

        for i in range(table.num_rows):
            row = {c: table[c][i].as_py() for c in self.columns}
            if row["prompt"] is None or row["response"] is None:
                continue
            yield row

    def __iter__(self):
        worker_info = torch.utils.data.get_worker_info()
        if worker_info is None:
            files = self.files
        else:
            per_worker = len(self.files) // worker_info.num_workers
            files = self.files[worker_info.id * per_worker : (worker_info.id + 1) * per_worker]

        for url in files:
            try:
                yield from self._stream_parquet(url)
            except Exception as exc:
                # Log and skip corrupt/unavailable files; don't kill training
                print(f"[WARN] Failed to load {url}: {exc}")
                continue


class CDNParquetDataset(Dataset):
    """
    Lightweight map-style wrapper for small deterministic runs / validation.
    Prefetch all rows into memory (use only for val/small sets).
    """

    def __init__(self, manifest_path: str, columns=("prompt", "response"), max_files: Optional[int] = None):
        super().__init__()
        with open(manifest_path) as f:
            manifest = json.load(f)

        self.rows = []
        files = [f["cdn_url"] for f in manifest["files"]]
        if max_files:
            files = files[:max_files]

        for url in files:
            try:
                resp = requests.get(url, timeout=60)
                resp.raise_for_status()
                with pa.BufferReader(resp.content) as buf:
                    table = pq.read_table(buf, columns=columns)

                if "prompt" not in table.column_names or "response" not in table.column_names:
                    continue

                for i in range(table.num_rows):
                    prompt = table["prompt"][i].as_py()
                    response = table["response"][i].as_py()
                    if prompt is None or response is None:
                        continue
                    self.rows.append({"prompt": prompt, "response": response})
            except Exception as exc:
                print(f"[WARN] Failed to load {url}: {exc}")
                continue

    def __len__(self):
        return len(self.rows)

    def __getitem__(self, idx):
        return self.rows[idx]
```

### 3) Minimal Lightning training script integration

File: `/opt/axentx/airship/surrogate/train_cdn.py` (excerpt)

```python
from pathlib import Path
from torch.utils.data import DataLoader
from surrogate.data.cdn_parquet_dataset import CDNParquetIterable

MANIFEST = Path("surrogate/file_manifest_2026-04-29.json")

def train_dataloader():
    dataset = CDNParquetIterable(
        manifest_path=str(MANIFEST),
        columns=("prompt", "response"),
        max_files=None
    )
    return DataLoader(dataset, batch_size=8, num_workers=4)
