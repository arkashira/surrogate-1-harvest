# Costinel / ops

## Final Implementation Plan — Top-Hub Signal Panel (CDN-first, <2h, telemetry-aware)

### What we ship
- A **non-blocking Top-Hub Signal Panel** on the Costinel dashboard.
- Defaults to hub **MOC** (configurable via `HUB_NAME`).
- Shows **3 actionable, cost-impact proposals** (title, impact, confidence, action).
- **CDN-first data fetch** using a pre-generated static JSON served from CDN (with optional server-side fetch script to refresh it).
- Telemetry events (`panel_impression`, `proposal_click`) for observability.
- Graceful fallback to static placeholder if CDN unavailable.
- Respects idle-stop / quota patterns and requires **zero model execution** on developer Macs.

---

### Why this is highest-value (<2h)
- Pure frontend + optional one-shot fetcher; no backend or core pipeline changes.
- Reuses existing MOC hub insight pattern and CDN-bypass/quota-saving patterns.
- Immediate UX value for ops users.
- Avoids HF API rate limits during panel updates by relying on CDN-hosted JSON.
- Keeps Mac compute unused; any server-side refresh runs in CI/Lightning Studio.

### File changes

#### 1) Add panel component: `src/components/TopHubSignalPanel.tsx`
```tsx
import { useEffect, useState } from "react";
import "./TopHubSignalPanel.css";

interface Proposal {
  id: string;
  title: string;
  impact: string;      // e.g. "-$12k/mo"
  confidence: number;  // 0-1
  action: string;      // e.g. "Convert to Savings Plan"
  href?: string;
}

interface HubData {
  hub: string;
  updated: string;
  proposals: Proposal[];
}

const CDN_BASE = "https://huggingface.co/datasets/axentx/costinel-hubs/resolve/main";
const HUB_NAME = (window as any).COSTINEL_HUB_NAME || "MOC";
// Primary: per-hub static file (CDN). Fallback to built-in placeholder.
const CDN_URL = `${CDN_BASE}/${HUB_NAME}.json`;

export default function TopHubSignalPanel() {
  const [data, setData] = useState<HubData | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(false);

  useEffect(() => {
    const load = async () => {
      try {
        const res = await fetch(CDN_URL, { cache: "no-cache" });
        if (!res.ok) throw new Error("CDN fetch failed");
        const json = await res.json();
        setData(json);
        track("panel_impression", { hub: json.hub, count: json.proposals?.length });
      } catch {
        setError(true);
        // Minimal, deterministic fallback so panel is always useful
        setData({
          hub: HUB_NAME,
          updated: new Date().toISOString(),
          proposals: [
            {
              id: "fallback-1",
              title: "Review idle EC2 instances in us-east-1",
              impact: "-$8k/mo",
              confidence: 0.78,
              action: "Schedule stop for dev nights",
            },
            {
              id: "fallback-2",
              title: "Increase RDS Aurora storage auto-grow threshold",
              impact: "-$3k/mo",
              confidence: 0.65,
              action: "Set 25% buffer policy",
            },
            {
              id: "fallback-3",
              title: "Right-size over-provisioned EKS node groups",
              impact: "-$5k/mo",
              confidence: 0.72,
              action: "Apply Cx4 to C3 migration",
            },
          ],
        });
      } finally {
        setLoading(false);
      }
    };

    load();
  }, []);

  function track(event: string, payload: any) {
    // Lightweight telemetry; replace with your analytics endpoint if needed
    try {
      navigator.sendBeacon?.("/_telemetry", JSON.stringify({ event, payload, ts: Date.now() }));
    } catch {}
  }

  const proposals = data?.proposals || [];

  return (
    <div className="top-hub-panel" role="region" aria-label="Top Hub signals">
      <div className="top-hub-header">
        <div>
          <span className="top-hub-badge">Top Hub</span>
          <span className="top-hub-name">{data?.hub || HUB_NAME}</span>
        </div>
        <div className="top-hub-meta">
          {loading && <span className="top-hub-loading">Loading signals…</span>}
          {error && <span className="top-hub-error">Using fallback signals</span>}
          {!loading && !error && <span className="top-hub-updated">Updated {new Date(data.updated).toLocaleDateString()}</span>}
        </div>
      </div>

      <div className="top-hub-list" aria-live="polite">
        {proposals.slice(0, 3).map((p) => (
          <a
            key={p.id}
            className="top-hub-item"
            href={p.href || "#"}
            onClick={() => track("proposal_click", { id: p.id, hub: data?.hub })}
            target={p.href ? "_blank" : undefined}
            rel={p.href ? "noopener noreferrer" : undefined}
          >
            <div className="top-hub-item-title">{p.title}</div>
            <div className="top-hub-item-row">
              <span className="top-hub-impact">{p.impact}</span>
              <span className="top-hub-confidence">{Math.round(p.confidence * 100)}%</span>
            </div>
            <div className="top-hub-action">{p.action}</div>
          </a>
        ))}
      </div>
    </div>
  );
}
```
