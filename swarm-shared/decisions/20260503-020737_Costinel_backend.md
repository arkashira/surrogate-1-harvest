### 3) Frontend: hook + component (TypeScript)

`src/hooks/useTopHubSignal.ts`

```ts
import { useEffect, useState, useCallback } from "react";

export interface Signal {
  id: string;
  type: "recommendation" | "anomaly" | "opportunity";
  severity: "low" | "medium" | "high";
  title: string;
  action: string;
  context: string;
}

export interface Proposal {
  id: string;
  title: string;
  status: string;
  createdBy: string;
  createdAt: string;
  impact: {
    estimatedMonthlySavingsUSD: number;
    estimatedCoverageIncreasePct: number;
  };
  auditTrail: Array<{ ts: string; actor: string; action: string }>;
}

export interface HubData {
  hub: string;
  title: string;
  description: string;
  lastUpdated: string;
  signals: Signal[];
  proposals: Proposal[];
}

const HUB_DATA_URL = `${process.env.PUBLIC_URL}/data/hubs/top-hub-moc.json`;

export default function useTopHubSignal() {
  const [data, setData] = useState<HubData | null>(null);
  const [loading, setLoading] = useState<boolean>(true);
  const [error, setError] = useState<string | null>(null);

  const fetchData = useCallback(async () => {
    setLoading(true);
    setError(null);
    try {
      const res = await fetch(HUB_DATA_URL, { cache: "no-store" });
      if (!res.ok) throw new Error(`Failed to load hub data: ${res.status}`);
      const json = (await res.json()) as HubData;
      setData(json);
    } catch (err: any) {
      setError(err.message || "Unknown error");
    } finally {
      setLoading(false);
    }
  }, []);

  useEffect(() => {
    fetchData();
  }, [fetchData]);

  return { data, loading, error, refetch: fetchData };
}
```
