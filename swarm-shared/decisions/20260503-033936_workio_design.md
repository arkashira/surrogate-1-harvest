# workio / design

## Final synthesized answer

### Diagnosis (merged, prioritized)
- **Idempotency gap**: LINE retries on 5xx/timeouts can replay webhooks and corrupt clock/leave/OT state.
- **Silent state divergence**: LINE-driven DB updates are not pushed to the frontend; UI shows stale data until refresh.
- **Signature verification fragility**: verification must run on the raw request body before any parser mutates it; otherwise security/reliability breaks.
- **Missing transactional boundary**: partial application (DB commit but downstream failure) leaves user-visible state inconsistent.
- **No audit/compliance trail**: no immutable log of LINE events or state changes for debugging or compliance.

### Proposed change (single coherent plan)
Implement an **idempotent, transactional LINE webhook processor** with:
- reusable signature verification on raw body,
- 10-minute deduplication window keyed by `(tenant_id, line_event_id)`,
- atomic DB updates + audit row,
- realtime push to frontend via WebSocket,
- centralized business-rule service for clock/leave/OT.

### Implementation (concrete, merged, corrected)

#### 2) Reusable signature verification (raw-body safe)
`server/src/middleware/lineAuth.ts`
```ts
import crypto from 'crypto';

export function verifyLineSignature(
  rawBody: Buffer,
  signature: string,
  channelSecret: string
): boolean {
  const expected = crypto
    .createHmac('sha256', channelSecret)
    .update(rawBody)
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}
```

#### 3) Lightweight WebSocket broadcast per tenant
`server/src/lib/ws.ts`
```ts
import { WebSocket } from 'ws';

const channels = new Map<string, Set<WebSocket>>(); // tenant_id -> sockets

export function joinTenant(tenantId: string, ws: WebSocket) {
  if (!channels.has(tenantId)) channels.set(tenantId, new Set());
  channels.get(tenantId)!.add(ws);
  ws.on('close', () => channels.get(tenantId)?.delete(ws));
}

export function broadcastTenant(tenantId: string, data: any) {
  const payload = JSON.stringify(data);
  for (const ws of channels.get(tenantId) || []) {
    if (ws.readyState === 1) ws.send(payload);
  }
}
```

#### 4) Centralized, transactional LINE processor
`server/src/services/lineProcessor.ts`
```ts
import { Pool, Client } from 'pg';
import { broadcastTenant } from '../lib/ws';
import { verifyLineSignature } from '../middleware/lineAuth';

const pool = new Pool();
const RETRY_WINDOW_MS = 10 * 60 * 1000; // 10m dedupe window

export async function processLineEvent(
  tenantId: string,
  rawBody: Buffer,
  signature: string,
  channelSecret: string,
  lineEvent: any
) {
  if (!verifyLineSignature(rawBody, signature, channelSecret)) {
    throw new Error('Invalid LINE signature');
  }

  const client: Client = await pool.connect();
  try {
    await client.query('BEGIN');

    // Idempotency check
    const dup = await client.query(
      `SELECT 1 FROM line_webhook_events
       WHERE tenant_id = $1 AND line_event_id = $2`,
      [tenantId, lineEvent.id]
    );
    if (dup.rows.length > 0) {
      await client.query('ROLLBACK');
      return { ok: true, duplicate: true };
    }

    // Record event (audit)
    await client.query(
      `INSERT INTO line_webhook_events
       (tenant_id, line_event_id, event_type, source_type, source_id, payload)
       VALUES ($1,$2,$3,$4,$5,$6)`,
      [
        tenantId,
        lineEvent.id,
        lineEvent.type,
        lineEvent.source?.type,
        lineEvent.source?.userId,
        lineEvent,
      ]
    );

    // Apply business rules and mutate domain state (clock/leave/OT)
    const result = await applyLineEvent(client, tenantId, lineEvent);

    // Mark processed
    await client.query(
      `UPDATE line_webhook_events SET processed_at = now()
       WHERE tenant_id = $1 AND line_event_id = $2`,
      [tenantId, lineEvent.id]
    );

    await client.query('COMMIT');

    // Realtime push to tenant clients
    broadcastTenant(tenantId, {
      type: 'line_event_processed',
      event: lineEvent.type,
      result,
      ts: new Date().toISOString(),
    });

    return { ok: true, result };
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  } finally {
    client.release();
  }
}

async function applyLineEvent(client: Client, tenantId: string, ev: any) {
  // Centralized business rules (examples)
  switch (ev.type) {
    case 'message':
      return handleMessageCommand(client, tenantId, ev);
    case 'postback':
      return handlePostback(client, tenantId, ev);
    default:
      return { ignored: true };
  }
}

async function handleMessageCommand(client: Client, tenantId: string, ev: any) {
  const text = ev.message?.text?.trim()?.toLowerCase();
  const userId = ev.source?.userId;
  if (!userId) return { error: 'invalid_source' };

  // Fetch employee
  const emp = await client.query(
    `SELECT id, role, last_clock_in, last_clock_out FROM employees
     WHERE tenant_id = $1 AND line_user_id = $2`,
    [tenantId, userId]
  );

  if (emp.rows.length === 0) {
    return { error: 'employee_not_found' };
  }

  const e = emp.rows[0];

  if (text === 'clock in') {
    // Rule: cannot clock in twice without clock out
    if (e.last_clock_in && !e.last_clock_out) {
      return { error: 'already_clocked_in' };
    }
    await client.query(
      `INSERT INTO time_logs (tenant_id, employee_id, type, ts, location)
       VALUES ($1,$2,'clock_in',now(),'LINE')`,
      [tenantId, e.id]
    );
    await client.query(
      `UPDATE employees SET last_clock_in = now() WHERE id = $1`,
      [e.id]
    );
    return { clocked: 'in' };
  }

  if (text === 'clock out') {
    if (!e.last_clock_in || e.last_clock_out) {
      return { error: 'cannot_clock_out' };
    }
    await client.query(
      `INSERT INTO time_logs (tenant_id, employee_id, type, ts, location)
       VALUES ($1,$2,'clock_out',now(),'LINE')`,
      [tenantId, e.id]
    );
    await client.query(
      `UPDATE employees SET last_clock_out = now() WHERE id = $1`,
      [e.id]
    );
    return { clocked: 'out' };
  }

  // Add leave/OT rules here similarly (centralized)
  return { ignored: true };
}

async function handlePostback(client: Client, tenantId: string, ev: any) {
  // Centralized postback handling (e.g., approve/reject OT/leave)
  return { ignored: true };
}
```
