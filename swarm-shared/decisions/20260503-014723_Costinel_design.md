# Costinel / design

## Final Implementation Plan — Top-Hub Signal Panel (Costinel Dashboard)

**Chosen stack**: Vue 3 + TypeScript + resilient fetch-first data layer  
**Scope**: Frontend-only, read-only panel that surfaces the most-connected hub (default “MOC”) and its actionable proposals from the knowledge graph. Resilient to missing backend; falls back to cache + local mock.  
**Estimated effort**: ~90–110 min (ship-ready)

### 3) Resilient data layer (`src/services/knowledgeGraph.ts`)

```ts
import type { TopHubResponse, HubInsight, Proposal } from '@/types/knowledgeGraph';

const CACHE_KEY = 'costinel:top-hub:cache';
const CACHE_TTL_MS = 5 * 60 * 1000; // 5m

function getCached(): TopHubResponse | null {
  try {
    const raw = localStorage.getItem(CACHE_KEY);
    if (!raw) return null;
    const parsed = JSON.parse(raw) as TopHubResponse & { _cachedAt?: number };
    if (!parsed._cachedAt || Date.now() - parsed._cachedAt > CACHE_TTL_MS) {
      localStorage.removeItem(CACHE_KEY);
      return null;
    }
    // Normalize shape
    return { hub: parsed.hub, proposals: parsed.proposals, updatedAt: parsed.updatedAt };
  } catch {
    return null;
  }
}

function setCached(data: TopHubResponse) {
  try {
    localStorage.setItem(
      CACHE_KEY,
      JSON.stringify({ ...data, _cachedAt: Date.now() })
    );
  } catch {
    // ignore storage failures
  }
}

export async function fetchTopHubInsight(params?: {
  hubName?: string;
  fallback?: boolean;
}): Promise<TopHubResponse> {
  const hubName = params?.hubName || 'MOC';
  const useFallback = params?.fallback !== false;

  // 1) Try cache first (fast, offline-friendly)
  const cached = getCached();
  if (cached) return cached;

  // 2) Try remote
  try {
    const res = await fetch(`/api/knowledge-graph/hubs/${encodeURIComponent(hubName)}/top`, {
      method: 'GET',
      headers: { 'Accept': 'application/json' },
      credentials: 'same-origin',
      signal: AbortSignal.timeout(8000),
    });

    if (!res.ok) throw new Error(`HTTP ${res.status}`);
    const json = (await res.json()) as TopHubResponse;
    // Basic normalization
    const normalized: TopHubResponse = {
      hub: {
        name: json.hub?.name || hubName,
        rank: json.hub?.rank,
        summary: json.hub?.summary || '',
        sourceLabel: json.hub?.sourceLabel || 'knowledge-graph',
        sourceUrl: json.hub?.sourceUrl,
        tags: json.hub?.tags,
        updatedAt: json.hub?.updatedAt || json.updatedAt || new Date().toISOString(),
      },
      proposals: Array.isArray(json.proposals)
        ? json.proposals.map((p) => ({
            id: p.id,
            title: p.title,
            description: p.description || p.summary || '',
            priority: p.priority,
            impact: p.impact,
            confidence: p.confidence,
            actions: p.actions,
            href: p.href,
          }))
        : [],
      updatedAt: json.updatedAt,
    };
    setCached(normalized);
    return normalized;
  } catch (err) {
    if (!useFallback) throw err;
    // 3) Fallback to mock
    return MOCK;
  }
}
```
