#### Step 3 — Add API route (src/api/v1/routes/cost_anomaly.py)

```python
from fastapi import APIRouter, Depends, Query
from datetime import date
from src.services.knowledge_rag import KnowledgeRagService
from src.models.schemas import TopHubSignal

router = APIRouter(prefix="/cost-anomaly", tags=["cost-anomaly"])

def get_knowledge_rag() -> KnowledgeRagService:
    # reuse or init service (consider DI if available)
    return KnowledgeRagService()

@router.get("/signal/top-hub", response_model=TopHubSignal)
def get_top_hub_signal(
    for_date: date = Query(None, description="Date (YYYY-MM-DD); defaults to today"),
    rag: KnowledgeRagService = Depends(get_knowledge_rag),
):
    """
    Return the strongest knowledge-graph hub as a cost-anomaly signal.
    Read-only. No execution. Sense + Signal.
    """
    return rag.get_top_hub_signal(for_date=for_date)
```
