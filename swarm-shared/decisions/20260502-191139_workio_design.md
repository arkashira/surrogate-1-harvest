**Core principles chosen across candidates**
- Use an **TTL-backed LRU tenant cache** keyed by `channelId` (Candidate 2’s TTL + eviction is safer than Candidate 1’s unbounded cache).  
- Cache lives in **tenant service**, not the webhook handler (clean separation; avoids per-handler cache duplication).

- Add **graceful shutdown** and **per-handler timeouts** (Candidate 2) to prevent mid-request truncation and cascading latency.

### 1) Add LRU cache utility

`server/src/lib/lruCache.ts`
```ts
type CacheEntry<T> = {
  value: T;
  expiresAt: number;
};

export class LRUCache<T> {
  private cache = new Map<string, CacheEntry<T>>();
  private maxSize: number;
  private defaultTtlMs: number;

  constructor(maxSize = 512, defaultTtlMs = 30_000) {
    this.maxSize = Math.max(1, maxSize);
    this.defaultTtlMs = Math.max(1, defaultTtlMs);
  }

  get(key: string): T | undefined {
    const entry = this.cache.get(key);
    if (!entry) return undefined;
    if (Date.now() > entry.expiresAt) {
      this.cache.delete(key);
      return undefined;
    }
    // refresh recency (LRU)
    this.cache.delete(key);
    this.cache.set(key, entry);
    return entry.value;
  }

  set(key: string, value: T, ttlMs = this.defaultTtlMs): void {
    if (this.cache.size >= this.maxSize) {
      // delete oldest (first inserted key)
      const oldestKey = this.cache.keys().next().value;
      if (oldestKey) this.cache.delete(oldestKey);
    }
    this.cache.set(key, { value, expiresAt: Date.now() + ttlMs });
  }

  delete(key: string): void {
    this.cache.delete(key);
  }

  clear(): void {
    this.cache.clear();
  }
}
```

### 2) Add cached tenant lookup in tenant service

`server/src/services/tenantService.ts`
```ts
import { LRUCache } from '../lib/lruCache';
import { pool } from '../db'; // adapt to your DB client

const tenantCache = new LRUCache<{ id: string; name: string; config?: any }>(512, 30_000);

export async function getTenantByChannelIdCached(channelId: string) {
  if (!channelId) return null;

  const cached = tenantCache.get(channelId);
  if (cached) return cached;

  // DB fallback
  const result = await pool.query(
    `SELECT id, name, config FROM tenants WHERE line_channel_id = $1 AND deleted_at IS NULL LIMIT 1`,
    [channelId]
  );

  if (result.rows.length === 0) return null;

  const tenant = result.rows[0];
  tenantCache.set(channelId, tenant);
  return tenant;
}

export function invalidateTenantCache(channelId: string) {
  tenantCache.delete(channelId);
}
```

### 3) Update LINE webhook handler (cached lookup + timeout + graceful-friendly)

```ts
import type { Request, Response } from 'express';
import { getTenantByChannelIdCached } from '../services/tenantService';
import { lineClientForTenant } from '../lib/lineClient'; // adapt

export async function handleLineWebhook(req: Request, res: Response) {
  const channelId =
    (req.headers['x-line-channel-id'] as string) ||
    (req.body && (req.body.destination as string));

  if (!channelId) {
    return res.status(400).json({ error: 'missing_channel_id' });
  }

  const tenant = await getTenantByChannelIdCached(channelId);
  if (!tenant) {
    return res.status(404).json({ error: 'tenant_not_found' });
  }

  // Defensive timeout for downstream LINE/DB calls
  const controller = new AbortController();
  const timeout = setTimeout(() => controller.abort(), 8_000);

  try {
    // TODO: process events (req.body.events)
    // Example:
    // await lineClientForTenant(tenant).replyMessage(..., { signal: controller.signal });
    res.status(200).json({ ok: true });
  } catch (err: any) {
    if (err.name === 'AbortError') {
      console.error('LINE webhook timeout', { channelId, tenantId: tenant.id });
      return res.status(504).json({ error: 'upstream_timeout' });
    }
    console.error('LINE webhook handler error', { channelId, tenantId: tenant.id, err });
    return res.status(500).json({ error: 'handler_error' });
  } finally {
    clearTimeout(timeout);
  }
}
```

### 4) Add health/readiness endpoints

`server/src/routes/health.ts`
```ts
import type { Router } from 'express';
import express from 'express';
import { pool } from '../db'; // adapt

const router = express.Router();

// Liveness: process is alive
router.get('/health', (_req, res) => {
  res.status(200).send('OK');
});

// Readiness: dependencies available
router.get('/ready', async (_req, res) => {
  try {
    // lightweight DB ping/query
    await pool.query('SELECT 1');
    res.status(200).send('OK');
  } catch (err) {
    console.error('Readiness check failed', err);
    res.status(503).send('Not ready');
  }
});

export default router;
```

Wire into app:

`server/src/app.ts` (or main Express file)
```ts
import express from 'express';
import healthRoutes from './routes/health';
import { handleLineWebhook } from './webhooks/lineWebhookHandler';

const app = express();
app.use(express.json());

// Health probes
app.use('/health', healthRoutes);
app.use('/ready', healthRoutes);

// Webhook route
app.post('/webhook/line', handleLineWebhook);

// Graceful shutdown
function shutdown(signal: string) {
  return async () => {
    console.log(`${signal} received: closing server...`);
    // Stop accepting new requests
    server.close(async (err) => {
      if (err) {
        console.error('Error during shutdown', err);
        process.exit(1);
      }
      // Close DB pool and other resources
      try {
        await pool.end();
        console.log('DB pool closed');
        process.exit(0);
      } catch (e) {
        console.error('Error closing resources', e);
        process.exit(1);
      }
    });
  };
}

const server = app.listen(process.env.PORT || 3000, () => {
  console.log(`Server listening on port ${process.env.PORT || 3000}`);
});

process.on('SIGTERM', shutdown('SIGTERM'));
process.on('SIGINT', shutdown('SIGINT'));
```
