## 1) Snapshot generator (Mac orchestration only)

`bin/make-snapshot.py`
```python
#!/usr/bin/env python3
"""
Generate deterministic snapshot for surrogate-1 dataset ingestion.
Usage:
  python bin/make-snapshot.py --repo axentx/surrogate-1-training-pairs \
    --date 2026-05-03 --out snapshot-2026-05-03.json
"""
import argparse
import json
import os
import sys
from datetime import datetime, timezone

from huggingface_hub import HfApi

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def build_snapshot(repo: str, date: str, out_path: str):
    api = HfApi()
    # list top-level for the date folder (non-recursive to stay within rate limits)
    entries = api.list_repo_tree(repo=repo, path=date, recursive=False)

    files = []
    for e in entries:
        if not e.path.endswith((".parquet", ".jsonl", ".csv")):
            continue
        files.append({
            "repo": repo,
            "path": e.path,
            "cdn_url": CDN_TEMPLATE.format(repo=repo, path=e.path),
            "size": getattr(e, "size", None),
        })

    snapshot = {
        "repo": repo,
        "date": date,
        "generated_at_utc": datetime.now(timezone.utc).isoformat(),
        "files": sorted(files, key=lambda x: x["path"]),
        "total_files": len(files),
    }

    os.makedirs(os.path.dirname(os.path.abspath(out_path)), exist_ok=True)
    with open(out_path, "w", encoding="utf-8") as f:
        json.dump(snapshot, f, indent=2)

    print(f"Snapshot written to {out_path} ({len(files)} files)")

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Create deterministic snapshot for CDN-only ingestion.")
    parser.add_argument("--repo", default="axentx/surrogate-1-training-pairs")
    parser.add_argument("--date", required=True, help="YYYY-MM-DD folder in dataset repo")
    parser.add_argument("--out", required=True, help="Output JSON path")
    args = parser.parse_args()
    try:
        build_snapshot(args.repo, args.date, args.out)
    except Exception as exc:
        print(f"Error: {exc}", file=sys.stderr)
        sys.exit(1)
```

Make executable:
```bash
chmod +x bin/make-snapshot.py
```

## 2) Worker script: CDN-only ingestion with snapshot

`bin/dataset-enrich.sh`
```bash
#!/usr/bin/env bash
# surrogate-1 dataset-enrich worker (GitHub Actions + HF Space compatible)
#
# Usage:
#   SHARD_ID=0 SHARD_TOTAL=16 SNAPSHOT=snapshot-2026-05-03.json bash bin/dataset-enrich.sh
#
# Behavior:
# - Reads deterministic snapshot (JSON) listing files and CDN URLs.
# - Each shard processes a deterministic slice by hash(slug) % SHARD_TOTAL.
# - Downloads via CDN (no HF API auth) and projects to {prompt,response}.
# - Outputs to batches/public-merged/<date>/shard<N>-<HHMMSS>.jsonl
#
# Required env:
#   SHARD_ID (0..N-1)
#   SHARD_TOTAL (>=1)
#   SNAPSHOT (path or URL to snapshot JSON)
#   HF_TOKEN (write permission to axentx/surrogate-1-training-pairs)
#   DATE (YYYY-MM-DD) — if not set, derived from snapshot

set -euo pipefail
export SHELL=/bin/bash

: "${SHARD_ID:?required}"
: "${SHARD_TOTAL:?required}"
: "${SNAPSHOT:?required}"
: "${HF_TOKEN:?required}"

REPO_DST="axentx/surrogate-1-training-pairs"
WORKDIR=$(mktemp -d)
OUTDIR="batches/public-merged"
mkdir -p "$WORKDIR"

echo "Worker SHARD_ID=$SHARD_ID/$SHARD_TOTAL | SNAPSHOT=$SNAPSHOT"

# Fetch snapshot if remote
if echo "$SNAPSHOT" | grep -qE '^https?://'; then
  curl -fsSL "$SNAPSHOT" -o "$WORKDIR/snapshot.json"
  SNAPSHOT="$WORKDIR/snapshot.json"
fi

DATE=$(jq -r '.date' "$SNAPSHOT")
: "${DATE:?snapshot must contain .date}"
TIMESTAMP=$(date -u +"%H%M%S")
OUTFILE="${OUTDIR}/${DATE}/shard${SHARD_ID}-${TIMESTAMP}.jsonl"

mkdir -p "$(dirname "$OUTFILE")"

# Deterministic shard assignment: hash(slug) mod SHARD_TOTAL
# slug = filename without extension (safe deterministic key)
assign_shard() {
  local slug=$1
  local hash
  # Use deterministic numeric hash (same across machines)
  hash=$(echo -n "$slug" | cksum | awk '{print $1}')
  echo $(( hash % SHARD_TOTAL ))
}

# Lightweight projection to {prompt,response} for common schemas
project_record() {
  local file=$1
  local ext=${file##*.}
  case "$ext" in
    jsonl)
      # Expect one JSON object per line; try common keys
      while IFS= read -r line; do
        [ -z "$line" ] && continue
        prompt=$(echo "$line" | jq -r '
          .prompt // .input // .question // .instruction // .text // empty
        ')
        response=$(echo "$line" | jq -r '
          .response // .output // .answer // .completion // .text // empty
        ')
        if [ -n "$prompt" ] && [ -n "$response" ]; then
          jq -nc --arg p "$prompt" --arg r "$response" '{prompt:$p, response:$r}'
        fi
      done < "$file"
      ;;
    parquet)
      # Use python for parquet projection (avoids heavy CLI deps)
      python3 -c "
import pyarrow.parquet as pq, sys, json
tbl = pq.read_table('$file')
cols = tbl.column_names
prompt_col = next((c for c in ['prompt','input','question','instruction','text'] if c in cols), None)
response_col = next((c for c in ['response','output','answer','completion','text'] if c in cols), None)
if not prompt_col or not response_col:
    sys.exit(0)
for i in range(tbl.num_rows):
    p = tbl[prompt_col][i].as_py()
    r = tbl[response_col][i].as_py()
    if p is not None and r is not None:
        print(json.dumps({'prompt': str(p), 'response': str(r)}))
" 2>/dev/null || true
      ;;
    csv)
      # Use python for consistent CSV handling
      python3 -c "
import csv, sys, json
with open('$file', newline='', encoding='utf-8') as f:
    r = csv.DictReader(f)
    for row in r:
        prompt = row.get('prompt') or row.get('input') or row.get('question') or row.get('instruction') or row
