### Diagnosis and Proposed Changes

The primary issues identified include:

1. **Lack of client-side idempotency key**: This leads to duplicate punches upon double-taps or LINE webhook replays.
2. **No optimistic state or debounce on the clock-in/out button**: Users can spam requests, leading to multiple queued requests.
3. **No local validation of valid state transitions**: This results in a confusing UX and wasted requests (e.g., attempting to clock-in while already clocked-in).
4. **Missing visual feedback during submission**: Users perceive slowness and are likely to retry, leading to potential duplicates.
5. **No persisted “last punch” client-side guard**: This omission allows obviously invalid actions to reach the API.

### Implementation

To address these issues, we propose modifying or creating the `TimeClockButton` component in `workio/src/components/TimeClock/TimeClockButton.tsx`. The changes will include:

- **Idempotency Key Generation**: Implement a unique idempotency key for each punch submission to prevent duplicates.
- **Optimistic UI**: Update the UI optimistically upon submitting a punch to reflect the expected outcome immediately.
- **Debounced Submission**: Implement a debounce mechanism to prevent multiple submissions within a short timeframe.
- **State Transition Validation**: Validate state transitions locally to prevent invalid actions (e.g., clocking in while already clocked in).
- **Loading/Disabled States**: Provide clear visual feedback during submission to prevent retries.

// workio/src/components/TimeClock/TimeClockButton.tsx
import React, { useState, useCallback, useEffect } from 'react';
import { useMutation, useQueryClient } from '@tanstack/react-query';
import { v4 as uuidv4 } from 'uuid';
import { clockIn, clockOut } from '../../api/timeclock';
import { useLastPunch } from '../../hooks/useLastPunch';

type PunchState = 'IN' | 'OUT';

export const TimeClockButton: React.FC = () => {
  const queryClient = useQueryClient();
  const { lastPunch, setLastPunch, canTransitionTo } = useLastPunch();
  const [idempotencyKey, setIdempotencyKey] = useState<string | null>(null);
  const [isSubmitting, setIsSubmitting] = useState(false);

  // Determine desired next state
  const nextState: PunchState = lastPunch?.state === 'IN' ? 'OUT' : 'IN';
  const canAct = canTransitionTo(nextState);

  // Mutations
  const punchMutation = useMutation({
    mutationFn: nextState === 'IN' ? clockIn : clockOut,
    onMutate: async () => {
      // Cancel/refetch invalidation handled after await
      queryClient.cancelQueries({ queryKey: ['todayPunches'] });
      
      // Optimistic update
      const optimisticPunch = {
        id: uuidv4(),
        state: nextState,
        timestamp: new Date().toISOString(),
        idempotencyKey: idempotencyKey!,
        _optimistic: true,
      };
      queryClient.setQueryData(['todayPunches'], (old: any[] = []) => [...old, optimisticPunch]);
      setLastPunch({ state: nextState, timestamp: optimisticPunch.timestamp });
      return { optimisticId: optimisticPunch.id };
    },
    onSettled: () => {
      queryClient.invalidateQueries({ queryKey: ['todayPunches'] });
      setIsSubmitting(false);
    },
  });

  const handlePress = useCallback(() => {
    if (!canAct || punchMutation.isPending || isSubmitting) return;
    const key = uuidv4();
    setIdempotencyKey(key);
    setIsSubmitting(true);
    punchMutation.mutate(
      { idempotencyKey: key },
      {
        onError: () => {
          // rollback optimistic state handled by refetch onSettled
        },
      }
    );
  }, [canAct, punchMutation, idempotencyKey, isSubmitting]);

  // Debounced click handler (simple 1s debounce to prevent double-taps)
  const [lastClicked, setLastClicked] = useState<number>(0);
  const onDebouncedPress = useCallback(() => {
    const now = Date.now();
    if (now - lastClicked < 1000) return;
    setLastClicked(now);
    handlePress();
  }, [handlePress, lastClicked]);

  // Keyboard/accessibility support
  const onKeyDown = useCallback(
    (e: React.KeyboardEvent) => {
      if (e.key === 'Enter' || e.key === ' ') {
        e.preventDefault();
        onDebouncedPress();
      }
    },
    [onDebouncedPress]
  );

  const isLoading = punchMutation.isPending || isSubmitting;
  const buttonLabel = nextState === 'IN' ? 'Clock In' : 'Clock Out';
  const buttonSub = isLoading ? 'Processing…' : lastPunch ? `Last: ${formatTime(lastPunch.timestamp)}` : '';

  return (
    <div className="flex flex-col items-center gap-2">
      <button
        type="button"
        onClick={onDebouncedPress}
        onKeyDown={onKeyDown}
        disabled={!canAct || isLoading}
        aria-busy={isLoading}
        aria-label={buttonLabel}
        className={`inline-flex min-h-[48px] items-center justify-center rounded-xl px-8 py-3 text-lg font-semibold transition-all focus:outline-none focus:ring-2 focus:ring-offset-2 ${
          !canAct || isLoading
            ? 'cursor-not-allowed bg-gray-200 text-gray-500'
            : nextState === 'IN'
            ? 'bg-green-600 text-white hover:bg-green-700 focus:ring-green-500'
            : 'bg-red-600 text-white hover:bg-red-700 focus:ring-red-500'
        } ${isLoading ? 'opacity-80' : ''}`}
      >
        {isLoading && (
          <svg
            className="-ml-1 mr-2 h-5 w-5 animate-spin text-white"
            xmlns="http://www.w3.org/2000/svg"
            fill="none"
            viewBox="0 0 24 24"
          >
            <circle className="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" strokeWidth="4" />
            <path
              className="opacity-75"
              fill="currentColor"
              d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4z"
            />
          </svg>
        )}
        {buttonLabel}
      </button>
      {buttonSub && <span className="text-sm text-gray-500">{buttonSub}</span>}
    </div>
  );
};

function formatTime(iso: string) {
  return new Date(iso).toLocaleTimeString(undefined, { hour: '2-digit', minute: '2-digit' });
}
