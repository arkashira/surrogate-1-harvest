# airship / design

**Final synthesized implementation (best of both proposals)**

Highest-value improvement (≤2h):  
**CDN-first deterministic ingestion + Lightning Studio lifecycle guard + deterministic HF sharding.**  
Eliminates HF API rate limits during training, prevents quota waste from duplicate Studio creation, enforces `{prompt, response}` projection only at parse/upload time, and spreads writes across repos to avoid per-repo commit caps.

## Code snippets (merged, hardened)

### 1. Export CDN file list (non-recursive, date-scoped)
```python
#!/usr/bin/env python3
# scripts/list_hf_files.py
import json, os, sys
from huggingface_hub import list_repo_tree

REPO = os.getenv("HF_DATASET_REPO", "axentx/surrogate-mirror")
FOLDER = sys.argv[1] if len(sys.argv) > 1 else "2026-05-03"
OUT = sys.argv[2] if len(sys.argv) > 2 else "file_list.json"

# non-recursive to avoid pagination explosion; keep flat per-date folder
tree = list_repo_tree(repo_id=REPO, path=FOLDER, recursive=False)
files = [f.rfilename for f in tree if f.type == "file"]

with open(OUT, "w") as f:
    json.dump({"repo": REPO, "folder": FOLDER, "files": files}, f, indent=2)

print(f"Wrote {len(files)} files to {OUT}")
```

### 2. Deterministic shard assignment (single source of truth)
```python
#!/usr/bin/env python3
# scripts/build_manifest.py
import hashlib, csv, json, os

N_SIBLINGS = int(os.getenv("HF_SIBLINGS", "5"))  # 5 siblings => 640 commits/hr aggregate

def pick_repo(slug: str) -> str:
    h = int(hashlib.sha256(slug.encode()).hexdigest(), 16)
    idx = h % N_SIBLINGS
    return f"axentx/surrogate-enriched-{idx}"

with open("file_list.json") as f:
    data = json.load(f)

with open("manifest.csv", "w", newline="") as csvf:
    w = csv.writer(csvf)
    w.writerow(["repo", "path", "slug"])
    for fn in data["files"]:
        slug = os.path.splitext(os.path.basename(fn))[0]
        repo = pick_repo(slug)
        w.writerow([repo, fn, slug])

print("manifest.csv ready")
```

### 3. CDN-only dataloader (zero HF API during training)
```python
# training/train.py  (excerpt)
import json, requests
from torch.utils.data import IterableDataset

CDN_ROOT = "https://huggingface.co/datasets"

class CDNJsonlDataset(IterableDataset):
    def __init__(self, file_list_path, repo):
        with open(file_list_path) as f:
            self.files = json.load(f)["files"]
        self.repo = repo

    def __iter__(self):
        for fn in self.files:
            url = f"{CDN_ROOT}/{self.repo}/resolve/main/{fn}"
            resp = requests.get(url, timeout=30)
            resp.raise_for_status()
            # Project to {prompt, response} only at parse time
            for line in resp.text.strip().splitlines():
                if not line.strip():
                    continue
                obj = json.loads(line)
                yield {
                    "prompt": obj.get("prompt") or obj.get("input") or "",
                    "response": obj.get("response") or obj.get("output") or "",
                }
```

### 4. Lightning Studio lifecycle guard (reuse + idle restart)
```python
# training/studio_lifecycle.py
from lightning import Studio, Teamspace, Machine
import time

def get_or_create_studio(name: str, machine=Machine.L40S):
    # Reuse running studio to save quota
    for s in Teamspace.studios:
        if s.name == name and s.status == "Running":
            print(f"Reusing running studio: {name}")
            return s

    print(f"Creating studio: {name}")
    return Studio(name=name, machine=machine, create_ok=True)

def run_with_idle_guard(studio_name, target, max_retries=3):
    for attempt in range(max_retries):
        studio = get_or_create_studio(studio_name)
        if studio.status != "Running":
            print(f"Studio not running (status={studio.status}), restarting...")
            studio.start(machine=Machine.L40S)
            time.sleep(30)
            continue
        try:
            studio.run(target)
            return
        except Exception as e:
            print(f"Run failed (attempt {attempt+1}): {e}")
            time.sleep(60)
    raise RuntimeError("Max retries exceeded")
```

### 5. Upload enriched (schema projection + deterministic repo)
```python
#!/usr/bin/env python3
# scripts/upload_enriched.py
import pandas as pd, hashlib, os

N_SIBLINGS = int(os.getenv("HF_SIBLINGS", "5"))

def pick_repo(slug: str) -> str:
    h = int(hashlib.sha256(slug.encode()).hexdigest(), 16)
    idx = h % N_SIBLINGS
    return f"axentx/surrogate-enriched-{idx}"

def upload_one(date: str, slug: str, rows):
    # Schema projection: only prompt,response
    df = pd.DataFrame(rows, columns=["prompt", "response"])
    out_dir = f"batches/mirror-merged/{date}"
    os.makedirs(out_dir, exist_ok=True)
    path = f"{out_dir}/{slug}.parquet"
    df.to_parquet(path, index=False)

    repo = pick_repo(slug)
    # Use huggingface_hub.upload_file(repo_id=repo, path_or_fileobj=path, path_in_repo=path)
    print(f"Upload {path} -> {repo}")
```

## Acceptance criteria
- Training runs with **zero HF API calls** after initial `file_list.json` generation.
- Studio reuse + idle restart guard reduces Lightning quota consumption; training no longer fails due to idle stops.
- Enriched parquet files written to deterministic sibling repos with minimal schema (`prompt`, `response` only).
- Writes spread across repos to avoid per-repo commit caps; deterministic sharding reproducible via `hash(slug) % N`.
