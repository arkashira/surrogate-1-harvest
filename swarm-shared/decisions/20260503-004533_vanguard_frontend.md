# vanguard / frontend

## 1. Diagnosis

- No frontend manifest cache: training UI re-enumerates HF repo via authenticated API on every load, burning quota and risking 429.
- No deterministic file list persisted to disk; schema heterogeneity can trigger pyarrow CastErrors during client-side previews.
- No reuse of Lightning Studio state in the UI; users can accidentally spawn duplicate studios and waste quota.
- No CDN-bypass download path in the frontend; large file previews use `/api/` endpoints instead of public CDN URLs.
- No lightweight, cached “available-dates” index for fast date-folder selection; each navigation triggers recursive API calls.

## 2. Proposed change

Add a frontend service + hook that:
- reads a local `manifest.json` (generated by ops) for a selected date folder,
- exposes CDN URLs for direct download,
- provides cached studio-reuse helpers,
- and falls back to authenticated API only when manifest is stale/missing.

Scope:
- `src/services/datasetManifest.ts` (new)
- `src/hooks/useDatasetFiles.ts` (new)
- `src/components/DatasetBrowser/DatasetBrowser.tsx` (modify to use hook)
- `public/manifests/` (expected folder; file added by ops, not code)

## 3. Implementation

```ts
// src/services/datasetManifest.ts
import axios from 'axios';

const CDN_ROOT = 'https://huggingface.co/datasets';
const API_ROOT = 'https://huggingface.co/api';

export interface DatasetFile {
  path: string;
  cdnUrl: string;
  size?: number;
  sha256?: string;
}

export interface Manifest {
  repo: string;
  dateFolder: string;
  files: DatasetFile[];
  generatedAt: string;
}

export async function loadLocalManifest(
  repo: string,
  dateFolder: string
): Promise<Manifest | null> {
  try {
    // public folder served as static asset
    const res = await axios.get<Manifest>(
      `/manifests/${repo}/${dateFolder}/manifest.json`,
      { timeout: 5000 }
    );
    // Basic validity check
    if (res.data?.repo === repo && Array.isArray(res.data.files)) {
      return res.data;
    }
    return null;
  } catch {
    return null;
  }
}

export async function fetchRepoFileList(
  repo: string,
  dateFolder: string,
  token?: string
): Promise<DatasetFile[]> {
  const headers = token ? { Authorization: `Bearer ${token}` } : {};
  // Non-recursive tree call to avoid quota burn and schema issues
  const url = `${API_ROOT}/datasets/${repo}/tree/${encodeURIComponent(
    dateFolder
  )}`;
  const { data } = await axios.get<Array<{ path: string; size?: number; sha256?: string }>>(
    url,
    { headers }
  );

  return (data || []).map((f) => ({
    path: f.path,
    cdnUrl: `${CDN_ROOT}/${repo}/resolve/main/${encodeURIComponent(
      `${dateFolder}/${f.path}`
    )}`,
    size: f.size,
    sha256: f.sha256,
  }));
}
```

```ts
// src/hooks/useDatasetFiles.ts
import { useEffect, useState, useCallback } from 'react';
import { loadLocalManifest, fetchRepoFileList, type DatasetFile } from '../services/datasetManifest';

export function useDatasetFiles(
  repo: string,
  dateFolder: string,
  hfToken?: string
) {
  const [files, setFiles] = useState<DatasetFile[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  const reload = useCallback(async () => {
    if (!repo || !dateFolder) {
      setFiles([]);
      setLoading(false);
      return;
    }

    setLoading(true);
    setError(null);

    try {
      // 1) Try local manifest first (CDN-only, zero API)
      const manifest = await loadLocalManifest(repo, dateFolder);
      if (manifest?.files?.length) {
        setFiles(manifest.files);
        return;
      }

      // 2) Fallback to authenticated API (non-recursive)
      const fetched = await fetchRepoFileList(repo, dateFolder, hfToken);
      setFiles(fetched);
    } catch (err: any) {
      setError(err?.message || 'Failed to load files');
      setFiles([]);
    } finally {
      setLoading(false);
    }
  }, [repo, dateFolder, hfToken]);

  useEffect(() => {
    reload();
  }, [reload]);

  return { files, loading, error, reload };
}
```

```tsx
// src/components/DatasetBrowser/DatasetBrowser.tsx
import React from 'react';
import { useDatasetFiles } from '../../hooks/useDatasetFiles';

interface Props {
  repo: string;
  dateFolder: string;
  hfToken?: string;
  onSelectFile?: (file: string) => void;
}

export const DatasetBrowser: React.FC<Props> = ({
  repo,
  dateFolder,
  hfToken,
  onSelectFile,
}) => {
  const { files, loading, error } = useDatasetFiles(repo, dateFolder, hfToken);

  if (loading) return <div className="p-4 text-sm text-gray-500">Loading files…</div>;
  if (error) return <div className="p-4 text-sm text-red-600">{error}</div>;
  if (!files.length) return <div className="p-4 text-sm text-gray-400">No files found.</div>;

  return (
    <ul className="divide-y divide-gray-100">
      {files.map((f) => (
        <li key={f.path} className="p-2 flex items-center justify-between hover:bg-gray-50">
          <span className="text-sm font-mono truncate">{f.path}</span>
          <div className="flex items-center gap-2">
            {f.size != null && (
              <span className="text-xs text-gray-400">{(f.size / 1024).toFixed(1)} KB</span>
            )}
            <a
              href={f.cdnUrl}
              target="_blank"
              rel="noopener noreferrer"
              className="text-xs text-blue-600 hover:underline"
              onClick={(e) => {
                // optional: allow parent to intercept selection before download
                if (onSelectFile) {
                  e.preventDefault();
                  onSelectFile(f.path);
                }
              }}
            >
              Preview
            </a>
          </div>
        </li>
      ))}
    </ul>
  );
};
```
