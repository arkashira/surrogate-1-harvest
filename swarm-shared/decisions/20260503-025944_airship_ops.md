```bash
# /opt/axentx/airship/scripts/list_hf_date_folder.sh
#!/usr/bin/env bash
set -euo pipefail
export HF_REPO="axentx/surrogate-training-data"
export DATE_FOLDER="${1:-2026-05-03}"  # e.g. batches/mirror-merged/2026-05-03
OUTPUT_JSON="${2:-./file_list.json}"

python3 - "$HF_REPO" "$DATE_FOLDER" "$OUTPUT_JSON" <<'PY'
import json, sys
from huggingface_hub import list_repo_tree

repo, path, out = sys.argv[1], sys.argv[2], sys.argv[3]
# non-recursive immediate listing for the date folder
items = list_repo_tree(repo=repo, path=path, recursive=False)
files = [f.rfilename for f in items if f.type == "file"]
with open(out, "w") as f:
    json.dump({"repo": repo, "base_path": path, "files": files}, f, indent=2)
print(f"Listed {len(files)} files -> {out}")
PY
```

### 2.2 Lightning: CDN-only Dataset (zero API during training)
```python
# /opt/axentx/airship/surrogate/data/cdn_dataset.py
import json
import os
from pathlib import Path
from typing import Dict

import requests
import pyarrow as pa
import pyarrow.parquet as pq
from torch.utils.data import Dataset

class CDNParquetDataset(Dataset):
    """
    Loads Parquet shards from HF CDN (no Authorization header).
    file_list.json must contain:
      {"repo": "axentx/surrogate-training-data", "base_path": "...", "files": [...]}
    """

    CDN_ROOT = "https://huggingface.co/datasets"

    def __init__(self, file_list_path: str, columns=("prompt", "response")):
        with open(file_list_path) as f:
            meta = json.load(f)
        self.repo = meta["repo"]
        self.base = meta["base_path"].rstrip("/")
        self.files = meta["files"]
        self.columns = columns
        self._cache: Dict[str, pa.Table] = {}

    def _fetch_parquet(self, file_rel: str) -> pa.Table:
        if file_rel in self._cache:
            return self._cache[file_rel]

        url = f"{self.CDN_ROOT}/{self.repo}/resolve/main/{self.base}/{file_rel}"
        resp = requests.get(url, timeout=30)
        resp.raise_for_status()
        table = pq.read_table(pa.BufferReader(resp.content), columns=self.columns)
        self._cache[file_rel] = table
        return table

    def __len__(self):
        return len(self.files)

    def __getitem__(self, idx):
        table = self._fetch_parquet(self.files[idx])
        # each shard expected single row; adapt if needed
        row = table.slice(0, 1).to_pydict()
        return {k: row[k][0] for k in self.columns}
```

### 2.3 Lightning: Studio lifecycle guard + launcher
```python
# /opt/axentx/airship/surrogate/train/lightning_launcher.py
import time
from lightning import LightningWork, LightningFlow, LightningApp, Machine
from lightning.app import BuildConfig
from surrogate.data.cdn_dataset import CDNParquetDataset

class SurrogateTrainer(LightningWork):
    def __init__(self, file_list_path: str):
        super().__init__(machine=Machine.L40S, cloud_build_config=BuildConfig(image="pytorch/pytorch:2.3.0-cuda12.1-cudnn8-runtime"))
        self.file_list_path = file_list_path
        self._studio = None

    def run(self):
        from lightning.pytorch import Trainer
        from lightning.pytorch.callbacks import ModelCheckpoint
        from surrogate.model.surrogate_module import SurrogateModule  # your LightningModule

        # Guard: ensure studio is running (restart if idle-stopped)
        from lightning import Teamspace
        for s in Teamspace.studios:
            if s.name == "surrogate-training-studio" and s.status == "Running":
                self._studio = s
                break

        if self._studio is None or self._studio.status != "Running":
            # restart
            from lightning import Studio
            self._studio = Studio(
                name="surrogate-training-studio",
                machine=Machine.L40S,
                create_ok=True
            )
            # wait until running
            while self._studio.status != "Running":
                time.sleep(10)
                self._studio.refresh()

        dataset = CDNParquetDataset(self.file_list_path)
        model = SurrogateModule()

        trainer = Trainer(
            max_epochs=1,
            devices=1,
            accelerator="gpu",
            callbacks=[ModelCheckpoint(monitor="loss")]
        )
        trainer.fit(model, train_dataloaders=[dataset])

class RootFlow(LightningFlow):
    def __init__(self):
        super().__init__()
        self.trainer = SurrogateTrainer(file_list_path="./file_list.json")

    def configure_layout(self):
        return []

    def run(self):
        self.trainer.run()

app = LightningApp(RootFlow())
```
