**Final consolidated implementation (correct + actionable, <2 h)**

**What we are doing**
- Eliminate HuggingFace API 429s during Surrogate training by replacing `load_dataset(streaming=True)` with **CDN-only data loading**.
- Add **Lightning Studio lifecycle resilience** so idle timeouts or stopped studios don’t kill training and we avoid unnecessary studio churn (saves quota).
- Keep schema/infra unchanged; no heavy migrations.

### 1) Pre-list files once (local/Mac)

Run this once per dataset version (after any rate-limit window clears). It produces a JSON file committed to the repo so training needs zero HF API calls.

`scripts/build_filelist.py`
```python
#!/usr/bin/env python3
"""
Pre-list HF repo files for a date folder and save to JSON.
Run once per dataset version.
"""
import json
import os
from huggingface_hub import HfApi

REPO = "axentx/surrogate-dataset"
DATE_FOLDER = "batches/mirror-merged/2026-05-03"
OUT = "training/filelist-2026-05-03.json"

api = HfApi()
tree = api.list_repo_tree(repo_id=REPO, path=DATE_FOLDER, recursive=False)
files = [f.rfilename for f in tree if f.rfilename.endswith((".parquet", ".jsonl"))]

os.makedirs(os.path.dirname(OUT), exist_ok=True)
with open(OUT, "w") as f:
    json.dump(files, f, indent=2)

print(f"Saved {len(files)} files to {OUT}")
```

Commit `training/filelist-2026-05-03.json` (or the appropriate date) so training can load it locally.

### 2) CDN-only dataset loader

`training/cdn_dataset.py`
```python
import io
import json
import requests
import pyarrow.parquet as pq
from concurrent.futures import ThreadPoolExecutor, as_completed
from tqdm import tqdm
from typing import List, Dict

REPO = "axentx/surrogate-dataset"
CDN_ROOT = f"https://huggingface.co/datasets/{REPO}/resolve/main"

def load_filelist(path: str) -> List[str]:
    with open(path) as f:
        return json.load(f)

def fetch_cdn_file(key: str) -> bytes:
    url = f"{CDN_ROOT}/{key}"
    resp = requests.get(url, timeout=30)
    resp.raise_for_status()
    return resp.content

def parse_parquet(content: bytes) -> List[Dict[str, str]]:
    table = pq.read_table(io.BytesIO(content))
    # Project only prompt/response; ignore heterogeneous extra columns
    has_prompt = "prompt" in table.column_names
    has_response = "response" in table.column_names

    prompts = table["prompt"].to_pylist() if has_prompt else [None] * len(table)
    responses = table["response"].to_pylist() if has_response else [None] * len(table)

    return [
        {"prompt": p, "response": r}
        for p, r in zip(prompts, responses)
        if p is not None and r is not None
    ]

def cdn_dataset(filelist_path: str, max_workers: int = 8) -> List[Dict[str, str]]:
    keys = load_filelist(filelist_path)
    records = []

    with ThreadPoolExecutor(max_workers=max_workers) as ex:
        future_to_key = {ex.submit(fetch_cdn_file, k): k for k in keys}
        for fut in tqdm(as_completed(future_to_key), total=len(keys), desc="CDN fetch"):
            try:
                content = fut.result()
                records.extend(parse_parquet(content))
            except Exception as exc:
                key = future_to_key[fut]
                print(f"Failed {key}: {exc}")
    return records
```

Notes:
- No authorization headers required; CDN is public and has much higher rate limits.
- Threaded downloads keep throughput high without schema assumptions.

### 3) Lightning Studio lifecycle guard + training entrypoint

`training/train.py`
```python
import os
import time
from typing import List, Dict

from lightning import LightningWork, LightningFlow, LightningApp, Machine
from lightning import Teamspace

from surrogate.cdn_dataset import cdn_dataset

# --
# Studio lifecycle helpers
# --
def get_or_create_studio(name: str, machine: Machine) -> LightningWork:
    """Reuse a running studio if present; otherwise create one."""
    for s in Teamspace.studios:
        if s.name == name and getattr(s, "status", None) == "running":
            print(f"Reusing running studio: {name}")
            return s

    print(f"Creating studio: {name}")
    return LightningWork(
        name=name,
        cloud_compute=machine,
        cloud_build_config=None,
        enable_app_server=False,
    )

# --
# Trainer Work
# --
class SurrogateTrainer(LightningWork):
    def __init__(self):
        super().__init__()
        # Prefer L40S for free-tier compatibility; change only if quota allows
        self.machine = Machine.L40S
        self.studio = None

    def run(self) -> None:
        # Ensure studio is alive
        if self.studio is None or getattr(self.studio, "status", None) != "running":
            self.studio = get_or_create_studio("surrogate-trainer", self.machine)
            if getattr(self.studio, "status", None) != "running":
                self.studio.start(machine=self.machine)
                time.sleep(10)  # allow warm-up

        # CDN-only dataset (zero HF API calls during training)
        filelist = "training/filelist-2026-05-03.json"
        if not os.path.exists(filelist):
            raise FileNotFoundError(
                f"Run scripts/build_filelist.py first -> {filelist}"
            )

        dataset: List[Dict[str, str]] = cdn_dataset(filelist, max_workers=8)
        print(f"Loaded {len(dataset)} prompt/response pairs via CDN")

        # --
        # Your training loop below (example placeholder)
        # --
        # train_model(dataset)
        # --

# --
# App
# --
class RootFlow(LightningFlow):
    def __init__(self):
        super().__init__()
        self.trainer = SurrogateTrainer()

    def run(self) -> None:
        self.trainer.run()

app = LightningApp(RootFlow())
```

- **Correctness**: CDN bypass is the only reliable way to avoid HF API 429s during long training runs.

- **Resilience**: Studio lifecycle guard prevents idle-timeout deaths and quota waste.
- **Performance**: Threaded CDN downloads saturate bandwidth without HF API bottlenecks.
