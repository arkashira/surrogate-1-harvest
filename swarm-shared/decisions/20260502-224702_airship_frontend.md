# airship / frontend

## Final Synthesized Implementation (Best of Both Candidates)

**Chosen Scope (<2h):** Ship a **static, CDN-safe `airship discover` orchestrator** that produces a deterministic status snapshot and asset manifest, enabling the frontend to render health/state with **zero SSR and zero backend calls**.

**Why this wins:**
- Immediate UX: status visible without backend/auth roundtrips.
- Removes runtime backend dependency from frontend (faster, cacheable, CDN-first).
- Fits <2h: one CLI orchestrator + one static frontend renderer + build hook.
- Resolves contradictions by **using Python for the CLI** (maintainability, config parsing) while keeping **Bash-style portability checks** (port/health probes), and outputs to `dist/` (clean separation) while optionally mirroring to `public/` for immediate web serving.

---

## 1) CLI Orchestrator: `airship/discover.py`

**Location:** `airship/discover.py` (add to `airship` CLI group)  
**Behavior:**
- Probes local services (docker ports) and optional remote health endpoints (with short timeouts).
- Never emits secrets; only public endpoints and deterministic state.
- Writes `dist/status.json`, `dist/manifest.json`, and `dist/status.html`.
- Accepts `--output-dir` and `--cdn-base` to control paths and CDN URLs.

```python
#!/usr/bin/env python3
"""
airship discover
Generates static status + manifest for CDN-frontend consumption.
"""
import argparse
import json
import socket
import sys
import urllib.request
import urllib.error
from datetime import datetime, timezone
from pathlib import Path

# Default ports (adjust via env/config if needed)
ARKSHIP_UI_PORT = int(os.getenv("ARKSHIP_UI_PORT", 3000))
ARKSHIP_API_PORT = int(os.getenv("ARKSHIP_API_PORT", 8000))
SURROGATE_PORT = int(os.getenv("SURROGATE_PORT", 8001))

# CDN / remote fallback (HF Space pattern)
SURROGATE_CDN_MANIFEST = os.getenv(
    "SURROGATE_CDN_MANIFEST",
    "https://huggingface.co/datasets/your-org/surrogate-manifest/resolve/main/airship-manifest.json"
)

def check_port(port: int, host: str = "localhost") -> bool:
    try:
        with socket.create_connection((host, port), timeout=2):
            return True
    except OSError:
        return False

def check_url(url: str, timeout: float = 3.0) -> bool:
    try:
        req = urllib.request.Request(url, method="GET", headers={"User-Agent": "airship-discover"})
        with urllib.request.urlopen(req, timeout=timeout):
            return True
    except Exception:
        return False

def build_status(ts: str) -> dict:
    arkship_ui = check_port(ARKSHIP_UI_PORT)
    arkship_api = check_port(ARKSHIP_API_PORT)
    surrogate_local = check_port(SURROGATE_PORT)

    surrogate_health_url = f"http://localhost:{SURROGATE_PORT}/health"
    surrogate_remote = check_url(surrogate_health_url)

    # Aggregate health
    if arkship_ui and arkship_api and surrogate_remote:
        overall = "healthy"
    elif not arkship_ui or not arkship_api:
        overall = "critical"
    else:
        overall = "degraded"

    return {
        "generatedAt": ts,
        "overall": overall,
        "services": {
            "arkshipUi": {"port": ARKSHIP_UI_PORT, "state": "up" if arkship_ui else "down"},
            "arkshipApi": {"port": ARKSHIP_API_PORT, "state": "up" if arkship_api else "down"},
            "surrogate": {
                "port": SURROGATE_PORT,
                "localState": "up" if surrogate_local else "down",
                "remoteState": "up" if surrogate_remote else "down",
                "localUrl": f"http://localhost:{SURROGATE_PORT}",
                "cdnManifestUrl": SURROGATE_CDN_MANIFEST,
            },
        },
        "notes": "Generated by airship discover — frontend should prefer CDN assets when available.",
    }

def build_manifest(ts: str, cdn_base: str) -> dict:
    # Normalize cdn_base
    cdn_base = cdn_base.rstrip("/")
    return {
        "version": "1.0.0",
        "generatedAt": ts,
        "assets": {
            "statusJson": f"{cdn_base}/status.json",
            "statusHtml": f"{cdn_base}/status.html",
            "surrogateCdnManifest": SURROGATE_CDN_MANIFEST,
        },
        "cache": {"maxAgeSeconds": 60},
        "services": {
            "arkshipUi": {"expectedPort": ARKSHIP_UI_PORT, "protocol": "http", "path": "/"},
            "arkshipApi": {"expectedPort": ARKSHIP_API_PORT, "protocol": "http", "path": "/health"},
            "surrogate": {
                "expectedPort": SURROGATE_PORT,
                "protocol": "http",
                "path": "/health",
                "cdnFallback": SURROGATE_CDN_MANIFEST,
            },
        },
    }

STATUS_HTML_TEMPLATE = """<!doctype html>
<html lang="en">
<head>
  <meta charset="utf-8" />
  <title>Arkship Status</title>
  <meta name="viewport" content="width=device-width,initial-scale=1" />
  <style>
    body {{ font-family: system-ui, sans-serif; padding: 2rem; }}
    .ok {{ color: #16a34a; }}
    .degraded {{ color: #ca8a04; }}
    .critical {{ color: #dc2626; }}
    pre {{ background: #f5f5f5; padding: 1rem; border-radius: 6px; overflow:auto; }}
  </style>
</head>
<body>
  <h1>Arkship Status</h1>
  <div id="status">Loading...</div>
  <noscript>
    <pre id="raw">Enable JavaScript to render status, or view the raw JSON.</pre>
  </noscript>
  <script>
    fetch('{cdn_base}/status.json').then(r => r.json()).then(s => {{
      document.getElementById('raw').textContent = JSON.stringify(s, null, 2);
      const cls = s.overall === 'healthy' ? 'ok' : s.overall === 'degraded' ? 'degraded' : 'critical';
      document.getElementById('status').innerHTML = '<p class=\"' + cls + '\"><strong>Status:</strong> ' + s.overall + '</p>';
    }}).catch(() => {{
      document.getElementById('status').innerHTML = '<p class=\"critical\">Unable to load status.</p>';
    }});
  </script>
</body>
</html>
"""

def main() -> None:
    parser = argparse.ArgumentParser(description="Generate static status + manifest for CDN.")
    parser.add_argument("--output-dir", default="dist", help="Output directory (default: dist)")
    parser.add_argument("--cdn-base", default="/", help="CDN base URL/path for asset references (default: /)")
    args = parser.parse_args()

    out_dir = Path(args.output_dir)
    out_dir.mkdir(parents=True, exist_ok=True)

    ts = datetime.now(timezone.utc).strftime("%Y-%m-%dT%H:%M:%SZ")
    status = build_status(ts)
    manifest = build_manifest(ts, args.cdn_base)

    (out_dir / "status.json").write_text(json.dumps(status, indent=2) + "\n")
    (out_dir / "manifest.json").write_text(json.dumps(manifest, indent=2) + "\n")
    (out_dir / "status.html").write_text(
        STATUS_HTML_TEMPLATE.format(cdn_base=args.cdn_base.rstrip("/") or ".")
    )

    print("✅ Generated:")
    print(f"  - {out_dir / 'status.json'}")
    print(f"  - {out_dir / 'manifest.json'}")
    print(f"  - {out_dir / 'status.html'}")

if __name__ == "__main__":
    main()
