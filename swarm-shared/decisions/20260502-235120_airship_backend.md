# airship / backend

## Final Synthesized Implementation (Correct + Actionable)

**Core deliverable**: a deterministic, CDN-only file manifest for a HuggingFace dataset `repo_id` and `date_folder`, exposed via:

- CLI: `airship discover <repo_id> <date_folder> [--out FILE]`
- HTTP: `GET /api/discover?repo_id=...&date_folder=...`
- Minimal frontend: `/discover` (HTML form + table)

This avoids HF API calls during training by embedding CDN URLs (`resolve/main/...`) in training scripts.

---

## Key Decisions (resolve contradictions)

- **Cache strategy**: use in-memory LRU in the API (fast, simple, safe for a small service) **and** optional local JSON cache in the CLI (avoids re-querying HF across CLI runs). Both are complementary; no conflict.
- **Path handling**: `date_folder` should not require trailing `/`; normalize internally. CDN URLs must point to `datasets/{repo_id}/resolve/main/{file_path}` (not nested under `date_folder` in the URL) because files in HF repos are stored at their full repo path. Candidate 1’s URL construction is correct; Candidate 2’s example URL was ambiguous/misleading — resolve in favor of Candidate 1.
- **Determinism**: sort entries by `path`. Include `size`. Output stable JSON (`sort_keys=True`, indent optional).
- **Scope**: non-recursive listing for the `date_folder` only (explicit requirement). Recursive listing can be added later if needed.
- **Error handling**: surface HF errors as HTTP 502 in API; CLI exits non-zero with clear message.

### 1. CLI: `airship/cli/discover.py`

```python
#!/usr/bin/env python3
"""
airship discover
Produce a deterministic CDN-only file manifest for a HuggingFace repo/date folder.

Usage:
  airship discover <repo_id> <date_folder> [--out=FILE] [--no-cache]
"""
import json
import sys
from datetime import datetime, timezone
from pathlib import Path
from typing import Dict, List

import click
from huggingface_hub import HfApi

API = HfApi()
CACHE_ROOT = Path(".cache/airship/discover")

def _normalize_date_folder(date_folder: str) -> str:
    return date_folder.strip("/")

def _cdn_base(repo_id: str) -> str:
    return f"https://huggingface.co/datasets/{repo_id}/resolve/main"

def build_manifest(repo_id: str, date_folder: str) -> Dict:
    """Return deterministic manifest for repo_id/date_folder (non-recursive)."""
    folder = _normalize_date_folder(date_folder)
    items = API.list_repo_tree(repo_id=repo_id, path=folder, recursive=False)
    files = sorted((it for it in items if it.type == "file"), key=lambda it: it.path)
    base = _cdn_base(repo_id)
    entries = [
        {
            "path": f.path,
            "cdn_url": f"{base}/{f.path}",
            "size": f.size,
        }
        for f in files
    ]
    return {
        "repo_id": repo_id,
        "date_folder": folder,
        "count": len(entries),
        "generated_at_utc": datetime.now(timezone.utc).isoformat(),
        "entries": entries,
    }

def _local_cache_path(repo_id: str, date_folder: str) -> Path:
    folder = _normalize_date_folder(date_folder)
    safe_repo = repo_id.replace("/", "_")
    return CACHE_ROOT / f"{safe_repo}_{folder}.json"

def load_cached_manifest(repo_id: str, date_folder: str) -> Dict | None:
    p = _local_cache_path(repo_id, date_folder)
    if p.is_file():
        try:
            return json.loads(p.read_text())
        except Exception:
            return None
    return None

def save_manifest_cache(manifest: Dict) -> None:
    p = _local_cache_path(manifest["repo_id"], manifest["date_folder"])
    p.parent.mkdir(parents=True, exist_ok=True)
    p.write_text(json.dumps(manifest, indent=2, sort_keys=True) + "\n")

@click.command()
@click.argument("repo_id")
@click.argument("date_folder")
@click.option("--out", "-o", type=click.Path(), help="Output file (default: stdout)")
@click.option(
    "--no-cache",
    is_flag=True,
    default=False,
    help="Ignore local cache and re-query HF API",
)
def discover(repo_id: str, date_folder: str, out: str, no_cache: bool) -> None:
    try:
        manifest = None
        if not no_cache:
            manifest = load_cached_manifest(repo_id, date_folder)
        if manifest is None:
            manifest = build_manifest(repo_id, date_folder)
            save_manifest_cache(manifest)

        payload = json.dumps(manifest, indent=2, sort_keys=True)
        if out:
            Path(out).write_text(payload + "\n")
            click.echo(f"Manifest written to {out}")
        else:
            click.echo(payload)
    except Exception as exc:
        click.echo(f"Error: {exc}", err=True)
        sys.exit(1)
```

def ttl_cache(seconds: int, maxsize: int = 32):
    def wrapper(func):
        func = lru_cache(maxsize=maxsize)(func)
        func.lru_ttl = seconds
        func.lru_timestamp = datetime.now(timezone.utc)

            now = datetime.now(timezone.utc)
            if (now - func.lru_timestamp).total_seconds() > func.lru_ttl:
                func.cache_clear()
                func.lru_timestamp = now
