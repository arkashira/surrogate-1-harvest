#### `surrogate/training/file_list.py`  (run on Mac)
```python
#!/usr/bin/env python3
"""
Generate file_list.json for a single date folder (CDN-only training).
Run after HF rate-limit window clears.
"""
import json
import os
import sys
from pathlib import Path

from huggingface_hub import HfApi

API_TOKEN = os.getenv("HF_TOKEN")
REPO_ID = os.getenv("HF_DATASET_REPO", "my-org/surrogate-data")
FOLDER = sys.argv[1] if len(sys.argv) > 1 else "batches/mirror-merged/2026-05-03"
OUTPUT = sys.argv[2] if len(sys.argv) > 2 else "file_list.json"

api = HfApi(token=API_TOKEN)
items = api.list_repo_tree(repo_id=REPO_ID, path=FOLDER, recursive=False)

files = []
for item in items:
    if not item.path.lower().endswith(".parquet"):
        continue
    cdn_url = f"https://huggingface.co/datasets/{REPO_ID}/resolve/main/{item.path}"
    files.append({"path": item.path, "cdn_url": cdn_url})

out_path = Path(OUTPUT)
out_path.write_text(json.dumps(files, indent=2))
print(f"Wrote {len(files)} files -> {out_path}")
```

#### `surrogate/training/dataset.py`
```python
import io
import json
from typing import Iterator

import pyarrow.parquet as pq
import requests
from torch.utils.data import IterableDataset

class CdnParquetIterableDataset(IterableDataset):
    """
    CDN-only Parquet streamer.
    Each file must contain at least prompt/response fields (extra columns ignored).
    """
    def __init__(self, file_list_path, max_files=None):
        with open(file_list_path) as f:
            self.files = [x for x in json.load(f) if x["path"].lower().endswith(".parquet")]
        if max_files:
            self.files = self.files[:max_files]

    def _stream_file(self, cdn_url: str) -> Iterator[dict]:
        resp = requests.get(cdn_url, timeout=60)
        resp.raise_for_status()
        buf = io.BytesIO(resp.content)
        table = pq.read_table(buf, columns=["prompt", "response"], use_threads=False)
        df = table.to_pandas()
        for _, row in df.iterrows():
            prompt = row.get("prompt") or ""
            response = row.get("response") or ""
            if not prompt or not response:
                continue
            yield {"prompt": str(prompt), "response": str(response)}

    def __iter__(self):
        for item in self.files:
            yield from self._stream_file(item["cdn_url"])
```

#### `surrogate/training/lightning_runner.py`
```python
import time
from pathlib import Path

from lightning import Lightning, Teamspace, Machine, Studio

LIGHTNING_ACCOUNT = "lightning-lambda-prod"  # H200 available (paid) or fallback handled below
MACHINE = Machine.L40S  # fallback if H200 unavailable in free tier

class LightningResilientTrainer:
    def __init__(self, studio_name, script_path, work_dir="."):
        self.studio_name = studio_name
        self.script_path = Path(script_path)
        self.work_dir = Path(work_dir)
        self.lightning = Lightning()

    def _find_running_studio(self):
        teamspace = Teamspace()
        for s in teamspace.studios:
            if s.name == self.studio_name and s.status == "Running":
                return s
        return None

    def _start_studio(self):
        # Try H200 first; fallback to L40S
        try:
            return Studio(
                name=self.studio_name,
                machine=Machine.H200,
                cloud=LIGHTNING_ACCOUNT,
                create_ok=True,
            )
        except Exception:
            return Studio(
                name=self.studio_name,
                machine=MACHINE,
                create_ok=True,
            )

    def run(self, args=None):
        studio = self._find_running_studio() or self._start_studio()
        cmd = ["python", str(self.script_path.name)] + (args or [])

        while True:
            if studio.status != "Running":
                print(f"Studio {self.studio_name} not running (status={studio.status}). Restarting...")
                studio = self._start_studio()
                time.sleep(30)
                continue

            try:
                run = studio.run(
                    command=" ".join(cmd),
                    cwd=str(self.work_dir.absolute()),
                    wait=False,
                )
                print(f"Started run {getattr(run, 'id', 'unknown')}")
                # Poll until completion or idle-stop
                while True:
                    if studio.status != "Running":
                        print("Studio stopped (idle or error). Will restart.")
                        break
                    time.sleep(60)
            except Exception as exc:
                print(f"Run error: {exc}")
                time.sleep(60)
```

#### `surrogate/training/hf_sibling_writer.py`
```python
import hashlib
import os

from huggingface_hub import HfApi

HF_TOKEN = os.getenv("HF_TOKEN")
BASE_REPO = os.getenv("HF_DATASET_REPO", "my-org/surrogate-data")
SIBLINGS = [f"{BASE_REPO}-sib-{i}" for i in range(5)]  # 5 repos -> 640/hr aggregate
api = HfApi(token=HF_TOKEN)

def pick_repo(slug: str) -> str:
    """Deterministic repo pick for commit-cap spread."""
    digest = hashlib.sha256(slug.encode()).hexdigest()
    idx = int(digest, 16) % len(SIBLINGS)
    return SIBLINGS[idx]

def upload_parquet(repo_path: str, local_path: str, commit_message: str = "upload") -> str:
    """Upload to sibling repo; CDN downloads during training are not rate-limited."""
    repo = pick_repo(repo_path)
    api.upload_file(
        path_or_fileobj=local_path,
        path_in_repo=repo_path,
        repo_id=repo,
        repo_type="dataset",
        commit_message=commit_message,
    )
    return f"https://huggingface.co/datasets/{repo}/resolve/main/{repo_path}"
```
