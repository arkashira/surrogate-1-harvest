### 1. CDN Manifest Generator
`/opt/axentx/airship/surrogate/ops/training_manifest.py`
```python
#!/usr/bin/env python3
"""
Generate CDN-only manifest for Surrogate training.
Usage:
  python training_manifest.py --repo org/surrogate-data --date 2026-05-03 --out manifest.json
"""
import argparse
import json
import os
import sys
from datetime import datetime

from huggingface_hub import list_repo_tree, hf_hub_download

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def build_manifest(repo: str, date: str, out_path: str):
    folder = f"batches/mirror-merged/{date}"
    tree = list_repo_tree(repo=repo, path=folder, recursive=False)
    files = [f.rfilename for f in tree if f.rfilename.endswith(".parquet")]

    manifest = {
        "repo": repo,
        "date": date,
        "generated_at": datetime.utcnow().isoformat() + "Z",
        "files": [
            {
                "path": p,
                "cdn_url": CDN_TEMPLATE.format(repo=repo, path=p),
                "local_cache": hf_hub_download(repo, p, cache_dir=os.getenv("HF_HOME", "~/.cache/huggingface"))
            }
            for p in sorted(files)
        ]
    }

    os.makedirs(os.path.dirname(out_path), exist_ok=True)
    with open(out_path, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Wrote {len(files)} files to {out_path}")

if __name__ == "__main__":
    p = argparse.ArgumentParser()
    p.add_argument("--repo", default="org/surrogate-data")
    p.add_argument("--date", required=True)
    p.add_argument("--out", default="manifest.json")
    args = p.parse_args()
    build_manifest(args.repo, args.date, args.out)
```

### 2. Lightning Studio Reuse Guard
`/opt/axentx/airship/surrogate/ops/lightning_guard.py`
```python
#!/usr/bin/env python3
"""
Reuse a running Lightning Studio or start L40S.
Guards against idle-stop killing training.
"""
import time
import lightning as L

STUDIO_NAME = "surrogate-train-l40s"
MACHINE = L.Machine.L40S

def get_or_start_studio():
    teamspace = L.Teamspace()
    for s in teamspace.studios:
        if s.name == STUDIO_NAME and s.status == "running":
            print(f"Reusing running studio: {s.name}")
            return s

    print(f"Starting new studio: {STUDIO_NAME}")
    return L.Studio(
        name=STUDIO_NAME,
        machine=MACHINE,
        create_ok=True,
    )

def run_with_idle_guard(studio, cmd):
    """Run cmd in studio; restart if idle-stopped."""
    for attempt in range(3):
        if studio.status != "running":
            print(f"Studio stopped (attempt {attempt}). Restarting...")
            studio.start(machine=MACHINE)
            time.sleep(30)
        try:
            studio.run(cmd)
            return
        except Exception as e:
            print(f"Run failed: {e}")
            time.sleep(60)
    raise RuntimeError("Training failed after 3 attempts")

if __name__ == "__main__":
    # Example usage from cron:
    # bash surrogate/ops/run_training.sh
    studio = get_or_start_studio()
    run_with_idle_guard(studio, "python surrogate/train.py --manifest manifest.json")
```

### 4. Cron-safe Wrapper
`/opt/axentx/airship/surrogate/ops/run_training.sh`
```bash
#!/usr/env/bin bash
# Cron-safe wrapper: ensures bash, sets env, handles 429 backoff
set -euo pipefail
export SHELL=/bin/bash
export HF_HOME="${HF_HOME:-$HOME/.cache/huggingface}"

cd "$(dirname "$0")/../.."

# Generate manifest (single API call; safe from 429 if within window)
python surrogate/ops/training_manifest.py --date "$(date +%Y-%m-%d)" --out manifest.json

# Launch guarded training
python surrogate/ops/lightning_guard.py
```
Make executable:
```bash
chmod +x /opt/axentx/airship/surrogate/ops/run_training.sh
```

Crontab entry (runs at 02:00 UTC):
```cron
SHELL=/bin/bash
0 2 * * * /opt/axentx/airship/surrogate/ops/run_training.sh >> /var/log/airship_training.log 2>&1
```
