#### 1. Add route handler (backend stub)

File: `backend/routes/cost_anomaly.py` (or equivalent)

```python
# backend/routes/cost_anomaly.py
from fastapi import APIRouter, HTTPException
from datetime import datetime, timezone
from typing import Optional
import logging

from services.knowledge_graph import KnowledgeGraphService

router = APIRouter()
logger = logging.getLogger(__name__)


@router.get("/api/v1/cost-anomaly/signal/top-hub", response_model=dict)
async def get_top_hub_cost_anomaly_signal() -> dict:
    """
    Deterministic, read-only endpoint.
    Queries knowledge graph for today's top hub (most-connected) and returns
    the strongest cost-anomaly signal with full context.
    """
    try:
        kg = KnowledgeGraphService()
        today_str = datetime.now(timezone.utc).strftime("%Y-%m-%d")

        # Step 1: Find top hub for today (most-connected)
        top_hub = kg.get_top_hub(date=today_str)
        if not top_hub:
            raise HTTPException(status_code=404, detail="No top hub found for today")

        # Step 2: Get strongest cost-anomaly signal for this hub
        signal = kg.get_strongest_cost_anomaly_signal(hub=top_hub["id"], date=today_str)
        if not signal:
            raise HTTPException(status_code=404, detail="No cost-anomaly signal found")

        response = {
            "request": {
                "endpoint": "/api/v1/cost-anomaly/signal/top-hub",
                "method": "GET",
                "timestamp_utc": datetime.now(timezone.utc).isoformat(),
            },
            "top_hub": top_hub,
            "cost_anomaly_signal": signal,
            "context": {
                "date": today_str,
                "read_only": True,
                "deterministic": True,
                "no_side_effects": True,
            },
        }

        logger.info(
            "Top-hub cost-anomaly signal returned",
            extra={"hub": top_hub.get("id"), "signal_strength": signal.get("strength")},
        )
        return response

    except Exception as exc:
        logger.exception("Failed to fetch top-hub cost-anomaly signal")
        raise HTTPException(status_code=500, detail=str(exc)) from exc
```

```python
# services/knowledge_graph.py
from datetime import date
from typing import Optional, Dict, Any


class KnowledgeGraphService:
    """
    Lightweight read-only interface to knowledge graph.
    Uses deterministic rules for top-hub selection and signal strength.
    """

    def __init__(self):
        # In real implementation, this would connect to graph DB or RAG index.
        # For ops/stub purposes, we simulate deterministic behavior.
        self._graph = self._load_deterministic_graph()

    def _load_deterministic_graph(self) -> Dict[str, Any]:
        # Simulated graph data for today. Deterministic and read-only.
        today = str(date.today())
        return {
            today: {
                "nodes": [
                    {"id": "MOC", "type": "hub", "connections": 42, "label": "MOC"},
                    {"id": "RI-Coverage", "type": "hub", "connections": 31, "label": "RI Coverage"},
                    {"id": "AWS-EC2-Overprovision", "type": "anomaly", "connections": 18},
                ],
                "signals": [
                    {
                        "id": "SIG-001",
                        "hub_id": "MOC",
                        "type": "cost-anomaly",
                        "strength": 0.93,
                        "description": "Spike in compute spend in MOC-linked accounts",
                        "details": {
                            "accounts": ["acct-1001", "acct-1002"],
                            "services": ["EC2", "EKS"],
                            "delta_pct": 47.2,
                            "window": "last_24h",
                        },
                        "recommendation": "Review reserved instance coverage and rightsizing opportunities",
                    },
                    {
                        "id": "SIG-002",
                        "hub_id": "RI-Coverage",
                        "type": "cost-anomaly",
                        "strength": 0.62,
                        "description": "Declining RI utilization",
                        "details": {
                            "accounts": ["acct-1001"],
                            "services": ["RDS"],
                            "delta_pct": -12.4,
                            "window": "last_7d",
                        },
                        "recommendation": "Consider modifying RI mix or enabling convertible RIs",
                    },
                ],
            }
        }

    def get_top_hub(self, date: str) -> Optional[Dict[str, Any]]:
        """Return most-connected hub for given date."""
        day_data = self._graph.get(date)
        if not day_data:
            return None

        hubs = [n for n in day_data["nodes"] if n["type"] == "hub"]
        if not hubs:
            return None

        # Deterministic: pick hub with highest connections
        top = max(hubs, key=lambda h: h["connections"])
        return top

    def get_strongest_cost_anomaly_signal(self, hub: str, date: str) -> Optional[Dict[str, Any]]:
        """Return strongest cost-anomaly signal for given hub and date."""
        day_data = self._graph.get(date)
        if not day_data:
            return None

        signals = [
            s
            for s in day_data["signals"]
            if s["hub_id"] == hub and s["type"] == "cost-anomaly"
        ]
        if not signals:
            return None

        # Deterministic: pick highest strength
        strongest = max(signals, key=lambda s: s["strength"])
        return strongest
```
