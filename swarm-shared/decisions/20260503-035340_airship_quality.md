### 1) `scripts/list_hf_files.py`

```python
#!/usr/bin/env python3
"""
List HF dataset files (non-recursive per folder) and emit CDN URLs.
Run on Mac after HF API rate-limit window clears.
Output: JSON list of {repo, path, url}
"""
import json
import os
import sys
from typing import List, Dict

try:
    from huggingface_hub import HfApi
except ImportError:
    print("Install: pip install huggingface_hub", file=sys.stderr)
    sys.exit(1)

HF_REPO = os.getenv("HF_REPO", "axentx/surrogate-dataset")  # datasets repo
FOLDERS = os.getenv("HF_FOLDERS", "batches/mirror-merged").split(",")  # comma-sep

def list_files() -> List[Dict[str, str]]:
    api = HfApi()
    out = []
    for folder in FOLDERS:
        folder = folder.strip()
        try:
            tree = api.list_repo_tree(repo_id=HF_REPO, path=folder, recursive=False)
        except Exception as e:
            print(f"WARN: failed to list {folder}: {e}", file=sys.stderr)
            continue
        for entry in tree:
            if entry.type != "file":
                continue
            path = entry.path
            cdn_url = f"https://huggingface.co/datasets/{HF_REPO}/resolve/main/{path}"
            out.append({"repo": HF_REPO, "path": path, "url": cdn_url})
    return out

if __name__ == "__main__":
    files = list_files()
    json.dump(files, sys.stdout, indent=2)
    print(file=sys.stderr)
    print(f"Listed {len(files)} files", file=sys.stderr)
```

Make executable:  
```bash
chmod +x scripts/list_hf_files.py
```

class CDNParquetDataset(IterableDataset):
    """
    Stream parquet files via CDN URLs (no HF API auth).
    Projects only prompt/response fields if schema is mixed.
    """
    def __init__(self, files: list[dict], buffer_size: int = 1024 * 1024):
        super().__init__()
        self.files = files
        self.buffer_size = buffer_size

    def _fetch_bytes(self, url: str) -> bytes:
        for attempt in range(1, MAX_RETRIES + 1):
            try:
                resp = requests.get(url, timeout=CDN_TIMEOUT, headers={})
                resp.raise_for_status()
                return resp.content
            except Exception as e:
                if attempt == MAX_RETRIES:
                    raise
                sleep_sec = 2 ** attempt
                print(f"WARN: CDN fetch failed ({e}), retry {attempt}/{MAX_RETRIES} in {sleep_sec}s", file=sys.stderr)
                time.sleep(sleep_sec)
        raise RuntimeError("unreachable")

    def _iter_table(self, url: str):
        raw = self._fetch_bytes(url)
        # Use temp file for pyarrow to memory-map if needed
        with tempfile.NamedTemporaryFile(suffix=".parquet", delete=False) as f:
            f.write(raw)
            f.flush()
            try:
                table = pq.read_table(f.name, columns=["prompt", "response"])
            except Exception:
                # fallback: read all and project
                table = pq.read_table(f.name)
                cols = [c for c in table.column_names if c in {"prompt", "response"}]
                if not cols:
                    # heuristic: first text-like column as prompt, second as response
                    cols = table.column_names[:2]
                table = table.select(cols)
            finally:
                os.unlink(f.name)
        return table

    def _iter_table_rows(self, url: str) -> Iterator[dict]:
        table = self._iter_table(url)
        for batch in table.to_batches(max_chunksize=1024):
            cols = batch.column_names
            for i in range(batch.num_rows):
                row = {c: batch[c][i].as_py() for c in cols}
                yield row

    def __iter__(self) -> Iterator[dict]:
        worker_info = torch.utils.data.get_worker_info()
        if worker_info is None:
            files = self.files
        else:
            per_worker = len(self.files) // worker_info.num_workers
            files = self.files[worker_info.id * per_worker : (worker_info.id + 1) * per_worker]

        for entry in files:
            url = entry["url"]
            try:
                yield from self._iter_table_rows(url)
            except Exception as e:
                print(f"ERROR: failed to process {url}: {e}", file=sys.stderr)
                continue
