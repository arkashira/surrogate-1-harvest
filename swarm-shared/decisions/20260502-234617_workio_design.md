# workio / design

## Final synthesized solution

Adopt a **frontend-first, strongly-typed route catalog** with **runtime validation** and **centralized role/navigation helpers**, plus **lightweight backend mirroring for critical endpoints** to close validation gaps. This resolves contradictions by prioritizing correctness (Zod validation at boundaries), concrete actionability (minimal footprint, clear rollout), and safety (typed hrefs, refactor-proof paths).

### 2. Typed router helpers (frontend)

`workio/src/lib/router.ts`

```ts
import { Route, type RouteDef } from '../routes';
import { z } from 'zod';

export function routeHref<T extends RouteDef>(
  key: keyof typeof Route,
  search?: Partial<z.infer<T['searchSchema']>>
): string {
  const def = Route[key as keyof typeof Route] as T;
  const validated = def.searchSchema.optional().parse(search);
  const sp = new URLSearchParams(validated as Record<string, string>);
  const qs = sp.toString();
  return def.path + (qs ? `?${qs}` : '');
}

export function validateSearch<T extends RouteDef>(
  key: keyof typeof Route,
  input: unknown
): z.infer<T['searchSchema']> {
  const def = Route[key as keyof typeof Route] as T;
  return def.searchSchema.parse(input);
}

export function routeAllowedFor<T extends RouteDef>(
  key: keyof typeof Route,
  userRole: string
): boolean {
  const def = Route[key as keyof typeof Route] as T;
  return def.roles.includes(userRole as (typeof def.roles)[number]);
}
```

### 3. Type-safe search params hook

`workio/src/hooks/useTypedSearchParams.ts`

```ts
import { useSearchParams } from 'react-router-dom';
import { validateSearch } from '../lib/router';
import { Route } from '../routes';
import { useCallback } from 'react';

export function useTypedSearchParams<T extends keyof typeof Route>(routeKey: T) {
  const [searchParams, setSearchParams] = useSearchParams();

  const parsed = validateSearch(routeKey, Object.fromEntries(searchParams));

  const set = useCallback(
    (next: Partial<z.infer<(typeof Route)[T]['searchSchema']>>) => {
      const nextSearch = new URLSearchParams(searchParams);
      Object.entries(next).forEach(([k, v]) => {
        if (v === undefined || v === null) nextSearch.delete(k);
        else nextSearch.set(k, String(v));
      });
      setSearchParams(nextSearch);
    },
    [searchParams, setSearchParams]
  );

  return [parsed, set] as const;
}
```

### 4. Guarded route component

`workio/src/components/GuardedRoute.tsx`

```tsx
import { ReactNode } from 'react';
import { routeAllowedFor } from '../lib/router';
import { Route } from '../routes';

interface GuardedRouteProps {
  routeKey: keyof typeof Route;
  userRole: string;
  children: ReactNode;
  fallback?: ReactNode;
}

export function GuardedRoute({
  routeKey,
  userRole,
  children,
  fallback = <div>Access denied</div>,
}: GuardedRouteProps) {
  if (!routeAllowedFor(routeKey, userRole)) {
    return fallback;
  }
  return <>{children}</>;
}
```
