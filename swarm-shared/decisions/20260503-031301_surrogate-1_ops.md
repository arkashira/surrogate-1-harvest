def parse_parquet_to_pairs(content: bytes) -> List[Dict[str, str]]:
    """
    Project heterogeneous parquet to {prompt,response}.
    Avoid load_dataset(streaming=True) on mixed schemas.
    """
    import io
    import pyarrow.parquet as pq
    table = pq.read_table(io.BytesIO(content))
    cols = table.column_names
    pairs = []
    # Heuristic field names
    prompt_col = next((c for c in cols if "prompt" in c.lower()), None)
    response_col = next((c for c in cols if "response" in c.lower() or "completion" in c.lower()), None)
    if prompt_col is None or response_col is None:
        # fallback: first two text-like columns
        text_cols = [c for c in cols if table.schema.field(c).type in (pa.string(), pa.large_string())]
        if len(text_cols) >= 2:
            prompt_col, response_col = text_cols[0], text_cols[1]
        else:
            return []
    for i in range(table.num_rows):
        row = table.slice(i, 1).to_pydict()
        prompt = row.get(prompt_col, [""])[0]
        response = row.get(response_col, [""])[0]
        if prompt and response:
            pairs.append({"prompt": str(prompt), "response": str(response)})
    return pairs

def parse_jsonl_to_pairs(content: bytes) -> List[Dict[str, str]]:
    pairs = []
    for line in content.decode("utf-8").strip().splitlines():
        line = line.strip()
        if not line:
            continue
        try:
            obj = json.loads(line)
        except Exception:
            continue
        prompt = obj.get("prompt") or obj.get("input") or obj.get("question")
        response = obj.get("response") or obj.get("output") or obj.get("answer")
        if prompt and response:
            pairs.append({"prompt": str(prompt), "response": str(response)})
    return pairs

def parse_file(content: bytes, file_path: str) -> List[Dict[str, str]]:
    if file_path.endswith(".parquet"):
        return parse_parquet_to_pairs(content)
    elif file_path.endswith(".jsonl"):
        return parse_jsonl_to_pairs(content)
    else:
        return []
