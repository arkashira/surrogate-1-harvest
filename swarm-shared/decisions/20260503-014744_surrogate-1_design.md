## Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** that:

- Uses a **single `list_repo_tree` snapshot** (JSON manifest) generated once per date on the Mac orchestrator and committed to the repo (or passed via `MANIFEST_URL`).  
- Each GitHub Actions shard (16 total) reads the manifest, filters by deterministic hash-bucket (`shard_id = hash(slug) % 16 == SHARD_ID`), then downloads only its assigned files via **HF CDN** (`https://huggingface.co/datasets/.../resolve/main/...`) with **no Authorization header** (bypasses API rate limits).  
- Projects heterogeneous schemas to `{prompt, response}` at parse time (avoids `pyarrow.CastError`).  
- Dedups via central `lib/dedup.py` SQLite store (same as HF Space) and writes `batches/public-merged/<date>/shard<N>-<HHMMSS>.jsonl`.  
- Runs in GitHub Actions with `HF_TOKEN` for dedup-store writes and final dataset push.

# Deterministic shard assignment
def shard_of(slug: str, n_shards: int = 16) -> int:
    return int(hashlib.md5(slug.encode()).hexdigest(), 16) % n_shards

def load_manifest(manifest_path: Path) -> List[Dict[str, Any]]:
    if not manifest_path.exists():
        raise FileNotFoundError(f"Manifest not found: {manifest_path}")
    with manifest_path.open() as f:
        data = json.load(f)
    if isinstance(data, dict) and "files" in data:
        return data["files"]
    if isinstance(data, list):
        return data
    raise ValueError("Manifest must be list of files or dict with 'files' key")

def cdn_download(client: httpx.Client, cdn_url: str, retries: int = 3) -> Optional[bytes]:
    for attempt in range(1, retries + 1):
        try:
            resp = client.get(cdn_url, timeout=30.0)
            resp.raise_for_status()
            return resp.content
        except httpx.HTTPStatusError as e:
            if e.response.status_code == 404:
                return None
            if attempt == retries:
                raise
        except (httpx.RequestError, httpx.HTTPError) as e:
            if attempt == retries:
                raise
        time.sleep(2**attempt)
    return None

def project_to_pair(raw_bytes: bytes, path: str) -> Optional[Dict[str, str]]:
    """
    Project heterogeneous file schemas to {prompt, response}.
    Supports:
      - JSONL lines with prompt/response fields (case-insensitive)
      - Parquet files with prompt/response columns
      - JSON objects with prompt/response keys
    """
    path_l = path.lower()
    try:
        if path_l.endswith(".parquet"):
            table = pq.read_table(pa.BufferReader(raw_bytes))
            df = table.to_pandas()
            prompt_col = next((c for c in df.columns if "prompt" in c.lower()), None)
            response_col = next((c for c in df.columns if "response" in c.lower()), None)
            if prompt_col is None or response_col is None:
                return None
            row = df.iloc[0]
            return {"prompt": str(row[prompt_col]), "response": str(row[response_col])}

        # Try JSONL first (line-oriented)
        text = raw_bytes.decode("utf-8", errors="replace").strip()
        if "\n" in text:
            for line in text.splitlines():
                line = line.strip()
                if not line:
                    continue
                obj = json.loads(line)
                prompt = obj.get("prompt") or obj.get("Prompt") or obj.get("input") or obj.get("Input")
                response = obj.get("response") or obj.get("Response") or obj.get("output") or obj.get("Output")
                if prompt is not None and response is not None:
                    return {"prompt": str(prompt), "response": str(response)}
            return None

        # Single JSON object
        obj = json.loads(text)
        prompt = obj.get("prompt") or obj.get("Prompt") or obj.get("input") or obj.get("Input")
        response = obj.get("response") or obj.get("Response") or obj.get("output") or obj.get("Output")
        if prompt is None or response is None:
            return None
        return {"prompt": str(prompt), "response": str(response)}
    except Exception:
        return None

def upload_batch(output_dir: Path, shard: int, date_str: str, pairs: List[Dict[str, str]], hf_token: str):
    if not pairs:
        return
    output_dir.mkdir(parents=True, exist_ok=True)
    ts = datetime.now(timezone.utc).strftime("%H%M%S")
    fname = f"shard{shard}-{ts}.jsonl"
    out_path = output_dir / fname
    with out_path.open("w") as f:
        for p in pairs:
            f.write(json.dumps(p, ensure_ascii=False) + "\n")

            ["git", "add", str(out_path)],

            capture_output=True,
        )
        subprocess.run(
            ["git", "config", "user.email", "bot@axentx.dev"],
            cwd=output_dir.parent.parent,
            check=True,
            capture_output=True,
        )
        subprocess.run(
            ["git", "config", "user.name", "surrogate-1-bot"],
            cwd=output_dir.parent.parent,
            check=True,
            capture_output=True,
        )
        subprocess.run(
            ["git", "commit", "-m", f"shard{shard} {date_str} {ts}"],
            cwd=output_dir.parent.parent,
            check=True,
            capture_output=True,
        )
        subprocess.run(

            cwd=output_dir.parent.parent,
            check=True,
            capture_output=True,
        )
    except subprocess.CalledProcessError as e:
        print(f"HF push failed: {e.stderr.decode() if e.stderr else e}", file=sys.stderr)
        raise
