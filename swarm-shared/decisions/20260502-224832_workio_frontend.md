# workio / frontend

## Final Synthesis — Resilient, Idempotent Attendance Queue (Frontend-only)

Chosen approach: **IndexedDB-backed queue with idempotency keys, durable retry state, exponential backoff + jitter, online-triggered + periodic flush, and cross-tab safe operations**.  
Contradictions resolved in favor of **correctness and concrete actionability**:

- **Storage**: Use IndexedDB (not localStorage) for durability, size, and structured retry state.
- **Idempotency**: Always generate and send a stable idempotency key to `/api/attendance`.
- **Retry**: Persist attempts and lastAttemptAt; cap retries (5) and keep exhausted items for manual review.
- **Backoff**: Use exponential backoff + jitter per-item to avoid thundering herd.
- **Cross-tab safety**: Prefer atomic DB operations; avoid shared in-memory state; rely on IndexedDB as source of truth.
- **Flush triggers**: periodic interval + online event + immediate attempt after enqueue when online.
- **UI feedback**: expose per-item status (pending/synced/failed) and connectivity state.

#### 2.1 IndexedDB queue (`src/lib/queue/db.ts`)

```ts
import { openDB, DBSchema, IDBPDatabase } from 'idb';

interface AttendanceQueueDB extends DBSchema {
  pending: {
    key: string;
    value: {
      id: string; // idempotency key
      type: 'CLOCK_IN' | 'CLOCK_OUT';
      payload: {
        tenantId: string;
        userId: string;
        timestamp: string;
        location?: { lat: number; lng: number };
      };
      attempts: number;
      createdAt: number;
      lastAttemptAt?: number;
      nextAttemptAt?: number; // for backoff scheduling
    };
  };
}

const DB_NAME = 'workio-attendance-queue';
const DB_VERSION = 1;

export async function getQueueDB(): Promise<IDBPDatabase<AttendanceQueueDB>> {
  return openDB<AttendanceQueueDB>(DB_NAME, DB_VERSION, {
    upgrade(db) {
      db.createObjectStore('pending', { keyPath: 'id' });
    },
  });
}

export async function enqueuePending(item: AttendanceQueueDB['pending']['value']) {
  const db = await getQueueDB();
  await db.put('pending', item);
}

export async function peekPending(limit = 20): Promise<AttendanceQueueDB['pending']['value'][]> {
  const db = await getQueueDB();
  const all = await db.getAll('pending');
  const now = Date.now();
  // prioritize oldest created, but respect nextAttemptAt when present
  return all
    .sort((a, b) => {
      const aTime = a.nextAttemptAt ?? a.createdAt;
      const bTime = b.nextAttemptAt ?? b.createdAt;
      return aTime - bTime;
    })
    .slice(0, limit);
}

export async function getPending(id: string) {
  const db = await getQueueDB();
  return db.get('pending', id);
}

export async function removePending(id: string) {
  const db = await getQueueDB();
  await db.delete('pending', id);
}

export async function updatePending(
  id: string,
  updates: Partial<Omit<AttendanceQueueDB['pending']['value'], 'id' | 'payload' | 'type'>>
) {
  const db = await getQueueDB();
  const item = await db.get('pending', id);
  if (!item) return;
  Object.assign(item, updates);
  await db.put('pending', item);
}
```

#### 2.2 Queue logic with idempotency and backoff (`src/lib/queue/attendanceQueue.ts`)

```ts
import { v4 as uuidv4 } from 'uuid';
import {
  enqueuePending,
  peekPending,
  removePending,
  updatePending,
  getPending,
} from './db';

const MAX_ATTEMPTS = 5;
const BASE_DELAY = 2000; // 2s

export type AttendanceCommand = {
  type: 'CLOCK_IN' | 'CLOCK_OUT';
  tenantId: string;
  userId: string;
  timestamp?: string;
  location?: { lat: number; lng: number };
};

function exponentialBackoff(attempt: number) {
  const jitter = Math.random() * 1000 + 500; // 0.5s - 1.5s jitter
  return Math.min(BASE_DELAY * 2 ** Math.max(0, attempt - 1), 30_000) + jitter;
}

export async function submitAttendance(command: AttendanceCommand) {
  // Deterministic-ish idempotency key while ensuring uniqueness across retries/collisions
  const idempotencyKey = `${command.userId}-${command.type}-${command.timestamp || Date.now()}-${uuidv4()}`;
  const item = {
    id: idempotencyKey,
    type: command.type,
    payload: {
      tenantId: command.tenantId,
      userId: command.userId,
      timestamp: command.timestamp || new Date().toISOString(),
      location: command.location,
    },
    attempts: 0,
    createdAt: Date.now(),
  } as const;

  await enqueuePending(item);
  // best-effort immediate flush if online
  if (navigator.onLine) {
    void flushQueue();
  }
  return idempotencyKey;
}

export async function flushQueue(signal?: AbortSignal) {
  const items = await peekPending(20);
  for (const item of items) {
    if (signal?.aborted) break;
    if (item.attempts >= MAX_ATTEMPTS) continue; // keep for inspection

    // honor backoff schedule
    const now = Date.now();
    if (item.nextAttemptAt && now < item.nextAttemptAt) continue;

    try {
      await updatePending(item.id, { attempts: item.attempts + 1, lastAttemptAt: now });

      const res = await fetch('/api/attendance', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          ...item.payload,
          idempotencyKey: item.id,
        }),
        signal,
      });

      if (!res.ok) {
        // schedule next attempt with backoff
        const nextDelay = exponentialBackoff(item.attempts + 1);
        await updatePending(item.id, { nextAttemptAt: Date.now() + nextDelay });
        continue;
      }

      await removePending(item.id);
    } catch (err) {
      // network or other error — schedule retry
      const nextDelay = exponentialBackoff(item.attempts + 1);
      await updatePending(item.id, { nextAttemptAt: Date.now() + nextDelay });
    }
  }
}
