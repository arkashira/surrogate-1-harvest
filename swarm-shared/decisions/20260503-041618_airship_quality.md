### 1. Deterministic CDN Manifest (Single API Call)

**Location:** `/opt/axentx/airship/scripts/generate_cdn_manifest.sh`  
**Run:** Once per day on Mac orchestration host (or in CI after new batch lands).

```bash
#!/usr/bin/env bash
set -euo pipefail

REPO="${1:-datasets/axentx/surrogate-shards}"
DATE_DIR="${2:-$(date +%Y-%m-%d)}"
OUT_DIR="/opt/axentx/airship/surrogate/config"
MANIFEST="${OUT_DIR}/cdn_manifest_${DATE_DIR}.json"

mkdir -p "${OUT_DIR}"

python3 - <<PY
import os, json, sys
from huggingface_hub import HfApi

api = HfApi()
repo_id = os.environ["REPO"]
folder = os.environ["DATE_DIR"]

try:
    items = api.list_repo_tree(repo_id=repo_id, path=folder, recursive=False)
except Exception as e:
    print(f"HF API error: {e}", file=sys.stderr)
    sys.exit(1)

files = []
for item in items:
    name = item["path"] if isinstance(item, dict) else getattr(item, "path", "")
    if name.lower().endswith(".parquet"):
        cdn_url = f"https://huggingface.co/datasets/{repo_id}/resolve/main/{folder}/{name}"
        files.append({"filename": name, "cdn_url": cdn_url, "local_key": f"{folder}/{name}"})

if not files:
    print("No parquet files found", file=sys.stderr)
    sys.exit(1)

with open(os.environ["MANIFEST"], "w") as f:
    json.dump({"date": folder, "repo": repo_id, "files": files}, f, indent=2)
print(f"Wrote {len(files)} files to {os.environ['MANIFEST']}")
PY

echo "Manifest generated: ${MANIFEST}"
```

### 2. CDN-Only PyTorch Dataset (No HF API During Training)

**Location:** `/opt/axentx/airship/surrogate/data/cdn_parquet_dataset.py`

```python
import json
import io
import pyarrow.parquet as pq
import requests
import torch
from torch.utils.data import Dataset, IterableDataset
from typing import List, Dict, Any

class CDNParquetDataset(IterableDataset):
    """
    Streams parquet shards directly from HuggingFace CDN.
    No HF API calls during training → immune to 429 rate limits.
    """

    def __init__(self, manifest_path: str, columns: List[str] = ("prompt", "response")):
        with open(manifest_path) as f:
            manifest = json.load(f)
        self.files = [f["cdn_url"] for f in manifest["files"]]
        self.columns = columns
        self.timeout = 30
        self.max_retries = 3

    def _fetch_shard(self, url: str) -> pq.Table:
        for attempt in range(self.max_retries):
            try:
                resp = requests.get(url, timeout=self.timeout)
                resp.raise_for_status()
                return pq.read_table(io.BytesIO(resp.content))
            except Exception as e:
                if attempt == self.max_retries - 1:
                    raise
                wait = 2 ** attempt
                print(f"Retry {url} in {wait}s ({e})")
                import time
                time.sleep(wait)

    def __iter__(self):
        worker_info = torch.utils.data.get_worker_info()
        if worker_info is None:
            files = self.files
        else:
            per_worker = len(self.files) // worker_info.num_workers
            files = self.files[
                worker_info.id * per_worker : (worker_info.id + 1) * per_worker
            ]

        for url in files:
            table = self._fetch_shard(url)
            df = table.select(self.columns).to_pandas()
            for _, row in df.iterrows():
                yield {col: row[col] for col in self.columns}
```

### 3. Resilient Lightning Launcher (Reuse + Auto-Restart)

**Location:** `/opt/axentx/airship/surrogate/training/launch.py`

```python
import os
import sys
import time
from lightning import Lightning, L40S

LIGHTNING_TEAMSPACE = os.getenv("LIGHTNING_TEAMSPACE", "default")
STUDIO_NAME = os.getenv("STUDIO_NAME", "surrogate-l40s-train")
CHECKPOINT_DIR = os.getenv("CHECKPOINT_DIR", "/opt/axentx/airship/surrogate/checkpoints")

os.makedirs(CHECKPOINT_DIR, exist_ok=True)

def get_or_create_studio():
    lightning = Lightning()
    teamspace = lightning.teamspace(LIGHTNING_TEAMSPACE)

    for studio in teamspace.studios():
        if studio.name == STUDIO_NAME:
            if studio.status == "running":
                print(f"Reusing running studio: {studio.id}")
                return studio
            else:
                print(f"Studio stopped ({studio.status}), restarting...")
                studio.start(machine=L40S())
                return studio

    print("Creating new studio...")
    return teamspace.studios().create(
        name=STUDIO_NAME,
        machine=L40S(),
    )

def run_training():
    studio = get_or_create_studio()

    # Ensure training script is executable
    train_script = os.path.join(os.path.dirname(__file__), "train.py")
    os.chmod(train_script, 0o755)

    # Launch with resilient settings
    run = studio.run(
        target=train_script,
        requirements=["torch", "pyarrow", "requests", "lightning", "pandas"],
        # Optional: checkpoint sync to HF dataset repo for resume
        env={
            "CHECKPOINT_DIR": CHECKPOINT_DIR,
            "MANIFEST_PATH": "/opt/axentx/airship/surrogate/config/cdn_manifest_latest.json",
        },
    )
    print(f"Started run: {run.id}")
    return run

if __name__ == "__main__":
    run_training()
```
