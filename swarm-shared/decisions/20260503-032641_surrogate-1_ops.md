## Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16` (matrix) and optional `DATE_FOLDER` (defaults to today `YYYY-MM-DD`)
- Uses **single API call** from the runner (or pre-generated manifest) to list one date folder via `list_repo_tree(..., recursive=False)` → saves list to `manifest.json`
- During training/ingest: **zero HF API calls** — downloads only via CDN (`https://huggingface.co/datasets/.../resolve/main/...`) to bypass 429 rate limits
- Projects every file to `{prompt, response}` at parse time (avoids `pyarrow.CastError` from mixed schemas)
- Deduplicates via central `lib/dedup.py` md5 store
- Writes deterministic output: `batches/public-merged/<date>/shard<N>-<HHMMSS>.jsonl`
- Exits non-zero on unrecoverable errors; logs structured JSON for Actions

CDN-bypass, manifest-driven shard worker for surrogate-1 public dataset ingest.

Usage (GitHub Actions matrix):

def iso_now() -> str:
    return datetime.now(timezone.utc).strftime("%Y-%m-%dT%H:%M:%SZ")


def build_manifest(repo: str, date_folder: str, manifest_path: Optional[str]) -> List[str]:
    """Return flat list of file paths under date_folder (non-recursive)."""
    if manifest_path and Path(manifest_path).exists():
        with open(manifest_path) as f:
            return json.load(f)

    # single API call (rate-limited) — do once per run
    tree = list_repo_tree(repo=repo, path=date_folder, recursive=False)
    files = [item.rfilename for item in tree if item.type == "file"]
    if manifest_path:
        Path(manifest_path).parent.mkdir(parents=True, exist_ok=True)
        with open(manifest_path, "w") as f:
            json.dump(files, f)
    return files


def assign_shard(path: str, shard_id: int, shard_total: int) -> bool:
    """Deterministic shard assignment by file path hash."""
    digest = int(hashlib.sha256(path.encode()).hexdigest(), 16)
    return (digest % shard_total) == shard_id


def project_to_pair(local_path: Path) -> Optional[Dict[str, str]]:
    """
    Parse heterogeneous files and return {prompt, response}.
    Implement per-schema adapters here to avoid pyarrow.CastError.
    """
    suffix = local_path.suffix.lower()
    try:
        if suffix == ".jsonl":
            with open(local_path) as f:
                for line in f:
                    line = line.strip()
                    if not line:
                        continue
                    obj = json.loads(line)
                    # heuristic projection; extend as needed
                    prompt = obj.get("prompt") or obj.get("input") or obj.get("question") or ""
                    response = obj.get("response") or obj.get("output") or obj.get("answer") or ""
                    if prompt and response:
                        return {"prompt": str(prompt), "response": str(response)}
        elif suffix == ".parquet":
            import pyarrow.parquet as pq
            tbl = pq.read_table(local_path, columns=["prompt", "response"])
            df = tbl.to_pandas()
            for _, row in df.iterrows():
                if row.get("prompt") and row.get("response"):
                    return {"prompt": str(row["prompt"]), "response": str(row["response"])}
        elif suffix == ".json":
            with open(local_path) as f:
                obj = json.load(f)
                prompt = obj.get("prompt") or obj.get("input") or ""
                response = obj.get("response") or obj.get("output") or ""
                if prompt and response:
                    return {"prompt": str(prompt), "response": str(response)}
    except Exception as exc:
        print(json.dumps({"level": "warn", "msg": "parse_error", "path": str(local_path), "error": str(exc)}))
        return None
    return None


def download_via_cdn(repo: str, path: str, target: Path) -> None:
    url = CDN_TEMPLATE.format(repo=repo, path=path)
    # CDN tier: no Authorization header -> bypasses /api/ rate limits
    with requests.get(url, timeout=60, stream=True) as r:
        r.raise_for_status()
        target.parent.mkdir(parents=True, exist_ok=True)
        with open(target, "wb") as f:
            for chunk in r.iter_content(chunk_size=8192):
                f.write(chunk)
