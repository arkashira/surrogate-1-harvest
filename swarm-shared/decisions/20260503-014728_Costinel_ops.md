# Costinel / ops

## Final Implementation Plan — Top-Hub Signal Panel (Costinel Dashboard)

**Scope & value**  
- Frontend-only, read-only panel that surfaces the most-connected hub (default “MOC”) and its actionable proposals from the knowledge graph.  
- Resilient to missing backend: falls back to cache / local fixture.  
- Ships in <2h as incremental ops improvement.

**Chosen stack**: Vue 3 + TypeScript + fetch-first resilient data layer (aligns with existing codebase patterns).

### 3) Service: resilient knowledge-graph fetcher (`src/services/knowledgeGraph.ts`)

```ts
import type { KnowledgeGraph } from '@/types/knowledgeGraph';

const CACHE_KEY = 'costinel:hub-signal:cache';
const CACHE_TTL_MS = 5 * 60 * 1000; // 5m

function getCached(): KnowledgeGraph.Hub | null {
  try {
    const raw = localStorage.getItem(CACHE_KEY);
    if (!raw) return null;
    const { data, ts } = JSON.parse(raw);
    if (Date.now() - ts > CACHE_TTL_MS) return null;
    return data as KnowledgeGraph.Hub;
  } catch {
    return null;
  }
}

function setCached(data: KnowledgeGraph.Hub) {
  try {
    localStorage.setItem(
      CACHE_KEY,
      JSON.stringify({ data, ts: Date.now() })
    );
  } catch {
    // ignore storage failures
  }
}

export async function fetchHubSignal(
  hubId: string
): Promise<KnowledgeGraph.Hub | null> {
  try {
    const res = await fetch(`/api/knowledge-graph/hubs/${hubId}`, {
      credentials: 'same-origin',
      cache: 'no-store',
    });

    if (!res.ok) throw new Error(`HTTP ${res.status}`);
    const json = (await res.json()) as KnowledgeGraph.Hub;
    setCached(json);
    return json;
  } catch {
    return getCached();
  }
}
```

### 4) Composables: `useTopHub.ts`

```ts
import { ref } from 'vue';
import type { KnowledgeGraph } from '@/types/knowledgeGraph';
import { fetchHubSignal } from '@/services/knowledgeGraph';

const FALLBACK_HUB: KnowledgeGraph.Hub = {
  id: 'MOC',
  label: 'MOC',
  connections: 184,
  proposals: [
    {
      id: 'moc-ri-2026-05',
      title: 'Reserved Instance coverage gap (m5.large)',
      summary: 'Detected 42% RI coverage gap in us-east-1 for m5.large workloads. Recommend 1-year convertible RI to reduce cost 32%.',
      priority: 'high',
      impact: '$12,400/mo',
      link: '/ops/proposals/moc-ri-2026-05',
    },
    {
      id: 'moc-az-bal-2026-05',
      title: 'AZ imbalance on EKS node groups',
      summary: 'us-east-1a over-utilized while us-east-1b under-utilized. Rebalance recommended.',
      priority: 'medium',
      impact: 'Performance risk',
      link: '/ops/proposals/moc-az-bal-2026-05',
    },
  ],
  updatedAt: new Date().toISOString(),
};

export function useTopHub(options: { hubId?: string } = {}) {
  const hub = ref<KnowledgeGraph.Hub>(FALLBACK_HUB);
  const loading = ref(false);
  const error = ref<string | null>(null);

  async function fetchHub() {
    loading.value = true;
    error.value = null;

    try {
      const result = await fetchHubSignal(options.hubId || 'MOC');
      if (result) {
        hub.value = result;
      } else {
        // keep fallback visible; log for observability
        console.warn('[useTopHub] Using fallback hub data');
        error.value = 'Using cached/fallback data';
      }
    } catch (err) {
      error.value = (err as Error).message;
      console.warn('[useTopHub] Fetch failed', err);
    } finally {
      loading.value = false;
    }
  }

  return { hub, loading, error, fetchHub };
}
```
