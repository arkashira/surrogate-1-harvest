# workio / design

## 1. Diagnosis

- LINE webhook state changes (clock/leave/OT) commit to DB but frontend state is stale until hard refresh → perceived latency and silent failures.
- No client-side subscription layer; UI relies on local React state + manual refresh, so multi-tab/devices diverge.
- Missing optimistic UI + rollback for LINE-driven events; users can’t distinguish “pending LINE confirmation” vs “failed”.
- No lightweight server-sent events (SSE) or WebSocket channel to push verified state from backend → forces polling or full reloads.
- Schema lacks `updated_at` on key tables (`clock_records`, `leave_requests`, `ot_requests`) → frontend can’t efficiently diff or cache.

## 2. Proposed change

Add a minimal, resilient SSE channel (`/api/stream`) and extend schema with `updated_at` so the frontend can subscribe to tenant-scoped updates and apply optimistic/pessimistic patches without full reloads.

Scope:
- `server/src/db/schema.sql` — add `updated_at` defaults and indexes.
- `server/src/routes/stream.ts` — new SSE route with tenant scoping, last-event-id, and keepalive.
- `server/src/middleware/tenantAuth.ts` (or equivalent) — ensure tenant context is available for stream.
- Frontend: `src/lib/stream.ts` + hook `useTenantStream()` to subscribe and merge updates into React Query/state.

## 3. Implementation

### 3.1 DB schema (add `updated_at`)

```sql
-- server/src/db/schema.sql
-- Add to clock_records, leave_requests, ot_requests
ALTER TABLE clock_records ADD COLUMN IF NOT EXISTS updated_at TIMESTAMPTZ DEFAULT now();
ALTER TABLE leave_requests ADD COLUMN IF NOT EXISTS updated_at TIMESTAMPTZ DEFAULT now();
ALTER TABLE ot_requests ADD COLUMN IF NOT EXISTS updated_at TIMESTAMPTZ DEFAULT now();

-- Indexes for efficient tenant+recent lookups
CREATE INDEX IF NOT EXISTS idx_clock_records_tenant_updated ON clock_records(tenant_id, updated_at DESC);
CREATE INDEX IF NOT EXISTS idx_leave_requests_tenant_updated ON leave_requests(tenant_id, updated_at DESC);
CREATE INDEX IF NOT EXISTS idx_ot_requests_tenant_updated ON ot_requests(tenant_id, updated_at DESC);

-- Trigger to auto-update updated_at (optional but recommended)
CREATE OR REPLACE FUNCTION set_updated_at()
RETURNS TRIGGER AS $$
BEGIN
  NEW.updated_at = now();
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DO $$
BEGIN
  IF NOT EXISTS (SELECT 1 FROM pg_trigger WHERE tgname = 'set_clock_records_updated_at') THEN
    CREATE TRIGGER set_clock_records_updated_at BEFORE UPDATE ON clock_records FOR EACH ROW EXECUTE FUNCTION set_updated_at();
  END IF;
  IF NOT EXISTS (SELECT 1 FROM pg_trigger WHERE tgname = 'set_leave_requests_updated_at') THEN
    CREATE TRIGGER set_leave_requests_updated_at BEFORE UPDATE ON leave_requests FOR EACH ROW EXECUTE FUNCTION set_updated_at();
  END IF;
  IF NOT EXISTS (SELECT 1 FROM pg_trigger WHERE tgname = 'set_ot_requests_updated_at') THEN
    CREATE TRIGGER set_ot_requests_updated_at BEFORE UPDATE ON ot_requests FOR EACH ROW EXECUTE FUNCTION set_updated_at();
  END IF;
END $$;
```

### 3.2 SSE stream route (Node + Express)

```ts
// server/src/routes/stream.ts
import express from 'express';
import { pool } from '../db';
import type { Request } from '../types';

const router = express.Router();

// Keep connections alive and allow last-event-id resumption
router.get('/stream', async (req: Request, res) => {
  const tenantId = req.tenantId;
  if (!tenantId) {
    res.status(401).end();
    return;
  }

  res.setHeader('Content-Type', 'text/event-stream');
  res.setHeader('Cache-Control', 'no-cache');
  res.setHeader('Connection', 'keep-alive');
  res.setHeader('Access-Control-Allow-Origin', '*');
  res.flushHeaders();

  // Send initial heartbeat so clients know stream is open
  res.write(`id: ${Date.now()}\n`);
  res.write('event: open\n');
  res.write('data: {"status":"connected"}\n\n');
  res.flush();

  // Helper to broadcast a change
  const broadcast = (table: string, payload: any) => {
    const id = Date.now();
    const data = JSON.stringify({ table, ...payload });
    res.write(`id: ${id}\n`);
    res.write(`event: update\n`);
    res.write(`data: ${data}\n\n`);
    res.flush();
  };

  // Poll interval (lightweight): check for recent updates per tenant
  const POLL_MS = 2000;
  let lastCheck = new Date(Date.now() - POLL_MS);

  const poll = async () => {
    try {
      const tables = ['clock_records', 'leave_requests', 'ot_requests'] as const;
      for (const table of tables) {
        const result = await pool.query(
          `SELECT id, tenant_id, updated_at FROM ${table} WHERE tenant_id = $1 AND updated_at > $2 ORDER BY updated_at ASC`,
          [tenantId, lastCheck]
        );
        for (const row of result.rows) {
          broadcast(table, { id: row.id, tenant_id: row.tenant_id, updated_at: row.updated_at });
          if (row.updated_at > lastCheck) lastCheck = row.updated_at;
        }
      }
    } catch (err) {
      // On error, send error event but keep stream alive
      res.write(`event: error\n`);
      res.write(`data: ${JSON.stringify({ error: String(err) })}\n\n`);
      res.flush();
    }
  };

  const interval = setInterval(poll, POLL_MS);

  // Client disconnect
  req.on('close', () => {
    clearInterval(interval);
    res.end();
  });
});

export default router;
```


### 3.3 Frontend hook (React + fetch EventSource)

```ts
// src/lib/stream.ts
export type StreamEvent = {
  table: 'clock_records' | 'leave_requests' | 'ot_requests';
  id: number;
  tenant_id: string;
  updated_at: string;
};

export function useTenantStream(tenantId: string | undefined, onUpdate: (e: StreamEvent) => void) {
  React.useEffect(() => {
    if (!tenantId) return;
    const source = new EventSource(`/api/stream`, { withCredentials: true });

    source.addEventListener('open', () => console.debug('SSE open'));
    source.addEventListener('update', (ev) => {
      try {
        const data = JSON.parse((ev as MessageEvent).data) as StreamEvent;
        onUpdate(data);
      } catch (err) {
        console.error('Failed to parse stream event', err);
      }
    });
    source.addEventListener('error', (ev) => console.warn('SSE error', ev));

    return () => {
      source.close();
    };
  }, [tenantId, onUpdate]);
}
```

Example usage in a dashboard component:

```tsx
// src/pages/Dashboard.tsx
import { useQueryClient } from '@tanstack/react-query';
import { useTenantStream } from '../lib/stream';

export default function Dashboard() {
  const tenantId = useTenantId(); // your existing hook
  const queryClient = useQueryClient();

  useTenantStream(tenantId, () => {
    // Invalidate relevant queries so UI refetches fresh state
    queryClient.invalidateQueries({ queryKey: ['clocks', tenantId] });
    queryClient.invalidateQueries({ queryKey: ['leaves', tenantId] });
    queryClient.invalidateQueries({ queryKey: ['ot', tenantId] });
  });

  // ... existing UI
}
```
