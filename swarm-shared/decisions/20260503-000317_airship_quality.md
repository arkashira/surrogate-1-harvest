## Final Synthesis: CDN-Only Data Loading to Eliminate HF 429s

**Core Problem**: Surrogate-1 training stalls because dataset loading repeatedly calls HF APIs (`list_repo_files`, `load_dataset`) and hits 429 rate limits. GPUs sit idle waiting for pagination.

**Unified Solution**: Generate a one-time CDN-only file manifest on your Mac, then patch the training data loader to fetch exclusively via CDN URLs with retry/backoff. This removes all HF API calls from the training loop, eliminates 429s, and keeps GPUs fed.

```python
# scripts/build_file_manifest.py
#!/usr/bin/env python3
"""
Generate CDN-only file manifest for Surrogate-1 training.
Run once on Mac.
"""
import argparse
import json
import time
from pathlib import Path

try:
    from huggingface_hub import list_repo_tree
except ImportError:
    raise SystemExit("pip install huggingface_hub")

CDN_BASE = "https://huggingface.co/datasets"

def build_manifest(repo: str, date_folder: str, out_path: Path, hf_token: str | None):
    headers = {"Authorization": f"Bearer {hf_token}"} if hf_token else {}
    items = list_repo_tree(
        repo=repo,
        path=date_folder,
        recursive=False,
        token=hf_token,
    )

    manifest = {
        "repo": repo,
        "date": date_folder,
        "generated_at": time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime()),
        "files": [],
    }

    for item in items:
        if item.get("type") != "file":
            continue
        path = item["path"]
        cdn_url = f"{CDN_BASE}/{repo}/resolve/main/{path}"
        manifest["files"].append({
            "path": path,
            "cdn_url": cdn_url,
            "size": item.get("size"),
        })

    out_path.parent.mkdir(parents=True, exist_ok=True)
    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"Manifest saved: {out_path} ({len(manifest['files'])} files)")

if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True, help="e.g. datasets/surrogate-1")
    parser.add_argument("--date", required=True, help="e.g. batches/mirror-merged/2026-04-29")
    parser.add_argument("--out", required=True, help="output JSON path")
    args = parser.parse_args()
    token = Path.home().joinpath(".huggingface", "token").read_text().strip() \
        if Path.home().joinpath(".huggingface", "token").exists() else None
    build_manifest(args.repo, args.date, Path(args.out), token)
```

```python
# surrogate/data/cdn_dataset.py
import json
import random
import time
from pathlib import Path
from typing import Dict, List

import requests
from datasets import Dataset
from tqdm import tqdm

class CDNDataset:
    """
    Surrogate-1 dataset loader using CDN URLs only.
    No HF API calls during training.
    """
    def __init__(self, manifest_path: Path, repo: str, siblings: int = 5):
        self.manifest_path = manifest_path
        self.repo = repo
        self.siblings = siblings
        self._manifest = None

    @property
    def manifest(self) -> Dict:
        if self._manifest is None:
            self._manifest = json.loads(self.manifest_path.read_text())
        return self._manifest

    def _pick_repo_for_slug(self, slug: str) -> str:
        """Deterministic repo selection to spread HF commit load."""
        idx = hash(slug) % self.siblings
        if idx == 0:
            return self.repo
        return f"{self.repo}-sibling-{idx}"

    def _stream_cdn(self, url: str, max_retries: int = 3) -> bytes:
        for attempt in range(max_retries):
            try:
                resp = requests.get(url, timeout=30)
                resp.raise_for_status()
                return resp.content
            except Exception as exc:
                if attempt == max_retries - 1:
                    raise
                wait = (2 ** attempt) + random.random()
                print(f"CDN retry {attempt+1}/{max_retries} for {url}: {exc} -> wait {wait:.1f}s")
                time.sleep(wait)

    def load(self, limit: int = None) -> Dataset:
        rows = []
        files = self.manifest["files"]
        if limit:
            files = random.sample(files, min(limit, len(files)))

        for item in tqdm(files, desc="Loading via CDN"):
            try:
                raw = self._stream_cdn(item["cdn_url"])
                # Project to {prompt, response} only (schema normalization)
                # Implement your per-file parser here; example for JSONL:
                # for line in raw.decode().splitlines():
                #     doc = json.loads(line)
                #     rows.append({"prompt": doc["prompt"], "response": doc["response"]})
                rows.append({
                    "path": item["path"],
                    "raw_size": len(raw),
                    "cdn_url": item["cdn_url"],
                })
            except Exception as exc:
                print(f"Skip {item['path']}: {exc}")

        return Dataset.from_list(rows)
```

import time
from pathlib import Path
from lightning import LightningStudio, Teamspace

def launch_training(manifest_path: str):
    # Reuse running studio to save quota (80hr/mo)
    studio = None
    for s in Teamspace.studios:
        if s.name == "surrogate-train" and s.status == "Running":
            studio = s
            break

    if not studio:
        studio = LightningStudio(
            name="surrogate-train",
            script_path="surrogate/train.py",
            machine="L40S",  # H200 only in lightning-lambda-prod (paid)
            cloud="lightning-public-prod",
        )
        studio.start()

    # Idle stop kills training — ensure running
    if studio.status != "Running":
        studio.start(machine="L40S")
        time.sleep(60)  # allow boot

    # Pass manifest path via CLI arg
    studio.run(
        args=[f"--manifest={manifest_path}"],
        env={"HF_TOKEN": "..."},  # optional for private repos
