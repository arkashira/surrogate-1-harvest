# vanguard / ops

## Final Synthesis (correctness + concrete actionability)

**Core diagnosis (merged)**
- Runtime HF API calls (frontend or backend) burn quota and cause 429s.
- No build-time/static manifest → every session re-enumerates repos and fetches via `/api/` instead of CDN `resolve/main/`.
- Missing ops automation to pre-compute manifests and embed them for both UI and training jobs.
- No deterministic sharding for HF ingestion (128 commits/hr/repo cap) and no reliable training-lifecycle guardrails.

**Chosen approach**
- Build a minimal, ops-friendly frontend scaffold with a build-time embedded manifest (CDN-only fetches).
- Add an ops manifest builder that pre-lists repo trees and produces a static JSON for UI + training jobs.
- Add a training launcher with Lightning-aware lifecycle handling (resume on available machine, idle-stop resilience).
- Add deterministic sibling-repo write sharding for HF ingestion to respect commit caps.

```js
const app = document.getElementById('app');

function renderManifest(manifest) {
  const list = document.createElement('ul');
  list.style.fontFamily = 'system-ui, sans-serif';
  (manifest.files || []).forEach((f) => {
    const li = document.createElement('li');
    const a = document.createElement('a');
    a.href = f.cdn;
    a.textContent = f.path;
    a.target = '_blank';
    a.rel = 'noopener';
    li.appendChild(a);
    list.appendChild(li);
  });
  app.appendChild(list);
}

function mount() {
  const manifest = window.APP_MANIFEST || { files: [] };
  app.innerHTML = '';
  const h1 = document.createElement('h1');
  h1.textContent = 'Vanguard — CDN-backed file browser';
  app.appendChild(h1);
  if (manifest.files && manifest.files.length) {
    renderManifest(manifest);
  } else {
    app.appendChild(document.createTextNode('No manifest found. Run ops/build.sh first.'));
  }
}

if (document.readyState === 'loading') {
  document.addEventListener('DOMContentLoaded', mount);
} else {
  mount();
}
```

## 3. Ops scripts

`/opt/axentx/vanguard/ops/build.sh`
```bash
#!/usr/bin/env bash
set -euo pipefail
# Build static frontend and embed file manifest for CDN-only access.
# Usage: ./ops/build.sh
# Requires: curl, jq

PROJECT_ROOT="$(cd "$(dirname "$0")/.." && pwd)"
MANIFEST_SRC="${MANIFEST_SRC:-datasets/my-org/my-repo}"
MANIFEST_DIR="${MANIFEST_DIR:-2026-05-03}"
OUT_DIR="${PROJECT_ROOT}/public"
STATIC_DIR="${PROJECT_ROOT}/static"
MANIFEST_OUT="${PROJECT_ROOT}/manifests/${MANIFEST_DIR}/file-list.json"

mkdir -p "${OUT_DIR}" "${STATIC_DIR}" "$(dirname "${MANIFEST_OUT}")"

# 1) Generate file manifest (single-level tree) using HF API.
#    This runs from ops machine where rate limits are manageable.
#    On 429, wait 360s before retry (per HF pattern).
echo "Fetching tree for ${MANIFEST_SRC}/${MANIFEST_DIR}..."
attempt=0
max_attempts=5
while [ $attempt -lt $max_attempts ]; do
  TREE_JSON=$(curl -sSfL "https://huggingface.co/api/datasets/${MANIFEST_SRC}/tree?path=${MANIFEST_DIR}&recursive=false" 2>/dev/null || true)
  if [ -n "${TREE_JSON}" ] && ! echo "${TREE_JSON}" | grep -q '"error"'; then
    break
  fi
  attempt=$((attempt + 1))
  if [ $attempt -ge $max_attempts ]; then
    echo "ERROR: HF API tree unavailable after retries."
    exit 1
  fi
  echo "Rate-limited or error; waiting 360s (attempt $attempt/$max_attempts)..."
  sleep 360
done

FILES_JSON=$(echo "${TREE_JSON}" | jq '[.[] | select(.type=="file") | {path: .path, cdn: ("https://huggingface.co/datasets/'${MANIFEST_SRC}'/resolve/main/'${MANIFEST_DIR}'/" + .path)}]')
MANIFEST_JSON=$(jq -n --argjson files "${FILES_JSON}" '{generated_at: now, source: "'${MANIFEST_SRC}/${MANIFEST_DIR}'", files: $files}')

# Save canonical manifest for training jobs and audits
echo "${MANIFEST_JSON}" > "${MANIFEST_OUT}"
echo "${MANIFEST_JSON}" > "${STATIC_DIR}/file-manifest.json"

# 2) Copy and inject manifest into index.html
cp "${PROJECT_ROOT}/index.html" "${OUT_DIR}/index.html"
MANIFEST_ESCAPED=$(echo "${MANIFEST_JSON}" | jq -c . | sed 's/"/\\"/g')
sed "s|%%MANIFEST_JSON%%|${MANIFEST_ESCAPED}|g" "${OUT_DIR}/index.html" > "${OUT_DIR}/index.html.tmp"
mv "${OUT_DIR}/index.html.tmp" "${OUT_DIR}/index.html"

# 3) Copy source for dev/prod serving
mkdir -p "${OUT_DIR}/src"
cp -r "${PROJECT_ROOT}/src"/* "${OUT_DIR}/src/" 2>/dev/null || true

echo "Build complete. Output in ${OUT_DIR}"
echo "Manifest: ${MANIFEST_OUT}"
```

`/opt/axentx/vanguard/ops/start-dev.sh`
```bash
#!/usr/bin/env bash
set -euo pipefail
# Minimal dev server for frontend iteration (ops-friendly).
# Serves ./public on port 8080.
# Usage: ./ops/start-dev.sh

PROJECT_ROOT="$(cd "$(dirname "$0")/.." && pwd)"
PORT="${PORT:-8080}"

cd "${PROJECT_ROOT}"

if [ ! -f "public/index.html" ]; then
  echo "No public/index.html — running ops/build.sh first..."
  bash ops/build.sh
fi

echo "Starting dev server on http://localhost:${PORT}"
if command -v python3 >/dev/null 2>&1; then
  python3 -m http.server "${PORT}" --directory public
elif command -v npx >/dev/null 2>&1; then
  npx http-server public -p "${PORT}" -c-1
else
  echo "ERROR: install python3 or npx to run dev server"
  exit 1
fi
```
