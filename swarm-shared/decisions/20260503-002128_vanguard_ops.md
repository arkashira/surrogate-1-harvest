# vanguard / ops

## Final Consolidated Solution

### 1. Diagnosis (merged, de-duplicated)
- **No persisted `(repo, dateFolder)` file manifest** → every training run re-enumerates via authenticated HF API → quota burn + 429 risk.
- **Training script likely uses `load_dataset(streaming=True)` or recursive `list_repo_files`** → PyArrow schema errors on heterogeneous repos and rate-limit amplification.
- **Lightning Studio reuse is not enforced** → idle-stop kills training and quota is wasted on repeated `create_ok=True` studio spawns.
- **No CDN-bypass strategy embedded** → workers perform authenticated fetches instead of zero-auth CDN downloads.
- **Missing ops guardrails**: no pre-flight file-list cache, no retry/backoff tuned for 429, no idempotent launcher.

### 2. Single Orchestrator + Manifest Layer (scope: new files only)
Add two new files; require only a small, optional training-script patch to accept `--file-list`.

### `/opt/axentx/vanguard/ops/manifest_utils.py`
```python
import json
import time
from datetime import datetime, timezone
from pathlib import Path
from typing import Dict, List, Optional

try:
    from huggingface_hub import HfApi, RepositoryNotFoundError
    from huggingface_hub.utils import RateLimitError
except ImportError as exc:
    raise RuntimeError("Install huggingface_hub to use manifest utilities") from exc


API = HfApi()


def _is_parquet_file(node) -> bool:
    # Support both rich node objects and dict-like shapes
    if hasattr(node, "type"):
        return node.type == "file" and getattr(node, "path", "").endswith(".parquet")
    if isinstance(node, dict):
        return node.get("type") == "file" and str(node.get("path", "")).endswith(".parquet")
    return False


def build_manifest(
    repo: str,
    date_folder: str,
    out_path: str,
    revision: str = "main",
    max_retries: int = 5,
    backoff_base: int = 60,
) -> List[str]:
    """
    Build persisted manifest for one date folder.
    Uses a single non-recursive repo-tree call.
    Retries with exponential backoff on RateLimitError (429).
    """
    out = Path(out_path)
    out.parent.mkdir(parents=True, exist_ok=True)

    last_exc: Optional[Exception] = None
    for attempt in range(1, max_retries + 1):
        try:
            tree = API.list_repo_tree(
                repo_id=repo,
                revision=revision,
                path=date_folder,
                repo_type="dataset",
                recursive=False,
            )
            files = sorted(
                str(f.path) for f in tree if _is_parquet_file(f)
            )
            if not files:
                # Try recursive once only if flat listing returned nothing
                tree_rec = API.list_repo_tree(
                    repo_id=repo,
                    revision=revision,
                    path=date_folder,
                    repo_type="dataset",
                    recursive=True,
                )
                files = sorted(
                    str(f.path) for f in tree_rec if _is_parquet_file(f)
                )

            manifest = {
                "repo": repo,
                "date_folder": date_folder,
                "revision": revision,
                "created_utc": datetime.now(timezone.utc).isoformat(),
                "files": files,
            }
            out.write_text(json.dumps(manifest, indent=2))
            return files

        except RateLimitError as exc:
            last_exc = exc
            wait = backoff_base * (2 ** (attempt - 1))
            time.sleep(wait)
        except RepositoryNotFoundError as exc:
            raise RuntimeError(f"Repo not found: {repo}") from exc
        except Exception as exc:
            last_exc = exc
            time.sleep(5 * attempt)

    raise RuntimeError(f"Failed to build manifest after {max_retries} attempts") from last_exc


def load_manifest(manifest_path: str) -> Dict:
    return json.loads(Path(manifest_path).read_text())
```

### `/opt/axentx/vanguard/ops/launch_surrogate1_training.py`
```python
#!/usr/bin/env python3
import argparse
import json
import subprocess
import sys
import time
from pathlib import Path

try:
    from lightning import Studio, Teamspace, Machine
    _LIGHTNING_AVAILABLE = True
except Exception:
    Studio = Teamspace = Machine = None
    _LIGHTNING_AVAILABLE = False

from manifest_utils import build_manifest, load_manifest


MANIFEST_DIR = Path("/opt/axentx/vanguard/ops/manifests")
TRAIN_SCRIPT = Path("/opt/axentx/vanguard/train/train_surrogate1.py")


def find_running_studio(name: str):
    if not _LIGHTNING_AVAILABLE or Teamspace is None:
        return None
    try:
        for s in Teamspace.studios:
            if getattr(s, "name", None) == name and getattr(s, "status", None) == "Running":
                return s
    except Exception:
        pass
    return None


def start_or_reuse_studio(studio_name: str, machine: str = "L40S"):
    """Idempotent studio reuse: running -> reuse; stopped -> restart; missing -> create."""
    if not _LIGHTNING_AVAILABLE or Studio is None:
        raise RuntimeError("Lightning not available; cannot manage Studio")

    existing = find_running_studio(studio_name)
    if existing:
        print(f"Reusing running studio: {studio_name}")
        return existing

    # stopped studio: restart
    try:
        for s in Teamspace.studios:
            if getattr(s, "name", None) == studio_name:
                print(f"Restarting stopped studio: {studio_name}")
                s.start(machine=Machine.L40S)
                return s
    except Exception:
        pass

    print(f"Creating new studio: {studio_name}")
    return Studio(
        name=studio_name,
        machine=Machine.L40S,
        create_ok=True,
    )


def build_and_run(repo: str, date_folder: str, force_rebuild: bool = False):
    MANIFEST_DIR.mkdir(parents=True, exist_ok=True)
    safe_name = repo.replace("/", "_")
    manifest_path = MANIFEST_DIR / f"{safe_name}_{date_folder.replace('/', '_')}.json"

    if force_rebuild or not manifest_path.exists():
        print(f"Building manifest: {repo}@{date_folder}")
        build_manifest(repo=repo, date_folder=date_folder, out_path=str(manifest_path))
    else:
        print(f"Using cached manifest: {manifest_path}")

    manifest = load_manifest(str(manifest_path))
    files = manifest.get("files", [])
    if not files:
        print("No files found in manifest. Exiting.")
        return

    # Pass CDN-only file list to training script
    cmd = [
        sys.executable,
        str(TRAIN_SCRIPT),
        "--repo",
        repo,
        "--file-list",
        json.dumps(files),
        "--date-folder",
        date_folder,
    ]

    # Prefer Lightning Studio execution when available
    if _LIGHTNING_AVAILABLE and Studio is not None:
        studio = start_or_reuse_studio(studio_name=f"surrogate1-{date_folder.replace('/', '-')}")
        # Studio.run executes on remote machine; adapt cwd as needed
        studio.run(
            command=" ".join(cmd),
            cwd="/workspace",
        )
    else:
        # Fallback: run locally
        print("Lightning not available; running locally")
        subprocess.run(cmd, check=True)


if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True, help="HF dataset repo")
    parser.add_argument("--date-folder", required=True, help="e.g. batches/mirror-merged/2026-04-29")
    parser.add_argument("--force-rebuild", action="store_true")
    args = parser.parse_args()

    build_and_run(repo=args.repo, date_folder=args.date_folder, force_rebuild=args.force_rebuild)
```
