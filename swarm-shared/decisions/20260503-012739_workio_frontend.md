# workio / frontend

## 1. Diagnosis

- Punch button lacks robust optimistic UI and safe retry: double-taps can create duplicate punches or leave UI stuck in “saving” state.
- No client-side idempotency key: retries on flaky networks can create duplicate server records.
- Clock-in/out state derived from last punch is racy when multiple devices/tabs are open.
- Pending/optimistic state is not reflected in the button (no immediate UI feedback).
- Error handling is basic: failed punches don’t show actionable retry and don’t rollback optimistic state.

## 2. Proposed change

File: `workio/src/features/punch/PunchButton.tsx`  
Scope: replace current click handler with React Query mutation + optimistic update + client-side idempotency key and in-flight lock. Keep API contract unchanged (same endpoint/payload) and add optional `clientId` for backend de-duping.

## 3. Implementation

```tsx
// workio/src/features/punch/PunchButton.tsx
import React, { useCallback, useMemo } from 'react';
import { useMutation, useQueryClient } from '@tanstack/react-query';
import { Check, Clock, AlertCircle } from 'lucide-react';
import { usePunch } from '../hooks/usePunch';
import { punchAPI } from '../api/punch';
import type { PunchType } from '../types';

const PENDING_TTL = 120_000; // 2m optimistic lock TTL

function generateClientId(): string {
  return `${Date.now()}-${Math.random().toString(36).slice(2, 9)}`;
}

export function PunchButton() {
  const queryClient = useQueryClient();
  const { lastPunch, isClockedIn, refetch: refetchPunch } = usePunch();

  const punchMutation = useMutation({
    mutationFn: punchAPI.create,
    onMutate: async (newPunch) => {
      // Cancel outgoing refetches so optimistic update is not overwritten
      await queryClient.cancelQueries({ queryKey: ['punches', 'latest'] });

      // Snapshot previous value
      const previous = queryClient.getQueryData(['punches', 'latest']);

      // Optimistic update
      const optimistic = {
        ...newPunch,
        id: `optimistic-${generateClientId()}`,
        clientId: newPunch.clientId,
        _optimistic: true as const,
        createdAt: new Date().toISOString(),
      };

      queryClient.setQueryData(['punches', 'latest'], optimistic);
      // Also update list if used elsewhere
      queryClient.setQueryData(['punches', 'list'], (old: any[] = []) => [
        optimistic,
        ...old,
      ]);

      return { previous };
    },
    onError: (err, _newPunch, context) => {
      // Rollback
      if (context?.previous) {
        queryClient.setQueryData(['punches', 'latest'], context.previous);
        queryClient.setQueryData(['punches', 'list'], (old: any[] = []) =>
          old?.map((item) =>
            item?._optimistic ? context.previous : item
          )
        );
      }
    },
    onSettled: () => {
      // Refetch to ensure consistency
      refetchPunch();
      queryClient.invalidateQueries({ queryKey: ['punches'] });
    },
  });

  const inFlight = useMemo(() => {
    // Prevent double submit within PENDING_TTL by checking recent optimistic punches
    const last = queryClient.getQueryData(['punches', 'latest']);
    if (!last || !('_optimistic' in last)) return false;
    const ts = Date.parse((last as any).createdAt);
    return !Number.isNaN(ts) && Date.now() - ts < PENDING_TTL;
  }, [queryClient]);

  const handlePunch = useCallback(async () => {
    if (punchMutation.isPending || inFlight) return;

    const clientId = generateClientId();
    punchMutation.mutate({
      type: (isClockedIn ? 'clock_out' : 'clock_in') as PunchType,
      location: {
        lat: 0, // Replace with real GPS
        lng: 0,
        address: '',
      },
      clientId,
    });
  }, [punchMutation, isClockedIn, inFlight]);

  const isLoading = punchMutation.isPending || inFlight;
  const isError = punchMutation.isError;

  return (
    <div className="flex flex-col items-center gap-3">
      <button
        onClick={handlePunch}
        disabled={isLoading}
        className={`
          relative flex h-20 w-20 items-center justify-center rounded-full font-semibold text-white
          transition-all focus:outline-none focus:ring-4 focus:ring-offset-2
          ${
            isError
              ? 'bg-red-600 hover:bg-red-700 focus:ring-red-300'
              : isClockedIn
              ? 'bg-red-500 hover:bg-red-600 focus:ring-red-300'
              : 'bg-green-600 hover:bg-green-700 focus:ring-green-300'
          }
          ${isLoading ? 'cursor-wait opacity-80' : ''}
        `}
        aria-busy={isLoading}
        aria-live="polite"
      >
        {isLoading && (
          <span className="absolute inset-0 rounded-full border-4 border-white/20 border-t-white animate-spin" />
        )}

        <span className="flex items-center gap-2">
          {isError ? (
            <AlertCircle className="h-6 w-6" />
          ) : isClockedIn ? (
            <Clock className="h-6 w-6" />
          ) : (
            <Check className="h-6 w-6" />
          )}
        </span>
      </button>

      <div className="text-sm font-medium text-gray-600">
        {isError ? (
          <span className="text-red-600">ไม่สามารถบันทึกได้</span>
        ) : isClockedIn ? (
          'กดเพื่อเลิกงาน'
        ) : (
          'กดเพื่อเข้างาน'
        )}
      </div>

      {isError && (
        <button
          onClick={() => punchMutation.reset()}
          className="text-xs text-blue-600 underline"
        >
          ลองใหม่
        </button>
      )}
    </div>
  );
}
```

```ts
// workio/src/features/punch/api/punch.ts
import axios from 'axios';
import type { PunchCreate } from '../types';

export const punchAPI = {
  create: async (payload: PunchCreate) => {
    const { data } = await axios.post('/api/punches', payload);
    return data;
  },
};
```

```ts
// workio/src/features/punch/types.ts
export interface Punch {
  id: string;
  type: 'clock_in' | 'clock_out';
  location: {
    lat: number;
    lng: number;
    address?: string;
  };
  createdAt: string;
  clientId?: string;
  _optimistic?: boolean;
}

export interface PunchCreate {
  type: 'clock_in' | 'clock_out';
  location: {
    lat: number;
    lng: number;
    address?: string;
  };
  clientId?: string;
}
```

```ts
// workio/src/features/punch/hooks/usePunch.ts
import { useQuery } from '@tanstack/react-query';
import { punchAPI } from '../api/punch';

export function usePunch() {
  const { data: lastPunch, refetch } = useQuery({
    queryKey: ['punches', 'latest'],
    queryFn: () => punchAPI.getLatest(),
    // Keep stale while editing; refetch after mutations
    staleTime: 60_000,
  });

  const isClockedIn = lastPunch?.type === 'clock_in';

  return { lastPunch, isClockedIn, refetch };
}
```

(If backend doesn’t accept `clientId` yet, it can ignore unknown fields; adding it is non-breaking.)
