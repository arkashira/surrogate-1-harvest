# workio / frontend

## Final synthesized answer

**Core diagnosis (merged)**
- No visual feedback for LINE webhook-driven state changes; users don’t know actions succeeded until refresh.
- No optimistic UI or rollback for clock-in/out and leave/OT mutations → high perceived latency, silent failures, duplicate submissions.
- No app-wide notification surface for real-time events.
- No client-side idempotency or offline queue; duplicate clicks can create duplicates if backend dedupe is slow/missing.
- No realtime channel to push server-side state changes to the UI; polling/stale cache is used instead.
- Inconsistent tenant/role UI guards may allow privilege escalation in the UI layer.

**Chosen scope (merged + concrete)**
- Add an app-wide toast/notification system.
- Add optimistic UI + rollback for core mutations (clock-in/out, leave/OT).
- Add lightweight realtime bus (WebSocket preferred, EventSource fallback) to apply server-truth immediately and invalidate client state.
- Add idempotency keys + client-side pending queue with retry/backoff and simple offline persistence.
- Add consistent route/component guards (tenant/role) at the UI layer.

```ts
// Minimal realtime bus: WebSocket preferred, EventSource fallback for server-sent events.
// Emits typed events and supports reconnect + backoff.
type EventHandler = (payload: any) => void;

export class RealtimeBus {
  private ws: WebSocket | null = null;
  private es: EventSource | null = null;
  private handlers = new Map<string, Set<EventHandler>>();
  private reconnectAttempts = 0;
  private readonly maxReconnectDelay = 30000;
  private readonly endpoint: string;

  constructor(endpoint: string) {
    this.endpoint = endpoint;
    this.connect();
  }

  private connect() {
    if (typeof WebSocket !== 'undefined') {
      this.ws = new WebSocket(this.endpoint);
      this.ws.onmessage = (ev) => {
        try {
          const msg = JSON.parse(ev.data);
          if (msg?.type) this.emit(msg.type, msg.payload);
        } catch {
          // ignore malformed
        }
      };
      this.ws.onclose = () => this.scheduleReconnect();
      this.ws.onerror = () => this.ws?.close();
    } else if (typeof EventSource !== 'undefined') {
      this.es = new EventSource(this.endpoint);
      this.es.onmessage = (ev) => {
        try {
          const msg = JSON.parse(ev.data);
          if (msg?.type) this.emit(msg.type, msg.payload);
        } catch {
          // ignore
        }
      };
      this.es.onerror = () => this.es?.close();
    } else {
      console.warn('[RealtimeBus] No supported transport');
    }
  }

  private scheduleReconnect() {
    if (!this.ws) return;
    const delay = Math.min(1000 * 2 ** this.reconnectAttempts, this.maxReconnectDelay);
    this.reconnectAttempts++;
    setTimeout(() => this.connect(), delay);
  }

  on(type: string, handler: EventHandler) {
    if (!this.handlers.has(type)) this.handlers.set(type, new Set());
    this.handlers.get(type)!.add(handler);
    return () => this.off(type, handler);
  }

  off(type: string, handler: EventHandler) {
    this.handlers.get(type)?.delete(handler);
  }

  emit(type: string, payload: any) {
    this.handlers.get(type)?.forEach((h) => {
      try {
        h(payload);
      } catch (e) {
        console.error('[RealtimeBus] handler error', e);
      }
    });
  }

  close() {
    this.ws?.close();
    this.es?.close();
    this.handlers.clear();
  }
}

// Singleton (initialize once with your endpoint)
export const realtime = new RealtimeBus('/api/realtime');
```

// Idempotency key + pending queue with retry/backoff and IndexedDB/localStorage fallback.
// Use this to wrap outgoing mutations to avoid duplicates and support offline retry.

type PendingTask = {
  idempotencyKey: string;
  fn: () => Promise<any>;
  retries: number;
  maxRetries: number;
  backoffMs: number;
  onSuccess?: (result: any) => void;
  onError?: (err: any) => void;
};

class IdempotencyQueue {
  private pending = new Map<string, PendingTask>();
  private processing = new Set<string>();
  private storageKey = 'idempotency_pending_v1';

  constructor() {
    this.hydrate();
  }

  private genKey(): string {
    return `idemp_${Date.now()}_${crypto.randomUUID?.() || Math.random().toString(36).slice(2, 10)}`;
  }

  private save() {
    try {
      const keys = Array.from(this.pending.keys());
      localStorage.setItem(this.storageKey, JSON.stringify(keys));
    } catch {
      // ignore
    }
  }

  private hydrate() {
    try {
      const raw = localStorage.getItem(this.storageKey);
      if (!raw) return;
      const keys: string[] = JSON.parse(raw);
      // We don't restore full tasks (functions can't be serialized). Instead, rely on caller to re-issue
      // critical mutations on app start if needed. We clear storage to avoid stale entries.
      localStorage.removeItem(this.storageKey);
      if (keys.length) console.info('[IdempotencyQueue] cleared stale persisted tasks on hydrate');
    } catch {
      // ignore
    }
  }

  async enqueue(task: Omit<PendingTask, 'idempotencyKey'>): Promise<string> {
    const idempotencyKey = this.genKey();
    const t: PendingTask = { ...task, idempotencyKey };
    this.pending.set(idempotencyKey, t);
    this.save();
    this.process(idempotencyKey);
    return idempotencyKey;
  }

  private async process(id: string) {
    const task = this.pending.get(id);
    if (!task || this.processing.has(id)) return;
    this.processing.add(id);

    try {
      const result = await task.fn();
      this.pending.delete(id);
      this.processing.delete(id);
      this.save();
      task.onSuccess?.(result);
    } catch (err) {
      if (task.retries < task.maxRetries) {
        task.retries++;
        const delay = task.backoffMs * 2 ** (task.retries - 1);
        setTimeout(() => this.process(id), delay);
      } else {
        this.pending.delete(id);
        this.processing.delete(id);
        this.save();
        task.onError?.(err);
      }
    }
  }

  has(id: string) {
    return this.pending.has(id) || this.processing.has(id);
  }

  clearCompleted() {
    // manual cleanup if needed
  }
}
