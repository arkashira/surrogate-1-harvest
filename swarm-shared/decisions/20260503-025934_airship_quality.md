# airship / quality

## Implementation Plan — Airship Quality (HF CDN Bypass + Lightning Idle Resilience)

**Estimated effort**: ≤2h  
**Scope**: Non-breaking ops improvements only — no schema changes  
**Value**: Eliminates HF API 429s during training, prevents Lightning idle-stop training loss, reduces Mac→Lightning iteration from ~15 min to <2 min per cycle.

---

### 1) HF CDN bypass + file-list pre-generation (Mac orchestration)

- Single API call (after rate-limit window) to list one date folder via `list_repo_tree(..., recursive=False)` → save to `file_list.json`.
- Embed `file_list.json` in training script; Lightning training does **CDN-only** fetches (`https://huggingface.co/datasets/{repo}/resolve/main/{path}`) with zero API/auth calls during data load.
- Avoid `load_dataset(streaming=True)` on mixed-schema repos; use `hf_hub_download` per file from CDN URLs and project `{prompt, response}` at parse time.

### 2) Lightning idle-stop resilience

- Before each `.run()`, check studio status; if stopped, restart with `target.start(machine=Machine.L40S)` (or fallback to free-tier L40S).
- Reuse running studios (`Teamspace.studios`) instead of create/destroy to save quota.

### 3) Fast Mac→Lightning iteration (<2 min)

- Lightweight launcher on Mac that:
  1. Generates/updates `file_list.json` (cached per date folder).
  2. Validates training script + file list.
  3. Starts/reuses Lightning Studio and runs training with CDN-only data loader.
- No model download/training on Mac; all compute on Lightning.

### `scripts/generate_file_list.py` (run on Mac)

```python
#!/usr/bin/env python3
"""
Generate file_list.json for a date folder in a HF dataset repo.
Run from Mac after rate-limit window clears.
"""
import json
import os
import sys
from datetime import datetime
from pathlib import Path

from huggingface_hub import HfApi

HF_REPO = os.getenv("HF_DATASET_REPO", "myorg/surrogate-data")
DATE_FOLDER = os.getenv("DATE_FOLDER", datetime.utcnow().strftime("%Y-%m-%d"))
OUT_PATH = Path(os.getenv("OUT_PATH", "file_list.json"))

def main() -> None:
    api = HfApi()
    # non-recursive per folder to avoid pagination explosion
    entries = api.list_repo_tree(
        repo_id=HF_REPO,
        path=DATE_FOLDER,
        repo_type="dataset",
        recursive=False,
    )
    files = [e.rfilename for e in entries if e.type == "file"]
    if not files:
        print(f"No files found in {HF_REPO}/{DATE_FOLDER}")
        sys.exit(1)

    payload = {
        "repo": HF_REPO,
        "date_folder": DATE_FOLDER,
        "generated_at_utc": datetime.utcnow().isoformat() + "Z",
        "files": sorted(files),
    }
    OUT_PATH.write_text(json.dumps(payload, indent=2))
    print(f"Wrote {len(files)} files to {OUT_PATH}")

if __name__ == "__main__":
    main()
```

### `surrogate/train.py` (Lightning training — CDN-only)

```python
import json
import os
from pathlib import Path
from typing import Dict, Any

import requests
import torch
from torch.utils.data import Dataset, DataLoader
from lightning import LightningModule, Trainer
from lightning.pytorch.strategies import DDPStrategy

HF_DATASET_REPO = os.getenv("HF_DATASET_REPO", "myorg/surrogate-data")
FILE_LIST_PATH = Path(os.getenv("FILE_LIST_PATH", "file_list.json"))
CDN_BASE = f"https://huggingface.co/datasets/{HF_DATASET_REPO}/resolve/main"

def load_file_list(path: Path) -> Dict[str, Any]:
    with path.open() as f:
        return json.load(f)

class CDNJsonlDataset(Dataset):
    def __init__(self, file_list: Dict[str, Any], max_files: int = -1):
        files = file_list["files"]
        if max_files > 0:
            files = files[:max_files]
        self.records = []
        for fn in files:
            url = f"{CDN_BASE}/{fn}"
            resp = requests.get(url, timeout=60)
            resp.raise_for_status()
            for line in resp.text.strip().splitlines():
                line = line.strip()
                if not line:
                    continue
                try:
                    obj = json.loads(line)
                    # project to {prompt, response} only
                    prompt = obj.get("prompt") or obj.get("input") or ""
                    response = obj.get("response") or obj.get("output") or ""
                    if prompt and response:
                        self.records.append({"prompt": prompt, "response": response})
                except json.JSONDecodeError:
                    continue

    def __len__(self) -> int:
        return len(self.records)

    def __getitem__(self, idx: int) -> Dict[str, str]:
        return self.records[idx]

class SurrogateModel(LightningModule):
    # Minimal placeholder — replace with your actual model/tokenizer/training step
    def __init__(self):
        super().__init__()
        self.model = torch.nn.Linear(10, 1)  # dummy

    def training_step(self, batch, batch_idx):
        return torch.tensor(0.0)

    def configure_optimizers(self):
        return torch.optim.Adam(self.parameters(), lr=1e-3)

def main() -> None:
    file_list = load_file_list(FILE_LIST_PATH)
    dataset = CDNJsonlDataset(file_list, max_files=100)  # tune as needed
    loader = DataLoader(dataset, batch_size=8, num_workers=4)

    model = SurrogateModel()
    trainer = Trainer(
        max_epochs=1,
        accelerator="gpu",
        devices=1,
        strategy=DDPStrategy(find_unused_parameters=False),
        log_every_n_steps=10,
    )
    trainer.fit(model, loader)

if __name__ == "__main__":
    main()
```
