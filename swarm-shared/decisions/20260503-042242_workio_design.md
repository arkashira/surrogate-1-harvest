### 2. Schema (run once; idempotent)

```sql
-- Idempotency and replay protection
CREATE TABLE IF NOT EXISTS webhook_idempotency (
  idempotency_key TEXT PRIMARY KEY,
  event_type      TEXT      NOT NULL,
  tenant_id       TEXT      NOT NULL,
  created_at      TIMESTAMPTZ NOT NULL
);

-- Attendance state with monotonic version to prevent replay/reordering
CREATE TABLE IF NOT EXISTS attendance (
  id              TEXT PRIMARY KEY,
  tenant_id       TEXT      NOT NULL,
  user_id         TEXT      NOT NULL,
  action          TEXT      NOT NULL, -- 'in' | 'out'
  clock_in_at     TIMESTAMPTZ,
  clock_out_at    TIMESTAMPTZ,
  version         BIGINT    NOT NULL DEFAULT 1,
  created_at      TIMESTAMPTZ NOT NULL,
  updated_at      TIMESTAMPTZ,
  CONSTRAINT attendance_valid_action CHECK (action IN ('in','out')),
  CONSTRAINT attendance_clock_order CHECK (
    (action = 'in'  AND clock_in_at IS NOT NULL) OR
    (action = 'out' AND clock_out_at IS NOT NULL)
  )
);

-- Daily summary to enforce one active clock-in per user per day
CREATE UNIQUE INDEX IF NOT EXISTS uq_attendance_active_in
  ON attendance (tenant_id, user_id, (clock_in_at::date))
  WHERE action = 'in' AND clock_out_at IS NULL;

-- Immutable audit log
CREATE TABLE IF NOT EXISTS audit_log (
  id              TEXT PRIMARY KEY,
  tenant_id       TEXT,
  user_id         TEXT,
  action          TEXT      NOT NULL,
  entity_type     TEXT      NOT NULL,
  entity_id       TEXT,
  payload         JSONB,
  created_at      TIMESTAMPTZ NOT NULL
);
```

import express, { Request, Response, NextFunction } from 'express';
import crypto from 'crypto';
import { Pool, Client } from 'pg';
import { v4 as uuidv4 } from 'uuid';

const router = express.Router();
const pool = new Pool({ connectionString: process.env.DATABASE_URL });

const LINE_CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';
if (!LINE_CHANNEL_SECRET) {
  throw new Error('LINE_CHANNEL_SECRET is required');
}

// Verify LINE signature (constant-time)
function verifySignature(rawBody: Buffer, signature: string): boolean {
  const expected = crypto
    .createHmac('sha256', LINE_CHANNEL_SECRET)
    .update(rawBody)
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

function nowUtc(): Date {
  return new Date();
}

// Idempotency check + insert (prevents races)
async function ensureIdempotency(
  client: Client,
  idemKey: string,
  eventType: string,
  tenantId: string
): Promise<boolean> {
  const exists = await client.query(
    `SELECT 1 FROM webhook_idempotency WHERE idempotency_key = $1`,
    [idemKey]
  );
  if (exists.rows.length > 0) return false;

  await client.query(
    `INSERT INTO webhook_idempotency (idempotency_key, event_type, tenant_id, created_at)
     VALUES ($1, $2, $3, $4)`,
    [idemKey, eventType, tenantId, nowUtc()]
  );
  return true;
}

// Resolve tenant from LINE user ID (strict)
async function resolveTenantIdByLineUserId(
  client: Client,
  lineUserId: string
): Promise<string> {
  const res = await client.query(
    `SELECT tenant_id FROM users WHERE line_user_id = $1`,
    [lineUserId]
  );
  if (res.rows.length === 0) {
    throw new Error('User not found');
  }
  return res.rows[0].tenant_id;
}

// Audit log (immutable)
async function writeAudit(
  client: Client,
  tenantId: string | null,
  userId: string | null,

  entityType: string,
  entityId: string | null,

    [uuidv4(), tenantId, userId, action, entityType, entityId, payload, nowUtc()]
  );
}

// Clock-in/out with state guards and grace window (optional)
async function applyAttendanceTransition(
  client: Client,
  tenantId: string,
  userId: string,
  desiredAction: 'in' | 'out',
  graceWindowMinutes: number = 5
) {
  const todayStart = new Date();
  todayStart.setUTCHours(0, 0, 0, 0);

  // Fetch latest record for this user+tenant
  const latest = await client.query(
    `SELECT id, action, clock_in_at, clock_out_at, version
     FROM attendance
     WHERE tenant_id = $1 AND user_id = $2
     ORDER BY created_at DESC
     LIMIT 1`,
    [tenantId, userId]
  );

  const now = nowUtc();
  const latestRecord = latest.rows[0];

  // Guard: invalid transitions
  if (desiredAction === 'in') {
    if (latestRecord && latestRecord.action === 'in' && latestRecord.clock_out_at === null) {
      // Already clocked in today — check grace window
      const lastIn = new Date(latestRecord.clock_in_at);
      const diffMin = (now.getTime() - lastIn.getTime()) / 60000;
      if (diffMin <= graceWindowMinutes) {
        // Grace: treat as no-op (idempotent)
        return { status: 'grace_no_op', recordId: latestRecord.id };
      }
      throw new Error('Invalid transition: already clocked in without clock-out');
    }
  } else if (desiredAction === 'out') {
    if (!latestRecord || latestRecord.action !== 'in' || latestRecord.clock_out_at !== null) {
      throw new Error('Invalid transition: no active clock-in to clock-out');
    }
  }

  // Apply transition transactionally
  if (desiredAction === 'in') {
    await client.query(
      `INSERT INTO attendance (id, tenant_id, user_id, action, clock_in_at, created_at)
       VALUES ($1, $2, $3, 'in', $4, $5)`,
      [uuidv4(), tenantId, userId, now, now]
    );
  } else {
    await client.query(
      `UPDATE attendance
       SET action = 'out', clock_out_at = $1, updated_at = $2, version = version + 1
       WHERE id = $3 AND tenant_id = $4 AND user_id = $5 AND action = 'in' AND clock_out_at IS NULL`,
      [now, now, latestRecord.id, tenantId, userId]
    );
    const updated = await client.query('SELECT ROW_COUNT as c');
    if (updated.rows[0].c === 0) {
      throw new Error('Failed to clock-out: state changed concurrently');
    }
  }

  return { status: 'ok', action: desiredAction };
}
