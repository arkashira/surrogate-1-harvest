### 1) Manifest generator (run once on orchestrator)

- **What:** Build `train_manifest.json` listing every parquet shard with its direct CDN URL.  
- **Why:** Removes all `load_dataset`/`list_repo_tree`/`hf_hub_download` calls during training.  
- **Correctness fix:** Use deterministic, recursive listing once per date folder; fall back to repo root only if needed.

```python
# scripts/build_cdn_manifest.py
#!/usr/bin/env python3
import json
import os
from pathlib import Path
from huggingface_hub import HfApi

REPO_ID = os.getenv("HF_DATASET_REPO", "your-org/surrogate-data")
BASE_PATH = "batches/mirror-merged"
OUT = Path(__file__).parent.parent / "surrogate" / "data" / "train_manifest.json"

def build() -> list[dict]:
    api = HfApi()
    try:
        top = api.list_repo_tree(REPO_ID, path=BASE_PATH, recursive=False)
        date_items = [d for d in top if d.type == "directory"]
    except Exception:
        all_items = api.list_repo_tree(REPO_ID, path="", recursive=False)
        date_items = [i for i in all_items if i.rfilename.startswith(BASE_PATH) and i.type == "directory"]

    manifest = []
    for d in date_items:
        try:
            files = api.list_repo_tree(REPO_ID, path=d.rfilename, recursive=False)
        except Exception:
            continue
        for f in files:
            if f.rfilename.lower().endswith(".parquet"):
                manifest.append({
                    "date": Path(d.rfilename).name,
                    "slug": Path(f.rfilename).name,
                    "cdn_url": f"https://huggingface.co/datasets/{REPO_ID}/resolve/main/{f.rfilename}"
                })
    return manifest

if __name__ == "__main__":
    OUT.parent.mkdir(parents=True, exist_ok=True)
    manifest = build()
    with open(OUT, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Wrote {len(manifest)} entries to {OUT}")
```

### 2) CDN-only dataset loader (no HF API during training)

- **What:** `CdnParquetIterableDataset` reads manifest and streams rows via `pyarrow`.  
- **Why:** Zero HF API calls; resilient to corrupt shards; compatible with distributed DataLoader.  
- **Correctness fix:** Use `pyarrow.parquet.read_table` (not pandas) for speed and memory safety; skip bad shards without crashing.

```python
# surrogate/data/cdn_parquet_loader.py
from __future__ import annotations
import json
from pathlib import Path
from typing import Iterator

import pyarrow.parquet as pq
import torch
from torch.utils.data import IterableDataset

class CdnParquetIterableDataset(IterableDataset):
    def __init__(self, manifest_path: str | Path, columns=("prompt", "response")):
        manifest_path = Path(manifest_path)
        if not manifest_path.exists():
            raise FileNotFoundError(f"Manifest not found: {manifest_path}")
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.columns = [c for c in columns if c]

    def _stream_shard(self, url: str) -> Iterator[dict]:
        try:
            table = pq.read_table(url, columns=self.columns)
            df = table.to_pandas()
            for _, row in df.iterrows():
                yield {k: row[k] for k in self.columns if k in row}
        except Exception as exc:
            print(f"Skipping shard {url}: {exc}")

    def __iter__(self) -> Iterator[dict]:
        for entry in self.manifest:
            yield from self._stream_shard(entry["cdn_url"])
```

### 3) Resilient Lightning launcher (reuse + restart studios)

- **What:** Reuse running studios; restart stopped ones; fallback to local Trainer if Studio unavailable.  
- **Why:** Prevents idle timeouts and lost progress; maximizes resource utilization.  
- **Correctness fix:** Do not block training if Studio unavailable; always provide a working Trainer path.

```python
# scripts/launch_surrogate_training.py
#!/usr/bin/env python3
import os
import sys
from pathlib import Path

from lightning.pytorch import Trainer
from lightning.pytorch.loggers import CSVLogger
from lightning.pytorch.strategies import DDPStrategy

sys.path.insert(0, str(Path(__file__).parent.parent))

from surrogate.lightning_module import SurrogateModule  # adjust as needed
from surrogate.data.cdn_parquet_loader import CdnParquetIterableDataset

try:
    from lightning.pytorch.studio import Studio, Machine, Teamspace
    STUDIO_AVAILABLE = True
except Exception:
    STUDIO_AVAILABLE = False

MANIFEST_PATH = os.getenv(
    "CDN_MANIFEST_PATH",
    str(Path(__file__).parent.parent / "surrogate" / "data" / "train_manifest.json")
)
STUDIO_NAME = os.getenv("LIGHTNING_STUDIO_NAME", "surrogate-training")

def get_or_create_studio():
    if not STUDIO_AVAILABLE:
        print("Studio unavailable; will use local Trainer.")
        return None

    teamspace = Teamspace()
    for s in teamspace.studios:
        if s.name == STUDIO_NAME:
            if s.status == "running":
                print(f"Reusing running studio: {s.name}")
                return s
            print(f"Restarting stopped studio: {s.name}")
            try:
                s.start(machine=Machine.L40S)
            except Exception:
                s.start()
            return s
    return None

def run_training():
    studio = get_or_create_studio()

    dataset = CdnParquetIterableDataset(MANIFEST_PATH)
    model = SurrogateModule()

    logger = CSVLogger(save_dir="logs", name="surrogate")
    trainer = Trainer(
        max_epochs=1,
        logger=logger,
        accelerator="auto",
        devices="auto",
        strategy=DDPStrategy(find_unused_parameters=False) if studio else "auto",
        log_every_n_steps=10,
    )

    if studio:
        studio.run(
            callable=trainer.fit,
            model=model,
            train_dataloaders=torch.utils.data.DataLoader(dataset, batch_size=8, num_workers=2),
        )
    else:
        trainer.fit(
            model=model,
            train_dataloaders=torch.utils.data.DataLoader(dataset, batch_size=8, num_workers=2),
        )

if __name__ == "__main__":
    run_training()
```
