# surrogate-1 / design

## Final Implementation Plan  
**Single Python worker (`bin/dataset-enrich.py`) — manifest-driven, CDN-bypass, production-ready**

---

### Core design decisions (resolved)
- **Manifest source**: Prefer local `MANIFEST_PATH` when provided; otherwise fetch `DATE/manifest.json` via CDN from the HF repo. Fail fast if manifest cannot be loaded.  
- **CDN bypass**: Use public `resolve/main/...` URLs with **no Authorization header** to avoid API rate limits. Add robust retries and streaming downloads to avoid OOM.  
- **Schema/projection**: Project to `{prompt, response}` **at parse time** per file. Tolerate extra columns and common aliases; skip files/rows that cannot yield valid pairs.  
- **Deduplication**: Reuse existing `lib/dedup.DedupStore` (central md5 store) to deduplicate across the shard’s stream.  
- **Output**: Deterministic shard assignment by `hash(slug) % 16`. Each worker writes to `batches/public-merged/<DATE>/shard<N>-<HHMMSS>.jsonl`. Keep GH Actions matrix (16 shards) unchanged; use `SHARD_ID` and `DATE` env vars.  
- **Safety & ops**: Validate manifest shape; cap memory via streaming parse and bounded batches; surface clear errors and non-zero exit codes on fatal failures.

def parse_file_to_pairs(local_path: Path) -> Iterable[Dict[str, str]]:
    suffix = local_path.suffix.lower()
    try:
        if suffix == ".parquet":
            try:
                pf = pq.read_table(local_path, columns=["prompt", "response"])
            except (pa.ArrowInvalid, KeyError, ValueError):
                pf = pq.read_table(local_path)
                col_map: Dict[str, str] = {}
                for c in pf.column_names:
                    cl = c.lower()
                    if "prompt" in cl:
                        col_map["prompt"] = c
                    elif any(x in cl for x in ("response", "completion", "answer", "output")):
                        col_map["response"] = c
                if "prompt" not in col_map or "response" not in col_map:
                    return
                pf = pf.select([col_map["prompt"], col_map["response"]])
            df = pf.to_pandas()
            for _, row in df.iterrows():
                prompt = str(row.get("prompt") or "")
                response = str(row.get("response") or "")
                if prompt.strip() and response.strip():
                    yield {"prompt": prompt.strip(), "response": response.strip()}

        elif suffix == ".jsonl":
            with open(local_path, "r", encoding="utf-8") as f:
                for line in f:
                    line = line.strip()
                    if not line:
                        continue
                    try:
                        obj = json.loads(line)
                    except json.JSONDecodeError:
                        continue
                    prompt = str(obj.get("prompt") or obj.get("input") or "")
                    response = str(obj.get("response") or obj.get("output") or obj.get("completion") or "")
                    if prompt.strip() and response.strip():
                        yield {"prompt": prompt.strip(), "response": response.strip()}

        elif suffix == ".json":
            with open(local_path, "r", encoding="utf-8") as f:
                data = json.load(f)
            items: List[Any] = []
            if isinstance(data, list):
                items = data
            elif isinstance(data, dict):
                items = data.get("examples", data.get("rows", data.get("data", [])))
                if not isinstance(items, list):
                    items = []
            for obj in items:
                if not isinstance(obj, dict):
                    continue
                prompt = str(obj.get("prompt") or obj.get("input") or "")
                response = str(obj.get("response") or obj.get("output") or obj.get("completion") or "")
                if prompt.strip() and response.strip():
                    yield {"prompt": prompt.strip(), "response": response.strip()}
        else:
            return
    except Exception as exc:
        print(f"WARN: failed to parse {local_path}: {exc}", file=sys.stderr)
        return

def fetch_via_cdn(url: str, timeout: int = TIMEOUT) -> bytes:
    for attempt in range(1, MAX_RETRIES + 1):
        try:
            resp = requests.get(url, timeout=timeout, stream=True)
            resp.raise_for_status()
            # Stream to avoid large in-memory spikes for big files
            chunks = []
            for chunk in resp.iter_content(chunk_size=8192):
                chunks.append(chunk)
            return b"".join(chunks)
        except Exception as exc:
            if attempt == MAX_RETRIES:
                raise
            wait = (2 ** attempt) * 5
            print(f"WARN: CDN fetch failed (attempt {attempt}): {exc} — retry in {wait}s", file=sys.stderr)
            time.sleep(wait)
    raise RuntimeError(f"Failed to fetch CDN resource after retries: {url}")
