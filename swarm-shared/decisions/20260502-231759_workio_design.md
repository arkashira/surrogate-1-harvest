# workio / design

## 1. Diagnosis

- **No idempotency key on LINE webhook payloads** → retries/redeliveries create duplicate punches for the same `employee_id` on the same date.
- **Non-atomic read-then-insert** in punch creation allows races between concurrent or redelivered webhooks.
- **Missing storage-level uniqueness constraint** for active punches (`employee_id, punch_type='in', date`) enables duplicates at the DB layer.
- **No traceability for webhook replays** — missing correlation ID/header propagation makes debugging and de-duping hard.
- **No idempotency table or lightweight dedupe cache** to enforce exactly-once semantics across retries within a time window.

## 2. Proposed change

File scope:  
- `server/src/routes/line/webhook.ts` (or equivalent) — add idempotency check and atomic upsert for punches.  
- `server/src/db/schema.sql` — add unique constraint/index for active punches and an idempotency table.  
- `server/src/services/punchService.ts` — implement `ensurePunch` with atomic `INSERT ... ON CONFLICT DO NOTHING` + idempotency key insert.

## 3. Implementation

### 3.1 DB schema changes (`server/src/db/schema.sql`)

```sql
-- Idempotency keys for LINE webhook events (24h+ retention)
CREATE TABLE IF NOT EXISTS webhook_idempotency (
  idempotency_key VARCHAR(255) PRIMARY KEY,
  event_type      VARCHAR(100) NOT NULL,
  employee_id     INTEGER      NOT NULL,
  created_at      TIMESTAMPTZ  DEFAULT NOW()
);

-- Index to prune old keys (optional, run via cron)
CREATE INDEX IF NOT EXISTS idx_webhook_idempotency_created ON webhook_idempotency (created_at);

-- Ensure only one active "in" punch per employee per date
-- (Assumes punches table has: id, employee_id, punch_type, punched_at, created_at)
CREATE UNIQUE INDEX IF NOT EXISTS idx_active_punch_employee_date
ON punches (employee_id, DATE(punched_at))
WHERE punch_type = 'in' AND deleted_at IS NULL;
```

### 3.2 Service: atomic punch creation (`server/src/services/punchService.ts`)

```ts
import { PoolClient } from 'pg';
import { db } from '../db';

export type PunchInput = {
  employeeId: number;
  punchType: 'in' | 'out';
  punchedAt: Date;
  lineUserId?: string;
  eventId?: string; // LINE webhook event id for idempotency
};

export async function ensurePunch(input: PunchInput, client?: PoolClient): Promise<{ punchId: number; created: boolean }> {
  const tx = client || db;
  const idempotencyKey = input.eventId ? `line:${input.eventId}` : `punch:${input.employeeId}:${input.punchType}:${input.punchedAt.toISOString()}`;

  return await tx.txIf(async (t) => {
    // 1) Idempotency check (fast fail)
    const exists = await t.oneOrNone<{ idempotency_key: string }>(
      `SELECT idempotency_key FROM webhook_idempotency WHERE idempotency_key = $1`,
      [idempotencyKey]
    );
    if (exists) {
      // Already processed — fetch existing punch if any
      const existing = await t.oneOrNone<{ id: number }>(
        `SELECT id FROM punches WHERE employee_id = $1 AND punch_type = $2 AND DATE(punched_at) = DATE($3) AND deleted_at IS NULL ORDER BY punched_at DESC LIMIT 1`,
        [input.employeeId, input.punchType, input.punchedAt]
      );
      if (existing) return { punchId: existing.id, created: false };
      // If missing punch but idempotency exists, proceed to create (safety)
    }

    // 2) Atomic punch insert with unique constraint protection
    const result = await t.oneOrNone<{ id: number }>(
      `INSERT INTO punches (employee_id, punch_type, punched_at, line_user_id, created_at)
       VALUES ($1, $2, $3, $4, NOW())
       ON CONFLICT DO NOTHING
       RETURNING id`,
      [input.employeeId, input.punchType, input.punchedAt, input.lineUserId || null]
    );

    // If unique constraint blocked insert (race), fetch existing
    if (!result) {
      const existing = await t.oneOrNone<{ id: number }>(
        `SELECT id FROM punches WHERE employee_id = $1 AND punch_type = $2 AND DATE(punched_at) = DATE($3) AND deleted_at IS NULL ORDER BY punched_at DESC LIMIT 1`,
        [input.employeeId, input.punchType, input.punchedAt]
      );
      if (!existing) throw new Error('Punch conflict but no row found');
      // Record idempotency to prevent future retries from reprocessing
      await t.none(`INSERT INTO webhook_idempotency (idempotency_key, event_type, employee_id) VALUES ($1, $2, $3) ON CONFLICT DO NOTHING`, [
        idempotencyKey,
        'punch',
        input.employeeId,
      ]);
      return { punchId: existing.id, created: false };
    }

    // 3) Record idempotency
    await t.none(`INSERT INTO webhook_idempotency (idempotency_key, event_type, employee_id) VALUES ($1, $2, $3) ON CONFLICT DO NOTHING`, [
      idempotencyKey,
      'punch',
      input.employeeId,
    ]);

    return { punchId: result.id, created: true };
  });
}
```

### 3.3 Webhook handler (`server/src/routes/line/webhook.ts`)

```ts
import { Router } from 'express';
import { ensurePunch } from '../../services/punchService';
import { verifyLineSignature } from '../../utils/line';

const router = Router();

router.post('/webhook/line', async (req, res) => {
  try {
    const signature = req.headers['x-line-signature'] as string;
    if (!verifyLineSignature(JSON.stringify(req.body), signature)) {
      return res.status(401).send('Invalid signature');
    }

    const events = req.body.events || [];
    for (const ev of events) {
      if (ev.type !== 'message' || ev.message?.type !== 'text') continue;

      // Example: parse clock-in/out command from text or use payload mapping
      const text = ev.message.text.trim().toLowerCase();
      if (!['clock in', 'clock out'].includes(text)) continue;

      const employeeId = await resolveEmployeeIdByLineUserId(ev.source.userId);
      if (!employeeId) continue;

      const punchType = text === 'clock in' ? 'in' : 'out';
      await ensurePunch({
        employeeId,
        punchType,
        punchedAt: new Date(),
        lineUserId: ev.source.userId,
        eventId: ev.source.userId + ':' + ev.timestamp, // or ev.webhookEventId if available
      });
    }

    res.status(200).json({ ok: true });
  } catch (err) {
    console.error('LINE webhook error', err);
    res.status(500).json({ error: 'internal' });
  }
});

export default router;
```
