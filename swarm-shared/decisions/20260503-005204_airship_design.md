### 1) Mac Orchestration Script (15 min)
**Path**: `scripts/build_file_manifest.py`  
**Purpose**: Single API call to list one date folder; outputs deterministic manifest.

```python
#!/usr/bin/env python3
"""
Build CDN file manifest for a single date folder.
Usage: python build_file_manifest.py --repo <repo> --date <YYYY-MM-DD> [--out <path>]
"""
import argparse, json, hashlib, sys, time
from pathlib import Path
from huggingface_hub import HfApi, HfFolder

HF_TOKEN = HfFolder.get_token() or ""
API = HfApi(token=HF_TOKEN)

def list_with_retry(repo_id: str, path: str, retries: int = 2, wait: int = 360):
    for attempt in range(retries):
        try:
            tree = API.list_repo_tree(repo_id=repo_id, path=path, recursive=False)
            files = [item.rfilename for item in tree if item.rfilename.endswith(".parquet")]
            # Include sha256 for change detection (Candidate 2)
            return [{"path": f, "sha256": hashlib.sha256(f.encode()).hexdigest()[:16]} for f in sorted(files)]
        except Exception as e:
            if "429" in str(e) and attempt < retries - 1:
                print(f"Rate limited. Waiting {wait}s...", file=sys.stderr)
                time.sleep(wait)
                continue
            raise
    return []

def main():
    parser = argparse.ArgumentParser(description="Build CDN file manifest.")
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g., org/surrogate-mirror)")
    parser.add_argument("--date", required=True, help="Date folder (YYYY-MM-DD)")
    parser.add_argument("--out", default="manifests/file-list.json", help="Output JSON path")
    args = parser.parse_args()

    folder = f"batches/mirror-merged/{args.date}"
    print(f"Listing {args.repo}/{folder} ...")
    files = list_with_retry(args.repo, folder)

    manifest = {
        "repo": args.repo,
        "date": args.date,
        "folder": folder,
        "files": files,
        "count": len(files),
    }

    out_path = Path(args.out)
    out_path.parent.mkdir(parents=True, exist_ok=True)
    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"Wrote {len(files)} files to {out_path}")

if __name__ == "__main__":
    main()
```

### 2) CDN-Only Dataset Loader (30 min)
**Path**: `surrogate/data/cdn_loader.py`  
**Purpose**: Stream parquet files via CDN; project only `{prompt, response}`; deterministic sibling repo hashing.

```python
#!/usr/bin/env python3
import json, hashlib
from pathlib import Path
from typing import Iterator, Tuple

import pyarrow.parquet as pq
import requests
from torch.utils.data import IterableDataset
from tqdm import tqdm

MANIFEST_PATH = Path(__file__).parent.parent.parent / "manifests" / "file-list.json"
CDN_BASE = "https://huggingface.co/datasets"

def pick_sibling_repo(slug: str, siblings: int = 5) -> str:
    """Deterministic sibling repo by hash (bypasses 128/hr HF commit cap)."""
    idx = int(hashlib.md5(slug.encode()).hexdigest(), 16) % siblings
    return f"surrogate-mirror-sib{idx}"

class CDNParquetDataset(IterableDataset):
    def __init__(self, manifest_path: Path = MANIFEST_PATH):
        manifest = json.loads(manifest_path.read_text())
        self.repo = manifest["repo"]
        self.files = [f["path"] for f in manifest["files"]]

    def _download_and_parse(self, rel_path: str) -> Iterator[Tuple[str, str]]:
        url = f"{CDN_BASE}/{self.repo}/resolve/main/{rel_path}"
        resp = requests.get(url, stream=True, timeout=60)
        resp.raise_for_status()
        tmp_path = Path("/tmp") / f"tmp_{hashlib.md5(rel_path.encode()).hexdigest()}.parquet"
        with open(tmp_path, "wb") as f:
            for chunk in resp.iter_content(chunk_size=8192):
                f.write(chunk)

        # Project only prompt/response to avoid pyarrow.CastError
        table = pq.read_table(tmp_path, columns=["prompt", "response"])
        df = table.to_pandas()
        for _, row in df.iterrows():
            prompt = str(row.get("prompt", ""))
            response = str(row.get("response", ""))
            if prompt and response:
                yield prompt, response
        tmp_path.unlink(missing_ok=True)

    def __iter__(self):
        for f in tqdm(self.files, desc="Loading CDN files"):
            yield from self._download_and_parse(f)
```

### 3) Lightning Studio Reuse Guard (5 min)
**Path**: `surrogate/utils/studio.py`  
**Purpose**: Prevent quota waste by rerunning existing studios; fallback to L40S.

```python
from lightning import Teamspace, Studio, Machine, L40S

def get_or_create_studio(name: str):
    """Reuse running studio; restart on idle with L40S free tier."""
    for s in Teamspace.studios():
        if s.name == name and s.status == "running":
            return s
    return Studio(name=name, machine=Machine.L40S, create_ok=True)
```
