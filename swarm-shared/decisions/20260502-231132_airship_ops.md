### 1. Mac/CDN Manifest Generator (30m)
Single orchestrator script that lists target folders once, emits `file_list.json` with CDN URLs, and ensures Lightning Studio is running.

```bash
#!/usr/bin/env bash
# airship/discover.sh
# Usage: ./discover.sh <repo> <date_folder> [output_json]
# Example: ./discover.sh axentx/surrogate-dataset 2026-05-02

set -euo pipefail

REPO="${1:?repo required}"
DATE_FOLDER="${2:?date_folder required}"
MAC_OUT="${3:-file_list.json}"

python3 - "$REPO" "$DATE_FOLDER" "$MAC_OUT" <<'PY'
import json, os, sys
from huggingface_hub import HfApi

repo = sys.argv[1]
date_folder = sys.argv[2]
out_path = sys.argv[3]

api = HfApi()
# Single non-recursive call per folder
tree = api.list_repo_tree(repo=repo, path=date_folder, recursive=False)

files = []
for item in tree:
    if item.type != "file":
        continue
    # CDN URL (no auth, no API rate limit during training)
    cdn_url = f"https://huggingface.co/datasets/{repo}/resolve/main/{item.path}"
    files.append({
        "repo": repo,
        "path": item.path,
        "sha": item.commit_id or "",
        "size": item.size or 0,
        "cdn_url": cdn_url
    })

with open(out_path, "w") as f:
    json.dump(files, f, indent=2)
print(f"Wrote {len(files)} files to {out_path}", file=sys.stderr)
PY
```

### 2. Training Loader: CDN-Only, Schema-Safe (45m)
Direct CDN fetches with strict `{prompt, response}` projection, eliminating PyArrow `CastError`.

```python
# surrogate/train.py (excerpt)
import json
from pathlib import Path
from typing import Dict, Any, Iterator

import datasets
from huggingface_hub import hf_hub_download

def load_file_list(manifest_path: str) -> list[Dict[str, Any]]:
    with open(manifest_path) as f:
        return json.load(f)

def parse_file_to_example(file_path: str) -> Iterator[Dict[str, str]]:
    # Lightweight projection: keep only prompt/response
    # Implement per-format (jsonl, parquet, csv, etc.)
    # Example for jsonl:
    import json
    with open(file_path, "r", encoding="utf-8") as f:
        for line in f:
            line = line.strip()
            if not line:
                continue
            obj = json.loads(line)
            # Keep only fields we need; drop everything else
            yield {
                "prompt": obj.get("prompt", ""),
                "response": obj.get("response", "")
            }

def cdn_generator(file_list: list[Dict[str, Any]]) -> Iterator[Dict[str, str]]:
    for meta in file_list:
        # Download via CDN (or hf_hub_download which uses CDN)
        local_path = hf_hub_download(
            repo_id=meta["repo"],
            filename=meta["path"],
            repo_type="dataset"
        )
        yield from parse_file_to_example(local_path)

def build_dataset(manifest_path: str) -> datasets.Dataset:
    file_list = load_file_list(manifest_path)
    return datasets.Dataset.from_generator(
        cdn_generator,
        gen_kwargs={"file_list": file_list},
        features=datasets.Features({
            "prompt": datasets.Value("string"),
            "response": datasets.Value("string")
        })
    )

# Usage in Lightning
if __name__ == "__main__":
    manifest = "file_list.json"
    ds = build_dataset(manifest)
    print(ds)
```

### 3. Lightning Studio: Idle-Resilient Runner (30m)
Reuses running studios; auto-restarts on stop with retry logic.

```python
# surrogate/lightning_launcher.py
from lightning_sdk import Studio, Teamspace, Machine
import time

def get_or_create_studio(name: str, machine: Machine = Machine.L40S) -> Studio:
    teamspace = Teamspace()
    for s in teamspace.studios:
        if s.name == name and s.status == "running":
            print(f"Reusing running studio: {name}")
            return s
    print(f"Creating studio: {name}")
    return Studio(name=name, create_ok=True, machine=machine)

def run_with_retry(studio: Studio, target, args=None, max_retries=3):
    args = args or []
    for attempt in range(max_retries):
        if studio.status != "running":
            print(f"Studio stopped (attempt {attempt+1}). Restarting...")
            studio.start(machine=Machine.L40S)
            time.sleep(30)  # allow startup
        try:
            run = studio.run(target=target, arguments=args)
            run.watch()
            return run
        except Exception as e:
            print(f"Run failed: {e}")
            if attempt == max_retries - 1:
                raise
            time.sleep(60)

if __name__ == "__main__":
    studio = get_or_create_studio("surrogate-train-l40s")
    run_with_retry(studio, target="train.py", args=["--manifest", "file_list.json"])
```
