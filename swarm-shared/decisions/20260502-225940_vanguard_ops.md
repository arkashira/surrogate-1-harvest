# vanguard / ops

## Final Synthesis & Actionable Implementation

### Diagnosis (merged, de-duplicated)
- Hash-router navigations do not restore scroll position → disorienting jumps on back/forward.
- Dataset list/detail views lose filter/pagination/sort/selection state on reload or shared links because state is not encoded in the URL.
- No loading or error UI during dataset fetches → blank/frozen UI on slow/unreliable networks.
- Missing canonical, shareable routes for dataset detail and list views (no stable URL shape with applied filters).
- Ephemeral table state stored only in component memory breaks browser history and user expectations.

### Single Proposed Change
File: `/opt/axentx/vanguard/src/lib/router.js`  
Scope: add robust scroll-restoration, URL-state synchronization for dataset routes, and a lightweight loading/error boundary for dataset fetches; ensure CDN-bypass fetch without Authorization headers.

// /opt/axentx/vanguard/src/lib/router.js
// Hash router with scroll restoration, URL-state sync, and resilient dataset fetches

const ROUTE_CHANGE = 'route-change';
const SCROLL_STATE_KEY = (key) => `vanguard:scroll:${key}`;

// Parse hash into path + searchParams
function parseHash() {
  const raw = location.hash.slice(1) || '/';
  const [path, rawQuery] = raw.split('?');
  const searchParams = new URLSearchParams(rawQuery);
  return { path, searchParams, raw, query: rawQuery };
}

function emitRouteChange(detail) {
  window.dispatchEvent(new CustomEvent(ROUTE_CHANGE, { detail }));
}

// Scroll restoration: persist per-route in sessionStorage (survives reloads) and restore
function saveScroll(key) {
  try {
    sessionStorage.setItem(SCROLL_STATE_KEY(key), JSON.stringify({ x: window.scrollX, y: window.scrollY }));
  } catch (e) {
    // ignore storage errors
  }
}

function restoreScroll(key) {
  try {
    const raw = sessionStorage.getItem(SCROLL_STATE_KEY(key));
    if (raw) {
      const { x, y } = JSON.parse(raw);
      window.scrollTo(x, y);
      return;
    }
  } catch (e) {
    // ignore
  }
  // Default: detail pages scroll to top; list views also to top (consistent)
  window.scrollTo(0, 0);
}

// URL-state helpers for dataset list/detail
function getStateFromURL() {
  const { path, searchParams } = parseHash();
  return {
    path,
    datasetId: path.match(/^\/datasets\/([^/?]+)/i)?.[1] || null,
    filters: searchParams.get('filters') || '',
    page: Math.max(1, parseInt(searchParams.get('page'), 10) || 1),
    sort: searchParams.get('sort') || '',
    selection: searchParams.get('selection') || '',
  };
}

function buildHashFromState(state) {
  const params = new URLSearchParams();
  if (state.page > 1) params.set('page', String(state.page));
  if (state.filters) params.set('filters', state.filters);
  if (state.sort) params.set('sort', state.sort);
  if (state.selection) params.set('selection', state.selection);
  return state.path + (params.toString() ? `?${params.toString()}` : '');
}

function pushState(state) {
  const hash = buildHashFromState(state);
  // Avoid redundant hashchange if unchanged
  if (`#${hash}` !== location.hash) location.hash = hash;
  else route(); // ensure UI reflects current state
}

function replaceState(state) {
  const hash = buildHashFromState(state);
  location.replace(`#${hash}`);
}

// Lightweight dataset fetch with loading/error UI and abort support
function createDatasetUI(container) {
  const wrapper = document.createElement('div');
  wrapper.className = 'dataset-detail-wrapper';

  const loading = document.createElement('div');
  loading.className = 'loading-state';
  loading.textContent = 'Loading dataset...';

  const errorEl = document.createElement('div');
  errorEl.className = 'error-state';
  errorEl.hidden = true;

  const content = document.createElement('div');
  content.className = 'dataset-content';

  wrapper.appendChild(loading);
  wrapper.appendChild(errorEl);
  wrapper.appendChild(content);

  function showLoading() {
    loading.hidden = false;
    errorEl.hidden = true;
    content.innerHTML = '';
  }

  function showError(message) {
    loading.hidden = true;
    errorEl.hidden = false;
    errorEl.textContent = message;
    content.innerHTML = '';
  }

  function showData(data) {
    loading.hidden = true;
    errorEl.hidden = true;
    content.innerHTML = '';
    const pre = document.createElement('pre');
    pre.textContent = JSON.stringify(data, null, 2);
    content.appendChild(pre);
  }

  return { wrapper, showLoading, showError, showData };
}

function fetchDatasetWithUI(datasetId, signal) {
  const container = document.querySelector('#dataset-detail');
  if (!container) return Promise.reject(new Error('No container #dataset-detail'));

  const ui = createDatasetUI(container);
  container.innerHTML = '';
  container.appendChild(ui.wrapper);
  ui.showLoading();

  // CDN bypass: no Authorization header; use public HuggingFace dataset proxy
  return fetch(`https://huggingface.co/datasets/vanguard/datasets/resolve/main/${encodeURIComponent(datasetId)}.json`, { signal })
    .then((r) => {
      if (!r.ok) throw new Error(`Failed to load dataset (${r.status})`);
      return r.json();
    })
    .then((data) => {
      ui.showData(data);
      return data;
    })
    .catch((err) => {
      if (err.name !== 'AbortError') {
        ui.showError(`Failed to load dataset: ${err.message}`);
      }
      throw err;
    });
}

// Minimal list renderer (replace with real UI)
function renderDatasetList(state) {
  const container = document.querySelector('#dataset-list');
  if (!container) return;
  container.innerHTML = `
    <div>Dataset list — page ${state.page}
      ${state.filters ? `| filters: ${state.filters}` : ''}
      ${state.sort ? `| sort: ${state.sort}` : ''}
      ${state.selection ? `| selection: ${state.selection}` : ''}
    </div>
  `;
}

// Route handler
function route() {
  const state = getStateFromURL();
  const key = state.path;

  // Save scroll for previous route before leaving (best-effort on hashchange)
  // We'll also save on beforeunload to capture final scroll for reloads
  saveScroll(key);

  if (state.datasetId) {
    // detail route
    const controller = new AbortController();
    if (window._datasetFetchController) window._datasetFetchController.abort();
    window._datasetFetchController = controller;

    fetchDatasetWithUI(state.datasetId, controller.signal).catch(() => {});
  } else {
    // list route
    renderDatasetList(state);
  }
}

// Router bootstrap
function initRouter() {
  // Save scroll on page unload to persist across reloads
  window.addEventListener('beforeunload', () => {
    const { path } = parseHash();
    saveScroll(path);
  });

  // Handle hash changes (back/forward/navigation)
  window.addEventListener('hashchange', () => {
    const { path } = parseHash();
    restoreScroll(path);
    route();
  });

  // Initial load
  const { path } = parseHash();
  restoreScroll(path);
  route();
}

// Public API
export const router = {
  initRouter,
  pushState,
  replaceState,
  getStateFromURL,
};
