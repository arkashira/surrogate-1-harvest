# Final synthesized solution

## Diagnosis (merged)
- No persisted `(repo, dateFolder) → file-list` manifest; every run calls authenticated `list_repo_tree`, burning HF API quota and risking 429s.
- Training/data loader likely uses `load_dataset(streaming=True)` or per-file API calls instead of CDN-only fetches, causing auth overhead and rate-limit exposure.
- No local cache layer for file manifests; re-runs recompute identical lists and re-download metadata.
- No retry/back-off strategy for 429s; bursts can stall the pipeline.
- No deterministic repo selection for commit-cap mitigation (all writes to one repo risk the 128/hr cap).

## Core proposal
Add two small, focused modules under `vanguard/ops/`:

1. **`file_manifest.py`** — generate and cache a `(repo, dateFolder) → file-list` JSON manifest with one authenticated `list_repo_tree` call, exponential back-off, 360s sleep on persistent 429, and deterministic sibling-repo selection for writes.
2. **`train_cdn.py`** — training helper that reads the manifest and streams files via CDN-only URLs (`https://huggingface.co/datasets/{repo}/resolve/main/{path}`) with zero auth and zero HF API calls during training.

Patch existing training scripts to accept a manifest and use the CDN-only dataset loader.

## Implementation (single, complete, production-ready)

```bash
# /opt/axentx/vanguard/ops/file_manifest.py
#!/usr/bin/env python3
"""
Generate and cache (repo, dateFolder) -> file-list manifest for HF datasets.

Usage:
  python -m vanguard.ops.file_manifest --repo org/name --date 2026-04-29 --out manifest.json
"""
import argparse
import hashlib
import json
import sys
import time
from pathlib import Path
from typing import Any, Dict, List

try:
    from huggingface_hub import HfApi, RateLimitError, RepositoryNotFoundError
except ImportError:
    print("Missing huggingface_hub. Install with: pip install huggingface_hub", file=sys.stderr)
    sys.exit(1)

API = HfApi()

def list_date_files(repo: str, date_folder: str, retries: int = 5, initial_backoff: float = 5.0) -> List[Dict[str, str]]:
    """
    List top-level files in repo/date_folder with exponential back-off.
    On persistent 429, sleep 360s once then raise.
    """
    backoff = initial_backoff
    for attempt in range(1, retries + 1):
        try:
            items = API.list_repo_tree(repo=repo, path=date_folder, recursive=False)
            files: List[Dict[str, str]] = []
            for it in items:
                if isinstance(it, dict):
                    path = it.get("path")
                    kind = it.get("type")
                else:
                    path = getattr(it, "path", None)
                    kind = getattr(it, "type", None)
                if path and kind == "file":
                    files.append({"repo": repo, "date": date_folder, "path": path})
            return files
        except RateLimitError:
            if attempt == retries:
                # Final mitigation: sleep 360s once, then fail so caller can retry later
                print("Rate limit exceeded after retries. Sleeping 360s.", file=sys.stderr)
                time.sleep(360)
                raise
            wait = backoff * (2 ** (attempt - 1))
            print(f"Rate limited (attempt {attempt}/{retries}). Wait {wait:.1f}s", file=sys.stderr)
            time.sleep(wait)
        except RepositoryNotFoundError:
            print(f"Repository not found: {repo}", file=sys.stderr)
            raise
    return []

def pick_write_repo(base_repo: str, n_siblings: int = 5, salt: str = "") -> str:
    """
    Deterministic sibling repo selection to avoid HF commit cap (128/hr/repo).
    base_repo format: org/name
    Returns org/name-suffix (e.g., org/name-03)
    """
    key = f"{base_repo}:{salt}"
    idx = int(hashlib.sha256(key.encode()).hexdigest(), 16) % n_siblings
    return f"{base_repo}-{idx:02d}"

def main() -> None:
    parser = argparse.ArgumentParser(description="Generate HF file manifest for a date folder.")
    parser.add_argument("--repo", required=True, help="HF dataset repo (org/name)")
    parser.add_argument("--date", required=True, help="Date folder (YYYY-MM-DD)")
    parser.add_argument("--out", required=True, help="Output JSON path")
    parser.add_argument("--siblings", type=int, default=5, help="Number of sibling repos for writes")
    parser.add_argument("--salt", default="", help="Salt for deterministic repo selection")
    args = parser.parse_args()

    files = list_date_files(args.repo, args.date)
    manifest = {
        "repo": args.repo,
        "date": args.date,
        "generated_at": time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime()),
        "files": files,
        "write_repo": pick_write_repo(args.repo, n_siblings=args.siblings, salt=args.salt),
    }

    out_path = Path(args.out)
    out_path.parent.mkdir(parents=True, exist_ok=True)
    with open(out_path, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Manifest written to {out_path} ({len(files)} files)")

if __name__ == "__main__":
    main()
```

# /opt/axentx/vanguard/ops/train_cdn.py
"""
Lightning-compatible training helper that uses a pre-generated manifest to fetch files via CDN only.

Usage:
  python -m vanguard.ops.train_cdn --manifest manifest.json --output-dir ./out

Notes:
- Manifest JSON must contain {"repo": "...", "files": [{"path": "..."}]}
- Each file is expected to be a JSONL with {"prompt": ..., "response": ...} (or input/output).
- All fetches use public CDN URLs; no HF API/auth during training.
"""
import argparse
import json
import sys
from pathlib import Path
from typing import Dict, Iterator, Tuple

import requests
import torch
from torch.utils.data import DataLoader, IterableDataset

CDN_ROOT = "https://huggingface.co/datasets"

class CDNTextDataset(IterableDataset):
    """
    Streams text pairs from manifest entries using CDN URLs (no auth).
    """
    def __init__(self, manifest_path: str, max_files: int = -1):
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.files = [f["path"] for f in self.manifest.get("files", [])]
        if max_files > 0:
            self.files = self.files[:max_files]
        self.repo = self.manifest.get("repo")
        if not self.repo:
            raise ValueError("Manifest must contain 'repo'")

    def _stream_file(self, rel_path: str) -> Iterator[Tuple[str, str]]:
        url = f"{CDN_ROOT}/{self.repo}/resolve/main/{rel_path}"
        resp = requests.get(url, timeout=30, stream=True)
        resp.raise_for_status()
        for line in resp.iter_lines(decode_unicode=True):
            if not line:
                continue
            line = line.strip()
            if not line:
                continue
            try:
                obj = json.loads(line)
                prompt = obj.get("prompt") or obj.get("input") or ""
                response = obj.get("response") or obj.get("output") or ""
                if prompt or response:
                    yield prompt, response
            except Exception:
                continue

    def __iter__(self):
        for rel_path in self.files:
            yield from self._stream_file(rel_path)

def build_dataloader(manifest_path: str, batch_size: int = 8, max_files: int = -1) -> DataLoader:
    dataset = CDNTextDataset(manifest_path, max_files=max_files)
    return DataLoader(dataset, batch_size=batch_size)
