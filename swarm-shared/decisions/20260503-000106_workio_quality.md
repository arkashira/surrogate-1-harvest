# workio / quality

## 1. Diagnosis
- Clock-in/out endpoints accept duplicate requests within the same minute (retries, double-taps, LINE duplicate webhooks) → multiple punches for same tenant/user/day with no idempotency key.
- No server-side state machine enforcement: employee can clock-in twice or clock-out without a prior clock-in, producing invalid daily punch sequences.
- No optimistic UI or local pending queue on the frontend — spinner ambiguity and zero feedback when LINE/webhook or API calls fail.
- Punch submission has no client-side idempotency key → retries (offline/poor network) can create duplicate punches on the backend.
- Missing last-known-state cache causes stale UI after failures; users can’t distinguish between “pending” and “failed” states.

## 2. Proposed change
- **Backend**: add idempotency enforcement and state-machine validation to `server/src/routes/punch.ts` (or equivalent) for `/punch/clock-in` and `/punch/clock-out`.
- **Frontend**: add optimistic UI + local pending queue + idempotency key to `workio/src/features/punch/usePunch.ts` (or equivalent hook/component) and surface clear pending/failed/synced states.

## 3. Implementation

### Backend: idempotency + state-machine (Node/Express + PostgreSQL)

File: `server/src/routes/punch.ts`

```ts
import express from 'express';
import { db } from '../db';
import { verifyTenant } from '../middleware/auth';

const router = express.Router();

// Simple in-memory short-term dedupe (per process) for same-millisecond retries.
// For multi-instance safety, rely on DB unique constraint below.
const recentIds = new Set<string>();

router.post('/clock-in', verifyTenant, async (req, res) => {
  const { userId, latitude, longitude, idempotencyKey } = req.body;
  const tenantId = (req as any).tenantId;

  if (!idempotencyKey) {
    return res.status(400).json({ error: 'idempotencyKey required' });
  }

  // Short-circuit recent in-process duplicates
  if (recentIds.has(idempotencyKey)) {
    return res.status(200).json({ ok: true, duplicate: true });
  }

  const today = new Date().toISOString().split('T')[0];

  try {
    await db.query('BEGIN');

    // Fetch latest punch for user today
    const latest = await db.query(
      `SELECT id, type, created_at FROM punches
       WHERE tenant_id = $1 AND user_id = $2 AND DATE(created_at) = $3
       ORDER BY created_at DESC LIMIT 1`,
      [tenantId, userId, today]
    );

    const last = latest.rows[0];

    // State machine: cannot clock-in if last punch today is already clock-in without clock-out
    if (last && last.type === 'clock-in') {
      await db.query('ROLLBACK');
      return res.status(409).json({ error: 'Already clocked in' });
    }

    // Idempotency: store key with punch; unique constraint prevents duplicates across instances
    const insert = await db.query(
      `INSERT INTO punches (tenant_id, user_id, type, latitude, longitude, idempotency_key, created_at)
       VALUES ($1, $2, $3, $4, $5, $6, NOW())
       ON CONFLICT (tenant_id, user_id, idempotency_key) DO UPDATE
         SET idempotency_key = EXCLUDED.idempotency_key
       RETURNING id`,
      [tenantId, userId, 'clock-in', latitude, longitude, idempotencyKey]
    );

    recentIds.add(idempotencyKey);
    setTimeout(() => recentIds.delete(idempotencyKey), 60_000);

    await db.query('COMMIT');
    res.json({ ok: true, punchId: insert.rows[0].id });
  } catch (err: any) {
    await db.query('ROLLBACK');
    if (err.code === '23505') {
      // Unique violation on idempotency_key — duplicate request
      return res.status(200).json({ ok: true, duplicate: true });
    }
    res.status(500).json({ error: 'Server error' });
  }
});

router.post('/clock-out', verifyTenant, async (req, res) => {
  const { userId, latitude, longitude, idempotencyKey } = req.body;
  const tenantId = (req as any).tenantId;

  if (!idempotencyKey) {
    return res.status(400).json({ error: 'idempotencyKey required' });
  }

  if (recentIds.has(idempotencyKey)) {
    return res.status(200).json({ ok: true, duplicate: true });
  }

  const today = new Date().toISOString().split('T')[0];

  try {
    await db.query('BEGIN');

    const latest = await db.query(
      `SELECT id, type FROM punches
       WHERE tenant_id = $1 AND user_id = $2 AND DATE(created_at) = $3
       ORDER BY created_at DESC LIMIT 1`,
      [tenantId, userId, today]
    );

    const last = latest.rows[0];

    // State machine: must have a clock-in today without a clock-out
    if (!last || last.type === 'clock-out') {
      await db.query('ROLLBACK');
      return res.status(409).json({ error: 'No active clock-in to clock out' });
    }

    const insert = await db.query(
      `INSERT INTO punches (tenant_id, user_id, type, latitude, longitude, idempotency_key, created_at)
       VALUES ($1, $2, $3, $4, $5, $6, NOW())
       ON CONFLICT (tenant_id, user_id, idempotency_key) DO UPDATE
         SET idempotency_key = EXCLUDED.idempotency_key
       RETURNING id`,
      [tenantId, userId, 'clock-out', latitude, longitude, idempotencyKey]
    );

    recentIds.add(idempotencyKey);
    setTimeout(() => recentIds.delete(idempotencyKey), 60_000);

    await db.query('COMMIT');
    res.json({ ok: true, punchId: insert.rows[0].id });
  } catch (err: any) {
    await db.query('ROLLBACK');
    if (err.code === '23505') {
      return res.status(200).json({ ok: true, duplicate: true });
    }
    res.status(500).json({ error: 'Server error' });
  }
});

export default router;
```

### Frontend: optimistic UI + idempotency key

File: `src/features/punch/usePunch.ts`

```ts
import { useState, useCallback } from 'react';
import { v4 as uuidv4 } from 'uuid';

type PunchState = 'idle' | 'pending' | 'synced' | 'failed';

export function usePunch() {
  const [state, setState] = useState<PunchState>('idle');
  const [lastError, setLastError] = useState<string | null>(null);
  const [queue, setQueue] = useState<Array<{ id: string; type: 'clock-in' | 'clock-out'; payload: any }>>([]);

  const postPunch = useCallback(async (type: 'clock-in' | 'clock-out', payload: any) => {
    const idempotencyKey = uuidv4();
    const item = { id: idempotencyKey, type, payload: { ...payload, idempotencyKey } };

    // Optimistic: add to pending queue and update local UI state immediately
    setQueue((q) => [...q, item]);
    setState('pending');
    setLastError(null);

    try {
      const res = await fetch(`/api/punch/${type}`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(item.payload),
      });
