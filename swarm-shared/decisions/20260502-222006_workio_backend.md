### 2. Implementation (merged + hardened)

#### `src/middleware/requestId.ts`
- Adds deterministic request ID and sets response header.
- Exposes ID on `req` and for logging.

```ts
import { Request, Response, NextFunction } from 'express';
import crypto from 'crypto';

declare global {
  namespace Express {
    interface Request {
      requestId: string;
    }
  }
}

export function requestIdMiddleware(req: Request, res: Response, next: NextFunction) {
  const id = (req.headers['x-request-id'] as string) || crypto.randomUUID();
  req.requestId = id;
  res.setHeader('X-Request-ID', id);
  next();
}
```

#### `src/middleware/lineSignature.ts`
- Validates `X-Line-Signature` using HMAC-SHA256.
- Uses raw body when available; otherwise falls back to `JSON.stringify` (ensure consistent serialization).
- Fails fast on missing/invalid signatures.

```ts
import { Request, Response, NextFunction } from 'express';
import crypto from 'crypto';

const LINE_CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';

export function lineSignatureMiddleware(req: Request, res: Response, next: NextFunction) {
  // Skip if secret not configured (dev only)
  if (!LINE_CHANNEL_SECRET) return next();

  const signature = req.headers['x-line-signature'] as string;
  if (!signature) return res.status(400).json({ error: 'Missing X-Line-Signature' });

  // Prefer raw body if available (recommended for signature checks)
  const body = (req as any).rawBody || JSON.stringify(req.body || {});
  const expected = crypto
    .createHmac('sha256', LINE_CHANNEL_SECRET)
    .update(body, 'utf8')
    .digest('base64');

  if (signature !== expected) {
    return res.status(401).json({ error: 'Invalid signature' });
  }
  next();
}
```

#### `src/middleware/tenantCache.ts`
- In-memory cache with TTL (Map).
- Exposes typed `req.tenant`.
- Safe for concurrent access; misses fall through to DB.
- For multi-instance, swap to Redis.

```ts
import { Request, Response, NextFunction } from 'express';
import { Pool } from 'pg';

declare global {
  namespace Express {
    interface Request {
      tenant: Tenant | null;
    }
  }
}

interface Tenant {
  id: string;
  channelId: string;
  name: string;
}

type CacheEntry = { tenant: Tenant; expiresAt: number };

const cache = new Map<string, CacheEntry>();
const TTL_MS = 60_000; // 60s

export function createTenantCache(pool: Pool) {
  async function getTenantByChannelId(channelId: string): Promise<Tenant | null> {
    const now = Date.now();
    const cached = cache.get(channelId);
    if (cached && cached.expiresAt > now) return cached.tenant;

    const result = await pool.query<Tenant>(
      'SELECT id, channel_id AS "channelId", name FROM tenants WHERE channel_id = $1 LIMIT 1',
      [channelId]
    );
    const tenant = result.rows[0] || null;
    if (tenant) {
      cache.set(channelId, { tenant, expiresAt: now + TTL_MS });
    }
    return tenant;
  }

  return async function tenantCacheMiddleware(req: Request, res: Response, next: NextFunction) {
    // LINE webhook destination is the bot's channel ID
    const channelId = req.body?.destination || (req.headers['x-line-channel-id'] as string);
    if (!channelId) {
      req.tenant = null;
      return next();
    }

    try {
      req.tenant = await getTenantByChannelId(channelId);
      next();
    } catch (err) {
      next(err);
    }
  };
}
```

#### `src/middleware/health.ts`
- Liveness: process is alive.
- Readiness: DB reachable.

```ts
import { Request, Response } from 'express';
import { Pool } from 'pg';

export function createHealthHandlers(pool: Pool) {
  async function liveness(req: Request, res: Response) {
    return res.json({ status: 'ok', type: 'liveness' });
  }

  async function readiness(req: Request, res: Response) {
    try {
      await pool.query('SELECT 1');
      return res.json({ status: 'ready', db: 'up' });
    } catch (err) {
      return res.status(503).json({ status: 'unready', db: 'down' });
    }
  }

  return { liveness, readiness };
}
```

### 3. Wire up in `src/app.ts`

- Add body parser with raw body capture for signature checks.
- Apply middleware stack in order: requestId → health → lineSignature → tenantCache → routes.
- Structured logs include requestId and tenant.

```ts
import express from 'express';
import bodyParser from 'body-parser';
import { Pool } from 'pg';
import { requestIdMiddleware } from './middleware/requestId';
import { lineSignatureMiddleware } from './middleware/lineSignature';
import { createTenantCache } from './middleware/tenantCache';
import { createHealthHandlers } from './middleware/health';

const app = express();
const pool = new Pool({ connectionString: process.env.DATABASE_URL });

// Capture raw body for LINE signature verification
app.use(
  bodyParser.json({
    limit: '1mb',
    verify: (req, res, buf) => {
      (req as any).rawBody = buf.toString('utf8');
    },
  })
);

app.use(requestIdMiddleware);

// Health endpoints
const health = createHealthHandlers(pool);
app.get('/health', health.liveness);
app.get('/ready', health.readiness);

// Tenant cache middleware
const tenantCache = createTenantCache(pool);

// Structured logger helper
function log(req: any, level: string, message: string, extra?: Record<string, any>) {
  const entry = {
    ts: new Date().toISOString(),
    level,
    requestId: req.requestId,
    tenantId: req.tenant?.id || null,
    message,
    ...extra,
  };
  console.log(JSON.stringify(entry));
}

// LINE webhook route
app.post(
  '/webhook/line',
  lineSignatureMiddleware,
  tenantCache,
  (req, res) => {
    const { tenant } = req;

    if (!tenant) {
      log(req, 'warn', 'No tenant found for webhook', { channelId: req.body?.destination });
      return res.status(404).json({ error: 'Tenant not found' });
    }

    log(req, 'info', 'Webhook received', {
      eventsCount: req.body?.events?.length || 0,
    });

    // Process events...
    // await processLineEvents(req.body.events, tenant);

    res.json({ status: 'accepted' });
  }
);

export default app;
```
