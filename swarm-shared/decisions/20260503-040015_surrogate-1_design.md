# surrogate-1 / design

## Final Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16` (matrix) and optional `DATE_FOLDER` (defaults to today `YYYY-MM-DD`).
- Uses a **single API call** from the runner (after rate-limit window) to fetch `list_repo_tree(path=DATE_FOLDER, recursive=False)` and saves the file list to `manifest.json`.
- Embeds the manifest in the worker; during execution each shard deterministically hashes `slug → shard` and only processes its slice.
- Downloads files via **HF CDN** (`https://huggingface.co/datasets/axentx/surrogate-1-training-pairs/resolve/main/{DATE_FOLDER}/{file}`) with no Authorization header — bypasses `/api/` rate limits entirely.
- Projects each file to `{prompt, response}` only at parse time (avoids `pyarrow.CastError` on mixed schemas).
- Dedups via central md5 store (`lib/dedup.py`) and writes output to:
  ```
  batches/public-merged/<DATE_FOLDER>/shard<N>-<HHMMSS>.jsonl
  ```
- Reuses existing GitHub Actions matrix (`ingest.yml`) — no workflow changes required.

def deterministic_shard(slug: str) -> int:
    h = int(hashlib.md5(slug.encode()).hexdigest(), 16)
    return h % SHARD_TOTAL

def cdn_url(repo: str, path: str) -> str:
    # repo format: datasets/owner/name
    _, owner, name = repo.split("/", 2)
    return f"https://huggingface.co/datasets/{owner}/{name}/resolve/main/{path}"

def list_date_files(date_folder: str) -> List[str]:
    """Single API call to list files in date folder (non-recursive)."""
    api = HfApi(token=HF_TOKEN or None)
    try:
        tree = api.list_repo_tree(repo_id=HF_REPO, path=date_folder, recursive=False)
    except Exception as e:
        print(f"[WARN] HF API list failed: {e}", file=sys.stderr)
        # fallback: try to read existing manifest
        if MANIFEST_PATH.exists():
            print("[INFO] Using existing manifest.json", file=sys.stderr)
            return json.loads(MANIFEST_PATH.read_text())
        raise
    files = [item.rfilename for item in tree if item.type == "file"]
    MANIFEST_PATH.write_text(json.dumps(files, indent=2))
    return files

def project_to_pair(file_path: Path) -> List[Dict[str, str]]:
    """
    Project file to {prompt, response} pairs.
    Supports .jsonl and .parquet; ignores schema mismatches by selecting
    only known columns and coercing to string.
    """
    suffix = file_path.suffix.lower()
    pairs = []

    try:
        if suffix == ".parquet":
            # Avoid mixed-schema CastError: read only metadata or try schema subset
            try:
                pf = pq.read_table(file_path, columns=["prompt", "response"])
            except (pa.ArrowInvalid, KeyError, OSError):
                # fallback: read all string columns and rename
                pf = pq.read_table(file_path)
                cols = [c for c in pf.column_names if c in ("prompt", "response", "instruction", "completion", "text")]
                if not cols:
                    return []
                tbl = pf.select(cols).cast(pa.schema([(c, pa.string()) for c in cols]))
                # heuristic: pick prompt/response
                prompt_col = next((c for c in ("prompt", "instruction", "text") if c in tbl.column_names), tbl.column_names[0])
                response_col = next((c for c in ("response", "completion") if c in tbl.column_names), None)
                if response_col is None:
                    return []
                prompts = tbl[prompt_col].to_pylist()
                responses = tbl[response_col].to_pylist()
                for p, r in zip(prompts, responses):
                    if p is not None and r is not None:
                        pairs.append({"prompt": str(p).strip(), "response": str(r).strip()})
                return pairs

            # normal path
            df = pf.to_pandas()
            prompt_col = "prompt" if "prompt" in df.columns else ("instruction" if "instruction" in df.columns else None)
            response_col = "response" if "response" in df.columns else ("completion" if "completion" in df.columns else None)
            if prompt_col is None or response_col is None:
                return []
            for _, row in df.iterrows():
                p, r = row.get(prompt_col), row.get(response_col)
                if p is not None and r is not None:
                    pairs.append({"prompt": str(p).strip(), "response": str(r).strip()})
            return pairs

        elif suffix == ".jsonl":
            for line in file_path.read_text().splitlines():
                line = line.strip()
                if not line:
                    continue
                try:
                    obj = json.loads(line)
                except json.JSONDecodeError:
                    continue
                # heuristic field names
                prompt = obj.get("prompt") or obj.get("instruction") or obj.get("text")
                response = obj.get("response") or obj.get("completion")
                if prompt is not None and response is not None:
                    pairs.append({"prompt": str(prompt).strip(), "response": str(response).strip()})
            return pairs

        else:
            return []
    except Exception as e:
        print(f"[WARN] Failed to project {file_path}: {e}", file=sys.stderr)
        return []

def slug_from_path(file_path: str) -> str:
    """Deterministic slug for dedup/sharding (relative path without extension)."""
    return Path(file_path).stem

def download_via_cdn(url: str, dest: Path) -> bool:
    """Download via CDN (no auth). Retry with backoff."""
    dest.parent.mkdir(parents=True, exist_ok=True)
    for attempt in range(3):
        try:
            resp = requests.get(url, timeout=30, stream=True)
            resp.raise_for_status()
            with open(dest, "wb") as f:
                for chunk in resp.iter_content(chunk_size=8192):
                    f.write(chunk)
            return True
        except Exception as e:
            wait = 2 ** attempt
            print(f"[WARN] CDN download failed ({url}) attempt={attempt}: {e}; retry in {wait}s", file=sys.stderr)
            time.sleep(wait)
    return False
