import { useCallback, useEffect, useRef, useState } from 'react';
import { v4 as uuidv4 } from 'uuid';

type PunchType = 'IN' | 'OUT';

type PendingPunch = {
  id: string; // idempotency key
  type: PunchType;
  timestamp: string; // ISO
  latitude?: number;
  longitude?: number;
  retries: number;
  createdAt: number; // monotonic for staleness checks
};

const MAX_RETRIES = 5;
const RETRY_DELAYS = [1, 2, 5, 10, 30]; // seconds
const PENDING_TTL_MS = 24 * 60 * 60 * 1000; // 1 day

export function usePunchQueue(tenantId: string, userId: string) {
  const [pending, setPending] = useState<PendingPunch[]>([]);
  const [isOnline, setIsOnline] = useState<boolean>(true);
  const processingRef = useRef<Set<string>>(new Set());
  const retryTimeouts = useRef<Map<string, NodeJS.Timeout>>(new Map());

  // Network status
  useEffect(() => {
    const onOnline = () => setIsOnline(true);
    const onOffline = () => setIsOnline(false);
    window.addEventListener('online', onOnline);
    window.addEventListener('offline', onOffline);
    setIsOnline(navigator.onLine);
    return () => {
      window.removeEventListener('online', onOnline);
      window.removeEventListener('offline', onOffline);
    };
  }, []);

  // Persistence helpers
  const persistQueue = useCallback(
    (q: PendingPunch[]) => {
      try {
        localStorage.setItem(
          `punchQueue:${tenantId}:${userId}`,
          JSON.stringify(q)
        );
      } catch {
        // ignore quota/incognito errors
      }
    },
    [tenantId, userId]
  );

  // Load persisted queue on mount
  useEffect(() => {
    try {
      const raw = localStorage.getItem(`punchQueue:${tenantId}:${userId}`);
      if (raw) {
        const parsed: unknown = JSON.parse(raw);
        if (Array.isArray(parsed)) {
          // Filter stale entries and normalize
          const now = Date.now();
          const cleaned = parsed.filter((item): item is PendingPunch => {
            if (!item || typeof item !== 'object') return false;
            const p = item as PendingPunch;
            if (!p.id || !p.type || !p.timestamp || typeof p.retries !== 'number')
              return false;
            // drop entries older than TTL
            const age = now - new Date(p.timestamp).getTime();
            return age >= 0 && age <= PENDING_TTL_MS;
          });
          setPending(cleaned);
        }
      }
    } catch {
      setPending([]);
    }
  }, [tenantId, userId]);

  // Persist when pending changes
  useEffect(() => {
    persistQueue(pending);
  }, [pending, persistQueue]);

  // Clear retry timeouts on unmount
  useEffect(() => {
    return () => {
      retryTimeouts.current.forEach(clearTimeout);
      retryTimeouts.current.clear();
    };
  }, []);

  // Core queue operations
  const enqueue = useCallback(
    (type: PunchType, location?: { latitude: number; longitude: number }) => {
      const item: PendingPunch = {
        id: uuidv4(),
        type,
        timestamp: new Date().toISOString(),
        latitude: location?.latitude,
        longitude: location?.longitude,
        retries: 0,
        createdAt: Date.now(),
      };
      setPending((prev) => [item, ...prev]);
      return item.id;
    },
    []
  );

  const remove = useCallback((id: string) => {
    setPending((prev) => prev.filter((p) => p.id !== id));
    const t = retryTimeouts.current.get(id);
    if (t) clearTimeout(t);
    retryTimeouts.current.delete(id);
  }, []);

  const scheduleRetry = useCallback(
    (item: PendingPunch) => {
      const t = retryTimeouts.current.get(item.id);
      if (t) clearTimeout(t);

      if (item.retries >= MAX_RETRIES) {
        retryTimeouts.current.delete(item.id);
        return;
      }

      const delayMs = RETRY_DELAYS[item.retries] * 1000;
      const timeout = setTimeout(() => {
        retryTimeouts.current.delete(item.id);
        // trigger processing by toggling a dependency or calling processQueue
        setPending((p) => [...p]); // shallow re-trigger
      }, delayMs);

      retryTimeouts.current.set(item.id, timeout);
    },
    []
  );

  // Send single punch
  const sendPunch = useCallback(
    async (item: PendingPunch): Promise<boolean> => {
      if (processingRef.current.has(item.id)) return false;
      processingRef.current.add(item.id);

      try {
        const res = await fetch('/api/punch', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            tenantId,
            userId,
            type: item.type,
            timestamp: item.timestamp,
            latitude: item.latitude,
            longitude: item.longitude,
            idempotencyKey: item.id,
          }),
        });

        if (res.ok) {
          remove(item.id);
          return true;
        }
        throw new Error(`HTTP ${res.status}`);
      } catch {
        setPending((prev) =>
          prev.map((p) =>
            p.id === item.id && p.retries < MAX_RETRIES
              ? { ...p, retries: p.retries + 1 }
              : p
          )
        );
        return false;
      } finally {
        processingRef.current.delete(item.id);
      }
    },
    [tenantId, userId, remove]
  );

  // Process queue
  const processQueue = useCallback(async () => {
    if (!isOnline) return;
    const toProcess = pending.filter(
      (p) => !processingRef.current.has(p.id) && p.retries < MAX_RETRIES
    );
    // Process serially to avoid thundering herd
    for (const item of toProcess) {
      await sendPunch(item);
    }
  }, [isOnline, pending, sendPunch]);

  // Retry scheduler effect
  useEffect(() => {
    pending.forEach((item) => {
      if (item.retries < MAX_RETRIES && !processingRef.current.has(item.id)) {
        scheduleRetry(item);
      }
    });
    return () => {
      retryTimeouts.current.forEach(clearTimeout);
    };
  }, [pending, scheduleRetry]);

  // Online-triggered processing
  useEffect(() => {
    if (isOnline) processQueue();
  }, [isOnline, processQueue]);

  // Status helper
  const getStatus = useCallback(
    (id: string) => {
      const item = pending.find((p) => p.id === id);
      if (!item) return 'unknown';
      if (processingRef.current.has(item.id)) return 'sending';
      if (item.retries >= MAX_RETRIES) return 'failed';
      return 'pending';
    },
    [pending]
  );

  return {
    enqueue,
    pending,
    getStatus,
    processQueue,
    isOnline,
    remove,
  };
}
