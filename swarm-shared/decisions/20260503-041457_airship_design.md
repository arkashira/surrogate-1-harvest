```python
# scripts/generate_cdn_manifest.py
import json
import os
import sys
from datetime import datetime
from huggingface_hub import HfApi

def main(repo: str, date: str, out: str):
    api = HfApi()
    folder = f"{date}"  # e.g. surrogate/datasets/batches/mirror-merged/2026-05-03
    tree = api.list_repo_tree(repo=repo, path=folder, recursive=False)
    files = [
        {
            "repo": repo,
            "path": f"{folder}/{node.path.split('/')[-1]}",
            "cdn_url": f"https://huggingface.co/datasets/{repo}/resolve/main/{folder}/{node.path.split('/')[-1]}",
            "size": getattr(node, "size", None),
        }
        for node in tree
        if node.path.lower().endswith(".parquet")
    ]
    os.makedirs(os.path.dirname(out), exist_ok=True)
    with open(out, "w") as f:
        json.dump({"generated_at": datetime.utcnow().isoformat(), "files": files}, f, indent=2)
    print(f"Wrote {len(files)} files to {out}")

if __name__ == "__main__":
    # usage: python generate_cdn_manifest.py --repo surrogate/datasets --date 2026-05-03 --out manifests/2026-05-03_files.json
    import argparse
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True)
    parser.add_argument("--date", required=True)
    parser.add_argument("--out", required=True)
    args = parser.parse_args()
    main(args.repo, args.date, args.out)
```

### 2) CDN-only Parquet streamer (Lightning-safe)

```python
# surrogate/data/cdn_parquet.py
import pyarrow.parquet as pq
import requests
from typing import Iterator, Dict, Any
import io

class IterableParquetCDN:
    """
    Streams rows from CDN-hosted parquet shards without HF API calls.
    Yields {prompt, response} only (projection at parse time).
    """
    def __init__(self, manifest_path: str, columns=("prompt", "response")):
        import json
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.files = [f["cdn_url"] for f in self.manifest["files"]]
        self.columns = columns

    def __iter__(self) -> Iterator[Dict[str, Any]]:
        for url in self.files:
            resp = requests.get(url, timeout=60)
            resp.raise_for_status()
            table = pq.read_table(io.BytesIO(resp.content), columns=self.columns)
            for batch in table.to_batches():
                df = batch.to_pandas()
                for _, row in df.iterrows():
                    yield {k: row[k] for k in self.columns}
```

```python
import os
import time
from lightning import LightningWork, LightningFlow, LightningApp, Machine
from lightning.app import BuildConfig

MANIFEST_PATH = os.getenv("CDN_MANIFEST", "manifests/2026-05-03_files.json")
STUDIO_NAME = os.getenv("STUDIO_NAME", "surrogate-train-l40s")

def get_or_create_studio(name: str) -> LightningWork:
    teamspace = Teamspace()
    for s in teamspace.studios:
        if s.name == name and s.status == "running":
            print(f"Reusing running studio: {name}")
            return s
    print(f"Creating studio: {name}")
    return LightningWork(
        name=name,
        cloud_compute=Machine.L40S,
        cloud_build=BuildConfig(packages=["pyarrow", "requests"]),
    )

def ensure_running(studio: LightningWork, timeout=300):
    waited = 0
    while studio.status != "running":
        if studio.status == "stopped":
            studio.start(machine=Machine.L40S)
        time.sleep(10)
        waited += 10
        if waited > timeout:
            raise RuntimeError(f"Studio {studio.name} failed to start within {timeout}s")

class SurrogateTrainer(LightningFlow):

        super().__init__()
        self.studio = get_or_create_studio(STUDIO_NAME)
```
