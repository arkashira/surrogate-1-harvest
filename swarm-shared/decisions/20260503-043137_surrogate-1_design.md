# surrogate-1 / design

## Implementation Plan (≤2h)

**Highest-value improvement**: Replace fragile shell-based ingestion with a **manifest-driven, CDN-bypass Python worker** that eliminates HF API rate limits during training data loads and prevents mixed-schema CastErrors.

### Changes

1. **Add `bin/manifest.py`** — single Mac-side script that:
   - Uses HF API once (after rate-limit window) to `list_repo_tree` for a date folder
   - Produces `manifest.json` with CDN URLs and expected schema hints
   - Commits nothing; only lists paths

2. **Replace `bin/dataset-enrich.sh` with `bin/worker.py`** — GitHub Actions runner script that:
   - Accepts `SHARD_ID`, `MANIFEST_PATH`
   - Downloads only assigned shard files via CDN (`resolve/main/...`) with no Authorization header
   - Projects to `{prompt, response}` at parse time (avoids pyarrow CastError)
   - Dedups via central md5 store (existing `lib/dedup.py`)
   - Outputs `batches/public-merged/<date>/shard<N>-<HHMMSS>.jsonl`

3. **Update `.github/workflows/ingest.yml`** to:
   - Add `MANIFEST_PATH` input (default `manifest.json`)
   - Keep 16-shard matrix
   - Pass manifest to each worker

4. **Add `requirements.txt`** entries if missing: `requests`, `pyarrow`, `numpy`, `tqdm`

## Code Snippets

### 1. `bin/manifest.py` (run on Mac)

```python
#!/usr/bin/env python3
"""
Generate manifest for a date folder.
Usage:
  python bin/manifest.py axentx/surrogate-1-training-pairs 2026-05-03 > manifest.json
"""
import json
import sys
from datetime import datetime
from huggingface_hub import HfApi

CDN_ROOT = "https://huggingface.co/datasets"

def main(repo: str, date_folder: str):
    api = HfApi()
    # Single API call; use recursive=False per folder to avoid 429
    tree = api.list_repo_tree(repo=repo, path=date_folder, recursive=False)
    files = [f.rfilename for f in tree if f.type == "file"]

    manifest = {
        "repo": repo,
        "date": date_folder,
        "generated_at": datetime.utcnow().isoformat() + "Z",
        "files": [
            {
                "path": f,
                "cdn_url": f"{CDN_ROOT}/{repo}/resolve/main/{f}",
                "shard_hint": hash(f) % 16  # deterministic shard assignment
            }
            for f in files
        ]
    }
    json.dump(manifest, sys.stdout, indent=2)

if __name__ == "__main__":
    if len(sys.argv) != 3:
        print("Usage: manifest.py <repo> <date_folder>", file=sys.stderr)
        sys.exit(1)
    main(sys.argv[1], sys.argv[2])
```

### 2. `bin/worker.py` (replaces `dataset-enrich.sh`)

```python
#!/usr/bin/env python3
"""
GitHub Actions worker for one shard.
Env:
  SHARD_ID=0..15
  MANIFEST_PATH=manifest.json
  HF_TOKEN=...
  DATE=2026-05-03
"""
import json
import os
import sys
import hashlib
import requests
import pyarrow as pa
import pyarrow.parquet as pq
from pathlib import Path

sys.path.insert(0, str(Path(__file__).parent))
from lib.dedup import DedupStore  # central md5 store

CDN_TIMEOUT = 30
CHUNK_SIZE = 8 * 1024 * 1024  # 8MB

def project_to_pair(raw_bytes, ext):
    """
    Project heterogeneous file to {prompt, response}.
    Extend per known schema.
    """
    ext = ext.lower()
    if ext == ".jsonl":
        # simplest: expect lines with prompt/response fields
        import json as _json
        out = []
        for line in raw_bytes.decode("utf-8").strip().split("\n"):
            if not line.strip():
                continue
            obj = _json.loads(line)
            out.append({
                "prompt": obj.get("prompt") or obj.get("input") or "",
                "response": obj.get("response") or obj.get("output") or "",
            })
        return out
    elif ext == ".parquet":
        tbl = pq.read_table(pa.BufferReader(raw_bytes))
        # project only these columns; ignore others to avoid CastError
        prompt_col = next((c for c in tbl.column_names if c in ("prompt", "input")), None)
        response_col = next((c for c in tbl.column_names if c in ("response", "output")), None)
        pairs = []
        for i in range(tbl.num_rows):
            pairs.append({
                "prompt": str(tbl[prompt_col][i].as_py()) if prompt_col else "",
                "response": str(tbl[response_col][i].as_py()) if response_col else "",
            })
        return pairs
    else:
        # fallback: treat as raw text prompt/response split by \n\n
        text = raw_bytes.decode("utf-8", errors="replace").strip()
        parts = [p.strip() for p in text.split("\n\n") if p.strip()]
        if len(parts) >= 2:
            return [{"prompt": parts[0], "response": parts[1]}]
        return []

def download_cdn(url):
    # No Authorization header -> bypasses /api/ rate limit
    resp = requests.get(url, timeout=CDN_TIMEOUT, stream=True)
    resp.raise_for_status()
    chunks = []
    for chunk in resp.iter_content(chunk_size=CHUNK_SIZE):
        chunks.append(chunk)
    return b"".join(chunks)

def main():
    shard_id = int(os.environ.get("SHARD_ID", 0))
    manifest_path = os.environ.get("MANIFEST_PATH", "manifest.json")
    date = os.environ.get("DATE", "2026-05-03")

    with open(manifest_path) as f:
        manifest = json.load(f)

    repo = manifest["repo"]
    files = [f for f in manifest["files"] if f["shard_hint"] == shard_id]

    out_dir = Path(f"batches/public-merged/{date}")
    out_dir.mkdir(parents=True, exist_ok=True)
    ts = datetime.utcnow().strftime("%H%M%S")
    out_path = out_dir / f"shard{shard_id}-{ts}.jsonl"

    store = DedupStore()
    written = 0

    with out_path.open("w", encoding="utf-8") as out_f:
        for item in files:
            try:
                data = download_cdn(item["cdn_url"])
                pairs = project_to_pair(data, Path(item["path"]).suffix)
                for pair in pairs:
                    prompt = (pair.get("prompt") or "").strip()
                    response = (pair.get("response") or "").strip()
                    if not prompt or not response:
                        continue
                    # dedup by content hash
                    md5 = hashlib.md5(f"{prompt}\n{response}".encode()).hexdigest()
                    if store.seen(md5):
                        continue
                    store.add(md5)
                    out_f.write(json.dumps({"prompt": prompt, "response": response}, ensure_ascii=False))
                    out_f.write("\n")
                    written += 1
            except Exception as exc:
                print(f"ERROR processing {item['path']}: {exc}", file=sys.stderr)

    print(f"Shard {shard_id}: wrote {written} pairs to {out_path}")

if __name__ == "__main__":
    main()
```
