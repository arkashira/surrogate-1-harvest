# workio / backend

## Final Synthesized Implementation

### Diagnosis (Consolidated)
- **Missing idempotency**: duplicate LINE events (retries, double-taps) corrupt attendance state and daily totals.
- **Weak signature enforcement**: no strict time-window enables replay attacks.
- **Non-transactional writes**: partial failures leave clock/leave/OT state inconsistent.
- **No durable deduplication**: restarts/horizontal scaling lose in-memory guards; duplicate side-effects (notifications, payroll entries) occur.
- **Insufficient input validation**: malformed events can crash handlers or bypass checks.

### 2. Idempotency + Signature Enforcement (Middleware)

`server/src/middleware/lineIdempotency.ts`
```ts
import { Request, Response, NextFunction } from 'express';
import crypto from 'crypto';
import { Pool } from 'pg';

const pool = new Pool({ connectionString: process.env.DATABASE_URL });
const SIGNATURE_TTL_MS = 5 * 60 * 1000; // 5 minutes

function nowMs() {
  return Date.now();
}

export async function lineSignatureCheck(
  req: Request,
  res: Response,
  next: NextFunction
) {
  const signature = req.get('X-Line-Signature');
  const timestamp = req.get('X-Line-Timestamp');
  const rawBody = JSON.stringify(req.body);

  if (!signature || !timestamp) {
    return res.status(400).json({ error: 'Missing LINE signature headers' });
  }

  const tsNum = Number(timestamp);
  if (Number.isNaN(tsNum) || Math.abs(nowMs() - tsNum * 1000) > SIGNATURE_TTL_MS) {
    return res.status(400).json({ error: 'Invalid or expired LINE timestamp' });
  }

  const expectedSig = crypto
    .createHmac('sha256', process.env.LINE_CHANNEL_SECRET || '')
    .update(rawBody, 'utf8')
    .digest('base64');

  // Constant-time-ish comparison to avoid timing hints
  if (!crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expectedSig))) {
    return res.status(401).json({ error: 'Invalid LINE signature' });
  }

  const events = req.body?.events;
  if (!Array.isArray(events) || events.length === 0) {
    return res.status(400).json({ error: 'No events in payload' });
  }

  // Attach validated metadata
  (req as any).lineMeta = { events, rawBody };
  next();
}

export async function deduplicateLineEvents(
  req: Request,
  _res: Response,
  next: NextFunction
) {
  const events = (req as any).lineMeta?.events;
  if (!Array.isArray(events) || events.length === 0) return next();

  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    const keep: any[] = [];
    for (const ev of events) {
      // Prefer stable event identifier from LINE; fallback to deterministic composite
      const eventId = ev?.webhookEventId || ev?.id || [
        ev?.source?.userId,
        ev?.type,
        ev?.timestamp,
        ev?.replyToken,
        ev?.mode
      ].filter(Boolean).join(':');

      if (!eventId) continue;

      const payloadHash = crypto
        .createHash('sha256')
        .update(JSON.stringify(ev))
        .digest('hex');

      // Check recent replays within TTL window
      const recent = await client.query(
        `SELECT 1 FROM line_event_replays
         WHERE event_id = $1 AND received_at > NOW() - INTERVAL '${SIGNATURE_TTL_MS}ms'`,
        [eventId]
      );

      if (recent.rowCount && recent.rowCount > 0) {
        // Duplicate within validity window: skip processing
        continue;
      }

      // Insert guard; on conflict keep newer received_at/payload_hash
      await client.query(
        `INSERT INTO line_event_replays (event_id, payload_hash, tenant_id, received_at)
         VALUES ($1, $2, $3, NOW())
         ON CONFLICT (event_id) DO UPDATE
         SET received_at = EXCLUDED.received_at,
             payload_hash = EXCLUDED.payload_hash,
             processed_at = NULL`,
        [eventId, payloadHash, (req as any).tenantId || 'unknown']
      );

      keep.push(ev);
    }

    (req as any).lineMeta.events = keep;
    await client.query('COMMIT');
    next();
  } catch (err) {
    await client.query('ROLLBACK');
    next(err);
  } finally {
    client.release();
  }
}
```

import { Pool, PoolClient } from 'pg';
const pool = new Pool({ connectionString: process.env.DATABASE_URL });

export async function upsertClockEvent(
  userId: string,
  tenantId: string,
  clockIn: Date,
  location?: { lat: number; lng: number }
) {
  const client: PoolClient = await pool.connect();

    await client.query('BEGIN');

    const day = clockIn.toISOString().slice(0, 10);

    await client.query(
      `SELECT id FROM daily_attendance
       WHERE user_id = $1 AND tenant_id = $2 AND day = $3
       FOR UPDATE`,
      [userId, tenantId, day]
    );

    await client.query(
      `INSERT INTO clock_events (user_id, tenant_id, clock_in, location)

       ON CONFLICT (user_id, tenant_id, DATE(clock_in))
