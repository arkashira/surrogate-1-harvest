# workio / quality

## 1. Diagnosis

- **Missing idempotency key on LINE webhook**: retries and at-least-once delivery create duplicate punch/leave/OT rows because no deduplication guard exists.
- **Non-atomic punch state transitions**: `find-then-insert/update` allows concurrent webhook deliveries to open multiple punches for the same user/date.
- **No transactional boundary**: punch writes and LINE reply are not atomic; partial failures leave DB inconsistent and user sees no confirmation.
- **No request fingerprinting**: duplicate detection relies only on `userId+type+date` which is too coarse and blocks legitimate same-day corrections.
- **No retry-safe reply**: LINE reply can be sent after DB commit but before client ack, causing user-visible duplicates when webhook is retried.

## 2. Proposed change

File: `/opt/axentx/workio/server/src/routes/line/webhook.ts`  
Scope: add idempotency middleware + atomic punch transition using `iid` (idempotency key) and row-level locking.

### 3.2 Idempotency helper

```ts
// server/src/lib/idempotency.ts
import { PoolClient } from 'pg';

export async function ensureIdempotent(
  client: PoolClient,
  iid: string,
  userId: number,
  entityType: string,
  entityId: number
): Promise<boolean> {
  const res = await client.query(
    `INSERT INTO idempotency_keys(iid, user_id, entity_type, entity_id)
     VALUES($1,$2,$3,$4)
     ON CONFLICT (iid) DO NOTHING
     RETURNING iid`,
    [iid, userId, entityType, entityId]
  );
  return res.rowCount === 1;
}
```

### 3.3 Update webhook route (atomic punch transition)

```ts
// server/src/routes/line/webhook.ts
import { Router } from 'express';
import { pool } from '../../db';
import { ensureIdempotent } from '../../lib/idempotency';

const router = Router();

router.post('/webhook/line', async (req, res) => {
  const events = req.body.events;
  if (!events) return res.sendStatus(200);

  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    for (const ev of events) {
      if (ev.type !== 'message' || ev.message.type !== 'text') continue;

      const userId = ev.source.userId; // LINE user id -> map to internal user_id via your user_line_mapping
      const text = ev.message.text.trim();
      const iid = ev.deliveryContext?.isRedelivery
        ? ev.webhookEventId // use LINE's event id as idempotency key for redeliveries
        : `${userId}:${Date.now()}:${text}`;

      // Map LINE userId -> internal user_id (example)
      const { rows: [userRow] } = await client.query(
        `SELECT id FROM users WHERE line_user_id = $1`,
        [userId]
      );
      if (!userRow) continue;
      const internalUserId = userRow.id;

      // Clock in/out logic with row-level lock
      const today = new Date().toISOString().split('T')[0];
      const { rows: openPunches } = await client.query(
        `SELECT id FROM punches
         WHERE user_id = $1 AND date = $2 AND clock_out IS NULL
         FOR UPDATE`,
        [internalUserId, today]
      );

      let entityId: number;
      if (openPunches.length === 0) {
        // clock in
        const { rows: inserted } = await client.query(
          `INSERT INTO punches(user_id, date, clock_in, location)
           VALUES($1,$2,NOW(),$3)
           RETURNING id`,
          [internalUserId, today, text] // text used as location stub; replace with GPS if available
        );
        entityId = inserted[0].id;
      } else {
        // clock out
        const punchId = openPunches[0].id;
        const { rows: updated } = await client.query(
          `UPDATE punches SET clock_out = NOW()
           WHERE id = $1
           RETURNING id`,
          [punchId]
        );
        entityId = updated[0].id;
      }

      // Idempotency guard
      const ok = await ensureIdempotent(client, iid, internalUserId, 'punch', entityId);
      if (!ok) {
        // duplicate delivery — skip side effects but still reply OK to LINE
        continue;
      }

      // Optionally send LINE reply (non-blocking)
      // await replyLine(ev.replyToken, `Punch recorded: ${entityId}`);
    }

    await client.query('COMMIT');
    res.sendStatus(200);
  } catch (err) {
    await client.query('ROLLBACK');
    console.error('Webhook error', err);
    res.status(500).json({ error: 'internal' });
  } finally {
    client.release();
  }
});

export default router;
```

### 3.4 Ensure LINE signature validation (quality hardening)

```ts
// server/src/middleware/line-signature.ts
import crypto from 'crypto';

export function verifyLineSignature(rawBody: Buffer, signature: string, channelSecret: string): boolean {
  const expected = crypto
    .createHmac('sha256', channelSecret)
    .update(rawBody)
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}
```

Wire into route:
```ts
router.post(
  '/webhook/line',
  (req, res, next) => {

      return res.status(401).json({ error: 'invalid-signature' });
    }
```
