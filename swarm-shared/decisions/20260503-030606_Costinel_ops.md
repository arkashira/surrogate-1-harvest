# Costinel / ops

## Final Implementation Plan — Top-Hub Signal Panel (CDN-first, <2h)

### What we ship
- A **non-blocking Top-Hub Signal Panel** mounted in the Costinel dashboard (sidebar or top banner).
- Defaults to hub **MOC** (configurable via `VITE_HUB_NAME`).
- Shows: hub title, short description, top 3 signals (anomalies/recommendations), last updated timestamp.
- **CDN-first data strategy**: file list and signal payloads are pre-built on Mac and deployed to CDN; **zero runtime HF API calls** to avoid 429/rate-limits.
- **Non-blocking & resilient**: panel gracefully degrades if CDN unavailable; never blocks dashboard rendering.

---

2. `src/composables/useHubSignals.ts` — CDN-first fetch logic with caching + fallback.

## Code Snippets

### 1. Orchestration script (Mac) — `scripts/orchestrator/generate-hub-filelist.js`
```js
#!/usr/bin/env node
/**
 * Generate hub filelist and minimal signals manifest for CDN-first loading.
 * Usage: node scripts/orchestrator/generate-hub-filelist.js MOC
 *
 * Prereqs: install huggingface_hub (pip install huggingface_hub) if using Python helper.
 * This script uses the HF REST API to avoid runtime deps during build.
 */

const fs = require('fs');
const path = require('path');
const { execSync } = require('child_process');

const REPO = 'axentx/costinel-hubs';
const HUB_NAME = process.argv[2] || 'MOC';
const OUT_DIR = path.resolve(__dirname, '../../public/data');
const FILELIST_OUT = path.join(OUT_DIR, 'hub-files.json');
const SIGNALS_OUT = path.join(OUT_DIR, 'hubs', HUB_NAME, 'signals.json');

function runPython(pythonCode) {
  try {
    return execSync(`python3 -c "${pythonCode.replace(/"/g, '\\"')}"`, { encoding: 'utf8' }).trim();
  } catch (e) {
    console.error('Python helper failed:', e.message);
    return null;
  }
}

function fetchTree(hub) {
  // Use huggingface_hub via Python for reliable pagination/tree listing
  const code = `
import json, os
from huggingface_hub import list_repo_tree
tree = list_repo_tree(repo="${REPO}", path="${hub}", recursive=False)
files = [f.rfilename for f in tree if f.type == "file"]
print(json.dumps(files))
  `;
  const out = runPython(code);
  if (!out) return [];
  try {
    return JSON.parse(out);
  } catch {
    return [];
  }
}

function buildSignalsFromFiles(files, hub) {
  // Heuristic: pick up to 3 files that look like signal/anomaly/recommendation
  const signalFiles = files.filter(f =>
    /signal|anomaly|recommend|alert|issue/i.test(f)
  ).slice(0, 3);

  const baseCDN = `https://huggingface.co/datasets/${REPO}/resolve/main/${hub}`;

  return signalFiles.map((f, i) => ({
    id: `${hub}-${i}`,
    title: f.replace(/\.[^/.]+$/, '').replace(/[_-]/g, ' '),
    severity: i === 0 ? 'critical' : (i === 1 ? 'warning' : 'info'),
    snippet: `Detected condition in ${f}. Review recommended.`,
    cta: { label: 'Open file', href: `${baseCDN}/${f}` },
    sourceFile: f
  }));
}

function main() {
  console.log(`Generating hub manifest for ${HUB_NAME}...`);

  const files = fetchTree(HUB_NAME);
  if (!files.length) {
    console.warn('No files found for hub:', HUB_NAME);
  }

  const hubFilesManifest = {
    hub: HUB_NAME,
    generated_at: new Date().toISOString(),
    files,
    cdn_base: `https://huggingface.co/datasets/${REPO}/resolve/main/${HUB_NAME}`
  };

  // Ensure dirs
  fs.mkdirSync(path.dirname(FILELIST_OUT), { recursive: true });
  fs.mkdirSync(path.dirname(SIGNALS_OUT), { recursive: true });

  // Write filelist (for repo reference)
  fs.writeFileSync(FILELIST_OUT, JSON.stringify(hubFilesManifest, null, 2));
  console.log('Saved', FILELIST_OUT);

  // Build and write signals payload (CDN)
  const signals = buildSignalsFromFiles(files, HUB_NAME);
  const signalsManifest = {
    hub: HUB_NAME,
    generated_at: new Date().toISOString(),
    signals,
    cdn_base: hubFilesManifest.cdn_base
  };

  fs.writeFileSync(SIGNALS_OUT, JSON.stringify(signalsManifest, null, 2));
  console.log('Saved', SIGNALS_OUT);
}

if (require.main === module) {
  main();
}
```

