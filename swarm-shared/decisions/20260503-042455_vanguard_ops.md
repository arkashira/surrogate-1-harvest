# vanguard / ops

## 1. Diagnosis

- No CDN-only ingestion path: training scripts still rely on `load_dataset`/`list_repo_files` which trigger HF API 429s and non-reproducible epochs.
- Missing content-addressed manifest (file list + SHA256) prevents deterministic CDN fetches and safe resume/restart.
- No Lightning Studio reuse logic: each run risks quota waste and idle-timeout deaths.
- No surrogate-1 schema guard: heterogeneous repo files can still hit `pyarrow.CastError` during ingestion.
- No ops wrapper shebang/shell hygiene: cron-launched scripts may fail with exec errors (seen in opus/active-learning patterns).

## 2. Proposed change

Add a small, high-leverage ops utility that fixes ingestion reliability and training quota in one shot:

- File: `/opt/axentx/vanguard/ops/ingest_manifest.py` (new)
- Scope:
  - Single-call `list_repo_tree` (non-recursive) for one date folder → JSON manifest with `path` and `sha256`.
  - CDN-only download helper using `https://huggingface.co/datasets/{repo}/resolve/main/{path}` (no auth).
  - Lightweight schema projection: keep only `{prompt,response}` fields; drop `source`, `ts`; move attribution to filename pattern `batches/mirror-merged/{date}/{slug}.parquet`.
  - Lightning Studio reuse + idle check before any training launch.
  - Bash shebang + `chmod +x`; safe for cron (`SHELL=/bin/bash`).

## 3. Implementation

Create `/opt/axentx/vanguard/ops/ingest_manifest.py`:

#!/usr/bin/env python3
"""
Generate a content-addressed manifest for a HuggingFace dataset folder
and provide CDN-only download + schema projection for surrogate-1.

Usage (cron-safe):
  SHELL=/bin/bash
  0 2 * * * cd /opt/axentx/vanguard && /usr/bin/env bash -c 'source .venv/bin/activate && python ops/ingest_manifest.py --repo org/surrogate-1 --date 2026-04-29 --out manifests/2026-04-29.json --mirror batches/mirror-merged/2026-04-29'
"""

from __future__ import annotations

import argparse
import hashlib
import json
import os
import sys
import time
from pathlib import Path
from typing import Any, Dict, List

import requests
from huggingface_hub import HfApi, list_repo_tree

# ---------- config ----------
HF_API_RATE_LIMIT_WAIT = 360  # seconds after 429
CDN_BASE = "https://huggingface.co/datasets"

# ---------- helpers ----------
def sha256_bytes(b: bytes) -> str:
    return hashlib.sha256(b).hexdigest()

def robust_get(url: str, max_retries: int = 5, backoff: float = 1.5) -> bytes:
    for attempt in range(1, max_retries + 1):
        try:
            resp = requests.get(url, timeout=30)
            if resp.status_code == 429:
                wait = HF_API_RATE_LIMIT_WAIT
                print(f"[rate-limited] waiting {wait}s", file=sys.stderr)
                time.sleep(wait)
                continue
            resp.raise_for_status()
            return resp.content
        except Exception as exc:
            if attempt == max_retries:
                raise
            sleep = backoff**attempt
            print(f"[retry {attempt}/{max_retries}] {exc} -> sleep {sleep:.1f}s", file=sys.stderr)
            time.sleep(sleep)
    raise RuntimeError("unreachable")

def project_to_prompt_response(raw: Dict[str, Any]) -> Dict[str, str]:
    """
    Surrogate-1 schema guard: keep only prompt/response.
    Accepts common key variants.
    """
    prompt = raw.get("prompt") or raw.get("instruction") or raw.get("input") or raw.get("question") or ""
    response = raw.get("response") or raw.get("output") or raw.get("answer") or raw.get("completion") or ""
    return {"prompt": str(prompt), "response": str(response)}

# ---------- manifest ----------
def build_manifest(repo: str, date_folder: str) -> List[Dict[str, str]]:
    """
    Single API call: list_repo_tree(non-recursive) for date folder.
    Returns [{"path": "...", "sha256": "..."}, ...]
    """
    api = HfApi()
    try:
        items = list_repo_tree(repo=repo, path=date_folder, recursive=False)
    except Exception as exc:
        # If date_folder is empty, list root
        items = list_repo_tree(repo=repo, path="", recursive=False)
        # Filter by prefix
        items = [it for it in items if it.rfilename.startswith(f"{date_folder}/")]

    manifest = []
    for it in items:
        if it.type != "file":
            continue
        manifest.append({"path": it.rfilename, "sha256": "pending"})
    return manifest

# ---------- CDN download + projection ----------
def mirror_to_parquet(
    repo: str,
    manifest: List[Dict[str, str]],
    out_dir: Path,
    max_files: int | None = None
) -> None:
    out_dir.mkdir(parents=True, exist_ok=True)
    from datasets import Dataset

    records = []
    count = 0
    for entry in manifest:
        if max_files is not None and count >= max_files:
            break
        path = entry["path"]
        url = f"{CDN_BASE}/{repo}/resolve/main/{path}"
        try:
            raw_bytes = robust_get(url)
            # naive JSONL assumption (common for mirror); fallback to json
            text = raw_bytes.decode("utf-8")
            for line in text.strip().splitlines():
                line = line.strip()
                if not line:
                    continue
                try:
                    obj = json.loads(line)
                except json.JSONDecodeError:
                    continue
                projected = project_to_prompt_response(obj)
                if projected["prompt"] and projected["response"]:
                    records.append(projected)
            count += 1
        except Exception as exc:
            print(f"[skip] {path} -> {exc}", file=sys.stderr)
            continue

    if records:
        ds = Dataset.from_list(records)
        slug = out_dir.name  # e.g. 2026-04-29
        parquet_path = out_dir.parent / f"{slug}.parquet"
        ds.to_parquet(parquet_path)
        print(f"[mirror] wrote {len(records)} rows -> {parquet_path}")
    else:
        print("[mirror] no valid records", file=sys.stderr)

# ---------- Lightning Studio reuse ----------
def reuse_or_start_studio(
    name: str = "vanguard-surrogate-train",
    machine: str = "L40S",
    target_script: str = "train_surrogate.py"
) -> None:
    try:
        from lightning import Studio, Machine, Teamspace
    except ImportError:
        print("[lightning] SDK not installed — skipping studio reuse", file=sys.stderr)
        return

    teamspace = Teamspace()
    running = None
    for s in teamspace.studios:
        if s.name == name and s.status == "Running":
            running = s
            break

    if running:
        print(f"[lightning] reusing running studio: {running.id}")
        studio = running
    else:
        print(f"[lightning] starting studio: {name}")
        studio = Studio(
            name=name,
            machine=Machine(machine),
            create_ok=True,
        )

    # Guard against idle stop
    if studio.status != "Running":
        print(f"[lightning] studio stopped — restarting")
        studio.start(machine=Machine(machine))

    # Lightweight run; adjust as needed
    studio.run(
        [sys.executable, target_script],
        cwd=str(Path.cwd()),
    )
    print("[lightning] run submitted")
