### 1) Pre-cache file list (Mac orchestration only)
```python
# scripts/pre-cache-filelist.py
#!/usr/bin/env python3
"""
Run once (or after rate-limit window clears) on Mac.
Produces filelist.json for a single date folder.
"""
import json
import os
import sys
from datetime import date
from huggingface_hub import HfApi

REPO_ID = os.getenv("HF_DATASET_REPO", "my-org/surrogate-dataset")
DATE_FOLDER = os.getenv("DATE_FOLDER", str(date.today()))  # e.g. "2026-04-29"
OUTPUT = os.getenv("OUTPUT", "filelist.json")

def main() -> None:
    api = HfApi()
    # Non-recursive, one folder — cheap, avoids 429 from recursive pagination.
    tree = api.list_repo_tree(repo_id=REPO_ID, path=DATE_FOLDER, recursive=False)
    files = [item.rfilename for item in tree if item.rfilename.endswith((".parquet", ".jsonl"))]
    payload = {
        "repo_id": REPO_ID,
        "date_folder": DATE_FOLDER,
        "files": files,
    }
    with open(OUTPUT, "w") as f:
        json.dump(payload, f, indent=2)
    print(f"Wrote {len(files)} files to {OUTPUT}")

if __name__ == "__main__":
    main()
```

### 2) Training script: CDN-only fetches
```python
# surrogate/train.py  (excerpt to integrate)
import json
import os
import requests
from pathlib import Path
from torch.utils.data import IterableDataset

HF_CDN_TEMPLATE = "https://huggingface.co/datasets/{repo_id}/resolve/main/{path}"

class CDNParquetDataset(IterableDataset):
    def __init__(self, filelist_path: str, repo_id: str, cache_dir: str = ".cache"):
        super().__init__()
        self.repo_id = repo_id
        self.cache_dir = Path(cache_dir)
        self.cache_dir.mkdir(parents=True, exist_ok=True)

        with open(filelist_path) as f:
            manifest = json.load(f)
        self.files = manifest["files"]

    def _download_cdn(self, rel_path: str) -> Path:
        out = self.cache_dir / Path(rel_path).name
        if out.exists():
            return out
        url = HF_CDN_TEMPLATE.format(repo_id=self.repo_id, path=rel_path)
        # No Authorization header -> bypasses /api/ rate limits; uses CDN tier.
        r = requests.get(url, timeout=60)
        r.raise_for_status()
        out.write_bytes(r.content)
        return out

    def __iter__(self):
        import pyarrow.parquet as pq
        for rel in self.files:
            local = self._download_cdn(rel)
            table = pq.read_table(local)
            # Project to {prompt, response} only at parse time (schema-safe).
            for batch in table.to_batches(columns=["prompt", "response"]):
                for i in range(batch.num_rows):
                    yield {
                        "prompt": batch["prompt"][i].as_py(),
                        "response": batch["response"][i].as_py(),
                    }
```

Usage:
```bash
python surrogate/train.py --filelist filelist.json --repo_id my-org/surrogate-dataset
```

### 3) Ensure Lightning Studio is running (restart if stopped)
```python
# scripts/ensure-studio.py
#!/usr/bin/env python3
"""
Check Teamspace for a running studio by name.
If stopped, restart on L40S (free tier fallback).
"""
import os
import sys
import time

from lightning.app import Studio, Teamspace
from lightning.app.core.constants import LightningMachineType

STUDIO_NAME = os.getenv("STUDIO_NAME", "surrogate-train")
MACHINE = LightningMachineType.L40S  # free tier falls to L40S; paid can use H200 in lightning-lambda-prod

def ensure_studio() -> Studio:
    teamspace = Teamspace()
    for s in teamspace.studios:
        if s.name == STUDIO_NAME:
            if s.status == "running":
                print(f"Studio '{STUDIO_NAME}' is running. Reusing.")
                return s
            else:
                print(f"Studio '{STUDIO_NAME}' is stopped. Restarting on {MACHINE}.")
                s.start(machine=MACHINE)
                # Poll briefly
                for _ in range(10):
                    time.sleep(6)
                    s.refresh()
                    if s.status == "running":
                        print("Studio is running.")
                        return s
                raise RuntimeError("Studio failed to start in time.")
    # Not found: create
    print(f"Creating studio '{STUDIO_NAME}' on {MACHINE}.")
    return Studio(name=STUDIO_NAME, create_ok=True, machine=MACHINE)

if __name__ == "__main__":
    studio = ensure_studio()
    print(f"Ready: {studio.name} ({studio.status})")
```

### 4) Lightweight run wrapper (cron-friendly)
```bash
#!/bin/bash
# scripts/run-iteration.sh
# Usage: bash scripts/run-iteration.sh
# Cron note: set SHELL=/bin/bash in crontab and invoke via bash.

set -euo pipefail
export SHELL=/bin/bash

cd "$(dirname "$0")/.."  # repo root

# 1) Pre-cache file list (idempotent; can be skipped if filelist.json is fresh)
if [[ ! -f filelist.json ]]; then
  echo "Pre-caching file list..."
  python3 scripts/pre-cache-filelist.py
fi

# 2) Ensure studio is running
echo "Ensuring Lightning Studio..."
python3 scripts/ensure-studio.py

# 3) Run training (Lightning SDK launcher) — uses CDN-only dataset
echo "Starting training iteration..."
python3 surrogate/train.py \
  --filelist filelist.json \
  --repo_id "${HF_DATASET_REPO:-my-org/surrogate-dataset}" \
  --max-steps 500

echo "Iteration complete."
```

Make executable:
```bash
chmod +x scripts/run-iteration.sh
```
