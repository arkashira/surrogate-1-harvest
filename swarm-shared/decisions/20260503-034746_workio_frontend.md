# workio / frontend

## 1. Diagnosis
- No real-time subscription layer: UI relies on local React state; LINE webhook–driven changes (clock/leave/OT) are invisible until hard refresh → perceived latency and silent failures.
- Multi-tab/device divergence: actions performed in one tab do not propagate to others; users see stale data and can double-submit.
- Missing optimistic UI + rollback: clock/leave/OT actions block on network with no instant feedback; errors leave UI stuck and require manual recovery.
- No idempotency keys on client requests: retries (e.g., flaky networks) can create duplicate punches or duplicate leave/OT requests.
- Polling-only dashboard/reports: periodic fetches waste bandwidth and delay updates; no server push for time-sensitive events.

## 2. Proposed change
Add a lightweight real-time layer using EventSource (SSE) + optimistic UI + idempotency keys. Scope:
- Frontend: `src/lib/api.ts` (add SSE client + idempotency helper), `src/hooks/useRealtime.ts` (subscribe + optimistic queue), update relevant UI components (clock/leave/OT) to use optimistic mutations.
- Backend: expose `/sse` endpoint (or reuse existing WebSocket/line webhook broadcast) — if not present, start with a minimal Node/Express route that streams entity changes (clock, leave, OT) for the tenant/user.

### src/lib/api.ts

```ts
// Generate stable idempotency key for mutations
export function makeIdempotencyKey(input: { action: string; entityId?: string; timestamp?: number }): string {
  const payload = `${input.action}:${input.entityId || ''}:${input.timestamp || Date.now()}`;
  // lightweight deterministic hash (not crypto)
  let hash = 0;
  for (let i = 0; i < payload.length; i++) {
    const chr = payload.charCodeAt(i);
    hash = ((hash << 5) - hash) + chr;
    hash |= 0;
  }
  return `wkio-${hash >>> 0}-${Date.now()}`;
}

// POST with idempotency header
export async function postWithIdempotency<T>(url: string, body: any, idempotencyKey?: string): Promise<T> {
  const key = idempotencyKey || makeIdempotencyKey({ action: url });
  const res = await fetch(url, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Idempotency-Key': key,
    },
    body: JSON.stringify(body),
  });
  if (!res.ok) throw new Error(`Request failed: ${res.status}`);
  return res.json();
}
```

### src/hooks/useRealtime.ts
```ts
import { useEffect, useRef } from 'react';

type Handler<T = any> = (data: T) => void;

export function useRealtime<T = any>(path: string, onMessage: Handler<T>, deps: any[] = []) {
  const handlersRef = useRef<Handler<T>[]>([]);
  const eventSourceRef = useRef<EventSource | null>(null);

  useEffect(() => {
    handlersRef.current = [onMessage];
  }, [onMessage, ...deps]);

  useEffect(() => {
    // Close existing
    if (eventSourceRef.current) {
      eventSourceRef.current.close();
      eventSourceRef.current = null;
    }

    const es = new EventSource(path, { withCredentials: true });
    eventSourceRef.current = es;

    es.onmessage = (ev) => {
      try {
        const payload = JSON.parse(ev.data);
        handlersRef.current.forEach((h) => {
          try { h(payload); } catch (e) { console.error('Realtime handler error', e); }
        });
      } catch (e) {
        console.error('Failed to parse realtime message', e);
      }
    };

    es.onerror = (ev) => {
      console.warn('SSE error, will reconnect', ev);
      // Let browser reconnect automatically; optionally implement backoff
    };

    return () => {
      es.close();
      if (eventSourceRef.current === es) eventSourceRef.current = null;
    };
  }, [path]);
}
```

### src/hooks/useOptimisticQueue.ts (simple optimistic queue)
```ts
import { useState, useCallback, useRef } from 'react';

type Mutation<TArgs, TResult> = (args: TArgs) => Promise<TResult>;

export function useOptimisticQueue<TArgs, TResult>(
  mutate: Mutation<TArgs, TResult>,
  optimisticApply: (state: any, args: TArgs) => any,
  optimisticRevert: (state: any, args: TArgs) => any
) {
  const [queue] = useState<Array<{ args: TArgs; id: string; resolve: (v: TResult) => void; reject: (e: any) => void }>>([]);
  const processing = useRef(false);

  const enqueue = useCallback((args: TArgs): Promise<TResult> => {
    return new Promise((resolve, reject) => {
      const id = crypto.randomUUID();
      queue.push({ args, id, resolve, reject });
      // optimistic apply to local state is caller's responsibility via optimisticApply
      if (!processing.current) process();
    });
  }, [queue]);

  async function process() {
    if (processing.current || queue.length === 0) return;
    processing.current = true;
    const item = queue[0];
    try {
      const result = await mutate(item.args);
      item.resolve(result);
    } catch (err) {
      item.reject(err);
      // caller should revert optimistic state using optimisticRevert
    } finally {
      queue.shift();
      processing.current = false;
      if (queue.length > 0) process();
    }
  }

  return { enqueue };
}
```

### Example usage in a clock-in component (src/components/ClockButton.tsx)
```tsx
import { useState } from 'react';
import { postWithIdempotency } from '../lib/api';
import { useRealtime } from '../hooks/useRealtime';
import { useOptimisticQueue } from '../hooks/useOptimisticQueue';

export function ClockButton({ tenantId, userId }: { tenantId: string; userId: string }) {
  const [lastPunch, setLastPunch] = useState<{ type: 'in' | 'out'; ts: string } | null>(null);

  // optimistic helpers
  const optimisticApply = (state: any, args: { type: 'in' | 'out' }) => {
    setLastPunch({ type: args.type, ts: new Date().toISOString() });
  };
  const optimisticRevert = (state: any, args: { type: 'in' | 'out' }) => {
    // revert to previous known server state (simplified)
    setLastPunch(null);
  };

  const { enqueue } = useOptimisticQueue(
    (args: { type: 'in' | 'out' }) =>
      postWithIdempotency(`/api/punch`, { tenantId, userId, type: args.type }),
    optimisticApply,
    optimisticRevert
  );

  // Subscribe to punches for this user
  useRealtime(`/sse/punches?userId=${userId}&tenantId=${tenantId}`, (msg) => {
    if (msg.userId === userId) {
      setLastPunch({ type: msg.type, ts: msg.ts });
    }
  });

  const handleClock = (type: 'in' | 'out') => {
    optimisticApply(null!, { type });
    enqueue({ type }).catch(() => {
      optimisticRevert(null!, { type });
      alert('打卡失敗，請重試');
    });
  };

  return (
    <div className="flex gap-4">
      <button onClick={() => handleClock('in')} className="btn btn-primary">上班打卡</button>
      <button onClick={() => handleClock('out')} className="btn btn-secondary">下班打卡</button>
      {lastPunch && <div className="text-sm text-gray-600">最近: {lastPunch.type} @ {new Date(lastPunch.ts).toLocaleTimeString()}</div>}
    </div>
  );
}
```
