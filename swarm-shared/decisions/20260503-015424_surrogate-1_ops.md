## Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Uses a **single `list_repo_tree` snapshot** (JSON manifest) generated once per date on the Mac orchestrator and committed to the repo (or passed via env).  
- Each GitHub Actions shard (0–15) deterministically picks its 1/16 slice from that manifest and downloads **only those files via HF CDN** (`resolve/main/...`) — zero API calls during training, bypasses 429 auth limits.  
- Projects heterogeneous schemas to `{prompt,response}` at parse time (no `load_dataset(streaming=True)`), dedups via central `lib/dedup.py` md5 store, and writes `batches/public-merged/<date>/shard<N>-<HHMMSS>.jsonl`.  
- Adds graceful idle handling and Studio reuse hooks for Lightning training (if invoked locally).

from lib.dedup import is_duplicate, mark_seen  # central md5 store

def shard_assign(slug: str, shard_id: int, shard_total: int) -> bool:
    """Deterministic shard assignment by slug hash."""
    h = int(hashlib.sha256(slug.encode("utf-8")).hexdigest(), 16)
    return (h % shard_total) == shard_id

def download_cdn(path: str, timeout: int = 30) -> bytes:
    url = f"{BASE_CDN}/{path}"
    resp = requests.get(url, timeout=timeout)
    resp.raise_for_status()
    return resp.content

def project_to_pair(data, fmt: str):
    """Project heterogeneous file to (prompt, response). Returns list of dicts."""
    pairs = []
    try:
        if fmt == "parquet":
            tbl = pq.read_table(pa.BufferReader(data))
            df = tbl.to_pandas()
        elif fmt == "jsonl":
            lines = data.decode("utf-8").strip().splitlines()
            df = pa.Table.from_pylist([json.loads(l) for l in lines if l.strip()]).to_pandas()
        elif fmt == "json":
            df = pa.Table.from_pylist(json.loads(data)).to_pandas()
        elif fmt == "csv":
            df = pa.Table.from_pandas(pa.csv.read_csv(pa.BufferReader(data))).to_pandas()
        else:
            return pairs

        # Normalize column names
        col_map = {c: c.strip().lower() for c in df.columns}
        df = df.rename(columns=col_map)

        prompt_col = None
        response_col = None
        for c in df.columns:
            if c in ("prompt", "question", "input", "instruction"):
                prompt_col = c
            if c in ("response", "answer", "output", "completion"):
                response_col = c

        if prompt_col is None or response_col is None:
            # Best-effort: first text col as prompt, second as response
            text_cols = [c for c in df.columns if df[c].dtype == "object"]
            if len(text_cols) >= 2:
                prompt_col, response_col = text_cols[0], text_cols[1]
            else:
                return pairs

        for _, row in df.iterrows():
            p = str(row[prompt_col]).strip()
            r = str(row[response_col]).strip()
            if not p or not r:
                continue
            # Build md5 for dedup
            raw = f"{p}\n{r}".encode("utf-8")
            md5 = hashlib.md5(raw).hexdigest()
            if is_duplicate(md5):
                continue
            mark_seen(md5)
            pairs.append({"prompt": p, "response": r})
    except Exception as exc:
        # Silently skip malformed files to avoid shard failure
        print(f"[WARN] project failed for {fmt}: {exc}", file=sys.stderr)
    return pairs

def main():
    shard_id = int(os.getenv("SHARD_ID", "0"))
    shard_total = int(os.getenv("SHARD_TOTAL", "16"))
    manifest_path = os.getenv("MANIFEST_JSON")
    date_str = os.getenv("DATE", datetime.utcnow().strftime("%Y-%m-%d"))

    if not manifest_path or not os.path.exists(manifest_path):
        print("[ERROR] MANIFEST_JSON not found", file=sys.stderr)
        sys.exit(1)

        manifest = json.load(f)  # list of relative paths

    out_dir.mkdir(parents=True, exist_ok=True)
    ts = datetime.utcnow().strftime("%H%M%S")
    out_file = out_dir / f"shard{shard_id}-{ts}.jsonl"
