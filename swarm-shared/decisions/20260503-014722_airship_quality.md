### 2.2 Deterministic sibling-repo selector
```python
# surrogate/ingest/shard.py
import hashlib

HF_REPO = "axentx/surrogate-ingest"
SIBLINGS = [f"{HF_REPO}-s{i}" for i in range(5)]

def pick_repo(slug: str) -> str:
    """Deterministic repo assignment from slug."""
    digest = hashlib.sha256(slug.encode()).hexdigest()
    idx = int(digest, 16) % len(SIBLINGS)
    return SIBLINGS[idx]
```

### 2.3 CDN-only downloader + projection
```python
# surrogate/data/cdn_loader.py
import os, hashlib
import pyarrow.parquet as pq
import pyarrow as pa
import requests
from pathlib import Path
from surrogate.ingest.shard import pick_repo

HF_REPO = os.getenv("HF_DATASET_REPO", "axentx/surrogate-ingest")

def cdn_url(path: str) -> str:
    return f"https://huggingface.co/datasets/{HF_REPO}/resolve/main/{path}"

def project_to_prompt_response(raw_path: str) -> pa.Table:
    """Download via CDN, keep only prompt/response."""
    url = cdn_url(raw_path)
    r = requests.get(url, timeout=30)
    r.raise_for_status()

    local_path = Path("/tmp") / Path(raw_path).name
    local_path.write_bytes(r.content)

    try:
        table = pq.read_table(local_path)
    except Exception:
        import pyarrow.json as pj
        table = pj.read_json(local_path)

    available = set(table.column_names)
    keep = [c for c in ("prompt", "response") if c in available]
    if not keep:
        raise ValueError(f"No prompt/response in {raw_path}")

    return table.select(keep)

def upload_sharded(date: str, slug: str, table: pa.Table) -> str:
    """Write to deterministic sibling repo."""
    from huggingface_hub import upload_file

    sibling = pick_repo(slug)
    out_dir = f"batches/mirror-merged/{date}"
    out_path = f"{out_dir}/{slug}.parquet"

    local_parquet = Path("/tmp") / f"{slug}.parquet"
    pq.write_table(table, local_parquet)

    upload_file(
        path_or_fileobj=str(local_parquet),
        path_in_repo=out_path,
        repo_id=sibling,
        repo_type="dataset",
    )
    return f"{sibling}/{out_path}"
```

### 2.4 Lightning training data module (CDN-only)
```python
# surrogate/train/data_module.py
import json
from torch.utils.data import Dataset, DataLoader
from surrogate.data.cdn_loader import project_to_prompt_response

class CDNTextDataset(Dataset):
    def __init__(self, file_list_path: str):
        with open(file_list_path) as f:
            self.files = json.load(f)

    def __len__(self):
        return len(self.files)

    def __getitem__(self, idx):
        table = project_to_prompt_response(self.files[idx])
        # For simplicity, take first row; extend collate_fn for batching rows.
        row = table.slice(0, 1).to_pydict()
        return {
            "prompt": row.get("prompt", [""])[0],
            "response": row.get("response", [""])[0],
        }

def build_dataloader(file_list_path: str, batch_size: int = 8, num_workers: int = 4) -> DataLoader:
    ds = CDNTextDataset(file_list_path)
    return DataLoader(ds, batch_size=batch_size, num_workers=num_workers, shuffle=True)
```

### 2.5 Lightning Studio launcher (reuse running studio)
```python
# surrogate/train/launch.py
from lightning import LightningFlow, LightningApp
from lightning.app import Teamspace
from lightning import Machine

class TrainFlow(LightningFlow):
    def __init__(self):
        super().__init__()
        self.studio = None

    def run(self):
        name = "surrogate-train-l40s"
        if self.studio is None or self.studio.status != "Running":
            for s in Teamspace.studios:
                if s.name == name and s.status == "Running":
                    self.studio = s
                    break
            else:
                self.studio = Teamspace.studios.create(
                    name=name,
                    machine=Machine.L40S,
                    cloud="lightning-public-prod",
                )
                self.studio.start(machine=Machine.L40S)
                return

        # Run training inside the studio
        self.studio.run(
            "python -m surrogate.train.fit --file-list batches/file-list-2026-05-03.json"
        )

app = LightningApp(TrainFlow())
```

## 3. Resolved contradictions / choices
- **Repo naming**: Use deterministic `{HF_REPO}-s0..4` pattern (from Candidate 1) but allow easy override via `SIBLINGS` list (from Candidate 2). This keeps behavior deterministic and configurable.
- **Hash function**: Prefer SHA-256 (Candidate 2) for lower collision risk; fallback to MD5 acceptable but SHA-256 is stronger.
- **Schema hygiene**: Explicitly drop all columns except `prompt`/`response` before upload (Candidate 1) to prevent `pyarrow.CastError`. This is non-negotiable for correctness.
- **Upload scope**: Use HF API only for uploads (writes) after sharding; CDN for all reads. This matches both proposals and eliminates 429s during training.
- **Lightning Studio**: Reuse a running studio if available; create only if missing. Avoids unnecessary spin-up costs and matches Candidate 1’s intent with Candidate 2’s clarity.
