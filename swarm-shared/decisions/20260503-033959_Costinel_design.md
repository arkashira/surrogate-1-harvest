# Costinel / design

## Final Implementation Plan — CDN-First Top-Hub Signal Panel (<2h)

**Chosen approach:**  
A non-blocking, CDN-first Top-Hub Signal Panel that surfaces the most-connected hub (e.g., "MOC") with **zero HuggingFace API calls at runtime**.

### 1. Architecture (CDN-first)
- **Build-time (Mac orchestration)**  
  - Single `list_repo_tree` call to `knowledge-rag/hubs/<date>` → lightweight scoring  
  - Compute top-hub by `inLinks` (edge-weight) → write `top-hub.json`  
  - Commit to repo (or upload to CDN) → versioned by date slug  
- **Runtime (Costinel dashboard)**  
  - Dashboard fetches `https://huggingface.co/datasets/{repo}/resolve/main/signals/top-hub.json` (no auth, no API, no rate limit)  
  - Hydrates a lightweight signal card with hub name, score, and link to full hub page  
  - Graceful fallback if fetch fails or stale (>7d)  
- **No runtime HF API** — satisfies CDN-bypass pattern and quota constraints.

### 3. Build script — `scripts/build-top-hub-signal.js`

```js
#!/usr/bin/env node
/**
 * Build-time script (run on Mac) to produce top-hub.json
 * Uses HF REST API minimally (one list_repo_tree) then CDN-only at runtime.
 *
 * Usage:
 *   HUGGING_FACE_TOKEN=hf_xxx node scripts/build-top-hub-signal.js \
 *     --repo knowledge-rag/hubs \
 *     --date 2026-04-27 \
 *     --out public/data/top-hub.json
 */

import { writeFileSync } from 'fs';
import { resolve } from 'path';
import { program } from 'commander';
import fetch from 'node-fetch';

program
  .requiredOption('--repo <repo>', 'HF repo (e.g. knowledge-rag/hubs)')
  .requiredOption('--date <YYYY-MM-DD>', 'Date folder')
  .option('--out <path>', 'Output JSON path', 'public/data/top-hub.json')
  .option('--token <token>', 'HF token (or env HUGGING_FACE_TOKEN)');

program.parse();
const opts = program.opts();
const token = opts.token || process.env.HUGGING_FACE_TOKEN;

if (!token) {
  console.error('Missing HUGGING_FACE_TOKEN');
  process.exit(1);
}

async function listRepoTree(path) {
  // HF API: list_repo_tree non-recursive per folder (rate-limit friendly)
  const url = `https://huggingface.co/api/models/${opts.repo}/tree?path=${encodeURIComponent(path)}&recursive=false`;
  const res = await fetch(url, {
    headers: { Authorization: `Bearer ${token}` },
  });
  if (res.status === 429) {
    console.warn('HF 429 — wait 360s recommended');
    throw new Error('HF rate limit');
  }
  if (!res.ok) throw new Error(`HF error ${res.status}`);
  return res.json();
}

async function main() {
  const base = opts.date; // e.g. 2026-04-27
  const entries = await listRepoTree(base);

  // Expect JSON files per hub: { hub, score, inLinks, outLinks, ... }
  // If files are raw JSON, we fetch each via CDN (no auth) to avoid recursive tree calls.
  // For <2h scope, assume index.json exists in folder summarizing hubs.
  // Fallback: try to fetch index.json via CDN first.
  const indexUrl = `https://huggingface.co/datasets/${opts.repo}/resolve/main/${base}/index.json`;
  let hubs = [];

  try {
    const idxRes = await fetch(indexUrl);
    if (idxRes.ok) {
      hubs = await idxRes.json();
    } else {
      // Build lightweight summary from tree entries (only names)
      hubs = entries
        .filter((e) => e.type === 'file' && e.path.endsWith('.json'))
        .map((e) => {
          const name = e.path.replace(/\.json$/, '');
          return { hub: name, score: 0, inLinks: 0, outLinks: 0 };
        });
    }
  } catch (e) {
    console.warn('Could not fetch index.json, using filenames', e.message);
    hubs = entries
      .filter((e) => e.type === 'file' && e.path.endsWith('.json'))
      .map((e) => {
        const name = e.path.replace(/\.json$/, '');
        return { hub: name, score: 0, inLinks: 0, outLinks: 0 };
      });
  }

  // If we have richer data, compute top by inLinks or score
  const top = hubs.reduce((best, cur) => {
    const curScore = cur.inLinks || cur.score || 0;
    const bestScore = best.inLinks || best.score || 0;
    return curScore > bestScore ? cur : best;
  }, hubs[0] || { hub: 'MOC', score: 0, inLinks: 0 });

  const out = {
    hub: top.hub || 'MOC',
    score: top.inLinks || top.score || 0,
    inLinks: top.inLinks || 0,
    outLinks: top.outLinks || 0,
    date: base,
    source: opts.repo,
    generatedAt: new Date().toISOString(),
  };

  const outPath = resolve(opts.out);
  writeFileSync(outPath, JSON.stringify(out, null, 2), 'utf8');
  console.log(`Wrote top-hub to ${outPath}`, out);
}

main().catch((err) => {
  console.error(err);
  process.exit(1);
});
```

Make executable:

```bash
chmod +x scripts/build-top-hub-signal.js
```
