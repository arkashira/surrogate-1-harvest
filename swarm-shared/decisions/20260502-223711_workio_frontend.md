# workio / frontend

## Final Synthesized Solution

### Diagnosis (merged)
- **No optimistic UI** for clock in/out → round-trip wait + double-tap duplicates on flaky networks.
- **No local-first draft state** → leave/OT form data lost on navigation/offline.
- **Clock state drift** between UI and server during hiccups; no retry/backoff or mutation queue.
- **Form resets on validation errors** → user rework.
- **Missing loading affordances** on action buttons → double submissions.
- **Clock-in status derived from remote fetch on every mount** → flicker + unnecessary network usage.
- **No offline queue** for clock/leave actions → silent failures when network drops.

### Scope
- `workio/src/features/attendance/ClockControls.tsx` (optimistic clock + offline queue + local-first status)
- `workio/src/features/leave/LeaveRequestForm.tsx` (draft persistence + optimistic submit + queue fallback)
- Shared utilities: `workio/src/lib/queue.ts`, `workio/src/lib/storage.ts`, `workio/src/lib/time.ts`

export const storage = {
  get<T>(key: string, fallback: T): T {
    try {
      const raw = localStorage.getItem(key);
      return raw ? JSON.parse(raw) : fallback;
    } catch {
      return fallback;
    }
  },
  set(key: string, value: unknown) {
    try {
      localStorage.setItem(key, JSON.stringify(value));
    } catch (e) {
      // quota exceeded or private mode; degrade gracefully
      console.warn('localStorage unavailable', e);
    }
  },
  remove(key: string) {
    try {
      localStorage.removeItem(key);
    } catch {}
  },
};

### `workio/src/lib/queue.ts`
```ts
type Task = {
  id: string;
  fn: () => Promise<unknown>;
  retries: number;
  maxRetries: number;
  backoff: number;
};

class OfflineQueue {
  private queue: Task[] = [];
  private processing = false;
  private listeners: Array<(q: Task[]) => void> = [];

  constructor() {
    this.load();
    window.addEventListener('online', () => this.process());
  }

  private save() {
    try {
      localStorage.setItem('workio:queue', JSON.stringify(this.queue));
    } catch {}
  }

  private load() {
    try {
      const raw = localStorage.getItem('workio:queue');
      this.queue = raw ? JSON.parse(raw) : [];
    } catch {
      this.queue = [];
    }
  }

  subscribe(cb: (q: Task[]) => void) {
    cb(this.queue);
    this.listeners.push(cb);
    return () => {
      this.listeners = this.listeners.filter((l) => l !== cb);
    };
  }

  add(task: Omit<Task, 'id' | 'retries'>) {
    const item: Task = { ...task, id: crypto.randomUUID(), retries: 0 };
    this.queue.push(item);
    this.save();
    this.emit();
    this.process();
    return item.id;
  }

  remove(id: string) {
    this.queue = this.queue.filter((t) => t.id !== id);
    this.save();
    this.emit();
  }

  private emit() {
    this.listeners.forEach((l) => l(this.queue));
  }

  async process() {
    if (this.processing || this.queue.length === 0 || !navigator.onLine) return;
    this.processing = true;

    while (this.queue.length > 0 && navigator.onLine) {
      const task = this.queue[0];
      try {
        await task.fn();
        this.queue.shift();
        this.save();
        this.emit();
      } catch (err) {
        task.retries += 1;
        if (task.retries >= task.maxRetries) {
          this.queue.shift(); // drop after max retries
        } else {
          // exponential backoff with jitter
          const delay = Math.min(task.backoff * 2 ** (task.retries - 1), 30000) + Math.random() * 500;
          await new Promise((r) => setTimeout(r, delay));
          continue;
        }
        this.save();
        this.emit();
      }
    }

    this.processing = false;
  }
}

export const offlineQueue = new OfflineQueue();
```

  const d1 = new Date(a);
  const d2 = new Date(b);
  return (
    d1.getFullYear() === d2.getFullYear() &&
