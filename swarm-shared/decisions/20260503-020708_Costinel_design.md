# Costinel / design

## Final Implementation Plan — Top-Hub Signal Panel (Costinel Dashboard)

**Scope & Value**  
- Frontend-only, read-only React panel that surfaces the highest-signal / most-connected hub (default “MOC”) and its actionable proposals from the knowledge graph.  
- CDN-first data: a single JSON file (`/data/hubs/moc/signals.json`) fetched at runtime (no backend/API calls).  
- Ships in <2h: one new component, one route, one JSON asset, light styling using existing design tokens.  
- Follows pattern: review most-connected hub (MOC) before planning tasks (#knowledge-rag #graph #hub).

### 2) Component (`src/components/TopHubSignalPanel/TopHubSignalPanel.tsx`)

```tsx
import React, { useEffect, useState } from "react";
import styles from "./TopHubSignalPanel.module.css";

type Signal = {
  id: string;
  severity: "critical" | "high" | "medium" | "low";
  type: "anomaly" | "coverage" | "opportunity";
  headline: string;
  context: string;
  action: string;
  proposalId: string;
  impactUSD: number;
  affectedAccounts: string[];
};

type HubData = {
  hub: string;
  title: string;
  description: string;
  lastUpdated: string;
  signals: Signal[];
};

const TopHubSignalPanel: React.FC = () => {
  const [data, setData] = useState<HubData | null>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    // CDN-first fetch; falls back gracefully.
    fetch("/data/hubs/moc/signals.json", { cache: "no-cache" })
      .then((res) => {
        if (!res.ok) throw new Error("Failed to load hub signals");
        return res.json();
      })
      .then((json) => setData(json))
      .catch((err) => {
        // eslint-disable-next-line no-console
        console.error(err);
      })
      .finally(() => setLoading(false));
  }, []);

  if (loading) {
    return (
      <section className={styles.panel} aria-busy="true">
        <p className={styles.loading}>Loading top-hub signals…</p>
      </section>
    );
  }

  if (!data || !data.signals.length) {
    return null;
  }

  const severityClass = (s: Signal["severity"]) => styles[s] || "";

  return (
    <section className={styles.panel} aria-label={`${data.title} — actionable signals`}>
      <header className={styles.header}>
        <h2 className={styles.title}>{data.title}</h2>
        <p className={styles.meta}>
          Updated {new Date(data.lastUpdated).toLocaleDateString(undefined, { dateStyle: "medium" })}
        </p>
        <p className={styles.description}>{data.description}</p>
      </header>

      <ul className={styles.list} role="list">
        {data.signals.map((s) => (
          <li key={s.id} className={styles.item}>
            <div className={styles.itemHeader}>
              <span className={`${styles.badge} ${severityClass(s)}`}>{s.severity}</span>
              <span className={styles.type}>{s.type}</span>
            </div>
            <h3 className={styles.headline}>{s.headline}</h3>
            <p className={styles.context}>{s.context}</p>
            <p className={styles.action}>
              <strong>Action:</strong> {s.action}
            </p>
            <p className={styles.impact}>
              <strong>Impact:</strong> ${s.impactUSD.toLocaleString()}/mo
            </p>
            <p className={styles.accounts}>
              <strong>Accounts:</strong> {s.affectedAccounts.join(", ")}
            </p>
            <p className={styles.proposal}>
              <a href={`/proposals/${s.proposalId}`}>View proposal →</a>
            </p>
          </li>
        ))}
      </ul>
    </section>
  );
};

export default TopHubSignalPanel;
```
