# workio / backend

## Final consolidated solution

**Core problem**: LINE webhook replays and missing atomicity corrupt time/leave/OT state.  
**Goal**: reject replays, bound signature lifetime, make event processing exactly-once (idempotent) and auditable without changing business logic.

---

### 1. Database (one migration)

Create an idempotency + audit table with a unique event key and short TTL for cleanup.

### 2. Signature verification + replay window (middleware)

Preserve raw body; verify HMAC-SHA256 and a tight timestamp window.

```ts
// middleware/lineSignature.ts
import { Request, Response, NextFunction } from 'express';
import crypto from 'crypto';

const SIGNATURE_TTL_MS = 5 * 60 * 1000; // 5 minutes

export function verifyLineSignature(channelSecret: string) {
  return (req: Request, res: Response, next: NextFunction) => {
    const signature = req.get('X-Line-Signature');
    const rawBody = (req as any).rawBody || JSON.stringify(req.body);

    if (!signature || !/^[0-9a-f]{64}$/i.test(signature)) {
      return res.status(400).json({ error: 'Missing or invalid X-Line-Signature' });
    }

    const expected = crypto
      .createHmac('sha256', channelSecret)
      .update(Buffer.from(rawBody))
      .digest('hex');

    if (!crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected))) {
      return res.status(401).json({ error: 'Invalid signature' });
    }

    // Basic event-level timestamp freshness when available
    const events = (req.body.events || []) as Array<{ timestamp?: number }>;
    const now = Date.now();
    for (const ev of events) {
      if (ev.timestamp && Math.abs(now - Number(ev.timestamp)) > SIGNATURE_TTL_MS) {
        return res.status(400).json({ error: 'Event timestamp outside validity window' });
      }
    }

    (req as any).lineSignature = { signature, rawBody };
    next();
  };
}
```

### 3. Idempotency + transactional processing (middleware)

Insert-or-skip duplicates; wrap business work in a transaction; mark outcome in audit.

```ts
// middleware/lineIdempotency.ts
import { Request, Response, NextFunction } from 'express';
import crypto from 'crypto';
import { Pool, PoolClient } from 'pg';

const pool = new Pool({ connectionString: process.env.DATABASE_URL });

function stableEventId(ev: any): string {
  // Prefer platform-provided id when available
  if (ev.webhookEventId) return ev.webhookEventId;
  // Deterministic fallback
  return crypto
    .createHash('sha256')
    .update(
      JSON.stringify({
        type: ev.type,
        timestamp: ev.timestamp,
        source: ev.source,
        message: ev.message,
      })
    )
    .digest('hex');
}

export async function deduplicateAndAudit(req: Request, res: Response, next: NextFunction) {
  const ctx = (req as any).lineSignature;
  if (!ctx) return next();

  const { signature, rawBody } = ctx;
  const events = (req.body.events || []) as any[];
  const client = await pool.connect();
  const bodySha = crypto.createHash('sha256').update(rawBody).digest('hex');
  const now = new Date();

  try {
    await client.query('BEGIN');

    const toProcess: Array<{ id: string; ev: any }> = [];

    for (const ev of events) {
      const id = stableEventId(ev);

      const ins = await client.query(
        `INSERT INTO line_webhook_events
           (line_event_id, signature, body_sha256, raw_event, status, received_at)
         VALUES ($1,$2,$3,$4,'received',$5)
         ON CONFLICT (line_event_id) DO UPDATE
         SET received_at = EXCLUDED.received_at
         WHERE line_webhook_events.status IN ('failed','duplicate')
         RETURNING status`,
        [id, signature, bodySha, JSON.stringify(ev), now]
      );

      const previous = ins.rows[0]?.status;
      if (previous === 'processed') {
        // Already done — skip processing
        await client.query(
          `UPDATE line_webhook_events SET status='duplicate' WHERE line_event_id=$1`,
          [id]
        );
        continue;
      }

      // Mark processing under this transaction
      await client.query(
        `UPDATE line_webhook_events SET status='processing' WHERE line_event_id=$1`,
        [id]
      );
      toProcess.push({ id, ev });
    }

    // Attach for downstream handlers and commit/rollback helpers
    (req as any).lineDb = { client, events: toProcess };
    next();
  } catch (err) {
    await client.query('ROLLBACK');
    client.release();
    return res.status(500).json({ error: 'Idempotency check failed', details: String(err) });
  }
}

export async function commitAuditOnSuccess(req: Request, res: Response, next: NextFunction) {
  const ctx = (req as any).lineDb;
  if (!ctx) return next();

  const { client, events } = ctx;
  try {
    for (const { id } of events) {
      await client.query(
        `UPDATE line_webhook_events SET status='processed', processed_at=NOW() WHERE line_event_id=$1`,
        [id]
      );
    }
    await client.query('COMMIT');
    client.release();
    next();
  } catch (err) {
    await client.query('ROLLBACK');
    client.release();
    console.error('Failed to commit audit', err);
    // Business operation already succeeded; avoid double failure response.
    next();
  }
}

export async function rollbackAuditOnFailure(client: PoolClient) {
  try {
    const ids = (client as any)._pendingIds as string[] | undefined;
    if (ids) {
      for (const id of ids) {
        await client.query(
          `UPDATE line_webhook_events SET status='failed' WHERE line_event_id=$1`,
          [id]
        );
      }
    }
    await client.query('ROLLBACK');
  } finally {
    client.release();
  }
}
```
