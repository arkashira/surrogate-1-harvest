# workio / backend

## 1. Diagnosis
- Missing **database-level partial unique index** to enforce at most one open punch (`clock_out_at IS NULL`) per `(user_id, date(clock_in_at))`, allowing duplicates under LINE webhook retries or client double-taps.
- App-layer **find-then-insert** clock-in logic is race-prone under concurrent LINE webhook deliveries; no locking or atomic upsert.
- No **idempotency key** on webhook handler to deduplicate LINE event deliveries with same `webhookEventId`.
- Clock-in validation (location/GPS) runs after row creation in some flows, allowing invalid punches to persist.
- No **compensating transaction or cleanup job** to close stale open punches (e.g., midnight boundary or missed clock-out).

## 2. Proposed change
- **Scope**: `server/src/db/schema.sql` — add partial unique index; `server/src/routes/punch.ts` — replace find-then-insert with atomic `INSERT ... ON CONFLICT DO NOTHING` + validation-before-insert; add idempotency table and check for LINE webhook event id.
- **Impact**: prevents duplicates at DB level, makes webhook handler idempotent, and keeps validation before row creation.

## 3. Implementation

### 3.1 DB schema — partial unique index and idempotency table
```sql
-- server/src/db/schema.sql
-- Add partial unique index: at most one open punch per user per calendar day
CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_open_per_day
ON punches (user_id, date_trunc('day', clock_in_at))
WHERE clock_out_at IS NULL;

-- Idempotency table for LINE webhook events (lightweight)
CREATE TABLE IF NOT EXISTS webhook_idempotency (
  webhook_event_id VARCHAR(255) PRIMARY KEY,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
```

### 3.2 Punch route — atomic clock-in with validation and idempotency
```ts
// server/src/routes/punch.ts
import { Router, Request, Response } from 'express';
import { db } from '../db';
import { verifyLocation } from '../utils/location';

const router = Router();

// Helper: idempotent LINE webhook handling
async function isAlreadyProcessed(webhookEventId: string): Promise<boolean> {
  const { rows } = await db.query(
    'SELECT 1 FROM webhook_idempotency WHERE webhook_event_id = $1',
    [webhookEventId]
  );
  return rows.length > 0;
}

async function markProcessed(webhookEventId: string): Promise<void> {
  await db.query(
    'INSERT INTO webhook_idempotency (webhook_event_id) VALUES ($1) ON CONFLICT DO NOTHING',
    [webhookEventId]
  );
}

// Clock-in: atomic with validation
router.post('/clock-in', async (req: Request, res: Response) => {
  const { userId, latitude, longitude, webhookEventId } = req.body;

  // Idempotency: if provided and already processed, return 200 idempotent response
  if (webhookEventId && (await isAlreadyProcessed(webhookEventId))) {
    // Return current open punch if exists (safe read)
    const { rows } = await db.query(
      `SELECT id, clock_in_at, latitude, longitude
       FROM punches
       WHERE user_id = $1 AND clock_out_at IS NULL
       ORDER BY clock_in_at DESC LIMIT 1`,
      [userId]
    );
    return res.json({ ok: true, punch: rows[0] || null, idempotent: true });
  }

  // Validate location before touching punches
  const locationOk = await verifyLocation(latitude, longitude);
  if (!locationOk) {
    return res.status(400).json({ error: 'Location not allowed' });
  }

  try {
    // Atomic insert: rely on partial unique index to prevent duplicates
    const { rows } = await db.query(
      `INSERT INTO punches (user_id, clock_in_at, latitude, longitude)
       VALUES ($1, NOW(), $2, $3)
       ON CONFLICT (user_id, date_trunc('day', clock_in_at))
       WHERE clock_out_at IS NULL
       DO NOTHING
       RETURNING id, clock_in_at, latitude, longitude`,
      [userId, latitude, longitude]
    );

    // If no row returned, there is already an open punch
    if (rows.length === 0) {
      const { rows: existing } = await db.query(
        `SELECT id, clock_in_at, latitude, longitude
         FROM punches
         WHERE user_id = $1 AND clock_out_at IS NULL
         ORDER BY clock_in_at DESC LIMIT 1`,
        [userId]
      );
      return res.json({ ok: true, punch: existing[0] || null, conflict: true });
    }

    // Mark webhook processed if provided
    if (webhookEventId) await markProcessed(webhookEventId);

    return res.json({ ok: true, punch: rows[0] });
  } catch (err) {
    // Unique violation or other DB error
    if ((err as any).code === '23505' || (err as any).constraint === 'idx_punches_open_per_day') {
      const { rows } = await db.query(
        `SELECT id, clock_in_at, latitude, longitude
         FROM punches
         WHERE user_id = $1 AND clock_out_at IS NULL
         ORDER BY clock_in_at DESC LIMIT 1`,
        [userId]
      );
      return res.json({ ok: true, punch: rows[0] || null, conflict: true });
    }
    console.error('Clock-in error', err);
    return res.status(500).json({ error: 'Internal error' });
  }
});

// Clock-out: idempotent and safe
router.post('/clock-out', async (req: Request, res: Response) => {
  const { userId, latitude, longitude, webhookEventId } = req.body;

  if (webhookEventId && (await isAlreadyProcessed(webhookEventId))) {
    const { rows } = await db.query(
      `SELECT id, clock_in_at, clock_out_at
       FROM punches
       WHERE user_id = $1 AND clock_out_at IS NOT NULL
       ORDER BY clock_out_at DESC LIMIT 1`,
      [userId]
    );
    return res.json({ ok: true, punch: rows[0] || null, idempotent: true });
  }

  const locationOk = await verifyLocation(latitude, longitude);
  if (!locationOk) {
    return res.status(400).json({ error: 'Location not allowed' });
  }

  try {
    const { rows } = await db.query(
      `UPDATE punches
       SET clock_out_at = NOW(), latitude_out = $2, longitude_out = $3
       WHERE id = (
         SELECT id
         FROM punches
         WHERE user_id = $4 AND clock_out_at IS NULL
         ORDER BY clock_in_at DESC
         LIMIT 1
       )
       RETURNING id, clock_in_at, clock_out_at`,
      [userId, latitude, longitude, userId]
    );

    if (webhookEventId) await markProcessed(webhookEventId);

    if (rows.length === 0) {
      return res.status(404).json({ error: 'No open punch to clock out' });
    }
    return res.json({ ok: true, punch: rows[0] });
  } catch (err) {
    console.error('Clock-out error', err);
    return res.status(500).json({ error: 'Internal error' });
  }
});

export default router;
```

## 4. Verification
- **Index present**: `psql workio -c "\d idx_punches_open_per_day"` should show the partial unique index.
- **Duplicate prevention**: Simulate concurrent clock-ins for same user same day (e.g., `ab -n 10 -c 5 -p payload.json http://localhost:3000/punch/clock-in`) — only one open punch row should exist and others return `conflict: true`.
- **Idempotency**: Send same `webhookEventId` twice — second request returns `idempotent: true` and same punch.
- **Validation**: Send invalid location — should return 400 before any punch row created.
- **Clock-out**: Ensure open punch is updated and no new rows created; repeated
