# workio / backend

## 1. Diagnosis

- **Non-idempotent LINE webhook handler**: retries on 5xx/timeouts create duplicate punches and invalid open/close state (confirmed in prior synthesis).
- **No atomic punch transition**: concurrent requests or retries can interleave, leaving `punches` without a matching `clock_out` or double `clock_in`.
- **Missing idempotency key**: LINE does not provide a built-in idempotency header; we must enforce at application level to deduplicate within a safe window.
- **Race-prone state checks**: `findLastPunch` + `create` is read-then-write; under concurrency this permits multiple `clock_in` rows.
- **No transactional boundary**: punch creation and related updates (daily summary, notifications) are not atomic; partial writes can occur on failure.

## 2. Proposed change

File: `/opt/axentx/workio/server/src/services/punchService.ts`  
Scope: add idempotency enforcement and atomic upsert for punch transitions using a single `INSERT ... ON CONFLICT` per (employee_id, date, type) with a unique constraint and transaction.  
Also add a small idempotency table for LINE event deduplication (event id + retry signature) to absorb LINE retries safely.

-- Idempotency keys for LINE events (absorb retries)
CREATE TABLE IF NOT EXISTS line_event_idempotency (
  event_id         TEXT NOT NULL,
  received_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  processed_at     TIMESTAMPTZ,
  handler          TEXT NOT NULL,
  payload_hash     TEXT,
  PRIMARY KEY (event_id, handler)
);

-- Ensure one clock_in per employee per date; allow one clock_out per clock_in via nullable clock_out_ts
-- Simplified punch model: one row per session (clock_in + clock_out)
CREATE TABLE IF NOT EXISTS punches (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  employee_id     UUID NOT NULL REFERENCES employees(id) ON DELETE CASCADE,
  tenant_id       UUID NOT NULL REFERENCES tenants(id) ON DELETE CASCADE,
  date            DATE NOT NULL,               -- date of clock_in (for partitioning/indexing)
  clock_in_ts     TIMESTAMPTZ NOT NULL,
  clock_out_ts    TIMESTAMPTZ,
  clock_in_lat    NUMERIC(9,6),
  clock_in_lon    NUMERIC(9,6),
  clock_out_lat   NUMERIC(9,6),
  clock_out_lon   NUMERIC(9,6),
  created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  UNIQUE (employee_id, date)  -- ensures at most one active session per day (adjust if multiple shifts needed)
);

-- Index for fast last-session lookup
CREATE INDEX IF NOT EXISTS idx_punches_employee_date ON punches (employee_id, date DESC);

// server/src/services/punchService.ts
import { pool } from '../db';
import { Row } from 'pg';

export interface PunchClockIn {
  employeeId: string;
  tenantId: string;
  timestamp: Date;
  lat?: number;
  lon?: number;
}

export interface PunchClockOut {
  employeeId: string;
  tenantId: string;
  timestamp: Date;
  lat?: number;
  lon?: number;
}

/**
 * Idempotent clock-in: ensures one clock_in per employee per date.
 * Uses INSERT ... ON CONFLICT to make transition atomic and safe under retries/concurrency.
 */
export async function clockIn(payload: PunchClockIn) {
  const { employeeId, tenantId, timestamp, lat, lon } = payload;
  const date = new Date(timestamp);
  date.setUTCHours(0, 0, 0, 0);

  const query = `
    INSERT INTO punches (employee_id, tenant_id, date, clock_in_ts, clock_in_lat, clock_in_lon, clock_out_ts)
    VALUES ($1, $2, $3, $4, $5, $6, NULL)
    ON CONFLICT (employee_id, date) DO UPDATE
      SET clock_in_ts = EXCLUDED.clock_in_ts,
          clock_in_lat = EXCLUDED.clock_in_lat,
          clock_in_lon = EXCLUDED.clock_in_lon,
          updated_at = NOW()
      WHERE punches.clock_in_ts IS NULL OR punches.clock_in_ts = EXCLUDED.clock_in_ts
    RETURNING *;
  `;

  const res = await pool.query(query, [
    employeeId,
    tenantId,
    date,
    timestamp,
    lat ?? null,
    lon ?? null,
  ]);

  return rowToPunch(res.rows[0]);
}

/**
 * Idempotent clock-out: closes the open session for the given date.
 * Safe under retries: multiple identical clock_out_ts will not create extra rows.
 */
export async function clockOut(payload: PunchClockOut) {
  const { employeeId, tenantId, timestamp, lat, lon } = payload;
  const date = new Date(timestamp);
  date.setUTCHours(0, 0, 0, 0);

  const query = `
    UPDATE punches
    SET clock_out_ts = $4,
        clock_out_lat = $5,
        clock_out_lon = $6,
        updated_at = NOW()
    WHERE employee_id = $1 AND date = $2 AND tenant_id = $3 AND clock_out_ts IS NULL
    RETURNING *;
  `;

  const res = await pool.query(query, [employeeId, date, tenantId, timestamp, lat ?? null, lon ?? null]);

  if (res.rowCount === 0) {
    // Either already clocked out or no clock_in row; return current state
    const getQuery = `
      SELECT * FROM punches
      WHERE employee_id = $1 AND date = $2 AND tenant_id = $3
      ORDER BY clock_in_ts DESC LIMIT 1;
    `;
    const getRes = await pool.query(getQuery, [employeeId, date, tenantId]);
    return rowToPunch(getRes.rows[0] || null);
  }

  return rowToPunch(res.rows[0]);
}

/**
 * Deduplicate LINE event processing. Returns true if event should be processed.
 * Uses event_id + handler name as composite key. Short TTL cleanup can be added later.
 */
export async function shouldProcessLineEvent(eventId: string, handler: string): Promise<boolean> {
  const insertQuery = `
    INSERT INTO line_event_idempotency (event_id, handler, processed_at)
    VALUES ($1, $2, NOW())
    ON CONFLICT (event_id, handler) DO NOTHING
    RETURNING event_id;
  `;
  const res = await pool.query(insertQuery, [eventId, handler]);
  return res.rowCount === 1;
}

function rowToPunch(row: Row | null) {
  if (!row) return null;
  return {
    id: row.id,
    employeeId: row.employee_id,
    tenantId: row.tenant_id,
    date: row.date,
    clockInTs: row.clock_in_ts,
    clockOutTs: row.clock_out_ts,
    clockInLat: row.clock_in_lat,
    clockInLon: row.clock_in_lon,
    clockOutLat: row.clock_out_lat,
    clockOutLon: row.clock_out_lon,
    createdAt: row.created_at,
    updatedAt: row.updated_at,
  };
}
