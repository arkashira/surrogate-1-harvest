# airship / design

## Highest-Value Incremental Improvement (<2h)

**Improvement**: Harden `airship discover` into a deterministic, CDN-only orchestrator that eliminates HF API rate limits and PyArrow schema errors while producing reproducible file lists.

**Why this ships fastest**:
- Pure orchestration change (no model training/infra spin-up)
- Reuses existing HF credentials + local tooling
- Fixes two known failure modes (429 rate limits, mixed-schema CastError) in one sweep
- Produces immediate artifact (`file-list.json`) that downstream training can consume with zero API calls

---

## Implementation Plan

### 1. Scope (90 minutes)
- Modify `airship/cli/discover.py` (or create if absent) to:
  - Accept repo, date-folder, and output path
  - Use `list_repo_tree(recursive=False)` per date folder (not recursive on entire repo)
  - Emit `file-list.json` with CDN-ready paths and metadata
  - Include fallback to CDN direct download template
- Add lightweight validation script to confirm CDN reachability
- Add usage to README snippet

### 2. Key Rules Enforced
- **No `load_dataset(streaming=True)` on heterogeneous repos** → avoid PyArrow CastError
- **No recursive `list_repo_files`** → avoid 100× pagination and 429
- **CDN-only during training** → embed file list; training uses `https://huggingface.co/datasets/{repo}/resolve/main/{path}` with no Authorization header
- **Deterministic slug → repo routing** → if writing across siblings, use hash mod N (placeholder)

### 3. Deliverables
- `airship/cli/discover.py`
- `airship/scripts/validate_cdn_urls.py`
- `README.md` snippet for usage

### `airship/cli/discover.py`
```python
#!/usr/bin/env python3
"""
Deterministic, CDN-only discovery for HF datasets.
Produces file-list.json for reproducible training without HF API rate limits.
"""

import argparse
import hashlib
import json
import os
import sys
from datetime import datetime
from pathlib import Path
from typing import List, Dict

from huggingface_hub import HfApi, hf_hub_download

API = HfApi()

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

SIBLING_REPOS = [
    "axentx/surrogate-data",
    "axentx/surrogate-data-1",
    "axentx/surrogate-data-2",
    "axentx/surrogate-data-3",
    "axentx/surrogate-data-4",
]

def pick_sibling_repo(slug: str) -> str:
    """Deterministic repo selection for commit-cap scaling."""
    digest = hashlib.sha256(slug.encode()).hexdigest()
    idx = int(digest, 16) % len(SIBLING_REPOS)
    return SIBLING_REPOS[idx]

def list_date_folder(repo: str, date_folder: str) -> List[Dict]:
    """
    List files in a single date folder (non-recursive) to avoid pagination + rate limits.
    Returns CDN-ready entries.
    """
    entries = []
    try:
        # list_repo_tree with recursive=False keeps pagination minimal
        tree = API.list_repo_tree(repo=repo, path=date_folder, recursive=False)
        for item in tree:
            if item.type != "file":
                continue
            # Skip non-parquet if desired; keep generic for flexibility
            if not item.path.lower().endswith((".parquet", ".jsonl", ".json")):
                continue

            entries.append({
                "repo": repo,
                "path": item.path,
                "cdn_url": CDN_TEMPLATE.format(repo=repo, path=item.path),
                "size": getattr(item, "size", None),
                "lfs": getattr(item, "lfs", None),
            })
    except Exception as exc:
        print(f"[WARN] Failed to list {repo}/{date_folder}: {exc}", file=sys.stderr)
    return entries

def build_file_list(repo: str, date_folder: str, output_path: Path) -> None:
    """Emit file-list.json for CDN-only training."""
    entries = list_date_folder(repo, date_folder)

    manifest = {
        "generated_at": datetime.utcnow().isoformat() + "Z",
        "repo": repo,
        "date_folder": date_folder,
        "total_files": len(entries),
        "cdn_only": True,
        "note": "Use CDN URLs during training to avoid HF API rate limits. Do not use load_dataset(streaming=True) on heterogeneous repos.",
        "files": entries,
    }

    output_path.parent.mkdir(parents=True, exist_ok=True)
    with open(output_path, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2, ensure_ascii=False)

    print(f"[OK] Wrote {len(entries)} files -> {output_path}")
    if not entries:
        print("[WARN] No files found; verify date_folder exists and contains parquet/jsonl/json.", file=sys.stderr)

def main() -> None:
    parser = argparse.ArgumentParser(description="CDN-only HF dataset discovery")
    parser.add_argument("--repo", default="axentx/surrogate-data", help="HF dataset repo")
    parser.add_argument("--date-folder", required=True, help="Folder (e.g. batches/mirror-merged/2026-04-29)")
    parser.add_argument("--output", default="file-list.json", help="Output JSON path")
    parser.add_argument("--sibling-slug", help="Optional slug to pick sibling repo deterministically")
    args = parser.parse_args()

    repo = args.repo
    if args.sibling_slug:
        repo = pick_sibling_repo(args.sibling_slug)
        print(f"[INFO] Selected sibling repo: {repo}")

    build_file_list(repo, args.date_folder, Path(args.output))

if __name__ == "__main__":
    main()
```

### `airship/scripts/validate_cdn_urls.py`
```python
#!/usr/bin/env python3
"""
Quick validation that CDN URLs from file-list.json are reachable (HEAD check).
No Authorization header used — bypasses API rate limits.
"""

import json
import sys
from concurrent.futures import ThreadPoolExecutor, as_completed
from pathlib import Path
from typing import List

import requests

TIMEOUT = 10
MAX_WORKERS = 20

def check_cdn(url: str) -> bool:
    try:
        resp = requests.head(url, timeout=TIMEOUT, allow_redirects=True)
        return resp.status_code == 200
    except Exception:
        return False

def main() -> None:
    if len(sys.argv) < 2:
        print("Usage: validate_cdn_urls.py file-list.json", file=sys.stderr)
        sys.exit(1)

    path = Path(sys.argv[1])
    with open(path, encoding="utf-8") as f:
        data = json.load(f)

    urls: List[str] = [f["cdn_url"] for f in data.get("files", [])]
    if not urls:
        print("[WARN] No CDN URLs to validate.", file=sys.stderr)
        return

    ok = 0
    failed = []

    with ThreadPoolExecutor(max_workers=MAX_WORKERS) as ex:
        future_to_url = {ex.submit(check_cdn, url): url for url in urls}
        for future in as_completed(future_to_url):
            url = future_to_url[future]
            if future.result():
                ok += 1
            else:
                failed.append(url)

    print(f"[OK] {ok}/{len(urls)} CDN URLs reachable")
    if failed:
        print("[FAIL] Unreachable:", file=sys.stderr)
        for u in failed[:10]:
            print(f"  {u}", file=sys.stderr)
        if len(failed) > 10:
            print(f"  ... and {len(failed)-10} more", file=sys.stderr)
        sys.exit(1)

if __name__ == "__main__":
    main()
```
