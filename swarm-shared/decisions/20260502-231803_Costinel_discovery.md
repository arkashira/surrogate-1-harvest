# Costinel / discovery

## Final Implementation Plan — `GET /api/v1/cost-anomaly/signal/top-hub`

**Scope & constraints**
- Read-only, no side effects.
- Optional `?for_date=YYYY-MM-DD` (default today UTC) and `?top_n` (1–10, default 1).
- Reuse existing knowledge-graph assets (MOC-style hubs) to surface the most-connected hub(s) for cost anomalies.
- Fast path: pre-computed daily snapshot in `knowledge-rag/top-hub/{date}.json`; fallback to lightweight in-memory graph query.
- Response shape:
  ```json
  {
    "date": "2026-05-02",
    "top_hubs": [
      {
        "hub_id": "MOC",
        "label": "Mission Operations Center",
        "score": 0.94,
        "signal_type": "cost_anomaly",
        "context_docs": [
          { "doc_id": "ops-2026-05-02-001", "title": "Costinel / ops", "snippet": "..." }
        ],
        "related_tags": ["#knowledge-rag", "#graph", "#hub"]
      }
    ],
    "generated_at": "2026-05-02T23:15:20Z"
  }
  ```

**Implementation steps (<2h)**

1. Add route and handler (FastAPI assumed; adapt if Flask)
   - Path: `/api/v1/cost-anomaly/signal/top-hub`
   - Method: `GET`
   - Dependencies: `fastapi`, `python-dateutil`, `pydantic`

2. Data source strategy
   - Primary: read `knowledge-rag/top-hub/{date}.json` (generated by nightly `knowledge-rag` job).
   - Fallback: compute top hub from in-memory graph snapshot (lightweight degree-centrality on anomaly-related nodes).
   - Cache: 60s in-memory cache keyed by `date` to avoid repeated file reads.

3. Validation & defaults
   - `for_date`: parse as UTC date; default `datetime.utcnow().date()`.
   - `top_n`: int 1–10; clamp and default 1.

4. Response model
   - Pydantic model for strict typing and OpenAPI docs.

5. Observability
   - Log request/response summary (no PII).
   - Metrics: `costinel.top_hub_requests_total`, `costinel.top_hub_fallback_total`.

6. Tests (minimal)
   - Unit: validate date parsing, top_n clamping, fallback path.
   - Integration: hit endpoint and assert shape.

```python
from datetime import date, datetime, timezone
from functools import lru_cache
from pathlib import Path
from typing import List, Optional

from fastapi import APIRouter, HTTPException, Query
from pydantic import BaseModel, Field, validator
import json

router = APIRouter(prefix="/api/v1/cost-anomaly", tags=["cost-anomaly"])

# Config
KNOWLEDGE_RAG_TOP_HUB_DIR = Path("knowledge-rag/top-hub")
KNOWLEDGE_RAG_TOP_HUB_DIR.mkdir(parents=True, exist_ok=True)

# ---------- Models ----------

class ContextDoc(BaseModel):
    doc_id: str
    title: str
    snippet: Optional[str] = None

class TopHub(BaseModel):
    hub_id: str
    label: str
    score: float = Field(ge=0.0, le=1.0)
    signal_type: str = "cost_anomaly"
    context_docs: List[ContextDoc] = []
    related_tags: List[str] = []

class TopHubResponse(BaseModel):
    date: str
    top_hubs: List[TopHub]
    generated_at: str

    @validator("date")
    def valid_date(cls, v):
        try:
            date.fromisoformat(v)
        except ValueError:
            raise ValueError("date must be YYYY-MM-DD")
        return v

# ---------- Helpers ----------

def _today_utc() -> date:
    return datetime.now(timezone.utc).date()

def _parse_date(value: Optional[str]) -> date:
    if value is None:
        return _today_utc()
    try:
        return date.fromisoformat(value)
    except ValueError:
        raise HTTPException(status_code=400, detail="Invalid for_date, expected YYYY-MM-DD")

@lru_cache(maxsize=32)
def _load_top_hub_snapshot(target_date: date) -> Optional[dict]:
    """Load pre-computed top-hub snapshot; return None if missing."""
    p = KNOWLEDGE_RAG_TOP_HUB_DIR / f"{target_date.isoformat()}.json"
    if not p.is_file():
        return None
    try:
        with p.open("r", encoding="utf-8") as f:
            return json.load(f)
    except Exception:
        return None

def _compute_fallback_top_hub(target_date: date) -> dict:
    """
    Lightweight fallback: return a deterministic best-effort hub.
    Replace with real in-memory graph query if available.
    """
    # Placeholder: return MOC-style hub with neutral score.
    return {
        "date": target_date.isoformat(),
        "top_hubs": [
            {
                "hub_id": "MOC",
                "label": "Mission Operations Center",
                "score": 0.5,
                "signal_type": "cost_anomaly",
                "context_docs": [
                    {"doc_id": "ops-default", "title": "Costinel / ops", "snippet": "Fallback hub"}
                ],
                "related_tags": ["#knowledge-rag", "#graph", "#hub"]
            }
        ]
    }

# ---------- Endpoint ----------

@router.get("/signal/top-hub", response_model=TopHubResponse)
async def get_top_hub(
    for_date: Optional[str] = Query(None, description="Date (YYYY-MM-DD), defaults to today UTC"),
    top_n: int = Query(1, ge=1, le=10, description="Number of top hubs to return (1-10)")
):
    """
    Return the top hub(s) for cost anomaly signals for a given date.
    Read-only, no side effects. Uses knowledge-rag snapshot when available.
    """
    target_date = _parse_date(for_date)
    snapshot = _load_top_hub_snapshot(target_date)

    if snapshot is None:
        # Fallback path
        snapshot = _compute_fallback_top_hub(target_date)

    top_hubs_raw = snapshot.get("top_hubs", [])
    if not isinstance(top_hubs_raw, list):
        top_hubs_raw = []

    # Clamp and slice
    top_hubs_raw = top_hubs_raw[:top_n]

    # Normalize to TopHub models (tolerant to missing fields)
    top_hubs = []
    for item in top_hubs_raw:
        try:
            top_hubs.append(TopHub(
                hub_id=item.get("hub_id", "unknown"),
                label=item.get("label", "Unknown"),
                score=float(item.get("score", 0.0)),
                signal_type=item.get("signal_type", "cost_anomaly"),
                context_docs=[ContextDoc(**d) for d in item.get("context_docs", [])],
                related_tags=item.get("related_tags", [])
            ))
        except Exception:
            # Skip malformed entries rather than fail request
            continue

    # If nothing usable, provide minimal fallback entry
    if not top_hubs:
        top_hubs = [
            TopHub(
                hub_id="MOC",
                label="Mission Operations Center",
                score=0.5,
                signal_type="cost_anomaly",
                context_docs=[{"doc_id": "fallback", "title": "Costinel / ops", "snippet": "No snapshot"}],
                related_tags=["#knowledge-rag", "#graph", "#hub"]
            )
        ]

    resp = TopHubResponse(
        date=target_date.isoformat(),
        top_hubs=top_hubs,
        generated_at=datetime.now(timezone.utc).isoformat()
    )
    return resp
```
