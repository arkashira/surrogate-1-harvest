# surrogate-1 / design

## Final Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16`, `DATE`, `HF_TOKEN`
- Uses **one** `list_repo_tree` call (per date folder) to enumerate parquet files under `batches/public-merged/{DATE}/`
- Saves the file list to `manifest-{DATE}.json` (cached for Lightning training)
- Each shard deterministically owns `hash(slug) % SHARD_TOTAL == SHARD_ID`
- Downloads assigned files via **HF CDN bypass** (`resolve/main/...` URLs, no Authorization header) → avoids 429 during training
- Streams each parquet → projects to `{prompt, response}` only (no mixed-schema columns)
- Dedups via central `lib/dedup.py` md5 store (same as Space)
- Outputs `batches/public-merged/{DATE}/shard{N}-{HHMMSS}.jsonl`
- Reuses existing Lightning Studio pattern: checks for running surrogate-1 training studio before launching new jobs

## 1) bin/dataset-enrich.py

```python
#!/usr/bin/env python3
"""
CDN-bypass ingestion worker for surrogate-1-training-pairs.

Usage:
  SHARD_ID=0 SHARD_TOTAL=16 DATE=2026-05-03 \
  HF_TOKEN=hf_xxx \
  python bin/dataset-enrich.py

Behavior:
- Single list_repo_tree call for the DATE folder (parquet files only)
- Deterministic shard assignment by hash(slug) % SHARD_TOTAL
- CDN-only file downloads (no auth header) to bypass HF API rate limits
- Projects heterogeneous schemas to {prompt, response}
- Deduplicates via lib.dedup
- Outputs batches/public-merged/{DATE}/shard{N}-{HHMMSS}.jsonl
- Writes manifest-{DATE}.json for Lightning training cache
"""

import json
import os
import sys
import hashlib
import time
from datetime import datetime, timezone
from pathlib import Path

import pyarrow.parquet as pq
import requests
from huggingface_hub import HfApi

# Local
from lib.dedup import is_duplicate, mark_seen

HF_REPO = "datasets/axentx/surrogate-1-training-pairs"
BASE_CDN = f"https://huggingface.co/{HF_REPO}/resolve/main"
DATE_FMT = "%Y-%m-%d"

# Accept env
SHARD_ID = int(os.getenv("SHARD_ID", "0"))
SHARD_TOTAL = int(os.getenv("SHARD_TOTAL", "16"))
DATE = os.getenv("DATE", datetime.utcnow().strftime(DATE_FMT))
HF_TOKEN = os.getenv("HF_TOKEN")

if not HF_TOKEN:
    print("ERROR: HF_TOKEN required", file=sys.stderr)
    sys.exit(1)

api = HfApi(token=HF_TOKEN)
out_dir = Path("batches/public-merged") / DATE
out_dir.mkdir(parents=True, exist_ok=True)
ts = datetime.utcnow().strftime("%H%M%S")
out_path = out_dir / f"shard{SHARD_ID}-{ts}.jsonl"
manifest_path = Path(f"manifest-{DATE}.json")

session = requests.Session()
# CDN does not require Authorization; omit header to bypass /api/ rate limits

def list_parquet_files(date_folder: str):
    """Single list_repo_tree call for parquet files under date folder."""
    try:
        tree = api.list_repo_tree(
            repo_id=HF_REPO,
            path=date_folder,
            recursive=True,
        )
        files = [
            item.rfilename
            for item in tree
            if item.type == "file" and item.rfilename.lower().endswith(".parquet")
        ]
        return files
    except Exception as e:
        print(f"list_repo_tree failed: {e}", file=sys.stderr)
        return []

def slug_from_path(path: str) -> str:
    """Derive stable slug from path for shard assignment."""
    return path.strip("/")

def belongs_to_shard(slug: str) -> bool:
    h = int(hashlib.sha256(slug.encode()).hexdigest(), 16)
    return (h % SHARD_TOTAL) == SHARD_ID

def project_to_pair(raw_obj) -> dict | None:
    """
    Project heterogeneous schemas to {prompt, response}.
    Avoids pyarrow CastError from mixed schemas in load_dataset(streaming=True).
    """
    if isinstance(raw_obj, dict):
        prompt = raw_obj.get("prompt") or raw_obj.get("input") or raw_obj.get("question") or ""
        response = raw_obj.get("response") or raw_obj.get("output") or raw_obj.get("answer") or raw_obj.get("completion") or ""
        if prompt or response:
            return {"prompt": str(prompt), "response": str(response)}
    return None

def process_parquet_cdn(file_path: str):
    """Stream parquet via CDN and project + dedup."""
    cdn_url = f"{BASE_CDN}/{file_path}"
    slug = slug_from_path(file_path)

    if not belongs_to_shard(slug):
        return []

    try:
        # Download parquet via CDN (no auth header)
        resp = session.get(cdn_url, timeout=60)
        resp.raise_for_status()
    except Exception as e:
        print(f"CDN fetch failed {file_path}: {e}", file=sys.stderr)
        return []

    results = []
    try:
        with open("temp.parquet", "wb") as tmp:
            tmp.write(resp.content)
        table = pq.read_table("temp.parquet")
        os.remove("temp.parquet")
    except Exception as e:
        print(f"Parquet read failed {file_path}: {e}", file=sys.stderr)
        return []

    for batch in table.to_batches():
        for row in batch.to_pylist():
            pair = project_to_pair(row)
            if not pair:
                continue

            content = json.dumps(pair, sort_keys=True)
            digest = hashlib.md5(content.encode()).hexdigest()
            if is_duplicate(digest):
                continue

            mark_seen(digest)
            results.append({
                "prompt": pair["prompt"],
                "response": pair["response"],
                "source": slug,
            })
    return results

def main():
    date_folder = DATE
    print(f"Listing parquet files for {date_folder} (shard {SHARD_ID}/{SHARD_TOTAL})...")
    files = list_parquet_files(date_folder)
    if not files:
        print("No parquet files found; exiting.", file=sys.stderr)
        sys.exit(0)

    # Write manifest for Lightning training cache
    manifest = {
        "date": DATE,
        "shard_total": SHARD_TOTAL,
        "files": files,
        "generated_at": datetime.utcnow().strftime("%Y-%m-%dT%H:%M:%SZ"),
    }
    with open(manifest_path, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2)
    print(f"Manifest written to {manifest_path}")

    print(f"Found {len(files)} parquet files; processing shard-assigned subset...")
    written = 0
    with out_path.open("w", encoding="utf-8") as fout:
        for file_path in files:
            pairs = process_parquet_cdn(file_path)
            if not pairs:
                continue
            for p in pairs:
                fout.write(json.dumps(p, ensure_ascii=False) + "\n")
                written += 1

    print(f"Wrote {written} pairs to {out_path}")

if __name__ == "__main__":
    main()
```
