## Final Decision (highest-value ops fix, <2h)

**Replace every HF API–driven ingestion path in the surrogate training pipeline with a manifest‑driven, CDN‑only iterable loader.**  
- Generate a flat `file_manifest.json` once on the Mac orchestrator for the target date folder.  
- Train exclusively via public CDN URLs (`huggingface.co/datasets/.../resolve/main/...`).  
- Eliminates 429 rate limits and guarantees **zero HF API calls during data loading** in Lightning Studio.

### 1) Manifest generator (run once per date folder)

```bash
# scripts/generate_manifest.sh
# Usage:
#   HF_REPO=datasets/your/repo \
#   FOLDER=enriched/2026-05-03 \
#   ./scripts/generate_manifest.sh

#!/usr/bin/env bash
set -euo pipefail

HF_REPO="${HF_REPO:-datasets/your/repo}"
FOLDER="${FOLDER:-enriched/$(date +%Y-%m-%d)}"
OUT="${OUT:-file_manifest.json}"

python3 - "$HF_REPO" "$FOLDER" "$OUT" <<'PY'
import json, sys
from datetime import datetime, timezone
from huggingface_hub import HfApi

repo_id, folder = sys.argv[1], sys.argv[2]
out_path = sys.argv[3]

api = HfApi()
tree = api.list_repo_tree(repo_id=repo_id, path=folder, recursive=False)
files = [
    {
        "repo_id": repo_id,
        "path": node.path,
        "url": f"https://huggingface.co/datasets/{repo_id}/resolve/main/{node.path}"
    }
    for node in tree
    if node.type == "file" and node.path.lower().endswith(".parquet")
]

with open(out_path, "w", encoding="utf-8") as f:
    json.dump({
        "generated_at": datetime.now(timezone.utc).isoformat(),
        "repo_id": repo_id,
        "folder": folder,
        "files": files
    }, f, indent=2)

print(f"Wrote {len(files)} files to {out_path}")
PY
```

### 2) CDN-only iterable dataset

```python
# surrogate/training/data.py
import json
import pyarrow.parquet as pq
import requests
import tempfile
from pathlib import Path
from typing import Dict, Iterator, Optional

import torch
from torch.utils.data import IterableDataset


class CdnIterableDataset(IterableDataset):
    """
    HF CDN-only dataset loader.
    - Uses file_manifest.json produced by scripts/generate_manifest.sh
    - Downloads each parquet via public CDN URL (no auth)
    - Yields {"prompt": str, "response": str}
    """

    def __init__(
        self,
        manifest_path: str,
        max_files: Optional[int] = None,
        shuffle_files: bool = True,
    ):
        super().__init__()
        manifest_path = Path(manifest_path)
        with manifest_path.open(encoding="utf-8") as f:
            manifest = json.load(f)
        self.files = [f["url"] for f in manifest["files"]]
        if max_files:
            self.files = self.files[:max_files]
        self.shuffle_files = shuffle_files

    def _download_parquet(self, url: str) -> bytes:
        resp = requests.get(url, timeout=60)
        resp.raise_for_status()
        return resp.content

    def _iter_file(self, url: str) -> Iterator[Dict[str, str]]:
        raw = self._download_parquet(url)
        with tempfile.NamedTemporaryFile(suffix=".parquet", delete=False) as tmp:
            tmp.write(raw)
            tmp_path = tmp.name
        try:
            table = pq.read_table(tmp_path, columns=["prompt", "response"])
            df = table.to_pandas()
            for _, row in df.iterrows():
                prompt = row.get("prompt")
                response = row.get("response")
                if isinstance(prompt, str) and isinstance(response, str):
                    yield {"prompt": prompt.strip(), "response": response.strip()}
        finally:
            Path(tmp_path).unlink(missing_ok=True)

    def __iter__(self) -> Iterator[Dict[str, str]]:
        files = list(self.files)
        if self.shuffle_files:
            worker_info = torch.utils.data.get_worker_info()
            seed = worker_info.id if worker_info else 0
            import random
            random.Random(seed).shuffle(files)

        for url in files:
            yield from self._iter_file(url)
```

### 3) Wire into DataModule

```python
# surrogate/training/datamodule.py
from torch.utils.data import DataLoader
from .data import CdnIterableDataset


class SurrogateDataModule:
    def __init__(
        self,
        manifest_path: str,
        batch_size: int = 8,
        max_files: Optional[int] = None,
        num_workers: int = 4,
    ):
        self.manifest_path = manifest_path
        self.batch_size = batch_size
        self.max_files = max_files
        self.num_workers = num_workers

    def train_dataloader(self):
        dataset = CdnIterableDataset(
            manifest_path=self.manifest_path,
            max_files=self.max_files,
            shuffle_files=True,
        )
        return DataLoader(
            dataset,
            batch_size=self.batch_size,
            num_workers=self.num_workers,
        )
```
