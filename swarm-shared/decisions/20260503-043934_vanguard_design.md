# vanguard / design

## Final Synthesized Design (Correct + Actionable)

**Core principle:** Eliminate runtime HF API calls during training. Replace them with a **content-addressed, CDN-only** data path that is deterministic, resumable, and reproducible.

---

### 1. Diagnosis (resolved)
- **Problem:** Runtime repo enumeration (`list_repo_files`/`load_dataset`) causes 429s and non-reproducible epochs.
  - **Fix:** Pre-build a per-date manifest once; embed or reference it at runtime.
- **Problem:** No `{path, sha256}` snapshot → CDN fetches can’t be validated or resumed.
  - **Fix:** Manifest includes `sha256` and `size`; validate before yielding batches.
- **Problem:** No deterministic shard→repo mapping → ingestion bursts hit HF 128/hr/repo cap.
  - **Fix:** Deterministic hash-based shard-to-sibling-repo mapping.
- **Problem:** Mixed-parquet/schema drift causes mid-run `pyarrow.CastError` and wasted quota.
  - **Fix:** Validate + project columns at load time; skip malformed shards.
- **Problem:** Unclear Mac vs Studio boundary; risk of running model code locally and idle-stop quota waste.
  - **Fix:** Mac orchestrates manifest build + uploads; Studio runs **CDN-only** training with zero HF API calls.

---

### 2. Single-file artifact + minimal code

Create `/opt/axentx/vanguard/manifest.py` (single source of truth, <150 LOC):

```python
# /opt/axentx/vanguard/manifest.py
from __future__ import annotations
import json, hashlib, os, io
from dataclasses import dataclass, asdict
from typing import List, Dict, Any, Iterator
import requests
import pyarrow.parquet as pq

HF_DATASETS = "https://huggingface.co/datasets"

@dataclass
class FileEntry:
    path: str
    sha256: str
    size: int
    url: str

def _sha256_bytes(data: bytes) -> str:
    return hashlib.sha256(data).hexdigest()

def build_manifest(repo: str, date_folder: str, out_path: str, siblings: List[str] | None = None) -> List[FileEntry]:
    """
    One-time build (run on Mac or HF Space after rate-limit window).
    Produces content-addressed manifest for one date folder.
    Optionally maps shards across siblings to bypass 128/hr/repo cap.
    """
    from huggingface_hub import list_repo_tree

    entries: List[FileEntry] = []
    prefix = f"{date_folder}/"
    for item in list_repo_tree(repo, recursive=True, prefix=prefix):
        if item.type != "file" or not item.path.endswith(".parquet"):
            continue

        path = item.path
        # If siblings provided, map shard to sibling repo deterministically
        target_repo = repo
        if siblings:
            idx = hash(path) % len(siblings)
            target_repo = siblings[idx]

        url = f"{HF_DATASETS}/{target_repo}/resolve/main/{path}"
        r = requests.head(url, timeout=10)
        r.raise_for_status()
        size = int(r.headers.get("content-length", 0))

        # Download once to compute sha256 (acceptable for manifest build)
        data = requests.get(url, timeout=30).content
        sha256 = _sha256_bytes(data)
        if len(data) != size:
            raise RuntimeError(f"Size mismatch during manifest build: {path}")

        entries.append(FileEntry(path=path, sha256=sha256, size=size, url=url))

    os.makedirs(os.path.dirname(out_path), exist_ok=True)
    with open(out_path, "w") as f:
        for e in entries:
            f.write(json.dumps(asdict(e)) + "\n")
    return entries

def load_manifest(path: str) -> List[FileEntry]:
    entries: List[FileEntry] = []
    with open(path) as f:
        for line in f:
            line = line.strip()
            if not line:
                continue
            entries.append(FileEntry(**json.loads(line)))
    return entries

def cdn_dataset(manifest_path: str, project_cols: tuple[str, ...] = ("prompt", "response")) -> Iterator[Dict[str, Any]]:
    """
    Lightning Studio compatible: zero HF API calls, CDN-only, validated.
    Skips malformed shards and projects columns to avoid schema issues.
    """
    entries = load_manifest(manifest_path)
    for entry in entries:
        r = requests.get(entry.url, timeout=30)
        r.raise_for_status()
        data = r.content

        if entry.size and len(data) != entry.size:
            print(f"Skipping {entry.path}: size mismatch")
            continue
        if entry.sha256 and _sha256_bytes(data) != entry.sha256:
            print(f"Skipping {entry.path}: sha256 mismatch")
            continue

        try:
            pf = pq.ParquetFile(io.BytesIO(data))
            available = [c for c in project_cols if c in pf.schema.names]
            if not available:
                print(f"Skipping {entry.path}: no project columns")
                continue
            table = pf.read(columns=available)
        except Exception as exc:
            print(f"Skipping {entry.path}: {exc}")
            continue

        df = table.to_pandas()
        for _, row in df.iterrows():
            yield {k: row[k] for k in available}
```

# /opt/axentx/vanguard/train.py
from lightning import Fabric, LightningModule
import torch
from torch.utils.data import IterableDataset
from vanguard.manifest import cdn_dataset

class CDNIterableDataset(IterableDataset):
    def __init__(self, manifest_path):
        self.manifest_path = manifest_path

    def __iter__(self):
        yield from cdn_dataset(self.manifest_path)

class SimpleModel(LightningModule):
    def __init__(self):
        super().__init__()
        self.ln = torch.nn.Linear(10, 1)  # placeholder

    def training_step(self, batch, batch_idx):
        # adapt to your schema; this is illustrative
        return torch.tensor(0.0)

    def configure_optimizers(self):
        return torch.optim.Adam(self.parameters(), lr=1e-3)

if __name__ == "__main__":
    fabric = Fabric(devices=1)
    dataset = CDNIterableDataset("manifests/2026-04-29.jsonl")
    loader = fabric.DataLoader(dataset, batch_size=8)
    model = SimpleModel()
    fabric.fit(model, loader)
