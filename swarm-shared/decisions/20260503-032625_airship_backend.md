### 2.1 CDN-only file-list generator (Mac orchestration)
Create `surrogate/training/make_filelist.py` (clear name; run once per date folder).

```python
#!/usr/bin/env python3
"""
Generate CDN-only file list for a specific date folder.
Run on Mac (or any dev machine) when HF rate-limit window is clear.
Result (filelist.json) is committed or passed to training to avoid any HF API usage during training.
"""
import json
import argparse
from pathlib import Path
from huggingface_hub import HfApi

HF_REPO = "datasets/axentx/surrogate-mirror"

def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("--date", required=True, help="Date folder, e.g. 2026-04-29")
    parser.add_argument("--out", default="filelist.json", help="Output JSON path")
    args = parser.parse_args()

    api = HfApi()
    # Single non-recursive call to avoid pagination and reduce rate-limit risk
    entries = api.list_repo_tree(
        repo_id=HF_REPO,
        path=args.date,
        repo_type="dataset",
        recursive=False,
    )

    files = sorted(e.path for e in entries if e.type == "file")
    if not files:
        raise RuntimeError(f"No files found in {HF_REPO}/{args.date}")

    out_path = Path(args.out).resolve()
    out_path.write_text(json.dumps({"date": args.date, "files": files}, indent=2), encoding="utf-8")
    print(f"Wrote {len(files)} files -> {out_path}")

if __name__ == "__main__":
    main()
```

Make executable:
```bash
chmod +x surrogate/training/make_filelist.py
```

### 2.2 CDN-only IterableDataset with deterministic sharding and schema projection
Update `surrogate/training/train.py` (or create a clean data module). Key points:
- Load local `filelist.json`.
- Stream via CDN (`resolve/main/...`) — no HF API, no auth headers.
- Deterministic sharding by `hash(path) % num_shards`.
- Project only `prompt` and `response` at parse time (ignore heterogeneous schemas).
- Use `requests` with timeouts and robust error handling.

```python
# surrogate/training/train.py  (data module excerpt)
import json
import hashlib
import requests
import pyarrow.parquet as pq
from io import BytesIO
from torch.utils.data import IterableDataset
from typing import Iterator, Dict

CDN_ROOT = "https://huggingface.co/datasets/axentx/surrogate-mirror/resolve/main"

class CDNParquetStream(IterableDataset):
    def __init__(self, filelist_path: str, shard_rank: int = 0, num_shards: int = 1) -> None:
        with open(filelist_path, encoding="utf-8") as f:
            manifest = json.load(f)
        self.files = sorted(manifest["files"])
        # Deterministic sharding
        self.files = [
            f for f in self.files
            if hashlib.sha256(f.encode()).digest()[0] % num_shards == shard_rank
        ]
        if not self.files:
            raise ValueError("No files assigned to this shard")

    def __iter__(self) -> Iterator[Dict[str, str]]:
        for path in self.files:
            url = f"{CDN_ROOT}/{path}"
            resp = requests.get(url, timeout=60)
            resp.raise_for_status()
            table = pq.read_table(BytesIO(resp.content))
            # Strict projection to avoid schema mismatches
            df = table.select(["prompt", "response"]).to_pandas()
            for _, row in df.iterrows():
                yield {"prompt": row["prompt"], "response": row["response"]}
```

### 2.3 Lightning Studio reuse + idle-stop resilience
Create `surrogate/training/launch_studio.py` to:
- Reuse a running Studio (saves quota).
- Restart if stopped (idle-timeout resilience).
- Prefer L40S in paid cloud with public-cloud fallback.
- Run training deterministically inside the Studio.

```python
#!/usr/bin/env python3
import lightning as L
from lightning.pytorch import Trainer

# Import after ensuring environment is correct inside Studio
def trainer_main() -> None:
    from train import MyModel, CDNParquetStream, DataLoader

    model = MyModel()
    # Example setup; adapt to your actual model/training loop
    global_rank = 0  # will be set properly by Lightning in distributed runs
    world_size = 1
    train_dataset = CDNParquetStream(
        filelist_path="filelist.json",
        shard_rank=global_rank,
        num_shards=world_size,
    )
    train_loader = DataLoader(train_dataset, batch_size=8, num_workers=4)

    trainer = Trainer(max_epochs=1, devices="auto", accelerator="gpu")
    trainer.fit(model, train_loader)

def launch_training() -> None:
    teamspace = L.Teamspace()
    studio_name = "surrogate-train-l40s"
    studio = None

    # Reuse running studio if available
    for s in teamspace.studios:
        if s.name == studio_name and s.status == "running":
            studio = s
            print(f"Reusing running Studio: {studio_name}")
            break

    clouds = ["lightning-lambda-prod", "lightning-public-prod"]
    machine = L.Machine.L40S

    if studio is None:
        print(f"Creating Studio: {studio_name}")
        for cloud in clouds:
            try:
                studio = L.Studio(
                    name=studio_name,
                    machine=machine,
                    cloud=cloud,
                    create_ok=True,
                )
                print(f"Launched on {cloud} with {machine}")
                break
            except Exception as e:
                print(f"Failed on {cloud}: {e}")
                continue
        else:
            raise RuntimeError("Could not launch Studio on any cloud")

    # Ensure studio is running (restart on idle-stop)
    if studio.status != "running":
        print("Studio stopped (idle-timeout). Restarting...")
        studio.start(machine=machine)

    # Run training inside Studio
    studio.run(fn=trainer_main, name="surrogate-cdn-train")

if __name__ == "__main__":
    launch_training()
```
