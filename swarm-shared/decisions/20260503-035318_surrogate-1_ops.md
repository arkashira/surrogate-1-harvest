Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16` (matrix) and optional `DATE_FOLDER` (defaults to today `YYYY-MM-DD`).
- Uses a **single pre-listed file manifest** (generated by the Mac orchestrator; fallback to one-time non-recursive tree call) to avoid recursive `list_repo_files` and HF API rate limits.

- Projects heterogeneous schemas to `{prompt, response}` at parse time (avoids `pyarrow.CastError`).
- Deduplicates via central `lib/dedup.py` md5 store and writes to `batches/public-merged/{date}/shard{N}-{HHMMSS}.jsonl`.

def shard_assign(slug: str, total: int) -> int:
    """Deterministic shard assignment."""
    digest = hashlib.sha256(slug.encode()).digest()
    return int.from_bytes(digest, "little") % total


def download_cdn(url: str, stream=True):
    """Download via HF CDN (no auth)."""
    resp = requests.get(url, stream=stream, timeout=60)

    return resp


def parse_file(path: Path, cdn_url: str):
    """Parse file and yield {prompt, response} dicts."""
    suffix = path.suffix.lower()
    try:
        if suffix == ".jsonl":
            resp = download_cdn(cdn_url, stream=True)
            for line in resp.iter_lines(decode_unicode=True):
                if not line.strip():
                    continue
                obj = json.loads(line)
                prompt = obj.get("prompt") or obj.get("input") or ""
                response = obj.get("response") or obj.get("output") or ""
                if prompt or response:
                    yield {"prompt": str(prompt), "response": str(response)}
        elif suffix == ".parquet":
            # Stream not supported; download to temp file
            resp = download_cdn(cdn_url, stream=False)
            import io
            buf = io.BytesIO(resp.content)
            table = pq.read_table(buf)

                prompt = row.get("prompt") or row.get("input") or ""
                response = row.get("response") or row.get("output") or ""
                if prompt or response:
                    yield {"prompt": str(prompt), "response": str(response)}

def build_file_list(repo: str, date_folder: str, manifest_path: Path | None):
    """Return list of (relative_path, slug) for files under date_folder."""
    if manifest_path and manifest_path.exists():
        with open(manifest_path) as f:
            data = json.load(f)
        # Expect list of relative paths
        items = [(p.strip("/"), Path(p).stem) for p in data if p.strip()]
        return items

    # Fallback: single non-recursive tree call (avoid list_repo_files)
    from huggingface_hub import list_repo_tree
    items = []
    try:
        tree = list_repo_tree(repo=repo, path=date_folder, recursive=False)
        for entry in tree:
            if entry.type == "file":
                rel = f"{date_folder}/{entry.path}"
                items.append((rel, Path(entry.path).stem))
    except Exception as e:
        print(f"[ERROR] Cannot list repo tree: {e}", file=sys.stderr)
        sys.exit(1)
    return items
