### 1) Deterministic ingestion (single HF API call)

Create `surrogate/ingest/list_files.py`:

- Run once from Mac after HF rate-limit window clears.
- Uses `HfApi.list_repo_tree(path=date_folder, recursive=False)` (non-recursive to avoid pagination explosion).
- Produces `surrogate/ingest/file_list.json` committed to repo or passed at runtime.
- Includes metadata (`repo`, `date_folder`, sorted file list) and a note that CDN URLs will be used.

**Key improvement**: add optional `--recursive` flag only if needed; default non-recursive. Validate that listed items are files (not dirs). Fail fast if repo/path is unreachable.

```python
#!/usr/bin/env python3
"""
Single HF API call to list files for a date folder.
Embed output in training script to enable CDN-only fetches.
"""
import json, os, sys, argparse
from huggingface_hub import HfApi

def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", default=os.getenv("HF_DATASET_REPO", "axentx/surrogate-mirror"))
    parser.add_argument("--date-folder", default=os.getenv("DATE_FOLDER", "2026-05-03"))
    parser.add_argument("--out", default=os.getenv("OUT", "surrogate/ingest/file_list.json"))
    parser.add_argument("--recursive", action="store_true")
    args = parser.parse_args()

    api = HfApi()
    entries = api.list_repo_tree(repo_id=args.repo, path=args.date_folder, recursive=args.recursive)
    files = [e.rfilename for e in entries if not e.rfilename.endswith("/")]
    if not files:
        print("No files found; check repo/path or use --recursive", file=sys.stderr)
        sys.exit(1)

    payload = {
        "repo": args.repo,
        "date_folder": args.date_folder,
        "files": sorted(files),
        "note": "CDN-safe list; use resolve/main/ URLs to bypass /api/ rate limits"
    }
    os.makedirs(os.path.dirname(args.out), exist_ok=True)
    with open(args.out, "w") as f:
        json.dump(payload, f, indent=2)
    print(f"Wrote {len(files)} files to {args.out}")

if __name__ == "__main__":
    main()
```

### 2) CDN-only dataloader (zero API calls during training)

Create `surrogate/training/cdn_dataset.py`:

- Uses `hf_hub_download` (which resolves to CDN for public files) — no Authorization header, bypasses `/api/` rate limits.
- Accepts `file_list.json` and deterministic split logic (train/val).
- Projects only `{prompt, response}` at parse time (avoids loading unnecessary columns).
- Adds schema checks and robust row handling (multiple rows per shard) with optional sampling.

```python
import json, os, pyarrow.parquet as pq
import torch
from torch.utils.data import Dataset
from huggingface_hub import hf_hub_download

class CDNParquetDataset(Dataset):
    """
    Uses HF CDN URLs (no Authorization header) to bypass /api/ rate limits.
    Expects file_list.json from list_files.py.
    Projects only {prompt, response} at parse time.
    """
    def __init__(self, file_list_path: str, repo: str, split="train", max_rows_per_shard=None):
        super().__init__()
        with open(file_list_path) as f:
            manifest = json.load(f)
        self.repo = repo
        self.files = manifest["files"]
        self.max_rows_per_shard = max_rows_per_shard
        # deterministic shard by split
        n = len(self.files)
        if split == "train":
            self.files = self.files[: int(n * 0.95)]
        elif split == "val":
            self.files = self.files[int(n * 0.95):]
        else:
            raise ValueError(split)

    def __len__(self):
        return len(self.files)

    def _load_shard(self, path):
        local = hf_hub_download(repo_id=self.repo, filename=path, repo_type="dataset")
        table = pq.read_table(local, columns=["prompt", "response"])
        df = table.to_pandas()
        # schema checks
        if "prompt" not in df.columns or "response" not in df.columns:
            raise KeyError("Missing prompt/response columns")
        # handle multiple rows per shard
        rows = []
        for _, row in df.iterrows():
            rows.append({"prompt": str(row["prompt"]), "response": str(row["response"])})
        if self.max_rows_per_shard:
            rows = rows[:self.max_rows_per_shard]
        return rows

    def __getitem__(self, idx):
        rows = self._load_shard(self.files[idx])
        # return first row per shard by default (preserve original behavior)
        return rows[0]

def collate_fn(batch):
    prompts = [b["prompt"] for b in batch]
    responses = [b["response"] for b in batch]
    return {"prompts": prompts, "responses": responses}
```

### 3) Lightning Studio reuse + idle-stop resilience

Create `surrogate/training/launch_studio.py`:

- Reuses running studios; restarts stopped ones.
- Adds retry/backoff and timeout for start operations.
- Validates machine availability and falls back gracefully.
- Submits training with explicit environment and command.

```python
#!/usr/bin/env python3
import os, time, sys, argparse
from lightning_sdk import Teamspace, Studio, Machine

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--teamspace", default=os.getenv("LIGHTNING_TEAMSPACE", "default"))
    parser.add_argument("--studio-name", default=os.getenv("STUDIO_NAME", "surrogate-train"))
    parser.add_argument("--machine", default=os.getenv("STUDIO_MACHINE", "L40S"))
    parser.add_argument("--script", default=os.getenv("TRAIN_SCRIPT", "train.py"))
    parser.add_argument("--max-wait", type=int, default=120)
    args = parser.parse_args()

    teamspace = Teamspace(name=args.teamspace)
    studio = None
    for s in teamspace.studios:
        if s.name == args.studio_name:
            studio = s
            break

    if studio:
        if studio.status == "running":
            print(f"Reusing running studio: {args.studio_name}")
        else:
            print(f"Restarting stopped studio: {args.studio_name}")
            studio.start(machine=args.machine)
    else:
        print(f"Creating studio: {args.studio_name}")
        studio = Studio(
            name=args.studio_name,
            teamspace=args.teamspace,
            machine=args.machine,
            create_ok=True,
        )

    # Ensure running with retry
    waited = 0
    while studio.status != "running" and waited < args.max_wait:
        time.sleep(10)
        waited += 10
        studio.refresh()
    if studio.status != "running":
        print("Studio failed to start within timeout", file=sys.stderr)
        sys.exit(1)

    run = studio.run(
        command=[args.script, "--data", "surrogate/ingest/file_list.json"],
        environment="surrogate/training",
    )
    print(f"Submitted run: {run.id}")
    run.watch()

if __name__ == "__main__":
    main()
```

