# workio / frontend

## Final Synthesized Solution

Below is the single, consolidated implementation that merges the strongest, most correct, and most actionable parts of both proposals. Contradictions are resolved in favor of correctness and concrete actionability.

### Key Decisions & Rationale

1. **Idempotency**  
   - **Correctness**: Idempotency must be stable for the *same logical action* within a short window (not a new key on every click).  
   - **Action**: Generate a deterministic key per `tenant:employee:type:calendarDay` (or short TTL) and reuse it while a request is in-flight. This prevents duplicate submissions from UI double-clicks and webhook replays.

2. **Optimistic UI + Rollback**  
   - **Correctness**: Optimistic updates must be reversible on conflict/error; otherwise state diverges.  
   - **Action**: Store the previous server state before optimistic update; on error, restore it (not just invalidate). Combine with invalidation for eventual consistency.

3. **GPS Caching & UX**  
   - **Correctness**: GPS acquisition is slow and can fail; caching reduces latency and prevents repeated prompts.  
   - **Action**: Cache location for a short TTL (e.g., 60s) and expose status (acquiring/failed). On failure, allow punch with fallback (null location + clear UX) rather than blocking indefinitely.

4. **Throttling & Disabled States**  
   - **Correctness**: Prevent rapid clicks and retry storms at the UI level.  
   - **Action**: Disable button during in-flight requests and suppress duplicate intents within a short TTL (client-side guard).

5. **Error Handling & Feedback**  
   - **Correctness**: Users need clear, actionable feedback; silent failures cause repeated attempts.  
   - **Action**: Surface toast/errors globally; distinguish between duplicate suppression (silent) and real failures (toast + recovery).

import { useState, useCallback, useRef } from 'react';
import { useMutation, useQueryClient } from '@tanstack/react-query';
import { useLocation } from '../hooks/useLocation';
import { clockIn, clockOut } from '../api/clock';
import type { PunchType } from '../types';
import { toast } from '../lib/toast';

const IDEMPOTENCY_TTL_MS = 2 * 60 * 1000; // 2m same-action suppression
const DAY_MS = 24 * 60 * 60 * 1000;

function getDayId() {
  return Math.floor(Date.now() / DAY_MS);
}

export function ClockButton() {
  const queryClient = useQueryClient();
  const { getCurrentPosition, lastPosition, clearCache, status: gpsStatus } = useLocation({
    enableHighAccuracy: true,
    maxAge: 60_000,
  });

  const idempotencyRef = useRef<{
    key?: string;
    ts?: number;
    inFlight?: boolean;
  }>({});

  const mutation = useMutation({
    mutationFn: async ({
      type,
      tenantId,
      employeeId,
    }: {
      type: PunchType;
      tenantId: string;
      employeeId: string;
    }) => {
      const position = await getCurrentPosition();
      const payload = {
        tenantId,
        employeeId,
        type,
        latitude: position?.coords.latitude ?? null,
        longitude: position?.coords.longitude ?? null,
        accuracy: position?.coords.accuracy ?? null,
      };

      const idempotencyKey = `${tenantId}:${employeeId}:${type}:${getDayId()}`;
      return (type === 'IN' ? clockIn(payload, idempotencyKey) : clockOut(payload, idempotencyKey)).then(
        (res) => ({
          ...res,
          _idempotencyKey: idempotencyKey,
        })
      );
    },
    onMutate: async ({ type, tenantId, employeeId }) => {
      // Cancel outgoing refetches to keep optimistic state stable
      await queryClient.cancelQueries({ queryKey: ['punches', 'open', tenantId, employeeId] });

      const idempotencyKey = `${tenantId}:${employeeId}:${type}:${getDayId()}`;
      const now = Date.now();

      // Throttle: suppress duplicate intent within TTL or if already in-flight
      if (
        (idempotencyRef.current.key === idempotencyKey &&
          idempotencyRef.current.ts &&
          now - idempotencyRef.current.ts < IDEMPOTENCY_TTL_MS) ||
        idempotencyRef.current.inFlight
      ) {
        throw new Error('DUPLICATE_INTENT_SUPPRESSED');
      }

      idempotencyRef.current = { key: idempotencyKey, ts: now, inFlight: true };

      // Snapshot previous state for rollback
      const previous = queryClient.getQueryData(['punches', 'open', tenantId, employeeId]);

      // Optimistic punch
      const optimistic = {
        id: `optimistic-${Date.now()}`,
        tenantId,
        employeeId,
        type,
        timestamp: new Date().toISOString(),
        latitude: lastPosition?.coords.latitude ?? null,
        longitude: lastPosition?.coords.longitude ?? null,
        _optimistic: true,
      };

      queryClient.setQueryData(['punches', 'open', tenantId, employeeId], optimistic);
      queryClient.setQueryData(['punches', 'recent', tenantId, employeeId], (old: any[] = []) => [
        optimistic,
        ...old,
      ]);

      return { optimisticId: optimistic.id, previous };
    },
    onError: (err, _variables, context) => {
      if (err.message === 'DUPLICATE_INTENT_SUPPRESSED') return;

      // Rollback to previous server state on error
      if (context?.previous) {
        const { tenantId, employeeId } = _variables;
        queryClient.setQueryData(['punches', 'open', tenantId, employeeId], context.previous);
      }

      queryClient.invalidateQueries({ queryKey: ['punches'] });
      toast.error('ไม่สามารถบันทึกเวลาได้ กรุณาลองใหม่อีกครั้ง');
    },
    onSuccess: (data, _variables, context) => {
      const { tenantId, employeeId } = data;
      // Replace optimistic with server truth
      queryClient.removeQueries({ queryKey: ['punches', 'open', tenantId, employeeId] });
      queryClient.invalidateQueries({ queryKey: ['punches', 'recent'] });
      queryClient.invalidateQueries({ queryKey: ['reports'] });
      clearCache();
    },
    onSettled: () => {
      idempotencyRef.current.inFlight = false;
      queryClient.invalidateQueries({ queryKey: ['punches', 'open'] });
    },
  });

  const handleClock = useCallback(
    async (type: PunchType) => {
      const tenantId = (window as any).__WORKIO_TENANT_ID;
      const employeeId = (window as any).__WORKIO_EMPLOYEE_ID;
      if (!tenantId || !employeeId) return;

      try {
        await mutation.mutateAsync({ type, tenantId, employeeId });
      } catch (e: any) {
        if (e.message !== 'DUPLICATE_INTENT_SUPPRESSED') {
          console.warn(e);
        }
      }
    },
    [mutation]
  );

  const tenantId = (window as any).__WORKIO_TENANT_ID;
  const employeeId = (window as any).__WORKIO_EMPLOYEE_ID;
  const isOpen = queryClient.getQueryData(['punches', 'open', tenantId, employeeId]);
  const isPending = mutation.isPending;
  const isDisabled = isPending || mutation.isLoading || gpsStatus === 'acquiring';
