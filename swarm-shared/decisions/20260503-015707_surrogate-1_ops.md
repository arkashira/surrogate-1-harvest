- Accepts `SHARD_ID` and `SHARD_TOTAL` (16) from the matrix workflow  
- Loads a pre-generated `manifest-YYYYMMDD.json` (created once per date by the Mac orchestrator via `list_repo_tree`) instead of calling `list_repo_files`/`list_repo_tree` per shard → eliminates HF API pressure and 429 risk  
- Downloads only assigned shard’s files via **HF CDN direct URLs** (`https://huggingface.co/datasets/.../resolve/main/...`) with no Authorization header → bypasses `/api/` rate limits entirely

- Deduplicates via the existing `lib/dedup.py` central md5 store  
- Writes `batches/public-merged/<date>/shard<N>-<HHMMSS>.jsonl` with shard id + timestamp to avoid collisions  
- Includes retry/backoff for CDN 429/5xx and HF commit-cap awareness (filename uniqueness prevents collisions)

def _now_slug() -> str:

def deterministic_shard(filename: str, total_shards: int) -> int:
    h = hashlib.sha256(filename.encode("utf-8")).hexdigest()
    return int(h, 16) % total_shards

def load_manifest(date_str: str) -> List[str]:
    manifest_path = MANIFEST_DIR / f"manifest-{date_str}.json"
    if not manifest_path.is_file():
        print(f"ERROR: manifest not found: {manifest_path}", file=sys.stderr)
        sys.exit(1)
    with open(manifest_path, "r", encoding="utf-8") as f:
        data = json.load(f)
    # Expected format: {"files": ["path1", "path2", ...]} or flat list
    if isinstance(data, dict) and "files" in data:
        return data["files"]
    if isinstance(data, list):
        return data
    print(f"ERROR: unexpected manifest format in {manifest_path}", file=sys.stderr)
    sys.exit(1)

def cdn_get(url: str) -> Optional[bytes]:
    for attempt in range(1, MAX_RETRIES + 1):
        try:
            resp = requests.get(url, timeout=TIMEOUT, headers={})
            if resp.status_code == 429:
                # CDN rate-limited; wait longer
                wait = INITIAL_BACKOFF * (BACKOFF_FACTOR ** (attempt - 1))
                print(f"CDN 429, retry {attempt}/{MAX_RETRIES} after {wait:.1f}s: {url}", file=sys.stderr)
                time.sleep(wait)
                continue
            resp.raise_for_status()
            return resp.content
        except requests.RequestException as exc:
            wait = INITIAL_BACKOFF * (BACKOFF_FACTOR ** (attempt - 1))
            print(f"Request error (attempt {attempt}/{MAX_RETRIES}): {exc}", file=sys.stderr)
            if attempt == MAX_RETRIES:
                return None
            time.sleep(wait)
    return None
