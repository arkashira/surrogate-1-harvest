# vanguard / ops

## Final Synthesis & Action Plan

**Chosen diagnosis (merged, de-duplicated):**
- No persistent ingestion run state → retries are manual and error-prone.
- No CDN-bypass ingestion path → training-time HF API calls trigger 429s.
- Dataset list view state is not URL-encoded → reloads/shared links lose filters/sort/page.
- Hash-router does not restore scroll position → disorienting jumps.
- No lightweight ops dashboard → blind spots on ingestion health and HF API usage.

**Resolution priority:** Correctness + concrete actionability.  
- Persist ingestion params and generate a CDN-bypass file list server-side (or via ops UI) so training consumes only CDN URLs.  
- Encode dataset list state in URL hash (query-style) and restore scroll position on hash navigation.  
- Add a minimal ops panel showing last run, file count, CDN vs API usage, and rate-limit headroom.

/**
 * Ops ingestion controller
 * - Persists ingestion params (repo, subpath, dateFolder, status) to localStorage
 * - Generates CDN-bypass file-list.json so Lightning training uses only CDN URLs
 * - Lightweight status panel (last run, file count, API vs CDN usage)
 */

const STORAGE_KEY = 'vanguard:ingest:params';
const CDN_BASE = 'https://huggingface.co/datasets';

function getDefaultParams() {
  return {
    repo: '',
    subpath: '',
    dateFolder: '',
    status: 'idle', // idle | listing | downloading | done | error
    fileCount: 0,
    lastRun: null,
    lastError: null,
    useCdnBypass: true,
    fileListPath: 'file-list.json',
    hfApiCalls: 0,
    cdnOnlyFetches: 0
  };
}

function loadParams() {
  try {
    const raw = localStorage.getItem(STORAGE_KEY);
    return raw ? { ...getDefaultParams(), ...JSON.parse(raw) } : getDefaultParams();
  } catch {
    return getDefaultParams();
  }
}

function saveParams(patch) {
  const current = loadParams();
  const next = { ...current, ...patch };
  localStorage.setItem(STORAGE_KEY, JSON.stringify(next));
  return next;
}

function escapeHtml(str) {
  if (!str) return '';
  return String(str)
    .replace(/&/g, '&amp;')
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;')
    .replace(/"/g, '&quot;')
    .replace(/'/g, '&#39;');
}

function buildFileListUrl(repo, subpath) {
  // HF Hub REST: non-recursive tree per folder
  return `https://huggingface.co/api/datasets/${repo}/tree${subpath ? '/' + subpath : ''}?recursive=true`;
}

async function listRepoTree(repo, subpath, hfToken) {
  const url = buildFileListUrl(repo, subpath);
  const headers = hfToken ? { Authorization: `Bearer ${hfToken}` } : {};
  const res = await fetch(url, { headers });
  if (!res.ok) {
    const text = await res.text().catch(() => '');
    throw new Error(`HF tree API ${res.status}: ${text}`);
  }
  const items = await res.json();
  if (!Array.isArray(items)) throw new Error('Unexpected HF tree response');
  return items;
}

function buildCdnDownloadUrl(repo, filePath) {
  return `${CDN_BASE}/${repo}/resolve/main/${filePath}`;
}

function formatTime(t) {
  if (!t) return '—';
  try {
    return new Date(t).toLocaleString();
  } catch {
    return String(t);
  }
}
