## Final consolidated solution (correct + actionable)

### 1. DB schema (single source of truth)

File: `/opt/axentx/workio/server/src/db/schema.sql`

```sql
-- Idempotency for deduplication across webhook + API
CREATE TABLE IF NOT EXISTS punch_idempotency (
  tenant_id      UUID NOT NULL,
  idempotency_key TEXT NOT NULL,
  created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  PRIMARY KEY (tenant_id, idempotency_key)
);

-- At-most-one-open punch per tenant+employee (DB-enforced)
CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_open_unique
  ON punches (tenant_id, employee_id)
  WHERE clock_out_at IS NULL;
```

Why:  
- `punch_idempotency` prevents duplicate processing of the same logical event regardless of origin (webhook or API).  
- The partial unique index enforces the business rule at the DB level, removing race windows that app-only checks can’t close.

### 2. Core service (transactional + idempotent)

File: `/opt/axentx/workio/server/src/services/punchService.ts`

```ts
import { PoolClient } from 'pg';
import { getDb } from './db';

export async function processClockInOut({
  tenantId,
  employeeId,
  clockInAt,
  clockOutAt,
  location,
  idempotencyKey,
  client,
}: {
  tenantId: string;
  employeeId: string;
  clockInAt?: Date;
  clockOutAt?: Date;
  location?: { lat: number; lng: number };
  idempotencyKey?: string;
  client?: PoolClient;
}) {
  const db = client ?? (await getDb().connect());
  const shouldRelease = !client;
  try {
    await db.query('BEGIN');

    // Idempotency guard (if key provided)
    if (idempotencyKey) {
      const exists = await db.query(
        'SELECT 1 FROM punch_idempotency WHERE tenant_id = $1 AND idempotency_key = $2',
        [tenantId, idempotencyKey]
      );
      if (exists.rows.length > 0) {
        await db.query('ROLLBACK');
        return { applied: false, reason: 'duplicate' };
      }
      await db.query(
        'INSERT INTO punch_idempotency (tenant_id, idempotency_key) VALUES ($1, $2)',
        [tenantId, idempotencyKey]
      );
    }

    if (clockInAt) {
      // One open punch enforced by DB unique index
      await db.query(
        `INSERT INTO punches (tenant_id, employee_id, clock_in_at, location)
         VALUES ($1, $2, $3, $4)`,
        [tenantId, employeeId, clockInAt, location ? JSON.stringify(location) : null]
      );
    } else if (clockOutAt) {
      // Close the open punch for this tenant+employee
      const res = await db.query(
        `UPDATE punches
         SET clock_out_at = $1, location_out = $2
         WHERE tenant_id = $3 AND employee_id = $4 AND clock_out_at IS NULL
         RETURNING *`,
        [clockOutAt, location ? JSON.stringify(location) : null, tenantId, employeeId]
      );
      if (res.rowCount === 0) {
        await db.query('ROLLBACK');
        return { applied: false, reason: 'no_open_punch' };
      }
    } else {
      await db.query('ROLLBACK');
      return { applied: false, reason: 'no_action' };
    }

    await db.query('COMMIT');
    return { applied: true };
  } catch (err: any) {
    await db.query('ROLLBACK');
    // Unique violation on open punch or idempotency -> treat as duplicate
    if (err.code === '23505') {
      return { applied: false, reason: 'duplicate' };
    }
    throw err;
  } finally {
    if (shouldRelease && client === undefined) {
      db.release();
    }
  }
}
```

Key choices:  
- All writes occur inside a transaction.  
- Idempotency key is recorded only after checks and before side effects, and uniqueness is DB-enforced.  
- Duplicate or constraint violations map to deterministic `{ applied: false, reason: 'duplicate' }`.  
- Accepts optional `client` to compose into larger transactions when needed.

### 3. LINE webhook (retry-safe)

File: `/opt/axentx/workio/server/src/routes/lineWebhook.ts`

```ts
import express from 'express';
import { processClockInOut } from '../services/punchService';

const router = express.Router();

router.post('/webhook/line', async (req, res) => {
  try {
    const events = req.body.events || [];
    for (const ev of events) {
      if (ev.type !== 'message' || ev.message.type !== 'text') continue;

      // Resolve tenant by channel/secret in production; env used for example
      const tenantId = process.env.LINE_TENANT_ID;
      if (!tenantId) {
        console.error('LINE_TENANT_ID not configured');
        continue;
      }

      const employeeId = ev.source.userId;
      const text = ev.message.text.trim();

      // Deterministic idempotency key from LINE (stable across retries)
      const idempotencyKey = ev.webhookEventId || ev.message.id;

      if (!idempotencyKey) {
        // If LINE doesn't provide one, reject to avoid unsafe retries
        continue;
      }

      if (text.toLowerCase().includes('clock in')) {
        await processClockInOut({
          tenantId,
          employeeId,
          clockInAt: new Date(),
          idempotencyKey,
        });
      } else if (text.toLowerCase().includes('clock out')) {
        await processClockInOut({
          tenantId,
          employeeId,
          clockOutAt: new Date(),
          idempotencyKey,
        });
      }
    }
    res.status(200).json({ ok: true });
  } catch (err) {
    console.error('LINE webhook error', err);
    // Return 5xx so LINE retries; idempotency prevents duplicates on retry
    res.status(500).json({ error: 'internal' });
  }
});

export default router;
```

Why:  
- Uses LINE’s stable `webhookEventId` (preferred) or `message.id` as idempotency key.  
- Returns 5xx on internal errors so LINE retries, while idempotency prevents double punches.  
- Keeps handler simple; all concurrency/deduplication logic lives in the service + DB.
