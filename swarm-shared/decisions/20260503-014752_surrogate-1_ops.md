- Uses a **single `list_repo_tree` snapshot** (JSON manifest) generated once per date and committed to the repo (or passed via `file_list.json`) so each GitHub Actions shard does **zero HF API calls** during data loading.
- Downloads files directly from **HF CDN** (`https://huggingface.co/datasets/.../resolve/main/...`) with no Authorization header, bypassing `/api/` rate limits entirely.
- Projects heterogeneous schemas to `{prompt, response}` only at parse time (avoids `pyarrow.CastError`).
- Writes output as `batches/public-merged/<date>/shard<N>-<HHMMSS>.jsonl` with deterministic shard assignment via `hash(slug) % 16`.

### bin/manifest.py
```python
#!/usr/bin/env python3
"""
Generate a file list snapshot for a given date folder.
Run once per date (on Mac) to avoid HF API rate limits during ingestion.
"""
import argparse
import json
import os
from huggingface_hub import HfApi

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", default="axentx/surrogate-1-training-pairs")
    parser.add_argument("--date", required=True, help="YYYY-MM-DD")
    parser.add_argument("--out", default="file_list.json")
    args = parser.parse_args()

    api = HfApi()
    # Single API call; non-recursive to minimize pagination
    tree = api.list_repo_tree(
        repo_id=args.repo,
        path=args.date,
        recursive=False,
        repo_type="dataset",
    )

    files = [
        os.path.join(args.date, f.path.split("/")[-1])
        for f in tree
        if f.type == "file"
    ]

    with open(args.out, "w") as f:
        json.dump({"date": args.date, "files": files}, f, indent=2)

    print(f"Wrote {len(files)} files to {args.out}")

if __name__ == "__main__":
    main()
```

### bin/dataset-enrich.py
```python
#!/usr/bin/env python3
"""
Manifest-driven, CDN-bypass ingestion worker.
Usage (in GitHub Actions matrix):
  SHARD_ID=0 TOTAL_SHARDS=16 DATE=2026-05-03 FILE_LIST=file_list.json python bin/dataset-enrich.py
"""
import argparse
import hashlib
import json
import os
import sys
import time
from pathlib import Path

import numpy as np
import pyarrow as pa
import pyarrow.parquet as pq
import requests

# Local dedup store
sys.path.insert(0, str(Path(__file__).parent))
from lib.dedup import DedupStore  # noqa: E402

HF_DATASETS = "axentx/surrogate-1-training-pairs"
CDN_ROOT = f"https://huggingface.co/datasets/{HF_DATASETS}/resolve/main"

def shard_for(slug: str, total: int) -> int:
    return int(hashlib.md5(slug.encode()).hexdigest(), 16) % total

def download_cdn(path: str, timeout: int = 60):
    url = f"{CDN_ROOT}/{path}"
    resp = requests.get(url, timeout=timeout, stream=True)
    resp.raise_for_status()
    return resp

def parse_file(path: str, dedup: DedupStore):
    """
    Project heterogeneous files to {prompt,response}.
    Supports .jsonl and .parquet.
    """
    ext = Path(path).suffix.lower()
    if ext == ".parquet":
        # Use pyarrow with projection to avoid mixed-schema CastError
        try:
            pf = pq.read_table(path, columns=["prompt", "response"])
        except (pa.ArrowInvalid, KeyError):
            # Fallback: read all and select
            table = pq.read_table(path)
            cols = [c for c in table.column_names if c in ("prompt", "response")]
            if len(cols) != 2:
                return []
            pf = table.select(cols)
        df = pf.to_pandas()
        for _, row in df.iterrows():
            yield row["prompt"], row["response"]
    else:
        # Assume line-oriented JSON or text
        with open(path, "r", encoding="utf-8", errors="ignore") as f:
            for line in f:
                line = line.strip()
                if not line:
                    continue
                try:
                    obj = json.loads(line)
                    prompt = obj.get("prompt") or obj.get("input") or ""
                    response = obj.get("response") or obj.get("output") or ""
                except Exception:
                    prompt, response = line, ""
                if prompt and response:
                    yield prompt, response

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--shard-id", type=int, required=True)
    parser.add_argument("--total-shards", type=int, default=16)
    parser.add_argument("--date", required=True)
    parser.add_argument("--file-list", required=True)
    parser.add_argument("--out-dir", default="batches/public-merged")
    args = parser.parse_args()

    with open(args.file_list) as f:
        manifest = json.load(f)

    files = [f for f in manifest["files"] if f.startswith(args.date)]
    my_files = [
        f for f in files
        if shard_for(f, args.total_shards) == args.shard_id
    ]

    ts = time.strftime("%H%M%S")
    out_path = Path(args.out_dir) / args.date / f"shard{args.shard_id}-{ts}.jsonl"
    out_path.parent.mkdir(parents=True, exist_ok=True)

    dedup = DedupStore()
    accepted = 0
    skipped_dup = 0

    # Download and process each file via CDN
    for rel_path in my_files:
        try:
            resp = download_cdn(rel_path)
            tmp = Path("tmp") / Path(rel_path).name
            tmp.parent.mkdir(exist_ok=True)
            with open(tmp, "wb") as f:
                for chunk in resp.iter_content(chunk_size=8192):
                    f.write(chunk)

            for prompt, response in parse_file(str(tmp), dedup):
                pair_hash = hashlib.md5(f"{prompt}\n{response}".encode()).hexdigest()
