# Costinel / design

## Final Implementation Plan — Top Hub Signal Panel (CDN-first, frontend-only)

**Scope**: Add a lightweight, resilient Top Hub signal card to the Costinel dashboard.  
**Effort**: ~60–90 minutes (frontend only).  
**Mechanism**:  
- Single orchestration call (Mac/CI) to `list_repo_tree(path, recursive=False)` for today’s folder → save `hub-files.json` to CDN/public.  
- Frontend loads `hub-files.json` + file contents via **HF CDN** (`/resolve/main/...`) — zero runtime API calls, bypassing 429 limits.  
- Display the top N most-connected hubs (e.g., "MOC") with key insights and links.  
- Graceful fallback if CDN fails (cached stale data + toast).

### 2) Data contract (public/data/hub-files.json)

```json
{
  "generatedAt": "2026-05-03T12:00:00Z",
  "source": "knowledge-rag",
  "dateFolder": "2026-05-03",
  "hubs": [
    {
      "path": "2026-05-03/MOC.json",
      "size": 1234,
      "last_modified": "2026-05-03T11:45:00Z"
    }
  ]
}
```

Each referenced hub file (e.g., `2026-05-03/MOC.json`) must contain:

```json
{
  "slug": "MOC",
  "title": "MOC",
  "summary": "Key insight summary",
  "score": 0.92,
  "links": [
    { "url": "https://huggingface.co/datasets/...", "title": "Dataset" },
    { "url": "https://huggingface.co/models/...", "title": "Model" }
  ],
  "updated": "2026-05-03T11:45:00Z"
}
```

### 3) Composable: CDN-first hub insights loader

`src/composables/useHubCdnInsights.js`

```js
import { ref, onMounted } from 'vue'

export function useHubCdnInsights({
  listPath = 'data/hub-files.json',
  hubBaseUrl = '/data',
  retryMs = 2000,
  maxRetries = 2,
  topN = 5
}) {
  const hubs = ref([])
  const loading = ref(false)
  const error = ref(null)

  async function fetchJsonWithRetry(url, attempt = 0) {
    try {
      const res = await fetch(url, { cache: 'no-store' })
      if (!res.ok) throw new Error(`HTTP ${res.status}`)
      return await res.json()
    } catch (err) {
      if (attempt < maxRetries) {
        await new Promise((r) => setTimeout(r, retryMs * (attempt + 1)))
        return fetchJsonWithRetry(url, attempt + 1)
      }
      throw err
    }
  }

  async function load() {
    loading.value = true
    error.value = null
    try {
      const listUrl = `${hubBaseUrl}/${listPath.replace(/^\/+/, '')}`
      const manifest = await fetchJsonWithRetry(listUrl)

      if (!manifest?.hubs?.length) {
        throw new Error('No hub entries in manifest')
      }

      const top = manifest.hubs
        .filter((e) => e.path && e.path.endsWith('.json'))
        .sort((a, b) => b.path.localeCompare(a.path))
        .slice(0, topN)

      const hubData = await Promise.all(
        top.map(async (t, idx) => {
          try {
            const url = `${hubBaseUrl}/${t.path.replace(/^\/+/, '')}`
            const meta = await fetchJsonWithRetry(url)
            return {
              slug: meta.slug || t.path.split('/').pop().replace('.json', ''),
              path: t.path,
              title: meta.title || meta.slug || t.path.split('/').pop().replace('.json', ''),
              summary: meta.summary || '',
              score: Number.isFinite(meta.score) ? Number(meta.score) : (Number.isFinite(t.score) ? Number(t.score) : 0),
              links: Array.isArray(meta.links) ? meta.links.slice(0, 8) : [],
              updated: meta.updated || t.last_modified || null,
              _index: idx + 1
            }
          } catch (e) {
            return {
              slug: t.path,
              title: 'Unavailable',
              summary: 'Failed to load hub details',
              score: 0,
              links: [],
              updated: t.last_modified || null,
              error: true,
              _index: idx + 1
            }
          }
        })
      )

      hubs.value = hubData
    } catch (err) {
      error.value = err.message || 'Failed to load hub signals'
    } finally {
      loading.value = false
    }
  }

  onMounted(load)

  return { hubs, loading, error, load }
}
```
