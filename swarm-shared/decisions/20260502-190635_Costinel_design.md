# Costinel / design

## Costinel — Design Increment (≤2h)

**Highest-value incremental improvement:**  
Ship a **deterministic discovery module** that produces machine-readable manifests (`manifests/sources.json`, `manifests/discovery.json`) and a **top-hub insight file** (`insights/top-hub.md`) so onboarding, training, and governance scripts can run without manual wiring.

This enables:
- Reproducible environment bootstraps (CI/CD, local dev, training pipelines).
- Knowledge-rag and graph pipelines to auto-select the most-connected hub (e.g., MOC) before planning.
- Surrogate-1 training and HF ingestion scripts to resolve sources deterministically (avoids runtime discovery/heuristics).

---

## Implementation Plan (<2h)

1. **Create directory scaffold** (2m)  
   `mkdir -p manifests insights`

2. **Add deterministic discovery script** `scripts/discover.py` (60–75m)  
   - Reads environment variables and optional `.env` for credentials.
   - Supports AWS, GCP, Azure connectors (lightweight stubs with validation).
   - Produces `manifests/sources.json` (connectors + cadence + auth scopes).
   - Produces `manifests/discovery.json` (discovered accounts, regions, tags, last-seen).
   - Deterministic output: sorted keys, stable UUIDs derived from deterministic slugs (hash of account+region).

3. **Add top-hub insight generator** `scripts/top_hub_insight.py` (15–20m)  
   - Reads `manifests/discovery.json`.
   - Picks most-connected hub by simple heuristic (highest unique service/region links).
   - Writes `insights/top-hub.md` with summary and actionable tags.

4. **Add lightweight validation + CLI entrypoint** (10m)  
   - `python -m scripts.discover --out-dir .` runs both steps.
   - Validate JSON schema with minimal inline checks.

5. **Update README snippet** (5m)  
   - Add quick usage and expected outputs.

### `scripts/discover.py`

```python
#!/usr/bin/env python3
"""
Deterministic discovery for Costinel.
Produces:
  - manifests/sources.json
  - manifests/discovery.json
"""

import json
import os
import hashlib
import sys
from datetime import datetime, timezone
from typing import Dict, List

MANIFESTS_DIR = "manifests"

def _hash_slug(*parts: str) -> str:
    return hashlib.sha256("::".join(parts).encode()).hexdigest()[:16]

def _now_iso() -> str:
    return datetime.now(timezone.utc).isoformat()

def build_sources() -> Dict:
    sources = []

    # AWS
    aws_access_key = os.getenv("AWS_ACCESS_KEY_ID")
    aws_secret = os.getenv("AWS_SECRET_ACCESS_KEY")
    aws_profile = os.getenv("AWS_PROFILE")
    if aws_access_key or aws_profile:
        sources.append({
            "type": "aws",
            "provider": "aws",
            "auth": {
                "method": "keys" if aws_access_key else "profile",
                "profile": aws_profile or "default"
            },
            "collection": {
                "cadence": "hourly",
                "scope": ["cost", "usage", "tags"],
                "regions": os.getenv("AWS_REGIONS", "us-east-1,eu-west-1").split(",")
            },
            "tags": ["cloud", "aws", "cost-governance"]
        })

    # GCP
    gcp_project = os.getenv("GCP_PROJECT")
    gcp_creds = os.getenv("GOOGLE_APPLICATION_CREDENTIALS")
    if gcp_project or gcp_creds:
        sources.append({
            "type": "gcp",
            "provider": "gcp",
            "auth": {
                "method": "adc" if not gcp_creds else "service-account",
                "credentials_path": gcp_creds or None
            },
            "collection": {
                "cadence": "hourly",
                "scope": ["cost", "usage", "labels"],
                "projects": [gcp_project] if gcp_project else []
            },
            "tags": ["cloud", "gcp", "cost-governance"]
        })

    # Azure
    azure_sub = os.getenv("AZURE_SUBSCRIPTION_ID")
    azure_tenant = os.getenv("AZURE_TENANT_ID")
    if azure_sub:
        sources.append({
            "type": "azure",
            "provider": "azure",
            "auth": {
                "method": "sp" if os.getenv("AZURE_CLIENT_ID") else "cli",
                "tenant_id": azure_tenant
            },
            "collection": {
                "cadence": "hourly",
                "scope": ["cost", "usage", "tags"],
                "subscriptions": [azure_sub]
            },
            "tags": ["cloud", "azure", "cost-governance"]
        })

    return {"sources": sources, "generated_at": _now_iso(), "version": "4.2.0"}

def build_discovery(sources: Dict) -> Dict:
    discovered = []
    for src in sources["sources"]:
        pid = _hash_slug(src["type"], src.get("provider", "unknown"))
        regions = src.get("collection", {}).get("regions", []) or []
        projects = src.get("collection", {}).get("projects", []) or []
        subs = src.get("collection", {}).get("subscriptions", []) or []

        accounts = []
        if regions:
            accounts.extend([{"id": _hash_slug(pid, r), "region": r.strip()} for r in regions])
        if projects:
            accounts.extend([{"id": _hash_slug(pid, p), "project": p.strip()} for p in projects])
        if subs:
            accounts.extend([{"id": _hash_slug(pid, s), "subscription": s.strip()} for s in subs])

        if not accounts:
            accounts = [{"id": pid, "label": "default"}]

        discovered.append({
            "source_id": pid,
            "type": src["type"],
            "provider": src.get("provider"),
            "accounts": accounts,
            "discovered_at": _now_iso(),
            "tags": src.get("tags", [])
        })

    # Deterministic ordering
    discovered.sort(key=lambda x: x["source_id"])
    for d in discovered:
        d["accounts"].sort(key=lambda a: a["id"])

    return {
        "discovered": discovered,
        "generated_at": _now_iso(),
        "version": "4.2.0"
    }

def write_manifest(name: str, payload: Dict, out_dir: str) -> str:
    os.makedirs(out_dir, exist_ok=True)
    path = os.path.join(out_dir, name)
    with open(path, "w", encoding="utf-8") as f:
        json.dump(payload, f, indent=2, sort_keys=True, ensure_ascii=False)
    return path

def main(out_dir: str = MANIFESTS_DIR) -> int:
    try:
        sources = build_sources()
        discovery = build_discovery(sources)

        sources_path = write_manifest("sources.json", sources, out_dir)
        discovery_path = write_manifest("discovery.json", discovery, out_dir)

        print(f"OK: wrote {sources_path}")
        print(f"OK: wrote {discovery_path}")

        # Generate top-hub insight
        from scripts.top_hub_insight import generate_top_hub_insight
        insights_path = generate_top_hub_insight(discovery, out_dir="insights")
        print(f"OK: wrote {insights_path}")
        return 0
    except Exception as exc:
        print(f"ERROR: {exc}", file=sys.stderr)
        return 1

if __name__ == "__main__":
    args = sys.argv[1:]
    target_dir = args[0] if args else MANIFESTS_DIR
    sys.exit(main(target_dir))
```
