# workio / quality

## 1. Diagnosis

- **No idempotency key enforcement on LINE webhook**: duplicate punches can occur from rapid clicks, retries, or LINE duplicate deliveries; no `X-Line-Signature` replay window or idempotency table.
- **Race condition on clock-in/out**: concurrent requests for same user can create multiple punch rows; no DB unique constraint or atomic upsert.
- **Missing optimistic UI + rollback**: frontend allows multiple clicks before response; no local pending state or error rollback, so UI can diverge from server truth.
- **No structured logging / audit trail**: punch events lack correlation IDs and consistent schema; hard to debug duplicates or investigate incidents.
- **Weak validation on location/time**: GPS and time checks exist but are not atomic with punch creation, allowing invalid punches to slip through under concurrency.

## 2. Proposed change

File scope:
- `workio/server/src/routes/webhook/line.ts` (or equivalent) — add idempotency and atomic punch upsert.
- `workio/server/src/db/schema.sql` — add `punches.idempotency_key` unique index and optional `punches.location_verified_at` for atomic validation.
- `workio/src/components/TimeClock.tsx` — add optimistic UI with pending state and rollback.

## 3. Implementation

### 3.1 DB schema change

```sql
-- workio/server/src/db/schema.sql
-- Add idempotency key to prevent duplicate processing
ALTER TABLE punches ADD COLUMN IF NOT EXISTS idempotency_key TEXT;
CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_idempotency_key
  ON punches (idempotency_key)
  WHERE idempotency_key IS NOT NULL;

-- Optional: mark when location was verified (atomic with punch)
ALTER TABLE punches ADD COLUMN IF NOT EXISTS location_verified_at TIMESTAMPTZ;
```

### 3.2 Atomic upsert + idempotency in webhook handler

```ts
// workio/server/src/routes/webhook/line.ts
import { Request, Response } from 'express';
import crypto from 'crypto';
import { pool } from '../../db';
import { verifyLineSignature } from '../../utils/line';

const IDEMPOTENCY_WINDOW_MS = 1000 * 60 * 5; // 5m

export async function handleLineWebhook(req: Request, res: Response) {
  const sig = req.headers['x-line-signature'] as string;
  const body = JSON.stringify(req.body);

  if (!verifyLineSignature(body, sig)) {
    return res.status(401).json({ error: 'Invalid signature' });
  }

  const events = req.body.events || [];
  const results = [];

  for (const ev of events) {
    if (ev.type !== 'message' || ev.message.type !== 'text') continue;

    // Derive idempotency key from event + userId + short timestamp bucket
    const bucket = Math.floor(ev.timestamp / 10000); // 10s bucket
    const idemKey = crypto
      .createHash('sha256')
      .update(`line:${ev.source.userId}:${ev.message.id}:${bucket}`)
      .digest('hex');

    try {
      // Fast path: skip if already processed
      const exists = await pool.query(
        'SELECT 1 FROM punches WHERE idempotency_key = $1',
        [idemKey]
      );
      if (exists.rows.length > 0) {
        results.push({ status: 'duplicate', event: ev });
        continue;
      }

      // Parse intent (clock-in/out) from text
      const text = ev.message.text.trim().toLowerCase();
      const isClockIn = text.includes('เข้า') || text.includes('in') || text.includes('clock in');
      const isClockOut = text.includes('ออก') || text.includes('out') || text.includes('clock out');

      if (!isClockIn && !isClockOut) {
        results.push({ status: 'ignored', event: ev });
        continue;
      }

      // Atomic upsert: one punch per user per session (no overlapping active punches for same session)
      // For clock-in: create new punch if no open punch (location_verified_at can be populated by separate flow)
      // For clock-out: close latest open punch for user
      const client = await pool.connect();
      try {
        await client.query('BEGIN');

        if (isClockIn) {
          await client.query(
            `INSERT INTO punches (
              user_id, tenant_id, clock_in, location_verified_at, idempotency_key, created_at
            ) VALUES ($1, $2, NOW(), NOW(), $3, NOW())`,
            [ev.source.userId, ev.source.tenantId || null, idemKey]
          );
        } else {
          // Close latest open punch (where clock_out IS NULL) for this user
          await client.query(
            `UPDATE punches
             SET clock_out = NOW(), updated_at = NOW()
             WHERE user_id = $1 AND clock_out IS NULL
             ORDER BY clock_in DESC
             LIMIT 1`,
            [ev.source.userId]
          );
          // Record idempotency for clock-out action too (separate key)
          const clockOutIdemKey = crypto.createHash('sha256').update(`${idemKey}:out`).digest('hex');
          await client.query(
            `INSERT INTO punches (
              user_id, tenant_id, clock_in, clock_out, idempotency_key, created_at
            ) VALUES ($1, $2, NOW(), NOW(), $3, NOW())
            ON CONFLICT DO NOTHING`,
            [ev.source.userId, ev.source.tenantId || null, clockOutIdemKey]
          );
        }

        await client.query('COMMIT');
        results.push({ status: 'ok', event: ev });
      } catch (err) {
        await client.query('ROLLBACK');
        // If unique violation on idempotency_key, treat as duplicate
        if ((err as any)?.code === '23505') {
          results.push({ status: 'duplicate', event: ev });
        } else {
          throw err;
        }
      } finally {
        client.release();
      }
    } catch (err) {
      console.error('LINE webhook processing error', { err, event: ev });
      results.push({ status: 'error', event: ev, error: String(err) });
    }
  }

  return res.json({ results });
}
```
