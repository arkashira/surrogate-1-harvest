# workio / ops

## 1. Diagnosis

- LINE retries on 5xx/timeouts with exponential backoff; webhook handler has no idempotency key enforcement → duplicate clock-in/out events create multiple open punches.
- No tenant+employee uniqueness constraint for open punches (nullable `clock_out_at`) → race conditions allow multiple concurrent “open” rows per person.
- Webhook processing is not atomic: read-then-insert pattern allows TOCTOU races under load.
- No deduplication store (idempotency key) for LINE event IDs → retries bypass app-level duplicate detection.
- Missing operational alerting/metrics for webhook error rates and duplicate events.

## 2. Proposed change

File: `/opt/axentx/workio/server/src/routes/webhook/line.ts` (or equivalent)  
Scope:  
- Add idempotency table `line_webhook_events` keyed by `line_event_id` (unique) with processed payload hash.  
- Add unique partial index on `punches(tenant_id, employee_id, clock_out_at)` where `clock_out_at IS NULL`.  
- Wrap punch handling in a single transaction that first attempts idempotency insert, then upserts punch state atomically.  
- Return 200 quickly after commit; defer heavy work (notifications) to queue/background.

## 3. Implementation

```sql
-- server/src/db/schema.sql additions
CREATE TABLE IF NOT EXISTS line_webhook_events (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  line_event_id   TEXT NOT NULL,
  tenant_id       UUID NOT NULL REFERENCES tenants(id),
  payload_hash    TEXT NOT NULL,
  processed_at    TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  CONSTRAINT uq_line_event_id UNIQUE (line_event_id)
);

-- Prevent multiple open punches per employee per tenant
CREATE UNIQUE INDEX IF NOT EXISTS uq_open_punch
ON punches (tenant_id, employee_id)
WHERE clock_out_at IS NULL;
```

```typescript
// server/src/routes/webhook/line.ts
import { Router, Request, Response } from 'express';
import { Pool } from 'pg';
import crypto from 'crypto';

const router = Router();
const pool = new Pool({ connectionString: process.env.DATABASE_URL });

function hashPayload(body: any): string {
  return crypto.createHash('sha256').update(JSON.stringify(body)).digest('hex');
}

router.post('/line', async (req: Request, res: Response) => {
  const body = req.body;
  const events = body?.events;
  if (!Array.isArray(events) || events.length === 0) return res.sendStatus(200);

  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    for (const ev of events) {
      const lineEventId = ev?.eventId;
      const type = ev?.type;
      const sourceId = ev?.source?.userId;
      const tenantId = req.tenantId; // assume set by prior middleware (e.g., via channel secret lookup)
      if (!lineEventId || !type || !sourceId || !tenantId) continue;

      const payloadHash = hashPayload(ev);

      // Idempotency guard
      const idempotencyResult = await client.query(
        `INSERT INTO line_webhook_events (line_event_id, tenant_id, payload_hash)
         VALUES ($1, $2, $3)
         ON CONFLICT (line_event_id) DO UPDATE
         SET payload_hash = EXCLUDED.payload_hash
         WHERE line_webhook_events.payload_hash <> EXCLUDED.payload_hash
         RETURNING id, (xmax = 0) AS inserted`,
        [lineEventId, tenantId, payloadHash]
      );

      const inserted = idempotencyResult.rows[0]?.inserted === true;
      // If not inserted and hash unchanged, skip processing (already handled)
      if (!inserted && idempotencyResult.rowCount) {
        const existingHash = await client.query(
          `SELECT payload_hash FROM line_webhook_events WHERE line_event_id = $1`,
          [lineEventId]
        );
        if (existingHash.rows[0]?.payload_hash === payloadHash) {
          continue;
        }
        // Hash differs (replay with different payload?) — still process but log
      }

      // Resolve employee by line_user_id (mapping table assumed)
      const empResult = await client.query(
        `SELECT id FROM employees WHERE line_user_id = $1 AND tenant_id = $2`,
        [sourceId, tenantId]
      );
      if (empResult.rows.length === 0) continue;
      const employeeId = empResult.rows[0].id;

      if (type === 'message' || type === 'postback') {
        // Atomic punch upsert: close open punch if exists, else open new one
        await client.query(`
          WITH closed AS (
            UPDATE punches
            SET clock_out_at = NOW(), updated_at = NOW()
            WHERE tenant_id = $1 AND employee_id = $2 AND clock_out_at IS NULL
            RETURNING id
          )
          INSERT INTO punches (tenant_id, employee_id, clock_in_at, clock_out_at, location, line_event_id, created_at, updated_at)
          SELECT $1, $2, NOW(), NULL, $3, $4, NOW(), NOW()
          WHERE NOT EXISTS (SELECT 1 FROM closed)
        `, [tenantId, employeeId, ev?.location?.address || '', lineEventId]);
      }
    }

    await client.query('COMMIT');
    res.sendStatus(200);
  } catch (err) {
    await client.query('ROLLBACK');
    console.error('LINE webhook processing failed', err);
    // Return 200 to prevent LINE retries on transient DB errors if desired,
    // or 500 to allow retry. Prefer 200 after idempotency to avoid double-punch on retry.
    res.sendStatus(200);
  } finally {
    client.release();
  }
});

export default router;
```

Middleware sketch to set `req.tenantId` (example):

```typescript
// server/src/middleware/lineTenant.ts
import { Request, Response, NextFunction } from 'express';
export async function resolveLineTenant(req: Request, res: Response, next: NextFunction) {
  const channelSecret = req.header('X-Line-Signature');
  // map channelSecret -> tenantId via DB (cached)
  const tenantId = await getTenantIdByChannelSecret(channelSecret);
  if (!tenantId) return res.sendStatus(401);
  req.tenantId = tenantId;
  next();
}
```

Add to route registration:

```typescript
import lineWebhook from './routes/webhook/line';
import { resolveLineTenant } from './middleware/lineTenant';
app.use('/webhook/line', resolveLineTenant, lineWebhook);
```
