# workio / frontend

## 1. Diagnosis

- **No optimistic UI for clock in/out** — every punch waits for LINE→backend→polling cycle, creating perceived lag and encouraging double-taps that trigger duplicate webhooks.
- **No local pending state or retry queue** — if the LINE webhook or network fails, the UI silently drops the punch with no feedback or recovery path.
- **No visual distinction between “last punch today” vs “open shift”** — employees can’t see at a glance whether they are currently clocked in, increasing accidental double-clocks.
- **No debounce/throttle on punch button** — rapid taps can queue multiple requests before the first resolves, compounding duplicates.
- **No offline-first posture** — on flaky mobile connections, punches can be lost instead of persisted locally and retried.

## 2. Proposed change

**Scope:** `workio/src/features/punch/PunchButton.tsx` (new) + `workio/src/features/punch/punchSlice.ts` (new) + integrate into dashboard/home view.  
**Goal:** Add optimistic UI + local retry queue for clock in/out with immediate feedback and safe de-duping.

## 3. Implementation

Create a small Redux slice for pending punches and an optimistic button that updates instantly, syncs in background, and retries on failure.

```tsx
// workio/src/features/punch/punchSlice.ts
import { createSlice, createAsyncThunk, PayloadAction } from '@reduxjs/toolkit';
import { RootState } from '../../app/store';

type Punch = {
  id: string; // local uuid
  userId: string;
  type: 'IN' | 'OUT';
  timestamp: string; // ISO
  status: 'pending' | 'synced' | 'failed';
  attempts: number;
};

type PunchState = {
  queue: Punch[];
  lastLocalPunchByUser: Record<string, Punch | undefined>;
};

const initialState: PunchState = {
  queue: [],
  lastLocalPunchByUser: {},
};

export const submitPunch = createAsyncThunk(
  'punch/submitPunch',
  async (payload: { userId: string; type: 'IN' | 'OUT' }, { dispatch, getState }) => {
    // optimistic ID
    const localId = crypto.randomUUID();
    const now = new Date().toISOString();

    const optimistic: Punch = {
      id: localId,
      userId: payload.userId,
      type: payload.type,
      timestamp: now,
      status: 'pending',
      attempts: 0,
    };

    dispatch(punchSlice.actions.addPending(optimistic));

    // Try send to backend (webhook or REST endpoint)
    try {
      const res = await fetch('/api/punch', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ userId: payload.userId, type: payload.type, timestamp: now, localId }),
      });

      if (!res.ok) throw new Error(`HTTP ${res.status}`);

      dispatch(punchSlice.actions.markSynced({ localId }));
      return { localId };
    } catch (err) {
      dispatch(punchSlice.actions.markFailed({ localId }));
      throw err;
    }
  }
);

export const punchSlice = createSlice({
  name: 'punch',
  initialState,
  reducers: {
    addPending: (state, action: PayloadAction<Punch>) => {
      state.queue.push(action.payload);
      state.lastLocalPunchByUser[action.payload.userId] = action.payload;
    },
    markSynced: (state, action: PayloadAction<{ localId: string }>) => {
      const item = state.queue.find((p) => p.id === action.payload.localId);
      if (item) {
        item.status = 'synced';
        item.attempts += 1;
      }
    },
    markFailed: (state, action: PayloadAction<{ localId: string }>) => {
      const item = state.queue.find((p) => p.id === action.payload.localId);
      if (item) {
        item.status = 'failed';
        item.attempts += 1;
      }
    },
    retryFailed: (state, action: PayloadAction<{ localId: string }>) => {
      const item = state.queue.find((p) => p.id === action.payload.localId);
      if (item) item.status = 'pending';
    },
    clearSynced: (state) => {
      state.queue = state.queue.filter((p) => p.status !== 'synced');
    },
  },
});

export const selectPendingByUser = (state: RootState, userId: string) =>
  state.punch.queue.filter((p) => p.userId === userId && p.status === 'pending');

export const selectLastPunchByUser = (state: RootState, userId: string) =>
  state.punch.lastLocalPunchByUser[userId] || null;

export default punchSlice.reducer;
```

```tsx
// workio/src/features/punch/PunchButton.tsx
import React, { useCallback } from 'react';
import { useDispatch, useSelector } from 'react-redux';
import { RootState } from '../../app/store';
import { submitPunch, selectPendingByUser, selectLastPunchByUser } from './punchSlice';
import { debounce } from 'lodash-es';

interface Props {
  userId: string;
}

export const PunchButton: React.FC<Props> = ({ userId }) => {
  const dispatch = useDispatch();
  const pending = useSelector((s: RootState) => selectPendingByUser(s, userId));
  const last = useSelector((s: RootState) => selectLastPunchByUser(s, userId));

  const isPending = pending.length > 0;
  const lastType = last?.type;
  const nextType = lastType === 'IN' ? 'OUT' : 'IN';
  const isClockedIn = lastType === 'IN' && last?.status !== 'failed';

  const handlePunch = useCallback(
    debounce(() => {
      if (isPending) return;
      dispatch(submitPunch({ userId, type: nextType }));
    }, 300),
    [isPending, nextType, userId, dispatch]
  );

  return (
    <div className="flex flex-col items-center gap-2">
      <button
        onClick={handlePunch}
        disabled={isPending}
        className={`
          px-6 py-3 rounded-lg font-semibold text-white shadow-md transition-all
          ${isPending ? 'bg-gray-400 cursor-wait' : isClockedIn ? 'bg-red-500 hover:bg-red-600' : 'bg-green-500 hover:bg-green-600'}
        `}
      >
        {isPending ? 'กำลังบันทึก...' : isClockedIn ? 'เลิกงาน (OUT)' : 'เข้างาน (IN)'}
      </button>

      {last && (
        <div className="text-xs text-gray-500">
          ครั้งล่าสุด: {lastType} {new Date(last.timestamp).toLocaleTimeString()}
          {last.status === 'failed' && (
            <button
              onClick={() => dispatch({ type: 'punch/retryFailed', payload: { localId: last.id } })}
              className="ml-2 text-red-600 underline"
            >
              retry
            </button>
          )}
        </div>
      )}
    </div>
  );
};
```
