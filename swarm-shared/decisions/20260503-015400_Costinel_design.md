# Costinel / design

## Final Implementation Plan — Top-Hub Signal Panel (Costinel Dashboard)

**Unified scope & value**  
- Frontend-only, read-only panel that surfaces the most-connected hub (default **MOC**) and its actionable proposals from the knowledge graph.  
- Ships in **<2h** as a resilient, self-contained component with **zero backend changes**.  
- Gracefully degrades when graph data is unavailable.  
- Directly applies pattern: **top-hub doc insight** (#knowledge-rag #graph #hub).

### 3. Data composable (`src/composables/useKnowledgeGraph.ts`)

```ts
import { ref, computed } from 'vue';
import type { HubNode, Proposal } from '@/types/knowledge-graph';

const cache = new Map<string, HubNode>();
const loading = ref(false);
const error = ref<Error | null>(null);

async function fetchHub(name: string): Promise<HubNode | null> {
  const cacheKey = `hub:${name}`;
  if (cache.has(cacheKey)) return cache.get(cacheKey)!;

  loading.value = true;
  error.value = null;

  try {
    // Primary: CDN-bypass JSON (fast, no auth) — aligns with HF CDN bypass pattern
    const cdnRes = await fetch(
      `https://huggingface.co/datasets/axentx/knowledge-rag/resolve/main/hubs/${name}.json`,
      { cache: 'no-store' }
    );

    if (cdnRes.ok) {
      const data = (await cdnRes.json()) as HubNode;
      cache.set(cacheKey, data);
      return data;
    }

    // Fallback: local static JSON (bundled or served from /public)
    const localRes = await fetch(`/data/hubs/${name}.json`, { cache: 'no-store' });
    if (localRes.ok) {
      const data = (await localRes.json()) as HubNode;
      cache.set(cacheKey, data);
      return data;
    }

    throw new Error(`Hub "${name}" not found on CDN or locally.`);
  } catch (err) {
    error.value = err instanceof Error ? err : new Error(String(err));
    return null;
  } finally {
    loading.value = false;
  }
}

export function useKnowledgeGraph(hubName: string) {
  const hub = ref<HubNode | null>(null);
  const proposals = computed<Proposal[]>(() => hub.value?.proposals ?? []);

  async function load() {
    hub.value = await fetchHub(hubName);
    return hub.value;
  }

  return {
    hub,
    proposals,
    loading,
    error,
    load,
  };
}
```
