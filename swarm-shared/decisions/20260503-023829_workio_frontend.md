# workio / frontend

## Final Synthesis & Actionable Implementation

**Chosen approach:** Merge the strongest, non-redundant pieces from both candidates into a single, production-ready solution.  
**Priority:** Correctness first (state-machine enforcement, idempotency, geofence), then concrete actionability (copy-paste-ready code), then UX (optimistic UI + offline queue).

---

### 1. Diagnosis (Consolidated)
- No optimistic UI → perceived latency → double-taps and race conditions.
- Missing client-side transition guards → illegal state transitions allowed in UI.
- No offline-first queue → punches lost on network failure.
- Geofence check is late/racy → rejected punches and poor UX.
- No persisted idempotency key reuse on retry → duplicate risk on retry storms and webhook replays.

---

### 2. File Changes (Minimal, focused)
- `src/features/clock/ClockButton.tsx` — optimistic UI + transition guard + geofence pre-check + queue submission.
- `src/lib/queue.ts` — offline queue with idempotency, retry, and TTL/drop policy.
- `src/hooks/useClockQueue.ts` — queue processor (online-triggered, exponential backoff).
- `src/lib/geofence.ts` — geofence helpers.
- (Optional) `src/api/clock.ts` — ensure backend call accepts `idempotencyKey`.

#### 3.1 Offline queue with idempotency and retry (`src/lib/queue.ts`)
```ts
// src/lib/queue.ts
type QueueItem = {
  id: string; // idempotency key
  action: 'clock-in' | 'clock-out';
  lat: number;
  lng: number;
  attempts: number;
  createdAt: number;
};

const STORAGE_KEY = 'workio_clock_queue';
const MAX_ATTEMPTS = 5;
const MAX_AGE_MS = 24 * 60 * 60 * 1000; // 1 day TTL

class ClockQueue {
  private items: QueueItem[] = [];

  constructor() {
    this.load();
  }

  private load() {
    try {
      const raw = localStorage.getItem(STORAGE_KEY);
      this.items = raw ? JSON.parse(raw) : [];
    } catch {
      this.items = [];
    }
  }

  private save() {
    try {
      localStorage.setItem(STORAGE_KEY, JSON.stringify(this.items));
    } catch {}
  }

  push(item: Omit<QueueItem, 'attempts' | 'createdAt'>) {
    const entry: QueueItem = { ...item, attempts: 0, createdAt: Date.now() };
    this.items.push(entry);
    this.save();
  }

  peek(): QueueItem | undefined {
    this.pruneExpired();
    return this.items[0];
  }

  pop() {
    const item = this.items.shift();
    this.save();
    return item;
  }

  removeById(id: string) {
    this.items = this.items.filter((it) => it.id !== id);
    this.save();
  }

  incrementAttempts(id: string) {
    const item = this.items.find((it) => it.id === id);
    if (item) {
      item.attempts += 1;
      this.save();
    }
  }

  shouldDrop(id: string): boolean {
    const item = this.items.find((it) => it.id === id);
    if (!item) return true;
    if (item.attempts >= MAX_ATTEMPTS) return true;
    if (Date.now() - item.createdAt > MAX_AGE_MS) return true;
    return false;
  }

  size() {
    return this.items.length;
  }

  private pruneExpired() {
    const before = this.items.length;
    this.items = this.items.filter((it) => Date.now() - it.createdAt <= MAX_AGE_MS);
    if (this.items.length !== before) this.save();
  }

  // For re-queueing with same idempotency key (e.g., after location refresh)
  updateLocation(id: string, lat: number, lng: number) {
    const item = this.items.find((it) => it.id === id);
    if (item) {
      item.lat = lat;
      item.lng = lng;
      this.save();
    }
  }

  getAll() {
    return this.items.slice();
  }
}

export const clockQueue = new ClockQueue();
```

#### 3.2 Queue processor hook (`src/hooks/useClockQueue.ts`)
```ts
// src/hooks/useClockQueue.ts
import { useEffect } from 'react';
import { clockQueue } from '../lib/queue';
import { postPunch } from '../api/clock'; // (action, lat, lng, idempotencyKey) => Promise
import { useOnline } from './useOnline';

export function useClockQueue() {
  const online = useOnline();

  useEffect(() => {
    if (!online) return;

    let active = true;
    const process = async () => {
      while (active && clockQueue.peek()) {
        const item = clockQueue.peek()!;
        if (clockQueue.shouldDrop(item.id)) {
          clockQueue.pop();
          continue;
        }

        try {
          await postPunch(item.action, item.lat, item.lng, item.id);
          clockQueue.pop(); // success -> remove
          // small delay to avoid tight loop on immediate subsequent items
          await new Promise((r) => setTimeout(r, 200));
        } catch (err: any) {
          // Network/server error -> backoff and stop this run
          clockQueue.incrementAttempts(item.id);
          const delay = Math.min(30_000, 1000 * 2 ** item.attempts);
          await new Promise((r) => setTimeout(r, delay));
          break;
        }
      }
    };

    process();
    return () => {
      active = false;
    };
  }, [online]);
}
```

// src/lib/geofence.ts
type Bounds = { north: number; south: number; east: number; west: number };

export function isInsideBounds(pos: GeolocationPosition, bounds: Bounds | null): boolean {
  if (!bounds) return true;
  const { latitude, longitude } = pos.coords;
  return latitude <= bounds.north && latitude >= bounds.south && longitude <= bounds.east && longitude >= bounds.west;
}

export function getCurrentPosition(): Promise<GeolocationPosition> {
  return new Promise((resolve, reject) => {
    if (!navigator.geolocation) {
      reject(new Error('Geolocation not supported'));
      return;
    }
    navigator.geolocation.getCurrentPosition(resolve, reject, {
      enableHighAccuracy: true,
      timeout: 10000,
      maximumAge: 30_000,
    });
  });
}
