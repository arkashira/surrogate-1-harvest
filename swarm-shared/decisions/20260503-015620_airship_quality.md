### 2.1 `scripts/list_hf_files.py`
```python
#!/usr/bin/env python3
"""
Mac-side: run once per date folder after rate-limit window.
Outputs file_list.json for downstream CDN ingestion.
"""
import json
import os
import sys
from huggingface_hub import HfApi

REPO = os.getenv("HF_DATASET_REPO", "my-org/surrogate-mirror")
DATE_FOLDER = sys.argv[1] if len(sys.argv) > 1 else "2026-05-01"
OUT = sys.argv[2] if len(sys.argv) > 2 else "file_list.json"

api = HfApi()
tree = api.list_repo_tree(repo_id=REPO, path=DATE_FOLDER, recursive=False)
files = [item.rfilename for item in tree if item.type == "file"]

with open(OUT, "w") as f:
    json.dump({"date": DATE_FOLDER, "repo": REPO, "files": files}, f, indent=2)

print(f"Wrote {len(files)} files to {OUT}")
```

### 2.2 `scripts/build_shard_map.py`
```python
#!/usr/bin/env python3
"""
Deterministic sibling sharding to bypass HF 128/hr commit cap.
Usage: python build_shard_map.py N_SIBLINGS file_list.json shard_map.json
"""
import hashlib
import json
import sys

N_SIBLINGS = int(sys.argv[1]) if len(sys.argv) > 1 else 5
FILE_LIST = sys.argv[2] if len(sys.argv) > 2 else "file_list.json"
OUT = sys.argv[3] if len(sys.argv) > 3 else "shard_map.json"

with open(FILE_LIST) as f:
    data = json.load(f)

BASE_REPO = data["repo"]
shard_map = {}
for file in data["files"]:
    digest = hashlib.sha256(file.encode()).hexdigest()
    idx = int(digest, 16) % N_SIBLINGS
    target = BASE_REPO if idx == 0 else f"{BASE_REPO}-shard{idx}"
    shard_map[file] = target

with open(OUT, "w") as f:
    json.dump(shard_map, f, indent=2)

print(f"Mapped {len(shard_map)} files across {N_SIBLINGS} siblings")
```

### 2.3 `surrogate/ingest/mirror.py` (CDN-only ingestion)
```python
import json
import os
from pathlib import Path
from typing import Any, Dict, List

import pandas as pd
import requests
from huggingface_hub import HfApi

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def download_via_cdn(file_path: str, repo: str, out_dir: Path) -> Path:
    out_dir.mkdir(parents=True, exist_ok=True)
    url = CDN_TEMPLATE.format(repo=repo, path=file_path)
    out_path = out_dir / Path(file_path).name
    if out_path.exists():
        return out_path
    resp = requests.get(url, timeout=30)
    resp.raise_for_status()
    out_path.write_bytes(resp.content)
    return out_path

def project_to_prompt_response(raw_path: Path) -> List[Dict[str, str]]:
    """
    Project heterogeneous files to canonical {prompt, response}.
    Extend per your formats (JSONL, JSON, etc.).
    """
    examples = []
    text = raw_path.read_text().strip()
    if not text:
        return examples

    # JSONL
    if raw_path.suffix == ".jsonl":
        for line in text.splitlines():
            line = line.strip()
            if not line:
                continue
            obj = json.loads(line)
            examples.append({
                "prompt": obj.get("prompt", ""),
                "response": obj.get("response", "")
            })
        return examples

    # Single JSON array or object
    if raw_path.suffix == ".json":
        obj = json.loads(text)
        if isinstance(obj, list):
            for item in obj:
                examples.append({
                    "prompt": item.get("prompt", ""),
                    "response": item.get("response", "")
                })
        else:
            examples.append({
                "prompt": obj.get("prompt", ""),
                "response": obj.get("response", "")
            })
        return examples

    # Fallback: plain text heuristics (customize as needed)
    # For now, return empty to avoid garbage.
    return examples

def ingest_cdn_only(file_list_json: str, shard_map_json: str, out_root: str = "enriched") -> None:
    with open(file_list_json) as f:
        file_data = json.load(f)
    with open(shard_map_json) as f:
        shard_map = json.load(f)

    out_root = Path(out_root)
    api = HfApi()

    for file_path in file_data["files"]:
        src_repo = file_data["repo"]
        target_repo = shard_map.get(file_path, src_repo)

        # 1) CDN download (no auth/API rate limits)
        raw_file = download_via_cdn(file_path, src_repo, out_root / "raw")

        # 2) Project to canonical {prompt, response}
        examples = project_to_prompt_response(raw_file)
        if not examples:
            continue

        # 3) Save as parquet (HF-friendly)
        df = pd.DataFrame(examples)
        parquet_name = raw_file.with_suffix(".parquet").name
        parquet_path = (
            out_root / "batches" / "mirror-merged" / file_data["date"] / parquet_name
        )
        parquet_path.parent.mkdir(parents=True, exist_ok=True)
        df.to_parquet(parquet_path, index=False)

        # 4) Upload to sibling repo (deterministic sharding)
        try:
            api.upload_file(
                path_or_fileobj=str(parquet_path),
                path_in_repo=f"batches/mirror-merged/{file_data['date']}/{parquet_name}",
                repo_id=target_repo,
                commit_message=f"Add {parquet_name}"
            )
        except Exception as e:
            # If 429/128 cap hit, log and continue; CDN downloads already succeeded
            print(f"Upload skipped for {parquet_path} to {target_repo}: {e}")

if __name__ == "__main__":
    import sys
    if len(sys.argv) != 3:
        print("Usage: python -m surrogate.ingest.mirror file_list.json shard_map.json")
        sys.exit(1)
