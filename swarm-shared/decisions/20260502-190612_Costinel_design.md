# Costinel / design

## Final Synthesis — Deterministic Discovery Module (≤2h)

**Goal:**  
Ship a lightweight, deterministic, read-only discovery module that immediately unblocks visibility and governance workflows while strictly adhering to “Sense + Signal — ไม่ Execute.”

Produces:
- `manifests/sources.json` — declared cloud sources + auth hints (no secrets).
- `manifests/discovery.json` — discovered resources with topology links, deterministic ordering, and stable hashes.
- `manifests/top-hub-insight.json` — single most-connected hub snapshot (degree centrality) for human review and downstream prioritization.

## 3) `src/discovery/sources.py`
```python
from __future__ import annotations

import json
from pathlib import Path
from typing import Any, Dict, List
from pydantic import BaseModel, Field, field_validator

MANIFESTS_DIR = Path(__file__).parents[2] / "manifests"

class Auth(BaseModel):
    method: str
    env_prefix: str | None = None

class Source(BaseModel):
    id: str
    provider: str
    account_id: str | None = None
    project_id: str | None = None
    regions: List[str]
    auth: Auth
    tags: List[str] = Field(default_factory=list)

    @field_validator("provider")
    def provider_must_be_supported(cls, v: str) -> str:
        allowed = {"aws", "gcp", "azure"}
        if v not in allowed:
            raise ValueError(f"provider must be one of {allowed}")
        return v

class SourcesManifest(BaseModel):
    version: str = "1.0"
    sources: List[Source]

def load_sources(manifest_path: Path | None = None) -> SourcesManifest:
    p = manifest_path or (MANIFESTS_DIR / "sources.json")
    if not p.is_file():
        raise FileNotFoundError(f"sources manifest not found: {p}")
    return SourcesManifest.model_validate_json(p.read_text())
```

## 4) Deterministic Graph Utilities (`src/discovery/graph.py`)
```python
from __future__ import annotations

import hashlib
import json
from collections import Counter
from typing import Any, Dict, List

def stable_hash(obj: Any) -> str:
    payload = json.dumps(obj, sort_keys=True, ensure_ascii=False).encode()
    return hashlib.sha256(payload).hexdigest()[:16]

def build_topology_links(discovery: Dict[str, Any]) -> List[Dict[str, str]]:
    """
    Build deterministic topology links between entities.
    Returns list of {src, dst, type}.
    """
    links: List[Dict[str, str]] = []
    discoveries = discovery.get("discoveries", [])

    for d in discoveries:
        provider = d.get("provider")
        account_id = d.get("account_id") or d.get("declared_account_id")
        regions = d.get("regions", [])
        services = d.get("services_discovered", [])

        if not account_id:
            continue

        # account node
        account_node = {"type": "account", "id": account_id, "provider": provider}

        for region in sorted(regions):
            region_node = {"type": "region", "id": region, "provider": provider}
            links.append({"src": stable_hash(account_node), "dst": stable_hash(region_node), "type": "contains"})

            for svc in sorted(services):
                svc_node = {"type": "service", "id": svc, "provider": provider, "region": region}
                links.append({"src": stable_hash(region_node), "dst": stable_hash(svc_node), "type": "hosts"})

    return links

def degree_centrality(links: List[Dict[str, str]]) -> Dict[str, int]:
    counts: Counter = Counter()
    for link in links:
        counts[link["src"]] += 1
        counts[link["dst"]] += 1
    return dict(counts)

def top_hub_from_degree(centrality: Dict[str, int]) -> Dict[str, Any]:
    if not centrality:
        return {"node_id": None, "degree": 0, "note": "no topology available"}
    top_id = max(centrality, key=centrality.get)
    return {"node_id": top_id, "degree": centrality[top_id]}
```

## 5) Cloud Discovery Stubs and AWS Example (`src/discovery/cloud/aws.py`)
```python
from __future__ import annotations

import os
from typing import Any, Dict, List

try:
    import boto3
    from botocore.exceptions import ClientError
except ImportError:
    boto3 = None

def discover_aws(source: Dict[str, Any]) -> Dict[str, Any]:
    if boto3 is None:
        return {"error": "boto3 not installed", "provider": "aws"}

    method = source["auth"]["method"]
    env_prefix = source["auth"].get("env_prefix", "AWS")

    session = boto3.Session(
        region_name=source["regions"][0] if source["regions"] else None
    )

    sts = session.client("sts")
    try:
        identity = sts.get_caller_identity()
    except ClientError as exc:
        return {"error": str(exc), "provider": "aws"}

    account_id = identity["Account"]
    regions = source["regions"]

    services: List[str] = []
    buckets: List[Dict[str, Any]] = []

    # Lightweight region/service probe (read-only)
    ec2 = session.client("ec2", region_name=regions[0]) if regions else None
    if ec2:
        try:
            ec2.describe_regions(AllRegions=True)
            services.append("ec2")
        except ClientError:
            pass

    s3 = session.client("s3")
    try:
        out = s3.list_buckets()
        buckets = [{"name": b["Name"], "creation_date": str(b.get("CreationDate"))} for b in out.get("Buckets", [])]
    except ClientError:
        pass

    return {
        "provider": "aws",
        "account_id": account_id,
        "declared_account_id": source.get("account_id"),
        "regions": regions,
        "services_discovered": services,
        "buckets": buckets,
        "tags": source.get("tags", []),
    }
```

(GCP/Azure stubs return `{"provider": "...", "discovered": []}` until implemented.)
