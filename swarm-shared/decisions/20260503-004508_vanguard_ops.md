# vanguard / ops

## Final Synthesis (Best Parts + Correctness + Actionability)

**Core diagnosis (merged, de-duplicated):**
- No persisted `(repo, dateFolder)` file manifest → every training run re-enumerates via authenticated HF API → burns quota and risks 429.
- Recursive `list_repo_files` / `load_dataset(streaming=True)` on heterogeneous repo schema triggers `pyarrow.CastError` and amplifies rate-limit pressure.
- Lightning Studio lifecycle is not idempotent: scripts recreate studios instead of reusing running ones → wastes ~80 hr/mo quota.
- Training script likely performs HF API data loads (or CDN fetches without a precomputed file list) during training → stalls on any API hiccup.
- No deterministic repo selection for commit-cap mitigation (all writes target one repo; risk hitting 128/hr cap).
- No guardrails for rate-limit or schema heterogeneity.

**Chosen strategy (resolved in favor of correctness + concrete actionability):**
1. Add a single, deterministic manifest layer that is generated once (out-of-band) and consumed during training (zero authenticated HF API calls).
2. Patch the Lightning launcher to reuse running studios (idempotent lifecycle).
3. Use CDN-only URLs for data fetches; avoid recursive listing and heterogeneous schema parsing during training.
4. Add deterministic repo selection (5 siblings) to spread commit traffic and avoid per-repo commit cap.
5. Keep verification minimal and reproducible.

## Implementation

### 1) New file: `/opt/axentx/vanguard/ops/training_manifest.py`
```python
#!/usr/bin/env python3
"""
Generate and consume a persisted manifest for (repo, dateFolder) so training
does zero authenticated HF API calls during data loading.

Usage:
  # One-time (after rate-limit window clears):
  python ops/training_manifest.py --repo surrogate-1 --date 2026-04-29 --out manifest.json

  # In Lightning training script (zero API calls):
  from ops.training_manifest import load_manifest, iter_cdn_urls
  urls = list(iter_cdn_urls("manifest.json"))
"""

from __future__ import annotations

import argparse
import hashlib
import json
import os
import sys
from pathlib import Path
from typing import Any, Dict, Iterable, List

import requests

HF_API_BASE = "https://huggingface.co/api"
HF_CDN_BASE = "https://huggingface.co/datasets"

# Deterministic repo selection for commit-cap mitigation (5 siblings)
SIBLING_REPOS = [
    "axentx/surrogate-1",
    "axentx/surrogate-1-sib1",
    "axentx/surrogate-1-sib2",
    "axentx/surrogate-1-sib3",
    "axentx/surrogate-1-sib4",
]

def pick_repo(slug: str) -> str:
    """Deterministically pick a sibling repo by hash of slug."""
    idx = int(hashlib.sha256(slug.encode()).hexdigest(), 16) % len(SIBLING_REPOS)
    return SIBLING_REPOS[idx]

def list_date_folder(repo: str, date_folder: str, token: str | None = None) -> List[str]:
    """
    Non-recursive folder listing (single API call).
    Returns repo-relative paths.
    """
    url = f"{HF_API_BASE}/{repo}/tree/{date_folder}"
    headers = {}
    if token:
        headers["Authorization"] = f"Bearer {token}"
    resp = requests.get(url, headers=headers, timeout=30)
    if resp.status_code == 429:
        print("Rate limited (429). Wait 360s before retry.", file=sys.stderr)
        raise RuntimeError("HF API 429")
    resp.raise_for_status()
    items: List[Dict[str, Any]] = resp.json()
    # Only top-level files in this date folder (no recursion)
    paths = [item["path"] for item in items if item.get("type") == "file"]
    return paths

def build_manifest(
    repo: str,
    date_folder: str,
    out_path: Path,
    token: str | None = None,
) -> Dict[str, Any]:
    """Generate and persist manifest."""
    paths = list_date_folder(repo, date_folder, token=token)
    manifest = {
        "repo": repo,
        "date_folder": date_folder,
        "paths": sorted(paths),
        "cdn_base": HF_CDN_BASE,
    }
    out_path.write_text(json.dumps(manifest, indent=2))
    return manifest

def load_manifest(manifest_path: Path | str) -> Dict[str, Any]:
    with open(manifest_path) as f:
        return json.load(f)

def iter_cdn_urls(manifest_path: Path | str) -> Iterable[str]:
    manifest = load_manifest(manifest_path)
    repo = manifest["repo"]
    date_folder = manifest["date_folder"]
    base = manifest["cdn_base"].rstrip("/")
    for p in manifest["paths"]:
        # CDN download bypasses authenticated API rate limits
        yield f"{base}/{repo}/resolve/main/{date_folder}/{p}"

def main() -> None:
    parser = argparse.ArgumentParser(description="Build training manifest to avoid HF API during training.")
    parser.add_argument("--repo", default="surrogate-1", help="HF dataset repo (default: surrogate-1)")
    parser.add_argument("--date", required=True, help="Date folder (e.g., 2026-04-29)")
    parser.add_argument("--out", default="manifest.json", help="Output manifest path")
    parser.add_argument("--token", default=os.getenv("HF_TOKEN"), help="HF token (optional; only for listing)")
    args = parser.parse_args()

    repo = pick_repo(args.date) if "surrogate-1" in args.repo else args.repo
    out = Path(args.out).expanduser().resolve()
    out.parent.mkdir(parents=True, exist_ok=True)

    print(f"Building manifest for {repo}/{args.date} -> {out}")
    build_manifest(repo=repo, date_folder=args.date, out_path=out, token=args.token)
    print("Done. Use iter_cdn_urls(manifest_path) in training for CDN-only fetches.")

if __name__ == "__main__":
    main()
```

### 2) Patch: Lightning launcher (idempotent reuse)

If no launcher exists, create `/opt/axentx/vanguard/ops/launch_training.py`:

```python
#!/usr/bin/env python3
"""
Idempotent Lightning Studio launcher: reuse running studios to save quota.
"""

from pathlib import Path
import lightning as L

def launch_training(
    script_path: Path,
    studio_name: str = "vanguard-surrogate-train",
    machine: str = "L40S",
    cloud: str = "lightning-public-prod",
) -> None:
    teamspace = L.Teamspace()
    running = None
    for s in teamspace.studios():
        if s.name == studio_name and s.status == "running":
            running = s
            break

    if running:
        print(f"Reusing running studio: {studio_name}")
        studio = running
    else:
        print(f"Creating studio: {studio_name}")
        studio = L.Studio(
            name=studio_name,
            script_path=str(script_path),
            machine=machine,
            cloud=cloud,
            create_ok=True,
        )

    # Ensure studio is running before submitting work
    if studio.status != "running":
        print(f"Studio not running (status={studio.status}); starting...")
        studio.start(machine=machine)

    # Submit run (idempotent; will queue)
    run = studio.run(script_args=[])
    print(f"Submitted run: {run.name}")

if __name__ == "__main__":
    repo_root = Path(__file__).parent.parent
    script = repo_root / "train.py"
    launch_training(script_path=script)
```

If an existing launcher is present, patch it to include the reuse logic above (list `Teamspace.studios()`, reuse if running, else create).
