## 1) Mac-side: prepare CDN file list (run once per date folder)

`surrogate/training/prepare_filelist.py`

```python
#!/usr/bin/env python3
"""
Run on Mac (or any dev host) after HF rate-limit window clears.
Produces train_filelist.json for a single date folder.
Usage:
  python prepare_filelist.py --repo surrogate-dataset --date 2026-04-29 --out train_filelist.json
"""

import argparse
import json
import os
import time
from huggingface_hub import HfApi

HF_TOKEN = os.getenv("HF_TOKEN")

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g., org/surrogate-dataset)")
    parser.add_argument("--date", required=True, help="Date folder under batches/mirror-merged/YYYY-MM-DD")
    parser.add_argument("--out", default="train_filelist.json", help="Output JSON path")
    args = parser.parse_args()

    api = HfApi(token=HF_TOKEN)
    folder = f"batches/mirror-merged/{args.date}"
    # Non-recursive list to avoid pagination explosion
    entries = api.list_repo_tree(repo_id=args.repo, path=folder, recursive=False)

    files = []
    for e in entries:
        if not e.path.endswith(".parquet"):
            continue
        # CDN URL (no auth, bypasses API rate limits during training)
        cdn_url = f"https://huggingface.co/datasets/{args.repo}/resolve/main/{e.path}"
        files.append({"path": e.path, "cdn_url": cdn_url})

    payload = {
        "repo": args.repo,
        "date": args.date,
        "generated_at": time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime()),
        "files": files,
    }

    os.makedirs(os.path.dirname(os.path.abspath(args.out)) if os.path.dirname(args.out) else ".", exist_ok=True)
    with open(args.out, "w") as f:
        json.dump(payload, f, indent=2)

    print(f"Wrote {len(files)} files to {args.out}")

if __name__ == "__main__":
    main()
```

Make it executable (optional):
```bash
chmod +x surrogate/training/prepare_filelist.py
```

## 2) Training: zero-API dataloader (CDN-only)

Patch `surrogate/training/train.py` (or create if missing). Key points:

- Load `train_filelist.json` produced above.
- Use `pd.read_parquet(cdn_url)` or `pyarrow.parquet` with HTTP filesystem (streaming).
- No `load_dataset` or recursive HF API calls during training.

Minimal patch snippet:

```python
import json
import pandas as pd
import pyarrow.parquet as pq
import requests
from io import BytesIO
from torch.utils.data import Dataset, DataLoader

class CDNParquetDataset(Dataset):
    def __init__(self, filelist_path, max_rows=None):
        with open(filelist_path) as f:
            manifest = json.load(f)
        self.rows = []
        for item in manifest["files"]:
            url = item["cdn_url"]
            # Simple HTTP fetch (CDN, no auth)
            resp = requests.get(url, timeout=60)
            resp.raise_for_status()
            table = pq.read_table(BytesIO(resp.content))
            df = table.to_pandas()
            # Project to {prompt, response} only
            for _, row in df.iterrows():
                prompt = row.get("prompt")
                response = row.get("response")
                if prompt and response:
                    self.rows.append({"prompt": prompt, "response": response})
                    if max_rows and len(self.rows) >= max_rows:
                        return

    def __len__(self):
        return len(self.rows)

    def __getitem__(self, idx):
        return self.rows[idx]
```

Use in training loop:
```python
train_ds = CDNParquetDataset("train_filelist.json")
loader = DataLoader(train_ds, batch_size=8, shuffle=True)
```

## 3) Lightning launcher with auto-recovery and studio reuse

`surrogate/training/lightning_launcher.py`

```python
#!/usr/bin/env python3
"""
Lightning launcher with:
- Studio reuse (running ones preferred)
- Idle-stop recovery (restart if stopped)
- Runs training with CDN-only filelist
"""

import time
import subprocess
import lightning as L

def find_running_studio(name="surrogate-train"):
    teamspace = L.Teamspace()
    for s in teamspace.studios:
        if s.name == name and s.status == "running":
            return s
    return None

def run_training_on_lightning(filelist_path="train_filelist.json", max_epochs=1):
    studio = find_running_studio()
    if studio is None:
        # Start new on L40S (free tier fallback to available)
        studio = L.Studio(
            name="surrogate-train",
            create_ok=True,
            machine=L.Machine.L40S,
        )

    script = "train.py"  # expects train.py in same folder
    cmd = [
        "python", script,
        "--filelist", filelist_path,
        "--epochs", str(max_epochs),
    ]

    # Ensure studio is running
    if studio.status != "running":
        print(f"Studio stopped ({studio.status}). Restarting...")
        studio.start(machine=L.Machine.L40S)
        # Wait until running
        for _ in range(30):
            time.sleep(10)
            studio.refresh()
            if studio.status == "running":
                break
        else:
            raise RuntimeError("Studio failed to start")

    # Run training
    job = studio.run(cmd, cwd=".", detach=False)
    return job

if __name__ == "__main__":
    run_training_on_lightning()
```

Notes:
- Uses `L.Studio.run` (or `subprocess` to `lightning run` depending on SDK version).
- Checks status and restarts on idle-stop.
- Prefers running studio to save quota.

## 4) Cron / scheduling guard (wrapper best practices)

If you schedule via cron, ensure:

- Shebang `#!/usr/bin/env bash` on wrapper scripts.
- `chmod +x` on scripts.
- Set `SHELL=/bin/bash` in crontab.
- Invoke via `bash script.sh "$@"` if calling from cron.

Example wrapper `surrogate/training/run_training.sh`:

```bash
#!/usr/bin/env bash
set -euo pipefail
cd "$(dirname "$0")/.."

# Optional: prepare filelist only if missing or stale
if [ ! -f train_filelist.json ]; then
  echo "Preparing CDN filelist..."
  python surrogate/training/prepare_filelist.py --repo surrogate-dataset --date 2026-04-29 --out train_filelist.json
fi

# Launch Lightning training with auto-recovery
python surrogate/training/lightning_launcher.py
```

Crontab line (example):
```
SHELL=/bin/bash
0 2 * * * /opt/axentx/airship/surrogate/training/run_training.sh >> /var/log/airship_training.log 2>&1
```
