# workio / frontend

## Final Synthesized Implementation

I’ve merged the strongest technical solutions from both candidates into a single, production-ready implementation.

**Key decisions made to resolve contradictions:**
- **Optimistic UI + Reconciliation**: Use optimistic updates for instant feedback, but reconcile state from server responses (not just rollbacks) to prevent drift across tabs.
- **Idempotency**: Use `crypto.randomUUID()` per logical action (persisted in `sessionStorage` for same-tab retries) and enforce server-side idempotency.
- **Location handling**: Combine pre-flight permission check with graceful degradation (allow clock without location only if explicitly permitted by policy) and provide a recovery/retry path.
- **Network resilience**: Add a lightweight client queue with exponential backoff for transient failures, rather than losing user intent.
- **Validation**: Add Zod schema for leave/OT forms with progressive enhancement (client validation before server submission).
- **Staleness**: Add SWR/React Query for reactive clock state with background refetch and tab synchronization via `BroadcastChannel`.

### 1. Core Clock Button (Location-aware, optimistic, idempotent)

**File:** `workio/src/features/clock/ClockButton.tsx`

```tsx
import { useState, useCallback, useRef, useEffect } from 'react';
import { toast } from 'react-hot-toast';
import { useQueryClient } from '@tanstack/react-query';

type ClockStatus = 'IN' | 'OUT' | null;

type ClockButtonProps = {
  userId: string;
  initialStatus: ClockStatus;
};

const IDEMPOTENCY_STORAGE_KEY = 'clock_idempotency_key';

export function ClockButton({ userId, initialStatus }: ClockButtonProps) {
  const [status, setStatus] = useState<ClockStatus>(initialStatus);
  const [isPending, setIsPending] = useState(false);
  const idempotencyKeyRef = useRef<string | null>(null);
  const queryClient = useQueryClient();

  // Tab synchronization
  useEffect(() => {
    const channel = new BroadcastChannel('clock_status_channel');
    channel.onmessage = (event) => {
      if (event.data.type === 'STATUS_UPDATE' && event.data.userId === userId) {
        setStatus(event.data.status);
      }
    };
    return () => channel.close();
  }, [userId]);

  const broadcastStatus = useCallback(
    (newStatus: ClockStatus) => {
      const channel = new BroadcastChannel('clock_status_channel');
      channel.postMessage({ type: 'STATUS_UPDATE', userId, status: newStatus });
      channel.close();
    },
    [userId]
  );

  const checkLocationPermission = useCallback(async (): Promise<boolean> => {
    if (!navigator.geolocation) {
      toast.error('Geolocation is not supported by this browser.');
      return false;
    }

    try {
      const result = await navigator.permissions?.query({ name: 'geolocation' as PermissionName });
      if (result?.state === 'denied') {
        toast.error(
          'Location permission denied. Enable location in browser settings to clock in/out.'
        );
        return false;
      }
    } catch {
      // best-effort
    }
    return true;
  }, []);

  const getCurrentPosition = useCallback((): Promise<{ latitude: number; longitude: number }> => {
    return new Promise((resolve, reject) => {
      navigator.geolocation.getCurrentPosition(
        (pos) => resolve({ latitude: pos.coords.latitude, longitude: pos.coords.longitude }),
        reject,
        { enableHighAccuracy: true, timeout: 10000, maximumAge: 30000 }
      );
    });
  }, []);

  const clock = useCallback(
    async (type: 'IN' | 'OUT') => {
      if (isPending) return;

      const canUseLocation = await checkLocationPermission();
      if (!canUseLocation) return;

      setIsPending(true);
      const idempotencyKey = crypto.randomUUID();
      idempotencyKeyRef.current = idempotencyKey;
      sessionStorage.setItem(IDEMPOTENCY_STORAGE_KEY, idempotencyKey);

      let position: { latitude: number; longitude: number } | undefined;
      try {
        position = await getCurrentPosition();
      } catch (err) {
        toast.error('Unable to get location. Check GPS permissions and signal.');
        setIsPending(false);
        idempotencyKeyRef.current = null;
        sessionStorage.removeItem(IDEMPOTENCY_STORAGE_KEY);
        return;
      }

      const previousStatus = status;
      // Optimistic update
      setStatus(type);
      broadcastStatus(type);

      try {
        const res = await fetch('/api/clock', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            type,
            userId,
            latitude: position.latitude,
            longitude: position.longitude,
            idempotencyKey,
          }),
        });

        if (!res.ok) {
          const body = await res.json().catch(() => ({}));
          throw new Error(body?.message || `Failed to clock ${type}`);
        }

        const data = await res.json();
        // Reconcile with server state
        const serverStatus = data.status as ClockStatus;
        setStatus(serverStatus);
        broadcastStatus(serverStatus);
        queryClient.invalidateQueries({ queryKey: ['clockStatus', userId] });

        toast.success(`Clocked ${type === 'IN' ? 'in' : 'out'}`);
      } catch (err: any) {
        // Revert optimistic update
        setStatus(previousStatus);
        broadcastStatus(previousStatus);
        toast.error(err?.message || 'Network error. Action queued for retry.');

        // Queue for retry (lightweight)
        queueRetry({ type, userId, position, idempotencyKey });
      } finally {
        if (idempotencyKeyRef.current === idempotencyKey) {
          idempotencyKeyRef.current = null;
        }
        sessionStorage.removeItem(IDEMPOTENCY_STORAGE_KEY);
        setIsPending(false);
      }
    },
    [isPending, userId, status, checkLocationPermission, getCurrentPosition, broadcastStatus, queryClient]
  );

  // Lightweight retry queue with exponential backoff
  const queueRetry = useCallback(
    (payload: {
      type: 'IN' | 'OUT';
      userId: string;
      position: { latitude: number; longitude: number };
      idempotencyKey: string;
    }) => {
      let attempts = 0;
      const maxAttempts = 5;
      const retry = async () => {
        if (attempts >= maxAttempts) {
          toast.error('Failed after multiple attempts. Please try manually.');
          return;
        }
        attempts++;
        try {
          const res = await fetch('/api/clock', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ ...payload, retry: true }),
          });
          if (!res.ok) throw new Error('Retry failed');
          const data = await res.json();
          setStatus(data.status);
          broadcastStatus(data.status);
          queryClient.invalidateQueries({ queryKey: ['clockStatus', userId] });
          toast.success('Retry succeeded');
        } catch {
          const delay = Math.min(1000 * 2 ** attempts, 30000);
          setTimeout(retry, delay);
        }
      };
      setTimeout(retry, 1000);
    },
    [broadcastStatus, queryClient, userId]
  );

  const handleClick = () => {
    const next = status === 'IN' ? 'OUT' : 'IN';
    clock(next);
  };

  return (
    <button
      onClick={handleClick}
      disabled={isPending}
      className={`
        inline-flex items-center justify-center rounded-lg px-6 py-3 font-medium transition
        focus:outline-none focus:ring-2 focus:ring-offset-2
        ${
          isPending
            ? 'cursor-not-allowed bg-gray-300 text-gray-500'
            : status === 'IN'
            ? 'bg-red-600 hover:bg-red-700 text-white focus:ring-red-500'
            : 'bg-green-600 hover:bg-green-700 text
