# workio / frontend

## Final Synthesized Solution

### Diagnosis (merged)
- **Offline/network loss**: current `fetch` has no retry/backoff; flaky mobile networks drop punches.
- **No client-side idempotency**: double-taps and retries can create duplicate punches for the same `(employee, punch_type, time_window)`.
- **No durable queue**: in-memory/`localStorage` state is lost on page reload/app kill.
- **No visibility/background recovery**: requests don’t resume when app returns to foreground.
- **No observability**: users/ops can’t see pending/failed counts.

`workio/src/lib/services/attendance.ts`
```ts
import { openDB, DBSchema, IDBPDatabase } from 'idb';

interface AttendanceDB extends DBSchema {
  punches: {
    key: string; // idempotency key: `${employeeId}-${punchType}-${yyyyMMddHHmm}`
    value: {
      key: string;
      employeeId: string;
      punchType: 'in' | 'out';
      timestamp: string; // ISO
      location?: { lat: number; lng: number };
      retries: number;
      createdAt: number;
    };
  };
}

const DB_NAME = 'workio-attendance';
const MAX_RETRIES = 5;
const BASE_DELAY = 1000; // 1s

let flushPromise: Promise<void> | null = null;

async function getDB() {
  return openDB<AttendanceDB>(DB_NAME, 1, {
    upgrade(db) {
      db.createObjectStore('punches', { keyPath: 'key' });
    },
  });
}

function sleep(ms: number) {
  return new Promise((r) => setTimeout(r, ms));
}

// Deterministic idempotency key (minute granularity)
function makeIdempotencyKey(employeeId: string, punchType: 'in' | 'out', timestamp: string) {
  return `${employeeId}-${punchType}-${new Date(timestamp).toISOString().slice(0, 16).replace(/[-:T]/g, '')}`;
}

// Enqueue punch locally and attempt immediate flush
export async function enqueuePunch(
  employeeId: string,
  punchType: 'in' | 'out',
  timestamp: string,
  location?: { lat: number; lng: number }
) {
  const db = await getDB();
  const key = makeIdempotencyKey(employeeId, punchType, timestamp);
  const record = { key, employeeId, punchType, timestamp, location, retries: 0, createdAt: Date.now() };
  await db.put('punches', record);
  // best-effort immediate flush (non-blocking)
  void flushPunches();
  return key;
}

// Try to send all pending punches; safe to call concurrently (idempotent server + local dedupe)
export async function flushPunches() {
  // Prevent concurrent flushes from stepping on each other's transaction state
  if (flushPromise) return flushPromise;

  flushPromise = (async () => {
    const db = await getDB();
    const tx = db.transaction('punches', 'readwrite');
    const store = tx.objectStore('punches');
    const all = await store.getAll();

    // Process sequentially to avoid interleaving delete/put races
    for (const item of all) {
      try {
        const res = await fetch('/api/attendance/punch', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            employeeId: item.employeeId,
            punchType: item.punchType,
            timestamp: item.timestamp,
            location: item.location,
            idempotencyKey: item.key,
          }),
        });

        if (res.ok) {
          await store.delete(item.key);
        } else {
          item.retries += 1;
          if (item.retries >= MAX_RETRIES) {
            // give up and remove to avoid unbounded growth; UI can surface failed items separately if desired
            await store.delete(item.key);
          } else {
            await store.put(item);
          }
        }
      } catch (err) {
        // network error
        item.retries += 1;
        if (item.retries >= MAX_RETRIES) {
          await store.delete(item.key);
        } else {
          await store.put(item);
        }
      }
    }

    await tx.done;

    // If anything remains, schedule retry with backoff
    const remaining = await store.count();
    if (remaining > 0) {
      // jittered backoff per flush cycle
      const delay = BASE_DELAY * Math.pow(2, Math.min(6, remaining)) + Math.random() * 500;
      setTimeout(() => void flushPunches(), delay);
    }
  })().finally(() => {
    flushPromise = null;
  });

  return flushPromise;
}

// Visibility change replay
if (typeof document !== 'undefined') {
  document.addEventListener('visibilitychange', () => {
    if (document.visibilityState === 'visible') void flushPunches();
  });
  // periodic background flush while page open
  setInterval(flushPunches, 15000);
}

// UI helper: get pending count
export async function getPendingCount(): Promise<number> {
  const db = await getDB();
  return db.getAllKeys('punches').then((keys) => keys.length);
}
```

`workio/src/components/AttendanceButton.tsx`
```tsx
'use client';
import { useState, useEffect } from 'react';
import { enqueuePunch, getPendingCount } from '@/lib/services/attendance';

export function AttendanceButton({ employeeId }: { employeeId: string }) {
  const [pending, setPending] = useState(0);
  const [processing, setProcessing] = useState(false);

  useEffect(() => {
    const update = () => void getPendingCount().then(setPending);
    update();
    const id = setInterval(update, 5000);
    return () => clearInterval(id);
  }, []);

  const punch = async (type: 'in' | 'out') => {
    setProcessing(true);
    try {
      const now = new Date().toISOString();
      // location optional; could pass geolocation here
      await enqueuePunch(employeeId, type, now);
      // optimistic local UI update; flush runs automatically
    } finally {
      setProcessing(false);
    }
  };

  return (
    <div className="space-y-2">
      <div className="flex gap-2">
        <button
          onClick={() => punch('in')}
          disabled={processing}
          className="btn btn-primary"
        >
          {processing ? 'กำลังบันทึก...' : 'Clock In'}
        </button>
        <button
          onClick={() => punch('out')}
          disabled={processing}
          className="btn btn-secondary"
        >
          {processing ? 'กำลังบันทึก...' : 'Clock Out'}
        </button>
      </div>
      {pending > 0 && (
        <div className="text-sm text-warning">
          ⚠️ รอการส่ง {pending} รายการ (ออฟไลน์/รีลอง)
        </div>
      )}
    </div>
  );
}
```

### Server-side idempotency (recommended)
Accept `idempotencyKey` in the request body (or header) and enforce uniqueness on:
- DB unique index on `(employee_id, punch_type, date_trunc('minute', timestamp))`, or
- Redis/database record keyed by `idempotencyKey` with a TTL (e.g., 24–72h).

Return `200 OK` for duplicate idempotency keys with the original response to prevent duplicates even under retries.

### Verification
1. Start dev servers (`npm run dev` frontend + backend).
2. Stop backend or disconnect network.
3. Tap “Clock In” — pending count > 0 and warning text appears.
4. Re-enable backend/network.
5. Within ~15s (or switch tabs away/back), pending count returns to 0 and a single punch appears on the backend
