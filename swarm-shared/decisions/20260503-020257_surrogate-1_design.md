def shard_assign(slug: str, total: int) -> int:
    """Deterministic shard assignment."""
    digest = hashlib.md5(slug.encode()).hexdigest()
    return int(digest, 16) % total


def load_manifest(manifest_path: Path):

def cdn_download(url: str):
    for attempt in range(1, MAX_RETRIES + 1):
        try:
            resp = requests.get(
                url,
                headers=HEADERS,
                timeout=TIMEOUT,
                stream=True
            )
            if resp.status_code == 429:
                print(f"429 rate-limited, waiting {RETRY_429_WAIT}s (attempt {attempt})")
                time.sleep(RETRY_429_WAIT)
                continue
            resp.raise_for_status()
            return resp
        except requests.RequestException as e:
            if attempt == MAX_RETRIES:
                raise
            sleep_sec = 2 ** attempt
            print(f"Request error {e}, retry {attempt}/{MAX_RETRIES} in {sleep_sec}s")
            time.sleep(sleep_sec)
    raise RuntimeError("Unreachable")


def project_to_pair(obj):
    """Extract {prompt, response} from heterogeneous file objects."""
    prompt_keys = {"prompt", "instruction", "input", "question", "text"}
    response_keys = {"response", "completion", "output", "answer"}

    prompt = None
    response = None

    if isinstance(obj, dict):
        for k, v in obj.items():
            if k in prompt_keys and prompt is None:
                prompt = v
            if k in response_keys and response is None:
                response = v
        # Fallback: first/second values if keys not matched
        if prompt is None or response is None:
            values = list(obj.values())
            if len(values) >= 2:
                if prompt is None:
                    prompt = values[0]
                if response is None:
                    response = values[1]
    elif isinstance(obj, (list, tuple)) and len(obj) >= 2:
        prompt, response = obj[0], obj[1]

    # Normalize to string
    if prompt is not None:
        prompt = str(prompt).strip()
    if response is not None:
        response = str(response).strip()

    return {"prompt": prompt, "response": response}


def parse_file_content(content, suffix):
    """Parse JSONL or Parquet content into iterable rows."""
    suffix = suffix.lower()
    if suffix == ".jsonl" or suffix == ".json":
        for line in content.splitlines():
            line = line.strip()
            if not line:
                continue
            try:
                yield json.loads(line)
            except json.JSONDecodeError:
                continue
    elif suffix == ".parquet":
        table = pq.read_table(pa.BufferReader(content))
        for batch in table.to_batches(max_chunksize=1000):
            for i in range(batch.num_rows):
                yield {k: batch[k][i].as_py() for k in batch.schema.names}
    else:
        # Try JSONL fallback
        for line in content.splitlines():
            line = line.strip()
            if line:
                try:
                    yield json.loads(line)
                except Exception:
                    pass
