# workio / design

## Final Synthesized Implementation (Best of Both Candidates)

### Diagnosis (Consensus)
- **DB Pressure**: Repeated `channelId → tenant` lookups on every webhook (burst-sensitive).
- **Observability**: Missing correlation IDs, inconsistent structured logging, and no trace propagation.
- **Resilience**: No k8s/PM2 readiness/liveness probes; signature verification lacks timing-attack safety.
- **Security**: Missing defense-in-depth for LINE signature verification.

#### 2. Secure Signature Verification (`workio/server/src/lib/line-signature.ts`)
```ts
import crypto from 'crypto';

export function verifyLineSignature(
  body: string | Buffer,
  signature: string,
  channelSecret: string
): boolean {
  if (!signature || !channelSecret) return false;

  const expected = crypto
    .createHmac('sha256', channelSecret)
    .update(Buffer.isBuffer(body) ? body : Buffer.from(body))
    .digest('base64');

  // Constant-time comparison to prevent timing attacks
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}
```

#### 3. Structured Logger with Correlation ID (`workio/server/src/lib/logger.ts`)
```ts
import { v4 as uuidv4 } from 'uuid';

export type RequestWithMeta = any & { correlationId: string; rawBody?: string };

export function reqLogger(req: RequestWithMeta, res: any, next: () => void) {
  const cid = req.headers['x-correlation-id'] || uuidv4();
  req.correlationId = cid;
  res.setHeader('x-correlation-id', cid);

  const start = Date.now();

  res.on('finish', () => {
    console.log(
      JSON.stringify({
        time: new Date().toISOString(),
        correlationId: cid,
        method: req.method,
        path: req.originalUrl,
        status: res.statusCode,
        durationMs: Date.now() - start,
        ip: req.ip,
        ua: req.get('User-Agent'),
      })
    );
  });

  next();
}
```

#### 4. Health & Readiness Probes (`workio/server/src/routes/health.ts`)
```ts
import { Router } from 'express';

const router = Router();

// Liveness: process is alive
router.get('/health', (_, res) => {
  res.json({ status: 'ok', uptime: process.uptime() });
});

// Readiness: critical dependencies available
router.get('/ready', async (req, res) => {
  try {
    // Lightweight DB ping
    await req.app.locals.db?.query('SELECT 1');

    const hasLineConfig = !!(
      process.env.LINE_CHANNEL_ACCESS_TOKEN &&
      process.env.LINE_CHANNEL_SECRET
    );

    res.json({
      status: hasLineConfig ? 'ready' : 'degraded',
      db: 'up',
      lineConfig: hasLineConfig,
    });
  } catch (err) {
    res.status(503).json({ status: 'unavailable', db: 'down' });
  }
});

export default router;
```

#### 5. LINE Webhook Handler (`workio/server/src/routes/line.ts`)
```ts
import { Router, Request, Response } from 'express';
import { tenantCache } from '../lib/cache';
import { verifyLineSignature } from '../lib/line-signature';
import { reqLogger, RequestWithMeta } from '../lib/logger';
import db from '../db';

const router = Router();
router.use(reqLogger);

async function resolveTenantId(channelId: string): Promise<string | null> {
  const cached = tenantCache.get(channelId);
  if (cached) return cached.tenantId;

  const result = await db.query(
    'SELECT tenant_id FROM channels WHERE channel_id = $1 LIMIT 1',
    [channelId]
  );

  if (result.rows.length === 0) return null;

  const tenantId = result.rows[0].tenant_id;
  tenantCache.set(channelId, { tenantId, updatedAt: Date.now() });
  return tenantId;
}

router.post('/webhook/line', async (req: RequestWithMeta, res: Response) => {
  const sig = req.headers['x-line-signature'] as string;
  const secret = process.env.LINE_CHANNEL_SECRET;

  if (!secret) {
    console.warn(
      JSON.stringify({
        correlationId: req.correlationId,
        error: 'LINE_CHANNEL_SECRET missing',
      })
    );
    return res.status(500).json({ error: 'server configuration error' });
  }

  const raw = req.rawBody || JSON.stringify(req.body);
  if (!verifyLineSignature(raw, sig, secret)) {
    console.warn(
      JSON.stringify({
        correlationId: req.correlationId,
        error: 'invalid line signature',
      })
    );
    return res.status(401).json({ error: 'invalid signature' });
  }

  const events = req.body.events || [];

  for (const ev of events) {
    try {
      const channelId = ev.source?.userId; // Adjust per schema
      if (!channelId) continue;

      const tenantId = await resolveTenantId(channelId);
      if (!tenantId) {
        console.warn(
          JSON.stringify({
            correlationId: req.correlationId,
            channelId,
            error: 'tenant not found',
          })
        );
        continue;
      }

      ev.tenantId = tenantId;
      // await handleLineEvent(ev, tenantId);
    } catch (err) {
      console.error(
        JSON.stringify({
          correlationId: req.correlationId,
          error: 'line event processing failed',
          details: err instanceof Error ? err.message : err,
        })
      );
    }
  }

  res.status(200).json({ ok: true });
});

export default router;
```

#### 6. App Entry Wiring (`workio/server/src/app.ts` or `index.ts`)
```ts
import express from 'express';
import lineRouter from './routes/line';
import healthRouter from './routes/health';

const app = express();

// Raw body capture for LINE signature verification
app.use(
  '/webhook/line',
  (req, res, next) => {
    let data = '';
    req.setEncoding('utf8');
    req.on('data', (chunk) => (data += chunk));
    req.on('end', () => {
      (req as any).rawBody = data;
      try {
        req.body = JSON.parse(data);
      } catch {
        req.body = {};
      }
      next();
    });
  },
  lineRouter
);

app.use(express.json());
app.use('/health', healthRouter);

export default app;
```
