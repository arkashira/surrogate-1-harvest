# surrogate-1 / design

## Final Implementation Plan (≤2 h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID` and `SHARD_TOTAL` (16) from GitHub Actions matrix.
- Uses a **single `list_repo_tree` call per date folder** to produce `file-list.json` (cached on disk).  
  → Avoids recursive listing and HF API pagination/rate-limits.  
  → Falls back to a pre-generated `file-list.json` committed in the repo (generated once on Mac) when listing is unavailable or too slow.
- Downloads only assigned shard files via **HF CDN** (`resolve/main/...`) with **no Authorization header** → bypasses `/api/` 429 limits.
- Streams each file by downloading individually and projecting to `{prompt, response}` at parse time (avoids `pyarrow.CastError` from heterogeneous repos).
- Deduplicates via central `lib/dedup.py` (SQLite md5 store) and writes to:
  ```
  batches/public-merged/<YYYY-MM-DD>/shard<SHARD_ID>-<HHMMSS>.jsonl
  ```
  with **no extra metadata columns**.
- Uploads via HF Hub (single commit per shard) with deterministic filename to avoid collisions.
- Reuses existing HF token (`HF_TOKEN`) and respects HF commit cap by sharding across filenames (already done).
- Fails fast on 429 with **360 s backoff**; retries network errors with exponential backoff; logs summary metrics.

def list_date_files(date_folder: str) -> List[str]:
    """List immediate files in date_folder (non-recursive)."""
    # 1) try cached
    if MANIFEST_PATH.exists():
        log.info("Using cached manifest: %s", MANIFEST_PATH)
        return json.loads(MANIFEST_PATH.read_text())

    # 2) try repo-level fallback (pre-generated on Mac)
    if FALLBACK_MANIFEST.exists():
        log.info("Using fallback manifest: %s", FALLBACK_MANIFEST)
        return json.loads(FALLBACK_MANIFEST.read_text())

    # 3) live listing (single call)
    log.info("Listing repo tree: %s/%s", DATASET_REPO, date_folder)
    items = api.list_repo_tree(
        repo_id=DATASET_REPO,
        path=date_folder,
        repo_type="dataset",
        recursive=False,
    )
    files = [it.rfilename for it in items if it.type == "file"]
    MANIFEST_PATH.parent.mkdir(parents=True, exist_ok=True)
    MANIFEST_PATH.write_text(json.dumps(files, indent=2))
    log.info("Cached %d files to %s", len(files), MANIFEST_PATH)
    return files

def assign_shard(files: List[str], shard_id: int, shard_total: int) -> List[str]:
    """Deterministic shard assignment by filename hash."""
    assigned = []
    for f in sorted(files):
        h = int(hashlib.md5(f.encode()).hexdigest(), 16)
        if (h % shard_total) == shard_id:
            assigned.append(f)
    return assigned

def cdn_download(file_path: str, dest: Path) -> Path:
    """Download via CDN (no auth). Retries with backoff on 429/5xx."""
    url = f"{CDN_BASE}/{file_path}"
    backoff = 5
    max_backoff = 360
    for attempt in range(1, 11):
        try:
            resp = requests.get(url, stream=True, timeout=60)
            if resp.status_code == 429:
                wait = max(backoff, 360)
                log.warning("CDN 429, waiting %ss (attempt %d)", wait, attempt)
                time.sleep(wait)
                backoff = min(backoff * 2, max_backoff)
                continue
            resp.raise_for_status()
            dest.parent.mkdir(parents=True, exist_ok=True)
            with open(dest, "wb") as f:
                for chunk in resp.iter_content(chunk_size=8192):
                    f.write(chunk)
            return dest
        except requests.RequestException as exc:
            wait = min(backoff * (2 ** (attempt - 1)), max_backoff)
            log.warning("Download failed (%s): %s — retry in %ss", url, exc, wait)
            time.sleep(wait)
    raise RuntimeError(f"Failed to download {url} after retries")

def project_to_pair(record: Dict[str, Any]) -> Dict[str, Any]:
    """
    Best-effort projection to {prompt,response}.
    Adjust heuristics here per observed schemas.
    """
    prompt_keys = {"prompt", "instruction", "question", "input", "text"}
    response_keys = {"response", "completion", "answer", "output"}

    prompt = None
    response = None

    rk = set(record.keys())
    for k in rk:
        if k in prompt_keys and prompt is None:
            prompt = record[k]
        if k in response_keys and response is None:
            response = record[k]

    # Fallbacks
    if prompt is None and response is None:
        str_vals = [v for v in record.values() if isinstance(v, str)]
        if len(str_vals) >= 2:
            prompt, response = str_vals[0], str_vals[1]
        elif len(str_vals) == 1:
            prompt = str_vals[0]
            response = ""
    elif response is None:
        response = ""
    elif prompt is None:
        prompt = ""

    return {"prompt": str(prompt), "response": str(response)}
