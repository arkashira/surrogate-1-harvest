# vanguard / design

## 1. Diagnosis
- Frontend still triggers runtime HF API calls (`list_repo_tree`, `load_dataset`) during dataset selection/preview → 429 risk and non-reproducible runs.
- No deterministic, content-addressed manifest for dataset files; training scripts can’t guarantee CDN-only fetches.
- Missing pre-list-and-embed workflow: no JSON file listing date-scoped folder paths to enable Lightning training with zero API calls.
- No fallback for Lightning idle-stop: training dies if studio stops; no auto-restart guard before `.run()`.
- Ingestion likely writes mixed-schema enriched files (extra `source`, `ts` cols) instead of strict `{prompt,response}` parquet in `batches/mirror-merged/{date}/{slug}.parquet`.

## 2. Proposed change
- Add `/opt/axentx/vanguard/ingest/manifest.py` and `/opt/axentx/vanguard/train/prepare_manifest.py` to implement:
  - Mac-side one-off `prepare_manifest.py` that calls `list_repo_tree(path, recursive=False)` for a date folder, saves `manifest-{date}.json` (content-addressed by folder hash).
  - `manifest.py` provides `load_manifest(date)` and `stream_files(manifest, hf_dataset_repo)` that downloads via CDN URLs only (`resolve/main/...`) with no Authorization header.
- Update training launcher to require a manifest and use `stream_files` in the Lightning `DataModule` (zero HF API calls during training).
- Add small guard in launcher to reuse running studio and restart idle ones.

```bash
# /opt/axentx/vanguard/ingest/manifest.py
import json, hashlib, os, requests
from pathlib import Path
from typing import List, Dict, Any

HF_CDN = "https://huggingface.co/datasets"

def folder_hash(repo: str, folder: str, items: List[str]) -> str:
    h = hashlib.sha256()
    h.update(f"{repo}::{folder}".encode())
    for it in sorted(items):
        h.update(it.encode())
    return h.hexdigest()[:16]

def save_manifest(repo: str, folder: str, items: List[str], out_dir: Path) -> Path:
    manifest = {
        "repo": repo,
        "folder": folder.rstrip("/"),
        "items": items,
        "hash": folder_hash(repo, folder, items)
    }
    out_dir.mkdir(parents=True, exist_ok=True)
    out = out_dir / f"manifest-{manifest['hash']}.json"
    out.write_text(json.dumps(manifest, indent=2))
    return out

def load_manifest(path: Path) -> Dict[str, Any]:
    return json.loads(path.read_text())

def cdn_url(repo: str, filepath: str) -> str:
    return f"{HF_CDN}/{repo}/resolve/main/{filepath.lstrip('/')}"

def stream_files(manifest: Dict[str, Any], local_dir: Path):
    repo = manifest["repo"]
    folder = manifest["folder"].lstrip("/")
    local_dir.mkdir(parents=True, exist_ok=True)
    for item in manifest["items"]:
        remote = f"{folder}/{item}" if folder else item
        url = cdn_url(repo, remote)
        out = local_dir / item
        if out.exists():
            continue
        with requests.get(url, stream=True, timeout=60) as r:
            r.raise_for_status()
            with open(out, "wb") as f:
                for chunk in r.iter_content(chunk_size=8192):
                    f.write(chunk)
        yield out
```

```bash
# /opt/axentx/vanguard/train/prepare_manifest.py
#!/usr/bin/env bash
set -euo pipefail
# Run on Mac after HF API window clears.
# Usage: HF_TOKEN=... python prepare_manifest.py <repo> <date-folder>

python3 - "$@" <<'PY'
import os, sys, json
from huggingface_hub import HfApi
from pathlib import Path

sys.path.insert(0, str(Path(__file__).parent.parent / "ingest"))
import manifest as mf

def main():
    repo = sys.argv[1]               # e.g. "axentx/surrogate-1-ingest"
    folder = sys.argv[2].rstrip("/") # e.g. "batches/mirror-merged/2026-05-03"
    api = HfApi(token=os.getenv("HF_TOKEN"))
    # list one level only (avoids recursive pagination)
    items = [Path(f.rfilename).name for f in api.list_repo_tree(repo, path=folder, recursive=False)]
    if not items:
        print("No items found", file=sys.stderr)
        sys.exit(1)
    out = mf.save_manifest(repo, folder, items, Path(__file__).parent.parent / "manifests")
    print(f"Saved {out}")

if __name__ == "__main__":
    main()
PY
```

```bash
# /opt/axentx/vanguard/train/launcher.py  (additions/updates)
# Ensure executable and Bash shebang for cron safety
#!/usr/bin/env bash
set -euo pipefail
export SHELL=/bin/bash

# Reuse running studio or restart idle
python3 - "$@" <<'PY'
import lightning as L
from pathlib import Path
import json, sys

def get_studio(name="vanguard-train"):
    teamspace = L.Teamspace()
    for s in teamspace.studios:
        if s.name == name:
            return s
    return None

def run_with_manifest(manifest_path: Path):
    manifest = json.loads(manifest_path.read_text())
    # Pass manifest path to Lightning run so DataModule uses CDN-only loader
    target = L.LightningWork(
        cloud_compute=L.CloudCompute("L40S"),
        cloud_build=L.CloudBuild(requirements=["torch", "lightning", "requests"])
    )
    # simplified: in real code, pass manifest_path to your DataModule
    job = L.LightningTrainer(
        target=target,
        task=... # your training task configured to use manifest_path
    )
    studio = get_studio()
    if studio is None or studio.status != "running":
        if studio and studio.status != "running":
            print("Restarting idle studio")
        studio = L.Studio(create_ok=True, name="vanguard-train")
    job.run(studio=studio)

if __name__ == "__main__":
    manifest = Path(sys.argv[1]) if len(sys.argv) > 1 else next(Path(__file__).parent.parent.glob("manifests/manifest-*.json"))
    run_with_manifest(manifest)
PY
```
