### 2. Build script (Node) — primary path

`scripts/build-assets.js`

```js
#!/usr/bin/env node
// Build deterministic CDN asset manifest for surrogate-1 dataset.
// Usage:
//   HF_TOKEN=<token> node scripts/build-assets.js <repo> <dateFolder>
//
// Produces: public/asset-manifest.json
// Strategy: one list_repo_tree call per dateFolder (non-recursive) to avoid pagination/rate-limits.

import { writeFileSync, mkdirSync } from 'fs';
import { resolve, dirname } from 'path';
import { fileURLToPath } from 'url';
import { HfApi } from 'huggingface-hub';

const __dirname = dirname(fileURLToPath(import.meta.url));
const root = resolve(__dirname, '..');

const [, , REPO = 'surrogate-1', DATEFOLDER] = process.argv;
if (!DATEFOLDER) {
  console.error('Usage: HF_TOKEN=<token> node build-assets.js <repo> <dateFolder>');
  process.exit(1);
}

const HF_TOKEN = process.env.HF_TOKEN || '';
const outDir = resolve(root, 'public');
const outFile = resolve(outDir, 'asset-manifest.json');

mkdirSync(outDir, { recursive: true });

async function main() {
  const api = new HfApi({ accessToken: HF_TOKEN });

  // Single non-recursive call per folder to minimize API usage.
  const tree = await api.listRepoTree({
    repo: REPO,
    path: DATEFOLDER,
    recursive: false,
  });

  const files = tree
    .map((item) => item.rfilename)
    .filter((f) => f.endsWith('.parquet'))
    .sort();

  const manifest = {
    repo: REPO,
    date: DATEFOLDER,
    cdnPrefix: `https://huggingface.co/datasets/${REPO}/resolve/main/${DATEFOLDER}`,
    files,
  };

  writeFileSync(outFile, JSON.stringify(manifest, null, 2), 'utf8');
  console.log(`Wrote ${files.length} files to ${outFile}`);
}

main().catch((err) => {
  console.error(err);
  process.exit(1);
});
```

Notes:
- Uses `huggingface-hub` npm package (`@huggingface/hub` or `huggingface-hub` depending on your stack — pick one; if you prefer, swap to `hf.js`/`axios` + REST). If Node tooling is undesirable, use the provided Python script instead.
- One API call per folder keeps rate-limit exposure minimal.
- Output is deterministic and CDN-first.

```jsx
import { useEffect, useState } from 'react';

const MANIFEST_PATH = '/asset-manifest.json';

function createFileLink(cdnPrefix, file) {
  return `${cdnPrefix}/${file}`;
}

export default function App() {
  const [manifest, setManifest] = useState(null);
  const [error, setError] = useState(null);

  useEffect(() => {
    fetch(MANIFEST_PATH)
      .then((res) => {
        if (!res.ok) throw new Error('Manifest missing — run build:assets first.');
        return res.json();
      })
      .then(setManifest)
      .catch(setError);
  }, []);

  if (error) {
    return <p style={{ color: 'red' }}>Error: {error.message}</p>;
  }

  if (!manifest) {
    return <p>Loading manifest...</p>;
  }

  return (
    <div style={{ fontFamily: 'system-ui, sans-serif', padding: 20 }}>
      <h2>{manifest.repo} — {manifest.date}</h2>
      <ul>
        {manifest.files.map((f) => (
          <li key={f}>
            <a href={createFileLink(manifest.cdnPrefix, f)} target="_blank" rel="noopener noreferrer">
              {f}
            </a>
          </li>
        ))}
      </ul>
      <p><small>Resolved via CDN (no HF API calls from browser).</small></p>
    </div>
  );
}
```
