## Decision

Implement a single backend orchestration endpoint `/api/v1/sense/top-hub-signal` that:

- Runs `granite-business-research.sh` idempotently (respecting a 24-hour cache/lock).
- Queries the knowledge-RAG graph for the top-connected hub (e.g., "MOC") and returns contextual insights.
- **Never executes** any change — strictly adheres to Costinel’s *Sense + Signal* philosophy.
- Uses HF CDN bypass for any dataset fetches (zero API calls during inference) and Lightning Studio reuse patterns for any model runs.

### 1. Route: `/api/v1/sense/top-hub-signal`

```python
# File: app/api/v1/endpoints/sense.py
import asyncio
import os
import time
import subprocess
from datetime import datetime, timedelta
from pathlib import Path
from typing import Any

from fastapi import APIRouter, HTTPException
from pydantic import BaseModel

router = APIRouter()

LOCKFILE = Path("/tmp/costinel_granite.lock")
CACHEFILE = Path("/tmp/costinel_granite.cache.json")
CACHE_TTL = 24 * 60 * 60  # 24h

class TopHubSignalResponse(BaseModel):
    signal: str
    hub: str
    insights: list[str]
    cache_status: str  # "fresh" | "stale" | "generated"
    ts: str

def _is_cache_fresh() -> bool:
    if not CACHEFILE.exists():
        return False
    age = time.time() - CACHEFILE.stat().st_mtime
    return age < CACHE_TTL

def _run_granite_business_research() -> None:
    script = Path("/opt/axentx/Costinel/scripts/granite-business-research.sh")
    if not script.exists():
        raise RuntimeError("granite-business-research.sh not found")
    # Ensure executable and shebang
    script.chmod(0o755)
    result = subprocess.run(
        ["/bin/bash", str(script)],
        capture_output=True,
        text=True,
        env={**os.environ, "SHELL": "/bin/bash"},
        cwd=Path("/opt/axentx/Costinel"),
    )
    if result.returncode != 0:
        raise RuntimeError(f"granite-business-research failed: {result.stderr}")
    # Persist minimal cache marker
    CACHEFILE.write_text('{"generated": true}')

async def _ensure_granite_fresh() -> str:
    if _is_cache_fresh():
        return "fresh"
    # Try lock
    try:
        with LOCKFILE.open("x"):
            _run_granite_business_research()
            return "generated"
    except FileExistsError:
        # Another process is running; wait briefly and accept stale or fresh
        await asyncio.sleep(2)
        if _is_cache_fresh():
            return "fresh"
        return "stale"

def _query_top_hub_and_insights() -> tuple[str, list[str]]:
    # Reuse existing knowledge-RAG client (assumed available)
    from app.services.knowledge_rag import KnowledgeRAG
    rag = KnowledgeRAG()
    top_hub = rag.get_most_connected_hub() or "MOC"
    insights = rag.query_hub_insights(top_hub, limit=5)
    return top_hub, insights

@router.get("/top-hub-signal", response_model=TopHubSignalResponse)
async def top_hub_signal() -> Any:
    cache_status = await _ensure_granite_fresh()
    hub, insights = _query_top_hub_and_insights()
    signal = (
        f"Top hub '{hub}' identified with actionable insights. "
        "No execution performed — signal only."
    )
    return TopHubSignalResponse(
        signal=signal,
        hub=hub,
        insights=insights,
        cache_status=cache_status,
        ts=datetime.utcnow().isoformat() + "Z",
    )
```

### 3. Ensure script wrapper correctness (idempotency + shebang)

```bash
# File: scripts/granite-business-research.sh
#!/usr/bin/env bash
set -euo pipefail

cd "$(dirname "$0")/.."

LOCK="/tmp/granite_business.lock"
CACHE="/tmp/granite_business.cache.json"
TTL=$((24*60*60))

if [[ -f "$CACHE" ]]; then
  age=$(( $(date +%s) - $(stat -c %Y "$CACHE") ))
  if (( age < TTL )); then
    echo "Cache fresh, skipping."
    exit 0
  fi
fi

if (set -C; echo $$ > "$LOCK") 2>/dev/null; then
  trap 'rm -f "$LOCK"' EXIT
  echo "Running granite business research..."
```
