## 1. CDN Manifest Generation (run once, outside rate-limit window)

# scripts/generate_cdn_manifest.sh
#!/usr/bin/env bash
set -euo pipefail
REPO="datasets/your-org/surrogate-mirror"
DATE_FOLDER="2026-05-03"
OUT="data/cdn_manifest_${DATE_FOLDER}.json"

python - <<PY "$REPO" "$DATE_FOLDER" "$OUT"
import json, os, sys
from huggingface_hub import HfApi

repo_id, date_folder = sys.argv[1], sys.argv[2]
out_path = sys.argv[3]

api = HfApi()
# Recursive to capture nested date partitions; filter size>0 to skip placeholders
tree = api.list_repo_tree(repo_id, path=date_folder, recursive=True)
files = sorted(
    f.rfilename for f in tree
    if f.rfilename.endswith(".parquet") and getattr(f, "size", 0) > 0
)

os.makedirs(os.path.dirname(out_path), exist_ok=True)
manifest = {
    "repo_id": repo_id,
    "date_folder": date_folder,
    "files": files,
    "cdn_base": f"https://huggingface.co/datasets/{repo_id}/resolve/main",
    "generated_at": __import__("datetime").datetime.utcnow().isoformat() + "Z"
}
with open(out_path, "w") as f:
    json.dump(manifest, f, indent=2)
print(f"Manifest: {len(files)} shards -> {out_path}")
PY

## 2. CDN-Only Dataset Loader (robust + memory-safe)

**Improvement over Candidate 1**: Add retries, backoff, and optional sharding for large manifests to avoid single-point failures and reduce memory pressure.

```python
# surrogate/training/cdn_dataset.py
import json
import pyarrow.parquet as pq
import requests
from io import BytesIO
from typing import Iterator, Dict, List
import time

class CDNParquetLoader:
    def __init__(self, manifest_path: str, columns=("prompt", "response"), max_retries: int = 3):
        with open(manifest_path) as f:
            cfg = json.load(f)
        self.repo_id = cfg["repo_id"]
        self.files: List[str] = cfg["files"]
        self.cdn_base = cfg.get("cdn_base", f"https://huggingface.co/datasets/{self.repo_id}/resolve/main")
        self.columns = columns
        self.max_retries = max_retries

    def _download_cdn(self, path: str) -> bytes:
        url = f"{self.cdn_base}/{path}"
        for attempt in range(1, self.max_retries + 1):
            try:
                resp = requests.get(url, timeout=60)
                resp.raise_for_status()
                return resp.content
            except Exception as exc:
                if attempt == self.max_retries:
                    raise RuntimeError(f"Failed to fetch {url}: {exc}") from exc
                time.sleep(2 ** attempt)
        raise RuntimeError("Unreachable")

    def stream_rows(self, shard_limit: int | None = None) -> Iterator[Dict[str, str]]:
        files = self.files[:shard_limit] if shard_limit else self.files
        for f in files:
            raw = self._download_cdn(f)
            table = pq.read_table(BytesIO(raw), columns=self.columns)
            for batch in table.to_batches():
                df = batch.to_pandas()
                for _, row in df.iterrows():
                    yield {k: str(row[k]) for k in self.columns if k in row}
```

## 3. Lightning Studio Reuse + Idle-Timeout Guard (deterministic)

**Improvement over Candidate 1**: Remove ambiguous retry loops; instead, enforce explicit state transitions and fail-fast on unrecoverable states to avoid quota burn.

```python
# surrogate/training/lightning_utils.py
from lightning import Lightning, Teamspace, Studio, Machine
import time

LIGHTNING_ACCOUNT = "your-teamspace"

def get_or_create_studio(name: str, machine: Machine = Machine.L40S) -> Studio:
    team = Teamspace(LIGHTNING_ACCOUNT)
    for s in team.studios:
        if s.name == name:
            if s.status == "Running":
                print(f"Reusing running studio: {name}")
                return s
            elif s.status in ("Stopped", "IdleStopped"):
                print(f"Restarting stopped studio: {name}")
                s.start(machine=machine)
                # Wait for running
                for _ in range(30):
                    s.refresh()
                    if s.status == "Running":
                        return s
                    time.sleep(10)
                raise RuntimeError(f"Studio {name} failed to start")
            else:
                raise RuntimeError(f"Studio {name} in unexpected state: {s.status}")

    print(f"Creating studio: {name}")
    return Studio(
        name=name,
        machine=machine,
        teamspace=LIGHTNING_ACCOUNT,
        create_ok=True,
    )

def run_training_in_studio(studio: Studio, target_script: str, script_args: list[str]):
    """Run training; fail-fast on non-recoverable states to protect quota."""
    if studio.status != "Running":
        raise RuntimeError(f"Studio not running (status={studio.status})")

    run = studio.run(
        target=target_script,
        args=script_args,
        cwd="/workspace",
    )
    # Stream logs; raise on failure to avoid silent quota waste
    for line in run.stream_logs():
        print(line, end="")
    if run.status != "Succeeded":
        raise RuntimeError(f"Training failed with status={run.status}")
    print("Training succeeded.")
```

if __name__ == "__main__":
    p = argparse.ArgumentParser()
    p.add_argument("--manifest", type=Path, required=True, help="CDN manifest JSON")
    p.add_argument("--studio", action="store_true", help="Run in Lightning Studio")
    p.add_argument("--limit", type=int, help="Limit shards for testing")
    args = p.parse_args()
