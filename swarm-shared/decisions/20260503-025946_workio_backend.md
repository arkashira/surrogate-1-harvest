# workio / backend

## 1. Diagnosis

- **No idempotency key enforcement on LINE webhook**: duplicate punches can occur from client retries or LINE delivery retries; backend accepts duplicates.
- **Race condition on clock-in/out**: concurrent requests from same user can create two punch records for the same session window.
- **Missing request traceability**: no correlation ID or idempotency-key logging on webhook ingress → hard to debug duplicates or retries.
- **Hardcoded tenant resolution**: webhook path `/webhook/line` lacks tenant context; multi-tenancy relies on channel binding only, which is brittle at scale.
- **No dedupe window for recent punches**: backend does not check for a recent punch (e.g., ±5s) before inserting, allowing near-duplicate rows.

## 2. Proposed change

File: `/opt/axentx/workio/server/src/routes/lineWebhook.ts` (or equivalent webhook handler)  
Scope: add idempotency middleware + dedupe check + correlation logging + tenant-aware path.

```ts
// /opt/axentx/workio/server/src/routes/lineWebhook.ts
import { Router, Request, Response, NextFunction } from 'express';
import { pool } from '../db';
import crypto from 'crypto';

const router = Router();

// In-memory short-lived idempotency cache (production: use Redis)
const IDEMPOTENCY_TTL_MS = 60_000; // 60s
const idempotencyCache = new Map<string, { status: number; body: any; ts: number }>();

function idempotencyMiddleware(req: Request, res: Response, next: NextFunction) {
  const key = req.headers['idempotency-key'] as string || req.headers['Idempotency-Key'] as string;
  const correlationId = req.headers['x-correlation-id'] as string || crypto.randomUUID();
  (req as any).correlationId = correlationId;

  if (!key) {
    // For LINE webhook, derive deterministic key from event + timestamp to absorb retries
    const body = req.body;
    const events = body?.events || [];
    if (events.length > 0) {
      const stable = events.map((e: any) => `${e.type}:${e.source?.userId || ''}:${e.timestamp || ''}`).join('|');
      (req as any).idempotencyKey = crypto.createHash('sha256').update(stable).digest('hex');
    } else {
      (req as any).idempotencyKey = crypto.randomUUID();
    }
  } else {
    (req as any).idempotencyKey = key;
  }

  const now = Date.now();
  const cached = idempotencyCache.get((req as any).idempotencyKey);
  if (cached && now - cached.ts < IDEMPOTENCY_TTL_MS) {
    res.setHeader('X-Correlation-Id', correlationId);
    res.setHeader('X-Idempotency-Replay', 'true');
    return res.status(cached.status).json(cached.body);
  }

  // capture response to cache
  const originalSend = res.json;
  res.json = function (body) {
    idempotencyCache.set((req as any).idempotencyKey, { status: res.statusCode, body, ts: Date.now() });
    return originalSend.call(this, body);
  };

  next();
}

async function dedupeRecentPunch(userId: string, tenantId: string, type: 'CLOCK_IN' | 'CLOCK_OUT', windowMs = 5_000) {
  const cutoff = new Date(Date.now() - windowMs);
  const q = `
    SELECT id FROM punches
    WHERE user_id = $1 AND tenant_id = $2 AND type = $3 AND created_at >= $4
    ORDER BY created_at DESC LIMIT 1
  `;
  const { rows } = await pool.query(q, [userId, tenantId, type, cutoff]);
  return rows.length > 0 ? rows[0].id : null;
}

router.post('/line', idempotencyMiddleware, async (req: Request, res: Response) => {
  const correlationId = (req as any).correlationId;
  res.setHeader('X-Correlation-Id', correlationId);

  const events = req.body?.events || [];
  if (!events.length) return res.status(200).json({ ok: true });

  for (const ev of events) {
    if (ev.type !== 'message' || ev.message?.type !== 'text') continue;
    const userId = ev.source?.userId;
    const text = ev.message.text?.trim().toLowerCase();
    const replyToken = ev.replyToken;

    if (!userId) continue;

    // Resolve tenant by channel binding (simplified; extend as needed)
    const channelId = ev.source?.channelId || '';
    const { rows: tenantRows } = await pool.query(
      'SELECT id FROM tenants WHERE line_channel_id = $1 LIMIT 1',
      [channelId]
    );
    const tenantId = tenantRows[0]?.id || 'default';

    // Resolve user
    let { rows: userRows } = await pool.query(
      'SELECT id FROM users WHERE line_user_id = $1 AND tenant_id = $2 LIMIT 1',
      [userId, tenantId]
    );
    if (!userRows.length) {
      // auto-provision employee (optional)
      const insert = await pool.query(
        'INSERT INTO users (tenant_id, line_user_id, name, role) VALUES ($1,$2,$3,$4) RETURNING id',
        [tenantId, userId, `User_${userId.slice(-6)}`, 'employee']
      );
      userRows = insert.rows;
    }
    const userIdDb = userRows[0].id;

    // Clock in/out by text command (e.g., "in", "out")
    const isIn = text.includes('in') || text.includes('เข้า') || text.includes('clock in');
    const isOut = text.includes('out') || text.includes('ออก') || text.includes('clock out');
    if (!isIn && !isOut) continue;

    const type = isIn ? 'CLOCK_IN' : 'CLOCK_OUT';
    const dup = await dedupeRecentPunch(userIdDb, tenantId, type, 5_000);
    if (dup) {
      console.log(`[${correlationId}] deduped ${type} for user=${userIdDb} recent=${dup}`);
      continue;
    }

    // Insert with idempotency guard (unique constraint recommended on (user_id, type, created_at_rounded) in prod)
    await pool.query(
      `INSERT INTO punches (id, tenant_id, user_id, type, created_at, source, meta)
       VALUES (gen_random_uuid(), $1, $2, $3, NOW(), 'line', $4)`,
      [tenantId, userIdDb, type, JSON.stringify({ correlationId, lineEvent: ev.type })]
    );

    // Reply via LINE (simplified; queue for async reply in production)
    if (replyToken) {
      // Note: actual LINE reply requires LINE SDK and channel secret/token from .env
      // Placeholder: log intent
      console.log(`[${correlationId}] would reply to ${replyToken}: ${type} recorded`);
    }
  }

  res.status(200).json({ ok: true });
});

export default router;
```
