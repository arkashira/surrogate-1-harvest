# vanguard / design

## Final Synthesis (Best of Both Candidates)

Diagnosis and scope are identical across candidates; the differences are minor wording and style. The following is the single, authoritative plan that preserves all correctness and maximizes concrete actionability.

---

## 1. Diagnosis (condensed)

- `load_dataset` at runtime triggers HF `/api/` file enumeration and per-file metadata → 429 rate limits and non-reproducible epochs.
- No content-addressed manifest (file list + SHA256) → no CDN-only fetch and no integrity verification.
- Mixed-schema `enriched/` files (extra cols: `source`, `ts`) cause `pyarrow.CastError`; schema must be projected at parse time.
- No deterministic HF repo selection → ingestion hits 128 commit/hr cap with no sharding.
- No Lightning Studio reuse/idle guard → idle timeout kills training and wastes quota.

---

## 2. Proposed change (single scope)

Create/add:
- `/opt/axentx/vanguard/manifest.py`
- `/opt/axentx/vanguard/train.py` (create or patch)
- `/opt/axentx/vanguard/ingest.py` (light helper)

Goals:
- Build a content-addressed manifest (path, size, SHA256) per date folder.
- Embed manifest in training; data loader uses CDN-only URLs (`resolve/main/...`) with zero HF API calls.
- Project `enriched/*` to `{prompt, response}` at parse time; store attribution in filename pattern (`batches/mirror-merged/{date}/{slug}.parquet`).
- Deterministic repo selection via hash-slug modulo siblings to spread commits.
- Lightning Studio reuse + idle restart guard to preserve quota.

# /opt/axentx/vanguard/manifest.py
import json, hashlib, os
from pathlib import Path
from typing import Dict, List

def sha256_file(path: str) -> str:
    h = hashlib.sha256()
    with open(path, "rb") as f:
        for chunk in iter(lambda: f.read(1 << 20), b""):
            h.update(chunk)
    return h.hexdigest()

def build_manifest(date_folder: str, out_path: str) -> Dict:
    """
    Build content-addressed manifest for one date folder.
    Expects enriched/*.parquet files.
    """
    root = Path(date_folder)
    files: List[Dict] = []
    for p in sorted(root.rglob("*.parquet")):
        rel = str(p.relative_to(root))
        files.append({
            "path": rel,
            "size": p.stat().st_size,
            "sha256": sha256_file(str(p)),
        })
    manifest = {
        "date_folder": str(date_folder),
        "files": files,
        "total_files": len(files),
    }
    Path(out_path).write_text(json.dumps(manifest, indent=2))
    return manifest

def pick_repo(slug: str, siblings: int = 5) -> int:
    """Deterministic repo index 0..siblings-1 for HF commit spreading."""
    return hash(slug) % siblings

def cdn_url(repo: str, filepath: str) -> str:
    """CDN-only URL that bypasses HF /api/ rate limits."""
    return f"https://huggingface.co/datasets/{repo}/resolve/main/{filepath}"
```

# /opt/axentx/vanguard/train.py  (create or patch)
import json, os, pyarrow.parquet as pq
from pathlib import Path
import torch
from torch.utils.data import IterableDataset
from manifest import cdn_url, pick_repo

HF_REPO = os.getenv("HF_DATASET_REPO", "axentx/surrogate-1")
MANIFEST_PATH = os.getenv("MANIFEST_PATH", "manifest.json")

class CDNParquetDataset(IterableDataset):
    def __init__(self, manifest_path: str):
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.files = [
            cdn_url(HF_REPO, item["path"]) for item in self.manifest["files"]
        ]

    def __iter__(self):
        for url in self.files:
            # stream remote parquet without HF API
            import fsspec
            with fsspec.open(url, "rb") as f:
                table = pq.read_table(f)
                # project to {prompt, response} only
                df = table.select(["prompt", "response"]).to_pandas()
                for _, row in df.iterrows():
                    yield {"prompt": row["prompt"], "response": row["response"]}

# Lightning launcher with studio reuse + idle restart guard
def launch_studio(train_script: str, studio_name: str = "vanguard-train"):
    from lightning import Studio, Teamspace, Machine
    # reuse running studio
    for s in Teamspace.studios:
        if s.name == studio_name and s.status == "Running":
            print(f"Reusing running studio: {studio_name}")
            return s
    # else create
    studio = Studio(
        name=studio_name,
        machine=Machine.L40S,
        script=train_script,
        create_ok=True,
    )
    return studio

if __name__ == "__main__":
    dataset = CDNParquetDataset(MANIFEST_PATH)
    # minimal dataloader for local test (real training via Lightning)
    for i, sample in enumerate(dataset):
        if i >= 10:
            break
        print(sample)

# /opt/axentx/vanguard/ingest.py  (light helper)
import pyarrow.parquet as pq
from pathlib import Path
from manifest import pick_repo

def project_and_write(input_path: str, output_dir: str, date: str, slug: str):
    """
    Project enriched/ file to {prompt, response} and write to
    batches/mirror-merged/{date}/{slug}.parquet
    Attribution stored in filename pattern only.
    """
    tbl = pq.read_table(input_path)
    # keep only required cols; drop source/ts to avoid schema drift
    if "prompt" not in tbl.column_names or "response" not in tbl.column_names:
        raise ValueError("Missing prompt/response in input")
    proj = tbl.select(["prompt", "response"])
    out = Path(output_dir) / f"batches/mirror-merged/{date}/{slug}.parquet"
    out.parent.mkdir(parents=True, exist_ok=True)
    pq.write_table(proj, str(out))
    return str(out)
```
