## Core Design Decisions (Resolved)

1. **Manifest generation**  
   - Use `list_repo_tree(..., recursive=True)` (Candidate 2) so the manifest is complete and training never blocks on directory traversal.  
   - Keep generation on Mac (or any dev machine) to stay outside rate-limited training loops.

2. **Schema projection / `pyarrow.CastError`**  
   - Adopt Candidate 1’s field-name heuristics (`prompt_keys`, `response_keys`) plus Candidate 2’s **strict validation + logging**:  
     - Reject rows where prompt/response are empty after projection.  
     - Log rejected counts per file for data-quality triage.  
   - Add **UTF-8 coercion** and truncation guard to prevent tokenizer crashes.

3. **Streaming & memory**  
   - Use Candidate 2’s **true streaming** (`iter_content(chunk_size=...)`) instead of `r.content` to avoid OOM on large Parquet files.  
   - Keep Candidate 1’s `IterableDataset` + `max_chunksize` for Arrow batch efficiency.

4. **Lightning Studio lifecycle**  
   - Prefer Candidate 1’s **reuse-before-create** flow to save quota.  
   - Add Candidate 2’s **startup timeout + status polling** so automation is robust against slow studio boot.

5. **CLI / entrypoint**  
   - Accept `--manifest-path` (default `dataset_manifest.json`) and optional `--date-folder` to auto-generate if manifest missing (defensive UX).

## Implementation

### 1. `scripts/build_dataset_manifest.py`
```python
#!/usr/bin/env python3
"""
Generate dataset_manifest.json for CDN-only training.
Run on dev machine to avoid HF API rate limits during training.
"""
import json
from datetime import datetime, timezone
from pathlib import Path
from huggingface_hub import HfApi

API = HfApi()
REPO = "axentx/surrogate-datasets"
BASE_CDN = f"https://huggingface.co/datasets/{REPO}/resolve/main"

def build_manifest(date_folder: str, out_path: Path):
    # recursive=True ensures we capture all nested parquet/jsonl in one pass
    entries = API.list_repo_tree(REPO, path=date_folder, recursive=True)
    files = []
    for e in entries:
        if not e.path.endswith((".parquet", ".jsonl")):
            continue
        cdn_url = f"{BASE_CDN}/{e.path}"
        files.append({
            "path": e.path,
            "cdn_url": cdn_url,
            "size": getattr(e, "size", None),
            "date_folder": date_folder,
        })

    manifest = {
        "generated_at": datetime.now(timezone.utc).isoformat(),
        "repo": REPO,
        "date_folder": date_folder,
        "files": files,
        "total_files": len(files),
    }
    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"Manifest written to {out_path} ({len(files)} files)")

if __name__ == "__main__":
    import argparse
    parser = argparse.ArgumentParser()
    parser.add_argument("--date-folder", required=True, help="e.g. batches/mirror-merged/2026-05-03")
    parser.add_argument("--out", default="dataset_manifest.json")
    args = parser.parse_args()
    build_manifest(args.date_folder, Path(args.out))
```

### 2. `surrogate/data/cdn_dataset.py`
```python
import json
import pyarrow.parquet as pq
import pyarrow as pa
from torch.utils.data import IterableDataset
import requests
from io import BytesIO
from typing import Dict, Any, Iterator
import logging

logger = logging.getLogger(__name__)

class CdnDataset(IterableDataset):
    """
    CDN-only dataset loader.
    Zero HuggingFace API calls during training.
    Projects heterogeneous schemas to {prompt, response} with strict validation.
    """
    PROMPT_KEYS = {"prompt", "instruction", "input", "question", "user"}
    RESPONSE_KEYS = {"response", "output", "answer", "assistant", "completion"}

    def __init__(self, manifest_path: str, max_text_len: int = 8192):
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.files = self.manifest["files"]
        self.max_text_len = max_text_len
        self._rejected = 0

    def _project_record(self, raw: Dict[str, Any]) -> Dict[str, str]:
        prompt = None
        response = None

        for k, v in raw.items():
            if v is None:
                continue
            v = str(v).strip()
            if not v:
                continue
            if k in self.PROMPT_KEYS:
                prompt = v
            if k in self.RESPONSE_KEYS:
                response = v

        # Fallback to positional if standard keys missing
        if prompt is None or response is None:
            vals = [str(v).strip() for v in raw.values() if v is not None and str(v).strip()]
            if len(vals) >= 2:
                prompt = vals[0] if prompt is None else prompt
                response = vals[1] if response is None else response

        # Truncate to avoid tokenizer/OOM issues
        if prompt and len(prompt) > self.max_text_len:
            prompt = prompt[: self.max_text_len]
        if response and len(response) > self.max_text_len:
            response = response[: self.max_text_len]

        return {"prompt": prompt or "", "response": response or ""}

    def _stream_file(self, file_info: Dict[str, Any]) -> Iterator[Dict[str, str]]:
        url = file_info["cdn_url"]
        try:
            with requests.get(url, stream=True, timeout=30) as r:
                r.raise_for_status()
                buf = BytesIO()
                for chunk in r.iter_content(chunk_size=8192):
                    buf.write(chunk)
                buf.seek(0)

                table = pq.read_table(buf, use_threads=False)
        except Exception as exc:
            logger.warning("Failed to read %s: %s", url, exc)
            return

        rejected_in_file = 0
        for batch in table.to_batches(max_chunksize=512):
            for row in batch.to_pylist():
                projected = self._project_record(row)
                if projected["prompt"] and projected["response"]:
                    yield projected
                else:
                    rejected_in_file += 1

        if rejected_in_file:
            logger.info(
                "Rejected %d rows in %s (schema/projection mismatch)",
                rejected_in_file,
                file_info["path"],
            )
            self._rejected += rejected_in_file

    def __iter__(self):
        for file_info in self.files:
            yield from self._stream_file(file_info)
```

### 3. `surrogate/train.py` (minimal diff, robust)
```python
import argparse
import time
import torch
from pathlib import Path
from lightning.pytorch import Trainer
from lightning.pytorch import Studio
from surrogate.data.cdn_dataset import CdnDataset

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--manifest-path", default="dataset_manifest.json")
    parser.add_argument("--studio-name", default="surrogate-training")
    parser.add_argument("--date-folder", help="Auto-generate manifest if missing")
    args, remaining = parser.parse_known_args()

    # Auto-generate manifest if missing and date_folder provided
    manifest_path = Path(args.manifest_path)
    if not manifest_path.exists() and args.date_folder:
        from scripts.build_dataset_manifest import build_manifest
        build_manifest(args.date_folder, manifest_path)

    dataset = CdnDataset(str(manifest_path))
    dataloader = torch.utils.data.DataLoader(dataset, batch_size=8, num_workers=4)

    # Reuse or create studio with lifecycle guard
    studio = None
    for s in Studio.list():
        if s.name == args.studio_name:
            studio = s
            break

    if studio is None:
        studio = Studio.create(name=args.studio_name, machine="L40S")

    # Wait for running status with timeout
    for _ in range(30):
        studio.refresh()
        if studio.status == "running":
            break
