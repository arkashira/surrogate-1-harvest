# surrogate-1 / design

## Implementation Plan (≤2h)

**Highest-value improvement**: Replace fragile shell-based ingestion with a **manifest-driven, CDN-bypass Python worker** that eliminates HF API rate limits during training data loads and prevents mixed-schema CastErrors.

### Changes

1. **`bin/dataset-enrich.sh`** → **`bin/dataset-enrich.py`**  
   - Single API call (post-rate-limit window) to `list_repo_tree(path, recursive=False)` for one date folder.  
   - Save file list to `manifest.json`.  
   - Workers read manifest, download via CDN (`resolve/main/…`) with **no Authorization header** → bypasses `/api/` rate limits entirely.  
   - Stream-parse each file, project to `{prompt, response}` only, compute md5, emit normalized JSONL.  
   - Deterministic shard assignment via `hash(slug) % 16`.

2. **`lib/dedup.py`** (unchanged interface)  
   - Keep central md5 dedup store; workers call `is_duplicate(md5)` before emit.

3. **`.github/workflows/ingest.yml`**  
   - Matrix `shard_id: [0..15]`.  
   - Each job runs `python bin/dataset-enrich.py --shard $SHARD_ID --date $DATE`.  
   - Upload shard output to `batches/public-merged/<date>/shard<N>-<HHMMSS>.jsonl`.

4. **`requirements.txt`**  
   - Add `requests`, keep `huggingface_hub`, `pyarrow`, `numpy`, `tqdm`.

### `bin/dataset-enrich.py`

```python
#!/usr/bin/env python3
"""
Manifest-driven, CDN-bypass ingestion worker for surrogate-1.
Usage:
  python bin/dataset-enrich.py --shard 0 --date 2026-05-03
"""
import argparse
import hashlib
import json
import os
import sys
from datetime import datetime, timezone
from pathlib import Path

import pyarrow as pa
import pyarrow.parquet as pq
import requests
from huggingface_hub import list_repo_tree, hf_hub_download

REPO = "axentx/surrogate-1-training-pairs"
BASE_CDN = f"https://huggingface.co/datasets/{REPO}/resolve/main"
HF_TOKEN = os.getenv("HF_TOKEN")  # only needed for upload, not CDN reads

# Local imports
sys.path.insert(0, str(Path(__file__).parent.parent))
from lib.dedup import is_duplicate, mark_seen  # noqa

def build_manifest(date_folder: str, out_path: Path) -> list[str]:
    """Single API call to list files for date_folder; save manifest."""
    files = list_repo_tree(REPO, path=date_folder, recursive=False)
    paths = [f.rfilename for f in files if f.rfilename.endswith((".parquet", ".jsonl"))]
    out_path.write_text(json.dumps(paths, indent=2))
    return paths

def cdn_download(url: str, stream=True):
    """Bypass HF API auth; CDN has much higher rate limits."""
    return requests.get(url, stream=stream, timeout=30)

def hash_slug(slug: str) -> int:
    return int(hashlib.sha256(slug.encode()).hexdigest(), 16)

def project_to_pair(row) -> dict:
    """Normalize heterogeneous schemas to {prompt, response}."""
    prompt = row.get("prompt") or row.get("input") or row.get("question") or ""
    response = row.get("response") or row.get("output") or row.get("answer") or ""
    return {"prompt": str(prompt).strip(), "response": str(response).strip()}

def process_file(path: str, shard_id: int) -> list[dict]:
    """Download, parse, project, dedup; return accepted pairs."""
    url = f"{BASE_CDN}/{path}"
    out = []
    try:
        if path.endswith(".parquet"):
            with cdn_download(url, stream=True) as r:
                r.raise_for_status()
                with open("/tmp/tmp.parquet", "wb") as f:
                    for chunk in r.iter_content(chunk_size=8192):
                        f.write(chunk)
            table = pq.read_table("/tmp/tmp.parquet")
            # Avoid mixed-schema CastError: project only needed cols
            rows = table.select(["prompt", "response"]).to_pylist()
            for row in rows:
                pair = project_to_pair(row)
                if not pair["prompt"] or not pair["response"]:
                    continue
                md5 = hashlib.md5(json.dumps(pair, sort_keys=True).encode()).hexdigest()
                if is_duplicate(md5):
                    continue
                pair["md5"] = md5
                pair["source_file"] = path
                out.append(pair)
                mark_seen(md5)
        else:  # jsonl
            with cdn_download(url, stream=True) as r:
                r.raise_for_status()
                for line in r.iter_lines(decode_unicode=True):
                    if not line:
                        continue
                    try:
                        row = json.loads(line)
                    except json.JSONDecodeError:
                        continue
                    pair = project_to_pair(row)
                    if not pair["prompt"] or not pair["response"]:
                        continue
                    md5 = hashlib.md5(json.dumps(pair, sort_keys=True).encode()).hexdigest()
                    if is_duplicate(md5):
                        continue
                    pair["md5"] = md5
                    pair["source_file"] = path
                    out.append(pair)
                    mark_seen(md5)
    except Exception as e:
        print(f"Error processing {path}: {e}", file=sys.stderr)
    return out

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--shard", type=int, required=True, help="0..15")
    parser.add_argument("--date", default=datetime.now(timezone.utc).strftime("%Y-%m-%d"))
    parser.add_argument("--rebuild-manifest", action="store_true")
    args = parser.parse_args()

    date_folder = f"batches/public-merged/{args.date}"
    manifest_path = Path(f"manifest-{args.date}.json")

    if args.rebuild_manifest or not manifest_path.exists():
        print("Building manifest (single API call)...")
        paths = build_manifest(date_folder, manifest_path)
    else:
        paths = json.loads(manifest_path.read_text())

    shard_paths = [p for p in paths if hash_slug(p) % 16 == args.shard]
    print(f"Shard {args.shard}: processing {len(shard_paths)} files")

    out_rows = []
    for p in shard_paths:
        out_rows.extend(process_file(p, args.shard))

    ts = datetime.now(timezone.utc).strftime("%H%M%S")
    out_path = Path(f"shard{args.shard}-{ts}.jsonl")
    with out_path.open("w") as f:
        for row in out_rows:
            f.write(json.dumps(row) + "\n")
    print(f"Wrote {len(out_rows)} pairs to {out_path}")

if __name__ == "__main__":
    main()
```

### `lib/dedup.py` (unchanged interface)

```python
import sqlite3
from pathlib import Path

DB_PATH = Path(__file__).parent.parent / "dedup.db"

def _ensure_db():
    DB_PATH.parent.mkdir(parents=True, exist_ok=True)
    conn = sqlite3.connect(DB_PATH)
    conn.execute("CREATE TABLE IF NOT EXISTS seen (md5 TEXT PRIMARY KEY, ts DATETIME DEFAULT CURRENT_TIMESTAMP)")
    conn.commit()
    conn.close()

def is_duplicate(md5: str) -> bool:
    _ensure_db()
    conn = sqlite3.connect(DB_PATH)
    cur = conn.execute("SELECT 1 FROM seen WHERE md5 = ?", (md5,))
    exists = cur.fetchone() is not None
    conn.close()
    return exists

def mark_seen(md5: str):
    _ensure_db()
    conn = sqlite3.connect(DB_PATH)
    conn.execute("INSERT OR IGNORE INTO seen (md5) VALUES (?)", (md5,))
    conn.commit()
    conn.close()
```
