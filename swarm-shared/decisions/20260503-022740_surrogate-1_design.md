# surrogate-1 / design

## Final Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16`, `DATE`, `HF_TOKEN`, `REPO_ID` (default: `axentx/surrogate-1-training-pairs`)
- Single API call from runner (after rate-limit window) → `list_repo_tree(path=f"batches/public-merged/{DATE}", recursive=False)` → save `file-list.json`
- Deterministic shard assignment: `hash(slug) % SHARD_TOTAL == SHARD_ID`
- Per-file CDN download only: `https://huggingface.co/datasets/{REPO_ID}/resolve/main/{path}` (no Authorization header → bypasses API rate limit)
- Stream-parse each file (parquet/jsonl) → project to `{prompt, response}` only at parse time (avoid mixed-schema `pyarrow.CastError`)
- Central dedup via `lib/dedup.py` (md5 store) — same interface used by HF Space
- Upload shard output to `batches/public-merged/{DATE}/shard{SHARD_ID}-{HHMMSS}.jsonl`
- Retry/backoff on 429: wait 360s; on HF commit cap: deterministic sibling repo fallback (hash slug → pick repo)
- Keep Bash wrapper for cron compatibility: `#!/usr/bin/env bash`, `chmod +x`, invoke via `bash bin/dataset-enrich.py "$@"`, set `SHELL=/bin/bash` in crontab

def deterministic_shard(key: str) -> int:
    return int(hashlib.md5(key.encode()).hexdigest(), 16) % SHARD_TOTAL

def sibling_repo_for(slug: str) -> str:
    idx = int(hashlib.md5(slug.encode()).hexdigest(), 16) % len(SIBLING_REPOS)
    return SIBLING_REPOS[idx]

def backoff_429(attempt: int) -> None:
    wait = 360
    print(f"[{datetime.utcnow().isoformat()}] Rate limited (429). Waiting {wait}s...", file=sys.stderr)
    time.sleep(wait)

def list_date_files(date: str) -> List[str]:
    """Single API call to list files for DATE folder."""
    for attempt in range(5):
        try:
            items = API.list_repo_tree(
                repo_id=REPO_ID,
                path=f"batches/public-merged/{date}",
                repo_type="dataset",
                token=HF_TOKEN,
                recursive=False,
            )
            # items can be dict or FileInfo depending on hf_hub version
            paths = []
            for it in items:
                if isinstance(it, dict):
                    paths.append(it["path"])
                else:
                    paths.append(it.rfilename)
            return [p for p in paths if p]
        except Exception as exc:
            if hasattr(exc, "response") and getattr(exc.response, "status_code", None) == 429:
                backoff_429(attempt)
                continue
            raise
    raise RuntimeError("Exhausted retries listing repo tree.")

def cdn_download(path: str) -> bytes:
    url = CDN_TEMPLATE.format(repo=REPO_ID, path=path)
    resp = requests.get(url, timeout=60)
    resp.raise_for_status()
    return resp.content

def parse_and_project(content: bytes, path: str):
    """Yield {prompt, response} dicts; ignore other schema fields."""
    path = path.lower()
    if path.endswith(".jsonl"):
        for line in content.decode("utf-8").splitlines():
            line = line.strip()
            if not line:
                continue
            try:
                obj = json.loads(line)
            except Exception:
                continue
            prompt = obj.get("prompt") or obj.get("input") or obj.get("text")
            response = obj.get("response") or obj.get("output") or obj.get("completion")
            if prompt is not None and response is not None:
                yield {"prompt": str(prompt), "response": str(response)}
    elif path.endswith(".parquet"):
        import io
        import pyarrow.parquet as pq

        table = pq.read_table(io.BytesIO(content))
        cols = set(table.column_names)
        prompt_col = next((c for c in ("prompt", "input", "text") if c in cols), None)
        response_col = next((c for c in ("response", "output", "completion") if c in cols), None)
        if prompt_col and response_col:
            for i in range(table.num_rows):
                prompt = table.column(prompt_col)[i].as_py()
                response = table.column(response_col)[i].as_py()
                if prompt is not None and response is not None:
                    yield {"prompt": str(prompt), "response": str(response)}
    else:
        # skip unknown
        return
