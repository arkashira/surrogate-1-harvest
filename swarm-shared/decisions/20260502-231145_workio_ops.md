# workio / ops

## 1. Diagnosis

- No storage-level uniqueness constraint on active punches → `(employee_id, punch_type, date)` duplicates on LINE webhook retries and client retries.
- No idempotency mechanism → duplicate `INSERT`s treated as new punches instead of idempotent upserts.
- Punch write path uses plain `INSERT` without conflict handling; races between concurrent clock-ins create multiple rows.
- No deterministic punch-idempotency key → retries cannot be de-duplicated at DB or application level.
- Missing defensive index for fast duplicate detection on write path (no covering index on `(employee_id, punch_type, date)`).

## 2. Proposed change

File: `/opt/axentx/workio/server/src/db/schema.sql`  
Scope:  
- Add unique constraint/index on `(employee_id, punch_type, date)` where `clock_out_time IS NULL` (active punches only).  
- Add `punch_idempotency_key` column (deterministic hash of `employee_id|date|punch_type|date_window`) to support idempotent upserts.  
- Replace `INSERT` path in `/opt/axentx/workio/server/src/routes/punch.ts` (or equivalent) with `INSERT ... ON CONFLICT DO UPDATE` using idempotency key and clock-out merge semantics.

## 3. Implementation

```sql
-- /opt/axentx/workio/server/src/db/schema.sql
-- Add idempotency column (nullable for migration)
ALTER TABLE punches ADD COLUMN IF NOT EXISTS punch_idempotency_key TEXT;

-- Deterministic key: employee_id|date|punch_type|YYYY-MM-DD (date part only)
-- Populate existing rows (safe default)
UPDATE punches
SET punch_idempotency_key = employee_id || '|' || date::text || '|' || punch_type || '|' || date::date::text
WHERE punch_idempotency_key IS NULL;

-- Make NOT NULL after backfill
ALTER TABLE punches ALTER COLUMN punch_idempotency_key SET NOT NULL;

-- Unique constraint on active punches (only one active punch per type per day)
-- Allows multiple closed punches (clock_out_time set) but prevents duplicate actives
CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_active_unique
ON punches (employee_id, punch_type, date::date)
WHERE clock_out_time IS NULL;

-- Fast lookup for upsert path
CREATE INDEX IF NOT EXISTS idx_punches_idempotency
ON punches (punch_idempotency_key);
```

```ts
// /opt/axentx/workio/server/src/routes/punch.ts
// Replace raw INSERT with idempotent upsert
import { Request, Response } from 'express';
import { pool } from '../db';

export async function recordPunch(req: Request, res: Response) {
  const { employee_id, punch_type, latitude, longitude, timestamp } = req.body;

  if (!employee_id || !punch_type) {
    return res.status(400).json({ error: 'Missing required fields' });
  }

  const date = new Date(timestamp || Date.now());
  const dateOnly = date.toISOString().split('T')[0]; // YYYY-MM-DD
  const idempotencyKey = `${employee_id}|${dateOnly}|${punch_type}|${dateOnly}`;

  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // Idempotent upsert:
    // - If no active punch exists (clock_out_time IS NULL), update location/ts if newer.
    // - If active punch exists and this is a clock-out, set clock_out_time.
    // - If closed punch exists and this is a new clock-in, allow new row (handled by partial index on actives).
    const result = await client.query(
      `INSERT INTO punches (
         employee_id, punch_type, date, latitude, longitude,
         clock_in_time, clock_out_time, punch_idempotency_key
       ) VALUES (
         $1, $2, $3, $4, $5,
         CASE WHEN $6 = 'clock_in' THEN $7 ELSE NULL END,
         CASE WHEN $6 = 'clock_out' THEN $7 ELSE NULL END,
         $8
       )
       ON CONFLICT (employee_id, punch_type, date)
       WHERE clock_out_time IS NULL
       DO UPDATE SET
         latitude = EXCLUDED.latitude,
         longitude = EXCLUDED.longitude,
         clock_in_time = CASE
           WHEN EXCLUDED.punch_type = 'clock_in' AND punches.clock_in_time IS NULL THEN EXCLUDED.clock_in_time
           ELSE punches.clock_in_time
         END,
         clock_out_time = CASE
           WHEN EXCLUDED.punch_type = 'clock_out' THEN EXCLUDED.clock_out_time
           ELSE punches.clock_out_time
         END,
         punch_idempotency_key = EXCLUDED.punch_idempotency_key
       RETURNING *`,
      [
        employee_id,
        punch_type,
        date,
        latitude,
        longitude,
        punch_type,
        date,
        idempotencyKey,
      ]
    );

    await client.query('COMMIT');
    return res.json({ ok: true, punch: result.rows[0] });
  } catch (err) {
    await client.query('ROLLBACK');
    console.error('Punch upsert failed', err);
    return res.status(500).json({ error: 'Punch upsert failed' });
  } finally {
    client.release();
  }
}
```
