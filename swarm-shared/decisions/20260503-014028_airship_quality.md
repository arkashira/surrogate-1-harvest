### 1. Pre-list dataset files (mac orchestrator)

Use the correct `rfilename` field and support nested date paths. Run once per date folder.

```bash
# /opt/axentx/airship/scripts/list_hf_files.py
#!/usr/bin/env python3
"""
Usage: python list_hf_files.py <repo> <date_folder> [output.json]
Example: python list_hf_files.py axentx/surrogate-dataset batches/mirror-merged/2026-05-03
"""
import json
import sys
from huggingface_hub import HfApi

def main():
    if len(sys.argv) < 3:
        print("Usage: list_hf_files.py <repo> <date_folder> [output.json]", file=sys.stderr)
        sys.exit(1)
    repo, date_folder = sys.argv[1], sys.argv[2]
    out_path = sys.argv[3] if len(sys.argv) > 3 else "file_manifest.json"

    api = HfApi()
    # Non-recursive to avoid pagination and extra API calls
    tree = api.list_repo_tree(repo=repo, path=date_folder, recursive=False)
    files = [node.rfilename for node in tree if node.type == "file"]

    manifest = {"repo": repo, "date_folder": date_folder, "files": files}
    with open(out_path, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Wrote {len(files)} files to {out_path}")

if __name__ == "__main__":
    main()
```

Run and commit manifest:

```bash

cd /opt/axentx/airship/surrogate
python /opt/axentx/airship/scripts/list_hf_files.py axentx/surrogate-dataset batches/mirror-merged/2026-05-03 file_manifest.json
```

### 2. Update Surrogate training script for CDN-only fetches

Replace `load_dataset(streaming=True)` with direct CDN downloads using the manifest. Zero HF API calls during training; robust to bad files.

```python
# /opt/axentx/airship/surrogate/train.py  (excerpt)
import json
import os
import time
from pathlib import Path
from typing import Iterator, Tuple

import pyarrow.parquet as pq
import requests
from torch.utils.data import IterableDataset

CDN_BASE = "https://huggingface.co/datasets"

class CDNParquetIterable(IterableDataset):
    """
    CDN-only parquet loader.
    file_manifest.json format: {"repo": "...", "date_folder": "...", "files": [...]}
    """
    def __init__(self, manifest_path: str, max_files: int = None, cache_root: str = "cache"):
        with open(manifest_path) as f:
            manifest = json.load(f)
        self.repo = manifest["repo"]
        self.files = manifest["files"]
        if max_files:
            self.files = self.files[:max_files]
        self.cache_root = Path(cache_root)

    def _download_parquet(self, remote_path: str) -> Path:
        local_path = self.cache_root / remote_path.replace("/", "_")
        if local_path.exists():
            return local_path
        local_path.parent.mkdir(parents=True, exist_ok=True)
        url = f"{CDN_BASE}/{self.repo}/resolve/main/{remote_path}"
        resp = requests.get(url, timeout=60)
        resp.raise_for_status()
        local_path.write_bytes(resp.content)
        return local_path

    def _parse_parquet_to_pairs(self, local_path: Path) -> Iterator[Tuple[str, str]]:
        try:
            table = pq.read_table(local_path, columns=["prompt", "response"])
        except Exception as exc:
            print(f"Failed to read {local_path}: {exc}")
            return
        for row in table.to_pylist():
            prompt = row.get("prompt")
            response = row.get("response")
            if isinstance(prompt, str) and isinstance(response, str):
                yield prompt.strip(), response.strip()

    def __iter__(self) -> Iterator[dict]:
        for remote_file in self.files:
            try:
                local_file = self._download_parquet(remote_file)
                for prompt, response in self._parse_parquet_to_pairs(local_file):
                    yield {"prompt": prompt, "response": response}
            except Exception as exc:
                print(f"Skipping {remote_file}: {exc}")
                continue

# Example usage in training loop
if __name__ == "__main__":
    dataset = CDNParquetIterable("file_manifest.json", max_files=50)
    for sample in dataset:
        # feed to surrogate model
        print(sample["prompt"][:60], "...")
```

### 3. Lightning Studio reuse + idle handling

Reuse running studios; restart only if stopped. Avoid quota waste and idle-timeout deaths.

```python
# /opt/axentx/airship/surrogate/launch_studio.py
#!/usr/bin/env python3
import time
import sys
from lightning_sdk import Teamspace, Studio, Machine

TEAMSPACE = "axentx"
STUDIO_NAME = "surrogate-train-l40s"
MACHINE = Machine.L40S

def get_or_create_studio():
    team = Teamspace(TEAMSPACE)
    for s in team.studios:
        if s.name == STUDIO_NAME:
            if s.status == "running":
                print(f"Reusing running studio: {STUDIO_NAME}")
                return s
            else:
                print(f"Studio stopped. Restarting {STUDIO_NAME}...")
                s.start(machine=MACHINE)
                return s
    print(f"Creating studio: {STUDIO_NAME}")
    return Studio.create(
        teamspace=TEAMSPACE,
        name=STUDIO_NAME,
        machine=MACHINE,
        create_ok=True,
    )

def run_training_script(studio: Studio):
    if studio.status != "running":
        studio.start(machine=MACHINE)
        time.sleep(10)  # allow startup
    job = studio.run(
        command="python train.py",
        environment="lightning-thunder",
    )
    print(f"Submitted job: {job.id}")
    return job

if __name__ == "__main__":
    try:
        studio = get_or_create_studio()
        run_training_script(studio)
    except Exception as exc:
        print(f"Failed to launch studio/job: {exc}", file=sys.stderr)
        sys.exit(1)
```
