# Airship Quality Sprint – 2h Incremental Plan

## Highest-Value Increment (< 2h)
**Implement CDN-bypass dataset manifest + Lightning Studio reuse for Surrogate training**  
- **Why**: Eliminates HF 429s during training and saves Lightning quota (top pain points from swarm decisions and past patterns).  
- **Outcome**: Training jobs fetch via CDN only; zero API calls during data load; reuse running studios to save quota.

---

## Implementation Plan

### 1. Add dataset manifest generator (Mac orchestration script)
- Single API call to `list_repo_tree` for a date folder → save `manifest.json`.
- Embed `manifest.json` in Lightning training script.
- Training uses CDN URLs (`https://huggingface.co/datasets/.../resolve/main/...`) with no auth.

### 2. Lightning Studio reuse + idle handling
- Before `Studio(create_ok=True)`, list `Teamspace.studios` and reuse a running studio with matching name.
- If stopped, restart with `target.start(machine=Machine.L40S)`.

### 3. Schema-safe ingestion (project-to-prompt/response only)
- Ensure only `{prompt, response}` are projected before upload; move attribution to filename (`batches/mirror-merged/{date}/{slug}.parquet`).
- Avoid mixed-schema `load_dataset(streaming=True)`.

### `scripts/generate_manifest.py`
```python
#!/usr/bin/env python3
"""
Generate dataset manifest for a date folder.
Run from Mac (or CI) after rate-limit window clears.
"""
import json
import os
import sys
from pathlib import Path

from huggingface_hub import HfApi

HF_REPO = os.getenv("HF_DATASET_REPO", "axentx/surrogate-dataset")
DATE_FOLDER = sys.argv[1] if len(sys.argv) > 1 else "2026-05-03"
OUT_PATH = Path(sys.argv[2]) if len(sys.argv) > 2 else Path("manifest.json")

api = HfApi()
# Non-recursive per folder to minimize pagination/requests
tree = api.list_repo_tree(repo_id=HF_REPO, path=DATE_FOLDER, recursive=False)

files = []
for item in tree:
    if item.type == "file":
        # CDN URL (no auth, bypasses API rate limits)
        cdn_url = f"https://huggingface.co/datasets/{HF_REPO}/resolve/main/{DATE_FOLDER}/{item.path}"
        files.append({
            "path": item.path,
            "cdn_url": cdn_url,
            "size": getattr(item, "size", None)
        })

manifest = {
    "repo": HF_REPO,
    "date_folder": DATE_FOLDER,
    "files": files
}

OUT_PATH.write_text(json.dumps(manifest, indent=2))
print(f"Manifest written to {OUT_PATH} ({len(files)} files)")
```

### `surrogate/train.py` (excerpt)
```python
import json
from pathlib import Path
import torch
from torch.utils.data import IterableDataset, DataLoader
import requests

MANIFEST_PATH = Path("manifest.json")
BATCH_SIZE = 8

class CDNParquetDataset(IterableDataset):
    def __init__(self, manifest_path):
        manifest = json.loads(manifest_path.read_text())
        self.file_urls = [f["cdn_url"] for f in manifest["files"] if f["cdn_url"].endswith(".parquet")]

    def __iter__(self):
        for url in self.file_urls:
            # Stream parquet from CDN (no HF API calls)
            resp = requests.get(url, stream=True, timeout=60)
            resp.raise_for_status()
            with open("/tmp/temp.parquet", "wb") as f:
                for chunk in resp.iter_content(chunk_size=8192):
                    f.write(chunk)

            # Project to {prompt, response} only (schema-safe)
            import pyarrow.parquet as pq
            table = pq.read_table("/tmp/temp.parquet", columns=["prompt", "response"])
            for row in table.to_pylist():
                if row.get("prompt") and row.get("response"):
                    yield row

def main():
    dataset = CDNParquetDataset(MANIFEST_PATH)
    loader = DataLoader(dataset, batch_size=BATCH_SIZE, num_workers=2)

    # Minimal training loop placeholder
    for batch in loader:
        prompts, responses = batch["prompt"], batch["response"]
        # tokenize & train...
        print(f"Batch: {len(prompts)} samples")

if __name__ == "__main__":
    main()
```

### `surrogate/lightning_launcher.py`
```python
#!/usr/bin/env python3
"""
Lightning Studio launcher with reuse + idle handling.
"""
import os
import time
from lightning_sdk import Studio, Teamspace, Machine

TEAMSPACE = os.getenv("LIGHTNING_TEAMSPACE", "default")
STUDIO_NAME = os.getenv("LIGHTNING_STUDIO_NAME", "surrogate-train")
MACHINE = Machine.L40S  # Free tier fallback; paid can use H200 in lightning-lambda-prod

def get_or_create_studio():
    teamspace = Teamspace(name=TEAMSPACE)
    # Reuse running studio if exists
    for s in teamspace.studios:
        if s.name == STUDIO_NAME:
            if s.status == "running":
                print(f"Reusing running studio: {s.name}")
                return s
            else:
                print(f"Studio {s.name} is {s.status}. Restarting...")
                s.start(machine=MACHINE)
                # Wait until running
                while s.status != "running":
                    time.sleep(10)
                    s.refresh()
                return s

    # Create if not found
    print(f"Creating studio {STUDIO_NAME}...")
    return Studio(
        name=STUDIO_NAME,
        teamspace=TEAMSPACE,
        machine=MACHINE,
        create_ok=True
    )

def run_training(script_path="train.py"):
    studio = get_or_create_studio()
    # Ensure studio is running before run
    if studio.status != "running":
        studio.start(machine=MACHINE)
        while studio.status != "running":
            time.sleep(10)
            studio.refresh()

    run = studio.run(
        script=script_path,
        requirements=["torch", "pyarrow", "requests"],
        # Avoid recreating environment each time to save quota
        create_new_environment=False,
    )
    print(f"Started run: {run.id}")
    return run

if __name__ == "__main__":
    run_training()
```
