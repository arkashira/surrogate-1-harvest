# vanguard / design

## Final Consolidated Solution (Correct + Actionable)

### 1) Diagnosis (merged, de-duplicated)
- **No persistent file manifest**: repeated `list_repo_tree`/`load_dataset` calls cause HF API 429s.
- **Lightning Studio not reused**: quota burned on redundant creates/idle/stop cycles.
- **No CDN-only data path**: training uses HF API during data loading instead of direct CDN fetches.
- **Missing deterministic sibling-repo write strategy**: ingestion bursts risk 128 commits/hr/repo cap.
- **No guardrails for rate limits / lifecycle reuse / deterministic routing**.

---

### 2) Proposed Change (merged)
Add a **manifest-first, CDN-only training contract** and **Studio reuse guard**:
- `manifest.py`: one-time Mac/low-cost run → deterministic JSON listing with CDN URLs only.
- `train.py`: accepts `--manifest`; streams via `webdataset`/`aiohttp`/`torchdata`; zero HF API calls during training.
- Studio lifecycle: detect and attach to a running Studio; reuse across restarts.
- Ingestion side: deterministic sibling-repo selection to spread HF commit load.

---

#### `/opt/axentx/vanguard/manifest.py`
```python
#!/usr/bin/env python3
"""
Generate a CDN-only file manifest for a HuggingFace dataset repo.
Run from Mac or low-cost box during rate-limit windows.
"""
import json
import os
import sys
from datetime import datetime, timezone
from pathlib import Path

from huggingface_hub import HfApi

HF_REPO = os.getenv("HF_DATASET_REPO", "datasets/your-dataset")
DATE_FOLDER = os.getenv("DATE_FOLDER", datetime.now(timezone.utc).strftime("%Y-%m-%d"))
OUT_PATH = Path(os.getenv("MANIFEST_OUT", "manifest.json"))

def build_manifest() -> None:
    api = HfApi()
    # Non-recursive per folder to minimize API calls/pagination
    entries = api.list_repo_tree(
        repo_id=HF_REPO,
        path=DATE_FOLDER,
        repo_type="dataset",
        recursive=False,
    )
    files = [e.rfilename for e in entries if e.type == "file" and e.rfilename.endswith(".parquet")]
    manifest = {
        "repo": HF_REPO,
        "date": DATE_FOLDER,
        "generated_utc": datetime.now(timezone.utc).isoformat(),
        "files": [
            {
                "path": f,
                "cdn_url": f"https://huggingface.co/datasets/{HF_REPO}/resolve/main/{f}",
            }
            for f in sorted(files)
        ],
    }
    OUT_PATH.write_text(json.dumps(manifest, indent=2))
    print(f"Wrote {len(files)} files to {OUT_PATH}")

if __name__ == "__main__":
    build_manifest()
```

#### `/opt/axentx/vanguard/train.py`
```python
#!/usr/bin/env python3
"""
CDN-only training launcher for Lightning Studio.
Usage: python train.py --manifest manifest.json
"""
import argparse
import json
import os
import sys
from pathlib import Path

import torch
import torchdata
import webdataset as wds
from lightning import Fabric

try:
    from huggingface_hub import Teamspace
    HAS_TEAMSPACE = True
except Exception:
    Teamspace = None
    HAS_TEAMSPACE = False

def reuse_running_studio(name: str = "vanguard-train"):
    if not HAS_TEAMSPACE:
        return None
    try:
        for s in Teamspace.studios:
            if getattr(s, "name", None) == name and getattr(s, "status", None) == "Running":
                print(f"Reusing running studio: {s.name}")
                return s
    except Exception as e:
        print(f"Studio reuse check skipped: {e}")
    return None

def build_cdn_dataloader(manifest_path: Path, batch_size: int = 32, num_workers: int = 4):
    with open(manifest_path) as f:
        manifest = json.load(f)

    urls = [item["cdn_url"] for item in manifest["files"]]
    if not urls:
        raise RuntimeError("No files found in manifest")

    # WebDataset over HTTP (CDN) — zero HF API calls during training
    dataset = (
        wds.WebDataset(urls, shardshuffle=1000, handler=wds.handlers.warn_and_continue)
        .decode("pil")
        .to_tuple("json", "npy")  # adapt to your shard format
        .map_dict(
            prompt=lambda x: x,
            response=lambda x: x,
        )
        .batched(batch_size, partial=False)
    )
    loader = torchdata.DataLoader(
        dataset,
        batch_size=None,
        num_workers=num_workers,
        persistent_workers=num_workers > 0,
    )
    return loader

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--manifest", type=Path, required=True, help="Path to manifest.json")
    parser.add_argument("--batch-size", type=int, default=32)
    parser.add_argument("--epochs", type=int, default=1)
    parser.add_argument("--limit-train-batches", type=int, default=None)
    args = parser.parse_args()

    # Reuse studio if running
    studio = reuse_running_studio()
    if studio is not None:
        print("Running inside reused studio context")

    fabric = Fabric(accelerator="cuda", devices=1, precision="bf16-mixed")
    fabric.launch()

    loader = build_cdn_dataloader(args.manifest, batch_size=args.batch_size)
    # Replace with your model
    model = torch.nn.Linear(10, 1)  # placeholder
    optimizer = torch.optim.AdamW(model.parameters(), lr=1e-4)

    model, optimizer = fabric.setup(model, optimizer)
    loader = fabric.setup_dataloaders(loader)

    step = 0
    for epoch in range(args.epochs):
        for batch in loader:
            if args.limit_train_batches and step >= args.limit_train_batches:
                break
            # adapt batch to your schema
            loss = model(batch[0].float().sum(-1)).mean()
            fabric.backward(loss)
            optimizer.step()
            optimizer.zero_grad()
            if step % 10 == 0:
                print(f"epoch={epoch} step={step} loss={loss.item():.4f}")
            step += 1

    print("Training complete (CDN-only path)")

if __name__ == "__main__":
    main()
```

```python
import hashlib

def pick_sibling_repo(slug: str, n_siblings: int = 5) -> str:
    """Deterministic sibling repo selection to spread HF commit load."""
    digest = int(hashlib.sha256(slug.encode()).hexdigest(), 16)
    idx = digest % n_siblings
    return f"datasets/your-dataset-sibling-{idx}"
```
