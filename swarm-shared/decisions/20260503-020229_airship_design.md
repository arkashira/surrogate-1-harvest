# airship / design

## Final Synthesis: CDN-First Deterministic Ingestion + Lightning Studio Guard

**Core Thesis**: Eliminate Hugging Face API rate limits during training and prevent quota waste by (1) listing files once on the Mac side, (2) projecting to `{prompt, response}` only, (3) downloading exclusively via CDN during training, (4) deterministically sharding across 5 sibling repos to respect the 128/hr cap, and (5) reusing or safely restarting a Lightning Studio instead of creating duplicates.

### 1. `scripts/list_hub_files.py` (Mac orchestration only)
```python
#!/usr/bin/env python3
"""
Single Mac-side call to list files for a date folder.
Run after HF API rate-limit window clears.
Output: file_list.json
"""
import json
import os
import sys
from huggingface_hub import HfApi

REPO = os.getenv("HF_DATASET_REPO", "axentx/surrogate-mirror")
DATE_FOLDER = sys.argv[1] if len(sys.argv) > 1 else "2026-05-03"
OUT = sys.argv[2] if len(sys.argv) > 2 else "file_list.json"

api = HfApi()
# Non-recursive per folder to minimize pagination/requests
tree = api.list_repo_tree(repo_id=REPO, path=DATE_FOLDER, recursive=False)
files = [f.rfilename for f in tree if f.type == "file"]

with open(OUT, "w") as f:
    json.dump({"repo": REPO, "date": DATE_FOLDER, "files": files}, f, indent=2)

print(f"Wrote {len(files)} files to {OUT}")
```

### 2. `scripts/build_cdn_train_manifest.py`
```python
#!/usr/bin/env python3
"""
Embed file list into a training manifest with CDN URLs only.
Usage: python build_cdn_train_manifest.py file_list.json > cdn_manifest.json
"""
import json
import sys

def build_manifest(file_list_path):
    with open(file_list_path) as f:
        manifest = json.load(f)

    repo = manifest["repo"]
    cdn_files = []
    for fn in manifest["files"]:
        if not fn.endswith(".parquet"):
            continue
        cdn_url = f"https://huggingface.co/datasets/{repo}/resolve/main/{fn}"
        cdn_files.append({"path": fn, "cdn_url": cdn_url})

    out = {
        "repo": repo,
        "date": manifest["date"],
        "files": cdn_files,
    }
    return out

if __name__ == "__main__":
    file_list = sys.argv[1] if len(sys.argv) > 1 else "file_list.json"
    manifest = build_manifest(file_list)
    json.dump(manifest, sys.stdout, indent=2)
```

### 3. Ingestion projection (dataset-mirror)
```python
import pyarrow as pa
import pyarrow.parquet as pq
from pathlib import Path

def project_to_prompt_response(raw: dict) -> dict:
    # Keep only what surrogate training needs
    prompt = raw.get("prompt") or raw.get("input") or ""
    response = raw.get("response") or raw.get("output") or ""
    return {"prompt": prompt, "response": response}

def write_shard(records, date, slug, shard_id, sibling_repos):
    # Deterministic HF sharding to respect 128/hr cap
    import hashlib
    h = int(hashlib.sha256(f"{slug}".encode()).hexdigest(), 16)
    repo_idx = h % len(sibling_repos)
    target_repo = sibling_repos[repo_idx]

    table = pa.Table.from_pylist([project_to_prompt_response(r) for r in records])
    out_dir = Path(f"batches/mirror-merged/{date}")
    out_dir.mkdir(parents=True, exist_ok=True)
    out_path = out_dir / f"{slug}_{shard_id}.parquet"
    pq.write_table(table, out_path)

    # Upload to selected sibling repo (HF API)
    # api.upload_file(path_or_fileobj=out_path, path_in_repo=f"batches/mirror-merged/{date}/{slug}_{shard_id}.parquet", repo_id=target_repo)
    return target_repo, out_path
```

### 4. `train.py` — CDN-only data loader
```python
import json
import torch
from torch.utils.data import IterableDataset
import requests
from tqdm import tqdm
import pyarrow.parquet as pq
import pyarrow as pa
from io import BytesIO

class CDNParquetDataset(IterableDataset):
    def __init__(self, manifest_path):
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.files = self.manifest["files"]

    def _download_cdn(self, url):
        # CDN downloads bypass /api/ auth and rate limits
        r = requests.get(url, timeout=30)
        r.raise_for_status()
        return r.content

    def __iter__(self):
        for item in self.files:
            buf = self._download_cdn(item["cdn_url"])
            table = pq.read_table(pa.BufferReader(buf))
            for row in table.to_pylist():
                yield row["prompt"], row["response"]
```

### 5. Lightning Studio lifecycle guard
```python
from lightning import Studio, Teamspace, Machine

def get_or_create_studio(name, machine_type="L40S"):
    # Reuse running studio to save quota (80hr/mo)
    for s in Teamspace.studios:
        if s.name == name and s.status == "Running":
            print(f"Reusing running studio: {name}")
            return s

    # Lightning H200 only in lambda-prod (paid); fallback to public-prod L40S
    clouds_to_try = ["lightning-lambda-prod", "lightning-public-prod"]
    studio = None
    for cloud in clouds_to_try:
        try:
            studio = Studio.create(
                name=name,
                machine=Machine(machine_type, cloud=cloud),
                create_ok=True,
            )
            print(f"Created studio on {cloud}: {name}")
            break
        except Exception as e:
            print(f"Failed on {cloud}: {e}")
            continue
    return studio

def run_training_with_guard(studio_name, train_script):
    studio = get_or_create_studio(studio_name)
    if studio is None:
        raise RuntimeError("Could not create studio on any cloud")

    # Idle stop kills training; restart if stopped
    if studio.status != "Running":
        print("Studio stopped; restarting...")
        studio.start(machine=studio.machine)

    # Run training
    run = studio.run(strategy="dp", devices=1)
    run.run(train_script)
    return run
```
