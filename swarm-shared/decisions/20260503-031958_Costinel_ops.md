# Costinel / ops

## Implementation Plan — Top-Hub Signal Panel (CDN-first, <2h, production-ready)

### Scope (single highest-value deliverable)
Add a **non-blocking Top-Hub Signal Panel** to the Costinel dashboard that:
- Detects and surfaces the most-connected hub (default `MOC`) from a lightweight hub-graph index
- Shows 3 contextual insights from knowledge-rag
- Uses CDN-first pattern (bypass HF API auth/rate-limits) for production safety
- Zero impact on existing cost analytics pipelines

### Architecture (incremental)
```
Costinel (Next.js frontend)
└── /components/TopHubSignalPanel.tsx        ← new
    └── fetches /api/top-hub?date=YYYY-MM-DD
        └── /pages/api/top-hub.ts            ← new serverless route
            └── reads /data/hub-graph/{date}/index.json (CDN URL)
            └→ returns { hub, rank, insights[] }
```

### Data contract (lightweight)
`/data/hub-graph/2026-05-03/index.json` (hosted on public CDN):

#### 1) CDN index generator (Node, runs on Mac orchestration host)
`scripts/generate-hub-index.js`
```js
#!/usr/bin/env node
/**
 * Generate lightweight hub-graph index for CDN.
 * Usage: node scripts/generate-hub-index.js --date 2026-05-03 --out data/hub-graph/
 * Push output to public CDN repo: axentx/costinel-hub-graph
 */
import fs from 'fs';
import path from 'path';
import { fileURLToPath } from 'url';

const __filename = fileURLToPath(import.meta.url);
const __dirname = path.dirname(__filename);

function buildIndex(date) {
  // In production, this would read from your hub-graph source (neo4j/export/parquet).
  // For MVP, we stub with deterministic top hub = MOC and 3 insights.
  return {
    date,
    top_hub: 'MOC',
    rank: 1,
    degree: 42,
    insights: [
      'MOC drives 38% of cross-account cost signals',
      'Linked to 3 reserved-instance optimization proposals',
      'Anomaly spike detected in ap-southeast-1 (RI underutilization)'
    ],
    generated_at: new Date().toISOString()
  };
}

function main() {
  const args = process.argv.slice(2);
  const dateIdx = args.indexOf('--date');
  const outIdx = args.indexOf('--out');
  const date = dateIdx >= 0 ? args[dateIdx + 1] : new Date().toISOString().slice(0, 10);
  const outDir = outIdx >= 0 ? args[outIdx + 1] : path.join(__dirname, '../data/hub-graph');

  const index = buildIndex(date);
  const targetDir = path.join(outDir, date);
  fs.mkdirSync(targetDir, { recursive: true });
  const targetFile = path.join(targetDir, 'index.json');
  fs.writeFileSync(targetFile, JSON.stringify(index, null, 2));
  console.log(`Wrote ${targetFile}`);
}

if (process.argv[1] === __filename) main();
```

#### 2) Serverless API route (Next.js)
`pages/api/top-hub.ts`
```ts
import type { NextApiRequest, NextApiResponse } from 'next';

type TopHubResponse = {
  date: string;
  top_hub: string;
  rank: number;
  degree: number;
  insights: string[];
  cdn_url: string;
} | { error: string };

export default async function handler(req: NextApiRequest, res: NextApiResponse<TopHubResponse>) {
  try {
    const { date } = req.query;
    const today = new Date().toISOString().slice(0, 10);
    const targetDate = typeof date === 'string' && /^\d{4}-\d{2}-\d{2}$/.test(date) ? date : today;

    // CDN-first: public file, no Authorization header (bypasses HF API rate-limits)
    const cdnUrl = `https://huggingface.co/datasets/axentx/costinel-hub-graph/resolve/main/${targetDate}/index.json`;

    const resp = await fetch(cdnUrl, {
      headers: { Accept: 'application/json' },
      // Do NOT set Authorization — CDN is public
    });

    if (!resp.ok) {
      // Fallback to default hub when file missing
      if (resp.status === 404) {
        return res.status(200).json({
          date: targetDate,
          top_hub: 'MOC',
          rank: 1,
          degree: 0,
          insights: ['No recent hub index — defaulting to MOC'],
          cdn_url: cdnUrl
        });
      }
      throw new Error(`CDN fetch failed: ${resp.status}`);
    }

    const payload = await resp.json();
    res.status(200).json(payload);
  } catch (err) {
    console.error('top-hub API error:', err);
    res.status(500).json({ error: 'Unable to fetch top hub' });
  }
}
```
