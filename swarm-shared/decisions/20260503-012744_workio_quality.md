# workio / quality

## 1. Diagnosis

- **LINE webhook retries create duplicate punches** — no idempotency key or deduplication on `webhookEventId`, so 5xx/timeouts from DB or downstream cause double clock-ins/outs.
- **Race condition on punch create** — `find-then-insert/update` for open punch allows concurrent/close requests to create two open records for same user/tenant.
- **No optimistic UI for clock in/out** — users wait on network and can double-tap; no local pending-state or error recovery, so retries compound duplicates.
- **Missing DB uniqueness constraint** — `punches` table has no unique index to enforce one open punch per user/tenant, relying on app logic only.
- **No idempotency layer for leave/OT requests** — same webhook retry risk applies to leave/OT endpoints; duplicate requests create duplicate approvals/records.

## 2. Proposed change

- **File scope**: `workio/server/src/routes/webhook/line.ts` (webhook handler), `workio/server/src/db/schema.sql` (add constraint + idempotency table), `workio/server/src/services/punchService.ts` (upsert with idempotency), `workio/src/components/AttendanceButton.tsx` (optimistic UI + pending state).
- **Goal**: single-shot idempotent punch processing + optimistic UI to prevent double-tap; minimal surface, high leverage.

## 3. Implementation

### 3.1 DB: idempotency table + uniqueness

```sql
-- workio/server/src/db/schema.sql
-- Add idempotency table for LINE webhook events
CREATE TABLE IF NOT EXISTS line_webhook_idempotency (
  idempotency_key VARCHAR(255) PRIMARY KEY,
  event_type VARCHAR(50) NOT NULL,
  entity_type VARCHAR(50) NOT NULL,     -- e.g., 'punch', 'leave', 'ot'
  entity_id INTEGER NOT NULL,
  tenant_id INTEGER NOT NULL,
  processed_at TIMESTAMPTZ DEFAULT NOW(),
  payload_hash VARCHAR(64),
  UNIQUE (tenant_id, entity_type, entity_id)
);

-- Ensure one open punch per user per tenant
CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_one_open
ON punches (user_id, tenant_id)
WHERE status = 'in';
```

### 3.2 Service: idempotent punch upsert

```ts
// workio/server/src/services/punchService.ts
import { db } from '../db';
import { eq, and } from 'drizzle-orm';
import { punches, lineWebhookIdempotency } from '../db/schema';

type ProcessPunchOpts = {
  tenantId: number;
  userId: number;
  lineWebhookEventId: string;
  type: 'in' | 'out';
  location?: { lat: number; lng: number };
  timestamp: Date;
};

export async function processPunch(opts: ProcessPunchOpts) {
  const { tenantId, userId, lineWebhookEventId, type, location, timestamp } = opts;

  // Idempotency check
  const existing = await db
    .select()
    .from(lineWebhookIdempotency)
    .where(eq(lineWebhookIdempotency.idempotency_key, lineWebhookEventId))
    .limit(1);

  if (existing.length > 0) {
    // Already processed — return stored entity id
    return { punchId: existing[0].entity_id, processed: false };
  }

  return await db.transaction(async (tx) => {
    // Find open punch
    const openPunch = await tx
      .select()
      .from(punches)
      .where(and(
        eq(punches.user_id, userId),
        eq(punches.tenant_id, tenantId),
        eq(punches.status, 'in')
      ))
      .limit(1);

    let punchId: number;

    if (type === 'in') {
      if (openPunch.length > 0) {
        // Already clocked in — update location/timestamp if needed
        const [updated] = await tx
          .update(punches)
          .set {
            location: location ? JSON.stringify(location) : null,
            updated_at: new Date()
          }
          .where(eq(punches.id, openPunch[0].id))
          .returning({ id: punches.id });
        punchId = updated.id;
      } else {
        const [inserted] = await tx
          .insert(punches)
          .values({
            user_id: userId,
            tenant_id: tenantId,
            status: 'in',
            clock_in_at: timestamp,
            location: location ? JSON.stringify(location) : null,
            created_at: new Date(),
            updated_at: new Date()
          })
          .returning({ id: punches.id });
        punchId = inserted.id;
      }
    } else {
      // clock out
      if (openPunch.length === 0) {
        throw new Error('No open punch to clock out');
      }
      const [updated] = await tx
        .update(punches)
        .set({
          status: 'out',
          clock_out_at: timestamp,
          location: location ? JSON.stringify(location) : null,
          updated_at: new Date()
        })
        .where(eq(punches.id, openPunch[0].id))
        .returning({ id: punches.id });
      punchId = updated.id;
    }

    // Record idempotency
    await tx.insert(lineWebhookIdempotency).values({
      idempotency_key: lineWebhookEventId,
      event_type: 'punch',
      entity_type: 'punch',
      entity_id: punchId,
      tenant_id: tenantId,
      payload_hash: '' // optional: hash of stable payload subset
    });

    return { punchId, processed: true };
  });
}
```

### 3.3 Webhook handler: use service + return fast

```ts
// workio/server/src/routes/webhook/line.ts
import { Router } from 'express';
import { processPunch } from '../../services/punchService';
import { verifyLineSignature } from '../../utils/line';

const router = Router();

router.post('/line', verifyLineSignature, async (req, res) => {
  const { events } = req.body;

  // Best-effort async processing; respond 200 quickly to prevent LINE retries
  (async () => {
    for (const ev of events) {
      try {
        if (ev.type === 'message' && ev.message?.type === 'text') {
          // Example: parse clock command or rely on location callback
          continue;
        }
        if (ev.type === 'postback' && ev.postback?.data) {
          // Example: handle clock in/out postback
          const data = JSON.parse(ev.postback.data);
          if (data.action === 'punch') {
            await processPunch({
              tenantId: data.tenantId,
              userId: ev.source.userId,
              lineWebhookEventId: ev.webhookEventId,
              type: data.value, // 'in' | 'out'
              timestamp: new Date(ev.timestamp),
              location: ev.postback.params?.area
            });
          }
        }
      } catch (err) {
        console.error('Webhook processing error', err);
        // Do not throw — we already replied 200; log for retry investigation
      }
    }
  })();

  res.status(200).send('');
});

export default router;
```
