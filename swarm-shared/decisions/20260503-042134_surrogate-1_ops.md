# surrogate-1 / ops

## Final Implementation Plan (≤2h)

**Highest-value improvement**: Replace fragile shell-based ingestion with a **manifest-driven, CDN-bypass Python worker** that eliminates HF API rate limits during training data loads and prevents mixed-schema CastErrors.

### Changes

1. Add `bin/manifest.py` — Mac-side script that lists one date folder via HF API (after rate-limit window), writes `manifest.json` with CDN URLs, sizes, and deterministic shard assignments so workers require zero API calls.
2. Replace `bin/dataset-enrich.sh` with `bin/worker.py` — deterministic shard worker that:
   - Reads `manifest.json` (no API calls)
   - Filters rows by `slug_hash % 16 == SHARD_ID`
   - Downloads assigned files via CDN (`https://huggingface.co/datasets/.../resolve/main/...`) with no Authorization header
   - Projects to `{prompt, response}` only at parse time (avoids pyarrow CastError on mixed schemas)
   - Dedups via central md5 store (reuse existing `lib/dedup.py`)
   - Outputs `batches/public-merged/<date>/shard<N>-<HHMMSS>.jsonl`
3. Update `.github/workflows/ingest.yml` to:
   - Accept `MANIFEST_URL` input (or embed manifest in the workflow via `gh workflow run` dispatch)
   - Run 16 matrix shards using `python bin/worker.py --shard $SHARD_ID --manifest manifest.json`
4. Keep HF token only for final `git push` of produced JSONL (or use `huggingface_hub` upload with retries).

#### bin/manifest.py
```python
#!/usr/bin/env python3
"""
Generate manifest for one date folder.
Run from Mac (or any machine) after rate-limit window clears.
"""
import json, os, hashlib, datetime, sys
from huggingface_hub import HfApi

API = HfApi()
REPO = "datasets/axentx/surrogate-1-training-pairs"
DATE = os.getenv("DATE", datetime.date.today().isoformat())
FOLDER = f"raw/{DATE}"
OUT = os.getenv("OUT", "manifest.json")

def list_files(path: str):
    items = API.list_repo_tree(repo_id=REPO, path=path, recursive=False)
    files = [i for i in items if i.type == "file"]
    folders = [i for i in items if i.type == "dir"]
    result = []
    for f in files:
        result.append({
            "path": f"{path}/{f.name}",
            "cdn": f"https://huggingface.co/datasets/{REPO}/resolve/main/{path}/{f.name}",
            "size": getattr(f, "size", None)
        })
    for d in folders:
        result.extend(list_files(f"{path}/{d.name}"))
    return result

def slug_hash(slug: str) -> int:
    return int(hashlib.md5(slug.encode()).hexdigest(), 16)

def main():
    print(f"Listing {FOLDER}...", file=sys.stderr)
    files = list_files(FOLDER)
    if not files:
        print(f"No files found in {FOLDER}", file=sys.stderr)
        sys.exit(1)

    manifest = {
        "date": DATE,
        "generated_at": datetime.datetime.utcnow().isoformat() + "Z",
        "files": [
            {
                **f,
                "shard": slug_hash(f["path"].split("/")[-1].split(".")[0]) % 16
            }
            for f in files
        ]
    }

    with open(OUT, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Manifest written to {OUT} ({len(files)} files)", file=sys.stderr)

if __name__ == "__main__":
    main()
```

#### bin/worker.py
```python
#!/usr/bin/env python3
"""
Shard worker: CDN-only ingestion with schema projection + dedup.
Usage:
  SHARD_ID=0 python bin/worker.py --manifest manifest.json --out-dir batches/public-merged
"""
import argparse, json, os, sys, hashlib, datetime, requests, pyarrow.parquet as pq, pyarrow as pa
from pathlib import Path

sys.path.insert(0, str(Path(__file__).parent))
from lib.dedup import DedupStore  # existing central md5 store

CDN_TIMEOUT = (10, 30)
CHUNK_SIZE = 8192

def slug_hash(slug: str) -> int:
    return int(hashlib.md5(slug.encode()).hexdigest(), 16)

def project_to_pair(obj, filename: str):
    """
    Project heterogeneous file to {prompt, response}.
    Accepts dict-like row or parquet table.
    """
    prompt_keys = {"prompt", "input", "question", "instruction", "context"}
    response_keys = {"response", "output", "answer", "completion", "result"}

    if isinstance(obj, dict):
        pk = next((k for k in prompt_keys if k in obj), None)
        rk = next((k for k in response_keys if k in obj), None)
        prompt = str(obj[pk]) if pk else ""
        response = str(obj[rk]) if rk else ""
        return {"prompt": prompt.strip(), "response": response.strip()}
    return {"prompt": "", "response": ""}

def read_parquet_cdn(cdn_url: str):
    resp = requests.get(cdn_url, timeout=CDN_TIMEOUT, stream=True)
    resp.raise_for_status()
    data = resp.content
    table = pq.read_table(pa.BufferReader(data))
    return table.to_pylist()

def read_jsonl_cdn(cdn_url: str):
    resp = requests.get(cdn_url, timeout=CDN_TIMEOUT, stream=True)
    resp.raise_for_status()
    rows = []
    for line in resp.iter_lines(decode_unicode=True):
        if line:
            rows.append(json.loads(line))
    return rows

def process_file(cdn_url: str, dedup: DedupStore):
    rows = []
    if cdn_url.endswith(".parquet"):
        rows = read_parquet_cdn(cdn_url)
    elif cdn_url.endswith(".jsonl"):
        rows = read_jsonl_cdn(cdn_url)
    else:
        return []

    out = []
    for row in rows:
        pair = project_to_pair(row, cdn_url)
        if not pair["prompt"] or not pair["response"]:
            continue
        blob = f"{pair['prompt']}\n{pair['response']}".strip()
        md5 = hashlib.md5(blob.encode()).hexdigest()
        if dedup.seen(md5):
            continue
        dedup.add(md5)
        out.append(pair)
    return out

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--shard", type=int, required=True)
    parser.add_argument("--manifest", required=True)
    parser.add_argument("--out-dir", default="batches/public-merged")
    args = parser.parse_args()

    with open(args.manifest) as f:
        manifest = json.load(f)

    date = manifest["date"]
    out_dir = Path(args.out_dir) / date
    out_dir.mkdir(parents=True, exist_ok=True)
    ts = datetime.datetime.utcnow().strftime("%H%M%S")
    out_path = out_dir / f"shard{args.shard}-{ts}.jsonl"

    dedup = DedupStore()

    assigned = [f for f in manifest["files"] if f["shard"] == args.shard]

    print(f"Shard {args.shard}: processing {len(assigned)} files -> {out_path}", file=sys.stderr)

    written = 0
    with out_path.open("w") as out_f:
        for f in assigned:
            try:
                pairs = process_file(f["cdn"], dedup)
                for p in pairs:
                    out_f.write(json.dumps(p, ensure_ascii=False) + "\n")
                    written += 1
            except Exception as e:
                print(f"Error processing {f['cdn']}: {e}", file=sys.stderr)

    print(f"Shard {args.shard}: wrote {written} pairs to {out_path}", file=sys.stderr)

if __name__ == "__main__":
    main()
```
