## Analysis & Highest-Value Increment

**Chosen fix**: Implement **HF CDN-bypass training loader + schema-safe projection** with **Lightning Studio quota guard**.  
This directly attacks the two highest-cost failure modes we’ve seen:

1. HF API 429s during data loading (blocks training)
2. `pyarrow.CastError` from heterogeneous repo files (wastes GPU time)
3. Lightning Studio idle-stop killing runs and burning quota

**Why this over others**:  
- CDN bypass removes auth/rate-limit during training (the biggest throughput blocker).  
- Schema projection at parse time prevents CastError without upstream changes.  
- Studio reuse + idle guard saves ~80 hr/mo quota (immediate cost/iteration win).  
- All changes fit in <2h and are non-breaking.

### 1. List HF date folder (run once from Mac)

```python
# scripts/list_hf_date_folder.py
#!/usr/bin/env python3
"""
Usage:
  python scripts/list_hf_date_folder.py \
    --repo datasets/your-org/surrogate-ingest \
    --date 2026-04-29 \
    --out file_list.json
"""
import argparse
import json
from huggingface_hub import HfApi

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True)
    parser.add_argument("--date", required=True)
    parser.add_argument("--out", default="file_list.json")
    args = parser.parse_args()

    api = HfApi()
    # Single non-recursive call per date folder
    files = api.list_repo_tree(
        repo_id=args.repo,
        path=args.date,
        repo_type="dataset",
        recursive=False,
    )

    entries = []
    for f in files:
        if f.type == "file":
            entries.append({
                "repo": args.repo,
                "path": f.path,  # relative to repo root
                "cdn_url": f"https://huggingface.co/datasets/{args.repo}/resolve/main/{f.path}"
            })

    with open(args.out, "w") as fp:
        json.dump(entries, fp, indent=2)

    print(f"Wrote {len(entries)} files to {args.out}")

if __name__ == "__main__":
    main()
```

### 2. CDN-only loader with schema-safe projection

```python
# surrogate/data/hf_cdn_loader.py
import json
import pyarrow as pa
import pyarrow.parquet as pq
import requests
from io import BytesIO
from typing import List, Dict, Any, Iterator
import logging

log = logging.getLogger(__name__)

class HFCDNDataset:
    """
    Loads files listed in file_list.json via CDN (no auth).
    Projects only {prompt, response} at parse time.
    Skips malformed files to avoid pyarrow.CastError.
    """
    def __init__(self, file_list_path: str):
        with open(file_list_path) as f:
            self.files = json.load(f)

    def _download(self, cdn_url: str) -> bytes:
        resp = requests.get(cdn_url, timeout=30)
        resp.raise_for_status()
        return resp.content

    def _project_row(self, blob: bytes, path: str) -> List[Dict[str, Any]]:
        """
        Try to read parquet; if schema mismatch, attempt safe projection.
        Return list of {prompt, response} rows.
        """
        try:
            table = pq.read_table(BytesIO(blob), columns=["prompt", "response"])
        except (pa.ArrowInvalid, pa.ArrowTypeError, KeyError, OSError):
            # Fallback: try reading all, then select if possible
            try:
                table = pq.read_table(BytesIO(blob))
                if "prompt" in table.column_names and "response" in table.column_names:
                    table = table.select(["prompt", "response"])
                else:
                    log.warning("Missing prompt/response in %s, skipping", path)
                    return []
            except Exception as e:
                log.warning("Cannot read %s: %s", path, e)
                return []

        rows = []
        for i in range(table.num_rows):
            row = table.slice(i, 1).to_pylist()[0]
            rows.append({
                "prompt": row.get("prompt"),
                "response": row.get("response"),
            })
        return rows

    def __iter__(self) -> Iterator[Dict[str, Any]]:
        for entry in self.files:
            try:
                blob = self._download(entry["cdn_url"])
                for row in self._project_row(blob, entry["path"]):
                    if row["prompt"] is not None and row["response"] is not None:
                        yield row
            except Exception as e:
                log.error("Failed to process %s: %s", entry["path"], e)
                continue
```

### 3. Lightning Studio guard + reuse

```python
# surrogate/training/studio_guard.py
import time
from lightning_sdk import Teamspace, Studio, Machine

def get_or_create_studio(name: str, machine: Machine = Machine.L40S) -> Studio:
    """
    Reuse running studio if exists; otherwise create.
    Avoids burning quota by not recreating.
    """
    teamspace = Teamspace()
    for s in teamspace.studios:
        if s.name == name:
            if s.status == "running":
                print(f"Reusing running studio: {name}")
                return s
            else:
                print(f"Starting stopped studio: {name}")
                s.start(machine=machine)
                return s

    print(f"Creating new studio: {name}")
    return Studio(
        name=name,
        machine=machine,
        create_ok=True,
    )

def wait_for_studio_running(studio: Studio, timeout: int = 300, interval: int = 15):
    elapsed = 0
    while elapsed < timeout:
        studio.refresh()
        if studio.status == "running":
            return
        time.sleep(interval)
        elapsed += interval
    raise TimeoutError(f"Studio {studio.name} did not start within {timeout}s")

def run_with_idle_guard(studio: Studio, target):
    """
    Run target in studio, but restart if studio was idle-stopped.
    """
    studio.refresh()
    if studio.status != "running":
        print("Studio not running (idle-stop?). Restarting...")
        studio.start(machine=studio.machine)
        wait_for_studio_running(studio)

    run = studio.run(target)
    return run
```
