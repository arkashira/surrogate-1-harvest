## Final Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16`, `DATE`, `HF_TOKEN` via env
- Single `list_repo_tree(path, recursive=False)` per date folder (avoids 429)
- Writes deterministic file list JSON → embeds it so Lightning training does **CDN-only** fetches (zero API calls during data load)
- Downloads via CDN URLs (`https://huggingface.co/datasets/.../resolve/main/...`) with no Authorization header (bypasses API rate limits)
- Projects each file to `{prompt, response}` at parse time (avoids pyarrow CastError on mixed schemas)
- Deterministic shard assignment: `hash(slug) % SHARD_TOTAL == SHARD_ID`
- Central md5 dedup via existing `lib/dedup.py`
- Outputs: `batches/public-merged/{DATE}/shard{N}-{HHMMSS}.jsonl`
- Returns exit code 0 with summary counts (processed, new, dup, failed)

def init_dedup() -> sqlite3.Connection:
    conn = sqlite3.connect(str(DEDUP_DB))
    conn.execute("CREATE TABLE IF NOT EXISTS seen (md5 TEXT PRIMARY KEY, ts TEXT)")
    conn.commit()

def is_duplicate(conn: sqlite3.Connection, md5: str) -> bool:
    cur = conn.execute("SELECT 1 FROM seen WHERE md5 = ?", (md5,))
    return cur.fetchone() is not None

def mark_seen(conn: sqlite3.Connection, md5: str) -> None:
    ts = datetime.datetime.utcnow().isoformat()
    conn.execute("INSERT OR IGNORE INTO seen (md5, ts) VALUES (?, ?)", (md5, ts))

def download_via_cdn(repo: str, path: str) -> bytes:

    url = f"https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def hash_slug(path: str) -> int:
    return int(hashlib.sha256(path.encode()).hexdigest(), 16)

def belongs_to_shard(path: str, shard_id: int, shard_total: int) -> bool:
    return (hash_slug(path) % shard_total) == shard_id

# ---------- schema projection ----------
def project_to_pair(raw_bytes: bytes, path: str) -> List[Dict[str, str]]:
    """
    Best-effort projection to {prompt,response}.
    Supports:
      - JSONL lines with 'prompt'/'response' (or 'instruction'/'output')
      - JSON array of objects
      - plain text treated as prompt with empty response
    """
    import io
    results = []
    text = raw_bytes.decode("utf-8", errors="replace").strip()
    if not text:
        return results

    # Try JSONL
    for line in io.StringIO(text):
        line = line.strip()
        if not line:
            continue
        try:
            obj = json.loads(line)
        except json.JSONDecodeError:
            continue

        if isinstance(obj, dict):
            prompt = obj.get("prompt") or obj.get("instruction") or obj.get("input") or ""
            response = obj.get("response") or obj.get("output") or obj.get("completion") or ""
            if prompt or response:
                results.append({"prompt": str(prompt), "response": str(response)})
        elif isinstance(obj, list):
            for item in obj:
                if isinstance(item, dict):
                    prompt = item.get("prompt") or item.get("instruction") or ""
                    response = item.get("response") or item.get("output") or ""
                    if prompt or response:
                        results.append({"prompt": str(prompt), "response": str(response)})
    if results:
        return results

    # Fallback: treat whole file as prompt
    return [{"prompt": text[:4000], "response": ""}]
