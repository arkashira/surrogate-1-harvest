### 1) Concrete changes (highest value, <2h)

1. **Deterministic file listing (run once per date folder)**  
   - Use `list_repo_tree(repo_id, path, recursive=False)` per folder (never recursive on big repos).  
   - Save `file_list.json` next to training code or in repo.  
   - On 429: wait 360 s then retry once; fail fast thereafter.

2. **Training script uses CDN-first fetches**  
   - Replace `load_dataset(streaming=True)` with direct CDN URLs or `hf_hub_download` (cached).  
   - Parse only `{prompt, response}` at parse time; ignore extra columns.  
   - Attribution via filename pattern: `batches/mirror-merged/{date}/{slug}.parquet`.

3. **Lightning Studio reuse + idle handling**  
   - Before create, list `Teamspace.studios()` and reuse any running studio with matching name.  
   - Before each `.run()`, check status; if stopped, restart with explicit machine (priority: `lightning-lambda-prod` H200 → L40S → free-tier fallback).  
   - Ensure `target.start(machine=...)` is used on restart.

4. **Wrapper hygiene**  
   - All scripts: `#!/usr/bin/env bash`, `chmod +x`, invoked via `bash <script> "$@"`.  
   - Crontab: set `SHELL=/bin/bash`.

api = HfApi()

for attempt in range(3):

    except Exception as e:

files = [it.rfilename for it in items if not it.rfilename.endswith("/")]
os.makedirs(os.path.dirname(out_path) or ".", exist_ok=True)
with open(out_path, "w") as f:

#### b) CDN-first dataset loader (embedded file list)

```python
# surrogate/data/cdn_loader.py
import json
import pyarrow.parquet as pq
from pathlib import Path
from huggingface_hub import hf_hub_download
from datasets import Dataset

def load_from_cdn(file_list_path: str, cache_dir: str = ".cache/hf"):
    with open(file_list_path) as f:
        manifest = json.load(f)

    rows = []
    for rel in manifest["files"]:
        try:
            local = hf_hub_download(
                repo_id=manifest["repo"],
                filename=rel,
                cache_dir=cache_dir,
            )
        except Exception:
            # CDN direct fallback (no auth, no API rate limit)
            url = f"https://huggingface.co/{manifest['repo']}/resolve/main/{rel}"
            import requests
            r = requests.get(url, timeout=60)
            r.raise_for_status()
            local = Path(cache_dir) / rel.replace("/", "_")
            local.parent.mkdir(parents=True, exist_ok=True)
            local.write_bytes(r.content)

        try:
            tbl = pq.read_table(local, columns=["prompt", "response"])
        except Exception:
            tbl = pq.read_table(local)
            if "prompt" not in tbl.column_names or "response" not in tbl.column_names:
                continue
            tbl = tbl.select(["prompt", "response"])

        for batch in tbl.to_batches(max_chunksize=10_000):
            for prompt, response in zip(batch["prompt"].to_pylist(), batch["response"].to_pylist()):
                if prompt is None or response is None:
                    continue
                rows.append({"prompt": prompt, "response": response})

    return Dataset.from_list(rows)
```

#### c) Lightning Studio reuse + cloud sweep

```python
# surrogate/train/launch.py
import os
import time
from lightning import LightningWork, LightningFlow, LightningApp, Machine

MACHINE_PRIORITY = [
    ("lightning-lambda-prod", "H200"),
    ("lightning-lambda-prod", "L40S"),
    (None, Machine.L40S),
    (None, Machine.CPU),
]

class SurrogateTrainer(LightningWork):
    def __init__(self, studio_name: str = "surrogate-trainer"):
        super().__init__(cloud=LightningWork.CloudChoice.AUTO)
        self.studio_name = studio_name
        self._studio = None

    def _find_running(self):
        from lightning.fabric.utilities.cloud import _get_teamspace
        try:
            team = _get_teamspace()
            for s in team.studios:
                if s.name == self.studio_name and s.status == "running":
                    return s
        except Exception:
            pass
        return None

    def _start_studio(self):
        from lightning import Studio
        running = self._find_running()
        if running:
            self._studio = Studio(running.id)
            return

        # create new with priority machines
        for cloud_name, machine in MACHINE_PRIORITY:
            try:
                if cloud_name:
                    self._studio = Studio.create(
                        self.studio_name,
                        machine=machine,
                        cloud=cloud_name,
                    )
                else:
                    self._studio = Studio.create(
                        self.studio_name,
                        machine=machine,
                    )
                return
            except Exception:
                continue
        raise RuntimeError("Could not start any studio machine.")

    def run(self):
        if self._studio is None:
            self._start_studio()

        # ensure running
        status = self._studio.status
        if status != "running":
            self._studio.start(machine=self._studio.machine or Machine.L40S)
            # poll briefly
            for _ in range(10):
                time.sleep(10)
                if self._studio.status == "running":
                    break

        # execute training command inside studio
        self._studio.run("python -m surrogate.train.main --data-cdn")
```
