## 1. Problem & Goal (resolved)
- **Problem**: Surrogate-1 training blocked by HF API 429 during dataset ingestion (`list_repo_files`/recursive calls).
- **Goal**: Eliminate **all** HF API calls during training. Generate a static CDN file manifest once (on Mac or any dev machine). Training must fetch exclusively via CDN URLs (`resolve/main/...`), never touching `/api/`.

**Why this wins (<2h)**:
- One manifest generation avoids recursive rate-limited calls.
- CDN downloads bypass auth checks and have much higher rate limits.
- No schema/mixing issues — project `{prompt,response}` only at parse time.
- Reuses existing Lightning Studio instead of recreating infrastructure (saves quota).

python3 - <<PY
import json, os, sys
from huggingface_hub import list_files_info

repo = os.environ.get("HF_REPO", "$REPO")
folder = os.environ.get("HF_FOLDER", "$DATE_FOLDER").rstrip("/")

try:
    files_info = list_files_info(repo=repo, path=folder)
except Exception as e:
    print(f"Error listing repo files: {e}", file=sys.stderr)
    sys.exit(1)

files = [
    f.rfilename for f in files_info
    if f.rfilename.endswith(('.parquet', '.jsonl', '.json'))
]

if not files:
    print("No parquet/json/jsonl files found in folder.", file=sys.stderr)
    sys.exit(1)

base = f"https://huggingface.co/datasets/{repo}/resolve/main/{folder}"
manifest = {
    "base_url": base,
    "date_folder": folder,
    "files": files,
    "cdn_urls": [f"{base}/{f}" for f in files]
}

os.makedirs(os.path.dirname("$OUTPUT"), exist_ok=True)
with open("$OUTPUT", "w") as f:
    json.dump(manifest, f, indent=2)

print(f"Manifest written: $OUTPUT ({len(files)} files)")
PY

### 2.2 CDN-Only Data Loader (corrected + robust)
**Critical fixes**:
- Use streaming-friendly iteration with deterministic ordering.
- Validate URLs and handle CDN errors gracefully (retry/backoff optional).
- Ensure schema: only yield rows with non-empty `prompt` and `response`.
- Avoid loading entire parquet into memory at once.

```python
# /opt/axentx/airship/surrogate/train/cdn_dataset.py
import json
import pyarrow.parquet as pq
import requests
from io import BytesIO
from pathlib import Path
from typing import Iterator, Dict, Any, List

MANIFEST_PATH = Path(__file__).parent.parent / "manifests" / "cdn_files.json"

class CDNParquetDataset:
    """Zero HF API calls during training. CDN-only streaming."""
    
    def __init__(self, manifest_path: Path = MANIFEST_PATH, columns=("prompt", "response")):
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.columns = columns
        self.urls: List[str] = self.manifest.get("cdn_urls", [])
        if not self.urls:
            raise ValueError("No CDN URLs found in manifest.")
    
    def _stream_parquet(self, url: str) -> Iterator[Dict[str, Any]]:
        """Download single parquet via CDN and project to {prompt, response}."""
        try:
            resp = requests.get(url, timeout=30)
            resp.raise_for_status()
        except Exception as e:
            raise RuntimeError(f"Failed to fetch {url}: {e}")

        table = pq.read_table(BytesIO(resp.content), columns=self.columns)
        for batch in table.to_batches(max_chunksize=1024):
            for row in batch.to_pylist():
                prompt = row.get("prompt")
                response = row.get("response")
                if prompt and response:
                    yield {"prompt": prompt, "response": response}
    
    def __iter__(self) -> Iterator[Dict[str, Any]]:
        for url in self.urls:
            yield from self._stream_parquet(url)

import pytorch_lightning as pl
from torch.utils.data import IterableDataset, DataLoader

class CDNIterableDataset(IterableDataset):
    def __init__(self, cdn_dataset):
        self.cdn_dataset = cdn_dataset
    
    def __iter__(self):
        # Deterministic order across workers (shuffle externally if needed)
        return iter(self.cdn_dataset)

class SurrogateDataModule(pl.LightningDataModule):
    def __init__(self, batch_size: int = 8, num_workers: int = 4):
        super().__init__()
        self.batch_size = batch_size
        self.num_workers = num_workers
        self.cdn_dataset = CDNParquetDataset()
    
    def train_dataloader(self):
        return DataLoader(
            CDNIterableDataset(self.cdn_dataset),
            batch_size=self.batch_size,
            num_workers=self.num_workers,
            pin_memory=True
        )

# /opt/axentx/airship/surrogate/train/launch_cdn_training.py
#!/usr/bin/env python3
import os
from lightning import LightningWork, LightningFlow, LightningApp
from lightning.pytorch import Trainer
from lightning.pytorch.callbacks import ModelCheckpoint

from cdn_datamodule import SurrogateDataModule
from train_config import SurrogateConfig  # your model config

class SurrogateTrainer(LightningWork):
    def __init__(self):
        super().__init__(
            cloud_compute="L40S",  # or fallback to available
            cloud_build_commands=[
                "pip install pyarrow requests huggingface_hub -q"
            ]
        )
    
    def run(self):
        dm = SurrogateDataModule(batch_size=4)
        model = SurrogateConfig.build_model()
        
        checkpoint = ModelCheckpoint(
            dirpath="/shared/checkpoints",
            filename="surrogate-{epoch:02d}",
            save_top_k=2,
            monitor="train_loss"
        )
        
        trainer = Trainer(
            max_epochs=1,
            accelerator="gpu",
            devices=1,
            callbacks=[checkpoint],
            limit_train_batches=100  # quick iteration
        )
        trainer.fit(model, dm)

class SurrogateFlow(LightningFlow):
    def __init__(self):
        super().__init__()
        self.trainer = SurrogateTrainer()
    
    def run(self):
        self.trainer.run()

app = LightningApp(SurrogateFlow())
```
