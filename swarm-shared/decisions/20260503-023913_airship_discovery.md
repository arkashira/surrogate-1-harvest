#### A. Mac orchestration script — list once, save JSON  
*Strongest from Candidate 1 (explicit, minimal, correct).*

```python
# surrogate/training/cdn_file_list.py
import json
import os
from datetime import datetime, timezone
from huggingface_hub import HfApi

HF_REPO = "datasets/axentx/surrogate-slice"   # change as needed
DATE_FOLDER = "2026-05-03"                    # change per run
OUT_PATH = os.path.join(os.path.dirname(__file__), "dataset", "file_list.json")

def main():
    api = HfApi()
    entries = api.list_repo_tree(
        repo_id=HF_REPO,
        path=DATE_FOLDER,
        repo_type="dataset",
        recursive=False
    )

    files = []
    for e in entries:
        if getattr(e, "type", None) == "file":
            fname = getattr(e, "path", None)
            if fname:
                files.append(fname)

    payload = {
        "repo": HF_REPO,
        "date_folder": DATE_FOLDER,
        "listed_at_utc": datetime.now(timezone.utc).isoformat(),
        "files": sorted(files)
    }

    os.makedirs(os.path.dirname(OUT_PATH), exist_ok=True)
    with open(OUT_PATH, "w", encoding="utf-8") as f:
        json.dump(payload, f, indent=2)

    print(f"Wrote {len(files)} files to {OUT_PATH}")

if __name__ == "__main__":
    main()
```

Run on Mac (once per slice):
```bash
cd /opt/axentx/airship/surrogate/training
python cdn_file_list.py
```

```python
# surrogate/training/train_cdn_slice.py
import json
import os
import io
import torch
import pyarrow.parquet as pq
import requests
from typing import Dict, Any, Iterator

import lightning as L
from lightning.pytorch import Trainer
from lightning.pytorch.loggers import CSVLogger

HF_REPO = "datasets/axentx/surrogate-slice"
FILE_LIST_PATH = os.path.join(os.path.dirname(__file__), "dataset", "file_list.json")
CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

# Optional model import; keep dummy fallback for scaffold runs
try:
    from surrogate.model import SurrogateModel
    MODEL_CLS = SurrogateModel
except Exception:
    MODEL_CLS = None

def load_file_list() -> Dict[str, Any]:
    with open(FILE_LIST_PATH, "r", encoding="utf-8") as f:
        return json.load(f)

def cdn_url(path: str) -> str:
    return CDN_TEMPLATE.format(repo=HF_REPO, path=path)

def stream_parquet_rows(url: str) -> Iterator[Dict[str, str]]:
    """Download single parquet via CDN and yield {prompt, response} rows."""
    resp = requests.get(url, timeout=60)
    resp.raise_for_status()
    table = pq.read_table(io.BytesIO(resp.content))

    # Resolve prompt/response columns robustly
    col_map = {}
    for c in table.column_names:
        low = c.lower()
        if "prompt" in low:
            col_map["prompt"] = c
        elif any(k in low for k in ("response", "completion", "answer")):
            col_map["response"] = c

    if "prompt" not in col_map or "response" not in col_map:
        raise ValueError(f"Missing prompt/response in {url}: {table.column_names}")

    for batch in table.to_batches(max_chunksize=1024):
        cols = batch.columns
        prompt_idx = table.schema.get_field_index(col_map["prompt"])
        response_idx = table.schema.get_field_index(col_map["response"])
        prompts = cols[prompt_idx].to_pylist()
        responses = cols[response_idx].to_pylist()
        for p, r in zip(prompts, responses):
            if p is None or r is None:
                continue
            yield {"prompt": str(p), "response": str(r)}

class CDNDataModule(L.LightningDataModule):
    def __init__(self, batch_size: int = 8):
        super().__init__()
        self.batch_size = batch_size
        self.file_list = load_file_list()
        self.urls = [cdn_url(p) for p in self.file_list["files"] if p.endswith(".parquet")]

    def train_dataloader(self):
        # Materialize small slice for deterministic validation; replace with streaming for scale
        items = []
        for url in self.urls:
            items.extend(stream_parquet_rows(url))
        return torch.utils.data.DataLoader(items, batch_size=self.batch_size, shuffle=True)

class SurrogateCDNTrainer(L.LightningModule):
    def __init__(self, model=None):
        super().__init__()
        self.model = model or (MODEL_CLS() if MODEL_CLS else torch.nn.Linear(10, 1))

    def training_step(self, batch, batch_idx):
        # Replace with real surrogate training logic
        # Minimal scaffold: ensure gradient flow for correctness checks
        dummy = torch.tensor(0.0, device=self.device, requires_grad=True)
        return dummy

    def configure_optimizers(self):
        return torch.optim.Adam(self.parameters(), lr=1e-4)

def main():
    # Prefer reuse pattern to save Lightning quota (Candidate 2 intent)
    try:
        from lightning.pytorch.studio import Studio
        running = [s for s in Studio.list() if s.status == "running"]
        if running:
            print("Reusing running Lightning Studio to save quota.")
            studio = running[0]
            # Attach fit to existing studio context if supported; fallback to local Trainer
            trainer = Trainer(
                max_epochs=1,
                logger=CSVLogger(save_dir="logs/"),
                limit_train_batches=10,
                accelerator="auto",
                devices=1,
            )
        else:
            trainer = Trainer(
                max_epochs=1,
                logger=CSVLogger(save_dir="logs/"),
                limit_train_batches=10,
                accelerator="auto",
                devices=1,
            )
    except Exception:
        trainer = Trainer(
            max_epochs=1,
            logger=CSVLogger(save_dir="logs/"),
            limit_train_batches=10,
            accelerator="auto",
            devices=1,
        )

    data = CDNDataModule(batch_size=4)
    model = SurrogateCDNTrainer()
    trainer.fit(model, datamodule=data)
    print("CDN slice training complete (zero HF API calls during data load).")

if __name__ == "__main__":
    main()
```
