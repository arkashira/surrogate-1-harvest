# workio / backend

## Final consolidated solution (best parts, corrected + actionable)

### 1. Diagnosis (merged, corrected)
- **Race-prone app-layer find-then-insert** under concurrent LINE webhook deliveries or client double-taps.
- **No database-level enforcement** preventing multiple open punches per user per calendar day.
- **No idempotency/deduplication** for LINE retries within the retry window.
- **Success returned before commit is durable**, encouraging client retries that create duplicates.
- **Overlapping punches can occur across day boundaries** if timezone or manual edits shift dates.
- **Missing compensating audit/rejection log** makes reconciliation and debugging hard.

### 2. Scope and constraints
- **Backend-only change**; incremental, <2h.
- **Correctness priority**: enforce exactly-once open punch per user per calendar day (based on `clock_in_at` date in UTC unless business requires local date — choose one and be explicit).
- **Actionability**: atomic upsert, partial unique index, idempotency, safe clock-out, observable rejections.

### 3. Implementation

#### 3.1 DB schema (`/opt/axentx/workio/server/src/db/schema.sql`)
```sql
-- Idempotency for webhook deduplication (retain 24h+ for debug)
CREATE TABLE IF NOT EXISTS webhook_idempotency (
  idempotency_key TEXT NOT NULL PRIMARY KEY,
  handler         TEXT NOT NULL,
  payload_hash    TEXT NOT NULL,
  created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
CREATE INDEX IF NOT EXISTS idx_webhook_idempotency_created
  ON webhook_idempotency (created_at);

-- Optional: lightweight audit for rejected duplicates (helps reconciliation)
CREATE TABLE IF NOT EXISTS punch_rejections (
  id              BIGSERIAL PRIMARY KEY,
  user_id         INT NOT NULL,
  attempted_at    TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  handler         TEXT NOT NULL,
  reason          TEXT NOT NULL,
  client_payload  JSONB,
  idempotency_key TEXT
);
CREATE INDEX IF NOT EXISTS idx_punch_rejections_user_at
  ON punch_rejections (user_id, attempted_at);

-- Partial unique index: at most one open punch per user per calendar day (UTC).
-- If business uses local date, store timezone or compute date in that zone consistently.
CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_one_open_per_day
  ON punches (user_id, DATE(clock_in_at))
  WHERE clock_out_at IS NULL;
```

#### 3.2 Idempotency utility (`server/src/lib/idempotency.ts`)
```ts
import { PoolClient } from 'pg';
import crypto from 'crypto';

export async function enforceIdempotency(
  client: PoolClient,
  key: string,
  handler: string,
  payload: unknown
): Promise<{ isNew: boolean; sameHash: boolean }> {
  const hash = crypto.createHash('sha256').update(JSON.stringify(payload)).digest('hex');
  const res = await client.query(
    `INSERT INTO webhook_idempotency (idempotency_key, handler, payload_hash)
     VALUES ($1, $2, $3)
     ON CONFLICT (idempotency_key) DO UPDATE
     SET payload_hash = EXCLUDED.payload_hash
     WHERE webhook_idempotency.payload_hash <> EXCLUDED.payload_hash
     RETURNING (xmax = 0) AS inserted,
               (webhook_idempotency.payload_hash = EXCLUDED.payload_hash) AS same_hash`,
    [key, handler, hash]
  );
  const row = res.rows[0];
  return { isNew: row?.inserted === true, sameHash: row?.same_hash === true };
}
```

#### 3.3 Atomic punch upsert (`server/src/routes/punch.ts`)
```ts
import { Router } from 'express';
import { pool } from '../db';
import { enforceIdempotency } from '../lib/idempotency';
import crypto from 'crypto';

const router = Router();

/**
 * POST /punch/clock-in
 * Body: { userId: number, latitude?: number, longitude?: number }
 * Idempotency header: X-Idempotency-Key (preferred) or X-Line-Retry-Id
 */
router.post('/clock-in', async (req, res) => {
  const client = await pool.connect();
  try {
    const { userId, latitude, longitude } = req.body;
    const idempotencyKey = req.header('X-Idempotency-Key') || req.header('X-Line-Retry-Id') || crypto.randomUUID();

    await client.query('BEGIN');

    const { isNew, sameHash } = await enforceIdempotency(client, idempotencyKey, 'clock-in', req.body);
    if (!isNew) {
      // Duplicate request; if payload differs, surface conflict, otherwise safe 200
      if (!sameHash) {
        await client.query('ROLLBACK');
        return res.status(409).json({ ok: false, error: 'Idempotency key collision (payload differs)' });
      }
      // Safe duplicate: return existing open punch if any, else last known state
      const dup = await client.query(
        `SELECT id, clock_in_at, latitude, longitude FROM punches
         WHERE user_id = $1 AND clock_out_at IS NULL AND DATE(clock_in_at) = DATE(NOW())
         ORDER BY clock_in_at DESC LIMIT 1`,
        [userId]
      );
      await client.query('ROLLBACK');
      return res.status(200).json({ ok: true, duplicate: true, punch: dup.rows[0] || null });
    }

    // Safety: auto-close any stale open punch for this user+day (should be prevented by unique index)
    await client.query(`
      UPDATE punches
      SET clock_out_at = NOW(), updated_at = NOW(), note = COALESCE(note, '') || ' auto-closed by new punch'
      WHERE user_id = $1
        AND clock_out_at IS NULL
        AND DATE(clock_in_at) = DATE(NOW())
    `, [userId]);

    // Insert new open punch
    const insertRes = await client.query(`
      INSERT INTO punches (user_id, clock_in_at, latitude, longitude)
      VALUES ($1, NOW(), $2, $3)
      RETURNING id, clock_in_at, latitude, longitude
    `, [userId, latitude, longitude]);

    await client.query('COMMIT');
    res.json({ ok: true, punch: insertRes.rows[0] });
  } catch (err: any) {
    await client.query('ROLLBACK');
    // Unique violation from partial index => duplicate open punch exists
    if (err.code === '23505') {
      // Record rejection for audit
      try {
        await pool.query(
          `INSERT INTO punch_rejections (user_id, handler, reason, client_payload, idempotency_key)
           VALUES ($1, $2, $3, $4, $5)`,
          [req.body.userId, 'clock-in', 'Duplicate open punch', req.body, idempotencyKey]
        );
      } catch (_) {
        // best-effort audit
      }
      return res.status(409).json({ ok: false, error: 'Already clocked in for today' });
    }
    console.error('Clock-in error', err);
    res.status(500).json({ ok: false, error: 'Internal error' });
  } finally {
    client.release();
  }
});
