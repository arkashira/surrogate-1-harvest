# surrogate-1 / design

## Final Implementation Plan (≤2h)

**Highest-value improvement**: Replace fragile shell-based ingestion with a **manifest-first, CDN-bypass Python worker** that eliminates HF API rate limits during training data loads and prevents mixed-schema CastErrors.

### Changes

1. **`bin/dataset-enrich.sh`** → **`bin/dataset-enrich.py`**
   - Single API call (`list_repo_tree`) per date folder → save `manifest.json`
   - Workers read manifest, download via CDN URLs (`resolve/main/...`) with no auth
   - Project to `{prompt, response}` only at parse time; skip mixed columns
   - Deterministic shard assignment via `hash(slug) % 16`
   - Output: `batches/public-merged/{date}/shard{N}-{HHMMSS}.jsonl`

2. **`lib/dedup.py`** (no change) — keep central md5 store as source of truth.

3. **`.github/workflows/ingest.yml`**
   - Update matrix runner to invoke `python bin/dataset-enrich.py`
   - Pass `SHARD_ID`, `DATE`, `MANIFEST_PATH` as env vars.

4. **`requirements.txt`**
   - Add `requests` (for CDN downloads), keep `datasets`, `huggingface_hub`, `pyarrow`, `numpy`.

### `bin/dataset-enrich.py`
```python
#!/usr/bin/env python3
"""
Manifest-first, CDN-bypass worker for surrogate-1 public dataset ingestion.

Usage (via GitHub Actions matrix):
    SHARD_ID=0 DATE=2026-04-29 python bin/dataset-enrich.py
"""
import json
import hashlib
import os
import sys
import time
from datetime import datetime, timezone
from pathlib import Path

import requests
from huggingface_hub import HfApi, list_repo_tree

HF_REPO = "datasets/axentx/surrogate-1-training-pairs"
API = HfApi()
CDN_ROOT = f"https://huggingface.co/{HF_REPO}/resolve/main"

# Deterministic shard assignment
def shard_for_slug(slug: str, n_shards: int = 16) -> int:
    return int(hashlib.md5(slug.encode()).hexdigest(), 16) % n_shards

def build_manifest(date: str) -> list[str]:
    """Single API call: list files for one date folder (non-recursive)."""
    folder = f"public-raw/{date}"
    try:
        items = list_repo_tree(repo_id=HF_REPO, path=folder, recursive=False)
    except Exception as exc:
        print(f"ERROR listing {folder}: {exc}", file=sys.stderr)
        return []
    # Keep only files we want to process (parquet/jsonl)
    paths = [it.rfilename for it in items if it.type == "file"]
    return paths

def download_via_cdn(path: str) -> bytes:
    """CDN download — no Authorization header, bypasses API rate limits."""
    url = f"{CDN_ROOT}/{path}"
    resp = requests.get(url, timeout=30)
    resp.raise_for_status()
    return resp.content

def parse_parquet_project(content: bytes):
    """Project to {prompt, response} only; ignore mixed schema columns."""
    import pyarrow.parquet as pq
    import io
    table = pq.read_table(io.BytesIO(content))
    cols = set(table.column_names)
    # Keep only known target fields; drop everything else
    keep = {"prompt", "response"}
    present = [c for c in keep if c in cols]
    if not present:
        return []
    df = table.select(present).to_pandas()
    # Normalize column names
    if "prompt" not in df.columns:
        df = df.rename(columns={"Prompt": "prompt", "prompt_text": "prompt"})
    if "response" not in df.columns:
        df = df.rename(columns={"response": "response", "completion": "response"})
    # Drop nulls
    df = df.dropna(subset=["prompt", "response"])
    return df.to_dict(orient="records")

def parse_jsonl_project(content: bytes):
    import json
    out = []
    for line in content.decode().splitlines():
        line = line.strip()
        if not line:
            continue
        try:
            obj = json.loads(line)
        except json.JSONDecodeError:
            continue
        prompt = obj.get("prompt") or obj.get("Prompt") or obj.get("prompt_text")
        response = obj.get("response") or obj.get("Response") or obj.get("completion")
        if prompt and response:
            out.append({"prompt": prompt, "response": response})
    return out

def process_file(path: str):
    content = download_via_cdn(path)
    if path.endswith(".parquet"):
        return parse_parquet_project(content)
    elif path.endswith(".jsonl"):
        return parse_jsonl_project(content)
    else:
        print(f"SKIP unknown file type: {path}")
        return []

def main():
    shard_id = int(os.environ.get("SHARD_ID", 0))
    date = os.environ.get("DATE", datetime.now(timezone.utc).strftime("%Y-%m-%d"))
    out_dir = Path(f"batches/public-merged/{date}")
    out_dir.mkdir(parents=True, exist_ok=True)

    manifest_path = os.environ.get("MANIFEST_PATH", f"manifest-{date}.json")
    manifest_file = Path(manifest_path)

    # Build or load manifest
    if manifest_file.exists():
        paths = json.loads(manifest_file.read_text())
    else:
        paths = build_manifest(date)
        manifest_file.write_text(json.dumps(paths, indent=2))

    # Filter to this shard's work
    my_paths = [p for p in paths if shard_for_slug(p, 16) == shard_id]
    print(f"SHARD {shard_id}: processing {len(my_paths)} files from {len(paths)} total")

    results = []
    seen_hashes = set()
    for i, path in enumerate(my_paths):
        try:
            pairs = process_file(path)
        except Exception as exc:
            print(f"WARN {path}: {exc}")
            continue
        for pair in pairs:
            # Lightweight dedup by content hash (local to worker; central store is authoritative)
            blob = f"{pair['prompt']}\n{pair['response']}".encode()
            h = hashlib.md5(blob).hexdigest()
            if h in seen_hashes:
                continue
            seen_hashes.add(h)
            results.append(pair)

        # Periodic flush to avoid OOM on large shards
        if (i + 1) % 10 == 0:
            print(f"  processed {i+1}/{len(my_paths)}")

    # Write output
    ts = datetime.now(timezone.utc).strftime("%H%M%S")
    out_file = out_dir / f"shard{shard_id}-{ts}.jsonl"
    with out_file.open("w") as f:
        for pair in results:
            f.write(json.dumps(pair, ensure_ascii=False) + "\n")

    print(f"WROTE {out_file}  records={len(results)}")

if __name__ == "__main__":
    main()
```
