# surrogate-1 / design

## Final Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16`, `DATE`, `HF_TOKEN`, `REPO_ID` (default: `axentx/surrogate-1-training-pairs`)
- Uses **manifest-first strategy**: single `list_repo_tree` call (per date folder) → embed file list → worker does **CDN-only downloads** (no API/auth during training/ingest) to bypass HF 429 limits
- Projects heterogeneous HF datasets to `{prompt, response}` only at parse time (avoids pyarrow `CastError` on mixed schemas)
- Deterministic sharding via `hash(slug) % SHARD_TOTAL` so same file always maps to same shard across runs
- Central dedup via `lib/dedup.py` (SQLite md5 store) — cross-run dedup survives across GitHub Action invocations
- Outputs `batches/public-merged/{DATE}/shard{N}-{HHMMSS}.jsonl` with no extra metadata columns (attribution via filename pattern)
- Reuses running HF Space when possible (Lightning Studio reuse pattern) for any downstream compute
- Mac-only orchestration; all heavy compute on remote (Lightning/Kaggle/Cerebras)

def read_parquet_to_pairs(path: str) -> List[Dict[str, str]]:
    import pyarrow.parquet as pq
    table = pq.read_table(path, columns=["prompt", "response"])
    out = []
    for i in range(table.num_rows):
        row = table.slice(i, 1).to_pydict()
        prompt = row.get("prompt", [None])[0]
        response = row.get("response", [None])[0]
        if prompt and response:
            out.append({"prompt": str(prompt), "response": str(response)})
    return out

def read_jsonl_to_pairs(path: str) -> List[Dict[str, str]]:
    out = []
    with open(path, "r", encoding="utf-8") as f:
        for line in f:
            line = line.strip()
            if not line:
                continue
            obj = json.loads(line)
            prompt = obj.get("prompt") or obj.get("input") or obj.get("question")
            response = obj.get("response") or obj.get("output") or obj.get("answer")
            if prompt and response:
                out.append({"prompt": str(prompt), "response": str(response)})
    return out

def read_csv_to_pairs(path: str) -> List[Dict[str, str]]:
    import csv
    out = []
    with open(path, "r", encoding="utf-8") as f:
        reader = csv.DictReader(f)
        for row in reader:
            prompt = row.get("prompt") or row.get("input") or row.get("question")
            response = row.get("response") or row.get("output") or row.get("answer")
            if prompt and response:
                out.append({"prompt": str(prompt), "response": str(response)})
    return out

def deterministic_shard(slug: str, total: int) -> int:
    digest = hashlib.sha256(slug.encode("utf-8")).hexdigest()
    return int(digest, 16) % total

def list_repo_tree(repo_id: str, folder: str = "", token: Optional[str] = None) -> List[str]:
    """
    Single API call: list top-level or folder tree non-recursive.
    Caller should call per-date folder to avoid 100x pagination on big repos.
    """
    files = HF_API.list_repo_tree(repo_id=repo_id, path=folder, recursive=False, token=token)
    paths = []
    for f in files:
        if f.type == "file":
            paths.append(f.path)
        else:
            # If you want recursion, call per subfolder separately (avoid recursive=True on big repos)
            subpaths = list_repo_tree(repo_id, f.path, token=token)
            paths.extend(subpaths)
    return paths

def download_via_cdn(repo: str, path: str, dest: str, token: Optional[str] = None) -> str:
    url = HF_CDN_TEMPLATE.format(repo=repo, path=path)
    headers = {"Authorization": f"Bearer {token}"} if token else {}
    # CDN downloads typically don't require auth for public datasets; include token only if provided
    r = requests.get(url, headers=headers, stream=True, timeout=60)
    r.raise_for_status()
    with open(dest, "wb") as f:
        for chunk in r.iter_content(chunk_size=8192):
            f.write(chunk)
    return dest

def reuse_running_studio(name: str) -> Optional["Studio"]:

    except Exception:
        return None
    return None

def ensure_studio_running(name: str, machine: str = "L40S") -> Optional["Studio"]:
