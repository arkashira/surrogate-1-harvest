**Chosen improvement**  
Replace any remaining recursive HF ingestion / `load_dataset(streaming=True)` in surrogate training with a **manifest-driven, CDN-only iterable loader** — generated once on the Mac orchestrator and consumed by Lightning Studio training with zero HF API calls during data load.

This eliminates 429 rate limits, avoids pyarrow schema mismatches, and fits the established pattern (HF CDN bypass + pre-list + studio reuse).

### `scripts/generate_hf_manifest.py`

```python
#!/usr/bin/env python3
"""
Generate a manifest for HuggingFace dataset files using non-recursive tree calls
and CDN URLs to avoid HF API rate limits during training.
"""
import json
import os
import sys
import time
from pathlib import Path
from typing import List, Dict

import requests

HF_API_BASE = "https://huggingface.co/api"
RETRY_WAIT = 360  # seconds on 429

def list_tree_nonrecursive(repo: str, path: str = "") -> List[Dict]:
    """
    Use list_repo_tree(path, recursive=False) per folder to avoid pagination blowup.
    Returns list of {'path': ..., 'type': 'file'|'dir', 'size': ...}
    """
    url = f"{HF_API_BASE}/datasets/{repo}/tree"
    params = {"path": path, "recursive": "false"}
    while True:
        resp = requests.get(url, params=params, timeout=30)
        if resp.status_code == 429:
            print(f"Rate limited. Waiting {RETRY_WAIT}s...")
            time.sleep(RETRY_WAIT)
            continue
        resp.raise_for_status()
        return resp.json()

def build_manifest(repo: str, root: str, out_path: Path):
    manifest = []
    stack = [root]

    while stack:
        folder = stack.pop()
        entries = list_tree_nonrecursive(repo, folder)
        for e in entries:
            if e["type"] == "dir":
                stack.append(e["path"])
            else:
                # Only include likely parquet/jsonl files; adapt as needed
                if e["path"].endswith((".parquet", ".jsonl", ".json")):
                    manifest.append(
                        {
                            "repo": repo,
                            "path": e["path"],
                            "size": e.get("size", 0),
                            "sha": e.get("oid", ""),
                            "url": f"https://huggingface.co/datasets/{repo}/resolve/main/{e['path']}",
                        }
                    )

    out_path.parent.mkdir(parents=True, exist_ok=True)
    with out_path.open("w") as f:
        json.dump(manifest, f, indent=2)

    # Also write simple filelist for shell iteration
    list_path = out_path.with_suffix(".txt")
    with list_path.open("w") as f:
        for item in manifest:
            f.write(f"{item['repo']}/{item['path']}\n")

    print(f"Wrote {len(manifest)} files to {out_path} and {list_path}")

if __name__ == "__main__":
    if len(sys.argv) < 3:
        print("Usage: generate_hf_manifest.py <repo> <root_folder> [out_json]")
        sys.exit(1)
    repo = sys.argv[1]
    root = sys.argv[2]
    out = Path(sys.argv[3]) if len(sys.argv) > 3 else Path("manifest.json")
    build_manifest(repo, root, out)
```
