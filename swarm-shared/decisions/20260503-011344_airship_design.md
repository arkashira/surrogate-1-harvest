### Key Decisions (resolve contradictions)

- **Manifest + CDN only**: Both candidates agree; use CDN URLs exclusively to avoid 429s. No HF `load_dataset` during training.
- **Studio reuse**: Reuse running Studio first; create only if needed. Use `Machine.L40S` as primary; on failure, surface error clearly (do not silently retry same machine). If org policy allows a free-tier fallback, document it explicitly; otherwise fail fast so operator can choose machine.
- **Schema projection**: Use heuristic column matching (`prompt/instruction/input` × `response/output/completion`) with deterministic fallback to first two text columns. Reject files that cannot be projected (do not guess). This prevents `pyarrow.CastError` and silent schema drift.
- **Frontend scope**: Minimal control (date picker + trigger). Do not build full training UI now.

# scripts/list_hf_date_folder.py
import os
import json
import argparse
from huggingface_hub import HfApi

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", default="axentx/surrogate-mirror")
    parser.add_argument("--date", required=True, help="YYYY-MM-DD under batches/mirror-merged/")
    parser.add_argument("--out", default=None)
    args = parser.parse_args()

    api = HfApi()
    path = f"batches/mirror-merged/{args.date}"
    tree = api.list_repo_tree(repo_id=args.repo, path=path, recursive=False)
    files = [f.rfilename for f in tree if f.type == "file"]

    out_path = args.out or f"manifests/file-list-{args.date}.json"
    os.makedirs(os.path.dirname(out_path), exist_ok=True)
    with open(out_path, "w") as f:
        json.dump({"repo": args.repo, "date": args.date, "path": path, "files": files}, f, indent=2)
    print(f"Wrote {len(files)} files to {out_path}")

if __name__ == "__main__":
    main()
```

# scripts/build_training_manifest.py
import json
import argparse
import os

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--file-list", required=True)
    parser.add_argument("--out", default=None)
    args = parser.parse_args()

    with open(args.file_list) as f:
        meta = json.load(f)

    entries = []
    for fname in meta["files"]:
        cdn_url = f"https://huggingface.co/datasets/{meta['repo']}/resolve/main/{fname}"
        entries.append({"filename": fname, "cdn_url": cdn_url})

    manifest = {
        "repo": meta["repo"],
        "date": meta["date"],
        "path": meta["path"],
        "entries": entries
    }

    out_path = args.out or f"manifests/train-manifest-{meta['date']}.json"
    os.makedirs(os.path.dirname(out_path), exist_ok=True)
    with open(out_path, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Wrote manifest with {len(entries)} entries to {out_path}")

if __name__ == "__main__":
    main()
```

#### 3. Safe projection to `{prompt, response}`

```python
# surrogate/data.py
import pyarrow.parquet as pq
import requests
import io
from typing import List, Dict

def project_to_prompt_response(file_path_or_url: str) -> List[Dict[str, str]]:
    """
    Download a single file (CDN or local) and project to {prompt, response}.
    Avoids load_dataset(streaming=True) on heterogeneous repos.
    Raises if projection is impossible.
    """
    if file_path_or_url.startswith("http"):
        resp = requests.get(file_path_or_url, timeout=60)
        resp.raise_for_status()
        buf = io.BytesIO(resp.content)
        table = pq.read_table(buf)
    else:
        table = pq.read_table(file_path_or_url)

    cols = set(table.column_names)
    prompt_col = next((c for c in ["prompt", "instruction", "input"] if c in cols), None)
    response_col = next((c for c in ["response", "output", "completion"] if c in cols), None)

    if prompt_col is None or response_col is None:
        text_cols = [c for c in cols if table.schema.field(c).type in (pq.string(), pq.large_string())]
        if len(text_cols) >= 2:
            prompt_col, response_col = text_cols[0], text_cols[1]
        else:
            raise ValueError(f"Cannot project {file_path_or_url}: no prompt/response columns found")

    prompts = table.column(prompt_col).to_pylist()
    responses = table.column(response_col).to_pylist()
    return [{"prompt": str(p), "response": str(r)} for p, r in zip(prompts, responses)]
```

# scripts/launch_studio_training.py
import os

from lightning_sdk import Teamspace, Studio, Machine

def find_running_studio(name: str):
    teamspace = Teamspace()
    running = [s for s in teamspace.studios if s.name == name and s.status == "running"]
    return running[0] if running else None

def ensure_studio_running(name: str, preferred_machine=Machine.L40S):
    studio = find_running_studio(name)
    if studio:
        print(f"Reusing running studio: {studio.id}")
        return studio

    # Try preferred machine; fail fast if unavailable (operator can retry with another)
