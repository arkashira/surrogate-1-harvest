# workio / frontend

## 1. Diagnosis

- **No resilient offline-first clock-in/out**: current optimistic UI likely relies on localStorage queue (per past decisions) which is fragile and size-limited; IndexedDB queue was recommended but probably not implemented.
- **Missing idempotency key on attendance mutations**: duplicate submissions on flaky networks can create phantom punches or double entries.
- **No GPS pre-check UX**: users can attempt clock-in and only then discover location mismatch, causing friction and support load.
- **No local draft persistence for leave/OT requests**: partially filled forms are lost on navigation/refresh.
- **Missing sync-status indicator**: users can’t tell whether their punch/request is pending, synced, or failed.

## 2. Proposed change

**Scope**: `src/hooks/useAttendance.ts` (create) + `src/components/ClockInOutButton.tsx` (modify) + `src/lib/idbQueue.ts` (create).  
Target: replace localStorage queue with IndexedDB, add idempotency keys, add GPS pre-check, add sync-status UI, and persist leave/OT drafts to IndexedDB.

### 3.1 Create `src/lib/idbQueue.ts`

```ts
// src/lib/idbQueue.ts
import { openDB, DBSchema, IDBPDatabase } from 'idb';

interface WorkioDB extends DBSchema {
  queue: {
    key: string;
    value: {
      id: string;
      type: 'attendance' | 'leave' | 'ot';
      payload: unknown;
      retries: number;
      createdAt: number;
    };
    indexes: { type: string; createdAt: number };
  };
  drafts: {
    key: string;
    value: { formKey: string; value: unknown; updatedAt: number };
  };
}

const DB_NAME = 'workio';
const DB_VERSION = 1;

async function getDB(): Promise<IDBPDatabase<WorkioDB>> {
  return openDB<WorkioDB>(DB_NAME, DB_VERSION, {
    upgrade(db) {
      db.createObjectStore('queue', { keyPath: 'id' });
      const queueStore = db.transaction.objectStore('queue');
      queueStore.createIndex('type', 'type', { unique: false });
      queueStore.createIndex('createdAt', 'createdAt', { unique: false });

      db.createObjectStore('drafts', { keyPath: 'formKey' });
    },
  });
}

export async function enqueueJob(
  type: 'attendance' | 'leave' | 'ot',
  payload: unknown
): Promise<string> {
  const id = `${type}_${Date.now()}_${crypto.randomUUID()}`;
  const db = await getDB();
  await db.add('queue', {
    id,
    type,
    payload,
    retries: 0,
    createdAt: Date.now(),
  });
  return id;
}

export async function peekQueue(
  limit = 50
): Promise<WorkioDB['queue']['value'][]> {
  const db = await getDB();
  return db.getAllFromIndex('queue', 'createdAt', undefined, limit);
}

export async function dequeueJob(id: string): Promise<void> {
  const db = await getDB();
  await db.delete('queue', id);
}

export async function incrementRetries(id: string): Promise<void> {
  const db = await getDB();
  const job = await db.get('queue', id);
  if (job) {
    job.retries += 1;
    await db.put('queue', job);
  }
}

export async function saveDraft(formKey: string, value: unknown): Promise<void> {
  const db = await getDB();
  await db.put('drafts', { formKey, value, updatedAt: Date.now() });
}

export async function loadDraft(formKey: string): Promise<unknown | undefined> {
  const db = await getDB();
  const draft = await db.get('drafts', formKey);
  return draft?.value;
}

export async function clearDraft(formKey: string): Promise<void> {
  const db = await getDB();
  await db.delete('drafts', formKey);
}
```

### 3.2 Create `src/hooks/useAttendance.ts`

```ts
// src/hooks/useAttendance.ts
import { useState, useEffect, useCallback } from 'react';
import { enqueueJob, peekQueue, dequeueJob, incrementRetries } from '../lib/idbQueue';
import { useSyncWorker } from './useSyncWorker';

type AttendancePayload = {
  type: 'in' | 'out';
  lat: number;
  lng: number;
  accuracy?: number;
  idempotencyKey: string;
};

export function useAttendance() {
  const [status, setStatus] = useState<'idle' | 'pending' | 'synced' | 'error'>('idle');
  const [lastIdempotencyKey, setLastIdempotencyKey] = useState<string | null>(null);
  const { startSync } = useSyncWorker();

  // Start background sync processor on mount
  useEffect(() => {
    startSync();
  }, [startSync]);

  const requestLocation = useCallback(
    (): Promise<GeolocationPosition> =>
      new Promise((resolve, reject) => {
        if (!navigator.geolocation) {
          reject(new Error('Geolocation not supported'));
          return;
        }
        navigator.geolocation.getCurrentPosition(resolve, reject, {
          enableHighAccuracy: true,
          timeout: 10000,
          maximumAge: 30000,
        });
      }),
    []
  );

  const clock = useCallback(
    async (type: 'in' | 'out') => {
      setStatus('pending');
      try {
        const pos = await requestLocation();
        const idempotencyKey = crypto.randomUUID();
        const payload: AttendancePayload = {
          type,
          lat: pos.coords.latitude,
          lng: pos.coords.longitude,
          accuracy: pos.coords.accuracy,
          idempotencyKey,
        };

        // Optimistic local state
        setLastIdempotencyKey(idempotencyKey);

        // Persist to IndexedDB queue
        await enqueueJob('attendance', payload);
        setStatus('pending');

        // Trigger immediate sync attempt
        startSync();
      } catch (err) {
        console.error(err);
        setStatus('error');
      }
    },
    [requestLocation, startSync]
  );

  return { clockIn: () => clock('in'), clockOut: () => clock('out'), status, lastIdempotencyKey };
}
```
