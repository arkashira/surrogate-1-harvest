# workio / frontend

## Final Synthesized Solution

**Chosen approach:** Adopt Candidate 1’s concrete implementation as the baseline, augmented with Candidate 2’s operational safeguards (queue observability, visibilitychange handling, strict server-side idempotency, and clear migration path). Contradictions are resolved in favor of durability, idempotency, and immediate actionability.

---

## 1. Diagnosis (merged)

- `localStorage` buffering is ephemeral and eviction-prone; flaky networks and reloads cause duplicates/loss.
- No idempotency or retry/backoff; transient failures become data loss.
- Cross-tab races cause duplicate submissions.
- Offline-first is brittle without durable persistence and online-triggered flush.
- Missing observability (pending counts, errors, retry schedule) hinders debugging/support.

```ts
import { openDB, DBSchema, IDBPDatabase } from 'idb';

// Stable event shape
interface AttendanceEvent {
  id: string; // idempotency key (uuid)
  type: 'clock_in' | 'clock_out' | 'break_start' | 'break_end';
  payload: Record<string, unknown>;
  createdAt: number;
  attempts: number;
  lastAttemptAt?: number;
  nextRetryAt?: number;
  failed?: boolean;
  error?: string | null;
}

interface WorkioDB extends DBSchema {
  attendanceQueue: {
    key: string;
    value: AttendanceEvent;
    indexes: { by_nextRetryAt: number; by_failed: boolean };
  };
}

const DB_NAME = 'workio';
const STORE = 'attendanceQueue';
const MAX_ATTEMPTS = 5;
const BASE_DELAY_MS = 1000;
const JITTER = 0.3;
const FLUSH_INTERVAL_MS = 15000;

function backoff(attempt: number) {
  const exp = Math.min(attempt, 10);
  const jitter = 1 + JITTER * (Math.random() * 2 - 1);
  return Math.round(Math.pow(2, exp) * BASE_DELAY_MS * jitter);
}

let dbPromise: Promise<IDBPDatabase<WorkioDB>> | null = null;

function getDB() {
  if (!dbPromise) {
    dbPromise = openDB<WorkioDB>(DB_NAME, 1, {
      upgrade(db) {
        const store = db.createObjectStore(STORE, { keyPath: 'id' });
        store.createIndex('by_nextRetryAt', 'nextRetryAt');
        store.createIndex('by_failed', 'failed');
      },
    });
  }
  return dbPromise;
}

// Observability helpers
let pendingCountCache: number | null = null;
let lastErrorCache: string | null = null;

export async function getQueueStats() {
  const db = await getDB();
  const tx = db.transaction(STORE, 'readonly');
  const store = tx.objectStore(STORE);
  const all = await store.getAll();
  const pending = all.filter((e) => !e.failed && e.nextRetryAt && e.nextRetryAt <= Date.now()).length;
  const failed = all.filter((e) => e.failed).length;
  pendingCountCache = pending;
  return { total: all.length, pending, failed, lastError: lastErrorCache };
}

// Core submission
export async function enqueueAttendance(
  type: AttendanceEvent['type'],
  payload: Record<string, unknown>
): Promise<string> {
  const id = crypto.randomUUID();
  const event: AttendanceEvent = {
    id,
    type,
    payload,
    createdAt: Date.now(),
    attempts: 0,
    nextRetryAt: Date.now(),
    failed: false,
    error: null,
  };
  const db = await getDB();
  await db.add(STORE, event);
  void flushQueue();
  return id;
}

// Server submission with idempotency key
async function submitEvent(event: AttendanceEvent): Promise<void> {
  const res = await fetch('/api/attendance/events', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      idempotencyKey: event.id,
      type: event.type,
      ...event.payload,
    }),
  });

  if (!res.ok) {
    const text = await res.text().catch(() => '');
    const err: any = new Error(`HTTP ${res.status}: ${text}`);
    err.status = res.status;
    throw err;
  }
}

// Flush with concurrency control and observability
let flushLock = false;

export async function flushQueue(): Promise<void> {
  if (flushLock) return;
  flushLock = true;

  try {
    const db = await getDB();
    const tx = db.transaction(STORE, 'readwrite');
    const store = tx.objectStore(STORE);
    const idx = store.index('by_nextRetryAt');
    const now = Date.now();

    let cursor = await idx.openCursor(IDBKeyRange.upperBound(now));
    const toProcess: AttendanceEvent[] = [];
    while (cursor) {
      toProcess.push(cursor.value);
      cursor = await cursor.continue();
    }

    for (const event of toProcess) {
      if (event.attempts >= MAX_ATTEMPTS) {
        await store.update(event.id, { failed: true, error: event.error ?? 'Max attempts exceeded' });
        lastErrorCache = `Event ${event.id} failed permanently`;
        continue;
      }

      try {
        await submitEvent(event);
        await store.delete(event.id);
      } catch (err: any) {
        const nextDelay = backoff(event.attempts + 1);
        const msg = err?.message ?? String(err);
        await store.update(event.id, {
          attempts: event.attempts + 1,
          lastAttemptAt: Date.now(),
          nextRetryAt: Date.now() + nextDelay,
          error: msg,
        });
        lastErrorCache = `Event ${event.id} attempt ${event.attempts + 1} failed: ${msg}`;
      }
    }

    await tx.done;
  } finally {
    flushLock = false;
  }
}

// Cross-tab coordination
let bc: BroadcastChannel | null = null;
if (typeof BroadcastChannel !== 'undefined') {
  bc = new BroadcastChannel('workio_attendance_flush');
  bc.onmessage = (ev) => {
    if (ev.data === 'flush') void flushQueue();
  };
}

export function broadcastFlush() {
  bc?.postMessage('flush');
}

// Periodic + event-driven flush
if (typeof window !== 'undefined') {
  setInterval(() => void flushQueue(), FLUSH_INTERVAL_MS);
  window.addEventListener('online', () => void flushQueue());
  window.addEventListener('visibilitychange', () => {
    if (document.visibilityState === 'visible') void flushQueue();
  });
}
```
