# airship / design

## Incremental Improvement (≤2h)

**Goal**: Eliminate HF API 429 and Lightning quota waste during Surrogate training by implementing **CDN-first deterministic ingestion + Lightning Studio reuse/guard + Arkship UI idle-stop resilience**.

**Why this ships fastest**:
- Removes the primary blocker (HF API rate limits) by switching to CDN-only fetches during training.
- Recovers ~80h/mo Lightning quota by reusing running studios instead of recreating.
- Prevents lost training progress when Lightning idle-stops.
- All changes are additive/config-level (no breaking refactors) and can be implemented in <2h.

---

## Implementation Plan

### 1) CDN-first deterministic file list (Mac orchestration)
- Single API call from Mac (after rate-limit window) to list one date folder.
- Save list to JSON committed to repo or passed to Lightning job.
- Training script uses only CDN URLs (`https://huggingface.co/datasets/{repo}/resolve/main/{path}`) with no Authorization header.

### 2) Lightning Studio reuse + idle-stop guard
- Before `Studio(create_ok=True)`, enumerate `Teamspace.studios` and reuse a running studio with matching name.
- If studio exists but stopped, restart with `target.start(machine=Machine.L40S)` (or fallback to L40S if H200 unavailable).
- Wrap each `.run()` with status check; if stopped, restart then run.

### 3) Arkship UI idle-stop resilience (optional but high value)
- Ensure Arkship containers have restart policies and healthchecks so they survive idle-stop/reboot.
- Add lightweight readiness probe to avoid routing traffic to starting containers.

### 4) Surrogate training script changes
- Replace `load_dataset(streaming=True)` for heterogeneous repos with per-file CDN downloads via `hf_hub_download` or raw CDN GET.
- Project to `{prompt, response}` only at parse time; no mixed-schema reads.
- Use deterministic repo selection for commit-cap spreading (hash slug → sibling repo index).

### Lightning launcher with reuse + idle-stop guard
```python
# scripts/run_lightning_training.py
import os
import time
from lightning import LightningWork, LightningApp, Machine
from lightning.app import LightningFlow
from huggingface_hub import HfApi

class SurrogateTrainer(LightningWork):
    def __init__(self, studio_name: str = "surrogate-trainer", machine_type: str = "lightning-lambda-prod/L40S:1"):
        super().__init__()
        self.studio_name = studio_name
        self.machine_type = machine_type
        self._studio = None

    def _find_running_studio(self):
        from lightning.app import Teamspace
        for s in Teamspace.studios:
            if s.name == self.studio_name:
                return s
        return None

    def _ensure_studio(self):
        if self._studio is not None:
            return self._studio

        existing = self._find_running_studio()
        if existing is not None:
            self._studio = existing
            print(f"Reusing running studio: {self.studio_name}")
            return self._studio

        # Studio not running; create (or start) one
        from lightning.app import Studio
        # Try to reuse stopped studio object by name if Studio API allows; fallback to create
        self._studio = Studio(
            name=self.studio_name,
            machine=Machine(self.machine_type),
            create_ok=True,
            start_ok=True,
        )
        print(f"Created/started studio: {self.studio_name}")
        return self._studio

    def _wait_for_ready(self, timeout=300):
        for _ in range(timeout // 10):
            if self._studio and getattr(self._studio, "status", None) == "Running":
                return True
            time.sleep(10)
        return False

    def run(self):
        studio = self._ensure_studio()
        if not self._wait_for_ready():
            raise RuntimeError("Studio failed to become ready")

        # Guard: if studio stopped during run, restart and continue
        def guarded_run(script_path, args=None):
            if getattr(studio, "status", None) != "Running":
                print("Studio stopped; restarting...")
                studio.start(machine=Machine(self.machine_type))
                if not self._wait_for_ready():
                    raise RuntimeError("Studio restart failed")
            return studio.run(script_path, args or {})

        # Example: run training with CDN-only file list
        guarded_run(
            script_path="train.py",
            args={
                "file_list": "file_list_2026-05-03.json",
                "use_cdn_only": True,
                "output_repo": "my-org/surrogate-enriched",
            },
        )

class RootFlow(LightningFlow):
    def __init__(self):
        super().__init__()
        self.trainer = SurrogateTrainer()

    def run(self):
        self.trainer.run()

app = LightningApp(RootFlow())
```

### Surrogate training: CDN-only dataset loader (no HF API during training)
```python
# surrogate/train.py
import json
import os
import requests
import pyarrow as pa
import pyarrow.parquet as pq
from typing import List, Dict

def load_file_list(path: str) -> Dict:
    with open(path) as f:
        return json.load(f)

def cdn_fetch(repo: str, file_path: str, local_cache: str = "/tmp/cache") -> str:
    os.makedirs(local_cache, exist_ok=True)
    local_file = os.path.join(local_cache, os.path.basename(file_path))
    if os.path.exists(local_file):
        return local_file

    url = f"https://huggingface.co/datasets/{repo}/resolve/main/{file_path}"
    resp = requests.get(url, timeout=60)
    resp.raise_for_status()
    with open(local_file, "wb") as f:
        f.write(resp.content)
    return local_file

def parse_to_prompt_response(local_path: str) -> List[Dict]:
    # Implement per-file parsing that projects to {prompt, response}
    # Example for parquet:
    table = pq.read_table(local_path, columns=["prompt", "response"])
    return table.to_pylist()

def main():
    file_list = load_file_list(os.environ.get("FILE_LIST", "file_list_2026-05-03.json"))
    repo = file_list["repo"]
    files = file_list["files"]

    records = []
    for f in files:
        local = cdn_fetch(repo, f)
        try:
            rows = parse_to_prompt_response(local)
            records.extend(rows)
        except Exception as exc:
            print(f"Skipping {f}: {exc}")
            continue

    # Save enriched parquet with deterministic naming (no source/ts cols)
    if records:
        table = pa.Table.from_pylist(records, schema=pa.schema([
            pa.field("prompt", pa.string()),
            pa.field("response", pa.string()),
        ]))
        out_path = f"batches/mirror-merged/{file_list['folder'].split('/')[-1]}/merged.parquet"
        os.makedirs(os.path.dirname(out_path), exist_ok=True)
        pq.write_table(table, out_path)
        print(f"Wrote {len(records)} records to {out_path}")

if __name__ == "__main__":
    main()
```
