# workio / quality

## 1) Diagnosis

- **No idempotency key enforcement at DB layer**: clock-in/out endpoints accept duplicate requests for the same `(employee_id, punch_type, date)` when clients retry on flaky mobile networks → double punches and payroll drift.
- **Race condition on concurrent punches**: two near-simultaneous requests can both pass “last punch” checks and insert duplicate/out-of-order punches.
- **Missing immutable audit trail**: punch updates/overwrites lose provenance; compliance and payroll reconciliation cannot prove what actually happened.
- **Webhook replay vulnerability**: LINE may redeliver webhooks; without idempotency keys and upsert semantics, replays create duplicate punches.
- **Soft deletes/updates without history**: current schema allows punches to be silently updated or removed with no before/after record.

## 2) Proposed change

- **File**: `workio/server/src/db/schema.sql` — add unique constraint + audit table.
- **File**: `workio/server/src/controllers/punchController.ts` — add idempotency key and upsert logic with transaction.
- **Scope**: single atomic endpoint change (clock-in/out) + schema migration; <2h.

## 3) Implementation

### 3.1 Schema change (`workio/server/src/db/schema.sql`)

```sql
-- Add idempotency table for punch requests (webhook/client retries)
CREATE TABLE IF NOT EXISTS punch_idempotency (
  idempotency_key TEXT NOT NULL,
  employee_id    INTEGER NOT NULL,
  punch_type     TEXT NOT NULL CHECK (punch_type IN ('IN', 'OUT')),
  punch_date     DATE NOT NULL,
  punch_id       INTEGER NOT NULL,
  created_at     TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  PRIMARY KEY (idempotency_key)
);

-- Ensure one punch per employee per punch_type per calendar day (business rule)
CREATE UNIQUE INDEX IF NOT EXISTS uq_employee_punch_type_date
ON punches (employee_id, punch_type, DATE(timestamp AT TIME ZONE 'UTC'));

-- Immutable audit table for punches
CREATE TABLE IF NOT EXISTS punch_audit (
  audit_id      BIGSERIAL PRIMARY KEY,
  punch_id      INTEGER NOT NULL,
  employee_id   INTEGER NOT NULL,
  action        TEXT NOT NULL CHECK (action IN ('INSERT', 'UPDATE', 'DELETE')),
  old_data      JSONB,
  new_data      JSONB,
  changed_by    TEXT,
  changed_at    TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Audit trigger function
CREATE OR REPLACE FUNCTION fn_punch_audit()
RETURNS TRIGGER AS $$
BEGIN
  IF TG_OP = 'INSERT' THEN
    INSERT INTO punch_audit (punch_id, employee_id, action, new_data, changed_by)
    VALUES (NEW.id, NEW.employee_id, 'INSERT', to_jsonb(NEW), current_user);
    RETURN NEW;
  ELSIF TG_OP = 'UPDATE' THEN
    INSERT INTO punch_audit (punch_id, employee_id, action, old_data, new_data, changed_by)
    VALUES (NEW.id, NEW.employee_id, 'UPDATE', to_jsonb(OLD), to_jsonb(NEW), current_user);
    RETURN NEW;
  ELSIF TG_OP = 'DELETE' THEN
    INSERT INTO punch_audit (punch_id, employee_id, action, old_data, changed_by)
    VALUES (OLD.id, OLD.employee_id, 'DELETE', to_jsonb(OLD), current_user);
    RETURN OLD;
  END IF;
  RETURN NULL;
END;
$$ LANGUAGE plpgsql;

-- Attach trigger (idempotent)
DO $$
BEGIN
  IF NOT EXISTS (
    SELECT 1 FROM pg_trigger WHERE tgname = 'trg_punch_audit'
  ) THEN
    CREATE TRIGGER trg_punch_audit
      AFTER INSERT OR UPDATE OR DELETE ON punches
      FOR EACH ROW EXECUTE FUNCTION fn_punch_audit();
  END IF;
END $$;
```

### 3.2 Controller change (`workio/server/src/controllers/punchController.ts`)

```ts
import { Request, Response } from 'express';
import { pool } from '../db';

// POST /api/punches/clock-in  (or /clock-out)
// Body: { employeeId: number, punchType: 'IN'|'OUT', idempotencyKey: string, timestamp?: string, location?: { lat, lng } }
export const recordPunch = async (req: Request, res: Response) => {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    const { employeeId, punchType, idempotencyKey, timestamp = new Date().toISOString(), location } = req.body;

    if (!idempotencyKey) {
      return res.status(400).json({ error: 'idempotencyKey is required' });
    }

    // 1) Idempotency check: if key exists, return existing punch
    const idemRes = await client.query(
      `SELECT punch_id FROM punch_idempotency WHERE idempotency_key = $1`,
      [idempotencyKey]
    );

    if (idemRes.rows.length > 0) {
      const punch = await client.query(`SELECT * FROM punches WHERE id = $1`, [idemRes.rows[0].punch_id]);
      await client.query('COMMIT');
      return res.status(200).json({ punch: punch.rows[0], idempotent: true });
    }

    const punchDate = new Date(timestamp).toISOString().split('T')[0]; // YYYY-MM-DD UTC

    // 2) Upsert: try insert; on unique violation, do nothing and fetch existing
    // We use the unique index on (employee_id, punch_type, date(timestamp)) to enforce rule
    const insertRes = await client.query(
      `INSERT INTO punches (employee_id, punch_type, timestamp, location)
       VALUES ($1, $2, $3, $4)
       ON CONFLICT DO NOTHING
       RETURNING id`,
      [employeeId, punchType, timestamp, location || null]
    );

    let punchId: number;
    if (insertRes.rows.length > 0) {
      punchId = insertRes.rows[0].id;
    } else {
      // Fetch existing punch for this employee/punchType/date
      const existing = await client.query(
        `SELECT id FROM punches
         WHERE employee_id = $1 AND punch_type = $2 AND DATE(timestamp AT TIME ZONE 'UTC') = $3
         ORDER BY timestamp DESC LIMIT 1`,
        [employeeId, punchType, punchDate]
      );

      if (existing.rows.length === 0) {
        // Should not happen if unique index exists; defensive fallback insert with explicit ON CONFLICT target
        const fallback = await client.query(
          `INSERT INTO punches (employee_id, punch_type, timestamp, location)
           VALUES ($1, $2, $3, $4)
           ON CONFLICT (employee_id, punch_type, (DATE(timestamp AT TIME ZONE 'UTC'))) DO UPDATE
           SET timestamp = EXCLUDED.timestamp, location = EXCLUDED.location
           RETURNING id`,
          [employeeId, punchType, timestamp, location || null]
        );
        punchId = fallback.rows[0].id;
      } else {
        punchId = existing.rows[0].id;
      }
    }

    // 3) Record idempotency mapping
    await client.query(
      `INSERT INTO punch_idempotency (idempotency_key, employee_id, punch_type, punch_date, punch_id)
       VALUES ($1, $2, $3, $4, $5)`,
      [idempotencyKey, employeeId, punchType, punchDate, punchId]
    );

    await client.query('COMMIT');

    const punch = await client.query(`SELECT * FROM punches WHERE id = $1`, [punchId]);
    return res.status(201).json({ punch: punch.rows[0], idempotent: false });
  } catch (err) {
    await client.query('ROLLBACK');
    console.error('recordPunch error:', err);
    return res.status(500).json({ error: 'Internal server error' });
  } finally {
    client.release();
  }
};
```
