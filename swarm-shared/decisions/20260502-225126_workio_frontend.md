# workio / frontend

## Final synthesized solution

### Core diagnosis (accepted)
- No durable offline queue (localStorage is unsafe/limited).
- No idempotency → duplicate attendance records on retries.
- No exponential backoff + jitter → thundering herd and wasted retries.
- No cross-tab coordination → double-send or lost state.
- No online-triggered + periodic flush → stuck actions until manual refresh.

### Chosen approach
- Durable queue in IndexedDB (idb-keyval).
- Idempotency keys (crypto.randomUUID) stored with each item and sent as `Idempotency-Key` header.
- Exponential backoff with jitter and max attempts.
- Online-triggered + periodic flush coordinated across tabs via BroadcastChannel.
- Single source of truth for queue state; optimistic UI is allowed but queue is authoritative.

### Queue primitives
`src/lib/queue/IndexedDBQueue.ts`
```ts
import { get, set, del, entries, createStore } from 'idb-keyval';

const store = createStore('workio-attendance', 'queue');

export interface QueueItem<T = any> {
  id: string;
  action: 'clock-in' | 'clock-out' | 'leave-request' | 'ot-request';
  payload: T;
  idempotencyKey: string;
  attempts: number;
  maxAttempts: number;
  nextRetryAt: number;
  createdAt: number;
  updatedAt: number;
}

const RETRY_BASE_MS = 2_000;
const MAX_ATTEMPTS = 5;

function jitter(base: number) {
  return base * (0.5 + Math.random() * 0.5); // 0.5x–1.0x
}

function backoff(attempt: number) {
  return jitter(RETRY_BASE_MS * Math.pow(2, Math.min(attempt, 6)));
}

export async function enqueue<T>(item: Omit<QueueItem<T>, 'id' | 'attempts' | 'nextRetryAt' | 'createdAt' | 'updatedAt'>) {
  const now = Date.now();
  const queueItem: QueueItem<T> = {
    ...item,
    id: crypto.randomUUID(),
    attempts: 0,
    maxAttempts: MAX_ATTEMPTS,
    nextRetryAt: now,
    createdAt: now,
    updatedAt: now,
  };
  const all = await getAll();
  const exists = all.some((e) => e.idempotencyKey === item.idempotencyKey);
  if (exists) return queueItem;
  await set(queueItem.id, queueItem, store);
  return queueItem;
}

export async function dequeue(id: string) {
  await del(id, store);
}

export async function peekReady(): Promise<QueueItem[]> {
  const all = await getAll();
  const now = Date.now();
  return all.filter((item) => item.nextRetryAt <= now && item.attempts < item.maxAttempts)
            .sort((a, b) => a.nextRetryAt - b.nextRetryAt);
}

export async function markAttempt(id: string, success: boolean) {
  const item = await get<QueueItem>(id, store);
  if (!item) return;
  if (success) {
    await del(id, store);
    return;
  }
  item.attempts += 1;
  item.updatedAt = Date.now();
  if (item.attempts >= item.maxAttempts) {
    await del(id, store);
    return;
  }
  item.nextRetryAt = Date.now() + backoff(item.attempts);
  await set(id, item, store);
}

export async function getAll(): Promise<QueueItem[]> {
  return (await entries<string, QueueItem>(store))
    .map(([, v]) => v)
    .filter(Boolean);
}
```

### Hook to manage flush
`src/hooks/useAttendanceQueue.ts`
```ts
import { useEffect, useRef } from 'react';
import * as Queue from '../lib/queue/IndexedDBQueue';
import { sendAttendance } from '../api/attendance';

const FLUSH_INTERVAL_MS = 15_000;
const BROADCAST_CHANNEL = 'workio-queue-flush';

export function useAttendanceQueue() {
  const processing = useRef<Set<string>>(new Set());

  const flush = async () => {
    const ready = await Queue.peekReady();
    for (const item of ready) {
      if (processing.current.has(item.id)) continue;
      processing.current.add(item.id);
      try {
        await sendAttendance(item);
        await Queue.markAttempt(item.id, true);
      } catch {
        await Queue.markAttempt(item.id, false);
      } finally {
        processing.current.delete(item.id);
      }
    }
  };

  useEffect(() => {
    const channel = new BroadcastChannel(BROADCAST_CHANNEL);
    channel.onmessage = () => flush();

    const onOnline = () => {
      flush();
      channel.postMessage('flush');
    };

    window.addEventListener('online', onOnline);
    const interval = setInterval(() => {
      flush();
      channel.postMessage('flush');
    }, FLUSH_INTERVAL_MS);

    // initial flush
    flush();

    return () => {
      window.removeEventListener('online', onOnline);
      clearInterval(interval);
      channel.close();
    };
  }, []);

  return { enqueue: Queue.enqueue };
}
```

### API adapter
`src/api/attendance.ts`
```ts
import type { QueueItem } from '../lib/queue/IndexedDBQueue';

export async function sendAttendance(item: QueueItem) {
  const res = await fetch('/api/attendance', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Idempotency-Key': item.idempotencyKey,
    },
    body: JSON.stringify({ action: item.action, payload: item.payload }),
  });

  if (!res.ok) {
    const text = await res.text().catch(() => '');
    throw new Error(`Attendance failed: ${res.status} ${text}`);
  }
  return res.json();
}
```

### UI integration (clock button)
`src/components/ClockButton.tsx`
```tsx
import { useAttendanceQueue } from '../hooks/useAttendanceQueue';

export function ClockButton() {
  const { enqueue } = useAttendanceQueue();

  const handleClockIn = async () => {
    await enqueue({
      action: 'clock-in',
      payload: { location: getCurrentLocation(), ts: Date.now() },
      idempotencyKey: crypto.randomUUID(),
    });
  };

  return <button onClick={handleClockIn}>Clock In</button>;
}
```

## Verification checklist
- **Offline**: Disable network, click Clock In multiple times; confirm single queue entry per idempotency key in DevTools → Application → IndexedDB. Re-enable network and confirm items flush and disappear; backend shows one record per idempotency key.
- **Retry/backoff**: Mock server returning 500; verify attempts increment and `nextRetryAt` increases with backoff; after `MAX_ATTEMPTS`, item is removed.
- **Cross-tab**: Open two tabs; enqueue in one, verify BroadcastChannel triggers flush in the other and item is processed once.
- **Periodic flush**: With network available and pending items, wait ~15s and confirm periodic flush sends them.
- **Idempotency**: Resend same idempotency key (simulate retry) and confirm backend deduplicates.
