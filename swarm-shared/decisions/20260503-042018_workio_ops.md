# workio / ops

Below is the **single, consolidated implementation** that keeps every correct insight from both candidates, removes duplication, resolves contradictions in favor of **correctness + concrete actionability**, and adds production-grade safeguards (strict signature check, idempotency, atomic daily-state guard, and tamper-evident audit).

---

## 1. Diagnosis (merged)
- **Missing `X-Line-Signature` verification** → spoofed clock-in/out/leave/OT possible.
- **No idempotency** on webhook ingestion → LINE retries or client double-taps create duplicate attendance rows.
- **No per-employee per-day intent guard** → races can create invalid daily state (e.g., two clock-ins, missing clock-out).
- **Non-atomic upsert** for daily state → concurrent webhooks corrupt daily summaries.
- **No audit trail** (request-id, signature result, before/after) → cannot reconstruct who/when/why for compliance/debugging.

import crypto from 'crypto';

const router = Router();
const pool = new Pool({ connectionString: process.env.DATABASE_URL });

const LINE_CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';
if (!LINE_CHANNEL_SECRET) {
  throw new Error('LINE_CHANNEL_SECRET is required for webhook security');

// Strict HMAC-SHA256 signature verification
function verifyLineSignature(req: Request, _res: Response, next: NextFunction) {
  const signature = req.get('X-Line-Signature') || '';
  // Use raw body for signature; ensure body-parser does NOT mutate encoding
  const body = typeof req.body === 'string' ? req.body : JSON.stringify(req.body);
  const expected = crypto
    .createHmac('sha256', LINE_CHANNEL_SECRET)
    .update(body, 'utf8')
    .digest('base64');

  if (!signature || !crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected))) {
    return void next(new Error('Invalid signature'));
  }
  next();
}

// Deterministic idempotency key from LINE event (preferred) or deterministic fallback
function buildIdempotencyKey(event: any): string {
  if (event.webhookEventId) return `line:${event.webhookEventId}`;
  const userId = event.source?.userId || 'unknown';
  const ts = event.timestamp || Date.now();
  const intent = eventTypeToIntent(event);
  return `line:fallback:${userId}:${event.eventDate || new Date(ts).toISOString().slice(0, 10)}:${intent}:${ts}`;
}

function eventTypeToIntent(event: any): 'clock_in' | 'clock_out' | 'leave' | 'other' {
  const text = (event.message?.text || event.postback?.data || '').toLowerCase();
  if (text.includes('ออก') || text.includes('clock_out') || text.includes('leave')) return 'clock_out';
  if (text.includes('clock_in')) return 'clock_in';
  if (text.includes('leave')) return 'leave';
  return 'other';
}

// Atomic upsert of daily attendance with row-level locking to prevent races
async function upsertDailyAttendance(
  client: PoolClient,
  tenantId: string,
  userId: string,
  eventDate: string,
  intent: 'clock_in' | 'clock_out' | 'leave' | 'other',
  occurredAt: Date
) {
  // Lock row for update to serialize concurrent webhooks for same user/day
  await client.query(
    `SELECT id FROM daily_attendance
     WHERE tenant_id = $1 AND user_id = $2 AND event_date = $3
     FOR UPDATE`,
    [tenantId, userId, eventDate]
  );

  if (intent === 'clock_in') {
    await client.query(
      `INSERT INTO daily_attendance (tenant_id, user_id, event_date, first_clock_in, last_clock_in, updated_at)
       VALUES ($1, $2, $3, $4, $4, NOW())
       ON CONFLICT (tenant_id, user_id, event_date)
       DO UPDATE SET
         first_clock_in = LEAST(COALESCE(EXCLUDED.first_clock_in, daily_attendance.first_clock_in), EXCLUDED.last_clock_in),
         last_clock_in = GREATEST(COALESCE(EXCLUDED.last_clock_in, daily_attendance.last_clock_in), EXCLUDED.last_clock_in),
         updated_at = NOW()`,
      [tenantId, userId, eventDate, occurredAt]
    );
  } else if (intent === 'clock_out') {
    await client.query(
      `INSERT INTO daily_attendance (tenant_id, user_id, event_date, first_clock_out, last_clock_out, updated_at)
       VALUES ($1, $2, $3, $4, $4, NOW())
       ON CONFLICT (tenant_id, user_id, event_date)
       DO UPDATE SET
         first_clock_out = LEAST(COALESCE(EXCLUDED.first_clock_out, daily_attendance.first_clock_out), EXCLUDED.last_clock_out),
         last_clock_out = GREATEST(COALESCE(EXCLUDED.last_clock_out, daily_attendance.last_clock_out), EXCLUDED.last_clock_out),
         updated_at = NOW()`,
      [tenantId, userId, eventDate, occurredAt]
    );
