### 1. One-time Mac orchestration script (≤10 min)

`scripts/list_hf_files.py`

```python
#!/usr/bin/env python3
"""
List parquet files for a date folder in HF dataset repo and emit train_files.json
with CDN URLs and local cache paths.

Usage:
    python scripts/list_hf_files.py \
        --repo "axentx/surrogate-mirror" \
        --date-folder "2026-05-03" \
        --out "train_files.json" \
        --cache-dir ".cache/hf"
"""

import argparse
import json
import os
import sys
import time
from pathlib import Path

from huggingface_hub import HfApi, RepositoryNotFoundError, HFValidationError

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def list_parquet_files(repo: str, date_folder: str):
    api = HfApi()
    prefix = f"batches/mirror-merged/{date_folder.rstrip('/')}/"
    try:
        items = api.list_repo_tree(repo=repo, path=prefix, recursive=False)
    except Exception as exc:
        if getattr(exc, "status_code", None) == 429:
            retry_after = getattr(exc, "headers", {}).get("Retry-After")
            wait = int(retry_after) if retry_after and retry_after.isdigit() else 360
            print(f"HF API 429 rate-limited. Wait {wait}s before retry.", file=sys.stderr)
            sys.exit(1)
        raise

    files = []
    for item in items:
        path = item.get("path") if isinstance(item, dict) else getattr(item, "path", None)
        if path and path.lower().endswith(".parquet"):
            files.append(path)
    return sorted(files)

def main():
    parser = argparse.ArgumentParser(description="List HF parquet files for training")
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g., axentx/surrogate-mirror)")
    parser.add_argument("--date-folder", required=True, help="Date folder under batches/mirror-merged/")
    parser.add_argument("--out", default="train_files.json", help="Output JSON path")
    parser.add_argument("--cache-dir", default=".cache/hf", help="Local cache directory")
    args = parser.parse_args()

    print(f"Listing parquet files in {args.repo} at batches/mirror-merged/{args.date_folder}/ ...")
    files = list_parquet_files(args.repo, args.date_folder)

    if not files:
        print("No parquet files found.", file=sys.stderr)
        sys.exit(1)

    cache_dir = Path(args.cache_dir)
    cache_dir.mkdir(parents=True, exist_ok=True)

    entries = []
    for f in files:
        local_path = cache_dir / Path(f).name
        entries.append({
            "filename": Path(f).name,
            "cdn_url": CDN_TEMPLATE.format(repo=args.repo, path=f),
            "hf_path": f,
            "local_path": str(local_path),
        })

    out_path = Path(args.out)
    out_path.write_text(json.dumps(entries, indent=2))
    print(f"Wrote {len(entries)} files to {out_path}")

if __name__ == "__main__":
    main()
```

### 2. Training script changes (surrogate/train.py)

Key behaviors:
- Accept `--train-files-json` (default: `train_files.json`).
- Prefer local files; fall back to CDN URLs.
- Project to `{prompt, response}` only at parse time; ignore extra columns.
- Avoid recursive HF API calls and `streaming=True` for mixed-schema repos.

```python
import json
import argparse
from pathlib import Path
from datasets import load_dataset, Features, Value, IterableDataset
from torch.utils.data import DataLoader

def load_train_files(train_files_json: str):
    with open(train_files_json) as f:
        return json.load(f)

def build_dataset(train_files_json, streaming=False, local_only=False):
    entries = load_train_files(train_files_json)

    # Prefer local files if present
    local_paths = [e["local_path"] for e in entries if Path(e["local_path"]).exists()]
    if local_only or local_paths:
        data_files = local_paths if local_paths else []
    else:
        # Fallback to CDN URLs (datasets supports http/https in data_files)
        data_files = [e["cdn_url"] for e in entries]

    if not data_files:
        raise FileNotFoundError("No local files or CDN URLs available for training.")

    features = Features({
        "prompt": Value("string"),
        "response": Value("string"),
    })

    ds = load_dataset(
        "parquet",
        data_files=data_files,
        features=features,
        streaming=streaming,
    )
    return ds["train"] if not streaming else ds

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--train-files-json", default="train_files.json")
    parser.add_argument("--output-dir", default="checkpoints")
    parser.add_argument("--local-only", action="store_true", help="Require local files")
    args = parser.parse_args()

    train_ds = build_dataset(args.train_files_json, streaming=False, local_only=args.local_only)

    # Minimal DataLoader example
    loader = DataLoader(train_ds, batch_size=8, shuffle=True)
    print(f"Loaded dataset; first batch example:")
    for batch in loader:
        print({k: v[:2] for k, v in batch.items()})
        break

if __name__ == "__main__":
```
