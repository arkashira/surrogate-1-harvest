# workio / design

## Final consolidated solution

**Core diagnosis (merged)**
- Missing **database-level enforcement** for at most one open punch per user per calendar day (`clock_out_at IS NULL`), causing duplicates under LINE retries or client double-taps.
- App-layer “find-then-insert/update” is race-prone; no atomic upsert or proper locking.
- No idempotency guard on LINE webhook handler → duplicate processing on LINE retries/timeouts.
- Clock-in/out boundary invariants are not enforced at the DB level (e.g., `clock_out_at >= clock_in_at`, no negative durations).
- Minimal audit context (no LINE delivery id, no trace id) makes duplicate detection and replay debugging hard.
- Missing index on the time column will hurt time-range reports as data grows.

**Chosen approach (correctness + actionability)**
- Enforce invariants **at the database level** where possible (partial unique index, check constraint).
- Use **atomic upserts and locking** in a single transaction to eliminate races.
- Implement **idempotency** with a dedicated table keyed by LINE `deliveryId` (or message id) so retries are safe and fast.
- Store **audit metadata** (`line_delivery_id`, `request_trace_id`, `created_at`, `updated_at`) on punches for traceability.
- Add a **covering index** on `user_id, clock_in_at` for fast time-range queries and report scans.
- Keep application logic simple and defensive; treat unique violations as idempotent successes.

### 1) Database schema changes  
File: `/opt/axentx/workio/server/src/db/schema.sql`

```sql
-- Audit/trace columns on punches
ALTER TABLE punches
  ADD COLUMN IF NOT EXISTS line_delivery_id VARCHAR(255),
  ADD COLUMN IF NOT EXISTS request_trace_id VARCHAR(255),
  ADD COLUMN IF NOT EXISTS created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  ADD COLUMN IF NOT EXISTS updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW();

-- Partial unique index: at most one open punch per user per calendar day
CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_one_open_per_day
  ON punches (user_id, DATE(clock_in_at))
  WHERE clock_out_at IS NULL;

-- Enforce non-negative punch durations
ALTER TABLE punches
  ADD CONSTRAINT chk_clock_out_after_in
  CHECK (clock_out_at IS NULL OR clock_out_at >= clock_in_at);

-- Idempotency table for LINE webhook events (primary enforcement for duplicates)
CREATE TABLE IF NOT EXISTS line_webhook_idempotency (
  idempotency_key TEXT PRIMARY KEY,
  event_type      TEXT NOT NULL,
  punch_id        BIGINT,             -- optional FK to punches for traceability
  processed_at    TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  payload_hash    TEXT
);

-- Optional TTL helper (run periodically)
-- DELETE FROM line_webhook_idempotency WHERE processed_at < NOW() - INTERVAL '7 days';

-- Indexes for fast lookups and time-range reports
CREATE INDEX IF NOT EXISTS idx_punches_user_clock_in
  ON punches (user_id, clock_in_at);

CREATE INDEX IF NOT EXISTS idx_punches_line_delivery_id
  ON punches (line_delivery_id)
  WHERE line_delivery_id IS NOT NULL;
```

### 2) Idempotency middleware for LINE webhooks  
File: `/opt/axentx/workio/server/src/middleware/lineWebhook.ts`

```ts
import { pool } from '../db';

export async function withLineIdempotency(
  deliveryId: string,
  eventType: string,
  handler: () => Promise<{ punchId?: number }>
) {
  // Fast-path: reject obvious duplicates before heavier work
  const exists = await pool.query(
    `SELECT 1 FROM line_webhook_idempotency WHERE idempotency_key = $1`,
    [deliveryId]
  );
  if (exists.rows.length > 0) return { duplicate: true };

  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // Lock the idempotency row to serialize concurrent retries
    await client.query(
      `INSERT INTO line_webhook_idempotency (idempotency_key, event_type)
       VALUES ($1, $2)
       ON CONFLICT (idempotency_key) DO NOTHING`,
      [deliveryId, eventType]
    );

    const inserted = (await client.query(
      `SELECT 1 FROM line_webhook_idempotency WHERE idempotency_key = $1`,
      [deliveryId]
    )).rows.length > 0;

    // If another worker inserted first, treat as duplicate
    if (!inserted) {
      await client.query('ROLLBACK');
      return { duplicate: true };
    }

    const result = await handler();

    // Optionally link punch for traceability
    if (result?.punchId) {
      await client.query(
        `UPDATE line_webhook_idempotency
         SET punch_id = $1 WHERE idempotency_key = $2`,
        [result.punchId, deliveryId]
      );
    }

    await client.query('COMMIT');
    return { duplicate: false, ...result };
  } catch (err: any) {
    await client.query('ROLLBACK');
    // Unique violation means duplicate processed concurrently -> treat as success
    if (err.code === '23505') return { duplicate: true };
    throw err;
  } finally {
    client.release();
  }
}
```

### 3) Atomic clock-in/clock-out handler  
File: `/opt/axentx/workio/server/src/routes/line.ts`

```ts
import { Router } from 'express';
import { pool } from '../db';
import crypto from 'crypto';
import { withLineIdempotency } from '../middleware/lineWebhook';

const router = Router();

router.post('/webhook/line', async (req, res) => {
  const traceId = crypto.randomUUID();
  const events = req.body.events || [];

  try {
    for (const ev of events) {
      if (ev.type !== 'message' || ev.message?.type !== 'text') continue;

      const userId = ev.source.userId; // map to local user_id via your tenant logic
      const text = ev.message.text.trim().toLowerCase();
      const deliveryId = ev.deliveryId || ev.message.id || traceId;

      const isClockIn = ['in', 'เข้า', 'clock in'].includes(text);
      const isClockOut = ['out', 'ออก', 'clock out'].includes(text);
      if (!isClockIn && !isClockOut) continue;

      await withLineIdempotency(deliveryId, isClockIn ? 'clock_in' : 'clock_out', async () => {
        const client = await pool.connect();
        try {
          await client.query('BEGIN');

          if (isClockIn) {
            // Atomic insert; partial index prevents second open punch same day
            await client.query(
              `INSERT INTO punches (user_id, clock_in_at, line_delivery_id, request_trace_id)
               VALUES ($1, NOW(), $2, $3)`,
              [userId, deliveryId, traceId]
            );
          } else {
            // Atomically close the latest open punch for this user
            const result = await client.query(
              `UPDATE punches
               SET clock_out_at = NOW(),
                   updated_at = NOW()
               WHERE id = (
                 SELECT id FROM punches
                 WHERE user_id = $1 AND clock_out_at IS NULL
                 ORDER BY clock_in_at DESC
                 LIMIT 1
                 FOR UPDATE SKIP LOCKED
               )
               RETURNING id`,
              [userId]
            );

            // If no open punch exists, ignore (do not create orphaned out)
            // Optionally log for observability.
          }

          await client.query('COMMIT');
        } catch (err: any) {
          await client.query('ROLLBACK');
          // Unique violation from partial index -> duplicate open punch; treat as idempotent
          if (err.code === '23505') return;
          throw err;
        } finally {
          client.release();
        }
      });
    }

    res.status(200).json({ message: 'ok' });
  } catch (err) {
    console.error('LINE webhook error', { traceId, err });
    res.status(500).json({ error: 'internal' });
  }
});

export default router;
```
