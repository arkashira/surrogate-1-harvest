# workio / design

## 1. Diagnosis
- **No idempotency guard on LINE webhook ingestion** — LINE retries on 5xx/timeouts; duplicate clock-in/out or leave/OT records are created on replay.
- **Missing `X-Line-Signature` verification** — webhook accepts spoofed events; enables attendance fraud and phantom records.
- **No request deduplication key** — same event delivered multiple times creates multiple attendance rows instead of upsert.
- **Clock-in/out state logic is vulnerable to race conditions** — concurrent webhook deliveries can produce invalid state transitions (e.g., double clock-in without intervening clock-out).
- **No audit trail for webhook processing** — failures are opaque and retries are blind; operators cannot distinguish fraud from transient LINE retry storms.

## 2. Proposed change
File: `/opt/axentx/workio/server/src/routes/webhook/line.ts` (or equivalent route handling `POST /webhook/line`).  
Scope: add signature verification, idempotency (dedupe by `X-Line-Signature` + event `webhookEventId`), and atomic state transition checks within a single DB transaction.

## 3. Implementation

```ts
// server/src/routes/webhook/line.ts
import express from 'express';
import crypto from 'crypto';
import { db } from '../../db/index.js';
import { attendance, attendanceAudit } from '../../db/schema.js';
import { eq, and, desc } from 'drizzle-orm';
import type { LINEEvent, LineWebhookRequest } from '../../types/line.js';

const router = express.Router();
const LINE_CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';

function verifySignature(rawBody: Buffer, signature: string): boolean {
  const expected = crypto
    .createHmac('sha256', LINE_CHANNEL_SECRET)
    .update(rawBody)
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

function isDuplicateEvent(eventId: string, deliveredAt: number): boolean {
  // Simple in-memory guard for same process; persistent guard handled in DB unique constraint.
  // Use a small LRU if needed; here we rely on DB uniqueness.
  return false;
}

async function getLatestAttendanceForUser(userId: string, tenantId: string) {
  const rows = await db
    .select()
    .from(attendance)
    .where(and(eq(attendance.userId, userId), eq(attendance.tenantId, tenantId)))
    .orderBy(desc(attendance.timestamp))
    .limit(1);
  return rows[0] || null;
}

router.post('/line', async (req, res) => {
  const signature = req.headers['x-line-signature'] as string;
  const rawBody = Buffer.from(JSON.stringify(req.body));

  // 1) Verify signature
  if (!signature || !verifySignature(rawBody, signature)) {
    await db.insert(attendanceAudit).values({
      tenantId: 'system',
      userId: 'system',
      eventType: 'webhook_invalid_signature',
      details: { receivedSignature: signature, body: req.body },
      status: 'rejected',
    });
    return res.status(401).json({ error: 'Invalid signature' });
  }

  const payload = req.body as LineWebhookRequest;
  const events: LINEEvent[] = payload.events || [];

  // 2) Process events transactionally
  const result = await db.transaction(async (tx) => {
    const results: Array<{ eventId: string; status: 'ok' | 'duplicate' | 'rejected' }> = [];

    for (const ev of events) {
      const eventId = ev.webhookEventId;
      const sourceId = ev.source?.userId || ev.source?.groupId || ev.source?.roomId;
      const timestamp = ev.timestamp || Date.now();

      // Idempotency: unique constraint on (eventId, tenantId) in attendanceAudit prevents replays.
      // Try insert audit row first; if duplicate, skip processing.
      try {
        await tx.insert(attendanceAudit).values({
          tenantId: 'default', // derive from source or payload; placeholder
          userId: sourceId || 'unknown',
          eventType: 'webhook_event',
          details: { event: ev.type, source: ev.source, mode: ev.mode },
          externalEventId: eventId,
          status: 'processing',
        });
      } catch (e: any) {
        // Unique violation -> duplicate event
        results.push({ eventId, status: 'duplicate' });
        continue;
      }

      // 3) Handle clock in/out via message (example: user texts "in" or "out")
      if (ev.type === 'message' && ev.message.type === 'text') {
        const text = ev.message.text.trim().toLowerCase();
        const userId = ev.source?.userId;
        const tenantId = 'default'; // derive from mapping

        if (!userId) {
          results.push({ eventId, status: 'rejected' });
          continue;
        }

        const latest = await getLatestAttendanceForUser(userId, tenantId);
        const now = new Date(timestamp);

        if (text === 'in') {
          // Prevent double clock-in without clock-out
          if (latest && !latest.clockOut) {
            await tx.update(attendanceAudit)
              .set({ status: 'rejected', details: { reason: 'already_clocked_in', latestId: latest.id } })
              .where(eq(attendanceAudit.externalEventId, eventId));
            results.push({ eventId, status: 'rejected' });
            continue;
          }

          const [record] = await tx.insert(attendance).values({
            userId,
            tenantId,
            clockIn: now,
            clockOut: null,
            location: null, // populate from location event if available
            status: 'active',
          }).returning({ id: attendance.id });

          await tx.update(attendanceAudit)
            .set({ status: 'ok', relatedId: record.id })
            .where(eq(attendanceAudit.externalEventId, eventId));

          results.push({ eventId, status: 'ok' });
        } else if (text === 'out') {
          if (!latest || latest.clockOut) {
            await tx.update(attendanceAudit)
              .set({ status: 'rejected', details: { reason: 'no_active_clock_in' } })
              .where(eq(attendanceAudit.externalEventId, eventId));
            results.push({ eventId, status: 'rejected' });
            continue;
          }

          await tx.update(attendance)
            .set({ clockOut: now, status: 'completed' })
            .where(eq(attendance.id, latest.id));

          await tx.update(attendanceAudit)
            .set({ status: 'ok', relatedId: latest.id })
            .where(eq(attendanceAudit.externalEventId, eventId));

          results.push({ eventId, status: 'ok' });
        } else {
          await tx.update(attendanceAudit)
            .set({ status: 'rejected', details: { reason: 'unknown_command' } })
            .where(eq(attendanceAudit.externalEventId, eventId));
          results.push({ eventId, status: 'rejected' });
        }
      } else {
        // Non-message events: mark processed but no attendance action
        await tx.update(attendanceAudit)
          .set({ status: 'ok' })
          .where(eq(attendanceAudit.externalEventId, eventId));
        results.push({ eventId, status: 'ok' });
      }
    }

    return results;
  });

  // LINE expects 200 even if some events were duplicates/rejected (idempotency)
  return res.status(200).json({});
});

export default router;
```
