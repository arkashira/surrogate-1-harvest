## Final Synthesized Solution

### Diagnosis (merged)
- **Race conditions**: rapid toggles, double-taps, retries, and network replays create duplicate attendance records.
- **Missing idempotency**: no per-tenant/per-day/per-employee intent guard at the write path; no client-side in-flight deduplication.
- **No optimistic UI**: users spam buttons under latency, increasing races.
- **Temporal ambiguity**: “same-day” edits use ambiguous windows, risking split-shift misclassification and broken OT/leave logic.
- **Missing deterministic latest-valid-clock rule**: back-dated or out-of-sequence inserts can corrupt state under concurrency.
- **Schema weak**: no unique constraint to enforce one IN and one OUT per day per employee.

---

### Proposed Change (merged)
- Add tenant-aware, idempotent `clockInOut` service with:
  - client-provided requestId fast-path,
  - calendar-day normalization (configurable timezone),
  - deterministic latest-valid-clock guard,
  - unique intent constraint `(tenant_id, employee_id, clock_date, clock_type)` to enforce single intent per day.
- Keep LINE webhook fast and non-blocking; commit before response.
- Add optimistic UI + client-side in-flight deduplication and request-id generation.
- Extend schema with unique partial index for intent integrity.

---

### Implementation

#### 1) DB schema (add constraint)

```sql
-- server/src/db/schema.sql
-- One IN and one OUT per employee per calendar day (tenant-scoped)
CREATE UNIQUE INDEX IF NOT EXISTS idx_attendance_intent
ON attendance_records (tenant_id, employee_id, clock_date, clock_type)
WHERE deleted_at IS NULL;
```

#### 2) Idempotent service method
```ts
// server/src/services/attendanceService.ts
import { pool } from '../db';

export type ClockIntent = {
  tenantId: string;
  employeeId: string;
  clockType: 'IN' | 'OUT';
  timestamp: Date;           // event time (for location/tolerance)
  location?: { lat: number; lng: number };
  lineUserId: string;
  requestId?: string;        // client-generated idempotency key
};

export async function clockInOut(intent: ClockIntent) {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // Calendar-day normalization (UTC midnight; replace with tenant TZ later)
    const clockDate = new Date(intent.timestamp);
    clockDate.setUTCHours(0, 0, 0, 0);

    // 1) Fast-path: client-provided idempotency key
    if (intent.requestId) {
      const existing = await client.query(
        `SELECT id, clock_type, recorded_at FROM attendance_records
         WHERE tenant_id = $1 AND employee_id = $2 AND request_id = $3 LIMIT 1`,
        [intent.tenantId, intent.employeeId, intent.requestId]
      );
      if (existing.rows.length) {
        await client.query('ROLLBACK');
        return { record: existing.rows[0], alreadyProcessed: true };
      }
    }

    // 2) Deterministic latest-valid-clock guard
    const latest = await client.query(
      `SELECT id, clock_type, recorded_at
       FROM attendance_records
       WHERE tenant_id = $1 AND employee_id = $2 AND clock_date = $3 AND deleted_at IS NULL
       ORDER BY recorded_at DESC LIMIT 1`,
      [intent.tenantId, intent.employeeId, clockDate]
    );

    if (latest.rows.length) {
      const latestType = latest.rows[0].clock_type;
      const latestTime = new Date(latest.rows[0].recorded_at);
      // Reject older duplicate of same type to preserve sequence integrity
      if (latestType === intent.clockType && latestTime > intent.timestamp) {
        await client.query('ROLLBACK');
        return { record: null, ignored: 'older_duplicate' };
      }
    }

    // 3) Insert with intent columns
    const result = await client.query(
      `INSERT INTO attendance_records
        (tenant_id, employee_id, clock_date, clock_type, recorded_at, location, line_user_id, request_id, created_at, updated_at)
       VALUES ($1,$2,$3,$4,$5,$6,$7,$8,NOW(),NOW())
       RETURNING id, clock_type, recorded_at`,
      [
        intent.tenantId,
        intent.employeeId,
        clockDate,
        intent.clockType,
        intent.timestamp,
        intent.location ? JSON.stringify(intent.location) : null,
        intent.lineUserId,
        intent.requestId || null,
      ]
    );

    await client.query('COMMIT');
    return { record: result.rows[0], alreadyProcessed: false };
  } catch (err: any) {
    await client.query('ROLLBACK');
    // Unique violation: intent already exists (race or retry)
    if (err.code === '23505') {
      const existing = await pool.query(
        `SELECT id, clock_type, recorded_at FROM attendance_records
         WHERE tenant_id = $1 AND employee_id = $2 AND clock_date = $3 AND clock_type = $4 LIMIT 1`,
        [intent.tenantId, intent.employeeId, clockDate, intent.clockType]
      );
      return { record: existing.rows[0] || null, alreadyProcessed: true };
    }
    throw err;
  } finally {
    client.release();
  }
}
```

#### 3) LINE webhook handler (fast)
```ts
// server/src/routes/lineWebhook.ts
import { Router } from 'express';
import { clockInOut } from '../services/attendanceService';

const router = Router();

router.post('/line', async (req, res) => {
  try {
    const { tenantId, employeeId, lineUserId, type, timestamp, location, requestId } = req.body;
    if (!tenantId || !employeeId || !lineUserId || !type) {
      return res.status(400).json({ error: 'invalid_payload' });
    }

    const result = await clockInOut({
      tenantId,
      employeeId,
      clockType: type.toUpperCase() === 'OUT' ? 'OUT' : 'IN',
      timestamp: new Date(timestamp || Date.now()),
      location,
      lineUserId,
      requestId,
    });

    return res.json({ ok: true, record: result.record, alreadyProcessed: result.alreadyProcessed });
  } catch (err) {
    console.error('LINE webhook error', err);
    return res.status(500).json({ error: 'internal' });
  }
});

export default router;
```
