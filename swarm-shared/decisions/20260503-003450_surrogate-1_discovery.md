## 1) snapshot_manifest.py

```python
#!/usr/bin/env python3
"""
Mac-side snapshot: list one date-partition of axentx/surrogate-1-training-pairs
and emit file_manifest.json for CDN-only training.

Usage:
    HF_TOKEN=<token> python snapshot_manifest.py --date 2026-04-29 --out file_manifest.json
"""

import argparse
import json
import os
import sys
from datetime import datetime

from huggingface_hub import HfApi, login

REPO_ID = "axentx/surrogate-1-training-pairs"


def main() -> None:
    parser = argparse.ArgumentParser(description="Create CDN file manifest for a date partition.")
    parser.add_argument("--date", required=True, help="Date folder, e.g. 2026-04-29")
    parser.add_argument("--out", default="file_manifest.json", help="Output JSON path")
    parser.add_argument("--repo", default=REPO_ID, help="HF dataset repo")
    args = parser.parse_args()

    token = os.environ.get("HF_TOKEN")
    if not token:
        print("ERROR: HF_TOKEN environment variable is required.", file=sys.stderr)
        sys.exit(1)

    login(token=token)
    api = HfApi()

    # List only the target date folder (non-recursive).
    folder_path = f"batches/public-merged/{args.date}"
    try:
        items = api.list_repo_tree(repo_id=args.repo, path=folder_path, recursive=False)
    except Exception as e:
        print(f"ERROR listing repo tree: {e}", file=sys.stderr)
        sys.exit(1)

    # Keep only file paths (exclude subfolders).
    files = sorted(
        item.rfilename for item in items if item.type == "file" and item.rfilename
    )

    manifest = {
        "date": args.date,
        "repo": args.repo,
        "snapshot_ts": datetime.utcnow().isoformat() + "Z",
        "files": files,
        "folder_path": folder_path,
        "cdn_prefix": f"https://huggingface.co/datasets/{args.repo}/resolve/main",
    }

    with open(args.out, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2, sort_keys=True)

    print(f"Manifest written to {args.out} ({len(files)} files)")


if __name__ == "__main__":
    main()
```

## 2) train_cdn.py (Lightning training, CDN-only)

```python
#!/usr/bin/env python3
"""
Lightning training using CDN-only fetches.

Expects file_manifest.json (generated by snapshot_manifest.py) in cwd or via MANIFEST_PATH.
Each file is downloaded via CDN URL (no HF API/auth during data loading).
"""

import json
import os
import sys
from pathlib import Path
from typing import Dict, List, Optional

import pyarrow as pa
import pyarrow.parquet as pq
import requests
import torch
from torch.utils.data import Dataset, DataLoader

try:
    import lightning as L
except ImportError:
    print("ERROR: lightning not installed.", file=sys.stderr)
    sys.exit(1)

MANIFEST_PATH = os.environ.get("MANIFEST_PATH", "file_manifest.json")
CDN_TIMEOUT = int(os.environ.get("CDN_TIMEOUT", "60"))


def cdn_url(repo: str, file_path: str) -> str:
    return f"https://huggingface.co/datasets/{repo}/resolve/main/{file_path}"


class CDNParquetDataset(Dataset):
    """
    Lightweight dataset that downloads parquet files via CDN and projects
    to {prompt, response}. Handles heterogeneous schemas by reading only
    required columns when possible, or fallback to full read + projection.
    """

    def __init__(self, manifest_path: str = MANIFEST_PATH, cache_dir: str = ".cdn_cache"):
        super().__init__()
        self.cache_dir = Path(cache_dir)
        self.cache_dir.mkdir(parents=True, exist_ok=True)

        with open(manifest_path, "r", encoding="utf-8") as f:
            manifest = json.load(f)

        self.repo = manifest["repo"]
        self.files: List[str] = manifest["files"]
        if not self.files:
            raise ValueError("No files found in manifest.")

        self._rows: List[Dict[str, str]] = []
        self._build_index()

    def _download_cdn(self, file_path: str) -> Path:
        url = cdn_url(self.repo, file_path)
        out_path = self.cache_dir / file_path.replace("/", "_")
        if out_path.is_file():
            return out_path

        resp = requests.get(url, timeout=CDN_TIMEOUT)
        resp.raise_for_status()
        out_path.write_bytes(resp.content)
        return out_path

    def _project_row(self, batch: Dict[str, pa.Array]) -> List[Dict[str, str]]:
        """Project batch to {prompt, response}. Tolerant to schema variants."""
        prompt_candidates = ("prompt", "input", "question", "instruction")
        response_candidates = ("response", "output", "answer", "completion")

        prompt_col = next((c for c in prompt_candidates if c in batch), None)
        response_col = next((c for c in response_candidates if c in batch), None)

        rows = []
        length = len(next(iter(batch.values())))
        for i in range(length):
            prompt = str(batch[prompt_col][i]) if prompt_col else ""
            response = str(batch[response_col][i]) if response_col else ""
            if prompt or response:
                rows.append({"prompt": prompt, "response": response})
        return rows

    def _build_index(self) -> None:
        for file_path in self.files:
            local_path = self._download_cdn(file_path)
            try:
                want = [
                    "prompt", "input", "question", "instruction",
                    "response", "output", "answer", "completion",
                ]
                available = pq.read_schema(local_path).names
                cols = [c for c in want if c in available]
                if not cols:
                    table = pq.read_table(local_path)
                else:
                    table = pq.read_table(local_path, columns=cols)

                for batch in table.to_batches():
                    rows = self._project_row(batch.to_pydict())
                    self._rows.extend(rows)
            except Exception as exc:
                print(f"WARNING: failed to process {file_path}: {exc}", file=sys.stderr)

    def __len__(self) -> int:
        return len(self._rows)

    def __getitem__(self, idx: int) -> Dict[str, str]:
        return self._rows[idx]
