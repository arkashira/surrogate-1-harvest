```python
#!/usr/bin/env python3
"""
CDN-bypass ingestion for Surrogate training.
Downloads dataset files directly from HF CDN (no API auth/rate-limit).
"""
import json
import hashlib
import time
import requests
from pathlib import Path
from datetime import datetime, timezone
from typing import List, Dict
from huggingface_hub import HfApi

HF_DATASET_REPO = "axentx/surrogate-training-data"
CDN_BASE = "https://huggingface.co/datasets"

api = HfApi()

def list_date_folder(date_str: str) -> List[str]:
    """List files in date folder (single API call; run from Mac only)."""
    tree = api.list_repo_tree(
        repo_id=HF_DATASET_REPO,
        path=date_str,
        recursive=False
    )
    return [item.path for item in tree if item.type == "file"]

def _sha256(path: Path) -> str:
    h = hashlib.sha256()
    with open(path, "rb") as f:
        for chunk in iter(lambda: f.read(8192), b""):
            h.update(chunk)
    return h.hexdigest()

def download_via_cdn(file_path: str, output_dir: Path, max_retries: int = 3) -> Path:
    """Download via CDN with retries and integrity check."""
    url = f"{CDN_BASE}/{HF_DATASET_REPO}/resolve/main/{file_path}"
    output_dir.mkdir(parents=True, exist_ok=True)
    output_path = output_dir / Path(file_path).name

    for attempt in range(1, max_retries + 1):
        try:
            resp = requests.get(url, timeout=30)
            resp.raise_for_status()
            output_path.write_bytes(resp.content)
            # Basic integrity: non-zero size
            if output_path.stat().st_size > 0:
                return output_path
            raise RuntimeError("Zero-length file")
        except Exception as exc:
            if attempt == max_retries:
                raise
            time.sleep(2 ** attempt)
    raise RuntimeError("Unreachable")

def project_to_prompt_response(parquet_path: Path) -> List[Dict]:
    """Project parquet to {prompt, response} only at parse time."""
    import pyarrow.parquet as pq
    table = pq.read_table(parquet_path)
    df = table.to_pandas()
    return df[["prompt", "response"]].to_dict("records")

def generate_file_list_json(date_str: str, output_path: Path) -> Dict:
    """Generate file list JSON for embedding in training script."""
    files = list_date_folder(date_str)
    manifest = {
        "date": date_str,
        "generated_at": datetime.now(timezone.utc).isoformat(),
        "files": files,
        "checksum": hashlib.sha256(json.dumps(files, sort_keys=True).encode()).hexdigest()
    }
    output_path.write_text(json.dumps(manifest, indent=2))
    return manifest

if __name__ == "__main__":
    import argparse
    parser = argparse.ArgumentParser()
    parser.add_argument("--date", required=True, help="YYYY-MM-DD")
    parser.add_argument("--output-dir", default="./batches")
    parser.add_argument("--generate-manifest", action="store_true")
    args = parser.parse_args()

    root = Path(args.output_dir)
    manifest_path = root / f"file_list_{args.date}.json"

    if args.generate_manifest:
        manifest = generate_file_list_json(args.date, manifest_path)
        print(f"Manifest saved: {len(manifest['files'])} files")
    else:
        manifest = json.loads(manifest_path.read_text())

    # Download all files via CDN
    date_dir = root / args.date
    for file_path in manifest["files"]:
        print(f"Downloading {file_path} via CDN...")
        p = download_via_cdn(file_path, date_dir)
        print(f"  -> {p.name} ({p.stat().st_size} bytes)")
```

#!/usr/bin/env python3
"""
Lightning Studio manager with reuse logic to save quota and survive idle timeouts.
"""
import time
from lightning_sdk import Studio, Teamspace, Machine
from typing import Optional

MACHINE = Machine.L40S
MAX_RETRIES = 3
IDLE_GRACE_SECONDS = 30
STARTUP_WAIT_SECONDS = 45

class StudioManager:
    def __init__(self, teamspace_name: str = "default"):
        self.teamspace = Teamspace(teamspace_name)

    def _find_studio(self, name: str) -> Optional[Studio]:
        for s in self.teamspace.studios:
            if s.name == name:
                return s
        return None

    def get_or_create_studio(self, name: str) -> Studio:
        """Reuse running studio; restart stopped; else create."""
        studio = self._find_studio(name)
        if studio:
            if studio.status == "running":
                print(f"Reusing running studio: {name}")
                return studio
            if studio.status == "stopped":
                print(f"Restarting stopped studio: {name}")
                studio.start(machine=MACHINE)
                time.sleep(STARTUP_WAIT_SECONDS)
                return studio

        print(f"Creating new studio: {name}")
        return Studio(
            name=name,
            teamspace=self.teamspace.name,
            machine=MACHINE,
            create_ok=True,
        )

    def run_training_with_retry(self, studio: Studio, script_path: str) -> bool:
        """Run training with pre-flight status check and bounded retries."""
