# Costinel / design

## Final Implementation Plan — Costinel read-only top-hub signal (≤2 h)

**Goal:** Add a single, production-ready, read-only endpoint that surfaces the strongest knowledge-graph hub as a cost-anomaly signal (Sense + Signal; no execution). Reuses existing knowledge-rag/graph patterns and fits the “top-hub doc insight” workflow.

---

### 1) API contract (fixed, minimal, correct)

- **Method / path:** `GET /api/v1/cost-anomaly/signal/top-hub`
- **Query param:** `for_date` (optional, `YYYY-MM-DD`; defaults to today)
- **Response shape:**
  ```json
  {
    "hub": "MOC",
    "score": 0.92,
    "summary": "Key hub driving cost anomalies; linked to coverage gaps and billing spikes.",
    "related_docs": [
      { "id": "ri-coverage-report", "title": "RI Coverage Report", "score": 0.88 },
      { "id": "aws-billing-anomaly-2026-04", "title": "AWS Billing Anomaly Apr 2026", "score": 0.85 }
    ],
    "for_date": "2026-06-25"
  }
  ```
- **Semantics:** Read-only, no side effects, no execution.

```python
      - nodes: {id, title?, summary?}
      - edges: {weight}

    G.add_node("ri-coverage-report", title="RI Coverage Report")
    G.add_node("aws-billing-anomaly-2026-04", title="AWS Billing Anomaly Apr 2026")
    G.add_edge("MOC", "ri-coverage-report", weight=0.88)
    G.add_edge("MOC", "aws-billing-anomaly-2026-04", weight=0.85)
    return G

def get_top_hub_signal(for_date: Optional[date] = None) -> Dict[str, Any]:
    """
    Resolve top hub from cost-related knowledge graph.
    Uses weighted degree (or PageRank if preferred) + top related docs.
    """
    G = _load_cost_graph(for_date=for_date)

    # Prefer weighted degree (simpler, stable, fast).
    # If graph is large/directed, switch to nx.pagerank(G, weight='weight').
    scores = {n: G.degree(n, weight="weight") for n in G.nodes()}
    top_hub = max(scores, key=scores.get)

    # Top related docs by edge weight
    neighbors = sorted(
        G[top_hub].items(),
        key=lambda kv: kv[1].get("weight", 1.0),
        reverse=True
    )[:5]

    related_docs = [
        {
            "id": n,
            "title": G.nodes[n].get("title", n),
            "score": float(d.get("weight", 1.0)),
        }
        for n, d in neighbors
    ]

    return {
        "hub": top_hub,
        "score": float(scores[top_hub]),
        "summary": G.nodes[top_hub].get("summary", f"Key hub: {top_hub}"),
        "related_docs": related_docs,
        "for_date": str(for_date or date.today()),
    }
```

### 5) FastAPI route (minimal, correct, documented)

File: `app/api/v1/endpoints/cost_anomaly.py`

```python
from fastapi import APIRouter, Query
from datetime import date
from typing import Optional
from app.services.knowledge_rag import get_top_hub_signal
from app.models.schemas import TopHubSignal

router = APIRouter(prefix="/cost-anomaly", tags=["cost-anomaly"])

@router.get("/signal/top-hub", response_model=TopHubSignal)
async def top_hub_signal(
    for_date: Optional[date] = Query(
        None,
        description="Date to evaluate (YYYY-MM-DD); defaults to today"
    ),
) -> TopHubSignal:
    """
    Return the strongest knowledge-graph hub as a cost-anomaly signal.
    Sense + Signal — no execution, no side effects.
    """
    signal = get_top_hub_signal(for_date=for_date)
    return TopHubSignal(**signal)
```

### 6) Frontend card (minimal, non-blocking)

File: `frontend/components/TopHubSignalCard.tsx`

```tsx
import useSWR from "swr";

export default function TopHubSignalCard({ forDate }: { forDate?: string }) {
  const { data, error } = useSWR(
    `/api/v1/cost-anomaly/signal/top-hub${forDate ? `?for_date=${forDate}` : ""}`
  );

  if (!data) return <div className="card">Loading signal…</div>;
  if (error) return null; // Fail silently; don’t block dashboard

  return (
    <div className="card">
      <h4>Top anomaly signal</h4>
      <p><strong>{data.hub}</strong> — {data.summary}</p>
      <small>Score: {data.score.toFixed(2)}</small>
      <ul>
        {data.related_docs.slice(0, 3).map((d: any) => (
          <li key={d.id}>{d.title}</li>
        ))}
      </ul>
    </div>
  );
}
```
