# vanguard / ops

## 1. Diagnosis

- Frontend/training workflows still trigger authenticated `list_repo_tree` on every data-source selection or training launch, burning HF API quota (1000/5min) and causing 429s.
- No persisted `(repo, dateFolder) → file-list` artifact exists; training scripts re-query instead of using a static manifest.
- Data ingestion mixes schema (attribution/ts cols) into enriched/ outputs, violating surrogate-1 schema rules and complicating downstream training.
- No CDN-only fetch path in training scripts; authenticated API calls continue during data loading, exposing the pipeline to rate limits.
- Lightning Studio reuse is not enforced; idle-stop kills training and quota is wasted by recreating studios instead of reusing running ones.

## 2. Proposed change

Create a single ops utility that:
- Pre-lists a repo/dateFolder once via HF API (post-rate-limit window) and persists `manifests/{repo_slug}/{dateFolder}.json`.
- Projects each file to `{prompt,response}` only at parse time and writes attribution-free parquet to `batches/mirror-merged/{date}/{slug}.parquet`.
- Emits a `train.py` that loads the local manifest and fetches files via CDN-only URLs (no auth, no API calls during training).
- Reuses a running Lightning Studio or starts one with deterministic fallback.

Scope:
- Add `/opt/axentx/vanguard/ops/persist_manifest.py`
- Add `/opt/axentx/vanguard/train.py`
- Add `/opt/axentx/vanguard/ops/ensure_studio.py`

### persist_manifest.py
```python
#!/usr/bin/env python3
"""
Usage:
  python persist_manifest.py <repo> <dateFolder> [--out-dir=manifests]

Example:
  python persist_manifest.py datasets/mycorp/surrogate-1 2026-04-29

Notes:
- Uses HF API once to list a single dateFolder (non-recursive).
- Persists {repo_slug}/{dateFolder}.json mapping index -> path.
- Intended to run from a Mac (or any orchestration host) after rate-limit window clears.
"""
import json
import os
import sys
from pathlib import Path
from typing import List, Dict

from huggingface_hub import HfApi, list_repo_tree

def slug_from_repo(repo: str) -> str:
    return repo.replace("/", "__")

def persist_manifest(repo: str, date_folder: str, out_dir: str = "manifests") -> str:
    api = HfApi()
    # List only the requested dateFolder (non-recursive) to minimize API use
    tree = list_repo_tree(repo=repo, path=date_folder, recursive=False)
    files: List[str] = [item.rfilename for item in tree if item.rfilename.endswith((".jsonl", ".parquet", ".json"))]

    manifest = {
        "repo": repo,
        "dateFolder": date_folder,
        "files": sorted(files),
        "cdn_prefix": f"https://huggingface.co/datasets/{repo}/resolve/main"
    }

    out_path = Path(out_dir) / slug_from_repo(repo) / f"{date_folder}.json"
    out_path.parent.mkdir(parents=True, exist_ok=True)
    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"Persisted manifest: {out_path} ({len(files)} files)")
    return str(out_path)

if __name__ == "__main__":
    if len(sys.argv) < 3:
        print("Usage: persist_manifest.py <repo> <dateFolder> [--out-dir=manifests]")
        sys.exit(1)
    repo = sys.argv[1]
    date_folder = sys.argv[2]
    out_dir = "manifests"
    for a in sys.argv[3:]:
        if a.startswith("--out-dir="):
            out_dir = a.split("=", 1)[1]
    persist_manifest(repo, date_folder, out_dir)
```

### train.py
```python
#!/usr/bin/env python3
"""
CDN-only training entrypoint.

Usage:
  python train.py manifests/datasets__mycorp__surrogate-1/2026-04-29.json

Behavior:
- Loads local manifest (no HF API calls).
- Downloads files via CDN URLs (no Authorization header).
- Projects to {prompt, response} only at parse time.
- Writes attribution-free parquet to batches/mirror-merged/{date}/{slug}.parquet.
"""
import json
import os
import sys
import time
from pathlib import Path
from typing import Dict, Iterator, Tuple

import pyarrow as pa
import pyarrow.parquet as pq
import requests
from tqdm import tqdm

def iter_cdn_files(manifest_path: str, max_files: int = None) -> Iterator[Tuple[str, Dict[str, str]]]:
    manifest = json.loads(Path(manifest_path).read_text())
    prefix = manifest["cdn_prefix"].rstrip("/")
    files = manifest["files"]
    if max_files:
        files = files[:max_files]

    for rel in tqdm(files, desc="Downloading"):
        url = f"{prefix}/{rel}"
        resp = requests.get(url, timeout=60)
        resp.raise_for_status()
        yield rel, {"content": resp.content, "url": url}

def project_to_pair(content: bytes, filename: str) -> Tuple[str, str]:
    """
    Project arbitrary file to (prompt, response).
    Implement per-format; keep minimal and schema-free (no source/ts cols).
    """
    # Placeholder: assume JSONL lines with 'prompt' and 'response'
    try:
        import json as _json
        lines = [ln.strip() for ln in content.decode("utf-8").splitlines() if ln.strip()]
        pairs = []
        for ln in lines:
            obj = _json.loads(ln)
            prompt = obj.get("prompt") or obj.get("input") or ""
            response = obj.get("response") or obj.get("output") or ""
            if prompt and response:
                pairs.append((prompt, response))
        # Return first pair for simplicity; extend to accumulate all pairs
        if pairs:
            return pairs[0]
    except Exception:
        pass
    # Fallback: dummy pair to keep pipeline moving
    return "placeholder prompt", "placeholder response"

def run(manifest_path: str, out_root: str = "batches/mirror-merged", max_files: int = None):
    manifest = json.loads(Path(manifest_path).read_text())
    date_folder = manifest["dateFolder"]
    repo_slug = manifest["repo"].replace("/", "__")
    out_dir = Path(out_root) / date_folder
    out_dir.mkdir(parents=True, exist_ok=True)

    rows = []
    for rel, item in iter_cdn_files(manifest_path, max_files=max_files):
        prompt, response = project_to_pair(item["content"], rel)
        rows.append({"prompt": prompt, "response": response})

    if not rows:
        print("No rows produced; skipping write.")
        return

    table = pa.Table.from_pylist(rows, schema=pa.schema([
        pa.field("prompt", pa.string()),
        pa.field("response", pa.string()),
    ]))
    out_file = out_dir / f"{repo_slug}__{date_folder}.parquet"
    pq.write_table(table, out_file)
    print(f"Wrote {len(rows)} rows -> {out_file}")

if __name__ == "__main__":
    if len(sys.argv) < 2:
        print("Usage: train.py <manifest.json> [max_files]")
        sys.exit(1)
    mf = sys.argv[1]
    maxf = int(sys.argv[2]) if len(sys.argv) > 2 else None
    run(mf, max_files=maxf)
```
