# workio / frontend

Below is the **single, merged, production-ready implementation** that keeps the strongest ideas from both candidates while resolving contradictions in favor of correctness and concrete actionability.

---

## 1. Chosen architecture (merged)

| Concern | Chosen solution (why) |
|--------|-----------------------|
| **Optimistic UI + rollback** | Adopt Candidate 1’s per-action optimistic state (fast feedback) **plus** Candidate 2’s persistent queue for offline/retry resilience. |
| **Duplicate prevention** | Idempotency key on every request (Candidate 1) + queue deduplication (Candidate 2). |
| **Offline/retry** | Persistent queue with exponential backoff and max attempts (Candidate 2) is strictly necessary for mobile/field usage. |
| **Visual state** | Clear status pills and toast with inline retry (Candidate 2) + optimistic button state (Candidate 1). |
| **Location** | Optional location capture; failure to get location does not block punch attempt (Candidate 1). |
| **Error handling** | Centralized toast with retry affordance (Candidate 2) + per-action rollback (Candidate 1). |
| **Storage** | `localStorage` queue (Candidate 2) with safe hydration and TTL to avoid stale data. |

  };

  attempts: number;
  lastError?: string | null;

// workio/src/features/punch/usePunchQueue.ts
import { useCallback, useEffect, useRef, useState } from 'react';
import { toast } from 'sonner';
import type { Punch, PunchType } from './types';

const STORAGE_KEY = 'workio:punch-queue';
const MAX_ATTEMPTS = 3;
const BASE_DELAY_MS = 2000;

function now() {
  return Date.now();
}

function loadQueue(): Punch[] {
  try {
    const raw = localStorage.getItem(STORAGE_KEY);
    if (!raw) return [];
    const parsed = JSON.parse(raw) as Punch[];
    // Basic safety: ensure required fields
    return Array.isArray(parsed)
      ? parsed.filter((p) => p && typeof p.id === 'string' && ['IN', 'OUT'].includes(p.type))
      : [];
  } catch {
    return [];
  }
}

function saveQueue(queue: Punch[]) {
  try {
    localStorage.setItem(STORAGE_KEY, JSON.stringify(queue));
  } catch {
    // ignore quota/incognito failures
  }
}

export function usePunchQueue() {
  const [queue, setQueue] = useState<Punch[]>(loadQueue);
  const processing = useRef(false);
  const retryTimeouts = useRef<Record<string, number>>({});

  // persist
  useEffect(() => {
    saveQueue(queue);
  }, [queue]);

  // process queue
  const processQueue = useCallback(async () => {
    if (processing.current) return;
    processing.current = true;

    setQueue((prev) => {
      const next = [...prev];
      let changed = false;

      (async () => {
        for (const item of next.filter((p) => p.status === 'pending')) {
          // avoid parallel processing of same item across re-entrancy
          if (retryTimeouts.current[item.id]) continue;

          try {
            const body: Record<string, any> = {
              type: item.type,
              idempotencyKey: item.id,
              timestamp: item.timestamp,
            };
            if (item.location) body.location = item.location;

            const res = await fetch('/api/punch', {
              method: 'POST',
              headers: { 'Content-Type': 'application/json' },
              body: JSON.stringify(body),
            });

            if (!res.ok) {
              const text = await res.text().catch(() => `HTTP ${res.status}`);
              throw new Error(text || `HTTP ${res.status}`);
            }

            // success
            setQueue((q) =>
              q.map((p) => (p.id === item.id ? { ...p, status: 'synced' as const } : p))
            );
            toast.success(item.type === 'IN' ? 'Clocked in' : 'Clocked out');
          } catch (err: any) {
            const attempt = item.attempts + 1;
            const lastError = err?.message || 'Network error';

            if (attempt >= MAX_ATTEMPTS) {
              setQueue((q) =>
                q.map((p) =>
                  p.id === item.id ? { ...p, status: 'failed' as const, attempts: attempt, lastError } : p
                )
              );
              toast.error(
                `Failed to ${item.type === 'IN' ? 'clock in' : 'clock out'}`,
                {
                  description: lastError,
                  action: {
                    label: 'Retry',
                    onClick: () => retryPunch(item.id),
                  },
                }
              );
            } else {
              // exponential backoff retry
              const delay = BASE_DELAY_MS * 2 ** (attempt - 1);
              retryTimeouts.current[item.id] = window.setTimeout(() => {
                setQueue((q) =>
                  q.map((p) =>
                    p.id === item.id
                      ? { ...p, attempts: attempt, lastError, status: 'pending' as const }
                      : p
                  )
                );
                delete retryTimeouts.current[item.id];
                // trigger next loop
                setTimeout(processQueue, 100);
              }, delay);
            }
          }
        }

        processing.current = false;
        // if items remain pending, re-run
        setQueue((q) => {
          if (q.some((p) => p.status === 'pending')) {
            setTimeout(processQueue, 300);
          }
          return q;
        });
      })();

      return next;
    });
  }, []);
