### 1. tools/build_cdn_manifest.py
```python
#!/usr/bin/env python3
"""
Mac-safe tool: build CDN-only manifest for a HuggingFace dataset folder.
Run once per date folder after rate-limit window clears.
"""
import json
import sys
from datetime import datetime
from pathlib import Path

from huggingface_hub import HfApi

API = HfApi()
REPO = "your-org/surrogate-dataset"  # override via env if needed

def build_manifest(date_folder: str, out_dir: Path):
    out_dir.mkdir(parents=True, exist_ok=True)
    tree = API.list_repo_tree(repo_id=REPO, path=date_folder, recursive=True)

    files = []
    for item in tree:
        if item.type != "file":
            continue
        cdn_url = f"https://huggingface.co/datasets/{REPO}/resolve/main/{item.path}"
        files.append({
            "path": item.path,
            "cdn_url": cdn_url,
            "size": getattr(item, "size", None),
        })

    manifest = {
        "repo": REPO,
        "date_folder": date_folder,
        "generated_at": datetime.utcnow().isoformat() + "Z",
        "files": files,
    }

    out_file = out_dir / f"manifest_{date_folder.replace('/', '_')}.json"
    out_file.write_text(json.dumps(manifest, indent=2))
    print(f"Wrote {len(files)} entries -> {out_file}")
    return out_file

if __name__ == "__main__":
    if len(sys.argv) < 2:
        print("Usage: build_cdn_manifest.py <date_folder> [out_dir]")
        sys.exit(1)
    df = sys.argv[1]
    od = Path(sys.argv[2]) if len(sys.argv) > 2 else Path("manifests")
    build_manifest(df, od)
```

class CDNParquetIterable(IterableDataset):
    def __init__(self, manifest_path: Path):
        manifest = json.loads(manifest_path.read_text())
        self.urls = [f["cdn_url"] for f in manifest["files"] if f["path"].endswith(".parquet")]

    def __iter__(self) -> Iterable[dict]:
        for url in self.urls:

### 3. lightning_idle_guard.py (bash + Python)
```bash
#!/usr/bin/env bash
set -euo pipefail
export SHELL=/bin/bash

python3 - <<'PY'
import os
from lightning import Studio, Machine, Teamspace

STUDIO_NAME = os.getenv("LIGHTNING_STUDIO", "surrogate-train")
MACHINE = Machine.L40S

teamspace = Teamspace()
studio = None

# Reuse running
running = [s for s in teamspace.studios if s.name == STUDIO_NAME and s.status == "running"]
if running:
    studio = running[0]
    print(f"Reusing running studio: {studio.name}")
else:
    # Restart if stopped
    stopped = [s for s in teamspace.studios if s.name == STUDIO_NAME and s.status == "stopped"]
    if stopped:
        studio = stopped[0]
        print(f"Restarting stopped studio: {studio.name}")
        studio.start(machine=MACHINE)
    else:
        # Create if missing
        print(f"Creating new studio: {STUDIO_NAME}")
        studio = Studio.create(name=STUDIO_NAME, machine=MACHINE)

# Safe to run training
studio.run(
    target="train.py",
    arguments=["--manifest", "manifests/manifest_2026-05-03.json"],
)
PY
```
