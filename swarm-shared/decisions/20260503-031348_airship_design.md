# airship / design

## Implementation Plan — airship/surrogate (≤2h)

Highest-value incremental improvement: **make surrogate training HF-rate-limit-proof and Lightning-idle-resilient** by embedding a CDN-only file list and adding auto-recovery for idle studio stops.

### Why this wins
- Removes 429 risk during data load (uses CDN bypass).
- Saves Lightning quota (reuse running studio; restart on idle stop).
- Fits <2h: single script + small train.py patch + one cron/systemd helper.

## 1) Pre-list file paths once (Mac orchestration)

Run this from your Mac (or any dev box) **after** the HF rate-limit window clears. It produces `filelist.json` to be committed/embedded in the training job.

```bash
# surrogate/scripts/build_filelist.sh
#!/usr/bin/env bash
set -euo pipefail

REPO="datasets/your-org/your-mirror"   # change me
DATE_FOLDER="2026-05-03"               # change me (or parameterize)
OUT="surrogate/train/filelist.json"

echo "Listing ${REPO}/${DATE_FOLDER} (non-recursive)..."
# Requires: pip install huggingface_hub
python3 - "$REPO" "$DATE_FOLDER" "$OUT" <<'PY'
import json, sys, os
from huggingface_hub import HfApi

repo = sys.argv[1]
folder = sys.argv[2]
out_path = sys.argv[3]

api = HfApi()
# One API call; non-recursive keeps pagination small
items = api.list_repo_tree(repo=repo, path=folder, recursive=False)
files = [f.rfilename for f in items if f.type == "file"]

os.makedirs(os.path.dirname(out_path), exist_ok=True)
with open(out_path, "w") as f:
    json.dump({"repo": repo, "folder": folder, "files": sorted(files)}, f, indent=2)

print(f"Wrote {len(files)} files -> {out_path}")
PY

echo "Done."
```

Make executable:
```bash
chmod +x surrogate/scripts/build_filelist.sh
```

## 2) CDN-only DataLoader (zero API calls during training)

Replace any `load_dataset(streaming=True)` or recursive HF API usage with this loader. It uses only CDN URLs (`resolve/main/...`) and projects to `{prompt,response}` at parse time (avoids pyarrow schema issues).

```python
# surrogate/train/data.py
import json
import pyarrow as pa
import pyarrow.parquet as pq
import requests
from typing import List, Dict, Iterator
from io import BytesIO

class CDNParquetStream:
    """
    Stream rows from HF-hosted parquet files using CDN URLs only.
    No Authorization header; no HF API calls during training.
    """
    BASE = "https://huggingface.co/datasets"

    def __init__(self, filelist_path: str):
        with open(filelist_path) as f:
            meta = json.load(f)
        self.repo = meta["repo"]
        self.folder = meta["folder"]
        self.files: List[str] = meta["files"]
        self._idx = 0

    def _cdn_url(self, path: str) -> str:
        return f"{self.BASE}/{self.repo}/resolve/main/{path}"

    def __iter__(self) -> Iterator[Dict[str, str]]:
        for rel in self.files:
            full = f"{self.folder}/{rel}" if self.folder else rel
            url = self._cdn_url(full)
            resp = requests.get(url, timeout=30)
            resp.raise_for_status()
            table = pq.read_table(BytesIO(resp.content))
            # Project to minimal schema; ignore mixed/extra columns
            if "prompt" not in table.column_names or "response" not in table.column_names:
                continue
            prompts = table["prompt"].to_pylist()
            responses = table["response"].to_pylist()
            for p, r in zip(prompts, responses):
                if p is None or r is None:
                    continue
                yield {"prompt": str(p), "response": str(r)}
```

Usage in train.py (minimal patch):

## 3) Lightning studio reuse + idle-stop recovery

Avoid recreating studios and restart automatically if idle-stop killed training.

```python
# surrogate/train/lightning_utils.py
import time
from lightning import Lightning, Teamspace, Machine, Studio

LIGHTNING_CLOUD = "lightning-public-prod"  # or "lightning-lambda-prod" for H200
MACHINE = Machine.L40S

def get_or_create_studio(name: str, max_retries: int = 3) -> Studio:
    lightning = Lightning()
    ts = Teamspace()

    # Reuse running studio
    for s in ts.studios:
        if s.name == name and s.status == "Running":
            print(f"Reusing running studio: {name}")
            return s

    # Create if not exists
    for attempt in range(max_retries):
        try:
            s = Studio(
                name=name,
                cloud=LIGHTNING_CLOUD,
                machine=MACHINE,
                create_ok=True
            )
            print(f"Created studio: {name}")
            return s
        except Exception as e:
            wait = 30 * (attempt + 1)
            print(f"Studio creation failed (attempt {attempt+1}): {e}. Retrying in {wait}s")
            time.sleep(wait)
    raise RuntimeError("Failed to create studio after retries")

def run_with_recovery(studio_name: str, target):
    """
    Run target in studio; auto-restart if studio stopped (idle timeout).
    """
    while True:
        s = get_or_create_studio(studio_name)
        try:
            if s.status != "Running":
                print(f"Studio {studio_name} not running (status={s.status}). Restarting...")
                s.start(machine=MACHINE)
                time.sleep(60)  # allow boot
            s.run(target)
            break  # finished
        except Exception as e:
            print(f"Training run failed: {e}")
            print("Will restart studio and retry in 60s...")
            time.sleep(60)
```

Usage:
```python
# surrogate/train/run.py
from lightning_utils import run_with_recovery
from train import train_step  # your training loop callable

if __name__ == "__main__":
    run_with_recovery("surrogate-train-l40s", target=train_step)
```
