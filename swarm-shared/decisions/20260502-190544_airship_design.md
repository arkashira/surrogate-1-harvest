# airship / design

## Incremental Improvement: Zero-config discovery CLI (`airship discover`)

**Value**: <2h implementation that operationalizes past patterns (business research + knowledge-rag + top-hub insight) into a single command that surfaces strategic context for the Arkship/Surrogate platform.

---

## Implementation Plan

### 1. Create `airship/cli/discover.py` (main entry)
```python
#!/usr/bin/env python3
"""
airship discover - Zero-config discovery CLI
Executes business research → knowledge-rag → top-hub insight pipeline.
"""
import subprocess
import sys
import os
import json
from pathlib import Path

PROJECT_ROOT = Path(__file__).parent.parent.parent

def run(cmd, cwd=None, capture=True):
    """Run shell command with error handling."""
    print(f"→ {cmd}")
    result = subprocess.run(
        cmd, shell=True, cwd=cwd or PROJECT_ROOT,
        capture_output=capture, text=True
    )
    if result.returncode != 0:
        print(f"⚠️  Non-zero exit {result.returncode}: {result.stderr.strip()}")
    return result

def step_business_research():
    """Pattern: business research with knowledge-rag pipeline."""
    print("\n== 📊 Business Research ==")
    script = PROJECT_ROOT / "scripts" / "granite-business-research.sh"
    if script.exists():
        run(f"bash {script}")
    else:
        print(f"⚠️  {script} not found, skipping business research")

def step_knowledge_rag():
    """Pattern: knowledge-rag query for contextual insights."""
    print("\n== 🧠 Knowledge RAG ==")
    # Look for knowledge-rag CLI or python module
    rag_script = PROJECT_ROOT / "scripts" / "knowledge-rag.sh"
    if rag_script.exists():
        run(f"bash {rag_script} --top-hub")
    else:
        # Fallback: try python module
        result = run("python -m knowledge_rag --top-hub 2>/dev/null || true")
        if result.returncode != 0:
            print("⚠️  knowledge-rag not available, skipping")

def step_top_hub_insight():
    """Pattern: top-hub doc insight — review most-connected hub."""
    print("\n== 🔝 Top Hub Insight ==")
    # Try to get top hub from knowledge graph
    result = run("python -c \"\nimport json, os, sys\nfrom pathlib import Path\n# Look for graph data or cached insights\ngraph_dir = Path.home() / '.airship' / 'knowledge-graph'\ninsight_file = graph_dir / 'top-hub.json'\nif insight_file.exists():\n    data = json.loads(insight_file.read_text())\n    print(json.dumps(data, indent=2))\nelse:\n    print('ℹ️  No cached top-hub insight found')\n\" 2>/dev/null || echo 'ℹ️  Skipping top-hub lookup'")

def step_generate_summary():
    """Generate consolidated discovery summary."""
    print("\n== 🎯 Discovery Summary ==")
    summary = {
        "project": "airship",
        "scope": "Arkship + Surrogate AI Platform",
        "services": ["Arkship (8000)", "Surrogate AI (8001)"],
        "architecture": "2 independent microservices",
        "key_features": [
            "Workflow Orchestration (Temporal)",
            "Incident Management (Perception Layer)",
            "Intent Language Compiler (YAML → IaC)",
            "Knowledge Graph (Neo4j, 250+ tools)",
            "Vector Store (Qdrant, 130+ repos)",
            "Multi-model ensemble (Qwen + Mistral + DeepSeek-R1)"
        ],
        "quick_start": {
            "full_stack": "docker-compose -f docker-compose.microservices.yml up -d",
            "arkship_only": "cd arkship && docker-compose up -d",
            "surrogate_only": "cd surrogate && docker-compose up -d"
        }
    }
    print(json.dumps(summary, indent=2))
    # Save to cache
    cache_dir = PROJECT_ROOT / ".airship"
    cache_dir.mkdir(exist_ok=True)
    (cache_dir / "discovery-summary.json").write_text(json.dumps(summary, indent=2))
    print(f"\n✅ Summary saved to {cache_dir / 'discovery-summary.json'}")

def main():
    print("🚀 airship discover — Zero-config discovery pipeline")
    print("=" * 55)
    step_business_research()
    step_knowledge_rag()
    step_top_hub_insight()
    step_generate_summary()
    print("\n✨ Done!")

if __name__ == "__main__":
    main()
```

### 2. Create wrapper script `scripts/airship-discover.sh`
```bash
#!/usr/bin/env bash
# Wrapper for airship discover CLI
set -euo pipefail
SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
PROJECT_ROOT="$(cd "$SCRIPT_DIR/.." && pwd)"

export PYTHONPATH="$PROJECT_ROOT:$PYTHONPATH"

exec python3 "$PROJECT_ROOT/airship/cli/discover.py" "$@"
```

### 3. Update `Makefile` (or create one) with discover target
```makefile
# Add to existing Makefile or create new one
.PHONY: discover
discover:
	@bash scripts/airship-discover.sh
```

### 4. Make executable and add to PATH convenience
```bash
chmod +x scripts/airship-discover.sh
chmod +x airship/cli/discover.py
```

### 5. Optional: Add to `pyproject.toml` console scripts (if using poetry/pip)
```toml
[tool.poetry.scripts]
airship-discover = "airship.cli.discover:main"
```
