# workio / backend

## Final synthesized implementation (best of both proposals)

This version keeps the strongest, most production-ready choices from both candidates and resolves contradictions in favor of correctness and concrete actionability.

- **Tenant cache**: Use `lru-cache` with 10m TTL (Candidate 1) and typed cache values + DB fallback (Candidate 2).
- **Health/readiness**: Keep both `/health` (liveness) and `/ready` (readiness) with real DB checks.
- **Request ID**: Use `uuid` (not crypto.randomUUID) for guaranteed RFC4122 IDs and allow `X-Request-ID` override.
- **LINE API client**: Use `axios-retry` with exponential backoff + 429 handling (Candidate 1) and export a dedicated `lineClient` for reuse.
- **Graceful shutdown**: Add 30s drain + in-flight tracking (Candidate 2) — critical for k8s/PM2.
- **Clock-in throttle**: Keep lightweight in-memory throttle (Candidate 1) with 5m cooldown.

### 2. Tenant cache service
`server/src/services/tenantService.ts`
```ts
import { LRUCache } from 'lru-cache';
import { pool } from '../db';

type Tenant = { id: string; name: string; channelId: string };

const tenantCache = new LRUCache<string, Tenant>({
  max: 500,
  ttl: 1000 * 60 * 10, // 10m
  allowStale: false,
  updateAgeOnGet: true,
});

export async function getTenantByChannelId(channelId: string): Promise<Tenant | null> {
  const cached = tenantCache.get(channelId);
  if (cached) return cached;

  const { rows } = await pool.query(
    'SELECT id, name, "channelId" FROM tenants WHERE "channelId" = $1 LIMIT 1',
    [channelId]
  );
  const tenant = rows[0] || null;
  if (tenant) tenantCache.set(channelId, tenant);
  return tenant;
}

export function invalidateTenantCache(channelId: string) {
  tenantCache.delete(channelId);
}
```

### 3. Request ID middleware
`server/src/middleware/requestId.ts`
```ts
import { Request, Response, NextFunction } from 'express';
import { v4 as uuidv4 } from 'uuid';

export function requestId(req: Request, _res: Response, next: NextFunction) {
  req.id = (req.headers['x-request-id'] as string) || uuidv4();
  next();
}

declare module 'express-serve-static-core' {
  interface Request {
    id: string;
  }
}
```

### 4. LINE API client with retry + circuit-breaker guard
`server/src/middleware/lineRetry.ts`
```ts
import axios, { AxiosInstance } from 'axios';
import axiosRetry from 'axios-retry';

export const lineClient: AxiosInstance = axios.create();

axiosRetry(lineClient, {
  retries: 2,
  retryDelay: (retryCount) => Math.min(1000 * 2 ** retryCount, 4000),
  retryCondition: (error) =>
    axiosRetry.isNetworkOrIdempotentRequestError(error) ||
    error.response?.status === 429 ||
    (error.response?.status >= 500 && error.response.status < 600),
});
```

### 5. Clock-in throttle middleware
`server/src/middleware/clockThrottle.ts`
```ts
import { Request, Response, NextFunction } from 'express';

const lastClockIn = new Map<string, number>();
const COOLDOWN_MS = 1000 * 60 * 5; // 5m

export function clockInThrottle(req: Request, res: Response, next: NextFunction) {
  const userId = req.body?.userId || (req as any)?.user?.id;
  if (!userId) return next();

  const now = Date.now();
  const prev = lastClockIn.get(userId);
  if (prev && now - prev < COOLDOWN_MS) {
    return res.status(429).json({
      error: 'too_soon',
      retryAfter: Math.ceil((prev + COOLDOWN_MS - now) / 1000),
    });
  }
  lastClockIn.set(userId, now);
  next();
}
```

### 6. Main app with health/ready, graceful shutdown, and webhook
`server/src/index.ts`
```ts
import express, { Request, Response } from 'express';
import http from 'http';
import { requestId } from './middleware/requestId';
import { lineClient } from './middleware/lineRetry';
import { getTenantByChannelId } from './services/tenantService';
import { clockInThrottle } from './middleware/clockThrottle';
import { pool } from './db';

const app = express();
app.use(express.json());
app.use(requestId);

// Health & readiness
app.get('/health', (_req: Request, res: Response) =>
  res.json({ status: 'ok', uptime: process.uptime() })
);

app.get('/ready', async (_req: Request, res: Response) => {
  try {
    await pool.query('SELECT 1');
    res.json({ status: 'ready' });
  } catch (err) {
    res.status(503).json({ status: 'not ready', db: 'down' });
  }
});

// Example clock-in route with throttle
const clocksRouter = express.Router();
clocksRouter.post('/clock-in', clockInThrottle, async (req: Request, res: Response) => {
  // Your clock-in logic here
  res.json({ ok: true });
});
app.use('/api/clocks', clocksRouter);

// LINE webhook with tenant cache + retry client + request-id logging
app.post('/webhook/line', async (req: Request, res: Response) => {
  const reqId = (req as any).id;
  const channelId = req.body?.destination;
  if (!channelId) return res.status(400).send();

  try {
    const tenant = await getTenantByChannelId(channelId);
    if (!tenant) return res.status(404).send();

    const events = req.body.events || [];
    for (const ev of events) {
      if (!ev.replyToken) continue;
      await lineClient.post(
        'https://api.line.me/v2/bot/message/reply',
        { replyToken: ev.replyToken, messages: [{ type: 'text', text: 'Received' }] },
        { headers: { Authorization: `Bearer ${process.env.LINE_CHANNEL_ACCESS_TOKEN}` } }
      );
    }

    res.json({ requestId: reqId, tenantId: tenant.id, processed: events.length });
  } catch (err) {
    console.error(`[${reqId}] webhook error`, err);
    res.status(500).send();
  }
});

// Mount other routes
// app.use('/api/leaves', leavesRouter);

const server = http.createServer(app);

function gracefulShutdown(signal: string) {
  return () => {
    console.log(`[shutdown] ${signal} received. Draining for 30s...`);
    server.close(() => {
      console.log('[shutdown] HTTP server closed.');
      process.exit(0);
    });

    // Force exit after drain
    setTimeout(() => {
      console.error('[shutdown] Force exit after timeout.');
      process.exit(1);
    }, 30_000).unref();
  };
}

const PORT = process.env.PORT || 3000;
server.listen(PORT, () => {
  console.log(`Server listening on port ${PORT}`);
});

process.on('SIGTERM', gracefulShutdown('SIGTERM'));
process.on('SIGINT', gracefulShutdown('SIGINT'));

export default app;
```
