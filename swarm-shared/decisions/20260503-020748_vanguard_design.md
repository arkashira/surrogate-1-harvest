# vanguard / design

## 1. Diagnosis

- No persisted `(repo, dateFolder) → file-list` manifest: every orchestration/training run triggers authenticated `list_repo_tree`, burning HF API quota and risking 429s.
- Data loader likely uses `load_dataset(streaming=True)` or repeated per-file authenticated fetches instead of CDN-only downloads, amplifying rate-limit exposure.
- No deterministic repo-sharding for HF commit cap (128/hr/repo): ingestion writes risk hitting limit and blocking pipeline.
- Training jobs may recreate Lightning Studio instead of reusing running ones, wasting 80hr/mo quota.
- No guard against Lightning idle-stop killing training; missing status-check + restart logic causes silent job death.

## 2. Proposed change

Add a lightweight manifest + CDN-only data loader for Surrogate-1 training:

- Create `vanguard/data/manifest.py` — one API call to `list_repo_tree` per `(repo, dateFolder)`, save JSON mapping `{file_path, cdn_url, sha}` to `manifests/{repo}/{dateFolder}.json`.
- Create `vanguard/data/cdn_loader.py` — deterministic repo shard selector and CDN-only fetch (zero auth) using `hf_hub_download`-style URL or direct `requests.get(cdn_url, stream=True)` with byte-range for parquet projection.
- Update training launcher (`vanguard/train.py` or equivalent) to accept `--manifest` and use `cdn_loader` for data; add Lightning Studio reuse + idle-restart guard.

Scope: ~150–200 lines across 3 files; <2h to ship.

### vanguard/data/manifest.py
```python
from __future__ import annotations
import json
import os
from datetime import datetime
from pathlib import Path
from typing import Dict, List

from huggingface_hub import HfApi, Repository

HF_API = HfApi()
MANIFEST_ROOT = Path(__file__).parent.parent / "manifests"

def list_and_persist(repo_id: str, date_folder: str, out_root: Path = MANIFEST_ROOT) -> Path:
    """
    Single authenticated list_repo_tree call for one date folder.
    Saves manifest: {repo}/{date_folder}.json
    """
    out_root.mkdir(parents=True, exist_ok=True)
    out_file = out_root / repo_id.replace("/", "--") / f"{date_folder}.json"
    out_file.parent.mkdir(parents=True, exist_ok=True)

    # Non-recursive top-level of date folder; keeps API usage minimal
    tree = HF_API.list_repo_tree(repo_id=repo_id, path=date_folder, recursive=False)
    entries = []
    for item in tree:
        if item.type != "file":
            continue
        path = item.path
        entries.append(
            {
                "path": path,
                "cdn_url": f"https://huggingface.co/datasets/{repo_id}/resolve/main/{path}",
                "size": item.size if hasattr(item, "size") else None,
                "sha": getattr(item, "oid", None),
            }
        )

    manifest = {
        "repo_id": repo_id,
        "date_folder": date_folder,
        "generated_at": datetime.utcnow().isoformat() + "Z",
        "entries": entries,
    }

    with open(out_file, "w") as f:
        json.dump(manifest, f, indent=2)
    return out_file


def load_manifest(repo_id: str, date_folder: str, manifest_root: Path = MANIFEST_ROOT) -> Dict:
    p = manifest_root / repo_id.replace("/", "--") / f"{date_folder}.json"
    if not p.exists():
        raise FileNotFoundError(f"Manifest missing: {p}. Run manifest.list_and_persist first.")
    with open(p) as f:
        return json.load(f)
```

### vanguard/data/cdn_loader.py
```python
from __future__ import annotations
import hashlib
import json
from pathlib import Path
from typing import Iterator, Tuple

import pyarrow.parquet as pq
import requests
from tqdm import tqdm

from .manifest import load_manifest

# Deterministic repo shard by hash(slug) % N to respect HF commit cap
def pick_shard(slug: str, n_shards: int = 5) -> int:
    digest = int(hashlib.sha256(slug.encode()).hexdigest(), 16)
    return digest % n_shards

def cdn_parquet_projection(
    manifest_path: Path,
    columns: Tuple[str, ...] = ("prompt", "response"),
    max_files: int | None = None,
) -> Iterator[dict]:
    """
    CDN-only parquet projection (zero auth).
    Reads manifest, streams each file via CDN URL, projects columns.
    """
    with open(manifest_path) as f:
        manifest = json.load(f)

    entries = manifest["entries"]
    if max_files:
        entries = entries[:max_files]

    for item in tqdm(entries, desc="CDN fetch"):
        url = item["cdn_url"]
        # Stream download; seekable via Range requests supported by HF CDN
        resp = requests.get(url, stream=True, timeout=60)
        resp.raise_for_status()
        # Use temp file to satisfy pq.read_table file-like requirement
        import tempfile

        with tempfile.NamedTemporaryFile(suffix=".parquet") as tmp:
            for chunk in resp.iter_content(chunk_size=8192):
                tmp.write(chunk)
            tmp.flush()
            try:
                table = pq.read_table(tmp.name, columns=list(columns))
            except Exception as exc:
                # Skip malformed files; log and continue
                print(f"Skipping {url}: {exc}")
                continue
            df = table.to_pandas()
            for _, row in df.iterrows():
                yield {k: row[k] for k in columns if k in row}
```

### vanguard/train.py (minimal diff / integration)
```python
from pathlib import Path
import lightning as L
from axentx.data.manifest import list_and_persist
from axentx.data.cdn_loader import cdn_parquet_projection

def train(
    repo_id: str,
    date_folder: str,
    manifest_root: Path = Path(__file__).parent / "manifests",
    max_files: int | None = None,
):
    # 1) Reuse running studio if exists
    teamspace = L.Teamspace()
    studio_name = f"surrogate1-{repo_id.replace('/', '-')}-{date_folder}"
    studio = None
    for s in teamspace.studios:
        if s.name == studio_name and s.status == "running":
            studio = s
            print(f"Reusing running studio: {studio_name}")
            break

    if studio is None:
        studio = L.Studio(
            name=studio_name,
            machine=L.Machine.L40S,
            create_ok=True,
        )

    # 2) Generate manifest once (Mac orchestrator) or load existing
    manifest_file = manifest_root / repo_id.replace("/", "--") / f"{date_folder}.json"
    if not manifest_file.exists():
        print("Generating manifest (single API call)...")
        manifest_file = list_and_persist(repo_id, date_folder, manifest_root)

    # 3) Build dataset via CDN-only loader (zero auth during training)
    def dataset_generator():
        yield from cdn_parquet_projection(manifest_file, columns=("prompt", "response"), max_files=max_files)

    # 4) Guard against idle-stop: check status before run; restart if stopped
    if studio.status != "running":
        print("Studio not running; restarting...")
        studio.start(machine=L.Machine.L40S)
```
