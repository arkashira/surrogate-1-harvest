```python
#!/usr/bin/env python3
"""
CDN-only dataset loader for Surrogate-1 training.
Bypasses HF API rate limits by using direct CDN downloads.
Handles heterogeneous schemas via per-file download + projection.
"""
import json
import logging
from pathlib import Path
from typing import Dict, List, Optional, Iterator
from dataclasses import dataclass

import pyarrow as pa
import pyarrow.parquet as pq
import requests
from tqdm import tqdm

logger = logging.getLogger(__name__)


@dataclass
class TrainingExample:
    prompt: str
    response: str
    source_file: str


class CDNDatasetLoader:
    """
    Loads datasets from HuggingFace CDN without API authentication.
    Uses manifest file listing to avoid recursive API calls.
    """
    
    CDN_BASE = "https://huggingface.co/datasets"
    
    def __init__(
        self,
        repo_id: str,
        manifest_path: Path,
        cache_dir: Path = Path("./.cache/hf_cdn"),
    ):
        """
        Args:
            repo_id: HF dataset repo (e.g., "axentx/surrogate-1-mirror")
            manifest_path: Path to JSON manifest with file listing
            cache_dir: Local cache for downloaded files
        """
        self.repo_id = repo_id
        self.cache_dir = Path(cache_dir)
        self.cache_dir.mkdir(parents=True, exist_ok=True)
        
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        
        self._validate_manifest()
    
    def _validate_manifest(self):
        """Ensure manifest has required structure."""
        if "date_folder" not in self.manifest:
            raise ValueError("Manifest missing 'date_folder'")
        if "files" not in self.manifest or not isinstance(self.manifest["files"], list):
            raise ValueError("Manifest missing 'files' list")
    
    def _cdn_url(self, file_path: str) -> str:
        """Build CDN URL for a dataset file (no auth required)."""
        return f"{self.CDN_BASE}/{self.repo_id}/resolve/main/{file_path}"
    
    def _cache_path(self, file_path: str) -> Path:
        """Get local cache path for a file."""
        # Flatten path to avoid deep nesting, use hash for uniqueness
        safe_name = file_path.replace("/", "_").replace("\\", "_")
        return self.cache_dir / safe_name
    
    def _download_file(self, file_path: str, force: bool = False) -> Path:
        """
        Download a single file from CDN.
        
        Args:
            file_path: Path within dataset repo
            force: Redownload even if cached
            
        Returns:
            Path to cached file
        """
        cache_file = self._cache_path(file_path)
        
        if cache_file.exists() and not force:
            logger.debug(f"Using cached: {file_path}")
            return cache_file
        
        url = self._cdn_url(file_path)
        logger.info(f"Downloading from CDN: {file_path}")
        
        try:
            response = requests.get(url, stream=True, timeout=30)
            response.raise_for_status()
            
            cache_file.parent.mkdir(parents=True, exist_ok=True)
            with open(cache_file, "wb") as f:
                for chunk in response.iter_content(chunk_size=8192):
                    f.write(chunk)
            
            logger.debug(f"Cached: {cache_file}")
            return cache_file
        
        except requests.RequestException as e:
            logger.error(f"Failed to download {file_path}: {e}")
            raise
    
    def _project_to_training_pair(
        self,
        parquet_path: Path,
        source_file: str,
    ) -> List[TrainingExample]:
        """
        Load a parquet file and project to {prompt, response} pairs.
        Handles heterogeneous schemas by trying common field names.
        
        Args:
            parquet_path: Path to parquet file
            source_file: Original path in dataset repo
            
        Returns:
            List of training examples
        """
        try:
            table = pq.read_table(parquet_path)
        except Exception as e:
            logger.warning(f"Failed to read {parquet_path}: {e}")
            return []
        
        # Find prompt/response columns (handle schema variations)
        schema_names = [name.lower() for name in table.column_names]
        
        prompt_col = None
        response_col = None
        
        # Common prompt field names
        for candidate in ["prompt", "instruction", "input", "question", "context"]:
            if candidate in schema_names:
                prompt_col = table.column_names[schema_names.index(candidate)]
                break
        
        # Common response field names
        for candidate in ["response", "output", "completion", "answer", "text"]:
            if candidate in schema_names:
                response_col = table.column_names[schema_names.index(candidate)]
                break
        
        if not prompt_col or not response_col:
            logger.warning(
                f"Could not find prompt/response columns in {source_file}. "
                f"Available: {table.column_names}"
            )
            return []
        
        # Extract pairs
        examples = []
        prompts = table.column(prompt_col).to_pylist()
        responses = table.column(response_col).to_pylist()
        
        for prompt, response in zip(prompts, responses):
            if prompt and response:  # Skip empty
                examples.append(TrainingExample(
                    prompt=str(prompt).strip(),
                    response=str(response).strip(),
                    source_file=source_file,
                ))
        
        logger.info(f"Extracted {len(examples)} pairs from {source_file}")
        return examples
    
    def load_dataset(
        self,
        max_files: Optional[int] = None,
        force_download: bool = False,
    ) -> Iterator[List[TrainingExample]]:
        """
        Stream training examples from CDN.
        
        Yields:
            Batches of training examples (one batch per file)
        """
        files = self.manifest["files"]
        date_folder = self.manifest["date_folder"]
        
        if max_files:
            files = files[:max_files]
        
        logger.info(f"Loading {len(files)} files from {date_folder}")
        
        for file_path in tqdm(files, desc="Loading dataset files"):
            try:
                # Full path includes date folder
                full_path = f"{date_folder}/{file_path}"
                cached = self._download_file(full_path, force=force_download)
                
                examples = self._project_to_training_pair(cached, file_path)
                if examples:
                    yield examples
            
            except Exception as e:
                logger.error(f"Failed to process {file_path}: {e}")
                continue
    
    def load_all(self, max_files: Optional[int] = None) -> List[TrainingExample]:
        """Load all examples into memory (for small datasets)."""
        all_examples = []
        for batch in self.load_dataset(max_files=max_files):
            all_examples.extend(batch)
        return all_examples
