### `surrogate/data/hf_cdn_loader.py`
```python
import json
import os
import pyarrow.parquet as pq
import pyarrow as pa
import requests
from typing import List, Dict, Iterator
from pathlib import Path

HF_DATASET_REPO = os.getenv("HF_DATASET_REPO", "org/surrogate-dataset")
CDN_ROOT = f"https://huggingface.co/datasets/{HF_DATASET_REPO}/resolve/main"

def list_date_folder(date_folder: str) -> List[str]:
    """
    Run once (or during image build) from a non-rate-limited environment.
    Returns relative file paths under date_folder.
    """
    from huggingface_hub import list_repo_tree
    tree = list_repo_tree(repo_id=HF_DATASET_REPO, path=date_folder, recursive=False)
    return sorted(item.path for item in tree if item.type == "file")

def cdn_lines(file_paths: List[str], columns: tuple = ("prompt", "response")) -> Iterator[Dict[str, str]]:
    for path in file_paths:
        url = f"{CDN_ROOT}/{path}"
        resp = requests.get(url, timeout=30)
        resp.raise_for_status()
        table = pq.read_table(pa.BufferReader(resp.content))

        available = set(table.column_names)
        proj_cols = [c for c in columns if c in available]
        if not proj_cols:
            continue

        tbl = table.select(proj_cols)
        for batch in tbl.to_batches(max_chunksize=8192):
            df = batch.to_pydict()
            n = len(next(iter(df.values()))) if df else 0
            for i in range(n):
                yield {k: v[i] for k, v in df.items()}

def write_mirror_parquet(rows: Iterator[Dict[str, str]], out_path: Path, batch_size: int = 16384):
    out_path.parent.mkdir(parents=True, exist_ok=True)
    schema = pa.schema([pa.field("prompt", pa.string()), pa.field("response", pa.string())])
    with pq.ParquetWriter(out_path, schema) as writer:
        batch_buf = {k: [] for k in schema.names}
        for row in rows:
            for k in schema.names:
                batch_buf[k].append(row.get(k, ""))
            if len(batch_buf["prompt"]) >= batch_size:
                writer.write_table(pa.table(batch_buf, schema=schema))
                for k in batch_buf:
                    batch_buf[k].clear()
        if batch_buf["prompt"]:
            writer.write_table(pa.table(batch_buf, schema=schema))
```

### `surrogate/train/studio_manager.py`
```python
import os
from lightning_sdk import Studio, Teamspace, Machine
from typing import Optional

TEAMSPACE = os.getenv("LIGHTNING_TEAMSPACE", "default")
STUDIO_NAME = os.getenv("LIGHTNING_STUDIO_NAME", "surrogate-train")

def get_or_create_studio(machine: Machine = Machine.L40S) -> Studio:
    team = Teamspace(TEAMSPACE)
    for s in team.studios:
        if s.name == STUDIO_NAME and s.status == "running":
            return Studio(id=s.id)
    return Studio.create(name=STUDIO_NAME, teamspace=TEAMSPACE, machine=machine, create_ok=True)

def run_with_lifecycle(studio: Studio, script: str, args: list):
    if studio.status != "running":
        studio.start(machine=Machine.L40S)
    return studio.run(script, args=args)
```

### `surrogate/api/train.py`
```python
from fastapi import APIRouter
from ..data.hf_cdn_loader import list_date_folder, cdn_lines, write_mirror_parquet
from ..train.studio_manager import get_or_create_studio, run_with_lifecycle
from pathlib import Path

router = APIRouter()

@router.post("/api/v1/train/start")
async def start_training(repo: str, date: str, machine: str = "L40S"):
    # 1) File list (ideally precomputed and embedded; fallback to on-demand)
    files = list_date_folder(date)
    # 2) Mirror parquet via CDN
    out_path = Path(f"batches/mirror-merged/{date}/train.parquet")
    rows = cdn_lines(files)
    write_mirror_parquet(rows, out_path)
    # 3) Launch with studio reuse
    studio = get_or_create_studio(Machine[machine] if hasattr(Machine, machine) else Machine.L40S)
    run = run_with_lifecycle(studio, script="train.py", args=["--data", str(out_path), "--epochs", "1"])
    return {"run_id": run.id, "studio_id": studio.id, "status": "started"}
```
