## Final Synthesis (Best Parts + Correctness + Actionability)

**Diagnosis (merged, corrected)**
- No content-addressed manifest exists → training/frontend hit HF API at runtime, causing 429s and non-reproducible epochs.
- Mixed-schema ingestion writes extra `source`/`ts` columns into `enriched/` instead of projecting to `{prompt, response}` only, breaking downstream `load_dataset` and training stability.
- HF API rate-limit (1000 req/5min) is exhausted by recursive `list_repo_tree`/`list_repo_files` and repeated file metadata calls during ingestion/training.
- No CDN-bypass strategy: training scripts use `load_dataset(streaming=True)` or runtime API calls instead of raw CDN fetches, triggering auth/rate-limit overhead.
- No precomputed file manifest embedded in training jobs, forcing Lightning Studio to hit the API during data loading and wasting quota.

**Proposed change (merged, prioritized)**
Create a single `/opt/axentx/vanguard/ops/manifest.py` (~120–150 lines) that:
- Scans local `batches/mirror-merged/{date}/` and emits a content-addressed manifest JSON mapping `{sha256: {path, size, prompt_col, response_col}}`.
- Generates `train_filelist.json` containing only CDN URLs (`https://huggingface.co/datasets/{repo}/resolve/main/{path}`) for the selected date folder.
- Adds a tiny CLI (`python -m vanguard.ops.manifest --date 2026-05-03 --repo org/surrogate-1`) to run from the Mac orchestration box after the rate-limit window clears.
- Exposes a `load_local_or_cdn(filelist)` helper and a `project_to_prompt_response` transform so training scripts can do zero-API, schema-resilient data loading.
- Includes a minimal `train.py` diff to accept `--filelist` and use CDNParquetDataset.

Scope: one new file + one small update to `train.py` to accept `--filelist`.

```python
# /opt/axentx/vanguard/ops/manifest.py
#!/usr/bin/env python3
"""
Generate content-addressed manifest and CDN filelist for HF dataset ingestion.

Usage:
  python -m vanguard.ops.manifest --date 2026-05-03 --repo org/surrogate-1
"""
import argparse
import hashlib
import json
import os
import sys
from datetime import datetime, timezone
from pathlib import Path
from typing import Dict, List, Any

HF_DATASETS_CDN = "https://huggingface.co/datasets"

def sha256_file(path: Path, block_kb: int = 64) -> str:
    h = hashlib.sha256()
    with path.open("rb") as f:
        for block in iter(lambda: f.read(block_kb * 1024), b""):
            h.update(block)
    return h.hexdigest()

def build_manifest_local(batches_root: Path, date_str: str) -> Dict[str, Dict[str, Any]]:
    """
    Scan batches/mirror-merged/{date}/ for parquet files.
    Produce manifest: {sha256: {path, size, prompt_col, response_col}}
    """
    folder = batches_root / date_str
    if not folder.exists():
        raise FileNotFoundError(f"Folder not found: {folder}")

    manifest: Dict[str, Dict[str, Any]] = {}
    for p in folder.rglob("*.parquet"):
        rel = p.relative_to(batches_root)
        manifest[sha256_file(p)] = {
            "path": str(rel).replace("\\", "/"),
            "size": p.stat().st_size,
            "prompt_col": "prompt",
            "response_col": "response",
        }
    return manifest

def build_cdn_filelist(repo: str, folder_path: str) -> List[str]:
    """
    Single API call: list_repo_tree(non-recursive) for one folder.
    Returns CDN URLs (no auth/rate-limit during training).
    """
    try:
        from huggingface_hub import list_repo_tree
    except ImportError as exc:
        raise RuntimeError("Install: pip install huggingface_hub") from exc

    items = list_repo_tree(repo=repo, path=folder_path, recursive=False)
    urls = []
    for item in items:
        if item.rfilename.endswith(".parquet"):
            urls.append(f"{HF_DATASETS_CDN}/{repo}/resolve/main/{item.rfilename}")
    return urls

def main() -> None:
    parser = argparse.ArgumentParser(description="Generate manifest + CDN filelist")
    parser.add_argument("--date", required=True, help="YYYY-MM-DD")
    parser.add_argument("--repo", required=True, help="org/dataset")
    parser.add_argument("--batches-root", default="batches", help="Local mirror root")
    parser.add_argument("--out-dir", default="ops/out", help="Output directory")
    args = parser.parse_args()

    out_dir = Path(args.out_dir)
    out_dir.mkdir(parents=True, exist_ok=True)

    # 1) Local manifest (content-addressed)
    manifest = build_manifest_local(Path(args.batches_root), args.date)
    manifest_path = out_dir / f"manifest-{args.date}.json"
    manifest_path.write_text(json.dumps(manifest, indent=2))
    print(f"Manifest: {manifest_path} ({len(manifest)} files)")

    # 2) CDN filelist (single API call)
    folder_path = f"batches/mirror-merged/{args.date}"
    try:
        urls = build_cdn_filelist(args.repo, folder_path)
    except Exception as exc:
        print(f"HF API call failed (possibly rate-limited): {exc}")
        print("Fallback: generate filelist from local manifest paths.")
        urls = [
            f"{HF_DATASETS_CDN}/{args.repo}/resolve/main/{m['path']}"
            for m in manifest.values()
        ]

    filelist_path = out_dir / f"train_filelist-{args.date}.json"
    filelist_path.write_text(json.dumps(urls, indent=2))
    print(f"CDN filelist: {filelist_path} ({len(urls)} URLs)")

    # 3) Small metadata for reproducibility
    meta = {
        "created_at": datetime.now(timezone.utc).isoformat(),
        "date": args.date,
        "repo": args.repo,
        "n_files": len(urls),
        "manifest": str(manifest_path),
    }
    (out_dir / f"meta-{args.date}.json").write_text(json.dumps(meta, indent=2))

if __name__ == "__main__":
    main()
```

import json
from pathlib import Path
from typing import Dict

import pyarrow.parquet as pq
import requests
from torch.utils.data import Dataset
from io import BytesIO

class CDNParquetDataset(Dataset):
    """
    Zero-API dataset that loads parquet files via public CDN URLs.
    Projects only prompt/response columns for schema resilience.
    """
    def __init__(self, filelist_path: str, prompt_col: str = "prompt", response_col: str = "response"):
        with open(filelist_path) as f:
            self.urls: list[str] = json.load(f)
        self.prompt_col = prompt_col
        self.response_col = response_col

    def __len__(self) -> int:
        return len(self.urls)

    def _project_row(self, table) -> Dict[str, str]:
        # Robust projection: if expected columns missing, try common alternatives
        prompt_col = self.prompt_col if self.prompt_col in table.column_names else "prompt"
        response_col = self.response_col if self.response_col in table.column_names else "response"
        prompts = table[prompt_col].to_pylist()
        responses = table[response_col].to_pylist()
        return {"prompt": prompts, "response": responses}

    def __getitem__(self, idx: int) -> Dict[str, list]:
        url = self.urls[idx]
