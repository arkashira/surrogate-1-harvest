## Incremental Improvement (≤2h)

**Title**: HF CDN-bypass dataset loader + schema-safe projection + Lightning Studio reuse guard  
**Scope**: `surrogate/` backend — training data pipeline only  
**Value**:  
- Eliminates HF API 429s during training by using CDN-only downloads (`resolve/main/`).  
- Prevents `pyarrow.CastError` from heterogeneous repo files by projecting to `{prompt,response}` at parse time.  
- Cuts Lightning quota burn by reusing running studios and guarding against idle-stop restarts.  

---

## Implementation Plan

1. **Add CDN-only file fetcher**  
   - Single API call (from Mac orchestrator) to `list_repo_tree(path, recursive=False)` for one date folder → save `file_list.json`.  
   - Embed `file_list.json` in training script; Lightning training uses only CDN URLs (no Authorization, no `/api/` rate limits).  

2. **Schema-safe loader**  
   - Download each file via `hf_hub_download` (or raw CDN) → parse with best-effort JSON/Parquet → project to `{prompt, response}`.  
   - Skip files that can’t be projected; log and continue.  

3. **Lightning Studio reuse + idle-stop guard**  
   - Before `.run()`, list `Teamspace.studios`; reuse running studio with matching name.  
   - If studio status is stopped, restart with `target.start(machine=Machine.L40S)` (or fallback to L40S if H200 unavailable).  

4. **Integrate into training entrypoint**  
   - Expose CLI: `python train.py --file-list file_list.json --studio-name surrogate-1-l40s`.

# generate_file_list.py
import json
import os
from huggingface_hub import HfApi

HF_REPO = "axentx/surrogate-1-dataset"
OUT_PATH = "file_list.json"
FOLDER = "batches/mirror-merged/2026-05-03"  # one date folder

api = HfApi()
tree = api.list_repo_tree(repo_id=HF_REPO, path=FOLDER, recursive=False)
files = [f.rfilename for f in tree if f.rfilename.endswith((".json", ".jsonl", ".parquet"))]

with open(OUT_PATH, "w") as f:
    json.dump({"repo": HF_REPO, "folder": FOLDER, "files": files}, f, indent=2)

print(f"Wrote {len(files)} files to {OUT_PATH}")

### 2. CDN-bypass dataset loader (schema-safe)

```python
# surrogate/data/cdn_loader.py
import json
import logging
from pathlib import Path
from typing import Iterator, Dict, Any

import pyarrow.parquet as pq
import requests
from huggingface_hub import hf_hub_download

logger = logging.getLogger(__name__)

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def project_to_prompt_response(obj: Dict[str, Any]) -> Dict[str, str]:
    """Best-effort projection to {prompt, response}. Returns empty on failure."""
    prompt = obj.get("prompt") or obj.get("input") or obj.get("question")
    response = obj.get("response") or obj.get("output") or obj.get("answer")
    if not prompt or not response:
        return {}
    return {"prompt": str(prompt), "response": str(response)}

def cdn_download(repo: str, rfilename: str, cache_dir: str = ".cache") -> Path:
    """Download via CDN (no auth) or fallback to hf_hub_download."""
    cache_dir = Path(cache_dir)
    cache_dir.mkdir(parents=True, exist_ok=True)
    out_path = cache_dir / Path(rfilename).name
    if out_path.exists():
        return out_path

    url = CDN_TEMPLATE.format(repo=repo, path=rfilename)
    try:
        r = requests.get(url, timeout=30)
        r.raise_for_status()
        out_path.write_bytes(r.content)
        logger.info("CDN downloaded %s", rfilename)
        return out_path
    except Exception as e:
        logger.warning("CDN failed for %s: %s; falling back to hf_hub_download", rfilename, e)
        return Path(hf_hub_download(repo_id=repo, filename=rfilename, cache_dir=str(cache_dir)))

def load_file(path: Path) -> Iterator[Dict[str, str]]:
    """Load JSON/JSONL/Parquet and project to {prompt, response}."""
    suffix = path.suffix.lower()
    try:
        if suffix == ".parquet":
            table = pq.read_table(path)
            for col in table.column_names:
                if "prompt" in col.lower() or "response" in col.lower():
                    # attempt projection-friendly read
                    df = table.select([col for col in table.column_names]).to_pandas()
                    for _, row in df.iterrows():
                        proj = project_to_prompt_response(row.to_dict())
                        if proj:
                            yield proj
                    return
            # fallback: try generic dict rows
            df = table.to_pandas()
            for _, row in df.iterrows():
                proj = project_to_prompt_response(row.to_dict())
                if proj:
                    yield proj
        else:
            # JSON/JSONL
            text = path.read_text(encoding="utf-8")
            lines = [ln.strip() for ln in text.splitlines() if ln.strip()]
            for ln in lines:
                try:
                    obj = json.loads(ln)
                except Exception:
                    # maybe whole file is one JSON array
                    if ln.startswith("["):
                        objs = json.loads(ln)
                        for obj in objs:
                            proj = project_to_prompt_response(obj)
                            if proj:
                                yield proj
                        return
                    continue
                proj = project_to_prompt_response(obj)
                if proj:
                    yield proj
    except Exception as e:
        logger.error("Failed to load %s: %s", path, e, exc_info=True)

def load_dataset_from_manifest(manifest_path: str, cache_dir: str = ".cache") -> Iterator[Dict[str, str]]:
    manifest = json.loads(Path(manifest_path).read_text())
    repo = manifest["repo"]
    files = manifest["files"]
    for rfn in files:
        local_path = cdn_download(repo, rfn, cache_dir=cache_dir)
        try:
            for item in load_file(local_path):
                yield item
        except Exception as e:
            logger.error("Skipping %s after load error: %s", rfn, e)
            continue
```

import logging

logger = logging.getLogger(__name__)

class SurrogateTrainer(LightningWork):
    def __init__(self, file_list_path: str, studio_name: str = "surrogate-1-l40s", **kwargs):
        super().__init__(**kwargs)
        self.file_list_path = file_list_path
        self.studio_name = studio_name
        self._studio = None

    def run(self):
        # This runs inside the Lightning Work (remote)
        from surrogate.train.trainer import train
        train(file_list_path=self.file_list_path)

    def configure(self) -> Machine:
        # Prefer L40S (free tier); fallback to other available machines
        return Machine(name="L40S")
