# workio / backend

## Final synthesized solution (correct + actionable)

### 1. Diagnosis (merged, prioritized)
- **Missing idempotency key** on punch records → LINE retries and concurrent deliveries create duplicates.
- **Non-atomic read-then-insert** for punch state → races cause invalid transitions (double clock-in, lost clock-out).
- **No database-level uniqueness guard** for tenant+employee active punch and for idempotency → application checks are unsafe under concurrency.
- **Webhook handler does not deduplicate** already-processed events → retries re-apply side effects (notifications, state changes).
- **No stable request identifier persisted** → no reliable replay detection across retries.

### 2. Scope
- `server/src/db/schema.sql` — add idempotency column and active-punch uniqueness constraint.
- `server/src/services/punch.ts` — atomic, idempotent state transition with upsert.
- `server/src/routes/line.ts` — verify signature, derive/stabilize idempotency key, early return for duplicates.

### 3. Implementation

#### 3.1 DB schema (add idempotency + enforce one active punch)
```sql
-- server/src/db/schema.sql
-- Idempotency and processing metadata
ALTER TABLE punches
  ADD COLUMN IF NOT EXISTS idempotency_key VARCHAR(255) NULL,
  ADD COLUMN IF NOT EXISTS processed_at TIMESTAMPTZ NULL;

-- One idempotent write per tenant+idempotency_key
CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_idempotency_key
  ON punches (tenant_id, idempotency_key)
  WHERE idempotency_key IS NOT NULL;

-- At most one active (unprocessed) punch per tenant+employee.
-- processed_at IS NULL means "this is the current open punch".
-- On clock-out we set processed_at = now() on the previously active row.
CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_one_active
  ON punches (tenant_id, employee_id)
  WHERE processed_at IS NULL;
```

#### 3.2 Service — atomic, idempotent transition
```ts
// server/src/services/punch.ts
import { Pool } from 'pg';

const pool = new Pool({ connectionString: process.env.DATABASE_URL });

export type PunchType = 'CLOCK_IN' | 'CLOCK_OUT';

/**
 * Apply a punch event exactly once under retries/concurrency.
 * Returns the active punch row after handling this event.
 */
export async function applyPunch(params: {
  tenantId: string;
  employeeId: string;
  type: PunchType;
  idempotencyKey: string;
  location?: { lat: number; lng: number } | null;
  timestamp?: Date;
}): Promise<any> {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // 1) Try to claim this idempotency key first (safe under concurrency)
    const idempotencyInsert = `
      INSERT INTO punches (tenant_id, employee_id, idempotency_key, type, location, created_at, processed_at)
      VALUES ($1, $2, $3, $4, $5, $6, $6)
      ON CONFLICT (tenant_id, idempotency_key) DO NOTHING
      RETURNING *
    `;
    const ts = params.timestamp || new Date();
    const loc = params.location ? JSON.stringify(params.location) : null;
    const { rows: inserted } = await client.query(idempotencyInsert, [
      params.tenantId,
      params.employeeId,
      params.idempotencyKey,
      params.type,
      loc,
      ts,
    ]);

    if (inserted.length > 0) {
      // First time seeing this idempotency key — perform state transition atomically
      await applyStateTransition(client, params.tenantId, params.employeeId, params.type, ts, loc);
      await client.query('COMMIT');
      return await getActivePunch(client, params.tenantId, params.employeeId);
    }

    // 2) Already processed — return current active state
    const active = await getActivePunch(client, params.tenantId, params.employeeId);
    await client.query('COMMIT');
    return active;
  } catch (err: any) {
    await client.query('ROLLBACK');
    // Surface concurrency conflict explicitly for callers/retry logic
    if (err.code === '23505' && err.constraint && err.constraint.includes('idx_punches_one_active')) {
      throw new Error('CONCURRENT_ACTIVE_PUNCH_CONFLICT');
    }
    throw err;
  } finally {
    client.release();
  }
}

async function applyStateTransition(
  client: any,
  tenantId: string,
  employeeId: string,
  type: PunchType,
  ts: Date,
  location: string | null
) {
  // Close any existing active punch (set processed_at)
  const closePrevious = `
    UPDATE punches
    SET processed_at = $1
    WHERE tenant_id = $2 AND employee_id = $3 AND processed_at IS NULL
  `;
  await client.query(closePrevious, [ts, tenantId, employeeId]);

  // If CLOCK_IN, open a new active punch (processed_at = NULL)
  if (type === 'CLOCK_IN') {
    const insertActive = `
      INSERT INTO punches (tenant_id, employee_id, type, location, created_at, processed_at)
      VALUES ($1, $2, $3, $4, $5, NULL)
    `;
    await client.query(insertActive, [tenantId, employeeId, type, location, ts]);
  }
}

async function getActivePunch(client: any, tenantId: string, employeeId: string) {
  const { rows } = await client.query(
    `SELECT * FROM punches WHERE tenant_id = $1 AND employee_id = $2 AND processed_at IS NULL ORDER BY created_at DESC LIMIT 1`,
    [tenantId, employeeId]
  );
  return rows[0] || null;
}
```
