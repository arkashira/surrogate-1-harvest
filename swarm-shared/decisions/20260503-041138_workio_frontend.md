# workio / frontend

## Final Synthesis (single, actionable plan)

### 1. Diagnosis (merged, de-duplicated)
- **Double-tap & perceived latency**: no optimistic/pending UI; users retry before idempotency settles.
- **Location failures**: no inline guard or recovery for missing/denied GPS; failures are silent.
- **Stale state**: clock/shift state fetched only on mount; background/foreground or server-side rule changes produce invalid attempts.
- **Duplicate requests**: no client idempotency key; retries or rapid taps can create duplicates while prior requests are in-flight.
- **Unclear progress**: no visual affordance for “recording location” / “submitting” / “retrying”.

### 2. Scope (unchanged, clarified)
- Frontend-only changes; server contract unchanged (idempotency header optional).
- Files:
  - `workio/src/lib/api.ts`
  - `workio/src/features/clock/useClock.ts`
  - `workio/src/features/clock/ClockButton.tsx`
  - `workio/src/features/clock/ClockStatus.tsx`

### 3. Implementation (merged best parts, resolved contradictions)

```ts
import { api } from './api'; // assumed axios/ky-like instance

export const clockIn = async (lat: number, lng: number, idempotencyKey?: string) => {
  const res = await api.post('/clock/in', { lat, lng }, {
    headers: idempotencyKey ? { 'Idempotency-Key': idempotencyKey } : undefined,
  });
  return res.data;
};

export const clockOut = async (lat: number, lng: number, idempotencyKey?: string) => {
  const res = await api.post('/clock/out', { lat, lng }, {
    headers: idempotencyKey ? { 'Idempotency-Key': idempotencyKey } : undefined,
  });
  return res.data;
};
```

#### `workio/src/features/clock/useClock.ts`
```ts
import { useState, useEffect, useCallback, useRef } from 'react';
import { clockIn, clockOut } from '@/lib/api';

type Punch = { type: 'IN' | 'OUT'; ts: string; lat: number; lng: number };
type ClockState = {
  lastPunch: Punch | null;
  canPunch: boolean;
  nextAllowedAt: number | null; // epoch ms
  isPending: boolean;
  isRetrying: boolean;
  error: string | null;
  locationError: string | null;
};

const COOLDOWN_MS = 5 * 60 * 1000; // optimistic cooldown (server may override)
const LOCATION_MAX_AGE = 30_000;
const POLL_INTERVAL = 30_000;

export function useClock() {
  const [state, setState] = useState<ClockState>({
    lastPunch: null,
    canPunch: true,
    nextAllowedAt: null,
    isPending: false,
    isRetrying: false,
    error: null,
    locationError: null,
  });

  const mountedRef = useRef(true);
  useEffect(() => {
    mountedRef.current = true;
    return () => { mountedRef.current = false; };
  }, []);

  const fetchStatus = useCallback(async () => {
    try {
      const res = await fetch('/api/clock/status');
      if (!res.ok) throw new Error('Status fetch failed');
      const data = await res.json();
      if (!mountedRef.current) return;
      setState((s) => ({
        ...s,
        lastPunch: data.lastPunch || null,
        canPunch: Boolean(data.canPunch),
        nextAllowedAt: data.nextAllowedAt ? new Date(data.nextAllowedAt).getTime() : null,
        // preserve pending/retry UI flags; only clear error if fetch succeeded
        error: null,
      }));
    } catch {
      // keep existing state; surface no error for background poll failures
    }
  }, []);

  useEffect(() => {
    fetchStatus();
    const id = setInterval(fetchStatus, POLL_INTERVAL);
    return () => clearInterval(id);
  }, [fetchStatus]);

  const getLocation = useCallback(async (): Promise<{ lat: number; lng: number }> => {
    if (!navigator.geolocation) {
      throw new Error('Geolocation not supported');
    }

    return new Promise((resolve, reject) => {
      navigator.geolocation.getCurrentPosition(
        (pos) => {
          resolve({ lat: pos.coords.latitude, lng: pos.coords.longitude });
        },
        (err) => {
          switch (err.code) {
            case err.PERMISSION_DENIED:
              reject(new Error('Location permission denied'));
              break;
            case err.POSITION_UNAVAILABLE:
              reject(new Error('Location unavailable'));
              break;
            case err.TIMEOUT:
              reject(new Error('Location request timed out'));
              break;
            default:
              reject(new Error('Failed to get location'));
          }
        },
        { enableHighAccuracy: true, timeout: 10_000, maximumAge: LOCATION_MAX_AGE }
      );
    });
  }, []);

  const requestPunch = useCallback(
    async (type: 'IN' | 'OUT') => {
      if (!state.canPunch || state.isPending) return;

      setState((s) => ({ ...s, isPending: true, error: null, locationError: null }));

      let loc: { lat: number; lng: number };
      try {
        loc = await getLocation();
      } catch (err: any) {
        setState((s) => ({
          ...s,
          isPending: false,
          locationError: err.message || 'Location unavailable',
          error: err.message || 'Location unavailable',
        }));
        return;
      }

      const idempotencyKey = `${type}-${Date.now()}-${crypto.randomUUID()}`;

      try {
        if (type === 'IN') await clockIn(loc.lat, loc.lng, idempotencyKey);
        else await clockOut(loc.lat, loc.lng, idempotencyKey);

        const now = new Date().toISOString();
        const newPunch: Punch = { type, ts: now, lat: loc.lat, lng: loc.lng };
        setState((s) => ({
          ...s,
          lastPunch: newPunch,
          canPunch: false,
          nextAllowedAt: Date.now() + COOLDOWN_MS,
          isPending: false,
          error: null,
        }));
        // reconcile with server shortly
        setTimeout(fetchStatus, 2_000);
      } catch (err: any) {
        // retry once automatically for transient network/server errors
        setState((s) => ({ ...s, isPending: false, isRetrying: true, error: null }));
        try {
          if (type === 'IN') await clockIn(loc.lat, loc.lng, `${idempotencyKey}-retry`);
          else await clockOut(loc.lat, loc.lng, `${idempotencyKey}-retry`);

          const now = new Date().toISOString();
          const newPunch: Punch = { type, ts: now, lat: loc.lat, lng: loc.lng };
          setState((s) => ({
            ...s,
            lastPunch: newPunch,
            canPunch: false,
            nextAllowedAt: Date.now() + COOLDOWN_MS,
            isRetrying: false,
            error: null,
          }));
          setTimeout(fetchStatus, 2_000);
        } catch (retryErr: any) {
          setState((s) => ({
            ...s,
            isRetrying: false,
            error: retryErr.message || 'Failed to record punch after retry',
          }));
          setTimeout(fetchStatus, 3_000);
        }
      }
    },
    [state.canPunch, state.isPending, getLocation, fetchStatus]
  );

  return { ...state, requestPunch, refresh: fetchStatus };
}
```
