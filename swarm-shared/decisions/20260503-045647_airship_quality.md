### 1. tools/build_manifest.py
```python
#!/usr/bin/env python3
"""
Build a CDN-only manifest for Surrogate-1 training.
Run from Mac orchestrator after rate-limit window clears.
"""
import json, os, sys, hashlib
from datetime import datetime, timezone
from pathlib import Path

try:
    from huggingface_hub import HfApi
except ImportError:
    print("pip install huggingface_hub")
    sys.exit(1)

REPO = "axentx/surrogate-1-ingest"
DATE_FOLDER = datetime.now(timezone.utc).strftime("%Y-%m-%d")
OUT_DIR = Path(__file__).parent.parent / "manifests" / DATE_FOLDER
OUT_DIR.mkdir(parents=True, exist_ok=True)

api = HfApi()

def build_manifest() -> dict:
    # Non-recursive per folder to avoid pagination explosion
    entries = api.list_repo_tree(REPO, path=DATE_FOLDER, recursive=False)
    files = []
    for e in entries:
        if getattr(e, "type", None) != "file":
            continue
        path = e.path
        # CDN URL (no auth, no API rate limit)
        cdn_url = f"https://huggingface.co/datasets/{REPO}/resolve/main/{path}"
        files.append({
            "path": path,
            "cdn_url": cdn_url,
            "size": getattr(e, "size", 0),
        })

    manifest = {
        "repo": REPO,
        "date": DATE_FOLDER,
        "generated_utc": datetime.now(timezone.utc).isoformat(),
        "files": files,
        "total_files": len(files),
    }
    out_path = OUT_DIR / "file_manifest.json"
    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"Manifest written: {out_path} ({len(files)} files)")
    return manifest

if __name__ == "__main__":
    build_manifest()
```

### 2. surrogate/data/cdn_dataset.py
```python
import json, io, logging
from pathlib import Path
from typing import Optional, Dict, Any, Iterator

import torch
from torch.utils.data import IterableDataset
import pyarrow as pa
import pyarrow.parquet as pq
import requests

logger = logging.getLogger(__name__)

class CDNParquetDataset(IterableDataset):
    """
    Zero HF API during training.
    Reads manifest produced by tools/build_manifest.py and streams
    parquet files directly from HF CDN.
    Projects only {prompt, response} at parse time.
    """
    def __init__(self, manifest_path: str | Path, split: str = "train", start: int = 0, end: Optional[int] = None):
        manifest_path = Path(manifest_path)
        if not manifest_path.exists():
            raise FileNotFoundError(f"Manifest not found: {manifest_path}")
        manifest = json.loads(manifest_path.read_text())
        self.files = [f["cdn_url"] for f in manifest["files"] if f["path"].endswith(".parquet")]
        if not self.files:
            raise ValueError("No parquet files in manifest")
        self.split = split
        self.start = start
        self.end = end if end is not None else len(self.files)
        self.total_files = len(self.files)

    def _stream_file(self, url: str) -> Iterator[Dict[str, Any]]:
        try:
            resp = requests.get(url, timeout=30)
            resp.raise_for_status()
            buf = io.BytesIO(resp.content)
            table = pq.read_table(buf, columns=["prompt", "response"])
            df = table.to_pandas()
            for _, row in df.iterrows():
                prompt = str(row.get("prompt", ""))
                response = str(row.get("response", ""))
                if prompt.strip() and response.strip():
                    yield {"prompt": prompt, "response": response}
        except Exception as exc:
            logger.warning("Failed to stream %s: %s", url, exc)

    def __iter__(self) -> Iterator[Dict[str, Any]]:
        worker_info = torch.utils.data.get_worker_info()
        if worker_info is None:
            iter_files = range(self.start, self.end)
        else:
            per_worker = (self.end - self.start) // worker_info.num_workers
            worker_id = worker_info.id
            iter_files = range(
                self.start + worker_id * per_worker,
                self.start + (worker_id + 1) * per_worker,
            )

        for idx in iter_files:
            if idx >= self.total_files:
                break
            url = self.files[idx]
            yield from self._stream_file(url)
```
