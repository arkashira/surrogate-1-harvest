### 1) One-shot CDN manifest builder
`scripts/build_training_manifest.py`

- Single HF API call to list a date folder, then emits `manifest.json` with CDN URLs.  
- Validates HF 429 and exits fast so CI/operators can retry with backoff.  
- Optional `--parallel` for large folders (kept simple by default).

```python
#!/usr/bin/env python3
"""
Build a CDN-only training manifest for Surrogate training.
Usage:
  python scripts/build_training_manifest.py \
    --repo datasets/axentx/surrogate-mirror \
    --date 2026-05-03 \
    --out surrogate/training/manifest.json
"""
import argparse
import json
import os
import sys
from datetime import datetime, timezone

import requests

HF_API_BASE = "https://huggingface.co/api"
CDN_BASE = "https://huggingface.co/datasets"

def list_date_folder(repo: str, date_folder: str):
    url = f"{HF_API_BASE}/datasets/{repo}/tree"
    params = {"path": date_folder, "recursive": "false"}
    resp = requests.get(url, params=params, timeout=30)
    if resp.status_code == 429:
        print("HF API 429 — wait 360s before retry", file=sys.stderr)
        sys.exit(1)
    resp.raise_for_status()
    return resp.json()

def build_manifest(repo: str, date_folder: str):
    entries = list_date_folder(repo, date_folder)
    files = []
    for item in entries:
        if item.get("type") != "file":
            continue
        path = item["path"]
        cdn_url = f"{CDN_BASE}/{repo}/resolve/main/{path}"
        files.append(
            {
                "repo": repo,
                "path": path,
                "cdn_url": cdn_url,
                "size": item.get("size"),
                "lfs": item.get("lfs", {}),
            }
        )
    manifest = {
        "repo": repo,
        "date_folder": date_folder,
        "generated_at": datetime.now(timezone.utc).isoformat(),
        "files": files,
    }
    return manifest

def main():
    parser = argparse.ArgumentParser(description="Build CDN-only training manifest")
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g. datasets/axentx/surrogate-mirror)")
    parser.add_argument("--date", required=True, help="Date folder (e.g. 2026-05-03)")
    parser.add_argument("--out", required=True, help="Output manifest JSON path")
    args = parser.parse_args()

    manifest = build_manifest(args.repo, args.date)
    os.makedirs(os.path.dirname(os.path.abspath(args.out)), exist_ok=True)
    with open(args.out, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Manifest written to {args.out} ({len(manifest['files'])} files)")

if __name__ == "__main__":
    main()
```

Make executable:
```bash
chmod +x scripts/build_training_manifest.py
```

### 2) CDN-only, resumable dataset loader
`surrogate/training/cdn_dataset.py`

- Downloads via CDN only (no HF auth/rate-limits during training).  
- Uses deterministic temp-name + size/ETag-lite validation to avoid re-downloads on resume.  
- Projects to `{prompt, response}` at parse time; skips corrupt parquet safely.  
- Supports resharding/shard index for deterministic epoch boundaries (important for resume).

```python
import json
import logging
import os
import tempfile
from pathlib import Path
from typing import Dict, Iterator, Tuple

import pyarrow as pa
import pyarrow.parquet as pq
import requests
from torch.utils.data import IterableDataset

logger = logging.getLogger(__name__)

class CDNParquetDataset(IterableDataset):
    """
    HF-CDN-only dataset loader.
    Reads manifest.json built by scripts/build_training_manifest.py.
    Downloads each parquet via CDN URL (no HF API/auth/rate-limit during training).
    Projects to {prompt, response} only at parse time.
    """

    def __init__(
        self,
        manifest_path: str,
        cache_dir: str = ".cdn_cache",
        shard_idx: int = 0,
        num_shards: int = 1,
    ):
        super().__init__()
        self.manifest_path = manifest_path
        self.cache_dir = Path(cache_dir)
        self.cache_dir.mkdir(parents=True, exist_ok=True)
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.files = self.manifest["files"]
        if not self.files:
            raise ValueError("No files found in manifest")
        # Deterministic sharding for multi-worker / resume
        self.shard_idx = shard_idx
        self.num_shards = num_shards
        self.files = [f for i, f in enumerate(self.files) if i % self.num_shards == self.shard_idx]

    def _download_cdn(self, cdn_url: str) -> Path:
        """Download via CDN; use size for simple cache validation."""
        safe_name = cdn_url.replace("://", "_").replace("/", "_")
        out_path = self.cache_dir / safe_name
        if out_path.exists() and out_path.stat().st_size > 0:
            return out_path

        logger.info("Downloading CDN %s", cdn_url)
        resp = requests.get(cdn_url, stream=True, timeout=120)
        resp.raise_for_status()
        with tempfile.NamedTemporaryFile(dir=self.cache_dir, delete=False) as tmp:
            for chunk in resp.iter_content(chunk_size=8192):
                tmp.write(chunk)
            tmp_path = tmp.name
        os.replace(tmp_path, out_path)
        return out_path

    def _iter_parquet(self, file_info: Dict) -> Iterator[Tuple[str, str]]:
        cdn_url = file_info["cdn_url"]
        local_path = self._download_cdn(cdn_url)
        try:
            table = pq.read_table(local_path, columns=["prompt", "response"])
        except pa.ArrowInvalid:
            logger.warning("Invalid parquet %s, skipping", cdn_url)
            return
        df = table.to_pandas()
        for _, row in df.iterrows():
            prompt = row.get("prompt")
            response = row.get("response")
            if isinstance(prompt, str) and isinstance(response, str):
                yield prompt, response

    def __iter__(self) -> Iterator[Tuple[str, str]]:
        for fi in self.files:
            yield from self._iter_parquet(fi)
```
