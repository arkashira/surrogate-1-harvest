## 1. Diagnosis (merged)

- No database-level uniqueness constraint → duplicates `(employee_id, punch_date, punch_type)` can be created by retries, double-taps, or concurrent webhook deliveries.
- Application logic is non-atomic read-then-insert → race condition under concurrency.
- No idempotency key stored → safe replay / deduplication is impossible at DB level.
- Observability gaps: no request/trace ID, no structured logs, no metrics (latency/duplicates/errors).
- Missing safe retry/backoff for outbound LINE API calls (notifications) can leave users without confirmation.

---

## 2. Concrete changes (scope + priority)

**File:** `/opt/axentx/workio/server/src/routes/line/webhook.ts` (or equivalent)  
**DB migration:** add partial unique constraint and idempotency column/index.  
**Runtime:** atomic upsert, deterministic idempotency key, structured logging, request tracing, safe retry for LINE client calls.

### 3.2 Route handler (atomic, idempotent, observable)

```ts
// /opt/axentx/workio/server/src/routes/line/webhook.ts
import { Router, Request, Response } from 'express';
import { v4 as uuidv4 } from 'uuid';
import crypto from 'crypto';
import { db } from '../db';
import { logger } from '../lib/logger';
import { retryBackoff } from '../lib/retry';
import type { Client } from '@line/bot-sdk';

const router = Router();

function hashBody(body: unknown): string {
  return crypto.createHash('sha256').update(JSON.stringify(body)).digest('hex');
}

function idempotencyKey(event: any, retryId?: string): string {
  // Prefer LINE retry id when present; otherwise deterministic content key
  if (retryId) return `line-retry:${retryId}`;
  const ts = event.timestamp || Date.now();
  return `line-event:${ts}:${hashBody(event)}`;
}

function parsePunchType(text: string): 'in' | 'out' {
  return text.includes('ออก') ? 'out' : 'in';
}

export function createLineWebhookHandler(lineClient: Client) {
  return async (req: Request, res: Response) => {
    const requestId = req.header('x-request-id') || uuidv4();
    const log = (msg: string, extra?: Record<string, unknown>) =>
      logger.info({ requestId, route: 'line/webhook', msg, ...extra });

    const start = Date.now();
    const retryId = req.header('x-line-retry-id');

    try {
      const events = req.body?.events;
      if (!events?.length) {
        log('no_events');
        res.setHeader('x-request-id', requestId);
        return res.sendStatus(200);
      }

      // Process first valid punch event in webhook (common LINE pattern)
      const event = events[0];
      const { userId } = event.source || {};
      const text = (event.message?.text || '').trim();
      if (!userId || !text) {
        log('invalid_event', { event });
        res.setHeader('x-request-id', requestId);
        return res.sendStatus(200);
      }

      const punchType = parsePunchType(text);
      const punchedAt = new Date(event.timestamp || Date.now());
      const punchDate = punchedAt.toISOString().slice(0, 10); // YYYY-MM-DD
      const key = idempotencyKey(event, retryId);

      // Atomic upsert using DB unique constraint
      const result = await db.query(
        `INSERT INTO punches (employee_id, punch_date, punch_type, punched_at, idempotency_key)
         SELECT id, $1, $2, $3, $4
         FROM employees
         WHERE line_user_id = $5 AND deleted_at IS NULL
         ON CONFLICT (employee_id, punch_date, punch_type)
         WHERE deleted_at IS NULL
         DO NOTHING
         RETURNING id`,
        [punchDate, punchType, punchedAt, key, userId]
      );

      const inserted = !!(result.rowCount && result.rowCount > 0);
      if (inserted) {
        log('punch_inserted', { employee_line_id: userId, punchDate, punchType });
      } else {
        log('duplicate_ignored', { employee_line_id: userId, punchDate, punchType });
      }

      // Optional: notify user via LINE (safe retry)
      const notify = async () => {
        const msg = {
          type: 'text',
          text: punchType === 'in' ? 'เข้างานเรียบร้อยแล้ว' : 'เลิกงานเรียบร้อยแล้ว',
        };
        await lineClient.pushMessage(userId, msg);
      };

      // Fire-and-forget notification with safe retry; don't block webhook response
      retryBackoff(notify, { maxAttempts: 3, baseMs: 500 }).catch((err) => {
        log('line_notify_failed', { employee_line_id: userId, error: String(err) });
      });

      res.setHeader('x-request-id', requestId);
      return res.sendStatus(200);
    } catch (err) {
      log('error', { error: String(err) });
      res.setHeader('x-request-id', requestId);
      return res.status(500).json({ error: 'internal_error' });
    } finally {
      logger.info({
        requestId,
        route: 'line/webhook',
        duration_ms: Date.now() - start,
      });
    }
  };
}

router.post('/webhook/line', createLineWebhookHandler({} as Client)); // inject real client at composition root

export default router;
```

// /opt/axentx/workio/server/src/lib/retry.ts
export async function retryBackoff<T>(
  fn: () => Promise<T>,
  { maxAttempts = 3, baseMs = 500 }: { maxAttempts?: number; baseMs?: number } = {}
): Promise<T> {
  let lastErr: unknown;
  for (let i = 0; i < maxAttempts; i++) {
    try {
      return await fn();
    } catch (err) {
      lastErr = err;
      if (i < maxAttempts - 1) {
        await new Promise((r) => setTimeout(r, baseMs * 2 ** i));
      }
    }
  }
  throw lastErr;
}
