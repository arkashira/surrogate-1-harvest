### 1) Library: `/opt/axentx/airship/lib/airship-discover.sh`

```bash
#!/usr/bin/env bash
# airship-discover library
# Usage: source this file; call airship_discover_main "$@"

set -euo pipefail

readonly AIRSHIP_ROOT="/opt/axentx/airship"
readonly LOG_FILE="${AIRSHIP_ROOT}/logs/discover.log"
readonly TIMESTAMP="$(date -u +"%Y-%m-%dT%H:%M:%SZ")"

# Ensure log directory exists
mkdir -p "$(dirname "${LOG_FILE}")"

log() {
  local level="$1"
  shift
  echo "${TIMESTAMP} [${level}] airship-discover: $*" | tee -a "${LOG_FILE}"
}

run_granite_business_research() {
  local script="${AIRSHIP_ROOT}/scripts/granite-business-research.sh"
  if [[ -x "${script}" ]]; then
    log INFO "Running granite-business-research.sh"
    if "${script}"; then
      log INFO "granite-business-research.sh completed"
      # Tag outputs in-place where possible (example: append tags to result files)
      find "${AIRSHIP_ROOT}/outputs" -type f -newer "${script}" -exec \
        sh -c 'echo "#business-research #knowledge-rag #graph" >> "$1"' _ {} \; || true
    else
      log WARN "granite-business-research.sh exited non-zero (continuing)"
    fi
  else
    log INFO "granite-business-research.sh not found or not executable — skipping"
  fi
}

run_knowledge_rag_top_hub() {
  local rag_cmd="${AIRSHIP_ROOT}/bin/knowledge-rag"
  if [[ -x "${rag_cmd}" ]]; then
    log INFO "Querying top hub via knowledge-rag"
    # Query top hub (e.g., MOC) and emit tagged insights
    if "${rag_cmd}" --query "top hub" --format json 2>/dev/null | tee -a "${LOG_FILE}" | grep -q .; then
      log INFO "knowledge-rag top-hub query completed"
    else
      # Fallback: generic top-hub prompt
      "${rag_cmd}" --query "most connected hub" --format text 2>/dev/null | tee -a "${LOG_FILE}" || true
      log INFO "knowledge-rag fallback query completed"
    fi
  else
    log WARN "knowledge-rag executable not found — skipping"
  fi
}

airship_discover_main() {
  log INFO "Starting airship-discover"
  run_granite_business_research
  run_knowledge_rag_top_hub
  log INFO "Finished airship-discover"
}
```

### 2) Orchestrator: `/opt/axentx/airship/bin/airship-discover`

```bash
#!/usr/bin/env bash
# airship-discover — cron-safe orchestrator
# Invoke via bash to ensure consistent behavior in cron.

set -euo pipefail

readonly AIRSHIP_ROOT="/opt/axentx/airship"
readonly LIB="${AIRSHIP_ROOT}/lib/airship-discover.sh"

# Ensure we run with bash
if [[ -z "${BASH_VERSION:-}" ]]; then
  exec /bin/bash "$0" "$@"
fi

# Source library
if [[ -f "${LIB}" ]]; then
  # shellcheck source=/dev/null
  source "${LIB}"
else
  echo "ERROR: library not found: ${LIB}" >&2
  exit 1
fi

# Main
airship_discover_main "$@"
```

Make executable:
```bash
chmod +x /opt/axentx/airship/bin/airship-discover
chmod +x /opt/axentx/airship/lib/airship-discover.sh
```
