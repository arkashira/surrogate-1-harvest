# airship / backend

## Highest-Value Incremental (<2h)

**Chosen improvement**: Implement `airship discover` backend CLI that produces a deterministic CDN-only file manifest for a given HuggingFace `repo_id` and `date_folder`, and exposes a tiny HTTP endpoint (`GET /discover/manifest`) so the frontend `/discover` page can fetch and render it + offer “Copy CDN train script”.

Why this ships fast and unlocks training:
- Single-purpose, <200 LOC.
- Avoids HF API rate limits by using CDN URLs and a single `list_repo_tree` snapshot.
- Produces the exact file list the training script needs to do zero-API CDN fetches on Lightning.
- Enables immediate frontend integration (already planned).

---

## Implementation Plan

1. Add CLI entrypoint: `python -m airship.backend.discover` (or `airship discover` via `__main__`).
2. Accept args: `--repo_id`, `--date_folder`, `--out` (default `./manifest.json`).
3. Use HF API **once** to call `list_repo_tree(repo_id, path=date_folder, recursive=False)`.
4. Build manifest entries:
   - `cdn_url`: `https://huggingface.co/datasets/{repo_id}/resolve/main/{date_folder}/{file}`
   - `path`, `size`, `sha256` (if available).
5. Write deterministic JSON (sorted keys, sorted entries).
6. Add FastAPI route `GET /discover/manifest?repo_id=...&date_folder=...` that runs the same logic (cached 5m) and returns manifest.
7. Add minimal OpenAPI docs for the endpoint.
8. Include a `cdn_train_script.sh` template in response metadata so frontend can offer “Copy”.

```python
from __future__ import annotations

import argparse
import hashlib
import json
import os
import time
from dataclasses import asdict, dataclass
from datetime import datetime, timezone
from pathlib import Path
from typing import Any, Dict, List

import httpx
from huggingface_hub import HfApi, HfFolder

CDN_BASE = "https://huggingface.co/datasets"
API_BASE = "https://huggingface.co/api"

# Single API call; cache token locally to avoid repeated lookups.
def _get_token() -> str | None:
    try:
        return HfFolder.get_token()
    except Exception:
        return os.getenv("HF_TOKEN") or os.getenv("HUGGINGFACE_HUB_TOKEN")

@dataclass(frozen=True, order=True)
class FileEntry:
    path: str            # relative to repo root, e.g. "2026-04-29/file.parquet"
    cdn_url: str
    size: int | None = None
    sha256: str | None = None
    etag: str | None = None
    last_modified: str | None = None  # ISO UTC

    def to_dict(self) -> Dict[str, Any]:
        d = asdict(self)
        # Remove None for compactness
        return {k: v for k, v in d.items() if v is not None}

def list_repo_tree_cdn(repo_id: str, date_folder: str) -> List[FileEntry]:
    """
    Single HF API call to list files in date_folder (non-recursive).
    Returns CDN-capable entries.
    """
    api = HfApi(token=_get_token())
    # recursive=False keeps pagination minimal and avoids heavy list_repo_files.
    files = api.list_repo_tree(repo_id=repo_id, path=date_folder, recursive=False)

    entries: List[FileEntry] = []
    for f in files:
        if not f.rfilename:
            continue
        # f.rfilename is relative to repo root.
        cdn_url = f"{CDN_BASE}/{repo_id}/resolve/main/{f.rfilename}"
        entries.append(
            FileEntry(
                path=f.rfilename,
                cdn_url=cdn_url,
                size=f.size,
                sha256=f.lfs.get("sha256") if getattr(f, "lfs", None) else None,
                etag=f.etag,
                last_modified=f.last_modified.isoformat() if getattr(f, "last_modified", None) else None,
            )
        )

    # Deterministic ordering
    entries.sort(key=lambda e: e.path)
    return entries

def build_manifest(repo_id: str, date_folder: str) -> Dict[str, Any]:
    t0 = time.time()
    entries = list_repo_tree_cdn(repo_id=repo_id, date_folder=date_folder)

    manifest = {
        "repo_id": repo_id,
        "date_folder": date_folder.rstrip("/"),
        "generated_at_utc": datetime.now(timezone.utc).isoformat(),
        "generated_ts": int(time.time()),
        "count": len(entries),
        "total_size": sum(e.size for e in entries if e.size),
        "cdn_base": CDN_BASE,
        "entries": [e.to_dict() for e in entries],
        "elapsed_ms": int((time.time() - t0) * 1000),
    }
    return manifest

def generate_train_script(manifest: Dict[str, Any]) -> str:
    repo_id = manifest["repo_id"]
    date_folder = manifest["date_folder"]
    out = [
        "#!/usr/bin/env bash",
        "# CDN-only training file fetches — zero HF API calls during data loading",
        f"# Generated by airship discover for {repo_id}/{date_folder}",
        f"# Manifest generated at: {manifest['generated_at_utc']}",
        "set -euo pipefail",
        "",
        "OUTDIR=${1:-./data}",
        "mkdir -p \"$OUTDIR\"",
        "",
    ]

    for e in manifest["entries"]:
        fname = os.path.basename(e["path"])
        out.append(f'curl -L -o "$OUTDIR/{fname}" "{e["cdn_url"]}"')

    out.extend([
        "",
        "echo \"Downloaded ${#manifest[\"entries\"]} files to $OUTDIR\"",
        "ls -lh \"$OUTDIR\"",
    ])
    return "\n".join(out)

def main() -> None:
    parser = argparse.ArgumentParser(description="airship discover — CDN-only manifest generator")
    parser.add_argument("--repo_id", required=True, help="HF dataset repo id (e.g. 'org/repo')")
    parser.add_argument("--date_folder", required=True, help="Folder path in repo (e.g. '2026-04-29')")
    parser.add_argument("--out", default="./manifest.json", help="Output JSON path")
    parser.add_argument("--with-script", action="store_true", help="Also write cdn_train_script.sh")
    args = parser.parse_args()

    manifest = build_manifest(args.repo_id, args.date_folder)

    out_path = Path(args.out)
    out_path.write_text(json.dumps(manifest, indent=2, sort_keys=True))
    print(f"Manifest written to {out_path} ({len(manifest['entries'])} entries)")

    if args.with_script:
        script_path = out_path.parent / "cdn_train_script.sh"
        script_path.write_text(generate_train_script(manifest))
        script_path.chmod(0o755)
        print(f"Training script written to {script_path}")

if __name__ == "__main__":
    main()
```
