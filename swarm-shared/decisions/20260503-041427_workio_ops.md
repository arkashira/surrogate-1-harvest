#### Retry with backoff (exponential) + jitter  
File: `src/utils/retry.ts`

```ts
function sleep(ms: number) {
  return new Promise((resolve) => setTimeout(resolve, ms));
}

export async function retry<T>(
  fn: () => Promise<T>,
  attempts = 3,
  baseDelayMs = 300,
  backoff = 2,
  jitter = true
): Promise<T> {
  let lastError: unknown;
  let delay = baseDelayMs;

  for (let i = 0; i < attempts; i++) {
    try {
      return await fn();
    } catch (err) {
      lastError = err;
      const isLast = i === attempts - 1;
      if (!isLast) {
        const jitterMs = jitter ? Math.random() * 100 : 0;
        await sleep(delay + jitterMs);
        delay *= backoff;
      }
    }
  }
  throw lastError;
}
```

#### Idempotency guard (memory + optional Redis)  
File: `src/middleware/lineIdempotency.ts`

```ts
import { Request, Response, NextFunction } from 'express';
import createLogger from '../utils/logger';

const logger = createLogger();

// In-memory for dev/startup; replace with Redis in prod for multi-instance safety
const recentIds = new Set<string>();
const TTL_MS = 60_000;

export function lineEventIdempotency(req: Request, res: Response, next: NextFunction) {
  const eventId =
    req.body?.events?.[0]?.webhookEventId ||
    (req.headers['x-line-webhook-event-id'] as string) ||
    '';
  const tenantId = req.body?.tenantId || (req as any).tenantId || 'unknown';
  const key = `${tenantId}:${eventId}`;

  if (!eventId) {
    logger.warn({ tenantId, msg: 'missing_line_webhook_event_id' });
    return next();
  }

  if (recentIds.has(key)) {
    logger.info({ tenantId, eventId, key, msg: 'duplicate_line_event_dropped' });
    return res.status(200).json({ ok: true, reason: 'duplicate' });
  }

  recentIds.add(key);
  setTimeout(() => recentIds.delete(key), TTL_MS);
  (req as any)._lineEventId = eventId;
  next();
}
```

### 3. Logging and correlation middleware

File: `src/middleware/loggingMiddleware.ts`

```ts
import { Request, Response, NextFunction } from 'express';
import createLogger from '../utils/logger';

const logger = createLogger();

export function correlationIdMiddleware(req: Request, res: Response, next: NextFunction) {
  const cid =
    (req.headers['x-correlation-id'] as string) ||
    `cid-${Date.now()}-${Math.random().toString(36).slice(2, 9)}`;
  (req as any).correlationId = cid;
  res.setHeader('x-correlation-id', cid);
  next();
}

export function requestLoggingMiddleware(req: Request, res: Response, next: NextFunction) {
  const start = Date.now();
  const { method, url, correlationId } = req as any;
  const log = logger.child({ correlationId, method, url });

  res.on('finish', () => {
    log.info({
      status: res.statusCode,
      durationMs: Date.now() - start,
      msg: 'request_completed',
    });
  });

  (req as any).log = log;
  next();
}
```

### 4. LINE webhook service (resilient API calls)

File: `src/services/lineWebhookService.ts`

```ts
import { retry } from '../utils/retry';
import createLogger from '../utils/logger';

const logger = createLogger();

export async function replyLineMessage(replyToken: string, message: string, tenantId: string) {
  const url = 'https://api.line.me/v2/bot/message/reply';
  const token = process.env.LINE_CHANNEL_ACCESS_TOKEN;

  if (!token) {
    throw new Error('LINE_CHANNEL_ACCESS_TOKEN missing');
  }

  return retry(
    async () => {
      const res = await fetch(url, {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          Authorization: `Bearer ${token}`,
        },
        body: JSON.stringify({
          replyToken,
          messages: [{ type: 'text', text: message }],
        }),
      });

      if (!res.ok) {
        const text = await res.text().catch(() => '');
        const err: any = new Error(`LINE API ${res.status}`);
        err.status = res.status;
        err.body = text;
        throw err;
      }

      return res.json();
    },
    3,
    300,
    2,
    true
  ).catch((err) => {
    logger.error({ tenantId, err: String(err), status: err?.status, msg: 'line_reply_failed' });
    throw err;
  });
}
```
