# vanguard / design

## 1. Diagnosis
- No canonical discovery entrypoint exists to surface top-hub insights (e.g., MOC) before planning — violates established `#knowledge-rag #graph #hub` pattern and forces manual exploration.
- Missing CDN-bypass file-list strategy for HF datasets; any future surrogate-1 training ingestion will immediately hit API rate limits and commit caps.
- No reusable Lightning Studio orchestration wrapper — teams will recreate studios and burn quota instead of reusing running instances.
- No guardrails for heterogeneous HF dataset schemas (mixed file schemas) — surrogate-1 training will fail with pyarrow CastError on ingestion.
- No Mac=CLI + remote compute boundary enforcement — risk of running `.from_pretrained()` or heavy training locally on dev machines.

## 2. Proposed change
Create `/opt/axentx/vanguard/design/entrypoints.py` (single file, ~120 lines) that provides:
- `discover_top_hub()` — queries knowledge-rag for the most-connected hub (e.g., MOC) and returns actionable context.
- `build_hf_cdn_filelist(repo_id, date_folder)` — single-tree API call to list files for one date folder, returns JSON-serializable list of CDN URLs (no auth) and local paths.
- `get_or_create_lightning_studio(name, machine_spec, reuse=True)` — lists existing Teamspace studios, reuses a running one if found, else creates with target machine (L40S by default; H200 only on paid account).
- `safe_hf_dataset_pull(repo_id, file_infos, columns=("prompt","response"))` — downloads via `hf_hub_download` per file, projects only required columns at parse time to avoid pyarrow schema issues.
- `enforce_remote_compute()` — raises clear error if heavy ops are attempted locally; prints guidance to use Lightning/Kaggle/Cerebras.

# /opt/axentx/vanguard/design/entrypoints.py
from __future__ import annotations

import json
import os
import sys
from dataclasses import dataclass
from pathlib import Path
from typing import Any, Dict, List, Optional, Tuple

try:
    import requests
    from huggingface_hub import HfApi, hf_hub_download, list_repo_tree
except ImportError as e:
    print(f"[vanguard] Missing dependency: {e}. Install: pip install requests huggingface_hub")
    sys.exit(1)

# ---- HF CDN bypass helpers ----
HF_CDN_TEMPLATE = "https://huggingface.co/datasets/{repo_id}/resolve/main/{path}"

@dataclass
class HfFileInfo:
    repo_id: str
    path: str
    cdn_url: str
    local_name: str  # safe filename for caching

def build_hf_cdn_filelist(repo_id: str, date_folder: str) -> List[HfFileInfo]:
    """
    Single API call to list files in date_folder (non-recursive).
    Returns CDN URLs that bypass auth/rate limits during training.
    """
    api = HfApi()
    tree = list_repo_tree(repo_id=repo_id, path=date_folder, recursive=False)
    items: List[HfFileInfo] = []
    for node in tree:
        if node.type != "file":
            continue
        cdn = HF_CDN_TEMPLATE.format(repo_id=repo_id, path=node.path)
        items.append(
            HfFileInfo(
                repo_id=repo_id,
                path=node.path,
                cdn_url=cdn,
                local_name=os.path.basename(node.path),
            )
        )
    return items

def safe_hf_dataset_pull(
    repo_id: str,
    file_infos: List[HfFileInfo],
    cache_dir: Optional[str] = None,
    columns: Tuple[str, ...] = ("prompt", "response"),
) -> List[Dict[str, Any]]:
    """
    Download each file individually and project to required columns only.
    Avoids load_dataset(streaming=True) on heterogeneous schemas.
    """
    if cache_dir is None:
        cache_dir = os.path.join(os.getcwd(), ".hf_cache")
    os.makedirs(cache_dir, exist_ok=True)

    results: List[Dict[str, Any]] = []
    for info in file_infos:
        local_path = hf_hub_download(
            repo_id=repo_id,
            filename=info.path,
            cache_dir=cache_dir,
            force_download=False,
        )
        # Lightweight projection: only keep required keys; ignore others.
        # Supports JSONL/JSON/parquet via extension dispatch.
        ext = Path(local_path).suffix.lower()
        try:
            if ext == ".jsonl":
                import jsonlines
                with jsonlines.open(local_path) as reader:
                    for obj in reader:
                        projected = {k: obj.get(k) for k in columns if k in obj}
                        if projected:
                            results.append(projected)
            elif ext == ".json":
                with open(local_path, "r", encoding="utf-8") as f:
                    data = json.load(f)
                if isinstance(data, list):
                    for obj in data:
                        projected = {k: obj.get(k) for k in columns if k in obj}
                        if projected:
                            results.append(projected)
                elif isinstance(data, dict):
                    projected = {k: data.get(k) for k in columns if k in data}
                    if projected:
                        results.append(projected)
            elif ext in (".parquet", ".pq"):
                import pyarrow.parquet as pq
                table = pq.read_table(local_path, columns=list(columns) if columns else None)
                batch = table.to_pylist()
                for obj in batch:
                    projected = {k: obj.get(k) for k in columns if k in obj}
                    if projected:
                        results.append(projected)
            else:
                print(f"[vanguard] Skipping unsupported file: {local_path}")
        except Exception as exc:
            print(f"[vanguard] Failed to process {local_path}: {exc}")
            continue
    return results

# ---- Knowledge-rag top-hub discovery ----
def discover_top_hub(knowledge_rag_endpoint: str = "http://localhost:8000") -> Dict[str, Any]:
    """
    Query knowledge-rag for the most-connected hub (e.g., MOC).
    Returns actionable context to inform planning.
    """
    try:
        resp = requests.get(f"{knowledge_rag_endpoint}/top-hub", timeout=10)
        resp.raise_for_status()
        return resp.json()
    except Exception as exc:
        print(f"[vanguard] Knowledge-rag top-hub query failed: {exc}")
        return {"hub": "MOC", "insight": "Manual review recommended", "connections": 0}

# ---- Lightning Studio orchestration ----
try:
    from lightning import Fabric, LightningFlow, LightningWork, LightningApp
    from lightning.app import LightningStudio
    from lightning.app.utilities.cloud import get_cloud
    from lightning.app.utilities.state import AppState
    _LIGHTNING_AVAILABLE = True
except Exception:
    _LIGHTNING_AVAILABLE = False

@dataclass
class StudioSpec:
    name: str
    machine: str = "L40S"  # default; H200 requires lightning-lambda-prod
    cloud: str = "lightning-public-prod"
    requirements: Optional[List[str]] = None

def get_or_create_lightning_studio(
    spec: StudioSpec,
    reuse: bool = True,
    teamspace: Optional[Any] = None,
) -> Any:
    """
    Reuse running studio if available; otherwise create one.
    Returns a LightningStudio-like object with .run() capability.
    """
    if not _LIGHTNING_AVAILABLE:
        raise RuntimeError("Lightning SDK not available. Install lightning.")

    # Try to find running studio first
    if reuse:
        try:
            # Teamspace.studios may vary by SDK version; best-effort discovery
            from lightning.app import Teamspace
            studios = Teamspace.studios
            for s in studios:
                if getattr(s, "name", None) == spec.name and getattr(s, "status", None) == "Running":
                    print(f"[vanguard] Reusing running studio: {spec.name}")
                    return s
        except Exception:
            pass

    # Create new studio
    print(f"[vanguard] Creating studio: {spec.name} on {spec.machine} ({spec.cloud})")
    studio = LightningStudio(
        name=spec.name,
        machine=spec.machine,
        cloud=spec.cloud,
        create_ok=True,
    )
    return studio
