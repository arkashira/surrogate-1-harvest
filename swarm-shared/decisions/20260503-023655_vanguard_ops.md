### 2) `/opt/axentx/vanguard/ops/prepare_manifest.py`
```python
#!/usr/bin/env python3
import argparse
import json
import os
import fcntl          # simple flock to avoid concurrent runs
from huggingface_hub import HfApi
from utils import ensure_dir

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True)
    parser.add_argument("--date-folder", required=True)
    parser.add_argument("--out-file", required=True)
    args = parser.parse_args()

    ensure_dir(args.out_file)

    # Single API call; non-recursive to avoid pagination explosion
    api = HfApi()
    files = api.list_repo_tree(repo_id=args.repo, path=args.date_folder, recursive=False)
    paths = [f.rfilename for f in files if f.rfilename.endswith((".parquet", ".jsonl"))]

    out_path = os.path.abspath(args.out_file)
    lock_path = out_path + ".lock"
    with open(lock_path, "w") as lock_f:
        fcntl.flock(lock_f, fcntl.LOCK_EX)
        with open(out_path, "w") as f:
            json.dump({"repo": args.repo, "date_folder": args.date_folder, "files": paths}, f, indent=2)
        fcntl.flock(lock_f, fcntl.LOCK_UN)

    print(f"Wrote {len(paths)} files to {out_path}")

if __name__ == "__main__":
    main()
```

import argparse

import requests

import pyarrow.parquet as pq
import pyarrow as pa
from torch.utils.data import IterableDataset, DataLoader
from utils import load_manifest

class CDNParquetIterable(IterableDataset):
    def __init__(self, manifest, files=None):
        self.repo = manifest["repo"]
        self.date_folder = manifest["date_folder"]
        self.files = files or manifest["files"]
        self.base = f"https://huggingface.co/datasets/{self.repo}/resolve/main/{self.date_folder}"

    def _stream_parquet(self, url):
        # Stream to temp file to avoid memory blow-up on large parquet files
        with tempfile.NamedTemporaryFile(suffix=".parquet", delete=False) as tmp:
            tmp_path = tmp.name
            with requests.get(url, stream=True, timeout=120) as r:
                r.raise_for_status()
                for chunk in r.iter_content(chunk_size=8192):
                    tmp.write(chunk)
        try:
            table = pq.read_table(tmp_path)
        finally:
            os.unlink(tmp_path)
        return table

    def __iter__(self):
        for f in self.files:
            url = f"{self.base}/{f}"
            table = self._stream_parquet(url)
            # Expect columns including prompt/response; adapt as needed
            for batch in table.to_batches(max_chunksize=1024):
                df = batch.to_pandas()
                for _, row in df.iterrows():
                    yield {"prompt": row["prompt"], "response": row["response"]}
