## Implementation (≤2h)

### 1. Create `bin/dataset-enrich.py`

```python
#!/usr/bin/env python3
"""
Manifest-driven, CDN-bypass ingestion worker for surrogate-1.

Eliminates HF API rate limits and PyArrow CastError by:
- Pre-listing once via HF API (respects 429/128 caps)
- Downloading via CDN direct URLs (no auth, bypasses /api/ limits)
- Projecting heterogeneous schemas to {prompt, response} at parse time
- Deduplicating via centralized SQLite md5 store
"""

import os
import sys
import json
import hashlib
import sqlite3
import datetime
from pathlib import Path
from typing import Iterator, Dict, Any

import pandas as pd
import pyarrow.parquet as pq
import requests
from huggingface_hub import HfApi, hf_hub_download

# ---------- config ----------
SHARD_ID = int(os.getenv("SHARD_ID", -1))
SHARD_TOTAL = int(os.getenv("SHARD_TOTAL", 16))
DATE = os.getenv("DATE", "")
HF_TOKEN = os.getenv("HF_TOKEN", "")
DATASET_REPO = os.getenv("DATASET_REPO", "axentx/surrogate-1-training-pairs")
DEDUP_DB_URL = os.getenv("DEDUP_DB_URL", "file:dedup.db?mode=rwc")
OUT_DIR = Path(os.getenv("OUT_DIR", "."))

if SHARD_ID < 0 or SHARD_ID >= SHARD_TOTAL:
    print("ERROR: SHARD_ID must be 0..SHARD_TOTAL-1", file=sys.stderr)
    sys.exit(1)
if not DATE:
    print("ERROR: DATE is required (YYYY-MM-DD)", file=sys.stderr)
    sys.exit(1)
if not HF_TOKEN:
    print("ERROR: HF_TOKEN is required", file=sys.stderr)
    sys.exit(1)

API = HfApi(token=HF_TOKEN)
os.environ["HF_HUB_ENABLE_HF_TRANSFER"] = "1"

# ---------- dedup ----------
def get_conn() -> sqlite3.Connection:
    if DEDUP_DB_URL.startswith("file:"):
        path = DEDUP_DB_URL.split("file:")[1].split("?")[0]
        Path(path).parent.mkdir(parents=True, exist_ok=True)
    return sqlite3.connect(DEDUP_DB_URL, uri=True)

def init_db(conn: sqlite3.Connection) -> None:
    conn.execute("CREATE TABLE IF NOT EXISTS seen_md5 (md5 TEXT PRIMARY KEY)")
    conn.commit()

def is_duplicate(conn: sqlite3.Connection, md5: str) -> bool:
    cur = conn.execute("SELECT 1 FROM seen_md5 WHERE md5 = ?", (md5,))
    return cur.fetchone() is not None

def mark_seen(conn: sqlite3.Connection, md5: str) -> None:
    try:
        conn.execute("INSERT INTO seen_md5 (md5) VALUES (?)", (md5,))
    except sqlite3.IntegrityError:
        pass

# ---------- manifest ----------
def list_folder_files(path: str) -> list[str]:
    try:
        items = API.list_repo_tree(repo_id=DATASET_REPO, path=path, repo_type="dataset")
    except Exception as e:
        print(f"WARNING: list_repo_tree failed: {e}", file=sys.stderr)
        return []
    return [item["path"] for item in items if item.get("type") == "file"]

def build_manifest(date_folder: str) -> list[str]:
    files = list_folder_files(date_folder)
    manifest_path = OUT_DIR / "manifest.json"
    manifest_path.write_text(json.dumps(files, indent=2))
    return sorted(files)

# ---------- shard assignment ----------
def shard_for_file(file_path: str) -> int:
    slug = Path(file_path).stem
    return hash(slug) % SHARD_TOTAL

# ---------- download + project ----------
def content_md5(content: bytes) -> str:
    return hashlib.md5(content).hexdigest()

def project_row(raw: Dict[str, Any]) -> Dict[str, str]:
    prompt = raw.get("prompt") or raw.get("input") or raw.get("question") or ""
    response = raw.get("response") or raw.get("output") or raw.get("answer") or ""
    return {"prompt": str(prompt), "response": str(response)}

def download_parquet(cdn_url: str) -> Iterator[Dict[str, str]]:
    local_path = hf_hub_download(
        repo_id=DATASET_REPO, repo_type="dataset", filename=cdn_url, token=HF_TOKEN
    )
    table = pq.read_table(local_path)
    df = table.to_pandas()
    for _, row in df.iterrows():
        yield project_row(row.to_dict())

def download_jsonl(cdn_url: str) -> Iterator[Dict[str, str]]:
    local_path = hf_hub_download(
        repo_id=DATASET_REPO, repo_type="dataset", filename=cdn_url, token=HF_TOKEN
    )
    with open(local_path) as f:
        for line in f:
            if line.strip():
                yield project_row(json.loads(line))

def download_file(file_path: str) -> Iterator[Dict[str, str]]:
    if file_path.endswith(".parquet"):
        yield from download_parquet(file_path)
    elif file_path.endswith(".jsonl"):
        yield from download_jsonl(file_path)
    else:
        print(f"WARNING: skipping unsupported file {file_path}", file=sys.stderr)

# ---------- main ----------
def main() -> None:
    date_folder = f"public-merged/{DATE}"
    OUT_DIR.mkdir(parents=True, exist_ok=True)
    
    # Build manifest
    files = build_manifest(date_folder)
    if not files:
        print(f"ERROR: no files found in {date_folder}", file=sys.stderr)
        sys.exit(1)
    
    # Assign files to this shard
    my_files = [f for f in files if shard_for_file(f) == SHARD_ID]
    print(f"Shard {SHARD_ID}/{SHARD_TOTAL}: processing {len(my_files)} files")
    
    # Setup dedup
    conn = get_conn()
    init_db(conn)
    
    # Process files
    timestamp = datetime.datetime.utcnow().strftime("%H%M%S")
    out_file = OUT_DIR / f"shard{SHARD_ID}-{timestamp}.jsonl"
    
    accepted = 0
    duplicates = 0
    
    with open(out_file, "w") as f_out:
        for file_path in my_files:
            try:
                for row in download_file(file_path):
                    # Create content-based md5 for dedup
                    content = json.dumps(row, sort_keys=True).encode()
                    md5 = content_md5(content)
                    
                    if is_duplicate(conn, md5):
                        duplicates += 1
                        continue
                    
                    mark_seen(conn, md5)
                    f_out.write(json.dumps(row) + "\n")
                    accepted += 1
                    
            except Exception as e:
                print(f"ERROR processing {file_path}: {e}", file=sys.stderr)
                conn.rollback()
                continue
    
    conn.commit()
    conn.close()
    
    print(f"Shard {SHARD_ID}: {accepted} rows written, {duplicates} duplicates skipped")
    print(f"Output: {out_file}")
    
    # Upload via HF dataset API
    try:
        API.upload_file(
            path_or_fileobj=str(out_file),
            path_in_repo=f"batches/public-merged/{DATE}/{out_file.name}",
            repo_id=DATASET_REPO,
            repo_type="dataset",
            commit_message=f"Add shard {SHARD_ID} for {DATE}",
        )
        print("Upload successful")
    except Exception as e:
        print(f"ERROR: upload failed: {e}", file=sys.stderr)
        sys.exit(1)

if __name__ == "__main__":
    main()
```
