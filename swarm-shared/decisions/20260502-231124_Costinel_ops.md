# Costinel / ops

## Final Implementation Plan — `GET /api/v1/cost-anomaly/signal/top-hub`

**Scope & Constraints**
- Read-only endpoint (Sense + Signal). No mutations, no side effects.
- Optional `?for_date=YYYY-MM-DD` (default today).
- Optional `?top_k=int` (default 1) and `?max_hops=int` (default 2).
- Reuse existing knowledge-graph snapshots; avoid recomputing centrality on every request.
- Return compact, stable JSON suitable for dashboards/alerts.
- Ship in <2h with minimal footprint and tests.

---

### 1) Design

**Endpoint**
```
GET /api/v1/cost-anomaly/signal/top-hub
Query params:
  for_date   (optional) YYYY-MM-DD, default today
  top_k      (optional) int >=1, default 1
  max_hops   (optional) int >=1, default 2
```

**Response shape (canonical)**
```json
{
  "signal_id": "cost-anomaly:top-hub:2026-05-03",
  "generated_at": "2026-05-03T14:23:00Z",
  "for_date": "2026-05-03",
  "top_hubs": [
    {
      "node_id": "MOC",
      "node_type": "service",
      "label": "MOC",
      "centrality": {
        "degree": 42,
        "betweenness": 0.31,
        "pagerank": 0.12
      },
      "anomaly_score": 0.87,
      "cost_delta_7d": 0.42,
      "reason": "Highest composite centrality with rising cost delta"
    }
  ],
  "neighborhood": [
    { "node_id": "aws-ec2", "relation": "depends_on", "weight": 0.92 },
    { "node_id": "gcp-bigquery", "relation": "correlates_with", "weight": 0.81 }
  ],
  "recommendation": {
    "action": "review",
    "message": "Inspect MOC-linked resources for cost anomalies and reserved-instance coverage gaps."
  },
  "audit": {
    "graph_version": "kg-2026-05-03-v1",
    "read_only": true
  }
}
```

**Behavior**
- Load precomputed graph snapshot for `for_date` (or latest available ≤ today).
- If snapshot missing, return a minimal fallback signal (no exception).
- Compute lightweight composite centrality × (1 + max(0, cost_delta_7d)) to rank nodes.
- Select top_k hubs (default 1) and include 1–2 hop neighborhood for context.
- Time-box graph operations (fail fast) and keep memory footprint small.

### 3) Code Snippets

**`app/lib/graph_loader.py`**
```python
from __future__ import annotations

import json
from datetime import date
from pathlib import Path
from typing import Any, Dict, Optional
import logging

logger = logging.getLogger(__name__)

GRAPH_SNAPSHOTS_ROOT = Path("data/knowledge_graph")

def load_graph_snapshot(for_date: date) -> Optional[Dict[str, Any]]:
    """
    Load precomputed graph snapshot for a date.
    Expected file: data/knowledge_graph/{for_date}/graph.json
    Falls back to latest available snapshot <= today.
    """
    snapshot_path = GRAPH_SNAPSHOTS_ROOT / f"{for_date.isoformat()}" / "graph.json"
    if snapshot_path.exists():
        try:
            with snapshot_path.open("r", encoding="utf-8") as fh:
                return json.load(fh)
        except Exception as exc:
            logger.warning("Failed to load graph snapshot %s: %s", snapshot_path, exc)

    # fallback: latest available <= today
    parent = GRAPH_SNAPSHOTS_ROOT
    if not parent.exists():
        return None
    candidates = sorted(
        (p for p in parent.iterdir() if p.is_dir() and p.name <= date.today().isoformat()),
        reverse=True,
    )
    for cand in candidates:
        fallback = cand / "graph.json"
        if fallback.exists():
            logger.info("Using fallback snapshot: %s", fallback)
            try:
                with fallback.open("r", encoding="utf-8") as fh:
                    return json.load(fh)
            except Exception as exc:
                logger.warning("Failed to load fallback snapshot %s: %s", fallback, exc)
    return None
```

**`app/services/top_hub_signal.py`**
```python
from __future__ import annotations

from datetime import date, datetime, timezone
from typing import Any, Dict, List
import logging

logger = logging.getLogger(__name__)

from app.lib.graph_loader import load_graph_snapshot

def _composite_score(node: Dict[str, Any]) -> float:
    centrality = node.get("centrality", {})
    degree = centrality.get("degree", 0)
    betweenness = centrality.get("betweenness", 0.0)
    pagerank = centrality.get("pagerank", 0.0)
    # normalize betweenness/pagerank by approximating typical scale; weights are tunable
    return 0.4 * degree + 0.4 * (betweenness * 100) + 0.2 * (pagerank * 100)

def _neighborhood(graph: Dict[str, Any], node_id: str, max_hops: int = 2, limit: int = 10) -> List[Dict[str, Any]]:
    edges = graph.get("edges", [])
    visited = {node_id}
    current = {node_id}
    result: List[Dict[str, Any]] = []

    for _ in range(max_hops):
        next_level: set[str] = set()
        for e in edges:
            src, dst = e.get("source"), e.get("target")
            if src in current and dst not in visited:
                result.append({"node_id": dst, "relation": e.get("relation", "related"), "weight": e.get("weight", 0.0)})
                visited.add(dst)
                next_level.add(dst)
            elif dst in current and src not in visited:
                result.append({"node_id": src, "relation": e.get("relation", "related"), "weight": e.get("weight", 0.0)})
                visited.add(src)
                next_level.add(src)
        current = next_level
        if not current:
            break
    # dedupe while preserving order
    seen = set()
    deduped: List[Dict[str, Any]] = []
    for item in result:
        key = item["node_id"]
        if key not in seen:
            seen.add(key)
            deduped.append(item)
    return deduped[:limit]

def build_top_hub_signal(for_date: date, top_k: int = 1, max_hops: int = 2) -> Dict[str, Any]:
    graph = load_graph_snapshot(for_date)
    generated_at = datetime.now(timezone.utc).isoformat()
    signal_id = f"cost-anomaly:top-hub:{for_date.isoformat()}"

    if not graph:
        return {
            "signal_id": signal_id,
            "generated_at": generated_at,
            "for_date": for_date.isoformat(),
            "top_hubs": [],
            "neighborhood": [],
            "recommendation": {"action": "none", "message": "Knowledge graph snapshot unavailable."},
            "audit": {"graph_version": None
