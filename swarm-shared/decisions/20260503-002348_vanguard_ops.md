# Final Consolidated Solution

## 1. Diagnosis (merged)
- **No persisted manifest**: Every training run re-enumerates via authenticated HF API → quota burn + 429 risk.
- **Recursive enumeration during training**: Likely uses `list_repo_tree` or `load_dataset(streaming=True)` causing repeated API calls and mixed-schema parse failures.
- **No CDN-only fetch path**: Authenticated API calls during data load are rate-limited and unnecessary; CDN (`resolve/main/`) bypasses auth and quota.
- **No Lightning Studio reuse guard**: Idle-stop + recreation burns ~80hr/mo quota and risks H200/L40S allocation churn.
- **No idle-stop resilience**: Lightning idle timeout kills long-running training; no auto-restart on stopped studio.
- **Mac orchestration risk**: May run model code locally (accidental `from_pretrained`) instead of delegating training to Lightning Studio / HF Space.

## 2. Proposed Change (merged + strengthened)
Create **two** artifacts:

1. `/opt/axentx/vanguard/ops/prepare_manifest.py` (single-purpose, <150 LOC)
   - Accepts `REPO`, `DATE_FOLDER` (e.g. `2026-04-29`) via CLI.
   - Calls `list_repo_tree(path=DATE_FOLDER, recursive=False)` **once** (paginated safe).
   - Emits `manifests/{REPO}/{DATE_FOLDER}.json` containing CDN URLs, relative paths, sizes, and LFS hints.
   - Exits 0 only if manifest created and contains ≥1 file.

2. `/opt/axentx/vanguard/train/train.py` patch + runtime guard
   - Reads pre-generated manifest (**zero** HF API calls during training).
   - Uses raw CDN fetch per file in manifest; projects to `{prompt, response}` only; streams into training loader.
   - Wraps Lightning `Studio` usage with **reuse + idle-stop guard + auto-restart**.
   - Adds fallback to `hf_hub_download` for LFS files when CDN raw fetch fails.

# /opt/axentx/vanguard/ops/prepare_manifest.py
#!/usr/bin/env python3
"""
Generate a CDN-only manifest for a repo+dateFolder to avoid HF API calls during training.
Usage:
  python prepare_manifest.py --repo surrogate-1/datasets --date 2026-04-29
"""
import argparse
import json
import sys
from pathlib import Path

try:
    from huggingface_hub import list_repo_tree
except ImportError:
    print("ERROR: huggingface_hub not installed. Run: pip install huggingface_hub")
    sys.exit(1)

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def build_manifest(repo: str, date_folder: str, out_dir: Path):
    # Non-recursive, single folder listing (safe for rate limits)
    entries = list_repo_tree(repo=repo, path=date_folder, recursive=False)
    files = [e for e in entries if e.get("type") == "file"]
    if not files:
        print(f"WARN: no files found in {repo}/{date_folder}")
        return []

    manifest = []
    for f in files:
        rel = f["path"]
        manifest.append({
            "repo": repo,
            "path": rel,
            "cdn_url": CDN_TEMPLATE.format(repo=repo, path=rel),
            "size": f.get("size"),
            "lfs": f.get("lfs", {}).get("oid") is not None
        })

    out_dir.mkdir(parents=True, exist_ok=True)
    safe_repo = repo.replace("/", "_")
    out_path = out_dir / f"{safe_repo}__{date_folder}.json"
    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"Manifest written: {out_path} ({len(manifest)} files)")
    return manifest

def main():
    parser = argparse.ArgumentParser(description="Generate CDN manifest for HF dataset folder.")
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g. surrogate-1/datasets)")
    parser.add_argument("--date", required=True, help="Date folder (e.g. 2026-04-29)")
    parser.add_argument("--out-dir", default="manifests", help="Output directory for manifests")
    args = parser.parse_args()

    manifest = build_manifest(args.repo, args.date, Path(args.out_dir))
    if not manifest:
        sys.exit(1)

if __name__ == "__main__":
    main()

"""
Training loader that uses pre-generated manifest (zero HF API during training).
Includes Studio reuse + idle-stop guard + auto-restart.
"""
import json
import time
from pathlib import Path
from typing import Dict, Iterator

import requests
from huggingface_hub import hf_hub_download

MANIFEST_DIR = Path(__file__).parent.parent / "manifests"

def load_manifest(repo: str, date_folder: str) -> list[Dict]:
    safe_repo = repo.replace("/", "_")
    manifest_path = MANIFEST_DIR / f"{safe_repo}__{date_folder}.json"
    if not manifest_path.is_file():
        raise FileNotFoundError(
            f"Manifest missing: {manifest_path}. Run prepare_manifest.py first."
        )
    return json.loads(manifest_path.read_text())

def cdn_records(repo: str, date_folder: str) -> Iterator[Dict]:
    for item in load_manifest(repo, date_folder):
        # Prefer raw CDN fetch (no auth, no API quota)
        try:
            resp = requests.get(item["cdn_url"], timeout=30)
            resp.raise_for_status()
            yield {
                "raw": resp.content,
                "path": item["path"],
                "cdn_url": item["cdn_url"],
                "source": "cdn"
            }
        except Exception:
            # Fallback for LFS or CDN edge cases
            local_path = hf_hub_download(
                repo_id=item["repo"],
                filename=item["path"].replace(f"{date_folder}/", "", 1),
                repo_type="dataset"
            )
            with open(local_path, "rb") as f:
                yield {
                    "raw": f.read(),
                    "path": item["path"],
                    "cdn_url": item["cdn_url"],
                    "source": "hf_hub_download"
                }

# Studio reuse + idle-stop guard + auto-restart
def get_or_create_studio(name: str, machine: str = "L40S"):
    from lightning import Studio, Teamspace  # optional import

    max_retries = 3
    for attempt in range(max_retries):
        try:
            running = [
                s for s in Teamspace.studios
                if s.name == name and s.status == "Running"
            ]
            if running:
                print(f"Reusing running studio: {name}")
                return running[0]

            # Check for stopped studio and restart
            stopped = [
                s for s in Teamspace.studios
                if s.name == name and s.status == "Stopped"
            ]
            if stopped:
                studio = stopped[0]
                print(f"Restarting stopped studio: {name}")
                studio.start()
                # Wait for running
                for _ in range(30):
                    studio.refresh()
                    if studio.status == "Running":
                        return studio
                    time.sleep(10)
                raise RuntimeError(f"Studio {name} failed to start after restart")

            # Create new
            print(f"Creating studio: {name}")
            return Studio(
                name=name,
                machine=machine,
                create_ok=True,
            )
        except Exception as e:
            if attempt == max_retries - 1:
                raise
            print(f"Studio operation failed (attempt {attempt+1}/{max_retries}): {e}")
            time.sleep(5)
