### 1. `scripts/list_hf_files.py`
```bash
#!/usr/bin/env bash
# list_hf_files.sh — run from Mac after rate-limit window clears
# Usage: bash scripts/list_hf_files.sh <dataset_repo> <date_folder> > file-list.json
# Example: bash scripts/list_hf_files.sh axentx/surrogate-data 2026-05-03 > file-list.json

set -euo pipefail
REPO="${1:-axentx/surrogate-data}"
FOLDER="${2:-}"

if [ -z "$FOLDER" ]; then
  echo "Usage: $0 <dataset_repo> <date_folder>" >&2
  exit 1
fi

python3 - "$REPO" "$FOLDER" <<'PY'
import json, os, sys, time, requests
from huggingface_hub import HfApi, hf_hub_download

REPO = sys.argv[1]
FOLDER = sys.argv[2].strip("/")
API = HfApi()

# Single non-recursive call per folder (avoids 100x pagination)
entries = API.list_repo_tree(repo_id=REPO, path=FOLDER, recursive=False)

files = []
for e in entries:
    if e.path.lower().endswith((".parquet", ".pq")):
        # CDN URL (no auth, bypasses /api/ rate limit)
        cdn = f"https://huggingface.co/datasets/{REPO}/resolve/main/{e.path}"
        files.append({
            "path": e.path,
            "cdn": cdn,
            "size": getattr(e, "size", None)
        })

print(json.dumps({"repo": REPO, "folder": FOLDER, "files": files}, indent=2))
PY
```

Make executable:
```bash
chmod +x scripts/list_hf_files.py
```

class CDNParquetDataset(torch.utils.data.IterableDataset):
    """
    Streams parquet files directly from CDN (no HF API auth).
    Projects raw columns to {prompt, response} at parse time.
    """

    def __init__(self, file_list, batch_size=64, max_rows=None):
        super().__init__()
        self.file_list = file_list
        self.batch_size = batch_size
        self.max_rows = max_rows

    def _project_row(self, row):
        # Heuristic projection; adapt to your schema
        prompt = row.get("prompt") or row.get("input") or row.get("text") or ""
        response = row.get("response") or row.get("output") or row.get("completion") or ""
        if not prompt or not response:
            # Fallback: try to find any two text cols
            text_cols = [c for c in row.keys() if isinstance(row[c], str) and len(row[c]) > 10]
            if len(text_cols) >= 2:
                prompt, response = text_cols[0], text_cols[1]
        return {"prompt": str(prompt), "response": str(response)}

    def _stream_file(self, cdn_url):
        retries = 0
        while retries < MAX_RETRIES:
            try:
                # CDN download (no auth) via pandas/pyarrow
                df = pd.read_parquet(cdn_url, engine="pyarrow")
                for _, row in df.iterrows():
                    yield self._project_row(row)
                return
            except Exception as exc:
                retries += 1
                print(f"[WARN] CDN fetch failed {cdn_url}: {exc} (retry {retries}/{MAX_RETRIES})")
                time.sleep(RETRY_WAIT * retries)
        print(f"[ERROR] Exhausted retries for {cdn_url}")

    def __iter__(self):
        worker_info = torch.utils.data.get_worker_info()
        if worker_info is None:
            files = self.file_list
        else:
            per_worker = max(1, len(self.file_list) // worker_info.num_workers)
            files = self.file_list[worker_info.id * per_worker : (worker_info.id + 1) * per_worker]

        rows_yielded = 0
        for entry in files:
            cdn = entry["cdn"] if isinstance(entry, dict) else entry
            for item in self._stream_file(cdn):
                yield item
                rows_yielded += 1
                if self.max_rows and rows_yielded >= self.max_rows:
                    return

# ---------- HF Sibling Selector ----------
def select_sibling_repo(slug: str) -> str:
    """Deterministic repo selection for HF commit-cap scaling."""
    digest = hashlib.sha256(slug.encode()).hexdigest()
    idx = int(digest, 16) % len(SIBLING_REPOS)
    return SIBLING_REPOS[idx]
