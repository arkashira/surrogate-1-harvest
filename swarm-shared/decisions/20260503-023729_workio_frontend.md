# workio / frontend

## Final synthesized implementation (best of both proposals)

**Scope**:  
- `workio/src/hooks/usePunch.ts` (new)  
- `workio/src/features/attendance/PunchButton.tsx` (new)  
- Light edit to the page that renders the punch control (e.g., `Dashboard.tsx`)  
- Backend: enforce idempotency key uniqueness and upsert on conflict  

**Goal**:  
- Immediate optimistic UI with guarded double-tap  
- Non-blocking location handling with graceful fallback  
- Client-side idempotency + local pending queue for offline/retry  
- Clear success/error feedback (toast + button state)  
- No duplicate punches from UI or webhook retries

### `workio/src/hooks/usePunch.ts`

```ts
import { useCallback, useState, useRef } from 'react';
import { toast } from 'sonner';

type PunchType = 'clock_in' | 'clock_out';
type PunchState = 'idle' | 'pending' | 'success' | 'error';

interface PunchResult {
  id: string;
  employee_id: string;
  punch_type: PunchType;
  punch_date: string; // ISO date
  timestamp: string;   // ISO timestamp
  latitude?: number | null;
  longitude?: number | null;
}

interface PendingPunch {
  idempotencyKey: string;
  type: PunchType;
  timestamp: string;
  latitude?: number;
  longitude?: number;
}

export function usePunch(employeeId: string) {
  const [state, setState] = useState<PunchState>('idle');
  const [lastResult, setLastResult] = useState<PunchResult | null>(null);
  const pendingRef = useRef<Record<string, PendingPunch>>({});
  const controllerRef = useRef<AbortController | null>(null);

  const punch = useCallback(
    async (type: PunchType, location?: { latitude: number; longitude: number }) => {
      if (state === 'pending') return;

      const now = new Date();
      const datePart = now.toISOString().slice(0, 10); // YYYY-MM-DD
      const idempotencyKey = `${employeeId}:${datePart}:${type}:${crypto.randomUUID()}`;
      const timestamp = now.toISOString();

      const pendingPunch: PendingPunch = {
        idempotencyKey,
        type,
        timestamp,
        latitude: location?.latitude,
        longitude: location?.longitude,
      };

      // Optimistic local queue + UI state
      pendingRef.current[idempotencyKey] = pendingPunch;
      setState('pending');

      const prevController = controllerRef.current;
      const controller = new AbortController();
      controllerRef.current = controller;

      try {
        const res = await fetch('/api/attendance/punch', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            employee_id: employeeId,
            punch_type: type,
            punch_date: datePart,
            timestamp,
            latitude: location?.latitude ?? null,
            longitude: location?.longitude ?? null,
            idempotency_key: idempotencyKey,
          }),
          signal: controller.signal,
          credentials: 'include',
        });

        if (!res.ok) {
          const err = (await res.json().catch(() => ({}))) || {};
          throw new Error(err.message || `Failed to ${type}`);
        }

        const data: PunchResult = await res.json();
        setLastResult(data);
        setState('success');
        toast.success(
          type === 'clock_in' ? 'Clock in recorded' : 'Clock out recorded',
          { description: new Date(timestamp).toLocaleTimeString() }
        );

        // Remove from pending queue on success
        delete pendingRef.current[idempotencyKey];
        return data;
      } catch (err: any) {
        if (err.name !== 'AbortError') {
          setState('error');
          toast.error(
            type === 'clock_in' ? 'Clock in failed' : 'Clock out failed',
            { description: err.message || 'Please try again' }
          );
          // Keep in pending queue so caller can retry
        }
        throw err;
      } finally {
        if (controllerRef.current === controller) {
          controllerRef.current = null;
        }
      }
    },
    [employeeId, state]
  );

  const retryPending = useCallback(async (idempotencyKey: string) => {
    const pending = pendingRef.current[idempotencyKey];
    if (!pending) return;

    setState('pending');
    try {
      const res = await fetch('/api/attendance/punch', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          employee_id: employeeId,
          punch_type: pending.type,
          punch_date: pending.timestamp.slice(0, 10),
          timestamp: pending.timestamp,
          latitude: pending.latitude ?? null,
          longitude: pending.longitude ?? null,
          idempotency_key: idempotencyKey,
        }),
        credentials: 'include',
      });

      if (!res.ok) throw new Error('Retry failed');

      const data: PunchResult = await res.json();
      setLastResult(data);
      setState('success');
      toast.success('Punch saved', { description: new Date(pending.timestamp).toLocaleTimeString() });
      delete pendingRef.current[idempotencyKey];
    } catch {
      setState('error');
      toast.error('Retry failed');
    }
  }, [employeeId]);

  const reset = useCallback(() => setState('idle'), []);
  const cancel = useCallback(() => {
    controllerRef.current?.abort();
    controllerRef.current = null;
  }, []);

  return {
    punch,
    state,
    reset,
    cancel,
    lastResult,
    pending: Object.values(pendingRef.current),
    retryPending,
  };
}
```
