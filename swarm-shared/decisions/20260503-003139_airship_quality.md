## Implementation Plan (≤2h)

**Goal**: Eliminate HF API 429s and mixed-schema CastErrors in surrogate training by switching to a manifest-driven, CDN-only iterable loader.

**Scope**:
- Add a Mac-side generator script (`scripts/generate_cdn_manifest.py`) that lists one date folder via `list_repo_tree` (non-recursive) and emits `file_manifest.json` with CDN URLs and slugs.
- Add a Lightning-compatible IterableDataset (`surrogate/data/cdn_dataset.py`) that reads the manifest and streams files directly from CDN (no Authorization, no HF dataset API).
- Project each file to `{prompt, response}` at parse time; drop extra columns.
- Update surrogate training entrypoint to use the new dataset and reuse running Studio when possible.
- Add a small integration test to verify CDN fetch and schema projection.

**Estimated effort**: ~90–110 minutes.

### 1) Manifest generator (Mac orchestrator)

```python
# scripts/generate_cdn_manifest.py
#!/usr/bin/env python3
"""
Generate file_manifest.json for a single date folder in a HF dataset repo.
Run on Mac (or any orchestrator) when rate-limit window is clear.

Usage:
  python scripts/generate_cdn_manifest.py \
    --repo datasets/your-org/your-repo \
    --date-folder 2026-05-03 \
    --out surrogate/data/file_manifest.json
"""

import argparse
import json
import os
import time
from pathlib import Path
from typing import List, Dict

from huggingface_hub import HfApi

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def build_manifest(repo: str, date_folder: str, out_path: Path) -> List[Dict]:
    api = HfApi()
    # Non-recursive list to avoid pagination explosion
    entries = api.list_repo_tree(repo=repo, path=date_folder, recursive=False)

    files = []
    for entry in entries:
        if entry.type != "file":
            continue
        path = f"{date_folder}/{entry.path}"
        slug = Path(entry.path).stem  # e.g., abc123 from abc123.parquet
        files.append(
            {
                "repo": repo,
                "path": path,
                "slug": slug,
                "cdn_url": CDN_TEMPLATE.format(repo=repo, path=path),
            }
        )

    out_path.parent.mkdir(parents=True, exist_ok=True)
    with open(out_path, "w") as f:
        json.dump(files, f, indent=2)

    print(f"Wrote {len(files)} entries to {out_path}")
    return files

def main() -> None:
    parser = argparse.ArgumentParser(description="Generate CDN manifest for HF dataset folder.")
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g. datasets/org/name)")
    parser.add_argument("--date-folder", required=True, help="Date folder inside repo (e.g. 2026-05-03)")
    parser.add_argument("--out", required=True, help="Output JSON path")
    args = parser.parse_args()

    # Basic retry for 429 (should be rare if run after window clear)
    for attempt in range(3):
        try:
            build_manifest(args.repo, args.date_folder, Path(args.out))
            break
        except Exception as exc:
            wait = 30 * (attempt + 1)
            print(f"Attempt {attempt+1} failed: {exc}. Retrying in {wait}s")
            time.sleep(wait)
    else:
        raise RuntimeError("Failed to generate manifest after retries")

if __name__ == "__main__":
    main()
```

### 2) CDN-only IterableDataset (Lightning training)

```python
# surrogate/data/cdn_dataset.py
import json
import pyarrow as pa
import pyarrow.parquet as pq
import requests
import torch
from torch.utils.data import IterableDataset
from typing import Dict, Any, Optional

class CDNParquetIterableDataset(IterableDataset):
    """
    Streams Parquet files from HF CDN using a pre-generated manifest.
    No HF API calls during training. Projects each row to {prompt, response}.
    """

    def __init__(self, manifest_path: str, start_index: int = 0, max_files: Optional[int] = None):
        manifest_path = str(manifest_path)
        with open(manifest_path) as f:
            self.files = json.load(f)

        if max_files:
            self.files = self.files[:max_files]
        self.start_index = start_index

    def _stream_file(self, entry: Dict[str, Any]):
        # CDN fetch: no Authorization header (bypasses API rate limits)
        resp = requests.get(entry["cdn_url"], stream=True, timeout=60)
        resp.raise_for_status()

        # Use pyarrow to read projected columns only
        try:
            table = pq.read_table(pa.BufferReader(resp.content), columns=["prompt", "response"])
        except pa.ArrowInvalid as exc:
            # Fallback: read full file and project (handles mixed schemas)
            table = pq.read_table(pa.BufferReader(resp.content))
            if "prompt" not in table.column_names or "response" not in table.column_names:
                raise ValueError(f"Missing prompt/response in {entry['path']}") from exc
            table = table.select(["prompt", "response"])

        for row in table.to_pylist():
            yield {
                "prompt": row["prompt"],
                "response": row["response"],
                "_slug": entry["slug"],
            }

    def __iter__(self):
        worker_info = torch.utils.data.get_worker_info()
        if worker_info is None:
            files = self.files[self.start_index:]
        else:
            per_worker = len(self.files) // worker_info.num_workers
            worker_id = worker_info.id
            files = self.files[worker_id * per_worker : (worker_id + 1) * per_worker]

        for entry in files:
            yield from self._stream_file(entry)
```

```python
# surrogate/train.py  (minimal diff)
import lightning as L
from lightning.pytorch.utilities import rank_zero_only
from surrogate.data.cdn_dataset import CDNParquetIterableDataset
from torch.utils.data import DataLoader

MANIFEST_PATH = "surrogate/data/file_manifest.json"

def maybe_reuse_studio(name: str, machine: L.Machine) -> L.Studio:
    teamspace = L.Teamspace()
    for s in teamspace.studios:
        if s.name == name and s.status == "running":
            print(f"Reusing running studio: {s.name}")
            return s
    print(f"Creating new studio: {name}")
    return L.Studio(
        name=name,
        machine=machine,
        create_ok=True,
    )

class SimpleSurrogateModel(L.LightningModule):
    # ... existing model code unchanged ...

class SurrogateDataModule(L.LightningDataModule):
    def train_dataloader(self):
        dataset = CDNParrogateIterableDataset(
            manifest_path=MANIFEST_PATH,
            max_files=None,
        )
        return DataLoader(dataset, batch_size=8, num_workers=4)

@rank_zero_only
def main():
    # Prefer L40S; fall back to free-tier compatible machine
    try:
        machine = L.Machine.L40S
        studio = maybe_reuse_studio("surrogate-train", machine=machine)
    except Exception:
        machine = L.Machine.L40  # or other free-tier compatible
        studio = maybe_reuse_studio("surrogate-train-free", machine=machine)

    model = SimpleSurrogateModel()
    dm = SurrogateDataModule()

    trainer = L.Trainer(max_epochs=1, devices=1, accelerator="gpu")
    trainer.fit(model, dm)

if __name__ == "__main__":
    main()
```
