## Implementation Plan (≤2h)

1. **Create CLI entrypoint** (`bin/airship-discover`)  
   - POSIX-compliant Bash, `#!/usr/bin/env bash`, `chmod +x`.
   - Accepts `<repo> <folder> [outdir]` and `--help`.
   - Uses `curl` to HF API **once** (non-recursive tree) with retries.

2. **Deterministic discovery & outputs**  
   - `manifest.json`: sorted file list (path, size, lfs) + metadata.
   - `status.json`: repo, folder, timestamp, manifest sha256, file count, status.
   - `file_urls.txt`: CDN URLs (`https://huggingface.co/datasets/{repo}/resolve/main/{folder}/{path}`) for auth-free, rate-limit-free fetching.

3. **Makefile integration**  
   - `make airship-discover REPO=datasets/myorg/surrogate-1 DATE=2026-05-02 OUT=./var/discover`
   - Idempotent; safe for CI.

4. **Lightweight validation & robustness**  
   - Check `curl`, `jq`, `sha256sum` availability.
   - Retry with exponential backoff on HF API failures.
   - Validate outputs exist and are well-formed.

5. **Documentation**  
   - Inline `--help`.
   - One-line usage in README.

## Code Snippets

### `bin/airship-discover`

```bash
#!/usr/bin/env bash
# airship-discover — CDN-safe deterministic file discovery for HF datasets
# Usage: ./airship-discover <repo> <folder> [outdir]
#        ./airship-discover --help
#
# Emits (in outdir):
#   manifest.json  — sorted file list + metadata
#   status.json    — discovery status + checksums
#   file_urls.txt  — CDN URLs for auth-free fetching
#
# Requires: curl, jq, sha256sum

set -euo pipefail

show_help() {
  cat <<EOF
airship-discover — CDN-safe deterministic HF dataset discovery

Usage:
  $0 <repo> <folder> [outdir]
  $0 --help

Arguments:
  repo     Hugging Face dataset repo (e.g. datasets/myorg/surrogate-1)
  folder   Folder/date within repo (e.g. 2026-05-02)
  outdir   Output directory (default: ./airship-discover-out)

Outputs:
  outdir/manifest.json   Deterministic file manifest
  outdir/status.json     Discovery status + metadata
  outdir/file_urls.txt   CDN URLs for training/inference (no auth)

Notes:
  - Uses HF API once (non-recursive tree) and caches result.
  - CDN URLs avoid API auth/rate limits during training/inference.
  - Idempotent; safe to rerun.
EOF
}

# Parse args
if [[ "${1:-}" == "--help" || "${1:-}" == "-h" ]]; then
  show_help
  exit 0
fi

REPO="${1:-}"
FOLDER="${2:-}"
OUTDIR="${3:-./airship-discover-out}"

if [[ -z "$REPO" || -z "$FOLDER" ]]; then
  echo "ERROR: repo and folder are required" >&2
  show_help
  exit 1
fi

mkdir -p "$OUTDIR"

HF_API="https://huggingface.co/api"
TIMESTAMP="$(date -u +"%Y-%m-%dT%H:%M:%SZ")"

MANIFEST="$OUTDIR/manifest.json"
STATUS="$OUTDIR/status.json"
URLS="$OUTDIR/file_urls.txt"

# Basic deps check
for cmd in curl jq sha256sum; do
  if ! command -v "$cmd" >/dev/null 2>&1; then
    echo "ERROR: required command not found: $cmd" >&2
    exit 1
  fi
done

# Fetch tree (non-recursive) with retries
MAX_RETRIES=3
RETRY=0
WAIT=5
RESPONSE=""
while [ $RETRY -lt $MAX_RETRIES ]; do
  if RESPONSE=$(curl -sSf --retry 2 --retry-delay 1 \
    "${HF_API}/datasets/${REPO}/tree?path=${FOLDER}&recursive=false" 2>/tmp/hf_err); then
    break
  else
    RETRY=$((RETRY+1))
    if [ $RETRY -ge $MAX_RETRIES ]; then
      echo "ERROR: HF API request failed after $MAX_RETRIES attempts" >&2
      cat /tmp/hf_err >&2
      exit 1
    fi
    echo "WARN: HF API error, retrying in ${WAIT}s (attempt $RETRY/$MAX_RETRIES)" >&2
    sleep $WAIT
    WAIT=$((WAIT*2))
  fi
done

# Deterministic ordering by path; keep only files
echo "$RESPONSE" | jq -r 'sort_by(.path) | .[] | select(.type=="file")' > "$OUTDIR/_files_raw.jsonl"

# Build manifest
jq -s \
  --arg repo "$REPO" \
  --arg folder "$FOLDER" \
  --arg ts "$TIMESTAMP" \
  '{
    repo: $repo,
    folder: $folder,
    generated_at: $ts,
    file_count: length,
    files: [.[] | {path, size, lfs}]
  }' "$OUTDIR/_files_raw.jsonl" > "$MANIFEST"

# Build CDN URLs (auth-free, rate-limit-free)
jq -r '.files[].path' "$MANIFEST" | while read -r p; do
  echo "https://huggingface.co/datasets/${REPO}/resolve/main/${FOLDER}/${p}"
done > "$URLS"

# Manifest checksum
MANIFEST_SHA256="$(sha256sum "$MANIFEST" | awk '{print $1}')"

# Emit status
jq -n \
  --arg repo "$REPO" \
  --arg folder "$FOLDER" \
  --arg ts "$TIMESTAMP" \
  --arg sha "$MANIFEST_SHA256" \
  --argjson count "$(jq '.file_count' "$MANIFEST")" \
  '{
    repo: $repo,
    folder: $folder,
    generated_at: $ts,
    manifest_sha256: $sha,
    file_count: $count,
    status: "ok",
    outputs: {
      manifest: "manifest.json",
      urls: "file_urls.txt",
      status: "status.json"
    }
  }' > "$STATUS"

# Cleanup
rm -f "$OUTDIR/_files_raw.jsonl"

echo "OK: discovery complete"
echo "  repo:     $REPO"
echo "  folder:   $FOLDER"
echo "  outdir:   $OUTDIR"
echo "  files:    $(jq '.file_count' "$MANIFEST")"
echo "  manifest: $MANIFEST"
echo "  urls:     $URLS"
echo "  status:   $STATUS"
```

### `Makefile`

```make
AIRSHIP_DISCOVER := ./bin/airship-discover

.PHONY: airship-discover
airship-discover:
	@if [ -z "$(REPO)" ] || [ -z "$(DATE)" ]; then \
		echo "Usage: make airship-discover REPO=<repo> DATE=<folder> [OUT=<outdir>]"; \
		exit 1; \
	fi
	@mkdir -p $(OUT)
	$(AIRSHIP_DISCOVER) $(REPO) $(DATE) $(OUT)
```
