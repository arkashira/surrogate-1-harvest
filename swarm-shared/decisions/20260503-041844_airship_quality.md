### 1. Generate manifest (run once from Mac)
```bash
# Install if needed
pip install huggingface_hub pyyaml

# Generate manifest for a specific date folder
python - <<'PY'
import json, os
from huggingface_hub import HfApi

api = HfApi()
repo_id = "axentx/surrogate-dataset"  # adjust
date_folder = "2026-05-03"            # adjust

# Single API call: non-recursive per folder
tree = api.list_repo_tree(repo_id, path=date_folder, recursive=False)
files = [f.rfilename for f in tree if f.type == "file"]

manifest = {
    "repo_id": repo_id,
    "date_folder": date_folder,
    "files": sorted(files),
    "cdn_base": f"https://huggingface.co/datasets/{repo_id}/resolve/main/{date_folder}"
}

out_path = "training_manifest.json"
with open(out_path, "w") as f:
    json.dump(manifest, f, indent=2)
print(f"Manifest saved to {out_path} ({len(files)} files)")
PY
```

### 2. `train.py` — CDN-only data loader
```python
import json, os, requests, hashlib
from pathlib import Path
from torch.utils.data import IterableDataset

MANIFEST_PATH = os.getenv("MANIFEST_PATH", "training_manifest.json")
SIBLING_REPOS = [
    "axentx/surrogate-dataset",
    "axentx/surrogate-dataset-s1",
    "axentx/surrogate-dataset-s2",
    "axentx/surrogate-dataset-s3",
    "axentx/surrogate-dataset-s4",
    "axentx/surrogate-dataset-s5",
]

def pick_sibling(slug: str) -> str:
    """Deterministic sibling repo by hash."""
    h = int(hashlib.sha256(slug.encode()).hexdigest(), 16)
    return SIBLING_REPOS[h % len(SIBLING_REPOS)]

def cdn_url(repo_id: str, date_folder: str, filename: str) -> str:
    return f"https://huggingface.co/datasets/{repo_id}/resolve/main/{date_folder}/{filename}"

class CDNParquetDataset(IterableDataset):
    def __init__(self, manifest_path=MANIFEST_PATH, max_retries=3):
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.repo_id = self.manifest["repo_id"]
        self.date_folder = self.manifest["date_folder"]
        self.files = [f for f in self.manifest["files"] if f.endswith(".parquet")]
        self.max_retries = max_retries

    def _stream_file(self, url: str):
        for attempt in range(self.max_retries):
            try:
                with requests.get(url, stream=True, timeout=30) as r:
                    r.raise_for_status()
                    # stream to temp file or direct pyarrow read
                    import tempfile, pyarrow.parquet as pq
                    with tempfile.NamedTemporaryFile(suffix=".parquet", delete=False) as tmp:
                        for chunk in r.iter_content(chunk_size=8192):
                            tmp.write(chunk)
                        tmp_path = tmp.name
                    table = pq.read_table(tmp_path, columns=["prompt", "response"])
                    os.unlink(tmp_path)
                    return table
            except Exception as e:
                if attempt == self.max_retries - 1:
                    raise
                # simple backoff
                import time; time.sleep(2 ** attempt)

    def __iter__(self):
        for fname in self.files:
            slug = fname.removesuffix(".parquet")
            repo = pick_sibling(slug)
            url = cdn_url(repo, self.date_folder, fname)
            table = self._stream_file(url)
            for row in table.to_pylist():
                # project to {prompt, response} only
                yield {
                    "prompt": row.get("prompt", ""),
                    "response": row.get("response", "")
                }
```

### 3. Lightning Studio lifecycle guard
```python
from lightning import Studio, Machine, Teamspace
import time

STUDIO_NAME = "surrogate-train-studio"
MACHINE = Machine.L40S

def get_or_start_studio():
    ts = Teamspace()
    for s in ts.studios:
        if s.name == STUDIO_NAME:
            if s.status == "running":
                return s
            elif s.status == "stopped":
                print(f"Restarting stopped studio: {STUDIO_NAME}")
                s.start(machine=MACHINE)
                # wait until running
                while s.status != "running":
                    time.sleep(10)
                    s.refresh()
                return s
    # create if not exists
    studio = Studio(
        name=STUDIO_NAME,
        machine=MACHINE,
        create_ok=True
    )
    return studio

def run_training_job(script_path="train.py", args=None):
    studio = get_or_start_studio()
    # ensure studio is running before submit
    if studio.status != "running":
        raise RuntimeError(f"Studio {STUDIO_NAME} not running (status={studio.status})")

    run = studio.run(
        script_path,
        arguments=args or [],
        enable_pip_install_on_build=True
    )
    return run
```
