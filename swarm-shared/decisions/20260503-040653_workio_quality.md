# workio / quality

## 1. Diagnosis

- Clock-in/out race conditions: rapid taps or retries can create duplicate time records because there is no per-user, per-day idempotency guard at the DB layer.
- No optimistic UI or pending state on clock actions → perceived latency and duplicate requests when network is slow.
- Missing location validation resilience: if GPS fails or user denies permission, the frontend blocks the action instead of offering fallback (manual confirm with audit note).
- Clock API does not return the created record with full timestamps and status → frontend cannot optimistically update or show confirmation details.
- No idempotency key in request/response contract → retries (LINE webhook retries, user double-taps) are not safe.

## 2. Proposed change

- **Files**:  
  - `server/src/routes/clock.ts` (add idempotency + return created record)  
  - `server/src/db/queries/clockQueries.ts` (add `getOrCreateClockRecord` with `ON CONFLICT DO NOTHING` + select)  
  - `src/features/clock/ClockButton.tsx` (add optimistic UI + disabled/pending state + idempotency key)  
- **Scope**: minimal API contract extension and frontend behavior; no schema changes (uses existing unique constraint on `user_id, date, type`).

## 3. Implementation

### Backend: add idempotent upsert and return record

```ts
// server/src/db/queries/clockQueries.ts
import { pool } from '../db';

export interface ClockRecord {
  id: number;
  user_id: number;
  date: string;       // YYYY-MM-DD
  type: 'IN' | 'OUT';
  timestamp: Date;
  latitude: number | null;
  longitude: number | null;
  created_at: Date;
}

export async function upsertClockRecord(
  userId: number,
  date: string,
  type: 'IN' | 'OUT',
  latitude: number | null,
  longitude: number | null,
  idempotencyKey?: string
): Promise<ClockRecord> {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');
    // If idempotencyKey provided, try to fetch existing by key first
    if (idempotencyKey) {
      const existing = await client.query<ClockRecord>(
        `SELECT id, user_id, date, type, timestamp, latitude, longitude, created_at
         FROM clock_records
         WHERE user_id = $1 AND date = $2 AND type = $3
         ORDER BY created_at DESC LIMIT 1`,
        [userId, date, type]
      );
      if (existing.rows[0]) {
        await client.query('COMMIT');
        return existing.rows[0];
      }
    }

    // Try insert; if unique violation, select existing
    const insertRes = await client.query<ClockRecord>(
      `INSERT INTO clock_records (user_id, date, type, timestamp, latitude, longitude)
       VALUES ($1, $2, $3, $4, $5, $6)
       ON CONFLICT (user_id, date, type) DO NOTHING
       RETURNING id, user_id, date, type, timestamp, latitude, longitude, created_at`,
      [userId, date, type, new Date(), latitude, longitude]
    );

    if (insertRes.rows[0]) {
      await client.query('COMMIT');
      return insertRes.rows[0];
    }

    const selectRes = await client.query<ClockRecord>(
      `SELECT id, user_id, date, type, timestamp, latitude, longitude, created_at
       FROM clock_records
       WHERE user_id = $1 AND date = $2 AND type = $3
       ORDER BY created_at DESC LIMIT 1`,
      [userId, date, type]
    );

    await client.query('COMMIT');
    if (!selectRes.rows[0]) {
      throw new Error('Failed to retrieve clock record after conflict');
    }
    return selectRes.rows[0];
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  } finally {
    client.release();
  }
}
```

```ts
// server/src/routes/clock.ts
import express from 'express';
import { upsertClockRecord } from '../db/queries/clockQueries';
import { authMiddleware } from './authMiddleware';

const router = express.Router();

router.post('/api/clock', authMiddleware, async (req, res) => {
  try {
    const { type, latitude, longitude, idempotencyKey } = req.body as {
      type: 'IN' | 'OUT';
      latitude?: number | null;
      longitude?: number | null;
      idempotencyKey?: string;
    };
    const userId = (req as any).user.id;
    const today = new Date().toISOString().split('T')[0];

    const record = await upsertClockRecord(
      userId,
      today,
      type,
      latitude ?? null,
      longitude ?? null,
      idempotencyKey
    );

    return res.json({
      ok: true,
      data: {
        id: record.id,
        type: record.type,
        timestamp: record.timestamp,
        latitude: record.latitude,
        longitude: record.longitude,
        created_at: record.created_at,
      },
    });
  } catch (err) {
    console.error('Clock error:', err);
    return res.status(500).json({ ok: false, error: 'Internal server error' });
  }
});

export default router;
```

### Frontend: optimistic UI + idempotency

```tsx
// src/features/clock/ClockButton.tsx
import { useState } from 'react';
import { useLocation } from './useLocation';

type ClockType = 'IN' | 'OUT';

export function ClockButton({ type }: { type: ClockType }) {
  const [pending, setPending] = useState(false);
  const [lastRecord, setLastRecord] = useState<{
    id: number;
    timestamp: string;
    type: 'IN' | 'OUT';
  } | null>(null);
  const { getLocation, permissionDenied } = useLocation();

  const handleClock = async () => {
    if (pending) return;
    setPending(true);

    let latitude: number | null = null;
    let longitude: number | null = null;
    try {
      const loc = await getLocation();
      if (loc) {
        latitude = loc.latitude;
        longitude = loc.longitude;
      }
    } catch (err) {
      // continue without location; backend accepts nulls
      console.warn('Location unavailable', err);
    }

    const idempotencyKey = `${Date.now()}-${Math.random().toString(36).slice(2)}`;
    const optimisticTs = new Date().toISOString();

    // optimistic update
    setLastRecord({
      id: -1,
      timestamp: optimisticTs,
      type,
    });

    try {
      const res = await fetch('/api/clock', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ type, latitude, longitude, idempotencyKey }),
      });
      const json = await res.json();
      if (json.ok && json.data) {
        setLastRecord({
          id: json.data.id,
          timestamp: json.data.timestamp,
          type: json.data.type,
        });
      } else {
        // revert optimistic on failure
        setLastRecord(null);
      }
    } catch (err) {
      setLastRecord(null);
      console.error('Clock request failed', err);
    } finally {
      setPending(false);
    }
  };

  return (
    <div className="space-y-2">
      <button
        onClick={handleClock}
        disabled={pending}
        className={`px-6 py-3 rounded-lg font-medium text-white transition ${
          pending
            ? 'bg-gray-400 cursor-wait'
            : type === 'IN'
            ? 'bg-green-600 hover:bg-green-700'
            : 'bg-red-600 hover:bg-red-700'
        }`}
      >
        {pending
          ? 'กำลังบันทึก...'
          : type === 'IN'
          ? 'Clock In'
          : 'Clock Out'}
      </button
