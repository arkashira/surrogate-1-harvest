**Highest-value incremental improvement (≤2h)**  
Harden `airship discover` into a deterministic, CDN-safe, zero-runtime orchestrator that eliminates HF API rate limits and PyArrow schema errors while keeping ingestion reproducible and training scripts CDN-only.

**Why this wins**  
- Removes HF API calls during training (bypasses 429/128-hr cap).  
- Eliminates `load_dataset(streaming=True)` schema explosions by projecting `{prompt,response}` at parse time.  
- Makes `discover` idempotent and deterministic (date-folder → file-list JSON → embed in train).  
- Fits <2h: single orchestrator change + small ingestion/training guardrails.

```python
#!/usr/bin/env python3
"""
airship discover
Deterministic, CDN-safe file lister for HF datasets.
Writes file-list JSON for downstream training (zero API calls during training).
"""
import argparse
import json
import sys
from pathlib import Path
from typing import List, Dict

import requests
from huggingface_hub import HfApi  # corrected import

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"


def list_date_folder(repo: str, date_folder: str) -> List[Dict]:
    api = HfApi()
    entries = api.list_repo_tree(repo=repo, path=date_folder, recursive=False)
    files = [e for e in entries if e.type == "file"]
    out = []
    for f in files:
        url = CDN_TEMPLATE.format(repo=repo, path=f"{date_folder}/{f.path}")
        out.append(
            {
                "path": f"{date_folder}/{f.path}",
                "size": f.size or 0,
                "url": url,
                "lfs": getattr(f, "lfs", None),
            }
        )
    return out


def validate_cdn_urls(files: List[Dict], timeout: float = 5.0) -> List[Dict]:
    ok = []
    for f in files:
        try:
            r = requests.head(f["url"], timeout=timeout, allow_redirects=True)
            if r.status_code == 200:
                ok.append(f)
            else:
                print(f"WARN: CDN unavailable for {f['path']} ({r.status_code})", file=sys.stderr)
        except Exception as exc:
            print(f"WARN: CDN check failed for {f['path']}: {exc}", file=sys.stderr)
    return ok


def write_manifest(files: List[Dict], out_dir: Path):
    out_dir.mkdir(parents=True, exist_ok=True)
    (out_dir / "files.json").write_text(json.dumps([f["path"] for f in files], indent=2))
    (out_dir / "manifest.json").write_text(json.dumps(files, indent=2))


def main():
    parser = argparse.ArgumentParser(description="Discover dataset files for a date folder (CDN-safe).")
    parser.add_argument("--repo", required=True, help="HF dataset repo, e.g. org/name")
    parser.add_argument("--date", required=True, help="Date folder, e.g. 2026-05-02")
    parser.add_argument("--out-dir", default="discover_out", help="Output directory")
    parser.add_argument("--skip-cdn-check", action="store_true")
    args = parser.parse_args()

    try:
        files = list_date_folder(args.repo, args.date)
        if not files:
            print(f"No files found in {args.repo}/{args.date}", file=sys.stderr)
            sys.exit(1)

        if not args.skip_cdn_check:
            files = validate_cdn_urls(files)
            if not files:
                print("No files passed CDN validation.", file=sys.stderr)
                sys.exit(1)

        write_manifest(files, Path(args.out_dir))
        print(f"Discovered {len(files)} files -> {args.out_dir}")
    except Exception as exc:
        print(f"Discover failed: {exc}", file=sys.stderr)
        sys.exit(1)


if __name__ == "__main__":
    main()
```

### 2) Ingestion projection (parquet writer)

```python
def project_and_save(raw_path: str, out_dir: Path, date: str):
    """
    Read raw file (jsonl/parquet/csv), keep only {prompt,response}, write parquet.
    """
    import pandas as pd
    from slugify import slugify

    if raw_path.endswith(".parquet"):
        df = pd.read_parquet(raw_path)
    elif raw_path.endswith(".jsonl"):
        df = pd.read_json(raw_path, lines=True)
    else:
        df = pd.read_csv(raw_path)

    # Keep only required fields; tolerate different column names
    prompt_col = next((c for c in df.columns if "prompt" in c.lower()), None)
    response_col = next((c for c in df.columns if "response" in c.lower()), None)

    if not prompt_col or not response_col:
        raise ValueError(f"Missing prompt/response in {raw_path}")

    projected = df[[prompt_col, response_col]].rename(columns={prompt_col: "prompt", response_col: "response"})
    # drop rows with nulls
    projected = projected.dropna(subset=["prompt", "response"]).reset_index(drop=True)

    slug = slugify(Path(raw_path).stem)
    out_path = out_dir / f"batches/mirror-merged/{date}/{slug}.parquet"
    out_path.parent.mkdir(parents=True, exist_ok=True)
    projected.to_parquet(out_path, index=False)
    return out_path
```
