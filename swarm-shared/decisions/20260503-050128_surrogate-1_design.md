# surrogate-1 / design

## Final Implementation Plan (≤2h)

**Highest-value improvement**: Replace fragile shell-based ingestion with a **manifest-first, CDN-bypass Python worker** that eliminates HF API rate limits during training data loads and prevents mixed-schema `CastError`.

### Changes

1. **`bin/dataset-enrich.sh`** → **`bin/dataset-enrich.py`**
   - Single API call (post-rate-limit window) to `list_repo_tree` for the target date folder; save manifest JSON.
   - Deterministic shard assignment: `hash(slug) % 16 == SHARD_ID`.
   - Per-file CDN download via `https://huggingface.co/datasets/.../resolve/main/...` (no auth, no API rate limit).
   - Stream-parse each file into `{prompt, response}` only; drop all other columns to avoid mixed-schema `CastError`.
   - Compute per-row md5 for dedup; skip if already in central store.
   - Write output as newline JSON (`.jsonl`) to `batches/public-merged/{date}/shard{N}-{HHMMSS}.jsonl`.
   - Preserve existing behavior: 16 parallel GitHub Actions runners, no cross-run state.

2. **`lib/dedup.py`**
   - Keep SQLite-backed central md5 store; add `seen(md5: str) -> bool` and `add(md5: str) -> None`.
   - Lightweight connection pooling so each worker process can check/add safely.

3. **`.github/workflows/ingest.yml`**
   - Update matrix runner to invoke `python bin/dataset-enrich.py` instead of shell script.
   - Pass `SHARD_ID`, `DATE`, `HF_TOKEN` as env vars.
   - Keep 16-shard matrix; no other changes.

4. **`requirements.txt`**
   - Add `requests` (for CDN bypass) if not present; keep existing deps.

#!/usr/bin/env python3
"""
Manifest-first, CDN-bypass ingestion worker for surrogate-1.
Usage: SHARD_ID=0 DATE=2026-05-03 python bin/dataset-enrich.py
"""

import os
import sys

from pathlib import Path

import pyarrow as pa
import pyarrow.parquet as pq
import requests
from huggingface_hub import list_repo_tree, upload_file

BASE_CDN = f"https://huggingface.co/datasets/{REPO}/resolve/main"

SHARD_ID = int(os.getenv("SHARD_ID", 0))
DATE = os.getenv("DATE", datetime.now(timezone.utc).strftime("%Y-%m-%d"))
HF_TOKEN = os.getenv("HF_TOKEN")
if not HF_TOKEN:
    print("HF_TOKEN required", file=sys.stderr)
    sys.exit(1)

HEADERS = {"Authorization": f"Bearer {HF_TOKEN}"}
OUTPUT_DIR = Path(f"batches/public-merged/{DATE}")
OUTPUT_DIR.mkdir(parents=True, exist_ok=True)
TIMESTAMP = datetime.now(timezone.utc).strftime("%H%M%S")
OUTFILE = OUTPUT_DIR / f"shard{SHARD_ID}-{TIMESTAMP}.jsonl"

# Central dedup store (shared across runners via mounted volume or HF Space SQLite)
sys.path.insert(0, str(Path(__file__).parent))
from lib.dedup import DedupStore

dedup = DedupStore()

def list_date_files() -> list[str]:
    """Single API call: list files for the target date folder only."""
    tree = list_repo_tree(REPO, path=DATE, recursive=False, token=HF_TOKEN)
    files = [item.path for item in tree if item.type == "file"]
    return files

def belongs_to_shard(path: str) -> bool:
    slug = path
    return (int(hashlib.md5(slug.encode()).hexdigest(), 16) % 16) == SHARD_ID

def cdn_url(path: str) -> str:
    return f"{BASE_CDN}/{path}"

def parse_file_to_rows(path: str):
    """
    Download via CDN (no auth) and project to {prompt,response} only.
    Supports common formats: jsonl, parquet, json.
    """
    url = cdn_url(path)
    resp = requests.get(url, timeout=60)
    resp.raise_for_status()
    data = resp.content

    suffix = Path(path).suffix.lower()
    try:
        if suffix == ".parquet":
            table = pq.read_table(pa.BufferReader(data))
            # Project only required columns; tolerate missing/extra
            cols = set(table.column_names)
            prompt_col = next((c for c in ("prompt", "input", "question") if c in cols), None)
            response_col = next((c for c in ("response", "output", "answer") if c in cols), None)
            if prompt_col is None or response_col is None:
                # fallback: first two string/text columns
                text_cols = [c for c in cols if table.schema.field(c).type in (pa.string(), pa.large_string())]
                if len(text_cols) >= 2:
                    prompt_col, response_col = text_cols[0], text_cols[1]
                else:
                    return  # skip
            prompts = table.column(prompt_col).to_pylist()
            responses = table.column(response_col).to_pylist()
            for p, r in zip(prompts, responses):
                if isinstance(p, str) and isinstance(r, str) and p.strip() and r.strip():
                    yield {"prompt": p.strip(), "response": r.strip()}
        else:
            # jsonl or json
            if suffix == ".jsonl":
                lines = data.decode("utf-8").splitlines()
                records = (json.loads(l) for l in lines if l.strip())
            else:
                records = json.loads(data.decode("utf-8"))
                if not isinstance(records, list):
                    records = [records]
            for rec in records:
                if not isinstance(rec, dict):
                    continue
                p = rec.get("prompt") or rec.get("input") or rec.get("question")
                r = rec.get("response") or rec.get("output") or rec.get("answer")
                if isinstance(p, str) and isinstance(r, str) and p.strip() and r.strip():
                    yield {"prompt": p.strip(), "response": r.strip()}
    except Exception as exc:
        print(f"Parse error {path}: {exc}", file=sys.stderr)
        return

def row_md5(row: dict) -> str:
    payload = f"{row['prompt']}\0{row['response']}".encode("utf-8")
    return hashlib.md5(payload).hexdigest()

def main():
    files = list_date_files()
    shard_files = [f for f in files if belongs_to_shard(f)]
    print(f"Shard {SHARD_ID}: processing {len(shard_files)} files for {DATE}", file=sys.stderr)

    written = 0
    skipped_dup = 0
    skipped_parse = 0

    with OUTFILE.open("w", encoding="utf-8") as fout:
        for path in shard_files:
            for row in parse_file_to_rows(path):
                md5 = row_md5(row)
                if dedup.seen(md5):
                    skipped_dup += 1
                    continue
                dedup.add(md5)
                fout.write(json.dumps(row, ensure_ascii=False) + "\n")
                written += 1

    print(
        f"Shard {SHARD_ID} done: written={written} dup_skipped={skipped_dup} parse_skipped={skipped_parse} -> {OUTFILE}",
        file=sys.stderr,
    )

    # Upload via huggingface_hub (one commit per shard per run)
    if written > 0:
        try:
            upload_file(
                path_or_fileobj=str(OUTFILE),
                path_in_repo=str(OUTFILE.relative_to(Path.cwd())),
                repo_id=REPO,
                token=HF_TOKEN,
                commit_message=f"public-merge shard{SHARD_ID} {DATE} {TIMESTAMP}",
            )
            print("Upload succeeded", file=sys.stderr)
        except Exception as exc:
            print(f"Upload failed: {exc}", file=sys.stderr)
            sys.exit(1)
    else:
        print("No rows to upload", file=sys.stderr)
