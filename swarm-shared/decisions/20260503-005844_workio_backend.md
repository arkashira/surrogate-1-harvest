# workio / backend

## Final consolidated solution

**Core principle**: enforce “one open punch per user” at the database level, make endpoints safe under concurrency and LINE retries, and keep changes minimal + production-ready.

---

### 1. Database schema (migration-safe)

Add audit columns if missing, then create a **partial unique index** that ignores soft-deleted rows. Prefer a migration file; if your project applies schema directly, place this in `src/db/schema.sql` and run it once.

```sql
-- Ensure audit columns exist (run conditionally or via migration)
ALTER TABLE punch
  ADD COLUMN IF NOT EXISTS created_at timestamptz NOT NULL DEFAULT now(),
  ADD COLUMN IF NOT EXISTS updated_at timestamptz NOT NULL DEFAULT now(),
  ADD COLUMN IF NOT EXISTS deleted_at timestamptz;

-- One open punch per user (ignores soft-deleted rows)
-- This is the authoritative guardrail.
CREATE UNIQUE INDEX IF NOT EXISTS idx_punch_one_open_per_user
  ON punch (user_id)
  WHERE clock_out_at IS NULL AND deleted_at IS NULL;

-- Optional: fast lookup for clock-out and housekeeping
CREATE INDEX IF NOT EXISTS idx_punch_user_open
  ON punch (user_id)
  WHERE clock_out_at IS NULL AND deleted_at IS NULL;
```

If you use migrations, create a timestamped migration (e.g. `migrations/20250601000000_one_open_punch_per_user.sql`) with the same statements and run it.

### 2. Idempotency (lightweight, optional but recommended)

Use a simple idempotency table keyed by a hash of the client-provided idempotency key (or LINE `deliveryId`/event hash). Keep TTL short (5–10 minutes) and store only the response for replay.

```sql
CREATE TABLE IF NOT EXISTS idempotency_key (
  key_hash      text PRIMARY KEY,
  response_payload jsonb,
  created_at    timestamptz NOT NULL DEFAULT now(),
  last_seen_at  timestamptz,
  expires_at    timestamptz NOT NULL
);

-- Optional cleanup job (run periodically)
-- DELETE FROM idempotency_key WHERE expires_at < now();
```

Middleware (Express):

```ts
// src/middleware/idempotency.ts
import { Request, Response, NextFunction } from 'express';
import crypto from 'crypto';
import { getClient } from '../db';

const TTL_MS = 5 * 60 * 1000; // 5 minutes

export async function idempotencyGuard(req: Request, res: Response, next: NextFunction) {
  const key = req.header('Idempotency-Key') || req.body.idempotency_key;
  if (!key) return next();

  const hash = crypto.createHash('sha256').update(key).digest('hex');
  const client = getClient();
  const now = new Date();
  const expiresAt = new Date(now.getTime() + TTL_MS);

  try {
    const { rows } = await client.query(
      `INSERT INTO idempotency_key (key_hash, created_at, expires_at)
       VALUES ($1, $2, $3)
       ON CONFLICT (key_hash) DO UPDATE
       SET last_seen_at = $2
       WHERE idempotency_key.expires_at > $2
       RETURNING response_payload`,
      [hash, now, expiresAt]
    );

    // If we have a cached response, return it immediately
    if (rows.length && rows[0].response_payload) {
      return res.json(rows[0].response_payload);
    }

    (req as any).idempotencyKey = hash;
    next();
  } catch (err) {
    // On conflict+expired or other errors, continue without idempotency cache
    (req as any).idempotencyKey = hash;
    next();
  }
}
```

### 3. Atomic punch endpoints

Use a single transaction and `INSERT ... ON CONFLICT` for clock-in. For clock-out, update the most recent open punch. Both must respect `deleted_at IS NULL`.

```ts
// src/routes/punch.ts
import { Router } from 'express';
import { getClient } from '../db';
import { idempotencyGuard } from '../middleware/idempotency';

const router = Router();

// Clock-in: atomic upsert constrained by partial unique index
router.post('/clock-in', idempotencyGuard, async (req, res) => {
  const { user_id, latitude, longitude, line_user_id } = req.body;
  const client = getClient();
  const now = new Date();

  try {
    await client.query('BEGIN');

    const result = await client.query(
      `INSERT INTO punch (user_id, clock_in_at, latitude, longitude, line_user_id, created_at, updated_at)
       VALUES ($1, $2, $3, $4, $5, $6, $6)
       ON CONFLICT (user_id)
       WHERE clock_out_at IS NULL AND deleted_at IS NULL
       DO UPDATE SET updated_at = EXCLUDED.updated_at
       RETURNING *`,
      [user_id, now, latitude, longitude, line_user_id, now]
    );

    await client.query('COMMIT');

    // Store idempotency response if key provided
    if ((req as any).idempotencyKey) {
      await client.query(
        `UPDATE idempotency_key SET response_payload = $1 WHERE key_hash = $2`,
        [result.rows[0], (req as any).idempotencyKey]
      ).catch(() => {});
    }

    return res.json({ ok: true, punch: result.rows[0] });
  } catch (err: any) {
    await client.query('ROLLBACK').catch(() => {});
    // Unique violation from partial index => one-open rule enforced
    if (err.code === '23505') {
      // Return existing open punch for idempotency-like behavior
      const { rows } = await client.query(
        `SELECT * FROM punch
         WHERE user_id = $1 AND clock_out_at IS NULL AND deleted_at IS NULL
         ORDER BY clock_in_at DESC LIMIT 1`,
        [user_id]
      );
      if (rows.length) {
        return res.json({ ok: true, punch: rows[0] });
      }
    }
    console.error('clock-in failed', err);
    return res.status(500).json({ ok: false, error: 'clock-in failed' });
  } finally {
    client.release();
  }
});

// Clock-out: close the most recent open punch
router.post('/clock-out', async (req, res) => {
  const { user_id, latitude, longitude } = req.body;
  const client = getClient();
  const now = new Date();

  try {
    await client.query('BEGIN');

    const result = await client.query(
      `UPDATE punch
       SET clock_out_at = $1,
           updated_at = $1,
           latitude_out = $2,
           longitude_out = $3
       WHERE id = (
         SELECT id FROM punch
         WHERE user_id = $4 AND clock_out_at IS NULL AND deleted_at IS NULL
         ORDER BY clock_in_at DESC
         LIMIT 1
       )
       RETURNING *`,
      [now, latitude, longitude, user_id]
    );

    await client.query('COMMIT');

    if (result.rowCount === 0) {
      return res.status(404).json({ ok: false, error: 'no open punch to close' });
    }
    return res.json({ ok: true, punch: result.rows[0] });
  } catch (err) {
    await client.query('ROLLBACK').catch(() => {});
    console.error('clock-out failed', err);
    return res.status(500).json({ ok: false, error: 'clock-out failed' });
  } finally {
    client.release();
  }
});

export default router;
```
