```python
#!/usr/bin/env python3
"""
List files in a HuggingFace dataset folder (non-recursive) for a given date.
Run from Mac after rate-limit window clears. Saves file_list.json for training.
"""
import argparse
import json
import os
from huggingface_hub import HfApi

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True, help="e.g., axentx/surrogate-data")
    parser.add_argument("--date", required=True, help="Date folder, e.g. 2026-04-29")
    parser.add_argument("--out", default="file_list.json")
    args = parser.parse_args()

    api = HfApi()
    tree = api.list_repo_tree(repo_id=args.repo, path=args.date, recursive=False)
    files = [item.rfilename for item in tree if item.type == "file"]

    out_path = os.path.join(os.path.dirname(__file__), "..", "surrogate", "ingest", args.out)
    with open(out_path, "w") as f:
        json.dump({"repo": args.repo, "date": args.date, "files": files}, f, indent=2)
    print(f"Wrote {len(files)} files to {out_path}")

if __name__ == "__main__":
    main()
```

#### 2. `surrogate/ingest/cdn_loader.py`
```python
#!/usr/bin/env python3
"""
Download dataset files via public CDN (no auth) and project to {prompt, response}.
"""
import json
import time
from pathlib import Path
from typing import Iterator, Dict
import requests
import pyarrow as pa
import pyarrow.parquet as pq
import pandas as pd
from tqdm import tqdm

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def download_with_retry(url: str, max_retries: int = 3) -> bytes:
    for attempt in range(max_retries):
        try:
            resp = requests.get(url, timeout=30, stream=True)
            if resp.status_code == 429:
                wait = 360
                print(f"Rate-limited (429). Waiting {wait}s")
                time.sleep(wait)
                continue
            resp.raise_for_status()
            return resp.content
        except requests.RequestException as e:
            if attempt == max_retries - 1:
                raise
            sleep_sec = 2 ** attempt
            print(f"Download failed ({e}), retry in {sleep_sec}s")
            time.sleep(sleep_sec)
    raise RuntimeError("Max retries exceeded")

def project_row(raw: Dict) -> Dict[str, str]:
    """
    Project heterogeneous schema to {prompt, response}.
    Adjust mapping per dataset convention if needed.
    """
    prompt = raw.get("prompt") or raw.get("input") or raw.get("question") or ""
    response = raw.get("response") or raw.get("output") or raw.get("answer") or ""
    return {"prompt": str(prompt), "response": str(response)}

def stream_cdn_files(file_list_path: Path) -> Iterator[Dict[str, str]]:
    with open(file_list_path) as f:
        meta = json.load(f)

    repo = meta["repo"]
    date = meta["date"]
    for fname in tqdm(meta["files"], desc="Downloading"):
        path = f"{date}/{fname}"
        url = CDN_TEMPLATE.format(repo=repo, path=path)
        try:
            data = download_with_retry(url)
            # Try parquet first
            try:
                table = pq.read_table(pa.BufferReader(data))
                df = table.to_pandas()
            except Exception:
                # Fallback: newline JSON
                df = pd.read_json(data, lines=True, dtype=str)

            for _, row in df.iterrows():
                yield project_row(row.to_dict())
        except Exception as e:
            print(f"Skipping {path} due to error: {e}")
            continue
```

#### 3. `surrogate/ingest/shard_writer.py`
```python
#!/usr/bin/env python3
"""
Deterministic sharding across sibling repos to bypass HF commit cap.
"""
import hashlib
import pandas as pd
import pyarrow as pa
import pyarrow.parquet as pq
from typing import List, Dict
from huggingface_hub import HfApi

SIBLINGS = [
    "axentx/surrogate-data-00",
    "axentx/surrogate-data-01",
    "axentx/surrogate-data-02",
    "axentx/surrogate-data-03",
    "axentx/surrogate-data-04",
]

def pick_repo(slug: str) -> str:
    h = hashlib.sha256(slug.encode()).hexdigest()
    idx = int(h, 16) % len(SIBLINGS)
    return SIBLINGS[idx]

def upload_sharded(date: str, slug: str, rows: List[Dict], api: HfApi):
    repo = pick_repo(slug)
    df = pd.DataFrame(rows)
    # Ensure schema
    df = df[["prompt", "response"]].astype(str)
    out_path = f"batches/mirror-merged/{date}/{slug}.parquet"
    buf = pa.BufferWriter()
    pq.write_table(pa.Table.from_pandas(df), buf)
    api.upload_file(
        path_or_fileobj=buf.getvalue(),
        path_in_repo=out_path,
        repo_id=repo,
    )
    print(f"Uploaded {len(rows)} rows to {repo}/{out_path}")
```
