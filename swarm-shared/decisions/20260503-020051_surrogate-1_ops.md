# surrogate-1 / ops

## Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID` / `SHARD_TOTAL` from the matrix.
- Loads a pre-generated `manifest-YYYYMMDD.json` (produced once per day by a Mac orchestrator after `list_repo_tree`) containing public file paths for that date.
- Assigns each file to a deterministic shard: `hash(slug) % SHARD_TOTAL == SHARD_ID`.
- Downloads assigned files via **HF CDN bypass** (`https://huggingface.co/datasets/.../resolve/main/...`) with no Authorization header (avoids `/api/` 429).
- Projects each file to `{prompt, response}` only at parse time (avoids pyarrow CastError on mixed schemas).
- Deduplicates via the existing `lib/dedup.py` central md5 store.
- Emits normalized pairs to `batches/public-merged/<date>/shard<N>-<HHMMSS>.jsonl` (one file per shard run) and streams upload to HF dataset repo via `huggingface_hub` (one commit per shard).
- Logs progress and exits cleanly; GitHub Actions matrix runs 16 of these in parallel.

### Why this is the highest-value incremental (<2h)
- Directly applies the **HF CDN bypass** and **manifest pre-list** patterns (eliminates API rate limits during training).
- Reuses existing dedup logic and output layout (no schema changes).
- Small, focused Python script (~120 lines) + tiny workflow tweak = safe, fast, and immediately improves ingestion reliability.

#!/usr/bin/env python3
"""
surrogate-1 CDN-bypass ingestion worker (manifest-driven).

Usage (GH Actions matrix):
  SHARD_ID=0 SHARD_TOTAL=16 python bin/dataset-enrich.py \
    --repo axentx/surrogate-1-training-pairs \
    --date 20260503 \
    --out-dir batches/public-merged

Environment:
  HF_TOKEN         write token for uploads
  MANIFEST_URL     optional; defaults to repo's manifest-YYYYMMDD.json at CDN
"""

import os
import sys
import json
import hashlib
import datetime
import argparse
from pathlib import Path
from typing import List, Dict, Any

import requests
import pyarrow as pa
import pyarrow.parquet as pq
from huggingface_hub import HfApi, hf_hub_download

# Local dedup helper
sys.path.insert(0, str(Path(__file__).parent.parent))
from lib.dedup import DedupStore  # type: ignore

HF_DATASETS_CDN = "https://huggingface.co/datasets"

def _iso_date() -> str:
    return datetime.datetime.utcnow().strftime("%Y%m%d")

def _slug_from_path(repo: str, path: str) -> str:
    """Deterministic slug for dedup/sharding."""
    return f"{repo}/{path}"

def _shard_for(slug: str, total: int) -> int:
    return int(hashlib.md5(slug.encode()).hexdigest(), 16) % total

def load_manifest(repo: str, date: str, manifest_url: str | None = None) -> List[str]:
    """
    Load manifest JSON listing public file paths for date.
    Expected manifest format:
      { "date": "20260503", "files": ["folder/file1.parquet", ...] }
    """
    if manifest_url:
        resp = requests.get(manifest_url, timeout=30)
        resp.raise_for_status()
        data = resp.json()
    else:
        # CDN fallback: repo-root manifest file
        url = f"{HF_DATASETS_CDN}/{repo}/resolve/main/manifest-{date}.json"
        resp = requests.get(url, timeout=30)
        resp.raise_for_status()
        data = resp.json()

    files = data.get("files") or data.get("paths") or []
    if not isinstance(files, list):
        raise ValueError("Manifest must contain 'files' or 'paths' list.")
    return files

def cdn_download(repo: str, path: str, dest: Path) -> Path:
    """Download via HF CDN (no auth). Returns local path."""
    url = f"{HF_DATASETS_CDN}/{repo}/resolve/main/{path}"
    resp = requests.get(url, timeout=120)
    resp.raise_for_status()
    dest.parent.mkdir(parents=True, exist_ok=True)
    dest.write_bytes(resp.content)
    return dest

def project_to_pair(local_path: Path) -> List[Dict[str, str]]:
    """
    Project heterogeneous file to {prompt,response} pairs.
    Supports:
      - JSONL with 'prompt'/'response' (or 'instruction'/'output')
      - Parquet with any schema: tries to find text columns
    """
    suffix = local_path.suffix.lower()
    if suffix == ".jsonl":
        pairs = []
        for line in local_path.read_text().splitlines():
            if not line.strip():
                continue
            obj = json.loads(line)
            prompt = obj.get("prompt") or obj.get("instruction") or ""
            response = obj.get("response") or obj.get("output") or ""
            if prompt and response:
                pairs.append({"prompt": str(prompt), "response": str(response)})
        return pairs

    if suffix == ".parquet":
        # Avoid mixed-schema issues: read only string columns, then try to pair
        table = pq.read_table(local_path, columns=None)
        # Heuristic: pick two text-ish columns
        col_names = [n for n in table.column_names if pa.types.is_string(table.schema.field(n).type)]
        if len(col_names) >= 2:
            prompts = table.column(col_names[0]).to_pylist()
            responses = table.column(col_names[1]).to_pylist()
            return [
                {"prompt": str(p), "response": str(r)}
                for p, r in zip(prompts, responses)
                if p and r
            ]
        # Fallback: dump all rows as prompt-only (response empty) — will be filtered later
        rows = table.to_pylist()
        out = []
        for r in rows:
            txt = " ".join(str(v) for v in r.values() if v)
            if txt:
                out.append({"prompt": txt, "response": ""})
        return out

    # Unknown: try to parse as JSON lines again
    return []

def upload_shard(
    api: HfApi,
    repo: str,
    shard_id: int,
    date: str,
    pairs: List[Dict[str, str]],
) -> None:
    """Stream upload shard file to dataset repo."""
    ts = datetime.datetime.utcnow().strftime("%H%M%S")
    filename = f"shard{shard_id}-{ts}.jsonl"
    remote_path = f"batches/public-merged/{date}/{filename}"
    local_path = Path("tmp") / filename
    local_path.parent.mkdir(parents=True, exist_ok=True)
    local_path.write_text("\n".join(json.dumps(p) for p in pairs))

    api.upload_file(
        path_or_fileobj=str(local_path),
        path_in_repo=remote_path,
        repo_id=repo,
        repo_type="dataset",
    )
    local_path.unlink(missing_ok=True)

def main() -> None:
    parser = argparse.ArgumentParser(description="CDN-bypass surrogate-1 ingestion worker")
    parser.add_argument("--repo", default="axentx/surrogate-1-training-pairs")
    parser.add_argument("--date", default=_iso_date())
    parser.add_argument("--out-dir", default="batches/public-merged")
    parser.add_argument("--manifest-url", default=None)
    args = parser.parse_args()

    shard_id = int(os.environ.get("SHARD_ID", 0))
    shard_total = int(os.environ.get("SHARD_TOTAL", 16))
    hf_token = os.environ.get("HF_TOKEN")
    if not hf_token:
        print("ERROR: HF_TOKEN required", file=sys.stderr)
        sys.exit(1)

    api = HfApi(token=hf_token)
    dedup = DedupStore()

    print(f"[shard {shard_id}/{shard_total}] loading manifest for {args.date}")
