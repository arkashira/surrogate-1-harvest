## Highest-Value Incremental Improvement (≤2h)

**What**: Add a CDN-first training slice to Surrogate that:
1. Exposes a tiny API + UI to trigger “list one date folder from HF” (Mac orchestration) → saves `file-list.json`
2. Uses `file-list.json` in Lightning training to fetch via CDN-only URLs (bypasses HF API 429)
3. Projects each file to `{prompt, response}` on-the-fly (avoids mixed-schema `load_dataset`)
4. Reuses a running Lightning Studio instead of recreating (saves quota + prevents idle-timeout churn)

**Why**: Directly eliminates the two highest-risk patterns for Surrogate training (HF API 429 and Lightning idle-timeout quota waste) and can ship in <2h with minimal surface area.

### surrogate/scripts/list_hf_date_folder.py
```python
#!/usr/bin/env python3
"""
List one date folder from HF dataset repo and produce file-list.json for CDN training.
Usage: python list_hf_date_folder.py --repo <repo> --date-folder 2026-05-03 --out file-list.json
"""
import argparse
import json
import os
import sys
from pathlib import Path

from huggingface_hub import list_repo_tree

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def build_file_list(repo: str, date_folder: str) -> list[dict]:
    entries = list_repo_tree(repo_path=repo, path=date_folder, recursive=False)
    files = []
    for e in entries:
        if e.type != "file":
            continue
        repo_path = f"{date_folder}/{e.path}"
        files.append({
            "repo_path": repo_path,
            "cdn_url": CDN_TEMPLATE.format(repo=repo, path=repo_path),
            "filename": os.path.basename(e.path),
        })
    return files

def main() -> None:
    parser = argparse.ArgumentParser(description="List HF date folder for CDN training")
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g., org/surrogate-data)")
    parser.add_argument("--date-folder", required=True, help="Date folder (e.g., 2026-05-03)")
    parser.add_argument("--out", default="file-list.json", help="Output JSON path")
    args = parser.parse_args()

    try:
        files = build_file_list(args.repo, args.date_folder)
        out_path = Path(args.out)
        out_path.write_text(json.dumps({"repo": args.repo, "date_folder": args.date_folder, "files": files}, indent=2))
        print(f"Wrote {len(files)} files to {out_path}")
    except Exception as exc:
        print(f"Error: {exc}", file=sys.stderr)
        sys.exit(1)

if __name__ == "__main__":
    main()
```

### surrogate/data/cdn_dataset.py
```python
import json
import requests
from typing import Dict, Iterator, List

class CDNDataset:
    """
    CDN-only dataset loader that projects files to {prompt, response}.
    Supports JSON/JSONL/parquet (parquet requires pyarrow; kept optional).
    """
    def __init__(self, file_list_path: str):
        with open(file_list_path) as f:
            manifest = json.load(f)
        self.files: List[Dict] = manifest.get("files", [])

    def _download_json(self, url: str) -> dict:
        resp = requests.get(url, timeout=60)
        resp.raise_for_status()
        return resp.json()

    def _project_to_pair(self, obj: dict) -> Dict[str, str]:
        # Heuristic projection; adapt per actual schema
        prompt = obj.get("prompt") or obj.get("input") or obj.get("question") or ""
        response = obj.get("response") or obj.get("output") or obj.get("answer") or ""
        return {"prompt": str(prompt), "response": str(response)}

    def __iter__(self) -> Iterator[Dict[str, str]]:
        for item in self.files:
            url = item["cdn_url"]
            try:
                data = self._download_json(url)
                # Handle JSONL-like list or single object
                if isinstance(data, list):
                    for row in data:
                        yield self._project_to_pair(row)
                else:
                    yield self._project_to_pair(data)
            except Exception as exc:
                # Log and skip bad files to avoid crashing training
                print(f"Skipping {url}: {exc}")
                continue
```

#!/usr/bin/env python3
"""
Lightning training launcher with Studio reuse and idle-timeout resilience.
"""
import time
from lightning import LightningWork, LightningFlow, LightningApp, Machine
from lightning.app import BuildConfig

# Minimal Work that runs training script; adapt entrypoint as needed
class SurrogateTrainWork(LightningWork):
    def __init__(self, file_list_path: str, **kwargs):
        super().__init__(**kwargs)
        self.file_list_path = file_list_path
        self.machine = Machine.L40S  # Free tier fallback; paid tier can use H200 in lightning-lambda-prod

    def run(self):
        # Import here so it only runs inside the Work
        import subprocess, sys, os
        script = os.path.join(os.path.dirname(__file__), "train_cdn.py")
        cmd = [sys.executable, script, "--file-list", self.file_list_path]
        subprocess.run(cmd, check=True)

class SurrogateTrainFlow(LightningFlow):
    def __init__(self):
        super().__init__()
        self.work = None

    def configure_layout(self):
        return [{"name": "Train", "content": "Training running in Lightning Studio."}]

    def run(self):
        if self.work is None:
            # Reuse running studio if exists (pseudo-code; adapt to your org/teamspace)
            # from lightning import Teamspace
            # for s in Teamspace.studios:
            #     if s.name == "surrogate-train" and s.status == "running":
            #         self.work = s  # attach to running
            #         return
            self.work = SurrogateTrainWork(file_list_path="file-list.json", cloud_compute=self.machine)
        if self.work and not self.work.has_started:
            self.work.start()
        if self.work and self.work.has_stopped and not self.work.has_started:
            # Restart if idle-stop killed it
