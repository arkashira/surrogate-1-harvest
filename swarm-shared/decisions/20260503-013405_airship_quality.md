## 1) Backend (surrogate/training/cdn_loader.py)

- Single API call to list one date folder (non-recursive).  
- Save manifest locally; training workers use **CDN-only URLs** (no Authorization header).  
- Reuse running Studio by name; fallback to L40S.

```python
# surrogate/training/cdn_loader.py
import json
from pathlib import Path
from typing import List, Dict
from huggingface_hub import list_repo_tree

def list_date_folder(repo: str, date_folder: str) -> List[Dict]:
    items = list_repo_tree(repo_id=repo, path=date_folder, recursive=False)
    return [{"path": str(p.path), "size": p.size} for p in items if p.size and str(p.path).endswith(".parquet")]

def save_manifest(repo: str, date_folder: str, out_dir: Path) -> Path:
    out_dir.mkdir(parents=True, exist_ok=True)
    files = list_date_folder(repo, date_folder)
    manifest_path = out_dir / f"file_manifest_{date_folder.replace('/', '_')}.json"
    manifest_path.write_text(json.dumps({"repo": repo, "date_folder": date_folder, "files": files}, indent=2))
    return manifest_path

def cdn_url(repo: str, file_path: str) -> str:
    return f"https://huggingface.co/datasets/{repo}/resolve/main/{file_path}"
```

## 2) Training launcher & Studio reuse (surrogate/training/launch_studio.py)

- Reuse pattern: match Studio name and `status == "running"`; restart only if stopped.  
- Pass manifest path to training script.  
- Keep interface simple for Mac/CI orchestrator.

```python
# surrogate/training/launch_studio.py
import os
from pathlib import Path
from lightning import LightningWork, LightningClient, Machine
from surrogate.training.cdn_loader import save_manifest
from surrogate.training.train import train

class SurrogateTrainer(LightningWork):
    def __init__(self, repo: str, date_folder: str, use_cdn_bypass: bool = True):
        super().__init__(machine=Machine.L40S)
        self.repo = repo
        self.date_folder = date_folder
        self.use_cdn_bypass = use_cdn_bypass

    def run(self):
        client = LightningClient()
        studio_name = f"surrogate-{self.repo.rstrip('/').split('/')[-1]}"
        studio = None

        for s in client.teamspace.studios:
            if s.name == studio_name:
                studio = s
                break

        if studio and studio.status == "running":
            print(f"Reusing running studio: {studio.name}")
        elif studio and studio.status == "stopped":
            print(f"Restarting stopped studio: {studio.name}")
            studio.start(machine=Machine.L40S)
        else:
            print(f"Creating new studio: {studio_name}")
            # LightningWork self-hosts; name is set implicitly via app/project config.
            # If explicit naming is required, use LightningApp-level naming or project defaults.

        if self.use_cdn_bypass:
            manifest_path = save_manifest(self.repo, self.date_folder, Path("training"))
            train(manifest_path=str(manifest_path))
        else:
            raise NotImplementedError("Non-CDN path not implemented")
```

## 3) Training script — CDN-only streaming (surrogate/training/train.py)

- Accept `--manifest_json`.  
- Stream parquet via CDN; project to `{prompt,response}`.  
- Avoid `load_dataset` and `list_repo_files` during training.

```python
# surrogate/training/train.py
import argparse
import json
import pyarrow.parquet as pq
import requests
from io import BytesIO
from datasets import Dataset

def stream_parquet_via_cdn(repo: str, file_path: str):
    url = f"https://huggingface.co/datasets/{repo}/resolve/main/{file_path}"
    resp = requests.get(url, timeout=30)
    resp.raise_for_status()
    table = pq.read_table(BytesIO(resp.content))
    cols = [c for c in table.column_names if c in ("prompt", "response")]
    if not cols:
        # fallback: first two text-like columns
        cols = [c for c in table.column_names if str(table.schema.field(c).type).lower() in ("string", "large_string")][:2]
    return table.select(cols).to_pylist()

def train(manifest_path: str):
    manifest = json.loads(open(manifest_path).read())
    repo = manifest["repo"]
    rows = []
    for f in manifest["files"]:
        try:
            rows.extend(stream_parquet_via_cdn(repo, f["path"]))
        except Exception as e:
            print(f"Skipping {f['path']}: {e}")
    ds = Dataset.from_list(rows)
    print(f"Loaded {len(ds)} examples via CDN-bypass")
    # Continue with your training loop (e.g., SFTTrainer)
    # trainer = SFTTrainer(...)
    # trainer.train()

if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--manifest_json", required=True)
    args = parser.parse_args()
    train(args.manifest_json)
```

## 4) Frontend (arkship/ui/src/components/TrainingLauncher.vue)

- Add toggle “Use CDN-bypass (recommended)”.  
- On launch, POST `/api/training/start`.  
- Show reused Studio name and status; auto-refresh every 30 s.

```vue
<template>
  <div>
    <h3>Surrogate Training Launcher</h3>
    <input v-model="repo" placeholder="datasets/username/repo" />
    <input v-model="dateFolder" placeholder="2026-05-03" />
    <label><input type="checkbox" v-model="useCdnBypass" checked /> Use CDN-bypass (recommended)</label>
    <button @click="launch" :disabled="loading">Launch Training</button>
    <div v-if="studioName">Studio: {{ studioName }} ({{ studioStatus }})</div>
  </div>
</template>

<script>
export default {
  data() {
    return {
      repo: "",
      dateFolder: "",
      useCdnBypass: true,
      loading: false,
      studioName: "",
      studioStatus: "",
      pollTimer: null,
    };
  },
  methods: {
    async launch() {
      this.loading = true;
      try {
        const res = await fetch("/api/training/start", {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({
            repo: this.repo,
            date_folder: this.dateFolder,
            use_cdn_bypass: this.useCdnBypass,
          }),
        });
        const data = await res.json();
        this.studioName = data.studio_name;
        this.studioStatus = data.status;
        this.pollStatus(data.studio_name);
      } finally {
        this.loading = false;
      }
    },
    pollStatus(name) {
      if (this.pollTimer) clearInterval(this.pollTimer);
      const tick = async () => {
        try {
          const r = await fetch(`/api/training/status?studio_name=${name}`);
          const d = await r.json();
          this.studioStatus = d.status;
          if (d.status === "running" || d.status === "failed") {
            // keep showing; stop polling only on terminal states if desired
          }
        } catch (e) {
          this.studioStatus = "error";
        }
      };
      tick();
      this.pollTimer = setInterval(tick, 30000);
    },
  },
  beforeUnmount() {
    if (this.pollTimer) clearInterval(this.pollTimer);
  },
};
</script>
```
