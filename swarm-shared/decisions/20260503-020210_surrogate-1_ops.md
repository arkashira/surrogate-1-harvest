Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID` and `SHARD_TOTAL` (16) from the GitHub Actions matrix.
- Loads a pre-generated `manifest-YYYYMMDD.json` (created once on the Mac orchestrator) containing the list of files for a single date folder from the public dataset repo.
- Deterministically assigns each file to a shard by `hash(slug) % SHARD_TOTAL`.
- Downloads only assigned files via **HF CDN bypass** (`https://huggingface.co/datasets/.../resolve/main/...`) — zero API calls during training.
- Projects each file to `{prompt, response}` at parse time (handles mixed schemas per surrogate-1 lesson).
- Deduplicates via the central `lib/dedup.py` md5 store.
- Writes output as `batches/public-merged/YYYYMMDD/shardN-HHMMSS.jsonl` with one JSON object per line.
- Exits with non-zero on fatal errors; logs progress to stdout for Actions.

HF_DATASETS_CDN = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"
RETRY_WAIT = 360  # seconds after 429
MAX_RETRIES = 5

def hash_slug(slug: str) -> int:
    return int(hashlib.sha256(slug.encode("utf-8")).hexdigest(), 16)

def assign_shard(slug: str, total: int) -> int:
    return hash_slug(slug) % total

def download_cdn(url: str, timeout: int = 30) -> bytes:
    for attempt in range(1, MAX_RETRIES + 1):
        try:
            resp = requests.get(url, timeout=timeout, stream=True)
            if resp.status_code == 429:
                wait = RETRY_WAIT
                print(f"[{url}] 429 rate-limited, waiting {wait}s (attempt {attempt})", file=sys.stderr)
                time.sleep(wait)
                continue
            resp.raise_for_status()
            return resp.content
        except requests.RequestException as exc:
            if attempt == MAX_RETRIES:
                raise
            wait = 2 ** attempt
            print(f"[{url}] error: {exc}, retry in {wait}s (attempt {attempt})", file=sys.stderr)
            time.sleep(wait)
    raise RuntimeError(f"Failed to download {url}")

def project_to_pair(raw: Dict[str, Any], filename: str) -> Dict[str, str]:
    """
    Project heterogeneous file schemas to {prompt, response}.
    Follow surrogate-1 pattern: do not add 'source'/'ts' columns.
    """
    # Common patterns seen in surrogate-1 training pairs
    if "prompt" in raw and "response" in raw:
        return {"prompt": str(raw["prompt"]), "response": str(raw["response"])}

    # Alternate field names
    prompt_keys = {"instruction", "input", "question", "prompt_text", "user"}
    response_keys = {"output", "answer", "completion", "response_text", "assistant"}

    p = None
    r = None
    for k in raw:
        if p is None and k in prompt_keys:
            p = raw[k]
        if r is None and k in response_keys:
            r = raw[k]

    if p is not None and r is not None:
        return {"prompt": str(p), "response": str(r)}

    # Fallback: try to split on common separators in text fields
    text_val = None
    for k in ("text", "content", "text_data"):
        if k in raw:
            text_val = raw[k]
            break

    if isinstance(text_val, str):
        # Very simple split; customize per dataset if needed
        parts = text_val.split("\n\n")
        if len(parts) >= 2:
            return {"prompt": parts[0].strip(), "response": parts[1].strip()}

    # Final fallback: encode entire raw as prompt and empty response
    # (will likely be filtered later or by dedup)
    return {"prompt": json.dumps(raw, ensure_ascii=False), "response": ""}
