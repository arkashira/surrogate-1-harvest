# workio / backend

## 1. Diagnosis
- **Non-idempotent LINE webhook handler**: no deduplication key or idempotency check → LINE retries on 5xx/timeouts create duplicate clock/leave/OT events.
- **No transactional upsert or unique constraint**: concurrent retries can insert duplicate rows instead of safely ignoring repeats.
- **Missing request signature verification**: webhook handler does not validate `X-Line-Signature` → accepts spoofed/malformed events.
- **No idempotency key storage**: no `event_id`/`message_id` column to track processed events; retries re-process identical payloads.
- **Race-prone status transitions**: leave/OT approval state changes can be applied multiple times if webhook retries overlap.

## 2. Proposed change
- **Files**:  
  - `server/src/db/schema.sql` (add unique constraint + processed_events table)  
  - `server/src/routes/webhook/line.ts` (add idempotency + signature verification + transactional upsert)  
- **Scope**:  
  - Add `processed_events` table keyed by `line_event_id` (unique).  
  - Wrap clock/leave/OT handling in a transaction that first attempts to insert into `processed_events`; on conflict → skip processing.  
  - Verify `X-Line-Signature` using HMAC-SHA256 before parsing body.

## 3. Implementation

### `server/src/db/schema.sql`
```sql
-- Idempotency table for LINE webhook events
CREATE TABLE IF NOT EXISTS processed_events (
  line_event_id VARCHAR(255) PRIMARY KEY,
  tenant_id     INTEGER NOT NULL,
  event_type    VARCHAR(50) NOT NULL,
  processed_at  TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Ensure clock_records are idempotent by natural key if not using processed_events fallback
ALTER TABLE clock_records
  ADD CONSTRAINT uniq_employee_date_type
  UNIQUE (employee_id, date, clock_type)
  DEFERRABLE INITIALLY DEFERRED;

-- Ensure leave/ot requests have stable external idempotency
ALTER TABLE leave_requests
  ADD COLUMN IF NOT EXISTS request_line_event_id VARCHAR(255),
  ADD CONSTRAINT uniq_line_event_leave UNIQUE (request_line_event_id) DEFERRABLE INITIALLY DEFERRED;

ALTER TABLE ot_requests
  ADD COLUMN IF NOT EXISTS request_line_event_id VARCHAR(255),
  ADD CONSTRAINT uniq_line_event_ot UNIQUE (request_line_event_id) DEFERRABLE INITIALLY DEFERRED;
```

### `server/src/routes/webhook/line.ts`
```ts
import express from 'express';
import crypto from 'crypto';
import { pool } from '../../db';
import { processClockEvent, processLeaveEvent, processOtEvent } from '../../handlers/lineEvents';

const router = express.Router();
const LINE_CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';

function verifySignature(body: string, signature: string): boolean {
  const expected = crypto
    .createHmac('sha256', LINE_CHANNEL_SECRET)
    .update(body, 'utf8')
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

async function isAlreadyProcessed(lineEventId: string): Promise<boolean> {
  const res = await pool.query(
    'SELECT 1 FROM processed_events WHERE line_event_id = $1',
    [lineEventId]
  );
  return res.rowCount > 0;
}

async function markProcessed(lineEventId: string, tenantId: number, eventType: string) {
  await pool.query(
    'INSERT INTO processed_events(line_event_id, tenant_id, event_type) VALUES($1,$2,$3) ON CONFLICT DO NOTHING',
    [lineEventId, tenantId, eventType]
  );
}

router.post('/line', async (req, res) => {
  const signature = req.get('X-Line-Signature') || '';
  const rawBody = JSON.stringify(req.body);

  if (!verifySignature(rawBody, signature)) {
    return res.status(401).json({ error: 'Invalid signature' });
  }

  const events = req.body?.events;
  if (!Array.isArray(events) || events.length === 0) {
    return res.status(200).json({ ok: true });
  }

  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    for (const ev of events) {
      const lineEventId = ev?.eventId;
      if (!lineEventId) continue;

      // Idempotency guard
      if (await isAlreadyProcessed(lineEventId)) {
        continue;
      }

      const sourceType = ev?.source?.type;
      const userId = ev?.source?.userId;
      const tenantId = ev?.tenantId || null; // ensure your upstream attaches tenantId or resolve from userId
      const eventType = ev?.type;

      // Route and process
      if (eventType === 'message' && ev.message?.type === 'text') {
        // Example: parse commands like "clock in", "leave", "ot"
        const text = ev.message.text.trim().toLowerCase();
        if (text.includes('clock')) {
          await processClockEvent(client, tenantId, userId, ev);
        } else if (text.includes('leave')) {
          await processLeaveEvent(client, tenantId, userId, ev);
        } else if (text.includes('ot') || text.includes('overtime')) {
          await processOtEvent(client, tenantId, userId, ev);
        }
      } else if (eventType === 'postback') {
        // Handle interactive postbacks (approve/reject etc.)
        // Similar idempotent handling with processed_events
      }

      // Mark processed within same transaction
      await markProcessed(lineEventId, tenantId || 0, eventType);
    }

    await client.query('COMMIT');
    res.status(200).json({ ok: true });
  } catch (err) {
    await client.query('ROLLBACK');
    console.error('Webhook processing failed', err);
    // Return 5xx so LINE retries (but idempotency prevents duplicates)
    res.status(500).json({ error: 'Processing failed' });
  } finally {
    client.release();
  }
});

export default router;
```

### `server/src/handlers/lineEvents.ts` (minimal example)
```ts
import { PoolClient } from 'pg';

export async function processClockEvent(client: PoolClient, tenantId: number, userId: string, ev: any) {
  const date = new Date().toISOString().split('T')[0];
  const now = new Date();
  const clockType = await inferClockType(client, tenantId, userId, date); // 'in' or 'out'

  await client.query(
    `INSERT INTO clock_records(tenant_id, employee_id, clock_type, clock_time, date, line_event_id)
     VALUES($1,$2,$3,$4,$5,$6)
     ON CONFLICT (employee_id, date, clock_type) DO UPDATE
     SET clock_time = EXCLUDED.clock_time, line_event_id = EXCLUDED.line_event_id`,
    [tenantId, userId, clockType, now, date, ev.eventId]
  );
}

async function inferClockType(client: PoolClient, tenantId: number, userId: string, date: string): Promise<'in' | 'out'> {
  const res = await client.query(
    'SELECT clock_type FROM clock_records WHERE tenant_id=$1 AND employee_id=$2 AND date=$3 ORDER BY clock_time DESC LIMIT 1',
    [tenantId, userId, date]
  );
  if (res.rowCount === 0) return 'in';
  return res.rows[0].clock_type === 'in' ? 'out' : 'in';
}

export async function processLeaveEvent(client: PoolClient, tenantId: number, userId: string, ev: any) {
  await client.query(
    `INSERT INTO leave_requests(tenant_id, employee_id, status, request_line_event_id, created_at)
     VALUES($1,$2,'pending',$3,NOW())
     ON CONFLICT (request_line_event_id) DO NOTHING`,
    [tenantId, userId, ev.eventId]
  );
}
