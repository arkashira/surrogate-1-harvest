# airship / design

## Final Implementation Plan — Manifest-Driven CDN-Only Dataset Loader

**Goal**: Eliminate HF API 429 rate limits and mixed-schema `pyarrow.CastError` failures by replacing `load_dataset`/`list_repo_files` with a manifest-driven, CDN-only iterable loader. No model or infra changes; pure data-loader swap deliverable in ≤2 hours.

---

### Why this is highest-value (≤2h)
- Removes the #1 training blocker (HF API 429) and #2 blocker (mixed-schema CastError).
- Uses a proven pattern: pre-list once on Mac, embed JSON manifest, Lightning training performs CDN-only fetches.
- Zero model/infra changes; minimal risk; immediate unblock.

#### `tools/build_manifest.py`
```python
#!/usr/bin/env python3
"""
One-shot manifest builder.
Usage:
  HF_TOKEN=hf_xxx python tools/build_manifest.py \
    --repo datasets/myorg/surrogate-prompt-responses \
    --date-folder 2026-05-03 \
    --out manifest.json
"""
import argparse
import hashlib
import json
import os
import sys

try:
    from huggingface_hub import HfApi
except ImportError:
    print("Error: huggingface_hub required (pip install huggingface_hub).", file=sys.stderr)
    sys.exit(1)

def build_manifest(repo: str, date_folder: str, out_path: str, recursive: bool = False):
    api = HfApi(token=os.getenv("HF_TOKEN"))
    entries = api.list_repo_tree(repo=repo, path=date_folder, recursive=recursive)

    files = []
    for e in entries:
        if not getattr(e, "path", "").endswith(".parquet"):
            continue
        files.append({
            "path": e.path,
            "size": getattr(e, "size", None),
            # Optional: store sha256 if available; skipped to avoid extra requests
        })

    manifest = {
        "repo": repo,
        "date_folder": date_folder,
        "files": files,
        "total_files": len(files),
    }

    with open(out_path, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Wrote {len(files)} files to {out_path}")

if __name__ == "__main__":
    p = argparse.ArgumentParser()
    p.add_argument("--repo", required=True, help="Dataset repo (e.g. datasets/myorg/repo)")
    p.add_argument("--date-folder", required=True, help="Folder to list (e.g. 2026-05-03)")
    p.add_argument("--out", default="manifest.json", help="Output manifest path")
    p.add_argument("--recursive", action="store_true", help="List recursively (use with care)")
    args = p.parse_args()
    build_manifest(args.repo, args.date_folder, args.out, recursive=args.recursive)
```

#### `surrogate/data/cdn_dataset.py`
```python
import json
import logging
import os
from typing import Dict, Any, Iterator, Optional

import pyarrow.parquet as pq
import requests
from torch.utils.data import IterableDataset, get_worker_info

logger = logging.getLogger(__name__)

CDN_ROOT = "https://huggingface.co/datasets"

class CDNDataset(IterableDataset):
    """
    Manifest-driven, CDN-only iterable dataset.

    Manifest format:
    {
      "repo": "datasets/org/repo",
      "date_folder": "2026-05-03",
      "files": [
        {"path": "2026-05-03/part-00000.parquet", "size": 12345}
      ],
      "total_files": 42
    }

    Notes:
    - No HF API calls during iteration.
    - Projects only `columns`; ignores extra/mixed-schema fields.
    - Retries failed CDN downloads; skips corrupt parquet files.
    """

    def __init__(
        self,
        manifest_path: str,
        columns: tuple = ("prompt", "response"),
        max_retries: int = 3,
        timeout: int = 60,
        cdn_root: str = CDN_ROOT,
    ):
        if not os.path.isfile(manifest_path):
            raise FileNotFoundError(f"Manifest not found: {manifest_path}")
        with open(manifest_path) as f:
            self.manifest = json.load(f)

        self.repo = self.manifest.get("repo")
        if not self.repo:
            raise ValueError("Manifest missing 'repo'")
        self.files = [f["path"] for f in self.manifest.get("files", [])]
        self.columns = columns
        self.max_retries = max_retries
        self.timeout = timeout
        self.cdn_root = cdn_root.rstrip("/")

    def _download_parquet(self, path: str) -> bytes:
        url = f"{self.cdn_root}/{self.repo}/resolve/main/{path}"
        for attempt in range(1, self.max_retries + 1):
            try:
                resp = requests.get(url, timeout=self.timeout, stream=True)
                resp.raise_for_status()
                return resp.content
            except Exception as exc:
                wait = 2 ** attempt
                logger.warning(
                    "CDN download failed (attempt %s/%s) for %s: %s — retrying in %ss",
                    attempt, self.max_retries, path, exc, wait
                )
                if attempt == self.max_retries:
                    raise
                import time
                time.sleep(wait)
        raise RuntimeError("Unreachable")

    def _iter_parquet(self, content: bytes, path: str) -> Iterator[Dict[str, Any]]:
        try:
            table = pq.read_table(pq.ParquetFile(pq.ParquetBuffer(content)))
            missing = [c for c in self.columns if c not in table.column_names]
            if missing:
                raise KeyError(f"Missing columns {missing} in {path}")
            df = table.select_columns(self.columns).to_pandas()
            for _, row in df.iterrows():
                yield {k: row[k] for k in self.columns}
        except Exception as exc:
            logger.error("Failed to parse parquet from %s: %s", path, exc)
            return

    def __iter__(self) -> Iterator[Dict[str, Any]]:
        worker_info = None
        try:
            worker_info = get_worker_info()
        except Exception:
            pass

        if worker_info is None:
            files = self.files
        else:
            # Deterministic per-worker shard to avoid duplicates in multi-worker DataLoader
            files = self.files[worker_info.id::worker_info.num_workers]

        for path in files:
            try:
                content = self._download_parquet(path)
                yield from self._iter_parquet(content, path)
            except Exception as exc:
                logger.error("Skipping file %s due to error: %s", path, exc)
                continue
```
