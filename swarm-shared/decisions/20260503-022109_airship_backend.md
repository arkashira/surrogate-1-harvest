# airship / backend

## Final Consolidated Implementation (≤2h)

**Core objective**: eliminate HF API 429s and Lightning quota waste during Surrogate training by:

1. **CDN-first deterministic ingestion** — pre-list once, embed file list in training script; Lightning training fetches via CDN only (zero API calls during data load).
2. **Lightning Studio reuse + idle-stop guard** — list running studios; reuse if running, else start; restart if stopped before each run.
3. **Arkship idle-stop resilience** — lightweight watchdog to restart Surrogate Studio when stopped (prevents idle-timeout death).
4. **Deterministic repo/commit selection** — shard repos by hash to mitigate commit-cap throttling; prefer CDN URLs; retry with backoff on 429.

### scripts/list_hf_date_folder.py
```python
#!/usr/bin/env python3
"""
Generate CDN file manifest for a date folder to avoid HF API calls during training.
Usage: python list_hf_date_folder.py --repo org/surrogate-mirror --date 2026-05-03 --out configs/file_manifest.json
"""
import argparse
import json
import os
from pathlib import Path

from huggingface_hub import HfApi

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True, help="HF repo (e.g. org/surrogate-mirror)")
    parser.add_argument("--date", required=True, help="Date folder (YYYY-MM-DD)")
    parser.add_argument("--out", required=True, help="Output JSON path")
    parser.add_argument("--path", default="batches/mirror-merged", help="Base path in repo")
    args = parser.parse_args()

    api = HfApi()
    folder = f"{args.path}/{args.date}"
    # Single API call; recursive=False keeps it small and fast
    tree = api.list_repo_tree(repo_id=args.repo, path=folder, recursive=False)
    files = [entry.path for entry in tree if entry.path.lower().endswith(".parquet")]

    manifest = {
        "repo": args.repo,
        "date": args.date,
        "folder": folder,
        "files": sorted(files),
        "cdn_prefix": f"https://huggingface.co/datasets/{args.repo}/resolve/main"
    }

    out_path = Path(args.out)
    out_path.parent.mkdir(parents=True, exist_ok=True)
    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"Wrote manifest with {len(files)} files to {out_path}")

if __name__ == "__main__":
    main()
```

### surrogate/train.py (CDN-only IterableDataset snippet)
```python
import io
import json
import random
import time
import requests
from pathlib import Path
from typing import Dict, Iterator, Optional

import pyarrow.parquet as pq
import torch
from torch.utils.data import IterableDataset

class CDNParquetIterable(IterableDataset):
    def __init__(self, manifest_path: str, seed: int = 42):
        super().__init__()
        manifest = json.loads(Path(manifest_path).read_text())
        self.files = [f"{manifest['cdn_prefix']}/{f}" for f in manifest["files"]]
        if not self.files:
            raise ValueError("No parquet files found in manifest")
        self.seed = seed

    def _download_stream(self, url: str, retries: int = 3, backoff: int = 360) -> bytes:
        for attempt in range(1, retries + 1):
            try:
                resp = requests.get(url, timeout=60, stream=True)
                if resp.status_code == 429:
                    # CDN rarely 429, but honor retry-after if present
                    wait = int(resp.headers.get("retry-after", backoff))
                    print(f"CDN 429, sleeping {wait}s (attempt {attempt})")
                    time.sleep(wait)
                    continue
                resp.raise_for_status()
                buf = io.BytesIO()
                for chunk in resp.iter_content(chunk_size=8192):
                    buf.write(chunk)
                return buf.getvalue()
            except Exception as e:
                if attempt == retries:
                    raise
                time.sleep(2 ** attempt)
        raise RuntimeError("Unreachable")

    def __iter__(self) -> Iterator[Dict[str, str]]:
        worker_info = torch.utils.data.get_worker_info()
        if worker_info is None:
            it_files = self.files
        else:
            it_files = self.files[worker_info.id::worker_info.num_workers]

        for url in it_files:
            try:
                data = self._download_stream(url)
                table = pq.read_table(io.BytesIO(data))
                df = table.to_pandas()
                # Project to {prompt, response} only
                for _, row in df.iterrows():
                    prompt = row.get("prompt") or row.get("input") or ""
                    response = row.get("response") or row.get("output") or ""
                    if prompt and response:
                        yield {"prompt": str(prompt), "response": str(response)}
            except Exception as exc:
                print(f"Failed to process {url}: {exc}")
                continue
```
