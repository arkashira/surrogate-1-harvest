def hf_tree():
    """Single list_repo_tree call for the date folder."""
    try:
        items = API.list_repo_tree(
            repo_id=HF_REPO,
            path=DATE_FOLDER,
            repo_type="dataset",
            recursive=False,
        )
        files = [it["path"] for it in items if it.get("type") == "file"]
        return sorted(files)
    except Exception as exc:
        print(f"[ERROR] list_repo_tree failed: {exc}", file=sys.stderr)
        raise

def slug_for(path: str) -> str:
    """Deterministic slug from path (e.g. raw/2026-04-29/abc.parquet -> abc)."""
    stem = Path(path).stem
    return stem

def shard_for(path: str) -> int:
    """Assign shard by hash(slug) % SHARD_TOTAL."""
    slug = slug_for(path)
    return hash(slug) % SHARD_TOTAL

def cdn_url(path: str) -> str:
    """Public CDN URL (bypasses API auth/rate limits)."""
    return f"https://huggingface.co/datasets/{HF_REPO}/resolve/main/{path}"

def download_cdn(path: str, timeout: int = 30) -> bytes:
    """Download via CDN (no Authorization header)."""
    url = cdn_url(path)
    resp = requests.get(url, timeout=timeout)
    resp.raise_for_status()
    return resp.content

def project_to_pair(raw_bytes: bytes, path: str) -> dict | None:
    """
    Project file to {prompt, response} only at parse time.
    Supports:
      - JSONL lines with prompt/response fields
      - Parquet via pyarrow (projected columns)
    Returns one pair per file (first valid pair) or None.
    """
    # Try JSONL first
    try:
        text = raw_bytes.decode("utf-8")
        for line in text.splitlines():
            line = line.strip()
            if not line:
                continue
            obj = json.loads(line)
            prompt = obj.get("prompt") or obj.get("input") or obj.get("text")
            response = obj.get("response") or obj.get("output")
            if prompt and response:
                return {"prompt": str(prompt), "response": str(response)}
    except Exception:
        pass

    # Fallback: parquet projection (avoids mixed-schema CastError)
    try:
        import pyarrow as pa
        import pyarrow.parquet as pq
        table = pq.read_table(pa.BufferReader(raw_bytes), columns=["prompt", "response"])
        if table.num_rows > 0:
            prompt = table["prompt"][0].as_py()
            response = table["response"][0].as_py()
            if prompt and response:
                return {"prompt": str(prompt), "response": str(response)}
    except Exception:
        pass

    return None

def is_duplicate(md5_hex: str) -> bool:
    """Check central dedup store; fallback to local SQLite."""
    try:
        from lib.dedup import DedupStore
        store = DedupStore()
        if store.exists(md5_hex):
            return True
    except Exception:
        pass

    # SQLite fallback
    conn = sqlite3.connect(DB_PATH)
    try:
        conn.execute("CREATE TABLE IF NOT EXISTS seen (md5 TEXT PRIMARY KEY)")
        cur = conn.execute("SELECT 1 FROM seen WHERE md5 = ?", (md5_hex,))
        return cur.fetchone() is not None
    finally:
        conn.close()

def mark_seen(md5_hex: str):
    """Mark in central dedup store; fallback to SQLite."""
    try:
        from lib.dedup import DedupStore
        store = DedupStore()
        store.add(md5_hex)
        return
    except Exception:
        pass

    # SQLite fallback
    conn = sqlite3.connect(DB_PATH)
    try:
        conn.execute("CREATE TABLE IF NOT EXISTS seen (md5 TEXT PRIMARY KEY)")
        conn.execute("INSERT OR IGNORE INTO seen (md5) VALUES (?)", (md5_hex,))
        conn.commit()
    finally:
        conn.close()
