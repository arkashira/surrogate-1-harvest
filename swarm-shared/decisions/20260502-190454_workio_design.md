# workio / design

## 1. Diagnosis

- **No tenant cache on LINE webhook** — every webhook hits the DB to resolve `channelId → tenant`; slow, noisy, and fails fast under load or cold DB.
- **Missing health/readiness endpoints** — no `/health` or `/ready` for k8s/PM2/liveness checks; deployments risk traffic before DB/tenant readiness.
- **No graceful degradation on DB outage** — LINE webhook returns 5xx on transient DB errors; LINE will retry and can amplify load.
- **Cold-start latency on tenant resolution** — repeated tenant queries on every clock-in/out event add unnecessary roundtrips for active tenants.
- **No observability hooks** — missing request-id / correlation-id and minimal structured logging for webhook flows; hard to trace bursts.

## 2. Proposed change

- **File scope**: `workio/server/src/routes/line-webhook.ts` (or equivalent webhook handler) and `workio/server/src/routes/health.ts` (new).
- **Add**:
  - LRU cache (`channelId → tenant`) with 5-minute TTL and 100-item max in the LINE webhook route.
  - `/health` (liveness) and `/ready` (readiness) endpoints.
  - Short-circuit on cache hit; fallback to DB on miss with cache population.
  - Graceful 503 on DB failures in readiness; 200 on liveness.
  - Minimal structured logging (timestamp, level, route, channelId, tenantId, cacheHit).

Create `workio/server/src/routes/health.ts`:

```ts
import express, { Request, Response } from 'express';
import { Pool } from 'pg';
import { logger } from '../utils/logger';

export function createHealthRouter(dbPool: Pool) {
  const router = express.Router();

  // Liveness: process is alive
  router.get('/health', (req: Request, res: Response) => {
    res.json({ status: 'ok', uptime: process.uptime() });
  });

  // Readiness: dependencies (DB + at least one tenant) are available
  router.get('/ready', async (req: Request, res: Response) => {
    try {
      const result = await dbPool.query('SELECT 1 as ok');
      // optional: verify at least one tenant exists
      const tenantCheck = await dbPool.query(
        'SELECT id FROM tenants WHERE active = true LIMIT 1'
      );
      if (tenantCheck.rowCount && tenantCheck.rowCount > 0) {
        res.json({ status: 'ready', db: 'connected', tenants: tenantCheck.rowCount });
      } else {
        res.status(503).json({ status: 'not_ready', reason: 'no_active_tenants' });
      }
    } catch (err) {
      logger.error({ err, msg: 'readiness_check_failed' });
      res.status(503).json({ status: 'not_ready', reason: 'db_unavailable' });
    }
  });

  return router;
}
```

Update `workio/server/src/routes/line-webhook.ts` (or create if missing) to add cache + logging:

```ts
import express, { Request, Response } from 'express';
import { Pool } from 'pg';
import { LRUCache } from 'lru-cache';
import { logger } from '../utils/logger';

// Simple structured logger fallback if not present
const log = logger || console;

// Cache: channelId -> { tenant, ts }
const tenantCache = new LRUCache<string, { tenant: any; ts: number }>({
  max: 100,
  ttl: 1000 * 60 * 5, // 5 minutes
});

export function createLineWebhookRouter(dbPool: Pool) {
  const router = express.Router();

  router.post('/webhook/line', async (req: Request, res: Response) => {
    const channelId = req.header('x-line-channel-id') || (req.body as any)?.destination;
    const requestId = req.header('x-request-id') || Math.random().toString(36).slice(2, 9);
    const correlationId = requestId;

    const logMeta = { route: '/webhook/line', channelId, correlationId };

    if (!channelId) {
      log.warn({ ...logMeta, msg: 'missing_channel_id' });
      return res.status(400).json({ error: 'missing_channel_id' });
    }

    // Try cache first
    let cached = tenantCache.get(channelId);
    if (cached) {
      log.info({ ...logMeta, cacheHit: true, tenantId: cached.tenant?.id, msg: 'webhook_received' });
      // continue processing with cached.tenant
      return handleWebhook(req, res, cached.tenant, logMeta);
    }

    // Cache miss: resolve from DB
    try {
      const result = await dbPool.query(
        'SELECT * FROM tenants WHERE line_channel_id = $1 AND active = true LIMIT 1',
        [channelId]
      );

      if (!result.rowCount) {
        log.warn({ ...logMeta, cacheHit: false, msg: 'tenant_not_found' });
        return res.status(404).json({ error: 'tenant_not_found' });
      }

      const tenant = result.rows[0];
      tenantCache.set(channelId, { tenant, ts: Date.now() });

      log.info({ ...logMeta, cacheHit: false, tenantId: tenant.id, msg: 'webhook_received' });
      return handleWebhook(req, res, tenant, logMeta);
    } catch (err: any) {
      log.error({ ...logMeta, err: err?.message, cacheHit: false, msg: 'tenant_resolution_failed' });
      // Graceful degradation: return 503 so LINE can retry later
      return res.status(503).json({ error: 'service_unavailable' });
    }
  });

  return router;
}

// Minimal handler stub — replace with your actual clock-in/out/leave/OT logic
async function handleWebhook(req: Request, res: Response, tenant: any, logMeta: any) {
  try {
    const events = req.body.events || [];
    // Example: process events and respond quickly
    // await processLineEvents(events, tenant, logMeta);
    log.info({ ...logMeta, tenantId: tenant.id, eventCount: events.length, msg: 'events_processed' });
    return res.json({ status: 'ok' });
  } catch (err: any) {
    log.error({ ...logMeta, err: err?.message, msg: 'webhook_handle_failed' });
    return res.status(500).json({ error: 'processing_failed' });
  }
}
```

Wire into your main server file (e.g., `workio/server/src/index.ts` or `app.ts`):

```ts
import express from 'express';
import { Pool } from 'pg';
import { createHealthRouter } from './routes/health';
import { createLineWebhookRouter } from './routes/line-webhook';

const app = express();
app.use(express.json());

const dbPool = new Pool({
  connectionString: process.env.DATABASE_URL,
});

// Health & readiness
app.use('/api', createHealthRouter(dbPool));

// LINE webhook
app.use('/webhook', createLineWebhookRouter(dbPool));

const PORT = process.env.PORT || 3000;
app.listen(PORT, () => {
  console.log(`Workio server listening on ${PORT}`);
});
```

If you don’t have a logger, add a minimal one in `workio/server/src/utils/logger.ts`:

```ts
import pino from 'pino';

});
```
