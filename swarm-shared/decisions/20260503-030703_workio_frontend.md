# workio / frontend

## 1. Diagnosis

- Clock-in/out UX lacks immediate feedback and optimistic state; users can double-tap and create duplicate punches while waiting on LINE/webhook round-trip.
- No local pending queue for offline/weak-network punches; failed requests lose user intent and require manual retry.
- Clock status (in/out/on-break) is derived from remote calls on every view load — slow and janky on mobile.
- No visual time-to-break or shift-progress indicator; employees can’t see schedule context at a glance.
- Missing skeleton/loading states and inline validation for leave/OT forms → perceived latency and error-prone submissions.

## 2. Proposed change

File: `src/components/Attendance/TimerCard.tsx` (new) + `src/hooks/useAttendanceQueue.ts` (new) + light edits to `src/pages/Dashboard.tsx` and `src/App.tsx` (types).  
Scope: add optimistic punch UI + local pending queue + shift-progress indicator; keep backend contract unchanged (POST `/api/attendance/punch`).

```tsx
// src/hooks/useAttendanceQueue.ts
import { useEffect, useRef, useState } from 'react';

type Punch = {
  id: string; // client uuid
  type: 'CLOCK_IN' | 'CLOCK_OUT' | 'BREAK_START' | 'BREAK_END';
  timestamp: string;
  location?: { lat: number; lng: number };
  retries: number;
};

const MAX_RETRIES = 3;

export function useAttendanceQueue(tenantId: string, employeeId: string) {
  const [pending, setPending] = useState<Punch[]>([]);
  const processing = useRef(false);

  // persist to localStorage for reload resilience
  useEffect(() => {
    try {
      const raw = localStorage.getItem(`attendance-queue-${tenantId}-${employeeId}`);
      if (raw) setPending(JSON.parse(raw));
    } catch {}
  }, [tenantId, employeeId]);

  useEffect(() => {
    try {
      localStorage.setItem(`attendance-queue-${tenantId}-${employeeId}`, JSON.stringify(pending));
    } catch {}
  }, [pending, tenantId, employeeId]);

  const push = (punch: Omit<Punch, 'id' | 'retries'>) => {
    setPending((q) => [...q, { ...punch, id: crypto.randomUUID(), retries: 0 }]);
  };

  const flush = async () => {
    if (processing.current || pending.length === 0) return;
    processing.current = true;

    const toSend = pending[0];
    try {
      const res = await fetch('/api/attendance/punch', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          tenantId,
          employeeId,
          type: toSend.type,
          timestamp: toSend.timestamp,
          location: toSend.location,
        }),
      });

      if (!res.ok) throw new Error(`HTTP ${res.status}`);
      setPending((q) => q.filter((p) => p.id !== toSend.id));
    } catch {
      setPending((q) =>
        q.map((p) => (p.id === toSend.id && p.retries < MAX_RETRIES ? { ...p, retries: p.retries + 1 } : p))
      );
    } finally {
      processing.current = false;
    }
  };

  // exponential backoff flush
  useEffect(() => {
    if (pending.length === 0) return;
    const timer = setTimeout(flush, 500);
    const interval = setInterval(flush, 5000);
    return () => {
      clearTimeout(timer);
      clearInterval(interval);
    };
  }, [pending]);

  // online flush
  useEffect(() => {
    const onOnline = () => flush();
    window.addEventListener('online', onOnline);
    return () => window.removeEventListener('online', onOnline);
  }, []);

  return { pending, push };
}
```
