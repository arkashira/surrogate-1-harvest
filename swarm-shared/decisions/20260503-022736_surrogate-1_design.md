# surrogate-1 / design

## Final Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16`, `DATE`, `HF_TOKEN`, `REPO_ID` (default: `axentx/surrogate-1-training-pairs`)
- Uses **one API call** (`list_repo_tree` on the date folder) → saves `manifest.json`
- Each shard deterministically hashes `file_path` → picks its slice; only processes its slice
- Downloads via **CDN bypass** (`https://huggingface.co/datasets/{repo}/resolve/main/{path}`) — no Authorization header, avoids 429 API limits
- Projects to `{prompt, response}` only at parse time (avoids pyarrow CastError on mixed schemas)
- Dedups via content-based md5 store (`lib/dedup.py`)
- Uploads to `batches/public-merged/{DATE}/shard{N}-{HHMMSS}.jsonl`
- Reuses existing HF token and repo layout; keeps GitHub Actions matrix unchanged

CDN-bypass ingestion worker for surrogate-1 public dataset.

Usage:
  SHARD_ID=0 SHARD_TOTAL=16 DATE=2026-05-03 \
  HF_TOKEN=hf_xxx \

import os

import hashlib
import datetime
import argparse

from huggingface_hub import HfApi

# -- config --
HF_API = HfApi()
CDN_ROOT = "https://huggingface.co/datasets"
DEFAULT_REPO = "axentx/surrogate-1-training-pairs"
DEFAULT_SHARD_TOTAL = 16
# --

def deterministic_shard(slug: str, total: int) -> int:
    return int(hashlib.md5(slug.encode()).hexdigest(), 16) % total

def list_date_folder(repo: str, date: str, token: str) -> List[str]:
    """Single API call: list files under date/ (non-recursive)."""
    try:
        items = HF_API.list_repo_tree(
            repo=repo,
            path=date,
            recursive=False,
            token=token,
        )
    except Exception:
        # fallback: list root and filter by prefix
        items = HF_API.list_repo_tree(
            repo=repo,
            path="",
            recursive=False,
            token=token,
        )
        items = [i for i in items if i.startswith(f"{date}/")]
    # keep only leaf files (skip subfolders)
    return [f for f in items if not f.endswith("/")]

def build_manifest(repo: str, date: str, token: str, out_path: Path) -> List[str]:
    files = list_date_folder(repo, date, token)
    manifest = {
        "repo": repo,
        "date": date,
        "files": sorted(files),
        "generated_at": datetime.datetime.utcnow().isoformat() + "Z",
    }
    out_path.write_text(json.dumps(manifest, indent=2))
    return manifest["files"]

def cdn_download_url(repo: str, path: str) -> str:
    return f"{CDN_ROOT}/{repo}/resolve/main/{path}"

def safe_download(url: str, timeout: int = 30) -> bytes:
    resp = requests.get(url, timeout=timeout)
    resp.raise_for_status()
    return resp.content

def parse_to_pair(raw: bytes, filename: str) -> Dict[str, str]:
    """
    Best-effort projection to {prompt,response}.
    Extend per known schema as needed.
    """
    # Try parquet first (minimal column projection)
    try:
        import pyarrow.parquet as pq
        import io
        table = pq.read_table(io.BytesIO(raw), columns=["prompt", "response"])
        df = table.to_pandas()
        row = df.iloc[0]
        prompt = str(row["prompt"]).strip()
        response = str(row["response"]).strip()
        if prompt and response:
            return {"prompt": prompt, "response": response}
    except Exception:
        pass

    # Try JSON/JSONL
    try:
        import json as jsonlib
        text = raw.decode()
        # if JSONL, take first non-empty line
        lines = [ln.strip() for ln in text.splitlines() if ln.strip()]
        for ln in lines:
            data = jsonlib.loads(ln)
            if isinstance(data, dict):
                prompt = str(data.get("prompt", data.get("input", ""))).strip()
                response = str(data.get("response", data.get("output", ""))).strip()
                if prompt and response:
                    return {"prompt": prompt, "response": response}
            # if list of dicts, try first item
            if isinstance(data, list) and len(data) > 0:
                item = data[0]
                if isinstance(item, dict):
                    prompt = str(item.get("prompt", item.get("input", ""))).strip()
                    response = str(item.get("response", item.get("output", ""))).strip()
                    if prompt and response:
                        return {"prompt": prompt, "response": response}
    except Exception:
        pass

    # Fallback: treat as plain text blob
    text = raw.decode(errors="replace").strip()
    if text:
        # heuristic split if prompt/response separated by newline or delimiter
        parts = text.split("\n", 1)
        if len(parts) == 2:
            return {"prompt": parts[0].strip(), "response": parts[1].strip()}
        return {"prompt": filename, "response": text[:4000]}
    return {"prompt": filename, "response": ""}
