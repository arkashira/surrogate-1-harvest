## Final Synthesis (Best Parts + Correctness + Actionability)

I merged the strongest, most actionable elements from both candidates, removed contradictions, and hardened the implementation for reliability and CDN-only training.

Key decisions:
- Use **one Python-first manifest generator** (not a Bash wrapper around Python) for clarity, error handling, and retries.
- Manifest must be **deterministic and CDN-only** so training never touches HF API.
- Add **idempotent state directory** and **resume/retry logic** with exponential backoff for 429 (360 s wait honored).
- Provide a **lightweight, optional Studio reuse helper** (no forced Lightning App in the manifest script) so orchestration stays explicit and safe.
- Training script accepts `--manifest` and streams via `fsspec`/`aiohttp` (no mandatory `datasets` dependency) to guarantee CDN-only fetches.

### `/opt/axentx/vanguard/ops/ingest_manifest.py`

```python
#!/usr/bin/env python3
"""
Generate a durable, CDN-only manifest for a date folder to avoid HF API during training.

Usage:
  ./ingest_manifest.py 2026-04-29
  HF_DATASET_REPO=axentx/surrogate-mirror ./ingest_manifest.py 2026-04-29

Outputs:
  batches/mirror-merged/{date}/manifest.json
  .ingest_state/manifest-{date}.done  (idempotency/resume marker)
"""

import os
import sys
import json
import time
import logging
from datetime import datetime
from pathlib import Path
from typing import Dict, List

from huggingface_hub import HfApi, HfFolder, RepositoryError

REPO = os.getenv("HF_DATASET_REPO", "axentx/surrogate-mirror")
HF_TOKEN = os.getenv("HF_TOKEN", HfFolder().get_token()) or None
API = HfApi(token=HF_TOKEN)

STATE_DIR = Path(__file__).parent / ".ingest_state"
OUT_BASE = Path(__file__).parent / "batches" / "mirror-merged"

logging.basicConfig(level=logging.INFO, format="%(asctime)s %(levelname)s %(message)s")

def _backoff(attempt: int, base: float = 1.0, cap: float = 360.0) -> float:
    return min(cap, base * (2 ** attempt))

def list_folder_safe(path: str, max_retries: int = 5) -> List[Dict]:
    for attempt in range(max_retries + 1):
        try:
            tree = API.list_repo_tree(repo_id=REPO, path=path, recursive=False)
            return [t for t in tree if getattr(t, "type", None) == "file" and t.path.endswith(".parquet")]
        except Exception as exc:
            if attempt == max_retries:
                raise
            wait = _backoff(attempt)
            logging.warning("list_repo_tree failed (attempt %s/%s): %s. Retrying in %.1fs",
                            attempt + 1, max_retries + 1, exc, wait)
            time.sleep(wait)
    raise RuntimeError("Unreachable")

def build_manifest(date_str: str) -> Path:
    folder = OUT_BASE / date_str
    manifest_path = folder / "manifest.json"
    done_marker = STATE_DIR / f"manifest-{date_str}.done"

    if done_marker.exists() and manifest_path.exists():
        logging.info("Idempotent hit: %s already exists. Skipping.", manifest_path)
        return manifest_path

    # Single non-recursive API call per folder (strongest reduction in API usage).
    entries = list_folder_safe(str(folder))
    if not entries:
        raise FileNotFoundError(f"No parquet files found in {REPO}/{folder} (check path/date).")

    manifest = []
    for e in entries:
        cdn_url = f"https://huggingface.co/datasets/{REPO}/resolve/main/{e.path}"
        manifest.append({
            "path": e.path,
            "cdn_url": cdn_url,
            "size": getattr(e, "size", None),
        })

    os.makedirs(os.path.dirname(manifest_path), exist_ok=True)
    with open(manifest_path, "w") as f:
        json.dump(manifest, f, indent=2)

    os.makedirs(STATE_DIR, exist_ok=True)
    done_marker.touch()
    logging.info("Wrote %d files to %s", len(manifest), manifest_path)
    return manifest_path

if __name__ == "__main__":
    date_arg = sys.argv[1] if len(sys.argv) > 1 else datetime.utcnow().strftime("%Y-%m-%d")
    try:
        p = build_manifest(date_arg)
        print(str(p))
    except Exception as exc:
        logging.error("Manifest generation failed: %s", exc)
        sys.exit(1)
```

class CDNParquetIterable(IterableDataset):
    """Stream parquet files from CDN URLs listed in manifest."""
    def __init__(self, manifest_path: str):
        super().__init__()
        with open(manifest_path) as f:
            self.items = [m["cdn_url"] for m in json.load(f)]
        if not self.items:
            raise ValueError("No CDN URLs found in manifest.")

    def _stream_one(self, url: str) -> Iterator[Dict[str, Any]]:
        # Lightweight streaming: read parquet with fsspec -> pyarrow
        try:
            import pyarrow.parquet as pq
        except ImportError:
            raise ImportError("pyarrow is required for parquet reading. pip install pyarrow")
        with fsspec.open(url, mode="rb") as f:
            table = pq.read_table(f)
            for batch in table.to_batches(max_chunksize=1024):
                # Convert to dict-of-lists for simple training steps
                yield {col: batch.column(col).to_pylist() for col in batch.schema.names}

    def __iter__(self) -> Iterator[Dict[str, Any]]:
        for url in self.items:
            yield from self._stream_one(url)
