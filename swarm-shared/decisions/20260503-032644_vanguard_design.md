# vanguard / design

## 1. Diagnosis
- No CDN-first manifest: ingestion/training scripts can still trigger `list_repo_tree`/`load_dataset` at runtime → 429s and non-reproducible runs.
- Missing deterministic file list keyed by date/slug; training jobs re-enumerate the repo and pay API costs.
- No content-addressed pin for dataset shards; cache invalidation and silent schema drift possible.
- Lightning Studio reuse/idle-stop handling absent; quota burned by recreation and jobs killed by idle timeout.
- No clear boundary between Mac orchestration and remote compute; risk of accidental local `from_pretrained` or ingestion on laptop.

## 2. Proposed change
Create `/opt/axentx/vanguard/design/cdn_manifest.py` (single file, <150 LOC) plus minimal project-level wiring:
- Expose `build_manifest(repo, date_folder, out_path)` → produces `manifest-{date}.json` with CDN URLs and sha256.
- Expose `train_loader(manifest_path)` → yields file streams via CDN only (zero HF API calls).
- Add `lightning/run_studio.py` helper to reuse Running studios and survive idle-stop.
- Add `mac_orchestrator.sh` wrapper to enforce remote-only training/ingestion.

#!/usr/bin/env python3
"""
CDN-first manifest builder + loader for HF datasets.
Goal: eliminate HF Hub API calls during training.
"""
from __future__ import annotations

import hashlib
import json
import os
import sys
from datetime import datetime
from pathlib import Path
from typing import Dict, List, Any

import requests
from huggingface_hub import HfApi, list_repo_tree

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def sha256_file(path: str, chunk_kb: int = 1024) -> str:
    h = hashlib.sha256()
    with open(path, "rb") as f:
        for chunk in iter(lambda: f.read(chunk_kb * 1024), b""):
            h.update(chunk)
    return h.hexdigest()

def build_manifest(
    repo: str,
    date_folder: str,
    out_path: str | Path,
    token: str | None = None,
) -> Dict[str, Any]:
    """
    Build a content-addressed manifest for one date folder.
    date_folder example: '2026-04-29' or 'batches/mirror-merged/2026-04-29'
    """
    api = HfApi(token=token)
    # Single shallow list (no recursion into subfolders beyond date_folder)
    tree = list_repo_tree(repo=repo, path=date_folder, recursive=False)

    entries: List[Dict[str, str]] = []
    for node in tree:
        if node.type != "file":
            continue
        cdn_url = CDN_TEMPLATE.format(repo=repo, path=node.path)
        # Optional: pre-compute sha256 by downloading HEAD or small range if needed.
        # For now, store path-based content-address via repo path + size.
        entries.append(
            {
                "repo": repo,
                "path": node.path,
                "size": node.size,
                "cdn_url": cdn_url,
                "lfs": node.lfs is not None,
            }
        )

    manifest = {
        "repo": repo,
        "date_folder": date_folder,
        "generated_at_utc": datetime.utcnow().isoformat() + "Z",
        "entries": entries,
    }

    out_path = Path(out_path)
    out_path.parent.mkdir(parents=True, exist_ok=True)
    out_path.write_text(json.dumps(manifest, indent=2))
    return manifest

def train_loader(
    manifest_path: str | Path,
    project_cols: tuple = ("prompt", "response"),
    max_files: int | None = None,
):
    """
    Generator yielding file streams from CDN (zero HF Hub API calls).
    Caller should project to {prompt,response} and parse per-file schema.
    """
    manifest = json.loads(Path(manifest_path).read_text())
    entries = manifest["entries"]
    if max_files is not None:
        entries = entries[:max_files]

    for i, e in enumerate(entries):
        if e.get("lfs"):
            # LFS pointer files require resolving; skip or handle separately.
            # For mirror-merged/ enriched/ we expect real parquet, not pointers.
            continue
        resp = requests.get(e["cdn_url"], timeout=30)
        resp.raise_for_status()
        yield {
            "index": i,
            "path": e["path"],
            "size": e["size"],
            "content": resp.content,
            "cols": project_cols,
        }

if __name__ == "__main__":
    # CLI: python cdn_manifest.py <repo> <date_folder> <out_json>
    if len(sys.argv) != 4:
        print("Usage: cdn_manifest.py <repo> <date_folder> <out_json>")
        sys.exit(1)
    repo, date_folder, out_json = sys.argv[1], sys.argv[2], sys.argv[3]
    build_manifest(repo=repo, date_folder=date_folder, out_path=out_json, token=os.getenv("HF_TOKEN"))
    print(f"Manifest written to {out_json}")
```

#!/usr/bin/env python3
"""
Lightning Studio reuse + idle-stop resilience.
"""
import time
from lightning import Lightning, L40S, Machine, Teamspace

LIGHTNING_ACCOUNT = "lightning-lambda-prod"  # H200-capable; fallback to public if needed

def get_or_start_studio(
    name: str,
    machine: Machine = Machine.L40S,
    account: str = LIGHTNING_ACCOUNT,
    max_wait: int = 300,
):
    lt = Lightning()
    teamspace = Teamspace()

    # Reuse running studio
    for s in teamspace.studios:
        if s.name == name and s.status == "Running":
            print(f"Reusing running studio: {name}")
            return s

    # Start new studio
    print(f"Starting studio {name} on {machine} in {account}")
    studio = lt.Studio(
        name=name,
        machine=machine,
        cloud_account=account,
        create_ok=True,
    )

    # Wait until running
    for _ in range(max_wait // 5):
        studio.refresh()
        if studio.status == "Running":
            return studio
        time.sleep(5)
    raise RuntimeError(f"Studio {name} failed to start within {max_wait}s")

def run_with_idle_resilience(studio_name: str, script: str, args: list[str] | None = None):
    """
    Run script on studio; if studio stopped, restart and rerun.
    """
    studio = get_or_start_studio(studio_name)
    try:
        studio.run(script, args=args or [])
    except Exception as exc:
        # If idle-stop killed training, restart and retry once
        studio.refresh()
        if studio.status != "Running":
            print("Studio not running (likely idle-stop). Restarting...")
            studio = get_or_start_studio(studio_name)
            studio.run(script, args=args or [])
        else:
            raise exc

if __name__ == "__main__":
    import sys
    if len(sys.argv) < 3:
        print("Usage: run_studio.py <studio_name> <script.py> [args...]")
        sys.exit(1)
    run_with_idle_resilience(sys.argv[1], sys.argv[2], sys.argv[3:])
```
