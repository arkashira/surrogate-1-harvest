# surrogate-1 / design

### Final Implementation Plan (≤2h)

**Highest-value change**  
Add a Mac-side `tools/snapshot_manifest.py` that lists **one date-partition with a single HF API call**, emits a deterministic `file_manifest.json` containing **CDN URLs + integrity metadata**, and update training to use **CDN-only streaming fetches with retries** (zero HF API calls during data loading). This directly prevents 429/128-hour commit limits from blocking training while remaining reproducible and actionable.

### 2) `tools/snapshot_manifest.py`  *(30–40 min)*
```python
#!/usr/bin/env python3
"""
Generate a CDN-only file manifest for one date-partition in a HuggingFace dataset repo.

Usage:
    python tools/snapshot_manifest.py \
        --repo axentx/surrogate-1-training-pairs \
        --date 2026-04-29 \
        --out file_manifest.json
"""

import argparse
import hashlib
import json
import os
import sys
from datetime import datetime, timezone

from huggingface_hub import list_repo_tree
from tqdm import tqdm

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def build_manifest(repo: str, date_partition: str, out_path: str) -> None:
    # Single API call: list one folder (date partition)
    entries = list(
        list_repo_tree(
            repo_id=repo,
            path=date_partition,
            recursive=True,
            repo_type="dataset",
        )
    )
    files = [e for e in entries if e.type == "file"]
    if not files:
        print(f"No files found under {repo}/{date_partition}", file=sys.stderr)
        sys.exit(1)

    manifest_files = []
    for f in tqdm(files, desc="Building manifest"):
        manifest_files.append(
            {
                "repo": repo,
                "path": f.path,
                "cdn_url": CDN_TEMPLATE.format(repo=repo, path=f.path),
                "size": getattr(f, "size", None),
            }
        )

    payload = {
        "generated_at": datetime.now(timezone.utc).isoformat(),
        "repo": repo,
        "date_partition": date_partition,
        "files": manifest_files,
    }

    # Deterministic sha256 for reproducibility
    canonical = json.dumps(payload, sort_keys=True, separators=(",", ":")).encode()
    payload["manifest_sha256"] = hashlib.sha256(canonical).hexdigest()

    os.makedirs(os.path.dirname(os.path.abspath(out_path)) or ".", exist_ok=True)
    with open(out_path, "w", encoding="utf-8") as fp:
        json.dump(payload, fp, indent=2, sort_keys=True)

    print(f"Wrote {len(manifest_files)} files -> {out_path}")

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Generate CDN file manifest.")
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g. axentx/surrogate-1-training-pairs)")
    parser.add_argument("--date", required=True, help="Date partition folder (e.g. 2026-04-29)")
    parser.add_argument("--out", default="file_manifest.json", help="Output JSON path")
    args = parser.parse_args()
    build_manifest(args.repo, args.date, args.out)
```

### 3) CDN-only loader for `train.py`  *(40–50 min)*
```python
import argparse
import io
import json
import time
import urllib.request
from typing import Dict, List

import pyarrow.parquet as pq

def load_cdn_parquet(cdn_url: str, columns: List[str] = None, max_retries: int = 3, backoff: float = 2.0):
    """
    Fetch and read a Parquet file from CDN with retries.
    Returns list of dicts projected to `columns` (or all).
    """
    columns = columns or []
    for attempt in range(1, max_retries + 1):
        try:
            with urllib.request.urlopen(cdn_url, timeout=60) as resp:
                buf = io.BytesIO(resp.read())
            table = pq.read_table(buf, columns=columns or None)
            return table.to_pylist()
        except Exception as exc:
            if attempt == max_retries:
                print(f"FAILED to fetch {cdn_url}: {exc}")
                return []
            sleep = backoff ** attempt
            print(f"Retry {attempt}/{max_retries} for {cdn_url} in {sleep:.1f}s: {exc}")
            time.sleep(sleep)
    return []

def load_dataset_from_manifest(manifest_path: str) -> List[Dict[str, str]]:
    with open(manifest_path, "r", encoding="utf-8") as f:
        manifest = json.load(f)

    records = []
    for item in manifest["files"]:
        rows = load_cdn_parquet(item["cdn_url"], columns=["prompt", "response"])
        for r in rows:
            # Project to expected schema; ignore corrupt/missing rows
            prompt = r.get("prompt")
            response = r.get("response")
            if prompt and response:
                records.append({"prompt": prompt, "response": response})
    return records

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--manifest", default="file_manifest.json", help="Path to file_manifest.json")
    args = parser.parse_args()

    records = load_dataset_from_manifest(args.manifest)
    print(f"Loaded {len(records)} valid records from manifest.")
    # Continue training with `records`

if __name__ == "__main__":
    main()
```

### Key Benefits (resolved in favor of correctness + actionability)
- **Single HF API call** per partition → avoids 429 and 128-hour commit limits.  
- **CDN-only streaming** during training → faster, no HF API dependency at load time.  
- **Deterministic manifest + sha256** → reproducible snapshots.  
- **Retry/backoff on CDN 429** and graceful skip of corrupt files → robust in production.  
- **Minimal dependencies** and clear CLI → actionable on a Mac or any orchestration host within ~2 hours.
