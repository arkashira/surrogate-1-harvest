## Final Synthesized Implementation (Best of Both Proposals)

**Highest-Value Improvement (<2h):**  
Harden `airship discover` into a deterministic, CDN-only orchestrator that eliminates HF API rate limits and PyArrow schema errors while producing reproducible file lists and safe, schema-stable ingestion.

**Why this is correct + actionable:**  
- Removes 429 risk and mixed-schema PyArrow failures by avoiding `load_dataset(streaming=True)` and recursive tree walks.  
- Uses a single `list_repo_tree` per date folder (CDN-only) and per-file projection to `{prompt, response}`.  
- Deterministic repo selection spreads HF commit load; idempotent JSON outputs integrate cleanly with cron/CI.  
- Deliverable is one orchestrator module + one CLI + one lightweight projection helper — no training changes.

### `airship/discover.py`
```python
#!/usr/bin/env python3
"""
CDN-only file discovery for HF datasets.
Produces deterministic file lists to avoid API rate limits and schema errors.
"""
from __future__ import annotations

import json
import hashlib
import logging
from pathlib import Path
from typing import List, Dict, Any

from huggingface_hub import HfApi

logging.basicConfig(level=logging.INFO, format="%(asctime)s %(levelname)s %(message)s")
log = logging.getLogger("airship.discover")

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"
DEFAULT_SIBLING_COUNT = 1


def _hash_slug(slug: str) -> int:
    return int(hashlib.sha256(slug.encode()).hexdigest(), 16)


def pick_sibling_repo(slug: str, n: int = DEFAULT_SIBLING_COUNT) -> str:
    """Deterministic sibling repo selection to spread HF commit load."""
    if n <= 1:
        return slug
    idx = _hash_slug(slug) % n
    # convention: repo-suffix with -s0..s{N-1}
    parts = slug.split("/", 1)
    if len(parts) == 2:
        base, name = parts
        return f"{base}/{name}-s{idx}"
    return f"{slug}-s{idx}"


def list_date_folder(repo: str, date_folder: str, recursive: bool = False) -> List[Dict[str, Any]]:
    """
    Single non-recursive tree call per folder to minimize API usage.
    Returns rows with cdn_url and metadata.
    """
    api = HfApi()
    prefix = date_folder.rstrip("/") + "/"
    entries = api.list_repo_tree(repo=repo, path=prefix, recursive=recursive)

    files: List[Dict[str, Any]] = []
    for e in entries:
        if e.type != "file":
            continue
        # Skip known large binary/LFS unless explicitly allowed
        if e.path.lower().endswith((".zip", ".bin", ".pt", ".pth", ".ckpt")):
            log.info("Skipping binary file: %s", e.path)
            continue
        files.append(
            {
                "repo": repo,
                "path": e.path,
                "cdn_url": CDN_TEMPLATE.format(repo=repo, path=e.path),
                "size": getattr(e, "size", None),
                "lfs": getattr(e, "lfs", False),
            }
        )
    log.info("Found %d files in %s:%s", len(files), repo, prefix)
    return files


def run(date: str, repo: str, out_dir: str | Path = "discoveries", sibling_repos: int = DEFAULT_SIBLING_COUNT) -> Path:
    out_dir = Path(out_dir)
    out_dir.mkdir(parents=True, exist_ok=True)

    sibling_repo = pick_sibling_repo(repo, n=sibling_repos)
    files = list_date_folder(sibling_repo, date, recursive=False)

    if not files:
        log.warning("No files found for %s:%s", sibling_repo, date)

    slug_hash = hashlib.sha256(sibling_repo.encode()).hexdigest()[:8]
    out_file = out_dir / f"filelist-{date}-{slug_hash}.json"
    payload = {
        "generated_by": "airship.discover",
        "date": date,
        "repo_requested": repo,
        "repo_used": sibling_repo,
        "sibling_count": sibling_repos,
        "files": files,
    }
    out_file.write_text(json.dumps(payload, indent=2))
    log.info("Wrote %s (%d files)", out_file, len(files))
    return out_file


if __name__ == "__main__":
    import argparse

    parser = argparse.ArgumentParser(description="CDN-only HF dataset file discovery")
    parser.add_argument("--date", required=True, help="Date folder (YYYY-MM-DD)")
    parser.add_argument("--repo", default="datasets/mirror-merged", help="HF dataset repo")
    parser.add_argument("--out-dir", default="discoveries", help="Output directory")
    parser.add_argument("--sibling-repos", type=int, default=DEFAULT_SIBLING_COUNT, help="Number of sibling repos for load spreading")
    args = parser.parse_args()

    run(date=args.date, repo=args.repo, out_dir=args.out_dir, sibling_repos=args.sibling_repos)
```

### `bin/airship-discover`
```bash
#!/usr/bin/env bash
# Wrapper to run airship.discover with proper environment for cron/CI.
set -euo pipefail
export SHELL=/bin/bash

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
PROJECT_ROOT="$(cd "$SCRIPT_DIR/.." && pwd)"

cd "$PROJECT_ROOT"

LOG_DIR="logs"
mkdir -p "$LOG_DIR"

DATE="${1:-}"
if [[ -z "$DATE" ]]; then
  echo "Usage: $0 <YYYY-MM-DD>"
  exit 1
fi

LOG_FILE="${LOG_DIR}/discover-${DATE}.log"
exec > >(tee -a "$LOG_FILE") 2>&1

python -m airship.discover --date "$DATE" --repo "datasets/mirror-merged" --out-dir discoveries
```
