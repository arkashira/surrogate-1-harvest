# Costinel / ops

**Final consolidated implementation** (best of Candidate 1 + Candidate 2)

- **Scope**: Frontend-only, read-only panel that surfaces the most-connected hub and actionable proposals from the knowledge graph.  
- **Timebox**: <2h  
- **Stack**: React + TypeScript + Tailwind  
- **Entry point**: `src/components/TopHubSignalPanel.tsx` + integrate into `CostinelDashboard`.

### 1) Data model (minimal, extendable)

```ts
// src/types/knowledgeGraph.ts
export interface HubNode {
  id: string;
  label: string;           // e.g. "MOC"
  description?: string;
  connections: number;
  pagerank?: number;
}

export interface Proposal {
  id: string;
  title: string;
  summary: string;
  priority: "critical" | "high" | "medium" | "low";
  hubIds: string[];        // which hubs this relates to
  href?: string;           // optional deep link
}
```

### 2) Component implementation (graceful fallbacks, correct selection)

```tsx
// src/components/TopHubSignalPanel.tsx
import React from "react";
import { HubNode, Proposal } from "../types/knowledgeGraph";

function priorityColor(p: string) {
  switch (p) {
    case "critical":
      return "bg-red-100 text-red-800 border-red-200";
    case "high":
      return "bg-orange-100 text-orange-800 border-orange-200";
    case "medium":
      return "bg-yellow-100 text-yellow-800 border-yellow-200";
    default:
      return "bg-gray-100 text-gray-800 border-gray-200";
  }
}

export interface TopHubSignalPanelProps {
  graphData?: {
    hubs: HubNode[];
    proposals: Proposal[];
  };
  className?: string;
}

export const TopHubSignalPanel: React.FC<TopHubSignalPanelProps> = ({
  graphData,
  className = "",
}) => {
  const topHub = graphData?.hubs?.length
    ? graphData.hubs.reduce((a, b) =>
        (b.pagerank ?? b.connections) > (a.pagerank ?? a.connections) ? b : a
      )
    : null;

  const relatedProposals = topHub
    ? (graphData?.proposals ?? []).filter((p) => p.hubIds.includes(topHub.id))
    : [];

  const visibleProposals = relatedProposals.slice(0, 3);

  if (!topHub) {
    return (
      <div
        className={`p-4 border border-dashed border-gray-300 rounded-lg bg-gray-50 ${className}`}
      >
        <p className="text-sm text-gray-600">
          No knowledge-graph data available. Run{" "}
          <code className="text-xs bg-gray-200 px-1 rounded">knowledge-rag</code>{" "}
          to populate top-hub insights.
        </p>
      </div>
    );
  }

  return (
    <div className={`p-5 border rounded-lg bg-white shadow-sm ${className}`}>
      <div className="flex items-start justify-between mb-3">
        <div>
          <h3 className="text-lg font-semibold text-gray-900">{topHub.label}</h3>
          <p className="text-sm text-gray-600">
            {topHub.description || "Knowledge hub"}
          </p>
        </div>
        <span className="text-xs font-medium text-gray-500">
          {topHub.connections} connections
        </span>
      </div>

      {visibleProposals.length > 0 ? (
        <ul className="space-y-2" role="list">
          {visibleProposals.map((p) => (
            <li key={p.id} className="flex items-start gap-2 text-sm">
              <span
                className={`inline-flex items-center px-2 py-0.5 text-xs font-medium rounded border mt-0.5 whitespace-nowrap ${priorityColor(
                  p.priority
                )}`}
              >
                {p.priority}
              </span>
              <div className="min-w-0">
                <p className="font-medium text-gray-900 truncate">{p.title}</p>
                <p className="text-gray-600 truncate">{p.summary}</p>
              </div>
            </li>
          ))}
        </ul>
      ) : (
        <p className="text-sm text-gray-500">
          No actionable proposals for this hub.
        </p>
      )}

      <div className="mt-3">
        <a
          href={`/knowledge-graph?hub=${encodeURIComponent(topHub.id)}`}
          className="text-sm text-blue-600 hover:underline"
        >
          View in graph →
        </a>
      </div>
    </div>
  );
};
```

### 3) Data fetching (SWR, resilient, no redundant fetches)

```tsx
// src/components/TopHubSignalPanelWithSWR.tsx
import useSWR from "swr";
import { TopHubSignalPanel, type TopHubSignalPanelProps } from "./TopHubSignalPanel";

const fetcher = (url: string) => fetch(url).then((r) => r.json());

export const TopHubSignalPanelWithSWR: React.FC<Omit<TopHubSignalPanelProps, "graphData">> = (
  props
) => {
  const { data, error } = useSWR<{ hubs: any[]; proposals: any[] }>(
    "/api/knowledge-graph/top-hub",
    fetcher,
    {
      revalidateOnFocus: false,
      shouldRetryOnError: false,
    }
  );

  const graphData = data && !error ? data : undefined;

  return <TopHubSignalPanel {...props} graphData={graphData} />;
};
```

### 4) API endpoint (backend)

```ts
// src/api/knowledge-graph.ts
import express from "express";
import { getTopHubAndProposals } from "../services/knowledge-graph";

const router = express.Router();

router.get("/top-hub", async (req, res) => {
  try {
    const result = await getTopHubAndProposals();
    res.json(result);
  } catch (err) {
    res.status(500).json({ error: "Failed to fetch top-hub data" });
  }
});

export default router;
```

### 5) Service (backend)

```ts
// src/services/knowledge-graph.ts
import { queryKnowledgeGraph } from "../utils/knowledge-graph";

export const getTopHubAndProposals = async () => {
  // Single query or two small queries depending on your graph API
  const [topHub, proposals] = await Promise.all([
    queryKnowledgeGraph("topHub"),
    queryKnowledgeGraph("proposals"),
  ]);

  return {
    hubs: Array.isArray(topHub) ? topHub : [topHub].filter(Boolean),
    proposals: Array.isArray(proposals) ? proposals : [],
  };
};
```
