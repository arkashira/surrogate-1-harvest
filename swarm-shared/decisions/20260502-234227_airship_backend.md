# airship / backend

## Final Implementation Plan — Deterministic CDN-Only Manifest for Surrogate-1

**Scope (≤2h):**  
Add a single CLI command `airship discover` that produces a deterministic, CDN-only file manifest for a given `repo_id` and `date_folder`.  
- Uses **one** `list_repo_tree(path, recursive=False)` from the orchestrator (Mac) to seed a JSON manifest.  
- Training scripts consume the manifest and fetch via CDN URLs only (`https://huggingface.co/datasets/{repo}/resolve/main/{path}`) — **zero HF API calls during training**.  
- Fits existing surrogate-1 ingestion patterns: project to `{prompt,response}` at parse time; store attribution in filename.

### 1) CLI entrypoint — `airship/cli/discover.py`

```python
#!/usr/bin/env python3
"""
airship discover
Produce a deterministic CDN-only manifest for a repo + date_folder.

Usage:
  airship discover owner/repo 2025-01-01 \
    --out manifests/owner-repo-2025-01-01.json
"""

import argparse
import json
import sys
from pathlib import Path
from typing import List, Dict

try:
    from huggingface_hub import HfApi
except ImportError:
    print("error: huggingface_hub required", file=sys.stderr)
    sys.exit(1)

CDN_BASE = "https://huggingface.co/datasets"


def normalize_repo_id(repo_id: str) -> str:
    repo_id = repo_id.removeprefix("huggingface.co/datasets/").strip("/")
    if repo_id.count("/") != 1:
        raise ValueError("repo_id must be 'owner/repo'")
    return repo_id


def build_manifest(repo_id: str, date_folder: str, api: HfApi) -> List[Dict]:
    """
    Single API call: list_repo_tree(non-recursive) for the date folder.
    Returns CDN-only entries sorted deterministically.
    """
    try:
        tree = api.list_repo_tree(repo_id=repo_id, path=date_folder, recursive=False)
    except Exception as exc:
        raise RuntimeError(f"HF API list_repo_tree failed: {exc}") from exc

    entries = []
    for item in tree:
        if getattr(item, "type", None) != "file":
            continue
        cdn_url = f"{CDN_BASE}/{repo_id}/resolve/main/{item.path}"
        entries.append(
            {
                "repo_id": repo_id,
                "path": item.path,
                "cdn_url": cdn_url,
                "size": getattr(item, "size", None),
                "lfs": getattr(item, "lfs", None),
            }
        )

    entries.sort(key=lambda e: e["path"])
    return entries


def main() -> None:
    parser = argparse.ArgumentParser(description="Build CDN-only manifest")
    parser.add_argument("repo_id", help="owner/repo or huggingface.co/datasets/owner/repo")
    parser.add_argument("date_folder", help="Folder under datasets root (e.g. 2025-01-01)")
    parser.add_argument("--out", default="manifest.json", help="Output JSON path")
    parser.add_argument("--token", default=None, help="HF token (optional; public datasets need none)")
    args = parser.parse_args()

    repo_id = normalize_repo_id(args.repo_id)
    api = HfApi(token=args.token)

    try:
        manifest = build_manifest(repo_id, args.date_folder, api)
    except Exception as exc:
        print(f"error: {exc}", file=sys.stderr)
        sys.exit(1)

    out_path = Path(args.out)
    out_path.parent.mkdir(parents=True, exist_ok=True)
    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"manifest written to {out_path} ({len(manifest)} files)")


if __name__ == "__main__":
    main()
```

from pathlib import Path
import requests

CDN_BASE = "https://huggingface.co/datasets"


def fetch_via_cdn(entry):
    resp = requests.get(entry["cdn_url"], timeout=30)
    resp.raise_for_status()
    return resp.content


def load_manifest(manifest_path):
    with open(manifest_path) as f:
        return json.load(f)


def build_dataset_from_manifest(manifest_path, cache_dir):
    entries = load_manifest(manifest_path)
    cache_dir = Path(cache_dir)
    cache_dir.mkdir(parents=True, exist_ok=True)

    rows = []
    for e in entries:
        out_path = cache_dir / Path(e["path"]).name
        if not out_path.exists():
            data = fetch_via_cdn(e)
            out_path.write_bytes(data)
        # Project to {prompt,response} at parse time (surrogate-1 pattern)
        rows.extend(parse_file(out_path))
    return rows
