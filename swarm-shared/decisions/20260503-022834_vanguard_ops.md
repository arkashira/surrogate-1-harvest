# vanguard / ops

## 1. Diagnosis
- No CDN-bypass for dataset fetches: training scripts still use authenticated HF API (`load_dataset`/`list_repo_tree`) which burns quota and risks 429s.
- No pre-listed file manifest: each training run re-enumerates repo files via API instead of embedding a static file list for CDN-only fetches.
- No graceful 429 fallback: no retry/backoff or CDN fallback when HF API rate-limits; training jobs fail instead of degrade.
- Studio reuse missing: training launcher recreates Lightning Studio instead of reusing running instances, wasting ~80hr/mo quota.
- Idle-timeout fragility: no pre-run status check for Lightning Studio; idle stop kills training without restart.

## 2. Proposed change
File: `/opt/axentx/vanguard/ops/train_launcher.py` (create or update)  
Scope: add CDN-bypass data loader, manifest embedding, 429 fallback, Studio reuse, and idle restart logic.

import json, os, time, random
import requests
import torch
from torch.utils.data import IterableDataset, DataLoader

REPO = os.environ["REPO"]
MANIFEST_FILE = os.environ["MANIFEST_FILE"]
HF_DATA_DIR = os.environ["HF_DATA_DIR"]
os.makedirs(HF_DATA_DIR, exist_ok=True)

with open(MANIFEST_FILE) as f:
    manifest = json.load(f)

FILES = manifest["files"]
BASE_URL = f"https://huggingface.co/datasets/{REPO}/resolve/main/{manifest['date']}"

class CDNParquetIterable(IterableDataset):
    def __init__(self, file_urls, max_retries=5):
        self.file_urls = file_urls
        self.max_retries = max_retries

    def _download_with_retry(self, url, dest):
        for attempt in range(self.max_retries):
            try:
                resp = requests.get(url, timeout=30)
                if resp.status_code == 429:
                    wait = 360 + random.uniform(0, 60)
                    print(f"429 hit, waiting {wait:.0f}s")
                    time.sleep(wait)
                    continue
                resp.raise_for_status()
                with open(dest, "wb") as f:
                    f.write(resp.content)
                return dest
            except Exception as e:
                wait = (2 ** attempt) + random.random()
                print(f"Download failed ({url}): {e}, retry in {wait:.1f}s")
                time.sleep(wait)
        raise RuntimeError(f"Failed to download {url}")

    def __iter__(self):
        for fn in self.file_urls:
            url = f"{BASE_URL}/{fn}"
            local_path = os.path.join(HF_DATA_DIR, fn)
            if not os.path.exists(local_path):
                os.makedirs(os.path.dirname(local_path), exist_ok=True)
                self._download_with_retry(url, local_path)
            # Minimal projection to {prompt, response} happens here
            # Replace with actual parquet read + projection
            import pyarrow.parquet as pq
            table = pq.read_table(local_path, columns=["prompt", "response"])
            for row in table.to_pylist():
                yield row

# Example dataset and loader (replace with real model/train loop)
dataset = CDNParquetIterable([f"{BASE_URL}/{f}" for f in FILES[:10]])  # subset for demo
loader = DataLoader(dataset, batch_size=4)

for i, batch in enumerate(loader):
    print(f"Batch {i}: {len(batch)} rows")
    if i >= 2:
        break

print("CDN-bypass data loader OK")
PY
