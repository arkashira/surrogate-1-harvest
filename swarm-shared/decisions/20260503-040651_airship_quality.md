Ship a **CDN-only parquet loader** + **Lightning idle-resilient runner** to eliminate HuggingFace API rate limits and Studio idle-timeout failures.

- Reuse running studios to save quota.

### 2) CDN-only parquet loader (surrogate/training/cdn_loader.py)
```python
import pandas as pd
import pyarrow as pa
import json
import os
from typing import List, Dict
from huggingface_hub import hf_hub_download

HF_REPO = "datasets/your-org/surrogate-mirror"
CDN_ROOT = f"https://huggingface.co/datasets/{HF_REPO}/resolve/main"

def deterministic_repo_index(slug: str, n_siblings: int = 5) -> int:
    """Map slug to sibling repo index to spread HF commit load."""
    return hash(slug) % n_siblings

def cdn_parquet_loader(file_list_path: str, batch_size: int = 1000):
    """
    Yield {prompt, response} rows from parquet files via CDN (no HF API auth).
    """
    with open(file_list_path) as f:
        files = json.load(f)

    for rfilename in files:
        # CDN URL bypasses API rate limits
        url = f"{CDN_ROOT}/{rfilename}"
        try:
            df = pd.read_parquet(url)
        except Exception as e:
            # fallback: download via hf_hub_download (uses CDN under the hood)
            local_path = hf_hub_download(HF_REPO, filename=rfilename)
            df = pd.read_parquet(local_path)

        # Project to minimal schema expected by surrogate training
        if "prompt" not in df.columns or "response" not in df.columns:
            continue

        for _, row in df.iterrows():
            yield {"prompt": row["prompt"], "response": row["response"]}

def buffered_loader(file_list_path: str, buffer_size: int = 10_000) -> List[Dict]:
    buffer = []
    for item in cdn_parquet_loader(file_list_path):
        buffer.append(item)
        if len(buffer) >= buffer_size:
            yield buffer
            buffer = []
    if buffer:
        yield buffer
```

### 3) Lightning idle-resilient runner (surrogate/training/lightning_runner.py)
```python
import time
from lightning import LightningWork, LightningFlow, LightningApp, Machine
from lightning.app.storage import Drive

class TrainingWork(LightningWork):
    def __init__(self, file_list_path: str, **kwargs):
        super().__init__(**kwargs)
        self.file_list_path = file_list_path
        self.machine = Machine.L40S
        self._studio = None

    def run(self):
        # Import here to avoid Mac-side heavy deps
        from surrogate.training.train import train_step
        from surrogate.training.cdn_loader import buffered_loader

        for batch in buffered_loader(self.file_list_path):
            train_step(batch)
            # Optional: yield metrics to UI

class ResilientFlow(LightningFlow):
    def __init__(self):
        super().__init__()
        self.training = TrainingWork(file_list_path="file_list_2026-04-29.json")

    def configure_layout(self):
        return [{"name": "Train", "content": self.training}]

    def run_once(self):
        # Reuse running studio; restart if stopped (idle timeout)
        if self.training.has_stopped:
            self.training.start(machine=self.training.machine)
        elif not self.training.is_running:
            self.training.run()

app = LightningApp(ResilientFlow())
```
