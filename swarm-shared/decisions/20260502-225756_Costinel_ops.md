```python
# routes/cost_anomaly.py  (or create if missing)
from datetime import datetime, date
from fastapi import APIRouter, Query
from services.knowledge_rag import get_top_hub_for_date  # existing or new thin wrapper

router = APIRouter(prefix="/api/v1/cost-anomaly", tags=["cost-anomaly"])

@router.get("/signal/top-hub")
async def top_hub_signal(
    for_date: str = Query(None, regex=r"^\d{4}-\d{2}-\d{2}$", description="YYYY-MM-DD (defaults today)")
):
    """
    Sense + Signal: return the strongest hub insight for the requested date.
    No execution. Read-only.
    """
    target_date = date.fromisoformat(for_date) if for_date else date.today()
    insight = get_top_hub_for_date(target_date)

    return {
        "signal_type": "top_hub",
        "hub": insight.get("hub"),
        "score": insight.get("score", 0.0),
        "context": insight.get("context", ""),
        "related_docs": insight.get("related_docs", []),
        "for_date": target_date.isoformat(),
        "generated_at": datetime.utcnow().isoformat() + "Z",
    }
```

from typing import Dict, List, Any
from datetime import date

def get_top_hub_for_date(target_date: date) -> Dict[str, Any]:
    """
    Query knowledge graph for the most-connected hub and related docs for target_date.
    Falls back gracefully if graph unavailable.
    """
    # If existing client exposes something like `graph.top_hub(date=...)`, use it.
    # Below is a minimal safe implementation.
    try:
        # Example placeholder — replace with real graph query
        from graph.client import GraphClient  # hypothetical existing import
        g = GraphClient()
        node = g.top_hub(for_date=target_date.isoformat())
        return {
            "hub": node.get("name"),
            "score": float(node.get("centrality", 0.0)),
            "context": node.get("summary", f"Top hub {node.get('name')} for cost signals on {target_date}"),
            "related_docs": node.get("related_docs", []),
        }
    except Exception:
        # Graceful fallback: return neutral signal
        return {
            "hub": "MOC",
            "score": 0.0,
            "context": "Default hub (MOC) — graph unavailable",
            "related_docs": [],
        }
