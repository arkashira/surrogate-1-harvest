def deterministic_shard(key: str, total: int) -> int:
    return int(hashlib.sha256(key.encode()).hexdigest(), 16) % total

def deterministic_repo_slug(key: str, repos: List[str]) -> str:
    """

    idx = int(hashlib.sha256(key.encode()).hexdigest(), 16) % len(repos)

def safe_request(method: str, url: str, **kwargs):
    for attempt, wait in enumerate(RETRY_BACKOFF):
        try:
            resp = requests.request(method, url, timeout=TIMEOUT, **kwargs)
            resp.raise_for_status()
            return resp
        except Exception as exc:
            log.warning("Request failed (attempt %s): %s — %s", attempt + 1, url, exc)
            if attempt == len(RETRY_BACKOFF) - 1:
                raise
            time.sleep(wait)
    raise RuntimeError("unreachable")

def list_date_files(repo_id: str, date_folder: str) -> List[str]:
    """
    Single API call to list files in date_folder (non-recursive preferred).
    Falls back to recursive if needed.
    """
    try:
        items = list_repo_tree(repo_id=repo_id, path=date_folder, recursive=False)
        files = [it.rfilename for it in items if it.type == "file"]
        if files:
            return files
    except Exception as exc:
        log.warning("Non-recursive list failed, trying recursive: %s", exc)

    try:
        items = list_repo_tree(repo_id=repo_id, path=date_folder, recursive=True)
        return [it.rfilename for it in items if it.type == "file"]
    except Exception as exc:
        log.error("Failed to list repo tree: %s", exc)
        raise

def project_to_pair(raw_obj: Dict) -> Optional[Dict[str, str]]:
    """
    Project heterogeneous HF dataset objects to {prompt, response}.
    Heuristic: look for common field names.
    """
    if not isinstance(raw_obj, dict):
        return None

    # If already shaped
    if "prompt" in raw_obj and "response" in raw_obj:
        return {"prompt": str(raw_obj["prompt"]), "response": str(raw_obj["response"])}

    # Common alternates
    prompt_keys = {"prompt", "instruction", "input", "question", "text"}
    response_keys = {"response", "output", "answer", "completion", "result"}

    pk = next((k for k in raw_obj if k in prompt_keys), None)
    rk = next((k for k in raw_obj if k in response_keys), None)

    if pk and rk:
        return {"prompt": str(raw_obj[pk]), "response": str(raw_obj[rk])}

    # Fallback: if exactly two fields, treat first as prompt, second as response
    keys = [k for k in raw_obj if isinstance(raw_obj[k], (str, int, float, bool))]
    if len(keys) == 2:
        return {"prompt": str(raw_obj[keys[0]]), "response": str(raw_obj[keys[1]])}

    return None
