```python
#!/usr/bin/env python3
"""
Enforce remote-only execution for training/ingest.
Raises immediately on local/Mac runs or direct HF model downloads.
"""
import os
import platform
import sys
from pathlib import Path


def _fail(msg: str) -> None:
    print(f"GUARDRAIL VIOLATION: {msg}", file=sys.stderr)
    sys.exit(1)


def enforce_remote_only() -> None:
    """
    Allow only known remote executors (Lightning, Kaggle, Cerebras).
    Block local laptops and interactive notebooks.
    """
    is_local = (
        platform.system() == "Darwin"
        or platform.system() == "Windows"
        or os.getenv("KAGGLE_CONTAINER_NAME") is None
        and os.getenv("LIGHTNING_CLOUD_BUILD_ID") is None
        and os.getenv("CEREBRAS_RUNNER") is None
        and os.getenv("CI") is None
    )
    if is_local:
        _fail(
            "Local/Mac execution blocked. "
            "Run via Lightning/Kaggle/Cerebras or CI only."
        )

    # Prevent accidental model downloads in user code
    try:
        import transformers  # noqa: F401
        _fail(
            "transformers import detected. "
            "Use CDN manifest + safetensors only; no from_pretrained()."
        )
    except Exception:
        pass  # OK: not installed


def enforce_lightning_quota_safe() -> None:
    """
    Prevent quota burn by refusing to create a studio
    when one with the same name is already running.
    """
    studio_name = os.getenv("STUDIO_NAME")
    if not studio_name:
        return
    try:
        from lightning import Lightning
        cloud = Lightning()
        running = [
            c
            for c in cloud.client.lightningapp_instance_list()
            if c.name == studio_name and c.phase.value in ("RUNNING", "STARTING")
        ]
        if running:
            print(f"Reusing running studio: {studio_name}", file=sys.stderr)
            sys.exit(0)
    except Exception:
        pass  # Best-effort; let launcher proceed if API unavailable
```

### 2) Shared utilities

`/opt/axentx/vanguard/ops/utils.py`
```python
#!/usr/bin/env python3
"""
Shared ops utilities for vanguard.
"""
import json
import hashlib
import os
import subprocess
import sys
from pathlib import Path
from typing import Any, Dict, List, Optional

HF_TOKEN = os.getenv("HF_TOKEN", "")
REPO_ROOT = Path(__file__).parent.parent


def sha256_file(path: str) -> str:
    h = hashlib.sha256()
    with open(path, "rb") as f:
        for chunk in iter(lambda: f.read(16 * 1024), b""):
            h.update(chunk)
    return h.hexdigest()


def run_cmd(cmd: List[str], cwd: Optional[str] = None, check: bool = True) -> str:
    result = subprocess.run(
        cmd, capture_output=True, text=True, cwd=cwd or REPO_ROOT
    )
    if check and result.returncode != 0:
        raise RuntimeError(f"Command failed: {cmd}\n{result.stderr}")
    return result.stdout


def load_manifest(manifest_path: Path) -> List[Dict[str, Any]]:
    if not manifest_path.exists():
        return []
    with open(manifest_path) as f:
        return [json.loads(line) for line in f if line.strip()]


def save_manifest(entries: List[Dict[str, Any]], manifest_path: Path) -> None:
    manifest_path.parent.mkdir(parents=True, exist_ok=True)
    with open(manifest_path, "w") as f:
        for e in entries:
            f.write(json.dumps(e, sort_keys=True) + "\n")
```

#!/usr/bin/env python3
"""
Generate a CDN-first manifest for a HuggingFace dataset folder.

Usage:
  HF_TOKEN=hf_xxx python generate_cdn_manifest.py \
    --repo datasets/mycorp/vanguard-ingest \
    --folder 2026-05-03 \
    --out manifests/2026-05-03.manifest.jsonl

Behavior:
- ONE HF API call (list_repo_tree, non-recursive) per folder.
- Emits {path, cdn_url, sha256?, size}.
- SHA-256 computed only when file already exists locally.
- Idempotent: exits 0 if manifest exists and non-empty.
"""
import argparse
import json
import os
import sys
from pathlib import Path
from typing import Dict, Any, List

import requests

from ops.utils import HF_TOKEN, save_manifest, run_cmd


def list_folder_files(repo: str, folder: str) -> List[Dict[str, Any]]:
    """
    Single API call: list_repo_tree(path=folder, recursive=False).
    Falls back to REST when huggingface_hub unavailable.
    """
    try:
        from huggingface_hub import HfApi

        api = HfApi(token=HF_TOKEN)
        tree = api.list_repo_tree(
            repo=repo, path=folder, recursive=False, repo_type="dataset"
        )
        files = [t for t in tree if t.type == "file"]
        return [{"path": str(f.path), "size": f.size} for f in files]
    except Exception:
        url = f"https://huggingface.co/api/datasets/{repo}/tree/{folder}"
        params = {"recursive": "false", "expand": "false"}
        headers = {"Authorization": f"Bearer {HF_TOKEN}"} if HF_TOKEN else {}
        resp = requests.get(url, params=params, headers=headers, timeout=30)
        if resp.status_code == 429:
            print("HF API 429: wait 360s before retry", file=sys.stderr)
            sys.exit(1)
        resp.raise_for_status()
        items = resp.json()
        return [{"path": i["path"], "size": i.get("size")} for i in items if i["type"] == "file"]


def build_manifest(repo: str, folder: str, local_cache_root: Path) -> List[Dict[str, Any]]:
    entries: List[Dict[str, Any]] = []
    infos = list_folder_files(repo, folder)
    for info in infos:
        rel = info["path"]
        cdn_url = f"https://huggingface.co/datasets/{repo}/resolve/main/{rel}"
        entry: Dict[str, Any] = {
            "path": rel,
            "cdn_url": cdn_url,
            "size": info.get("size"),
        }
        local_path = local_cache_root / rel
        if local_path.exists() and local_path.is_file():
            entry["sha256"] = run_cmd(["sha256sum", str(local_path)]).split()[0]
        entries.append(entry)
    return entries
