## Code Snippets

### 1) `scripts/build_cdn_manifest.py`
```python
#!/usr/bin/env python3
"""
Mac-only orchestration script.
Run after rate-limit window clears.
Produces: training_manifest_YYYYMMDD.json
"""
import json
import os
from datetime import datetime, timezone
from pathlib import Path

from huggingface_hub import HfApi

HF_REPO = os.getenv("HF_DATASET_REPO", "axentx/surrogate-dataset")
DATE_FOLDER = os.getenv("DATE_FOLDER", datetime.now(timezone.utc).strftime("%Y-%m-%d"))
OUT_DIR = Path(__file__).parent.parent / "surrogate" / "training" / "manifests"
OUT_DIR.mkdir(parents=True, exist_ok=True)

def build_manifest() -> None:
    api = HfApi()
    # Single API call; non-recursive to avoid pagination explosion
    entries = api.list_repo_tree(
        repo_id=HF_REPO,
        path=DATE_FOLDER,
        repo_type="dataset",
        recursive=False,
    )
    files = [e.rfilename for e in entries if e.rfilename.endswith(".parquet")]
    manifest = {
        "repo": HF_REPO,
        "date": DATE_FOLDER,
        "created_utc": datetime.now(timezone.utc).isoformat(),
        "files": sorted(files),
        # CDN URLs bypass HF API auth/rate limits
        "cdn_base": f"https://huggingface.co/datasets/{HF_REPO}/resolve/main",
    }
    out_path = OUT_DIR / f"training_manifest_{DATE_FOLDER.replace('-', '')}.json"
    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"Manifest written: {out_path} ({len(files)} files)")

if __name__ == "__main__":
    build_manifest()
```

### 2) `surrogate/training/train.py` (CDN-only dataloader excerpt)
```python
import json
import os
from pathlib import Path
from typing import List, Dict

import pyarrow.parquet as pq
import requests
from torch.utils.data import IterableDataset

class CDNParquetDataset(IterableDataset):
    """
    Zero HF API calls during training.
    Uses CDN URLs only (no Authorization header).
    """
    def __init__(self, manifest_path: str):
        manifest_path = Path(manifest_path)
        self.manifest = json.loads(manifest_path.read_text())
        self.cdn_base = self.manifest["cdn_base"]
        self.files: List[str] = self.manifest["files"]

    def _stream_file(self, rel_path: str):
        url = f"{self.cdn_base}/{rel_path}"
        # Streaming download; retry/backoff handled by requests default
        with requests.get(url, stream=True, timeout=60) as r:
            r.raise_for_status()
            # Write to temp file for pyarrow (avoids mixed schema in memory)
            import tempfile
            with tempfile.NamedTemporaryFile(suffix=".parquet", delete=False) as f:
                for chunk in r.iter_content(chunk_size=8192):
                    f.write(chunk)
                f.flush()
                os.fsync(f.fileno())
                temp_path = f.name
            try:
                table = pq.read_table(temp_path, columns=["prompt", "response"])
                for batch in table.to_batches(max_chunksize=1024):
                    df = batch.slice(0, batch.num_rows).to_pydict()
                    for prompt, response in zip(df["prompt"], df["response"]):
                        yield {"prompt": prompt, "response": response}
            finally:
                os.unlink(temp_path)

    def __iter__(self):
        for rel_path in self.files:
            yield from self._stream_file(rel_path)
```

### 3) `scripts/studio_guard.py`
```python
#!/usr/bin/env python3
"""
Lightning Studio lifecycle guard.
Prevents idle-stop training loss by restarting stopped studios.
Run via cron every 10-15 minutes.
"""
import os
import time
from lightning_sdk import Client, Studio, Machine

LIGHTNING_EMAIL = os.getenv("LIGHTNING_EMAIL")
LIGHTNING_PASS  = os.getenv("LIGHTNING_PASS")
STUDIO_NAME     = os.getenv("STUDIO_NAME", "surrogate-train")
MACHINE         = Machine.L40S  # fallback if H200 unavailable

def guard() -> None:
    client = Client()
    client.login(email=LIGHTNING_EMAIL, password=LIGHTNING_PASS)
    teamspace = client.teamspace()

    studio = None
    for s in teamspace.studios:
        if s.name == STUDIO_NAME:
            studio = s
            break

    if studio is None:
        print(f"Studio {STUDIO_NAME} not found; creating...")
        studio = teamspace.studios.create(
            name=STUDIO_NAME,
            machine=MACHINE,
            create_ok=True,
        )
        studio.start(machine=MACHINE)
        return

    # Refresh state
    studio = teamspace.studios.get(STUDIO_NAME)
    if studio.status != "Running":
        print(f"Studio {STUDIO_NAME} is {studio.status}; restarting...")
        studio.start(machine=MACHINE)
    else:
        print(f"Studio {STUDIO_NAME} is Running (OK)")

if __name__ == "__main__":
    guard()
```

### 4) Cron + compose updates

`docker-compose.microservices.yml` (add to surrogate service):
```yaml
    volumes:
      - ./surrogate/training/manifests:/app/surrogate/training/manifests
```

Crontab (on Mac orchestration host):
```bash
SHELL=/bin/bash
# Build manifest daily at 02:00 UTC (after HF rate-limit window)
0 2 * * * cd /opt/axentx/airship && /usr/bin/env bash scripts/build_cdn_manifest.sh >> /var/log/airship_cdn_manifest.log 2>&1
# Guard studio every 10 minutes
*/10 * * * * cd /opt/axentx/airship && /usr/bin/env bash scripts/studio_guard.sh >> /var/log/airship_studio_guard.log 2>&1
```

Wrapper scripts (`build_cdn_manifest.sh`, `studio_guard.sh`) must include:
```bash
#!/usr/bin/env bash
set -euo pipefail
cd "$(dirname "$0")/.."
exec python3 -m scripts."${0##*/}" "$@"
```
and be `chmod +x`.
