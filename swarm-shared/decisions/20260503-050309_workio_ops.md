# workio / ops

## Final consolidated solution

### Diagnosis (merged)
- **Missing idempotency**: LINE retries and client double-taps can deliver the same `webhookEventId` multiple times, creating duplicate clock rows and corrupting daily totals.
- **No replay-window / TTL**: Stale retries beyond the business day can still insert records and confuse attendance state.
- **Race conditions under concurrency**: Parallel deliveries for the same user can interleave clock-in/out rows and produce invalid state transitions.
- **Missing guardrails**: No unique index on the deduplication key causes full scans on duplicate checks; no audit trail for accepted/rejected events; no operational metric for duplicate ingestion rate.
- **Schema drift risk**: Ad-hoc columns added later could complicate mirroring and break deterministic projections.

### Scope
- Minimal schema diff + single middleware function.
- No frontend changes.
- Deterministic, testable behavior with concrete rollback/verification steps.

### Schema (`workio/server/src/db/schema.sql`)
```sql
-- Idempotent LINE webhook event log (24h effective window)
CREATE TABLE IF NOT EXISTS webhook_events (
  id              BIGSERIAL PRIMARY KEY,
  event_id        TEXT NOT NULL,            -- deterministic key: userId:webhookEventId
  source_user_id  TEXT NOT NULL,
  tenant_id       TEXT NOT NULL,
  event_type      TEXT NOT NULL,
  raw_payload     JSONB NOT NULL,
  created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Prevent duplicate processing (same LINE event)
CREATE UNIQUE INDEX IF NOT EXISTS uq_webhook_events_event_id ON webhook_events (event_id);

-- Fast lookup for TTL/retention and duplicate checks within window
CREATE INDEX IF NOT EXISTS ix_webhook_events_event_id_created_at ON webhook_events (event_id, created_at);
CREATE INDEX IF NOT EXISTS ix_webhook_events_created_at ON webhook_events (created_at)
WHERE created_at < NOW() - INTERVAL '24 hours';

-- Clock records with valid state transitions
CREATE TABLE IF NOT EXISTS clock_records (
  id            BIGSERIAL PRIMARY KEY,
  tenant_id     TEXT NOT NULL,
  user_id       TEXT NOT NULL,
  event_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  type          TEXT NOT NULL CHECK (type IN ('in', 'out')),
  latitude      NUMERIC(9,6),
  longitude     NUMERIC(9,6),
  created_at    TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Prevent multiple open "in" rows for same user (best-effort concurrency guard)
-- Allows one row with type='in' where no later row (by event_at, id) exists for that user.
CREATE UNIQUE INDEX IF NOT EXISTS ux_clock_records_one_open_in
ON clock_records (user_id)
WHERE type = 'in'
  AND NOT EXISTS (
    SELECT 1 FROM clock_records later
    WHERE later.user_id = clock_records.user_id
      AND (later.event_at > clock_records.event_at OR (later.event_at = clock_records.event_at AND later.id > clock_records.id))
      AND later.type IN ('in', 'out')
  );

-- Optional daily cleanup job (run via cron/systemd timer)
-- DELETE FROM webhook_events WHERE created_at < NOW() - INTERVAL '24 hours';
```

### Middleware (`workio/server/src/middleware/lineWebhook.ts`)
```ts
import { Request, Response, NextFunction } from 'express';
import crypto from 'crypto';
import { Pool } from 'pg';

const pool = new Pool({ connectionString: process.env.DATABASE_URL });

const CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';
const REPLAY_WINDOW_MS = 24 * 60 * 60 * 1000; // 24h

function verifySignature(body: string, signature: string): boolean {
  const expected = crypto
    .createHmac('sha256', CHANNEL_SECRET)
    .update(body)
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

export async function lineWebhookHandler(req: Request, res: Response, next: NextFunction) {
  const signature = req.headers['x-line-signature'] as string;
  const rawBody = JSON.stringify(req.body);

  if (!signature || !verifySignature(rawBody, signature)) {
    return res.status(401).json({ error: 'Invalid signature' });
  }

  const events = req.body.events;
  if (!Array.isArray(events) || events.length === 0) {
    return res.status(200).json({ ok: true });
  }

  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    for (const ev of events) {
      const userId = ev?.source?.userId;
      const webhookEventId = ev?.webhookEventId;
      if (!userId || !webhookEventId) continue;

      // Deterministic idempotency key
      const eventId = `${userId}:${webhookEventId}`;
      const tenantId = ev.source.tenantId || 'default';

      // Upsert idempotency record; unique index guarantees exactly-one semantics under races.
      // ON CONFLICT DO NOTHING avoids throwing on duplicates; we then skip processing.
      const upserted = await client.query(
        `INSERT INTO webhook_events (event_id, source_user_id, tenant_id, event_type, raw_payload)
         VALUES ($1, $2, $3, $4, $5)
         ON CONFLICT (event_id) DO NOTHING
         RETURNING id`,
        [eventId, userId, tenantId, ev.type, ev]
      );

      // If not inserted, this is a duplicate within the unique-key scope -> skip processing
      if (upserted.rowCount === 0) {
        continue;
      }

      // Process clock in/out messages only
      if (ev.type === 'message' && ev.message?.type === 'text') {
        const text = (ev.message.text || '').trim().toLowerCase();
        const type = text === 'in' || text === 'clock in' ? 'in' : text === 'out' || text === 'clock out' ? 'out' : null;
        if (type) {
          // Insert new record; state correctness enforced by application logic + partial unique index.
          await client.query(
            `INSERT INTO clock_records (tenant_id, user_id, type, latitude, longitude, event_at)
             VALUES ($1, $2, $3, $4, $5, NOW())`,
            [tenantId, userId, type, null, null]
          );
        }
      }
    }

    await client.query('COMMIT');
    res.status(200).json({ ok: true });
  } catch (err: any) {
    await client.query('ROLLBACK');
    // Unique violation on uq_webhook_events_event_id is safe duplicate -> accept
    if (err.code === '23505') {
      return res.status(200).json({ ok: true });
    }
    next(err);
  } finally {
    client.release();
  }
}
```

### Verification (concrete, actionable)
1. **Duplicate delivery test**  
   - POST the same payload (same `webhookEventId` and `userId`) twice to `/webhook/line`.  
   - Expect: only one row in `clock_records` for that user, second request returns 200 with no error.

2. **Concurrent duplicate race test**  
   - Fire two parallel requests with identical `eventId`.  
   - Expect: one succeeds with insert into `webhook_events`, one hits unique constraint and is treated as duplicate (200 OK). No duplicate `clock_records`.

3. **Replay-window / TTL test**  
   - Manually insert a `webhook_events` row with `created_at` 25h old.  
   - Verify index `ix_webhook_events_created_at` exists and run:  
     `DELETE FROM webhook_events WHERE created_at < NOW() - INTERVAL '24 hours';`  
   - Expect: old row is removable via that predicate (used by retention job).

4. **State transition smoke test**  
   - Send valid `in` then `out` texts from same LINE user.  
   - Expect: two rows in `clock_records` with types `in` and `out` and increasing timestamps.  
   - Attempt to send two consecutive `in` texts without an intervening `out`; second insert should be rejected by partial unique index (or handled gracefully by application logic) to prevent ambiguous open state.
