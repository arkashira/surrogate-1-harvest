### 2) Tenant cache (LRU, 10m TTL, stale-while-revalidate)
`src/middleware/tenantCache.ts`
```ts
import { LRUCache } from 'lru-cache';
import { getTenantByChannelId } from '../db/repositories/tenantRepository';

export type Tenant = { id: string; name: string; channelId: string };

const tenantCache = new LRUCache<string, Tenant>({
  max: 500,
  ttl: 1000 * 60 * 10, // 10 minutes
  allowStale: true,
  updateAgeOnGet: true,
});

export async function resolveTenant(channelId: string): Promise<Tenant | null> {
  const cached = tenantCache.get(channelId);
  if (cached) return cached;

  try {
    const tenant = await getTenantByChannelId(channelId);
    if (tenant) tenantCache.set(channelId, tenant);
    return tenant || null;
  } catch {
    // On DB failure, return stale value if present (graceful degradation)
    return tenantCache.get(channelId) || null;
  }
}

export function invalidateTenant(channelId: string) {
  tenantCache.delete(channelId);
}
```

### 3) Request tracing (correlation IDs + structured logs)
`src/middleware/requestTracing.ts`
```ts
import { Request, Response, NextFunction } from 'express';
import crypto from 'crypto';
import logger from '../lib/logger';

export function requestTracing(req: Request, res: Response, next: NextFunction) {
  const traceId = (req.headers['x-trace-id'] as string) || crypto.randomUUID();
  const spanId = crypto.randomUUID().slice(0, 8);
  (req as any).trace = { traceId, spanId };

  res.setHeader('x-trace-id', traceId);
  res.setHeader('x-span-id', spanId);

  const start = Date.now();
  res.on('finish', () => {
    logger.info({
      traceId,
      spanId,
      method: req.method,
      path: req.path,
      status: res.statusCode,
      durationMs: Date.now() - start,
      userAgent: req.get('User-Agent'),
    });
  });

  next();
}
```

### 4) LINE signature verification (fast-fail, constant-time)
`src/middleware/lineSignature.ts`
```ts
import { Request, Response, NextFunction } from 'express';
import crypto from 'crypto';

const LINE_CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';

export function verifyLineSignature(req: Request, res: Response, next: NextFunction) {
  const signature = req.headers['x-line-signature'] as string;
  if (!signature) return res.status(401).json({ error: 'Missing x-line-signature' });

  const body = JSON.stringify(req.body);
  const expected = crypto
    .createHmac('sha256', LINE_CHANNEL_SECRET)
    .update(body, 'utf8')
    .digest('base64');

  // constant-time comparison to prevent timing attacks
  const expectedBuf = Buffer.from(expected);
  const receivedBuf = Buffer.from(signature);
  const valid =
    expectedBuf.length === receivedBuf.length &&
    crypto.timingSafeEqual(expectedBuf, receivedBuf);

  if (!valid) return res.status(401).json({ error: 'Invalid signature' });
  next();
}
```

const logger = winston.createLogger({

### 6) Health & readiness endpoints
`src/routes/health.ts`
```ts
import { Router } from 'express';
import { getPool } from '../db';
import { resolveTenant } from '../middleware/tenantCache';

const router = Router();

// Liveness: process is up
router.get('/health', (_req, res) => {
  res.json({ status: 'ok', uptime: process.uptime() });
});

// Readiness: dependencies available (DB + minimal tenant resolution)
router.get('/ready', async (_req, res) => {
  try {
    const pool = getPool();
    const client = await pool.connect();
    await client.query('SELECT 1');
    client.release();

    // optional: validate at least one tenant can be resolved (if applicable)
    const sampleChannelId = process.env.SAMPLE_CHANNEL_ID;
    if (sampleChannelId) {
      const tenant = await resolveTenant(sampleChannelId);
      if (!tenant) {
        return res.status(503).json({ status: 'not ready', reason: 'sample tenant not found' });
      }
    }

    res.json({ status: 'ok' });
  } catch (err) {
    res.status(503).json({ status: 'not ready', error: String(err) });
  }
});

export default router;
```

### 7) Wire everything in the main entry point
`src/index.ts` (or `src/app.ts`)
```ts
import express from 'express';
import bodyParser from 'body-parser';
import logger from './lib/logger';
import { requestTracing } from './middleware/requestTracing';
import { verifyLineSignature } from './middleware/lineSignature';
import lineWebhookRouter from './routes/lineWebhook';
import healthRouter from './routes/health';

const app = express();

app.use(bodyParser.json({ limit: '1mb' }));
app.use(requestTracing);

// Health endpoints (no tracing/signature required)
app.use('/health', healthRouter);

// LINE webhook: signature -> tenant resolution -> handler
app.use('/webhook/line', verifyLineSignature, lineWebhookRouter);

// Central error handler
app.use((err: Error, _req: express.Request, res: express.Response, _next: express.NextFunction) => {
  logger.error({ error: err.message, stack: err.stack });
  res.status(500).json({ error: 'Internal Server Error' });
});

const PORT = process.env.PORT || 3000;
app.listen(PORT, () => {
  logger.info({ msg: 'Server started', port: PORT });
});
```

### 8) Update LINE webhook handler to use tenant cache
`src/routes/lineWebhook.ts`
```ts
import { Router } from 'express';
import { resolveTenant } from '../middleware/tenantCache';
import logger from '../lib/logger';

const router = Router();

router.post('/', async (req, res) => {
  const channelId = req.body.channelId;
  if (!channelId) return res.status(400).json({ error: 'channelId required' });

  const tenant = await resolveTenant(channelId);
  if (!tenant) return res.status(404).json({ error: 'Tenant not found' });

  try {
    // process webhook for tenant...
    logger.info({ tenantId: tenant.id, event: 'line.webhook.received' });
    res.json({ ok: true });
  } catch (err) {
    logger.error({ tenantId: tenant.id, error: String(err) });
    res.status(500).json({ error: 'Processing failed' });
  }
});

export default router;
```
