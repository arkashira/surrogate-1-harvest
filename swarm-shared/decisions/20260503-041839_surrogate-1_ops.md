## Implementation Plan (≤2h)

**Highest-value improvement**: Replace fragile shell-based ingestion with a **manifest-driven, CDN-bypass Python worker** that eliminates HF API rate limits during training data loads and prevents mixed-schema CastErrors.

### Changes

1. Add `bin/ingest_worker.py` — single-file, manifest-first worker:
   - Mac/Lightning runner calls HF API **once** (after rate-limit window) to list a date folder via `list_repo_tree(path, recursive=False)` → writes `manifest.json` (list of `{path, sha256, size}`).
   - Worker embeds that manifest and downloads **only via CDN** (`https://huggingface.co/datasets/.../resolve/main/...`) with no Authorization header — bypasses `/api/` rate limits entirely.
   - Per-file schema projection: reads raw bytes, detects schema (JSONL/Parquet/CSV), projects to `{prompt, response}` only, adds deterministic `id = hash(slug+ts)`, drops all other columns (no `source`, no `ts`).
   - Dedup via central `lib/dedup.py` md5 store (unchanged).
   - Outputs `batches/public-merged/{date}/shard{N}-{HHMMSS}.jsonl` with one object per line.

2. Update `bin/dataset-enrich.sh` → thin wrapper that:
   - Exports `PYTHONUNBUFFERED=1`, `SHELL=/bin/bash`.
   - Invokes `python3 bin/ingest_worker.py "$@"` with proper quoting.
   - Falls back to `bash` invocation if called from cron (shebang + executable bit already set).

3. Update `.github/workflows/ingest.yml` matrix to:

   - Use `actions/setup-python@v5` (3.11).

"""
Manifest-driven, CDN-bypass ingest worker for surrogate-1.

Usage:

HF_CDN = "https://huggingface.co/datasets"
CHUNK_SIZE = 8 * 1024 * 1024  # 8 MB

# Minimal schema projection keys
PROMPT_KEYS = {"prompt", "instruction", "question", "input"}
RESPONSE_KEYS = {"response", "answer", "output", "completion"}

# HF API rate-limit handling (only used once per worker for listing)
HF_TOKEN = os.getenv("HF_TOKEN")
HF_HEADERS = {"Authorization": f"Bearer {HF_TOKEN}"} if HF_TOKEN else {}


def hf_api_list_tree(repo: str, path: str) -> List[Dict[str, Any]]:
    """
    List files in a folder (non-recursive).
    Falls back to local manifest.json if present (dev convenience).
    """
    manifest_path = Path(path) / "manifest.json"
    if manifest_path.exists():
        with open(manifest_path) as f:
            return json.load(f)

    url = f"https://huggingface.co/api/datasets/{repo}/tree"
    params = {"path": path, "recursive": "false"}
    resp = requests.get(url, headers=HF_HEADERS, params=params, timeout=30)
    if resp.status_code == 429:
        wait = int(resp.headers.get("retry-after", 360))
        print(f"HF API 429, waiting {wait}s", file=sys.stderr)
        time.sleep(wait)
        return hf_api_list_tree(repo, path)
    resp.raise_for_status()
    return resp.json()


def cdn_download(url: str, dst: Path) -> None:
    """Download via CDN with no Authorization header."""
    with requests.get(url, stream=True, timeout=60) as r:
        r.raise_for_status()
        with open(dst, "wb") as f:
            for chunk in r.iter_content(chunk_size=CHUNK_SIZE):
                f.write(chunk)


def normalize_row(obj: Dict[str, Any]) -> Dict[str, Any]:
    """Project raw row to {prompt, response} only."""
    keys = {k.lower() for k in obj.keys()}
    prompt_key = next((k for k in obj.keys() if k.lower() in PROMPT_KEYS), None)
    response_key = next((k for k in obj.keys() if k.lower() in RESPONSE_KEYS), None)

    prompt = obj[prompt_key] if prompt_key else ""
    response = obj[response_key] if response_key else ""

    # Coerce to string
    if not isinstance(prompt, str):
        prompt = json.dumps(prompt, ensure_ascii=False)
    if not isinstance(response, str):
        response = json.dumps(response, ensure_ascii=False)

    return {"prompt": prompt.strip(), "response": response.strip()}


def process_file(file_path: Path, rel_key: str) -> List[Dict[str, Any]]:
    """Load file and return list of normalized rows."""
    suffix = file_path.suffix.lower()
    rows: List[Dict[str, Any]] = []

    try:
        if suffix == ".jsonl":
            with open(file_path) as f:
                for line in f:
                    line = line.strip()
                    if not line:
                        continue
                    obj = json.loads(line)
                    rows.append(normalize_row(obj))

        elif suffix == ".parquet":
            table = pq.read_table(file_path)
            df = table.to_pandas()
            for _, obj in df.iterrows():
                rows.append(normalize_row(obj.to_dict()))

        elif suffix == ".csv":
            df = pd.read_csv(file_path)
            for _, obj in df.iterrows():
                rows.append(normalize_row(obj.to_dict()))

        else:
            print(f"Unsupported file {rel_key} (suffix {suffix})", file=sys.stderr)
            return []
    except Exception as e:
        print(f"Error processing {rel_key}: {e}", file=sys.stderr)
        return []
    finally:
        # Clean up temp file
        try:
            file_path.unlink()
        except Exception:
            pass

    return rows


def build_manifest(repo: str, date_folder: str) -> List[Dict[str, Any]]:
    """List files in date folder via HF API (once per worker)."""
    entries = hf_api_list_tree(repo, date_folder)
    # Keep only files (not subfolders)
    files = [e for e in entries if e.get("type") == "file"]
    print(f"Found {len(files)} files in {date_folder}", file=sys.stderr)
    return files
