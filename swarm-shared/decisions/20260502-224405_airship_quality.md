- Use **`nohup`/background for training** so cron job finishes quickly (Candidate 1).
- Add **locking + retries + timeouts** (Candidate 2) to avoid overlapping cron runs and transient failures.
- Provide **concrete crontab + systemd timer** options so you can pick one and deploy immediately.

## 3) Orchestrator (cron-safe) — `bin/airship-discover`

```bash
#!/usr/bin/env bash
# airship-discover — cron-safe orchestrator for airship quality pipeline
# Usage: /opt/axentx/airship/bin/airship-discover [--date YYYY-MM-DD]
# Cron:  SHELL=/bin/bash
#        0 2 * * * /opt/axentx/airship/bin/airship-discover >> /var/log/airship-discover.log 2>&1

set -euo pipefail
export SHELL=/bin/bash

BASE_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")/.." && pwd)"
# shellcheck source=/dev/null
source "${BASE_DIR}/config/airship.conf" 2>/dev/null || true

OUT_DIR="${OUT_DIR:-/opt/axentx/airship/out}"
MANIFEST_FILE="${OUT_DIR}/manifest.json"
STATUS_FILE="${OUT_DIR}/status.json"
INSIGHTS_FILE="${OUT_DIR}/insights.json"
DATE="${1:-$(date +"${DATE_FMT}")}"
LOCKFILE="${LOCKFILE:-/var/lock/airship-discover.lock}"
MAX_RUNTIME_SECONDS="${MAX_RUNTIME_SECONDS:-3600}"

mkdir -p "${OUT_DIR}"

log() {
  echo "[$(date -u +"%Y-%m-%dT%H:%M:%SZ")] $*"
}

update_status() {
  local state="$1"
  local message="$2"
  jq -n \
    --arg state "$state" \
    --arg message "$message" \
    --arg date "$DATE" \
    --arg ts "$(date -u +"%Y-%m-%dT%H:%M:%SZ")" \
    '{state: $state, message: $message, date: $date, updated_at: $ts}' > "${STATUS_FILE}"
}

http_probe() {
  # Optional: tiny file-based probes for readiness/liveness
  local probe_type="$1"
  case "$probe_type" in
    ready)
      if [[ -f "${MANIFEST_FILE}" ]]; then
        echo "OK"
        return 0
      else
        echo "NOT_READY"
        return 1
      fi
      ;;
    live)
      echo "OK"
      return 0
      ;;
    status)
      if [[ -f "${STATUS_FILE}" ]]; then
        cat "${STATUS_FILE}"
      else
        echo '{"state":"unknown"}'
      fi
      return 0
      ;;
  esac
}

main() {
  # Handle probes if requested (for cron/monitoring)
  if [[ "${1:-}" == "--probe" ]]; then
    http_probe "${2:-status}"
    return 0
  fi

  # Locking to prevent overlapping cron runs
  exec 200>"${LOCKFILE}"
  if ! flock -n 200; then
    log "Another instance is running. Exiting."
    exit 0
  fi

  # Timeout guard
  if command -v timeout >/dev/null 2>&1; then
    # Re-exec with timeout if not already applied
    if [[ -z "${TIMEOUT_APPLIED:-}" ]]; then
      export TIMEOUT_APPLIED=1
      exec timeout "${MAX_RUNTIME_SECONDS}" "$0" "$@"
    fi
  fi

  log "Starting airship discover for ${DATE}"
  update_status "running" "discover started"

  # 1) Market research (optional)
  if command -v granite-business-research.sh >/dev/null 2>&1; then
    log "Running market research..."
    granite-business-research.sh --date "${DATE}" --out "${OUT_DIR}/research.json" || true
  fi

  # 2) Python orchestrator: RAG + HF manifest
  log "Running discover orchestrator..."
  if ! python3 "${BASE_DIR}/scripts/discover.py" --date "${DATE}" --out-dir "${OUT_DIR}" --hf-repo "${HF_REPO}"; then
    update_status "failed" "discover script failed"
    exit 1
  fi

  # 3) If manifest produced, trigger CDN-only training (async)
  if [[ -f "${MANIFEST_FILE}" ]]; then
    log "Manifest produced. Triggering CDN-only training (async)..."
    nohup bash "${BASE_DIR}/scripts/run-lightning-train.sh" \
      --manifest "${MANIFEST_FILE}" \
      --out-dir "${OUT_DIR}" \
      >> "${OUT_DIR}/lightning-train.log" 2>&1 &
    update_status "training_started" "CDN-only training launched"
  else
    update_status "failed" "manifest not produced"
    exit 1
  fi

  log "airship discover completed (async training running)"
}

main "$@"
```

Make executable:

```bash

```
