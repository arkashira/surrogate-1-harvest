# Costinel / design

## Implementation Plan — `GET /api/v1/cost-anomaly/signal/top-hub`

**Goal**: Ship a read-only endpoint in <2h that surfaces the most-connected knowledge-graph hub for cost-anomaly signals (e.g., top service/region/account driving anomalies) so dashboards and reviewers can triage faster.

**Why this is highest-value**
- Immediate UX win for Costinel’s “Sense + Signal” philosophy (no execution).
- Reuses existing knowledge-graph assets (no new ingestion/training).
- Small surface area → low risk, fast to review/test.

---

`costinel/api/v1/endpoints/cost_anomaly.py`
```python
from datetime import datetime, timezone
from typing import List
from fastapi import APIRouter, HTTPException, Query
from pydantic import BaseModel, Field, conint

from costinel.services.knowledge_rag import KnowledgeRAGService
from costinel.schemas.signal import HubSignal

router = APIRouter()
rag = KnowledgeRAGService()


class TopHubRequest(BaseModel):
    for_date: str = Field(default_factory=lambda: datetime.now(timezone.utc).strftime("%Y-%m-%d"))
    top_n: conint(ge=1, le=10) = 1


class TopHubResponse(BaseModel):
    request: TopHubRequest
    top_hubs: List[HubSignal]
    generated_at: str


@router.get("/cost-anomaly/signal/top-hub", response_model=TopHubResponse)
async def get_top_hub_signal(
    for_date: str = Query(None, regex=r"^\d{4}-\d{2}-\d{2}$"),
    top_n: int = Query(1, ge=1, le=10),
):
    try:
        target_date = for_date or datetime.now(timezone.utc).strftime("%Y-%m-%d")
        hubs = rag.top_hub_signals(for_date=target_date, top_n=top_n)
        return TopHubResponse(
            request=TopHubRequest(for_date=target_date, top_n=top_n),
            top_hubs=hubs,
            generated_at=datetime.now(timezone.utc).isoformat(),
        )
    except ValueError as exc:
        raise HTTPException(status_code=400, detail=str(exc))
    except Exception as exc:
        raise HTTPException(status_code=500, detail=f"Graph unavailable: {exc}")
```

`costinel/services/knowledge_rag.py` (minimal addition)
```python
from typing import List
from datetime import datetime, timedelta
from costinel.schemas.signal import HubSignal


class KnowledgeRAGService:
    def __init__(self, graph_client=None):
        self.graph = graph_client or self._default_graph()

    def _default_graph(self):
        # Replace with your actual graph client (Neo4j / NetworkX / etc.)
        from costinel.graph.client import get_graph_client
        return get_graph_client()

    def top_hub_signals(self, for_date: str, top_n: int = 1) -> List[HubSignal]:
        """
        Query the existing knowledge graph for the most-connected hub(s)
        related to cost-anomaly signals on `for_date`.
        """
        # Example Cypher-like query (adapt to your store):
        # MATCH (h:Hub)-[r:EMITS]->(s:Signal {type:'cost-anomaly'})
        # WHERE date(s.detected_at) = date($for_date)
        # RETURN h.id AS hub_id, h.type AS hub_type, h.label AS label,
        #        count(s) AS signal_count, sum(r.strength) AS score
        # ORDER BY score DESC
        # LIMIT $top_n

        raw = self.graph.query(
            """
            MATCH (h:Hub)-[r:EMITS]->(s:Signal {type: 'cost-anomaly'})
            WHERE date(s.detected_at) = date($for_date)
            RETURN
              h.id AS hub_id,
              h.type AS hub_type,
              h.label AS label,
              count(s) AS signal_count,
              sum(r.strength) AS score,
              h.context AS context
            ORDER BY score DESC
            LIMIT $top_n
            """,
            for_date=for_date,
            top_n=top_n,
        )

        hubs = []
        for row in raw:
            hubs.append(
                HubSignal(
                    hub_id=row["hub_id"],
                    hub_type=row["hub_type"],
                    label=row["label"],
                    score=float(row["score"] or 0.0),
                    signal_count=int(row["signal_count"] or 0),
                    context=row["context"] or {},
                    recommendations=self._recommendations_for(row["hub_id"], row["hub_type"]),
                )
            )
        return hubs

    def _recommendations_for(self, hub_id: str, hub_type: str):
        # Lightweight rule-based recommendations; can be expanded.
        recs = []
        if hub_type == "service":
            recs.append(
                {
                    "id": "rec-ri-coverage",
                    "type": "reserved_instance_coverage",
                    "title": f"Review RI/SP coverage for {hub_id}",
                    "severity": "high",
                    "action": "review_ri_purchase",
                }
            )
        return recs
```
