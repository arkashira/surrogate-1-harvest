#### `scripts/build_manifest.py`
```python
#!/usr/bin/env python3
"""
Generate manifest for one date folder in a HuggingFace dataset repo.
Usage:
  HF_REPO=datasets/your/repo python scripts/build_manifest.py --date 2026-04-29 --out data/manifest-2026-04-29.json
"""
import argparse
import json
import os
import sys
from datetime import datetime

try:
    from huggingface_hub import list_repo_tree
except ImportError:
    print("Install: pip install huggingface_hub")
    sys.exit(1)

def iso_date(s: str) -> str:
    datetime.strptime(s, "%Y-%m-%d")
    return s

def main() -> None:
    parser = argparse.ArgumentParser(description="Build CDN manifest for HF dataset repo folder")
    parser.add_argument("--repo", default=os.getenv("HF_REPO"), help="HF repo (e.g. datasets/your/repo)")
    parser.add_argument("--date", type=iso_date, required=True, help="Date folder (YYYY-MM-DD)")
    parser.add_argument("--out", required=True, help="Output JSON path")
    parser.add_argument("--subpath", default="", help="Optional subpath inside date folder")
    args = parser.parse_args()

    if not args.repo:
        parser.error("--repo or HF_REPO env required")

    base_path = f"{args.date}/{args.subpath}".rstrip("/")
    print(f"Listing repo tree: {args.repo} @ {base_path or '(root)'}")

    # Use recursive=False per folder to avoid pagination explosion; we rely on one date folder.
    items = list_repo_tree(repo_id=args.repo, path=base_path, recursive=False)

    manifest = []
    for item in items:
        # Only include parquet files (or add others if needed)
        if not item.path.endswith(".parquet"):
            continue
        manifest.append({
            "repo": args.repo,
            "path": item.path,
            "sha": getattr(item, "sha", None),
            "size": getattr(item, "size", None),
        })

    os.makedirs(os.path.dirname(os.path.abspath(args.out)), exist_ok=True)
    with open(args.out, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2)

    print(f"Wrote {len(manifest)} entries to {args.out}")

if __name__ == "__main__":
    main()
```

#### `surrogate/data/cdn_dataset.py`
```python
import io
import json
import logging
import random
from typing import Dict, Iterator, List, Optional

import pyarrow.parquet as pq
import requests
from torch.utils.data import IterableDataset

logger = logging.getLogger(__name__)

CDN_ROOT = "https://huggingface.co"

class CdnIterableDataset(IterableDataset):
    """
    CDN-only dataset loader.
    Manifest format: [{"repo": "datasets/...", "path": "...parquet", ...}, ...]
    Each parquet file is expected to contain at least one of:
      - prompt/response
      - instruction/output
      - text (used as prompt)
    """

    def __init__(self, manifest_path: str, columns=("prompt", "response"), shuffle: bool = True, seed: int = 42):
        super().__init__()
        self.manifest_path = manifest_path
        self.columns = columns
        self.shuffle = shuffle
        self.seed = seed
        self._entries: List[Dict] = []

    def _load_manifest(self) -> List[Dict]:
        with open(self.manifest_path, "r", encoding="utf-8") as f:
            data = json.load(f)
        if not isinstance(data, list):
            raise ValueError("Manifest must be a list of file entries")
        return data

    def _fetch_parquet(self, repo: str, path: str, retries: int = 3, backoff: float = 2.0) -> Optional[bytes]:
        url = f"{CDN_ROOT}/{repo}/resolve/main/{path}"
        for attempt in range(1, retries + 1):
            try:
                resp = requests.get(url, timeout=30)
                if resp.status_code == 429:
                    wait = backoff ** attempt
                    logger.warning("CDN 429, waiting %.1fs: %s", wait, url)
                    time.sleep(wait)
                    continue
                resp.raise_for_status()
                return resp.content
            except Exception as exc:
                if attempt == retries:
                    logger.error("Failed to fetch %s: %s", url, exc)
                    return None
                time.sleep(backoff ** attempt)
        return None

    def _project_record(self, rec: Dict) -> Optional[Dict]:
        # Try candidate column pairs
        prompt_col, response_col = self.columns
        if prompt_col in rec and response_col in rec:
            return {"prompt": str(rec[prompt_col]), "response": str(rec[response_col])}
        # fallback: instruction/output
        if "instruction" in rec and "output" in rec:
            return {"prompt": str(rec["instruction"]), "response": str(rec["output"])}
        # fallback: text only
        if "text" in rec:
            return {"prompt": str(rec["text"]), "response": ""}
        return None

    def _iter_entries(self, entries: List[Dict]) -> Iterator[Dict]:
        for entry in entries:
            repo = entry["repo"]
            path = entry["path"]
            blob = self._fetch_parquet(repo, path)
            if blob is None:
                continue
            try:
                table = pq.read_table(io.BytesIO(blob))
                df = table.to_pandas()
                for _, row in df.iterrows():
                    rec = self._project_record(row.to_dict())
                    if rec:
                        yield rec
            except Exception as exc:
                logger.warning("Skipping malformed parquet %s/%s: %s", repo, path, exc)
                continue

    def __iter__(self) -> Iterator[Dict]:
        # Lazy load manifest per worker
        if not self._entries:
            self._entries = self._load_manifest()
        entries = self._entries.copy()
        if self.shuffle:
            # Use worker-aware seed if available
            worker_info = torch.utils.data.get_worker_info()
            seed = self.seed + (worker_info.id if worker_info else 0)
            random.Random(seed).shuffle(entries)
        return self._iter_entries(entries)

import time
import torch
```
