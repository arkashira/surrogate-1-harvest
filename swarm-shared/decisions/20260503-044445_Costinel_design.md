# Costinel / design

**Final Implementation (merged & hardened)**

Add a resilient, CDN-first “Top Hub” signal panel to Costinel’s dashboard that surfaces the most-connected hub (e.g., “MOC”) and related docs with **zero model compute**, **zero API-rate-limit risk**, and **offline-first** UX. Ship in <2 hours.

---

### Architecture (minimal, production-ready)

```
Costinel Dashboard
 └─ TopHubPanel (React)
     ├─ useTopHubSignal() — CDN fetcher + localStorage cache + stale-while-revalidate
     │    ├─ CDN: https://huggingface.co/datasets/axentx/costinel-knowledge/resolve/main/hubs/top.json
     │    └─ fallback → localStorage (last good signal)
     └─ Offline behavior:
          - immediate cache render (if valid)
          - background revalidate when stale (>5 min) or missing
          - 24h CDN TTL; no Authorization header (HF CDN bypass)
```

#### 2) React component

`src/components/TopHubPanel.tsx`

```tsx
import React, { useEffect, useState, useCallback } from 'react';
import { ExternalLink, AlertCircle } from 'lucide-react';

type RelatedDoc = {
  slug: string;
  title: string;
  url: string;
};

type TopHub = {
  hub: string;
  title: string;
  summary: string;
  related_docs: RelatedDoc[];
  updated_at: string;
};

const STORAGE_KEY = 'costinel:top-hub:cached';
const STALE_TTL = 5 * 60 * 1000; // 5m client stale
const CDN_URL = 'https://huggingface.co/datasets/axentx/costinel-knowledge/resolve/main/hubs/top.json';

export function TopHubPanel() {
  const [hub, setHub] = useState<TopHub | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  const loadFromCache = useCallback(() => {
    try {
      const raw = localStorage.getItem(STORAGE_KEY);
      if (!raw) return null;
      const parsed = JSON.parse(raw);
      if (parsed?.hub && parsed?.title) return parsed as TopHub;
      return null;
    } catch {
      return null;
    }
  }, []);

  const saveToCache = useCallback((data: TopHub) => {
    try {
      localStorage.setItem(STORAGE_KEY, JSON.stringify(data));
    } catch {
      // ignore quota errors
    }
  }, []);

  const fetchFromCDN = useCallback(async () => {
    try {
      const res = await fetch(CDN_URL, {
        method: 'GET',
        cache: 'no-store',
      });

      if (!res.ok) throw new Error(`CDN fetch failed: ${res.status}`);
      const data = (await res.json()) as TopHub;
      saveToCache(data);
      setHub(data);
      setError(null);
    } catch (err: any) {
      setError(err?.message || 'Failed to load top hub');
      // keep stale cache visible if available
    } finally {
      setLoading(false);
    }
  }, [saveToCache]);

  useEffect(() => {
    const cached = loadFromCache();
    const isStale = cached
      ? Date.now() - new Date(cached.updated_at).getTime() > STALE_TTL
      : true;

    if (cached) {
      setHub(cached);
      setLoading(false);
    } else {
      setLoading(true);
    }

    if (isStale || !cached) {
      fetchFromCDN();
    }
  }, [fetchFromCDN, loadFromCache]);

  if (loading && !hub) {
    return (
      <div className="p-4 border rounded-lg bg-gray-50 dark:bg-gray-800 border-gray-200 dark:border-gray-700">
        <div className="animate-pulse flex items-center gap-3">
          <div className="h-10 w-10 rounded bg-gray-300 dark:bg-gray-600" />
          <div className="h-4 w-32 rounded bg-gray-300 dark:bg-gray-600" />
        </div>
      </div>
    );
  }

  if (!hub) {
    return (
      <div className="p-4 border rounded-lg bg-red-50 dark:bg-red-900/30 border-red-200 dark:border-red-800">
        <div className="flex items-start gap-2 text-red-800 dark:text-red-200">
          <AlertCircle className="h-5 w-5 mt-0.5 flex-shrink-0" />
          <div>
            <p className="font-medium">Top hub unavailable</p>
            <p className="text-sm">{error || 'No cached data found.'}</p>
            <button
              onClick={() => {
                setLoading(true);
                fetchFromCDN();
              }}
              className="mt-2 text-sm text-red-700 dark:text-red-300 underline hover:text-red-900 dark:hover:text-red-100"
            >
              Retry
            </button>
          </div>
        </div>
      </div>
    );
  }

  return (
    <div className="p-4 border rounded-lg bg-white dark:bg-gray-800 border-gray-200 dark:border-gray-700">
      <div className="flex items-start justify-between gap-3">
        <div className="min-w-0 flex-1">
          <div className="flex items-center gap-2">
            <span className="inline-flex items-center rounded bg-blue-50 dark:bg-blue-900/30 px-2 py-0.5 text-xs font-medium text-blue-700 dark:text-blue-300 border border-blue-100 dark:border-blue-800">
              Top Hub
            </span>
            <span className="font-semibold text-gray-900 dark:text-gray-100 truncate">
              {hub.hub}
            </span>
          </div>
          <p className="mt-1 text-sm text-gray-600 dark:text-gray-400 line-clamp-2">
            {hub.summary}
          </p>
          <p className="mt-2 text-xs text-gray-400 dark:text-gray-500">
            Updated {new Date(hub.updated_at).toLocaleDateString()}
          </p>
        </div>
      </div>

      {hub.related_docs && hub.related_docs.length > 0 && (
        <div className="mt-3 pt-3 border-t border-gray-100 dark:border-gray-700">
          <p className="text-xs font-medium text-gray-500 dark:text-gray-400 uppercase tracking-wide mb-2">
            Related docs
          </p>
          <ul className="space-y-1" role="list">
            {hub.related_docs.map((doc) =>
