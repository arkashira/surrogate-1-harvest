# vanguard / design

## 1. Diagnosis

- No canonical discovery entrypoint exists to surface top-hub insights (e.g., MOC) before planning — violates established `#knowledge-rag #graph #hub` pattern and forces ad-hoc exploration.
- Missing CDN-bypass file-list strategy for HF datasets; any future training ingestion will immediately hit API rate limits instead of using `resolve/main/` CDN paths.
- No reusable Lightning Studio orchestration guardrails (reuse-running + idle-stop handling) to protect quota and avoid duplicate studio spin-ups.
- No explicit pattern for HF commit-cap mitigation (sibling repos) or schema-projection discipline for mixed-schema repos in the design layer.
- No single script/module that composes these patterns into a safe, repeatable “ingest → list → train” workflow for Surrogate-1.

## 2. Proposed change

Add a lightweight design module and CLI entrypoint that:
- Discovers and prints the top-hub (e.g., MOC) via knowledge-rag before planning.
- Generates a CDN-only file-list JSON for a given HF dataset folder (single API call) to be embedded in training scripts.
- Provides a safe Lightning Studio launcher that reuses running studios and handles idle-stop restarts.
- Exposes these as importable functions and a CLI: `python -m vanguard.design`.

Scope: new file `/opt/axentx/vanguard/design.py` + optional `__main__` block. No changes to existing files.

# ---- 1) Top-hub discovery ----
def discover_top_hub(knowledge_rag_script: str = "knowledge-rag") -> Optional[str]:
    """
    Run knowledge-rag to query top hub (e.g., MOC).
    Returns hub name or None.
    """
    script = Path(knowledge_rag_script)
    if not script.exists():
        print(f"[WARN] knowledge-rag script not found at {knowledge_rag_script}", file=sys.stderr)
        return None

    try:
        result = subprocess.run(
            ["bash", str(script), "top-hub"],
            capture_output=True,
            text=True,
            timeout=60,
        )
        if result.returncode == 0 and result.stdout.strip():
            hub = result.stdout.strip().splitlines()[-1]
            print(f"[INFO] Top hub: {hub}")
            return hub
    except Exception as exc:
        print(f"[WARN] Failed to run knowledge-rag: {exc}", file=sys.stderr)
    return None

# ---- 2) HF CDN file-list ----
def list_hf_folder_for_cdn(
    repo: str,
    folder: str,
    token: Optional[str] = None,
) -> List[str]:
    """
    Single API call to list files in one folder (non-recursive).
    Returns list of paths suitable for CDN fetch:
      https://huggingface.co/datasets/{repo}/resolve/main/{path}
    """
    if requests is None:
        raise RuntimeError("requests is required for HF operations")

    api_url = f"https://huggingface.co/api/datasets/{repo}/tree"
    params = {"path": folder, "recursive": "false"}
    headers = {"Authorization": f"Bearer {token}"} if token else {}

    resp = requests.get(api_url, params=params, headers=headers, timeout=30)
    if resp.status_code == 429:
        wait = 360
        print(f"[WARN] HF API 429 rate-limited. Wait {wait}s before retry.", file=sys.stderr)
        raise RuntimeError("HF API rate-limited")
    resp.raise_for_status()
    items: List[Dict[str, Any]] = resp.json()

    # Keep only files (exclude subfolders)
    paths = [item["path"] for item in items if item.get("type") == "file"]
    print(f"[INFO] Found {len(paths)} files in datasets/{repo}/{folder}")
    return paths

def save_file_list(repo: str, folder: str, out_path: Path, token: Optional[str] = None) -> Path:
    """
    Generate and save file-list JSON for embedding in training scripts.
    """
    paths = list_hf_folder_for_cdn(repo, folder, token=token)
    payload = {
        "repo": repo,
        "folder": folder,
        "files": paths,
        "cdn_prefix": f"https://huggingface.co/datasets/{repo}/resolve/main",
    }
    out_path.parent.mkdir(parents=True, exist_ok=True)
    out_path.write_text(json.dumps(payload, indent=2))
    print(f"[INFO] File-list saved to {out_path}")
    return out_path

# ---- 3) Lightning Studio guardrails ----
def find_running_studio(name: str, client: Optional[LightningClient] = None) -> Optional[Dict[str, Any]]:
    """
    Reuse running studio to save quota (pattern: #lightning-ai #quota).
    """
    if LightningClient is None or client is None:
        return None
    try:
        studios = client.studios()
        for s in studios:
            if isinstance(s, dict):
                if s.get("name") == name and s.get("status") == "Running":
                    return s
            else:
                # fallback attribute access
                if getattr(s, "name", None) == name and getattr(s, "status", None) == "Running":
                    return {"name": s.name, "status": s.status, "id": getattr(s, "id", None)}
    except Exception as exc:
        print(f"[WARN] Failed to list studios: {exc}", file=sys.stderr)
    return None

def ensure_studio_running(
    name: str,
    machine: str = "L40S",
    target: Optional[Work] = None,
    client: Optional[LightningClient] = None,
) -> Dict[str, Any]:
    """
    Return studio info; if stopped, restart it.
    Pattern: #lightning-ai #idle-timeout
    """
    studio = find_running_studio(name, client=client)
    if studio:
        print(f"[INFO] Reusing running studio: {name}")
        return studio

    if target is None:
        raise ValueError("target Work required to start studio")

    print(f"[INFO] Starting studio {name} on {machine}")
    # Simplified launcher pattern; adapt to your Lightning SDK usage.
    target.run(machine=machine)
    return {"name": name, "status": "Starting", "machine": machine}
