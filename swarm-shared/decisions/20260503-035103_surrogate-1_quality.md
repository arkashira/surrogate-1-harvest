- Accepts `SHARD_ID`, `SHARD_TOTAL=16` (matrix) and optional `DATE_FOLDER` (defaults to today `YYYY-MM-DD`).

- Projects each file to `{prompt, response}` at parse time (avoids `load_dataset(streaming=True)` on mixed schemas).
- Deduplicates via central `lib/dedup.py` md5 store.
- Writes output to `batches/public-merged/{date}/shard{N}-{HHMMSS}.jsonl` with deterministic naming to prevent collisions.
- Reuses existing GitHub Actions matrix setup; only the worker script and optional manifest generation change.

#!/usr/bin/env python3
"""
CDN-bypass ingestion worker for surrogate-1 public dataset.
Usage (GH Actions matrix):
  SHARD_ID=3 SHARD_TOTAL=16 python bin/dataset-enrich.py [DATE_FOLDER]
"""
import os
import sys
import json
import hashlib
import datetime
import pathlib
import time
from typing import List, Dict, Any, Optional

import requests
import pyarrow.parquet as pq
import pyarrow as pa
from huggingface_hub import hf_hub_download, list_repo_tree

# Local dedup module (shared with HF Space)
sys.path.insert(0, str(pathlib.Path(__file__).parent.parent))
from lib.dedup import is_duplicate, mark_seen  # noqa: E402

HF_REPO = "datasets/axentx/surrogate-1-training-pairs"
BASE_CDN = f"https://huggingface.co/{HF_REPO}/resolve/main"

SHARD_ID = int(os.getenv("SHARD_ID", "0"))
SHARD_TOTAL = int(os.getenv("SHARD_TOTAL", "16"))
DATE_FOLDER = os.getenv("DATE_FOLDER") or datetime.date.today().isoformat()

OUT_DIR = pathlib.Path("batches/public-merged") / DATE_FOLDER
OUT_DIR.mkdir(parents=True, exist_ok=True)
TIMESTAMP = datetime.datetime.utcnow().strftime("%H%M%S")
OUT_FILE = OUT_DIR / f"shard{SHARD_ID}-{TIMESTAMP}.jsonl"

# Optional manifest; if present, use it to avoid any API calls during worker run.
MANIFEST_PATH = os.getenv("MANIFEST_PATH") or f"manifest-{DATE_FOLDER}.json"

def load_manifest() -> List[str]:
    """Load pre-listed file paths for DATE_FOLDER from JSON manifest."""
    if os.path.exists(MANIFEST_PATH):
        with open(MANIFEST_PATH) as f:
            data = json.load(f)
        # manifest format: list of relative paths under DATE_FOLDER
        return [p for p in data if isinstance(p, str)]
    # Fallback: minimal safe list via hf_hub_download tree (single folder, non-recursive)
    # This still uses HF API but only one call per worker if manifest missing.
    tree = list_repo_tree(
        repo_id=HF_REPO,
        path=DATE_FOLDER,
        repo_type="dataset",
        recursive=False,
    )
    files = []
    for entry in tree:
        if entry.type == "file" and entry.path.endswith((".parquet", ".jsonl")):
            files.append(entry.path)
    return files

def assign_shard(paths: List[str]) -> List[str]:
    """Deterministic shard assignment by hash(path)."""
    assigned = []
    for p in paths:
        h = int(hashlib.md5(p.encode()).hexdigest(), 16)
        if h % SHARD_TOTAL == SHARD_ID:
            assigned.append(p)
    return assigned

def cdn_download(path: str) -> bytes:
    """Download via CDN (no auth) with retry on 429/5xx."""
    url = f"{BASE_CDN}/{path}"
    for attempt in range(5):
        try:
            resp = requests.get(url, timeout=30)
            if resp.status_code == 429:
                wait = 360
                print(f"CDN 429, waiting {wait}s", file=sys.stderr)
                time.sleep(wait)
                continue
            resp.raise_for_status()
            return resp.content
        except Exception as exc:
            if attempt == 4:
                raise
            print(f"Download retry {attempt+1} for {path}: {exc}", file=sys.stderr)
            time.sleep(5 * (attempt + 1))
    raise RuntimeError("Unreachable")

def extract_pairs(content: bytes, path: str) -> List[Dict[str, str]]:
    """Project file to {prompt, response} pairs; schema-agnostic."""
    pairs = []
    try:
        if path.endswith(".parquet"):
            table = pq.read_table(pa.BufferReader(content))
            cols = table.column_names
            # Try common field names; keep only prompt/response projection.
            prompt_col = next((c for c in cols if "prompt" in c.lower()), None)
            response_col = next((c for c in cols if "response" in c.lower()), None)
            if prompt_col and response_col:
                df = table.select([prompt_col, response_col]).to_pydict()
                for p, r in zip(df[prompt_col], df[response_col]):
                    if isinstance(p, str) and isinstance(r, str) and p.strip() and r.strip():
                        pairs.append({"prompt": p.strip(), "response": r.strip()})
            else:
                # Fallback: attempt to read as dict-like rows and guess fields.
                for batch in table.to_batches():
                    for i in range(batch.num_rows):
                        row = {n: batch.column(n)[i].as_py() for n in batch.schema.names}
                        # Heuristic: pick two longest string fields
                        strs = {k: v for k, v in row.items() if isinstance(v, str) and v.strip()}
                        if len(strs) >= 2:
                            keys = sorted(strs, key=lambda k: len(strs[k]), reverse=True)
                            pairs.append({"prompt": strs[keys[0]].strip(), "response": strs[keys[1]].strip()})
        elif path.endswith(".jsonl"):
            for line in content.decode("utf-8", errors="replace").splitlines():
                line = line.strip()
                if not line:
                    continue
                try:
                    row = json.loads(line)
                except Exception:
                    continue
                prompt = row.get("prompt") or row.get("input") or row.get("question")
                response = row.get("response") or row.get("output") or row.get("answer")
                if isinstance(prompt, str) and isinstance(response, str) and prompt.strip() and response.strip():
                    pairs.append({"prompt": prompt.strip(), "response": response.strip()})
    except Exception as exc:
        print(f"Schema projection failed for {path}: {exc}", file=sys.stderr)
    return pairs

def main() -> None:
    print(f"Shard {SHARD_ID}/{SHARD_TOTAL} | date={DATE_FOLDER}", file=sys.stderr)
    paths = load_manifest()
    if not paths:
        print("No files found for manifest/date; exiting.", file=sys.stderr)
        return
    assigned = assign_shard(paths)
    print(f"Assigned {len(assigned)} files out of {len(paths)}", file=sys.stderr)
