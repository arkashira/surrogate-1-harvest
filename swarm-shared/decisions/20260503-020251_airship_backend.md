### 1. CDN-First Data Ingestion (Eliminate HF API 429s)
**Action:** Decouple training data loading from Hugging Face Hub API limits by pre-generating a deterministic file manifest and fetching raw bytes via public CDN.

**Implementation:**
*   **Step 1 (Orchestrator/Mac):** Generate the manifest *once* per training cycle (after rate-limit reset).
*   **Step 2 (Studio):** Training script reads the manifest and downloads `.parquet` files directly via `requests` (no `huggingface_hub` client overhead).

**Code: `tools/generate_file_list.py`**
```python
#!/usr/bin/env python3
"""
Run this ONCE before training (e.g., on your local machine or CI)
to avoid HF API rate limits during dataset loading.
"""
import json
import os
from huggingface_hub import HfApi

HF_REPO = os.getenv("HF_DATASET_REPO", "myorg/surrogate-dataset")
DATE_FOLDER = os.getenv("DATE_FOLDER", "2026-05-03")
OUT_PATH = os.getenv("OUT_PATH", "file_list.json")

api = HfApi()
# Critical: recursive=False prevents pagination explosion and keeps call <2h budget
tree = api.list_repo_tree(
    repo_id=HF_REPO,
    path=DATE_FOLDER,
    recursive=False,
    repo_type="dataset",
)

files = sorted([f.rfilename for f in tree if f.rfilename.endswith(".parquet")])

with open(OUT_PATH, "w") as f:
    json.dump({"repo": HF_REPO, "date": DATE_FOLDER, "files": files}, f, indent=2)

print(f"✅ Manifest ready: {len(files)} files -> {OUT_PATH}")
```

**Code: `surrogate/data/cdn_dataset.py`**
```python
import os
import json
import pyarrow.parquet as pq
import requests
import torch
from torch.utils.data import Dataset, DataLoader
from tqdm import tqdm

class CDNParquetDataset(Dataset):
    def __init__(self, manifest_path="file_list.json", cache_dir="/tmp/hf_cdn_cache"):
        os.makedirs(cache_dir, exist_ok=True)
        self.cache_dir = cache_dir
        self.entries = []

        with open(manifest_path) as f:
            cfg = json.load(f)

        for fname in tqdm(cfg["files"], desc="Loading CDN shards"):
            url = f"https://huggingface.co/datasets/{cfg['repo']}/resolve/main/{cfg['date']}/{fname}"
            local_path = os.path.join(cache_dir, fname.replace("/", "_"))
            
            if not os.path.exists(local_path):
                r = requests.get(url, timeout=60)
                r.raise_for_status()
                with open(local_path, "wb") as f_out:
                    f_out.write(r.content)
            
            # Project schema on-load: Enforce {prompt, response}
            tbl = pq.read_table(local_path, columns=["prompt", "response"])
            for i in range(tbl.num_rows):
                self.entries.append({
                    "prompt": tbl["prompt"][i].as_py(),
                    "response": tbl["response"][i].as_py()
                })

    def __len__(self):
        return len(self.entries)

    def __getitem__(self, idx):
        return self.entries[idx]

# loader = DataLoader(dataset, batch_size=8, shuffle=True, num_workers=4)
```

### 2. HF Sibling Sharding (Eliminate 128 Commits/HR Cap)
**Action:** Distribute write operations across multiple deterministic sibling repos to multiply the effective commit rate limit (128/hr * N siblings).

**Resolution:** Both candidates proposed identical logic. We adopt the `hash(slug) % N` strategy.

**Code: `tools/hf_shard.py`**
```python
import hashlib
import os
from huggingface_hub import HfApi, upload_file

api = HfApi()
BASE_REPO = os.getenv("HF_BASE_REPO", "myorg/surrogate-dataset")
SIBLINGS = [f"{BASE_REPO}-shard{i}" for i in range(5)]  # 640 commits/hr aggregate

def pick_repo(slug: str) -> str:
    """Deterministic routing to avoid rate limits."""
    h = hashlib.sha256(slug.encode()).hexdigest()
    idx = int(h, 16) % len(SIBLINGS)
    return SIBLINGS[idx]

# Example Usage:

# upload_file(..., repo_id=target_repo, ...)
```

### 3. Lightning Studio Guard (Eliminate Quota Waste)
**Action:** Prevent redundant Studio startups (which consume free-tier quota) by reusing running instances and gracefully restarting idle-stopped ones.

**Resolution:** Candidate 1 used raw `lightning` SDK iteration; Candidate 2 abstracted it into a function. We merge them into a robust utility function.

**Code: `surrogate/lightning/studio_guard.py`**
```python
#!/usr/bin/env python3
import os
import sys
from lightning import Studio, Machine, Teamspace

TEAMSPACE = os.getenv("LIGHTNING_TEAMSPACE", "my-team")
STUDIO_NAME = os.getenv("STUDIO_NAME", "surrogate-train-l40s")
# Use L40S for free tier (lightning-public-prod) or H200 for paid (lightning-lambda-prod)
MACHINE = Machine.L40S  

def get_or_create_studio(name: str, team: str) -> Studio:
    ts = Teamspace(team)
    studio = None

    # 1. Reuse if already running
    for s in ts.studios:
        if s.name == name:
            studio = s
            break

    # 2. Create if missing
    if studio is None:
        print(f"🆕 Creating Studio: {name}")
        studio = Studio.create(
            name=name,
            teamspace=team,
            machine=MACHINE,
            create_ok=True,
        )
    elif studio.status != "running":
        # 3. Guard against idle-stop (common in free tier)
        print(f"⚠️ Studio '{name}' is stopped. Restarting...")
        studio.start(machine=MACHINE)
    
    return studio

if __name__ == "__main__":
    studio = get_or_create_studio(STUDIO_NAME, TEAMSPACE)
    print(f"🚀 Using Studio: {studio.name} ({studio.status})")
    
    # Execute the training command inside the studio
    # This replaces direct `studio.run()` calls scattered in code
    cmd = " ".join(sys.argv[1:]) if len(sys.argv) > 1 else "python surrogate/train.py"
    studio.run(cmd, cwd="/workspace")
```

### 4. Schema Projection (Ingestion Standardization)
**Action:** Ensure only `{prompt, response}` pairs are stored in the final training corpus, moving metadata to filenames to simplify downstream tokenization.

**Code: `surrogate/ingest/project_schema.py`**
```python
#!/usr/bin/env python3
"""
Project heterogeneous raw files to strict {prompt, response} Parquet.
Attribution is moved to the filename (batches/mirror-merged/{date}/{slug}.parquet).
"""
import pyarrow as pa
import pyarrow.parquet as pq
from pathlib import Path

def write_mirror(date: str, slug: str, pairs: list[dict]):
    """
    pairs: List of dicts [{"prompt": "...", "response": "..."}, ...]
    """
    out_dir = Path(f"batches/mirror-merged/{date}")
    out_dir.mkdir(parents=True, exist_ok=True)
    
    safe_slug = slug.replace("/", "_")[:120]  # Sanitize for FS
    fname = out_dir / f"{safe_slug}.parquet"
    
    tbl = pa.Table.from_pylist(pairs)
    pq.write_table(tbl, fname)
    return fname
```
