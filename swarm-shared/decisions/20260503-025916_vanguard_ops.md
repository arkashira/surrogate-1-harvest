# vanguard / ops

## Final Synthesized Answer

### 1. Diagnosis (Consensus)
- **No deterministic asset manifest** at build time forces runtime HF API calls → invites 429s and breaks CDN-first strategy.
- **Missing mount-point / entrypoint boundary** between orchestration host (Mac) and compute target (Lightning Studio) → unclear contract for dataset/file handoff and risk of accidental local training.
- **No pre-listed file-path cache** for date-partitioned datasets → each training run risks HF API pagination/rate limits instead of pure CDN fetches.
- **No studio reuse guard** → wastes Lightning quota by recreating running studios.
- **No idle-stop resilience** → Lightning idle timeout kills training and isn’t auto-recovered.

### 2. Proposed Change (Unified)
Create `/opt/axentx/vanguard/ops/` with two small, high-leverage files (new files only; no changes to existing app code):

- `ops/build_manifest.py` — run on orchestration host (Mac). For a given `datasets/{repo}/{date}/`, calls HF API **once** (non-recursive `list_repo_tree`), emits `static/manifest-{repo_slug}-{date}.json` containing `{path, cdn_url, sha256, size}`. Uses `HfApi.list_repo_tree` (efficient, non-recursive by default; set `recursive=True` only if nested files are required) and avoids downloading dataset content.
- `ops/lightning_launcher.py` — imported by training notebooks/scripts. Provides `ensure_studio(name, machine)` that reuses running studios, and `run_with_retry(target, ...)` that checks studio status before each run and restarts on idle-stop.

```python
#!/usr/bin/env python3
"""
Build a deterministic CDN-first manifest for a date-partitioned dataset.
Run on orchestration host (Mac). Emits static/manifest-{repo_slug}-{date}.json
"""
import json, hashlib, os, sys
from pathlib import Path
from datetime import datetime, timezone

try:
    from huggingface_hub import HfApi
except ImportError:
    print("Install: pip install huggingface_hub")
    sys.exit(1)

REPO = os.getenv("HF_DATASET_REPO", "datasets/mirror-merged")
DATE_FOLDER = os.getenv("HF_DATE_FOLDER", datetime.now(timezone.utc).strftime("%Y-%m-%d"))
OUT_DIR = Path(os.getenv("OUT_DIR", "/opt/axentx/vanguard/static"))
OUT_DIR.mkdir(parents=True, exist_ok=True)

# Slugify repo for filename: datasets/mirror-merged -> datasets-mirror-merged
repo_slug = REPO.replace("/", "-")
out_path = OUT_DIR / f"manifest-{repo_slug}-{DATE_FOLDER}.json"

api = HfApi()

def build_manifest():
    # Single API call: list one folder (non-recursive by default)
    # Set recursive=True only if nested files are required
    items = api.list_repo_tree(REPO, path=DATE_FOLDER, repo_type="dataset", recursive=False)
    files = [it.rfilename for it in items if it.rfilename.endswith(".parquet")]

    manifest = {
        "repo": REPO,
        "date": DATE_FOLDER,
        "generated_utc": datetime.now(timezone.utc).isoformat(),
        "files": []
    }

    for f in sorted(files):
        # CDN URL (resolve via HuggingFace CDN)
        cdn_url = f"https://huggingface.co/datasets/{REPO}/resolve/main/{f}"
        # Note: sha256 requires local read; if not feasible at build time,
        # defer verification to compute target using HF-provided ETag or size.
        # For build-time integrity, download once and hash:
        try:
            local_path = api.hf_hub_download(repo_id=REPO, filename=f, repo_type="dataset")
            sha256 = hashlib.sha256(open(local_path, "rb").read()).hexdigest()
            size = os.path.getsize(local_path)
        except Exception:
            # Fallback: skip hash/size if download fails; log warning
            sha256 = None
            size = None
            print(f"Warning: could not hash/size {f}; skipping integrity for this file")

        manifest["files"].append({
            "path": f,
            "sha256": sha256,
            "cdn_url": cdn_url,
            "size": size
        })

    OUT_DIR.mkdir(parents=True, exist_ok=True)
    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"Manifest written to {out_path} ({len(files)} files)")

if __name__ == "__main__":
    build_manifest()
```

`/opt/axentx/vanguard/ops/lightning_launcher.py`
```python
#!/usr/bin/env python3
"""
Lightning Studio launcher with reuse + manifest-aware run and idle-stop resilience.
"""
import os, json, time
from pathlib import Path
from lightning_sdk import Studio, Machine, Teamspace

MANIFEST_PATH = Path(os.getenv("MANIFEST_PATH", "manifest.json"))
STUDIO_NAME = os.getenv("STUDIO_NAME", "vanguard-train")
SCRIPT = os.getenv("TRAIN_SCRIPT", "train.py")

def ensure_studio(name, machine=Machine.L40S):
    team = Teamspace()
    running = None
    for s in team.studios:
        if s.name == name and s.status == "Running":
            running = s
            break

    if running:
        print(f"Reusing running studio: {running.name}")
        return running

    print(f"Creating studio: {name}")
    return Studio.create(name=name, machine=machine, cloud="lightning-public-prod")

def run_with_retry(studio, command, env=None, max_retries=3, retry_delay=30):
    for attempt in range(1, max_retries + 1):
        studio.refresh()
        if studio.status != "Running":
            print(f"Studio not running (status={studio.status}); restarting...")
            studio.start(machine=Machine.L40S)
            # Wait for running
            for _ in range(10):
                studio.refresh()
                if studio.status == "Running":
                    break
                time.sleep(6)

        env = env or {}
        run = studio.run(command=command, env=env, sync=False)
        print(f"Started run {run.id} in studio {studio.name} (attempt {attempt}/{max_retries})")

        # Basic run monitoring: wait a short time and check if run is alive
        time.sleep(10)
        run.refresh()
        if run.status in ("running", "queued"):
            return run

        print(f"Run failed or stopped (status={run.status}); retrying in {retry_delay}s...")
        time.sleep(retry_delay)

    raise RuntimeError(f"Run failed after {max_retries} attempts")

def launch():
    if not MANIFEST_PATH.exists():
        raise FileNotFoundError(f"Manifest missing: {MANIFEST_PATH}")

    studio = ensure_studio(STUDIO_NAME)

    env = {
        "MANIFEST_PATH": str(MANIFEST_PATH),
        "HF_DATASET_REPO": os.getenv("HF_DATASET_REPO", "datasets/mirror-merged"),
    }

    return run_with_retry(studio, command=f"python {SCRIPT}", env=env)

if __name__ == "__main__":
    launch()
```
