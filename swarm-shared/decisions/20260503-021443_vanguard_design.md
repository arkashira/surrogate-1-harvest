# vanguard / design

## 1. Diagnosis

- No persisted `(repo, dateFolder) → file-list` manifest exists; every training run triggers authenticated `list_repo_tree` calls, burning HF API quota and risking 429s.
- Training uses `load_dataset(streaming=True)` or HF API reads on heterogeneous repos, causing `pyarrow.CastError` schema mismatches.
- Lightning training scripts do not reuse running Studios, wasting ~80hr/mo quota on recreation and risking idle-stop training death.
- Data ingestion writes mixed-schema files into `enriched/` with extra metadata columns (`source`, `ts`) instead of projecting to `{prompt, response}` only.
- No CDN-only data path; authenticated API calls continue during training data loading instead of using `https://huggingface.co/datasets/{repo}/resolve/main/...` bypass.

## 2. Proposed change

Create a lightweight manifest generator + CDN-only training stub under `/opt/axentx/vanguard/training/`:

- `training/manifest.py` — one-time Mac-side script: `list_repo_tree` per date folder → save `manifest-{repo}-{date}.json`
- `training/train.py` — Lightning Studio entrypoint: reads manifest, downloads via CDN URLs only, projects to `{prompt, response}`, yields HF Dataset without API/auth
- `training/reuse_studio.py` — helper: list running studios and attach/reuse instead of create

Scope: add these three files; modify existing launcher (if any) to use manifest path and Studio reuse.

### training/manifest.py
```python
#!/usr/bin/env python3
"""
Generate manifest for (repo, date_folder) -> list of file paths.
Run from Mac (or any env with HF token) once per folder.
Avoids recursive listing on big repos; one call per date folder.
"""
import json
import os
import sys
from datetime import datetime

from huggingface_hub import HfApi

API = HfApi()

def build_manifest(repo_id: str, date_folder: str, out_dir: str = "manifests"):
    """
    repo_id: e.g. 'datasets/myorg/surrogate-1'
    date_folder: e.g. '2026-04-29'
    """
    folder_path = f"{date_folder}"
    print(f"Listing {repo_id}/{folder_path} (non-recursive)...")
    items = API.list_repo_tree(repo_id=repo_id, path=folder_path, recursive=False)

    files = [
        item.rfilename
        for item in items
        if not item.rfilename.endswith("/")
    ]

    manifest = {
        "repo_id": repo_id,
        "date_folder": date_folder,
        "generated_at": datetime.utcnow().isoformat() + "Z",
        "files": sorted(files),
    }

    os.makedirs(out_dir, exist_ok=True)
    safe_repo = repo_id.replace("/", "_")
    out_path = os.path.join(out_dir, f"manifest-{safe_repo}-{date_folder}.json")
    with open(out_path, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Saved {len(files)} files -> {out_path}")
    return out_path

if __name__ == "__main__":
    if len(sys.argv) != 3:
        print("Usage: manifest.py <repo_id> <date_folder>")
        sys.exit(1)
    build_manifest(sys.argv[1], sys.argv[2])
```

### training/train.py
```python
#!/usr/bin/env python3
"""
Lightning Studio entrypoint.
- Reads manifest JSON
- Downloads via CDN URLs (no auth/API calls)
- Projects to {prompt, response}
- Returns HF Dataset for training
"""
import json
import os
import io
import pyarrow as pa
import pyarrow.parquet as pq
import requests
from datasets import Dataset, DatasetDict
from typing import Dict, Any

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo_id}/resolve/main/{file_path}"

def cdn_fetch(repo_id: str, file_path: str) -> bytes:
    url = CDN_TEMPLATE.format(repo_id=repo_id, file_path=file_path)
    resp = requests.get(url, timeout=60)
    resp.raise_for_status()
    return resp.content

def project_to_prompt_response(batch_bytes: bytes) -> Dict[str, Any]:
    """
    Project parquet bytes to {prompt, response}.
    Adjust projection logic per your schema.
    """
    table = pq.read_table(pa.BufferReader(batch_bytes))
    # Keep only prompt/response; drop others to avoid schema drift
    cols = set(table.column_names)
    keep = []
    if "prompt" in cols:
        keep.append("prompt")
    if "response" in cols:
        keep.append("response")
    if not keep:
        # fallback: first two string/text columns
        candidates = [c for c in table.column_names if table.schema.field(c).type in (pa.string(), pa.large_string(), pa.utf8())]
        keep = candidates[:2]
    projected = table.select(keep)
    # Rename to canonical names if needed
    names = projected.column_names
    if len(names) == 2:
        projected = projected.rename_columns(["prompt", "response"])
    return projected.to_pydict()

def load_from_manifest(manifest_path: str) -> Dataset:
    with open(manifest_path) as f:
        manifest = json.load(f)

    repo_id = manifest["repo_id"]
    records = []
    for file_path in manifest["files"]:
        try:
            blob = cdn_fetch(repo_id, file_path)
            projected = project_to_prompt_response(blob)
            for i in range(len(projected.get("prompt", []))):
                records.append({
                    "prompt": projected.get("prompt", [""])[i],
                    "response": projected.get("response", [""])[i],
                })
        except Exception as exc:
            print(f"Skipping {file_path}: {exc}")
            continue

    return Dataset.from_list(records)

if __name__ == "__main__":
    # Example local test
    import sys
    if len(sys.argv) != 2:
        print("Usage: train.py <manifest.json>")
        sys.exit(1)
    ds = load_from_manifest(sys.argv[1])
    print(ds)
    print(f"Loaded {len(ds)} examples")
```

### training/reuse_studio.py
```python
#!/usr/bin/env python3
"""
Reuse a running Lightning Studio if available.
"""
from lightning import Studio, Machine, Teamspace

def get_or_create_studio(name: str, machine: Machine = Machine.L40S):
    studios = Teamspace.studios()
    for s in studios:
        if s.name == name and s.status == "Running":
            print(f"Reusing running studio: {s.name}")
            return s
    print(f"Creating studio: {name} on {machine}")
    return Studio(name=name, machine=machine, create_ok=True)

def ensure_running(studio: Studio, machine: Machine = Machine.L40S) -> Studio:
    if studio.status != "Running":
        print(f"Studio {studio.name} not running (status={studio.status}). Restarting...")
        studio.start(machine=machine)
    return studio
```

studio = get_or_create_studio('vanguard-train', Machine.L40S)
studio = ensure_running(studio, Machine.L40S)
studio.run(
    work='training/',
    entrypoint='python train.py manifests/manifest-datasets_myorg_surrogate-1-2026-04-29.json',
    requirements=['pyarrow','datasets','requests'],
)
"
