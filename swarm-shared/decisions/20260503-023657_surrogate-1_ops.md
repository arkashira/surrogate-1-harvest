def load_manifest() -> list[str]:
    if Path(MANIFEST_PATH).exists():
        print(f"Using cached manifest: {MANIFEST_PATH}")
        return json.loads(Path(MANIFEST_PATH).read_text())

    # Fallback: single API call to list date folder (non-recursive)
    print(f"Listing {REPO_ID} tree for folder: {DATE}/")
    tree = api.list_repo_tree(repo_id=REPO_ID, path=DATE, recursive=False)
    files = [
        f"{DATE}/{entry.path}"
        for entry in tree
        if entry.type == "file" and entry.path.lower().endswith((".jsonl", ".json", ".parquet"))
    ]
    Path(MANIFEST_PATH).write_text(json.dumps(files, indent=2))
    return files

def extract_pair(obj) -> dict | None:
    # Try common surrogate-1 keys
    prompt = obj.get("prompt") or obj.get("instruction") or obj.get("input")
    response = obj.get("response") or obj.get("output") or obj.get("completion")
    if prompt is None or response is None:
        return None
    return {"prompt": str(prompt).strip(), "response": str(response).strip()}

# ---- CDN download (no auth header) ----
CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def stream_cdn_lines(path: str):
    url = CDN_TEMPLATE.format(repo=REPO_ID, path=path)
    with requests.get(url, timeout=30, stream=True) as r:
        r.raise_for_status()
        # iterate lines without loading whole file
        for chunk in r.iter_lines(decode_unicode=True):
            if chunk:
                yield chunk

def process_jsonl(path: str):
    for line in stream_cdn_lines(path):
        line = line.strip()
        if not line:
            continue
        try:
            obj = json.loads(line)
        except json.JSONDecodeError:
            continue
        pair = extract_pair(obj)
        if pair:
            yield pair
