---

### 1) Core Objective (non-negotiable)
Expose **one read-only, side-effect-free** endpoint:  
`GET /api/v1/cost-anomaly/signal/top-hub`  

- **Purpose**: surface the most-connected hub (e.g., `"MOC"`) and its top contextual signals for cost-anomaly triage.  
- **Philosophy**: Sense + Signal — ไม่ Execute. No writes, no mutations, no training-pipeline coupling.  
- **SLA target**: ≤100 ms p95 in production (cached path); graceful fallback if graph unavailable.

---

### 2) Design Decisions (resolved)

| Decision | Resolution |
|----------|------------|
| **Data source priority** | 1) Precomputed `top_hub.json` from knowledge-rag pipeline (strongest signal). 2) Lightweight in-memory graph snapshot. 3) Cached fallback. Never recurse deeply or run heavy scans on request. |
| **Caching** | 60 s in-memory TTL (coarse timestamp bucket). Refresh only on TTL expiry. |
| **Schema stability** | Minimal, version-tolerant response. Include `generated_at`, `ttl_seconds`, `source` for cache/debuggability. |
| **Observability** | Emit timing metric (`costinel.top_hub.latency_ms`) and log `top_hub.slug`, `degree`, `source`. |
| **Rate limiting** | 30/min per client (protect scan path). |
| **Permissions** | Read-only scopes only. No mutation scopes required. |

#### `api/cost_anomaly/signal.py`

```python
"""
Top-hub signal endpoint (read-only).
Sense + Signal — ไม่ Execute
"""
from datetime import datetime, timezone
from functools import lru_cache
from typing import Dict, List, Optional
import json
import logging
import os

from fastapi import APIRouter, HTTPException
from pydantic import BaseModel

router = APIRouter()
logger = logging.getLogger("costinel.top_hub")

# ---- Types ----
class HubNode(BaseModel):
    slug: str
    label: str
    degree: int
    description: Optional[str] = None
    related_docs: List[str]

class TopHubSignalResponse(BaseModel):
    top_hub: HubNode
    generated_at: str  # ISO UTC
    ttl_seconds: int
    source: str  # "knowledge_graph" | "cached_fallback"

# ---- Fallback ----
_FALLBACK_TOP_HUB = HubNode(
    slug="MOC",
    label="MOC",
    degree=-1,
    description="Manual override / cached fallback — review knowledge graph",
    related_docs=["cost-anomaly-playbook.md", "moc-operational-rules.md"],
)

# ---- Core logic ----
def _load_precomputed_top_hub(cache_path: str) -> Optional[HubNode]:
    """Load precomputed top_hub.json from knowledge-rag pipeline."""
    try:
        if os.path.exists(cache_path):
            with open(cache_path) as f:
                data = json.load(f)
            return HubNode(**data)
    except Exception as exc:
        logger.warning("Precomputed load failed: %s", exc, exc_info=True)
    return None

def _lightweight_graph_fallback(docs_root: str) -> Optional[HubNode]:
    """Lightweight directory scan for most-linked markdown files (naive)."""
    try:
        if not os.path.isdir(docs_root):
            return None

        link_counts: Dict[str, int] = {}
        for fn in os.listdir(docs_root):
            if not fn.endswith(".md"):
                continue
            path = os.path.join(docs_root, fn)
            with open(path, errors="ignore") as f:
                content = f.read()
            links = [p for p in content.split() if p.startswith("[[") and p.endswith("]]")]
            link_counts[fn] = len(links)

        if not link_counts:
            return None

        top_slug = max(link_counts, key=link_counts.get)
        return HubNode(
            slug=top_slug.replace(".md", ""),
            label=top_slug.replace(".md", ""),
            degree=link_counts[top_slug],
            description="Inferred from local docs link density",
            related_docs=[top_slug],
        )
    except Exception as exc:
        logger.warning("Lightweight scan failed: %s", exc, exc_info=True)
        return None

@lru_cache(maxsize=1)
def _cached_top_hub(ttl_token: int) -> HubNode:
    """
    In-memory cache keyed by rounded timestamp (coarse TTL).
    Priority: precomputed > lightweight scan > fallback.
    """
    cache_path = os.getenv("COSTINEL_TOP_HUB_CACHE", "/opt/axentx/Costinel/data/top_hub.json")
    docs_root = "/opt/axentx/Costinel/docs"

    node = _load_precomputed_top_hub(cache_path)
    if node is None:
        node = _lightweight_graph_fallback(docs_root)
    if node is None:
        node = _FALLBACK_TOP_HUB
    return node

@router.get(
    "/cost-anomaly/signal/top-hub",
    response_model=TopHubSignalResponse,
    tags=["cost-anomaly", "signal"],
)
def get_top_hub_signal() -> TopHubSignalResponse:
    """
    Return the most-connected hub and related context for triage.
    Read-only. No side effects.
    """
    t0 = datetime.now(timezone.utc)
    ttl = 60
    cache_key = int(t0.timestamp() // ttl)
    top = _cached_top_hub(cache_key)

    generated_at = t0.isoformat()
    source = "knowledge_graph" if top.degree >= 0 else "cached_fallback"

    logger.info("top_hub=%s degree=%d source=%s", top.slug, top.degree, source)

    return TopHubSignalResponse(
        top_hub=top,
        generated_at=generated_at,
        ttl_seconds=ttl,
        source=source,
    )
```
