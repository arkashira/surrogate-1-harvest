#### 1) Ingestion: produce shards + CDN file list
File: `surrogate/ingest/mirror_to_hf.py`

```python
#!/usr/bin/env python3
"""
After merging parquet for a date, upload deterministic shards
and emit filelist_YYYYMMDD.json with CDN URLs.
"""
import hashlib, json, os, sys
from pathlib import Path
from huggingface_hub import HfApi, hf_hub_upload

API = HfApi()
BASE_REPO = "axentx/surrogate-mirror"   # will use siblings: surrogate-mirror-0..4
N_SHARDS = 5
DATE_ROOT = "batches/mirror-merged"

def shard_id_for_slug(slug: str) -> int:
    return int(hashlib.sha256(slug.encode()).hexdigest(), 16) % N_SHARDS

def repo_for_shard(shard: int) -> str:
    return f"{BASE_REPO}-{shard}" if shard > 0 else BASE_REPO

def upload_shards_for_date(date_dir: Path):
    # date_dir contains {slug}.parquet
    filelist = []
    for parquet_path in date_dir.glob("*.parquet"):
        slug = parquet_path.stem
        shard = shard_id_for_slug(slug)
        repo = repo_for_shard(shard)
        rel_path = f"{DATE_ROOT}/{date_dir.name}/{slug}.parquet"

        # upload to HF sibling repo
        hf_hub_upload(
            repo_id=repo,
            path_in_repo=rel_path,
            path_or_fileobj=str(parquet_path),
            commit_message=f"Add {rel_path}",
        )

        cdn_url = f"https://huggingface.co/datasets/{repo}/resolve/main/{rel_path}"
        filelist.append(cdn_url)

    # write and commit filelist
    out_name = f"filelist_{date_dir.name.replace('-','')}.json"
    out_path = date_dir.parent / out_name
    with open(out_path, "w") as f:
        json.dump(filelist, f, indent=2)

    # commit filelist to primary repo for discoverability
    hf_hub_upload(
        repo_id=repo_for_shard(0),
        path_in_repo=out_name,
        path_or_fileobj=str(out_path),
        commit_message=f"Add {out_name}",
    )
    print(f"Wrote {len(filelist)} entries to {out_name}")

if __name__ == "__main__":
    if len(sys.argv) != 2:
        print("Usage: mirror_to_hf.py <YYYY-MM-DD>")
        sys.exit(1)
    upload_shards_for_date(Path(sys.argv[1]))
```

#### 2) Training loader: CDN-first with single file-list fetch
File: `surrogate/train/data.py`

```python
import json
import pyarrow.parquet as pq
import requests
from io import BytesIO
from pathlib import Path
from typing import List, Union

def load_file_list(source: Union[str, Path]) -> List[str]:
    """source: local path or HF raw URL (filelist_YYYYMMDD.json)"""
    if isinstance(source, Path) or (isinstance(source, str) and Path(source).exists()):
        with open(source) as f:
            return json.load(f)
    # assume URL
    resp = requests.get(source, timeout=30)
    resp.raise_for_status()
    return resp.json()

def cdn_fetch(cdn_url: str) -> BytesIO:
    resp = requests.get(cdn_url, timeout=30)
    resp.raise_for_status()
    return BytesIO(resp.content)

def project_to_prompt_response(table):
    keep = [c for c in table.column_names if c in {"prompt", "response"}]
    if not keep:
        raise ValueError("No prompt/response columns in parquet")
    return table.select(keep)

def load_dataset_cdn(filelist_source: Union[str, Path]):
    urls = load_file_list(filelist_source)
    tables = []
    for url in urls:
        try:
            buf = cdn_fetch(url)
            table = pq.read_table(buf)
            tables.append(project_to_prompt_response(table))
        except Exception:
            # minimal fallback: direct HF download only for this shard
            from huggingface_hub import hf_hub_download
            # derive repo+path from CDN URL (simple parsing)
            parts = url.replace("https://huggingface.co/datasets/", "").split("/resolve/main/")
            repo_id = parts[0]
            file_path = parts[1]
            local_path = hf_hub_download(repo_id=repo_id, filename=file_path)
            table = pq.read_table(local_path)
            tables.append(project_to_prompt_response(table))

    # combine chunks for downstream use
    if not tables:
        raise ValueError("No data loaded")
    combined = tables[0]
    for t in tables[1:]:
        combined = combined.append(t)
    return combined
```

#### 3) Lightning Studio reuse + idle handling
File: `surrogate/train/lightning_studio.py`

```python
from lightning.pai import Studio, Machine, Teamspace
from typing import Optional

STUDIO_NAME = "surrogate-train-l40s"

def get_or_create_studio() -> Studio:
    # Prefer already-running studio to save quota
    for s in Teamspace.studios:
        if s.name == STUDIO_NAME and s.status == "Running":
            print(f"Reusing running studio: {STUDIO_NAME}")
            return s

    # Restart stopped studio
    for s in Teamspace.studios:
        if s.name == STUDIO_NAME:
            print(f"Restarting stopped studio: {STUDIO_NAME}")
            s.start(machine=Machine.L40S)
            return s

    # Create new
    print(f"Creating new studio: {STUDIO_NAME}")
    return Studio(name=STUDIO_NAME, machine=Machine.L40S, create_ok=True)

def run_training(script_path: str, args: Optional[list] = None):
    studio = get_or_create_studio()
    if studio.status != "Running":
        studio.start(machine=Machine.L40S)

    job = studio.run(
        name="surrogate-train-run",
        command=f"python {script_path} {' '.join(args or [])}",
        requirements=["pyarrow", "requests"],
    )
    print(f"Started training job: {job.id}")
    return job
```
