# workio / frontend

## Final Synthesized Implementation

Below is the single, production-ready plan that merges the strongest, non-contradictory parts of both proposals and resolves all conflicts in favor of **correctness + concrete actionability**.

---

## 1. Core Principles (resolve contradictions)

| Conflict | Resolution |
|---|---|
| **Idempotency: client vs backend** | Client **always** generates a stable idempotency key (`crypto.randomUUID()`) per logical punch and persists it locally. Backend **must** honor `Idempotency-Key` and return the same record on retry. |
| **Optimistic UI vs server truth** | Optimistic UI is allowed, but always tagged with the client idempotency key and `clientTimestamp`. On sync, reconcile with server record; never silently overwrite user-visible times without confirmation if skew > threshold. |
| **Offline/retry strategy** | Use **persistent queue (IndexedDB)** for punches that fail. In-memory retry for transient errors; queue for offline or 5xx/429. |
| **Feedback & last-known-state** | Show immediate pending state, then synced/error. Always surface last punch time and today’s hours in UI; update optimistically, then reconcile. |
| **Geolocation & spoof detection** | Capture location at punch time; show inline reason if rejected (GPS mismatch/off-site). Do not rely on LINE webhook roundtrip for user feedback. |

### `workio/src/lib/queue/indexeddb.ts`
```ts
import { openDB, DBSchema, IDBPDatabase } from 'idb';

interface AttendanceQueueDB extends DBSchema {
  punches: {
    key: string; // idempotencyKey
    value: {
      idempotencyKey: string;
      type: 'clock_in' | 'clock_out';
      latitude?: number;
      longitude?: number;
      clientTimestamp: string;
      retries: number;
      createdAt: number;
    };
  };
}

const DB_NAME = 'attendance_queue';
const DB_VERSION = 1;

async function getDB(): Promise<IDBPDatabase<AttendanceQueueDB>> {
  return openDB<AttendanceQueueDB>(DB_NAME, DB_VERSION, {
    upgrade(db) {
      db.createObjectStore('punches', { keyPath: 'idempotencyKey' });
    },
  });
}

export async function enqueuePunch(
  punch: Omit<AttendanceQueueDB['value'], 'retries' | 'createdAt'>
) {
  const db = await getDB();
  await db.add('punches', {
    ...punch,
    retries: 0,
    createdAt: Date.now(),
  });
}

export async function dequeuePunch(idempotencyKey: string) {
  const db = await getDB();
  await db.delete('punches', idempotencyKey);
}

export async function peekPendingPunches(): Promise<AttendanceQueueDB['value'][]> {
  const db = await getDB();
  return db.getAll('punches');
}

export async function updatePunchRetries(idempotencyKey: string, retries: number) {
  const db = await getDB();
  const tx = db.transaction('punches', 'readwrite');
  const record = await tx.store.get(idempotencyKey);
  if (record) {
    record.retries = retries;
    await tx.store.put(record);
  }
  await tx.done;
}
```

### `workio/src/lib/api/attendance.ts`
```ts
import { AttendanceRecord } from '../types/attendance';

const API_BASE = '/api/attendance';

export async function punch(
  type: 'clock_in' | 'clock_out',
  options: {
    latitude?: number;
    longitude?: number;
    idempotencyKey: string;
    clientTimestamp: string;
    signal?: AbortSignal;
  }
): Promise<AttendanceRecord> {
  const res = await fetch(`${API_BASE}/punch`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Idempotency-Key': options.idempotencyKey,
    },
    body: JSON.stringify({
      type,
      latitude: options.latitude,
      longitude: options.longitude,
      clientTimestamp: options.clientTimestamp,
    }),
    signal: options.signal,
  });

  if (!res.ok) {
    const err = await res.json().catch(() => ({}));
    const error = new Error(err.message || 'Punch failed');
    (error as any).status = res.status;
    throw error;
  }

  return res.json();
}

export async function getTodaysSummary(signal?: AbortSignal) {
  const res = await fetch(`${API_BASE}/today`, { signal });
  if (!res.ok) throw new Error('Failed to load today summary');
  return res.json();
}
```

### `workio/src/hooks/useGeolocation.ts`
```ts
import { useState, useEffect } from 'react';

export function useGeolocation({ enabled = true } = {}) {
  const [location, setLocation] = useState<{ latitude: number; longitude: number } | null>(null);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    if (!enabled || !navigator.geolocation) {
      setError('ไม่รองรับการระบุตำแหน่ง');
      return;
    }

    const watchId = navigator.geolocation.watchPosition(
      (pos) => {
        setLocation({
          latitude: pos.coords.latitude,
          longitude: pos.coords.longitude,
        });
        setError(null);
      },
      (err) => {
        setError(err.message === 'User denied Geolocation' ? 'ไม่อนุญาตระบุตำแหน่ง' : 'ไม่สามารถระบุตำแหน่งได้');
      },
      { enableHighAccuracy: true, maximumAge: 10000, timeout: 10000 }
    );

    return () => navigator.geolocation.clearWatch(watchId);
  }, [enabled]);

  return { location, error };
}
```
