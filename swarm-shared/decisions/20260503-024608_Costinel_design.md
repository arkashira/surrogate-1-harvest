# Costinel / design

## Final Implementation Plan — Top-Hub Signal Panel (CDN-first, <2h, telemetry-aware)

### What we ship (single source of truth)
- **Non-blocking Top-Hub Signal Panel** on the Costinel dashboard.
- Defaults to hub **MOC** (configurable via `HUB_NAME`).
- Shows **exactly 3 actionable cost-impact proposals** (title, impact, confidence, source).
- **CDN-first data strategy**:
  - Build-time: CI/mac orchestrator runs **once per refresh window** to fetch `list_repo_tree` and write `public/hub-signals/index.json` with CDN paths.
  - Runtime: panel fetches `https://huggingface.co/datasets/axentx/costinel-hub-signals/resolve/main/{HUB_NAME}.json` (no auth, no HF API, no rate limits).
  - Fallback: if CDN fails, silently hide panel (non-blocking) or use last bundled JSON if you prefer resilience.
- **Telemetry-aware**:
  - Respects idle/stop; reuses running Lightning Studio (no extra model load on Mac).
  - Logs panel impressions and load outcomes (success/failure/latency) via existing telemetry hook.
- **Lightning integration**: optional notebook in `lightning/` to refresh signals via CDN only (zero HF API during training).

---

### Architecture (merged, minimal)
- **Frontend**: `src/components/TopHubSignalPanel.tsx`
- **Types**: `src/types/hubSignals.ts`
- **Data loader**: `src/lib/hubSignals.ts`
- **Build/CI script**: `scripts/embed-hub-list.js` (mac orchestrator)
- **Telemetry**: thin wrapper in panel (reuse existing `useTelemetry` or similar).

#### 1) Frontend component — `src/components/TopHubSignalPanel.tsx`
```tsx
import { useEffect, useState } from "react";
import { ExternalLink, TrendingUp, AlertCircle } from "lucide-react";
import { cn } from "@/lib/utils";
import type { HubSignal } from "@/types/hubSignals";
import { useTelemetry } from "@/lib/telemetry";

const HUB_NAME = import.meta.env.VITE_HUB_NAME || "moc";
const CDN_ROOT = "https://huggingface.co/datasets/axentx/costinel-hub-signals/resolve/main";

export function TopHubSignalPanel() {
  const [signals, setSignals] = useState<HubSignal[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);
  const telemetry = useTelemetry();

  useEffect(() => {
    let mounted = true;
    const start = performance.now();

    async function load() {
      try {
        const res = await fetch(`${CDN_ROOT}/${HUB_NAME}.json`, { cache: "no-cache" });
        if (!res.ok) throw new Error(`CDN fetch failed: ${res.status}`);
        const data: HubSignal[] = await res.json();
        if (!mounted) return;
        // Validate and take top 3 by confidence+impact
        const valid = data.filter((s) => s && s.id && typeof s.confidence === "number");
        const top3 = valid
          .sort((a, b) => {
            const scoreA = (a.estimatedSavings || 0) * a.confidence;
            const scoreB = (b.estimatedSavings || 0) * b.confidence;
            return scoreB - scoreA;
          })
          .slice(0, 3);
        setSignals(top3);
        telemetry.record("hub_signals_loaded", { hub: HUB_NAME, count: top3.length, latencyMs: performance.now() - start });
      } catch (err) {
        if (!mounted) return;
        const msg = err instanceof Error ? err.message : "Unknown error";
        setError(msg);
        telemetry.record("hub_signals_failed", { hub: HUB_NAME, error: msg, latencyMs: performance.now() - start });
      } finally {
        if (mounted) setLoading(false);
      }
    }

    load();
    return () => { mounted = false; };
  }, [HUB_NAME, telemetry]);

  if (loading) {
    return (
      <div className="rounded-lg border bg-card p-4">
        <div className="flex items-center gap-2 text-sm text-muted-foreground">
          <TrendingUp className="h-4 w-4 animate-pulse" />
          Loading top-hub signals...
        </div>
      </div>
    );
  }

  // Non-blocking: hide on error or empty
  if (error || signals.length === 0) {
    return null;
  }

  return (
    <div className="rounded-lg border bg-card p-4">
      <div className="mb-3 flex items-center justify-between">
        <div className="flex items-center gap-2">
          <TrendingUp className="h-4 w-4 text-primary" />
          <span className="font-semibold text-sm uppercase tracking-wide">
            Top signals — {HUB_NAME.toUpperCase()}
          </span>
        </div>
        <a
          href={`https://huggingface.co/datasets/axentx/costinel-hub-signals/tree/main/${HUB_NAME}`}
          target="_blank"
          rel="noopener noreferrer"
          className="text-xs text-muted-foreground hover:text-foreground flex items-center gap-1"
        >
          View hub <ExternalLink className="h-3 w-3" />
        </a>
      </div>

      <div className="space-y-3">
        {signals.map((s, i) => (
          <div
            key={s.id}
            className={cn(
              "rounded-md border p-3 text-sm",
              i === 0 ? "border-primary/30 bg-primary/5" : "border-border"
            )}
          >
            <div className="flex items-start justify-between gap-2">
              <div className="flex-1">
                <div className="font-medium text-foreground">{s.title}</div>
                <div className="text-xs text-muted-foreground mt-1">{s.description}</div>
                <div className="flex items-center gap-3 mt-2 text-xs">
                  <span className="flex items-center gap-1">
                    <span
                      className={cn(
                        "h-2 w-2 rounded-full",
                        s.impact === "high"
                          ? "bg-destructive"
                          : s.impact === "medium"
                          ? "bg-warning"
                          : "bg-muted-foreground"
                      )}
                    />
                    Impact: {s.impact}
                  </span>
                  <span>Confidence: {Math.round(s.confidence * 100)}%</span>
                  {typeof s.estimatedSavings === "number" && (
                    <span className="font-mono text-xs text-emerald-600">
                      Est. ${s.estimatedSavings.toLocaleString()}/mo
                    </span>
                  )}
                </div>
              </div>
              <div className="flex flex-col items-end gap-1">
                <span className="text-xs text-muted-foreground">{s.category}</span>
                {s.source && (
                  <a
                    href={s.source}
                    target="_blank"
                    rel="noopener noreferrer"
                    className="text-xs text-primary hover:underline flex items-center gap-1"
                  >
                    Source <ExternalLink className="h-3 w-3" />
                  </a>
                )}
              </div>
            </div>
          </div>
        ))}
      </div>

      {error && (
        <div className="mt-3 flex items-center gap-2 text-xs text-muted-foreground">
          <AlertCircle className="h-3 w-3" />
          Could not load latest signals — showing cached or hidden.
        </div>
      )}
    </div>
  );
}
```

export interface HubSignal {

  impact: "low" | "medium" | "high";
  confidence: number; // 0-1
