#### 2) Add CDN fetcher utility
File: `/src/lib/fetch-cdn.ts`

```ts
const CDN_ROOT = 'https://huggingface.co/datasets/axentx/costinel-knowledge/resolve/main';

export async function fetchCdnJson<T>(path: string, localFallbackPath?: string): Promise<T | null> {
  const url = `${CDN_ROOT}/${path}`;
  try {
    const res = await fetch(url, { cache: 'no-store' });
    if (!res.ok) throw new Error(`CDN fetch failed: ${res.status}`);
    return (await res.json()) as T;
  } catch (err) {
    console.warn('[CDN] fetch failed, using local fallback', err);
    if (localFallbackPath) {
      try {
        const localRes = await fetch(localFallbackPath);
        if (!localRes.ok) throw new Error('Local fallback failed');
        return (await localRes.json()) as T;
      } catch (localErr) {
        console.error('[CDN] local fallback failed', localErr);
        return null;
      }
    }
    return null;
  }
}
```

#### 3) Add TopHubSignalPanel component
File: `/src/components/TopHubSignalPanel.tsx`

```tsx
import React, { useEffect, useState } from 'react';
import { fetchCdnJson } from '../lib/fetch-cdn';

type Manifest = {
  generatedAt: string;
  topHub: {
    id: string;
    label: string;
    description: string;
    connections: number;
    lastUpdated: string;
    cdnPayloadPath: string;
  };
  fallback: {
    label: string;
    description: string;
  };
};

type HubSummary = {
  id: string;
  label: string;
  description: string;
  connections: number;
  keySignals: string[];
  lastUpdated: string;
};

export const TopHubSignalPanel: React.FC = () => {
  const [summary, setSummary] = useState<HubSummary | null>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    let mounted = true;
    setLoading(true);

    // Step 1: load manifest (local, fast)
    fetch('/knowledge-rag/top-hub-manifest.json')
      .then((r) => r.json())
      .then(async (manifest: Manifest) => {
        if (!mounted) return;

        // Step 2: try CDN for fresh payload (zero API auth; CDN-only)
        const payload = await fetchCdnJson<HubSummary>(
          manifest.topHub.cdnPayloadPath,
          `/knowledge-rag/${manifest.topHub.cdnPayloadPath}`
        );

        if (payload && mounted) {
          setSummary(payload);
        } else if (mounted) {
          // fallback to baked manifest data
          setSummary({
            id: manifest.topHub.id,
            label: manifest.topHub.label,
            description: manifest.topHub.description,
            connections: manifest.topHub.connections,
            keySignals: [],
            lastUpdated: manifest.topHub.lastUpdated,
          });
        }
        setLoading(false);
      })
      .catch((err) => {
        console.error('[TopHubSignalPanel] failed to load', err);
        if (mounted) setLoading(false);
      });

    return () => {
      mounted = false;
    };
  }, []);

  if (loading) {
    return (
      <div className="rounded-lg border bg-card/50 p-4 animate-pulse">
        <div className="h-5 w-32 bg-muted rounded mb-2" />
        <div className="h-4 w-full bg-muted rounded mb-1" />
        <div className="h-4 w-5/6 bg-muted rounded" />
      </div>
    );
  }

  if (!summary) {
    return (
      <div className="rounded-lg border bg-card/50 p-4 text-sm text-muted-foreground">
        Top hub signal unavailable.
      </div>
    );
  }

  return (
    <div className="rounded-lg border bg-card/50 p-4">
      <div className="flex items-start justify-between mb-2">
        <div>
          <h3 className="font-semibold text-sm">Top Hub Signal</h3>
          <p className="text-xs text-muted-foreground">{summary.lastUpdated}</p>
        </div>
        <span className="inline-flex items-center rounded-full bg-primary/10 px-2.5 py-0.5 text-xs font-medium text-primary">
          {summary.connections} connections
        </span>
      </div>

      <div className="mb-3">
        <p className="font-medium text-base">{summary.label}</p>
        <p className="text-sm text-muted-foreground">{summary.description}</p>
      </div>

      {summary.keySignals && summary.keySignals.length > 0 && (
        <div className="flex flex-wrap gap-1.5">
          {summary.keySignals.map((sig) => (
            <span
              key={sig}
              className="inline-flex items-center rounded-md bg-muted px-2 py-0.5 text-xs font-medium text-foreground"
            >
              {sig}
            </span>
          ))}
        </div>
      )}
    </div>
  );
};
```
