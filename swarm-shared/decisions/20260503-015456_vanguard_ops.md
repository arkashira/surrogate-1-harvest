# vanguard / ops

## 1. Diagnosis

- No persisted `(repo, dateFolder) → file-list` manifest: every training run triggers authenticated `list_repo_tree` against HF API, burning quota and risking 429s.
- Data loader likely uses `load_dataset(streaming=True)` or repeated per-file API calls on heterogeneous schemas, causing `pyarrow.CastError` and wasted API budget.
- No CDN-only fetch path: training jobs perform authenticated calls during data loading instead of using public CDN URLs (`https://huggingface.co/datasets/.../resolve/main/...`), which bypass auth rate limits.
- Missing Lightning Studio reuse logic: scripts probably create new studios each run, burning 80+ hours/month of quota instead of reusing running instances.
- No HF commit-cap mitigation: ingestion writes all files to one repo, risking the 128/hr/repo cap instead of spreading across sibling repos.

## 2. Proposed change

Create `/opt/axentx/vanguard/ops/file_manifest.py` + update training launcher to:
- Generate and cache a `(repo, dateFolder) → [file_paths]` JSON manifest via a single authenticated `list_repo_tree` call (Mac-side, after rate-limit window).
- Embed that manifest in Lightning training scripts so data loading uses CDN-only fetches (zero API calls during training).
- Add Lightning Studio reuse check before launching training.

Scope: new file + small edits to launcher/train script (if present); no schema changes.

```python
#!/usr/bin/env python3
"""
Generate and cache HF dataset file manifests to avoid API calls during training.
Usage:
    python file_manifest.py --repo <repo> --date <YYYY-MM-DD> --out ./manifest.json
"""
import argparse
import json
import os
import sys
from datetime import datetime
from pathlib import Path

try:
    from huggingface_hub import HfApi
except ImportError:
    print("Install: pip install huggingface-hub")
    sys.exit(1)

API = HfApi()

def list_date_files(repo_id: str, date_folder: str, out_path: Path):
    """
    Single authenticated list_repo_tree call for one date folder (non-recursive).
    CDN downloads (resolve/main/...) are NOT counted against API limits.
    """
    print(f"Listing {repo_id}/{date_folder} (non-recursive)...")
    try:
        tree = API.list_repo_tree(repo_id=repo_id, path=date_folder, recursive=False)
    except Exception as e:
        print(f"HF API error: {e}")
        # If rate-limited, wait 360s and retry once (caller should orchestrate timing)
        raise

    files = [item.path for item in tree if item.type == "file"]
    manifest = {
        "repo_id": repo_id,
        "date_folder": date_folder,
        "generated_at": datetime.utcnow().isoformat() + "Z",
        "files": sorted(files),
        "note": "CDN-only paths: use https://huggingface.co/datasets/{repo_id}/resolve/main/{file}"
    }

    out_path.parent.mkdir(parents=True, exist_ok=True)
    with open(out_path, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Manifest written to {out_path} ({len(files)} files)")
    return manifest

def main():
    parser = argparse.ArgumentParser(description="Generate HF dataset file manifest")
    parser.add_argument("--repo", required=True, help="HF dataset repo id (user/ds)")
    parser.add_argument("--date", required=True, help="Date folder (YYYY-MM-DD)")
    parser.add_argument("--out", default="./manifest.json", help="Output JSON path")
    args = parser.parse_args()

    out_path = Path(args.out).resolve()
    list_date_files(args.repo, args.date, out_path)

if __name__ == "__main__":
    main()
```

Create `/opt/axentx/vanguard/ops/lightning_reuse.py`:

```python
#!/usr/bin/env python3
"""
Reuse a running Lightning Studio if available; otherwise create one.
Saves quota by avoiding duplicate studios.
"""
import argparse
import time
from pathlib import Path

try:
    from lightning import Lightning, Studio, Machine
except ImportError:
    print("Install: pip install lightning")
    sys.exit(1)

def get_running_studio(lightning, name: str):
    for s in lightning.studios():
        if s.name == name and s.status == "running":
            return s
    return None

def launch_or_reuse(
    repo: str,
    date_folder: str,
    manifest_path: str,
    studio_name: str = "vanguard-train",
    machine: str = "lightning-public-prod/L40S",
    script: str = "train.py",
):
    lightning = Lightning()
    studio = get_running_studio(lightning, studio_name)

    if studio is None:
        print(f"No running studio '{studio_name}'; creating...")
        studio = Studio(
            name=studio_name,
            machine=Machine(machine),
            create_ok=True,
        )
    else:
        print(f"Reusing running studio '{studio_name}' (id={studio.id})")

    # Ensure studio is running before .run()
    if studio.status != "running":
        print("Studio not running; starting...")
        studio.start(machine=Machine(machine))
        # Poll briefly
        for _ in range(10):
            studio.refresh()
            if studio.status == "running":
                break
            time.sleep(6)
        if studio.status != "running":
            raise RuntimeError("Studio failed to start")

    # Pass manifest so train.py does CDN-only fetches
    run_result = studio.run(
        str(Path(script).resolve()),
        arguments=[
            "--repo", repo,
            "--date", date_folder,
            "--manifest", manifest_path,
        ],
    )
    print("Run submitted:", run_result)
    return run_result

if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True)
    parser.add_argument("--date", required=True)
    parser.add_argument("--manifest", default="./manifest.json")
    parser.add_argument("--studio", default="vanguard-train")
    parser.add_argument("--machine", default="lightning-public-prod/L40S")
    parser.add_argument("--script", default="train.py")
    args = parser.parse_args()

    launch_or_reuse(
        repo=args.repo,
        date_folder=args.date,
        manifest_path=args.manifest,
        studio_name=args.studio,
        machine=args.machine,
        script=args.script,
    )
```
