# vanguard / ops

## Final Synthesized Solution

**Core diagnosis (unified):**  
- No content-addressed manifest → runtime repo re-listing causes 429s and non-reproducible runs.  
- Mixed-schema files land in `enriched/` without projection to `{prompt,response}` → breaks surrogate-1 training.  
- No CDN-bypass strategy → training jobs hit `/api/` endpoints and trigger rate-limits.  
- Lightning quota waste + idle-stop fragility → scripts create new Studios instead of reusing running ones, and idle timeouts kill long-running training without auto-restart.

**Single concrete change:**  
Add `/opt/axentx/vanguard/ops/manifest.py` (single file, <150 LOC) and update `/opt/axentx/vanguard/ops/train_surrogate1.py` to:

1. **Build once, use everywhere**  
   - `build_manifest(repo, folder, out_path)` performs a single `list_repo_tree` call and writes `manifest-{date}.json` containing CDN URLs (`resolve/main/...`) for every file in that folder.  
   - Accept `--manifest`; if absent, build it once (Mac-side) then embed the path.

2. **CDN-only data loading with schema projection**  
   - `iter_cdn_urls(manifest_path)` yields `(path, cdn_url)`.  
   - Use a custom `IterableDataset` that streams via CDN URLs (`requests.get(..., stream=True, timeout=60)`), parses Parquet in-memory, and projects to `{prompt, response}` at parse time.  
   - Avoid all `/api/` calls during training; fall back to `hf_hub_download` only on CDN failure (should be rare).

3. **Lightning quota + uptime guard**  
   - `reuse_or_create_studio(name, machine)` reuses a running studio or creates one (respects free-tier limits).  
   - `ensure_running(studio, machine)` checks status and restarts stopped studios (auto-restart on idle timeout).  
   - Call both before `.run()` to prevent quota burn and training interruption.

**Implementation (merged, corrected, actionable):**

# /opt/axentx/vanguard/ops/manifest.py
#!/usr/bin/env python3
"""
Content-addressed manifest + CDN-bypass utilities for HF datasets.
"""
import json
import time
from pathlib import Path
from typing import Iterator, Tuple

import requests
from hugginggingface_hub import HfApi, list_repo_tree

try:
    from lightning import Lightning, Machine, Studio, Teamspace
except ImportError:
    Lightning = None  # type: ignore

HF_API = HfApi()
CDN_ROOT = "https://huggingface.co/datasets"

def build_manifest(repo: str, folder: str, out_path: str) -> str:
    """
    Single API call: list_repo_tree (non-recursive) for one folder.
    Writes manifest JSON with CDN URLs.
    """
    entries = list_repo_tree(repo=repo, path=folder, recursive=False)
    files = [e for e in entries if e.type == "file"]
    manifest = {
        "repo": repo,
        "folder": folder,
        "generated_at": time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime()),
        "files": [],
    }
    for f in files:
        cdn_url = f"{CDN_ROOT}/{repo}/resolve/main/{f.path}"
        manifest["files"].append(
            {"path": f.path, "cdn_url": cdn_url, "size": getattr(f, "size", None)}
        )
    Path(out_path).parent.mkdir(parents=True, exist_ok=True)
    with open(out_path, "w") as fp:
        json.dump(manifest, fp, indent=2)
    return out_path

def iter_cdn_urls(manifest_path: str) -> Iterator[Tuple[str, str]]:
    """Yield (path, cdn_url) for CDN-only fetches."""
    with open(manifest_path) as fp:
        manifest = json.load(fp)
    for f in manifest["files"]:
        yield f["path"], f["cdn_url"]

def stream_cdn_file(cdn_url: str, chunk_size: int = 1024 * 1024) -> Iterator[bytes]:
    """Stream from CDN (no auth, no API rate-limit)."""
    resp = requests.get(cdn_url, stream=True, timeout=60)
    resp.raise_for_status()
    for chunk in resp.iter_content(chunk_size=chunk_size):
        yield chunk

def reuse_or_create_studio(name: str, machine: str = "L40S", create_ok: bool = True):
    """Reuse running studio; avoids quota waste."""
    if Lightning is None:
        raise RuntimeError("lightning not installed")
    teamspace = Teamspace()
    for s in teamspace.studios:
        if s.name == name and s.status == "Running":
            return s
    if not create_ok:
        raise RuntimeError(f"No running studio {name}")
    return Studio.create(name=name, machine=Machine(machine), create_ok=True)

def ensure_running(studio, machine: str = "L40S"):
    """Restart stopped studio (Lightning idle timeout kills training)."""
    if studio.status != "Running":
        studio.start(machine=Machine(machine))
        for _ in range(10):
            studio.refresh()
            if studio.status == "Running":
                return
            time.sleep(6)
    return

#!/usr/bin/env python3
import argparse
import io
import json
from pathlib import Path

import pyarrow.parquet as pq
import torch
from torch.utils.data import IterableDataset, DataLoader

from manifest import iter_cdn_urls, reuse_or_create_studio, ensure_running, build_manifest, stream_cdn_file

class CDNParquetIterable(IterableDataset):
    """CDN-only fetches, project to {prompt,response}."""
    def __init__(self, manifest_path: str, max_files: int = -1):
        self.manifest_path = manifest_path
        self.max_files = max_files

    def __iter__(self):
        count = 0
        for path, url in iter_cdn_urls(self.manifest_path):
            if self.max_files > 0 and count >= self.max_files:
                break
            try:
                raw = b"".join(stream_cdn_file(url))
            except Exception:
                # fallback (should be rare)
                with open(self.manifest_path) as fp:
                    repo = json.load(fp)["repo"]
                raw = Path(__file__).parent.joinpath(
                    # delegate to hf_hub_download for robustness
                    __import__("huggingface_hub").hf_hub_download(repo, path, repo_type="dataset")
                ).read_bytes()
            table = pq.read_table(io.BytesIO(raw))
            df = table.select(["prompt", "response"]).to_pandas()
            for _, row in df.iterrows():
                yield {"prompt": row["prompt"], "response": row["response"]}
            count += 1

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", default="axentx/dataset-mirror")
    parser.add_argument("--folder", default="batches/mirror-merged/2026-04-29")
    parser.add_argument("--manifest", default=None)
    parser.add_argument("--studio", default="surrogate1-train")
    args = parser.parse_args()

    # Build manifest once (Mac-side) if missing
    if not args.manifest or not Path(args.manifest).exists():
        out = args.manifest or f"manifest-{Path(args.folder).name}.json"
        print(f"Building manifest -> {out}")
        build_manifest(args.repo, args.folder, out)
        args.manifest = out

    # Lightning: reuse studio + guard idle-stop
    studio = reuse_or_create_studio(args.studio, machine="L40S")
    ensure_running(studio, machine="L40S")

    dataset = CDNParquetIterable(args.manifest, max_files=50)
    loader = DataLoader(dataset, batch_size=8, num_workers=0)

        # surrogate-1 training step
        pass

if __name__ == "__main__":
    main()
