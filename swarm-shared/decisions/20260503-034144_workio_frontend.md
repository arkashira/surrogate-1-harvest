# workio / frontend

## 1. Diagnosis
- Frontend state is stale after LINE webhook state changes (clock/leave/OT) until hard refresh → perceived latency and silent failures.
- No client-side subscription layer; UI relies on local React state + manual refresh, so multi-tab/devices diverge.
- Polling is absent or inefficient; users cannot see real-time updates from coworkers or their own LINE-triggered events.
- No lightweight optimistic UI path for common actions (clock in/out), so perceived performance is poor on high-latency mobile networks.
- Missing reconnect/backoff and tab-coordination for real-time channel, risking duplicate connections or missed updates.

## 2. Proposed change
Add a realtime subscription layer using EventSource (SSE) from `/events` (backend) and a React context + hook (`useRealtime`) that:
- Subscribes to tenant-scoped events (clock, leave, ot, attendance)
- Updates global state (React Query cache) optimistically where appropriate
- Coordinates across tabs via BroadcastChannel
- Exposes `useOptimisticClock` for immediate local feedback on clock in/out

Scope:
- New file: `src/lib/realtime.ts` (SSE client + BroadcastChannel coordination)
- New file: `src/context/RealtimeProvider.tsx` (React context/provider)
- New hook: `src/hooks/useRealtime.ts` (typed event subscription + React Query invalidation)
- New hook: `src/hooks/useOptimisticClock.ts` (optimistic UI for clock actions)
- Modify: `src/main.tsx` to wrap app with `RealtimeProvider`
- Modify: clock/leave/OT mutation calls to use optimistic helpers

## 3. Implementation

### src/lib/realtime.ts
```ts
export type RealtimeEvent =
  | { type: 'clock'; payload: { userId: string; tenantId: string; ts: string; status: 'in' | 'out' } }
  | { type: 'leave'; payload: { userId: string; tenantId: string; leaveId: string; status: string } }
  | { type: 'ot'; payload: { userId: string; tenantId: string; otId: string; status: string } }
  | { type: 'attendance'; payload: { tenantId: string; date: string } };

const CHANNEL = 'workio-realtime';

export class RealtimeClient {
  private es: EventSource | null = null;
  private bc = new BroadcastChannel(CHANNEL);
  private listeners = new Set<(ev: RealtimeEvent) => void>();
  private tenantId: string | null = null;

  constructor(private baseUrl: string) {
    this.bc.onmessage = (msg) => {
      if (msg.data?.type === 'FORWARD') this.forwardToListeners(msg.data.payload);
    };
  }

  connect(tenantId: string) {
    if (this.tenantId === tenantId && this.es && this.es.readyState === EventSource.OPEN) return;
    this.disconnect();
    this.tenantId = tenantId;
    const url = `${this.baseUrl}/events?tenantId=${encodeURIComponent(tenantId)}`;
    this.es = new EventSource(url);
    this.es.onmessage = (e) => {
      try {
        const payload = JSON.parse(e.data) as RealtimeEvent;
        this.forwardToListeners(payload);
        // broadcast to other tabs
        this.bc.postMessage({ type: 'FORWARD', payload });
      } catch {
        // ignore malformed
      }
    };
    this.es.onerror = () => {
      // reconnect handled by browser; we can optionally add backoff by closing/reopening
    };
  }

  private forwardToListeners(payload: RealtimeEvent) {
    this.listeners.forEach((l) => {
      try {
        l(payload);
      } catch {}
    });
  }

  subscribe(fn: (ev: RealtimeEvent) => void) {
    this.listeners.add(fn);
    return () => this.listeners.delete(fn);
  }

  disconnect() {
    if (this.es) {
      this.es.close();
      this.es = null;
    }
    this.tenantId = null;
  }

  close() {
    this.disconnect();
    this.bc.close();
  }
}
```

### src/context/RealtimeProvider.tsx
```tsx
import React, { createContext, useContext, useEffect, useState } from 'react';
import { RealtimeClient } from '../lib/realtime';

const RealtimeContext = createContext<{ client: RealtimeClient | null }>({ client: null });

export const RealtimeProvider: React.FC<{ children: React.ReactNode; baseUrl?: string }> = ({
  children,
  baseUrl = '/',
}) => {
  const [client] = useState(() => new RealtimeClient(baseUrl));

  useEffect(() => {
    return () => client.close();
  }, [client]);

  return <RealtimeContext.Provider value={{ client }}>{children}</RealtimeContext.Provider>;
};

export const useRealtimeContext = () => useContext(RealtimeContext);
```

### src/hooks/useRealtime.ts
```ts
import { useEffect } from 'react';
import { useQueryClient } from '@tanstack/react-query';
import { RealtimeEvent } from '../lib/realtime';
import { useRealtimeContext } from '../context/RealtimeProvider';

export const useRealtime = (tenantId: string | undefined) => {
  const queryClient = useQueryClient();
  const { client } = useRealtimeContext();

  useEffect(() => {
    if (!client || !tenantId) return;
    client.connect(tenantId);

    return client.subscribe((ev) => {
      // Invalidate relevant queries so UI updates
      if (ev.type === 'clock' && ev.payload.tenantId === tenantId) {
        queryClient.invalidateQueries({ queryKey: ['attendance', tenantId] });
        queryClient.invalidateQueries({ queryKey: ['todayStatus', ev.payload.userId, tenantId] });
      }
      if (ev.type === 'leave' && ev.payload.tenantId === tenantId) {
        queryClient.invalidateQueries({ queryKey: ['leaves', tenantId] });
      }
      if (ev.type === 'ot' && ev.payload.tenantId === tenantId) {
        queryClient.invalidateQueries({ queryKey: ['otRequests', tenantId] });
      }
      if (ev.type === 'attendance' && ev.payload.tenantId === tenantId) {
        queryClient.invalidateQueries({ queryKey: ['attendance', tenantId] });
      }
    });
  }, [client, tenantId, queryClient]);
};
```

### src/hooks/useOptimisticClock.ts
```ts
import { useMutation } from '@tanstack/react-query';
import { queryClient } from '../lib/queryClient';

export const useOptimisticClock = (tenantId: string, userId: string) => {
  return useMutation({
    mutationFn: (action: 'in' | 'out') =>
      fetch(`/api/clock`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ tenantId, userId, action }),
      }).then((r) => r.json()),

    onMutate: async (action) => {
      await queryClient.cancelQueries({ queryKey: ['todayStatus', userId, tenantId] });
      const previous = queryClient.getQueryData(['todayStatus', userId, tenantId]);
      queryClient.setQueryData(['todayStatus', userId, tenantId], (old: any) => ({
        ...old,
        lastStatus: action,
        updatedAt: new Date().toISOString(),
        optimistic: true,
      }));
      return { previous };
    },

    onError: (err, _variables, context) => {
      if (context?.previous) {
        queryClient.setQueryData(['todayStatus', userId, tenantId], context.previous);
      }
    },

    onSettled: () => {
      queryClient.invalidateQueries({ queryKey: ['todayStatus', userId, tenantId] });
    },
  });
};
```
