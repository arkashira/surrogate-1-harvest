### Core Decisions (resolved)
- **CDN-only during training** — eliminates HF API rate limits and quota waste.  
- **Lightning idle-resilient runner** — checks studio status before each run and auto-restarts on stopped/idle studio.  
- **Deterministic sibling-repo selection** — mitigates HF commit-cap by hashing a slug to pick one repo for the session.

#!/usr/bin/env python3
"""
List HF dataset files (non-recursive per date folder) and save CDN filelist JSON.
Usage:
    python scripts/list_hf_files.py \
        --repo my-org/surrogate-data \
        --date 2026-04-29 \
        --out filelist.json
"""
import argparse
import json
import os
import time
from pathlib import Path

from huggingface_hub import HfApi, login

HF_TOKEN = os.getenv("HF_TOKEN")
if HF_TOKEN:
    login(token=HF_TOKEN)

api = HfApi()

def list_date_files(repo_id: str, date_folder: str) -> list[str]:
    """List files in repo/date_folder (non-recursive) and return CDN paths."""
    try:
        tree = api.list_repo_tree(repo_id=repo_id, path=date_folder, recursive=False)
    except Exception as e:
        # If 429, wait 360s as per pattern
        print(f"HF API error (possibly 429): {e}. Waiting 360s...")
        time.sleep(360)
        tree = api.list_repo_tree(repo_id=repo_id, path=date_folder, recursive=False)

    files = [item.path for item in tree if item.type == "file"]
    # CDN paths (no auth required)
    cdn_paths = [
        f"https://huggingface.co/datasets/{repo_id}/resolve/main/{path}"
        for path in files
    ]
    return cdn_paths

def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True, help="HF dataset repo, e.g. org/surrogate-data")
    parser.add_argument("--date", required=True, help="Date folder, e.g. 2026-04-29")
    parser.add_argument("--out", required=True, help="Output JSON path")
    args = parser.parse_args()

    cdn_paths = list_date_files(args.repo, args.date)
    out_path = Path(args.out)
    out_path.parent.mkdir(parents=True, exist_ok=True)
    with open(out_path, "w", encoding="utf-8") as f:
        json.dump({"repo": args.repo, "date": args.date, "files": cdn_paths}, f, indent=2)
    print(f"Saved {len(cdn_paths)} CDN file paths to {out_path}")

if __name__ == "__main__":
    main()

def pick_repo_by_slug(slug: str) -> str:
    """Deterministic sibling repo selection to mitigate HF commit cap."""
    idx = hash(slug) % len(SIBLING_REPOS)
    return SIBLING_REPOS[idx]

def cdn_parquet_loader(filelist_json: str) -> Iterator[Dict[str, str]]:
    """
    Yield {prompt, response} examples from CDN parquet files.
    filelist_json: output of scripts/list_hf_files.py
    """
    with open(filelist_json, encoding="utf-8") as f:
        manifest = json.load(f)

    for url in manifest.get("files", []):
        if not isinstance(url, str) or not url.endswith(".parquet"):
            continue
        try:
            resp = requests.get(url, timeout=30)
            resp.raise_for_status()
        except Exception as e:
            print(f"Failed to download {url}: {e}")
            continue

        tmp = tempfile.NamedTemporaryFile(suffix=".parquet", delete=False)
        try:
            tmp.write(resp.content)
            tmp.close()
            table = pq.read_table(tmp.name)

            # Project only needed columns; tolerate mixed schemas
            col_map: Dict[str, str] = {}
            for c in table.column_names:
                low = c.lower()
                if "prompt" in low or "input" in low or "question" in low:
                    col_map["prompt"] = c
                if "response" in low or "output" in low or "answer" in low:
                    col_map["response"] = c

            has_prompt = "prompt" in col_map
            has_response = "response" in col_map
            if not (has_prompt and has_response):
                # Fallback to exact names if aliases failed
                if "prompt" in table.column_names and "response" in table.column_names:
                    col_map.update({"prompt": "prompt", "response": "response"})
                    has_prompt = has_response = True

            if not (has_prompt and has_response):
                print(f"Skipping {url}: missing prompt/response columns")
                continue

            prompts = table.column(col_map["prompt"]).to_pylist()
            responses = table.column(col_map["response"]).to_pylist()

            for p, r in zip(prompts, responses):
                if p is None or r is None:
                    continue
                yield {"prompt": str(p), "response": str(r)}
        finally:
            try:
                os.unlink(tmp.name)
            except Exception:
                pass
