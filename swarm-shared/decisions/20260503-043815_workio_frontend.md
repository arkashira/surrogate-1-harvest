# workio / frontend

### Final synthesized plan (correctness + concrete actionability)

**Core problem**: duplicate attendance records and UI/data races caused by (1) missing webhook idempotency and signature verification (backend) and (2) missing optimistic UI + client-side deduplication + realtime updates (frontend).  
**Resolution priority**: backend correctness first (prevent fraud/replay), then frontend UX (prevent double-taps, show state immediately, reconcile with authoritative state).

### 1. Backend: secure, idempotent LINE webhook ingestion

**File**: `workio/server/src/controllers/webhookController.js` (or equivalent)  
**Key choices**:
- Verify `X-Line-Signature` with constant-time compare to prevent spoofing.
- Use a durable idempotency store keyed by **event id + event type** (not `replyToken`, which can be reused across event types). Fallback to `deliveryContext`/`timestamp` if needed.
- Return **200 OK quickly** and process asynchronously to avoid LINE retries.
- Persist idempotency keys in Redis (or DB) with TTL (e.g., 24–72h) rather than in-process memory.
- Add retry/backoff for outbound LINE notifications (approval alerts) with exponential backoff and jitter; persist notification attempts to avoid loss.

**Implementation sketch**:

```js
const crypto = require('crypto');
const express = require('express');
const router = express.Router();

// Use a real store in production (Redis, DB). Here: simple in-memory + TTL for illustration.
const seenEvents = new Map();
function isDuplicate(key) {
  if (seenEvents.has(key)) return true;
  seenEvents.set(key, true);
  return false;
}
setInterval(() => seenEvents.clear(), 24 * 60 * 60 * 1000); // 24h

function verifySignature(body, signature, secret) {
  const expected = crypto
    .createHmac('sha256', secret)
    .update(Buffer.isBuffer(body) ? body : JSON.stringify(body))
    .digest('hex');
  // constant-time-ish comparison
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

router.post('/webhook/line', express.raw({ type: 'application/json' }), (req, res) => {
  const signature = req.header('X-Line-Signature');
  const secret = process.env.LINE_CHANNEL_SECRET;

  if (!signature || !verifySignature(req.body, signature, secret)) {
    console.error('Invalid LINE signature');
    return res.status(401).send('Invalid signature');
  }

  let payload;
  try {
    payload = JSON.parse(req.body);
  } catch {
    return res.status(400).send('Invalid JSON');
  }

  if (!payload.events || !Array.isArray(payload.events)) {
    return res.status(400).send('No events');
  }

  // Accept events in batch but dedupe per event
  const results = payload.events.map((ev) => {
    const key = `${ev.type}:${ev.source?.userId}:${ev.timestamp}:${ev.id}`;
    if (isDuplicate(key)) {
      return { eventId: ev.id, status: 'duplicate' };
    }
    // Enqueue for async processing (queue, job, or await with timeout)
    // processWebhookEvent(ev).catch(console.error);
    return { eventId: ev.id, status: 'accepted' };
  });

  // Always 200 to stop LINE retries; log/metrics for duplicates
  console.log('Webhook processed', results);
  res.status(200).send('OK');
});

module.exports = router;
```

**ActionQueue** (TypeScript):

```ts
// src/lib/actionQueue.ts
type QueueItem<T = unknown> = {
  id: string;
  fn: () => Promise<T>;
  retries: number;
  maxRetries: number;
  backoffMs: number;
  onConflict?: (error: any) => void;
};

class ActionQueue {
  private queue = new Map<string, QueueItem>();
  private processing = new Set<string>();

  enqueue<T>(key: string, fn: () => Promise<T>, opts: Partial<Omit<QueueItem, 'id' | 'fn'>> = {}) {
    if (this.queue.has(key) || this.processing.has(key)) {
      return Promise.resolve(null as T | null); // dedupe in-flight
    }

    const item: QueueItem = {
      id: key,
      fn,
      retries: 0,
      maxRetries: 3,
      backoffMs: 500,
      onConflict: undefined,
      ...opts,
    };

    this.queue.set(key, item);
    this.process(key);
    return Promise.resolve(null as T | null);
  }

  private async process(key: string) {
    const item = this.queue.get(key);
    if (!item || this.processing.has(key)) return;
    this.queue.delete(key);
    this.processing.add(key);

    try {
      await item.fn();
    } catch (error: any) {
      if (error?.status === 409 && item.onConflict) {
        item.onConflict(error);
      } else if (item.retries < item.maxRetries) {
        item.retries += 1;
        await new Promise((r) => setTimeout(r, item.backoffMs * 2 ** (item.retries - 1)));
        this.queue.set(key, item);
        this.process(key);
      }
    } finally {
      this.processing.delete(key);
    }
  }

  revoke(key: string) {
    this.queue.delete(key);
    this.processing.delete(key);
  }
}

export const actionQueue = new ActionQueue();
```
