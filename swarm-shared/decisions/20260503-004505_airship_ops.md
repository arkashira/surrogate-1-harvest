## 1) One-time manifest generator (Mac/Linux)

File: `scripts/build_hf_cdn_manifest.py`

```python
#!/usr/bin/env python3
"""
Generate file_manifest.json for a single date folder in a HF dataset repo.
Run once per new folder when rate-limit window is clear.

Usage:
  HF_REPO="datasets/your-org/your-repo" python build_hf_cdn_manifest.py \
    --date-folder batches/mirror-merged/2026-05-03 \
    --out-file file_manifest.json
"""
import argparse
import json
import os
import sys
import time
from typing import List, Dict

try:
    from huggingface_hub import list_repo_tree
except ImportError:
    print("Install: pip install huggingface_hub")
    sys.exit(1)

HF_RETRY = 360  # seconds for 429

def list_folder_safe(repo_id: str, folder: str, retries: int = 3) -> List[Dict]:
    for attempt in range(1, retries + 1):
        try:
            items = list_repo_tree(repo_id=repo_id, path=folder, recursive=False)
            return [{"path": f.path, "size": getattr(f, "size", None)} for f in items if f.type == "file"]
        except Exception as exc:
            wait = HF_RETRY if "429" in str(exc) else (2 ** attempt)
            print(f"[attempt {attempt}/{retries}] list_repo_tree failed: {exc}. Waiting {wait}s", file=sys.stderr)
            time.sleep(wait)
    raise RuntimeError(f"Failed to list {repo_id}/{folder} after {retries} attempts")

def main() -> None:
    parser = argparse.ArgumentParser(description="Build CDN manifest for HF dataset folder")
    parser.add_argument("--repo", default=os.getenv("HF_REPO"), help="HF repo id (e.g. datasets/your-org/your-repo)")
    parser.add_argument("--date-folder", required=True, help="Folder path in repo (e.g. batches/mirror-merged/2026-05-03)")
    parser.add_argument("--out-file", default="file_manifest.json", help="Output JSON path")
    args = parser.parse_args()

    if not args.repo:
        parser.error("--repo or HF_REPO env required")

    print(f"Listing {args.repo}/{args.date_folder} (non-recursive)...")
    files = list_folder_safe(args.repo, args.date_folder)

    manifest = {
        "repo_id": args.repo,
        "folder": args.date_folder,
        "cdn_base": f"https://huggingface.co/datasets/{args.repo}/resolve/main",
        "files": files,
    }

    os.makedirs(os.path.dirname(os.path.abspath(args.out_file)) if os.path.dirname(args.out_file) else ".", exist_ok=True)
    with open(args.out_file, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2)
    print(f"Wrote {len(files)} files to {args.out_file}")

if __name__ == "__main__":
    main()
```

Make executable (optional):

```bash
chmod +x scripts/build_hf_cdn_manifest.py
```

## 2) CDN-only, manifest-driven dataset loader

File: `surrogate/train/cdn_manifest_dataset.py`

```python
"""
Dataset loader that uses a pre-built file_manifest.json to fetch parquet files
via HF CDN (no Authorization, no HF API calls during training).

Usage:
  from surrogate.train.cdn_manifest_dataset import CDNManifestDataset
  ds = CDNManifestDataset(manifest_path="file_manifest.json")
  for sample in ds:
      ...
"""
import json
import io
import pyarrow.parquet as pq
import requests
from typing import Dict, Any, Iterator, List
import datasets

class CDNManifestDataset:
    """
    Lightweight iterable dataset that:
    - Reads file_manifest.json once
    - Downloads each parquet via CDN URL (no auth)
    - Projects to {prompt, response} at parse time
    """

    def __init__(self, manifest_path: str, columns: List[str] = ("prompt", "response")):
        with open(manifest_path, "r", encoding="utf-8") as f:
            self.manifest = json.load(f)
        self.cdn_base = self.manifest.get(
            "cdn_base",
            f"https://huggingface.co/datasets/{self.manifest['repo_id']}/resolve/main"
        )
        self.folder = self.manifest.get("folder", "")
        self.files = [f["path"] for f in self.manifest.get("files", [])]
        self.columns = columns
        self._file_idx = 0
        self._row_idx = 0
        self._rows: List[Dict[str, str]] = []

    def _download_parquet(self, rel_path: str) -> bytes:
        url = f"{self.cdn_base}/{rel_path}"
        resp = requests.get(url, timeout=60)
        resp.raise_for_status()
        return resp.content

    def _project_row(self, row: Dict[str, Any]) -> Dict[str, str]:
        # Keep only required fields; coerce to string to avoid schema issues
        return {k: str(row.get(k, "")) for k in self.columns}

    def _load_next_file(self) -> bool:
        if self._file_idx >= len(self.files):
            return False
        rel_path = self.files[self._file_idx]
        self._file_idx += 1
        data = self._download_parquet(rel_path)
        table = pq.read_table(io.BytesIO(data))
        self._rows = [self._project_row(r) for r in table.to_pylist()]
        self._row_idx = 0
        return True

    def __iter__(self) -> Iterator[Dict[str, str]]:
        self._file_idx = 0
        self._row_idx = 0
        self._rows = []
        return self

    def __next__(self) -> Dict[str, str]:
        while True:
            if self._row_idx < len(self._rows):
                item = self._rows[self._row_idx]
                self._row_idx += 1
                return item
            if not self._load_next_file():
                raise StopIteration

    def to_hf_dataset(self):
        """Optional: convert to HF Dataset for trainer compatibility (still zero API calls during data fetch)."""
        rows = list(self)
        return datasets.Dataset.from_list(rows)
```

Key improvements vs candidates:
- Simpler, correct file/row indexing (avoids off-by-one and buffer bugs).
- Projects at parse time → prevents `pyarrow.CastError` from mixed schemas.
- CDN-only → zero HF API calls during training, eliminating 429 risk.

## 3) Update training script to use manifest loader

File: `surrogate/train.py` (excerpt)

```python
from surrogate.train.cdn_manifest_dataset import CDNManifestDataset

def get_dataloader(manifest_path: str, batch_size: int):
    dataset = CDNManifestDataset(manifest_path=manifest_path, columns=["prompt", "response"])
    # Optional: dataset.to_hf_dataset() if trainer requires HF Dataset
    return torch.utils.data.DataLoader(
        dataset,
        batch_size=batch_size,
        collate_fn=lambda batch: {
            k: [item[k] for item in batch]
            for k in batch[0].keys()
        },
        num_workers=0,
    )
```

Usage:

```bash
# One-time (when rate-limit window is clear)
python scripts/build_hf_cdn_manifest.py --repo datasets/your-org/your-repo --date-folder batches/mirror-merged/2026-05-03 --out-file file_manifest.json

# Training (no HF API calls, no 429s)
python surrogate/train.py --manifest file_manifest.json --batch-size 8
```

## Summary of resolved contradictions
- **Manifest scope**: Use non-recursive, single-folder listing (safe, minimal API) — avoids huge paginated trees.
- **Schema handling**: Project to `{prompt, response}` at parse time — prevents `pyarrow.CastError`.
- **Network/auth**: CDN-only URLs, no Authorization
