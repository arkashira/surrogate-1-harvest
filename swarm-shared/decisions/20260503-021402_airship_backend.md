Below is the **single, merged implementation** that keeps every high-leverage idea from both proposals and removes duplication, ambiguity, and contradiction.  
It favors **correctness + concrete actionability** and can ship in ≤2 hours with no schema migrations or new infra.

---

## 1) Core idea (merged)
- **CDN-first ingestion**: one HF API list call → `file_list.json` → all training epochs use **CDN URLs only** (no auth, no 429).
- **Lightning Studio reuse + idle-stop guard**: detect stopped Studio, restart automatically, and pin deterministic repo/commits to spread commit-cap.
- **Arkship UI resilience**: expose `/training/status` and `/training/restart` so the UI can resume after idle-stop without losing state.

```python
#!/usr/bin/env python3
"""
CDN-first deterministic ingestion for Surrogate training.
- Single HF API list call (after rate-limit window).
- Saves file_list.json.
- Downloads via CDN (no auth) and projects to {prompt, response}.
"""
import json
import os
import sys
from datetime import datetime, timezone
from pathlib import Path
from typing import Dict, List

import requests
from huggingface_hub import HfApi

HF_REPO = os.getenv("HF_DATASET_REPO", "axentx/surrogate-mirror")
DATE_FOLDER = os.getenv("INGEST_DATE_FOLDER", datetime.now(timezone.utc).strftime("%Y-%m-%d"))
OUTPUT_DIR = Path(os.getenv("INGEST_OUTPUT_DIR", "data/ingested"))
FILE_LIST_PATH = OUTPUT_DIR / "file_list.json"
CDN_BASE = f"https://huggingface.co/datasets/{HF_REPO}/resolve/main"

api = HfApi()

def list_date_folder() -> List[str]:
    """Single API call: list files in date folder (non-recursive)."""
    try:
        tree = api.list_repo_tree(
            repo_id=HF_REPO,
            path=DATE_FOLDER,
            repo_type="dataset",
            recursive=False,
        )
        return [item.rfilename for item in tree if item.type == "file"]
    except Exception as exc:
        print(f"HF API list failed: {exc}", file=sys.stderr)
        if FILE_LIST_PATH.exists():
            print("Using cached file_list.json", file=sys.stderr)
            return json.loads(FILE_LIST_PATH.read_text())
        raise

def build_file_entries(files: List[str]) -> List[Dict]:
    entries = []
    for f in files:
        if not f.startswith(DATE_FOLDER):
            f = f"{DATE_FOLDER}/{f}"
        entries.append({
            "repo": HF_REPO,
            "path": f,
            "cdn_url": f"{CDN_BASE}/{f}",
            "local_path": str(OUTPUT_DIR / Path(f).name),
        })
    return entries

def download_cdn(entry: Dict) -> bool:
    out = Path(entry["local_path"])
    out.parent.mkdir(parents=True, exist_ok=True)
    if out.exists() and out.stat().st_size > 0:
        return True
    try:
        resp = requests.get(entry["cdn_url"], timeout=30)
        resp.raise_for_status()
        out.write_bytes(resp.content)
        return True
    except Exception as exc:
        print(f"CDN download failed {entry['cdn_url']}: {exc}", file=sys.stderr)
        return False

def project_to_pair(local_path: Path) -> Dict:
    suffix = local_path.suffix.lower()

    if suffix == ".jsonl":
        pairs = []
        for line in local_path.read_text().strip().splitlines():
            if not line.strip():
                continue
            obj = json.loads(line)
            pairs.append({
                "prompt": obj.get("prompt") or obj.get("input") or "",
                "response": obj.get("response") or obj.get("output") or "",
            })
        return {"pairs": pairs, "source_file": local_path.name}

    if suffix == ".json":
        obj = json.loads(local_path.read_text())
        if isinstance(obj, list):
            pairs = [{"prompt": x.get("prompt", ""), "response": x.get("response", "")} for x in obj]
        else:
            pairs = [{"prompt": obj.get("prompt", ""), "response": obj.get("response", "")}]
        return {"pairs": pairs, "source_file": local_path.name}

    try:
        import pyarrow.parquet as pq
        table = pq.read_table(local_path, columns=["prompt", "response"])
        df = table.to_pandas()
        pairs = [{"prompt": r.prompt, "response": r.response} for r in df.itertuples()]
        return {"pairs": pairs, "source_file": local_path.name}
    except Exception:
        return {"pairs": [], "source_file": local_path.name}

def main() -> None:
    OUTPUT_DIR.mkdir(parents=True, exist_ok=True)

    print("Listing date folder via HF API (single call)...")
    files = list_date_folder()
    print(f"Found {len(files)} files")

    entries = build_file_entries(files)
    FILE_LIST_PATH.write_text(json.dumps(entries, indent=2))
    print(f"Saved file list to {FILE_LIST_PATH}")

    ok = 0
    for i, entry in enumerate(entries):
        print(f"Downloading [{i+1}/{len(entries)}] {Path(entry['path']).name}...")
        if download_cdn(entry):
            ok += 1

    print(f"CDN downloads: {ok}/{len(entries)}")

    shard_dir = OUTPUT_DIR / "shards"
    shard_dir.mkdir(exist_ok=True)
    all_pairs = []
    for entry in entries:
        lp = Path(entry["local_path"])
        if not lp.exists():
            continue
        proj = project_to_pair(lp)
        all_pairs.extend(proj.get("pairs", []))

    shard_path = shard_dir / f"train_{DATE_FOLDER}.jsonl"
    with shard_path.open("w") as f:
        for p in all_pairs:
            f.write(json.dumps(p) + "\n")
    print(f"Training shard written: {shard_path} ({len(all_pairs)} samples)")

if __name__ == "__main__":
    main()
```

#!/usr/bin/env python3
"""
Lightning Studio reuse + idle-stop guard.
- Checks Studio status; restarts if stopped.
- Pins deterministic repo/commits to spread commit-cap.
"""
import os
import sys
import time
from pathlib import Path
from typing import Dict, List, Optional

import requests

STUDIO_API_BASE = os.getenv("LIGHTNING_STUDIO_API", "https://lightning.ai/api/v1/studio")
STUDIO_ID = os.getenv("LIGHTNING_STUDIO_ID")
API_TOKEN = os.getenv("LIGHTNING_API_TOKEN")
HEADERS = {"Authorization": f"Bearer {API_TOKEN}"}

# Deterministic repos/commits to spread commit-cap
DETERMINISTIC_REPOS: List[Dict[str, str]] = [
    {"repo": "axentx/surrogate-base", "commit": "main"},
    {"repo": "axentx/surrogate-mirror", "commit": "main"},
]

def studio_status() -> Optional[str]:
    if not STUDIO_ID:
        print("LIGHTNING_STUDIO_ID not set", file=sys.stderr)
        return None
    try:
        resp = requests.get(f"{STUDIO_API_BASE}/{STUDIO_ID}", headers=HEADERS, timeout=15)
        resp.raise_for_status()
        return resp.json().get("status")
    except Exception as exc:
        print(f"Failed to get Studio status: {exc}", file=sys.stderr)
        return None

def start_studio() -> bool:
    if not STUDIO_ID:
        return False
    try:
        resp = requests.post(
            f"{STUDIO_API_BASE}/{STUDIO_ID}/start",
            headers=HEADERS,
            timeout=30,
        )
        resp.raise_for_status()
        return True
    except Exception as exc:
        print(f"Failed to start Studio: {exc}", file=sys.stderr)
