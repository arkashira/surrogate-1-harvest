## Implementation Plan (≤2h)

**Goal:** Deterministic CDN file manifest + Lightning Studio lifecycle resilience for Surrogate-1 training.

**Why:** Eliminates HF API 429s during Surrogate training and prevents quota loss from idle-stop studio kills.

### 1) Pre-list date folder → JSON manifest (Mac orchestration)
- Single `list_repo_tree(recursive=False)` for one date folder (e.g. `batches/mirror-merged/2026-05-03/`).
- Save list to `manifests/2026-05-03_files.json` (array of `{"path": "...", "size": ...}`).
- Commit manifest to repo (or upload to surrogate config bucket) so training script never calls HF API again.

**Script** (`scripts/build_manifest.py`):
```python
#!/usr/bin/env python3
"""
Usage: python build_manifest.py <date> <hf_repo>
Example: python build_manifest.py 2026-05-03 axentx/surrogate-mirror
"""
import json, os, sys
from huggingface_hub import HfApi

def main():
    date = sys.argv[1]
    repo = sys.argv[2]
    folder = f"batches/mirror-merged/{date}/"
    api = HfApi()
    entries = api.list_repo_tree(repo=repo, path=folder, recursive=False)
    files = [{"path": e.path, "size": getattr(e, "size", None)} for e in entries if not e.path.endswith("/")]
    out = {"date": date, "repo": repo, "folder": folder, "files": files}
    os.makedirs("manifests", exist_ok=True)
    out_path = f"manifests/{date}_files.json"
    with open(out_path, "w") as f:
        json.dump(out, f, indent=2)
    print(f"Wrote {len(files)} files -> {out_path}")

if __name__ == "__main__":
    main()
```

Make executable and add to cron (after rate-limit window):
```bash
chmod +x scripts/build_manifest.py
# Example cron (run once per date folder after ingestion finishes)
# 0 3 * * * cd /opt/axentx/airship && /usr/bin/env bash -c 'python scripts/build_manifest.py $(date -d yesterday +%F) axentx/surrogate-mirror'
```

### 2) Lightning Studio lifecycle resilience
- Before each `.run()`, check studio status; if stopped, restart with target machine.
- Reuse running studio to save quota.

**Module** (`surrogate/training/studio_lifecycle.py`):
```python
#!/usr/bin/env python3
import time
from lightning_sdk import Studio, Teamspace, Machine
from typing import Optional

def get_or_create_studio(
    name: str,
    machine: Machine = Machine.L40S,
    idle_timeout_minutes: int = 120,
) -> Studio:
    """
    Reuse running studio or create one.
    If studio exists but is stopped, restart it.
    """
    teamspace = Teamspace()
    running = [s for s in teamspace.studios if s.name == name and s.status == "running"]
    if running:
        print(f"Reusing running studio: {name}")
        return running[0]

    existing = [s for s in teamspace.studios if s.name == name]
    if existing:
        studio = existing[0]
        if studio.status == "stopped":
            print(f"Restarting stopped studio: {name}")
            studio.start(machine=machine)
            _wait_for_running(studio)
        return studio

    print(f"Creating studio: {name}")
    studio = Studio(
        name=name,
        machine=machine,
        idle_timeout_minutes=idle_timeout_minutes,
        create_ok=True,
    )
    _wait_for_running(studio)
    return studio

def _wait_for_running(studio: Studio, timeout: int = 300, interval: int = 10):
    elapsed = 0
    while elapsed < timeout:
        studio.refresh()
        if studio.status == "running":
            print(f"Studio {studio.name} is running")
            return
        print(f"Waiting... status={studio.status}")
        time.sleep(interval)
        elapsed += interval
    raise TimeoutError(f"Studio {studio.name} did not start within {timeout}s")

def run_with_resilience(
    studio: Studio,
    script_path: str,
    args: Optional[list] = None,
    max_retries: int = 1,
):
    """
    Run script on studio; if studio stopped, restart and retry once.
    """
    args = args or []
    for attempt in range(max_retries + 1):
        try:
            if studio.status != "running":
                print(f"Studio not running (status={studio.status}), restarting...")
                studio.start(machine=studio.machine)
                _wait_for_running(studio)
            print(f"Running {script_path} on studio {studio.name}")
            return studio.run(script_path, arguments=args)
        except Exception as e:
            if attempt >= max_retries:
                raise
            print(f"Run failed (attempt {attempt+1}): {e}. Retrying...")
            time.sleep(10)
```

### 3) Training script: CDN-only data loading using manifest
- Load manifest JSON.
- Fetch files directly from CDN (`https://huggingface.co/datasets/{repo}/resolve/main/{path}`) with `requests`/`wget`/`aiohttp`.
- No `load_dataset` or HF API calls during training.

**Snippet** (`surrogate/training/train_cdn.py`):
```python
#!/usr/bin/env python3
import json, os, requests, pyarrow.parquet as pq
from io import BytesIO
from surrogate.training.studio_lifecycle import get_or_create_studio, run_with_resilience

HF_REPO = "axentx/surrogate-mirror"
MANIFEST = "manifests/2026-05-03_files.json"
CDN_ROOT = f"https://huggingface.co/datasets/{HF_REPO}/resolve/main"

def load_parquet_from_cdn(path: str):
    url = f"{CDN_ROOT}/{path}"
    resp = requests.get(url, timeout=30)
    resp.raise_for_status()
    return pq.read_table(BytesIO(resp.content)).to_pydict()

def build_dataset(manifest_path: str):
    with open(manifest_path) as f:
        manifest = json.load(f)
    rows = []
    for fobj in manifest["files"]:
        if not fobj["path"].endswith(".parquet"):
            continue
        data = load_parquet_from_cdn(fobj["path"])
        # Project to {prompt, response} only
        for prompt, response in zip(data.get("prompt", []), data.get("response", [])):
            if prompt and response:
                rows.append({"prompt": prompt, "response": response})
    return rows

if __name__ == "__main__":
    # This file is intended to be run inside Lightning Studio.
    # On Mac, use studio_lifecycle to launch it.
    dataset = build_dataset(MANIFEST)
    print(f"Loaded {len(dataset)} rows from CDN")
    # Continue training with dataset...
```

### 4) Mac orchestration launcher (optional convenience)
```bash
#!/usr/bin/env bash
# scripts/run_training.sh
set -euo pipefail
cd /opt/axentx/airship

DATE="${1:-$(date +%F)}"
MANIFEST="manifests/${DATE}_files.json"

# Build manifest if missing
if [[ ! -f "$MANIFEST" ]]; then
  python scripts/build_manifest.py "$DATE" axentx/surrogate-mirror
fi

# Launch/resume studio and run training
python -c "
from surrogate.training.studio_lifecycle import get_or_create_studio, run_with_resilience
from lightning_sdk import Machine
studio = get_or_create_studio('surrogate-train-l40s', machine=Machine.L40S)
run_with_resilience(studio, 'surrogate/training/train_cdn.py', args=['--manifest', '$MANIFEST'])
"
```

Make executable:
```bash
chmod +x scripts/run_training.sh
```
