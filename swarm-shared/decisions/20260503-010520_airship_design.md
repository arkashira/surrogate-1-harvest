### 1. Manifest Generator (run from Mac orchestration host)

**Location**: `surrogate/scripts/gen_manifest.py`  
Uses a single `list_repo_tree` call per date folder. Produces a deterministic manifest with integrity verification.

```python
# surrogate/scripts/gen_manifest.py
import hashlib
import json
import os
import sys
from datetime import datetime, timezone
from pathlib import Path

from huggingface_hub import HfApi

REPO_ID = "axentx/surrogate-datasets"
OUT_DIR = Path("surrogate/data/manifests")
CDN_TEMPLATE = "https://huggingface.co/datasets/{repo_id}/resolve/main/{path}"

def sha256_bytes(data: bytes) -> str:
    return hashlib.sha256(data).hexdigest()

def build_manifest(date_folder: str, output_path: Path | None = None) -> dict:
    api = HfApi()
    tree = api.list_repo_tree(repo_id=REPO_ID, path=date_folder, recursive=True)

    files = []
    for node in tree:
        if getattr(node, "type", None) != "file":
            continue
        path = getattr(node, "path", "")
        size = getattr(node, "size", None)
        # Build CDN URL for later fetch; store minimal metadata
        files.append({
            "path": path,
            "size": size,
            "url": CDN_TEMPLATE.format(repo_id=REPO_ID, path=path),
        })

    manifest = {
        "repo_id": REPO_ID,
        "date_folder": date_folder,
        "generated_at": datetime.now(timezone.utc).isoformat(),
        "files": files,
    }

    if output_path is None:
        out = OUT_DIR / date_folder.replace("/", "_") / "file_manifest.json"
    else:
        out = Path(output_path)
    out.parent.mkdir(parents=True, exist_ok=True)

    with out.open("w") as f:
        json.dump(manifest, f, indent=2)

    print(f"Manifest written: {out} ({len(files)} files)")
    return manifest

if __name__ == "__main__":
    if len(sys.argv) < 2:
        print("Usage: python gen_manifest.py <date_folder> [output_path]")
        sys.exit(1)
    build_manifest(sys.argv[1], Path(sys.argv[2]) if len(sys.argv) > 2 else None)
```

### 2. CDN-Only Dataset Loader (Surrogate training code)

**Location**: `surrogate/data/cdn_dataset.py`  
- Reads the manifest.  
- Streams each file via CDN (no auth).  
- Parses per-file and projects to `{prompt, response}`.  
- Skips corrupt/unprojectable files; never assumes schema.  
- Returns a `datasets.Dataset`.

```python
# surrogate/data/cdn_dataset.py
import io
import json
import logging
from pathlib import Path
from typing import Iterator

import pyarrow.parquet as pq
import requests
from datasets import Dataset

logger = logging.getLogger(__name__)

def _safe_parquet_read(content: bytes) -> dict | None:
    try:
        return pq.read_table(io.BytesIO(content)).to_pydict()
    except Exception as exc:
        logger.debug("Parquet read failed: %s", exc)
        return None

def _find_column(cols, hints):
    lowered = [c.lower() for c in cols]
    for h in hints:
        for idx, lc in enumerate(lowered):
            if h in lc:
                return cols[idx]
    return None

def _project_rowwise(table_dict: dict) -> Iterator[dict]:
    if not table_dict:
        return

    cols = list(table_dict.keys())
    n_rows = len(next(iter(table_dict.values()))) if table_dict else 0

    prompt_col = _find_column(cols, ["prompt"])
    response_col = _find_column(cols, ["response", "completion", "answer"])

    if prompt_col is None or response_col is None:
        logger.debug("Missing prompt/response columns among: %s", cols)
        return

    prompts = table_dict[prompt_col]
    responses = table_dict[response_col]
    for i in range(n_rows):
        p = prompts[i]
        r = responses[i]
        if isinstance(p, str) and isinstance(r, str) and p.strip() and r.strip():
            yield {"prompt": p.strip(), "response": r.strip()}

def load_one_file(url: str) -> list[dict]:
    try:
        resp = requests.get(url, timeout=30)
        resp.raise_for_status()
    except Exception as exc:
        logger.debug("CDN fetch failed %s: %s", url, exc)
        return []

    table_dict = _safe_parquet_read(resp.content)
    return list(_project_rowwise(table_dict))

def load_dataset_from_manifest(manifest_path: Path | str) -> Dataset:
    manifest_path = Path(manifest_path)
    with manifest_path.open() as f:
        manifest = json.load(f)

    records = []
    total_files = len(manifest.get("files", []))
    for idx, entry in enumerate(manifest.get("files", []), 1):
        url = entry.get("url") or CDN_TEMPLATE.format(repo_id=manifest["repo_id"], path=entry["path"])
        try:
            rows = load_one_file(url)
            records.extend(rows)
        except Exception as exc:
            logger.warning("Failed processing %s: %s", entry.get("path"), exc)
            continue

        if idx % 100 == 0 or idx == total_files:
            logger.info("Processed %d/%d files, %d records", idx, total_files, len(records))

    logger.info("Loaded %d records from %d files via CDN", len(records), total_files)
    return Dataset.from_list(records)
```

### 3. Reuse Running Studio + Launcher

**Location**: `surrogate/scripts/launch_training.py`  
- Detects and reuses a running Studio.  
- Restarts only if not running.  
- Passes manifest path via environment variable (clean, explicit).

```python
# surrogate/scripts/launch_training.py
import os
import sys
from pathlib import Path

from lightning import Lightning, Teamspace, Machine

def reuse_running_studio(name: str = "surrogate-l40s-training"):
    teamspace = Teamspace()
    for studio in teamspace.studios:
        if studio.name == name and studio.status == "Running":
            print(f"Reusing running studio: {studio.id}")
            return studio
    return None

def ensure_running_studio(name: str = "surrogate-l40s-training") -> object:
    studio = reuse_running_studio(name)
    if studio is not None:
        return studio

    print("No running studio found; creating new...")
    return Lightning.Studio(
        name=name,
        machine=Machine.L40S,
        create_ok=True,
        max_idle_time=3600,
    )

def main():
    if len(sys.argv) < 2:
        print("Usage: python launch_training.py <manifest_path>")
        sys.exit(1)

    manifest_path = Path(sys.argv[1])
    if not manifest_path.exists():
        print(f"Manifest not found: {manifest_path}")
        sys.exit(1)

    studio = ensure_running_studio()
    if studio.status != "Running":
        print("Starting studio...")
        studio.start(machine=Machine.L40S)

    env = {**os.environ, "SURROGATE_MANIFEST_PATH": str(manifest_path)}
    job = studio.run(
        "python -m surrogate.training.train --manifest ${SURROGATE_MANIFEST_PATH}",
        environment=env,
    )
    print(f"Training job submitted: {job}")

if __name__ == "__main__":
    main()
```
