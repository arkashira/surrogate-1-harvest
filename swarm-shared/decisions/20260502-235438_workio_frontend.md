# workio / frontend

## Final synthesized implementation (best of both candidates)

**Guiding principles**
- Correctness first: idempotency key must be stable for retries within a short window (prevents duplicates) while still allowing legitimate sequential punches.
- Concrete actionability: optimistic UI + rollback, clear pending/failed states, minimal scope (punch flow only), and a simple persistent queue for offline/retry.
- Resolve contradictions:
  - Idempotency key: Candidate 1’s stable time-bucket key is safer for preventing accidental duplicates; Candidate 2’s random UUID per attempt prevents retries from being deduped (which can allow duplicates if user retries after a real failure). Use stable key + short bucket, but include a monotonic attempt counter to allow safe retries within the same bucket.
  - Queue storage: Candidate 2’s localStorage queue is better than Candidate 1’s “optional” in-memory queue because it survives page reloads and offline periods. Use localStorage queue with exponential backoff and replay on online/visibilitychange.
  - Error handling: unify network vs server errors with actionable messages and rollback.

### `workio/src/lib/api/punch.ts`
```ts
// Punch client with idempotency, optimistic queue, and retry
const PENDING_KEY = 'workio:punch:pending';

export type PunchPending = {
  idempotencyKey: string;
  type: 'IN' | 'OUT';
  timestamp: string;
  location?: { lat: number; lng: number };
  retries: number;
  lastAttemptAt?: string;
};

// Stable idempotency key per punch "logical attempt" within a time window.
// Includes attempt counter so retries in same window don't collide with other attempts.
export function makePunchIdempotencyKey(
  tenantId: string,
  userId: string,
  type: 'IN' | 'OUT',
  attempt = 0
): string {
  const now = new Date();
  const bucket = now.toISOString().slice(0, 13); // e.g. 2026-05-03T14
  return `punch:${tenantId}:${userId}:${type}:${bucket}:a${attempt}`;
}

export interface PunchRequest {
  tenantId: string;
  userId: string;
  type: 'IN' | 'OUT';
  latitude?: number;
  longitude?: number;
  idempotencyKey: string;
  ts: string;
}

function getPending(): PunchPending[] {
  try {
    const raw = localStorage.getItem(PENDING_KEY);
    return raw ? JSON.parse(raw) : [];
  } catch {
    return [];
  }
}

function setPending(list: PunchPending[]) {
  localStorage.setItem(PENDING_KEY, JSON.stringify(list));
}

function addPending(p: PunchPending) {
  const list = getPending();
  // replace if same key exists
  const idx = list.findIndex((x) => x.idempotencyKey === p.idempotencyKey);
  if (idx >= 0) list[idx] = p;
  else list.push(p);
  setPending(list);
}

function removePending(key: string) {
  const list = getPending().filter((p) => p.idempotencyKey !== key);
  setPending(list);
}

function incrementRetries(key: string) {
  const list = getPending().map((p) =>
    p.idempotencyKey === key ? { ...p, retries: p.retries + 1, lastAttemptAt: new Date().toISOString() } : p
  );
  setPending(list);
}

// Exponential backoff (ms)
function backoff(retries: number): number {
  return Math.min(30_000, 1_000 * 2 ** retries + Math.random() * 500);
}

// Core punch call (used by UI and queue replay)
export async function sendPunch(req: PunchRequest): Promise<Response> {
  return fetch('/api/punch', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(req),
  });
}

// UI-facing punch: creates pending entry and attempts request
export async function punch(
  tenantId: string,
  userId: string,
  type: 'IN' | 'OUT',
  location?: { lat: number; lng: number }
): Promise<{ ok: boolean; pending: PunchPending }> {
  const pendingList = getPending();
  // Avoid creating multiple pending for same logical punch within same bucket+attempt=0
  const now = new Date();
  const bucket = now.toISOString().slice(0, 13);
  const existing = pendingList.find(
    (p) => p.type === type && p.idempotencyKey.includes(bucket) && p.idempotencyKey.includes(':a0')
  );

  const attempt = existing ? existing.retries : 0;
  const idempotencyKey = existing?.idempotencyKey || makePunchIdempotencyKey(tenantId, userId, type, attempt);

  const pending: PunchPending = {
    idempotencyKey,
    type,
    timestamp: now.toISOString(),
    location,
    retries: attempt,
  };

  addPending(pending);

  const payload: PunchRequest = {
    tenantId,
    userId,
    type,
    latitude: location?.lat,
    longitude: location?.lng,
    idempotencyKey,
    ts: pending.timestamp,
  };

  try {
    const res = await sendPunch(payload);
    if (!res.ok) throw new Error(`HTTP ${res.status}`);
    removePending(idempotencyKey);
    return { ok: true, pending };
  } catch (err) {
    incrementRetries(idempotencyKey);
    throw err;
  }
}

// Queue replay for pending items (call on startup/online/visibilitychange)
export async function replayPending(
  getCredentials: () => { tenantId: string; userId: string } | null,
  signal?: AbortSignal
) {
  const list = getPending();
  if (!list.length) return;

  const creds = getCredentials();
  if (!creds) return;

  for (const p of list) {
    if (signal?.aborted) break;
    // respect backoff
    const wait = backoff(p.retries);
    const sinceLast = p.lastAttemptAt ? Date.now() - new Date(p.lastAttemptAt).getTime() : Infinity;
    if (sinceLast < wait) continue;

    const payload: PunchRequest = {
      tenantId: creds.tenantId,
      userId: creds.userId,
      type: p.type,
      latitude: p.location?.lat,
      longitude: p.location?.lng,
      idempotencyKey: p.idempotencyKey,
      ts: p.timestamp,
    };

    try {
      const res = await sendPunch(payload);
      if (res.ok) {
        removePending(p.idempotencyKey);
      } else {
        incrementRetries(p.idempotencyKey);
      }
    } catch {
      incrementRetries(p.idempotencyKey);
    }
  }
}
```
