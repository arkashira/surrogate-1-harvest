## Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID` and `SHARD_TOTAL` (16) from the GitHub Actions matrix.
- Loads a pre-generated `manifest-YYYYMMDD.json` (created by a one-time Mac orchestration script) containing the list of files for a single date folder (avoids recursive `list_repo_files` and HF API rate limits).
- Uses **CDN-bypass URLs** (`https://huggingface.co/datasets/.../resolve/main/...`) for all file downloads — zero Authorization header, no `/api/` rate limits during training.
- Projects each file to `{prompt, response}` at parse time (avoids `load_dataset(streaming=True)` on mixed-schema repos and pyarrow CastError).
- Deduplicates via the existing `lib/dedup.py` central md5 store.
- Writes shard output to `batches/public-merged/<date>/shard<N>-<HHMMSS>.jsonl` (deterministic filename prevents commit collisions).
- Keeps the HF Space as the source-of-truth dedup cache; workers remain stateless per run (acceptable trade-off: occasional duplicate uploads, no cross-run state).

HF_REPO = "datasets/axentx/surrogate-1-training-pairs"
BASE_CDN = f"https://huggingface.co/{HF_REPO}/resolve/main"

# Retry policy for CDN downloads
MAX_RETRIES = 5
BACKOFF = 2.0  # seconds

def shard_filter(path: str, shard_id: int, shard_total: int) -> bool:
    """Deterministic 1/N shard assignment by path hash."""
    digest = hashlib.md5(path.encode()).hexdigest()
    bucket = int(digest, 16) % shard_total
    return bucket == shard_id

def download_cdn(path: str, dest: Path) -> bool:
    url = f"{BASE_CDN}/{path}"
    for attempt in range(1, MAX_RETRIES + 1):
        try:
            resp = requests.get(url, timeout=30, stream=True)
            resp.raise_for_status()
            dest.parent.mkdir(parents=True, exist_ok=True)
            with open(dest, "wb") as f:
                for chunk in resp.iter_content(chunk_size=8192):
                    f.write(chunk)
            return True
        except Exception as exc:
            if attempt == MAX_RETRIES:
                print(f"[ERROR] CDN download failed for {path}: {exc}", file=sys.stderr)
                return False
            sleep = BACKOFF * (2 ** (attempt - 1))
            print(f"[WARN] CDN retry {attempt}/{MAX_RETRIES} for {path} in {sleep}s: {exc}", file=sys.stderr)
            time.sleep(sleep)
    return False

def project_to_pair(local_path: Path) -> Optional[Dict[str, str]]:
    """
    Schema-agnostic projector: extract {prompt, response} from common formats.
    Supports:
      - JSON/JSONL with 'prompt'/'response' or 'instruction'/'output' keys
      - Parquet files (single or multiple row groups) with string columns
    """
    suffix = local_path.suffix.lower()
    try:
        if suffix == ".parquet":
            tbl = pq.read_table(local_path, columns=["prompt", "response"], use_threads=False)
            if tbl.num_rows == 0:
                return None
            # Take first row
            prompt = tbl.column("prompt")[0].as_py()
            response = tbl.column("response")[0].as_py()
            if prompt is None or response is None:
                return None
            return {"prompt": str(prompt), "response": str(response)}

        # JSON/JSONL
        if suffix == ".jsonl":
            with open(local_path, "r", encoding="utf-8") as f:
                line = f.readline().strip()
                if not line:
                    return None
                obj = json.loads(line)
        else:
            with open(local_path, "r", encoding="utf-8") as f:
                obj = json.load(f)

        # Normalize keys
        prompt = obj.get("prompt") or obj.get("instruction") or obj.get("input")
        response = obj.get("response") or obj.get("output") or obj.get("completion")
        if prompt is None or response is None:
            return None
        return {"prompt": str(prompt), "response": str(response)}
    except Exception as exc:
        print(f"[WARN] Failed to project {local_path}: {exc}", file=sys.stderr)
        return None

def build_record(pair: Dict[str, str], source_path: str) -> Dict[str, Any]:
    # Attribution in filename pattern only (no source/ts cols)
    ts = datetime.now(timezone.utc).strftime("%Y%m%d%H%M%S")
    content_hash = hashlib.md5(f"{pair['prompt']}\n{pair['response']}".encode()).hexdigest()
    return {
        "prompt": pair["prompt"],
        "response": pair["response"],
        "md5": content_hash,
        "source_path": source_path,
        "ts": ts,
    }
