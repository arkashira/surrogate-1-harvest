# vanguard / ops

## 1. Diagnosis

- No persisted file-list cache exists; every training/data-selection run performs authenticated `list_repo_tree` calls, burning HF API quota and causing 429s.
- Training scripts likely use `load_dataset(streaming=True)` or recursive listing on heterogeneous repos, risking `pyarrow` schema errors and repeated rate-limit hits.
- No CDN-only data path; training still relies on authenticated `/api/` endpoints instead of public `resolve/main/` URLs, missing the key bypass opportunity.
- No reuse guard for Lightning Studio; scripts likely create new studios instead of reusing running ones, wasting 80+ hours/month of quota.
- No idle-stop resilience; if a Lightning Studio stops, training dies instead of restarting on an available machine.

## 2. Proposed change

Create `/opt/axentx/vanguard/ops/persist_filelist.py` (single script) and wire it into the training launcher so that:
- One authenticated `list_repo_tree` call per `(repo, dateFolder)` produces `manifests/{repo}__{date}.json`.
- Training uses only CDN URLs from that manifest (zero auth calls during data load).
- Lightning Studio reuse + idle restart is handled before `.run()`.

Scope: new file + small edits to launcher/train entrypoint (if present).

```python
# /opt/axentx/vanguard/ops/persist_filelist.py
#!/usr/bin/env python3
"""
Generate and cache repo file-lists to avoid HF API 429s.
Usage:
  python persist_filelist.py \
    --repo datasets/mycorp/surrogate-mirror \
    --date 2026-04-29 \
    --out manifests/

Produces: manifests/datasets.mycorp.surrogate-mirror__2026-04-29.json
"""
import argparse
import json
import os
import sys
import time
from pathlib import Path
from typing import List

import requests

HF_API_BASE = "https://huggingface.co/api"
CDN_BASE = "https://huggingface.co/datasets"

def list_repo_tree(repo: str, path: str = "", token: str = None) -> List[dict]:
    """Single non-recursive tree call per folder (paginated)."""
    headers = {}
    if token:
        headers["Authorization"] = f"Bearer {token}"
    url = f"{HF_API_BASE}/datasets/{repo}/tree"
    params = {"path": path or "", "recursive": "false"}
    items = []
    while url:
        resp = requests.get(url, headers=headers, params=params, timeout=30)
        if resp.status_code == 429:
            retry_after = int(resp.headers.get("Retry-After", 360))
            print(f"Rate-limited. Waiting {retry_after}s", file=sys.stderr)
            time.sleep(retry_after)
            continue
        resp.raise_for_status()
        batch = resp.json()
        if not isinstance(batch, list):
            break
        items.extend(batch)
        # pagination: Link header
        link = resp.headers.get("Link", "")
        url = None
        for part in link.split(","):
            if 'rel="next"' in part:
                url = part.split(";")[0].strip(" <>")
                break
        params = {}
    return items

def build_manifest(repo: str, date_folder: str, token: str = None) -> List[dict]:
    """
    Returns list of CDN-ready file entries for one date folder.
    Example entry:
      {"path": "batches/mirror-merged/2026-04-29/shard-0001.parquet",
       "cdn_url": "https://huggingface.co/datasets/.../resolve/main/...",
       "size": 123456}
    """
    items = list_repo_tree(repo, path=date_folder, token=token)
    out = []
    for item in items:
        if item.get("type") != "file":
            continue
        rel_path = os.path.join(date_folder, item["path"])
        cdn_url = f"{CDN_BASE}/{repo}/resolve/main/{rel_path}"
        out.append({
            "path": rel_path,
            "cdn_url": cdn_url,
            "size": item.get("size"),
            "lfs": item.get("lfs", {}).get("oid") is not None
        })
    return out

def main() -> None:
    parser = argparse.ArgumentParser(description="Cache HF repo file-list to avoid 429s.")
    parser.add_argument("--repo", required=True, help="HF dataset repo (user/name)")
    parser.add_argument("--date", required=True, help="Date folder (YYYY-MM-DD)")
    parser.add_argument("--out", default="manifests", help="Output directory")
    parser.add_argument("--token", default=os.getenv("HF_TOKEN"), help="HF token (optional)")
    args = parser.parse_args()

    out_dir = Path(args.out)
    out_dir.mkdir(parents=True, exist_ok=True)

    safe_repo = args.repo.replace("/", "__")
    manifest_path = out_dir / f"{safe_repo}__{args.date}.json"

    if manifest_path.exists():
        print(f"Manifest exists: {manifest_path}", file=sys.stderr)
        sys.exit(0)

    print(f"Listing {args.repo}/{args.date} ...", file=sys.stderr)
    manifest = build_manifest(args.repo, args.date, token=args.token)

    manifest_path.write_text(json.dumps(manifest, indent=2))
    print(f"Wrote {len(manifest)} entries to {manifest_path}", file=sys.stderr)

if __name__ == "__main__":
    main()
```

```python
# /opt/axentx/vanguard/ops/reuse_studio.py  (or inline)
from lightning import Studio, Machine
import time

def get_or_start_studio(name: str, machine: Machine = Machine.L40S):
    for s in Studio.list():
        if s.name == name and s.status == "running":
            print(f"Reusing running studio: {name}")
            return s
    print(f"Creating studio: {name}")
    return Studio(name=name, machine=machine, create_ok=True)

def ensure_running(studio):
    if studio.status != "running":
        print(f"Studio stopped. Restarting on L40S...")
        studio.start(machine=Machine.L40S)
        # wait briefly
        for _ in range(10):
            studio.refresh()
            if studio.status == "running":
                break
            time.sleep(6)
```
