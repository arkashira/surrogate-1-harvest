# workio / quality

## Final Synthesized Solution

**Core problems (agreed by both candidates):**
1. No idempotency → duplicate attendance records from LINE retries/double-taps.
2. No `X-Line-Signature` verification → spoofable webhook.
3. Race conditions on rapid/ concurrent clock toggles → inconsistent state.
4. Missing graceful shutdown/timeout handling → partial writes and resource leaks under load.

**Chosen approach:**  
- Add **signature verification** at the edge.  
- Add **deterministic idempotency** keyed by tenant+employee+date+intent with TTL.  
- Enforce **serializable/atomic clock transition** at the database level (not just app-level checks).  
- Add **graceful shutdown + request timeout** to the webhook handler.  
- Keep the solution **minimal, auditable, and deployable** with zero downtime.

## 1. Database changes (single migration)

```sql
-- 1) Idempotency table (short TTL, fast PK)
CREATE TABLE IF NOT EXISTS line_webhook_idempotency (
  id            TEXT PRIMARY KEY,            -- deterministic key
  tenant_id     TEXT NOT NULL,
  employee_id   TEXT NOT NULL,
  date          DATE NOT NULL,
  intent        TEXT NOT NULL,               -- clock-in | clock-out | leave-request | ...
  created_at    TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Keep recent keys fast; drop older ones cheaply
CREATE INDEX IF NOT EXISTS ix_line_idemp_created
  ON line_webhook_idempotency (created_at)
  WHERE created_at > NOW() - INTERVAL '1 day';

-- Optional: daily cleanup via cron/pg_cron
-- DELETE FROM line_webhook_idempotency
-- WHERE created_at < NOW() - INTERVAL '1 day';

-- 2) Attendance table (if not already strongly constrained)
-- Ensure one clock-in per tenant+employee+date without a matching clock-out
-- This is the source-of-truth invariant.
CREATE UNIQUE INDEX IF NOT EXISTS ux_attendance_open_clock_in
  ON attendance_events (tenant_id, employee_id, date)
  WHERE event_type = 'clock-in';
```

## 2. Service: atomic transition + idempotency

```ts
// server/src/services/attendance.ts
import { db } from '../db';

export async function processLineAttendance(
  tenantId: string,
  employeeId: string,
  date: string,
  intent: 'clock-in' | 'clock-out',
  meta: Record<string, unknown>
): Promise<{ ok: true } | { ok: false; reason: string }> {
  const client = await db.connect();
  try {
    await client.query('BEGIN');

    // 1) Idempotency insert (deterministic key)
    const idempKey = `workio:line:${tenantId}:${employeeId}:${date}:${intent}`;
    const idemp = await client.query(
      `INSERT INTO line_webhook_idempotency (id, tenant_id, employee_id, date, intent)
       VALUES ($1, $2, $3, $4, $5)
       ON CONFLICT (id) DO NOTHING
       RETURNING id`,
      [idempKey, tenantId, employeeId, date, intent]
    );

    if (idemp.rowCount === 0) {
      await client.query('ROLLBACK');
      return { ok: false, reason: 'duplicate_within_ttl' };
    }

    // 2) Transition validation + insert in one serializable step
    //    invariant: at most one open clock-in per tenant+employee+date
    if (intent === 'clock-in') {
      const open = await client.query(
        `SELECT 1 FROM attendance_events
         WHERE tenant_id = $1 AND employee_id = $2 AND date = $3
           AND event_type = 'clock-in'
         LIMIT 1`,
        [tenantId, employeeId, date]
      );
      if (open.rowCount > 0) {
        await client.query('ROLLBACK');
        return { ok: false, reason: 'already_clocked_in' };
      }
    } else {
      // clock-out requires an open clock-in
      const open = await client.query(
        `SELECT 1 FROM attendance_events
         WHERE tenant_id = $1 AND employee_id = $2 AND date = $3
           AND event_type = 'clock-in'
         LIMIT 1`,
        [tenantId, employeeId, date]
      );
      if (open.rowCount === 0) {
        await client.query('ROLLBACK');
        return { ok: false, reason: 'no_open_clock_in' };
      }
    }

    await client.query(
      `INSERT INTO attendance_events (tenant_id, employee_id, date, event_type, recorded_at, meta)
       VALUES ($1, $2, $3, $4, NOW(), $5)`,
      [tenantId, employeeId, date, intent, JSON.stringify(meta)]
    );

    await client.query('COMMIT');
    return { ok: true };
  } catch (err) {
    await client.query('ROLLBACK').catch(() => {});
    // If unique violation on ux_attendance_open_clock_in, treat as conflict
    if ((err as any)?.code === '23505') {
      return { ok: false, reason: 'already_clocked_in' };
    }
    throw err;
  } finally {
    client.release();
  }
}
```

import crypto from 'crypto';

const LINE_CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';

function verifySignature(body: string, signature: string): boolean {
  if (!LINE_CHANNEL_SECRET) return false;
  const expected = crypto
    .createHmac('sha256', LINE_CHANNEL_SECRET)
    .update(body, 'utf8')
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

function idempotencyKey(tenantId: string, employeeId: string, date: string, intent: string): string {
  return `workio:line:${tenantId}:${employeeId}:${date}:${intent}`;
}

async function parseLineEvent(
  ev: any
): Promise<{ employeeId: string; date: string; intent: 'clock-in' | 'clock-out' } | null> {
  // Map LINE userId -> employee (cached or DB). Simplified here.
  const employee = await global.db.query(
    `SELECT id, tenant_id FROM employees WHERE line_user_id = $1`,
    [ev.source?.userId]
  );
  if (employee.rowCount === 0) return null;

  const emp = employee.rows[0];
  const today = new Date().toISOString().slice(0, 10);

  const text = ev.postback?.data || ev.message?.text || '';
  let intent: 'clock-in' | 'clock-out' | null = null;
  if (/\bclock-?in\b/i.test(text)) intent = 'clock-in';
  else if (/\bclock-?out\b/i.test(text)) intent = 'clock-out';
  if (!intent) return null;

  return { employeeId: emp.id, date: today, intent };
}
