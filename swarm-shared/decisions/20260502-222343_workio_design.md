# workio / design

## 1. Diagnosis
- **Tenant cache missing on `/webhook/line`**: every webhook resolves `channelId → tenant` via DB; LINE retries/backoff amplify DB bursts under traffic spikes.
- **No correlation/trace IDs**: requests across webhook → service → DB are untraceable; incidents require slow log greps.
- **Signature verification depth weak**: single-point check without defense-in-depth (replay, clock skew, body integrity).
- **No structured logging**: ad-hoc `console.log` makes metrics/alerting and debugging unreliable in k8s/PM2.
- **Health/readiness endpoints absent or naive**: no DB/upstream checks; k8s/PM2 can’t detect partial failure or startup race.

## 2. Proposed change
File: `/opt/axentx/workio/server/src/middleware/lineWebhook.ts` (new) + `/opt/axentx/workio/server/src/routes/webhook/line.ts` (refactor).  
Scope: add tenant LRU cache, correlation IDs, stricter signature verification, structured logging, and wire into health/readiness.

`server/src/middleware/lineWebhook.ts`
```ts
import crypto from 'crypto';
import { LRUCache } from 'lru-cache';
import { v4 as uuidv4 } from 'uuid';
import pino from 'pino';
import type { Request, Response, NextFunction } from 'express';
import type { Tenant } from '../models/index.js';

const logger = pino({
  name: 'workio-line-webhook',
  level: process.env.LOG_LEVEL || 'info',
  redact: ['req.headers.authorization', 'req.body.secret'],
  base: { pid: process.pid },
});

// 5-min TTL, 100 tenants max (~few KB)
const tenantCache = new LRUCache<string, { tenant: Tenant; ts: number }>({
  max: 100,
  ttl: 1000 * 60 * 5,
});

const CLOCK_SKEW_MS = 5 * 60 * 1000; // 5m
const MAX_BODY_SIZE = 2 * 1024 * 1024; // 2MB

export function lineWebhookMiddleware(
  getTenantByChannelId: (channelId: string) => Promise<Tenant | null>
) {
  return async (req: Request, res: Response, next: NextFunction) => {
    const traceId = uuidv4();
    const start = Date.now();
    const childLogger = logger.child({ traceId, method: req.method, url: req.url });

    // Basic sanity
    if (req.method !== 'POST') {
      childLogger.warn({ status: 405 }, 'invalid method');
      return res.status(405).json({ error: 'Method Not Allowed' });
    }

    // Body already parsed by express.json() upstream; ensure raw body was captured for signature
    // (see server/src/index.ts for rawBody capture)
    const rawBody = (req as any).rawBody as Buffer;
    if (!rawBody || rawBody.length === 0) {
      childLogger.warn('missing rawBody for signature');
      return res.status(400).json({ error: 'Invalid request' });
    }

    // Signature verification (defense-in-depth)
    const signature = req.headers['x-line-signature'] as string | undefined;
    const secret = process.env.LINE_CHANNEL_SECRET;
    if (!signature || !secret) {
      childLogger.warn('missing signature or secret');
      return res.status(401).json({ error: 'Unauthorized' });
    }

    if (!verifyLineSignature(rawBody, signature, secret)) {
      childLogger.warn({ signature }, 'invalid signature');
      return res.status(401).json({ error: 'Unauthorized' });
    }

    // Replay/clock skew protection
    const body = req.body;
    const timestamp = body.timestamp as number | undefined;
    if (timestamp && Math.abs(Date.now() - timestamp) > CLOCK_SKEW_MS) {
      childLogger.warn({ timestamp, now: Date.now() }, 'replay or clock skew detected');
      return res.status(400).json({ error: 'Bad Request' });
    }

    // Tenant resolution with cache
    const channelId = body.destination as string | undefined; // LINE destination = channelId
    if (!channelId) {
      childLogger.warn('missing destination');
      return res.status(400).json({ error: 'Bad Request' });
    }

    let tenant: Tenant | null = null;
    const cached = tenantCache.get(channelId);
    if (cached) {
      tenant = cached.tenant;
      childLogger.debug({ tenantId: tenant.id, cached: true }, 'tenant cache hit');
    } else {
      tenant = await getTenantByChannelId(channelId);
      if (!tenant) {
        childLogger.warn({ channelId }, 'tenant not found');
        return res.status(404).json({ error: 'Tenant Not Found' });
      }
      tenantCache.set(channelId, { tenant, ts: Date.now() });
      childLogger.debug({ tenantId: tenant.id, cached: false }, 'tenant cached');
    }

    // Attach to request
    (req as any).tenant = tenant;
    (req as any).traceId = traceId;
    childLogger.info(
      { tenantId: tenant.id, channelId, durationMs: Date.now() - start },
      'webhook authorized'
    );
    next();
  };
}

function verifyLineSignature(rawBody: Buffer, signature: string, secret: string): boolean {
  const expected = crypto
    .createHmac('sha256', secret)
    .update(rawBody)
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}
```

`server/src/services/tenantService.ts` (add)
```ts
import db from '../db/index.js';
import type { Tenant } from '../models/index.js';

export async function getTenantByChannelId(channelId: string): Promise<Tenant | null> {
  // Minimal projection; index on line_channel_id recommended
  const row = await db.oneOrNone<Tenant>(
    'SELECT id, name, line_channel_id, line_access_token, settings FROM tenants WHERE line_channel_id = $1 AND active = true',
    [channelId]
  );
  return row;
}
```
