## 1. CDN-bypass File List (Mac orchestration + training)

Use a single HF API call to list files for a date folder, save JSON, and have training consume only CDN URLs.

```ts
// src/ops/cdnFileList.ts
import { HuggingFaceHubAPI } from './hfApi';

export interface RepoFile {
  path: string;
  type: 'file' | 'directory';
  size?: number;
}

export async function listDateFolderCDN(
  repo: string,
  dateFolder: string,
  token?: string
): Promise<RepoFile[]> {
  const tree = await HuggingFaceHubAPI.listRepoTree(repo, {
    path: dateFolder,
    recursive: false,
    token,
  });
  return tree.filter((n) => n.type === 'file');
}

export function buildCDNUrl(repo: string, filePath: string): string {
  return `https://huggingface.co/datasets/${repo}/resolve/main/${filePath}`;
}
```

```ts
// src/hfApi.ts
import axios from 'axios';

export class HuggingFaceHubAPI {
  static async listRepoTree(
    repo: string,
    opts: { path?: string; recursive?: boolean; token?: string } = {}
  ): Promise<any[]> {
    const url = `https://huggingface.co/api/datasets/${repo}/tree`;
    const params: any = { recursive: opts.recursive ? 1 : 0 };
    if (opts.path) params.path = opts.path;
    const headers = opts.token ? { Authorization: `Bearer ${opts.token}` } : {};
    const res = await axios.get(url, { params, headers });
    return res.data;
  }
}
```

## 2. Lightning Studio Reuse

Reuse or restart a named studio instead of recreating it.

```ts
// src/ops/useLightningStudio.ts
import { Lightning, Teamspace, Studio, Machine } from 'lightning-sdk';

export async function getOrCreateStudio(
  name: string,
  machine: Machine = Machine.L40S,
  createOk = false
): Promise<Studio> {
  const teamspace = await Teamspace.current();
  const running = teamspace.studios.find(
    (s) => s.name === name && s.status === 'Running'
  );
  if (running) return running;

  const stopped = teamspace.studios.find((s) => s.name === name && s.status === 'Stopped');
  if (stopped) {
    await stopped.start({ machine });
    return stopped;
  }

  if (!createOk) {
    throw new Error(`Studio ${name} not running and createOk=false`);
  }
  return Lightning.Studio({ name, createOk: true, machine });
}
```

## 3. Ingestion Ops State + Lightweight Dashboard

Persist ingestion run metadata and surface health.

```ts
// src/ops/ingestState.ts
import { atom, useAtom } from 'jotai';
import { useEffect } from 'react';

export type IngestRun = {
  id: string;
  repo: string;
  dateWindow: string;
  status: 'pending' | 'running' | 'success' | 'failed';
  startedAt: string;
  finishedAt?: string;
  cdnBypassFileList?: string;
  error?: string;
};

const runsAtom = atom<IngestRun[]>([]);

export function useIngestRuns() {
  const [runs, setRuns] = useAtom(runsAtom);

  useEffect(() => {
    // hydrate from localStorage (or server)
    try {
      const stored = localStorage.getItem('ingestRuns');
      if (stored) setRuns(JSON.parse(stored));
    } catch {}
  }, [setRuns]);

  const save = (next: IngestRun[]) => {
    setRuns(next);
    try {
      localStorage.setItem('ingestRuns', JSON.stringify(next));
    } catch {}
  };

  const add = (run: IngestRun) => save([run, ...runs]);
  const update = (id: string, patch: Partial<IngestRun>) =>
    save(runs.map((r) => (r.id === id ? { ...r, ...patch } : r)));

  return { runs, add, update };
}
```

```tsx
}
```
