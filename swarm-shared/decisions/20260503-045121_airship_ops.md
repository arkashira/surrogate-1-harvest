### 1. Manifest Generator (run once per date folder)  
**Path:** `scripts/generate_manifest.py`  

```python
#!/usr/bin/env python3
"""
Generate CDN-only manifest for Surrogate-1 training.
Run once per date folder after HF ingestion completes.
Bypasses HF API auth rate limits entirely.
"""
import json
import os
import sys
from datetime import datetime
from huggingface_hub import HfApi

HF_REPO = "axentx/surrogate-1-mirror"
DATE_FOLDER = os.getenv("DATE_FOLDER")  # e.g., "2026-04-29"

def main():
    date_folder = DATE_FOLDER or datetime.utcnow().strftime("%Y-%m-%d")
    api = HfApi()

    # Single API call: list one folder (non-recursive)
    files = api.list_repo_tree(
        repo_id=HF_REPO,
        path=f"batches/mirror-merged/{date_folder}",
        recursive=False,
    )

    # Filter to parquet only
    parquet_files = [f.rfilename for f in files if f.rfilename.endswith(".parquet")]

    manifest = {
        "date": date_folder,
        "repo": HF_REPO,
        "created_at": datetime.utcnow().isoformat(),
        "files": sorted(parquet_files),
        "cdn_base": f"https://huggingface.co/datasets/{HF_REPO}/resolve/main",
    }

    out_dir = f"batches/mirror-merged/{date_folder}"
    os.makedirs(out_dir, exist_ok=True)
    out_path = f"{out_dir}/manifest.json"

    with open(out_path, "w") as f:
        json.dump(manifest, f, indent=2)

    print(f"✅ Manifest written: {out_path}")
    print(f"   {len(parquet_files)} files, CDN-only training ready")

if __name__ == "__main__":
    main()
```

### 2. CDN-Only Training Script  
**Path:** `surrogate/train_cdn.py`

```python
#!/usr/bin/env python3
"""
Surrogate-1 training with CDN-only fetches.
Zero HF API calls during data loading.
"""
import json
import os
import sys
from pathlib import Path

import pyarrow.parquet as pq
import requests
import torch
from torch.utils.data import Dataset, DataLoader
from tqdm import tqdm

# Surrogate-1 model imports (adjust to your actual model code)
from src.model import SurrogateModel
from src.config import TrainingConfig

HF_REPO = "axentx/surrogate-1-mirror"
MANIFEST_PATH = os.getenv("MANIFEST_PATH", "batches/mirror-merged/latest/manifest.json")
CACHE_DIR = Path(os.getenv("CACHE_DIR", "/tmp/surrogate1_cdn_cache"))
CACHE_DIR.mkdir(parents=True, exist_ok=True)


class CDNParquetDataset(Dataset):
    def __init__(self, manifest_path: str):
        with open(manifest_path) as f:
            self.manifest = json.load(f)

        self.cdn_base = self.manifest["cdn_base"]
        self.files = self.manifest["files"]
        self.local_paths = []

        print(f"📦 CDN dataset: {len(self.files)} files")
        self._download_all()

    def _download_all(self):
        """Download all parquet files via CDN (no auth, no API rate limits)."""
        for rel_path in tqdm(self.files, desc="Downloading via CDN"):
            url = f"{self.cdn_base}/{rel_path}"
            local_file = CACHE_DIR / Path(rel_path).name

            if local_file.exists():
                self.local_paths.append(local_file)
                continue

            resp = requests.get(url, timeout=30)
            resp.raise_for_status()

            local_file.write_bytes(resp.content)
            self.local_paths.append(local_file)

    def __len__(self):
        return len(self.local_paths)

    def __getitem__(self, idx):
        table = pq.read_table(self.local_paths[idx])
        # Project to {prompt, response} only at parse time
        df = table.select(["prompt", "response"]).to_pandas()

        # Simple tokenization placeholder — replace with your tokenizer
        prompt = df.iloc[0]["prompt"]
        response = df.iloc[0]["response"]

        return {
            "input_ids": torch.tensor([1, 2, 3]),  # placeholder
            "labels": torch.tensor([1, 2, 3]),      # placeholder
        }


def train():
    print("🚀 Starting Surrogate-1 CDN training")

    dataset = CDNParquetDataset(MANIFEST_PATH)
    dataloader = DataLoader(dataset, batch_size=4, shuffle=True)

    model = SurrogateModel.from_pretrained(TrainingConfig.base_model)
    optimizer = torch.optim.AdamW(model.parameters(), lr=5e-5)

    device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
    model.to(device)

    model.train()
    for epoch in range(TrainingConfig.epochs):
        total_loss = 0
        for batch in tqdm(dataloader, desc=f"Epoch {epoch+1}"):
            batch = {k: v.to(device) for k, v in batch.items()}

            outputs = model(**batch)
            loss = outputs.loss

            optimizer.zero_grad()
            loss.backward()
            optimizer.step()

            total_loss += loss.item()

        print(f"Epoch {epoch+1} loss: {total_loss / len(dataloader):.4f}")

    # Save locally (Lightning studio will sync to HF via git)
    out_path = Path("surrogate_model_finetuned")
    model.save_pretrained(out_path)
    print(f"✅ Model saved to {out_path}")


if __name__ == "__main__":
    train()
```
