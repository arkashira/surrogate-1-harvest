# vanguard / ops

## 1. Diagnosis
- No persisted `(repo, dateFolder) → file-list` manifest: every training/data-selection run triggers authenticated `list_repo_tree`, burning HF API quota and risking 429s.
- Training/data loader likely uses `load_dataset(streaming=True)` or repeated per-file loads on heterogeneous repos, causing `pyarrow.CastError` from mixed schemas.
- No CDN bypass strategy: authenticated API calls used for file fetches during training instead of public CDN URLs, increasing rate-limit exposure.
- No studio reuse guard: training jobs may recreate Lightning Studio instead of reusing running instances, wasting 80hr/mo quota.
- No idle-stop resilience: Lightning Studio idle timeout kills long-running training; no pre-run status check + restart.

## 2. Proposed change
Create `/opt/axentx/vanguard/ops/file_manifest.py` (single utility) and wire it into the training orchestration path:
- Add `build_manifest(repo, date_folder)` → writes `manifests/{repo}/{date_folder}.json` with CDN URLs.
- Add `load_manifest(repo, date_folder)` → reads cached manifest; if missing/old, single authenticated list call then cache.
- Add `get_cdn_url(repo, path)` → returns `https://huggingface.co/datasets/{repo}/resolve/main/{path}`.
- Add lightweight CLI: `python -m vanguard.ops.file_manifest build <repo> <date_folder>` and `python -m vanguard.ops.file_manifest use <repo> <date_folder>`.
- Integrate into training launcher: consume manifest JSON and fetch via CDN (zero API calls during data load).

## 3. Implementation

```bash
# /opt/axentx/vanguard/ops/file_manifest.py
#!/usr/bin/env python3
"""
Persist (repo, date_folder) -> file-list manifest and provide CDN URLs.
Avoids repeated HF API list_repo_tree and enables CDN-only training fetches.
"""
from __future__ import annotations

import json
import os
import sys
from datetime import datetime, timezone
from pathlib import Path
from typing import Dict, List, Optional

import requests

# Optional: lazy import HF; keep CLI usable without HF auth for 'use' mode
try:
    from hugginggingface import HfApi  # type: ignore
except Exception:  # noqa: BLE001
    HfApi = None

BASE_DIR = Path(__file__).resolve().parents[2]  # /opt/axentx/vanguard
MANIFESTS_DIR = BASE_DIR / "manifests"
MANIFESTS_DIR.mkdir(parents=True, exist_ok=True)


def _manifest_path(repo: str, date_folder: str) -> Path:
    safe_repo = repo.replace("/", "_")
    return MANIFESTS_DIR / f"{safe_repo}_{date_folder}.json"


def get_cdn_url(repo: str, path: str) -> str:
    return f"https://huggingface.co/datasets/{repo}/resolve/main/{path}"


def build_manifest(
    repo: str,
    date_folder: str,
    *,
    hf_token: Optional[str] = None,
    recursive: bool = False,
) -> List[Dict[str, str]]:
    """
    Single authenticated list_repo_tree call for one date folder (non-recursive),
    persist manifest with CDN URLs + metadata.
    Returns list of entries: {"path": ..., "cdn_url": ..., "size": ..., "type": ...}
    """
    if HfApi is None:
        raise RuntimeError("huggingface_hub not available; install to build manifests.")

    api = HfApi(token=hf_token)
    # list_repo_tree per folder (non-recursive) to avoid 100x pagination on big repos
    tree = api.list_repo_tree(repo=repo, path=date_folder, recursive=recursive)

    entries: List[Dict[str, str]] = []
    for item in tree:
        if item.type != "file":
            continue
        cdn = get_cdn_url(repo, item.path)
        entries.append(
            {
                "path": item.path,
                "cdn_url": cdn,
                "size": str(item.size or 0),
                "type": item.type,
            }
        )

    manifest = {
        "repo": repo,
        "date_folder": date_folder,
        "generated_at_utc": datetime.now(timezone.utc).isoformat(),
        "recursive": recursive,
        "count": len(entries),
        "entries": entries,
    }

    out_path = _manifest_path(repo, date_folder)
    out_path.parent.mkdir(parents=True, exist_ok=True)
    with open(out_path, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2)
    return entries


def load_manifest(repo: str, date_folder: str) -> Optional[Dict]:
    p = _manifest_path(repo, date_folder)
    if not p.is_file():
        return None
    with open(p, encoding="utf-8") as f:
        return json.load(f)


def cli_build(repo: str, date_folder: str) -> None:
    entries = build_manifest(repo, date_folder, hf_token=os.getenv("HF_TOKEN"))
    print(f"Wrote {len(entries)} entries for {repo}/{date_folder}")


def cli_use(repo: str, date_folder: str) -> None:
    manifest = load_manifest(repo, date_folder)
    if manifest is None:
        print(f"No manifest for {repo}/{date_folder}. Build first.", file=sys.stderr)
        sys.exit(1)
    print(json.dumps(manifest, indent=2))


def main() -> None:
    if len(sys.argv) < 4:
        print("Usage: file_manifest.py build|use <repo> <date_folder>", file=sys.stderr)
        sys.exit(1)
    cmd, repo, date_folder = sys.argv[1], sys.argv[2], sys.argv[3]
    if cmd == "build":
        cli_build(repo, date_folder)
    elif cmd == "use":
        cli_use(repo, date_folder)
    else:
        print(f"Unknown command: {cmd}", file=sys.stderr)
        sys.exit(1)


if __name__ == "__main__":
    main()
```

```python
# In your train.py or data loader
import json
from pathlib import Path
import requests

def load_cdn_file_list(repo: str, date_folder: str):
    manifest_path = Path(__file__).parent.parent / "manifests" / f"{repo.replace('/', '_')}_{date_folder}.json"
    if not manifest_path.is_file():
        raise FileNotFoundError(f"Manifest missing: {manifest_path}")
    manifest = json.loads(manifest_path.read_text())
    return [e["cdn_url"] for e in manifest["entries"]]

# Use CDN URLs directly with requests/streaming download (zero HF API calls)
def stream_cdn_files(urls):
    for url in urls:
        with requests.get(url, stream=True, timeout=60) as r:
            r.raise_for_status()
            yield r.content  # or parse per your surrogate-1 projection logic
```
