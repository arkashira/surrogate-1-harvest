# Costinel / design

## Final Implementation Plan — Top-Hub Signal Panel (Costinel Dashboard)

**Scope & Value**  
- Frontend-only, read-only React panel that surfaces the highest-signal / most-connected hub (default “MOC”) and its actionable proposals from the knowledge graph.  
- CDN-first data: a single JSON file (`/data/hubs/moc/signals.json`) embedded at build time and refreshed nightly by CI; runtime fetches are cache-first with SWR.  
- Ships in <2h: one new component, one route, one JSON asset, light styling using existing design tokens, zero backend changes.

### 2) Types (`src/types/knowledgeGraph.ts`)

```ts
export interface Signal {
  id: string;
  title: string;
  summary: string;
  impact: "high" | "medium" | "low";
  category: "cost" | "governance" | "security" | "efficiency";
  actionUrl?: string;
  createdAt: string; // ISO
  updatedAt: string;
  tags: string[];
}

export interface HubSignals {
  hub: string;
  displayName: string;
  description: string;
  lastUpdated: string;
  signals: Signal[];
}

export interface HubMetrics {
  connectedHubs: number;
  activeSignals: number;
  openProposals: number;
  compositeScore: number;
}

export interface HubData extends HubSignals {
  rank: number;
  metrics: HubMetrics;
}
```

### 4) Component (`src/components/TopHubSignalPanel/TopHubSignalPanel.tsx`)

```tsx
import React from "react";
import useSWR from "swr";
import { HubData } from "../../types/knowledgeGraph";
import styles from "./TopHubSignalPanel.module.css";

const fetcher = (url: string) => fetch(url).then((r) => r.json());

const impactColor = (impact: string) => {
  switch (impact) {
    case "high":
      return "var(--impact-high, #dc2626)";
    case "medium":
      return "var(--impact-medium, #f59e0b)";
    default:
      return "var(--impact-low, #10b981)";
  }
};

const TopHubSignalPanel: React.FC = () => {
  const { data, error } = useSWR<HubData>("/data/hubs/moc/signals.json", fetcher, {
    revalidateOnFocus: false,
    dedupingInterval: 5 * 60 * 1000,
  });

  if (error)
    return (
      <div className={styles.panelError} role="status">
        Unable to load hub signals.
      </div>
    );
  if (!data)
    return (
      <div className={styles.panelLoading} aria-live="polite">
        Loading top hub...
      </div>
    );

  const { displayName, description, compositeScore, metrics, signals, lastUpdated } = data;

  return (
    <section className={styles.panel} aria-label={`Top hub: ${displayName}`}>
      <header className={styles.header}>
        <div>
          <h2 className={styles.title}>Top Hub</h2>
          <p className={styles.hubLabel}>{displayName}</p>
        </div>
        <div className={styles.scoreBadge} title="Composite signal score">
          {compositeScore.toFixed(1)}
        </div>
      </header>

      <p className={styles.summary}>{description}</p>

      <ul className={styles.metaRow} aria-label="Hub metrics">
        <li>Connected hubs: {metrics.connectedHubs}</li>
        <li>Active signals: {metrics.activeSignals}</li>
        <li>Open proposals: {metrics.openProposals}</li>
      </ul>

      <div className={styles.proposals}>
        <h3 className={styles.subtitle}>Actionable Signals</h3>
        {signals.length === 0 ? (
          <p className={styles.empty}>No open signals.</p>
        ) : (
          <ul className={styles.proposalList}>
            {signals.map((s) => (
              <li key={s.id} className={styles.proposalItem}>
                <div className={styles.proposalHeader}>
                  <strong className={styles.proposalTitle}>{s.title}</strong>
                  <span className={styles.proposalImpact} style={{ color: impactColor(s.impact) }}>
                    {s.impact.toUpperCase()}
                  </span>
                </div>
                <p className={styles.proposalSummary}>{s.summary}</p>
                <div className={styles.proposalMeta}>
                  <div className={styles.tags}>
                    {s.tags.map((t) => (
                      <span key={t} className={styles.tag}>
                        {t}
                      </span>
                    ))}
                  </div>
                  {s.actionUrl && (
                    <a
                      href={s.actionUrl}
                      target="_blank"
                      rel="noopener noreferrer"
                      className={styles.actionLink}
                    >
                      Take action
                    </a>
                  )}
                </div>
                <time className={styles.updated} dateTime={s.updatedAt}>
