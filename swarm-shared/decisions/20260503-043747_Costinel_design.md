## Goal
Ship in **<2h** a resilient, CDN-first **Top-Hub signal panel** for Costinel that:
- Shows the most-connected hub (e.g., “MOC”) and a short insight.
- Lists up to 4 related docs with links to the CDN.
- Performs **zero runtime HF API calls** (strict Sense + Signal, no automated actions).
- Handles loading, error, retry, and empty states.

## 1) CDN fetcher — `src/lib/cdn.ts`

```ts
// src/lib/cdn.ts
const CDN_ROOT = 'https://huggingface.co/datasets/axentx/costinel-knowledge/resolve/main';

export interface TopHubPayload {
  hub: string;
  insight: string;
  docs: Array<{
    title: string;
    path: string; // relative to CDN_ROOT/docs/
    score?: number;
  }>;
  generated_at: string; // ISO
}

export async function fetchTopHubSignal(): Promise<TopHubPayload> {
  // Stable filename updated by ops pipeline.
  const url = `${CDN_ROOT}/top-hub/latest.json`;
  const res = await fetch(url, { cache: 'no-store' });
  if (!res.ok) {
    throw new Error(`CDN fetch failed: ${res.status} ${res.statusText}`);
  }
  return res.json() as Promise<TopHubPayload>;
}
```

**Why this wins**:  
- Uses native `fetch` (no extra axios dep).  
- Explicit `cache: 'no-store'` avoids stale CDN reads.  
- Clear interface (`TopHubPayload`) for type safety.

## 2) Hook — `src/hooks/useTopHubSignal.ts`

```ts
// src/hooks/useTopHubSignal.ts
import { useEffect, useState, useCallback } from 'react';
import { fetchTopHubSignal, TopHubPayload } from '../lib/cdn';

export function useTopHubSignal() {
  const [data, setData] = useState<TopHubPayload | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  const load = useCallback(async () => {
    setLoading(true);
    setError(null);
    try {
      const payload = await fetchTopHubSignal();
      setData(payload);
    } catch (err: any) {
      setError(err.message || 'Failed to load top-hub signal');
    } finally {
      setLoading(false);
    }
  }, []);

  useEffect(() => {
    load();
  }, [load]);

  return {
    hub: data?.hub ?? null,
    insight: data?.insight ?? null,
    docs: data?.docs ?? [],
    generatedAt: data?.generated_at ?? null,
    loading,
    error,
    refresh: load,
  };
}
```

**Why this wins**:  
- Exposes `refresh` for manual retry.  
- Separates `loading`/`error` states for precise UI control.  
- Memoized `load` prevents infinite loops.

## 3) Component — `src/components/TopHubSignalPanel.tsx`

```tsx
// src/components/TopHubSignalPanel.tsx
import { useTopHubSignal } from '../hooks/useTopHubSignal';

export function TopHubSignalPanel() {
  const { hub, insight, docs, loading, error, refresh } = useTopHubSignal();

  if (loading) {
    return (
      <div className="rounded-lg border border-gray-200 bg-white p-4 shadow-sm">
        <p className="text-sm text-gray-500">Loading top-hub signal…</p>
      </div>
    );
  }

  if (error) {
    return (
      <div className="rounded-lg border border-yellow-100 bg-yellow-50 p-4 shadow-sm">
        <p className="text-sm text-yellow-800">{error}</p>
        <button
          onClick={refresh}
          className="mt-2 text-xs underline text-yellow-700 hover:text-yellow-900"
        >
          Retry
        </button>
      </div>
    );
  }

  if (!hub) {
    return null;
  }

  return (
    <div className="rounded-lg border border-gray-200 bg-white p-4 shadow-sm">
      <div className="flex items-start justify-between">
        <div>
          <h3 className="text-sm font-semibold text-gray-900">Top Hub</h3>
          <p className="mt-1 text-2xl font-bold text-blue-600">{hub}</p>
        </div>
        <span className="inline-flex items-center rounded-full bg-blue-50 px-2 py-1 text-xs font-medium text-blue-700">
          Signal
        </span>
      </div>

      {insight && <p className="mt-3 text-sm text-gray-600">{insight}</p>}

      {docs.length > 0 && (
        <div className="mt-3">
          <p className="text-xs font-medium text-gray-500">Related docs</p>
          <ul className="mt-1 space-y-1">
            {docs.slice(0, 4).map((doc, idx) => (
              <li key={idx}>
                <a
                  href={`https://huggingface.co/datasets/axentx/costinel-knowledge/resolve/main/docs/${doc.path}`}
                  target="_blank"
                  rel="noopener noreferrer"
                  className="text-xs text-blue-600 hover:underline"
                >
                  {doc.title || doc.path}
                </a>
              </li>
            ))}
          </ul>
          {docs.length > 4 && (
            <p className="mt-1 text-xs text-gray-400">
              +{docs.length - 4} more
            </p>
          )}
        </div>
      )}
    </div>
  );
}
```

**Why this wins**:  
- Clear, compact UI with Tailwind classes matching existing dashboard cards.  
- Shows only signals (no actions).  
- Truncates docs to 4 with a count for overflow.  
- Retry button wired to hook’s `refresh`.
