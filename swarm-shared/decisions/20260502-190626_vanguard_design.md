# vanguard / design

## Final Synthesis — One Correct, Actionable Design

**Core diagnosis (accepted from both candidates)**  
- Missing canonical discovery entrypoint that surfaces top-hub insights before planning (violates `#knowledge-rag #graph #hub` pattern).  
- Missing CDN-bypass file-list strategy for HF datasets → training will 429 on recursive `list_repo_files`.  
- No Lightning Studio reuse guard → quota burn.  
- No schema projection guard before upload → mixed-schema `enriched/` breaks surrogate-1 pyarrow loads.  
- No single executable discovery command that composes these patterns.

**Chosen approach**  
Create **one new CLI** at `/opt/axentx/vanguard/design/discover.py` plus minimal `requirements.txt`.  
Scope: additive only (no changes to existing code).  
Resolve contradictions in favor of **correctness + concrete actionability**:

1. **Top-hub insight**: prefer local `knowledge_rag/top_hubs.json` if present; otherwise run `granite-business-research.sh` if present; otherwise graceful fallback with clear guidance.  
2. **HF CDN-bypass**: non-recursive `list_repo_tree` → `file_list.json`; include repo+folder metadata; fail gracefully with actionable message if token/repo missing.  
3. **Lightning Studio reuse**: use Lightning SDK; find running studio by name; restart if stopped; create only if absent; explicit idle-stop warning in runbook.  
4. **Schema projection**: enforce `{prompt,response}` only before upload; attribution via filename; no extra columns in training artifacts.  
5. **Kaggle guidance**: include token-based push note only when `KAGGLE_TOKEN` present; otherwise skip.  
6. **Runbook**: emit human-readable markdown + stdout; include concrete next steps.

#!/usr/bin/env python3
"""
Vanguard design discovery
- surfaces top-hub insight (knowledge-rag pattern)
- pre-lists HF dataset folder for CDN-bypass training
- reuses running Lightning Studio (quota-safe)
- projects schema to {prompt,response} before any upload path
"""
import json
import os
import subprocess
import sys
from datetime import datetime
from pathlib import Path

from huggingface_hub import list_repo_tree

# ---------- config (override by env) ----------
HF_REPO = os.getenv("HF_DATASET_REPO", "datasets/example/mirror-merged")
HF_FOLDER = os.getenv("HF_FOLDER", "batches/mirror-merged/2026-05-02")
LIGHTNING_NAME = os.getenv("LIGHTNING_STUDIO_NAME", "vanguard-surrogate-train")
LIGHTNING_MACHINE = os.getenv("LIGHTNING_MACHINE", "L40S")
KAGGLE_TOKEN = os.getenv("KAGGLE_TOKEN", "")
OUTPUT_DIR = Path(__file__).parent
FILE_LIST_PATH = OUTPUT_DIR / "file_list.json"
RUNBOOK_PATH = OUTPUT_DIR / "discovery_output.md"

# ---------- helpers ----------
def run_cmd(cmd: str, cwd=None) -> subprocess.CompletedProcess:
    return subprocess.run(cmd, shell=True, capture_output=True, text=True, cwd=cwd)

def top_hub_insight() -> str:
    """
    Pattern: top-hub doc insight
    Priority:
      1) local knowledge_rag/top_hubs.json
      2) granite-business-research.sh
      3) graceful fallback
    """
    marker = OUTPUT_DIR.parent / "knowledge_rag" / "top_hubs.json"
    if marker.exists():
        try:
            data = json.loads(marker.read_text())
            if isinstance(data, list) and len(data) > 0:
                top = data[0]
                return (
                    f"Top hub: {top.get('name', '?')} "
                    f"(degree {top.get('degree', 0)}). "
                    f"Insight: {top.get('insight', 'Review before planning tasks.')}"
                )
        except Exception:
            pass

    script = OUTPUT_DIR.parent.parent / "scripts" / "granite-business-research.sh"
    if script.exists():
        r = run_cmd(f"bash {script} 2>/dev/null | head -20")
        if r.returncode == 0 and r.stdout.strip():
            return f"Market insight (granite-business-research): {r.stdout.strip()[:200]}..."

    return "Top hub insight unavailable. Run knowledge-rag or provide top_hubs.json."

def prefetch_hf_file_list() -> str:
    """
    Pattern: pre-list file paths once, embed in training script
    Non-recursive tree call -> CDN-only downloads during training.
    """
    try:
        tree = list_repo_tree(repo_id=HF_REPO, path=HF_FOLDER, recursive=False)
        files = [item.rfilename for item in tree if item.type == "file"]
        FILE_LIST_PATH.write_text(
            json.dumps({"repo": HF_REPO, "folder": HF_FOLDER, "files": files}, indent=2)
        )
        return f"Saved {len(files)} file(s) to {FILE_LIST_PATH.name} for CDN-bypass training."
    except Exception as exc:
        return (
            f"HF pre-list failed (may need token or repo access): {exc}. "
            "Set HF_TOKEN or verify repo/folder and rerun."
        )

def lightning_studio_reuse() -> str:
    """
    Pattern: existing Lightning Studio reuse + idle-stop guard
    """
    try:
        from lightning import Lightning, Studio, Teamspace, Machine
    except Exception as exc:
        return f"Lightning SDK unavailable: {exc}"

    try:
        ls = Lightning()
        teamspace = Teamspace()
        running = None
        for s in teamspace.studios:
            if s.name == LIGHTNING_NAME:
                running = s
                break

        if running and running.status == "running":
            return f"Reusing running studio '{LIGHTNING_NAME}' (id={running.id})."

        if running and running.status == "stopped":
            running.start(machine=Machine(LIGHTNING_MACHINE))
            return f"Restarted stopped studio '{LIGHTNING_NAME}' on {LIGHTNING_MACHINE}."

        studio = Studio(name=LIGHTNING_NAME, machine=Machine(LIGHTNING_MACHINE), create_ok=True)
        return f"Created new studio '{LIGHTNING_NAME}' on {LIGHTNING_MACHINE}."
    except Exception as exc:
        return f"Lightning studio action failed: {exc}"

def schema_projection_note() -> str:
    """
    Pattern: dataset-mirror writes mixed-schema files to enriched/
    Project to {prompt,response} only before upload; attribution via filename.
    """
    return (
        "Schema rule: project to {prompt,response} only before upload. "
        "Attribution via filename pattern: batches/mirror-merged/{date}/{slug}.parquet. "
        "Do not add source/ts columns to training artifacts."
    )

def kaggle_push_note() -> str:
    """
    Pattern: Kaggle KGAT token auth bypass CLI; use raw HTTP with Bearer.
    """
    if not KAGGLE_TOKEN:
        return "Kaggle: KAGGLE_TOKEN not set; skipping kernel push guidance."
    return (
        "Kaggle: Use raw HTTP POST to /api/v1/kernels/push with Bearer token, "
        "camelCase keys (slug, newTitle, text), and isPrivate:true for new accounts."
    )

# ---------- main ----------
def main() -> None:
    now = datetime.utcnow().isoformat(timespec="seconds") + "Z"
    report_lines = [
        f"# Vanguard discovery output - {now}",
        "",
        "## Top-hub insight",
        top_hub_insight(),
        "",
        "## HF CDN-bypass file list",
        prefetch_hf_file_list(),
        "",
        "## Lightning Studio",
        lightning_studio_reuse(),
        "",
        "## Schema guidance",
        schema_projection_note(),
        "",
        "## Kaggle guidance",
        kaggle_push_note(),
        "",
        "## Next steps",
        "- Embed file_list.json in train.py and use CDN-only downloads (no HF API during training).",
        "- Ensure training script projects to {prompt,response} only.",
