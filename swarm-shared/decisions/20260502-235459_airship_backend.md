# airship / backend

## Implementation Plan — `airship discover` (backend)

**Scope**: ≤2h  
**Goal**: Deterministic CLI command `airship discover` that emits a CDN-only file manifest for a given `repo_id` and `date_folder`, plus HTTP endpoint `/api/discover` for Arkship UI. Uses HF CDN bypass (no auth, no API rate limit during training).

### 1) Architecture
- CLI: `airship discover --repo <repo> --date <YYYY-MM-DD> [--out manifest.json]`
- HTTP: `GET /api/discover?repo=<repo>&date=<YYYY-MM-DD>`
- Output schema:
  ```json
  {
    "repo_id": "...",
    "date_folder": "...",
    "generated_at": "...",
    "cdn_prefix": "https://huggingface.co/datasets/{repo_id}/resolve/main/{date_folder}/",
    "files": [
      {
        "path": "2026-04-29/file1.parquet",
        "size": 12345,
        "cdn_url": "https://huggingface.co/datasets/{repo_id}/resolve/main/2026-04-29/file1.parquet"
      }
    ]
  }
  ```

### 2) Implementation Steps (≤2h)

1. Add CLI module `airship/cli/discover.py`
2. Add FastAPI route in `arkship/main.py` (or dedicated router)
3. Add HF tree listing utility with rate-limit awareness (single API call, fallback to CDN probe if 429)
4. Add tests (CLI + endpoint)
5. Update `pyproject.toml` entry points if needed

### 3) Code Snippets

#### `airship/cli/discover.py`
```python
#!/usr/bin/env python3
"""
airship discover
Deterministic CDN-only file manifest generator for HF datasets.
Usage:
  airship discover --repo <repo> --date <YYYY-MM-DD> [--out manifest.json]
"""

import json
import sys
from datetime import datetime
from pathlib import Path
from typing import List, Dict, Any

import click
import httpx
from huggingface_hub import HfApi, hf_hub_url

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo_id}/resolve/main/{path}"

def list_date_folder(repo_id: str, date_folder: str) -> List[Dict[str, Any]]:
    """
    List files in repo_id/date_folder/ using HF API (single non-recursive call).
    Falls back to CDN probe if rate-limited (429).
    """
    api = HfApi()
    try:
        items = api.list_repo_tree(
            repo_id=repo_id,
            path=date_folder,
            repo_type="dataset",
            recursive=False,
        )
        files = [i for i in items if i.type == "file"]
    except httpx.HTTPStatusError as exc:
        if exc.response.status_code == 429:
            # Rate-limited: fall back to CDN HEAD probing for common patterns.
            # This is conservative: probe known extensions in date folder.
            click.echo(
                f"Rate limited on repo tree; falling back to CDN HEAD probing for {date_folder}",
                err=True,
            )
            files = _cdn_probe(repo_id, date_folder)
        else:
            raise
    return files

def _cdn_probe(repo_id: str, date_folder: str) -> List[Dict[str, Any]]:
    """
    Conservative CDN HEAD probe for common parquet/jsonl extensions.
    Returns synthetic file entries for reachable URLs.
    """
    import httpx

    candidates = [
        f"{date_folder}/data.parquet",
        f"{date_folder}/data-00000-of-00001.parquet",
        f"{date_folder}/train.parquet",
        f"{date_folder}/data.jsonl",
    ]
    found = []
    with httpx.Client(timeout=10.0) as client:
        for cand in candidates:
            url = CDN_TEMPLATE.format(repo_id=repo_id, path=cand)
            try:
                r = client.head(url, follow_redirects=True)
                if r.status_code == 200:
                    found.append(
                        {
                            "path": cand,
                            "size": int(r.headers.get("content-length", 0)),
                        }
                    )
            except httpx.HTTPError:
                continue
    return found

def build_manifest(repo_id: str, date_folder: str) -> Dict[str, Any]:
    files = list_date_folder(repo_id, date_folder)
    file_entries = []
    for f in files:
        path = f.path if isinstance(f, dict) else f.rfilename
        # normalize to repo-root relative path
        if not path.startswith(date_folder):
            path = f"{date_folder}/{path}"
        entry = {
            "path": path,
            "size": f.size if isinstance(f, dict) and hasattr(f, "size") else (f.size if hasattr(f, "size") else 0),
            "cdn_url": CDN_TEMPLATE.format(repo_id=repo_id, path=path),
        }
        file_entries.append(entry)

    # Deterministic ordering
    file_entries.sort(key=lambda x: x["path"])

    return {
        "repo_id": repo_id,
        "date_folder": date_folder.rstrip("/"),
        "generated_at": datetime.utcnow().isoformat() + "Z",
        "cdn_prefix": CDN_TEMPLATE.format(repo_id=repo_id, path=date_folder.rstrip("/") + "/"),
        "files": file_entries,
    }

@click.command()
@click.option("--repo", required=True, help="HF dataset repo id (e.g., 'org/dataset')")
@click.option("--date", required=True, help="Date folder (YYYY-MM-DD)")
@click.option("--out", type=click.Path(), default="-", help="Output file (default: stdout)")
def discover(repo: str, date: str, out: str) -> None:
    """Generate deterministic CDN-only file manifest for HF dataset date folder."""
    try:
        manifest = build_manifest(repo, date)
    except Exception as exc:
        click.echo(f"Error: {exc}", err=True)
        sys.exit(1)

    payload = json.dumps(manifest, indent=2)
    if out == "-":
        click.echo(payload)
    else:
        Path(out).write_text(payload)
        click.echo(f"Manifest written to {out}")

if __name__ == "__main__":
    discover()
```

#### FastAPI route (add to `arkship/routes/discover.py` or `arkship/main.py`)
```python
# arkship/routes/discover.py
from fastapi import APIRouter, HTTPException, Query
from airship.cli.discover import build_manifest

router = APIRouter()

@router.get("/api/discover")
async def api_discover(
    repo: str = Query(..., description="HF dataset repo id"),
    date: str = Query(..., description="Date folder (YYYY-MM-DD)"),
):
    """
    Return CDN-only file manifest for deterministic surrogate/kaggle training.
    """
    try:
        return build_manifest(repo, date)
    except Exception as exc:
        raise HTTPException(status_code=500, detail=str(exc))
```

Register router in `arkship/main.py`:
```python
# arkship/main.py (excerpt)
```
