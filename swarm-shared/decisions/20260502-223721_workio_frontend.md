# workio / frontend

### Final synthesized plan (strongest parts, contradictions resolved)

**Core principles**
- **Optimistic UI first** for clock in/out so actions feel instant; reconcile server truth on completion and rollback on failure.  
- **Local-first drafts** for leave/OT with persistence across navigation, refresh, and tenant switches; clear recovery UX.  
- **Explicit, non-blocking feedback** (loading states, toast/inline errors, retry) to prevent double-taps and confusion.  
- **Defensive storage** (try/catch, quota handling) and **type-safe hooks** to avoid runtime crashes.

---

### 1. Diagnosis (merged)
- No optimistic UI for clock in/out → perceived latency and poor offline UX.  
- Missing local-first draft state → unsaved leave/OT data lost on navigation, refresh, offline, or tenant switch.  
- Stale button state after network errors; no retry/backoff or clear error state.  
- No loading indicators/skeletons → UI freezes and users double-tap, causing duplicate requests.  
- No recovery UX for failed submissions or persisted drafts.

---

### 3. Implementation

#### 3.1 `useLocalDraft` hook (typed, defensive)
```tsx
// src/hooks/useLocalDraft.ts
import { useEffect, useState, useCallback } from 'react';

export function useLocalDraft<T>(key: string, initialValue: T) {
  const [draft, setDraft] = useState<T>(() => {
    try {
      const raw = localStorage.getItem(key);
      return raw ? JSON.parse(raw) : initialValue;
    } catch {
      return initialValue;
    }
  });

  const save = useCallback((value: T) => {
    try {
      localStorage.setItem(key, JSON.stringify(value));
      setDraft(value);
    } catch (e) {
      console.warn('localStorage save failed', e);
    }
  }, [key]);

  const clear = useCallback(() => {
    try {
      localStorage.removeItem(key);
      setDraft(initialValue);
    } catch (e) {
      console.warn('localStorage clear failed', e);
    }
  }, [key, initialValue]);

  // Keep in sync with storage events (cross-tab)
  useEffect(() => {
    const onStorage = (e: StorageEvent) => {
      if (e.key === key && e.newValue) {
        try {
          setDraft(JSON.parse(e.newValue));
        } catch {
          // ignore malformed
        }
      } else if (e.key === key && e.newValue === null) {
        setDraft(initialValue);
      }
    };
    window.addEventListener('storage', onStorage);
    return () => window.removeEventListener('storage', onStorage);
  }, [key, initialValue]);

  return [draft, save, clear] as const;
}
```

#### 3.2 Optimistic clock controls (instant + rollback)
```tsx
// src/features/attendance/ClockControls.tsx
import { useState, useCallback } from 'react';
import { clockIn as apiClockIn, clockOut as apiClockOut } from '../lib/api';
import { toast } from '../lib/toast';

type Status = 'idle' | 'pending' | 'synced' | 'error';

export function ClockControls() {
  const [status, setStatus] = useState<Status>('idle');
  const [optimisticIn, setOptimisticIn] = useState<boolean | null>(null);

  const mutate = useCallback(async (action: 'in' | 'out') => {
    const prevStatus = status;
    const prevOptimistic = optimisticIn;
    const willBeIn = action === 'in';

    // Optimistic update
    setStatus('pending');
    setOptimisticIn(willBeIn);

    try {
      if (action === 'in') await apiClockIn();
      else await apiClockOut();
      setStatus('synced');
      setOptimisticIn(willBeIn);
      toast.success(`Clocked ${willBeIn ? 'in' : 'out'}`);
    } catch (err: any) {
      // Rollback
      setStatus('error');
      setOptimisticIn(prevOptimistic);
      toast.error(err.message || 'Network error — please retry');
      // Keep prevStatus allows retry UX to be clear
    }
  }, [status, optimisticIn]);

  const isPending = status === 'pending';

  return (
    <div>
      <button onClick={() => mutate('in')} disabled={isPending || optimisticIn === true}>
        {isPending && optimisticIn === true ? 'Clocking in…' : 'Clock in'}
      </button>
      <button onClick={() => mutate('out')} disabled={isPending || optimisticIn === false}>
        {isPending && optimisticIn === false ? 'Clocking out…' : 'Clock out'}
      </button>

      {status === 'error' && (
        <div style={{ color: 'crimson' }}>
          Last request failed — please retry.
        </div>
      )}
    </div>
  );
}
```

#### 3.3 Leave/OT form with local draft + optimistic submit
```tsx
// src/features/requests/LeaveRequestForm.tsx
import { useState, useEffect } from 'react';
import { useLocalDraft } from '../../hooks/useLocalDraft';
import { submitLeave } from '../../lib/api';
import { toast } from '../../lib/toast';

const DRAFT_KEY = 'leave_request_draft';

export function LeaveRequestForm() {
  const [draft, saveDraft, clearDraft] = useLocalDraft(DRAFT_KEY, { leave_type: '', start_date: '', end_date: '', note: '' });
  const [submitting, setSubmitting] = useState(false);

  // Persist on change (debounced in real apps)
  const onChange = (field: string, value: string) => {
    saveDraft({ ...draft, [field]: value });
  };

  const onSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    setSubmitting(true);
    try {
      await submitLeave(draft);
      clearDraft();
      toast.success('Leave request submitted');
    } catch (err: any) {
      toast.error(err.message || 'Submission failed — saved as draft');
    } finally {
      setSubmitting(false);
    }
  };

  // Optional: warn on leave with unsaved draft
  useEffect(() => {
    const onBeforeUnload = (e: BeforeUnloadEvent) => {
      if (JSON.stringify(draft) !== JSON.stringify({ leave_type: '', start_date: '', end_date: '', note: '' })) {
        e.preventDefault();
        e.returnValue = '';
      }
    };
    window.addEventListener('beforeunload', onBeforeUnload);
    return () => window.removeEventListener('beforeunload', onBeforeUnload);
  }, [draft]);

  return (
    <form onSubmit={onSubmit}>
      <input value={draft.leave_type} onChange={(e) => onChange('leave_type', e.target.value)} placeholder="Leave type" />
      <input type="date" value={draft.start_date} onChange={(e) => onChange('start_date', e.target.value)} />
      <input type="date" value={draft.end_date} onChange={(e) => onChange('end_date', e.target.value)} />
      <textarea value={draft.note} onChange={(e) => onChange('note', e.target.value)} placeholder="Note" />
      <button type="submit" disabled={submitting}>
        {submitting ? 'Submitting…' : 'Submit'}
      </button>
      {submitting && <div>Saving draft locally...</div>}
    </form>
  );
}
```
