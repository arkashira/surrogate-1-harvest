# vanguard / design

## Final Synthesis (Best Parts + Correctness + Actionability)

### 1. Diagnosis (resolved)
- **Problem**: runtime `list_repo_tree` / `load_dataset(..., streaming=True)` causes HF API 429s, non-deterministic shard order, and per-epoch metadata traffic.
- **Problem**: no content-addressed manifest per date folder → non-reproducible epochs, unreliable resumption, no offline validation.
- **Problem**: loader depends on HF datasets/list calls during training instead of CDN-only fetches → wastes quota and adds tail-latency stalls.
- **Risk**: future enrichment could hit HF commit-cap (128/hr) if training loop writes back.

**Resolution**: accept all candidates’ diagnosis; they agree.

---

### 2. Proposed change (merged)
- Add `manifests/` with per-date, content-addressed file lists (JSON) produced by a single `list_repo_tree` call (Mac side).
- Modify training to accept a manifest path and fetch files via HF CDN (`/resolve/main/...`) with **zero Authorization/API calls** during training.
- Add `tools/build_manifest.py` to generate and validate the manifest (one date folder pilot, e.g., `2026-04-29`).
- Manifest fields: repo, date_folder, generated_at, deterministic ordering, per-entry: path, url, size, sha256 (if available), etag.
- Scope: pilot one date folder; keep loader simple, deterministic, and auditable.

---

tools/build_manifest.py
```python
#!/usr/bin/env python3
"""
Generate a content-addressed manifest for one date folder in a HF dataset repo.
Usage:
  python tools/build_manifest.py \
    --repo "datasets/username/surrogate-1" \
    --date-folder "2026-04-29" \
    --out manifests/manifest-2026-04-29.json
"""
import argparse
import hashlib
import json
import os
import sys
import time
from typing import List, Dict, Optional

try:
    from huggingface_hub import list_repo_tree
except ImportError:
    print("pip install huggingface_hub")
    sys.exit(1)

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def _parse_lfs_sha(lfs_info):
    # lfs.oid may be "sha256:..." ; return hex sha256 or None
    if not lfs_info:
        return None
    oid = lfs_info.get("oid", "")
    if oid.startswith("sha256:"):
        return oid.split(":", 1)[1]
    return None

def build_manifest(repo: str, date_folder: str) -> List[Dict]:
    """
    Single API call to list items in date_folder (non-recursive),
    then list subfolders/files deterministically.
    """
    # list top-level in date_folder (non-recursive)
    top_items = list_repo_tree(repo=repo, path=date_folder, recursive=False)
    entries = []
    for item in sorted(top_items, key=lambda x: x.path):
        if item.type != "file":
            continue
        path = item.path
        url = CDN_TEMPLATE.format(repo=repo, path=path)
        entries.append({
            "repo": repo,
            "path": path,
            "url": url,
            "size": getattr(item, "size", None),
            "sha256": getattr(item, "sha256", None) or _parse_lfs_sha(getattr(item, "lfs", None)),
            "etag": getattr(item, "etag", None),
        })
    # deterministic ordering
    entries.sort(key=lambda e: e["path"])
    return entries

def main() -> None:
    parser = argparse.ArgumentParser(description="Build HF CDN manifest for a date folder.")
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g. datasets/username/surrogate-1)")
    parser.add_argument("--date-folder", required=True, help="Date folder inside repo (e.g. 2026-04-29)")
    parser.add_argument("--out", required=True, help="Output JSON path")
    parser.add_argument("--retry-wait", type=int, default=360, help="Wait on 429 (seconds)")
    args = parser.parse_args()

    for attempt in range(3):
        try:
            entries = build_manifest(args.repo, args.date_folder)
            break
        except Exception as exc:
            if "429" in str(exc) and attempt < 2:
                print(f"Rate limited, waiting {args.retry_wait}s (attempt {attempt+1})")
                time.sleep(args.retry_wait)
                continue
            raise

    manifest = {
        "repo": args.repo,
        "date_folder": args.date_folder,
        "generated_at": time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime()),
        "count": len(entries),
        "entries": entries,
    }

    os.makedirs(os.path.dirname(os.path.abspath(args.out)), exist_ok=True)
    with open(args.out, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Wrote {len(entries)} entries to {args.out}")

if __name__ == "__main__":
    main()
```

train_cdn.py (minimal, deterministic, CDN-only loader)

#!/usr/bin/env python3
"""
Train using CDN-only fetches. Pass --manifest manifests/manifest-YYYY-MM-DD.json
"""
import argparse
import json
import random
from typing import List, Dict

import torch
from torch.utils.data import IterableDataset, DataLoader
import requests

class CDNShardDataset(IterableDataset):
    def __init__(self, manifest_path: str, shuffle: bool = True):
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.entries: List[Dict] = self.manifest["entries"]
        self.shuffle = shuffle
        self.epoch = 0

    def set_epoch(self, epoch: int):
        self.epoch = epoch

    def __iter__(self):
        worker_info = torch.utils.data.get_worker_info()
        entries = list(self.entries)
        if self.shuffle:
            # deterministic per-epoch shuffle without API calls
            rng = random.Random(12345 + self.epoch)
            rng.shuffle(entries)

        if worker_info is None:
            it = entries
        else:
            per_worker = len(entries) // worker_info.num_workers
            start = worker_info.id * per_worker
            end = start + per_worker
            it = entries[start:end]

        for item in it:
            # CDN fetch: no Authorization header -> bypasses API rate limits
            resp = requests.get(item["url"], timeout=30)
            resp.raise_for_status()
            # TODO: parse parquet/jsonl bytes into {prompt, response} here
            # For now yield raw bytes + path for downstream processing
            yield {"path": item["path"], "raw": resp.content}

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--manifest", required=True)
    parser.add_argument("--epochs", type=int, default=1)
    parser.add_argument("--batch-size", type=int, default=8)
    parser.add_argument("--workers", type=int, default=4)
    args = parser.parse_args()

    dataset = CDNShardDataset(args.manifest, shuffle=True)
    loader = DataLoader(
        dataset,
        batch_size=args.batch_size,
        num_workers=args.workers,
        persistent_workers=args.workers > 0,
    )

    for epoch in range(args.epochs):
