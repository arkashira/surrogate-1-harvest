### 2. Tenant cache (LRU, 10m TTL)
`/opt/axentx/workio/workio/server/src/middleware/tenant.ts`
```ts
import { LRUCache } from 'lru-cache';
import { getTenantByChannelId } from '../db/queries/tenant-queries';

type Tenant = {
  id: string;
  name: string;
  channelId: string;
};

const tenantCache = new LRUCache<string, Tenant>({
  max: 500,
  ttl: 1000 * 60 * 10, // 10m
  allowStale: false,
  updateAgeOnGet: true,
});

export async function resolveTenantByChannelId(channelId: string): Promise<Tenant | null> {
  const cached = tenantCache.get(channelId);
  if (cached) return cached;

  const tenant = await getTenantByChannelId(channelId);
  if (tenant) tenantCache.set(channelId, tenant);
  return tenant || null;
}

export function invalidateTenantChannel(channelId: string) {
  tenantCache.delete(channelId);
}
```

### 4. Request tracing middleware
`/opt/axentx/workio/workio/server/src/middleware/tracing.ts`
```ts
import { Request, Response, NextFunction } from 'express';
import { v4 as uuidv4 } from 'uuid';

export function requestTracing(req: Request, res: Response, next: NextFunction) {
  const correlationId = (req.headers['x-correlation-id'] as string) || uuidv4();
  (req as any).correlationId = correlationId;
  res.setHeader('x-correlation-id', correlationId);
  next();
}
```

### 6. LINE signature verification (defense-in-depth)
`/opt/axentx/workio/workio/server/src/lib/line-signature.ts`
```ts
import crypto from 'crypto';

const REPLAY_WINDOW_MS = 5 * 60 * 1000; // 5m
const seenSignatures = new Set<string>();

export function verifyLineSignature(rawBody: Buffer, signature: string, secret: string): boolean {
  // Basic HMAC-SHA256 check
  const expected = crypto.createHmac('sha256', secret).update(rawBody).digest('base64');
  if (expected !== signature) return false;

  // Anti-replay: reject duplicate signatures within window
  if (seenSignatures.has(signature)) return false;
  seenSignatures.add(signature);
  setTimeout(() => seenSignatures.delete(signature), REPLAY_WINDOW_MS).unref();

  // Optional: body timestamp check if you include timestamp header (defensive)
  // const ts = Number(req.headers['x-line-request-timestamp']);
  // if (ts && Math.abs(Date.now() - ts) > 300_000) return false;

  return true;
}
```

### 7. Updated LINE webhook route
`/opt/axentx/workio/workio/server/src/routes/line-webhook.ts`
```ts
import express from 'express';
import { logger } from '../lib/logger';
import { resolveTenantByChannelId } from '../middleware/tenant';
import { requestTracing } from '../middleware/tracing';
import { verifyLineSignature } from '../lib/line-signature';

const router = express.Router();

router.use(requestTracing);

router.post('/webhook/line', express.raw({ type: 'application/json' }), async (req, res) => {
  const correlationId = (req as any).correlationId || 'unknown';
  const start = Date.now();
  const signature = req.headers['x-line-signature'] as string;
  const channelSecret = process.env.LINE_CHANNEL_SECRET || '';

  logger.info({
    correlationId,
    event: 'line.webhook.received',
    path: req.path,
    method: req.method,
    contentLength: req.body.length,
  });

  // 1) Signature verification (defense-in-depth)
  if (!signature || !verifyLineSignature(req.body, signature, channelSecret)) {
    logger.warn({ correlationId, event: 'line.webhook.invalid_signature' });
    res.status(401).json({ error: 'invalid signature' });
    return;
  }

  // 2) Parse events
  let events;
  try {
    events = JSON.parse(req.body.toString('utf8')).events || [];
  } catch (err) {
    logger.error({ correlationId, event: 'line.webhook.parse_error', err });
    res.status(400).json({ error: 'invalid json' });
    return;
  }

  // 3) Resolve tenant by channelId
  const channelId = req.headers['x-line-channel-id'] as string || events[0]?.source?.userId || '';
  if (!channelId) {
    logger.warn({ correlationId, event: 'line.webhook.missing_channel' });
    res.status(400).json({ error: 'missing channel' });
    return;
  }

  const tenant = await resolveTenantByChannelId(channelId);
  if (!tenant) {
    logger.warn({ correlationId, event: 'line.webhook.tenant_not_found', channelId });
    res.status(404).json({ error: 'tenant not found' });
    return;
  }

  (req as any).tenant = tenant;

  // 4) Acknowledge quickly; process events asynchronously (enqueue/handlers)
  try {
    // TODO: enqueue events per tenant for async processing
    logger.info({
      correlationId,
      tenantId: tenant.id,
      channelId,
      eventCount: events.length,
      ms: Date.now() - start,
      event: 'line.webhook.acknowledged',
    });
    res.status(200).json({ received: true });
