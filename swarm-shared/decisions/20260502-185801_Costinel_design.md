# Costinel / design

## Highest-Value Incremental Improvement (<2h)

**Goal**: Add a lightweight, machine-readable **manifest + readiness probe** so onboarding, training scripts, and ops can discover ingestible sources and validate Costinel’s runtime state without trial-and-error.

**Why this ships fast and unlocks downstream work**:
- Enables `knowledge-rag`/`top-hub` workflows to auto-discover context sources.
- Lets surrogate-1 training scripts pre-list HF paths once and embed a file manifest (CDN-only fetch) to avoid HF API rate limits.
- Provides `/ready` and `/live` probes for deployment automation and Lightning Studio reuse checks.
- Keeps Costinel’s “Sense + Signal — ไม่ Execute” philosophy (read-only probes, no mutating actions).

---

## Implementation Plan (≤2h)

1. **Add manifest schema and generator** (`manifest.yaml` + `scripts/build_manifest.py`)
   - Enumerate cloud accounts, HF repos/date-folders, and known dataset schemas.
   - Include `top_hub_hint` for RAG (e.g., `"MOC"` or most-connected doc).
   - Include `hf_file_list_path` for training scripts to embed.

2. **Add readiness/liveness endpoints** (`app/api/probes.py`)
   - `GET /live` — process is up.
   - `GET /ready` — critical deps reachable (object storage, HF CDN, DB) and manifest valid.

3. **Add CLI command to refresh manifest** (`costinel manifest refresh`)
   - Calls HF `list_repo_tree` once per configured repo/date (respect rate limits) and writes `manifest.json` used by training jobs.

4. **Update Docker entrypoint** to run `/ready` check before accepting traffic.

5. **Add lightweight smoke test** (`scripts/smoke_probes.py`) for CI/local dev.

### 2) Manifest builder — `scripts/build_manifest.py`

```python
#!/usr/bin/env python3
"""
Build/update manifest.yaml and optional per-date file lists for CDN-only training.
Run from Mac (or CI) after HF rate-limit window clears.
"""
import json
import os
import sys
import yaml
from datetime import datetime, timezone

try:
    from huggingface_hub import list_repo_tree
except ImportError:
    print("Install huggingface_hub: pip install huggingface_hub")
    sys.exit(1)

MANIFEST_PATH = os.path.join(os.path.dirname(__file__), "..", "manifest.yaml")
HF_REPO = "datasets/axentx/surrogate-1"
DATE_FOLDER = "2026-04-29"
OUTPUT_FILE_LIST = os.path.join(
    os.path.dirname(__file__), "..", "file_lists", f"{DATE_FOLDER}.json"
)

def build_manifest() -> dict:
    return {
        "version": "1.0",
        "generated_at": datetime.now(timezone.utc).isoformat(),
        "knowledge_rag": {
            "top_hub_hint": "MOC",
            "graph_source": "knowledge-rag",
        },
        "cloud_accounts": [
            {
                "provider": "aws",
                "account_id": os.getenv("AWS_ACCOUNT_ID", "123456789012"),
                "role_arn": os.getenv("AWS_ROLE_ARN", "arn:aws:iam::123456789012:role/CostinelReadOnly"),
                "regions": ["us-east-1", "eu-west-1"],
            },
            {
                "provider": "gcp",
                "project_id": os.getenv("GCP_PROJECT_ID", "axentx-costinel"),
                "dataset_location": "us-central1",
            },
        ],
        "datasets": {
            "surrogate_1": {
                "repo": HF_REPO,
                "date_folders": [DATE_FOLDER],
                "manifest_file": f"batches/mirror-merged/{{date}}/manifest.json",
                "schema_projection": {"prompt_field": "prompt", "response_field": "response"},
                "hf_api_strategy": {
                    "use_cdn_only_during_training": True,
                    "rate_limit_backoff_seconds": 360,
                },
            },
        },
        "training": {
            "lightning": {
                "reuse_running_studios": True,
                "preferred_clouds": ["lightning-lambda-prod", "lightning-public-prod"],
                "idle_timeout_minutes": 30,
            },
        },
    }

def cache_hf_folder_file_list(repo: str, folder: str, out_path: str) -> None:
    """
    Single API call to list one folder (non-recursive). Save JSON for training scripts.
    Training will use CDN-only URLs:
      https://huggingface.co/datasets/{repo}/resolve/main/{path}
    """
    os.makedirs(os.path.dirname(out_path), exist_ok=True)
    try:
        tree = list_repo_tree(repo=repo, path=folder, recursive=False)
        files = [item.rfilename for item in tree if item.type == "file"]
        payload = {
            "repo": repo,
            "folder": folder,
            "files": files,
            "cdn_prefix": f"https://huggingface.co/datasets/{repo}/resolve/main",
            "note": "Embed this in training script; use CDN URLs to avoid HF API auth rate limits.",
        }
        with open(out_path, "w") as f:
            json.dump(payload, f, indent=2)
        print(f"Saved HF file list to {out_path} ({len(files)} files)")
    except Exception as e:
        print(f"HF list_repo_tree failed (possibly rate-limited): {e}")
        # Fallback: create empty list so training script can still start
        with open(out_path, "w") as f:
            json.dump({"repo": repo, "folder": folder, "files": [], "error": str(e)}, f)

def main() -> None:
    manifest = build_manifest()
    os.makedirs(os.path.dirname(MANIFEST_PATH), exist_ok=True)
    with open(MANIFEST_PATH, "w") as f:
        yaml.dump(manifest, f, sort_keys=False, allow_unicode=True)
    print(f"Manifest written to {MANIFEST_PATH}")

    # Cache file list for CDN-only training
    cache_hf_folder_file_list(HF_REPO, DATE_FOLDER, OUTPUT_FILE_LIST)

if __name__ == "__main__":
    main()
```
