## Code snippets

### `scripts/build_hf_manifest.py`
```python
#!/usr/bin/env python3
"""
Generate manifest.json for a HuggingFace dataset folder (non-recursive).
Run on Mac orchestrator after rate-limit window clears.
"""
import json
import os
import sys
from pathlib import Path
from typing import List, Dict

from huggingface_hub import HfApi

HF_REPO = os.getenv("HF_DATASET_REPO", "axentx/surrogate-mirror")
DATE_FOLDER = os.getenv("HF_DATE_FOLDER", "batches/mirror-merged/2026-05-03")
OUT_PATH = Path(os.getenv("MANIFEST_OUT", "manifest.json"))

api = HfApi()

def build_manifest(repo: str, folder: str) -> List[Dict]:
    entries = api.list_repo_tree(repo=repo, path=folder, recursive=False)
    manifest = []
    for e in entries:
        if e.type != "file":
            continue
        cdn_url = f"https://huggingface.co/datasets/{repo}/resolve/main/{e.path}"
        manifest.append({
            "path": e.path,
            "cdn_url": cdn_url,
            "size": getattr(e, "size", None),
        })
    return manifest

if __name__ == "__main__":
    manifest = build_manifest(HF_REPO, DATE_FOLDER)
    OUT_PATH.write_text(json.dumps(manifest, indent=2))
    print(f"Wrote {len(manifest)} entries to {OUT_PATH}")
```

### `surrogate/data/hf_cdn_dataset.py`
```python
import json
import logging
from pathlib import Path
from typing import Dict, Iterator, Optional

import requests
from torch.utils.data import IterableDataset

logger = logging.getLogger(__name__)

class HFCDNDataset(IterableDataset):
    def __init__(self, manifest_path: str, max_retries: int = 3, timeout: int = 60):
        self.manifest_path = Path(manifest_path)
        self.max_retries = max_retries
        self.timeout = timeout
        with self.manifest_path.open() as f:
            self.manifest = json.load(f)

    def _download_and_parse(self, entry: Dict) -> Iterator[Dict]:
        url = entry["cdn_url"]
        for attempt in range(1, self.max_retries + 1):
            try:
                resp = requests.get(url, timeout=self.timeout, stream=True)
                resp.raise_for_status()
                # Project to {prompt, response} at parse time (avoid mixed-schema errors)
                # Replace with your actual parser (JSONL, JSON, Parquet projection, etc.)
                data = resp.json() if "json" in url else self._parse_lines(resp)
                for item in self._project_fields(data):
                    yield item
                return
            except Exception as exc:
                logger.warning("Attempt %s/%s failed for %s: %s", attempt, self.max_retries, url, exc)
                if attempt == self.max_retries:
                    logger.error("Failed to fetch %s after %s attempts", url, self.max_retries)
                    return

    def _parse_lines(self, resp):
        # Basic line-oriented JSONL fallback
        for line in resp.iter_lines(decode_unicode=True):
            if line:
                yield json.loads(line)

    def _project_fields(self, data):
        # Keep only {prompt, response}; adapt to your schema
        if isinstance(data, dict):
            data = [data]
        for item in data:
            yield {
                "prompt": item.get("prompt") or item.get("input") or "",
                "response": item.get("response") or item.get("output") or "",
            }

    def __iter__(self) -> Iterator[Dict]:
        worker_info = None
        # Optional: split across DataLoader workers
        try:
            from torch.utils.data import get_worker_info
            worker_info = get_worker_info()
        except Exception:
            pass

        if worker_info is None:
            entries = self.manifest
        else:
            per_worker = len(self.manifest) // worker_info.num_workers
            entries = self.manifest[worker_info.id * per_worker : (worker_info.id + 1) * per_worker]

        for entry in entries:
            yield from self._download_and_parse(entry)
```

from lightning import Teamspace, Studio, Machine

def get_or_start_studio(name: str, machine: Machine = Machine.L40S) -> Studio:
    for s in Teamspace.studios:
        if s.name == name and s.status == "Running":
            return s
    return Studio(name=name, create_ok=True).start(machine=machine)
