# workio / frontend

## 1. Diagnosis

- Clock-in/out UI has no optimistic update or rollback — failed LINE webhook calls leave spinner forever or silently fail, creating ambiguous state and duplicate retries.
- No local pending queue for offline/low-connectivity scenarios — users can’t punch when network blips and retry later.
- Missing visual feedback for GPS validation and LINE webhook latency — users don’t know whether delay is location check or backend.
- Punch state derived from remote fetch only — UI can optimistically reflect “clocked in” while syncing, reducing perceived latency.
- No idempotency key in client requests — retries from UI can create duplicate punches on backend retries.

## 2. Proposed change

File: `workio/src/features/punch/PunchButton.tsx` (create or update)  
Scope: add optimistic UI + offline queue + idempotency key + clear error/retry UX.  
Secondary: expose small `usePunchQueue` hook in `workio/src/lib/punch-queue.ts` to persist pending punches to IndexedDB/localStorage and replay.

### workio/src/lib/punch-queue.ts
```ts
// Simple persistent queue for pending punches (IndexedDB preferred; fallback localStorage)
type PendingPunch = {
  id: string; // idempotency key
  type: 'IN' | 'OUT';
  lat: number;
  lon: number;
  timestamp: string;
  attempts: number;
};

const STORAGE_KEY = 'workio:pending-punches';
const MAX_ATTEMPTS = 5;

export class PunchQueue {
  static getPending(): PendingPunch[] {
    try {
      const raw = localStorage.getItem(STORAGE_KEY);
      return raw ? JSON.parse(raw) : [];
    } catch {
      return [];
    }
  }

  static push(punch: Omit<PendingPunch, 'attempts'>) {
    const list = this.getPending();
    list.push({ ...punch, attempts: 0 });
    this.persist(list);
  }

  static remove(id: string) {
    const list = this.getPending().filter((p) => p.id !== id);
    this.persist(list);
  }

  static incrementAttempts(id: string) {
    const list = this.getPending().map((p) =>
      p.id === id ? { ...p, attempts: p.attempts + 1 } : p
    );
    this.persist(list);
  }

  static filterValid(): PendingPunch[] {
    return this.getPending().filter((p) => p.attempts < MAX_ATTEMPTS);
  }

  private static persist(list: PendingPunch[]) {
    try {
      localStorage.setItem(STORAGE_KEY, JSON.stringify(list));
    } catch (e) {
      console.warn('PunchQueue persist failed', e);
    }
  }
}
```

### workio/src/features/punch/PunchButton.tsx
```tsx
import React, { useState, useEffect } from 'react';
import { usePunchQueue } from '../../lib/usePunchQueue';

type PunchState = 'IDLE' | 'PENDING' | 'ERROR';

export function PunchButton() {
  const [state, setState] = useState<PunchState>('IDLE');
  const [lastError, setLastError] = useState<string | null>(null);
  const { punch, isOnline, queueSize } = usePunchQueue();

  const handlePunch = async (type: 'IN' | 'OUT') => {
    const id = crypto.randomUUID();
    const now = new Date().toISOString();

    // optimistic permission: assume GPS ok; real validation happens server-side
    setState('PENDING');
    setLastError(null);

    try {
      const pos = await getCurrentPosition();
      await punch({ type, lat: pos.coords.latitude, lon: pos.coords.longitude, id, now });
      setState('IDLE');
    } catch (err: any) {
      setState('ERROR');
      setLastError(err.message || 'Punch failed');
    }
  };

  return (
    <div className="flex flex-col items-center gap-3">
      <div className="flex gap-4">
        <button
          onClick={() => handlePunch('IN')}
          disabled={state === 'PENDING'}
          className="btn btn-primary"
        >
          {state === 'PENDING' ? 'Clocking in…' : 'Clock In'}
        </button>
        <button
          onClick={() => handlePunch('OUT')}
          disabled={state === 'PENDING'}
          className="btn btn-secondary"
        >
          {state === 'PENDING' ? 'Clocking out…' : 'Clock Out'}
        </button>
      </div>

      {lastError && (
        <div className="text-sm text-red-600">
          {lastError}{' '}
          <button
            onClick={() => window.location.reload()}
            className="underline text-blue-600"
          >
            Retry
          </button>
        </div>
      )}

      {!isOnline && (
        <div className="text-sm text-amber-600">
          Offline — punch queued ({queueSize})
        </div>
      )}
    </div>
  );
}

function getCurrentPosition(): Promise<GeolocationPosition> {
  return new Promise((resolve, reject) => {
    if (!navigator.geolocation) {
      reject(new Error('Geolocation not supported'));
      return;
    }
    navigator.geolocation.getCurrentPosition(resolve, (err) => {
      reject(new Error('Location required to punch'));
    }, { enableHighAccuracy: true, timeout: 10000 });
  });
}
```
