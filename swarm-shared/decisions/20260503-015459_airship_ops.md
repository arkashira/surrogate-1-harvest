## Implementation Plan

### 1. Pre-list file paths once (Mac orchestration)
Single API call after rate-limit window to list one date folder → save JSON. Training uses CDN-only.

```bash
# scripts/hf-pre-list.sh
#!/usr/bin/env bash
set -euo pipefail

REPO="${1:-datasets/axentx/surrogate-ingest}"
DATE="${2:-$(date +%Y-%m-%d)}"
OUT="${3:-file-list.json}"

python3 - "$REPO" "$DATE" "$OUT" <<'PY'
import json, os, sys
from huggingface_hub import HfApi

repo, date = sys.argv[1], sys.argv[2]
api = HfApi()
# non-recursive per folder to avoid pagination explosion
folders = api.list_repo_tree(repo, path=date, recursive=False)
files = [f.rfilename for f in folders if f.type == "file"]
with open(sys.argv[3], "w") as f:
    json.dump({"repo": repo, "date": date, "files": files}, f, indent=2)
print(f"Listed {len(files)} files -> {sys.argv[3]}")
PY
```

### 2. CDN-only data loader (Lightning training)
Embed file list; fetch via `https://huggingface.co/datasets/.../resolve/main/...` with zero API calls.

```python
# surrogate/training/data/cdn_loader.py
import json, requests, io
from pathlib import Path
import pyarrow.parquet as pq
from typing import List, Dict

class CDNLoader:
    def __init__(self, file_list_path: str):
        with open(file_list_path) as f:
            self.manifest = json.load(f)
        self.base = f"https://huggingface.co/datasets/{self.manifest['repo']}/resolve/main"

    def stream_parquet(self, file_rel: str):
        url = f"{self.base}/{file_rel}"
        r = requests.get(url, timeout=30)
        r.raise_for_status()
        return pq.read_table(io.BytesIO(r.content)).to_pydict()

    def iter_rows(self, columns=("prompt", "response")):
        for fn in self.manifest["files"]:
            if not fn.endswith(".parquet"):
                continue
            tbl = self.stream_parquet(fn)
            for i in range(len(tbl[columns[0]])):
                yield {k: tbl[k][i] for k in columns}
```

### 3. Deterministic sibling-repo sharding for writes
Hash slug → pick sibling repo to spread HF commit cap (128/hr/repo) across 5 siblings = 640/hr aggregate.

```python
# surrogate/ingestion/shard.py
import hashlib

SIBLINGS = [
    "datasets/axentx/surrogate-ingest",
    "datasets/axentx/surrogate-ingest-s1",
    "datasets/axentx/surrogate-ingest-s2",
    "datasets/axentx/surrogate-ingest-s3",
    "datasets/axentx/surrogate-ingest-s4",
]

def pick_repo(slug: str) -> str:
    h = int(hashlib.sha256(slug.encode()).hexdigest(), 16)
    return SIBLINGS[h % len(SIBLINGS)]
```

### 4. Studio lifecycle guard (Lightning quota saver)
Reuse running studios; restart if idle-stop killed training.

```python
# surrogate/training/studio_guard.py
from lightning import Teamspace, Studio, Machine
import time

def get_or_start_studio(name: str, machine=Machine.L40S):
    team = Teamspace()
    for s in team.studios:
        if s.name == name:
            if s.status == "running":
                return s
            # idle-stop killed training; restart
            print(f"Restarting stopped studio {name}")
            s.start(machine=machine)
            return s
    return Studio(name=name, machine=machine, create_ok=True)

def run_with_guard(studio_name, script_path, args):
    studio = get_or_start_studio(studio_name)
    # studio.run() dies if studio stops; check status before each run
    if studio.status != "running":
        studio.start(machine=Machine.L40S)
        time.sleep(10)  # allow startup
    return studio.run(script_path, arguments=args)
```
