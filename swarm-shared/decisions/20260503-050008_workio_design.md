-- workio/server/src/db/schema.sql

-- Idempotency for LINE webhook events (per-event granularity)
CREATE TABLE IF NOT EXISTS line_webhook_events (
  idempotency_key TEXT NOT NULL PRIMARY KEY,
  user_id         INTEGER NOT NULL,
  event_type      TEXT NOT NULL,
  payload_hash    TEXT NOT NULL,
  created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Index for fast expiry (48h > LINE retry window)
CREATE INDEX IF NOT EXISTS idx_line_webhook_events_created_at
  ON line_webhook_events(created_at);

-- Attendance current state (single row per user per day)
CREATE TABLE IF NOT EXISTS attendances (
  user_id     INTEGER NOT NULL,
  date        TEXT NOT NULL,               -- ISO date 'YYYY-MM-DD'
  clocked_in  BOOLEAN NOT NULL DEFAULT FALSE,
  last_clock_at TIMESTAMPTZ,
  PRIMARY KEY (user_id, date)
);

-- Immutable event log (audit trail)
CREATE TABLE IF NOT EXISTS attendance_logs (
  id           BIGSERIAL PRIMARY KEY,
  user_id      INTEGER NOT NULL,
  event_type   TEXT NOT NULL,              -- 'clock_in', 'clock_out'
  created_at   TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  request_id   TEXT,                       -- trace id from webhook
  client_ip    INET,
  idempotency_key TEXT
);

### 2. Idempotency + signature guard middleware

```ts
// workio/server/src/middleware/lineWebhook.ts
import { Request, Response, NextFunction } from 'express';
import { Pool } from 'pg';
import crypto from 'crypto';

const pool = new Pool({ connectionString: process.env.DATABASE_URL });
const LINE_SIGNATURE_TTL_MS = 5 * 60 * 1000; // 5m max clock skew + retry window

function hashPayload(body: unknown): string {
  return crypto.createHash('sha256').update(JSON.stringify(body)).digest('hex');
}

// Deterministic per-event key: <dateMinute>-<userId>-<eventType>
// dateMinute = yyyyMMddHHmm from event timestamp (not server time)
function makeIdempotencyKey(
  ts: number,
  userId: number,
  eventType: string
): string {
  const d = new Date(ts);
  const minuteKey = d.toISOString().slice(0, 16).replace(/[-:T]/g, ''); // yyyyMMddHHmm
  return `${minuteKey}-${userId}-${eventType}`;
}

export function lineWebhookGuard(
  req: Request & {
    webhookMeta?: {
      userId: number;
      events: Array<{ type: string; timestamp: number }>;
    };
  },
  res: Response,
  next: NextFunction
) {
  const { userId, events } = req.body || {};
  if (!userId || !Array.isArray(events) || events.length === 0) {
    return res.status(400).json({ error: 'invalid webhook payload' });
  }

  const now = Date.now();
  for (const ev of events) {
    if (Math.abs(now - ev.timestamp) > LINE_SIGNATURE_TTL_MS) {
      return res.status(400).json({ error: 'webhook timestamp expired' });
    }
  }

  req.webhookMeta = { userId, events };
  next();
}

export async function deduplicateWebhook(
  req: Request & {
    webhookMeta?: {
      userId: number;
      events: Array<{ type: string; timestamp: number }>;
    };
    idempotencyResults?: Array<{
      eventType: string;
      key: string;
      isDuplicate: boolean;
    }>;
  },
  res: Response,
  next: NextFunction
) {
  const meta = req.webhookMeta;
  if (!meta) return next();

  const client = await pool.connect();
  try {
    await client.query('BEGIN');
    const results = [];

    for (const ev of meta.events) {
      const key = makeIdempotencyKey(ev.timestamp, meta.userId, ev.type);
      const dup = await client.query(
        'SELECT 1 FROM line_webhook_events WHERE idempotency_key = $1',
        [key]
      );
      const isDuplicate = dup.rows.length > 0;

      if (!isDuplicate) {
        await client.query(
          'INSERT INTO line_webhook_events(idempotency_key, user_id, event_type, payload_hash) VALUES ($1,$2,$3,$4)',
          [key, meta.userId, ev.type, hashPayload(req.body)]
        );
      }
      results.push({ eventType: ev.type, key, isDuplicate });
    }

    await client.query('COMMIT');
    req.idempotencyResults = results;
    next();
  } catch (err) {
    await client.query('ROLLBACK');
    next(err);
  } finally {
    client.release();
  }
}
```

### 3. Atomic clock transition (service)

```ts
// workio/server/src/services/attendanceService.ts
import { Pool } from 'pg';
const pool = new Pool({ connectionString: process.env.DATABASE_URL });

export async function processClockEvent(
  userId: number,
  eventType: 'clock_in' | 'clock_out',
  at: Date,
  requestId?: string,
  clientIp?: string,
  idempotencyKey?: string
) {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');
    const date = at.toISOString().slice(0, 10);

    // Lock current day row
    const curr = await client.query(
      `SELECT clocked_in
       FROM attendances
       WHERE user_id = $1 AND date = $2
       FOR UPDATE`,
      [userId, date]
    );

    const expectedClockedIn = eventType === 'clock_in' ? false : true;
    const row = curr.rows[0];

    // If state already matches, treat as idempotent success
    if (row && row.clocked_in === expectedClockedIn) {
      await client.query('ROLLBACK');
      return { ok: true, duplicate: true, message: `already ${eventType}` };
    }

    // Upsert state
    await client.query(
      `INSERT INTO attendances(user_id, date, clocked_in, last_clock_at)
       VALUES ($1, $2, $3, $4)
       ON CONFLICT (user_id, date)
       DO UPDATE SET clocked_in = EXCLUDED.clocked_in, last_clock_at = EXCLUDED.last_clock_at`,
      [userId, date, eventType === 'clock_in', at]
    );

    // Immutable audit log
    await client.query(
      'INSERT INTO attendance_logs(user_id, event_type, created_at, request_id, client_ip, idempotency_key) VALUES ($1,$2,$3,$4,$5,$6)',
      [userId, eventType, at, requestId, clientIp, idempotencyKey]
    );

    await client.query('COMMIT');
    return { ok: true };
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  } finally {
    client.release();
  }
}
```
