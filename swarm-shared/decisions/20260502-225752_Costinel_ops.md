# Costinel / ops

## Final Implementation Plan — Costinel Quality Increment (<2h)

**Goal:** Add a deterministic, read-only endpoint that surfaces today’s top hub from the knowledge graph as a cost-anomaly signal (Sense + Signal; no Execute).

- **Endpoint:** `GET /api/v1/cost-anomaly/signal/top-hub`
- **Behavior:**  
  - Queries the knowledge graph (or lightweight fallback) for today’s top hub (e.g., `"MOC"`) and strongest connected insights.  
  - Returns structured signal with `hub`, `score`, `insights[]`, `generatedAt`, `fallback`.  
  - Read-only, no side effects, no writes to Hugging Face or commits.  
  - Uses CDN-bypass pattern for any dataset references — zero API calls during request.  
  - Fast (<200ms p95) and cacheable (1–5m) to avoid graph scans on every call.

# costinel/api/routes/cost_anomaly.py
from fastapi import APIRouter, HTTPException
from datetime import datetime, timezone
from typing import List, Optional
from pydantic import BaseModel
from ..services.top_hub_signal import get_top_hub_signal

router = APIRouter(prefix="/cost-anomaly", tags=["cost-anomaly"])

class InsightItem(BaseModel):
    text: str
    weight: float
    related: Optional[List[str]] = []

class TopHubSignalResponse(BaseModel):
    hub: str
    score: float
    insights: List[InsightItem]
    generatedAt: str
    fallback: bool = False

@router.get("/signal/top-hub", response_model=TopHubSignalResponse)
async def top_hub_signal():
    """
    Sense + Signal: return today's top hub and strongest connected insights.
    Read-only. No Execute.
    """
    try:
        signal = await get_top_hub_signal()
        return signal
    except Exception as exc:
        # Fail-safe: deterministic fallback
        fallback_signal = TopHubSignalResponse(
            hub="MOC",
            score=0.85,
            insights=[
                InsightItem(
                    text="High-cost anomaly cluster detected in MOC services",
                    weight=0.85,
                    related=["cost", "MOC", "spikes"]
                )
            ],
            generatedAt=datetime.now(timezone.utc).isoformat(),
            fallback=True,
        )
        return fallback_signal

### 2) Lightweight graph query utility (CDN-safe, no HF API in request path)

```python
# costinel/api/services/top_hub_signal.py
import asyncio
import aiohttp
import json
import os
from datetime import datetime, timezone
from typing import List, Optional
from pydantic import BaseModel

class InsightItem(BaseModel):
    text: str
    weight: float
    related: Optional[List[str]] = None

class TopHubSignalResponse(BaseModel):
    hub: str
    score: float
    insights: List[InsightItem]
    generatedAt: str
    fallback: bool = False

# Prefer local graph export; CDN used only as secondary read (no auth/API calls).
KNOWLEDGE_GRAPH_PATH = os.getenv(
    "KNOWLEDGE_GRAPH_PATH",
    "/opt/axentx/Costinel/data/graph/latest.json"
)
HF_DATASET_REPO = os.getenv("HF_DATASET_REPO", "axentx/costinel-knowledge")
HF_FOLDER = os.getenv("HF_FOLDER", "batches/mirror-merged")

async def _fetch_graph_from_cdn(date_str: str) -> Optional[dict]:
    """
    CDN-only fetch (no HF API/auth). Pre-listed file paths should be embedded/cached.
    """
    url = (
        f"https://huggingface.co/datasets/{HF_DATASET_REPO}"
        f"/resolve/main/{HF_FOLDER}/{date_str}/graph.json"
    )
    try:
        async with aiohttp.ClientSession(timeout=aiohttp.ClientTimeout(total=4)) as session:
            async with session.get(url) as resp:
                if resp.status == 200:
                    return await resp.json()
    except Exception:
        pass
    return None

async def _local_graph_fallback() -> Optional[dict]:
    try:
        if os.path.isfile(KNOWLEDGE_GRAPH_PATH):
            with open(KNOWLEDGE_GRAPH_PATH, "r", encoding="utf-8") as f:
                return json.load(f)
    except Exception:
        pass
    return None

async def get_top_hub_signal() -> TopHubSignalResponse:
    """
    Determine today's top hub and strongest insights.
    Preference: CDN file for today -> local graph -> deterministic fallback.
    """
    utc_now = datetime.now(timezone.utc)
    date_str = utc_now.strftime("%Y-%m-%d")
    generated_at = utc_now.isoformat()

    graph = None
    # 1) Try CDN (zero API calls)
    graph = await _fetch_graph_from_cdn(date_str)
    # 2) Try local graph
    if graph is None:
        graph = await _local_graph_fallback()

    if graph and "hubs" in graph and isinstance(graph["hubs"], list) and len(graph["hubs"]) > 0:
        # Pick max by score/connections
        top = max(
            graph["hubs"],
            key=lambda h: float(h.get("score", h.get("connections", 0)))
        )
        hub_name = str(top.get("name", top.get("id", "MOC")))
        score = float(top.get("score", top.get("connections", 0.85)))

        raw_insights = top.get("insights", []) or top.get("edges", []) or [f"Strong hub: {hub_name}"]
        insights: List[InsightItem] = []
        for r in raw_insights[:5]:
            if isinstance(r, str):
                insights.append(InsightItem(text=r, weight=0.7, related=[hub_name]))
            elif isinstance(r, dict):
                insights.append(
                    InsightItem(
                        text=r.get("text", r.get("label", str(r))),
                        weight=float(r.get("weight", r.get("score", 0.7))),
                        related=r.get("related", [hub_name])
                    )
                )
        if not insights:
            insights = [InsightItem(text=f"Top hub detected: {hub_name}", weight=0.85, related=[hub_name])]

        return TopHubSignalResponse(
            hub=hub_name,
            score=score,
            insights=insights,
            generatedAt=generated_at,
            fallback=False,
        )

    # Deterministic fallback (no Execute)
    return TopHubSignalResponse(
        hub="MOC",
        score=0.85,
        insights=[
            InsightItem(
                text="High-cost anomaly cluster detected in MOC services",
                weight=0.85,
                related=["cost", "MOC", "spikes"]
            )
        ],
        generatedAt=generated_at,
        fallback=True,
    )
```
