# vanguard / ops

### Final Consolidated Implementation Plan

**Core diagnosis (accepted from all candidates):**  
Runtime `list_repo_tree`/`load_dataset` calls during ingestion/training expose the pipeline to 429s and non-reproducible runs. There is no content-addressed `{date}/{slug}` manifest, so jobs cannot guarantee CDN-only fetches or avoid repeated enumeration. Lightning quota is wasted by duplicate studio creation and missing idle-status checks before `.run()`. Wrapper script hygiene is also missing, risking silent cron failures.

**Single source of truth:**  
Create `/opt/axentx/vanguard/ops/manifest_builder.py` (~120 lines) that produces a deterministic, content-addressed manifest keyed by `{date}/{slug}` and embeds a pre-listed file inventory for CDN-only training.

### 1. `manifest_builder.py` (canonical implementation)

```python
#!/usr/bin/env python3
"""
manifest_builder.py
Builds a deterministic, content-addressed manifest for {date}/{slug}
and emits a file inventory for CDN-only training.

Usage:
    python3 manifest_builder.py --repo <repo_id> --date 2026-04-29 --slug v1
"""

from __future__ import annotations

import argparse
import hashlib
import json
import os
import subprocess
import sys
from datetime import datetime
from pathlib import Path
from typing import Any, Dict, List

import requests
from huggingface_hub import list_repo_tree

CDN_BASE = "https://huggingface.co/datasets"


def content_hash_for_path(path: str) -> str:
    """Stable content-address for a path component."""
    return hashlib.sha256(path.encode()).hexdigest()[:16]


def build_manifest(repo_id: str, date: str, slug: str, out_dir: Path) -> Dict[str, Any]:
    """
    Build manifest for repo_id/date/slug:
      - content-addressed key: {date}/{slug}
      - file inventory via list_repo_tree
      - CDN URLs for each file
    """
    base_path = f"{date}/{slug}"
    tree = list_repo_tree(repo_id, base_path, recursive=True)

    files: List[Dict[str, str]] = []
    for node in tree:
        if node.get("type") != "file":
            continue
        rel = node["path"]
        files.append(
            {
                "path": rel,
                "sha": node.get("oid", ""),
                "size": node.get("size", 0),
                "cdn_url": f"{CDN_BASE}/{repo_id}/resolve/main/{rel}",
            }
        )

    if not files:
        raise RuntimeError(f"No files found at {repo_id}/{base_path}")

    manifest: Dict[str, Any] = {
        "repo_id": repo_id,
        "date": date,
        "slug": slug,
        "key": base_path,
        "content_address": content_hash_for_path(base_path),
        "created_at": datetime.utcnow().isoformat() + "Z",
        "files": files,
        "total_files": len(files),
        "total_bytes": sum(f["size"] for f in files),
    }

    out_dir.mkdir(parents=True, exist_ok=True)
    manifest_path = out_dir / f"manifest_{manifest['content_address']}.json"
    manifest_path.write_text(json.dumps(manifest, indent=2))

    # Also emit compact file_paths.json for training scripts
    file_paths_path = out_dir / "file_paths.json"
    file_paths_path.write_text(json.dumps([f["path"] for f in files], indent=2))

    return manifest


def main() -> None:
    parser = argparse.ArgumentParser(description="Build manifest for training run.")
    parser.add_argument("--repo", required=True, help="HF dataset repo id")
    parser.add_argument("--date", required=True, help="Date folder (YYYY-MM-DD)")
    parser.add_argument("--slug", required=True, help="Run slug")
    parser.add_argument(
        "--out-dir",
        type=Path,
        default=Path("/opt/axentx/vanguard/ops/manifests"),
        help="Output directory for manifests",
    )
    args = parser.parse_args()

    try:
        manifest = build_manifest(args.repo, args.date, args.slug, args.out_dir)
        print(json.dumps(manifest, indent=2))
    except Exception as exc:
        print(f"ERROR: {exc}", file=sys.stderr)
        sys.exit(1)


if __name__ == "__main__":
    main()
```

### 2. Training script changes (`train.py`)

Embed the pre-listed inventory and enforce CDN-only fetches:

```python
#!/usr/bin/env python3
import json
import os
import requests
from pathlib import Path
from typing import List

MANIFEST_DIR = Path(os.getenv("MANIFEST_DIR", "/opt/axentx/vanguard/ops/manifests"))


def load_file_paths(date: str, slug: str) -> List[str]:
    """
    Load deterministic file inventory for {date}/{slug}.
    Raises if manifest missing (fail-fast, reproducible).
    """
    manifest_path = next(
        MANIFEST_DIR.glob(f"manifest_*.json"), None
    )  # prefer content-addressed lookup in prod
    fallback = MANIFEST_DIR / "file_paths.json"

    if manifest_path and manifest_path.exists():
        manifest = json.loads(manifest_path.read_text())
        if manifest.get("date") == date and manifest.get("slug") == slug:
            return [f["path"] for f in manifest["files"]]

    if fallback.exists():
        return json.loads(fallback.read_text())

    raise FileNotFoundError(
        f"No manifest/file_paths.json found for {date}/{slug} in {MANIFEST_DIR}"
    )


def download_via_cdn(file_path: str, repo_id: str) -> bytes:
    url = f"https://huggingface.co/datasets/{repo_id}/resolve/main/{file_path}"
    resp = requests.get(url, timeout=30)
    resp.raise_for_status()
    return resp.content
```

### 3. Lightning studio reuse + lifecycle guard

Replace blind `Studio(create_ok=True)` with deterministic reuse and idle-resilience:

```python
import lightning
from typing import Optional

STUDIO_NAME = "axentx-vanguard-trainer"


def get_or_create_studio() -> lightning.Studio:
    # Reuse running studio if present
    for s in lightning.Teamspace.studios:
        if s.name == STUDIO_NAME and s.status == "Running":
            return s

    # Create exactly one; create_ok guards duplicates at API level
    return lightning.Studio.create(
        STUDIO_NAME,
        machine=lightning.Machine.L40S,
    )


def ensure_running(studio: lightning.Studio) -> lightning.Studio:
    if studio.status != "Running":
        studio.target.start(machine=lightning.Machine.L40S)
    return studio


def run_job():
    studio = get_or_create_studio()
    studio = ensure_running(studio)
    # Guard before expensive .run()
    if studio.status != "Running":
        raise RuntimeError("Studio not running after ensure_running")
    lightning.run(studio, ...)
```
