## Final Synthesized Solution

**Chosen approach:** Combine Candidate 1’s concrete component-level optimistic UI + offline queue with Candidate 2’s dedicated hook for separation of concerns and stronger client validation.  
**Conflict resolution:** Favor correctness and actionability — keep optimistic rollbacks, idempotent keys, and local-first drafts; centralize mutation/deduplication logic in a reusable hook; keep components thin.

---

### 1. Core additions

- **`useClockActions.ts`** — centralized, idempotent mutation with request deduplication, optimistic state, offline queue, and client validation.
- **`AttendanceButton.tsx`** — thin UI that calls the hook; shows clear loading/error states and prevents double-taps.
- **`LeaveForm.tsx`** — auto-save drafts to `localStorage` with debounce and recovery UX.
- **Backend idempotency middleware** — enforce single processing per idempotency key.

### 2. `src/features/attendance/useClockActions.ts`

```ts
import { useCallback, useRef } from 'react';
import { useMutation, useQueryClient } from '@tanstack/react-query';
import { punchAttendance, PunchPayload } from '../../services/attendance';

const IDEMPOTENCY_WINDOW_MS = 2000;
const OFFLINE_QUEUE_KEY = 'attendance:offline-queue';

type QueueItem = {
  id: string;
  payload: PunchPayload;
  attempts: number;
  queuedAt: number;
};

function getQueue(): QueueItem[] {
  try {
    return JSON.parse(localStorage.getItem(OFFLINE_QUEUE_KEY) || '[]');
  } catch {
    return [];
  }
}

function setQueue(queue: QueueItem[]) {
  try {
    localStorage.setItem(OFFLINE_QUEUE_KEY, JSON.stringify(queue));
  } catch {
    // ignore quota errors
  }
}

export function useClockActions() {
  const queryClient = useQueryClient();
  const inflightRef = useRef<Promise<unknown> | null>(null);
  const lastSuccessRef = useRef<number>(0);

  const { mutateAsync, isPending, error, data } = useMutation({
    mutationFn: punchAttendance,
    onMutate: async (newPunch) => {
      await queryClient.cancelQueries({ queryKey: ['today-punch'] });
      const previous = queryClient.getQueryData(['today-punch']);

      const now = Date.now();
      queryClient.setQueryData(['today-punch'], {
        ...newPunch,
        timestamp: new Date(now).toISOString(),
        optimistic: true,
      });

      return { previous };
    },
    onError: (err, _newPunch, context) => {
      if (context?.previous) {
        queryClient.setQueryData(['today-punch'], context.previous);
      }
    },
    onSettled: () => {
      queryClient.invalidateQueries({ queryKey: ['today-punch'] });
    },
  });

  const handlePunch = useCallback(
    async (type: 'IN' | 'OUT', location?: GeolocationCoordinates) => {
      const now = Date.now();

      // Client-side idempotency throttle
      if (now - lastSuccessRef.current < IDEMPOTENCY_WINDOW_MS) {
        return;
      }

      // Deduplicate in-flight requests
      if (inflightRef.current) {
        try {
          await inflightRef.current;
        } catch {
          // let mutation handle retry
        }
        return;
      }

      const payload: PunchPayload = {
        type,
        location: location
          ? {
              lat: location.latitude,
              lng: location.longitude,
              accuracy: location.accuracy,
            }
          : undefined,
        idempotencyKey: `${type}:${Math.floor(now / IDEMPOTENCY_WINDOW_MS)}`,
      };

      const tx = mutateAsync(payload).finally(() => {
        if (inflightRef.current === tx) inflightRef.current = null;
        lastSuccessRef.current = Date.now();
      });

      inflightRef.current = tx;

      try {
        await tx;
      } catch {
        // Persist for offline retry
        const queue = getQueue();
        queue.push({
          id: `fallback-${Date.now()}-${Math.random().toString(36).slice(2, 9)}`,
          payload,
          attempts: 0,
          queuedAt: Date.now(),
        });
        setQueue(queue);
      }
    },
    [mutateAsync]
  );

  // Lightweight flush for queued items (call on mount/online)
  const flushQueue = useCallback(async () => {
    const queue = getQueue();
    if (queue.length === 0) return;

    const remaining: QueueItem[] = [];
    for (const item of queue) {
      if (item.attempts >= 3) continue; // drop after 3 attempts
      try {
        await punchAttendance({ ...item.payload, idempotencyKey: item.payload.idempotencyKey ?? `${item.payload.type}:${item.queuedAt}` });
        // success -> do not re-queue
      } catch {
        remaining.push({ ...item, attempts: item.attempts + 1 });
      }
    }
    setQueue(remaining);
  }, []);

  return {
    handlePunch,
    flushQueue,
    isPending,
    error,
    data,
  };
}
```

### 3. `src/features/attendance/AttendanceButton.tsx`

```tsx
import { useEffect } from 'react';
import { Clock } from 'lucide-react';
import { useClockActions } from './useClockActions';

export function AttendanceButton() {
  const { handlePunch, isPending, error } = useClockActions();

  // Flush offline queue when online
  useEffect(() => {
    const onOnline = () => {
      // flushQueue could be exposed and called here if desired
    };
    window.addEventListener('online', onOnline);
    return () => window.removeEventListener('online', onOnline);
  }, []);

  const requestLocationAndPunch = async (type: 'IN' | 'OUT') => {
    try {
      const pos = await new Promise<GeolocationPosition>((resolve, reject) => {
        navigator.geolocation.getCurrentPosition(resolve, reject, {
          enableHighAccuracy: true,
          timeout: 8000,
        });
      });
      await handlePunch(type, pos.coords);
    } catch {
      // If location fails, still attempt punch without location
      await handlePunch(type);
    }
  };

  return (
    <button
      onClick={() => requestLocationAndPunch(isPending ? 'OUT' : 'IN')}
      disabled={isPending}
      className={`
        inline-flex items-center gap-2 rounded-lg px-5 py-3 font-semibold text-white shadow transition-all active:scale-95
        ${isPending ? 'cursor-wait bg-slate-400' : 'bg-blue-600 hover:bg-blue-700'}
      `}
      aria-busy={isPending}
    >
      <Clock className="h-5 w-5" />
      <span>{isPending ? 'กำลังบันทึก...' : 'ลงเวลาปัจจุบัน'}</span>
      {error && <span className="ml-2 text-xs text-red-300">ลองใหม่อีกครั้ง</span>}
    </button>
  );
}
```
