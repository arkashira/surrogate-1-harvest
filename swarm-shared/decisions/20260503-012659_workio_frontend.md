# workio / frontend

## Final synthesized implementation

### 1. Diagnosis (merged)
- No optimistic UI → perceived latency and double-tap/duplicate requests.
- No idempotency or client-side deduplication → race punches when retries or parallel clicks occur.
- No pending/confirmed state in the list → users can’t tell what’s saved.
- No recovery path when server succeeds but client times out.
- Missing debounce/throttle and disabled-state protection on the button.

### 2. Scope
- `workio/src/features/punch/PunchButton.tsx` (primary clock UI)
- `workio/src/features/punch/PunchList.tsx` + `PunchItem.tsx` (pending/confirmed states)
- `workio/src/hooks/usePunch.ts` (reusable mutation hook)
- `workio/src/lib/api/punch.ts` + `request.ts` (idempotency header + dedupe)
- Optional: `workio/src/lib/dedupe.ts` (client-side in-flight dedupe)

```ts
export const request = axios.create({
  baseURL: import.meta.env.VITE_API_URL || "/api",
  headers: { "Content-Type": "application/json" },
  timeout: 15000,
});

// Attach Idempotency-Key when provided
request.interceptors.request.use((config) => {
  const key = (config.data as any)?.idempotencyKey;
  if (key) {
    config.headers = config.headers || {};
    config.headers["Idempotency-Key"] = key;
  }
  return config;
});
```

```ts
// /opt/axentx/workio/src/lib/dedupe.ts
const inFlight = new Set<string>();

export function acquireIntentLock(intent: string): boolean {
  if (inFlight.has(intent)) return false;
  inFlight.add(intent);
  return true;
}

export function releaseIntentLock(intent: string) {
  inFlight.delete(intent);
}
```

```ts
import { request } from "./request";
import type { PunchRequest } from "./types";

export async function punchIn(payload: PunchRequest) {
  return request.post("/punches/in", payload);
}

export async function punchOut(payload: PunchRequest) {
  return request.post("/punches/out", payload);
}

export async function fetchPunches() {
  return request.get<Array<PunchResponse>>("/punches").then((r) => r.data);
}

// Types
export type PunchRequest = {
  idempotencyKey: string;
  // include lat/lng if needed

  id: string;
  type: "in" | "out";
  timestamp: string;
  status?: "pending" | "confirmed" | "error";
```

#### hooks/usePunch.ts
```ts
// /opt/axentx/workio/src/hooks/usePunch.ts
import { useCallback, useRef } from "react";
import { useMutation, useQueryClient } from "@tanstack/react-query";
import { v4 as uuidv4 } from "uuid";
import { punchIn, punchOut } from "../lib/api/punch";
import { acquireIntentLock, releaseIntentLock } from "../lib/dedupe";
import type { PunchRequest } from "../lib/api/punch";

const INTENT_WINDOW_MS = 5000;

export function usePunch() {
  const queryClient = useQueryClient();
  const intentTimeouts = useRef<Record<string, NodeJS.Timeout>>({});

  const clearIntentLock = useCallback((intent: string) => {
    clearTimeout(intentTimeouts.current[intent]);
    delete intentTimeouts.current[intent];
    releaseIntentLock(intent);
  }, []);

  const mutation = useMutation({
    mutationFn: async ({ type, idempotencyKey }: { type: "in" | "out"; idempotencyKey: string }) => {
      const fn = type === "in" ? punchIn : punchOut;
      return fn({ idempotencyKey });
    },
    onMutate: async ({ type, idempotencyKey }) => {
      await queryClient.cancelQueries({ queryKey: ["punches"] });
      const previous = queryClient.getQueryData<Array<any>>(["punches"]);

      const optimistic = {
        id: idempotencyKey,
        type,
        timestamp: new Date().toISOString(),
        status: "pending" as const,
        location: null,
      };

      queryClient.setQueryData<Array<any>>(["punches"], (old) => [optimistic, ...(old || [])]);
      return { previous };
    },
    onError: (err, _variables, context) => {
      queryClient.setQueryData(["punches"], context?.previous);
    },
    onSettled: () => {
      queryClient.invalidateQueries({ queryKey: ["punches"] });
    },
  });

  const punch = useCallback(
    (type: "in" | "out") => {
      const intent = `punch-${type}`;
      // Client-side dedupe: prevent parallel requests within window
      if (!acquireIntentLock(intent)) return;

      const idempotencyKey = uuidv4();
      mutation.mutate(
        { type, idempotencyKey },
        {
          onSetttle: () => clearIntentLock(intent),
          onError: () => clearIntentLock(intent),
        }
      );

      // Auto-release lock after window to allow retry after failure
      intentTimeouts.current[intent] = setTimeout(() => {
        clearIntentLock(intent);
      }, INTENT_WINDOW_MS);
    },
    [mutation, clearIntentLock]
  );

  return { punch, isPending: mutation.isPending };
}
```

#### features/punch/PunchButton.tsx
```tsx
// /opt/axentx/workio/src/features/punch/PunchButton.tsx
import { useState } from "react";
import { usePunch } from "../../hooks/usePunch";
import { Button } from "../../components/ui/button";
import { useToast } from "../../components/ui/use-toast";
import { Clock } from "lucide-react";

export function PunchButton() {
  const { punch, isPending } = usePunch();
  const { toast } = useToast();
  const [localPending, setLocalPending] = useState<"in" | "out" | null>(null);

  const handlePunch = (type: "in" | "out") => {
    setLocalPending(type);
    punch(type);
  };

  return (
    <div className="flex flex-col items-center gap-3">
      <div className="flex gap-3">
        <Button
          size="lg"
          className="w-32 h-16 text-lg"
          onClick={() => handlePunch("in")}
          disabled={isPending || localPending === "in"}
          variant="default"
        >
          <Clock className="mr-2 h-5 w-5" />
          เข้างาน
        </Button>
        <Button
          size="lg"
          className="w-32 h-16 text-lg"
          onClick={() => handlePunch("out")}
          disabled={isPending || localPending === "out"}
          variant="secondary"
        >
          <Clock className="mr-2 h-5 w-5" />
          ออกงาน
        </Button>
      </div>

      {isPending && (
        <span className="text-sm text-muted-foreground animate-pulse">
          กำลังบันทึก...
        </span>
      )}
    </div>
  );
}
```
