# workio / design

## Final Synthesis (correct + actionable)

**Core problem**: duplicate punches and invalid state transitions caused by non-atomic, non-idempotent webhook handling and missing server-side constraints.  
**Goal**: one canonical, safe endpoint that LINE webhooks and clients can call concurrently without double-punches or invalid sequences.

CREATE TABLE IF NOT EXISTS punch_idempotency (

  employee_id     INTEGER NOT NULL,
  punch_id        INTEGER NOT NULL REFERENCES punches(id) ON DELETE CASCADE,

-- Immutable audit trail
CREATE TABLE IF NOT EXISTS punch_audit (
  id              BIGSERIAL PRIMARY KEY,
  punch_id        INTEGER NOT NULL REFERENCES punches(id) ON DELETE CASCADE,
  employee_id     INTEGER NOT NULL,
  action          TEXT NOT NULL, -- 'create','update','delete'

  new_value       JSONB,
  performed_by    TEXT,
  ip_address      INET,
  user_agent      TEXT,

  ON punches (employee_id, punch_date, punch_type)

### 2. Service: atomic, idempotent, state-safe punch handler

File: `workio/server/src/services/punchService.ts`

```ts
import { Pool } from 'pg';

const pool = new Pool({ connectionString: process.env.DATABASE_URL });

export type PunchType = 'clock_in' | 'clock_out';

export type PunchInput = {
  employeeId: number;
  punchType: PunchType;
  timestamp?: Date;
  latitude?: number;
  longitude?: number;
  idempotencyKey?: string;
  performedBy?: string;
  ipAddress?: string;
  userAgent?: string;
};

// Validation stubs (replace with real shift/geofence checks)
function validateTimeWindow(ts: Date): void {
  // e.g. compare to scheduled shift; throw if outside policy
}
function validateGeofence(lat?: number, lon?: number): void {
  // e.g. ensure within GEO_RADIUS_METERS of expected location
}

export async function recordPunch(input: PunchInput) {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    const ts = input.timestamp || new Date();
    const punchDate = new Date(ts.getFullYear(), ts.getMonth(), ts.getDate());

    // 1) Idempotency (exact replay returns existing punch)
    if (input.idempotencyKey) {
      const idem = await client.query(
        'SELECT punch_id FROM punch_idempotency WHERE idempotency_key = $1',
        [input.idempotencyKey]
      );
      if (idem.rows.length > 0) {
        const punch = await client.query('SELECT * FROM punches WHERE id = $1', [idem.rows[0].punch_id]);
        await client.query('ROLLBACK');
        return { punch: punch.rows[0], created: false };
      }
    }

    // 2) State transition rules (strict)
    if (input.punchType === 'clock_in') {
      const last = await client.query(
        `SELECT punch_type FROM punches
         WHERE employee_id = $1 AND deleted_at IS NULL
         ORDER BY punch_date DESC, created_at DESC LIMIT 1`,
        [input.employeeId]
      );
      if (last.rows.length > 0 && last.rows[0].punch_type === 'clock_in') {
        throw new Error('Employee already clocked in. Clock out first.');
      }
    } else {
      const lastClockIn = await client.query(
        `SELECT id FROM punches
         WHERE employee_id = $1 AND punch_type = 'clock_in' AND deleted_at IS NULL
         ORDER BY punch_date DESC, created_at DESC LIMIT 1`,
        [input.employeeId]
      );
      if (lastClockIn.rows.length === 0) {
        throw new Error('No active clock-in to clock out.');
      }
    }

    // 3) Policy checks
    validateTimeWindow(ts);
    validateGeofence(input.latitude, input.longitude);

    // 4) Create punch (rely on DB unique index for final safety)
    const insertPunch = await client.query(
      `INSERT INTO punches (employee_id, punch_date, punch_type, timestamp, latitude, longitude, created_at)
       VALUES ($1, $2, $3, $4, $5, $6, NOW()) RETURNING *`,
      [input.employeeId, punchDate, input.punchType, ts, input.latitude || null, input.longitude || null]
    );
    const punch = insertPunch.rows[0];

    // 5) Idempotency record
    if (input.idempotencyKey) {
      await client.query(
        'INSERT INTO punch_idempotency (idempotency_key, employee_id, punch_id) VALUES ($1, $2, $3)',
        [input.idempotencyKey, input.employeeId, punch.id]
      );
    }

    // 6) Audit
    await client.query(
      `INSERT INTO punch_audit (punch_id, employee_id, action, new_value, performed_by, ip_address, user_agent)
       VALUES ($1, $2, 'create', $3, $4, $5, $6)`,
      [punch.id, input.employeeId, JSON.stringify(punch), input.performedBy || 'line-webhook', input.ipAddress, input.userAgent]
    );

    await client.query('COMMIT');
    return { punch, created: true };
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  } finally {
    client.release();
  }
}
```

### 3. Route: single endpoint for LINE + client

File: `workio/server/src/routes/punchRoutes.ts`

```ts
import express from 'express';
import { recordPunch } from '../services/punchService';

const router = express.Router();

router.post('/api/punches', async (req, res) => {
  try {
    const {
      employeeId,
      punchType,
      timestamp,
      latitude,
      longitude,
      idempotencyKey,
      performedBy = 'line-webhook',
    } = req.body;

    if (!employeeId || !punchType) {
      return res.status(400).json({ error: 'employeeId and punchType required' });
    }

    const result = await recordPunch({
      employeeId,
      punchType,
      timestamp: timestamp ? new Date(timestamp) : undefined,
      latitude,
      longitude,
      idempotencyKey,
      performedBy,
      ipAddress: req.ip,
      userAgent: req.get('User-Agent') || undefined,
    });

    return res.status(result.created ? 201 : 200).json({ punch: result.punch });
  } catch (err: any) {
    // Duplicate detected by unique index or business rule violation
    if (err.code === '23505' || /already clocked|No active clock-in/i.test(err.message)) {
      return res.status(409).json({ error: err.message });
    }
    console.error(err);
    return res.status(500).json({ error: 'Internal server error' });
  }
});

export default router;
```
