# workio / design

## Final Synthesis — Corrected, Prioritized, Actionable

**Core principles used to resolve contradictions:**
- Correctness first: enforce idempotency and state-machine correctness at the earliest safe point.
- Transactional integrity: one business operation = one DB transaction that covers side-effects (notifications, audit).
- Realtime consistency: webhook-driven changes must invalidate/push updates to the UI.
- Reuse and safety: signature verification and idempotency are middleware, not per-route logic.

---

## 1. Diagnosis (merged, de-duplicated)

- **LINE replay/duplicates**: no idempotency guard; LINE retries on timeouts/network errors can double-process clock/leave/OT events and corrupt state.
- **Implicit state machine**: transitions scattered in handlers; invalid transitions (clock-in while clocked-in) handled ad-hoc, risking inconsistent records.
- **Non-transactional multi-entity updates**: clock-in/out may create attendance + notification + tenant_audit; partial failures cause divergence between DB and LINE/frontend.
- **Silent/latent UI state**: LINE-driven changes do not push to frontend; users see stale data until hard refresh, inviting duplicate requests.
- **Signature verification reuse gap**: signature check implemented inline in routes; easy to forget on new endpoints and inconsistent error handling.

## 2. DB Changes (single source of truth)

```sql
-- Idempotency dedupe for LINE webhook deliveries (24h TTL recommended at app layer)
CREATE TABLE IF NOT EXISTS line_delivery_dedupe (
  delivery_id      VARCHAR(128) PRIMARY KEY,
  received_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
CREATE INDEX IF NOT EXISTS idx_line_delivery_dedupe_received_at
  ON line_delivery_dedupe (received_at);

-- Attendance state machine enforcement
ALTER TABLE attendances
  ADD COLUMN IF NOT EXISTS status VARCHAR(16) NOT NULL DEFAULT 'clocked_in'
  CHECK (status IN ('clocked_in', 'clocked_out', 'on_leave', 'on_ot'));

-- Prevent multiple active clock-ins for same user (soft constraint + app logic)
CREATE UNIQUE INDEX IF NOT EXISTS idx_attendances_user_active_clock_in
  ON attendances (user_id, date)
  WHERE status = 'clocked_in';
```

## 3. Idempotency + Signature Middleware (reusable)

`server/src/middleware/lineAuth.ts`

```ts
import { Request, Response, NextFunction } from 'express';
import crypto from 'crypto';
import { pool } from '../db';

const LINE_SIGNATURE_HEADER = 'x-line-signature';
const DELIVERY_ID_HEADER = 'x-line-delivery-id';

export function lineAuth(channelSecret: string) {
  return async (req: Request, res: Response, next: NextFunction) => {
    const signature = req.get(LINE_SIGNATURE_HEADER);
    const body = typeof req.body === 'string' ? req.body : JSON.stringify(req.body);
    const deliveryId = req.get(DELIVERY_ID_HEADER) || crypto.createHash('sha256').update(body).digest('hex');

    // 1) Idempotency check (fast fail for duplicates)
    try {
      const { rowCount } = await pool.query(
        `INSERT INTO line_delivery_dedupe (delivery_id, received_at) VALUES ($1, NOW()) ON CONFLICT DO NOTHING`,
        [deliveryId]
      );
      if (rowCount === 0) {
        // Duplicate delivery — accept but skip processing
        return res.status(200).json({ ok: true, duplicate: true });
      }
    } catch (err) {
      // If dedupe fails, continue but log — prefer availability
      console.warn('[lineAuth] dedupe insert failed', err);
    }

    // 2) Signature verification
    if (!signature || !channelSecret) {
      return res.status(401).json({ error: 'Unauthorized' });
    }

    const expected = crypto
      .createHmac('sha256', channelSecret)
      .update(body, 'utf8')
      .digest('base64');

    if (signature !== expected) {
      return res.status(401).json({ error: 'Invalid signature' });
    }

    // Attach deliveryId for handlers
    (req as any).lineDeliveryId = deliveryId;
    next();
  };
}
```

## 4. Transactional Service with State Machine

`server/src/services/lineWebhookService.ts`

```ts
import { PoolClient, QueryResult } from 'pg';
import { pool } from '../db';
import { v4 as uuidv4 } from 'uuid';

export type ClockAction = 'clock_in' | 'clock_out';
export type RequestAction = 'leave_request' | 'ot_request';

async function insertNotification(client: PoolClient, userId: string, type: string, message: string): Promise<QueryResult> {
  return client.query(
    `INSERT INTO notifications (id, user_id, type, message, created_at)
     VALUES ($1, $2, $3, $4, NOW())`,
    [uuidv4(), userId, type, message]
  );
}

export async function handleClockAction(
  client: PoolClient,
  userId: string,
  action: ClockAction,
  timestamp: Date,
  location?: { lat: number; lon: number }
) {
  if (action === 'clock_in') {
    const { rowCount } = await client.query(
      `SELECT 1 FROM attendances WHERE user_id = $1 AND status = 'clocked_in' LIMIT 1`,
      [userId]
    );
    if (rowCount && rowCount > 0) {
      throw new Error('Already clocked in');
    }

    await client.query(
      `INSERT INTO attendances (id, user_id, date, clock_in_at, location, status)
       VALUES ($1, $2, $3::date, $4, $5, 'clocked_in')`,
      [uuidv4(), userId, timestamp, timestamp, JSON.stringify(location)]
    );
    await insertNotification(client, userId, 'clock', 'Clock-in recorded');
  } else {
    const { rows } = await client.query(
      `SELECT id FROM attendances WHERE user_id = $1 AND status = 'clocked_in' ORDER BY clock_in_at DESC LIMIT 1`,
      [userId]
    );
    if (!rows.length) {
      throw new Error('No active clock-in to clock out');
    }

    await client.query(
      `UPDATE attendances SET clock_out_at = $1, status = 'clocked_out' WHERE id = $2`,
      [timestamp, rows[0].id]
    );
    await insertNotification(client, userId, 'clock', 'Clock-out recorded');
  }
}

export async function handleClockActionTx(
  userId: string,
  action: ClockAction,
  timestamp: Date,
  location?: { lat: number; lon: number }
) {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');
    await handleClockAction(client, userId, action, timestamp, location);
    // Add tenant_audit or other side-effects here within same transaction
    await client.query('COMMIT');
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  } finally {
    client.release();
  }
}
```
