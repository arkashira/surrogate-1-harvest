# ── helpers ──────────────────────────────────────────────────────────────────

def _now_iso() -> str:
    return datetime.now(timezone.utc).isoformat()

def _slug_for_path(path: str) -> str:
    """Stable slug for sharding: repo-relative path without extension."""
    p = Path(path)
    stem = p.stem or p.name
    return f"{p.parent.as_posix()}/{stem}".lstrip("/")

def _shard_of(slug: str, total: int) -> int:
    """Deterministic shard assignment by slug hash."""
    digest = hashlib.md5(slug.encode()).hexdigest()
    return int(digest, 16) % total

def _http_get(url: str, *, timeout: int = 60) -> bytes:
    """CDN fetch with retry/backoff."""
    for attempt in range(5):
        try:
            resp = requests.get(url, timeout=timeout, headers={"User-Agent": "axentx-surrogate-ingest/1.0"})
            if resp.status_code == 404:
                raise FileNotFoundError(f"CDN 404: {url}")
            resp.raise_for_status()
            return resp.content
        except Exception as exc:
            wait = 2 ** attempt
            log.warning("GET %s failed (attempt %s): %s — retry in %ss", url, attempt + 1, exc, wait)
            time.sleep(wait)
    raise RuntimeError(f"Failed to fetch after retries: {url}")

# ── schema projection ────────────────────────────────────────────────────────

def _project_to_pair(batch: pa.Table) -> Iterable[Dict[str, str]]:
    """
    Project heterogeneous HF parquet/json/csv to {prompt, response}.
    Accepts common field aliases.
    """
    col_names = {c.lower() for c in batch.column_names}

    prompt_col = None
    response_col = None

    for c in batch.column_names:
        cl = c.lower()
        if cl in {"prompt", "instruction", "question", "input", "user"}:
            prompt_col = c
        if cl in {"response", "completion", "answer", "output", "assistant", "text"}:
            response_col = c

    if prompt_col is None or response_col is None:
        # Fallback: first text column as prompt, second as response
        text_cols = [c for c in batch.column_names if pa.types.is_string(batch.schema.field(c).type)]
        if len(text_cols) >= 2:
            prompt_col, response_col = text_cols[0], text_cols[1]
        elif len(text_cols) == 1:
            prompt_col, response_col = text_cols[0], text_cols[0]
        else:
            raise ValueError(f"No prompt/response columns found in schema: {batch.column_names}")

    for i in range(batch.num_rows):
        prompt = str(batch.column(prompt_col)[i]).strip()
        response = str(batch.column(response_col)[i]).strip()
        if prompt and response:
            yield {"prompt": prompt, "response": response}

# ── file loader ──────────────────────────────────────────────────────────────

def _load_file_cdn(repo: str, path: str) -> Iterable[Dict[str, str]]:
    url = f"{CDN_BASE}/{path}"
    ext = Path(path).suffix.lower()
    raw = _http_get(url)

    try:
        if ext == ".parquet":
            table = pq.read_table(pa.BufferReader(raw))
        elif ext == ".json" or ext == ".jsonl":
            # Arrow JSON reader expects full JSON array; fallback to line-by-line for .jsonl
            if ext == ".jsonl" or raw.lstrip().startswith(b"{"):
                # line-delimited
                lines = raw.splitlines()
                rows = [json.loads(l) for l in lines if l.strip()]
                table = pa.Table.from_pylist(rows)
            else:
                table = paj.read_json(pa.BufferReader(raw))
        elif ext in {".csv", ".tsv"}:
            parse_options = pcsv.ParseOptions(delimiter="\t" if ext == ".tsv" else ",")
            table = pcsv.read_csv(pa.BufferReader(raw), parse_options=parse_options)
        else:
            log.warning("Unsupported extension %s for %s — skipping", ext, path)
            return
    except Exception as exc:
        log.error("Failed to decode %s: %s", path, exc)
        return

    try:
        yield from _project_to_pair(table)
    except Exception as exc:
        log.error("Projection failed for %s: %s", path, exc)

# ── dedup bridge ─────────────────────────────────────────────────────────────

def _md5_hex(payload: str) -> str:
    return hashlib.md5(payload.encode()).hexdigest()

def _dedup_key(prompt: str, response: str) -> str:
    # Deterministic canonical form
    canonical = json.dumps({"prompt": prompt, "response": response}, ensure_ascii=False, sort_keys=True)
    return _md5_hex(canonical)

def _sqlite_dedup_factory(db_path: Path):
    db_path.parent.mkdir(parents=True, exist_ok=True)
    conn = sqlite3.connect(db_path, isolation_level="EXCLUSIVE")
    conn.execute("PRAGMA journal_mode=WAL")
    conn.execute("CREATE TABLE IF NOT EXISTS seen (key TEXT PRIMARY KEY)")
    conn.commit()

    def exists(key: str) -> bool:
        cur = conn.execute("SELECT 1 FROM seen WHERE key = ?", (key,))
        return cur.fetchone() is not None

    def add(key: str) -> None:
        conn.execute("INSERT OR IGNORE INTO seen (key) VALUES (?)", (key,))
        conn.commit()

    def close() -> None:
        conn.close()

    return exists, add, close
