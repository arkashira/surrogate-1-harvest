## Highest-Value Incremental Improvement (<2h)

**Problem**: Surrogate training hits HF API 429 rate limits and Lightning idle timeouts kill 24/7 autonomous training.

**Solution**: CDN-only parquet loader + Lightning idle-resilient runner. Eliminates HF API calls during training and auto-recovers from idle stops.

---

### Implementation Plan

#### 1. CDN-only Parquet Loader (surrogate/training/cdm_loader.py)
- Single `list_repo_tree` call from Mac (after rate-limit window) → JSON file
- Lightning training uses CDN URLs only (zero API calls)
- Projects mixed-schema files to `{prompt, response}` at parse time

#### 2. Lightning Idle-Resilient Runner (surrogate/training/run_surrogate_training.py)
- Checks studio status before each run
- Auto-restarts stopped studios with L40S
- Uses existing running studio if available (quota save)

#### 3. Mac Orchestration Script (scripts/list_hf_files.py)
- Runs after rate-limit window clears
- Generates `file_list.json` for CDN-only training

#### `surrogate/training/cdm_loader.py`
```python
import json
import pyarrow.parquet as pq
import requests
from pathlib import Path
from typing import List, Dict, Iterator
import io

class CDNParquetLoader:
    """CDN-only parquet loader - zero HF API calls during training."""
    
    CDN_BASE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"
    
    def __init__(self, repo: str, file_list_path: Path):
        self.repo = repo
        with open(file_list_path) as f:
            self.files = json.load(f)  # [{"path": "...", "size": ...}, ...]
    
    def _download_cdn(self, rel_path: str) -> bytes:
        url = self.CDN_BASE.format(repo=self.repo, path=rel_path)
        resp = requests.get(url, timeout=30)
        resp.raise_for_status()
        return resp.content
    
    def _project_to_pair(self, table) -> List[Dict[str, str]]:
        """Project any schema to {prompt, response} only."""
        rows = []
        df = table.to_pandas()
        
        # Handle known schema variants
        if "prompt" in df.columns and "response" in df.columns:
            for _, row in df.iterrows():
                rows.append({"prompt": str(row["prompt"]), "response": str(row["response"])})
        elif "instruction" in df.columns and "output" in df.columns:
            for _, row in df.iterrows():
                rows.append({"prompt": str(row["instruction"]), "response": str(row["output"])})
        elif "text" in df.columns:
            # Split Q/A pairs if in single text column
            for _, row in df.iterrows():
                text = str(row["text"])
                if "### Response:" in text:
                    parts = text.split("### Response:")
                    rows.append({"prompt": parts[0].strip(), "response": parts[1].strip()})
        else:
            # Fallback: use first two text-like columns
            text_cols = [c for c in df.columns if df[c].dtype == "object"]
            if len(text_cols) >= 2:
                for _, row in df.iterrows():
                    rows.append({"prompt": str(row[text_cols[0]]), "response": str(row[text_cols[1]])})
        
        return rows
    
    def stream_pairs(self, limit: int = None) -> Iterator[Dict[str, str]]:
        """Stream prompt/response pairs via CDN only."""
        count = 0
        for file_info in self.files:
            if limit and count >= limit:
                break
            
            try:
                data = self._download_cdn(file_info["path"])
                table = pq.read_table(io.BytesIO(data))
                pairs = self._project_to_pair(table)
                
                for pair in pairs:
                    if pair["prompt"] and pair["response"]:
                        yield pair
                        count += 1
                        if limit and count >= limit:
                            break
            except Exception as e:
                print(f"Skipping {file_info['path']}: {e}")
                continue
```

#### `surrogate/training/run_surrogate_training.py`
```python
import time
from pathlib import Path
from lightning import Lightning, Teamspace, Machine, Studio
from surrogate.training.cdm_loader import CDNParquetLoader

# Lightning setup - use Lambda prod for H200, fallback to public
LIGHTNING_CLOUDS = ["lightning-lambda-prod", "lightning-public-prod"]

def get_running_studio(name: str) -> Studio:
    """Reuse existing running studio to save quota."""
    for cloud in LIGHTNING_CLOUDS:
        try:
            lightning = Lightning(cloud=cloud)
            teamspace = Teamspace(lightning)
            for studio in teamspace.studios:
                if studio.name == name and studio.status == "running":
                    print(f"Reusing running studio: {name} on {cloud}")
                    return studio
        except Exception:
            continue
    return None

def ensure_studio_running(name: str = "surrogate-train") -> Studio:
    """Ensure studio is running, restart if idle-stopped."""
    studio = get_running_studio(name)
    
    if studio:
        if studio.status == "stopped":
            print(f"Studio {name} was idle-stopped. Restarting...")
            studio.start(machine=Machine.L40S)
        return studio
    
    # Create new
    lightning = Lightning(cloud="lightning-lambda-prod")
    studio = lightning.studio(
        name=name,
        machine=Machine.L40S,
        create_ok=True
    )
    return studio

def run_training():
    """Run CDN-only training with idle-resilience."""
    studio = ensure_studio_running()
    
    # Verify studio is actually running
    max_wait = 300  # 5 min
    waited = 0
    while studio.status != "running" and waited < max_wait:
        print(f"Studio status: {studio.status}, waiting...")
        time.sleep(10)
        waited += 1
        studio.refresh()
    
    if studio.status != "running":
        raise RuntimeError("Studio failed to start")
    
    # Run training script via CDN loader
    script = """
import sys
sys.path.insert(0, "/app")

from surrogate.training.surrogate_trainer import train_model
from pathlib import Path

# CDN-only training - no HF API calls
file_list = Path("/app/data/file_list.json")
train_model(file_list=file_list, epochs=1)
"""
    
    result = studio.run(
        notebook="surrogate/training/train.ipynb",
        environment="lightning-ai",
        script=script
    )
    return result

if __name__ == "__main__":
    run_training()
```

#### `scripts/list_hf_files.py` (Mac orchestration)
```python
#!/usr/bin/env python3
"""
Mac-only orchestration: list HF files once after rate-limit window.
Generates file_list.json for CDN-only training.
"""
import json
from pathlib import Path
from huggingface_hub import list_repo_tree

REPO = "your-org/surrogate-dataset"
OUTPUT = Path("/opt/axentx/airship/surrogate/data/file_list.json")

def main():
    # Single API call - use after 429 window clears
    files = list_repo_tree(
        repo_id=REPO,
        path="batches/mirror-merged",  # Date folder
        recursive=False  # Avoid 100x pagination
    )
    
    file_list = [
        {"path": f.path, "size": f.size}
        for f in files
        if f.path.endswith(".parquet")
    ]
    
    OUTPUT.parent.mkdir(exist_ok=True)
    with open(OUTPUT, "w") as f:
        json.dump(file_list, f, indent=2)
    
    print(f"Saved {len(file_list)} files to {OUTPUT}")
    print("CDN-only training ready - zero API calls during training")

if __name__ == "__main__":
    main()
```
