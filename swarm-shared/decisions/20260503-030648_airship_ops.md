# airship / ops

## Highest-value incremental improvement (≤2h)

**Ship:** A zero-HF-API-during-training data loader + Lightning idle-resilient runner for Surrogate.

- Eliminates HF 429s during training by using CDN-only fetches.
- Survives Lightning idle-stop without losing progress.
- Keeps iteration cycle ≤2 min (no re-list, no re-upload, no recreation).

---

## Implementation plan

1. **Pre-list once, save to JSON** (Mac orchestration, one-time or cron)
   - Use `list_repo_tree(path, recursive=False)` for a single date folder.
   - Emit `file_list.json` with CDN URLs (`resolve/main/...`).

2. **Surrogate-side: CDN-only DataLoader**
   - Accept `file_list.json` at startup.
   - Stream files via `requests.get(cdn_url, timeout=30)` with retries.
   - Parse to `{prompt, response}` only (drop extra cols/attribution).
   - PyTorch IterableDataset to avoid materializing all files in RAM.

3. **Lightning runner: idle-resilient**
   - Before `.run()`, check studio status; if stopped, restart on `L40S` (free-tier fallback).
   - Resume training from latest checkpoint (Lightning automatic if `resume_from_checkpoint`).
   - Short `max_epochs=1` per run so idle-stop rarely interrupts an epoch.

4. **Validation**
   - Dry-run DataLoader for 100 files; confirm zero `huggingface_hub` calls.
   - Confirm studio restart + resume works after simulated stop.

import json, os

api = HfApi()
repo = os.environ["REPO"]
date = os.environ["DATE"]

# single folder, non-recursive to avoid pagination/rate
tree = api.list_repo_tree(repo=repo, path=date, recursive=False)
files = [
    {
        "filename": f.path,
        "cdn_url": f"https://huggingface.co/datasets/{repo}/resolve/main/{f.path}"
    }
    for f in tree if f.type == "file"
]

with open(os.environ["OUT"], "w") as fp:
    json.dump(files, fp, indent=2)

print(f"Wrote {len(files)} files to {os.environ['OUT']}")
PY

### 2) Surrogate: CDN-only IterableDataset

```python
# surrogate/data/cdn_dataset.py
import io
import json
import time
from pathlib import Path
from typing import Iterator, Dict, Any

import torch
import requests
from datasets import Features, Value
from torch.utils.data import IterableDataset

CDN_TIMEOUT = 30
MAX_RETRIES = 3
BACKOFF = 5

FEATURES = Features({"prompt": Value("string"), "response": Value("string")})

class CDNParquetIterableDataset(IterableDataset):
    def __init__(self, file_list_path: str, start_idx: int = 0):
        super().__init__()
        self.file_list_path = Path(file_list_path)
        with self.file_list_path.open() as f:
            self.files = json.load(f)
        self.start_idx = start_idx

    def _stream_rows(self, cdn_url: str) -> Iterator[Dict[str, Any]]:
        for attempt in range(1, MAX_RETRIES + 1):
            try:
                resp = requests.get(cdn_url, timeout=CDN_TIMEOUT, stream=True)
                resp.raise_for_status()
                buf = io.BytesIO(resp.content)
                import pyarrow.parquet as pq
                table = pq.read_table(buf, columns=["prompt", "response"])
                df = table.to_pandas()
                for _, row in df.iterrows():
                    yield {"prompt": str(row["prompt"]), "response": str(row["response"])}
                return
            except Exception as exc:
                if attempt == MAX_RETRIES:
                    raise
                time.sleep(BACKOFF * attempt)

    def __iter__(self) -> Iterator[Dict[str, Any]]:
        worker_info = torch.utils.data.get_worker_info()
        if worker_info is None:
            files = self.files[self.start_idx :]
        else:
            per_worker = len(self.files) // worker_info.num_workers
            files = self.files[
                worker_info.id * per_worker : (worker_info.id + 1) * per_worker
            ]

        for item in files:
            yield from self._stream_rows(item["cdn_url"])
```

### 3) Surrogate: idle-resilient Lightning runner

```python
# surrogate/run/train_cdn.py
import os
import time
from pathlib import Path

import lightning as L
from lightning.pytorch.callbacks import ModelCheckpoint
from torch.utils.data import DataLoader

from surrogate.data.cdn_dataset import CDNParquetIterableDataset
from surrogate.model.surrogate_module import SurrogateModule  # your existing module
from surrogate.lightning_utils import get_studio, L40S, LAMBDA_FREE

def main() -> None:
    # 1) pick latest file list
    list_dir = Path("configs/file_lists")
    latest = max(list_dir.glob("*.json"), key=lambda p: p.stat().mtime)
    print(f"Using file list: {latest}")

    dataset = CDNParquetIterableDataset(file_list_path=str(latest))
    loader = DataLoader(dataset, batch_size=8, num_workers=2)

    # 2) studio: reuse or create; survive idle-stop
    teamspace = L.Teamspace()
    studio = None
    for s in teamspace.studios:
        if s.name == "surrogate-train" and s.status == "Running":
            studio = s
            print("Reusing running studio")
            break

    if studio is None:
        # free-tier fallback -> L40S in lightning-public-prod
        studio = L.Studio(
            name="surrogate-train",
            machine=L40S,
            cloud=LAMBDA_FREE,
            create_ok=True,
        )

    model = SurrogateModule(learning_rate=1e-4)

    checkpoint_cb = ModelCheckpoint(
        dirpath="checkpoints",
        filename="surrogate-{epoch}-{step}",
        save_top_k=3,
        monitor="train_loss",
        mode="min",
    )

    trainer = L.Trainer(
        max_epochs=1,  # short runs survive idle-stop
        devices=1,
        accelerator="gpu",
        callbacks=[checkpoint_cb],
        log_every_n_steps=10,
    )

    # 3) resilient run: restart studio if stopped
    while True:
        try:
            if studio.status != "Running":
                print("Studio stopped; restarting on L40S")
                studio.start(machine=L40S, cloud=LAMBDA_FREE)
                time.sleep(30)  # allow boot
            trainer.fit(model, train_dataloaders=loader)
            break
        except Exception as exc:
            print(f"Training interrupted: {exc}")
            time.sleep(10)
            continue

if __name__ == "__main__":
    main()
```
