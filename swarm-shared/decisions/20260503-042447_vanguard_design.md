# vanguard / design

## 1. Diagnosis
- Runtime `load_dataset`/recursive file enumeration triggers HF API 429s and non-reproducible epochs.
- No content-addressed manifest (file list + SHA256) → CDN-only fetching impossible; training still hits API during data loading.
- Mixed-schema heterogeneous repo files risk `pyarrow.CastError` when loaded via streaming/multiprocessing.
- No deterministic repo-to-sibling mapping for writes → HF commit cap (128/hr/repo) throttles ingestion throughput.
- Mac/local runs may attempt `from_pretrained` or heavy data ops locally instead of delegating to Lightning/Kaggle/Cerebras.

## 2. Proposed change
Create `/opt/axentx/vanguard/design/manifest.py` (single file, <150 LOC) that:
- Exposes `build_manifest(repo, date_folder)` → saves `manifest-{date}.json` with `{path, sha256, size}` for one date folder (non-recursive tree + CDN URLs).
- Exposes `repo_to_sibling(slug, n=5)` → deterministic hash-mod mapping for HF sibling repos.
- Exposes `safe_rows(paths, project_to=None)` → per-file `hf_hub_download` + line-streamed projection to `{prompt, response}` to avoid schema issues.
- Adds a tiny CLI (`python -m vanguard.design.manifest build ...`) so Mac orchestration can pre-list once and embed the manifest in training scripts.

## 3. Implementation
File: `/opt/axentx/vanguard/design/manifest.py`

```python
#!/usr/bin/env python3
"""
Content-addressed manifest builder for HF datasets.
Avoids runtime load_dataset/API recursion and enables CDN-only training fetches.
"""
from __future__ import annotations

import hashlib
import json
import os
import sys
from pathlib import Path
from typing import Dict, List, Optional, Sequence

import requests

HF_API_BASE = "https://huggingface.co/api"
HF_CDN_BASE = "https://huggingface.co/datasets"

# Deterministic sibling spread for HF commit-cap avoidance
HF_SIBLINGS = [
    "axentx/vanguard-mirror-0",
    "axentx/vanguard-mirror-1",
    "axentx/vanguard-mirror-2",
    "axentx/vanguard-mirror-3",
    "axentx/vanguard-mirror-4",
]


def _hf_headers(token: Optional[str] = None) -> Dict[str, str]:
    token = token or os.getenv("HF_TOKEN") or os.getenv("HF_API_TOKEN")
    return {"Authorization": f"Bearer {token}"} if token else {}


def repo_to_sibling(slug: str, n: int = 5) -> str:
    """Deterministic repo -> sibling mapping (0..n-1)."""
    idx = int(hashlib.sha256(slug.encode()).hexdigest(), 16) % max(n, len(HF_SIBLINGS))
    return HF_SIBLINGS[idx % len(HF_SIBLINGS)]


def list_date_folder(repo: str, date_folder: str, token: Optional[str] = None) -> List[Dict]:
    """
    List files in a single date folder (non-recursive) via repo tree.
    Returns list[dict] with at least 'path', 'size', 'type'.
    """
    url = f"{HF_API_BASE}/models/{repo}/tree/{date_folder}"
    resp = requests.get(url, headers=_hf_headers(token), timeout=30)
    if resp.status_code == 404:
        # try datasets namespace
        url = f"{HF_API_BASE}/datasets/{repo}/tree/{date_folder}"
        resp = requests.get(url, headers=_hf_headers(token), timeout=30)
    resp.raise_for_status()
    items = resp.json()
    # Filter only files (exclude directories)
    files = [i for i in items if i.get("type") == "file"]
    return files


def file_sha256(path: Path, chunk_kb: int = 1024) -> str:
    h = hashlib.sha256()
    with open(path, "rb") as f:
        for chunk in iter(lambda: f.read(chunk_kb * 1024), b""):
            h.update(chunk)
    return h.hexdigest()


def build_manifest(
    repo: str,
    date_folder: str,
    out_dir: Path,
    token: Optional[str] = None,
) -> Path:
    """
    Build content-addressed manifest for one date folder.
    Manifest schema:
      {
        "repo": "...",
        "folder": "...",
        "files": [
          {"path": "...", "sha256": "...", "size": 123, "cdn_url": "..."}
        ]
      }
    """
    out_dir = Path(out_dir)
    out_dir.mkdir(parents=True, exist_ok=True)

    files = list_date_folder(repo, date_folder, token=token)
    manifest = {
        "repo": repo,
        "folder": date_folder,
        "files": [],
    }

    for f in files:
        p = f["path"]
        manifest["files"].append(
            {
                "path": p,
                "sha256": "",  # will be filled if local file provided; otherwise left blank
                "size": f.get("size", 0),
                "cdn_url": f"{HF_CDN_BASE}/{repo}/resolve/main/{p}",
            }
        )

    out_path = out_dir / f"manifest-{date_folder.replace('/', '-')}.json"
    out_path.write_text(json.dumps(manifest, indent=2))
    return out_path


def safe_rows(
    paths: Sequence[str],
    repo: str,
    project_to: Optional[List[str]] = None,
    token: Optional[str] = None,
):
    """
    Yield rows from HF files one-by-one with projection to selected fields.
    Avoids load_dataset and mixed-schema pyarrow issues.
    """
    import io
    import json as jsonlib

    from huggingface_hub import hf_hub_download

    project_to = project_to or ["prompt", "response"]
    headers = _hf_headers(token)

    for p in paths:
        local_path = hf_hub_download(repo_id=repo, filename=p, repo_type="dataset")
        # Heuristic: try JSONL first, then JSON array, then single JSON object per file
        try:
            with open(local_path, "r", encoding="utf-8") as f:
                for line in f:
                    line = line.strip()
                    if not line:
                        continue
                    try:
                        obj = jsonlib.loads(line)
                    except jsonlib.JSONDecodeError:
                        continue
                    projected = {k: obj.get(k) for k in project_to}
                    # skip rows missing all projected keys if at least one key requested
                    if project_to and all(projected.get(k) is None for k in project_to):
                        continue
                    yield projected
        except UnicodeDecodeError:
            # fallback: try parquet-like binary? skip for now
            continue


def main() -> int:
    import argparse

    parser = argparse.ArgumentParser(description="Build HF manifest for CDN-only training.")
    sub = parser.add_subparsers(dest="cmd", required=True)

    build = sub.add_parser("build", help="Build manifest for a date folder.")
    build.add_argument("--repo", required=True, help="HF dataset repo (user/repo).")
    build.add_argument("--folder", required=True, help="Date folder path (e.g. batches/mirror-merged/2026-05-03).")
    build.add_argument("--out", default=".", help="Output directory for manifest JSON.")
    build.add_argument("--token", default=None, help="HF token (optional; can use HF_TOKEN env).")

    args = parser.parse_args()

    if args.cmd == "build":
        out = build_manifest(args.repo, args.folder, Path(args.out), token=args.token)
        print(f"Manifest written to {out}")
        return 0

    return 1


if __name__ == "__main__":
    sys.exit(main())
```
