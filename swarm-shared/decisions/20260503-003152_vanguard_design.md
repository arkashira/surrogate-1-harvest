# vanguard / design

## 1. Diagnosis
- No persisted `(repo, dateFolder)` manifest exists → every training run re-enumerates via authenticated HF API → quota burn + 429 risk.
- Data loader likely uses recursive `list_repo_files` or `load_dataset` during training → amplifies rate-limit exposure and wastes quota.
- Lightning Studio reuse is not enforced → idle-stop kills training and quota is wasted on repeated studio creation.
- Mixed-schema files from `dataset-mirror` are probably uploaded with extra cols (`source`, `ts`) → downstream schema errors (pyarrow CastError).
- No CDN-only fetch strategy in training script → authenticated API calls during data loading instead of zero-auth CDN downloads.

## 2. Proposed change
Create `/opt/axentx/vanguard/training/manifest.py` and update `/opt/axentx/vanguard/training/train.py` (or create it) to:
- Add `build_manifest(repo, date_folder)` that calls HF API **once** to list top-level folder and persists `manifests/{repo}/{date_folder}.json`.
- Update data loader to read only from CDN URLs listed in the manifest (zero HF API calls during training).
- Add `get_or_create_studio(name)` that reuses a running Lightning Studio and restarts it if idle-stopped.
- Project mirror files to `{prompt, response}` only and store attribution in filename pattern `batches/mirror-merged/{date}/{slug}.parquet`.

# /opt/axentx/vanguard/training/manifest.py
#!/usr/bin/env python3
"""
Build and cache (repo, date_folder) manifests for CDN-only training.
"""
import json
import os
from pathlib import Path
from typing import Dict, List

import requests
from hugginggingface import HfApi  # if available; else use requests

MANIFEST_ROOT = Path(__file__).parent.parent / "manifests"
MANIFEST_ROOT.mkdir(parents=True, exist_ok=True)

HF_API_BASE = "https://huggingface.co/api"
HF_CDN_BASE = "https://huggingface.co/datasets"


def list_date_folder(repo: str, date_folder: str, token: str) -> List[str]:
    """Single authenticated API call to list files in repo/date_folder (non-recursive)."""
    url = f"{HF_API_BASE}/datasets/{repo}/tree/{date_folder}"
    headers = {"Authorization": f"Bearer {token}"}
    resp = requests.get(url, headers=headers, timeout=30)
    if resp.status_code == 429:
        # respect rate limit: wait 360s as per pattern
        import time
        time.sleep(360)
        resp = requests.get(url, headers=headers, timeout=30)
    resp.raise_for_status()
    items = resp.json()
    # keep only files (exclude subfolders) and use path relative to repo root
    return [item["path"] for item in items if item.get("type") == "file"]


def build_manifest(repo: str, date_folder: str, token: str) -> Dict:
    """Return and persist manifest for (repo, date_folder)."""
    manifest_path = MANIFEST_ROOT / repo / f"{date_folder}.json"
    manifest_path.parent.mkdir(parents=True, exist_ok=True)

    # reuse if already exists and is non-empty
    if manifest_path.exists() and manifest_path.stat().st_size > 0:
        with open(manifest_path) as f:
            return json.load(f)

    paths = list_date_folder(repo, date_folder, token)
    # build CDN-only URLs (no auth required)
    entries = []
    for p in paths:
        cdn_url = f"{HF_CDN_BASE}/{repo}/resolve/main/{p}"
        entries.append({
            "path": p,
            "cdn_url": cdn_url,
            "repo": repo,
            "date_folder": date_folder
        })

    manifest = {
        "repo": repo,
        "date_folder": date_folder,
        "created_by": "build_manifest",
        "entries": entries
    }

    with open(manifest_path, "w") as f:
        json.dump(manifest, f, indent=2)
    return manifest


def load_manifest(repo: str, date_folder: str) -> Dict:
    """Load cached manifest; raises FileNotFoundError if not built."""
    manifest_path = MANIFEST_ROOT / repo / f"{date_folder}.json"
    with open(manifest_path) as f:
        return json.load(f)

class CDNTextDataset(IterableDataset):
    """Yield {prompt, response} pairs from CDN files listed in manifest."""
    def __init__(self, manifest_path: str, split: str = "train"):
        with open(manifest_path) as f:
            manifest = json.load(f)
        self.entries = manifest["entries"]
        # simple deterministic split by hash
        if split == "train":
            self.entries = [e for i, e in enumerate(self.entries) if i % 10 != 0]
        else:
            self.entries = [e for i, e in enumerate(self.entries) if i % 10 == 0]

    def __iter__(self):
        for entry in self.entries:
            # download via CDN (no auth) — streaming-friendly
            import requests
            resp = requests.get(entry["cdn_url"], stream=True, timeout=30)
            resp.raise_for_status()
            # project to {prompt, response} only (mirror files may contain extra cols)
            # Assuming line-delimited JSONL with fields; adapt per actual schema.
            for line in resp.iter_lines(decode_unicode=True):
                if not line:
                    continue
                try:
                    obj = json.loads(line)
                except Exception:
                    continue
                # keep only prompt/response; ignore source/ts
                prompt = obj.get("prompt") or obj.get("input") or ""
                response = obj.get("response") or obj.get("output") or ""
                if prompt and response:
                    yield {"prompt": prompt, "response": response}

def get_or_create_studio(name: str, machine=L.Machine.L40S) -> L.Studio:
    """Reuse running studio; restart if idle-stopped."""
    teamspace = L.Teamspace()
    for s in teamspace.studios:
        if s.name == name:
            if s.status == "running":
                return s
            # idle-stop killed training; restart
            s.start(machine=machine)
            return s
    return L.Studio(
        name=name,
        machine=machine,
        create_ok=True
    )
