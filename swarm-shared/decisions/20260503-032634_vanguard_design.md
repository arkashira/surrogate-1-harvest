# vanguard / design

## 1. Diagnosis
- No CDN-first manifest: ingestion/training can still trigger `list_repo_tree`/`load_dataset` at runtime → 429s and non-reproducible runs.
- No content-addressed pinning: training uses logical paths instead of exact file hashes → silent drift across dataset updates.
- Lightning quota burn: scripts create new studios instead of reusing running ones; idle-stop kills training without restart logic.
- Mixed-schema ingestion risk: raw files with metadata columns can hit PyArrow `CastError` during HF dataset loads.
- No local file-list cache: each worker re-enumerates repo tree; wastes API quota and introduces flakiness.

## 2. Proposed change
Create `/opt/axentx/vanguard/design/cdn_manifest.py` + `/opt/axentx/vanguard/train/train.py` patch:
- Add `build_cdn_manifest(repo, date_folder)` → writes `manifest-{sha256(date_folder)}.json` with CDN URLs and sha256 per file.
- Update `train.py` to accept `--manifest` and use CDN-only fetches (zero HF API calls during training).
- Add studio reuse + idle-stop resilience in launcher.

### File: `/opt/axentx/vanguard/design/cdn_manifest.py`

```python
#!/usr/bin/env python3
"""
Build a CDN-first, content-addressed manifest for a HuggingFace dataset repo.
Usage:
    python cdn_manifest.py --repo datasets/my-org/surrogate-1 --date 2026-04-29 --out manifests/
"""
import argparse
import hashlib
import json
import os
import sys
from datetime import datetime
from pathlib import Path
from typing import Dict, List

import requests
from huggingface_hub import HfApi, list_repo_tree

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def sha256_hex(s: str) -> str:
    return hashlib.sha256(s.encode("utf-8")).hexdigest()

def build_cdn_manifest(repo: str, date_folder: str, out_dir: Path) -> Path:
    api = HfApi()
    # Single API call (do this from Mac when rate-limit window is clear)
    tree = list_repo_tree(repo=repo, path=date_folder, recursive=False)
    files = [t for t in tree if t.type == "file"]

    entries: List[Dict] = []
    for f in files:
        cdn_url = CDN_TEMPLATE.format(repo=repo, path=f.path)
        # Lightweight HEAD to get size and etag (optional; can skip if too heavy)
        try:
            r = requests.head(cdn_url, timeout=10)
            r.raise_for_status()
            size = int(r.headers.get("content-length", -1))
            etag = r.headers.get("etag", "").strip('"')
        except Exception:
            size = -1
            etag = ""

        entries.append(
            {
                "path": f.path,
                "cdn_url": cdn_url,
                "size": size,
                "etag": etag,
                "sha256": sha256_hex(f.path),
            }
        )

    manifest = {
        "repo": repo,
        "date_folder": date_folder,
        "created_at": datetime.utcnow().isoformat() + "Z",
        "entries": entries,
        "manifest_sha256": None,
    }
    manifest_bytes = json.dumps(manifest, sort_keys=True).encode("utf-8")
    manifest["manifest_sha256"] = hashlib.sha256(manifest_bytes).hexdigest()

    out_dir.mkdir(parents=True, exist_ok=True)
    out_path = out_dir / f"manifest-{sha256_hex(date_folder)}.json"
    out_path.write_bytes(json.dumps(manifest, indent=2, sort_keys=True))
    return out_path

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Build CDN manifest")
    parser.add_argument("--repo", required=True, help="HF dataset repo")
    parser.add_argument("--date", required=True, help="Date folder (e.g. 2026-04-29)")
    parser.add_argument("--out", default="manifests", help="Output directory")
    args = parser.parse_args()

    p = build_cdn_manifest(args.repo, args.date, Path(args.out))
    print(f"Manifest written: {p}")
```

### Launcher snippet (studio reuse + idle resilience)
```python
# launcher.py  (run from Mac)
from lightning import Lightning, Teamspace, Machine
import time

lightning = Lightning()
teamspace = Teamspace()

def get_or_start_studio(name: str):
    for s in teamspace.studios:
        if s.name == name and s.status == "Running":
            return s
    # create or restart
    studio = lightning.Studio(
        name=name,
        machine=Machine.L40S,
        create_ok=True,
    )
    studio.start()
    return studio

def run_training_with_retry(manifest_path: str, max_retries: int = 3):
    studio = get_or_start_studio("vanguard-train")
    for attempt in range(max_retries):
        if studio.status != "Running":
            studio.start(machine=Machine.L40S)
        try:
            job = studio.run(
                "python train/train.py --manifest " + manifest_path,
                wait=True,
            )
            return job
        except Exception as e:
            print(f"Attempt {attempt+1} failed: {e}")
            time.sleep(60)
            studio.refresh()
    raise RuntimeError("Training failed after retries")
```
