## Final Synthesis & Actionable Implementation

**Chosen approach**: CDN-first ingestion + Lightning Studio reuse (Candidate 1)  
**Why**: It is the only proposal that directly eliminates HF API 429s *and* Lightning quota waste with concrete, copy-paste code and a clear runbook. Candidate 2 was cut off and offered no executable fix.

**Core correctness decisions**:
- Use `list_repo_tree(recursive=False)` per date folder (not recursive) to avoid pagination/429.
- Resolve HF file access via **CDN URLs** (`https://huggingface.co/datasets/.../resolve/main/...`) — no auth, no rate limits.
- Project schema to `{prompt, response}` only at load time to tolerate mixed/evolving parquet schemas.
- Reuse or restart existing Lightning Studio instead of blind recreation (saves ~80 hrs/mo).
- Validate manifest existence before training launch; fail fast if missing.

### 2. CDN-first file listing (`scripts/list_hf_files.py`)
```python
#!/usr/bin/env python3
"""
List HF dataset files via tree API (no recursive pagination).
Save to JSON for CDN-only training.
"""
import json
import os
from datetime import datetime, timedelta
from huggingface_hub import HfApi

HF_REPO = "axentx/surrogate-training-data"
OUTPUT_DIR = "surrogate/file_manifests"

def list_date_folder(date_str: str):
    api = HfApi()
    tree = api.list_repo_tree(
        repo_id=HF_REPO,
        path=date_str,
        recursive=False  # Avoid pagination + reduce 429 surface
    )

    files = []
    for item in tree:
        if item.type == "file" and item.path.endswith((".parquet", ".jsonl")):
            cdn_url = f"https://huggingface.co/datasets/{HF_REPO}/resolve/main/{item.path}"
            files.append({
                "path": item.path,
                "cdn_url": cdn_url,
                "size": getattr(item, "size", None),
            })
    return files

def main():
    os.makedirs(OUTPUT_DIR, exist_ok=True)

    date_str = os.getenv(
        "HF_DATE_FOLDER",
        (datetime.now() - timedelta(days=1)).strftime("%Y-%m-%d")
    )

    print(f"Listing files for {date_str}...")
    files = list_date_folder(date_str)

    manifest = {
        "date": date_str,
        "repo": HF_REPO,
        "total_files": len(files),
        "files": files,
        "generated_at": datetime.now().isoformat(),
    }

    output_path = os.path.join(OUTPUT_DIR, f"manifest_{date_str}.json")
    with open(output_path, "w") as f:
        json.dump(manifest, f, indent=2)

    print(f"Saved manifest: {output_path} ({len(files)} files)")
    return 0

if __name__ == "__main__":
    raise SystemExit(main())
```

### 3. CDN-only data loader (`surrogate/training/data_loader.py`)
```python
import json
from io import BytesIO
from typing import Dict, Iterator
import pyarrow.parquet as pq
import requests

class CDNDataLoader:
    """Load training data via CDN URLs — zero HF API calls during training."""

    def __init__(self, manifest_path: str):
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.files = self.manifest["files"]

    def stream_file(self, file_entry: Dict) -> BytesIO:
        resp = requests.get(file_entry["cdn_url"], timeout=30)
        resp.raise_for_status()
        return BytesIO(resp.content)

    def load_dataset(self) -> Iterator[Dict]:
        for file_entry in self.files:
            try:
                buf = self.stream_file(file_entry)
                table = pq.read_table(buf)
                for batch in table.to_batches():
                    df = batch.to_pandas()
                    for _, row in df.iterrows():
                        yield {
                            "prompt": row.get("prompt") or row.get("input") or "",
                            "response": row.get("response") or row.get("output") or "",
                        }
            except Exception as exc:
                print(f"Skipping {file_entry['path']}: {exc}")
                continue
```

### 4. Lightning Studio reuse (`surrogate/training/lightning_launcher.py`)
```python
#!/usr/bin/env python3
import os
import sys
from lightning_sdk import Studio, Teamspace, Machine
from typing import Optional

def get_or_create_studio(studio_name: str, machine_type: str = "L40S") -> Studio:
    """
    Reuse running studio; restart stopped; else create.
    Avoids quota waste from repeated recreation.
    """
    teamspace = Teamspace()
    for studio in teamspace.studios:
        if studio.name == studio_name:
            if studio.status == "running":
                print(f"Reusing running studio: {studio_name}")
                return studio
            print(f"Restarting stopped studio: {studio_name}")
            studio.start(machine=Machine(machine_type))
            return studio

    print(f"Creating new studio: {studio_name}")
    return Studio(name=studio_name, machine=Machine(machine_type), create_ok=True)

def launch_training(
    script_path: str,
    manifest_path: str,
    studio_name: str = "surrogate-train",
    machine_type: str = "L40S",
):
    if not os.path.isfile(manifest_path):
        print(f"Manifest not found: {manifest_path}", file=sys.stderr)
        raise SystemExit(1)

    studio = get_or_create_studio(studio_name, machine_type)

    if studio.status != "running":
        print("Studio not running; starting...")
        studio.start(machine=Machine(machine_type))

    run = studio.run(
        script=open(script_path).read(),
        requirements=["pyarrow", "requests", "huggingface_hub"],
        name="surrogate-cdn-training",
    )
    print(f"Training started: {run.url}")
    return run

if __name__ == "__main__":
    # Example usage:
    # python3 lightning_launcher.py surrogate/training/train.py surrogate/file_manifests/manifest_2026-05-02.json
    if len(sys.argv) < 3:
        print("Usage: lightning_launcher.py <train_script> <manifest_path> [studio_name]")
        raise SystemExit(1)

    launch_training(
        script_path=sys.argv[1],
        manifest_path=sys.argv[2],
        studio_name=sys.argv[3] if len(sys.argv) > 3 else "surrogate-train",
    )
```
