# airship / quality

## Final consolidated implementation (best of both proposals)

**Chosen scope**: Ship a CDN-safe, zero-runtime-dependency `airship discover` orchestrator (Bash-first) that deterministically emits `status.json` + `manifest.json` for local or CI use and CDN upload. Frontends fetch these files directly (no SSR/backend) for global health and cache-busting.

**Why this now**
- Removes runtime coupling and HF API/rate-limit risk.
- Guarantees deterministic, cache-friendly artifacts (CDN-safe).
- Fits existing decisions: static-first, Bash-first orchestration, Mac=CLI only.
- Highest value in <2h with concrete, testable outputs.

### Single source of truth: `/opt/axentx/airship/scripts/discover.sh`

```bash
#!/usr/bin/env bash
# airship discover — produce status.json + manifest.json
# Usage: discover.sh [--root <path>] [--out <dir>] [--date-folder YYYY-MM-DD]
# Outputs deterministic JSON files safe for CI/CD and CDN upload.
# Exit codes: 0=ok, 1=validation failed, 2=execution error
set -euo pipefail

# ---- Args ----
ROOT=""
OUT_DIR=""
DATE_FOLDER="$(date -u +"%Y-%m-%d")"

while [[ $# -gt 0 ]]; do
  case "$1" in
    --root) ROOT="$2"; shift 2 ;;
    --out)  OUT_DIR="$2"; shift 2 ;;
    --date-folder) DATE_FOLDER="$2"; shift 2 ;;
    *) echo "Unknown option: $1" >&2; exit 2 ;;
  esac
done

: "${ROOT:=$(cd "$(dirname "$0")/../.." && pwd)}"
: "${OUT_DIR:=$(pwd)/dist}"

mkdir -p "$OUT_DIR"
TIMESTAMP="$(date -u +"%Y-%m-%dT%H:%M:%SZ")"

# ---- Helpers ----
file_hash() {
  if command -v shasum >/dev/null 2>&1; then
    shasum -a 256 "$1" | awk '{print $1}'
  else
    sha256sum "$1" | awk '{print $1}'
  fi
}

stable_json_from_pairs() {
  # Expects sorted KEY=VALUE lines on stdin; outputs stable JSON object.
  jq -Rn '
    [inputs | select(length>0) | split("=";2)]
    | map({(.[0]): .[1]})
    | add // {}
  '
}

# ---- File inventory (top-level deterministic snapshot) ----
# Save a lightweight file-list.json (top-level names only) for reproducibility.
# Keep this minimal and deterministic (sorted).
find "$ROOT" -maxdepth 1 -mindepth 1 -printf '%f\n' | sort | jq -R . | jq -s '.' > "$OUT_DIR/file-list.json"

# ---- Build service states ----
# Known services (name:port). Deterministic ordering by name.
readarray -t SERVICES <<< "$(printf '%s\n' "arkship:8000" "surrogate:8001" | sort)"

service_states="{}"
for svc_pair in "${SERVICES[@]}"; do
  name="${svc_pair%%:*}"
  port="${svc_pair##*:}"
  pid=""
  container=""
  status="unknown"

  if pgrep -f "$name" >/dev/null 2>&1; then
    pid="$(pgrep -f "$name" | head -n1)"
    status="running"
  fi

  if command -v docker >/dev/null 2>&1; then
    cid=$(docker ps --filter "name=$name" --format "{{.ID}}" 2>/dev/null | head -n1 || true)
    if [ -n "${cid:-}" ]; then
      container="$cid"
      status="running"
    fi
  fi

  service_states=$(jq -n \
    --argjson obj "$service_states" \
    --arg name "$name" \
    --arg port "$port" \
    --arg pid "${pid:-}" \
    --arg container "${container:-}" \
    --arg status "$status" \
    '$obj + { ($name): { port: $port, pid: (if $pid=="" then null else $pid end), container: (if $container=="" then null else $container end), status: $status } }')
done

# ---- Git state ----
GIT_BRANCH=""
GIT_COMMIT=""
GIT_DIRTY="false"
if [ -d "$ROOT/.git" ]; then
  GIT_BRANCH=$(cd "$ROOT" && git rev-parse --abbrev-ref HEAD 2>/dev/null || echo "")
  GIT_COMMIT=$(cd "$ROOT" && git rev-parse HEAD 2>/dev/null || echo "")
  if ! git -C "$ROOT" diff --quiet 2>/dev/null; then
    GIT_DIRTY="true"
  fi
fi

# ---- Critical file hashes ----
CRITICAL_PATHS=(
  "README.md"
  "arkship/README.md"
  "surrogate/README.md"
  "docker-compose.microservices.yml"
  "scripts/discover.sh"
)

critical_hashes="{}"
for p in "${CRITICAL_PATHS[@]}"; do
  full="$ROOT/$p"
  if [ -f "$full" ]; then
    h=$(file_hash "$full")
    critical_hashes=$(jq -n --argjson obj "$critical_hashes" --arg k "$p" --arg v "$h" '$obj + { ($k): $v }')
  fi
done

# ---- File checksums (streamed, deterministic order) ----
# Compute checksums for all regular files under ROOT (excluding .git) and sort paths.
# Use null-delimited find + sort to handle special filenames safely.
checksum_pairs=$(mktemp)
find "$ROOT" -type f -not -path '*/.git/*' -print0 | sort -z | while IFS= read -r -d '' f; do
  rel="${f#$ROOT/}"
  h=$(file_hash "$f")
  printf '%s=%s\n' "$rel" "$h"
done > "$checksum_pairs"

# ---- Status JSON ----
FILE_COUNT=$(wc -l < "$checksum_pairs" | tr -d ' ')
TOTAL_BYTES=$(awk -F= '{print $1}' "$checksum_pairs" | while read -r p; do stat -c%s "$ROOT/$p" 2>/dev/null || echo 0; done | awk '{s+=$1} END{print s}')

CDN_BASE="https://huggingface.co/datasets/axentx/airship/resolve/main/${DATE_FOLDER}/"

STATUS_JSON=$(jq -n \
  --arg ts "$TIMESTAMP" \
  --arg df "$DATE_FOLDER" \
  --arg root "$ROOT" \
  --argjson file_count "$FILE_COUNT" \
  --argjson total_bytes "$TOTAL_BYTES" \
  --argjson services "$service_states" \
  --argjson critical "$critical_hashes" \
  --arg cdn_base "$CDN_BASE" \
  --arg branch "$GIT_BRANCH" \
  --arg commit "$GIT_COMMIT" \
  --arg dirty "$GIT_DIRTY" \
  '{
    generated_at: $ts,
    date_folder: $df,
    project_root: $root,
    file_count: $file_count,
    total_bytes: $total_bytes,
    checksums: (reduce (inputs | split("=";2)) as [$k,$v] ({}; .[$k] = $v)),
    cdn_base: $cdn_base,
    git: {
      branch: (if $branch=="" then null else $branch end),
      commit: (if $commit=="" then null else $commit end),
      dirty: ($dirty == "true")
    },
    services: $services,
    critical_hashes: $critical,
    health: {
      status: (if ($services | to_entries | map(.value.status) | map(. == "running") | all) then "ok" elif ($services | length) > 0 then "degraded" else "unknown" end),
      checks: []
    },
    schema: "airship-discover-v1"
  }' < "$checksum_pairs")

echo "$STATUS_JSON" | jq . > "$OUT_DIR/status.json"

# ---- Manifest JSON ----
MANIFEST_JSON=$(jq -n \
  --arg ts "$TIMESTAMP" \
  --arg version "$GIT_COMMIT
