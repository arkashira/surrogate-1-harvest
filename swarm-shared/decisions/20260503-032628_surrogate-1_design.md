def hf_list_files(date_folder: str) -> List[str]:
    """Single API call: list files in date folder (non-recursive)."""
    retry = 0
    while True:
        try:
            items = list_repo_tree(
                repo_id=DATASET_REPO,
                path=date_folder,
                repo_type="dataset",
                recursive=False,
            )
            files = [it.rfilename for it in items if it.type == "file"]
            log.info("Listed %d files in %s", len(files), date_folder)
            return files
        except Exception as exc:
            # Handle 429
            if hasattr(exc, "response") and getattr(exc.response, "status_code", None) == 429:
                retry += 1
                wait = 360
                log.warning("HF API 429 (retry %d), waiting %ds", retry, wait)
                time.sleep(wait)
                continue
            log.exception("Failed to list repo tree")
            raise

def shard_assign(files: List[str]) -> List[str]:
    """Deterministic shard assignment by slug hash."""
    assigned = []
    for f in sorted(files):
        # derive slug from filename (remove extension)
        slug = Path(f).stem
        h = int(hashlib.sha256(slug.encode()).hexdigest(), 16)
        if h % SHARD_TOTAL == SHARD_ID:
            assigned.append(f)
    log.info("Shard %d assigned %d/%d files", SHARD_ID, len(assigned), len(files))
    return assigned

def download_via_cdn(rel_path: str, timeout: int = 30) -> Optional[bytes]:
    """Download via CDN (no auth counted against API rate limit)."""
    url = f"{CDN_ROOT}/{rel_path}"
    try:
        resp = requests.get(url, timeout=timeout)
        resp.raise_for_status()
        return resp.content
    except Exception as exc:
        log.warning("CDN download failed %s: %s", rel_path, exc)
        return None

def parse_parquet_to_pairs(data: bytes) -> List[Dict[str, str]]:
    """Project parquet bytes to [{prompt, response}] only."""
    import pyarrow.parquet as pq
    import io

    try:
        table = pq.read_table(io.BytesIO(data))
    except Exception as exc:
        log.warning("Parquet decode failed: %s", exc)
        return []

    pairs = []
    # Heuristic: find prompt/response-like columns
    cols = table.column_names
    prompt_col = next((c for c in cols if "prompt" in c.lower()), None)
    response_col = next((c for c in cols if "response" in c.lower()), None)

    if prompt_col and response_col:
        for i in range(table.num_rows):
            row = table.slice(i, 1).to_pydict()
            prompt = str(row[prompt_col][0]) if row[prompt_col][0] is not None else ""
            response = str(row[response_col][0]) if row[response_col][0] is not None else ""
            if prompt.strip() and response.strip():
                pairs.append({"prompt": prompt.strip(), "response": response.strip()})
    return pairs

def parse_jsonl_to_pairs(data: bytes) -> List[Dict[str, str]]:
    pairs = []
    for line in data.decode("utf-8", errors="ignore").strip().splitlines():
        line = line.strip()
        if not line:
            continue
        try:
            obj = json.loads(line)
        except Exception:
            continue
        prompt = obj.get("prompt") or obj.get("input") or ""
        response = obj.get("response") or obj.get("output") or ""
        if isinstance(prompt, str) and isinstance(response, str):
            if prompt.strip() and response.strip():
                pairs.append({"prompt": prompt.strip(), "response": response.strip()})
    return pairs

def compute_md5(data: bytes) -> str:
    return hashlib.md5(data).hexdigest()
