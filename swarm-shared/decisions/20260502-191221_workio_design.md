# workio / design

Candidate 3:

## 1. Diagnosis

- **No tenant cache on LINE webhook** — every webhook performs `channelId → tenant` PostgreSQL lookup; under burst traffic this amplifies DB load, adds latency, and increases failure surface.
- **Missing health/readiness endpoints** — no `/health` or `/ready` for k8s/PM2/liveness checks; complicates orchestration and auto-recovery.
- **No graceful shutdown handling** — in-flight LINE webhooks and DB queries can be cut off on restart/deploy; risk of partial writes and inconsistent clock-in/out state.
- **No request-id / trace correlation** — hard to debug webhook flows across services (LINE → backend → DB) when incidents occur.
- **Static config reload requires restart** — `.env`/tenant settings changes require process restart; small operational friction at low traffic.

## 2. Proposed change

Add a tenant-aware in-memory LRU cache for `channelId → tenant` in the LINE webhook handler and expose `/health` and `/ready` endpoints. Scope:

- File: `workio/server/src/app.ts` (or main Express entry)
- Add: `lru-cache` dependency, cache wrapper, and middleware for health/readiness.
- Modify: LINE webhook route (`/webhook/line`) to use cache-first lookup with DB fallback and TTL (5m).

## 3. Implementation

Install dependency:

```bash
cd /opt/axentx/workio/server
npm install lru-cache
```

Create cache module (`workio/server/src/cache.ts`):

```ts
import { LRUCache } from 'lru-cache';

type Tenant = {
  id: string;
  name: string;
  channelId: string;
  // include only what's needed by webhook/auth
};

const tenantByChannelCache = new LRUCache<string, Tenant>({
  max: 500,
  ttl: 1000 * 60 * 5, // 5 minutes
  allowStale: false,
  updateAgeOnGet: true,
});

export function setTenantByChannel(channelId: string, tenant: Tenant) {
  tenantByChannelCache.set(channelId, tenant);
}

export function getTenantByChannel(channelId: string): Tenant | undefined {
  return tenantByChannelCache.get(channelId);
}

export function deleteTenantByChannel(channelId: string) {
  tenantByChannelCache.delete(channelId);
}
```

Update main app (`workio/server/src/app.ts`) — add health/ready and wire cache into LINE webhook:

```ts
import express, { Request, Response, NextFunction } from 'express';
import { getTenantByChannel, setTenantByChannel } from './cache';
import db from './db'; // adjust import to your db client
import morgan from 'morgan';
import { v4 as uuidv4 } from 'uuid';

const app = express();

// request-id middleware for traceability
app.use((req, res, next) => {
  const requestId = req.headers['x-request-id'] as string || uuidv4();
  (req as any).requestId = requestId;
  res.setHeader('X-Request-Id', requestId);
  next();
});

app.use(express.json());
app.use(morgan('combined'));

// Health & readiness
let isReady = true;

app.get('/health', (req: Request, res: Response) => {
  // liveness: process is alive
  res.json({ status: 'ok', uptime: process.uptime(), requestId: (req as any).requestId });
});

app.get('/ready', async (req: Request, res: Response) => {
  // readiness: db reachable
  if (!isReady) {
    return res.status(503).json({ status: 'not_ready', requestId: (req as any).requestId });
  }
  try {
    await db.query('SELECT 1');
    res.json({ status: 'ready', requestId: (req as any).requestId });
  } catch (err) {
    res.status(503).json({ status: 'db_down', error: String(err), requestId: (req as any).requestId });
  }
});

// Graceful shutdown
function shutdown() {
  isReady = false;
  // allow in-flight requests ~30s then force exit
  const server = appServer.close(() => process.exit(0));
  setTimeout(() => process.exit(1), 30_000).unref();
}
process.on('SIGTERM', shutdown);
process.on('SIGINT', shutdown);

// LINE webhook route (example path)
app.post('/webhook/line', async (req: Request, res: Response) => {
  try {
    const channelId = req.header('x-line-channel-id') || '';
    if (!channelId) {
      return res.status(400).json({ error: 'missing_channel_id' });
    }

    let tenant = getTenantByChannel(channelId);
    if (!tenant) {
      // DB lookup with projection only
      const row = await db.query(
        `SELECT id, name, channel_id AS "channelId" FROM tenants WHERE channel_id = $1 LIMIT 1`,
        [channelId]
      );
      if (!row.rows.length) {
        return res.status(404).json({ error: 'tenant_not_found' });
      }
      tenant = row.rows[0];
      setTenantByChannel(channelId, tenant);
    }

    // Attach tenant to request for downstream handlers
    (req as any).tenant = tenant;

    // Process LINE event(s) here or delegate to controller
    // ...

    res.json({ ok: true });
  } catch (err) {
    console.error('LINE webhook error', err);
    res.status(500).json({ error: 'internal_error' });
  }
});

const PORT = process.env.PORT || 3000;
const appServer = app.listen(PORT, () => {
  console.log(`Workio server listening on ${PORT}`);
});
```

Notes:
- If your LINE route is elsewhere, move/apply the cache logic there.
- Ensure DB client (`db`) supports `.query` interface; adapt to your ORM/pool as needed.
- Keep cached tenant payload minimal (id, name, channelId) to avoid memory bloat.
