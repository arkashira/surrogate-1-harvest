# workio / frontend

## Final Synthesis & Action Plan

**Core diagnosis (unified):**  
- No real-time path from LINE webhooks → frontend; changes are invisible until refresh or poll.  
- Multi-tab/device state diverges, creating double-submissions and stale UI.  
- Polling is wasteful and jittery; transient failures surface as silent data loss.  
- Missing optimistic UI + rollback and server acknowledgment confuses users about success/failure.

**Chosen strategy:**  
Adopt Candidate 1’s SSE + SWR merge (lightweight, auto-reconnect, battery-friendly) and reinforce it with Candidate 2’s requirements: robust retry/backoff, clear server acknowledgment, and presence/optimistic UX with rollback.

### 1. Backend: resilient SSE endpoint with real pub/sub path

`workio/server/src/routes/events.ts`
```ts
import express from 'express';
import { getDb } from '../db';

const router = express.Router();

// Simple in-process pub/sub for immediate value; replace with pg_notify/Redis later.
const subscribers = new Map<string, Set<express.Response>>();

function publish(tenantId: string, event: string, payload: any) {
  const set = subscribers.get(tenantId);
  if (!set) return;
  for (const res of set) {
    try {
      res.write(`event: ${event}\ndata: ${JSON.stringify(payload)}\n\n`);
      res.flushHeaders?.();
    } catch {
      // will be cleaned up on close
    }
  }
}

// Expose for use in LINE/webhook handlers
export const __TENANT_PUBSUB = { publish };

router.get('/stream', (req, res) => {
  const tenantId = req.query.tenantId as string;
  if (!tenantId) {
    res.status(400).send('tenantId required');
    return;
  }

  res.setHeader('Content-Type', 'text/event-stream');
  res.setHeader('Cache-Control', 'no-cache');
  res.setHeader('Connection', 'keep-alive');
  res.flushHeaders?.();

  if (!subscribers.has(tenantId)) subscribers.set(tenantId, new Set());
  subscribers.get(tenantId)!.add(res);

  res.write(`event: connected\ndata: ${JSON.stringify({ tenantId })}\n\n`);

  (async () => {
    try {
      const db = getDb();
      const [clocks, leaves, ots] = await Promise.all([
        db.any(
          'SELECT id, employee_id, type, ts, location FROM clocks WHERE tenant_id = $1 ORDER BY ts DESC LIMIT 20',
          [tenantId]
        ),
        db.any(
          'SELECT id, employee_id, leave_type, start_date, end_date, status FROM leaves WHERE tenant_id = $1 AND status IN ($2, $3) ORDER BY updated_at DESC LIMIT 20',
          [tenantId, 'pending', 'approved']
        ),
        db.any(
          'SELECT id, employee_id, hours, date, status FROM ot_requests WHERE tenant_id = $1 AND status IN ($2, $3) ORDER BY updated_at DESC LIMIT 20',
          [tenantId, 'pending', 'approved']
        ),
      ]);
      res.write(`event: snapshot\ndata: ${JSON.stringify({ clocks, leaves, ots })}\n\n`);
    } catch (err) {
      res.write(`event: error\ndata: ${JSON.stringify({ message: 'snapshot failed' })}\n\n`);
    }
  })();

  const keepAlive = setInterval(() => {
    try {
      res.write(`event: ping\ndata: ${JSON.stringify({ ts: Date.now() })}\n\n`);
      res.flushHeaders?.();
    } catch {
      cleanup();
    }
  }, 25000);

  const cleanup = () => {
    clearInterval(keepAlive);
    subscribers.get(tenantId)?.delete(res);
    if (subscribers.get(tenantId)?.size === 0) subscribers.delete(tenantId);
    res.end();
  };

  req.on('close', cleanup);
});

export default router;
```

Register in `workio/server/src/app.ts`
```ts
import eventsRouter from './routes/events';
app.use('/events', eventsRouter);
```

**Emit from LINE webhook handlers (after successful DB write):**


### 2. Frontend: SSE hook with SWR + retry/backoff + acknowledgment

`workio/src/lib/hooks/useTenantEvents.ts`
```ts
import { useEffect, useRef } from 'react';
import useSWR from 'swr';

const fetcher = (url: string) => fetch(url).then((r) => {
  if (!r.ok) throw new Error(`HTTP ${r.status}`);
  return r.json();
});

export function useTenantEvents(tenantId: string | undefined) {
  const eventSourceRef = useRef<EventSource | null>(null);

  const { mutate: mutateClocks } = useSWR(
    tenantId ? `/api/clocks?tenantId=${tenantId}` : null,
    fetcher,
    { revalidateOnFocus: false, revalidateOnReconnect: false }
  );
  const { mutate: mutateLeaves } = useSWR(
    tenantId ? `/api/leaves?tenantId=${tenantId}` : null,
    fetcher,
    { revalidateOnFocus: false, revalidateOnReconnect: false }
  );
  const { mutate: mutateOt } = useSWR(
    tenantId ? `/api/ot-requests?tenantId=${tenantId}` : null,
    fetcher,
    { revalidateOnFocus: false, revalidateOnReconnect: false }
  );

  const apply = () => {
    mutateClocks?.();
    mutateLeaves?.();
    mutateOt?.();
  };

  useEffect(() => {
    if (!tenantId) return;

    const connect = () => {
      const es = new EventSource(
        `${import.meta.env.VITE_API_BASE_URL}/events/stream?tenantId=${encodeURIComponent(tenantId)}`
      );
      eventSourceRef.current = es;

      es.addEventListener('snapshot', apply);
      es.addEventListener('clock', apply);
      es.addEventListener('leave', apply);
      es.addEventListener('ot', apply);
      es.addEventListener('ping', () => {
        /* keepalive */
      });

      es.onerror = () => {
        // EventSource auto-reconnects; we rely on that.
        // Optionally log or surface connection status to UI.
        console.warn('[SSE] connection error (will retry)');
      };
    };

    connect();

    return () => {
      eventSourceRef.current?.close();
    };
  }, [tenantId]);

  return null;
}
```
