# workio / design

## 1. Diagnosis

- **No idempotency layer for LINE webhook deliveries**: duplicate `punch` rows on redelivery or concurrency because there is no durable `line_delivery_id` with a unique constraint.
- **Non-atomic state transition**: punch insert and state checks are separate operations; race conditions allow double-clock-in/out within the same tolerance window.
- **Missing tenant-scoped uniqueness**: uniqueness for `employee_id` + `date` + `type` (clock_in/out) is not enforced at the DB level, allowing invalid punch sequences.
- **No webhook replay protection**: no deduplication window or idempotency key storage to absorb LINE’s at-least-once delivery semantics.
- **Design inconsistency**: the multi-tenant schema lacks a `line_deliveries` table to record raw webhook events and correlate them to domain events (punches).

## 2. Proposed change

- **File scope**: `/opt/axentx/workio/server/src/db/schema.sql` (add tables + constraints)  
- **File scope**: `/opt/axentx/workio/server/src/services/punchService.ts` (add idempotent `handleLinePunch`)  
- **File scope**: `/opt/axentx/workio/server/src/routes/lineWebhook.ts` (use service with atomic upsert)

## 3. Implementation

### 3.1 DB schema (schema.sql)

```sql
-- Record raw LINE deliveries to provide idempotency and audit
CREATE TABLE IF NOT EXISTS line_deliveries (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id       UUID NOT NULL REFERENCES tenants(id) ON DELETE CASCADE,
  line_delivery_id TEXT NOT NULL,            -- LINE webhook delivery id (from X-Line-Signature or header)
  event_type      TEXT NOT NULL,             -- 'punch', 'leave_request', etc.
  payload         JSONB NOT NULL,
  processed_at    TIMESTAMPTZ,
  created_at      TIMESTAMPTZ DEFAULT now(),
  CONSTRAINT uq_line_delivery_per_tenant UNIQUE (tenant_id, line_delivery_id)
);

-- Strengthen punches with tenant+employee+time uniqueness and state machine
-- Add a deterministic idempotency slug to prevent duplicates from same logical event
ALTER TABLE punches
  ADD COLUMN IF NOT EXISTS idempotency_slug TEXT,
  ADD COLUMN IF NOT EXISTS line_delivery_id TEXT;

-- Index for fast idempotency checks
CREATE INDEX IF NOT EXISTS idx_punches_idempotency ON punches (tenant_id, idempotency_slug) WHERE idempotency_slug IS NOT NULL;
CREATE INDEX IF NOT EXISTS idx_punches_line_delivery ON punches (tenant_id, line_delivery_id) WHERE line_delivery_id IS NOT NULL;

-- Prevent duplicate active punches for same employee in same tenant
CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_unique_active
  ON punches (tenant_id, employee_id, type)
  WHERE status = 'active';
```

### 3.2 Punch service (punchService.ts)

```ts
// server/src/services/punchService.ts
import { pool } from '../db';

export interface HandleLinePunchInput {
  tenantId: string;
  employeeId: string;
  lineUserId: string;
  lineDeliveryId: string;   // from webhook header (X-Line-Signature + timestamp body)
  type: 'clock_in' | 'clock_out';
  latitude?: number;
  longitude?: number;
  occurredAt?: Date;
}

export async function handleLinePunch(input: HandleLinePunchInput) {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // 1) Idempotency: record line delivery
    const deliveryResult = await client.query(
      `INSERT INTO line_deliveries (tenant_id, line_delivery_id, event_type, payload, processed_at)
       VALUES ($1, $2, $3, $4, now())
       ON CONFLICT (tenant_id, line_delivery_id) DO UPDATE SET payload = EXCLUDED.payload
       RETURNING id, processed_at`,
      [input.tenantId, input.lineDeliveryId, 'punch', JSON.stringify({ lineUserId: input.lineUserId })]
    );

    // If already processed, skip domain action (idempotent)
    const alreadyProcessed = deliveryResult.rows[0].processed_at != null;
    if (alreadyProcessed) {
      await client.query('COMMIT');
      return { created: false, reason: 'already_processed' };
    }

    // 2) Build idempotency slug (deterministic for same logical event)
    const occurred = input.occurredAt || new Date();
    const datePart = occurred.toISOString().slice(0, 10); // YYYY-MM-DD
    const idempotencySlug = `${input.tenantId}:${input.employeeId}:${input.type}:${datePart}:${input.lineDeliveryId}`;

    // 3) Atomic punch upsert with state guard
    const punchResult = await client.query(
      `INSERT INTO punches (tenant_id, employee_id, type, status, latitude, longitude, occurred_at, line_delivery_id, idempotency_slug)
       VALUES ($1, $2, $3, 'active', $4, $5, $6, $7, $8)
       ON CONFLICT (tenant_id, employee_id, type)
       WHERE status = 'active'
       DO UPDATE SET line_delivery_id = EXCLUDED.line_delivery_id, idempotency_slug = EXCLUDED.idempotency_slug
       RETURNING id`,
      [
        input.tenantId,
        input.employeeId,
        input.type,
        input.latitude,
        input.longitude,
        occurred,
        input.lineDeliveryId,
        idempotencySlug,
      ]
    );

    // Mark delivery processed
    await client.query(
      `UPDATE line_deliveries SET processed_at = now() WHERE id = $1`,
      [deliveryResult.rows[0].id]
    );

    await client.query('COMMIT');
    return { created: true, punchId: punchResult.rows[0].id };
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  } finally {
    client.release();
  }
}
```

### 3.3 Webhook route (lineWebhook.ts)

```ts
// server/src/routes/lineWebhook.ts
import { Router } from 'express';
import { handleLinePunch } from '../services/punchService';

const router = Router();

router.post('/line', async (req, res) => {
  const lineDeliveryId = req.headers['x-line-signature'] as string; // or derive stable id from body+timestamp
  const { tenantId } = req.body; // assume tenant resolved earlier (e.g., via channel secret -> tenant mapping)

  // Minimal mapping for demo; expand with real LINE event parsing
  const events = req.body.events || [];
  for (const ev of events) {
    if (ev.type === 'message' && ev.message?.type === 'text') {
      // Example: simple text command "in" / "out"
      const type = ev.message.text.toLowerCase().includes('in') ? 'clock_in' : 'clock_out';
      await handleLinePunch({
        tenantId,
        employeeId: ev.source.userId, // LINE user id mapped to employee
        lineUserId: ev.source.userId,
        lineDeliveryId: `${lineDeliveryId}:${ev.destination?.[0] || ''}:${ev.timestamp}`,
        type,
        occurredAt: new Date(ev.timestamp),
      });
    }
  }

  res.status(200).send({ ok: true });
});

export default router;
```
