# workio / backend

## Final synthesized implementation (correct + actionable)

**Core principle**: merge the strongest, non-contradictory parts into a minimal, production-ready change that is correct and immediately actionable.

- **Cache**: use a single LRU (TTL 10 min, size 500) for `channelId → tenant` to collapse DB load.
- **Correlation ID**: request-scoped ID + structured logger (pino) attached to `req`.
- **Idempotency**: in-memory per-node guard for clock events (minute bucket) with clear upgrade path to Redis for multi-instance.
- **Tenant usage**: enforce `tenantId` on all attendance/leave/OT inserts.
- **Retry safety**: fast cache hits + idempotency + quick 2xx responses reduce LINE retry amplification.
- **Location**: keep changes in `workio/server/src/routes/webhook/line.ts` + two small shared modules (`cache.ts`, `logger.ts`).

```ts
import LRU from 'lru-cache';

export const tenantByChannelCache = new LRU<string, {
  tenantId: string;
  name: string;
  updatedAt: number;
}>({
  max: 500,
  ttl: 1000 * 60 * 10, // 10 minutes
  allowStale: false,
  updateAgeOnGet: true,
});
```

**`workio/server/src/lib/logger.ts`**

```ts
import pino from 'pino';
import { v4 as uuidv4 } from 'uuid';
import type { Request, Response, NextFunction } from 'express';

const logger = pino({
  level: process.env.LOG_LEVEL || 'info',
  base: { pid: process.pid },
  timestamp: pino.stdTimeFunctions.isoTime,
});

export const requestLogger = (req: Request, res: Response, next: NextFunction) => {
  const requestId = (req.headers['x-request-id'] as string) || uuidv4();
  (req as any).requestId = requestId;
  (req as any).log = logger.child({ requestId });
  next();
};
```

```ts
import express from 'express';
import crypto from 'crypto';
import { tenantByChannelCache } from '../../lib/cache';
import { requestLogger } from '../../lib/logger';
import type { Request, Response } from 'express';

// Optional Redis-backed idempotency (multi-node)
// import Redis from 'ioredis';
// const redis = new Redis(process.env.REDIS_URL);

const router = express.Router();

// In-memory per-node idempotency set (minute bucket). Replace with Redis for multi-instance.
const IDEM_TTL_MS = 120_000; // keep entries 2m to cover minute bucket edge cases
function globalIdemSet(): Set<string> {
  if (!(global as any)._lineWebhookIdemSet) {
    (global as any)._lineWebhookIdemSet = new Set<string>();
  }
  return (global as any)._lineWebhookIdemSet as Set<string>;
}

function hashIdemKey(key: string): string {
  return crypto.createHash('sha256').update(key).digest('hex').slice(0, 16);
}

async function isDuplicateIdem(key: string): Promise<boolean> {
  const hash = hashIdemKey(key);
  const set = globalIdemSet();
  if (set.has(hash)) return true;
  set.add(hash);
  setTimeout(() => set.delete(hash), IDEM_TTL_MS);
  return false;
}

// Redis version (uncomment if multi-node)
// async function isDuplicateIdemRedis(key: string): Promise<boolean> {
//   const added = await redis.set(`idem:${key}`, '1', 'EX', 120, 'NX');
//   return added === null;
// }

router.post('/webhook/line', requestLogger, async (req: Request, res: Response) => {
  const log = (req as any).log;
  const requestId = (req as any).requestId;
  const body = req.body;

  log.info({ event: 'line.webhook.received', destination: body?.destination }, 'webhook received');

  // Basic validation
  const channelId = body?.destination;
  if (!channelId) {
    log.warn({ body: JSON.stringify(body).slice(0, 500) }, 'missing destination');
    return res.status(400).json({ error: 'missing_destination' });
  }

  // Idempotency guard across events in this webhook payload
  if (body?.events && Array.isArray(body.events)) {
    for (const ev of body.events) {
      if (!ev?.source?.userId || !ev.type) continue;
      // Minute bucket key: user+type+YYYY-MM-DDTHH:MM (per-minute)
      const minuteKey = `${ev.source.userId}:${ev.type}:${new Date().toISOString().slice(0, 16)}`;
      if (await isDuplicateIdem(minuteKey)) {
        log.warn({ userId: ev.source.userId, type: ev.type, minuteKey }, 'duplicate event skipped');
        // skip processing this event but continue processing others in payload
        continue;
      }
    }
  }

  // Resolve tenant (cache-first)
  let tenant = tenantByChannelCache.get(channelId);
  if (!tenant) {
    log.info({ channelId }, 'cache miss: resolving tenant from db');
    // Replace with your actual DB client/query
    const rows = await (req as any).db?.query(
      'SELECT id, name FROM tenants WHERE line_channel_id = $1',
      [channelId]
    );
    if (!rows?.length) {
      log.warn({ channelId }, 'tenant not found for channel');
      return res.status(404).json({ error: 'tenant_not_found' });
    }
    tenant = { tenantId: rows[0].id, name: rows[0].name, updatedAt: Date.now() };
    tenantByChannelCache.set(channelId, tenant);
  } else {
    log.debug({ channelId, tenantId: tenant.tenantId }, 'cache hit');
  }

  // Attach for downstream handlers
  (req as any).tenant = tenant;

  // Process events (example: clock commands via text message)
  if (body?.events && Array.isArray(body.events)) {
    for (const ev of body.events) {
      if (ev.type === 'message' && ev.message?.type === 'text') {
        const userId = ev.source.userId;
        const text = ev.message.text;
        log.info({ userId, tenantId: tenant.tenantId, text, requestId }, 'processing clock command');

        // Example attendance insert — always include tenantId
        // Replace with your schema and DB client
        try {
          await (req as any).db?.query(
            `INSERT INTO attendances (tenant_id, user_id, event_type, recorded_at, request_id)
             VALUES ($1, $2, $3, $4, $5)`,
            [tenant.tenantId, userId, 'clock_in', new Date(), requestId]
          );
          log.info({ userId, tenantId: tenant.tenantId, requestId }, 'attendance recorded');
        } catch (err: any) {
          log.error({ err: err.message, userId, tenantId: tenant.tenantId, requestId }, 'attendance insert failed');
          // Do not expose internal errors to LINE; return 200 to stop retries if you've already handled business side-effects safely.
        }
      }
    }
  }

  log.info({ event: 'line.webhook.processed', channelId, tenantId: tenant.tenantId }, 'webhook processed');
  return res.json({ status: 'ok' });
});

export default router;
```
