## Code Snippets

### 1) Build CDN manifest (run from Mac/CI)

```python
# scripts/build_cdn_manifest.py
#!/usr/bin/env python3
"""
Generate CDN-only manifest for Surrogate training.
Run from Mac/CI (single API call per folder) after HF rate-limit window clears.
"""
import json
import os
from datetime import date
from huggingface_hub import HfApi

API = HfApi()
REPO_ID = "axentx/surrogate-datasets"
DATE_FOLDER = str(date.today())          # e.g. 2026-05-03
OUT_DIR = "batches/mirror-merged"
os.makedirs(OUT_DIR, exist_ok=True)

# Single non-recursive call per folder (avoids 100x pagination)
items = API.list_repo_tree(
    repo_id=REPO_ID,
    path=DATE_FOLDER,
    repo_type="dataset",
    recursive=False,
)

files = [it.rfilename for it in items if it.rfilename.endswith(".parquet")]

manifest = {
    "repo_id": REPO_ID,
    "date": DATE_FOLDER,
    "cdn_base": f"https://huggingface.co/datasets/{REPO_ID}/resolve/main",
    "files": [
        {
            "filename": f,
            "cdn_url": f"{REPO_ID}/resolve/main/{DATE_FOLDER}/{f}",
            "local_path": f"{DATE_FOLDER}/{f}",
        }
        for f in sorted(files)
    ],
}

out_path = os.path.join(OUT_DIR, f"manifest-{DATE_FOLDER}.json")
with open(out_path, "w") as fp:
    json.dump(manifest, fp, indent=2)

print(f"Wrote {len(files)} files -> {out_path}")
```

Make executable and set crontab to run once per day after ingestion finishes:

```bash
chmod +x scripts/build_cdn_manifest.py
# crontab -e
0 3 * * * cd /opt/axentx/airship && /usr/bin/env python3 scripts/build_cdn_manifest.py >> logs/build_manifest.log 2>&1
```

### 2) CDN-only dataset loader in training script

```python
# surrogate/train.py  (excerpt)
import json
import io
import pyarrow.parquet as pq
import requests
from torch.utils.data import IterableDataset

class CDNParquetIterable(IterableDataset):
    def __init__(self, manifest_path, max_files=None):
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.files = self.manifest["files"]
        if max_files:
            self.files = self.files[:max_files]

    def __iter__(self):
        for item in self.files:
            cdn_url = f"https://huggingface.co/datasets/{item['cdn_url']}"
            # CDN download — no Authorization header, bypasses /api/ rate limits
            resp = requests.get(cdn_url, stream=True, timeout=60)
            resp.raise_for_status()
            buf = io.BytesIO(resp.content)
            table = pq.read_table(buf)
            # Project to {prompt, response} only
            for batch in table.to_batches(columns=["prompt", "response"]):
                for prompt, response in zip(batch["prompt"], batch["response"]):
                    yield {"prompt": prompt.as_py(), "response": response.as_py()}
```

Usage:

```bash
python surrogate/train.py \
  --manifest batches/mirror-merged/manifest-2026-05-03.json \
  --output-dir runs/surrogate-2026-05-03
```

### 3) Lightning Studio watchdog (prevents idle-stop loss)

```python
# surrogate/lightning_watchdog.py
#!/usr/bin/env python3
"""
Ensure target Lightning Studio is Running before training.
If stopped, restart with L40S (free-tier fallback).
"""
import time
from lightning_sdk import Studio, Machine, Teamspace

TEAMSPACE = "axentx"
STUDIO_NAME = "surrogate-train"
MACHINE = Machine.L40S
MAX_RETRIES = 3
RETRY_WAIT = 60

def ensure_studio_running():
    ts = Teamspace(TEAMSPACE)
    for _ in range(MAX_RETRIES):
        studios = [s for s in ts.studios if s.name == STUDIO_NAME]
        if not studios:
            print(f"Creating Studio {STUDIO_NAME}...")
            studio = Studio.create(
                name=STUDIO_NAME,
                teamspace=TEAMSPACE,
                machine=MACHINE,
                create_ok=True,
            )
        else:
            studio = studios[0]

        if studio.status == "Running":
            print(f"Studio {STUDIO_NAME} is Running.")
            return studio

        print(f"Studio status={studio.status}. Starting...")
        studio.start(machine=MACHINE)

        # Wait for running
        for _ in range(30):
            studio.refresh()
            if studio.status == "Running":
                print("Studio is Running.")
                return studio
            time.sleep(10)

        print("Still not Running, retrying...")
        time.sleep(RETRY_WAIT)

    raise RuntimeError(f"Could not start Studio {STUDIO_NAME}")

if __name__ == "__main__":
    ensure_studio_running()
```

Run before training (or via cron every 5min):

```bash
chmod +x surrogate/lightning_watchdog.py
# Inline before training:
python surrogate/lightning_watchdog.py && python surrogate/train.py --manifest ...
```

Or cron:

```bash
*/5 * * * * cd /opt/axentx/airship && /usr/bin/env python3 surrogate/lightning_watchdog.py >> logs/watchdog.log 2>&1
```
