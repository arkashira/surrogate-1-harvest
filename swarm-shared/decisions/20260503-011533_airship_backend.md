## Final Synthesized Implementation (Best of Both Candidates)

**Chosen Fix**: Deploy a **HF CDN-bypass IterableDataset** with **schema-safe projection** and a **Lightning Studio reuse/idle-guard**.  
This eliminates the two highest-cost failure modes:

1. HF API 429s during training ingestion (solved by CDN-only streaming via `resolve/main/` URLs and a pre-built manifest).
2. Lightning quota burn from repeated Studio creation and idle-stop deaths (solved by strict reuse + status check + restart).

This is backend-only, requires no new infra, and can ship in ≤2 hours.

### 2) HF CDN-bypass IterableDataset (schema-safe)

```python
# training/dataset.py
import json
import random
import time
from typing import Iterator

import pyarrow.parquet as pq
import requests
import torch
from torch.utils.data import IterableDataset

HF_DATASET_REPO = "your-org/mirror-merged"
CDN_ROOT = f"https://huggingface.co/datasets/{HF_DATASET_REPO}/resolve/main"

class HFCDNDataset(IterableDataset):
    """
    Streams parquet shards via HF CDN (no API/auth).
    Projects to {prompt, response} only; schema-safe and tolerant.
    """
    def __init__(self, manifest_path: str, start_idx: int = 0, end_idx: int | None = None):
        with open(manifest_path) as f:
            self.files = json.load(f)[start_idx:end_idx]

    def _stream_shard(self, file_key: str) -> Iterator[dict]:
        url = f"{CDN_ROOT}/{file_key}"
        resp = requests.get(url, timeout=30)
        resp.raise_for_status()
        with open("/tmp/shard.parquet", "wb") as tmp:
            tmp.write(resp.content)
        table = pq.read_table("/tmp/shard.parquet", columns=["prompt", "response"])
        for row in table.to_pylist():
            prompt = row.get("prompt")
            response = row.get("response")
            # schema-safe: coerce to string; skip unusable rows
            if prompt is None or response is None:
                continue
            prompt = str(prompt).strip()
            response = str(response).strip()
            if not prompt or not response:
                continue
            yield {"prompt": prompt, "response": response}

    def __iter__(self) -> Iterator[dict]:
        # shuffle files per worker for better mixing
        files = self.files.copy()
        seed = torch.utils.data.get_worker_info().seed if torch.utils.data.get_worker_info() else None
        rng = random.Random(seed)
        rng.shuffle(files)
        for key in files:
            yield from self._stream_shard(key)
```

### 3) Schema-safe mirror writer (for ingestion side)

```python
# training/writer.py
import hashlib
import os

import pyarrow as pa
import pyarrow.parquet as pq

def write_mirror_batch(rows: list[dict], date_str: str, batch_slug: str) -> str:
    """
    rows: list[dict] with at least {prompt, response}
    date_str: YYYY-MM-DD
    batch_slug: short identifier
    Returns written path.
    """
    table = pa.Table.from_pylist(
        [{"prompt": r["prompt"], "response": r["response"]} for r in rows]
    )
    out_dir = f"batches/mirror-merged/{date_str}"
    os.makedirs(out_dir, exist_ok=True)
    slug = hashlib.sha256(batch_slug.encode()).hexdigest()[:12]
    out_path = f"{out_dir}/{slug}.parquet"
    pq.write_table(table, out_path)
    return out_path
```

### 4) Lightning Studio guard + quota-safe launcher

```python
# training/lightning_launcher.py
import lightning as L

HF_MANIFEST = "training/manifest/2026-05-03_files.json"
STUDIO_NAME = "surrogate-l40s-train"

def launch_studio() -> L.Studio:
    teamspace = L.Teamspace()
    running = [s for s in teamspace.studios if s.name == STUDIO_NAME and s.status == "Running"]
    if running:
        print(f"Reusing running studio: {STUDIO_NAME}")
        return running[0]

    existing = [s for s in teamspace.studios if s.name == STUDIO_NAME]
    if existing:
        studio = existing[0]
        if studio.status != "Running":
            print(f"Starting idle-stopped studio: {STUDIO_NAME}")
            studio.start(machine=L.Machine.L40S)
            return studio

    print(f"Creating studio: {STUDIO_NAME}")
    return L.Studio(
        name=STUDIO_NAME,
        machine=L.Machine.L40S,
        create_ok=True,
    )

def run_training():
    studio = launch_studio()

    # Final guard: ensure running
    if studio.status != "Running":
        print("Studio not running — restarting...")
        try:
            studio.start(machine=L.Machine.L40S)
        except Exception:
            # fallback machine if L40S unavailable
            print("L40S unavailable, falling back to lightning-public-prod")
            studio.start(machine=L.Machine.lightning_public_prod)

    # Use LightningCLI for actual training; pass manifest via CLI args
    from lightning.pytorch.cli import LightningCLI
    cli = LightningCLI(
        run=False,
        save_config_kwargs={"overwrite": True},
        args=[
            "--trainer.max_epochs=1",
            f"--data.manifest_path={HF_MANIFEST}",
            "--trainer.limit_train_batches=100",
        ],
    )
```
