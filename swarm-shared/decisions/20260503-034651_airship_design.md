### 1) `surrogate/dataset_loader.py`
```python
import json
import os
from pathlib import Path
from typing import List, Dict, Any
import requests
from datasets import Dataset, Features, Value

HF_CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def load_filelist(filelist_path: str) -> List[Dict[str, Any]]:
    with open(filelist_path, "r") as f:
        return json.load(f)

def cdn_text_iterator(repo: str, filelist: List[Dict[str, Any]], text_key: str = "response"):
    for item in filelist:
        url = HF_CDN_TEMPLATE.format(repo=repo, path=item["path"])
        resp = requests.get(url, timeout=30)
        resp.raise_for_status()
        # Minimal projection: expect JSONL lines with {prompt, response}
        # If parquet, this would be different — here we assume line-delimited JSON
        for line in resp.text.strip().splitlines():
            if not line.strip():
                continue
            obj = json.loads(line)
            yield {text_key: obj.get(text_key, "")}

def build_cdn_dataset(repo: str, filelist_path: str, text_key: str = "response") -> Dataset:
    filelist = load_filelist(filelist_path)
    features = Features({text_key: Value("string")})
    return Dataset.from_generator(
        lambda: cdn_text_iterator(repo, filelist, text_key=text_key),
        features=features,
    )
```

### 2) `surrogate/lightning_runner.py`
```python
import hashlib
import os
from typing import Optional
from lightning import LightningWork, LightningFlow, LightningApp, Machine
from lightning.app import BuildConfig
from lightning.app.utilities.state import AppState

# Deterministic sibling repo selector
SIBLING_REPOS = [
    "axentx/surrogate-data-0",
    "axentx/surrogate-data-1",
    "axentx/surrogate-data-2",
    "axentx/surrogate-data-3",
    "axentx/surrogate-data-4",
]

def pick_sibling_repo(slug: str) -> str:
    idx = int(hashlib.sha256(slug.encode()).hexdigest(), 16) % len(SIBLING_REPOS)
    return SIBLING_REPOS[idx]

class IdleResilientStudio(LightningWork):
    def __init__(
        self,
        name: str,
        script_path: str,
        repo: str,
        filelist_path: str,
        machine: str = "lightning-lambda-prod/L40S",
        cloud_account: str = "lightning-lambda-prod",
        max_retries: int = 3,
    ):
        super().__init__(cloud_compute=machine, cloud_build_config=BuildConfig(image="pytorch/pytorch:2.3.0-cuda12.1-cudnn8-runtime"))
        self.name = name
        self.script_path = script_path
        self.repo = repo
        self.filelist_path = filelist_path
        self.machine = machine
        self.cloud_account = cloud_account
        self.max_retries = max_retries
        self._retry_count = 0

    def run(self, *args, **kwargs):
        from lightning import Studio, Teamspace

        # Reuse running studio if exists
        for s in Teamspace.studios:
            if s.name == self.name and s.status == "running":
                print(f"Reusing running studio: {self.name}")
                # If stopped externally, restart
                if s.status != "running":
                    s.start(machine=Machine(self.machine, cloud_account=self.cloud_account))
                break
        else:
            print(f"Creating studio: {self.name}")
            studio = Studio(
                name=self.name,
                script_path=self.script_path,
                cloud_compute=Machine(self.machine, cloud_account=self.cloud_account),
                create_ok=True,
            )
            studio.start()

        # Execute training script inside studio context (simplified: assumes script is self-contained)
        # In practice, this would be orchestrated via LightningWork's run method on the cloud machine.
        # Here we provide a resilient launcher pattern:
        import subprocess, time

        env = os.environ.copy()
        env["HF_REPO"] = self.repo
        env["FILELIST_PATH"] = self.filelist_path

        cmd = ["python", self.script_path]
        for attempt in range(self.max_retries):
            try:
                proc = subprocess.run(cmd, env=env, check=True, capture_output=True, text=True)
                print(proc.stdout)
                return proc.stdout
            except subprocess.CalledProcessError as exc:
                self._retry_count += 1
                if self._retry_count >= self.max_retries:
                    raise
                wait = 360 if "429" in exc.stderr else 60
                print(f"Retry {attempt+1}/{self.max_retries} after {wait}s: {exc.stderr[-200:]}")
                time.sleep(wait)

class SurrogateTrainingFlow(LightningFlow):
    def __init__(self):
        super().__init__()
        self.studio = IdleResilientStudio(
            name="surrogate-train-cdn",
            script_path="train.py",
            repo="axentx/surrogate-dataset",
            filelist_path="filelist_latest.json",
        )

    def run(self):
        self.studio.run()

if __name__ == "__main__":
    app = LightningApp(SurrogateTrainingFlow())
```

### 4) `surrogate/prepare_filelist.py` (run on Mac once per folder)
```python
import json
import os
from huggingface_hub import list_repo_tree

def build_filelist(repo: str, folder: str, out_path: str):
    # Single API call per folder (non-recursive)
    tree = list_repo_tree(repo, path=folder, recursive=False)
    files = [{"path": f.path} for f in tree if f.type == "file"]
    with open(out_path, "w") as f:
        json.dump(files, f, indent=2)
    print(f"Wrote {len(files)} files to {out_path}")

if __name__ == "__main__":
    import sys
    repo = sys.argv[1]
    folder = sys.argv[2]
    out = sys.argv[3]
    build_filelist(repo, folder, out)
```
