# surrogate-1 / design

## Final Implementation Plan (≤2h)

**Goal:** Eliminate HF API 429s during training, enforce clean `{prompt,response}` schema, and make shard ingestion robust—without changing the 16-shard runner architecture or killing running Studios.

---

## 1) Architecture (single source of truth)

- **Mac orchestrator** (or CI job) performs **one** `list_repo_tree` per date folder after rate-limit window.
- Produces `manifest-YYYY-MM-DD.json` containing only filenames for that date.
- Training jobs (Lightning Studios) load **exclusively via CDN** using that manifest—**zero HF API calls** during training.
- Schema projection + dedup happens **once** during shard creation (`dataset-enrich.sh`), never during training.
- Existing 16-shard assignment (MD5 hash mod 16) is preserved for deterministic routing.

---

## 2) Concrete Changes

### A. `bin/dataset-enrich.sh` (minimal, production-ready)

Projects to `{prompt,response}` at parse time, writes clean JSONL, and performs cross-run dedup using a central sqlite file on the HF Space volume.

```bash
# 2) Deterministic shard assignment + schema projection + dedup
python3 - <<PY
import os, json, hashlib, sys, sqlite3
from pathlib import Path
from huggingface_hub import hf_hub_download
try:
    import pyarrow.parquet as pq
    import pyarrow as pa
except Exception as e:
    print("pyarrow required", file=sys.stderr)
    sys.exit(1)

HF_REPO = os.getenv("HF_REPO")
SHARD_ID = int(os.getenv("SHARD_ID"))
TOTAL_SHARDS = int(os.getenv("TOTAL_SHARDS"))
OUTFILE = os.getenv("OUTFILE")
FILE_LIST = json.loads(os.getenv("FILE_LIST_JSON", "[]"))
DEDUP_DB = os.getenv("DEDUP_DB")

def slug_hash(slug: str) -> int:
    return int(hashlib.md5(slug.encode()).hexdigest(), 16)

def belongs_to_shard(slug: str) -> bool:
    return (slug_hash(slug) % TOTAL_SHARDS) == SHARD_ID

conn = sqlite3.connect(DEDUP_DB)
conn.execute("CREATE TABLE IF NOT EXISTS seen (md5 TEXT PRIMARY KEY)")

def is_dup(md5: str) -> bool:
    return conn.execute("SELECT 1 FROM seen WHERE md5=?", (md5,)).fetchone() is not None

def mark_seen(md5: str):
    conn.execute("INSERT OR IGNORE INTO seen (md5) VALUES (?)", (md5,))

out_f = open(OUTFILE, "w", encoding="utf-8")
accepted = 0
skipped_dup = 0
skipped_shard = 0
skipped_bad = 0

for rfilename in FILE_LIST:
    slug = rfilename
    if not belongs_to_shard(slug):
        skipped_shard += 1
        continue

    try:
        local_path = hf_hub_download(repo_id=HF_REPO, filename=rfilename, repo_type="dataset")
    except Exception as e:
        print(f"Download failed {rfilename}: {e}", file=sys.stderr)
        continue

    try:
        table = pq.read_table(local_path, columns=["prompt", "response"])
        cols = set(table.column_names)
        if "prompt" not in cols or "response" not in cols:
            # fallback: pick two text-ish columns
            pick = [c for c in table.column_names if table.schema.field(c).type in (pa.string(), pa.large_string())][:2]
            if len(pick) < 2:
                skipped_bad += 1
                continue
            table = table.select(pick).rename_columns(["prompt", "response"])
        df = table.to_pandas()
    except Exception:
        skipped_bad += 1
        continue

    for _, row in df.iterrows():
        prompt = str(row.get("prompt") or "").strip()
        response = str(row.get("response") or "").strip()
        if not prompt or not response:
            continue
        md5 = hashlib.md5((prompt + "\0" + response).encode()).hexdigest()
        if is_dup(md5):
            skipped_dup += 1
            continue
        mark_seen(md5)
        out_f.write(json.dumps({"prompt": prompt, "response": response}, ensure_ascii=False))
        out_f.write("\n")
        accepted += 1

conn.commit()
conn.close()
out_f.close()
print(f"Shard {SHARD_ID}: accepted={accepted} dup={skipped_dup} skipped_shard={skipped_shard} bad={skipped_bad} -> {OUTFILE}")
PY

echo "[$(date)] Shard ${SHARD_ID} done: ${OUTFILE}"
```

HF_REPO = "axentx/surrogate-1-training-pairs"
CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

class CDNJsonlIterableDataset(IterableDataset):
    def __init__(self, manifest_files: List[str], repo: str = HF_REPO, buffer_size: int = 10_000):
        super().__init__()
        self.repo = repo
        self.files = manifest_files
        self.buffer_size = buffer_size

    def _stream_file(self, path: str):
        import urllib.request
        url = CDN_TEMPLATE.format(repo=self.repo, path=path)
        with urllib.request.urlopen(url) as resp:
            for line in resp:
                line = line.decode("utf-8").strip()
                if not line:
                    continue
                try:
                    item = json.loads(line)
                    prompt = str(item.get("prompt") or "").strip()
                    response = str(item.get("response") or "").strip()
                    if prompt and response:
                        yield {"prompt": prompt, "response": response}
                except Exception:
                    continue

    def __iter__(self):
        worker_info = torch.utils.data.get_worker_info()

            files = self.files
        else:
            files = [f for i, f in enumerate(self.files) if i % worker_info.num_workers == worker_info.id]

        for f in files:
            yield from self._stream_file(f)
