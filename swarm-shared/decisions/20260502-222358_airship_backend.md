# airship / backend

## Implementation Plan — `airship discover` (backend)

**Scope**: single command, <2h, zero-config, cron-safe, CDN-only ingestion for Surrogate training.  
**Location**: `/opt/axentx/airship/arkship/scripts/discover.sh` (new) + light Python helper.

### What it does (in priority order)
1. Runs `granite-business-research.sh` (if present) → tags outputs `#business-research #knowledge-rag #graph`.
2. Executes `knowledge-rag` → prints top hub (e.g., MOC) and related docs; tags `#knowledge-rag #graph #hub`.
3. Builds a **date-scoped file list** from a HuggingFace dataset repo (single API call) → saves JSON for Lightning training.
4. Validates Kaggle KGAT token reachability (raw HTTP Bearer) and prints actionable diagnostics.
5. Emits a concise report + next-steps (reuse running Lightning Studio, CDN-only training command).

### Why this is safe for cron
- All HF list calls are **one per run** and scoped to a single date folder.
- CDN URLs are pre-computed; training will use **zero HF API calls** during data load.
- Script uses `#!/usr/bin/env bash`, `set -euo pipefail`, validates executables, and exits non-zero on failure (cron-friendly).
- Avoids recursive `list_repo_files`; uses `list_repo_tree(path, recursive=False)` per folder.

# ---- Helpers ----
ensure_executable() {
  local cmd="$1"
  if ! command -v "$cmd" >/dev/null 2>&1; then
    fail "Required command not found: $cmd"
  fi
}

# ---- Step 1: Business research ----
run_business_research() {
  local script="granite-business-research.sh"
  local script_path
  script_path=$(find "${REPO_ROOT}" -maxdepth 2 -name "${script}" -type f | head -n1 || true)

  if [[ -n "${script_path}" && -x "${script_path}" ]]; then
    log "Running ${script} ..."
    # Run in subshell to avoid affecting main env
    (cd "$(dirname "${script_path}")" && bash "${script_path}") || {
      log "WARNING: ${script} exited non-zero (continuing)"
    }
    # Tag outputs heuristically
    find "$(dirname "${script_path}")" -maxdepth 1 -type f -newer "${script_path}" -print0 | while IFS= read -r -d '' f; do
      log "Tagged output: ${f}  #business-research #knowledge-rag #graph"
    done
  else
    log "No executable ${script} found — skipping business research"
  fi
}

# ---- Step 2: Knowledge RAG + top hub ----
run_knowledge_rag() {
  local cmd="knowledge-rag"
  if ! command -v "${cmd}" >/dev/null 2>&1; then
    log "WARNING: ${cmd} not in PATH — skipping knowledge-rag"
    return 0
  fi

  log "Running ${cmd} (top hub + related docs) ..."
  # Best-effort; do not fail entire discover if this errors
  if ! ${cmd} --top-hub 2>/dev/null; then
    log "WARNING: ${cmd} --top-hub failed or not supported"
  fi
}

# ---- Step 3: HF file list (single API call) ----
build_hf_file_list() {
  if [[ -z "${HF_DATASET_REPO}" ]]; then
    log "HF_DATASET_REPO not set — skipping HF file list"
    return 0
  fi

  ensure_executable "python3"
  local outdir="${REPO_ROOT}/hf-file-lists"
  mkdir -p "${outdir}"
  local out="${outdir}/file-list-$(date +%Y%m%d-%H%M%S).json"

  log "Building HF file list for ${HF_DATASET_REPO} :: ${HF_DATASET_PATH} ..."
  python3 - <<PY > "${out}"
import json, os, sys, datetime, subprocess, textwrap

HF_DATASET_REPO = os.environ.get("HF_DATASET_REPO", "")
HF_DATASET_PATH = os.environ.get("HF_DATASET_PATH", "")
HF_TOKEN = os.environ.get("HF_TOKEN", "")

if not HF_DATASET_REPO:
    sys.exit(0)

try:
    from huggingface_hub import HfApi, list_repo_tree
    api = HfApi(token=HF_TOKEN if HF_TOKEN.strip() else None)
    # Single API call: non-recursive per folder
    items = list_repo_tree(repo_id=HF_DATASET_REPO, path=HF_DATASET_PATH, repo_type="dataset", token=HF_TOKEN if HF_TOKEN.strip() else None)
    files = []
    for it in items:
        if it.type == "file":
            files.append({
                "path": it.path,
                "size": getattr(it, "size", None),
                "lfs": getattr(it, "lfs", None) is not None,
                "cdn_url": f"https://huggingface.co/datasets/{HF_DATASET_REPO}/resolve/main/{it.path}"
            })
    payload = {
        "repo": HF_DATASET_REPO,
        "path": HF_DATASET_PATH,
        "fetched_at": datetime.datetime.utcnow().isoformat() + "Z",
        "files": files,
        "note": "CDN-only list for surrogate training; use cdn_url to bypass HF API rate limits during data loading."
    }
    print(json.dumps(payload, indent=2))
except Exception as e:
    sys.stderr.write(f"HF file list failed: {e}\\n")
    # Emit minimal valid JSON so downstream steps don't crash parsing
    print(json.dumps({"error": str(e), "repo": HF_DATASET_REPO, "path": HF_DATASET_PATH, "files": []}))
PY

  if [[ -s "${out}" ]]; then
    log "Saved HF file list: ${out}"
    log "Embed this file in train.py and use CDN URLs to avoid HF API calls during training."
  else
    log "WARNING: HF file list is empty"
  fi
}
