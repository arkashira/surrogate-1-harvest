# vanguard / design

## Final Synthesis (Best Parts + Correctness + Actionability)

**Core diagnosis (merged, de-duplicated):**
- No persistent file-list cache → repeated `list_repo_tree`/`load_dataset` → HF API 429 (1000 req/5min).
- Lightning Studio lifecycle not reused → quota burn on recreation; idle-stop silently kills training.
- `load_dataset(streaming=True)` on heterogeneous repos → `pyarrow.CastError` on mixed schemas.
- No CDN bypass → every data load pays auth checks and counts against rate limits.
- Local/mac runs risk accidental heavy compute or `from_pretrained()` on laptop.
- Missing deterministic repo spread for HF commits → ingestion hits 128 commit/hr/repo cap when mirroring large batches.

**Proposed change (merged):**
Add a lightweight design-time manifest + training launcher that enforces known patterns:
- `vanguard/manifest.py` — single `list_repo_tree` snapshot → JSON cache per date folder; reused by training.
- `vanguard/train_launcher.py` — Lightning Studio reuse + idle-check + CDN-only data loader (no HF API calls during train).
- `vanguard/ingest.py` — deterministic repo spread + strict schema projection to `{prompt,response}`; attribution via filename.
- `vanguard/verify.py` — quick smoke tests to confirm CDN-only paths and Studio reuse.
- `vanguard/.env.example` — safe local defaults to prevent accidental laptop compute.

### manifest.py
```python
#!/usr/bin/env python3
"""
Generate and reuse HF repo file manifests to avoid API 429s.
Usage:
  HF_TOKEN=... python manifest.py --repo datasets/opus-mt --date 2026-05-02 --out manifests/opus-mt-2026-05-02.json
"""
import argparse
import json
import os
import sys
from datetime import datetime, timezone
from pathlib import Path

try:
    from huggingface_hub import list_repo_tree
except ImportError:
    print("pip install huggingface_hub")
    sys.exit(1)

HF_TOKEN = os.getenv("HF_TOKEN")
if not HF_TOKEN:
    print("HF_TOKEN required")
    sys.exit(1)

def snapshot_repo(repo_id: str, date_folder: str, out_path: Path):
    out_path.parent.mkdir(parents=True, exist_ok=True)
    tree = list_repo_tree(
        repo_id=repo_id,
        path=date_folder,
        recursive=False,
        token=HF_TOKEN,
    )
    files = [
        {"path": blob.path, "size": blob.size}
        for blob in tree if blob.type == "file"
    ]
    manifest = {
        "repo_id": repo_id,
        "date_folder": date_folder,
        "created_at": datetime.now(timezone.utc).isoformat(),
        "files": files,
    }
    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"Manifest saved: {out_path} ({len(files)} files)")
    return manifest

if __name__ == "__main__":
    p = argparse.ArgumentParser()
    p.add_argument("--repo", required=True)
    p.add_argument("--date", required=True)
    p.add_argument("--out", required=True)
    args = p.parse_args()
    snapshot_repo(args.repo, args.date, Path(args.out))
```

### train_launcher.py
```python
#!/usr/bin/env python3
"""
Lightning Studio launcher with reuse + idle checks + CDN-only data loading.
Usage:
  HF_TOKEN=... python train_launcher.py --manifest manifests/... --script train.py
"""
import json
import os
import sys
from pathlib import Path

try:
    import lightning as L
except ImportError:
    print("pip install lightning")
    sys.exit(1)

HF_TOKEN = os.getenv("HF_TOKEN")

def cdn_url(repo_id: str, file_path: str) -> str:
    """CDN bypass: no Authorization header required."""
    return f"https://huggingface.co/datasets/{repo_id}/resolve/main/{file_path}"

def find_running_studio(name: str):
    for studio in L.Teamspace.studios():
        if studio.name == name and studio.status == "Running":
            return studio
    return None

def run_training(
    manifest_path: str,
    script_path: str,
    studio_name: str = "vanguard-train",
    machine: str = "L40S",
):
    manifest = json.loads(Path(manifest_path).read_text())
    studio = find_running_studio(studio_name)

    if studio is None:
        print(f"Creating studio {studio_name}...")
        studio = L.Studio(
            name=studio_name,
            script_path=script_path,
            machine=machine,
            create_ok=True,
        )
    else:
        print(f"Reusing running studio {studio_name}")

    if studio.status != "Running":
        print("Studio stopped; restarting...")
        studio.start(machine=machine)

    file_urls = [cdn_url(manifest["repo_id"], f["path"]) for f in manifest["files"]]
    env = {
        "HF_TOKEN": HF_TOKEN,
        "FILE_URLS": ",".join(file_urls),
        "MANIFEST_PATH": str(Path(manifest_path).absolute()),
    }
    job = studio.run(
        script_path,
        arguments=[],
        env=env,
    )
    print(f"Launched job: {job}")
    return job

if __name__ == "__main__":
    import argparse
    ap = argparse.ArgumentParser()
    ap.add_argument("--manifest", required=True)
    ap.add_argument("--script", required=True)
    ap.add_argument("--studio", default="vanguard-train")
    ap.add_argument("--machine", default="L40S")
    args = ap.parse_args()
    run_training(args.manifest, args.script, args.studio, args.machine)
```

### ingest.py
```python
#!/usr/bin/env python3
"""
Deterministic HF repo spread + schema projection to {prompt,response}.
"""
import hashlib
import json
import os
from pathlib import Path

try:
    from huggingface_hub import upload_file
except ImportError:
    print("pip install huggingface_hub")
    raise

HF_TOKEN = os.getenv("HF_TOKEN")
if not HF_TOKEN:
    print("HF_TOKEN required")
    raise RuntimeError("HF_TOKEN required")

SIBLING_REPOS = [
    "axentx/vanguard-mirror-0",
    "axentx/vanguard-mirror-1",
    "axentx/vanguard-mirror-2",
    "axentx/vanguard-mirror-3",
    "axentx/vanguard-mirror-4",
]

def pick_repo(slug: str) -> str:
    """Deterministic repo assignment to spread 128/hr/repo cap."""
    digest = hashlib.sha256(slug.encode()).digest()
    idx = digest[0] % len(SIBLING_REPOS)
    return SIBLING_REPOS[idx]

def project_record(raw: dict) -> dict:
    """Keep only {prompt,response}; attribution via filename."""
    return {
        "prompt": raw.get("prompt") or raw.get("input") or "",
        "response": raw.get("response") or raw.get("output") or "",
    }

def upload_batch(records, date: str, slug: str, hf_token: str):
    import tempfile
    import pyarrow as pa
    import pyarrow.parquet as pq

    repo = pick_repo(slug)
    out_dir = Path("batches/mirror-merged") / date
    out_dir.mkdir(parents=True, exist_ok=True)
    local_path = out_dir / f"{slug}.parquet"

    table = pa.Table.from_pylist([project_record(r) for r in records])
    pq.write_table(table, local_path)

    upload_file(
        path_or_fileobj=str(local_path),
        path_in_repo=f"batches/mirror-merged/{date}/{slug}.parquet",
        repo_id=repo,
        token=hf_token,
    )
    print(f"Uploaded to {repo}/batches/mirror-merged/{date}/{slug}.parquet")
```
