# Costinel / design

## Implementation Plan — Costinel Top-Hub Signal (Backend + Frontend)

**Scope:** Highest-value, read-only, <2h  
**Deliverable:**  
- Backend: `GET /api/v1/cost-anomaly/signal/top-hub` (sense + signal, zero side effects)  
- Frontend: Dashboard TopHubSignal widget consuming the endpoint  

---

### 1. Backend — FastAPI endpoint

**File:** `app/api/v1/endpoints/cost_anomaly.py` (create or extend)

```python
# app/api/v1/endpoints/cost_anomaly.py
from fastapi import APIRouter, Depends, HTTPException
from sqlalchemy.ext.asyncio import AsyncSession
from typing import List
from datetime import datetime, timedelta, timezone

from app.core.database import get_db
from app.services.cost.anomaly import compute_top_hub_signals
from app.schemas.cost_anomaly import TopHubSignal, TopHubSignalResponse

router = APIRouter(prefix="/cost-anomaly", tags=["cost-anomaly"])

@router.get(
    "/signal/top-hub",
    response_model=TopHubSignalResponse,
    summary="Top-hub cost-anomaly signals (sense + signal, read-only)",
)
async def get_top_hub_signals(
    days: int = 7,
    limit: int = 10,
    db: AsyncSession = Depends(get_db),
) -> TopHubSignalResponse:
    """
    Sense + Signal endpoint.
    - No mutations, no side effects.
    - Returns top N hubs by anomaly score + context.
    """
    if not (1 <= days <= 90):
        raise HTTPException(status_code=400, detail="days must be 1-90")
    if not (1 <= limit <= 50):
        raise HTTPException(status_code=400, detail="limit must be 1-50")

    try:
        signals = await compute_top_hub_signals(
            db=db,
            lookback_days=days,
            limit=limit,
        )
        return TopHubSignalResponse(
            request_ts=datetime.now(timezone.utc).isoformat(),
            lookback_days=days,
            limit=limit,
            signals=signals,
        )
    except Exception as exc:
        raise HTTPException(status_code=500, detail=str(exc)) from exc
```

```python
# app/schemas/cost_anomaly.py
from pydantic import BaseModel, Field
from typing import List, Optional, Dict, Any
from datetime import datetime

class TopHubSignal(BaseModel):
    hub_id: str = Field(..., description="Hub identifier (e.g., MOC)")
    hub_name: Optional[str] = None
    anomaly_score: float = Field(..., ge=0, le=100, description="0-100 severity")
    total_cost: float = Field(..., description="Total cost in lookback window (USD)")
    delta_pct: float = Field(..., description="Cost change vs prior period (%)")
    top_services: List[Dict[str, Any]] = Field(default_factory=list, description="Top cost services in hub")
    top_accounts: List[Dict[str, Any]] = Field(default_factory=list, description="Top accounts in hub")
    suggested_action: Optional[str] = Field(None, description="Human-readable recommendation")
    evidence: List[str] = Field(default_factory=list, description="Short evidence bullets")

class TopHubSignalResponse(BaseModel):
    request_ts: str
    lookback_days: int
    limit: int
    signals: List[TopHubSignal]
```

**Service:** `app/services/cost/anomaly.py`

```python
# app/services/cost/anomaly.py
from typing import List, Dict, Any
from sqlalchemy import select, func, text
from sqlalchemy.ext.asyncio import AsyncSession
from app.models.cost import CostRecord  # adjust import to your models
from app.schemas.cost_anomaly import TopHubSignal

async def compute_top_hub_signals(
    db: AsyncSession,
    lookback_days: int = 7,
    limit: int = 10,
) -> List[TopHubSignal]:
    """
    Read-only computation:
    - Aggregate cost by hub for lookback window and prior window.
    - Score anomalies by delta + volatility + service concentration.
    """
    now = func.now()
    lookback_start = func.date_sub(now, text(f"INTERVAL {lookback_days} DAY"))
    prior_start = func.date_sub(lookback_start, text(f"INTERVAL {lookback_days} DAY"))

    # Current period: cost by hub
    stmt_current = (
        select(
            CostRecord.hub_id,
            func.coalesce(func.json_extract(CostRecord.tags, "$.hub_name"), CostRecord.hub_id).label("hub_name"),
            func.sum(CostRecord.cost_usd).label("total_cost"),
            func.array_agg(CostRecord.service).label("services"),
            func.array_agg(CostRecord.account_id).label("accounts"),
        )
        .where(CostRecord.timestamp >= lookback_start)
        .group_by(CostRecord.hub_id)
    )
    result_current = (await db.execute(stmt_current)).mappings().all()

    # Prior period: cost by hub
    stmt_prior = (
        select(
            CostRecord.hub_id,
            func.sum(CostRecord.cost_usd).label("total_cost"),
        )
        .where(CostRecord.timestamp >= prior_start, CostRecord.timestamp < lookback_start)
        .group_by(CostRecord.hub_id)
    )
    result_prior = {r[0]: float(r[1] or 0) for r in (await db.execute(stmt_prior)).all()}

    signals: List[TopHubSignal] = []
    for row in result_current:
        hub_id = row["hub_id"]
        hub_name = row["hub_name"] or hub_id
        total_cost = float(row["total_cost"] or 0)
        prior_cost = result_prior.get(hub_id, 0.0)
        delta_pct = ((total_cost - prior_cost) / prior_cost * 100) if prior_cost > 0 else 0.0

        # Simple anomaly score: blend delta and absolute scale (0-100)
        delta_score = min(abs(delta_pct), 200) / 2  # cap 200% -> 100
        scale_score = min(total_cost / 5000, 50)  # $5k+ => 50
        anomaly_score = min(delta_score + scale_score, 100)

        # Top services/accounts summaries
        services = row["services"] or []
        accounts = row["accounts"] or []
        from collections import Counter
        top_services = [{"service": k, "cost_usd": None} for k, _ in Counter(services).most_common(3)]
        top_accounts = [{"account_id": k, "cost_usd": None} for k, _ in Counter(accounts).most_common(3)]

        # Evidence & action
        evidence: List[str] = []
        if delta_pct > 50:
            evidence.append(f"Cost increased {delta_pct:+.1f}% vs prior period.")
        if total_cost > 3000:
            evidence.append(f"High absolute spend (${total_cost:,.0f}).")
        if anomaly_score > 60:
            evidence.append("Anomaly score indicates significant deviation.")

        suggested_action = None
        if anomaly_score >= 70:
            suggested_action = "Review hub spend and consider immediate governance actions."
        elif anomaly_score >= 40:
            suggested_action = "Monitor hub and validate expected usage."

        signals.append(
            TopHubSignal(
                hub_id=hub_id,
                hub_name=hub_name,
                anomaly_score=round(anomaly_score, 1),
                total_cost=round(total_cost, 2),
                delta_pct=round(delta_pct, 1),
                top_services=top_services,
                top_accounts=top_accounts,
                suggested_action=suggested_action,
                evidence=evidence,
            )
        )

    # Sort by anomaly_score desc, then total_cost desc
    signals.sort(key=lambda x: (-x.anomaly_score, -x.total_cost))
    return signals[:limit]
```
