#### 1) Pre-cache file list (Mac/Linux orchestration)
`scripts/pre-cache_filelist.py`
```python
#!/usr/bin/env python3
"""
Pre-cache HF dataset file list for a single date folder.
Run from Mac (or any orchestration host) when rate-limit window is clear.
"""
import json
import os
import sys
from datetime import datetime, timezone
from pathlib import Path

from huggingface_hub import HfApi

HF_REPO = os.getenv("HF_DATASET_REPO", "my-org/surrogate-dataset")
DATE_FOLDER = os.getenv("DATE_FOLDER", datetime.now(timezone.utc).strftime("%Y-%m-%d"))
OUT_PATH = Path(__file__).parent.parent / "training" / "filelist.json"

def main() -> None:
    api = HfApi()
    # Non-recursive per folder to avoid pagination explosion
    entries = api.list_repo_tree(repo_id=HF_REPO, path=DATE_FOLDER, recursive=False)
    files = [e.path for e in entries if e.type == "file"]

    payload = {
        "repo": HF_REPO,
        "date_folder": DATE_FOLDER,
        "generated_at_utc": datetime.now(timezone.utc).isoformat(),
        "files": sorted(files),
    }

    OUT_PATH.parent.mkdir(parents=True, exist_ok=True)
    OUT_PATH.write_text(json.dumps(payload, indent=2))
    print(f"Wrote {len(files)} files to {OUT_PATH}")

if __name__ == "__main__":
    main()
```

#### 2) CDN-only data loader in training script
`surrogate/train.py` (excerpt)
```python
import json
import os
from pathlib import Path
from typing import List, Tuple

import requests
from torch.utils.data import IterableDataset

HF_CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

class CDNTextDataset(IterableDataset):
    def __init__(self, filelist_path: str, repo: str, max_files: int = -1):
        super().__init__()
        with open(filelist_path) as f:
            manifest = json.load(f)
        self.repo = repo or manifest["repo"]
        files = manifest["files"]
        if max_files > 0:
            files = files[:max_files]
        self.files = files

    def _project_to_pair(self, raw: str) -> Tuple[str, str]:
        # Lightweight projection to {prompt, response}.
        # Replace with your actual schema logic.
        parts = raw.split("\n\n", 1)
        if len(parts) == 2:
            return parts[0].strip(), parts[1].strip()
        return "", raw.strip()

    def _stream_file(self, path: str):
        url = HF_CDN_TEMPLATE.format(repo=self.repo, path=path)
        # CDN downloads: no Authorization header -> bypasses /api/ rate limits.
        with requests.get(url, timeout=30, stream=True) as r:
            r.raise_for_status()
            # If parquet, stream via pyarrow; if text, iterate lines.
            if path.endswith(".parquet"):
                import pyarrow.parquet as pq
                import io
                buf = io.BytesIO(r.content)
                table = pq.read_table(buf, columns=["prompt", "response"])
                for row in table.to_pylist():
                    yield row.get("prompt", ""), row.get("response", "")
            else:
                # fallback: treat as line-oriented text
                for line in r.iter_lines(decode_unicode=True):
                    if line:
                        yield self._project_to_pair(line)

    def __iter__(self):
        for path in self.files:
            yield from self._stream_file(path)
```
