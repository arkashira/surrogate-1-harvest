# workio / design

## Final Synthesized Solution

### Diagnosis (merged, highest-confidence)
- **Duplicate root cause**: LINE webhook redelivery + concurrent deliveries produce duplicate `punch` rows for same `user_id`/`date`/`type`.
- **Race condition**: Non-atomic check-then-insert allows two active punches (two clock-ins) under concurrency.
- **Missing DB invariants**: No unique constraint/index to enforce at most one open punch per user and no durable deduplication key for replay protection.
- **Partial failure risk**: Punch creation and related state updates are not transactional; partial failures leave inconsistent state.

### Schema (`workio/server/src/db/schema.sql`)
Apply these **safe, additive** changes:

```sql
-- Durable idempotency key (nullable for non-LINE sources)
ALTER TABLE punches
  ADD COLUMN IF NOT EXISTS line_delivery_id TEXT;

-- Deduplication: one row per delivery id
CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_line_delivery
  ON punches (line_delivery_id)
  WHERE line_delivery_id IS NOT NULL;

-- Core invariant: at most one open punch per user
CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_one_open_per_user
  ON punches (user_id)
  WHERE clock_out_at IS NULL;
```

Notes:
- Do **not** add a composite `(user_id, date, type, line_delivery_id)` index; the single `line_delivery_id` unique index is sufficient for deduplication and avoids unnecessary constraint collisions.
- Keep `line_delivery_id` nullable so non-LINE sources can still create punches (they will generate their own idempotency key).

### Service layer (`workio/server/src/services/punchService.ts`)
Centralize atomic punch logic so routes and tests share one correct implementation.

```ts
import { PoolClient } from 'pg';
import { db } from '../db';

export type PunchInput = {
  userId: number;
  type: 'clock_in' | 'clock_out';
  lineDeliveryId?: string | null;
  latitude?: number | null;
  longitude?: number | null;
};

export class PunchService {
  /**
   * Create a punch with idempotency and atomic invariants.
   * Returns [punch, duplicate].
   */
  static async createPunch(
    input: PunchInput,
    client?: PoolClient
  ): Promise<[any, boolean]> {
    const id = input.lineDeliveryId || crypto.randomUUID();
    const date = new Date().toISOString().split('T')[0];
    const ownTx = !client;
    const pg = client || (await db.connect());

    try {
      if (ownTx) await pg.query('BEGIN');

      // Idempotent insert (exact duplicate delivery)
      const insertSQL = `
        INSERT INTO punches (user_id, date, type, line_delivery_id, clock_in_at, latitude, longitude)
        VALUES ($1, $2, $3, $4, NOW(), $5, $6)
        ON CONFLICT (line_delivery_id) DO NOTHING
        RETURNING *
      `;
      const insertRes = await pg.query(insertSQL, [
        input.userId,
        date,
        input.type,
        id,
        input.latitude || null,
        input.longitude || null,
      ]);

      if (insertRes.rowCount === 0) {
        if (ownTx) await pg.query('ROLLBACK');
        // Fetch existing for idempotent response
        const existing = await pg.query('SELECT * FROM punches WHERE line_delivery_id = $1', [id]);
        return [existing.rows[0] || null, true];
      }

      const punch = insertRes.rows[0];

      // If clock-in, close any other open punch for this user (defensive invariant)
      if (input.type === 'clock_in') {
        await pg.query(
          'UPDATE punches SET clock_out_at = NOW(), updated_at = NOW() WHERE user_id = $1 AND clock_out_at IS NULL AND id != $2',
          [input.userId, punch.id]
        );
      }

      if (ownTx) await pg.query('COMMIT');
      return [punch, false];
    } catch (err) {
      if (ownTx) await pg.query('ROLLBACK');
      throw err;
    } finally {
      if (ownTx && pg) pg.release();
    }
  }

  /**
   * Clock-out a punch with optional idempotency key.
   * If lineDeliveryId is provided and already exists, treat as idempotent duplicate.
   */
  static async clockOut(
    punchId: string,
    lineDeliveryId?: string | null,
    client?: PoolClient
  ): Promise<[any, boolean]> {
    const ownTx = !client;
    const pg = client || (await db.connect());

    try {
      if (ownTx) await pg.query('BEGIN');

      // Idempotency check when caller provides a delivery id
      if (lineDeliveryId) {
        const dup = await pg.query('SELECT 1 FROM punches WHERE line_delivery_id = $1', [lineDeliveryId]);
        if (dup.rowCount > 0) {
          // Fetch and return the existing row for idempotency
          const existing = await pg.query('SELECT * FROM punches WHERE line_delivery_id = $1', [lineDeliveryId]);
          if (ownTx) await pg.query('ROLLBACK');
          return [existing.rows[0] || null, true];
        }
      }

      const updateSQL = `
        UPDATE punches
        SET clock_out_at = NOW(),
            line_delivery_id = COALESCE(line_delivery_id, $2),
            updated_at = NOW()
        WHERE id = $1 AND clock_out_at IS NULL
        RETURNING *
      `;
      const updateRes = await pg.query(updateSQL, [punchId, lineDeliveryId || null]);

      if (updateRes.rowCount === 0) {
        if (ownTx) await pg.query('ROLLBACK');
        throw new Error('Already clocked out or not found');
      }

      if (ownTx) await pg.query('COMMIT');
      return [updateRes.rows[0], false];
    } catch (err) {
      if (ownTx) await pg.query('ROLLBACK');
      throw err;
    } finally {
      if (ownTx && pg) pg.release();
    }
  }
}
```

### Route (`workio/server/src/routes/punch.ts`)
Thin route layer that delegates to the service.

```ts
import { Router } from 'express';
import { PunchService } from '../services/punchService';

const router = Router();

/**
 * POST /punch
 * Body: { userId, type, lineDeliveryId?, latitude?, longitude? }
 */
router.post('/', async (req, res) => {
  try {
    const { userId, type, lineDeliveryId, latitude, longitude } = req.body;
    const [punch, duplicate] = await PunchService.createPunch(
      { userId, type, lineDeliveryId, latitude, longitude },
      undefined // no client -> service manages tx
    );

    if (duplicate) {
      return res.status(200).json({ ok: true, duplicate: true, punch });
    }
    return res.status(201).json({ ok: true, punch });
  } catch (err) {
    console.error('Create punch failed', err);
    return res.status(500).json({ ok: false, error: 'Internal server error' });
  }
});

/**
 * POST /punch/:id/clock-out
 * Body: { lineDeliveryId? }
 */
router.post('/:id/clock-out', async (req, res) => {
  try {
    const { id } = req.params;
    const { lineDeliveryId } = req.body;
    const [punch, duplicate] = await PunchService.clockOut(id, lineDeliveryId, undefined);

    if (duplicate) {
      return res.status(200).json({ ok: true, duplicate: true, punch });
    }
    return res.status(200).json({ ok: true, punch });
  } catch (err: any) {
    if (err.message === 'Already clocked out or not found') {
      return res.status(409).json({ ok: false, error: err.message });
    }
    console.error('Clock-out failed', err);
    return res.status(500).json({ ok: false, error: 'Internal server error' });
