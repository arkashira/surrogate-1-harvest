# workio / ops

## Final synthesized implementation (best of both candidates)

**Core principle**: DB-level uniqueness + atomic upsert + idempotency key = guaranteed exactly-once punches even under LINE retries and concurrency.

### 1. DB schema changes (run once)

```sql
-- workio/server/src/db/schema.sql

-- 1) Prevent duplicate punches per user per day per type (per-tenant safe)
ALTER TABLE punches
  ADD CONSTRAINT uniq_punch_user_date_type_tenant
  UNIQUE (user_id, punch_date, punch_type, tenant_id);

-- 2) Idempotency table for webhook retries (LINE events or API calls)
CREATE TABLE IF NOT EXISTS punch_idempotency (
  idempotency_key TEXT PRIMARY KEY,
  punch_id        BIGINT NOT NULL REFERENCES punches(punch_id) ON DELETE CASCADE,
  created_at      TIMESTAMPTZ DEFAULT NOW()
);

-- Optional: indexes for performance
CREATE INDEX IF NOT EXISTS idx_punches_user_date ON punches (user_id, punch_date);
CREATE INDEX IF NOT EXISTS idx_punches_user_type_date ON punches (user_id, punch_type, punch_date);
```

### 2. Atomic upsert endpoint (controller)

```ts
// workio/server/src/controllers/punchController.ts
import { Request, Response } from 'express';
import { pool } from '../db';
import { v4 as uuidv4 } from 'uuid';

export const processPunch = async (req: Request, res: Response) => {
  const { userId, tenantId, punchType, latitude, longitude, timestamp } = req.body;
  const idempotencyKey = req.header('Idempotency-Key') || uuidv4();
  const punchDate = new Date(timestamp || Date.now()).toISOString().split('T')[0];

  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // Fast idempotency check
    const idem = await client.query(
      `SELECT punch_id FROM punch_idempotency WHERE idempotency_key = $1`,
      [idempotencyKey]
    );
    if (idem.rows[0]) {
      await client.query('ROLLBACK');
      return res.status(200).json({ punchId: idem.rows[0].punch_id, duplicate: true });
    }

    // Atomic upsert on business key
    const upsert = await client.query(
      `INSERT INTO punches (user_id, tenant_id, punch_date, punch_type, latitude, longitude, created_at)
       VALUES ($1, $2, $3, $4, $5, $6, NOW())
       ON CONFLICT (user_id, punch_date, punch_type, tenant_id)
       DO UPDATE SET
         latitude = EXCLUDED.latitude,
         longitude = EXCLUDED.longitude,
         updated_at = NOW()
       RETURNING punch_id`,
      [userId, tenantId, punchDate, punchType, latitude, longitude]
    );

    const punchId = upsert.rows[0].punch_id;

    // Record idempotency (best-effort; don't fail punch if this fails)
    try {
      await client.query(
        `INSERT INTO punch_idempotency (idempotency_key, punch_id) VALUES ($1, $2)`,
        [idempotencyKey, punchId]
      );
    } catch {
      // duplicate key is fine
    }

    // Fire-and-forget notification (do not block commit)
    sendLinePunchNotification(userId, punchType, punchDate).catch((e) =>
      console.warn('LINE notify failed', e)
    );

    await client.query('COMMIT');
    return res.status(200).json({ punchId, duplicate: false });
  } catch (err: any) {
    await client.query('ROLLBACK');
    if (err.code === '23505') {
      return res.status(409).json({ error: 'Duplicate punch' });
    }
    console.error('Punch processing failed', err);
    return res.status(500).json({ error: 'Internal server error' });
  } finally {
    client.release();
  }
};
```

### 3. LINE webhook handler (idempotent by LINE event)

```ts
// workio/server/src/routes/line/webhook.ts
import { Request, Response } from 'express';
import { pool } from '../../db';
import crypto from 'crypto';

function verifyLineSignature(rawBody: string, signature: string, channelSecret: string): boolean {
  const expected = crypto
    .createHmac('sha256', channelSecret)
    .update(rawBody)
    .digest('base64');
  return signature === expected;
}

export async function lineWebhook(req: Request, res: Response) {
  const channelSecret = process.env.LINE_CHANNEL_SECRET || '';
  const signature = req.headers['x-line-signature'] as string;
  const rawBody = typeof req.body === 'string' ? req.body : JSON.stringify(req.body);

  if (!verifyLineSignature(rawBody, signature, channelSecret)) {
    return res.status(401).json({ error: 'Invalid signature' });
  }

  const events = req.body.events || [];
  const client = await pool.connect();

  try {
    await client.query('BEGIN');

    for (const ev of events) {
      if (ev.type !== 'message' && ev.type !== 'postback') continue;

      // Use stable LINE-derived idempotency key
      const idemKey = `${ev.source?.userId}:${ev.timestamp}:${ev.type}:${ev.message?.id || ev.postback?.data || ''}`;
      if (!idemKey) continue;

      const idemCheck = await client.query(
        `SELECT 1 FROM punch_idempotency WHERE idempotency_key = $1`,
        [idemKey]
      );
      if (idemCheck.rows.length > 0) continue;

      // Determine punch details from event (customize as needed)
      const userId = ev.source?.userId;
      const tenantId = ev.source?.groupId || 'default';
      const punchType = ev.postback?.data?.includes('in') ? 'IN' : 'OUT';
      const punchDate = new Date().toISOString().split('T')[0];

      // Atomic upsert
      const upsert = await client.query(
        `INSERT INTO punches (user_id, tenant_id, punch_date, punch_type, created_at)
         VALUES ($1, $2, $3, $4, NOW())
         ON CONFLICT (user_id, punch_date, punch_type, tenant_id)
         DO UPDATE SET updated_at = NOW()
         RETURNING punch_id`,
        [userId, tenantId, punchDate, punchType]
      );

      const punchId = upsert.rows[0].punch_id;

      try {
        await client.query(
          `INSERT INTO punch_idempotency (idempotency_key, punch_id) VALUES ($1, $2)`,
          [idemKey, punchId]
        );
      } catch {
        // duplicate is fine
      }
    }

    await client.query('COMMIT');
    return res.status(200).json({});
  } catch (err) {
    await client.query('ROLLBACK');
    console.error('LINE webhook failed', err);
    return res.status(500).json({ error: 'Internal server error' });
  } finally {
    client.release();
  }
}
```
