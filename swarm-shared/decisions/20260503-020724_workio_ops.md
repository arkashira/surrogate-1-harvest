# workio / ops

## Final Synthesized Implementation Plan

**Priority:** Production resilience + immediate duplicate-prevention + zero-downtime deployability.

---

### 1. Diagnosis (Consolidated)
- **Silent process death**: No supervision, no graceful shutdown, no crash handlers.
- **No observability**: Missing structured logs, request tracing, and health endpoints for load balancers/orchestration.
- **Duplicate LINE events**: Schema-level idempotency exists, but runtime handlers lack enforcement (no idempotency key check or retry dedupe).
- **Static secrets/config**: `.env` only; no hot-reload or per-tenant overrides for LINE tokens.
- **Operational runbook gaps**: No automated handling for LINE 429/5xx, DB loss, or queue backlogs.

---

### 2. Proposed Changes (Unified)

**New files:**
- `server/src/ops/logging.ts` — structured JSON logs (Pino) with redaction.
- `server/src/ops/health.ts` — `/healthz` (liveness) and `/readyz` (readiness) with DB + lightweight LINE token reachability.
- `server/src/middleware/observability.ts` — request/response logging + error/latency counters.
- `server/src/middleware/idempotency.ts` — enforce `X-Idempotency-Key` on state-changing endpoints; return `409` on duplicates.
- `server/src/config.ts` — hot-reloadable config with per-tenant LINE tokens and safe defaults.
- `server/src/ops/graceful-shutdown.ts` — graceful Express shutdown + uncaught/unhandled rejection handlers.

**Edits:**
- `server/src/index.ts` (or `app.ts`) — wire logging, middleware, health routes, graceful shutdown.
- `package.json` — add `start`, `dev`, `build`, and optional `start:ops`.
- `/opt/axentx/workio/server/start.sh` — deployment helper with log dir creation.

#### `server/src/config.ts`

```ts
import dotenv from 'dotenv';
dotenv.config();

function getTokensFromEnv(): Record<string, string> {
  // Expected: LINE_TOKEN_TENANT_A=..., LINE_TOKEN_TENANT_B=...
  const tokens: Record<string, string> = {};
  for (const [k, v] of Object.entries(process.env)) {
    if (k.startsWith('LINE_TOKEN_') && v) {
      const tenant = k.replace(/^LINE_TOKEN_/, '').toLowerCase();
      tokens[tenant] = v;
    }
  }
  return tokens;
}

export const CONFIG = {
  port: Number(process.env.PORT || 4000),
  nodeEnv: process.env.NODE_ENV || 'development',
  dbUrl: process.env.DATABASE_URL || '',
  logLevel: process.env.LOG_LEVEL || 'info',
  shutdownTimeoutMs: Number(process.env.SHUTDOWN_TIMEOUT_MS || 30000),
  lineTokens: getTokensFromEnv(),
  defaultLineChannelSecret: process.env.LINE_CHANNEL_SECRET || '',
};

// Hot-reload helper (call on SIGHUP or admin endpoint)
export function reloadConfig() {
  dotenv.config();
  const updated = getTokensFromEnv();
  Object.assign(CONFIG.lineTokens, updated);
  CONFIG.logLevel = process.env.LOG_LEVEL || CONFIG.logLevel;
  CONFIG.defaultLineChannelSecret = process.env.LINE_CHANNEL_SECRET || CONFIG.defaultLineChannelSecret;
  return CONFIG;
}
```

#### `server/src/ops/health.ts`
```ts
import { Router, Request, Response } from 'express';
import logger from './logging';

let isReady = false;

export function setReady(ready: boolean) {
  isReady = ready;
}

export const healthRouter = Router();

// Liveness: process alive
healthRouter.get('/healthz', (_req: Request, res: Response) => {
  res.json({ status: 'ok', uptime: process.uptime() });
});

// Readiness: app + db + minimal LINE token presence
healthRouter.get('/readyz', async (req: Request, res: Response) => {
  if (!isReady) {
    return res.status(503).json({ status: 'not_ready' });
  }

  // DB check (lightweight)
  try {
    const db = (req as any).db || require('../db').default;
    if (db) {
      if (typeof db.$queryRaw === 'function') {
        await db.$queryRaw`SELECT 1`;
      } else if (typeof db.query === 'function') {
        await db.query('SELECT 1');
      } else if (typeof db.authenticate === 'function') {
        await db.authenticate();
      }
    }
  } catch (err) {
    logger.error({ err }, 'DB readiness check failed');
    return res.status(503).json({ status: 'db_unavailable' });
  }

  // Minimal LINE token presence check (config-based)
  const { lineTokens } = require('../config').CONFIG;
  const hasTokens = Object.keys(lineTokens).length > 0;
  if (!hasTokens) {
    return res.status(503).json({ status: 'line_tokens_missing' });
  }

  return res.json({ status: 'ready' });
});
```

#### `server/src/middleware/observability.ts`
```ts
import { Request, Response, NextFunction } from 'express';
import logger from '../ops/logging';

export function observabilityMiddleware(req: Request, res: Response, next: NextFunction) {
  const start = Date.now();
  const originalSend = res.send;
  const originalJson = res.json;

  // Capture status and body size minimally
  res.send = function (...args) {
    logRequest(req, res, start);
    return originalSend.apply(this, args as any);
  };
  res.json = function (...args) {
    logRequest(req, res, start);
    return originalJson.apply(this, args as any);
  };

  next();
}

function logRequest(req: Request, res: Response, start: number) {
  const duration = Date.now() - start;
  logger.info({
    method: req.method,
    url: req.originalUrl || req.url,
    status: res.statusCode,
    durationMs: duration,
    ip: req.ip,
    userAgent: req.get('User-Agent'),
  }, 'request');
}
```
