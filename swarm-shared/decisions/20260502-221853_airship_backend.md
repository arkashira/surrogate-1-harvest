```bash
#!/usr/bin/env bash
# airship — Arkship + Surrogate AI Platform CLI
# Location: /opt/axentx/airship/bin/airship
# Purpose: Zero-config operational commands (discover, health)
# Cron-safe: non-interactive, timestamped logs, no auth prompts

set -euo pipefail

readonly SCRIPT_NAME="airship"
readonly PROJECT_ROOT="$(cd "$(dirname "${BASH_SOURCE[0]}")/.." && pwd)"
readonly LOG_PREFIX="[$(date -u +'%Y-%m-%dT%H:%M:%SZ')] $SCRIPT_NAME"

# ---- helpers ----
log_info()  { echo "$LOG_PREFIX [INFO] $*" >&2; }
log_warn()  { echo "$LOG_PREFIX [WARN] $*" >&2; }
log_error() { echo "$LOG_PREFIX [ERROR] $*" >&2; }

die() {
  log_error "$*"
  exit "${2:-1}"
}

require_cmd() {
  if ! command -v "$1" >/dev/null 2>&1; then
    die "required command not found: $1" 2
  fi
}

is_tty() {
  [[ -t 1 ]]
}

# ---- subcommands ----
discover() {
  log_info "Starting strategic discovery (business research + knowledge-rag + top-hub insight)"

  # 1) Business research (if available)
  local research_script="${PROJECT_ROOT}/scripts/granite-business-research.sh"
  if [[ -x "$research_script" ]]; then
    log_info "Running business research: $research_script"
    if ! bash "$research_script"; then
      log_warn "Business research script exited non-zero; continuing discovery"
    fi
  else
    log_warn "Business research script not found or not executable: $research_script (skipping)"
  fi

  # 2) Knowledge-RAG: query top hub and related docs
  local knowledge_rag_cmd=""
  for try_cmd in "knowledge-rag" "${PROJECT_ROOT}/scripts/knowledge-rag"; do
    if command -v "$try_cmd" >/dev/null 2>&1 || [[ -x "$try_cmd" ]]; then
      knowledge_rag_cmd="$try_cmd"
      break
    fi
  done

  if [[ -n "$knowledge_rag_cmd" ]]; then
    log_info "Querying knowledge-rag for top hub and related docs"
    # Try structured top-hub query first, then fallback
    if ! "$knowledge_rag_cmd" query --top-hub --limit 5 2>/dev/null; then
      "$knowledge_rag_cmd" query "top hub and related documents" --limit 5 || true
    fi

    # Most-connected hub (corrected: use query, not assignment from subshell that fails)
    local most_connected_hub
    most_connected_hub="$("$knowledge_rag_cmd" query "most connected hub" --limit 1 2>/dev/null | head -n 5 | tr -d '\n')" || true
    if [[ -n "$most_connected_hub" ]]; then
      log_info "Most connected hub: $most_connected_hub"
      # Surface contextual insights for that hub
      "$knowledge_rag_cmd" query "contextual insights for $most_connected_hub" --limit 5 2>/dev/null || true
    else
      log_warn "Could not determine most connected hub; skipping hub-specific insights"
    fi
  else
    log_warn "knowledge-rag not available; skipping RAG queries (install or add to PATH to enable)"
  fi

  # 3) Top-hub insight summary (heuristic)
  log_info "Top-hub insight: review the most-connected hub (e.g., MOC) before planning tasks (pattern: #knowledge-rag #graph #hub)"

  # 4) Print actionable summary
  if is_tty; then
    echo ""
    echo "== airship discover summary =="
    echo "  - Business research: ${research_script:+executed} ${research_script:-skipped}"
    echo "  - Knowledge-RAG: $([[ -n "$knowledge_rag_cmd" ]] && echo 'queried' || echo 'unavailable')"
    echo "  - Recommendation: validate top hub (MOC) connections and update runbooks"
    echo "  - Patterns applied: #business-research #knowledge-rag #graph #hub"
    echo "==============================="
  else
    echo "discover completed: research=${research_script:+executed} rag=$([[ -n "$knowledge_rag_cmd" ]] && echo queried || echo unavailable)"
  fi

  log_info "Discovery complete"
}

health() {
  log_info "Health check (basic)"
  local checks=0
  if command -v docker >/dev/null 2>&1; then
    log_info "docker: available"
  else
    log_warn "docker: not found"
    checks=$((checks+1))
  fi

  if pgrep -f "arkship" >/dev/null 2>&1 || pgrep -f "surrogate" >/dev/null 2>&1; then
    log_info "arkship/surrogate processes: detected"
  else
    log_warn "arkship/surrogate processes: not detected (may be stopped)"
    checks=$((checks+1))
  fi

  return $checks
}

usage() {
  cat <<EOF
Usage: $SCRIPT_NAME <command> [options]

Commands:
  discover   Run strategic discovery (business research + knowledge-rag + top-hub insight)
  health     Basic health checks
  help       Show this help

Examples:
  $SCRIPT_NAME discover
  $SCRIPT_NAME health

Patterns applied:
  - #business-research #knowledge-rag #graph #hub
  - Cron-safe: non-interactive, timestamped logs, no auth prompts

Notes:
  - Designed for automation (cron/CI) and local operator use.
  - Uses project root: $PROJECT_ROOT
EOF
}

# ---- main ----
main() {
  local cmd="${1:-help}"
  shift || true

  case "$cmd" in
    discover) discover "$@" ;;
    health)   health "$@" ;;
    help|--help|-h) usage; exit 0 ;;
    *)
      log_error "unknown command: $cmd"
      usage
      exit 1
      ;;
  esac
}

main "$@"
```
