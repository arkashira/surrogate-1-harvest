# vanguard / ops

## 1. Diagnosis
- No persisted `(repo, dateFolder) → file-list` manifest: every training/data-selection run triggers authenticated `list_repo_tree`, burning HF API quota and risking 429s.
- Data loader uses `load_dataset(streaming=True)` or repeated per-file API calls on heterogeneous repos, causing `pyarrow.CastError` and extra rate-limit pressure.
- No CDN-only data loading path: training still performs authenticated fetches instead of using public CDN URLs (`https://huggingface.co/datasets/{repo}/resolve/main/{path}`) which bypass auth and rate limits.
- No studio reuse guard: training scripts create new Lightning Studio sessions instead of reusing running ones, wasting ~80hr/mo quota.
- No idle-stop resilience: Lightning idle timeout kills long-running training; no pre-run status check + restart logic.

## 2. Proposed change
Add a small, ops-focused module that:
- Persists a `file-list` JSON for a given `(repo, dateFolder)` after a single authenticated `list_repo_tree` call.
- Exposes a CDN-only dataloader that reads parquet shards via direct HTTPS (no auth).
- Wraps Lightning Studio creation with reuse + idle-restart logic.

Scope:
- Create `/opt/axentx/vanguard/ops/hf_cdn_loader.py`
- Create `/opt/axentx/vanguard/ops/lightning_studio.py`
- Add CLI entrypoint script `/opt/axentx/vanguard/bin/build_file_list.py`
- Update existing training launcher (if any) to import and use these (minimal patch to main train script).

### ops/hf_cdn_loader.py
```python
import json
import os
import time
from datetime import datetime
from pathlib import Path
from typing import Dict, List, Optional

import pyarrow.parquet as pq
import requests
from huggingface_hub import HfApi, list_repo_tree

CDN_ROOT = "https://huggingface.co/datasets"

def build_file_list(
    repo: str,
    date_folder: str,
    out_path: Path,
    revision: str = "main",
    recursive: bool = True,
) -> List[str]:
    """
    Single authenticated call to list_repo_tree for one date folder.
    Persists JSON: {repo, date_folder, revision, ts, files: [...]}
    """
    api = HfApi()
    tree = list_repo_tree(
        repo_id=repo,
        path=date_folder,
        recursive=recursive,
        revision=revision,
    )
    files = [f.rfilename for f in tree if f.type == "file"]

    manifest = {
        "repo": repo,
        "date_folder": date_folder,
        "revision": revision,
        "ts": datetime.utcnow().isoformat() + "Z",
        "files": sorted(files),
    }
    out_path.parent.mkdir(parents=True, exist_ok=True)
    out_path.write_text(json.dumps(manifest, indent=2))
    return files


def load_file_list(repo: str, date_folder: str, cache_dir: Path) -> List[str]:
    cache_file = cache_dir / f"{repo.replace('/', '_')}_{date_folder.replace('/', '_')}.json"
    if cache_file.exists():
        data = json.loads(cache_file.read_text())
        # simple TTL 24h; rebuild if older
        ts = datetime.fromisoformat(data["ts"].rstrip("Z"))
        if (datetime.utcnow() - ts).total_seconds() < 86400:
            return data["files"]

    # rebuild
    return build_file_list(repo, date_folder, cache_file)


def cdn_parquet_reader(
    repo: str,
    file_paths: List[str],
    columns: Optional[List[str]] = None,
    batch_size: int = 10_000,
):
    """
    Yield record batches from parquet files via CDN (no auth).
    Projects only selected columns; avoids schema heterogeneity issues.
    """
    for fp in file_paths:
        url = f"{CDN_ROOT}/{repo}/resolve/main/{fp}"
        resp = requests.get(url, timeout=30)
        resp.raise_for_status()
        with pq.ParquetFile(pq.ParquetFile(pq.BufferReader(resp.content))) as pf:
            if columns:
                # project to avoid mixed-schema errors
                cols = [c for c in columns if c in pf.schema.names]
            else:
                cols = None
            for batch in pf.iter_batches(batch_size=batch_size, columns=cols):
                yield batch.to_pylist()
```

### ops/lightning_studio.py
```python
import time
from typing import Optional

from lightning import Studio, L, Machine, Teamspace

def get_or_create_studio(
    name: str,
    machine: Machine = Machine.L40S,
    teamspace: Optional[str] = None,
    max_retries: int = 3,
    retry_delay: int = 10,
) -> Studio:
    """
    Reuse running studio if exists; restart if stopped.
    Avoids quota waste from repeated creates.
    """
    ts = Teamspace(teamspace) if teamspace else Teamspace.current()
    for s in ts.studios:
        if s.name == name:
            if s.status == "running":
                return Studio(id=s.id)
            # stopped/idle: restart
            target = Studio(id=s.id)
            target.start(machine=machine)
            _wait_running(target, max_retries, retry_delay)
            return target

    # create new
    studio = Studio(
        name=name,
        machine=machine,
        teamspace=teamspace,
        create_ok=True,
    )
    _wait_running(studio, max_retries, retry_delay)
    return studio


def _wait_running(studio: Studio, max_retries: int, delay: int):
    for _ in range(max_retries):
        studio.refresh()
        if studio.status == "running":
            return
        time.sleep(delay)
    raise RuntimeError(f"Studio {studio.name} failed to reach running state")
```

### ops/build_file_list.py (module entry)
```python
if __name__ == "__main__":
    import argparse
    from .hf_cdn_loader import build_file_list
    from pathlib import Path

    p = argparse.ArgumentParser()
    p.add_argument("--repo", required=True)
    p.add_argument("--date-folder", required=True)
    p.add_argument("--out-dir", default=".ops/file_lists")
    p.add_argument("--revision", default="main")
    args = p.parse_args()

    out_dir = Path(args.out_dir)
    out_dir.mkdir(parents=True, exist_ok=True)
    slug = args.repo.replace("/", "_")
    date_slug = args.date_folder.replace("/", "_")
    out_path = out_dir / f"{slug}_{date_slug}.json"

    files = build_file_list(args.repo, args.date_folder, out_path, revision=args.revision)
    print(f"Wrote {len(files)} files to {out_path}")
```
