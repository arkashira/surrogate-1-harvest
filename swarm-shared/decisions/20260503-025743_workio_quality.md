# workio / quality

## Final Synthesis & Actionable Plan

**Core diagnosis (accepted from both candidates):**  
- LINE webhook lacks idempotency → duplicates on retries.  
- No tenant-scoped, one-open-punch guarantee → race conditions in clock-in/out.  
- App-level checks are insufficient; constraints and atomic operations are required.  
- Clock-in/out must be safe across webhook and API paths.

**Chosen approach:**  
- Enforce idempotency at the webhook boundary immediately (fast 200, then async).  
- Enforce one-open-punch at the database level with a partial unique index and atomic upsert/locking.  
- Centralize punch state transitions behind a single service used by both webhook processor and API routes.  
- Use tenant_id everywhere (never rely on global employee_id alone).

```sql
CREATE TABLE IF NOT EXISTS line_webhook_idempotency (
  tenant_id      uuid   NOT NULL,
  line_event_id  text   NOT NULL,
  created_at     timestamptz NOT NULL DEFAULT now(),
  processed_at   timestamptz,
  payload_hash   text,
  PRIMARY KEY (tenant_id, line_event_id)
);

-- At most one open punch per tenant+employee
CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_one_open_per_tenant_employee
  ON punches (tenant_id, employee_id)
  WHERE clock_out_at IS NULL;
```

## 2. Idempotency utility

`workio/server/src/lib/idempotency.ts`

```ts
import { PoolClient } from 'pg';

export async function ensureLineWebhookIdempotent(
  client: PoolClient,
  tenantId: string,
  lineEventId: string,
  payloadHash?: string
): Promise<boolean> {
  const res = await client.query(
    `INSERT INTO line_webhook_idempotency (tenant_id, line_event_id, payload_hash, processed_at)
     VALUES ($1, $2, $3, now())
     ON CONFLICT (tenant_id, line_event_id) DO NOTHING
     RETURNING 1`,
    [tenantId, lineEventId, payloadHash]
  );
  return res.rowCount === 1;
}
```

## 3. Punch service (atomic, tenant-scoped)

`workio/server/src/services/punchService.ts`

```ts
import { PoolClient } from 'pg';

export async function clockIn(
  client: PoolClient,
  tenantId: string,
  employeeId: string,
  clockInAt: Date,
  location?: { lat: number; lng: number }
) {
  // Atomic insert guarded by partial unique index
  const res = await client.query(
    `INSERT INTO punches (tenant_id, employee_id, clock_in_at, location)
     SELECT $1, $2, $3, $4
     WHERE NOT EXISTS (
       SELECT 1 FROM punches
       WHERE tenant_id = $1 AND employee_id = $2 AND clock_out_at IS NULL
     )
     RETURNING *`,
    [tenantId, employeeId, clockInAt, JSON.stringify(location || null)]
  );

  if (res.rowCount === 0) {
    const open = await client.query(
      `SELECT id, clock_in_at FROM punches
       WHERE tenant_id = $1 AND employee_id = $2 AND clock_out_at IS NULL
       LIMIT 1`,
      [tenantId, employeeId]
    );
    const err: any = new Error('Employee already has an open punch');
    err.existing = open.rows[0];
    throw err;
  }
  return res.rows[0];
}

export async function clockOut(
  client: PoolClient,
  tenantId: string,
  employeeId: string,
  clockOutAt: Date,
  location?: { lat: number; lng: number }
) {
  const res = await client.query(
    `UPDATE punches
     SET clock_out_at = $3, location_out = $4
     WHERE id = (
       SELECT id FROM punches
       WHERE tenant_id = $1 AND employee_id = $2 AND clock_out_at IS NULL
       ORDER BY clock_in_at DESC
       LIMIT 1
       FOR UPDATE SKIP LOCKED
     )
     RETURNING *`,
    [tenantId, employeeId, clockOutAt, JSON.stringify(location || null)]
  );

  if (res.rowCount === 0) {
    const err: any = new Error('No open punch found for employee');
    err.code = 'NO_OPEN_PUNCH';
    throw err;
  }
  return res.rows[0];
}
```

## 4. Webhook handler (fast ack + idempotent async processing)

`workio/server/src/routes/webhook/line.ts`

```ts
import express from 'express';
import { pool } from '../../db';
import { ensureLineWebhookIdempotent } from '../../lib/idempotency';
import { processLineEvent } from '../../services/lineEventProcessor';

const router = express.Router();

router.post('/line', async (req, res) => {
  const events = req.body?.events;
  if (!Array.isArray(events) || events.length === 0) {
    return res.sendStatus(200);
  }

  // Acknowledge quickly to prevent LINE retries
  res.sendStatus(200);

  for (const ev of events) {
    const tenantId = await resolveTenantByChannelId(ev.source?.channelId || ev.source?.userId);
    if (!tenantId) continue;

    const isNew = await runWithClient(async (client) => {
      return ensureLineWebhookIdempotent(client, tenantId, ev.id, hash(ev));
    });
    if (!isNew) continue;

    // Async processing; failures are logged but won't cause LINE retries
    (async () => {
      try {
        await processLineEvent(tenantId, ev);
      } catch (err) {
        console.error('Failed to process LINE event', ev.id, err);
      }
    })();
  }
});

async function runWithClient<T>(fn: (c: any) => Promise<T>) {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');
    const result = await fn(client);
    await client.query('COMMIT');
    return result;
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  } finally {
    client.release();
  }
}

function hash(obj: any): string {
  return require('crypto').createHash('sha256').update(JSON.stringify(obj)).digest('hex');
}

export default router;
```

## 5. API routes (clock-in/out) — use the same service

Example (clock-in):

```ts
import express from 'express';
import { pool } from '../db';
import { clockIn } from '../services/punchService';

const router = express.Router();

router.post('/clock-in', async (req, res) => {
  const { tenantId, employeeId, location } = req.body;
  if (!tenantId || !employeeId) return res.status(400).json({ error: 'tenantId and employeeId required' });

  const client = await pool.connect();
  try {
    await client.query('BEGIN');
    const punch = await clockIn(client, tenantId, employeeId, new Date(), location);
    await client.query('COMMIT');
    return res.json(punch);
  } catch (err: any) {
    await client.query('ROLLBACK');
    return res.status(409).json({ error: err.message, existing: err.existing });
  } finally {
    client.release();
  }
});

export default router;
```

Do the same pattern for clock-out using `clockOut`.
