# workio / quality

## 1) Diagnosis

- **No server-side idempotency key**: clock-in/out endpoints accept duplicate requests for the same `(employee_id, punch_type, date)` when clients retry on flaky mobile networks.
- **Race condition on concurrent punches**: two near-simultaneous requests can both pass the “last punch was > N minutes ago” check and create duplicate records.
- **Missing database uniqueness constraint**: the `attendance_punches` table has no unique index to prevent duplicates at the data layer.
- **Client retry logic is naive**: frontend uses simple `fetch` with no exponential backoff or idempotency header propagation.
- **No request deduplication window**: transient LINE webhook retries or user double-taps within seconds produce multiple punches instead of one.

## 2) Proposed change

Add server-side idempotency and database-level uniqueness for attendance punches.

- **Files**:
  - `workio/server/src/db/schema.sql` — add unique constraint on `(employee_id, punch_type, punch_date)` and an idempotency table.
  - `workio/server/src/routes/attendance.ts` — apply idempotency check before creating punches; use `ON CONFLICT DO NOTHING` and return existing record.
- **Scope**: ~30 lines changed; focused on the punch creation path only.

## 3) Implementation

### `workio/server/src/db/schema.sql`

```sql
-- Add unique constraint to prevent duplicate punches per employee per day per type
ALTER TABLE attendance_punches
  ADD CONSTRAINT uq_employee_punch_date_type
  UNIQUE (employee_id, punch_type, punch_date);

-- Idempotency table for request deduplication (48h retention)
CREATE TABLE IF NOT EXISTS request_idempotency (
  idempotency_key TEXT PRIMARY KEY,
  employee_id     INTEGER NOT NULL,
  endpoint        TEXT    NOT NULL,
  request_hash    TEXT    NOT NULL,
  response_body   JSONB   NOT NULL,
  created_at      TIMESTAMPTZ DEFAULT NOW()
);

-- Optional: index for cleanup
CREATE INDEX IF NOT EXISTS idx_idempotency_created ON request_idempotency (created_at);
```

### `workio/server/src/routes/attendance.ts`

```ts
import { Router, Request, Response } from 'express';
import { pool } from '../db';
import crypto from 'crypto';

const router = Router();

function hashRequestBody(body: any): string {
  return crypto.createHash('sha256').update(JSON.stringify(body)).digest('hex');
}

// Idempotent punch creation
router.post('/punch', async (req: Request, res: Response) => {
  const { employee_id, punch_type, punch_date, latitude, longitude } = req.body;
  const idempotencyKey = req.header('Idempotency-Key') || req.header('X-Idempotency-Key');

  if (!idempotencyKey) {
    return res.status(400).json({ error: 'Idempotency-Key header required' });
  }

  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // Check idempotency table
    const idem = await client.query(
      `SELECT response_body FROM request_idempotency WHERE idempotency_key = $1 AND endpoint = $2`,
      [idempotencyKey, '/punch']
    );

    if (idem.rows.length > 0) {
      await client.query('ROLLBACK');
      return res.status(200).json(idem.rows[0].response_body);
    }

    const requestHash = hashRequestBody(req.body);

    // Try insert; on conflict return existing punch
    const result = await client.query(
      `INSERT INTO attendance_punches (employee_id, punch_type, punch_date, latitude, longitude)
       VALUES ($1, $2, $3, $4, $5)
       ON CONFLICT (employee_id, punch_type, punch_date) DO UPDATE
         SET latitude = EXCLUDED.latitude,
             longitude = EXCLUDED.longitude,
             updated_at = NOW()
       RETURNING *`,
      [employee_id, punch_type, punch_date, latitude, longitude]
    );

    const punch = result.rows[0];

    // Store idempotent response
    await client.query(
      `INSERT INTO request_idempotency (idempotency_key, employee_id, endpoint, request_hash, response_body)
       VALUES ($1, $2, $3, $4, $5)`,
      [idempotencyKey, employee_id, '/punch', requestHash, JSON.stringify(punch)]
    );

    await client.query('COMMIT');
    return res.status(201).json(punch);
  } catch (err: any) {
    await client.query('ROLLBACK');
    // If unique violation still slips through, return existing punch
    if (err.code === '23505') {
      const exist = await pool.query(
        `SELECT * FROM attendance_punches WHERE employee_id = $1 AND punch_type = $2 AND punch_date = $3`,
        [employee_id, punch_type, punch_date]
      );
      if (exist.rows.length) return res.status(200).json(exist.rows[0]);
    }
    console.error(err);
    return res.status(500).json({ error: 'Internal server error' });
  } finally {
    client.release();
  }
});
```

### Frontend (optional, minimal)

In the punch caller (e.g., `workio/src/lib/services/attendance.ts`), attach an idempotency key:

```ts
export async function createPunch(payload: any) {
  const idempotencyKey = `punch-${payload.employee_id}-${payload.punch_type}-${payload.punch_date}-${Date.now()}`;
  return fetch(`${API_URL}/punch`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Idempotency-Key': idempotencyKey,
    },
    body: JSON.stringify(payload),
  }).then((r) => r.json());
}
```
