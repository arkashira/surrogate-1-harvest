## 1) One-time Mac script: deterministic CDN file list

- Uses `list_repo_tree` once per date folder (non-recursive) → avoids pagination and `/api/` rate limits.
- Embeds deterministic repo sharding for commit-cap scaling.
- Emits `file_list.json` consumed by training.

```python
# surrogate/training/gen_file_list.py
#!/usr/bin/env python3
"""
Generate CDN-only file list for a date folder in the mirror repo.
Usage:
  python gen_file_list.py --repo org/mirror-dataset --date 2026-05-03 --out file_list.json
"""

import argparse
import hashlib
import json
import os
import sys
from typing import List, Dict

from huggingface_hub import HfApi

HF_API = HfApi()
N_SIBLINGS = 5  # sibling repos for commit-cap scaling


def shard_for_slug(slug: str) -> int:
    return int(hashlib.sha256(slug.encode()).hexdigest(), 16) % N_SIBLINGS


def build_file_list(repo: str, date: str, out_path: str) -> None:
    prefix = f"batches/mirror-merged/{date}/"
    # non-recursive to avoid pagination explosion
    items = HF_API.list_repo_tree(repo=repo, path=prefix, recursive=False)

    files: List[Dict[str, str]] = []
    for item in items:
        if item.rfilename.endswith(".parquet"):
            slug = os.path.splitext(os.path.basename(item.rfilename))[0]
            files.append(
                {
                    "cdn_url": f"https://huggingface.co/datasets/{repo}/resolve/main/{item.rfilename}",
                    "path": item.rfilename,
                    "slug": slug,
                    "shard_repo": f"{repo}-shard-{shard_for_slug(slug)}",
                }
            )

    payload = {
        "repo": repo,
        "date": date,
        "prefix": prefix,
        "files": files,
    }

    os.makedirs(os.path.dirname(os.path.abspath(out_path)), exist_ok=True)
    with open(out_path, "w") as f:
        json.dump(payload, f, indent=2)

    print(f"Wrote {len(files)} files to {out_path}")


if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Generate CDN file list for training")
    parser.add_argument("--repo", default="org/mirror-dataset", help="HF dataset repo")
    parser.add_argument("--date", required=True, help="Date folder (YYYY-MM-DD)")
    parser.add_argument("--out", default="file_list.json", help="Output JSON path")
    args = parser.parse_args()
    build_file_list(args.repo, args.date, args.out)
```

Make executable:

```bash
chmod +x surrogate/training/gen_file_list.py
```

Optional cron wrapper (Bash best practice):

```bash

## 2) CDN-only dataloader (zero HF API during training)

- Loads `file_list.json`.
- Streams parquet directly from CDN URLs via `datasets` with `streaming=True`.
- Robust projection to `{prompt, response}`; ignores mixed schemas.

```python
# surrogate/training/data.py
import json
from pathlib import Path
from typing import Dict, Any

from datasets import load_dataset


def load_cdn_file_list(file_list_path: str) -> Dict[str, Any]:
    with open(file_list_path) as f:
        return json.load(f)


def build_cdn_streaming_dataset(file_list_path: str, split: str = "train"):
    """
    Returns a streaming dataset that reads parquet files directly from CDN.
    No HF API Authorization header is used; CDN-only.
    """
    payload = load_cdn_file_list(file_list_path)
    cdn_urls = [f["cdn_url"] for f in payload["files"]]

    if not cdn_urls:
        raise ValueError("No files found in file_list.json")

    ds = load_dataset(
        "parquet",
        name=split,
        data_files={"train": cdn_urls},
        split=split,
        streaming=True,
    )
    return ds


def project_to_prompt_response(example: Dict[str, Any]) -> Dict[str, str]:
    prompt = example.get("prompt") or example.get("input") or example.get("text") or ""
    response = example.get("response") or example.get("output") or example.get("completion") or ""
    return {"prompt": str(prompt), "response": str(response)}
```

## 3) Lightning-aware launcher: reuse + idle-resilient restart

- Reuses running studios when available.
- Checks studio status before `.run()`; restarts if stopped.
- Uses CDN dataloader; no `load_dataset(repo=...)` during training.
- Deterministic sharding for write rate-limit avoidance.

```python
# surrogate/training/train.py
import argparse
import os
import sys
from pathlib import Path

import lightning as L
from lightning.pytorch.studio import Studio
from transformers import AutoTokenizer, Trainer, TrainingArguments

from surrogate.training.data import build_cdn_streaming_dataset, project_to_prompt_response


def get_or_create_studio(name: str, machine: str = "L40S"):
    """
    Reuse running studio or create a new one.
    """
    teamspace = L.Teamspace()
    for s in teamspace.studios:
        if s.name == name and s.status == "Running":
            print(f"Reusing running studio: {s.name}")
            return s

    print(f"No running studio '{name}' found. Creating...")
    # Prefer lightning-lambda-prod for H200 when available; fallback to public-prod
    return Studio(
        name=name,
        machine=machine,
        create_ok=True,
        cloud="lightning-lambda-prod",
    )


def train(args):
    file_list_path = Path(args.file_list or "training/file_list.json")
    if not file_list_path.exists():
        print("file_list.json not found. Run gen_file_list.py first.", file=sys.stderr)
        sys.exit(1)

    # CDN-only streaming dataset
    ds = build_cdn_streaming_dataset(str(file_list_path))
    ds = ds.map(project_to_prompt_response, remove_columns=ds.features.keys())

    tokenizer = AutoTokenizer.from_pretrained(args.model)
    # model loading handled by Trainer below; keep lightweight here

    # Deterministic sharding for write rate-limit avoidance
    shard_id = hash(file_list_path) % 5
    output_dir = Path(args.output_dir) / f"shard-{shard_id}"
    output_dir.mkdir(parents=True, exist_ok=True)

    training_args = TrainingArguments(
        output_dir=str(output_dir),
        per_device_train_batch_size=args.batch_size,
        num_train_epochs=args.epochs,
        save_steps=args.save_steps,
        logging_steps=args.logging_steps,
        fp16=args.fp16,
        dataloader_num_workers=args.dataloader_workers,
        # Avoid HF write rate-limit: shard writes across siblings deterministically
    )

    trainer = Trainer(
        model=args.model,
        args=training_args,
        train_dataset=ds,
        tokenizer=tokenizer,
    )

    # Lightning studio lifecycle: reuse or restart idle/stopped
    studio = get_or_create_studio(name=args.studio_name, machine=args.machine)
    if studio.status != "Running":
        print(f"Restarting studio {studio.name} (was {studio.status})")
        studio.restart()

    # Run training inside studio session
    studio.run(trainer.train)


if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--model", default="gpt2", help="Model name or path")
    parser.add_argument("--file-list", help="Path to file_list.json")
    parser.add_argument("--output-dir", default="outputs", help="Training output dir")
