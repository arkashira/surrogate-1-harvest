# Costinel / design

## Final Implementation Plan — Top-Hub Signal Panel (CDN-first, <2h)

### Scope & Design Decisions (merged + resolved)
- **CDN-first**: Runtime fetches only from `https://huggingface.co/datasets/.../resolve/main/...` (no `/api/`, no auth, no 429).  
- **Non-blocking**: Panel renders immediately with cached snapshot; background refresh optional.  
- **Lightning reuse**: Detect running Lightning Studio; attach if alive, restart if idle-killed (only when needed for future updates).  
- **Minimal payload**: `{ hub, generatedAt, insights: [{ text, tags }] }` — no `source`, no `ts` columns.  
- **Graceful fallback**: If CDN 404, show skeleton with last-known timestamp (if available) or hide silently.

#### 2) Panel component (React/Next.js style — adapt to actual framework)
`/opt/axentx/Costinel/components/TopHubSignalPanel.tsx`
```tsx
import { useEffect, useState } from 'react';

type Insight = { text: string; tags: string[] };
type Snapshot = { hub: string; generatedAt: string; insights: Insight[] };

export default function TopHubSignalPanel() {
  const [snapshot, setSnapshot] = useState<Snapshot | null>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    // CDN-only fetch; no Authorization header
    const url =
      'https://huggingface.co/datasets/axentx/costinel-knowledge/resolve/main/data/hubs/top-hub.json';

    fetch(url, { cache: 'no-store' })
      .then((res) => {
        if (!res.ok) throw new Error('Failed to load top-hub snapshot');
        return res.json();
      })
      .then((data) => {
        setSnapshot(data);
        setLoading(false);
      })
      .catch(() => {
        // graceful fallback: render nothing (non-blocking)
        setLoading(false);
      });
  }, []);

  if (loading) {
    return (
      <div className="rounded border border-gray-200 bg-gray-50 p-4">
        <div className="h-4 w-32 animate-pulse rounded bg-gray-300" />
      </div>
    );
  }

  if (!snapshot) return null;

  return (
    <div className="rounded border border-blue-200 bg-blue-50 p-4">
      <div className="mb-2 flex items-center justify-between">
        <h3 className="text-sm font-semibold text-blue-900">Top-Hub Signal</h3>
        <span className="inline-flex items-center rounded bg-blue-100 px-2 py-0.5 text-xs font-medium text-blue-800">
          {snapshot.hub}
        </span>
      </div>
      <p className="mb-3 text-xs text-blue-700">
        Generated {new Date(snapshot.generatedAt).toLocaleDateString()}
      </p>
      <ul className="space-y-2">
        {snapshot.insights.map((insight, idx) => (
          <li key={idx} className="text-sm text-blue-800">
            • {insight.text}
            <div className="mt-1 flex flex-wrap gap-1">
              {insight.tags.map((tag) => (
                <span key={tag} className="text-xs text-blue-600">
                  {tag}
                </span>
              ))}
            </div>
          </li>
        ))}
      </ul>
    </div>
  );
}
```

#### 4) Sync script (orchestration machine / Mac / CI)
`/opt/axentx/Costinel/scripts/sync-top-hub.js`
```js
#!/usr/bin/env node
/**
 * Sync top-hub signal to CDN.
 * - Runs on orchestration machine (Mac/CI)
 * - Uses HF API once to list tree, then CDN-only downloads
 * - Outputs /data/hubs/top-hub.json
 */
import { promises as fs } from 'fs';
import { execSync } from 'child_process';
import path from 'path';

const REPO = 'axentx/costinel-knowledge';
const OUT_DIR = path.resolve('data/hubs');
const OUT_FILE = path.join(OUT_DIR, 'top-hub.json');

async function run() {
  try {
    // Ensure output dir
    await fs.mkdir(OUT_DIR, { recursive: true });

    // One API call: list non-recursive tree for top-hub/
    const listRaw = execSync(
      `gh api "repos/${REPO}/contents/top-hub?ref=main" --paginate`,
      { encoding: 'utf8' }
    );
    const list = JSON.parse(listRaw);
    const files = list.filter((f) => f.type === 'file').map((f) => f.name);

    // Pick latest snapshot (simple: pick first or by date)
    const latestFile = files
      .filter((f) => f.startsWith('snapshot-') && f.endsWith('.json'))
      .sort()
      .reverse()[0];

    if (!latestFile) {
      throw new Error('No snapshot file found in top-hub/');
    }

    // CDN download (no auth)
    const url = `https://huggingface.co/datasets/${REPO}/resolve/main/top-hub/${latestFile}`;
    const data = execSync(`curl -sSfL "${url}"`, { encoding: 'utf8' });
    const parsed = JSON.parse(data);

    // Normalize output shape
    const output = {
      hub: parsed.hub || 'MOC',
      generatedAt: parsed.generatedAt || new Date().toISOString(),
      insights: parsed.insights || []
    };

    await fs.writeFile(OUT_FILE, JSON.stringify(output, null, 2), 'utf8');
    console.log(`✅ top-hub snapshot saved to ${OUT_FILE}`);
  } catch (err) {
    console.error('❌ sync failed:', err.message);
    process.exit(1);
  }
}

run();
```
Make executable:
```bash
chmod +x /opt/axentx/Costinel/scripts/sync-top-hub.js
```
