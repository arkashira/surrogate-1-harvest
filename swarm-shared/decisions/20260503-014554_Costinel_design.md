# Costinel / design

## Final Implementation Plan — Top-Hub Signal Panel (Costinel Dashboard)

**Chosen stack**: Vue 3 (aligns with Candidate 2’s existing codebase) + TypeScript + fetch-first resilient data layer.  
**Scope**: frontend-only, read-only, graceful fallbacks, ship <2h.

---

### 1) Data contract (single source of truth)

```ts
// src/types/topHub.ts
export interface HubNode {
  id: string;           // e.g. "MOC"
  label: string;        // human-readable
  type: "hub" | "tag" | "account";
  weight: number;       // connection strength 0..1
}

export interface Proposal {
  id: string;
  title: string;
  summary: string;
  impact: "high" | "medium" | "low";
  tags: string[];
  href?: string;        // optional external/detail link
}

export interface TopHubPayload {
  hub: HubNode | null;
  proposals: Proposal[];
  generatedAt: string;   // ISO
  source: "api" | "cdn" | "local";
}
```

### 3) Resilient data layer (priority: correctness + actionability)

`src/api/topHub.ts`

```ts
import type { TopHubPayload } from "@/types/topHub";

const API_URL = "/api/top-hub";
const CDN_URL =
  "https://huggingface.co/datasets/axentx/costinel-signals/resolve/main/top-hub/latest.json";
const LOCAL_FIXTURE = "/data/fixtures/top-hub.json";

const FALLBACK_ORDER = [
  { url: API_URL, source: "api" as const, cache: "no-store" },
  { url: CDN_URL, source: "cdn" as const, cache: "no-store" },
  { url: LOCAL_FIXTURE, source: "local" as const, cache: "default" },
] as const;

async function tryFetch(
  url: string,
  options: RequestInit,
  timeoutMs = 5_000
): Promise<Response> {
  const controller = new AbortController();
  const timeout = setTimeout(() => controller.abort(), timeoutMs);
  try {
    const res = await fetch(url, { ...options, signal: controller.signal });
    clearTimeout(timeout);
    return res;
  } catch (err) {
    clearTimeout(timeout);
    throw err;
  }
}

export async function fetchTopHub(): Promise<TopHubPayload> {
  for (const { url, source, cache } of FALLBACK_ORDER) {
    try {
      const res = await tryFetch(url, { cache });
      if (res.ok) {
        const json = (await res.json()) as TopHubPayload;
        // Basic runtime validation (lightweight)
        if (json && typeof json === "object" && Array.isArray(json.proposals)) {
          return { ...json, source };
        }
      }
    } catch {
      // continue to next fallback
    }
  }

  // Final deterministic fallback (prevents runtime exceptions)
  return {
    hub: null,
    proposals: [],
    generatedAt: new Date().toISOString(),
    source: "local",
  };
}
```

### 4) Presentational component (reuse-friendly, accessible)

`src/components/TopHubSignalPanel.vue`

```vue
<template>
  <section class="panel" aria-label="Top hub signals">
    <header class="header">
      <h3 class="title">Top-hub signals</h3>
      <span class="badge" :data-source="source">{{ source }}</span>
    </header>

    <div v-if="loading" class="skeleton" aria-busy="true">
      <div class="skeleton-row"></div>
      <div class="skeleton-row"></div>
      <div class="skeleton-row"></div>
    </div>

    <div v-else-if="!payload?.hub" class="empty">No hub detected</div>

    <div v-else>
      <div class="hub">
        <span class="hub-label">{{ payload.hub.label }}</span>
        <span class="hub-type">{{ payload.hub.type }}</span>
        <span class="hub-weight">{{ Math.round(payload.hub.weight * 100) }}%</span>
      </div>

      <div v-if="!payload.proposals?.length" class="empty">No proposals available</div>

      <ul v-else class="proposals" role="list">
        <li v-for="p in payload.proposals" :key="p.id" class="proposal">
          <a
            class="proposal-link"
            :href="p.href || '#'"
            :target="p.href ? '_blank' : undefined"
            :rel="p.href ? 'noopener noreferrer' : undefined"
            @click="$emit('proposal-click', p)"
          >
            <div class="proposal-title">{{ p.title }}</div>
            <div class="proposal-summary">{{ p.summary }}</div>
            <div class="proposal-meta">
              <span :class="['impact', `impact-${p.impact}`]">{{ p.impact }}</span>
              <span v-for="t in p.tags.slice(0, 2)" :key="t" class="tag">{{ t }}</span>
            </div>
          </a>
        </li>
      </ul>
    </div>
  </section>
</template>

<script setup lang="ts">
import { ref, onMounted } from "vue";
import type { TopHubPayload } from "@/types/topHub";
import { fetchTopHub } from "@/api/topHub";

const emit = defineEmits<{
  (e: "proposal-click", p: TopHubPayload["proposals"][number]): void;
}>();

const payload = ref<TopHubPayload | null>(null);
const loading = ref(true);
const source = ref<"api" | "cdn" | "local">("api");

onMounted(async () => {
  const data = await fetchTopHub();
  payload.value = data;
  source.value = data.source;
  loading.value = false;
});
</script>
