# workio / frontend

## Final synthesized implementation

**Chosen approach**: Unified, production-ready hook + minimal, resilient components.  
Keeps the strongest parts from both candidates (optimistic UI, idempotency, offline queue, GPS pre-check, local drafts) and removes overlap/contradictions.

### 1. Core hook (single source of truth)

`src/features/attendance/hooks/useAttendance.ts`

```ts
// src/features/attendance/hooks/useAttendance.ts
import { useCallback, useRef, useState, useEffect } from 'react';
import { useMutation, useQueryClient } from '@tanstack/react-query';

const MAX_RADIUS_METERS = 100;
const OFFICE_CENTER = { lat: 37.7749, lng: -122.4194 }; // <— replace with config/env

type Mode = 'CLOCK_IN' | 'CLOCK_OUT';
type AttendanceState = { mode: Mode; ts: number; location: { lat: number; lng: number } };

function haversine(lat1: number, lon1: number, lat2: number, lon2: number) {
  const R = 6371e3;
  const φ1 = (lat1 * Math.PI) / 180;
  const φ2 = (lat2 * Math.PI) / 180;
  const Δφ = ((lat2 - lat1) * Math.PI) / 180;
  const Δλ = ((lon2 - lon1) * Math.PI) / 180;
  const a =
    Math.sin(Δφ / 2) * Math.sin(Δφ / 2) +
    Math.cos(φ1) * Math.cos(φ2) * Math.sin(Δλ / 2) * Math.sin(Δλ / 2);
  const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
  return R * c;
}

function makeIdempotencyKey(mode: Mode) {
  // Prevent duplicates within same session + short time window
  return `${mode}:${Date.now()}:${Math.random().toString(36).slice(2, 9)}`;
}

function persistQueue<T>(queue: T[]) {
  try {
    localStorage.setItem('attendance_queue', JSON.stringify(queue));
  } catch {}
}

function loadQueue<T>(): T[] {
  try {
    return JSON.parse(localStorage.getItem('attendance_queue') || '[]');
  } catch {
    return [];
  }
}

export function useAttendance() {
  const queryClient = useQueryClient();
  const idempotencyRef = useRef<string | null>(null);

  // Optimistic UI state for immediate feedback
  const [optimistic, setOptimistic] = useState<{
    mode: Mode | null;
    ts: number | null;
    status: 'idle' | 'pending' | 'success' | 'error';
  }>({ mode: null, ts: null, status: 'idle' });

  // Network-aware retry
  const [canRetry, setCanRetry] = useState(true);
  useEffect(() => {
    const onOnline = () => setCanRetry(true);
    const onOffline = () => setCanRetry(false);
    window.addEventListener('online', onOnline);
    window.addEventListener('offline', onOffline);
    setCanRetry(navigator.onLine);
    return () => {
      window.removeEventListener('online', onOnline);
      window.removeEventListener('offline', onOffline);
    };
  }, []);

  const mutation = useMutation({
    mutationFn: async (payload: { mode: Mode; location: { lat: number; lng: number }; idempotencyKey: string }) => {
      const res = await fetch('/api/attendance/clock', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(payload),
      });
      if (!res.ok) throw new Error('Clock action failed');
      return res.json() as Promise<AttendanceState>;
    },
    onMutate: async (payload) => {
      idempotencyRef.current = payload.idempotencyKey;
      setOptimistic({ mode: payload.mode, ts: Date.now(), status: 'pending' });
      await queryClient.cancelQueries({ queryKey: ['todayAttendance'] });
    },
    onSuccess: (data) => {
      setOptimistic({ mode: data.mode, ts: data.ts, status: 'success' });
      queryClient.invalidateQueries({ queryKey: ['todayAttendance'] });
      // Keep small local history for instant UI consistency if needed
      try {
        const history = JSON.parse(localStorage.getItem('attendance_history') || '[]');
        history.push({ mode: data.mode, ts: data.ts, location: data.location });
        localStorage.setItem('attendance_history', JSON.stringify(history.slice(-50)));
      } catch {}
    },
    onError: (_, payload) => {
      setOptimistic((s) => ({ ...s, status: 'error' }));
      // Queue for background retry
      const queue = loadQueue<typeof payload>();
      queue.push(payload);
      persistQueue(queue);
    },
  });

  // GPS pre-check
  const checkLocation = useCallback(
    (position: GeolocationPosition) => {
      const { latitude, longitude } = position.coords;
      const distance = haversine(latitude, longitude, OFFICE_CENTER.lat, OFFICE_CENTER.lng);
      return { ok: distance <= MAX_RADIUS_METERS, distance };
    },
    []
  );

  const clock = useCallback(
    async (mode: Mode) => {
      // 1) Get location first
      const pos = await new Promise<GeolocationPosition>((resolve, reject) => {
        navigator.geolocation.getCurrentPosition(resolve, reject, {
          enableHighAccuracy: true,
          timeout: 8000,
          maximumAge: 10_000,
        });
      });

      // 2) Validate radius
      const { ok, distance } = checkLocation(pos);
      if (!ok) {
        const err: any = new Error(
          `Location outside allowed area (${Math.round(distance || 0)}m). Please clock in near office.`
        );
        err.code = 'LOCATION_OUT_OF_RANGE';
        throw err;
      }

      // 3) Perform optimistic action
      const idempotencyKey = makeIdempotencyKey(mode);
      mutation.mutate({
        mode,
        location: { lat: pos.coords.latitude, lng: pos.coords.longitude },
        idempotencyKey,
      });
    },
    [mutation, checkLocation]
  );

  // Flush queued actions when online
  const flushQueue = useCallback(async () => {
    if (!canRetry) return;
    const queue = loadQueue<{ mode: Mode; location: { lat: number; lng: number }; idempotencyKey: string }>();
    if (!queue.length) return;
    const remaining: typeof queue = [];
    for (const item of queue) {
      try {
        await mutation.mutateAsync(item);
      } catch {
        remaining.push(item);
      }
    }
    persistQueue(remaining);
  }, [canRetry, mutation]);

  // Auto-retry on reconnect
  useEffect(() => {
    if (canRetry) {
      flushQueue();
    }
  }, [canRetry, flushQueue]);

  return {
    optimistic,
    clock,
    flushQueue,
    canRetry,
    isPending: mutation.isPending,
  };
}
```
