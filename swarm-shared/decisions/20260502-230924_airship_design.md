- CDN-only fetches eliminate HF API 429/128-hr caps during training.
- Schema projection to `{prompt,response}` prevents PyArrow `CastError` from mixed schemas.
- Mac=CLI rule preserved: orchestration only; heavy compute stays on Lightning/Kaggle.
- Deterministic, content-addressed outputs (manifest + parquet) enable reproducible surrogate training runs.
- Reuses proven patterns from `knowledge-rag`, `top-hub`, `dataset-mirror` projection rules.

### 1) CLI entrypoint (`airship/cli/discover.py`)

```python
#!/usr/bin/env python3
"""
airship discover
Produce CDN-only manifest and schema-projected parquet for surrogate training.
"""
import argparse
import hashlib
import json
from datetime import datetime, timezone
from pathlib import Path

import huggingface_hub
import pyarrow as pa
import pyarrow.parquet as pq
import requests
from tqdm import tqdm

HF_REPO = "datasets/example-repo"  # override via --repo or env
DATE_TAG = datetime.now(timezone.utc).strftime("%Y-%m-%d")
OUT_DIR = Path("batches/mirror-merged") / DATE_TAG
OUT_DIR.mkdir(parents=True, exist_ok=True)

def list_tree(repo: str, path: str = "") -> list[dict]:
    """Non-recursive listing; recurse manually to avoid large payloads."""
    api = huggingface_hub.HfApi()
    items = api.list_repo_tree(repo=repo, path=path, recursive=False)
    out = []
    for item in items:
        if item.type == "file":
            out.append({
                "path": item.path,
                "size": item.size,
                "lfs": getattr(item, "lfs", None),
            })
        elif item.type == "directory":
            out.extend(list_tree(repo, item.path))
    return out

def cdn_url(repo: str, path: str) -> str:
    return f"https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def extract_pair(text: str) -> dict:
    """Heuristic projection to {prompt,response}. Adapt per corpus."""
    lines = [ln.strip() for ln in text.splitlines() if ln.strip()]
    if len(lines) >= 2:
        return {"prompt": lines[0], "response": lines[1]}
    return {"prompt": text, "response": ""}

def build_manifest(repo: str) -> tuple[Path, list[dict]]:
    tree = list_tree(repo)
    manifest_path = OUT_DIR / "manifest.json"
    manifest_path.write_text(json.dumps(tree, indent=2))
    return manifest_path, tree

def project_to_parquet(repo: str, tree: list[dict], shards: int = 1) -> list[Path]:
    schema = pa.schema([
        pa.field("slug", pa.string()),
        pa.field("prompt", pa.string()),
        pa.field("response", pa.string()),
    ])
    shard_buffers = {i: [] for i in range(max(1, shards))}
    out_paths = []

    for item in tqdm(tree, desc="Projecting"):
        url = cdn_url(repo, item["path"])
        r = requests.get(url, timeout=30)
        r.raise_for_status()
        pair = extract_pair(r.text)
        row = {
            "slug": item["path"].rsplit(".", 1)[0].replace("/", "_"),
            "prompt": pair["prompt"],
            "response": pair["response"],
        }
        shard_id = hash(row["slug"]) % max(1, shards)
        shard_buffers[shard_id].append(row)

    for shard_id, rows in shard_buffers.items():
        if not rows:
            continue
        table = pa.Table.from_pylist(rows, schema=schema)
        shard_suffix = f"_shard{shard_id}" if shards > 1 else ""
        parquet_path = OUT_DIR / f"mirror-merged-{DATE_TAG}{shard_suffix}.parquet"
        pq.write_table(table, parquet_path, compression="snappy")
        out_paths.append(parquet_path)
    return out_paths

def main() -> None:
    parser = argparse.ArgumentParser(description="airship discover")
    parser.add_argument("--repo", default=HF_REPO, help="HF dataset repo")
    parser.add_argument("--shards", type=int, default=1, help="number of parquet shards")
    args = parser.parse_args()

    print("Building manifest...")
    manifest_path, tree = build_manifest(args.repo)
    print(f"Manifest: {manifest_path}")

    print("Projecting to parquet (CDN-only)...")
    parquet_paths = project_to_parquet(args.repo, tree, shards=args.shards)
    for p in parquet_paths:
        print(f"Parquet: {p}")

    print("Done. Surrogate training can now load local parquet with zero HF API calls.")

if __name__ == "__main__":
    main()
```

# surrogate/train_stub.py
import pyarrow.parquet as pq
from pathlib import Path
import pyarrow as pa

def load_local_dataset(parquet_glob: str = "batches/mirror-merged/*/*.parquet") -> pa.Table:
    files = sorted(Path().glob(parquet_glob))
    if not files:
        raise FileNotFoundError(f"No parquet files found for {parquet_glob}")
    tables = [pq.read_table(str(f)) for f in files]
    table = pa.concat_tables(tables).combine_chunks()
    # Optional dedup by slug
    return table
