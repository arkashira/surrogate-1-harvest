# airship-discover.sh
# Deterministic, CDN-safe discovery + manifest builder for HF ingestion.
# Zero runtime deps beyond bash, jq, curl, find, sort, date.
#
# Usage:
#   bash airship-discover.sh \

#     --date <YYYY-MM-DD> \
#     --out-dir <path> \
#     --cdn-base <cdn-base-url>

#   <out-dir>/manifest.json    - sorted file list for CDN fetches

Usage: $0 --repo <owner/repo> --date <YYYY-MM-DD> --out-dir <path> [--cdn-base <url>] [--dry-run]

list_tree() {
  if [[ $DRY_RUN -eq 1 ]]; then
    log "DRY_RUN: would list tree for $HF_REPO (date=$DATE)"
    printf '[]' > "$TREE_FILE"
    return 0
  fi

  # Prefer using HF CLI if available; otherwise fallback to raw CDN + saved list.
  if command -v huggingface-cli >/dev/null 2>&1; then
    log "Listing repo tree (non-recursive per top-level) for $HF_REPO..."
    # We'll do a single recursive call for simplicity; for very large repos switch to per-folder.
    huggingface-cli repo tree --repo-type dataset "$HF_REPO" --recursive > "$TREE_FILE" 2>/dev/null || {
      log "HF CLI tree failed; falling back to minimal manifest"
      printf '[]' > "$TREE_FILE"
    }
  else
    log "huggingface-cli not found. Create $TREE_FILE manually or provide pre-saved file."
    printf '[]' > "$TREE_FILE"
  fi

# ---- step 2: build deterministic manifest ----
# Extract paths, keep only files, sort deterministically, produce CDN URLs.
# We project to {prompt,response} at parse time later; here we only record paths.

build_manifest() {
  local tree_file="$1"
  local cdn_base="$2"
  local date="$3"

  if [[ ! -s "$tree_file" ]]; then
    log "Tree file empty or missing — producing empty manifest."
    printf '{"generated_at":"%s","repo":"%s","date":"%s","files":[],"cdn_base":"%s","total_files":0}\n' \
      "$TIMESTAMP" "$HF_REPO" "$date" "$cdn_base" > "$MANIFEST"
    return 0
  fi

  # jq: expect tree entries like {"path":"...","type":"file","size":...}
  # Keep only files, sort by path, produce stable CDN URLs.
  jq -r --arg repo "$HF_REPO" --arg date "$date" --arg cdn "$cdn_base" '
    if type == "array" then
      map(select(.type == "file"))
    else
      []
    end
    | sort_by(.path)
    | {
        generated_at: $TIMESTAMP,
        repo: $repo,
        date: $date,
        cdn_base: (if $cdn == "" then ("https://huggingface.co/datasets/" + $repo + "/resolve/main") else $cdn end),
        files: map({
          path: .path,
          size: (.size // 0),
          cdn_url: ((if $cdn == "" then ("https://huggingface.co/datasets/" + $repo + "/resolve/main") else $cdn end) + "/" + .path)
        }),
        total_files: length
      }' "$tree_file" > "$MANIFEST"
}

build_manifest "$TREE_FILE" "$CDN_BASE" "$DATE"
