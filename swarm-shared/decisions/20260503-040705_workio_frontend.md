# workio / frontend

## 1. Diagnosis

- No optimistic/pending UI on clock-in/out → double-taps and perceived latency; duplicates can slip before backend idempotency settles.
- No inline location guard or recovery when LINE/geolocation permission is missing or GPS fails; UX falls back to silent failure.
- Clock state (in/out) is not reflected immediately in the UI after successful request; user must refresh or wait for poll.
- No request-level idempotency key for clock mutations → retries (e.g., after network blip) can create duplicate punches.
- Missing local pending queue for offline/weak-signal scenarios; clock events can be lost when device is offline.

## 2. Proposed change

File scope:
- `src/features/clock/ClockButton.tsx` (or create if absent) — primary clock action
- `src/lib/api.ts` — add idempotency header and typed helpers
- `src/hooks/useClock.ts` — add optimistic state + pending queue
- `src/stores/clockStore.ts` (or `zustand` slice) — hold `lastPunch`, `pending`, `isClocking`

## 3. Implementation

### Add typed API helper with idempotency

```ts
// src/lib/api.ts
import axios from 'axios';

const api = axios.create({
  baseURL: '/api',
  timeout: 10_000,
});

export const clockInOut = async (
  type: 'IN' | 'OUT',
  userId: string,
  options?: { latitude?: number; longitude?: number }
) => {
  const idempotencyKey = `${userId}-${type}-${Date.now()}`;
  const { data } = await api.post<{ success: boolean; punch: any }>('/clock', {
    type,
    userId,
    ...options,
  }, {
    headers: {
      'Idempotency-Key': idempotencyKey,
    },
  });
  return data;
};
```

### Clock store (Zustand or simple module)

```ts
// src/stores/clockStore.ts
import { create } from 'zustand';

type Punch = { id: string; type: 'IN' | 'OUT'; ts: number; pending?: boolean };

interface ClockState {
  lastPunch: Punch | null;
  pending: Punch | null;
  isClocking: boolean;
  setPending: (punch: Punch | null) => void;
  commitPunch: (punch: Punch) => void;
  setClocking: (v: boolean) => void;
}

export const useClockStore = create<ClockState>((set) => ({
  lastPunch: null,
  pending: null,
  isClocking: false,
  setPending: (punch) => set({ pending: punch }),
  commitPunch: (punch) => set({ lastPunch: punch, pending: null, isClocking: false }),
  setClocking: (v) => set({ isClocking: v }),
}));
```

### Hook with optimistic + retry + queue

```ts
// src/hooks/useClock.ts
import { useCallback } from 'react';
import { clockInOut } from '../lib/api';
import { useClockStore } from '../stores/clockStore';

export const useClock = (userId: string) => {
  const { setPending, commitPunch, setClocking, pending } = useClockStore();

  const attempt = useCallback(async (type: 'IN' | 'OUT', geolocation?: { latitude: number; longitude: number }) => {
    if (useClockStore.getState().isClocking) return;

    const optimisticId = `local-${Date.now()}`;
    const optimisticPunch = { id: optimisticId, type, ts: Date.now(), pending: true };
    setClocking(true);
    setPending(optimisticPunch);

    try {
      const res = await clockInOut(type, userId, geolocation);
      commitPunch({ id: res.punch.id, type, ts: Date.now(), pending: false });
    } catch (err) {
      // keep pending in store for UI; caller can retry
      setClocking(false);
      throw err;
    }
  }, [userId, setPending, commitPunch, setClocking]);

  const retryPending = useCallback(async () => {
    const p = pending;
    if (!p || !p.pending) return;
    await attempt(p.type, undefined);
  }, [pending, attempt]);

  return { attempt, retryPending, pending };
};
```

### Clock button with optimistic UI + location guard

```tsx
// src/features/clock/ClockButton.tsx
import React, { useEffect, useState } from 'react';
import { useClock } from '../../hooks/useClock';
import { useClockStore } from '../../stores/clockStore';

export const ClockButton: React.FC<{ userId: string }> = ({ userId }) => {
  const lastPunch = useClockStore((s) => s.lastPunch);
  const pending = useClockStore((s) => s.pending);
  const isClocking = useClockStore((s) => s.isClocking);
  const { attempt, retryPending } = useClock(userId);
  const [locationError, setLocationError] = useState<string | null>(null);

  const type = lastPunch?.type === 'IN' ? 'OUT' : 'IN';
  const label = { IN: 'Clock In', OUT: 'Clock Out' }[type];

  const acquireLocation = (): Promise<{ latitude: number; longitude: number } | null> => {
    if (!navigator.geolocation) {
      setLocationError('Geolocation not supported');
      return Promise.resolve(null);
    }
    return new Promise((resolve) => {
      navigator.geolocation.getCurrentPosition(
        (pos) => {
          setLocationError(null);
          resolve({ latitude: pos.coords.latitude, longitude: pos.coords.longitude });
        },
        (err) => {
          setLocationError('Location permission required for clock actions.');
          resolve(null);
        },
        { timeout: 8000, enableHighAccuracy: true }
      );
    });
  };

  const handleClock = async () => {
    const loc = await acquireLocation();
    // allow clock without location? backend may require it — here we allow but warn
    try {
      await attempt(type, loc || undefined);
    } catch {
      // optimistic state remains; show retry
    }
  };

  // If there's a pending punch from a previous session, offer retry
  useEffect(() => {
    // no auto-retry to avoid surprise; user-initiated retry via UI
  }, []);

  return (
    <div className="flex flex-col items-center gap-3">
      <button
        onClick={handleClock}
        disabled={isClocking}
        className={`px-6 py-3 rounded-lg font-semibold text-white transition ${
          type === 'IN'
            ? 'bg-green-600 hover:bg-green-700 disabled:bg-green-400'
            : 'bg-red-600 hover:bg-red-700 disabled:bg-red-400'
        }`}
      >
        {isClocking ? (pending ? 'Processing...' : 'Clocking...') : label}
      </button>

      {locationError && <p className="text-sm text-amber-600">{locationError}</p>}

      {pending && pending.pending && (
        <div className="flex items-center gap-2 text-sm text-gray-600">
          <span>Pending ({pending.type}) — </span>
          <button onClick={retryPending} className="underline text-blue-600">
            Retry
          </button>
        </div>
      )}
    </div>
  );
};
```
