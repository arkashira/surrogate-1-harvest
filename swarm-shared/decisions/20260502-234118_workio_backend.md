# workio / backend

## Final Synthesis — One Correct, Actionable Solution

**Core problem**: LINE webhook retries + concurrent deliveries cause duplicate and inconsistent punch state because the system lacks **idempotency** and **atomic state transitions** at the database level.

**Guiding principle**: enforce correctness at the database (constraints + atomic CTEs) and guarantee exactly-once processing per external event using an idempotency table keyed by a deterministic event key.

## 1. Schema changes (`workio/server/src/db/schema.sql`)

```sql
-- Idempotency for external webhook events (LINE, etc.)
CREATE TABLE IF NOT EXISTS webhook_events (
  id           UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  source       TEXT NOT NULL,            -- e.g. 'line'
  event_key    TEXT NOT NULL,            -- deterministic key (hash of stable event fields)
  processed_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  payload_hash TEXT,
  created_at   TIMESTAMPTZ NOT NULL DEFAULT now(),
  CONSTRAINT uq_webhook_event_key UNIQUE (source, event_key)
);

-- Reference idempotency on punches (optional but useful for traceability)
ALTER TABLE punches
  ADD COLUMN IF NOT EXISTS event_id UUID NULL REFERENCES webhook_events(id);

-- One punch per event (when event_id is present)
CREATE UNIQUE INDEX IF NOT EXISTS uq_punch_event_id
  ON punches (event_id)
  WHERE event_id IS NOT NULL;

-- Prevent more than one active 'in' per employee (active = no matching 'out')
-- This is a safety guard; primary enforcement is via atomic CTE below.
CREATE UNIQUE INDEX IF NOT EXISTS uq_employee_active_in
  ON punches (employee_id)
  WHERE action = 'in' AND deleted_at IS NULL;
```

## 2. Deterministic event key (route layer)

Use a stable, collision-resistant key derived from the LINE event so retries and redeliveries map to the same key.

```ts
import crypto from 'crypto';

function computeLineEventKey(body: any): string {
  // Prefer a stable delivery identifier; fall back to content hash.
  const deliveryId = body?.events?.[0]?.deliveryId;
  if (deliveryId) {
    return `line:delivery:${deliveryId}`;
  }
  // Deterministic hash of the event payload (sorted)
  const payload = JSON.stringify(body, Object.keys(body).sort());
  return `line:hash:${crypto.createHash('sha256').update(payload).digest('hex')}`;
}
```

## 3. Atomic punch service (`workio/server/src/services/punch.ts`)

```ts
import { Pool } from 'pg';

const pool = new Pool({ connectionString: process.env.DATABASE_URL });

export type PunchAction = 'in' | 'out';

export interface ProcessPunchOpts {
  employeeId: number;
  action: PunchAction;
  lineUserId: string;
  eventKey: string;          // deterministic key from webhook
  source?: string;
  timestamp?: Date;
  latitude?: number;
  longitude?: number;
}

/**
 * Exactly-once punch processing:
 * 1) Insert/get idempotency record (event_key unique).
 * 2) If event already processed, return existing punch row.
 * 3) Otherwise perform atomic state transition:
 *    - 'in': insert new 'in' (partial unique index prevents concurrent active 'in').
 *    - 'out': atomically close latest 'in' without 'out' for this employee.
 * All within a single transaction using FOR UPDATE SKIP LOCKED to avoid deadlocks.
 */
export async function processPunch(opts: ProcessPunchOpts) {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // Idempotency record
    const eventRes = await client.query(
      `INSERT INTO webhook_events (source, event_key, created_at)
       VALUES ($1, $2, now())
       ON CONFLICT (source, event_key) DO UPDATE
       SET processed_at = webhook_events.processed_at
       RETURNING id`,
      [opts.source || 'line', opts.eventKey]
    );
    const eventId = eventRes.rows[0].id;

    // If already processed, return existing punch for this event
    const existing = await client.query(
      `SELECT id, employee_id, action, in_time, out_time
       FROM punches
       WHERE event_id = $1`,
      [eventId]
    );

    if (existing.rows.length > 0) {
      await client.query('COMMIT');
      client.release();
      return existing.rows[0];
    }

    const ts = opts.timestamp || new Date();
    let result;

    if (opts.action === 'in') {
      // Insert new 'in' — partial unique index prevents concurrent active 'in'
      const insertRes = await client.query(
        `INSERT INTO punches (employee_id, action, in_time, line_user_id, latitude, longitude, event_id, created_at)
         VALUES ($1, 'in', $2, $3, $4, $5, $6, now())
         RETURNING id, employee_id, action, in_time, out_time`,
        [opts.employeeId, ts, opts.lineUserId, opts.latitude, opts.longitude, eventId]
      );
      result = insertRes.rows[0];
    } else {
      // Atomically close the latest open 'in'
      const updateRes = await client.query(
        `UPDATE punches
         SET out_time = $1, updated_at = now()
         WHERE id = (
           SELECT id
           FROM punches
           WHERE employee_id = $2 AND action = 'in' AND out_time IS NULL AND deleted_at IS NULL
           ORDER BY in_time DESC
           LIMIT 1
           FOR UPDATE SKIP LOCKED
         )
         RETURNING id, employee_id, action, in_time, out_time`,
        [ts, opts.employeeId]
      );

      if (updateRes.rowCount === 0) {
        // No open 'in' — record explicit 'out' so event is idempotently stored
        const insertRes = await client.query(
          `INSERT INTO punches (employee_id, action, out_time, line_user_id, latitude, longitude, event_id, created_at)
           VALUES ($1, 'out', $2, $3, $4, $5, $6, now())
           RETURNING id, employee_id, action, in_time, out_time`,
          [opts.employeeId, ts, opts.lineUserId, opts.latitude, opts.longitude, eventId]
        );
        result = insertRes.rows[0];
      } else {
        result = updateRes.rows[0];
      }
    }

    await client.query('COMMIT');
    return result;
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  } finally {
    client.release();
  }
}
```
