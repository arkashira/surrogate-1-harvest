## Final Synthesis: CDN-only HF ingestion + deterministic sibling-repo sharding (ready to ship <2h)

**Goal**: Eliminate HF API 429s during Surrogate training by switching to a CDN-only ingestion pipeline with pre-listed file paths and deterministic sibling-repo sharding for writes.

---

### Unified Implementation Plan

1. **File-listing utility** (orchestrated from Mac after rate-limit window)  
   - Single `list_repo_tree(path, recursive=False)` per date folder.  
   - Save to `file_listings/{date}.json` and commit or pass to Lightning job.  
   - Non-recursive to avoid pagination explosion.

2. **CDN fetcher module** (training-side)  
   - Accepts repo, folder, and local file listing JSON.  
   - Downloads via `https://huggingface.co/datasets/{repo}/resolve/main/{path}` (no auth).  
   - Projects to `{prompt, response}` only at parse time.  
   - Retries with exponential backoff; respects CDN limits.

3. **Deterministic sibling-repo sharding writer**  
   - Shard selection: `hash(slug) % N` → sibling repo (e.g., `sibling-0..4`).  
   - Writes parquet to `batches/{sibling}/mirror-merged/{date}/{slug}.parquet`.  
   - No extra metadata columns (`source`, `ts`) to avoid mixed schema.

4. **Training launcher update** (Lightning Studio)  
   - Embed file listing JSON path in train args.  
   - Reuse running instance if available; restart if stopped.  
   - Prefer L40S; fallback to free-tier compatible machine.

5. **Ops hardening**  
   - Ensure wrapper scripts have `#!/usr/bin/env bash`, `chmod +x`.  
   - Set `SHELL=/bin/bash` in crontab for any cron jobs.

#### 1) File listing utility
```python
# scripts/list_hf_date_folder.py
#!/usr/bin/env python3
import json
import argparse
from pathlib import Path
from huggingface_hub import HfApi

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True)
    parser.add_argument("--folder", required=True)
    parser.add_argument("--out", required=True)
    args = parser.parse_args()

    api = HfApi()
    tree = api.list_repo_tree(repo_id=args.repo, path=args.folder, recursive=False)
    files = [item.rfilename for item in tree if item.type == "file"]

    out_path = Path(args.out)
    out_path.parent.mkdir(parents=True, exist_ok=True)
    out_path.write_text(json.dumps({"repo": args.repo, "folder": args.folder, "files": files}, indent=2))
    print(f"Wrote {len(files)} files to {out_path}")

if __name__ == "__main__":
    main()
```

#### 2) CDN fetcher (core)
```python
# surrogate/ingestion/cdn_fetcher.py
import json
import time
import requests
from pathlib import Path
from typing import Dict

CDN_ROOT = "https://huggingface.co/datasets"

def load_file_list(listing_path: str) -> Dict:
    return json.loads(Path(listing_path).read_text())

def cdn_download(repo: str, file_path: str, retries: int = 5) -> bytes:
    url = f"{CDN_ROOT}/{repo}/resolve/main/{file_path}"
    for attempt in range(1, retries + 1):
        try:
            resp = requests.get(url, timeout=30)
            resp.raise_for_status()
            return resp.content
        except Exception as exc:
            if attempt == retries:
                raise
            sleep_sec = 2 ** attempt
            print(f"Retry {attempt}/{retries} for {file_path}: {exc}. Sleep {sleep_sec}s")
            time.sleep(sleep_sec)

def parse_parquet_projection_bytes(raw: bytes):
    import pyarrow.parquet as pq
    import io
    table = pq.read_table(io.BytesIO(raw), columns=["prompt", "response"])
    return table.to_pydict()

def fetch_date_folder(listing_path: str, output_dir: str):
    data = load_file_list(listing_path)
    repo = data["repo"]
    out_root = Path(output_dir) / data["folder"].replace("/", "_")
    out_root.mkdir(parents=True, exist_ok=True)

    records = []
    for f in data["files"]:
        try:
            raw = cdn_download(repo, f)
            batch = parse_parquet_projection_bytes(raw)
            records.extend(batch)
        except Exception as exc:
            print(f"Skip {f}: {exc}")
            continue
    import pyarrow as pa
    tbl = pa.table(records)
    out_file = out_root / "projected.parquet"
    pq.write_table(tbl, out_file)
    print(f"Saved {len(records)} rows to {out_file}")
```

#### 3) Deterministic sharded writer
```python
# surrogate/ingestion/sharded_writer.py
import hashlib
from pathlib import Path
import pyarrow.parquet as pq

def shard_for_slug(slug: str, siblings: list) -> str:
    digest = int(hashlib.sha256(slug.encode()).hexdigest(), 16)
    return siblings[digest % len(siblings)]

def write_sharded(date: str, slug: str, table, siblings: list, base: str = "batches"):
    repo = shard_for_slug(slug, siblings)
    out_dir = Path(base) / repo / "mirror-merged" / date
    out_dir.mkdir(parents=True, exist_ok=True)
    out_path = out_dir / f"{slug}.parquet"
    pq.write_table(table, out_path)
    return str(out_path)
```

#### 4) Training launcher (Lightning reuse)
```python
# scripts/launch_surrogate_training.py
#!/usr/bin/env python3
import argparse
from lightning import Lightning, Teamspace, Machine

def reuse_or_create(studio_name: str):
    team = Teamspace()
    for s in team.studios:
        if s.name == studio_name and s.status == "Running":
            print(f"Reusing running studio: {studio_name}")
            return s
    print(f"Creating studio: {studio_name}")
    machine = Machine.L40S
    return Lightning.Studio(
        name=studio_name,
        machine=machine,
        create_ok=True,
    )

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--date", required=True)
    parser.add_argument("--file_list", required=True)
    parser.add_argument("--siblings", nargs="+", default=["sibling-0", "sibling-1", "sibling-2", "sibling-3", "sibling-4"])
    args = parser.parse_args()

    studio = reuse_or_create("surrogate-train")
    if studio.status != "Running":
        print("Studio stopped; restarting...")
        studio.start(machine=Machine.L40S)

    run = studio.run(
        "train.py",
        arguments=[
            "--date", args.date,
            "--file_list", args.file_list,
            "--siblings", ",".join(args.siblings),
        ],
    )
    print(f"Started run: {run}")

if __name__ == "__main__":
    main()
```
