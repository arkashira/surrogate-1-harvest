# workio / design

## 1. Diagnosis
- Missing **LINE webhook signature verification** (`X-Line-Signature`) on `/webhook/line` allows spoofed/replayed events to be accepted as valid clock-in/out, leave, or OT requests.
- No **idempotency/replay protection** — LINE retries on 5xx/timeouts can create duplicate `attendance_punches`, `leave_requests`, `ot_requests`.
- No **request deduplication key** in DB layer (no unique constraint/index on `(event_id, tenant_id, user_id, type)`), so duplicates are silently inserted.
- Clock/leave/OT handlers accept events without validating **timestamp tolerance** (replay window) or **event sequencing**, enabling time-shifted replays.
- Missing **defense-in-depth logging** of signature failures and duplicate attempts, reducing detectability of abuse.

## 2. Proposed change
- **File scope**: `workio/server/src/routes/webhook/line.ts` (or equivalent route handling `/webhook/line`).
- **Add**:
  - `X-Line-Signature` HMAC-SHA256 verification middleware.
  - Idempotency layer using `line_event_id` (from webhook body) + tenant/user context with unique constraint.
  - 5-minute replay window check on event timestamp.
  - Safe upsert/ignore for duplicates and structured audit logging.

import express, { Request, Response, NextFunction } from 'express';
import crypto from 'crypto';
import { pool } from '../../db';
import { logger } from '../../utils/logger';

const router = express.Router();
const CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';
const REPLAY_WINDOW_MS = 5 * 60 * 1000; // 5 minutes

// Middleware: verify LINE signature
function verifyLineSignature(req: Request, res: Response, next: NextFunction) {
  const signature = req.headers['x-line-signature'] as string | undefined;
  if (!signature) {
    logger.warn({ msg: 'line_webhook_missing_signature' });
    return res.status(401).json({ error: 'Missing signature' });
  }

  const body = JSON.stringify(req.body);
  const expected = crypto
    .createHmac('sha256', CHANNEL_SECRET)
    .update(body, 'utf8')
    .digest('base64');

  if (!crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected))) {
    logger.warn({ msg: 'line_webhook_invalid_signature', signature });
    return res.status(401).json({ error: 'Invalid signature' });
  }
  next();
}

// Middleware: reject stale events
function rejectStaleEvent(req: Request, res: Response, next: NextFunction) {
  const events = req.body.events || [];
  const now = Date.now();
  for (const ev of events) {
    const ts = ev.timestamp;
    if (ts && Math.abs(now - ts) > REPLAY_WINDOW_MS) {
      logger.warn({ msg: 'line_webhook_stale_event', event: ev });
      return res.status(400).json({ error: 'Stale event' });
    }
  }
  next();
}

// Idempotent handler for attendance/leave/OT events
async function handleLineEvent(req: Request, res: Response) {
  const events = req.body.events || [];
  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    for (const ev of events) {
      const lineEventId = ev.source?.userId ? `${ev.source.userId}:${ev.timestamp}:${ev.type}` : null;
      const tenantId = ev.source?.groupId || ev.source?.roomId || 'default'; // map to tenant appropriately
      const userId = ev.source?.userId;
      const eventType = ev.type; // 'message', 'postback', etc.
      const payload = ev;

      if (!lineEventId || !userId) {
        logger.warn({ msg: 'line_webhook_missing_ids', ev });
        continue;
      }

      // Try insert with unique constraint; ignore duplicate
      await client.query(
        `INSERT INTO line_event_log (line_event_id, tenant_id, user_id, event_type, payload, created_at)
         VALUES ($1, $2, $3, $4, $5, NOW())
         ON CONFLICT (line_event_id) DO NOTHING`,
        [lineEventId, tenantId, userId, eventType, JSON.stringify(payload)]
      );

      // If inserted (not duplicate), process business logic
      const { rowCount } = await client.query(
        `SELECT 1 FROM line_event_log WHERE line_event_id = $1`,
        [lineEventId]
      );

      // rowCount check is conservative; ON CONFLICT DO NOTHING means we can check via RETURNING in insert.
      // Simpler: attempt insert and check result.actualRows if using pg >= 8.7; here we re-select.
      // For brevity, we'll assume if no error and we got here, it's new if the SELECT returns a row.
      // Better: use RETURNING in INSERT and inspect rows.
      // We'll refactor insert to RETURNING to avoid extra roundtrip.
    }

    // Refactored single-pass insert+process (preferred)
    // For each event, attempt insert with RETURNING to know if new, then process.
    for (const ev of events) {
      const lineEventId = ev.source?.userId ? `${ev.source.userId}:${ev.timestamp}:${ev.type}` : null;
      const tenantId = ev.source?.groupId || ev.source?.roomId || 'default';
      const userId = ev.source?.userId;
      const eventType = ev.type;
      const payload = ev;

      if (!lineEventId || !userId) continue;

      const insertRes = await client.query(
        `INSERT INTO line_event_log (line_event_id, tenant_id, user_id, event_type, payload, created_at)
         VALUES ($1, $2, $3, $4, $5, NOW())
         ON CONFLICT (line_event_id) DO NOTHING
         RETURNING line_event_id`,
        [lineEventId, tenantId, userId, eventType, JSON.stringify(payload)]
      );

      const isNew = insertRes.rowCount > 0;
      if (!isNew) {
        logger.info({ msg: 'line_webhook_duplicate_ignored', lineEventId });
        continue;
      }

      // Process business event (clock/leave/OT) — keep minimal here; delegate to services
      await processBusinessEvent(client, ev, tenantId, userId);
    }

    await client.query('COMMIT');
    res.status(200).json({ status: 'ok' });
  } catch (err) {
    await client.query('ROLLBACK');
    logger.error({ msg: 'line_webhook_processing_error', error: err });
    // LINE will retry on 5xx — safe because duplicates are idempotent
    res.status(500).json({ error: 'Processing failed' });
  } finally {
    client.release();
  }
}

async function processBusinessEvent(client: any, ev: any, tenantId: string, userId: string) {
  // Minimal example: handle clock-in/out via postback or message text
  if (ev.type === 'postback') {
    const data = ev.postback?.data || '';
    // Example: data="action=clock&in=true"
    const params = new URLSearchParams(data);
    const action = params.get('action');
    if (action === 'clock') {
      const isIn = params.get('in') === 'true';
      // Validate GPS if required (omitted for brevity)
      await client.query(
        `INSERT INTO attendance_punches (tenant_id, user_id, punch_in, created_at, line_event_id)
         VALUES ($1, $2, $3, NOW(), $4)`,
        [tenantId, userId, isIn, `${userId}:${ev.timestamp}:${ev.type}`]
      );
    }
  }

  // Handle leave/OT requests similarly — delegate to domain services in future
}

router.post('/webhook/line', verifyLineSignature, rejectStaleEvent, handleLineEvent);

export default router;
