#### 1) `bin/list-date-files.py`
```python
#!/usr/bin/env python3
"""
Generate deterministic file list for a date folder in
axentx/surrogate-1-training-pairs to enable CDN-only ingestion.

Usage:
  python bin/list-date-files.py --date 2026-04-29 --out file-list-2026-04-29.json
"""
import argparse
import json
import sys
import time
from pathlib import Path

from huggingface_hub import HfApi, RateLimitError, RepositoryNotFoundError

REPO_ID = "datasets/axentx/surrogate-1-training-pairs"

def list_date_files(date_folder: str, out_path: Path, max_retries: int = 3) -> None:
    api = HfApi()
    for attempt in range(1, max_retries + 1):
        try:
            entries = api.list_repo_tree(
                repo_id=REPO_ID,
                path=date_folder,
                repo_type="dataset",
                recursive=False,
            )
            files = [e.rfilename for e in entries if e.type == "file"]
            out_path.write_text(json.dumps({"date": date_folder, "files": files}, indent=2))
            print(f"Wrote {len(files)} files to {out_path}")
            return
        except RateLimitError:
            wait = 360
            print(f"Rate limited (429). Waiting {wait}s (attempt {attempt}/{max_retries})", file=sys.stderr)
            time.sleep(wait)
        except RepositoryNotFoundError:
            print(f"Repo not found: {REPO_ID}", file=sys.stderr)
            sys.exit(1)
        except Exception as exc:
            print(f"Unexpected error: {exc}", file=sys.stderr)
            if attempt == max_retries:
                sys.exit(1)
            time.sleep(5 * attempt)

    print("Max retries exceeded.", file=sys.stderr)
    sys.exit(1)

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="List date folder files for CDN ingestion.")
    parser.add_argument("--date", required=True, help="Date folder (e.g., 2026-04-29)")
    parser.add_argument("--out", required=True, help="Output JSON path")
    args = parser.parse_args()
    list_date_files(args.date, Path(args.out))
```

#### 2) `bin/dataset-enrich.sh`
```bash
#!/usr/bin/env bash
# dataset-enrich.sh
# Deterministic shard worker with optional file-list to avoid HF API 429s.
#
# Usage:
#   # With pre-generated file list (preferred)
#   export FILE_LIST=/path/to/file-list-2026-04-29.json
#   ./bin/dataset-enrich.sh
#
#   # Without file list (fallback; may hit API limits)
#   ./bin/dataset-enrich.sh

set -euo pipefail

HF_REPO="datasets/axentx/surrogate-1-training-pairs"
DATE="${DATE:-$(date +%Y-%m-%d)}"
SHARD_ID="${SHARD_ID:-0}"
TOTAL_SHARDS="${TOTAL_SHARDS:-16}"
OUT_DIR="batches/public-merged/${DATE}"
TIMESTAMP=$(date +%H%M%S)
OUT_FILE="${OUT_DIR}/shard${SHARD_ID}-${TIMESTAMP}.jsonl"

mkdir -p "$(dirname "$OUT_FILE")"

# Central dedup store (SQLite) - reused across runs
DEDUP_DB="${DEDUP_DB:-/tmp/dedup_store.db}"

# Optional pre-computed file list to avoid list_repo_tree calls
FILE_LIST="${FILE_LIST:-}"

log() { echo "[$(date -Iseconds)] $*"; }

# Deterministic shard assignment by slug hash
shard_for_slug() {
  local slug="$1"
  python3 -c "import hashlib,sys; h=int(hashlib.md5(sys.argv[1].encode()).hexdigest(),16); print(h % ${TOTAL_SHARDS})" "$slug"
}

# Resolve files to process
resolve_files() {
  if [[ -n "$FILE_LIST" && -f "$FILE_LIST" ]]; then
    python3 -c "
import json, sys
with open(sys.argv[1]) as f:
    data = json.load(f)
for fn in data.get('files', []):
    print(fn)
" "$FILE_LIST"
  else
    # Fallback: single non-recursive folder list (minimizes API calls)
    python3 -c "
from huggingface_hub import HfApi
api = HfApi()
entries = api.list_repo_tree('$HF_REPO', path='$DATE', repo_type='dataset', recursive=False)
for e in entries:
    if e.type == 'file':
        print(e.rfilename)
" 2>/dev/null || {
      log "WARNING: HF API unavailable; falling back to empty file set"
      return 0
    }
  fi
}

# CDN-only download (no Authorization header -> bypasses /api/ rate limits)
download_via_cdn() {
  local rel_path="$1"
  local url="https://huggingface.co/datasets/${HF_REPO}/resolve/main/${rel_path}"
  curl -fsSL --retry 3 --retry-delay 5 "$url"
}

# Central dedup check/insert via Python helper
is_duplicate() {
  local md5="$1"
  python3 -c "
import sqlite3, sys
conn = sqlite3.connect(sys.argv[1])
conn.execute('CREATE TABLE IF NOT EXISTS seen (md5 TEXT PRIMARY KEY)')
try:
    conn.execute('INSERT INTO seen (md5) VALUES (?)', (sys.argv[2],))
    conn.commit()
    conn.close()
    sys.exit(0)
except sqlite3.IntegrityError:
    conn.close()
    sys.exit(1)
" "$DEDUP_DB" "$md5"
}

# Main processing loop
process_file() {
  local rel_path="$1"
  local slug
  slug=$(basename "$rel_path" | sed 's/\.[^.]*$//')
  local target_shard
  target_shard=$(shard_for_slug "$slug")
  if [[ "$target_shard" != "$SHARD_ID" ]]; then
    return 0
  fi

  local content
  content=$(download_via_cdn "$rel_path") || {
    log "Failed to download $rel_path"
    return 1
  }

  local md5
  md5=$(echo -n "$content" | md5sum | awk '{print $1}')
  if is_duplicate "$md5"; then
    log "Duplicate skipped: $rel_path"
    return 0
  fi

  # Append enriched record (example)
  jq -n --arg slug "$slug" --arg content "$content" --arg path "$rel_path" \
    '{slug:$slug, content:$content, path:$path}' >> "$OUT_FILE"
  log "Processed shard $SHARD_ID: $rel_path"
}

# Run
log "Starting shard $SHARD_ID/$TOTAL_SHARDS for $DATE"
while IFS= read -r rel_path; do
  [[ -n "$rel_path" ]] && process_file "$rel_path"
done < <(resolve_files)

log "Completed shard $SHARD_ID -> $OUT_FILE"
```
