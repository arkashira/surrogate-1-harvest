# workio / quality

## 1. Diagnosis

- **No idempotency key on LINE webhook payloads** → LINE retries on 5xx/timeouts create duplicate clock-in/out records.
- **No tenant-scoped, one-open-punch constraint** → race conditions allow multiple concurrent punches per employee (corrupts hours/OT).
- **App-level uniqueness checks are racy** → `SELECT` then `INSERT` under concurrent webhook calls leads to duplicates.
- **No database-level uniqueness/index to enforce invariants** → integrity relies on app logic; easy to bypass via direct DB edits or future code paths.
- **Missing audit trail for punch state transitions** → hard to debug duplicates or reconcile reports after retries.

## 2. Proposed change

- **File scope**: `workio/server/src/db/schema.sql` (add constraint + index)  
- **File scope**: `workio/server/src/routes/line.ts` (add idempotency handling + atomic upsert)  
- **File scope**: `workio/server/src/services/punch.service.ts` (add `recordPunch` with `ON CONFLICT DO NOTHING` + audit insert)

## 3. Implementation

### 3.1 DB schema: enforce tenant+employee uniqueness for open punches and add idempotency key

```sql
-- workio/server/src/db/schema.sql
-- Add idempotency key to punches (for LINE webhook retries)
ALTER TABLE punches ADD COLUMN IF NOT EXISTS idempotency_key VARCHAR(255);
CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_idempotency
  ON punches (idempotency_key)
  WHERE idempotency_key IS NOT NULL;

-- Ensure one open punch per tenant+employee (null ended_at means open)
-- Use partial index + unique constraint via function for safety
CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_one_open_per_employee
  ON punches (tenant_id, employee_id)
  WHERE ended_at IS NULL;
```

### 3.2 Punch service: atomic insert with idempotency

```ts
// workio/server/src/services/punch.service.ts
import { Pool } from 'pg';

const pool = new Pool({ connectionString: process.env.DATABASE_URL });

export interface RecordPunchInput {
  tenantId: string;
  employeeId: string;
  type: 'IN' | 'OUT';
  latitude: number;
  longitude: number;
  lineUserId: string;
  // idempotency key provided by caller (e.g., LINE webhook event id or hash)
  idempotencyKey: string;
}

export async function recordPunch(input: RecordPunchInput) {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // Try insert; if idempotency_key exists, do nothing (safe retry)
    const insertText = `
      INSERT INTO punches (tenant_id, employee_id, type, latitude, longitude, line_user_id, idempotency_key, created_at)
      VALUES ($1, $2, $3, $4, $5, $6, $7, NOW())
      ON CONFLICT (idempotency_key) DO NOTHING
      RETURNING id, ended_at
    `;
    const insertRes = await client.query(insertText, [
      input.tenantId,
      input.employeeId,
      input.type,
      input.latitude,
      input.longitude,
      input.lineUserId,
      input.idempotencyKey,
    ]);

    // If no row inserted, it's a duplicate — return existing open punch state
    if (insertRes.rowCount === 0) {
      const getOpen = await client.query(
        'SELECT id, type, created_at FROM punches WHERE tenant_id = $1 AND employee_id = $2 AND ended_at IS NULL ORDER BY created_at DESC LIMIT 1',
        [input.tenantId, input.employeeId]
      );
      await client.query('COMMIT');
      return { duplicate: true, punch: getOpen.rows[0] || null };
    }

    const punchId = insertRes.rows[0].id;

    // If this is an "OUT", close the most recent open punch for this employee
    if (input.type === 'OUT') {
      await client.query(
        `UPDATE punches
         SET ended_at = NOW()
         WHERE id = (
           SELECT id FROM punches
           WHERE tenant_id = $1 AND employee_id = $2 AND ended_at IS NULL AND id != $3
           ORDER BY created_at DESC
           LIMIT 1
         )`,
        [input.tenantId, input.employeeId, punchId]
      );
    }

    // Audit trail for state transition
    await client.query(
      `INSERT INTO punch_audit (punch_id, action, actor_line_user_id, created_at)
       VALUES ($1, $2, $3, NOW())`,
      [punchId, input.type, input.lineUserId]
    );

    await client.query('COMMIT');
    return { duplicate: false, punch: { id: punchId, type: input.type } };
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  } finally {
    client.release();
  }
}
```

### 3.3 LINE webhook route: attach idempotency key and handle safely

// workio/server/src/routes/line.ts
import { Router } from 'express';
import { recordPunch } from '../services/punch.service';

const router = Router();

router.post('/webhook/line', async (req, res) => {
  try {
    const events = req.body.events || [];
    for (const ev of events) {
      if (ev.type === 'message' && ev.message.type === 'text') {
        // Example: employee texts "IN" or "OUT"
        const text = ev.message.text.trim().toUpperCase();
        if (text !== 'IN' && text !== 'OUT') continue;

        // Resolve tenant/employee by lineUserId (simplified)
        const employee = await resolveEmployeeByLineId(ev.source.userId);
        if (!employee) continue;

        // Use LINE webhook event id + message id as idempotency key
        const idempotencyKey = `line:${ev.source.userId}:${ev.message.id}`;

        await recordPunch({
          tenantId: employee.tenant_id,
          employeeId: employee.id,
          type: text,
          latitude: 0, // GPS should come from location message or payload
          longitude: 0,
          lineUserId: ev.source.userId,
