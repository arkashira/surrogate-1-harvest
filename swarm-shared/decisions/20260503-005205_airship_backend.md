# airship / backend

## Final Synthesis — Best single improvement (<2h)

**Goal:** Eliminate runtime API calls and schema errors during Surrogate-1 training by switching to a manifest-driven, CDN-only dataset loader with deterministic upload routing.

---

## 1. What to build (single deliverable)

A small, production-ready package that:

- **Generates a manifest once per date folder** (`scripts/build_dataset_manifest.py`)  
  → one `list_repo_tree` call; writes `data/manifests/{date}/file-list.json`.

- **Loads training data from CDN only** (`surrogate/data/cdn_dataset.py`)  
  → `IterableDataset` that streams files listed in the manifest via raw CDN URLs.  
  → Projects each record to `{prompt, response}` at parse time (avoids `pyarrow.CastError` from mixed/extra columns).  
  → Retries + skips bad files instead of crashing training.

- **Routes enriched parquet to a deterministic sibling repo** (`surrogate/data/upload_utils.py`)  
  → `hash(slug) % 5` picks one of five mirror repos.  
  → Writes `batches/mirror-merged/{date}/{slug}.parquet` with only `prompt, response`.

- **Updates training launcher** (`surrogate/train_surrogate_1.py`)  
  → Uses `CdnDataset(manifest_path)` and removes any `load_dataset(streaming=True)` calls.  
  → Keeps performant `DataLoader(num_workers>0, prefetch_factor=2, pin_memory=True)`.

- **Lightning Studio reuse guard** (`surrogate/lightning/launcher.py`)  
  → Lists running studios and reuses a matching one; on idle stop, restarts on `Machine.L40S`.

- **Cron/orchestration hygiene**  
  → `SHELL=/bin/bash` in crontab; `#!/usr/bin/env bash` shebangs; invoke via `bash script.sh "$@"`.

## 2. Implementation (ready to copy)

### `scripts/build_dataset_manifest.py`
```python
#!/usr/bin/env python3
"""
Build CDN manifest for a date folder in a HuggingFace dataset repo.
Usage: python build_dataset_manifest.py --repo <repo> --date 2026-05-03 --out data/manifests/2026-05-03/file-list.json
"""
import argparse
import json
import sys
from pathlib import Path

from huggingface_hub import list_repo_tree


def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g., org/ds)")
    parser.add_argument("--date", required=True, help="Date folder (YYYY-MM-DD)")
    parser.add_argument("--out", required=True, help="Output JSON path")
    args = parser.parse_args()

    prefix = f"{args.date}/"
    out_path = Path(args.out)
    out_path.parent.mkdir(parents=True, exist_ok=True)

    try:
        tree = list_repo_tree(repo_id=args.repo, path=prefix, recursive=False)
        files = [
            item.rfilename
            for item in tree
            if item.rfilename.endswith((".json", ".jsonl", ".parquet"))
        ]
        manifest = {"repo": args.repo, "date": args.date, "prefix": prefix, "files": files}
        out_path.write_text(json.dumps(manifest, indent=2))
        print(f"Wrote {len(files)} files to {out_path}")
        sys.exit(0)
    except Exception as exc:
        print(f"Failed to list {args.repo}/{prefix}: {exc}", file=sys.stderr)
        sys.exit(1)


if __name__ == "__main__":
    main()
```

### `surrogate/data/cdn_dataset.py`
```python
from __future__ import annotations

import json
from io import BytesIO
from typing import Any, Dict, Iterator

import pyarrow.parquet as pq
import requests
from torch.utils.data import IterableDataset

CDN_ROOT = "https://huggingface.co/datasets"


class CdnDataset(IterableDataset):
    """
    CDN-only dataset loader for HuggingFace datasets.
    Manifest format: {"repo": "...", "date": "...", "files": ["date/file1.json", ...]}
    """

    def __init__(self, manifest_path: str, split: str = "train", max_retries: int = 3) -> None:
        manifest = json.loads(open(manifest_path).read())
        self.repo = manifest["repo"]
        self.files = manifest["files"]
        self.split = split
        self.max_retries = max_retries

    def _download_file(self, path: str) -> bytes:
        url = f"{CDN_ROOT}/{self.repo}/resolve/main/{path}"
        for attempt in range(self.max_retries):
            try:
                resp = requests.get(url, timeout=30)
                resp.raise_for_status()
                return resp.content
            except Exception as exc:
                if attempt == self.max_retries - 1:
                    raise RuntimeError(f"Failed to fetch {url}: {exc}") from exc
        raise RuntimeError("Unreachable")

    def _parse_to_record(self, path: str, content: bytes) -> Iterator[Dict[str, Any]]:
        # Project to {prompt, response} only — ignore heterogeneous schema
        if path.endswith(".json"):
            data = json.loads(content)
            items = data if isinstance(data, list) else [data]
            for item in items:
                yield {"prompt": item.get("prompt"), "response": item.get("response")}
        elif path.endswith(".jsonl"):
            for line in content.decode("utf-8").splitlines():
                line = line.strip()
                if not line:
                    continue
                item = json.loads(line)
                yield {"prompt": item.get("prompt"), "response": item.get("response")}
        elif path.endswith(".parquet"):
            table = pq.read_table(pq.ParquetFile(BytesIO(content)))
            df = table.to_pandas()
            for _, row in df.iterrows():
                yield {"prompt": row.get("prompt"), "response": row.get("response")}
        else:
            raise ValueError(f"Unsupported file type: {path}")

    def __iter__(self) -> Iterator[Dict[str, Any]]:
        for path in self.files:
            try:
                content = self._download_file(path)
                yield from self._parse_to_record(path, content)
            except Exception as exc:
                # Log and skip bad files to avoid crashing training
                print(f"Skipping {path}: {exc}")
                continue
```
