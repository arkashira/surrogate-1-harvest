# vanguard / ops

## 1. Diagnosis
- No persisted HF file manifest per `(repo, dateFolder)` — every training run re-enumerates via authenticated HF API, burning quota and risking 429.
- Training script likely uses `load_dataset(streaming=True)` on heterogeneous repos, causing `pyarrow.CastError` on mixed schemas.
- Lightning Studio reuse is not enforced — new runs create fresh studios, wasting 80+ hrs/mo quota.
- No resilience to Lightning idle-stop — if a studio stops, training dies instead of restarting on L40S.
- Mac orchestration may attempt local model loading instead of delegating training to Lightning (architecture rule violation).

## 2. Proposed change
Create `/opt/axentx/vanguard/ops/training_ops.py` (single orchestrator) and `/opt/axentx/vanguard/ops/train.py` (Lightning training stub).  
Key scope:
- Add `create_manifest(repo, date_folder) -> manifest.json` using one authenticated `list_repo_tree` call (non-recursive) and persist to `manifests/{repo}/{date_folder}.json`.
- Train script reads manifest and downloads via HF CDN (`resolve/main/...`) with zero authenticated calls.
- Project only `{prompt, response}` at parse time; ignore heterogeneous extras.
- Reuse running Lightning Studio or start L40S if stopped; never create duplicate studios.
- Guard against idle-stop by checking status before `.run()` and restarting if needed.

## 3. Implementation

```bash
# /opt/axentx/vanguard/ops/training_ops.py
#!/usr/bin/env python3
"""
Ops orchestrator for Surrogate-1 training.
- Creates CDN-only manifest per (repo, date_folder)
- Starts/reuses Lightning Studio (L40S)
- Launches training with manifest (zero HF API calls during data load)
"""
import json, os, sys, time, hashlib
from pathlib import Path

import requests
from lightning_sdk import Lightning, Teamspace, Studio, Machine

HF_API_BASE = "https://huggingface.co/api"
HF_CDN_BASE = "https://huggingface.co/datasets"

MANIFEST_DIR = Path(__file__).parent.parent / "manifests"
MANIFEST_DIR.mkdir(parents=True, exist_ok=True)

# Optional: set HF token via env; fallback to unauthenticated listing where possible
HF_TOKEN = os.getenv("HF_TOKEN", "")

def list_repo_files(repo: str, path: str = "") -> list[str]:
    """List files non-recursively (single page) to avoid 429."""
    headers = {"Authorization": f"Bearer {HF_TOKEN}"} if HF_TOKEN else {}
    url = f"{HF_API_BASE}/repos/datasets/{repo}/tree"
    params = {"path": path, "recursive": "false"}
    r = requests.get(url, headers=headers, params=params, timeout=30)
    if r.status_code == 429:
        wait = int(r.headers.get("Retry-After", 360))
        print(f"Rate limited. Waiting {wait}s")
        time.sleep(wait)
        return list_repo_files(repo, path)
    r.raise_for_status()
    items = r.json()
    # Return paths relative to repo root
    return [item["path"] for item in items if item.get("type") == "file"]

def create_manifest(repo: str, date_folder: str) -> Path:
    """Create persisted manifest for (repo, date_folder)."""
    files = list_repo_files(repo, date_folder)
    manifest = {
        "repo": repo,
        "date_folder": date_folder,
        "files": files,
        "cdn_base": f"{HF_CDN_BASE}/{repo}/resolve/main/{date_folder}",
        "created_at": time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime()),
    }
    out = MANIFEST_DIR / repo.replace("/", "_") / f"{date_folder}.json"
    out.parent.mkdir(parents=True, exist_ok=True)
    out.write_text(json.dumps(manifest, indent=2))
    print(f"Manifest written: {out}")
    return out

def find_or_start_studio(name: str = "vanguard-surrogate1") -> Studio:
    """Reuse running studio; otherwise start L40S on lightning-lambda-prod."""
    team = Teamspace.current()
    for s in team.studios:
        if s.name == name:
            if s.status == "running":
                print(f"Reusing running studio: {s.id}")
                return s
            else:
                print(f"Studio {s.id} exists but stopped. Restarting...")
                s.start(machine=Machine.L40S, cloud="lightning-lambda-prod")
                s.wait_until_ready()
                return s
    print("Creating new studio (L40S)...")
    return Studio.create(
        name=name,
        machine=Machine.L40S,
        cloud="lightning-lambda-prod",
        create_ok=True,
    )

def run_training(manifest_path: Path):
    studio = find_or_start_studio()
    train_script = Path(__file__).parent / "train.py"
    # Pass manifest path so train.py does CDN-only fetches
    run = studio.run(
        str(train_script),
        arguments=[str(manifest_path)],
        framework="lightning",
        requirements=["torch", "pyarrow", "requests"],
    )
    print(f"Training run submitted: {run.id}")
    # Monitor and handle idle-stop resilience
    while True:
        run.refresh()
        if run.status == "stopped":
            print("Studio stopped during training. Restarting studio and run...")
            studio.start(machine=Machine.L40S, cloud="lightning-lambda-prod")
            studio.wait_until_ready()
            run = studio.run(
                str(train_script),
                arguments=[str(manifest_path)],
                framework="lightning",
                requirements=["torch", "pyarrow", "requests"],
            )
        time.sleep(60)

if __name__ == "__main__":
    if len(sys.argv) < 3:
        print("Usage: training_ops.py <repo> <date_folder>")
        sys.exit(1)
    repo, date_folder = sys.argv[1], sys.argv[2]
    mf = create_manifest(repo, date_folder)
    run_training(mf)
```

# /opt/axentx/vanguard/ops/train.py
#!/usr/bin/env python3
"""
Lightning training script (runs inside Studio).
Expects one argument: path to manifest JSON.
Downloads via HF CDN only (zero authenticated API calls).
Projects heterogeneous files to {prompt, response}.
"""
import json, sys, csv, io, gzip, mimetypes
from pathlib import Path
from typing import Dict, Any

import pyarrow as pa
import pyarrow.parquet as pq
import requests
from datasets import Dataset

HF_CDN_BASE = "https://huggingface.co/datasets"

def cdn_url(repo: str, filepath: str) -> str:
    return f"{HF_CDN_BASE}/{repo}/resolve/main/{filepath}"

def project_to_pair(raw: Dict[str, Any]) -> Dict[str, str]:
    """Best-effort projection to {prompt, response}. Ignore extras."""
    prompt = raw.get("prompt") or raw.get("input") or raw.get("question") or ""
    response = raw.get("response") or raw.get("output") or raw.get("answer") or ""
    # If raw is a single text field, attempt simple split
    if not prompt and not response:
        text = raw.get("text", "")
        if "\n\n" in text:
            parts = text.split("\n\n", 1)
            prompt, response = parts[0], parts[1]
        else:
            prompt, response = "", text
    return {"prompt": str(prompt), "response": str(response)}

def fetch_and_parse(repo: str, filepath: str):
    url = cdn_url(repo, filepath)
    r = requests.get(url, timeout=60)
    r.raise_for_status()
    content = r.content

    # Decompress if needed
    if filepath.endswith(".gz") or mimetypes.guess_type(filepath)[1] == "gzip":
        content = gzip.decompress(content)

    name = Path(filepath).name.lower()
    if name.endswith(".jsonl") or name.endswith(".json"):
        lines = content.decode("utf-8").strip().splitlines()
        for line in lines:
            if not line.strip():
                continue
