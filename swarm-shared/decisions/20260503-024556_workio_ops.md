# workio / ops

## Final synthesized solution (correct + actionable)

### 1. Diagnosis (merged, resolved)
- **Root cause**: LINE retries on slow 2xx/network timeouts + missing idempotency + no DB uniqueness = duplicate punches.
- **Race**: read-then-insert gap under concurrent webhooks.
- **Missing defenses**: no DB constraint, no idempotency store, no observability.
- **Frontend risk**: double-tap and perceived latency (add idempotency key plumbing to clock UI).
- **Observability**: no metrics/logs to detect duplicates or retry storms.

### 2. Changes (file scope)
- `workio/server/src/db/schema.sql` — add DB-level uniqueness and idempotency table.
- `workio/server/src/routes/webhook/line.ts` (or `line.ts` if path differs) — atomic upsert + idempotency + observability.
- `workio/server/src/lib/idempotency.ts` — idempotency helpers.
- Frontend clock-in/out flow — generate and send `X-Idempotency-Key` header.

### 3. Implementation

#### 3.1 DB schema (single source of truth)
```sql
-- workio/server/src/db/schema.sql

-- Prevent duplicate punches at the persistence layer
-- Use a rounded timestamp (minute) to absorb microsecond/retry jitter
ALTER TABLE punches
  ADD CONSTRAINT uniq_tenant_user_timestamp_minute
  UNIQUE (tenant_id, user_id, date_trunc('minute', timestamp));

-- Lightweight idempotency store for LINE webhook retries (24h TTL)
CREATE TABLE IF NOT EXISTS webhook_idempotency (
  idempotency_key TEXT NOT NULL,
  tenant_id       INTEGER NOT NULL,
  user_id         INTEGER NOT NULL,
  payload_hash    TEXT NOT NULL,
  created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  processed_at    TIMESTAMPTZ,
  response_status INTEGER,
  response_body   JSONB,
  PRIMARY KEY (idempotency_key, tenant_id)
);

-- Fast expiry index for cleanup jobs / bounded scans
CREATE INDEX IF NOT EXISTS idx_webhook_idempotency_expiry
  ON webhook_idempotency (created_at)
  WHERE processed_at IS NULL;
```

#### 3.2 Idempotency helper (correct + minimal)
```ts
// workio/server/src/lib/idempotency.ts
import { PoolClient } from 'pg';
import crypto from 'crypto';

export function generateIdempotencyKey(headers: Record<string, any>): string | null {
  const key = headers['x-idempotency-key'] || headers['x-line-retry-id'];
  return typeof key === 'string' && key.trim() ? key.trim() : null;
}

export async function acquireIdempotency(
  client: PoolClient,
  idempotencyKey: string,
  tenantId: number,
  userId: number,
  payload: any
): Promise<{ isReplay: boolean; existing?: { status: number; body: any } }> {
  const payloadHash = crypto.createHash('sha256')
    .update(JSON.stringify(payload))
    .digest('hex');

  // Insert new attempt; if conflict and already processed -> replay
  const { rows } = await client.query(
    `INSERT INTO webhook_idempotency (idempotency_key, tenant_id, user_id, payload_hash)
     VALUES ($1, $2, $3, $4)
     ON CONFLICT (idempotency_key, tenant_id)
       DO UPDATE
         SET payload_hash = EXCLUDED.payload_hash
         WHERE webhook_idempotency.processed_at IS NULL
     RETURNING idempotency_key, processed_at, response_status, response_body`,
    [idempotencyKey, tenantId, userId, payloadHash]
  );

  const row = rows[0];
  if (!row) {
    // Should not happen; defensive fallback
    return { isReplay: true, existing: { status: 200, body: { ok: true } } };
  }

  if (row.processed_at) {
    return { isReplay: true, existing: { status: row.response_status, body: row.response_body } };
  }
  return { isReplay: false };
}

export async function markIdempotencyProcessed(
  client: PoolClient,
  idempotencyKey: string,
  tenantId: number,
  status: number,
  body: any
) {
  await client.query(
    `UPDATE webhook_idempotency
      SET processed_at = NOW(), response_status = $3, response_body = $4
      WHERE idempotency_key = $1 AND tenant_id = $2`,
    [idempotencyKey, tenantId, status, body]
  );
}
```

#### 3.3 Webhook handler (atomic, observable, correct)
```ts
// workio/server/src/routes/webhook/line.ts
import { Router, Request, Response } from 'express';
import { pool } from '../../db';
import { generateIdempotencyKey, acquireIdempotency, markIdempotencyProcessed } from '../lib/idempotency';
import { verifySignature } from '../lib/line-signature';
import crypto from 'crypto';

const router = Router();

router.post('/webhook/line', verifySignature, async (req: Request, res: Response) => {
  const start = Date.now();
  const tenantId = req.tenant?.id;
  if (!tenantId) return res.status(400).json({ error: 'tenant missing' });

  const events = req.body?.events;
  if (!Array.isArray(events) || !events.length) return res.status(200).json({ ok: true });

  const idempotencyKey = generateIdempotencyKey(req.headers) || crypto.randomUUID();
  const firstEvent = events[0];
  const userId = firstEvent?.source?.userId ? parseInt(firstEvent.source.userId.replace(/\D/g, ''), 10) || 0 : 0;

  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    const { isReplay, existing } = await acquireIdempotency(client, idempotencyKey, tenantId, userId, req.body);
    if (isReplay) {
      await client.query('COMMIT');
      console.log(`[line-webhook] replay key=${idempotencyKey} tenant=${tenantId} status=${existing?.status}`);
      return res.status(existing?.status || 200).json(existing?.body || { ok: true });
    }

    // Process events atomically within transaction
    for (const ev of events) {
      if (ev.type === 'message' && ev.message?.type === 'text' && /(ลงเวลา|clock)/i.test(ev.message.text || '')) {
        await upsertPunch(client, tenantId, userId, ev);
      }
      if (ev.type === 'postback' && ev.postback?.data?.startsWith('punch=')) {
        const [, action] = ev.postback.data.split('=');
        await upsertPunch(client, tenantId, userId, ev, action);
      }
    }

    await client.query('COMMIT');
    await markIdempotencyProcessed(client, idempotencyKey, tenantId, 200, { ok: true });
    console.log(`[line-webhook] processed key=${idempotencyKey} tenant=${tenantId} events=${events.length} ms=${Date.now() - start}`);
    res.status(200).json({ ok: true });
  } catch (err) {
    await client.query('ROLLBACK');
    await markIdempotencyProcessed(client, idempotencyKey, tenantId, 500, { error: String(err) });
    console.error('[line-webhook] error', err);
    res.status(500).json({ error: 'internal' });
  } finally {
    client.release();
  }
});
