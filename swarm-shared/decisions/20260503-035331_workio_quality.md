# workio / quality

## Final Synthesis (Best Parts + Correctness + Actionability)

**Core problem**: LINE retries on 5xx/timeouts; handler is not idempotent, commits can fail silently after 200, and frontend tabs diverge.  
**Non-negotiable fix**: make ingestion idempotent, durable (commit before 200), and observable in real time.

## 2. Idempotent webhook handler

`server/src/routes/line/webhook.ts`

```ts
import { Router, Request, Response } from 'express';
import { verifySignature } from '../../utils/lineCrypto';
import { db } from '../../db';
import { processClockEvent } from '../../services/clockService';
import { publishEvent } from '../../services/eventService';

const router = Router();

router.post('/line', async (req: Request, res: Response) => {
  const sig = req.headers['x-line-signature'] as string;
  const body = JSON.stringify(req.body);
  if (!verifySignature(body, sig)) {
    return res.status(401).send('Invalid signature');
  }

  const events = req.body.events;
  if (!Array.isArray(events) || events.length === 0) {
    return res.status(200).send('OK');
  }

  try {
    await db.tx(async (t) => {
      for (const ev of events) {
        if (ev.type !== 'message' && ev.type !== 'postback') continue;

        // Stable idempotency key: LINE delivery id preferred; fallback to deterministic composite
        const deliveryId = ev.deliveryId || ev.message?.id || ev.postback?.id || ev.timestamp;
        const idemKey = `line:${ev.source.userId}:${ev.type}:${deliveryId}`;

        const exists = await t.oneOrNone(
          'SELECT 1 FROM line_webhook_events WHERE idempotency_key = $1',
          [idemKey]
        );
        if (exists) continue; // already processed

        // Process domain logic
        await processClockEvent(t, ev);

        // Record idempotency and emit domain event (within same tx)
        await t.none(
          'INSERT INTO line_webhook_events(idempotency_key, event_type, payload) VALUES($1,$2,$3)',
          [idemKey, ev.type, ev]
        );

        await publishEvent(t, {
          event_type: 'line_webhook_processed',
          tenant_id: ev.source.userId, // map via user->tenant in real impl
          user_id: ev.source.userId,
          payload: { lineEvent: ev }
        });
      }
    });

    // Commit succeeded — return 200 only after durable write
    res.status(200).send('OK');
  } catch (err) {
    console.error('Webhook processing failed', err);
    // Return 5xx so LINE retries (per LINE spec)
    res.status(500).send('Processing failed');
  }
});

export default router;
```

## 3. Idempotent domain service

`server/src/services/clockService.ts`

```ts
import { ITask } from 'pg-promise';

export async function processClockEvent(t: ITask<any>, lineEvent: any) {
  const userId = lineEvent.source.userId;
  const tenantId = await resolveTenantId(t, userId);
  const ts = new Date(lineEvent.timestamp);

  const intent = lineEvent.postback?.data || lineEvent.message?.text || 'clock';

  if (intent.startsWith('clock')) {
    // Determine next valid action
    const last = await t.oneOrNone(
      `SELECT id, action FROM punches
       WHERE tenant_id = $1 AND user_id = $2 AND DATE(created_at) = DATE($3)
       ORDER BY created_at DESC LIMIT 1`,
      [tenantId, userId, ts]
    );

    const nextAction = !last ? 'clock_in' : (last.action === 'clock_in' ? 'clock_out' : 'clock_in');

    // Deterministic rounding to avoid near-simultaneous duplicates
    const rounded = new Date(Math.floor(ts.getTime() / 1000) * 1000);

    // Idempotent insert: rely on unique constraint at DB level (preferred) + existence check
    const exists = await t.oneOrNone(
      'SELECT 1 FROM punches WHERE tenant_id=$1 AND user_id=$2 AND action=$3 AND created_at=$4',
      [tenantId, userId, nextAction, rounded]
    );
    if (!exists) {
      await t.none(
        'INSERT INTO punches(tenant_id, user_id, action, created_at, line_event_id) VALUES($1,$2,$3,$4,$5)',
        [tenantId, userId, nextAction, rounded, lineEvent.message?.id || lineEvent.timestamp]
      );

      await t.none(
        'INSERT INTO domain_events(event_type, tenant_id, user_id, payload) VALUES($1,$2,$3,$4)',
        ['punch_recorded', tenantId, userId, { action: nextAction, ts: rounded }]
      );
    }
  }

  // leave/OT: apply same pattern — idempotency key + unique constraint + domain event
}

async function resolveTenantId(t: ITask<any>, userId: string): Promise<string> {
  const row = await t.oneOrNone('SELECT tenant_id FROM users WHERE id = $1', [userId]);
  if (!row?.tenant_id) throw new Error(`No tenant for user ${userId}`);
  return row.tenant_id;
}
```

## 4. Real-time broadcast to frontend (SSE)

`server/src/services/eventService.ts`

```ts
import { db } from '../db';

const listeners: Array<(data: string) => void> = [];

export async function publishEvent(
  t: any,
  ev: { event_type: string; tenant_id: string; user_id: string; payload: any }
) {
  await db.none(
    'INSERT INTO domain_events(event_type, tenant_id, user_id, payload) VALUES($1,$2,$3,$4)',
    [ev.event_type, ev.tenant_id, ev.user_id, ev.payload]
  );

  const msg = JSON.stringify(ev);
  listeners.forEach((fn) => {
    try { fn(msg); } catch { /* ignore dead clients */ }
  });
}

export function addListener(fn: (data: string) => void) {
  listeners.push(fn);
}

export function removeListener(fn: (data: string) => void) {
  const idx = listeners.indexOf(fn);
  if (idx > -1) listeners.splice(idx, 1);
}
```

`server/src/routes/events.ts`

```ts
import { Router } from 'express';
import { addListener, removeListener } from '../services/eventService';

const router = Router();

router.get('/sse', (req, res) => {
  res.setHeader('Content-Type', 'text/event-stream');
  res.setHeader('Cache-Control', 'no-cache');
  res.setHeader('Connection', 'keep-alive');
  res.flushHeaders();

  const send = (data: string) => res.write(`data: ${data}\n\n`);
  addListener(send);

  req.on('close', () => removeListener(send));
});

export default router;
```
