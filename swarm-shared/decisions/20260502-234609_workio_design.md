# workio / design

## Final Synthesis — Canonical Route Contract (Type-safe, Frontend + Backend)

**Goal**: One source of truth for routes, params, and search params; typed href builders; lightweight runtime validation for Express; zero URL drift; minimal surface area and safe defaults.

---

export type RouteName = keyof typeof routes;
export type RouteDef = (typeof routes)[RouteName];

// ── Typed href builder ────────────────────────────────────────
export function href<Name extends RouteName>(
  name: Name,
  params: z.input<(typeof routes)[Name]['params']>,
  search?: z.input<(typeof routes)[Name]['search']>
): string {
  const route = routes[name];
  // Validate params at runtime (cheap; catches runtime mismatches)
  route.params.parse(params);

  let url = route.path;
  for (const [k, v] of Object.entries(params)) {
    url = url.replace(`:${k}`, encodeURIComponent(String(v)));
  }

  const sp = new URLSearchParams();
  if (search) {
    // Use schema to coerce/validate search params
    const parsed = route.search.safeParse(search);
    if (!parsed.success) {
      throw new Error(`Invalid search params for route "${name}": ${parsed.error.message}`);
    }
    for (const [k, v] of Object.entries(parsed.data)) {
      if (v != null) sp.set(k, String(v));
    }
  }
  const qs = sp.toString();
  return qs ? `${url}?${qs}` : url;
}

// ── Express middleware (lightweight) ──────────────────────────
import type { Request, Response, NextFunction } from 'express';

export function validateSearch<Name extends RouteName>(name: Name) {
  const schema = routes[name].search;
  return (req: Request, res: Response, next: NextFunction) => {
    const result = schema.safeParse(req.query);
    if (!result.success) {
      return res.status(400).json({ error: 'Invalid query params', details: result.error.format() });
    }
    // Coerce validated values onto req.query for downstream handlers
    req.query = result.data;
    next();
  };
}

// Optional: param-only validator for routes that need path param validation
export function validateParams<Name extends RouteName>(name: Name) {
  const schema = routes[name].params;
  return (req: Request, res: Response, next: NextFunction) => {
    const result = schema.safeParse(req.params);
    if (!result.success) {
      return res.status(400).json({ error: 'Invalid path params', details: result.error.format() });
    }
    // Normalize req.params shape (keeps string values but ensures shape)
    req.params = result.data;
    next();
  };
}

### 2. Frontend integration (React Router v6 + typed navigation)

- Keep existing URLs unchanged.
- Use `href(...)` for link generation and programmatic navigation.
- Optionally map route catalog to React Router route objects for DRY paths.

`workio/src/router.tsx` (example):

```tsx
import { createBrowserRouter, useNavigate, Link } from 'react-router-dom';

// Example: typed navigation in components
export function useTypedNavigate() {
  const navigate = useNavigate();
  return {
    to: <Name extends keyof typeof routes>(name: Name, params: any, search?: any) => {
      navigate(href(name, params, search));
    },
  };
}
