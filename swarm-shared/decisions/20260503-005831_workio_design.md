# workio / design

### Final synthesized solution (best parts, corrected, fully actionable)

**Core diagnosis (merged, de-duplicated)**
- Missing **DB-level guardrails**: no enforcement of “one open punch per user” (`clock_out_at IS NULL`) under concurrency or LINE retries.
- **Race-prone app upsert**: find-then-insert/update allows multiple open punches.
- **No idempotency for LINE webhook retries**: duplicate events create duplicate punches or double state changes.
- **Weak clock-in validation**: stale open punches are not reliably rejected before new clock-ins.
- **No audit trail for corrections**: admin/auto-closes of punches are not recorded.

**Goal**
- Enforce single open punch at the database (hard guarantee).
- Make punch operations idempotent and safe under retries/concurrency.
- Add minimal audit for corrections.

### 1) Database schema changes (definitive)

**File:** `server/src/db/schema.sql` (or migration file)

```sql
-- 1) Enforce at most one open punch per user
-- Use partial unique index (works under concurrency, no locking pitfalls)
CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_one_open_per_user
ON punches (user_id)
WHERE clock_out_at IS NULL;

-- 2) Lightweight idempotency table for LINE webhook events
CREATE TABLE IF NOT EXISTS line_webhook_idempotency (
  idempotency_key TEXT PRIMARY KEY,
  event_id        TEXT NOT NULL,
  processed_at    TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX IF NOT EXISTS idx_line_webhook_idempotency_event_id
ON line_webhook_idempotency (event_id);

-- 3) Audit trail for punch corrections (admin/auto closes)
CREATE TABLE IF NOT EXISTS punch_corrections (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  punch_id        UUID NOT NULL REFERENCES punches(id),
  corrected_by    UUID, -- user id or system
  correction_type TEXT NOT NULL, -- e.g. 'auto_close_stale', 'admin_override'
  old_clock_out_at TIMESTAMPTZ,
  new_clock_out_at TIMESTAMPTZ,
  reason          TEXT,
  created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
```

Apply immediately (one-time):
```bash
psql $DATABASE_URL < server/src/db/schema.sql
```

### 2) Controller / route — safe, idempotent punch handling

**File:** `server/src/controllers/ClockController.js` (or route equivalent)

```javascript
const { v4: uuidv4 } = require('uuid');

/**
 * Idempotent clock-in
 * POST /clock/in
 * Body: { userId, lineEventId?, idempotencyKey? }
 */
async function clockIn(req, res) {
  const userId = req.user?.id || req.body.userId;
  const lineEventId = req.body.lineEventId;
  const idempotencyKey = req.body.idempotencyKey || lineEventId || uuidv4();

  if (!userId) {
    return res.status(400).json({ error: 'userId required' });
  }

  const client = await db.pool.connect();
  try {
    await client.query('BEGIN');

    // Idempotency check
    const idem = await client.query(
      `SELECT 1 FROM line_webhook_idempotency WHERE idempotency_key = $1`,
      [idempotencyKey]
    );
    if (idem.rows.length > 0) {
      await client.query('ROLLBACK');
      return res.status(200).json({ ok: true, message: 'already processed' });
    }

    // Insert new punch; partial unique index prevents multiple open punches
    const result = await client.query(
      `INSERT INTO punches (id, user_id, clock_in_at, clock_out_at)
       VALUES ($1, $2, NOW(), NULL)
       RETURNING *`,
      [uuidv4(), userId]
    );

    await client.query(
      `INSERT INTO line_webhook_idempotency (idempotency_key, event_id)
       VALUES ($1, $2)`,
      [idempotencyKey, lineEventId || null]
    );

    await client.query('COMMIT');
    return res.status(201).json({ punch: result.rows[0] });
  } catch (err) {
    await client.query('ROLLBACK');

    // Unique violation from partial index => multiple open punches blocked
    if (err.code === '23505') {
      return res.status(409).json({
        error: 'open_punch_exists',
        message: 'User already has an open punch'
      });
    }

    console.error('Clock-in error', err);
    return res.status(500).json({ error: 'clock_in_failed' });
  } finally {
    client.release();
  }
}

 * Idempotent clock-out

async function clockOut(req, res) {
  const userId = req.user?.id || req.body.userId;
  const lineEventId = req.body.lineEventId;
  const idempotencyKey = req.body.idempotencyKey || lineEventId || uuidv4();

  if (!userId) {
    return res.status(400).json({ error: 'userId required' });
  }

  const client = await db.pool.connect();
  try {
    await client.query('BEGIN');

    // Idempotency check
    const idem = await client.query(
      `SELECT 1 FROM line_webhook_idempotency WHERE idempotency_key = $1`,
      [idempotencyKey]
    );
    if (idem.rows.length > 0) {
      await client.query('ROLLBACK');
      return res.status(200).json({ ok: true, message: 'already processed' });
    }

    // Close latest open punch for user
    const result = await client.query(
      `UPDATE punches
       SET clock_out_at = NOW()
       WHERE id = (
         SELECT id FROM punches
         WHERE user_id = $1 AND clock_out_at IS NULL
         ORDER BY clock_in_at DESC
         LIMIT 1
       )
       RETURNING *`,
      [userId]
    );

    if (result.rowCount === 0) {
      await client.query('ROLLBACK');
      return res.status(404).json({ error: 'no_open_punch' });
    }

    await client.query(
      `INSERT INTO line_webhook_idempotency (idempotency_key, event_id)
       VALUES ($1, $2)`,
      [idempotencyKey, lineEventId || null]
    );

    await client.query('COMMIT');
    return res.status(200).json({ punch: result.rows[0] });
  } catch (err) {
    await client.query('ROLLBACK');
    console.error('Clock-out error', err);
    return res.status(500).json({ error: 'clock_out_failed' });
  } finally {
    client.release();
  }
}
