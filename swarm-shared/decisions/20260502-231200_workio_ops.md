# workio / ops

## 1. Diagnosis

- No storage-level uniqueness on `(employee_id, punch_type, date)` allows duplicate punches from retries/races.
- Punch write path is read-then-insert (non-idempotent) and vulnerable to concurrent requests and LINE webhook redelivery.
- No idempotency key or deterministic punch identifier to deduplicate retries at the API layer.
- Clock-in/out endpoint lacks conflict detection for overlapping punches (e.g., double clock-in without clock-out).
- No audit trail (created_at / source) to distinguish legitimate duplicates from retries.

## 2. Proposed change

- **Files**: `workio/server/src/db/schema.sql` (add constraint + index), `workio/server/src/routes/punch.ts` (add idempotency + conflict check), optionally `workio/server/src/middleware/idempotency.ts` (new).
- **Scope**: enforce uniqueness at DB, make punch writes idempotent via client-provided `idempotency_key` (or deterministic hash), and reject overlapping punches.

## 3. Implementation

### 3.1 DB schema (add constraint + index)

```sql
-- workio/server/src/db/schema.sql
-- Add unique constraint to prevent duplicates at source of truth
ALTER TABLE punches
  ADD CONSTRAINT punches_employee_punch_type_date_uniq
  UNIQUE (employee_id, punch_type, date);

-- Fast lookup for conflict checks and idempotency
CREATE INDEX IF NOT EXISTS idx_punches_employee_date
  ON punches (employee_id, date);
```

### 3.2 Idempotency middleware (lightweight)

```ts
// workio/server/src/middleware/idempotency.ts
import { Request, Response, NextFunction } from 'express';
import crypto from 'crypto';
import { Pool } from 'pg';

const pool = new Pool({ connectionString: process.env.DATABASE_URL });
const IDEMPOTENCY_TTL_MS = 24 * 60 * 60 * 1000; // 24h

export async function idempotencyGuard(
  req: Request,
  res: Response,
  next: NextFunction
) {
  const key = req.headers['idempotency-key'] as string;
  if (!key) {
    return res.status(400).json({ error: 'idempotency-key header required' });
  }

  const hash = crypto.createHash('sha256').update(key).digest('hex');
  const now = Date.now();

  // Try to fetch previous response
  const { rows } = await pool.query(
    `SELECT response_body, created_at FROM idempotency_keys WHERE key_hash = $1`,
    [hash]
  );

  if (rows.length > 0) {
    const record = rows[0];
    // If within TTL, replay previous response
    if (now - new Date(record.created_at).getTime() < IDEMPOTENCY_TTL_MS) {
      return res.json(JSON.parse(record.response_body));
    }
  }

  // Attach key to request for later storage
  (req as any).idempotencyKeyHash = hash;
  next();
}

export async function storeIdempotentResponse(
  req: Request,
  res: Response,
  next: NextFunction
) {
  const hash = (req as any).idempotencyKeyHash;
  if (!hash) return next();

  const originalSend = res.json;
  // eslint-disable-next-line @typescript-eslint/ban-ts-comment
  // @ts-ignore
  res.json = function (body) {
    const result = originalSend.call(this, body);
    // Best-effort store; don't fail request if storage fails
    pool.query(
      `INSERT INTO idempotency_keys (key_hash, response_body, created_at)
       VALUES ($1, $2, NOW())
       ON CONFLICT (key_hash) DO UPDATE
       SET response_body = EXCLUDED.response_body, created_at = NOW()`,
      [hash, JSON.stringify(body)]
    ).catch(() => {});
    return result;
  };
  next();
}
```

Create idempotency_keys table (run once):

```sql
CREATE TABLE IF NOT EXISTS idempotency_keys (
  key_hash CHAR(64) PRIMARY KEY,
  response_body JSONB NOT NULL,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
```

### 3.3 Punch route with conflict checks and idempotency

```ts
// workio/server/src/routes/punch.ts
import { Router, Request, Response } from 'express';
import { Pool } from 'pg';
import { idempotencyGuard, storeIdempotentResponse } from '../middleware/idempotency';

const router = Router();
const pool = new Pool({ connectionString: process.env.DATABASE_URL });

router.post(
  '/',
  idempotencyGuard,
  async (req: Request, res: Response) => {
    const { employee_id, punch_type, date, timestamp, location } = req.body;

    if (!employee_id || !punch_type || !date || !timestamp) {
      return res.status(400).json({ error: 'Missing required fields' });
    }

    const client = await pool.connect();
    try {
      await client.query('BEGIN');

      // Conflict check: prevent overlapping punches on same date
      // For clock-in, ensure no existing clock-in without clock-out
      if (punch_type === 'clock_in') {
        const { rows } = await client.query(
          `SELECT id FROM punches
           WHERE employee_id = $1 AND date = $2
           AND punch_type = 'clock_in'
           ORDER BY timestamp DESC LIMIT 1`,
          [employee_id, date]
        );
        if (rows.length > 0) {
          // Optional: allow if previous has matching clock_out
          const { rows: outs } = await client.query(
            `SELECT id FROM punches
             WHERE employee_id = $1 AND date = $2
             AND punch_type = 'clock_out'
             ORDER BY timestamp DESC LIMIT 1`,
            [employee_id, date]
          );
          if (rows.length > (outs.length || 0)) {
            await client.query('ROLLBACK');
            return res.status(409).json({ error: 'Already clocked in for this date' });
          }
        }
      }

      // Insert punch (uniqueness constraint will catch duplicates)
      const { rows: inserted } = await client.query(
        `INSERT INTO punches (employee_id, punch_type, date, timestamp, location)
         VALUES ($1, $2, $3, $4, $5)
         RETURNING *`,
        [employee_id, punch_type, date, timestamp, location || null]
      );

      await client.query('COMMIT');
      (req as any).punchResult = inserted[0];
    } catch (err: any) {
      await client.query('ROLLBACK');

      // Duplicate detected by unique constraint
      if (err.code === '23505') {
        // Fetch existing to return consistent response
        const { rows } = await pool.query(
          `SELECT * FROM punches
           WHERE employee_id = $1 AND punch_type = $2 AND date = $3`,
          [employee_id, punch_type, date]
        );
        return res.status(200).json({ duplicate: true, punch: rows[0] });
      }

      return res.status(500).json({ error: 'Failed to record punch', details: err.message });
    } finally {
      client.release();
    }
  },
  storeIdempotentResponse
);

export default router;
```
