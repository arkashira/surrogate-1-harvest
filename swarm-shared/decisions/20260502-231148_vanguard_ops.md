# vanguard / ops

## 1. Diagnosis
- No durable ingestion manifest: every training run re-lists HF repos and re-downloads, causing 429s and wasted bandwidth.
- Training loads via HF API (`load_dataset`/`list_repo_files`) instead of CDN bypass → guaranteed rate limits during data loading.
- No Lightning Studio reuse: scripts create new studios instead of reusing running ones, burning ~80hr/mo quota.
- No idle-stop guard: Lightning idle timeout kills training; no check before `.run()` calls.
- Mac used for heavy compute: local runs of `from_pretrained`/data loading violate Mac=CLI rule and risk rate limits.

## 2. Proposed change
Add a lightweight ops orchestration module at `/opt/axentx/vanguard/ops/train_orchestrator.py` plus a small JSON manifest schema in `/opt/axentx/vanguard/ops/manifests/`. Scope: single-file addition + one-line cron/startup integration. No changes to existing training code required — we inject file list and CDN-only fetches via environment/config.

## 3. Implementation
Create `/opt/axentx/vanguard/ops/train_orchestrator.py`:

```python
#!/usr/bin/env python3
"""
Ops orchestrator for Surrogate-1 training (vanguard).
- Produces durable manifest (date folder -> file list) via single HF API call.
- Reuses running Lightning Studio or starts one (L40S priority).
- Emits run command that uses CDN-only fetches (zero API calls during training).
"""

import json
import os
import subprocess
import sys
import time
from datetime import datetime, timezone
from pathlib import Path

try:
    from lightning_sdk import Client, Machine, Teamspace
except ImportError:
    print("Install lightning-sdk on orchestrator host: pip install lightning-sdk")
    sys.exit(1)

REPO_ROOT = Path(__file__).parent.parent.parent
MANIFEST_DIR = REPO_ROOT / "ops" / "manifests"
MANIFEST_DIR.mkdir(parents=True, exist_ok=True)

HF_DATASET_REPO = os.getenv("HF_DATASET_REPO", "datasets/your-org/surrogate-1")
HF_TOKEN = os.getenv("HF_TOKEN", "")  # only used for listing; CDN fetches require no token
LIGHTNING_TEAMSPACE = os.getenv("LIGHTNING_TEAMSPACE", "default")
STUDIO_NAME = os.getenv("STUDIO_NAME", "vanguard-surrogate1")

from huggingface_hub import HfApi, list_repo_tree

def build_manifest(date_folder: str) -> Path:
    """
    Single API call to list one date folder (non-recursive), then save manifest.
    Avoids recursive list_repo_files and rate limits.
    """
    api = HfApi(token=HF_TOKEN)
    folder_path = date_folder.rstrip("/")
    entries = list_repo_tree(
        repo_id=HF_DATASET_REPO,
        path=folder_path,
        recursive=False,
        token=HF_TOKEN,
    )
    files = [e.path for e in entries if e.type == "file"]
    manifest = {
        "created_utc": datetime.now(timezone.utc).isoformat(),
        "date_folder": folder_path,
        "repo": HF_DATASET_REPO,
        "files": sorted(files),
        "strategy": "cdn_only",
    }
    out = MANIFEST_DIR / f"manifest-{date_folder.replace('/', '_')}.json"
    out.write_text(json.dumps(manifest, indent=2))
    print(f"Manifest written: {out}")
    return out

def find_or_start_studio():
    client = Client()  # uses LIGHTNING_ACCOUNT_ID / API key from env
    ts = Teamspace(name=LIGHTNING_TEAMSPACE, client=client)

    # Reuse running studio
    for s in ts.studios:
        if s.name == STUDIO_NAME and s.status == "running":
            print(f"Reusing running studio: {s.name} (id={s.id})")
            return s

    # Start new (priority: H200 in lightning-lambda-prod, else L40S in public)
    clouds_to_try = [
        ("lightning-lambda-prod", "H200"),
        ("lightning-public-prod", "L40S"),
    ]
    for cloud, machine_name in clouds_to_try:
        try:
            m = Machine(name=machine_name, cloud=cloud)
            studio = ts.studios.create(
                name=STUDIO_NAME,
                machine=m,
                enable_apps=False,
            )
            print(f"Started studio: {studio.name} on {cloud}/{machine_name}")
            return studio
        except Exception as e:
            print(f"Could not start on {cloud}/{machine_name}: {e}")
            continue

    raise RuntimeError("Could not start or reuse any studio.")

def run_training_with_cdn(manifest_path: Path):
    studio = find_or_start_studio()
    if studio.status != "running":
        print(f"Studio not running (status={studio.status}). Restarting...")
        # stop + start to ensure fresh running state
        studio.stop()
        time.sleep(10)
        studio.start(machine=studio.machine or Machine(name="L40S", cloud="lightning-public-prod"))
        # poll until running (simple)
        for _ in range(30):
            studio.refresh()
            if studio.status == "running":
                break
            time.sleep(10)

    # Build run command that uses CDN-only fetches.
    # The training script should read MANIFEST_PATH and use:
    #   base = f"https://huggingface.co/datasets/{repo}/resolve/main/"
    #   for f in manifest["files"]: download(f"{base}{f}")
    env = os.environ.copy()
    env.update({
        "MANIFEST_PATH": str(manifest_path),
        "HF_DATASET_REPO": HF_DATASET_REPO,
        "HF_TOKEN": HF_TOKEN,
    })

    train_script = REPO_ROOT / "train.py"
    if not train_script.exists():
        raise FileNotFoundError(f"train.py not found at {train_script}")

    # Use Lightning's .run to execute training script inside studio
    run = studio.runs.create(
        command=[sys.executable, str(train_script)],
        environment_variables=env,
    )
    print(f"Started run {run.id}")
    return run

def main():
    # Example: use today's date folder; adapt as needed
    date_folder = datetime.utcnow().strftime("batches/mirror-merged/%Y-%m-%d")
    manifest = build_manifest(date_folder)
    run = run_training_with_cdn(manifest)
    print(f"Orchestrated run: {run.id}")

if __name__ == "__main__":
    main()
```

Create a minimal training-side loader snippet to be used by `train.py` (non-invasive):

```python
# train.py (add near top)
import os, json, requests
from pathlib import Path

MANIFEST_PATH = os.getenv("MANIFEST_PATH")
HF_DATASET_REPO = os.getenv("HF_DATASET_REPO", "")

def load_file_list_from_manifest():
    if not MANIFEST_PATH or not Path(MANIFEST_PATH).exists():
        return []
    with open(MANIFEST_PATH) as f:
        m = json.load(f)
    return m.get("files", [])

def cdn_fetch(file_path, local_dir="./data_cache"):
    os.makedirs(local_dir, exist_ok=True)
    url = f"https://huggingface.co/datasets/{HF_DATASET_REPO}/resolve/main/{file_path}"
    out = Path(local_dir) / Path(file_path).name
    if out.exists():
        return out
    resp = requests.get(url, timeout=60)
    resp.raise_for_status()
    out.write_bytes(resp.content)
    return out
```
