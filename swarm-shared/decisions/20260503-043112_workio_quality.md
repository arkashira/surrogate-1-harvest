# workio / quality

## Final Synthesized Solution

### Diagnosis (merged, highest severity first)
1. **Missing `X-Line-Signature` verification** — allows spoofed attendance events and fraud.
2. **No idempotency on `webhookEventId`** — LINE retries create duplicate clock-in/out and leave/OT rows.
3. **Non-atomic clock-state transitions** — concurrent webhook deliveries or retries can corrupt `clock_in`/`clock_out` pairing and produce invalid states (e.g., double clock-in).
4. **No short-term replay defense** — retries after deployment or during incidents create new rows unless deduplicated.
5. **No transactional consistency between state change and LINE reply** — partial failures leave users without confirmation and system state ambiguous.

---

### Proposed Change
File: `/opt/axentx/workio/server/src/routes/webhook/line.ts` (or equivalent).  
Scope:
- Verify `X-Line-Signature` before any processing.
- Enforce idempotency with `webhookEventId` via DB unique constraint + short TTL cache.
- Use atomic upsert and state-machine rules for clock-in/out (prevent invalid transitions).
- Wrap per-event processing in a transaction that includes idempotency marking and attendance update.
- Reply to LINE only after commit; return 5xx on failure so LINE retries, but idempotency prevents duplicates.

---

```ts
// /opt/axentx/workio/server/src/routes/webhook/line.ts
import express, { Request, Response } from 'express';
import crypto from 'crypto';
import { db } from '../../db'; // adjust to your db client
import { Row } from 'pg';

const router = express.Router();

// -- Signature verification --
function verifyLineSignature(rawBody: Buffer, signature: string, channelSecret: string): boolean {
  const expected = crypto
    .createHmac('sha256', channelSecret)
    .update(rawBody)
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

// -- Idempotency (DB-backed, optional Redis fast path) --
// Table: line_webhook_events(event_id TEXT PRIMARY KEY, created_at TIMESTAMPTZ, seen_at TIMESTAMPTZ)
async function markEventSeen(eventId: string, client = db): Promise<boolean> {
  // Returns true if inserted (first time), false if duplicate
  const result = await client.query(
    `INSERT INTO line_webhook_events (event_id, created_at)
     VALUES ($1, NOW())
     ON CONFLICT (event_id) DO UPDATE SET seen_at = NOW()
     RETURNING (xmax = 0) AS inserted`,
    [eventId]
  );
  return !!result.rows[0]?.inserted;
}

// -- State-safe clock operations --
// Rules:
// - Only one open clock-in (clock_out IS NULL) per employee.
// - clock_in must precede clock_out.
// - Prevent double clock-in without intervening clock-out.
async function upsertAttendanceClock(
  employeeId: number,
  clockInAt: Date | null,
  clockOutAt: Date | null,
  location: any,
  client = db
) {
  if (clockOutAt) {
    // Close latest open clock-in
    const result = await client.query(
      `UPDATE attendance
       SET clock_out = $1, updated_at = NOW(), location_out = $2
       WHERE employee_id = $3 AND clock_out IS NULL
       ORDER BY clock_in DESC
       LIMIT 1
       RETURNING id`,
      [clockOutAt, JSON.stringify(location), employeeId]
    );
    // If no open clock-in found, do not create phantom rows; caller may log/alert
    return !!result.rows.length;
  }

  if (!clockInAt) return false;

  // Ensure no existing open clock-in (defensive; unique partial index recommended)
  const open = await client.query(
    `SELECT id FROM attendance
     WHERE employee_id = $1 AND clock_out IS NULL
     ORDER BY clock_in DESC
     LIMIT 1`,
    [employeeId]
  );
  if (open.rows.length > 0) {
    // Invalid transition: already clocked in. Optionally auto-close stale record or reject.
    // Here we reject to avoid ambiguous state.
    throw new Error('Employee already has an open clock-in');
  }

  await client.query(
    `INSERT INTO attendance (employee_id, clock_in, location_in)
     VALUES ($1, $2, $3)`,
    [employeeId, clockInAt, JSON.stringify(location)]
  );
  return true;
}

// -- Leave/OT helpers (idempotent upserts) --
// Example: one active leave request per employee overlapping dates is prevented by app/db constraints.
// Use similar event-idempotent pattern and transactions.

// -- Route handler --
router.post('/line', async (req: Request, res: Response) => {
  const channelSecret = process.env.LINE_CHANNEL_SECRET || '';
  const signature = req.headers['x-line-signature'] as string;
  // Expect raw body preserved by express.raw() upstream
  const rawBody = (req as any).rawBody as Buffer | undefined;

  if (!rawBody || !Buffer.isBuffer(rawBody)) {
    return res.status(400).json({ error: 'Raw body required for LINE signature verification' });
  }

  if (!signature || !verifyLineSignature(rawBody, signature, channelSecret)) {
    return res.status(401).json({ error: 'Invalid LINE signature' });
  }

  let body;
  try {
    body = JSON.parse(rawBody.toString('utf8'));
  } catch (err) {
    return res.status(400).json({ error: 'Invalid JSON' });
  }

  const events = body.events || [];
  const client = await db.connect();

  try {
    await client.query('BEGIN');

    for (const ev of events) {
      const eventId = ev.webhookEventId;
      if (!eventId) continue;

      // Idempotency check within transaction
      const isNew = await markEventSeen(eventId, client);
      if (!isNew) continue; // skip duplicate

      const source = ev.source || {};
      const lineUserId = source.userId;
      if (!lineUserId) continue;

      const empResult = await client.query(
        `SELECT id FROM employees WHERE line_user_id = $1 LIMIT 1`,
        [lineUserId]
      );
      if (!empResult.rows.length) continue;
      const empId = empResult.rows[0].id;

      const type = ev.type;
      const now = new Date();

      if (type === 'message' && ev.message?.type === 'text') {
        const text = (ev.message.text || '').toLowerCase();
        const location = {
          latitude: ev?.location?.latitude || null,
          longitude: ev?.location?.longitude || null,
        };

        if (text.includes('clock in') || text.includes('เข้างาน')) {
          await upsertAttendanceClock(empId, now, null, location, client);
        } else if (text.includes('clock out') || text.includes('เลิกงาน')) {
          await upsertAttendanceClock(empId, null, now, location, client);
        }
      }

      // Extend similarly for leave/OT events with idempotent upserts
    }

    await client.query('COMMIT');
    return res.status(200).json({ ok: true });
  } catch (err) {
    await client.query('ROLLBACK').catch(() => {});
    console.error('LINE webhook processing error:', err);
    // 5xx triggers LINE retry; idempotency prevents duplicates
    return res.status(500).json({ error: 'Processing failed' });
  } finally {
    client.release();
  }
});

export default router;
```
