## Code Snippets

### `surrogate/training/cdn_loader.py`
```python
import aiohttp
import asyncio
import json
import os
from pathlib import Path
from huggingface_hub import list_repo_tree

HF_REPO = "axentx/surrogate-dataset-mirror"
DATE_FOLDER = "2026-04-29"  # parameterized in prod

async def download_file(session, url, dest):
    async with session.get(url) as resp:
        resp.raise_for_status()
        dest.parent.mkdir(parents=True, exist_ok=True)
        with open(dest, "wb") as f:
            while chunk := await resp.content.read(64 * 1024):
                f.write(chunk)

async def cdn_bypass_loader(file_list_path: str, output_dir: str, concurrency: int = 8):
    with open(file_list_path) as f:
        files = json.load(f)

    connector = aiohttp.TCPConnector(limit=concurrency)
    async with aiohttp.ClientSession(connector=connector) as session:
        tasks = []
        for entry in files:
            path = entry["path"]
            cdn_url = f"https://huggingface.co/datasets/{HF_REPO}/resolve/main/{path}"
            dest = Path(output_dir) / Path(path).name
            tasks.append(download_file(session, cdn_url, dest))
        await asyncio.gather(*tasks)

def prepare_file_list():
    """Single API call from Mac (run once per date folder)."""
    tree = list_repo_tree(HF_REPO, path=DATE_FOLDER, recursive=False)
    files = [{"path": f.path} for f in tree if f.type == "file"]
    out = Path("file_list.json")
    out.write_text(json.dumps(files, indent=2))
    return out

if __name__ == "__main__":
    fl = prepare_file_list()
    asyncio.run(cdn_bypass_loader(str(fl), "downloads/"))
```

### `surrogate/training/studio_pool.py`
```python
from lightning import Studio, L40S, Teamspace
import time

STUDIO_NAME = "surrogate-train-l40s"

def get_or_create_studio():
    # Reuse running studio
    for s in Teamspace.studios:
        if s.name == STUDIO_NAME and s.status == "running":
            print(f"Reusing running studio: {s.id}")
            return s

    # Restart stopped studio
    for s in Teamspace.studios:
        if s.name == STUDIO_NAME and s.status == "stopped":
            print(f"Restarting stopped studio: {s.id}")
            s.start(machine=L40S)
            wait_running(s)
            return s

    # Create new
    print("Creating new studio")
    return Studio(
        name=STUDIO_NAME,
        machine=L40S,
        create_ok=True
    )

def wait_running(studio, timeout=300):
    for _ in range(timeout):
        if studio.status() == "running":
            return
        time.sleep(1)
    raise TimeoutError("Studio failed to start")
```

### `surrogate/training/orchestrator.py` (excerpt)
```python
import os
from .cdn_loader import prepare_file_list
from .studio_pool import get_or_create_studio

def run_training():
    # 1. Prepare file list (Mac orchestration)
    fl = prepare_file_list()
    os.environ["TRAIN_FILE_LIST"] = str(fl)

    # 2. Get studio
    studio = get_or_create_studio()

    # 3. Run training (Lightning SDK)
    job = studio.run(
        "train.py",
        arguments=["--file-list", str(fl)],
        framework="pytorch"
    )
    return job
```
