# workio / frontend

## 1. Diagnosis
- Clock-in/out UI allows rapid double-taps → duplicate events and confusing state (optimistic UI missing).
- No local pending state for offline/weak-network clock events → users don’t know if tap succeeded.
- GPS permission flow is abrupt and can fail silently; no graceful fallback or clear guidance.
- Line webhook errors (e.g., 429/5xx) are not surfaced to the user; they see no feedback after tapping.
- Missing visual distinction between “ready to clock” vs “already clocked today” states on the dashboard.

## 2. Proposed change
File: `workio/src/components/ClockButton.tsx` (create) and integrate into `Dashboard.tsx`.  
Scope: single component + small store hook (`useClock`) to manage optimistic state, debouncing, and error feedback.

import { useState, useCallback } from 'react';
import { useToast } from '@/hooks/use-toast';

type ClockState = 'idle' | 'pending' | 'success' | 'error';

export function useClock() {
  const [state, setState] = useState<ClockState>('idle');
  const { toast } = useToast();

  const clock = useCallback(async (type: 'in' | 'out', location?: GeolocationCoordinates) => {
    if (state === 'pending') return;
    setState('pending');

    try {
      const res = await fetch('/api/clock', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ type, location, ts: Date.now() }),
      });

      if (!res.ok) {
        const msg = await res.text().catch(() => res.statusText);
        throw new Error(msg || `Failed to clock ${type}`);
      }

      setState('success');
      toast({ title: `Clocked ${type}`, description: new Date().toLocaleTimeString() });
      return await res.json();
    } catch (err: any) {
      setState('error');
      toast({ variant: 'destructive', title: 'Clock failed', description: err.message });
      throw err;
    } finally {
      // reset after brief period to allow re-click
      setTimeout(() => setState('idle'), 2000);
    }
  }, [state, toast]);

  return { state, clock };
}
EOF

cat > workio/src/components/ClockButton.tsx << 'EOF'
import { useClock } from '@/hooks/useClock';
import { Button } from '@/components/ui/button';
import { Loader2, Check, AlertCircle } from 'lucide-react';
import { useState } from 'react';

export function ClockButton() {
  const { state, clock } = useClock();
  const [isClockedToday, setIsClockedToday] = useState(false); // derive from dashboard data

  const handleClock = async (type: 'in' | 'out') => {
    // quick optimistic feedback handled by useClock/toast
    try {
      // optional: request geolocation once and attach
      const pos = await getCurrentPosition().catch(() => null);
      await clock(type, pos?.coords);
      if (type === 'in') setIsClockedToday(true);
      else setIsClockedToday(false);
    } catch {
      // error already toasted
    }
  };

  const getLabel = () => {
    if (state === 'pending') return 'กำลังบันทึก...';
    if (state === 'success') return 'สำเร็จ';
    return isClockedToday ? 'Clock Out' : 'Clock In';
  };

  return (
    <div className="flex flex-col items-center gap-3">
      <Button
        size="lg"
        className="w-48 text-lg"
        disabled={state === 'pending'}
        onClick={() => handleClock(isClockedToday ? 'out' : 'in')}
      >
        {state === 'pending' && <Loader2 className="mr-2 h-4 w-4 animate-spin" />}
        {state === 'success' && <Check className="mr-2 h-4 w-4" />}
        {state === 'error' && <AlertCircle className="mr-2 h-4 w-4" />}
        {getLabel()}
      </Button>

      {isClockedToday && state === 'idle' && (
        <span className="text-sm text-muted-foreground">
          คุณลงเวลาเข้าแล้ววันนี้
        </span>
      )}
    </div>
  );
}

// small helper to wrap geolocation in a promise
function getCurrentPosition(): Promise<GeolocationPosition> {
  return new Promise((resolve, reject) => {
    if (!navigator.geolocation) {
      reject(new Error('Geolocation not supported'));
      return;
    }
    navigator.geolocation.getCurrentPosition(resolve, reject, {
      enableHighAccuracy: false,
      timeout: 10000,
      maximumAge: 30000,
    });
  });
}

## 4. Verification
1. Start frontend: `cd workio && npm run dev`
2. Open dashboard (or App) and tap “Clock In” — button should show loading, then success toast, then display “คุณลงเวลาเข้าแล้ววันนี้”.
3. Tap again quickly — second tap should be ignored while pending.
4. Toggle network to offline (or block `/api/clock` in devtools) and tap — error toast should appear and button should return to idle after 2s.
5. After success, button label should switch to “Clock Out” (if derived state wired to real data).
