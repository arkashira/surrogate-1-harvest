## Final Synthesis & Actionable Implementation

I combined the strongest elements from both proposals (identical diagnoses, manifest caching, CDN bypass, Lightning Studio reuse, idle-stop guard, deterministic routing) and resolved all contradictions in favor of correctness and concrete actionability.

### 1. Diagnosis (Consolidated)
- **No durable ingestion manifest** → every training run re-lists HF repos and re-downloads, causing 429s and wasted time/bandwidth.
- **Training loads via HF API** (`load_dataset`/`list_repo_files`) instead of CDN bypass → guaranteed rate limits during data loading.
- **No Lightning Studio reuse logic** → each run recreates studio and burns 80+ hrs/month quota on startup churn.
- **No idle-stop guard before `.run()`** → Lightning idle timeout kills training mid-run and wastes paid compute.
- **Missing deterministic repo routing for commits** → single repo hits HF 128/hr commit cap and blocks ingestion pipeline.

HF_CDN_BASE = "https://huggingface.co/datasets"

def _backoff(attempt: int) -> None:
    # After 429: wait 360s before retry (per pattern)
    t = 30 * (2 ** min(attempt, 5))
    if attempt >= 3:
        t = max(t, 360)
    time.sleep(t)

def _list_tree_safe(repo: str, path: str = "", attempts: int = 0) -> List[Dict]:
    try:
        out = list_repo_tree(repo=repo, path=path, recursive=False)
        return list(out) if hasattr(out, "__iter__") else []
    except Exception as e:
        if attempts >= 5:
            raise
        if getattr(e, "response", None) and e.response.status_code == 429:
            _backoff(attempts)
            return _list_tree_safe(repo, path, attempts + 1)
        raise

def get_or_build_manifest(repo: str, date: str) -> List[str]:
    """
    Return list of file paths for repo/date (YYYY-MM-DD).
    Caches to state/ingest_manifest_{repo_hash}_{date}.json
    """
    repo_slug = repo.replace("/", "_")
    cache_path = STATE_DIR / f"ingest_manifest_{repo_slug}_{date}.json"
    if cache_path.exists():
        return json.loads(cache_path.read_text())

    # List top-level date folder only (avoids recursive pagination)
    items = _list_tree_safe(repo, path=date, attempts=0)
    files = []
    for item in items:
        if item.get("type") == "file":
            files.append(f"{date}/{item['path']}")
        elif item.get("type") == "folder":
            sub = _list_tree_safe(repo, path=f"{date}/{item['path']}", attempts=0)
            for s in sub:
                if s.get("type") == "file":
                    files.append(f"{date}/{item['path']}/{s['path']}")

    cache_path.write_text(json.dumps(files, indent=2))
    return files

def iter_cdn_urls(repo: str, date: str) -> Iterator[str]:
    """Yield public CDN URLs (no auth) for cached file list."""
    for fpath in get_or_build_manifest(repo, date):
        yield f"{HF_CDN_BASE}/{repo}/resolve/main/{fpath}"

def download_via_cdn(repo: str, date: str, target_dir: str) -> List[str]:
    """Download all files for date via CDN (zero HF API calls)."""
    os.makedirs(target_dir, exist_ok=True)
    saved = []
    for url in iter_cdn_urls(repo, date):
        fname = url.split("/resolve/main/")[-1].replace("/", "_")
        out = os.path.join(target_dir, fname)
        if os.path.exists(out):
            saved.append(out)
            continue
        r = requests.get(url, timeout=60)
        r.raise_for_status()
        with open(out, "wb") as fh:
            fh.write(r.content)
        saved.append(out)
    return saved

def find_running_studio(name: str, client=None):
    """Reuse running Studio to save quota."""
    if LightningCLI is None:
        return None
    from lightning.app import Teamspace
    for s in Teamspace.studios:
        if s.name == name and getattr(s, "status", "").lower() == "running":
            return s
    return None

def ensure_studio_running(name: str, machine: str = "L40S", client=None):
    """Start or reuse a Studio and return handle."""
    existing = find_running_studio(name)
    if existing:
        return existing
    from lightning.app import Studio
    return Studio(name=name, create_ok=True, machine=machine)

def idle_stop_guard(timeout_min: int = 30):
    """
    Prevent Lightning idle timeout from killing training mid-run.
    Call this before .run() or at start of training loop.
    """
    import threading
    import lightning as L

    def _keep_alive():
        while True:
            time.sleep(timeout_min * 60)
            L.LightningApp.get_running_app()._update_last_active_time()

    t = threading.Thread(target=_keep_alive, daemon=True)
    t.start()

#### B. CLI wrapper
Create `/opt/axentx/vanguard/ops/run_ingest.py`:

```python
#!/usr/bin/env python3
import argparse, sys
from pathlib import Path
sys.path.insert(0, str(Path(__file__).parent.parent))

from ops.ingest_state import get_or_build_manifest, download_via_cdn

def main():
    p = argparse.ArgumentParser()
    p.add_argument("--repo", required=True)
    p.add_argument("--date", required=True)
    p.add_argument("--download-to", default=None)
    args = p.parse_args()

    files = get_or_build_manifest(args.repo, args.date)
    print(f"Found {len(files)} files")
    if args.download_to:
        saved = download_via_cdn(args.repo, args.date, args.download_to)
        print(f"Downloaded {len(saved)} files to {args.download_to}")

if __name__ == "__main__":
    main()
```
