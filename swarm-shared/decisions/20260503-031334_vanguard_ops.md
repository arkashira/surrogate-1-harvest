# vanguard / ops

## 1. Diagnosis

- No CDN-first manifest exists; ingestion/training scripts likely still call HF API (`list_repo_tree`, `load_dataset`) at runtime, risking 429s and non-reproducible runs.
- Missing content-addressed file list keyed by date/slug; training jobs cannot run deterministically from a frozen snapshot.
- No guardrails to enforce Mac-only orchestration + remote-only compute; risk of accidental local model loading or training on laptop.
- No studio reuse logic; each run may create new Lightning Studio and burn quota instead of reusing running instances.
- No HF CDN bypass path in data loader; every epoch re-authenticates against API instead of using public CDN URLs.

## 2. Proposed change

Create `/opt/axentx/vanguard/ops/guardrails.py` (single file, ~120 lines) that:
- Generates a deterministic CDN manifest for a given `{date}` folder in a HF dataset repo.
- Enforces remote-only compute policy (Mac → Lightning/Kaggle/Colab only).
- Reuses running Lightning Studio by name or creates with `L40S` fallback.
- Exports a `run_surrogate_training(date, repo, hf_token=None)` that uses CDN-only data loading.

## 3. Implementation

# /opt/axentx/vanguard/ops/guardrails.py
#!/usr/bin/env python3
"""
Guardrails for Surrogate-1 training (ops).
- CDN-first manifest generation (no HF API during training)
- Remote-only compute enforcement
- Lightning Studio reuse
"""
import os
import json
import hashlib
import datetime
from pathlib import Path
from typing import List, Dict, Optional

try:
    from huggingface_hub import list_repo_tree, hf_hub_download, HfApi
except ImportError:
    HfApi = None  # soft fail for ops checks

# ---- Policy: Mac = orchestration only ----
def enforce_remote_compute_policy() -> None:
    """Raise if running training-grade workloads locally."""
    import platform
    system = platform.system()
    if system == "Darwin":
        # Mac allowed only for orchestration scripts; block direct training imports
        import traceback
        tb = "".join(traceback.format_stack())
        if "surrogate_train" in tb or "train" in tb:
            raise RuntimeError(
                "Mac policy violation: training must run on remote (Lightning/Kaggle/Colab). "
                "Mac is for orchestration only."
            )

# ---- CDN manifest ----
def build_cdn_manifest(repo: str, date: str, hf_token: Optional[str] = None) -> Dict:
    """
    Build content-addressed manifest for repo/{date}/*.parquet.
    Uses single API call (list_repo_tree) and produces CDN URLs.
    """
    if HfApi is None:
        raise RuntimeError("huggingface_hub not installed")

    api = HfApi(token=hf_token)
    # Non-recursive per date folder to minimize pagination/API usage
    tree = list_repo_tree(repo=repo, path=date, recursive=False, token=hf_token)
    files = [item.rfilename for item in tree if item.rfilename.endswith(".parquet")]
    files = sorted(files)

    manifest = {
        "repo": repo,
        "date": date,
        "generated_at_utc": datetime.datetime.utcnow().isoformat() + "Z",
        "snapshot_id": hashlib.sha256(f"{repo}:{date}:{','.join(files)}".encode()).hexdigest()[:16],
        "files": [],
    }

    for f in files:
        slug = Path(f).stem  # e.g., batches/mirror-merged/2026-04-27/abc123 -> abc123
        cdn_url = f"https://huggingface.co/datasets/{repo}/resolve/main/{f}"
        manifest["files"].append(
            {
                "path": f,
                "slug": slug,
                "cdn_url": cdn_url,
                "size_hint": None,  # could HEAD if needed, but avoid extra calls
            }
        )
    return manifest

def save_manifest(manifest: Dict, out_dir: str = "manifests") -> str:
    out_path = Path(out_dir) / f"{manifest['date']}_{manifest['snapshot_id']}.json"
    out_path.parent.mkdir(parents=True, exist_ok=True)
    out_path.write_text(json.dumps(manifest, indent=2))
    return str(out_path)

# ---- Lightning Studio reuse ----
try:
    from lightning import Studio, L40S, Machine

    def get_or_create_studio(name: str, machine: Machine = L40S) -> Studio:
        """Reuse running studio or create new one (L40S preferred)."""
        from lightning import Teamspace

        for s in Teamspace.studios:
            if s.name == name and s.status == "Running":
                return s
        # If stopped, restart; if not exist, create
        for s in Teamspace.studios:
            if s.name == name and s.status == "Stopped":
                s.start(machine=machine)
                return s
        return Studio(
            name=name,
            machine=machine,
            create_ok=True,
        )
except Exception:
    # Fallback for environments without lightning SDK
    def get_or_create_studio(name: str, machine=None):
        raise RuntimeError("Lightning SDK not available; cannot manage Studio.")

# ---- Training launcher ----
def run_surrogate_training(
    date: str,
    repo: str = "axentx/surrogate-1-mirror",
    hf_token: Optional[str] = None,
    studio_name: str = "surrogate-1-train",
) -> Dict:
    """
    End-to-end guardrail-enabled training run:
    1) Build CDN manifest (single API call)
    2) Save manifest
    3) Reuse/create Lightning Studio
    4) Launch training script with manifest path (CDN-only loader)
    """
    enforce_remote_compute_policy()

    manifest = build_cdn_manifest(repo=repo, date=date, hf_token=hf_token)
    manifest_path = save_manifest(manifest)

    # Studio reuse
    studio = get_or_create_studio(studio_name, machine=L40S)

    # Launch training (example command; adapt to your train.py)
    # train.py must accept --manifest and use CDN-only loading (no HF API calls).
    cmd = [
        "python",
        "train.py",
        "--manifest",
        manifest_path,
        "--repo",
        repo,
        "--date",
        date,
    ]

    # If running locally as orchestrator, submit to studio
    run = studio.run(
        cmd,
        cwd="/opt/axentx/vanguard",
        name=f"surrogate-train-{date}",
    )
    return {
        "manifest": manifest,
        "manifest_path": manifest_path,
        "studio_name": studio.name,
        "run_id": run.id if run else None,
    }

# ---- CLI ----
if __name__ == "__main__":
    import argparse

    parser = argparse.ArgumentParser(description="Guardrails for Surrogate-1 training")
    parser.add_argument("--date", required=True, help="Date folder (YYYY-MM-DD)")
    parser.add_argument("--repo", default="axentx/surrogate-1-mirror")
    parser.add_argument("--hf-token", default=None)
    parser.add_argument("--studio", default="surrogate-1-train")
    parser.add_argument("--dry-run", action="store_true", help="Only build manifest")

    args = parser.parse_args()

    if args.dry_run:
        m = build_cdn_manifest(repo=args.repo, date=args.date, hf_token=args.hf_token)
        p = save_manifest(m)
        print(json.dumps(m, indent=2))
        print(f"Manifest saved: {p}")
    else:
        result = run_surrogate_training(
            date=args.date,
            repo=args.repo,
            hf_token=args.hf_token,
            studio_name=args.studio,
        )
        print(json.dumps(result, indent=2))
