# vanguard / ops

## 1. Diagnosis
- No persisted `(repo, dateFolder) → file-list` manifest exists; training/data-selection re-runs authenticated `list_repo_tree` on every launch, burning HF API quota and causing 429s.
- Training scripts likely use `load_dataset(streaming=True)` or repeated HF API calls during data loading, risking mixed-schema CastErrors and rate limits.
- No CDN-only data path; authenticated API is used for file access instead of public CDN URLs (`https://huggingface.co/datasets/{repo}/resolve/main/{path}`) which bypass auth and rate limits.
- No guard to reuse existing Lightning Studio; training likely recreates studios and wastes 80hr/mo quota.
- No idle-stop handling; Lightning Studio timeouts kill long-running training jobs without restart logic.

## 2. Proposed change
Create `/opt/axentx/vanguard/ops/prepare_hf_snapshot.py` and update training launcher to:
- Accept `repo` and `date_folder` args.
- Call `list_repo_tree(recursive=False)` once per date folder and persist `snapshot-{repo}-{date}.json` with `{ "repo": "...", "date": "...", "files": ["path1", "path2", ...], "generated_at": "..." }`.
- Emit a companion `train.py` that reads the snapshot and downloads files via CDN URLs only (zero HF API calls during training).
- Add a small wrapper to reuse existing Lightning Studio or create one if missing, and check status before `.run()`.

Scope: single new file + optional launcher tweak (no existing training code modified yet).

#!/usr/bin/env bash
# Generates a CDN-only file manifest for a HuggingFace dataset repo/date folder.
# Usage:
#   ./prepare_hf_snapshot.py <repo> <date_folder> [--out-dir ./snapshots]
#
# Requires: python3 + huggingface_hf (huggingface_hub)
# Note: Run from Mac (or any orchestration host). Do NOT run heavy compute here.

set -euo pipefail

REPO="${1:-}"
DATE_FOLDER="${2:-}"
OUT_DIR="${3:-./snapshots}"

if [[ -z "$REPO" || -z "$DATE_FOLDER" ]]; then
  echo "Usage: $0 <repo> <date_folder> [out_dir]"
  exit 1
fi

mkdir -p "$OUT_DIR"

python3 - "$REPO" "$DATE_FOLDER" "$OUT_DIR" <<'PY'
import json
import os
import sys
from datetime import datetime, timezone

from huggingface_hub import HfApi, list_repo_tree

def main(repo: str, date_folder: str, out_dir: str) -> None:
    api = HfApi()

    # Single API call: list top-level of the date folder (non-recursive)
    # If files are nested deeper, adjust path accordingly.
    folder_path = f"{date_folder}"
    entries = list_repo_tree(repo=repo, path=folder_path, recursive=False)

    files = []
    for entry in entries:
        # Only include files (skip subdirs for now)
        if entry.type == "file":
            # path is relative to repo root
            files.append(entry.path)

    snapshot = {
        "repo": repo,
        "date": date_folder,
        "files": sorted(files),
        "generated_at": datetime.now(timezone.utc).isoformat(),
        "note": "CDN-only manifest. Use https://huggingface.co/datasets/{repo}/resolve/main/{file} for downloads."
    }

    out_path = os.path.join(out_dir, f"snapshot-{repo.replace('/', '-')}-{date_folder}.json")
    os.makedirs(out_dir, exist_ok=True)
    with open(out_path, "w", encoding="utf-8") as f:
        json.dump(snapshot, f, indent=2)

    print(f"Snapshot written: {out_path}")
    print(f"Total files: {len(files)}")

if __name__ == "__main__":
    if len(sys.argv) != 4:
        print("Usage: python prepare_hf_snapshot.py <repo> <date_folder> <out_dir>")
        sys.exit(1)
    main(sys.argv[1], sys.argv[2], sys.argv[3])
PY

class CDNTextDataset(Dataset):
    def __init__(self, repo: str, files: List[str], cache_dir: str = "./.cdn_cache"):
        self.repo = repo
        self.files = files
        self.cache_dir = Path(cache_dir)
        self.cache_dir.mkdir(parents=True, exist_ok=True)

        # Lightweight projection: expect line-delimited JSON with {prompt,response}
        # If schema varies, project at parse time (do not rely on HF dataset loader).
        self.items = self._build_index()

    def _download_file(self, file_path: str) -> Path:
        out = self.cache_dir / file_path.replace("/", "_")
        if out.is_file():
            return out
        url = HF_CDN_TEMPLATE.format(repo=self.repo, file=file_path)
        out.parent.mkdir(parents=True, exist_ok=True)
        resp = requests.get(url, timeout=60)
        resp.raise_for_status()
        out.write_bytes(resp.content)
        return out

    def _build_index(self):
        items = []
        for f in self.files:
            local_path = self._download_file(f)

                print(f"Skipping {f} due to parse error: {exc}")
        return items
