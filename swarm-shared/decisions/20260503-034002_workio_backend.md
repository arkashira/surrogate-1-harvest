# workio / backend

## 1. Diagnosis

- **Idempotency missing**: LINE retries on 5xx/timeouts can replay webhooks and corrupt clock-in/out or leave/OT state (duplicate punches, double approvals).
- **Signature verification absent**: Webhook handler does not verify `X-Line-Signature`, allowing spoofed events to mutate state.
- **No transactional boundary**: Business operations (clock/leave/OT) perform partial writes without rollback on downstream failure, risking divergence.
- **Tenant isolation weak**: Webhook path does not enforce tenant context before DB writes, enabling cross-tenant bleed via shared channel IDs.
- **Frontend staleness**: LINE-driven DB updates are not pushed to clients; UI shows stale state until manual refresh.

## 2. Proposed change

File: `/opt/axentx/workio/server/src/routes/webhook/line.ts` (create if missing) or update existing handler at `/opt/axentx/workio/server/src/routes/webhook/line.ts` (or equivalent).  
Scope: add middleware + handler that (a) verifies signature, (b) enforces idempotency via `X-Line-Webhook-Id` (or event-level dedupe), (c) wraps business logic in a transaction, (d) emits a tenant-scoped event for frontend sync.

// /opt/axentx/workio/server/src/routes/webhook/line.ts
import { Router, Request, Response, NextFunction } from 'express';
import crypto from 'crypto';
import { Pool } from 'pg';
import { Server } from 'socket.io';

const router = Router();
const pool = new Pool({ connectionString: process.env.DATABASE_URL });

// Shared idempotency window (e.g., 24h)
const IDEMPOTENCY_TTL_MS = 24 * 60 * 60 * 1000;

function verifyLineSignature(req: Request, res: Response, next: NextFunction) {
  const signature = req.headers['x-line-signature'] as string;
  const body = JSON.stringify(req.body);
  const channelSecret = process.env.LINE_CHANNEL_SECRET;

  if (!channelSecret) {
    return res.status(500).json({ error: 'LINE_CHANNEL_SECRET not configured' });
  }

  const expected = crypto
    .createHmac('sha256', channelSecret)
    .update(body, 'utf8')
    .digest('base64');

  if (!crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected))) {
    return res.status(401).json({ error: 'Invalid signature' });
  }
  next();
}

async function dedupeEvent(event: any, client: PoolClient): Promise<boolean> {
  // Use LINE webhook id or event id + type + timestamp as dedupe key
  const key = event.webhookId || `${event.type}:${event.source.userId}:${event.timestamp}`;
  const now = Date.now();

  const { rows } = await client.query(
    `SELECT 1 FROM webhook_events
     WHERE dedupe_key = $1 AND received_at >= to_timestamp($2 / 1000.0)
     LIMIT 1`,
    [key, (now - IDEMPOTENCY_TTL_MS) / 1000]
  );

  if (rows.length > 0) return false;

  await client.query(
    `INSERT INTO webhook_events (dedupe_key, event_type, payload, received_at)
     VALUES ($1, $2, $3, to_timestamp($4 / 1000.0))`,
    [key, event.type, event, now / 1000]
  );
  return true;
}

async function handleLineEvent(req: Request, res: Response) {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    const events = req.body.events || [];
    for (const event of events) {
      // enforce tenant via channelId -> tenant mapping (cached)
      const tenantId = await resolveTenantId(event.source.channelId, client);
      if (!tenantId) {
        // log and skip; do not ack unknown tenant
        continue;
      }

      const isNew = await dedupeEvent(event, client);
      if (!isNew) continue;

      // Dispatch to domain handlers with tenant isolation
      await dispatchToDomain(event, tenantId, client);
    }

    await client.query('COMMIT');
    res.status(200).json({ status: 'ok' });
  } catch (err) {
    await client.query('ROLLBACK');
    console.error('LINE webhook transaction failed', err);
    // LINE will retry on 5xx
    res.status(500).json({ error: 'Processing failed' });
  } finally {
    client.release();
  }
}

async function resolveTenantId(channelId: string, client: PoolClient): Promise<string | null> {
  const { rows } = await client.query(
    `SELECT tenant_id FROM line_channels WHERE channel_id = $1 LIMIT 1`,
    [channelId]
  );
  return rows[0]?.tenant_id || null;
}

async function dispatchToDomain(event: any, tenantId: string, client: PoolClient) {
  const { type } = event;

  if (type === 'message' && event.message.type === 'text') {
    const text = event.message.text.toLowerCase();
    const userId = event.source.userId;

    if (text.includes('เข้างาน') || text.includes('clock in')) {
      await handleClockIn(userId, tenantId, client, event);
    } else if (text.includes('ออกงาน') || text.includes('clock out')) {
      await handleClockOut(userId, tenantId, client, event);
    } else if (text.includes('ลา') || text.includes('leave')) {
      await handleLeaveRequest(userId, tenantId, client, event);
    } else if (text.includes('โอที') || text.includes('ot')) {
      await handleOtRequest(userId, tenantId, client, event);
    }
  }
}

async function handleClockIn(userId: string, tenantId: string, client: PoolClient, event: any) {
  // Idempotent upsert for same day + user + tenant; enforce one active session
  const now = new Date();
  const date = now.toISOString().split('T')[0];

  await client.query(
    `INSERT INTO attendances (tenant_id, user_id, clock_in, clock_in_event, date)
     VALUES ($1, $2, $3, $4, $5)
     ON CONFLICT (tenant_id, user_id, date)
     DO UPDATE SET clock_in = EXCLUDED.clock_in, clock_in_event = EXCLUDED.clock_in_event
     WHERE attendances.clock_in IS NULL`,
    [tenantId, userId, now, event]
  );

  // Emit realtime update
  emitTenantEvent(tenantId, { type: 'attendance_updated', userId, date, clockIn: now });
}

async function handleClockOut(userId: string, tenantId: string, client: PoolClient, event: any) {
  const now = new Date();
  const date = now.toISOString().split('T')[0];

  await client.query(
    `UPDATE attendances
     SET clock_out = $1, clock_out_event = $2
     WHERE tenant_id = $3 AND user_id = $4 AND date = $5 AND clock_out IS NULL`,
    [now, event, tenantId, userId, date]
  );

  emitTenantEvent(tenantId, { type: 'attendance_updated', userId, date, clockOut: now });
}

async function handleLeaveRequest(userId: string, tenantId: string, client: PoolClient, event: any) {
  await client.query(
    `INSERT INTO leave_requests (tenant_id, user_id, status, requested_at, line_event)
     VALUES ($1, $2, 'pending', NOW(), $3)`,
    [tenantId, userId, event]
  );

  emitTenantEvent(tenantId, { type: 'leave_requested', userId, at: new Date() });
}
