## Final Synthesized Implementation Plan  
**Highest-value improvement:** CDN-only training manifest + Lightning idle-resilient launcher.  
**Why:** Eliminates HF API rate limits during training, auto-recovers from Lightning idle stops with zero quota waste, and is deployable in ≤2h.

### 1) Manifest generator (run once on Mac)
```bash
# generate_manifest.py
import json, os, sys
from datetime import datetime
from huggingface_hub import HfApi

HF_REPO = os.getenv("HF_DATASET_REPO", "axentx/surrogate-mirror-merged")
DATE_FOLDER = os.getenv("DATE_FOLDER", datetime.utcnow().strftime("%Y-%m-%d"))
OUT_PATH = os.getenv("MANIFEST_OUT", "manifest.json")

api = HfApi()
entries = api.list_repo_tree(
    repo_id=HF_REPO,
    path=DATE_FOLDER,
    recursive=False
)

files = sorted([e.path for e in entries if e.path.endswith(".parquet")])
if not files:
    sys.exit(f"No parquet files found in {HF_REPO}/{DATE_FOLDER}")

manifest = {
    "repo_id": HF_REPO,
    "base_cdn": f"https://huggingface.co/datasets/{HF_REPO}/resolve/main",
    "date": DATE_FOLDER,
    "created_utc": datetime.utcnow().isoformat() + "Z",
    "files": [{"filename": f, "cdn_url": f"{base}/{f}"} for f in files]
}
manifest["base_cdn"] = base = f"https://huggingface.co/datasets/{HF_REPO}/resolve/main"

with open(OUT_PATH, "w") as f:
    json.dump(manifest, f, indent=2)

print(f"Wrote {len(files)} files to {OUT_PATH}")
```

Run:

### 2) CDN-only IterableDataset (surrogate/train.py)
```python
import argparse
import io
import json
from typing import Iterator, Dict

import pyarrow.parquet as pq
import requests
import torch
from torch.utils.data import IterableDataset

class CDNParquetDataset(IterableDataset):
    def __init__(self, manifest_path: str, start: int = 0, end: int = None):
        with open(manifest_path) as f:
            m = json.load(f)
        self.base = m["base_cdn"]
        files = m["files"]
        if isinstance(files[0], dict):
            self.files = [f["filename"] for f in files]
        else:
            self.files = files
        if end is not None:
            self.files = self.files[start:end]
        else:
            self.files = self.files[start:]

    def __iter__(self) -> Iterator[Dict[str, str]]:
        for path in self.files:
            url = f"{self.base}/{path}"
            resp = requests.get(url, timeout=30)
            resp.raise_for_status()
            buf = io.BytesIO(resp.content)
            table = pq.read_table(buf, columns=["prompt", "response"])
            for row in table.to_pylist():
                yield {
                    "prompt": row["prompt"],
                    "response": row["response"]
                }

# usage in Lightning DataModule
def train_dataloader(self):
    dataset = CDNParquetDataset("manifest.json")
    return torch.utils.data.DataLoader(dataset, batch_size=8, num_workers=0)
```

### 3) Lightning launcher with idle resilience (surrogate/launch_train.py)
```python
import time
import subprocess
from lightning import LightningWork, LightningApp, LightningFlow, Machine

class SurrogateTrainer(LightningWork):
    def __init__(self):
        super().__init__(machine=Machine.L40S, cloud_compute=None)
        self.studio_name = "surrogate-train-l40s"

    def run(self, manifest_path: str):
        cmd = [
            "python", "train.py",
            "--manifest", manifest_path,
            "--epochs", "1"
        ]
        subprocess.run(cmd, check=True)

class SurrogateFlow(LightningFlow):
    def __init__(self):
        super().__init__()
        self.trainer = SurrogateTrainer()

    def run(self):
        from lightning import Teamspace
        # reuse running studio if exists
        for s in Teamspace.studios:
            if s.name == self.trainer.studio_name and s.status == "running":
                print(f"Reusing running studio: {s.name}")
                self.trainer = s
                break
        else:
            print("Starting new studio...")
            self.trainer.start()

        # ensure alive before run
        if self.trainer.status != "running":
            self.trainer.start(machine=Machine.L40S)
            time.sleep(30)  # allow boot

        self.trainer.run("manifest.json")

app = LightningApp(SurrogateFlow())
```
