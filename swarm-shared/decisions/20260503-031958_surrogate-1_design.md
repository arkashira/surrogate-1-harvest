# surrogate-1 / design

## Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16`, `DATE`, `HF_TOKEN` via env
- Single `list_repo_tree(path, recursive=False)` per date folder (avoids recursive 429)
- Deterministic shard assignment via `hash(slug) % SHARD_TOTAL`
- Downloads eligible files via **HF CDN** (`resolve/main/...`) — zero API calls during data load, bypasses auth rate limits
- Projects to `{prompt, response}` only at parse time (avoids pyarrow CastError on mixed schemas)
- Writes `batches/public-merged/{DATE}/shard{N}-{HHMMSS}.jsonl`
- Uses central `lib/dedup.py` for cross-source dedup (md5 store)
- Exits 0 on success, non-zero on hard failure (GitHub Actions will retry)

### Why this is the highest-value incremental improvement
- Directly applies the **HF CDN bypass** insight (eliminates 429 during training)
- Fixes **pyarrow CastError** by deferring projection
- Keeps **shard isolation** and **commit-cap spread** patterns
- Small, focused change (<2h) with clear rollback path (keep old script renamed)

### 1) New worker: `bin/dataset-enrich.py`

```python
#!/usr/bin/env python3
"""
surrogate-1 CDN-bypass ingestion worker.

Environment:
  SHARD_ID        (int, 0..15)
  SHARD_TOTAL=16  (int)
  DATE            (YYYY-MM-DD)
  HF_TOKEN        (write token for axentx/surrogate-1-training-pairs)
"""
import os
import sys
import json
import hashlib
import datetime
from pathlib import Path
from typing import Dict, Any, Optional

import requests
from huggingface_hub import HfApi, hf_hub_download

REPO_ID = "axentx/surrogate-1-training-pairs"
API = HfApi()

# ── helpers ──────────────────────────────────────────────────────────────

def _hash_slug(slug: str) -> int:
    return int(hashlib.md5(slug.encode()).hexdigest(), 16)

def _belongs_to_shard(slug: str, shard_id: int, shard_total: int) -> bool:
    return _hash_slug(slug) % shard_total == shard_id

def _now_str() -> str:
    return datetime.datetime.utcnow().strftime("%Y%m%d-%H%M%S")

def _cdn_url(path: str) -> str:
    """Public CDN URL — no Authorization header required."""
    return f"https://huggingface.co/datasets/{REPO_ID}/resolve/main/{path}"

# ── schema projection ───────────────────────────────────────────────────

def _extract_pair(obj: Dict[str, Any]) -> Optional[Dict[str, str]]:
    """
    Best-effort projection to {prompt, response}.
    Supports common HF dataset shapes without loading full parquet schema.
    """
    if not isinstance(obj, dict):
        return None

    # Direct keys
    prompt = obj.get("prompt") or obj.get("input") or obj.get("question")
    response = obj.get("response") or obj.get("output") or obj.get("answer")

    # Nested chat format
    if prompt is None and "messages" in obj:
        msgs = obj["messages"]
        if isinstance(msgs, list) and len(msgs) >= 2:
            prompt = msgs[-2].get("content") if len(msgs) >= 2 else None
            response = msgs[-1].get("content")

    # JSONL row with single text field
    if prompt is None and "text" in obj:
        # heuristic: split on last Q/A separator if present
        prompt = obj["text"]
        response = None

    if prompt is None or response is None:
        return None
    return {"prompt": str(prompt), "response": str(response)}

# ── dedup bridge ────────────────────────────────────────────────────────

def _is_duplicate(md5_hex: str) -> bool:
    from lib.dedup import DedupStore  # lazy import
    store = DedupStore()
    return store.exists(md5_hex)

def _register_md5(md5_hex: str) -> None:
    from lib.dedup import DedupStore
    store = DedupStore()
    store.add(md5_hex)

# ── main worker ─────────────────────────────────────────────────────────

def run() -> int:
    shard_id = int(os.environ.get("SHARD_ID", 0))
    shard_total = int(os.environ.get("SHARD_TOTAL", 16))
    date = os.environ.get("DATE", datetime.date.today().isoformat())
    hf_token = os.environ.get("HF_TOKEN")

    if not hf_token:
        print("HF_TOKEN missing", file=sys.stderr)
        return 1

    headers = {"Authorization": f"Bearer {hf_token}"}
    out_dir = Path(f"batches/public-merged/{date}")
    out_dir.mkdir(parents=True, exist_ok=True)
    out_file = out_dir / f"shard{shard_id}-{_now_str()}.jsonl"

    # 1) single non-recursive tree call for the date folder
    folder_path = f"batches/public-merged/{date}"
    try:
        tree = API.list_repo_tree(
            repo_id=REPO_ID,
            path=folder_path,
            repo_type="dataset",
            recursive=False,
        )
    except Exception as exc:
        # folder may not exist yet — that's fine, we'll create it
        print(f"No existing tree at {folder_path}: {exc}", file=sys.stderr)
        tree = []

    # 2) collect existing slugs to avoid re-ingesting same files
    existing_slugs = set()
    for entry in tree:
        if entry.path.endswith(".jsonl"):
            # shard-N-<ts>.jsonl
            parts = Path(entry.path).stem.split("-")
            if len(parts) >= 3:
                existing_slugs.add(parts[-1])  # crude; good enough

    # 3) list source files (top-level or known source folders)
    # We ingest from upstream source folders (e.g. public-datasets/raw/)
    # For this repo we assume sources are in a known prefix.
    source_prefix = "sources/"
    try:
        src_tree = API.list_repo_tree(
            repo_id=REPO_ID,
            path=source_prefix,
            repo_type="dataset",
            recursive=True,
        )
    except Exception:
        src_tree = []

    eligible_files = []
    for entry in src_tree:
        if entry.type != "file":
            continue
        if not entry.path.lower().endswith((".jsonl", ".parquet", ".json")):
            continue
        slug = Path(entry.path).stem
        if _belongs_to_shard(slug, shard_id, shard_total):
            eligible_files.append(entry.path)

    if not eligible_files:
        print("No eligible files for this shard.")
        # still produce empty output so workflow continues cleanly
        out_file.touch()
        return 0

    # 4) stream-parse via CDN download + line-by-line projection
    written = 0
    skipped_dup = 0
    with out_file.open("w", encoding="utf-8") as fout:
        for src_path in eligible_files:
            print(f"Processing {src_path}...")
            url = _cdn_url(src_path)

            # For JSONL we can stream line-by-line.
            # For parquet we fallback to hf_hub_download (local file) then stream.
            if src_path.lower().endswith(".jsonl"):
                resp = requests.get(url, headers=headers, stream=True, timeout=60)
                resp.raise_for_status()
                lines = resp.iter_lines(decode_unicode=True)
            else:
                # Parquet or JSON: download once (small files)
                local_path = hf_hub_download(
                    repo_id=REPO_ID,
                    filename=src_path,
                    repo_type="dataset",
                )
                # Use pyarrow to stream rows without full schema load
                import pyarrow.parquet as pq
                table = pq.read_table(local_path, columns=["prompt", "response"]
