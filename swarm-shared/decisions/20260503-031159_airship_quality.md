## 1) High-level plan (≤2h)

1. **Mac orchestration (once per dataset update)**  
   - One non-recursive `list_repo_tree` call per date folder.  
   - Output: `data/train_files_{date}.json` (repo, date_folder, sorted parquet paths).  
   - On 429: wait 360s and retry once; fail fast otherwise.

2. **CDN-only DataLoader (zero HF API during training)**  
   - Fetch via `https://huggingface.co/datasets/{repo}/resolve/main/{path}` (no auth).  
   - Cache locally; fallback to `hf_hub_download` only if CDN fails.  
   - Stream/read parquet and project to `{prompt, response}` only.  
   - No extra columns (`source`, `ts`) added.

3. **Lightning idle-resilient runner**  
   - Before each `.run()`, reuse any running studio with matching name; else create.  
   - If studio is stopped, restart it and continue.  
   - Wrap training job in try/except; on failure/stop, restart studio and retry.

4. **Deployment**  
   - Commit generated JSON + new modules + updated `train.py`.  
   - Start via `lightning_runner.py` (or compose).  
   - Monitor; auto-recovery on idle-stop.

### `scripts/prep_train_files.sh`
```bash
#!/usr/bin/env bash
# Generate train_files_{date}.json with one HF API call.
# Usage: REPO=org/repo DATE=2026-04-29 ./scripts/prep_train_files.sh

set -euo pipefail

: "${REPO:?Set REPO=org/repo}"
: "${DATE_FOLDER:=$(date +%Y-%m-%d)}"
OUT_FILE="data/train_files_${DATE_FOLDER}.json"

mkdir -p data

python3 - <<PY
import json, os, sys, time
from huggingface_hub import HfApi, HfFolder

repo = os.environ["REPO"]
date_folder = os.environ["DATE_FOLDER"]
out_file = os.environ["OUT_FILE"]

api = HfApi()

def list_once():
    return api.list_repo_tree(repo=repo, path=date_folder, recursive=False)

try:
    entries = list_once()
except Exception as e:
    # crude 429 handling: wait and retry once
    if "429" in str(e):
        print("Rate limited; waiting 360s...")
        time.sleep(360)
        entries = list_once()
    else:
        raise

files = sorted(e.rfilename for e in entries if e.rfilename.endswith(".parquet"))

payload = {
    "repo": repo,
    "date_folder": date_folder,
    "files": files
}
with open(out_file, "w") as f:
    json.dump(payload, f, indent=2)
print(f"Wrote {len(files)} files to {out_file}")
PY
```

### `data/cdn_dataset.py` (PyTorch Dataset)
```python
import json
import os
import pyarrow.parquet as pq
import requests
from io import BytesIO
from torch.utils.data import Dataset
from huggingface_hub import hf_hub_download

CDN_BASE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

class CDNParquetDataset(Dataset):
    """
    CDN-only Parquet dataset yielding {"prompt": ..., "response": ...}.
    file_list_json format:
    {
      "repo": "org/repo",
      "date_folder": "batches/mirror-merged/2026-04-29",
      "files": ["batches/mirror-merged/2026-04-29/xxx.parquet", ...]
    }
    """

    def __init__(self, file_list_json, cache_dir=".cache_cdn", transform=None):
        super().__init__()
        with open(file_list_json) as f:
            meta = json.load(f)
        self.repo = meta["repo"]
        self.date_folder = meta["date_folder"]
        self.files = meta["files"]
        self.cache_dir = cache_dir
        self.transform = transform
        os.makedirs(self.cache_dir, exist_ok=True)

        # Build index of rows for optional deterministic access
        self._length = None

    def _cdn_url(self, path):
        return CDN_BASE.format(repo=self.repo, path=path)

    def _fetch(self, path):
        name = os.path.basename(path)
        local_path = os.path.join(self.cache_dir, name)
        if os.path.exists(local_path):
            return local_path

        # Try CDN first (no auth)
        url = self._cdn_url(path)
        try:
            r = requests.get(url, timeout=30)
            r.raise_for_status()
            with open(local_path, "wb") as f:
                f.write(r.content)
            return local_path
        except Exception:
            # Fallback to hf_hub_download (uses API)
            return hf_hub_download(
                repo_id=self.repo,
                filename=path,
                cache_dir=self.cache_dir,
                force_download=False,
            )

    def _load_parquet_rows(self, parquet_path):
        table = pq.read_table(parquet_path, columns=["prompt", "response"])
        return table.to_pylist()

    def __len__(self):
        if self._length is None:
            total = 0
            for p in self.files:
                local = self._fetch(p)
                rows = self._load_parquet_rows(local)
                total += len([r for r in rows if r.get("prompt") and r.get("response")])
            self._length = total
        return self._length

    def __getitem__(self, idx):
        # Simple linear scan; acceptable for small/medium cached datasets.
        # For large-scale training, pre-build a row index or use IterableDataset.
        current = 0
        for p in self.files:
            local = self._fetch(p)
            rows = [r for r in self._load_parquet_rows(local)
                    if r.get("prompt") and r.get("response")]
            if current + len(rows) > idx:
                item = rows[idx - current]
                if self.transform:
                    item = self.transform(item)
                return item
            current += len(rows)
        raise IndexError(idx)
```
