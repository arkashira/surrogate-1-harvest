## Implementation Plan

### 1. One-time Mac orchestration script (≤10 min)

`scripts/list_parquet_files.py`
```python
#!/usr/bin/env python3
"""
Run from Mac after HF rate-limit window clears.
Lists parquet files in a date folder and saves to JSON for CDN-only training.
"""
import json
from pathlib import Path
from huggingface_hub import HfApi

REPO_ID = "axentx/surrogate-datasets"
DATE_FOLDER = "batches/mirror-merged/2026-05-03"  # update per run
OUT_FILE = Path("airship/surrogate/configs/file_list.json")

def main():
    api = HfApi()
    # recursive=False avoids 429 on big repos; list per folder
    entries = api.list_repo_tree(
        repo_id=REPO_ID,
        path=DATE_FOLDER,
        recursive=True  # single call; acceptable for one date folder
    )
    parquet_files = [
        f"{DATE_FOLDER}/{e.path.split('/')[-1]}"
        for e in entries
        if e.path.endswith(".parquet")
    ]
    payload = {
        "repo_id": REPO_ID,
        "date_folder": DATE_FOLDER,
        "files": sorted(parquet_files),
        "total": len(parquet_files),
    }
    OUT_FILE.parent.mkdir(parents=True, exist_ok=True)
    OUT_FILE.write_text(json.dumps(payload, indent=2))
    print(f"Wrote {len(parquet_files)} files to {OUT_FILE}")

if __name__ == "__main__":
    main()
```

Make executable and run once:
```bash
chmod +x scripts/list_parquet_files.py
python3 scripts/list_parquet_files.py
```

### 2. CDN-only Parquet Dataset (zero API calls during training)

`surrogate/data/cdn_parquet_dataset.py`
```python
import pyarrow.parquet as pq
import pyarrow as pa
import requests
from pathlib import Path
from torch.utils.data import Dataset, IterableDataset
import io
import json
from typing import List, Dict, Any
import os

HF_CDN = "https://huggingface.co/datasets"

class CDNParquetDataset(IterableDataset):
    """
    Loads parquet files directly from HF CDN (no Authorization header).
    Uses file list generated by Mac orchestration script.
    Projects only {prompt, response} at parse time.
    """
    def __init__(self, file_list_path: str, repo_id: str, split: str = "train"):
        super().__init__()
        with open(file_list_path) as f:
            self.meta = json.load(f)
        self.repo_id = repo_id
        self.files: List[str] = self.meta["files"]
        if not self.files:
            raise ValueError("No parquet files found in file_list")

    def _stream_file(self, file_path: str) -> bytes:
        url = f"{HF_CDN}/{self.repo_id}/resolve/main/{file_path}"
        resp = requests.get(url, timeout=30)
        resp.raise_for_status()
        return resp.content

    def _project_record(self, batch: pa.Table) -> List[Dict[str, Any]]:
        records = []
        prompts = batch.column("prompt").to_pylist() if "prompt" in batch.column_names else [None] * len(batch)
        responses = batch.column("response").to_pylist() if "response" in batch.column_names else [None] * len(batch)
        for p, r in zip(prompts, responses):
            if p is not None and r is not None:
                records.append({"prompt": p, "response": r})
        return records

    def __iter__(self):
        worker_info = torch.utils.data.get_worker_info()
        if worker_info is None:
            iter_files = self.files
        else:
            per_worker = len(self.files) // worker_info.num_workers
            worker_id = worker_info.id
            iter_files = self.files[worker_id * per_worker : (worker_id + 1) * per_worker]

        for file_path in iter_files:
            try:
                buf = self._stream_file(file_path)
                table = pq.read_table(io.BytesIO(buf), columns=["prompt", "response"])
                for record in self._project_record(table):
                    yield record
            except Exception as e:
                print(f"Skipping {file_path}: {e}")
                continue
```

### 3. Lightning idle-resilient runner

`surrogate/training/lightning_runner.py`
```python
#!/usr/bin/env python3
"""
Lightning training runner with idle-resilience and studio reuse.
Checks status before each run; restarts if stopped by idle timeout.
"""
import time
import lightning as L
from pathlib import Path
from surrogate.training.train_surrogate import SurrogateTrainer  # your existing trainer
from surrogate.data.cdn_parquet_dataset import CDNParquetDataset
from torch.utils.data import DataLoader

class LightningRunner:
    def __init__(self, studio_name: str = "surrogate-l40s-run"):
        self.studio_name = studio_name
        self.teamspace = L.Teamspace()
        self.studio = None
        self.machine = L.Machine.L40S  # free tier falls to L40S; H200 requires prod account

    def _find_or_create_studio(self):
        for s in self.teamspace.studios:
            if s.name == self.studio_name:
                self.studio = s
                print(f"Reusing studio: {self.studio_name} (status: {s.status})")
                return
        self.studio = L.Studio(
            name=self.studio_name,
            machine=self.machine,
            create_ok=True,
        )
        print(f"Created studio: {self.studio_name}")

    def _ensure_running(self):
        if self.studio is None:
            self._find_or_create_studio()
        if self.studio.status != "Running":
            print(f"Studio stopped (idle timeout?). Restarting on {self.machine}...")
            self.studio.start(machine=self.machine)
            # wait for running
            while self.studio.status != "Running":
                time.sleep(10)
                self.studio.refresh()
            print("Studio is running.")

    def build_dataloader(self):
        file_list = Path("configs/file_list.json")
        dataset = CDNParquetDataset(
            file_list_path=str(file_list),
            repo_id="axentx/surrogate-datasets",
        )
        return DataLoader(dataset, batch_size=8, num_workers=2)

    def run(self, max_epochs: int = 1):
        self._ensure_running()
        dataloader = self.build_dataloader()
        trainer = SurrogateTrainer(max_epochs=max_epochs)

        with self.studio.run():
            print("Running training inside studio...")
            trainer.fit(dataloader)
            print("Training complete.")

if __name__ == "__main__":
    runner = LightningRunner()
    runner.run(max_epochs=1)
```

### 4. Cron-safe wrapper (prevents script-error patterns)

`scripts/run_surrogate_training.sh`
```bash
#!/usr/bin/env bash
# Cron-safe wrapper for surrogate training.
# Ensures proper environment and executable bits.

set -euo pipefail
export SHELL=/bin/bash
cd /opt/axentx/airship

# If HF API rate-limited, re-list files first (run manually after window clears)
if [ ! -f configs/file_list.json ]; then
    echo "file_list.json missing. Run list_parquet_files.py from Mac after rate-limit clears."
    exit 1
fi

python3 -m surrogate.training.lightning_runner --max-epochs 1
```

Make executable:
```bash
chmod +x scripts/run_surrogate_training.sh
```

```cron

0 */6 * * * /opt/axentx/airship/scripts/run_surrogate_training.sh >> /var/log/airship_training.log 2>&1
```
