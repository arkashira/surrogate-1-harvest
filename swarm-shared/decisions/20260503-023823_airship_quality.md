## Core Principles (resolve contradictions)

1. **One API call only** — use non-recursive folder listing to avoid 429.
2. **CDN-first, never streaming** — `hf_hub_download` per file; no `load_dataset` streaming.
3. **Deterministic sibling routing** — hash-based repo selection across 5 siblings to spread load and avoid per-repo limits.
4. **Schema-safe parsing** — read only `prompt`/`response` columns; fallback to line-by-line JSONL; never rely on global schema inference.
5. **Zero training-time API calls** — all fetches happen during download phase; training consumes local cache.
6. **Studio reuse + auto-restart** — detect running studio; restart L40S if idle-stopped; no redundant instance spin-ups.

### 1. Manifest Generator (`surrogate/scripts/build_manifest.py`)
```python
#!/usr/bin/env python3
"""
Single API call → CDN-only training afterward.
Run on Mac or after rate-limit window clears.
"""
import json
import hashlib
from pathlib import Path
from huggingface_hub import HfApi

API = HfApi()
REPO = "axentx/surrogate-data"
SIBLINGS = [f"axentx/surrogate-data-s{i}" for i in range(1, 6)]

def pick_repo(slug: str) -> str:
    h = int(hashlib.md5(slug.encode()).hexdigest(), 16)
    return SIBLINGS[h % len(SIBLINGS)]

def build(date_folder: str, out_path: Path):
    # Non-recursive folder list: one API call
    items = API.list_repo_tree(REPO, path=date_folder, recursive=False)
    files = [f for f in items if f.endswith(('.jsonl', '.parquet', '.json'))]

    manifest = {
        "date": date_folder,
        "repo_template": REPO,
        "sibling_repos": SIBLINGS,
        "files": [
            {
                "path": f,
                "cdn_url": f"https://huggingface.co/datasets/{REPO}/resolve/main/{f}",
                "sibling_repo": pick_repo(f)
            }
            for f in sorted(files)
        ]
    }
    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"Manifest: {len(files)} files → {out_path}")

if __name__ == "__main__":
    import sys
    date = sys.argv[1] if len(sys.argv) > 1 else "2026-04-29"
    build(date, Path("file_manifest.json"))
```

### 2. CDN-First Dataset Loader (`surrogate/data/cdn_dataset.py`)
```python
from pathlib import Path
import pyarrow.parquet as pq
import pyarrow as pa
import json
from huggingface_hub import hf_hub_download
from typing import Dict, Iterator, List

class CDNDataset:
    def __init__(self, manifest_path: str, cache_dir: str = ".cache"):
        self.manifest = json.loads(Path(manifest_path).read_text())
        self.cache_dir = Path(cache_dir)
        self.cache_dir.mkdir(exist_ok=True)
        self.api_calls = 0

    def _download(self, file_entry: Dict) -> Path:
        repo = file_entry["sibling_repo"]
        path = file_entry["path"]
        local = hf_hub_download(
            repo_id=repo,
            filename=path,
            cache_dir=str(self.cache_dir),
            force_download=False,
            resume_download=True
        )
        self.api_calls += 1  # Only during download
        return Path(local)

    def _project_to_pair(self, file_path: Path) -> List[Dict[str, str]]:
        # Schema-safe: read only prompt/response; fallback to JSONL
        try:
            table = pq.read_table(file_path, columns=["prompt", "response"])
            df = table.to_pandas()
            return df[["prompt", "response"]].to_dict(orient="records")
        except (pa.ArrowInvalid, KeyError, OSError):
            items = []
            for line in file_path.read_text().splitlines():
                line = line.strip()
                if not line:
                    continue
                try:
                    obj = json.loads(line)
                    if "prompt" in obj and "response" in obj:
                        items.append({"prompt": obj["prompt"], "response": obj["response"]})
                except json.JSONDecodeError:
                    continue
            return items

    def stream(self) -> Iterator[Dict[str, str]]:
        for entry in self.manifest["files"]:
            local = self._download(entry)
            yield from self._project_to_pair(local)

    def __len__(self):
        return len(self.manifest["files"])
```

### 3. Lightning Launcher with Studio Reuse (`surrogate/train.py`)
```python
#!/usr/bin/env python3
import argparse
import time
from pathlib import Path
import lightning as L
from torch.utils.data import IterableDataset, DataLoader
from surrogate.data.cdn_dataset import CDNDataset

class CDNIterable(IterableDataset):
    def __init__(self, manifest_path: str):
        self.manifest_path = manifest_path

    def __iter__(self):
        dataset = CDNDataset(self.manifest_path)
        return iter(dataset.stream())

def find_running_studio(name: str):
    teamspace = L.Teamspace()
    for s in teamspace.studios:
        if s.name == name and s.status == "running":
            return s
    return None

def train(manifest: str, epochs: int = 1):
    studio = find_running_studio("surrogate-train")
    if studio is None:
        studio = L.Studio(
            name="surrogate-train",
            create_ok=True,
            machine=L.Machine.L40S,
            cloud="lightning-public-prod"
        )

    if studio.status != "running":
        studio.start(machine=L.Machine.L40S)
        time.sleep(30)  # warmup

    dataset = CDNIterable(manifest)
    loader = DataLoader(dataset, batch_size=8, num_workers=0)

    for epoch in range(epochs):
        for i, batch in enumerate(loader):
            prompts = [b["prompt"] for b in batch]
            responses = [b["response"] for b in batch]
            # TODO: surrogate model forward/backward
            if i % 10 == 0:
                print(f"Epoch {epoch} step {i}: {len(prompts)} samples")

    print("Training complete (CDN-only, zero HF API during data load)")

if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--date", default="2026-04-29")
    parser.add_argument("--manifest", default="file_manifest.json")
    parser.add_argument("--epochs", type=int, default=1)
    args = parser.parse_args()

    manifest_path = Path(args.manifest)
    if not manifest_path.exists():
        from surrogate.scripts.build_manifest import build
        build(args.date, manifest_path)

    train(str(manifest_path), args.epochs)
```
