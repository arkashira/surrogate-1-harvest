## Final Synthesized Solution

**Core Diagnosis (Consolidated)**
- No content-addressed manifest per date folder forces runtime repo enumeration (`list_repo_tree`/`load_dataset`) during training, triggering HF API 429s and non-reproducible epochs.
- Training scripts call `load_dataset(streaming=True)` on heterogeneous repos, causing `pyarrow.CastError` on mixed-schema files.
- Lightning Studio reuse is not implemented; each run recreates a studio and burns quota.
- Data ingestion writes mixed-schema files into `enriched/` with extra metadata columns instead of projecting to `{prompt, response}` and using filename attribution.
- No CDN-only fallback path; every data load hits HF API and is subject to rate limits.

**Proposed Change (Unified)**
Create `/opt/axentx/vanguard/ops/cdn_manifest.py` (single file, <150 LOC) that:
- Accepts a repo + date folder
- Calls `list_repo_tree(recursive=False)` **once** (offline, after rate-limit window) and writes `manifest_{date}.json` with `{path, sha256, size, cdn_url}`
- Provides a `DataLoader` that streams files via CDN URLs with **zero API calls** during training
- Projects each file to `{prompt, response}` on-the-fly (no schema assumptions)
- Integrates Lightning Studio reuse: checks for running studio by name and reuses it

Scope: one new file + one small patch to existing training launcher (or a new launcher script).

## Implementation

```bash
# /opt/axentx/vanguard/ops/cdn_manifest.py
#!/usr/bin/env python3
"""
Content-addressed CDN-only manifest for HF datasets.
Eliminates runtime HF API calls during training.
"""
import json, hashlib, os, sys, time
from pathlib import Path
from typing import Dict, List, Iterator, Optional

import requests
from huggingface_hub import HfApi

CDN_ROOT = "https://huggingface.co/datasets"
API = HfApi()

# --
# 1) Manifest generation (run once per date folder on Mac)
# --
def build_manifest(repo_id: str, folder: str, out_path: Path) -> Path:
    """
    List files in repo/folder (non-recursive) and write manifest.
    Uses HF API once; safe to run after rate-limit window clears.
    """
    print(f"Building manifest for {repo_id}/{folder} ...")
    tree = API.list_repo_tree(repo_id=repo_id, path=folder, recursive=False)
    entries = []
    for item in tree:
        if item.type != "file":
            continue
        url = f"{CDN_ROOT}/{repo_id}/resolve/main/{folder}/{item.path}"
        r = requests.head(url, timeout=30)
        r.raise_for_status()
        size = int(r.headers.get("content-length", 0))
        etag = r.headers.get("etag", "")
        sha256 = etag.strip('"') if "sha256" in etag.lower() else None
        entries.append({
            "path": f"{folder}/{item.path}",
            "size": size,
            "sha256": sha256,
            "cdn_url": url,
        })
    manifest = {
        "repo_id": repo_id,
        "folder": folder,
        "created_ts": int(time.time()),
        "entries": entries,
    }
    out_path.parent.mkdir(parents=True, exist_ok=True)
    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"Manifest written to {out_path} ({len(entries)} files)")
    return out_path

# --
# 2) CDN-only streaming loader (zero API calls during training)
# --
def iter_cdn_records(manifest_path: Path, max_files: int = 0) -> Iterator[Dict[str, str]]:
    manifest = json.loads(manifest_path.read_text())
    entries = manifest["entries"]
    if max_files:
        entries = entries[:max_files]
    for e in entries:
        url = e["cdn_url"]
        try:
            r = requests.get(url, timeout=60, stream=True)
            r.raise_for_status()
            content = r.content
            text = content.decode("utf-8", errors="replace").strip()
            if text.startswith("{") or text.startswith("["):
                try:
                    obj = json.loads(text)
                    if isinstance(obj, dict):
                        prompt = obj.get("prompt") or obj.get("input") or obj.get("instruction") or ""
                        response = obj.get("response") or obj.get("output") or obj.get("completion") or ""
                    elif isinstance(obj, list):
                        prompt = str(obj[0]) if len(obj) > 0 else ""
                        response = str(obj[1]) if len(obj) > 1 else ""
                    else:
                        prompt, response = "", ""
                except Exception:
                    prompt, response = "", text
            else:
                parts = [p.strip() for p in text.replace("\r", "").split("\n\n") if p.strip()]
                if len(parts) >= 2:
                    prompt, response = parts[0], parts[1]
                else:
                    prompt, response = "", text
            yield {"prompt": prompt, "response": response}
        except Exception as exc:
            print(f"Failed to fetch {url}: {exc}", file=sys.stderr)
            continue

# --
# 3) Lightning Studio reuse helper
# --
def reuse_or_create_studio(name: str, machine: str = "lightning-ai/L40S-1"):
    """
    Reuse a running studio if exists; otherwise create one.
    Prevents quota burn from repeated recreation.
    """
    try:
        from lightning.pytorch import LightningModule
        from lightning.fabric.plugins import LightningStudio
        from lightning.fabric.plugins.lightning_studio import Teamspace, Machine
    except ImportError as e:
        print(f"Lightning SDK not available: {e}. Skipping studio reuse.", file=sys.stderr)
        return None

    teamspace = Teamspace()
    for s in teamspace.studios:
        if s.name == name and s.status == "Running":
            print(f"Reusing running studio: {name}")
            return s
    print(f"No running studio '{name}' found. Creating...")
    try:
        studio = LightningStudio(
            name=name,
            machine=Machine(machine),
            create_ok=True,
        )
        return studio
    except Exception as exc:
        print(f"Studio creation failed: {exc}", file=sys.stderr)
        return None

# --
# 4) CLI entrypoints
# --
if __name__ == "__main__":
    if len(sys.argv) < 4:
        print("Usage: cdn_manifest.py build <repo_id> <folder> [out.json]")
        print("       cdn_manifest.py stream <manifest.json> [max_files]")
        sys.exit(1)
    cmd = sys.argv[1]
    if cmd == "build":
        repo_id, folder = sys.argv[2], sys.argv[3]
        out = Path(sys.argv[4]) if len(sys.argv) > 4 else Path(f"manifest_{folder.replace('/', '_')}.json")
        build_manifest(repo_id, folder, out)
    elif cmd == "stream":
        manifest_path = Path(sys.argv[2])
        max_files = int(sys.argv[3]) if len(sys.argv) > 3 else 0
        for i, rec in enumerate(iter_cdn_records(manifest_path, max_files=max_files)):
            print(json.dumps(rec, ensure_ascii=False))
            if i >= 9 and max_files <= 0:
                break
    else:
        print(f"Unknown command: {cmd}")
        sys.exit(1)
```
