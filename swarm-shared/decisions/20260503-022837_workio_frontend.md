# workio / frontend

## 1. Diagnosis

- Clock-in/out UI has no optimistic feedback and no local pending state, so users can’t tell if a tap registered while offline or slow — leads to repeated taps and duplicate punches.
- No client-side idempotency guard for punch requests: rapid double-taps or retries can race and create duplicate records even if backend is fixed later.
- Missing visual confirmation (toast/sound) after successful punch; users often second-guess whether clock-in/out worked.
- GPS acquisition is blocking and can stall the punch flow; UX doesn’t surface location status or fallback when GPS is slow/denied.
- No offline queue for punches: if network drops, the action is lost and user must retry manually.

## 2. Proposed change

**Scope:** `workio/src/components/TimeClock.tsx` (or create if absent) + `workio/src/hooks/usePunch.ts` + `workio/src/lib/api.ts`  
**Goal:** Add optimistic UI, client idempotency key, non-blocking GPS, toast confirmation, and offline queue (best-effort) — all within <2h.

## 3. Implementation

### 3.1 Add idempotency + optimistic state + toast

```tsx
// workio/src/hooks/usePunch.ts
import { useCallback, useRef } = 'react';
import { toast } from 'react-hot-toast';
import { postPunch } from '../lib/api';

export function usePunch() {
  const inflightRef = useRef<Promise<unknown> | null>(null);
  const lastIdempotencyRef = useRef<string | null>(null);

  const punch = useCallback(async (type: 'in' | 'out', location?: GeolocationCoordinates) => {
    // prevent rapid double-tap races
    if (inflightRef.current) {
      toast('⏳ Punch in progress…', { id: 'punch-in-flight' });
      return;
    }

    const idempotencyKey = `${Date.now()}-${Math.random().toString(36).slice(2, 9)}`;
    lastIdempotencyRef.current = idempotencyKey;

    // optimistic local feedback
    toast.loading(`📍 Clock ${type === 'in' ? 'in' : 'out'}…`, { id: 'punch-status' });

    const payload = {
      type,
      timestamp: new Date().toISOString(),
      location,
      idempotencyKey,
    };

    try {
      inflightRef.current = postPunch(payload);
      await inflightRef.current;
      toast.success(`✅ Clocked ${type === 'in' ? 'in' : 'out'}`, { id: 'punch-status', duration: 2500 });
    } catch (err: any) {
      // queue for retry if offline-ish (best-effort)
      try {
        const queue = JSON.parse(localStorage.getItem('punchQueue') || '[]');
        queue.push(payload);
        localStorage.setItem('punchQueue', JSON.stringify(queue));
        toast('⚠️ Saved locally — will retry when online', { id: 'punch-status', duration: 3000 });
      } catch {
        toast.error('❌ Punch failed — please retry', { id: 'punch-status' });
      }
    } finally {
      inflightRef.current = null;
    }
  }, []);

  return { punch, lastIdempotency: lastIdempotencyRef.current };
}
```

### 3.2 Non-blocking GPS + TimeClock component

```tsx
// workio/src/components/TimeClock.tsx
import { useState, useCallback } from 'react';
import { usePunch } from '../hooks/usePunch';
import { toast } from 'react-hot-toast';

export function TimeClock() {
  const [acquiring, setAcquiring] = useState(false);
  const { punch } = usePunch();

  const getLocation = useCallback(
    (): Promise<GeolocationCoordinates | undefined> =>
      new Promise((resolve) => {
        if (!navigator.geolocation) {
          resolve(undefined);
          return;
        }
        setAcquiring(true);
        navigator.geolocation.getCurrentPosition(
          (pos) => {
            setAcquiring(false);
            resolve(pos.coords);
          },
          () => {
            setAcquiring(false);
            // soft fail: allow punch without GPS (backend may enforce)
            resolve(undefined);
          },
          { timeout: 8000, maximumAge: 30000 }
        );
      }),
    []
  );

  const handlePunch = async (type: 'in' | 'out') => {
    const coords = await getLocation();
    await punch(type, coords);
  };

  return (
    <div className="flex flex-col items-center gap-4 p-6">
      <div className="text-sm text-gray-500">
        {acquiring ? '📍 Getting location…' : 'Tap to clock in/out'}
      </div>
      <div className="flex gap-4">
        <button
          onClick={() => handlePunch('in')}
          className="rounded-lg bg-green-600 px-6 py-3 font-semibold text-white shadow hover:bg-green-700 focus:outline-none focus:ring-2 focus:ring-green-500"
        >
          Clock In
        </button>
        <button
          onClick={() => handlePunch('out')}
          className="rounded-lg bg-red-600 px-6 py-3 font-semibold text-white shadow hover:bg-red-700 focus:outline-none focus:ring-2 focus:ring-red-500"
        >
          Clock Out
        </button>
      </div>
    </div>
  );
}
```

### 3.3 API helper with idempotency header

```ts
// workio/src/lib/api.ts
export async function postPunch(payload: {
  type: 'in' | 'out';
  timestamp: string;
  location?: GeolocationCoordinates;
  idempotencyKey: string;
}) {
  const res = await fetch('/api/punch', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Idempotency-Key': payload.idempotencyKey,
    },
    body: JSON.stringify(payload),
  });

  if (!res.ok) {
    const err = new Error('Punch failed') as any;
    err.status = res.status;
    throw err;
  }
  return res.json();
}
```

### 3.4 Background retry for offline queue (simple)

```ts
// workio/src/lib/retryQueue.ts — call once at app startup
export function startPunchQueueRetry() {
  const process = async () => {
    const raw = localStorage.getItem('punchQueue');
    if (!raw) return;
    const queue = JSON.parse(raw);
    if (!queue.length) return;

    const remaining: typeof queue = [];
    for (const item of queue) {
      try {
        await fetch('/api/punch', {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json',
            'Idempotency-Key': item.idempotencyKey,
          },
          body: JSON.stringify(item),
        });
      } catch {
        remaining.push(item);
      }
    }
    localStorage.setItem('punchQueue', JSON.stringify(remaining));
  };

  // run immediately and then every 30s while online
  process();
  window.addEventListener('online', process);
  setInterval(process, 30000);
}
```
