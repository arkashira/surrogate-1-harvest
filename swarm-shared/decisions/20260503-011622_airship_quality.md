### 1. CDN-bypass file lister + downloader

```ts
// src/lib/hf-cdn.ts
import { HfApi } from "@huggingface/hub";

const HF_API = new HfApi({ token: process.env.NEXT_PUBLIC_HF_TOKEN });

export interface DatasetFile {
  path: string;
  size: number;
  type: "file" | "directory";
}

export async function listRepoFilesOnce(
  repo: string,
  path: string = ""
): Promise<{ folder: string; files: DatasetFile[] }> {
  const tree = await HF_API.listRepoTree({
    repo,
    path,
    recursive: false,
  });
  const files = tree.filter((item) => item.type === "file") as DatasetFile[];
  return { folder: path, files };
}

export function downloadViaCdn(repo: string, path: string): string {
  // CDN bypass: no Authorization header, no API rate limits
  return `https://huggingface.co/datasets/${repo}/resolve/main/${path}`;
}
```

### 2. Parquet projection to `{prompt, response}` (schema guard)

```ts
// src/lib/dataset/projection.ts
import * as arrow from "apache-arrow";

export function projectToPromptResponse(
  table: arrow.Table
): Array<{ prompt: string; response: string }> {
  const prompts = table.getChild("prompt");
  const responses = table.getChild("response");

  if (!prompts || !responses) {
    throw new Error("Parquet must contain 'prompt' and 'response' columns");
  }

  const out: Array<{ prompt: string; response: string }> = [];
  for (let i = 0; i < table.numRows; i++) {
    out.push({
      prompt: String(prompts.get(i) ?? ""),
      response: String(responses.get(i) ?? ""),
    });
  }
  return out;
}

export function validateSchema(schema: arrow.Schema): boolean {
  const names = schema.fields.map((f) => f.name);
  return names.includes("prompt") && names.includes("response");
}
```

### 3. Studio reuse hook (corrected + actionable)

```ts
// src/lib/lightning/studio-reuse.ts
import { Lightning } from "@lightningai/sdk";

export async function getOrCreateStudio(
  name: string,
  machine: string = "L40S"
) {
  const studios = await Lightning.Teamspace.studios();

  // Prefer running studio
  const running = studios.find(
    (s) => s.name === name && s.status === "Running"
  );
  if (running) return running;

  // Restart stopped studio to avoid idle-timeout churn
  const stopped = studios.find(
    (s) => s.name === name && s.status === "Stopped"
  );
  if (stopped) {
    await stopped.start({ machine });
    return stopped;
  }

  // Create new only if none exist
  return Lightning.Studio.create({
    name,
    machine,
    createOk: true,
  });
}
```

### 4. Training UI integration (final)

```tsx
// src/components/TrainingJobForm.tsx
import { useState, useEffect } from "react";
import {
  listRepoFilesOnce,
  downloadViaCdn,
} from "@/lib/hf-cdn";
import { getOrCreateStudio } from "@/lib/lightning/studio-reuse";

export function TrainingJobForm() {
  const [files, setFiles] = useState<Array<{ path: string; size: number }>>([]);
  const [studio, setStudio] = useState<any>(null);
  const [loading, setLoading] = useState(false);
  const [useCdn, setUseCdn] = useState(true);

  useEffect(() => {
    async function loadFiles() {
      try {
        const { files } = await listRepoFilesOnce(
          "org/surrogate-datasets",
          "batches/mirror-merged/2026-04-29"
        );
        setFiles(files);
      } catch (error) {
        console.error("Failed to list dataset files", error);
      }
    }
    loadFiles();
  }, []);

  const handleStartTraining = async () => {
    setLoading(true);
    try {
      const studio = await getOrCreateStudio("surrogate-training");
      setStudio(studio);

      const fileUrls = files.map((f) =>
        useCdn
          ? downloadViaCdn("org/surrogate-datasets", f.path)
          : `hf://datasets/org/surrogate-datasets/${f.path}`
      );

      await studio.run({
        script: "train.py",
        args: ["--data-files", JSON.stringify(fileUrls)],
      });
    } catch (error) {
      console.error("Training failed", error);
    } finally {
      setLoading(false);
    }
  };

  return (
    <div className="p-4 space-y-4">
      <h2 className="text-xl font-semibold">Training Job</h2>

      <label className="flex items-center gap-2">
        <input
          type="checkbox"
          checked={useCdn}
          onChange={(e) => setUseCdn(e.target.checked)}
        />
        Use CDN-bypass (no API limits)
      </label>

      <div>
        <h3 className="font-medium">Dataset Files ({files.length})</h3>
        <ul className="max-h-48 overflow-y-auto border rounded p-2 text-sm">
          {files.map((f) => (
            <li key={f.path} className="flex justify-between">
              <span>{f.path}</span>
              <span className="text-muted-foreground">{f.size} bytes</span>
            </li>
          ))}
        </ul>
      </div>

      <button
        onClick={handleStartTraining}
        disabled={loading || files.length === 0}
        className="px-4 py-2 bg-primary text-primary-foreground rounded disabled:opacity-50"
      >
        {loading ? "Starting..." : "Start Training"}
      </button>

      {studio && (
        <div className="text-sm text-muted-foreground">
          Studio: {studio.name} — Status: {studio.status}
        </div>
      )}
    </div>
  );
}
```
