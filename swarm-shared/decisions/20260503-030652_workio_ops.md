# workio / ops

## 1. Diagnosis

- **LINE webhook replay vulnerability**: no `X-Line-Signature` verification and no idempotency layer allows duplicate clock events when LINE retries (5xx/timeouts).
- **Race condition on open punches**: concurrent requests for same `(tenant_id, employee_id)` with nullable `clock_out_at` can create multiple active punches.
- **No uniqueness constraint**: database lacks `(tenant_id, employee_id, clock_out_at)` partial unique index to enforce at most one open punch.
- **No idempotency key**: webhook handler does not store/lookup idempotency keys (e.g., `X-Line-Delivery` or generated nonce) to deduplicate within a time window.
- **Missing audit trail**: no `created_at` precision or request-id correlation for ops debugging of duplicate events.

## 2. Proposed change

- **File scope**: `/opt/axentx/workio/server/src/routes/webhook/line.ts` (or equivalent webhook handler) and `/opt/axentx/workio/server/src/db/schema.sql`.
- **Changes**:
  1. Add partial unique index on punches for open punch constraint.
  2. Add idempotency table (key, payload hash, created_at) with TTL.
  3. Verify `X-Line-Signature` on incoming requests.
  4. Use idempotency key (`X-Line-Delivery` or generated) to deduplicate before processing.
  5. Wrap punch creation in `ON CONFLICT DO NOTHING` + application-level lock (advisory or row-level) to prevent races.

```sql
-- Ensure only one open punch per tenant+employee
CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_open_unique
ON punches (tenant_id, employee_id)
WHERE clock_out_at IS NULL;

-- Idempotency keys for LINE webhook deduplication (7d TTL)
CREATE TABLE IF NOT EXISTS webhook_idempotency (
  idempotency_key VARCHAR(255) PRIMARY KEY,
  payload_hash     VARCHAR(64) NOT NULL,
  created_at       TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
CREATE INDEX IF NOT EXISTS idx_webhook_idempotency_created
ON webhook_idempotency (created_at);
```

### 3.2 Webhook handler (add signature verification + idempotency)

```ts
// /opt/axentx/workio/server/src/routes/webhook/line.ts
import crypto from 'crypto';
import express from 'express';
import { pool } from '../../db';
import { verifyLineSignature } from '../../utils/lineCrypto';

const router = express.Router();

const LINE_CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';
const IDEMPOTENCY_TTL_MS = 7 * 24 * 60 * 60 * 1000; // 7d

function computePayloadHash(body: any): string {
  return crypto.createHash('sha256').update(JSON.stringify(body)).digest('hex');
}

async function isDuplicate(key: string, hash: string): Promise<boolean> {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');
    // Lock row for update to prevent race between select/insert
    const existing = await client.query(
      `SELECT payload_hash FROM webhook_idempotency WHERE idempotency_key = $1 FOR UPDATE`,
      [key]
    );
    if (existing.rows.length > 0) {
      await client.query('ROLLBACK');
      return existing.rows[0].payload_hash === hash;
    }
    await client.query(
      `INSERT INTO webhook_idempotency (idempotency_key, payload_hash) VALUES ($1, $2)`,
      [key, hash]
    );
    await client.query('COMMIT');
    return false;
  } catch (e) {
    await client.query('ROLLBACK');
    throw e;
  } finally {
    client.release();
  }
}

async function cleanupOldIdempotency() {
  await pool.query(
    `DELETE FROM webhook_idempotency WHERE created_at < NOW() - ($1 || ' ms')::INTERVAL`,
    [IDEMPOTENCY_TTL_MS.toString()]
  );
}

router.post('/line', async (req, res) => {
  const signature = req.headers['x-line-signature'] as string;
  const delivery = req.headers['x-line-delivery'] as string || crypto.randomUUID();
  const body = req.body;

  // 1) Verify signature
  if (!verifyLineSignature(LINE_CHANNEL_SECRET, JSON.stringify(body), signature)) {
    return res.status(401).json({ error: 'Invalid signature' });
  }

  // 2) Idempotency check
  const payloadHash = computePayloadHash(body);
  try {
    const duplicate = await isDuplicate(delivery, payloadHash);
    if (duplicate) {
      return res.status(200).json({ ok: true, reason: 'duplicate' });
    }
  } catch (e) {
    // If idempotency store fails, continue cautiously (log) but process to avoid outage
    console.warn('Idempotency store failure, processing anyway', e);
  }

  // 3) Process events with application-level advisory lock per tenant+employee
  try {
    for (const event of body.events || []) {
      if (event.type !== 'message' && event.type !== 'postback') continue;
      const { userId, source } = event;
      if (!source?.userId || !userId) continue;

      // Map LINE userId -> internal employee (implementation-specific)
      const employeeRow = await pool.query(
        `SELECT id, tenant_id FROM employees WHERE line_user_id = $1 LIMIT 1`,
        [userId]
      );
      if (employeeRow.rows.length === 0) continue;
      const { id: employeeId, tenant_id: tenantId } = employeeRow.rows[0];

      // Advisory lock to serialize concurrent punches for same employee
      const lockKey = ((BigInt(tenantId) << 32n) | BigInt(employeeId)) % BigInt(2147483647);
      await pool.query('SELECT pg_advisory_xact_lock($1)', [lockKey.toString()]);

      // Check for existing open punch
      const openPunch = await pool.query(
        `SELECT id FROM punches WHERE tenant_id = $1 AND employee_id = $2 AND clock_out_at IS NULL LIMIT 1`,
        [tenantId, employeeId]
      );

      if (openPunch.rows.length > 0) {
        // Clock out
        await pool.query(
          `UPDATE punches SET clock_out_at = NOW(), updated_at = NOW() WHERE id = $1`,
          [openPunch.rows[0].id]
        );
      } else {
        // Clock in (ignore if another concurrent insert happens — unique index protects)
        await pool.query(
          `INSERT INTO punches (tenant_id, employee_id, clock_in_at, created_at, updated_at)
           VALUES ($1, $2, NOW(), NOW(), NOW())
           ON CONFLICT DO NOTHING`,
          [tenantId, employeeId]
        );
      }
    }

    res.status(200).json({ ok: true });
  } catch (err) {
    console.error('LINE webhook processing error', err);
    res.status(500).json({ error: 'processing_error' });
  }
});

// Optional periodic cleanup (run via cron or setInterval in dev)
if (process.env.NODE_ENV === 'development') {
  setInterval(cleanupOldIdempotency, 60 * 60 * 1000);
}

export default router;
```

### 3.3 Crypto utility for signature verification

```ts
// /opt/axentx/workio/server/src/utils/lineCrypto.ts
import crypto from 'crypto';

export function verifyLineSignature(secret: string, body: string, signature: string): boolean {
  if (!secret || !signature) return false;
  const expected = crypto
    .createHmac('sha256', secret)
    .update(body, 'utf8')
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}
```
