# vanguard / ops

## Final Synthesis (single, correct, actionable)

**Core problem**: training repeatedly calls authenticated HF APIs (quota + 429 risk) and tries to stream heterogeneous repos directly, causing `pyarrow.CastError` schema mismatches.  
**Goal**: eliminate authenticated calls during training, enforce uniform `{prompt,response}` schema, persist a `(repo,dateFolder)→file-list` manifest, and make training resilient to studio idle-stop.

---

### 1. Manifest service (build once, reuse everywhere)

Create `/opt/axentx/vanguard/ops/training_manifest.py`:

```python
# /opt/axentx/vanguard/ops/training_manifest.py
import json
import os
from pathlib import Path
from typing import Dict, List, Tuple, Iterator
import requests
from huggingface_hub import list_repo_tree, HfApi

MANIFEST_DIR = Path(__file__).parent.parent / "manifests"
MANIFEST_DIR.mkdir(exist_ok=True)
HF_API = HfApi()

def _list_dir_safe(repo: str, folder: str) -> List[Dict]:
    return list(list_repo_tree(repo=repo, path=folder, recursive=False))

def build_manifest(repo: str, date_folder: str, out_path: Path = None) -> Path:
    """
    Single shallow walk under date_folder.
    Persists manifest for CDN-only training.
    """
    top = _list_dir_safe(repo, date_folder)
    file_paths: List[str] = []

    for entry in top:
        if entry.type == "file":
            file_paths.append(f"{date_folder}/{entry.path}")
        elif entry.type == "dir":
            sub = _list_dir_safe(repo, entry.path)
            for f in sub:
                if f.type == "file":
                    file_paths.append(f"{entry.path}/{f.path}")

    manifest = {
        "repo": repo,
        "date_folder": date_folder,
        "files": sorted(set(file_paths)),
        "cdn_prefix": f"https://huggingface.co/datasets/{repo}/resolve/main"
    }

    out_path = out_path or (MANIFEST_DIR / f"{repo.replace('/', '_')}_{date_folder}.json")
    out_path.parent.mkdir(parents=True, exist_ok=True)
    out_path.write_text(json.dumps(manifest, indent=2))
    return out_path


def load_manifest(repo: str, date_folder: str) -> Dict:
    p = MANIFEST_DIR / f"{repo.replace('/', '_')}_{date_folder}.json"
    if not p.exists():
        raise FileNotFoundError(f"Manifest missing: {p}. Run build_manifest first.")
    return json.loads(p.read_text())


def cdn_url(repo: str, path: str) -> str:
    return f"https://huggingface.co/datasets/{repo}/resolve/main/{path}"


def download_file_cdn(repo: str, path: str, dst: Path) -> Path:
    url = cdn_url(repo, path)
    r = requests.get(url, timeout=60)
    r.raise_for_status()
    dst.parent.mkdir(parents=True, exist_ok=True)
    dst.write_bytes(r.content)
    return dst


def project_to_pair(raw_bytes: bytes, path: str) -> Dict[str, str]:
    """
    Best-effort projection to {prompt,response}.
    Accepts JSONL with prompt/response (or input/output) keys.
    Falls back to filename-based stub.
    """
    import io
    try:
        lines = io.TextIOWrapper(io.BytesIO(raw_bytes), encoding="utf-8")
        for line in lines:
            line = line.strip()
            if not line:
                continue
            obj = json.loads(line)
            return {
                "prompt": obj.get("prompt") or obj.get("input") or "",
                "response": obj.get("response") or obj.get("output") or ""
            }
        return {"prompt": "", "response": ""}
    except Exception:
        return {"prompt": f"Extract from {path}", "response": raw_bytes.decode("utf-8", errors="replace")[:2000]}


def cdn_dataset(manifest: Dict) -> Iterator[Dict[str, str]]:
    """
    Generator suitable for DataLoader workers.
    Each worker should map CDN URLs -> tensors; this yields parsed pairs.
    """
    repo = manifest["repo"]
    for f in manifest["files"]:
        url = cdn_url(repo, f)
        r = requests.get(url, timeout=60)
        r.raise_for_status()
        yield project_to_pair(r.content, f)
```

### 4. What this fixes

- **No authenticated `list_repo_tree` during training** → manifest is built once; training uses CDN URLs only.
- **No `pyarrow.CastError` from heterogeneous repos** → each file is projected to uniform `{prompt,response}` at parse time.
- **No wasted studio quota** → reuse running studio; restart automatically if idle-stop occurs.
- **Actionable and minimal diff** → one new ops module, small changes to entrypoint, one pre-train manifest build step.
