## Final Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16`, `DATE`, `HF_TOKEN` via env
- Single `list_repo_tree` call per date folder → deterministic file list
- Worker keeps only its shard (`hash(filename) % 16 == SHARD_ID`)
- Downloads via HF CDN (`resolve/main/...`) **without Authorization header** to bypass API rate limits
- Projects each file to `{prompt, response}` at parse time (avoids pyarrow CastError on mixed schemas)
- Dedups via central md5 store (`lib/dedup.py`)
- Writes `batches/public-merged/{DATE}/shard{N}-{HHMMSS}.jsonl`
- GitHub Actions matrix: 16 shards; each runner uses the same pre-generated manifest
- Adds robust retry/backoff: 429 → wait 360s; 403/5xx → exponential backoff

def deterministic_shard(key: str, total: int) -> int:
    return int(hashlib.md5(key.encode()).hexdigest(), 16) % total

def list_date_files(date_folder: str) -> List[str]:
    """
    Single API call to list files in date folder (non-recursive).
    Returns relative paths under repo root.
    """
    try:
        tree = API.list_repo_tree(
            repo_id=REPO,
            path=date_folder,
            repo_type="dataset",
        )
        items = list(tree) if not isinstance(tree, list) else tree
        files = [item.rfilename for item in items if item.type == "file"]
        return files
    except Exception as e:
        print(f"list_repo_tree failed: {e}", file=sys.stderr)
        sys.exit(1)

def download_via_cdn_with_retry(rel_path: str, max_retries: int = 5) -> bytes:
    url = f"{BASE_CDN}/{rel_path}"
    for attempt in range(max_retries):
        try:
            # CDN fetch: no Authorization header -> bypasses /api/ rate limits
            resp = requests.get(url, timeout=30)
            if resp.status_code == 429:
                wait = 360
                print(f"429 on {rel_path}, waiting {wait}s")
                time.sleep(wait)
                continue
            if resp.status_code in (403, 500, 502, 503, 504):
                wait = 2 ** attempt
                print(f"{resp.status_code} on {rel_path}, waiting {wait}s")
                time.sleep(wait)
                continue
            resp.raise_for_status()
            return resp.content
        except requests.RequestException as exc:
            if attempt == max_retries - 1:
                raise
            wait = 2 ** attempt
            print(f"Request error {rel_path}: {exc}, retrying in {wait}s")
            time.sleep(wait)
    raise RuntimeError(f"Failed to download {rel_path} after {max_retries} retries")

def parse_to_pair(content: bytes, filename: str) -> Dict[str, str]:
    """
    Project raw file to {prompt, response} only.
    Supports common repo patterns (JSONL, JSON, text).
    """
    text = content.decode("utf-8", errors="replace").strip()
    if not text:
        return {"prompt": "", "response": ""}

    # JSONL: first line
    if filename.endswith(".jsonl"):
        first = json.loads(text.splitlines()[0])
        prompt = first.get("prompt") or first.get("input") or first.get("question") or ""
        response = first.get("response") or first.get("output") or first.get("answer") or ""
        return {"prompt": str(prompt), "response": str(response)}

    # JSON object
    if filename.endswith(".json"):
        obj = json.loads(text)
        prompt = obj.get("prompt") or obj.get("input") or obj.get("question") or ""
        response = obj.get("response") or obj.get("output") or obj.get("answer") or ""
        return {"prompt": str(prompt), "response": str(response)}

    # Plain text: treat first non-empty line as prompt, rest as response
    lines = [ln.strip() for ln in text.splitlines() if ln.strip()]
    if len(lines) == 1:
        return {"prompt": lines[0], "response": ""}
    return {"prompt": lines[0], "response": "\n".join(lines[1:])}
