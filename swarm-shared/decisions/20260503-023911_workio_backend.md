# workio / backend

## Final Synthesized Solution

### 1. Diagnosis (merged, prioritized by correctness)
- **Missing DB-level uniqueness** on punches allows duplicates when LINE retries webhooks.
- **Application-level race conditions**: two near-simultaneous webhooks can pass “last punch” checks and both insert.
- **No idempotency persistence** for LINE `deliveryId` (or composite key) means retries create duplicate punches.
- **No enforced state machine** for clock-in → clock-out transitions at DB or service layer; invalid transitions can be persisted.
- **Missing pre-insert validation** (GPS/schedule rules) causes invalid punches that require later cleanup.

### 2. Scope of Changes (concrete, minimal, high-leverage)
- **DB schema** (`workio/server/src/db/schema.sql`):
  - Idempotency table keyed by LINE delivery key.
  - Unique constraint/index on punches to prevent duplicates at DB level.
- **Service layer** (`workio/server/src/services/punchService.ts`):
  - Transactional upsert with idempotency check + state transition guard.
  - Enforce “no double clock-in” and “no clock-out without prior clock-in”.
- **Webhook route** (`workio/server/src/routes/lineWebhook.ts`):
  - Verify signature, derive deterministic idempotency key, call service.
  - Return 200 for duplicates/already-processed; 4xx/5xx only for real errors.

### 3. Implementation (resolved contradictions + actionability)

#### 3.1 DB schema (`workio/server/src/db/schema.sql`)
```sql
-- Idempotency table for LINE webhook deliveries (DB-level dedupe)
CREATE TABLE IF NOT EXISTS line_webhook_deliveries (
  delivery_key TEXT PRIMARY KEY,
  created_at    TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Prevent duplicate punches for same tenant/user/date/type at DB level.
-- This is the source-of-truth guard against duplicates and races.
ALTER TABLE punches
  ADD CONSTRAINT uniq_tenant_user_date_type
  UNIQUE (tenant_id, user_id, punch_date, punch_type);

-- Fast lookup for service-layer checks
CREATE INDEX IF NOT EXISTS idx_punches_user_date_type
  ON punches (tenant_id, user_id, punch_date, punch_type);
```

#### 3.2 Service layer (`workio/server/src/services/punchService.ts`)
```ts
import { db } from '../db';
import { punches, lineWebhookDeliveries } from '../db/schema';
import { and, eq, sql } from 'drizzle-orm';
import { PunchType } from '../types';

export async function upsertPunchWithIdempotency(params: {
  tenantId: string;
  userId: string;
  punchType: PunchType;
  punchDate: string; // YYYY-MM-DD
  timestamp: Date;
  location?: { lat: number; lng: number };
  deliveryKey: string;
}) {
  return db.transaction(async (tx) => {
    // 1) Idempotency check (DB primary key ensures uniqueness under concurrency)
    const existingDelivery = await tx
      .select()
      .from(lineWebhookDeliveries)
      .where(eq(lineWebhookDeliveries.delivery_key, params.deliveryKey))
      .limit(1);

    if (existingDelivery.length > 0) {
      // Already processed: return the punch that was created for this delivery
      const existing = await tx
        .select()
        .from(punches)
        .where(
          and(
            eq(punches.tenant_id, params.tenantId),
            eq(punches.user_id, params.userId),
            eq(punches.punch_date, params.punchDate),
            eq(punches.punch_type, params.punchType)
          )
        )
        .limit(1);
      return existing[0] || null;
    }

    // 2) Record delivery first (so concurrent retries will hit idempotency)
    await tx.insert(lineWebhookDeliveries).values({
      delivery_key: params.deliveryKey,
    });

    // 3) Transition guard: enforce valid state machine
    const latest = await tx
      .select()
      .from(punches)
      .where(
        and(
          eq(punches.tenant_id, params.tenantId),
          eq(punches.user_id, params.userId),
          eq(punches.punch_date, params.punchDate)
        )
      )
      .orderBy(punches.created_at)
      .limit(1);

    const lastType = latest.length > 0 ? latest[0].punch_type : null;

    if (params.punchType === 'clock_in' && lastType === 'clock_in') {
      throw new Error('Duplicate clock-in not allowed');
    }
    if (params.punchType === 'clock_out' && lastType !== 'clock_in') {
      throw new Error('Clock-out requires prior clock-in');
    }

    // 4) Insert punch (DB unique constraint is final safeguard)
    const [punch] = await tx
      .insert(punches)
      .values({
        tenant_id: params.tenantId,
        user_id: params.userId,
        punch_type: params.punchType,
        punch_date: params.punchDate,
        timestamp: params.timestamp,
        location: params.location ? JSON.stringify(params.location) : null,
      })
      .returning();

    return punch;
  });
}
```

#### 3.3 Webhook route (`workio/server/src/routes/lineWebhook.ts`)
```ts
import { Router } from 'express';
import { upsertPunchWithIdempotency } from '../services/punchService';
import { verifyLineSignature } from '../utils/line';

const router = Router();

router.post('/webhook/line', async (req, res) => {
  const sig = req.headers['x-line-signature'] as string;
  const body = req.body;

  if (!verifyLineSignature(JSON.stringify(body), sig)) {
    return res.status(401).send('Invalid signature');
  }

  try {
    for (const event of body.events) {
      if (event.type !== 'message' && event.type !== 'postback') continue;

      // Deterministic idempotency key from LINE fields
      const deliveryKey = `line:${event.source.userId}:${event.timestamp}:${event.type}:${event.message?.id || event.postback?.data || 'default'}`;

      // Resolve tenant from your mapping (example uses default)
      const tenantId = 'default';
      const userId = event.source.userId;
      const punchDate = new Date(event.timestamp).toISOString().split('T')[0];

      // Map event to punch type (adjust to your payload)
      const punchType = event.postback?.data === 'clock_in' ? 'clock_in' : 'clock_out';

      await upsertPunchWithIdempotency({
        tenantId,
        userId,
        punchType,
        punchDate,
        timestamp: new Date(event.timestamp),
        location: undefined, // parse from postback/message if available
        deliveryKey,
      });
    }

    return res.status(200).send('OK');
  } catch (err: any) {
    // Idempotency/validation errors from retries -> 200 so LINE stops retrying
    if (err.message && /(Duplicate|already|not allowed|requires prior)/i.test(err.message)) {
      return res.status(200).send('Already processed');
    }
    console.error(err);
    return res.status(500).send('Server error');
  }
});

export default router;
```
