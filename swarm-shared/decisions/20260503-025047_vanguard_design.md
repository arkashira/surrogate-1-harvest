# vanguard / design

## Final synthesized implementation (correct + actionable)

**Core diagnosis (merged, de-duplicated)**
- No frontend entrypoint or mount → nothing renders and no local iteration/HMR.
- No build/dev tooling → no deterministic asset hashing, fragile runtime path resolution.
- No manifest → frontend must call HF API at runtime (rate-limit + availability risk).
- No CDN-first data strategy wired into UI → training/inference and browsing hit HF API.
- Missing Mac→browser orchestration → no deterministic file list for CDN fetches.

**Chosen strategy**
- Minimal Vite frontend (index.html + src/) with CDN-only dataset links.
- Build-time manifest generated on Mac (date folder → files.json) so the browser never calls the HF datasets API.
- Frontend loads `manifest.json` and builds `https://huggingface.co/datasets/.../resolve/main/...` URLs only.
- One small Mac-side script to produce the manifest; no secrets or API calls from the browser.

### src/App.js
```js
const DATASET_OWNER = 'axentx';
const DATASET_REPO = 'vanguard-data';
const MANIFEST_PATH = '/manifest.json';

function el(tag, attrs = {}, ...children) {
  const node = document.createElement(tag);
  for (const [k, v] of Object.entries(attrs)) {
    if (k === 'className') node.className = v;
    else if (k.startsWith('on') && typeof v === 'function') node.addEventListener(k.slice(2).toLowerCase(), v);
    else node.setAttribute(k, v);
  }
  for (const c of children) {
    if (c == null) continue;
    node.append(typeof c === 'string' ? document.createTextNode(c) : c);
  }
  return node;
}

export default function App() {
  const root = el('div', { className: 'app-root' },
    el('header', {},
      el('h1', {}, 'Vanguard'),
      el('p', {}, 'CDN-first dataset browser')
    ),
    el('div', { className: 'status', id: 'status' }, 'Loading manifest...'),
    el('ul', { className: 'file-list', id: 'fileList' })
  );

  const statusEl = root.querySelector('#status');
  const listEl = root.querySelector('#fileList');

  function setStatus(msg, type = '') {
    statusEl.textContent = msg;
    statusEl.className = `status ${type}`.trim();
  }

  async function loadManifest() {
    try {
      const res = await fetch(MANIFEST_PATH, { cache: 'no-cache' });
      if (!res.ok) throw new Error(`HTTP ${res.status}`);
      return await res.json(); // expected: { date: "...", files: ["path/a.parquet", ...] }
    } catch (err) {
      setStatus('No manifest — run generate-manifest.js on Mac', 'error');
      return null;
    }
  }

  function cdnUrl(filePath) {
    return `https://huggingface.co/datasets/${DATASET_OWNER}/${DATASET_REPO}/resolve/main/${encodeURIComponent(filePath)}`;
  }

  async function renderFiles(manifest) {
    listEl.innerHTML = '';
    if (!manifest || !Array.isArray(manifest.files) || manifest.files.length === 0) {
      listEl.appendChild(el('li', {}, 'No files listed in manifest.'));
      return;
    }

    for (const filePath of manifest.files) {
      const url = cdnUrl(filePath);
      const li = el('li', {},
        el('a', { href: url, target: '_blank', rel: 'noopener', className: 'file-link' }, filePath),
        ' ',
        el('button', { className: 'head-btn' }, 'HEAD')
      );

      li.querySelector('.head-btn').addEventListener('click', async () => {
        setStatus(`Checking ${filePath}...`, 'info');
        try {
          const res = await fetch(url, { method: 'HEAD', mode: 'cors' });
          setStatus(`${filePath}: ${res.status} ${res.statusText}`, res.ok ? 'ok' : 'error');
        } catch (err) {
          setStatus(`${filePath}: network/CORS error`, 'error');
        }
      });

      listEl.appendChild(li);
    }

    if (manifest.date) {
      const meta = el('small', {}, `Manifest date: ${manifest.date}`);
      root.appendChild(meta);
    }
  }

  (async () => {
    const manifest = await loadManifest();
    await renderFiles(manifest);
  })();

  return root;
}
```
