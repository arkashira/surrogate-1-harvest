# workio / design

## Final synthesized answer

### Diagnosis (merged)
- LINE webhook retries cause duplicate punches because ingestion has no idempotency at the message-id level (no dedup table or idempotency-key enforcement).
- Punch upsert is non-atomic (`findOne` → conditional `insert`/`update`) across `(tenant, employee, date, event)` and is vulnerable to races under concurrency.
- No unique constraint or partial index enforces one active clock-in per employee per day, so duplicates can persist.
- Tenant isolation in punch writes is application-only and not enforced by schema, increasing risk of cross-tenant leakage.
- Clock-in/out endpoint mixes validation, business rules, and persistence, making idempotency and retries hard to reason about.

### Proposed changes (merged)
- **Schema** (`workio/server/src/db/schema.sql`): add idempotency table + partial unique index for open punches + FK for tenant isolation.
- **Idempotency middleware** (`workio/server/src/middleware/idempotency.ts`): new middleware to deduplicate LINE webhooks using `x-line-message-id` (or generated key) with transactional insert into `line_message_dedup`.
- **Punch controller** (`workio/server/src/controllers/punchController.ts`): replace non-atomic upsert with atomic `INSERT ... ON CONFLICT DO NOTHING` (clock-in) and targeted `UPDATE ... RETURNING` (clock-out) inside a transaction; reuse idempotency transaction when available.

### Implementation (merged + hardened)

#### 1) Schema (`workio/server/src/db/schema.sql`)
```sql
-- Idempotency table for LINE webhook messages (tenant-scoped)
CREATE TABLE IF NOT EXISTS line_message_dedup (
  tenant_id   uuid NOT NULL,
  message_id  text NOT NULL,        -- LINE x-line-message-id or generated idempotency key
  handler     text NOT NULL,        -- e.g. 'punch'
  created_at  timestamptz DEFAULT now(),
  PRIMARY KEY (tenant_id, message_id)
);

-- One active clock-in per employee per calendar day (tenant-scoped)
-- Allows multiple clock-outs after a clock-in; prevents second clock-in without intervening clock-out
CREATE UNIQUE INDEX IF NOT EXISTS idx_punch_one_clock_in_per_day
  ON punches (tenant_id, employee_id, date(clock_in_at))
  WHERE clock_out_at IS NULL;

-- Enforce tenant isolation at schema level
ALTER TABLE punches
  ADD CONSTRAINT fk_punches_tenant
  FOREIGN KEY (tenant_id) REFERENCES tenants(id) ON DELETE CASCADE;
```

#### 2) Idempotency middleware (`workio/server/src/middleware/idempotency.ts`)
```ts
import { Request, Response, NextFunction } from 'express';
import { pool } from '../db';

export type IdempotencyOpts = {
  handler: string;
  getId?: (req: Request) => string | null;
};

export function lineIdempotency(opts: IdempotencyOpts) {
  const { handler, getId } = opts;

  return async (req: Request, res: Response, next: NextFunction) => {
    const tenantId = (req as any).tenantId ?? (req.body as any)?.tenantId;
    const messageId = (getId?.(req) || req.header('x-line-message-id') || req.header('x-delivery-id'))?.trim();

    if (!tenantId || !messageId) {
      // No idempotency key: skip dedup but continue safely
      return next();
    }

    const client = await pool.connect();
    try {
      await client.query('BEGIN');
      const dedup = await client.query(
        `SELECT 1 FROM line_message_dedup WHERE tenant_id = $1 AND message_id = $2 FOR UPDATE`,
        [tenantId, messageId]
      );

      if (dedup.rows.length > 0) {
        await client.query('ROLLBACK');
        client.release();
        // Return 200 so LINE stops retrying
        return res.status(200).json({ ok: true, reason: 'duplicate' });
      }

      await client.query(
        `INSERT INTO line_message_dedup (tenant_id, message_id, handler) VALUES ($1, $2, $3)`,
        [tenantId, messageId, handler]
      );
      await client.query('COMMIT');
      // Attach client to request so handler can reuse same tx
      (req as any)._idemClient = client;
      next();
    } catch (err) {
      await client.query('ROLLBACK').catch(() => {});
      client.release();
      // On unique violation (race), treat as duplicate
      if ((err as any)?.code === '23505') {
        return res.status(200).json({ ok: true, reason: 'duplicate' });
      }
      next(err);
    }
  };
}

export function releaseIdemClient(req: Request, _res: Response, next: NextFunction) {
  const client = (req as any)._idemClient;
  if (client) {
    try { client.release(); } catch { /* ignore */ }
    (req as any)._idemClient = undefined;
  }
  next();
}
```

#### 3) Punch controller (`workio/server/src/controllers/punchController.ts`)
```ts
import { Request, Response } from 'express';
import { pool } from '../db';
import { releaseIdemClient } from '../middleware/idempotency';

// POST /api/punches/clock-in
export async function clockIn(req: Request, res: Response) {
  const { tenantId, employeeId, latitude, longitude } = req.body;
  const now = new Date();

  const client = (req as any)._idemClient || await pool.connect();
  try {
    const shouldManageTx = !(req as any)._idemClient;
    if (shouldManageTx) await client.query('BEGIN');

    // Try insert; unique index prevents double clock-in same day while open
    const insertResult = await client.query(
      `INSERT INTO punches (tenant_id, employee_id, clock_in_at, latitude, longitude)
       VALUES ($1, $2, $3, $4, $5)
       ON CONFLICT DO NOTHING
       RETURNING *`,
      [tenantId, employeeId, now, latitude, longitude]
    );

    let punch = insertResult.rows[0];

    // If conflict (already clocked in), return existing open punch
    if (!punch) {
      const selectResult = await client.query(
        `SELECT * FROM punches
         WHERE tenant_id = $1 AND employee_id = $2 AND clock_out_at IS NULL
         ORDER BY clock_in_at DESC LIMIT 1`,
        [tenantId, employeeId]
      );
      punch = selectResult.rows[0];
    }

    if (shouldManageTx) await client.query('COMMIT');
    return res.status(200).json({ ok: true, punch });
  } catch (err) {
    if (!(req as any)._idemClient) await client.query('ROLLBACK').catch(() => {});
    return res.status(500).json({ ok: false, error: String(err) });
  } finally {
    if (!(req as any)._idemClient) client.release();
  }
}
