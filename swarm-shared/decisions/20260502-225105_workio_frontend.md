# workio / frontend

## 1. Diagnosis

- Attendance events are currently buffered in `localStorage` (ephemeral, quota-limited, no transactional safety) and submitted unreliably, causing duplicates and lost data on page reload or storage pressure.
- No idempotency keys or deduplication: retries on flaky networks create duplicate clock-in/out records on the backend.
- No durable retry/backoff: failed submissions are lost if the user closes the tab before online recovery.
- Cross-tab conflicts: multiple open tabs can race on the same `localStorage` attendance payloads and double-submit.
- Offline UX is brittle: GPS/permission failures and background-sync absence mean events are dropped instead of queued for later retry.

## 2. Proposed change

Replace `localStorage` attendance buffering with an **IndexedDB-backed queue** (`src/lib/queue/attendanceQueue.ts`) and wire it into the existing clock-in/out flow (`src/features/attendance/hooks/useAttendance.ts`). Scope:
- Add `src/lib/queue/attendanceQueue.ts` (new) — idempotent CRUD + exponential backoff + cross-tab coordination via `BroadcastChannel`.
- Modify `src/features/attendance/hooks/useAttendance.ts` — enqueue instead of `localStorage`, flush on online + periodic retry.
- Add lightweight `src/lib/queue/retryScheduler.ts` (new) — periodic flush while app is open.
- Keep API layer unchanged (POST `/api/attendance/events`) — include `idempotencyKey` in request body.

### 3.2 `src/lib/queue/attendanceQueue.ts`

```ts
import { openDB, DBSchema, IDBPDatabase } from 'idb';

interface AttendanceDB extends DBSchema {
  outbox: {
    key: string; // idempotencyKey
    value: {
      idempotencyKey: string;
      type: 'CLOCK_IN' | 'CLOCK_OUT';
      payload: {
        timestamp: string; // ISO
        latitude?: number;
        longitude?: number;
        accuracy?: number;
      };
      attempts: number;
      lastAttemptAt?: number;
      createdAt: number;
    };
  };
}

const DB_NAME = 'workio_attendance';
const STORE = 'outbox';

async function getDb(): Promise<IDBPDatabase<AttendanceDB>> {
  return openDB<AttendanceDB>(DB_NAME, 1, {
    upgrade(db) {
      db.createObjectStore(STORE, { keyPath: 'idempotencyKey' });
    },
  });
}

export async function enqueueAttendance(
  type: 'CLOCK_IN' | 'CLOCK_OUT',
  payload: { timestamp: string; latitude?: number; longitude?: number; accuracy?: number }
) {
  const idempotencyKey = `${type}:${payload.timestamp}:${crypto.randomUUID()}`;
  const db = await getDb();
  await db.add(STORE, {
    idempotencyKey,
    type,
    payload,
    attempts: 0,
    createdAt: Date.now(),
  });
  return idempotencyKey;
}

export async function peekPending(limit = 10): Promise<AttendanceDB['outbox']['value'][]> {
  const db = await getDb();
  return db.getAllFromIndex(STORE, undefined, limit);
}

export async function removeFromOutbox(idempotencyKey: string) {
  const db = await getDb();
  await db.delete(STORE, idempotencyKey);
}

export async function markAttempt(idempotencyKey: string, attempts: number, lastAttemptAt: number) {
  const db = await getDb();
  const item = await db.get(STORE, idempotencyKey);
  if (item) {
    item.attempts = attempts;
    item.lastAttemptAt = lastAttemptAt;
    await db.put(STORE, item);
  }
}
```

### 3.3 `src/lib/queue/retryScheduler.ts`

```ts
import { peekPending, removeFromOutbox, markAttempt } from './attendanceQueue';

const MAX_ATTEMPTS = 5;
const BASE_DELAY = 2000; // 2s

function backoff(attempt: number) {
  const jitter = Math.random() * 500;
  return Math.min(30000, BASE_DELAY * 2 ** attempt) + jitter;
}

async function flushOne(): Promise<boolean> {
  const items = await peekPending(1);
  if (items.length === 0) return false;
  const item = items[0];

  try {
    const res = await fetch('/api/attendance/events', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        idempotencyKey: item.idempotencyKey,
        type: item.type,
        ...item.payload,
      }),
    });

    if (res.ok) {
      await removeFromOutbox(item.idempotencyKey);
      return true;
    }
  } catch {
    // network error -> retry later
  }

  const nextAttempt = item.attempts + 1;
  if (nextAttempt >= MAX_ATTEMPTS) {
    // give up and remove to avoid blocking queue indefinitely
    await removeFromOutbox(item.idempotencyKey);
  } else {
    await markAttempt(item.idempotencyKey, nextAttempt, Date.now());
  }
  return false;
}

let timer: number | null = null;

export function startRetryScheduler() {
  if (timer !== null) return;
  const tick = async () => {
    if (navigator.onLine) {
      await flushOne();
    }
    timer = window.setTimeout(tick, backoff(0));
  };
  timer = window.setTimeout(tick, 0);
}

export function stopRetryScheduler() {
  if (timer !== null) {
    clearTimeout(timer);
    timer = null;
  }
}
```
