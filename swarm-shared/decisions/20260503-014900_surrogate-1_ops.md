## Final Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** that:

- Uses a **single `list_repo_tree` snapshot** (JSON manifest) generated once per date on the Mac orchestrator and committed to the repo (or passed via `MANIFEST_URL`).  
- Downloads **only files assigned to this shard** via raw CDN URLs (`https://huggingface.co/datasets/.../resolve/main/...`) — zero HF API calls during training/ingest.  
- Projects each file to `{prompt, response, source, ts}` at parse time; writes normalized JSONL to `batches/public-merged/<date>/shard<N>-<HHMMSS>.jsonl`.  
- Deduplicates via central md5 store (`lib/dedup.py`) and commits with deterministic filenames to avoid collisions.  
- Runs in GitHub Actions matrix (16 shards) or locally with `SHARD_ID=0..15`.

def deterministic_shard(slug: str) -> int:
    return int(hashlib.sha256(slug.encode()).hexdigest(), 16) % TOTAL_SHARDS

def backoff_retry(fn, max_tries=5, base_delay=1):
    last_exc = None
    for attempt in range(max_tries):
        try:
            return fn()
        except Exception as exc:
            last_exc = exc
            if attempt == max_tries - 1:
                break
            delay = base_delay * (2 ** attempt)
            # if HF 429, wait at least 360s
            if hasattr(exc, "response") and getattr(exc.response, "status_code", None) == 429:
                delay = max(delay, 360)
            print(f"[retry {attempt+1}/{max_tries}] {exc} -> sleeping {delay}s", file=sys.stderr)
            time.sleep(delay)
    raise last_exc

def get_manifest() -> List[str]:
    """Return list of file paths for DATE_FOLDER from manifest or API (once)."""
    # 1) local manifest
    if LOCAL_MANIFEST and Path(LOCAL_MANIFEST).exists():
        data = json.loads(Path(LOCAL_MANIFEST).read_text())
        if isinstance(data, dict) and "files" in data:
            files = data["files"]
        else:
            files = data
        return [f for f in files if isinstance(f, str) and f.startswith(DATE_FOLDER)]

    # 2) remote manifest
    if MANIFEST_URL:
        resp = requests.get(MANIFEST_URL, timeout=60)
        resp.raise_for_status()
        data = resp.json()
        if isinstance(data, dict) and "files" in data:
            files = data["files"]
        else:
            files = data
        return [f for f in files if isinstance(f, str) and f.startswith(DATE_FOLDER)]

    # 3) fallback: single API list_repo_tree (non-recursive) for folder
    # Avoid recursive on big repos; we expect DATE_FOLDER to contain manageable file list.
    print(f"[manifest] listing {HF_REPO}/{DATE_FOLDER} (non-recursive)", file=sys.stderr)
    tree = backoff_retry(lambda: list_repo_tree(HF_REPO, path=DATE_FOLDER, recursive=False))
    files = [item.rfilename for item in tree if not item.rfilename.endswith("/")]
    return [f for f in files if f.startswith(DATE_FOLDER)]

def cdn_download(path: str, out_path: Path) -> Path:
    url = f"{CDN_BASE}/{path}"
    resp = requests.get(url, stream=True, timeout=120)
    resp.raise_for_status()
    out_path.parent.mkdir(parents=True, exist_ok=True)
    with open(out_path, "wb") as f:
        for chunk in resp.iter_content(chunk_size=8192):
            f.write(chunk)
    return out_path

def normalize_record(raw: Dict[str, Any], source: str) -> Dict[str, Any]:
    """Project heterogeneous schema -> {prompt, response, source, ts}."""
    # Common field names observed in surrogate-1 pipelines
    prompt_keys = {"prompt", "input", "question", "instruction", "text"}
    response_keys = {"response", "output", "answer", "completion", "target"}

    prompt = None
    response = None

    for k, v in raw.items():
        if k in prompt_keys and prompt is None:
            prompt = v
        if k in response_keys and response is None:
            response = v

    # Fallbacks
    if prompt is None:
        prompt = raw.get("prompt", raw.get("input", ""))
    if response is None:
        response = raw.get("response", raw.get("output", ""))

    return {
        "prompt": str(prompt),
        "response": str(response),
        "source": source,
        "ts": datetime.now(timezone.utc).isoformat(),
    }
