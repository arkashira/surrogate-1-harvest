## Highest-Value Incremental Improvement (<2h)

**Goal**: Eliminate HF API 429s during Surrogate-1 training by replacing `load_dataset(streaming=True)` with a CDN-only `IterableDataset` that reads from a pre-generated `file_list.json`. This removes pagination, auth, and API calls during data loading.

**Why this**: GPUs sit idle waiting on HF API rate limits; CDN downloads bypass auth and have much higher limits. Single API call from Mac (after window clears) → embed file list → Lightning training does zero API calls during data load.

---

## Implementation Plan

### 1. Create CDN-only dataset loader (`surrogate/data/cdn_dataset.py`)
- `CdnIterableDataset`: given repo, date folder, and local `file_list.json`, streams files via `hf_hub_download` or raw CDN URLs
- Projects each file to `{prompt, response}` only at parse time (avoids mixed-schema CastError)
- Attribution via filename pattern (`batches/mirror-merged/{date}/{slug}.parquet`) — no extra cols

### 2. Add file-list generator script (`surrogate/scripts/generate_file_list.py`)
- Runs on Mac after rate-limit window clears
- Uses `list_repo_tree(path, recursive=False)` per date folder (avoids 100x pagination)
- Saves `file_list.json` with relative paths and sizes
- Embeds into training script or copies to Lightning machine

### 3. Update training entrypoint (`surrogate/train.py`)
- Accept `--file-list` arg pointing to `file_list.json`
- Use `CdnIterableDataset` instead of `load_dataset`
- Keep existing model/training logic unchanged

### 4. Reuse Lightning Studio (quota-safe)
- Before `Studio(create_ok=True)`, list `Teamspace.studios` and reuse running ones
- On idle stop, restart with `target.start(machine=Machine.L40S)`

### 5. Make scripts executable + Bash shebang
- Ensure wrapper scripts have `#!/usr/bin/env bash`, `chmod +x`
- Set `SHELL=/bin/bash` in crontab if scheduled

### `surrogate/data/cdn_dataset.py`
```python
# surrogate/data/cdn_dataset.py
import json
import os
from typing import Dict, Iterator, Optional
from huggingface_hub import hf_hub_download, HfApi
import pyarrow.parquet as pq
import io
import torch
from torch.utils.data import IterableDataset

class CdnIterableDataset(IterableDataset):
    """
    CDN-only dataset loader for Surrogate-1 training.
    Bypasses HF API auth/rate limits by using:
      - hf_hub_download (cached) OR
      - raw CDN URLs: https://huggingface.co/datasets/{repo}/resolve/main/{path}
    Expects file_list.json with [{"path": "batches/mirror-merged/2026-04-27/slug.parquet", "size": 12345}, ...]
    Projects each file to {prompt, response} only at parse time.
    """

    def __init__(
        self,
        repo_id: str,
        file_list_path: str,
        use_cdn_url: bool = True,
        cache_dir: Optional[str] = None,
        split_by_worker: bool = True,
    ):
        self.repo_id = repo_id
        self.use_cdn_url = use_cdn_url
        self.cache_dir = cache_dir or os.getenv("HF_HOME", os.path.expanduser("~/.cache/huggingface"))

        with open(file_list_path, "r") as f:
            self.files = json.load(f)  # list[dict]

        if not self.files:
            raise ValueError("file_list.json is empty")

        # Optional: shard across DataLoader workers
        self.split_by_worker = split_by_worker
        self.worker_info = None

    def _get_worker_files(self) -> list:
        if not self.split_by_worker or self.worker_info is None:
            return self.files
        worker_id = self.worker_info.id
        num_workers = self.worker_info.num_workers
        return [f for i, f in enumerate(self.files) if i % num_workers == worker_id]

    def __iter__(self) -> Iterator[Dict[str, str]]:
        if self.split_by_worker:
            worker_info = torch.utils.data.get_worker_info()
            self.worker_info = worker_info

        for item in self._get_worker_files():
            path = item["path"]
            try:
                if self.use_cdn_url:
                    # CDN bypass: no Authorization header, much higher rate limits
                    url = f"https://huggingface.co/datasets/{self.repo_id}/resolve/main/{path}"
                    import requests
                    resp = requests.get(url, timeout=30)
                    resp.raise_for_status()
                    buf = io.BytesIO(resp.content)
                else:
                    # Use hf_hub_download (cached) — still avoids streaming API
                    local_path = hf_hub_download(
                        repo_id=self.repo_id,
                        filename=path,
                        cache_dir=self.cache_dir,
                        force_download=False,
                    )
                    buf = open(local_path, "rb")

                # Read parquet and project to {prompt, response} only
                table = pq.read_table(buf)
                if isinstance(buf, io.BytesIO):
                    buf.close()
                else:
                    buf.close()

                # Convert to dicts; keep only prompt/response fields
                df = table.to_pandas()
                for _, row in df.iterrows():
                    # Flexible field names (mirror schema may vary)
                    prompt = row.get("prompt") or row.get("input") or row.get("question") or ""
                    response = row.get("response") or row.get("output") or row.get("answer") or ""
                    if prompt and response:
                        yield {"prompt": str(prompt), "response": str(response)}
            except Exception as e:
                # Log and skip corrupt files to avoid crashing training
                print(f"[WARN] Failed to process {path}: {e}")
                continue

    def __len__(self) -> int:
        # Approximate length (number of rows unknown without reading)
        return len(self._get_worker_files())
```

### `surrogate/scripts/generate_file_list.py`
```bash
#!/usr/bin/env bash
# surrogate/scripts/generate_file_list.py
# Usage: python generate_file_list.py --repo datasets/your-repo --date-folder 2026-04-27 --out file_list.json
set -euo pipefail

python3 - "$@" <<'PY'
import argparse
import json
import os
from huggingface_hub import HfApi

def main():
    parser = argparse.ArgumentParser(description="Generate file_list.json for CDN-only dataset loading.")
    parser.add_argument("--repo", required=True, help="HuggingFace dataset repo (e.g., org/dataset)")
    parser.add_argument("--date-folder", required=True, help="Date folder under datasets (e.g., 2026-04-27)")
    parser.add_argument("--out", default="file_list.json", help="Output JSON path")
    parser.add_argument("--base-path", default="batches/mirror-merged", help="Base path in repo")
    args = parser.parse_args()

    api = HfApi()
    folder_path = f"{args.base_path}/{args.date_folder}"

    # Use recursive=False to avoid 100x pagination on large repos
    entries = api.list_repo_tree(repo_id=args.repo, path=folder_path, recursive=False)

    files = []
    for entry in entries:
        if entry.get("type") == "file" and entry["path"].endswith(".parquet"):
            files.append({"path": entry["path"], "size": entry.get("size", 0)})

    os.makedirs(os.path.dirname(os.path.abspath(args.out)) if os.path.dirname(args.out) else ".", exist_ok=True)
    with open(args.out, "w") as f:
        json.dump(files, f, indent=2)

    print(f"Generated {args.out} with {len(files)} files")

if __name__ == "__main__":
    main()
PY
```
