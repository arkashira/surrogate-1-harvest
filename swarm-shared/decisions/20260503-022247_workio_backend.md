# workio / backend

Candidate 3:
## 1. Diagnosis

- LINE webhook retries and client replays create duplicate punches because there is no stable, persisted deduplication key and no unique constraint at the persistence layer.
- Punch upsert is not atomic across `(tenant_id, employee_id, date)` and is vulnerable to race conditions under concurrent or retried requests.
- No request-level idempotency handling (LINE can redeliver events minutes apart) and no short TTL replay-protection cache.
- Business rules (max one in/out pair per day, no overlapping punches) are enforced in application code instead of at the database.
- Missing audit/source tracking for punches (webhook vs manual) makes debugging duplicates difficult.

## 2. Proposed change

File: `workio/server/src/services/punchService.ts`  
Scope: add idempotent punch ingestion with atomic upsert on `(tenant_id, employee_id, date, type, line_event_id)` and a unique constraint/index; add a lightweight idempotency table keyed by `(tenant_id, source, idempotency_key)`; wrap punch creation + idempotency + audit/event emission in a single transaction.

```sql
-- Idempotency keys for webhook events (LINE retries)
CREATE TABLE IF NOT EXISTS webhook_idempotency (
  id              BIGSERIAL PRIMARY KEY,
  tenant_id       BIGINT NOT NULL,
  source          TEXT   NOT NULL,        -- 'line'
  idempotency_key TEXT NOT NULL,          -- e.g. X-Line-Retry-Id or hash(payload)
  event_type      TEXT   NOT NULL,        -- 'punch'
  entity_id       BIGINT NOT NULL,        -- punch.id
  created_at      TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE (tenant_id, source, idempotency_key)
);

-- Prevent duplicate open punches per tenant/employee/date
-- Assumes punches table exists with employee_id, tenant_id, clock_in, clock_out, date
CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_unique_open
ON punches (tenant_id, employee_id, date)
WHERE clock_out IS NULL;

-- Optional: enforce one in/out pair per day (no overlapping punches)
-- If your app allows multiple in/out pairs per day, omit this constraint.
CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_unique_day
ON punches (tenant_id, employee_id, date);
```

### 3.2 Idempotency helper

```ts
// server/src/db/idempotency.ts
import { PoolClient } from 'pg';

export async function ensureIdempotent(
  client: PoolClient,
  opts: {
    tenantId: number;
    source: string;
    idempotencyKey: string;
    eventType: string;
    entityId: number;
  }
): Promise<boolean> {
  const { tenantId, source, idempotencyKey, eventType, entityId } = opts;
  const q = `
    INSERT INTO webhook_idempotency (tenant_id, source, idempotency_key, event_type, entity_id)
    VALUES ($1, $2, $3, $4, $5)
    ON CONFLICT (tenant_id, source, idempotency_key) DO NOTHING
    RETURNING id;
  `;
  const res = await client.query(q, [tenantId, source, idempotencyKey, eventType, entityId]);
  return res.rowCount > 0;
}
```

### 3.3 Punch service (atomic upsert + audit)

```ts
// workio/server/src/services/punchService.ts
import { pool } from '../db';
import { ensureIdempotent } from '../db/idempotency';

export interface CreatePunchOpts {
  tenantId: number;
  employeeId: number;
  date: string; // YYYY-MM-DD
  clockIn: Date;
  source: 'line' | 'manual';
  lineEventId?: string;
  lineRetryId?: string;
}

export async function createPunch(opts: CreatePunchOpts) {
  const { tenantId, employeeId, date, clockIn, source, lineEventId, lineRetryId } = opts;
  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // Idempotency key: prefer LINE retry id, else deterministic event hash
    const idempotencyKey = lineRetryId
      ? `line-retry-${lineRetryId}-${lineEventId || ''}`
      : `line-event-${date}-${employeeId}-${lineEventId || 'manual'}`;

    // Check idempotency first (fast fail for retries)
    const idemCheck = await client.query(
      `SELECT entity_id FROM webhook_idempotency WHERE tenant_id = $1 AND source = $2 AND idempotency_key = $3`,
      [tenantId, source, idempotencyKey]
    );
    if (idemCheck.rowCount > 0) {
      // Already processed; return existing punch id
      const punchId = idemCheck.rows[0].entity_id;
      await client.query('ROLLBACK');
      return { punchId, processed: false };
    }

    // Atomic upsert: update today's open punch if clock_out is NULL, else insert new row
    const upsertRes = await client.query(
      `
      INSERT INTO punches (tenant_id, employee_id, date, clock_in, clock_out, created_at, updated_at, source, line_event_id)
      VALUES ($1, $2, $3, $4, NULL, NOW(), NOW(), $5, $6)
      ON CONFLICT (tenant_id, employee_id, date)
      DO UPDATE SET
        clock_in = EXCLUDED.clock_in,
        updated_at = NOW(),
        source = EXCLUDED.source,
        line_event_id = EXCLUDED.line_event_id
      WHERE punches.clock_out IS NULL
      RETURNING id, clock_in, clock_out;
      `,
      [tenantId, employeeId, date, clockIn, source, lineEventId]
    );

    let punchId: number;
    if (upsertRes.rowCount === 0) {
      // Row exists and clock_out already set (or constraint prevented update) -> insert new punch row
      const insertRes = await client.query(
        `
        INSERT INTO punches (tenant_id, employee_id, date, clock_in, clock_out, created_at, updated_at, source, line_event_id)
        VALUES ($1, $2, $3, $4, NULL, NOW(), NOW(), $5, $6)
        RETURNING id, clock_in, clock_out;
        `,
        [tenantId, employeeId, date, clockIn, source, lineEventId]
      );
      punchId = insertRes.rows[0].id;
    } else {
      punchId = upsertRes.rows[0].id;
    }

    // Record idempotency
    await ensureIdempotent(client, {
      tenantId,
      source,
      idempotencyKey,
      eventType: 'punch',
      entityId: punchId,
    });

    // Optional: emit internal event / notification here (within same tx)

    await client.query('COMMIT');
    return { punchId, processed: true };
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  } finally {
    client.release();
  }
}
```
