# workio / design

## 1. Diagnosis

- **No tenant cache on LINE webhook** — every webhook hits PostgreSQL to resolve `channelId → tenant`; slow under burst traffic, noisy under load, and fails fast if DB is cold.
- **Missing health/readiness endpoints** — no `/health` or `/ready` for k8s/PM2/liveness checks; deploy/rollback confidence is low.
- **No structured logging or request correlation** — hard to trace webhook → tenant → action across async flows; debugging in prod is slow.
- **LINE signature verification is likely ad-hoc or missing on webhook** — increases risk of spoofed events and replay.
- **No graceful shutdown or in-flight drain** — SIGTERM can drop LINE webhook ACKs mid-flight and cause retries or lost clock events.

## 2. Proposed change

**Scope**: `workio/server/src/app.ts` (or main Express entry) + add `workio/server/src/middleware/tenantCache.ts` and `workio/server/src/middleware/logging.ts`.

- Add TTL-backed LRU tenant cache keyed by `channelId` (1000 entries, TTL 5m, refresh on hit) in tenant service.
- Add `/health` (liveness) and `/ready` (readiness; checks DB connectivity) endpoints.
- Add structured request logging + correlation ID middleware.
- Add LINE signature verification middleware (HMAC-SHA256) for webhook routes.
- Add graceful shutdown with in-flight drain (30s timeout) in server bootstrap.

Create tenant cache (`server/src/middleware/tenantCache.ts`):

```ts
import { LRUCache } from 'lru-cache';
import { getTenantByChannelId } from '../services/tenantService';

type Tenant = { id: string; name: string; config: Record<string, any> };

const tenantCache = new LRUCache<string, Tenant>({
  max: 1000,
  ttl: 1000 * 60 * 5, // 5m
  allowStale: false,
  updateAgeOnGet: true,
});

export async function resolveTenantByChannelId(channelId: string): Promise<Tenant | null> {
  const cached = tenantCache.get(channelId);
  if (cached) return cached;

  const tenant = await getTenantByChannelId(channelId);
  if (tenant) tenantCache.set(channelId, tenant);
  return tenant || null;
}

export function invalidateTenantCache(channelId: string) {
  tenantCache.delete(channelId);
}
```

Structured logging + correlation ID (`server/src/middleware/logging.ts`):

```ts
import { Request, Response, NextFunction } from 'express';
import pino from 'pino';
import pinoHttp from 'pino-http';

const logger = pino({
  level: process.env.LOG_LEVEL || 'info',
  transport: process.env.NODE_ENV !== 'production' ? { target: 'pino-pretty' } : undefined,
});

export const httpLogger = pinoHttp({ logger });

export function requestIdMiddleware(req: Request, res: Response, next: NextFunction) {
  const cid = req.headers['x-request-id'] || crypto.randomUUID();
  (req as any).requestId = cid;
  res.setHeader('x-request-id', cid as string);
  next();
}
```

LINE signature verification (`server/src/middleware/lineSignature.ts`):

```ts
import { Request, Response, NextFunction } from 'express';
import crypto from 'crypto';

const LINE_CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';

export function lineSignatureMiddleware(req: Request, res: Response, next: NextFunction) {
  if (!LINE_CHANNEL_SECRET) return next();
  const signature = req.headers['x-line-signature'] as string;
  if (!signature) return res.status(401).json({ error: 'Missing X-Line-Signature' });

  const body = JSON.stringify(req.body);
  const expected = crypto
    .createHmac('sha256', LINE_CHANNEL_SECRET)
    .update(body)
    .digest('base64');

  if (!crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected))) {
    return res.status(401).json({ error: 'Invalid signature' });
  }
  next();
}
```

Wire into Express (`server/src/app.ts`):

```ts
import express from 'express';
import { requestIdMiddleware, httpLogger } from './middleware/logging';
import { lineSignatureMiddleware } from './middleware/lineSignature';
import { resolveTenantByChannelId } from './middleware/tenantCache';
import db from './db';

const app = express();
app.use(express.json());
app.use(requestIdMiddleware);
app.use(httpLogger);

// Health & readiness
app.get('/health', (_, res) => res.json({ status: 'ok' }));
app.get('/ready', async (_, res) => {
  try {
    await db.query('SELECT 1');
    res.json({ status: 'ready' });
  } catch (err) {
    res.status(503).json({ status: 'not ready', error: String(err) });
  }
});

// Apply LINE signature only to webhook route(s)
app.post('/webhook/line', lineSignatureMiddleware, async (req, res) => {
  const channelId = req.body?.source?.userId ? undefined : req.body?.source?.channelId;
  // typical LINE webhook: events[].source.channelId or bot's channelId via header; adapt as needed
  const events = req.body.events || [];
  const results = [];

  for (const ev of events) {
    const cid = ev.source?.channelId || (req as any).lineChannelId;
    const tenant = cid ? await resolveTenantByChannelId(cid) : null;
    // enqueue or handle event with tenant context
    results.push({ eventId: ev.eventId, tenantId: tenant?.id || null });
  }

  res.json({ results });
});

// Graceful shutdown (server bootstrap)
function start() {
  const server = app.listen(process.env.PORT || 3000, () => {
    console.log('Server listening');
  });

  let shuttingDown = false;
  const shutdown = () => {
    if (shuttingDown) return;
    shuttingDown = true;
    console.log('Shutting down gracefully...');
    server.close(() => {
      console.log('Closed remaining connections');
      process.exit(0);
    });

    setTimeout(() => {
      console.error('Forced shutdown after timeout');
      process.exit(1);
    }, 30000);
  };

  process.on('SIGTERM', shutdown);
  process.on('SIGINT', shutdown);
}

if (require.main === module) start();

export default app;
```
