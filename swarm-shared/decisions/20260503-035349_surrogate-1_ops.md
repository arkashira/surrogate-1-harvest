## Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16` (matrix) and optional `DATE_FOLDER` (defaults to today `YYYY-MM-DD`).
- Uses a **single pre-listed manifest** (generated by the Mac orchestrator) to avoid recursive `list_repo_files` and HF API rate limits.
- Downloads only assigned shard files via **HF CDN direct URLs** (`https://huggingface.co/datasets/.../resolve/main/...`) — zero Authorization header, bypasses `/api/` 429 limits.
- Projects heterogeneous schemas to `{prompt, response}` at parse time (no `load_dataset(streaming=True)` on mixed schemas).
- Deduplicates via the existing `lib/dedup.py` central md5 store.
- Writes normalized JSONL to `batches/public-merged/{date}/shard{N}-{HHMMSS}.jsonl` and pushes to `axentx/surrogate-1-training-pairs` using the HF token.
- Reuses running Lightning Studio when present (saves quota) and falls back to L40S if H200 unavailable.
- Adds cron-safe Bash wrappers with proper shebang and `SHELL=/bin/bash` for any scheduled tasks.

def list_repo_tree(path: str = "") -> List[str]:
    """
    Single API call to list files in a folder (non-recursive).
    Used only once by orchestrator; worker uses provided manifest.
    """
    import huggingface_hub
    api = huggingface_hub.HfApi(token=HF_TOKEN)
    files = api.list_repo_tree(repo_id=HF_REPO, path=path, recursive=False)
    return [f.rfilename for f in files if not f.rfilename.endswith("/")]

def shard_assign(files: List[str], shard_id: int, shard_total: int) -> List[str]:
    """Deterministic shard assignment by hash(slug)."""
    assigned = []
    for f in files:
        slug = pathlib.Path(f).stem
        h = int(hashlib.md5(slug.encode()).hexdigest(), 16)
        if h % shard_total == shard_id:
            assigned.append(f)
    return assigned

def cdn_download(url: str, dest: pathlib.Path, retries: int = 3, backoff: int = 5) -> bool:
    for attempt in range(1, retries + 1):
        try:
            r = requests.get(url, timeout=60)
            r.raise_for_status()
            dest.write_bytes(r.content)
            return True
        except Exception as e:
            if attempt == retries:
                print(f"ERROR: CDN download failed after {retries} attempts: {e}", file=sys.stderr)
                return False
            wait = backoff * (2 ** (attempt - 1))
            print(f"WARN: CDN download failed (attempt {attempt}/{retries}), retry in {wait}s: {e}")
            time.sleep(wait)
    return False

def normalize_to_pair(raw: Dict[str, Any]) -> Dict[str, Any]:
    """
    Project heterogeneous schemas to {prompt, response}.
    Handles common HF dataset field names.
    """
    prompt = None
    response = None

    # Common field mappings
    prompt_candidates = ["prompt", "instruction", "question", "input", "text", "content"]
    response_candidates = ["response", "completion", "answer", "output", "target"]

    for k in raw:
        if k in prompt_candidates and prompt is None:
            prompt = raw[k]
        if k in response_candidates and response is None:
            response = raw[k]

    # Fallback: if only one text-like field, split by common separators
    if prompt is None or response is None:
        text_fields = [v for v in raw.values() if isinstance(v, str) and len(v) > 10]
        if len(text_fields) == 1:
            txt = text_fields[0]
            # Try split by Q/A patterns
            parts = [p.strip() for p in txt.split("###") if p.strip()]
            if len(parts) >= 2:
                prompt = parts[0]
                response = parts[1]
            else:
                # Last resort: use whole as prompt, empty response
                prompt = txt
                response = ""

    if prompt is None:
        prompt = ""
    if response is None:
        response = ""

    return {"prompt": str(prompt), "response": str(response)}

def process_parquet(file_url: str, dedup: DedupStore) -> List[Dict[str, Any]]:
    """Download parquet via CDN, normalize rows, dedup by md5."""
    with tempfile.TemporaryDirectory() as td:
        tmp = pathlib.Path(td) / "file.parquet"
        if not cdn_download(file_url, tmp):
            return []
        try:
            table = pq.read_table(tmp)
        except Exception as e:
            print(f"WARN: failed to read parquet {file_url}: {e}")
            return []

        rows = []
        for i in range(table.num_rows):
            raw = {col: table[col][i].as_py() for col in table.column_names}
            pair = normalize_to_pair(raw)
            # dedup by content hash
            digest = hashlib.md5(json.dumps(pair, sort_keys=True).encode()).hexdigest()
            if dedup.seen(digest):
                continue
            dedup.add(digest)
            pair["_md5"] = digest
            rows.append(pair)
        return rows

def upload_batch(rows: List[Dict[str, Any]], date_folder: str, shard_id: int):
    """Write JSONL and push to HF dataset repo."""
    if not rows:
        print("No rows to upload.")
        return

    ts = datetime.datetime.utcnow().strftime("%H%M%S")
    filename = f"shard{shard_id}-{ts}.jsonl"
    out_dir = pathlib.Path("batches") / "public-merged" / date_folder
    out_dir.mkdir(parents=True, exist_ok=True)
    out_path = out_dir / filename

    with out_path.open("w") as f:
        for row in rows:
            f.write(json.dumps(row, ensure_ascii=False) + "\n")

    # Git add + commit + push
    repo_dir = pathlib.cwd()
    try:
        subprocess.run(["git", "add", str(out_path.relative_to(repo_dir))], cwd=repo_dir, check=True)
        commit_msg = f"public-merged/{date_folder}/{filename} — shard {shard_id}"
        subprocess.run(["git", "commit", "-m", commit_msg], cwd=repo_dir, check=True)
        subprocess.run(["git", "push", "origin", "HEAD"], cwd=repo_dir, check=True)
        print(f"Uploaded {len(rows)} rows to {out_path}")
    except subprocess.CalledProcessError as e:
        print(f"ERROR: git push failed: {e}", file=sys.stderr)
