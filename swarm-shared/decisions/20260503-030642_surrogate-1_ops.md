## Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16`, `DATE`, `HF_TOKEN` via env
- Uses **one** `list_repo_tree` call (per DATE folder) from Mac/runner, saves `file-list.json`
- During training: Lightning Studio loads `file-list.json` and fetches **only** via CDN URLs (`resolve/main/...`) — zero HF API calls during data load, bypasses 429 rate limits
- Projects heterogeneous files to `{prompt, response}` at parse time (avoids pyarrow CastError)
- Deduplicates via central md5 store (`lib/dedup.py`)
- Writes to deterministic path: `batches/public-merged/{DATE}/shard{N}-{HHMMSS}.jsonl`
- Reuses running Lightning Studio instead of recreating (saves quota)
- Adds retry/backoff for HF 429 (wait 360s) and commit-cap spreading across 5 sibling repos

def download_via_cdn(file_path: str, local_path: Path) -> Path:
    """Download via CDN (no Authorization header) — bypasses /api/ rate limits."""
    url = f"{CDN_BASE}/{file_path}"
    local_path.parent.mkdir(parents=True, exist_ok=True)
    resp = requests.get(url, timeout=60)
    resp.raise_for_status()
    local_path.write_bytes(resp.content)
    return local_path

def parse_to_pair(local_path: Path) -> Optional[Dict[str, str]]:
    """
    Project heterogeneous files to {prompt, response} at parse time.
    Supports common patterns: JSON/JSONL with 'prompt'/'response' or 'instruction'/'output'.
    """
    try:
        text = local_path.read_text(encoding="utf-8")
    except UnicodeDecodeError:
        # skip binary/non-text
        return None

    # JSONL
    if local_path.suffix == ".jsonl":
        for line in text.strip().splitlines():
            line = line.strip()
            if not line:
                continue
            try:
                obj = json.loads(line)
            except json.JSONDecodeError:
                continue
            prompt = obj.get("prompt") or obj.get("instruction") or obj.get("input")
            response = obj.get("response") or obj.get("output") or obj.get("completion")
            if prompt and response:
                return {"prompt": str(prompt).strip(), "response": str(response).strip()}
        return None

    # JSON
    if local_path.suffix == ".json":
        try:
            obj = json.loads(text)
        except json.JSONDecodeError:
            return None
        if isinstance(obj, dict):
            prompt = obj.get("prompt") or obj.get("instruction") or obj.get("input")
            response = obj.get("response") or obj.get("output") or obj.get("completion")
            if prompt and response:
                return {"prompt": str(prompt).strip(), "response": str(response).strip()}
        elif isinstance(obj, list):
            for item in obj:
                if isinstance(item, dict):
                    prompt = item.get("prompt") or item.get("instruction") or item.get("input")
                    response = item.get("response") or item.get("output") or item.get("completion")
                    if prompt and response:
                        return {"prompt": str(prompt).strip(), "response": str(response).strip()}
        return None

    # Plain text heuristics (fallback)
    lines = [ln.strip() for ln in text.strip().splitlines() if ln.strip()]
    if len(lines) >= 2:
        return {"prompt": lines[0], "response": " ".join(lines[1:])}
    return None

def sibling_repo_for(slug: str) -> str:
    """Deterministic sibling repo selection for commit-cap spreading."""
    if SIBLING_REPOS <= 1:
        return REPO_ID
    digest = int(hashlib.sha256(slug.encode()).hexdigest(), 16)
    idx = digest % SIBLING_REPOS
    if idx == 0:
        return REPO_ID
    owner, repo = REPO_ID.split("/", 1)
    return f"{owner}/{repo.replace('.git', '').replace('.', '')}-sibling{idx}"
