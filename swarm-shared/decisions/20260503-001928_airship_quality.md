### Goal (unified)
Eliminate HF API 429s during Surrogate-1 training **today** by replacing `load_dataset(streaming=True)` with a **CDN-only IterableDataset** that reads from a pre-generated `file_list.json`.  
- Zero HF API calls during training.

- Works with heterogeneous parquet schemas.

### Core Design (corrected + actionable)
1. **CDN-only data path**  
   Use public URLs:  
   `https://huggingface.co/datasets/{repo_id}/resolve/main/{path}`  
   No auth, no API, no 429s while training.

2. **Schema robustness**  
   - Read parquet with `pyarrow` (not `datasets`).  
   - Project to `{prompt, response}` at parse time; ignore extra/missing columns.  
   - Skip corrupt files with clear logging (do not crash training).

3. **Deterministic sharding for multi-worker DataLoader**  
   - Shard files (not rows) across workers to avoid heavy `skip`/`take` on remote parquet.  
   - Preserve order deterministically; support resume via `start_idx`.

4. **One-off file list generation**  
   - Run from orchestration host (Mac) **once per snapshot**.  
   - Handle 429 on generation with a single retry after wait.  
   - Output `file_list.json` with `repo_id` and sorted file paths.

#### 1) CDN-only IterableDataset  
`surrogate/training/data/cdn_parquet_dataset.py`
```python
from torch.utils.data import IterableDataset
import pyarrow.parquet as pq
import requests
import io
import json
import logging
from typing import List, Dict, Optional

logger = logging.getLogger(__name__)


class CdnParquetIterableDataset(IterableDataset):
    """
    CDN-only dataset for HuggingFace parquet repos.
    Avoids HF API entirely by using public CDN URLs:
      https://huggingface.co/datasets/{repo_id}/resolve/main/{path}

    file_list.json format:
    {
      "repo_id": "org/surrogate-1-mirror",
      "files": [
        "batches/mirror-merged/2026-04-28/abc123.parquet",
        ...
      ]
    }

    Projects rows to {"prompt": str, "response": str}.
    Skips unreadable files/rows and logs warnings.
    """

    CDN_TEMPLATE = "https://huggingface.co/datasets/{repo_id}/resolve/main/{path}"

    def __init__(
        self,
        file_list_path: str,
        repo_id: Optional[str] = None,
        start_idx: int = 0,
        end_idx: Optional[int] = None,
    ):
        super().__init__()
        with open(file_list_path, "r") as f:
            meta = json.load(f)
        self.repo_id = repo_id or meta.get("repo_id")
        if not self.repo_id:
            raise ValueError("repo_id must be provided or present in file_list.json")
        files: List[str] = meta["files"]
        self.files = sorted(files)[start_idx:end_idx]
        logger.info(f"CDN dataset initialized with {len(self.files)} files (repo={self.repo_id})")

    def _download_parquet(self, path: str) -> bytes:
        url = self.CDN_TEMPLATE.format(repo_id=self.repo_id, path=path)
        resp = requests.get(url, timeout=30)
        resp.raise_for_status()
        return resp.content

    def _iter_file(self, path: str):
        try:
            data = self._download_parquet(path)
            table = pq.read_table(io.BytesIO(data))
            df = table.to_pandas()
            for _, row in df.iterrows():
                yield {
                    "prompt": str(row.get("prompt", "")),
                    "response": str(row.get("response", "")),
                }
        except Exception as exc:
            logger.warning(f"Skipping {path} due to error: {exc}")

    def __iter__(self):
        worker_info = torch.utils.data.get_worker_info()
        if worker_info is None:
            files = self.files
        else:
            # shard by files to avoid heavy remote skip/take
            nw = worker_info.num_workers
            wid = worker_info.id
            per_worker = len(self.files) // nw
            files = self.files[wid * per_worker : (wid + 1) * per_worker]

        for path in files:
            yield from self._iter_file(path)
```

#### 2) One-off file list generator  
`tools/generate_file_list.py`
```python
#!/usr/bin/env python3
"""
Generate file_list.json for CDN-only training.
Run from orchestration host (Mac) after rate-limit window clears.

Usage:
  python tools/generate_file_list.py \
    --repo-id org/surrogate-1-mirror \
    --date-folder 2026-04-28 \
    --out-file file_list.json
"""

import argparse
import json
import time
import sys
from huggingface_hub import HfApi

HF_API_RATE_LIMIT_WAIT = 360  # seconds after 429


def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo-id", required=True)
    parser.add_argument("--date-folder", required=True)
    parser.add_argument("--out-file", required=True)
    args = parser.parse_args()

    api = HfApi()
    base_path = f"batches/mirror-merged/{args.date_folder}"
    try:
        items = api.list_repo_tree(repo_id=args.repo_id, path=base_path, recursive=False)
    except Exception as exc:
        print(f"HF API error (likely 429): {exc}. Waiting {HF_API_RATE_LIMIT_WAIT}s...", file=sys.stderr)
        time.sleep(HF_API_RATE_LIMIT_WAIT)
        items = api.list_repo_tree(repo_id=args.repo_id, path=base_path, recursive=False)

    files = [
        f"{base_path}/{item.path}"
        for item in items
        if item.path.endswith(".parquet")
    ]

    out = {
        "repo_id": args.repo_id,
        "files": sorted(files),
    }

    with open(args.out_file, "w") as f:
        json.dump(out, f, indent=2)

    print(f"Wrote {len(files)} files to {args.out_file}")


if __name__ == "__main__":
    main()
```
