import { useState, useEffect, useCallback, useRef } from 'react';
import { useLocation } from '../hooks/useLocation';
import { punch } from '../api/punch';
import { punchStatus } from '../api/punchStatus';
import { useToast } from '../hooks/useToast';
import type { PunchResponse } from '../types/punch';

const IDEMPOTENCY_TTL_MS = 60_000;

export function PunchButton() {
  const [status, setStatus] = useState<'in' | 'out' | 'unknown'>('unknown');
  const [pending, setPending] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const idempotencyKeyRef = useRef<string | null>(null);
  const lastSuccessAtRef = useRef<number>(0);
  const { checkLocation, permissionState } = useLocation();
  const { showToast } = useToast();

  // Fetch current state once on mount
  useEffect(() => {
    fetchStatus();
  }, []);

  const fetchStatus = async () => {
    try {
      const data = await punchStatus();
      setStatus(data.openToday ? 'in' : 'out');
    } catch {
      setStatus('unknown');
    }
  };

  const clearIdempotencyKey = useCallback(() => {
    idempotencyKeyRef.current = null;
  }, []);

  const doPunch = useCallback(async () => {
    if (pending) return;

    // Idempotency guard: prevent retries within TTL of last success
    const now = Date.now();
    if (idempotencyKeyRef.current && now - lastSuccessAtRef.current < IDEMPOTENCY_TTL_MS) {
      return;
    }

    setError(null);

    const locOk = await checkLocation();
    if (!locOk) {
      setError('Location permission required to punch.');
      return;
    }

    const optimisticNext = status === 'in' ? 'out' : 'in';
    const prev = status;
    setPending(true);
    setStatus(optimisticNext);

    // Generate idempotency key for this session attempt
    const idempotencyKey = `${Date.now()}-${Math.random().toString(36).slice(2, 9)}`;
    idempotencyKeyRef.current = idempotencyKey;

    try {
      const data: PunchResponse = await punch(
        {
          type: optimisticNext === 'in' ? 'clock_in' : 'clock_out',
        },
        { idempotencyKey }
      );

      // Confirm with server state
      setStatus(data.openToday ? 'in' : 'out');
      lastSuccessAtRef.current = Date.now();

      // Keep key briefly to suppress duplicate clicks
      setTimeout(clearIdempotencyKey, 4_000);

      showToast(
        data.openToday
          ? 'Clocked in successfully'
          : 'Clocked out successfully',
        'success'
      );
    } catch (err: any) {
      // Rollback optimistic state
      setStatus(prev);
      setError(err?.message || 'Punch failed. Please try again.');
      idempotencyKeyRef.current = null;
    } finally {
      setPending(false);
    }
  }, [status, pending, checkLocation, clearIdempotencyKey, showToast]);

  const label =
    status === 'in' ? 'Clock Out' : status === 'out' ? 'Clock In' : 'Loading...';
  const isActionable = status !== 'unknown' && !pending;

  return (
    <div className="flex flex-col items-center gap-2">
      <button
        onClick={doPunch}
        disabled={!isActionable || pending}
        aria-busy={pending}
        className={`
          px-8 py-3 rounded-lg font-semibold text-white transition
          ${pending ? 'bg-gray-400 cursor-wait' : ''}
          ${!isActionable && !pending ? 'bg-gray-400 cursor-not-allowed' : ''}
          ${
            isActionable && !pending
              ? status === 'in'
                ? 'bg-red-600 hover:bg-red-700'
                : 'bg-green-600 hover:bg-green-700'
              : ''
          }
        `}
      >
        {pending ? 'Processing…' : label}
      </button>

      {permissionState === 'denied' && (
        <p className="text-sm text-red-500">
          Location access denied — enable to punch.
        </p>
      )}
      {error && <p className="text-sm text-red-500">{error}</p>}
    </div>
  );
}

import { useState, useCallback } from 'react';

export function useLocation() {
  const [permissionState, setPermissionState] = useState<
    'granted' | 'denied' | 'prompt'
  >('prompt');

  const checkLocation = useCallback(async (): Promise<boolean> => {
    if (!navigator.geolocation) {
      setPermissionState('denied');
      return false;
    }

    try {
      const perm = await (navigator.permissions as any)?.query?.({
        name: 'geolocation',
      });
      if (perm) {
        setPermissionState(perm.state);
        if (perm.state === 'denied') return false;
      }
    } catch {
      // ignore query failures
    }

    return new Promise((resolve) => {
      navigator.geolocation.getCurrentPosition(
        () => resolve(true),
        () => {
          setPermissionState('denied');
          resolve(false);
        },
        { timeout: 10_000, enableHighAccuracy: true }
      );
    });
  }, []);

  return { checkLocation, permissionState };
}

import { useCallback } from 'react';
import type { ToastVariant } from '../types/toast';

export function useToast() {
  const showToast = useCallback((message: string, variant: ToastVariant) => {
    // Replace with your toast system (e.g., Sonner, Toastify, custom)
    const el = document.createElement('div');
    el.className = `toast-notification ${variant}`;
    el.textContent = message;
    Object.assign(el.style, {
      position: 'fixed',
      bottom: '1rem',
      left: '50%',
      transform: 'translateX(-50%)',
      padding: '0.5rem 1rem',
      borderRadius: '0.375rem',
      color: '#fff',
      background: variant === 'success' ? '#16a34a' : '#dc2626',
      zIndex: 9999,
      animation: 'fadeSlideIn 0.2s ease',
    } as React.CSSProperties);

    document.body.appendChild(el);
    setTimeout(() => {
      el.style.opacity = '0';
      el.style.transition = 'opacity 0.2s';
      setTimeout(() => el.remove(), 200);
    }, 3000);
  }, []);

  return { showToast };
}
