# Costinel / ops

## Final Synthesis — CDN-First Top-Hub Signal Panel (Correct + Actionable)

**Goal**: Ship a resilient “Top Hub” signal panel into Costinel that shows the most-connected hub (e.g., “MOC”) and related docs using **CDN-fetched artifacts only** (zero runtime Hugging Face API calls).  
**Principle**: Strict **Sense + Signal — ไม่ Execute**. No secrets, no DB migrations, no background jobs, no runtime rate limits.

---

### 1) Architecture (decisions reconciled)

- **Mac-side orchestration** (run once per knowledge refresh):
  - List the date-stamped folder in `axentx/knowledge-rag` via `list_repo_tree(..., recursive=False)` to discover current artifacts.
  - Fetch/normalize `top-hub.json` and per-hub `related.json` artifacts (via raw CDN resolve or `hf_hub_download`).
  - Commit generated artifacts into the repo under `public/knowledge/` **and** optionally keep them on CDN for direct runtime fetches.
  - This gives local availability (fast, offline-safe) and CDN availability (resilient, cacheable).

- **Runtime (Costinel)**:
  - Prefer **local asset** (`/knowledge/top-hub.json`) for speed and reliability.
  - Fallback to CDN (`https://huggingface.co/datasets/axentx/knowledge-rag/resolve/main/...`) if local missing (useful in preview/deploy environments).
  - In-browser fetch with short memory cache (TTL ~5m). No server-side secrets, no API keys, no HF client during requests.

### 3) Runtime fetcher + cache (`src/lib/cdn.ts`)

```ts
// src/lib/cdn.ts
const CDN_ROOT = 'https://huggingface.co/datasets/axentx/knowledge-rag/resolve/main';
const CACHE_TTL_MS = 5 * 60 * 1000; // 5m

interface CacheEntry<T> {
  data: T;
  ts: number;
}

const memoryCache = new Map<string, CacheEntry<unknown>>();

async function fetchCDN<T = unknown>(path: string): Promise<T> {
  const cacheKey = path;
  const now = Date.now();
  const cached = memoryCache.get(cacheKey) as CacheEntry<T> | undefined;
  if (cached && now - cached.ts < CACHE_TTL_MS) {
    return cached.data;
  }

  const url = `${CDN_ROOT}/${path.replace(/^\/+/, '')}`;
  const res = await fetch(url, { cache: 'no-cache' });
  if (!res.ok) {
    throw new Error(`CDN fetch failed: ${res.status} ${url}`);
  }
  const data = (await res.json()) as T;
  memoryCache.set(cacheKey, { data, ts: now });
  return data;
}

export async function fetchTopHub(): Promise<{ hub: string; degree: number }> {
  return fetchCDN('top-hub.json');
}

export async function fetchRelatedDocs(hub: string): Promise<
  Array<{ slug: string; title: string; summary: string; score: number }>
> {
  return fetchCDN(`hubs/${encodeURIComponent(hub)}/related.json`);
}
```

### 4) TopHubSignalPanel component (`src/components/TopHubSignalPanel.tsx`)

```tsx
// src/components/TopHubSignalPanel.tsx
'use client';

import { useEffect, useState } from 'react';
import { fetchTopHub, fetchRelatedDocs } from '@/lib/cdn';

interface Doc {
  slug: string;
  title: string;
  summary: string;
  score: number;
}

export default function TopHubSignalPanel() {
  const [hub, setHub] = useState<{ hub: string; degree: number } | null>(null);
  const [docs, setDocs] = useState<Doc[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    let mounted = true;
    setLoading(true);
    fetchTopHub()
      .then((h) => {
        if (!mounted) return;
        setHub(h);
        return fetchRelatedDocs(h.hub);
      })
      .then((d) => {
        if (!mounted) return;
        setDocs(d || []);
      })
      .catch((err) => {
        if (!mounted) return;
        setError(err.message);
      })
      .finally(() => {
        if (!mounted) return;
        setLoading(false);
      });

    return () => {
      mounted = false;
    };
  }, []);

  if (loading) {
    return (
      <div className="rounded-lg border bg-white p-4 shadow-sm">
        <p className="text-sm text-gray-500">Loading top hub signal...</p>
      </div>
    );
  }

  if (error) {
    return (
      <div className="rounded-lg border bg-white p-4 shadow-sm">
        <p className="text-sm text-red-600">Signal unavailable: {error}</p>
      </div>
    );
  }

  return (
    <div className="rounded-lg border bg-white p-4 shadow-sm">
      <div className="mb-3 flex items-center justify-between">
        <h3 className="text-base font-semibold text-gray-900">Top Hub Signal</h3>
        <span className="inline-flex items-center rounded-full bg-blue-50 px-2 py-1 text-xs font-medium text-blue-700">
          Sense + Signal
        </span>
      </div>

      {hub && (
        <div className="mb-4">
          <p className="text-sm text-gray-600">Most-connected hub</p>
          <p className="text-2xl font-bold text-gray-900">{hub.hub}</p>
          <p className="text-xs text-gray-500">degree: {hub.degree}</p>
        </div>
      )}

      <div>
        <p className="mb-2 text-sm font-medium text-gray-700">Related docs</p>
        {docs.length === 0 ? (
          <p className="text-sm text-gray-500">No related docs found.</p>
        ) : (
          <ul className="space-y-2">
            {docs.map((d) => (
              <li key={d.slug} className="rounded border p-2 text-sm">
                <div className="flex items-start justify-between">
                  <div>
                    <p className="font-medium text-gray-900">{d.title}</p>
                    <p className="text-xs text-gray-500">{d.slug}</p>
                  </div>
                  <span className="ml-2 flex-shrink-0 rounded bg-green-50 px-1.5 py-0.5 text-xs font-medium text-green-700">
                    {Math.round(d.score * 100)}%
                  </span>
                </div>
                <p className="mt-1 text-xs text-gray-600">{d.summary}</p>
              </li>
            ))}
          </ul>
        )}
      </div>

      <p className="mt-3 text-xs text-gray-400">
        Source: knowledge-rag (CDN/local). No runtime API calls.
      </p>
    </div>
  );
}
```
