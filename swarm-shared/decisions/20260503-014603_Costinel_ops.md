# Costinel / ops

## Implementation Plan — Top-Hub Signal Panel (Costinel Dashboard)

**Scope & value**  
- Frontend-only, read-only panel that surfaces the most-connected hub (default “MOC”) and its actionable proposals from the knowledge graph.  
- Resilient to missing backend: falls back to cached/last-known hub data and gracefully degrades visuals.  
- Ships in <2h: modify existing dashboard layout, add panel component, wire lightweight fetch + cache, no backend changes.

---

### Steps (ordered for fastest safe delivery)

1. **Locate dashboard layout**  
   Identify where to insert panel (likely grid container in `pages/dashboard` or `components/DashboardGrid`).

2. **Create `TopHubSignalPanel` component**  
   - Accept `hubKey` prop (default `"MOC"`).  
   - Fetch `/api/knowledge-graph/hubs/{hubKey}/proposals` (or fallback JSON).  
   - Show: hub title, short description, top 3 proposals (title, impact, action link), last-updated timestamp.  
   - Skeleton loader + empty state.

3. **Add resilient data layer**  
   - Try live endpoint first; on 404/5xx, load `public/data/top-hub-fallback.json`.  
   - Cache successful response to `localStorage` (10-minute TTL) to survive navigation flaps.

4. **Wire into dashboard grid**  
   - Insert `<TopHubSignalPanel />` as first item in the priority row.  
   - Ensure responsive breakpoints match existing grid tokens.

5. **Static fallback content**  
   - Create `public/data/top-hub-fallback.json` with MOC hub and 3 example proposals (derived from last known insights).  
   - Include attribution and timestamp.

6. **Polish & test**  
   - Verify styles in light/dark mode.  
   - Test offline behavior (cached + fallback).  
   - No console errors on missing API.

#### 1) Component: `components/TopHubSignalPanel.tsx`

```tsx
'use client';

import { useEffect, useState } from 'react';
import type { HubData, Proposal } from '@/types/knowledge-graph';

interface TopHubSignalPanelProps {
  hubKey?: string;
  cacheTtlMs?: number;
}

const CACHE_KEY = 'costinel:top-hub:cached';
const CACHE_TTL_DEFAULT = 10 * 60 * 1000; // 10m

function isHubData(x: unknown): x is HubData {
  return (
    typeof x === 'object' &&
    x !== null &&
    'hubKey' in x &&
    'title' in x &&
    'proposals' in x &&
    Array.isArray((x as any).proposals)
  );
}

export default function TopHubSignalPanel({
  hubKey = 'MOC',
  cacheTtlMs = CACHE_TTL_DEFAULT,
}: TopHubSignalPanelProps) {
  const [data, setData] = useState<HubData | null>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    let mounted = true;

    async function load() {
      setLoading(true);
      try {
        // 1) try live API
        const res = await fetch(`/api/knowledge-graph/hubs/${hubKey}/proposals`, {
          cache: 'no-store',
        });

        if (res.ok) {
          const json = await res.json();
          if (isHubData(json)) {
            if (mounted) {
              setData(json);
              localStorage.setItem(
                CACHE_KEY,
                JSON.stringify({ ts: Date.now(), payload: json })
              );
            }
            setLoading(false);
            return;
          }
        }
      } catch {
        // continue to fallback
      }

      // 2) try cache
      try {
        const raw = localStorage.getItem(CACHE_KEY);
        if (raw) {
          const parsed = JSON.parse(raw);
          if (parsed?.payload && Date.now() - parsed.ts < cacheTtlMs) {
            if (isHubData(parsed.payload) && mounted) {
              setData(parsed.payload);
              setLoading(false);
              return;
            }
          }
        }
      } catch {
        // ignore cache errors
      }

      // 3) static fallback
      try {
        const res = await fetch('/data/top-hub-fallback.json', { cache: 'default' });
        if (res.ok) {
          const json = await res.json();
          if (isHubData(json) && mounted) {
            setData(json);
          }
        }
      } catch {
        // final fallback: minimal inline data
        if (mounted) {
          setData({
            hubKey,
            title: 'MOC — Mission Operations Center',
            description: 'Central hub for cloud governance proposals and change coordination.',
            proposals: [],
            updatedAt: new Date().toISOString(),
          });
        }
      } finally {
        if (mounted) setLoading(false);
      }
    }

    load();

    return () => {
      mounted = false;
    };
  }, [hubKey, cacheTtlMs]);

  if (loading) {
    return (
      <div className="rounded-xl border bg-card p-5 shadow-sm">
        <div className="h-6 w-32 rounded bg-muted/50 mb-3 animate-pulse" />
        <div className="h-4 w-full max-w-sm rounded bg-muted/30 mb-2 animate-pulse" />
        <div className="mt-4 space-y-3">
          {Array.from({ length: 3 }).map((_, i) => (
            <div key={i} className="h-10 rounded bg-muted/20 animate-pulse" />
          ))}
        </div>
      </div>
    );
  }

  if (!data || !data.proposals.length) {
    return (
      <div className="rounded-xl border bg-card p-5 shadow-sm">
        <h3 className="text-base font-semibold">{data?.title || hubKey}</h3>
        <p className="text-sm text-muted-foreground mt-1">
          {data?.description || 'No description available.'}
        </p>
        <p className="text-xs text-muted-foreground mt-3">No proposals available.</p>
      </div>
    );
  }

  return (
    <div className="rounded-xl border bg-card p-5 shadow-sm">
      <div className="flex items-start justify-between gap-3">
        <div>
          <h3 className="text-base font-semibold">{data.title}</h3>
          <p className="text-sm text-muted-foreground mt-1">{data.description}</p>
        </div>
        {data.updatedAt && (
          <time className="text-xs text-muted-foreground whitespace-nowrap">
            Updated {new Date(data.updatedAt).toLocaleDateString()}
          </time>
        )}
      </div>

      <ul className="mt-4 space-y-2" role="list">
        {data.proposals.slice(0, 3).map((p: Proposal, idx) => (
          <li key={p.id || idx}>
            <a
              href={p.url || '#'}
              target={p.url ? '_blank' : undefined}
              rel={p.url ? 'noopener noreferrer' : undefined}
              className="block rounded-lg border border-transparent bg-accent/50 p-3 text-sm hover:border-ring/40 transition-colors"
            >
              <span className="font-medium">{p.title}</span>
              {p.impact != null && (
                <span className="ml-2 inline-flex items-center rounded-full px-2 py-0.5 text-xs font-medium bg-primary/10 text-primary">
                  {p.impact}
                </span>
              )}
              {p.summary && (
                <p className="text-xs text-muted-foreground mt-1 line-clamp-2">{p.summary}</p>
              )}
            </a>
          </li>
        ))}
      </ul>
    </div>
  );
}
```
