import { useState, useCallback, useRef } from 'react';
import { useToast } from '@/hooks/use-toast';
import { Loader2, Check, AlertCircle } from 'lucide-react';

type PunchStatus = 'idle' | 'pending' | 'success' | 'error';

interface PunchButtonProps {
  type: 'in' | 'out';
  /** Submit punch. Return ok:true on success. Throw or return ok:false on failure. */
  onPunch: (type: 'in' | 'out', clientId: string) => Promise<{ ok: boolean; error?: string }>;
  disabled?: boolean;
  /** Optional: show last-committed punch in UI (for optimistic list sync) */
  lastCommittedAt?: string | null;
  /** Optional: callback to add optimistic punch to list immediately */
  onOptimisticPunch?: (punch: {
    id: string;
    type: 'IN' | 'OUT';
    timestamp: string;
    status: 'pending' | 'committed';
    clientId: string;
  }) => void;
  /** Optional: callback to update optimistic punch to committed */
  onCommitPunch?: (clientId: string, serverId?: string) => void;
  /** Optional: callback to remove failed optimistic punch */
  onRevertPunch?: (clientId: string) => void;
}

const MIN_SUCCESS_DURATION = 800; // ms (avoid success flicker)
const ERROR_RESET_DELAY = 3000; // ms

export function PunchButton({
  type,
  onPunch,
  disabled,
  lastCommittedAt,
  onOptimisticPunch,
  onCommitPunch,
  onRevertPunch,
}: PunchButtonProps) {
  const [status, setStatus] = useState<PunchStatus>('idle');
  const clientIdRef = useRef<string | null>(null);
  const { toast } = useToast();

  const handlePunch = useCallback(async () => {
    if (status === 'pending' || disabled) return;

    const clientId = `punch-ui-${Date.now()}-${Math.random().toString(36).slice(2)}`;
    clientIdRef.current = clientId;
    const startedAt = Date.now();

    // optimistic state
    setStatus('pending');
    const optimisticId = `pending-${clientId}`;
    const timestamp = new Date().toISOString();

    onOptimisticPunch?.({
      id: optimisticId,
      type: type.toUpperCase() as 'IN' | 'OUT',
      timestamp,
      status: 'pending',
      clientId,
    });

    try {
      const result = await onPunch(type, clientId);
      const elapsed = Date.now() - startedAt;

      if (result.ok) {
        // keep success visible for at least MIN_SUCCESS_DURATION
        if (elapsed < MIN_SUCCESS_DURATION) {
          await new Promise((r) => setTimeout(r, MIN_SUCCESS_DURATION - elapsed));
        }

        setStatus('success');
        onCommitPunch?.(clientId, optimisticId);

        toast({
          title: `Clock ${type.toUpperCase()} recorded`,
          description: 'Your punch was submitted successfully.',
          icon: <Check className="h-4 w-4" />,
        });

        // reset after brief success state
        setTimeout(() => setStatus('idle'), 2000);
      } else {
        throw new Error(result.error || 'Failed to record punch');
      }
    } catch (err) {
      setStatus('error');
      onRevertPunch?.(clientId);

      toast({
        variant: 'destructive',
        title: 'Unable to record punch',
        description: err instanceof Error ? err.message : String(err),
        icon: <AlertCircle className="h-4 w-4" />,
      });

      // allow retry after delay
      setTimeout(() => setStatus('idle'), ERROR_RESET_DELAY);
    }
  }, [type, onPunch, disabled, status, toast, onOptimisticPunch, onCommitPunch, onRevertPunch]);

  const isPending = status === 'pending';
  const isSuccess = status === 'success';
  const isError = status === 'error';

  return (
    <button
      onClick={handlePunch}
      disabled={disabled || isPending || isSuccess}
      className={`
        inline-flex items-center justify-center gap-2 rounded-lg px-6 py-3 font-medium transition-all
        disabled:opacity-60 disabled:cursor-not-allowed
        ${
          type === 'in'
            ? 'bg-green-600 hover:bg-green-700 text-white shadow-sm hover:shadow'
            : 'bg-red-600 hover:bg-red-700 text-white shadow-sm hover:shadow'
        }
        ${isPending ? 'cursor-wait' : ''}
        ${isSuccess ? 'bg-green-700/80' : ''}
        ${isError ? 'bg-red-700/80 ring-2 ring-red-400/40' : ''}
      `}
      aria-busy={isPending}
      data-client-id={clientIdRef.current ?? undefined}
    >
      {isPending && <Loader2 className="h-4 w-4 animate-spin" />}
      {isSuccess && <Check className="h-4 w-4" />}
      {isError && <AlertCircle className="h-4 w-4" />}
      <span className="truncate">
        {isPending && 'Recording...'}
        {isSuccess && 'Recorded'}
        {isError && 'Retry'}
        {!isPending && !isSuccess && !isError && (type === 'in' ? 'Clock In' : 'Clock Out')}
      </span>
    </button>
  );
}
