# workio / backend

Candidate 3:
## 1. Diagnosis

- LINE webhook retries create duplicate punches because ingestion is not idempotent (no dedupe key or idempotency token enforced).
- Clock-in/out path uses non-atomic `findOne` → conditional `insert`/`update`, allowing race conditions under concurrency.
- No tenant-scoped uniqueness constraint on `(tenant_id, user_id, punch_date, punch_type)` so duplicates can persist.
- Webhook handler commits side effects (notifications, state changes) before idempotency check, causing repeated LINE notifications.
- Missing request-level idempotency store (short TTL) to absorb retries within the same day.

### Step 2 — Idempotency middleware (Redis, 1h TTL)

Install:

```bash
cd /opt/axentx/workio/server
npm install ioredis
```

Create `server/src/middleware/idempotency.ts`:

```ts
import { Context, Next } from 'koa';
import Redis from 'ioredis';

const redis = new Redis({
  host: process.env.REDIS_HOST || '127.0.0.1',
  port: Number(process.env.REDIS_PORT || 6379),
  password: process.env.REDIS_PASSWORD || undefined,
});

const IDEMPOTENCY_TTL_SECONDS = 60 * 60; // 1h

function idempotencyKeyFromLineEvent(body: any): string | null {
  if (!body || !body.events || !Array.isArray(body.events)) return null;
  const ev = body.events[0];
  if (ev.deliveryId) return `line:${ev.deliveryId}`;
  const src = ev.source?.userId;
  const type = ev.type;
  const ts = ev.timestamp;
  if (!src || !type || !ts) return null;
  return `line:${src}:${type}:${ts}`;
}

export function lineIdempotency() {
  return async (ctx: Context, next: Next) => {
    const key = idempotencyKeyFromLineEvent(ctx.request.body);
    if (!key) return next();

    const exists = await redis.set(key, '1', 'PX', IDEMPOTENCY_TTL_SECONDS * 1000, 'NX');
    if (!exists) {
      ctx.status = 200;
      ctx.body = { ok: true, duplicate: true };
      return;
    }

    ctx.state.idempotencyKey = key;
    await next();
  };
}
```

### Step 3 — Controller: atomic upsert + transactional side effects

File: `server/src/controllers/lineWebhook.ts`

```ts
import { Context } from 'koa';
import { lineIdempotency } from '../middleware/idempotency';
import { db } from '../db';
import { Router } from 'koa-router';

const router = new Router();

router.post('/line', lineIdempotency(), async (ctx: Context) => {
  const body = ctx.request.body;
  if (!body.events) {
    ctx.status = 400;
    ctx.body = { error: 'invalid payload' };
    return;
  }

  const conn = await db.getConnection();
  try {
    await conn.query('BEGIN');

    for (const ev of body.events) {
      if (ev.type === 'message' && ev.message?.text === 'clock_in') {
        const tenantId = /* resolve tenant from source/user mapping */ 1;
        const userId = /* resolve internal user id from LINE source.userId */ 123;
        const punchDate = new Date().toISOString().slice(0, 10);
        const now = new Date().toISOString();

        // Atomic upsert: insert or do nothing on conflict
        await conn.query(
          `INSERT INTO attendance_punches (tenant_id, user_id, punch_date, punch_type, clock_in, clock_out, location, line_message_id, created_at, updated_at)
           VALUES ($1, $2, $3, $4, $5, $6, $7, $8, NOW(), NOW())
           ON CONFLICT (tenant_id, user_id, punch_date, punch_type) DO NOTHING`,
          [tenantId, userId, punchDate, 'clock_in', now, null, ev.location ? JSON.stringify(ev.location) : null, ev.message?.id || null]
        );

        // Optional notification (only if punch was inserted)
        const inserted = await conn.query(
          `SELECT 1 FROM attendance_punches WHERE tenant_id = $1 AND user_id = $2 AND punch_date = $3 AND punch_type = $4`,
          [tenantId, userId, punchDate, 'clock_in']
        );

        if (inserted.rows.length > 0) {
          await conn.query(
            `INSERT INTO notifications (tenant_id, user_id, type, message, read, created_at)
             VALUES ($1, $2, $3, $4, false, NOW())`,
            [tenantId, userId, 'clock_in', `Clock-in recorded at ${now}`]
          );
        }
      }
      // handle clock_out / leave / ot similarly
    }

    await conn.query('COMMIT');
    ctx.status = 200;
    ctx.body = { ok: true };
  } catch (err) {
    await conn.query('ROLLBACK');
    ctx.status = 500;
    ctx.body = { error: 'processing failed' };
    console.error(err);
  } finally {
    conn.release();
  }
});

export default router;
```
