### Final Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **deterministic, manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16`, `DATE`, `HF_TOKEN`, `REPO_ID` (default: `axentx/surrogate-1-training-pairs`)
- Single API call to list one date folder via `list_repo_tree(recursive=False)` → deterministic shard assignment by hash(slug)
- Downloads only assigned files via **HF CDN bypass** (`https://huggingface.co/datasets/.../resolve/main/...`) with no Authorization header to avoid 429 rate limits
- Streams, normalizes per-schema, dedups via central md5 store (`lib/dedup.py`), projects to `{prompt, response}`
- Writes `batches/public-merged/{DATE}/shard{N}-{HHMMSS}.jsonl` (one file per shard per run)
- Exits 0 on success; non-zero on fatal error (GitHub Actions will retry)

Why this is the highest-value incremental improvement:
- Eliminates `load_dataset(streaming=True)` mixed-schema `pyarrow.CastError` (known failure mode)
- Bypasses HF API rate limits during data loading (CDN tier has much higher limits)
- Keeps parallel 16-shard architecture and RAM isolation (7 GB/runner)
- Deterministic sharding prevents commit collisions and enables idempotent retries
- Fits <2h: small focused Python script + minimal workflow tweak

def _now() -> str:
    return datetime.now(timezone.utc).strftime("%Y-%m-%dT%H:%M:%SZ")


def _slug_from_path(path: str) -> str:
    """Stable slug used for sharding and dedup."""
    return path.strip("/").replace("/", "--")


def _hash_shard(slug: str, shard_total: int) -> int:
    """Deterministic shard assignment."""
    digest = int(hashlib.sha256(slug.encode()).hexdigest(), 16)
    return digest % shard_total


def list_date_files(
    repo_id: str,
    date_folder: str,
    token: Optional[str],
) -> List[str]:
    """
    Single API call to list files in a date folder (non-recursive).
    Retries on 429 with exponential-ish backoff.
    """
    headers = {"Authorization": f"Bearer {token}"} if token else {}
    url = f"{HF_API_BASE}/datasets/{repo_id}/tree/{date_folder}"
    params = {"recursive": "false"}

    for attempt in range(MAX_RETRIES):
        try:
            resp = httpx.get(url, headers=headers, params=params, timeout=30.0)
            if resp.status_code == 429:
                wait = RETRY_WAIT
                print(f"[{_now()}] Rate limited 429, waiting {wait}s", file=sys.stderr)
                time.sleep(wait)
                continue
            resp.raise_for_status()
            items = resp.json()
            # Expect list of {"path": "...", "type": "file"}
            files = [item["path"] for item in items if item.get("type") == "file"]
            print(f"[{_now()}] Listed {len(files)} files in {date_folder}", file=sys.stderr)
            return files
        except httpx.HTTPStatusError as e:
            if e.response.status_code == 429:
                wait = RETRY_WAIT
                print(f"[{_now()}] Rate limited 429, waiting {wait}s", file=sys.stderr)
                time.sleep(wait)
                continue
            raise
        except Exception as e:
            print(f"[{_now()}] Error listing files: {e}", file=sys.stderr)
            raise

    raise RuntimeError(f"Failed to list files after {MAX_RETRIES} attempts (429)")


def download_via_cdn(repo_id: str, file_path: str) -> bytes:
    """
    CDN bypass: no Authorization header. Public CDN endpoint.
    """
    url = f"{HF_CDN_BASE}/{repo_id}/resolve/main/{file_path}"
    for attempt in range(MAX_RETRIES):
        try:
            resp = httpx.get(url, timeout=60.0)
            resp.raise_for_status()
            return resp.content
        except httpx.HTTPStatusError as e:
            if e.response.status_code == 429:
                wait = RETRY_WAIT
                print(f"[{_now()}] CDN 429, waiting {wait}s", file=sys.stderr)
                time.sleep(wait)
                continue
            raise
        except Exception as e:
            print(f"[{_now()}] CDN download failed {file_path}: {e}", file=sys.stderr)
            raise

    raise RuntimeError(f"CDN download failed after {MAX_RETRIES} attempts")


def normalize_record(raw: Dict[str, Any], source_path: str) -> Optional[Dict[str, str]]:
    """
    Normalize mixed-schema records into {prompt, response}.
    Returns None for unrecognized/malformed records.
    """
    if not isinstance(raw, dict):
        return None

    # Common known schemas
    prompt = None
    response = None

    # Schema 1: explicit prompt/response
    if "prompt" in raw and "response" in raw:
        prompt = str(raw["prompt"]).strip()
        response = str(raw["response"]).strip()
    # Schema 2: instruction/input/output
    elif "instruction" in raw and "output" in raw:
        prompt = str(raw["instruction"]).strip()
        if "input" in raw and str(raw["input"]).strip():
            prompt = f"{prompt}\n\n{str(raw['input']).strip()}"
        response = str(raw["output"]).strip()
    # Schema 3: question/answer
    elif "question" in raw and "answer" in raw:
        prompt = str(raw["question"]).strip()
        response = str(raw["answer"]).strip()
    # Schema 4: text completion style
    elif "text" in raw:
        text = str(raw["text"]).strip()
        # Heuristic split on last newline or double newline if present
        parts = text.rsplit("\n\n", 1)
        if len(parts) == 2:
            prompt, response = parts[0].strip(), parts[1].strip()
        else:
            parts = text.rsplit("\n", 1)
            if len(parts) == 2:
                prompt, response = parts[0].strip(), parts[1].strip()
            else:
                return None
    else:
        return None

    if not prompt or not response:
        return None

    # Basic length sanity
    if len(prompt) < 4 or len(response) < 4:
        return None

    return {"prompt": prompt, "response": response, "_source": source_path}
