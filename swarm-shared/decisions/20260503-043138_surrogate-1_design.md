# surrogate-1 / design

## Implementation Plan (≤2h)

**Highest-value improvement**: Replace fragile shell-based ingestion with a **manifest-driven, CDN-bypass Python worker** that eliminates HF API rate limits during training data loads and prevents mixed-schema CastErrors.

### Changes

1. **Add `bin/manifest.py`** — single Mac-side script that lists one date folder via `list_repo_tree(recursive=False)` once, saves `manifest.json` with CDN URLs and expected slugs. Embeds this manifest into the repo so GitHub runners never call `list_repo_files`/`load_dataset` during ingestion.

2. **Replace `bin/dataset-enrich.sh` with `bin/worker.py`** — Python worker that:
   - Reads `manifest.json` (or accepts it via env)
   - Filters to its 1/16 shard by `hash(slug) % 16 == SHARD_ID`
   - Downloads each file directly from HF CDN (`resolve/main/...`) with no Authorization header (bypasses API rate limits)
   - Projects to `{prompt, response}` only at parse time (avoids pyarrow CastError from mixed schemas)
   - Streams normalized JSONL lines to stdout or direct upload
   - Uses `lib/dedup.py` for cross-run dedup via central md5 store

3. **Update `.github/workflows/ingest.yml`** — switch matrix runner steps to run `python bin/worker.py` instead of shell script; pass `SHARD_ID`, `DATE`, and optional `MANIFEST_PATH`.

4. **Add lightweight schema adapters** in `lib/schema.py` — per-source projection rules so new file types can be added without touching worker core.

5. **Keep backward compatibility** — if `manifest.json` is absent, worker falls back to `huggingface_hub` list calls (but warns). This lets the cron continue working during transition.

### Code Snippets

#### `bin/manifest.py` (run on Mac, commit or embed manifest)
```python
#!/usr/bin/env python3
"""
Generate manifest for one date folder to avoid HF API calls during ingestion.
Usage:
  python bin/manifest.py --repo axentx/surrogate-1-training-pairs --date 2026-05-03 > manifest.json
"""
import argparse
import json
import os
import sys
from huggingface_hub import HfApi

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", default="axentx/surrogate-1-training-pairs")
    parser.add_argument("--date", required=True, help="Folder name, e.g. 2026-05-03")
    parser.add_argument("--out", default="-")
    args = parser.parse_args()

    api = HfApi()
    prefix = f"batches/public-merged/{args.date}/"
    # Single non-recursive call (avoids pagination + rate limits)
    files = api.list_repo_tree(repo_id=args.repo, path=prefix, recursive=False)

    entries = []
    for f in files:
        if not f.path.startswith(prefix):
            continue
        slug = os.path.splitext(os.path.basename(f.path))[0]
        entries.append(
            {
                "path": f.path,
                "slug": slug,
                "cdn_url": CDN_TEMPLATE.format(repo=args.repo, path=f.path),
                "size": getattr(f, "size", None),
            }
        )

    manifest = {
        "repo": args.repo,
        "date": args.date,
        "prefix": prefix,
        "generated_by": "manifest.py",
        "entries": entries,
    }

    if args.out == "-":
        json.dump(manifest, sys.stdout, indent=2)
    else:
        with open(args.out, "w", encoding="utf-8") as fh:
            json.dump(manifest, fh, indent=2)

if __name__ == "__main__":
    main()
```

#### `lib/schema.py` (project mixed schemas safely)
```python
"""
Minimal schema adapters: project heterogeneous repo files to {prompt, response}.
Add new adapters as needed without touching worker core.
"""
import logging
from typing import Any, Dict, Optional

logger = logging.getLogger(__name__)

def project_record(raw: Dict[str, Any], source_slug: str) -> Optional[Dict[str, str]]:
    """
    Return {prompt, response} or None if record should be skipped.
    """
    if not raw:
        return None

    # Common patterns observed in surrogate-1 datasets
    prompt = raw.get("prompt") or raw.get("input") or raw.get("question") or raw.get("instruction")
    response = raw.get("response") or raw.get("output") or raw.get("answer") or raw.get("completion")

    # Fallback: try to find any text-like fields
    if prompt is None or response is None:
        text_keys = [k for k in raw if isinstance(raw[k], str) and len(raw[k]) > 20]
        if len(text_keys) >= 2:
            prompt, response = raw[text_keys[0]], raw[text_keys[1]]

    if not isinstance(prompt, str) or not isinstance(response, str):
        logger.debug("Skipping record from %s: missing prompt/response", source_slug)
        return None

    return {"prompt": prompt.strip(), "response": response.strip()}
```

#### `bin/worker.py` (replaces `dataset-enrich.sh`)
```python
#!/usr/bin/env python3
"""
CDN-bypass ingestion worker for surrogate-1.
Usage (GitHub Actions):
  python bin/worker.py --shard $SHARD_ID --date $DATE [--manifest manifest.json]
"""
import argparse
import hashlib
import json
import logging
import os
import sys
from typing import Dict, List
import requests

try:
    from lib.dedup import is_duplicate, mark_seen
    from lib.schema import project_record
except ImportError as e:
    print("ERROR: Missing lib modules. Ensure PYTHONPATH includes repo root.", file=sys.stderr)
    sys.exit(1)

CDN_TIMEOUT = int(os.getenv("CDN_TIMEOUT", "30"))
BATCH_SIZE = int(os.getenv("BATCH_SIZE", "500"))

logging.basicConfig(level=logging.INFO, format="%(asctime)s %(levelname)s %(message)s")
log = logging.getLogger("worker")

def shard_id(slug: str, n_shards: int = 16) -> int:
    return int(hashlib.md5(slug.encode()).hexdigest(), 16) % n_shards

def load_manifest(path: str) -> Dict:
    with open(path, "r", encoding="utf-8") as fh:
        return json.load(fh)

def download_cdn(url: str) -> bytes:
    # No Authorization header -> bypasses HF API rate limits
    resp = requests.get(url, timeout=CDN_TIMEOUT)
    resp.raise_for_status()
    return resp.content

def process_entries(entries: List[Dict], my_shard: int, date: str) -> None:
    out_lines = []
    total = len(entries)
    for idx, entry in enumerate(entries, 1):
        slug = entry["slug"]
        if shard_id(slug) != my_shard:
            continue

        url = entry["cdn_url"]
        try:
            raw_bytes = download_cdn(url)
            # Most files are JSON/JSONL; fallback to line-by-line if needed
            text = raw_bytes.decode("utf-8")
            # Try JSONL first
            lines = [ln.strip() for ln in text.splitlines() if ln.strip()]
            for ln in lines:
                try:
                    raw = json.loads(ln)
                except json.JSONDecodeError:
                    continue
                rec = project_record(raw, slug)
                if rec is None:
                    continue
                md5 = hashlib.md5(json.dumps(rec, sort_keys=True).encode()).hexdigest()
                if is_duplicate(md5):
                    continue
                mark_seen(md5)
                rec["_meta"] = {"slug": slug, "source": "public-merged", "date": date}
                out_lines.append(json.dumps(rec, ensure_ascii=False))
        except Exception as exc:
            log.warning("Failed to process %s: %s", url, exc)
            continue

        if idx % 50 == 0:
            log.info("Processed %s/%s entries, produced %s lines", idx, total, len(out_lines))
