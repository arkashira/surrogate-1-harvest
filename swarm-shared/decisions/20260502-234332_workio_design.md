# workio / design

## Final synthesized implementation

**Chosen scope and priorities**
- Create a single, canonical route catalog with **compile-time types + runtime validation** for params and search params.
- Keep URLs unchanged; only add contracts and typed helpers.
- Prefer **Zod** for search/location state validation (correctness + actionability) while keeping param types lightweight via `useParams<T>`.
- Provide **path builders** to eliminate string duplication and guard against missing params with early throws/redirects.
- Minimal surface: `src/routes/` (catalog + hooks + schemas) and updates to route modules.

---

### 1) Route catalog and types  
`src/routes/index.ts`

```ts
import { z } from 'zod';
import { Path } from 'history';

// IDs
export type TenantId = string;
export type EmployeeId = string;

// Param schemas (for useParams generics)
export interface AppParams {
  tenantId?: TenantId;
}

export interface EmployeeParams {
  id: EmployeeId;
}

export interface AttendanceParams {
  tenantId: TenantId;
}

// Location state contracts
export interface LocationStateFromDashboard {
  fromDashboard: true;
  employeeName?: string;
}

export type KnownLocationState = LocationStateFromDashboard | null;

// Search param schemas (runtime validation + defaults)
export const attendanceSearchSchema = z.object({
  from: z.string().regex(/^\d{4}-\d{2}-\d{2}$/).optional(),
  to: z.string().regex(/^\d{4}-\d{2}-\d{2}$/).optional(),
  page: z.coerce.number().int().positive().optional(),
  pageSize: z.coerce.number().int().positive().optional(),
  employeeId: z.string().optional(),
});

export type AttendanceSearchInput = z.input<typeof attendanceSearchSchema>;
export type AttendanceSearch = z.output<typeof attendanceSearchSchema>;

// Path templates + builders (single source of truth)
export const routes = {
  root: '/',
  login: '/login',

  tenantDashboard: '/tenant/:tenantId/dashboard',
  tenantEmployees: '/tenant/:tenantId/employees',
  employeeDetail: '/tenant/:tenantId/employees/:id',
  employeeCreate: '/tenant/:tenantId/employees/new',

  attendance: '/tenant/:tenantId/attendance',

  build: {
    root: () => '/',
    login: () => '/login',
    tenantDashboard: (tenantId: TenantId) => `/tenant/${tenantId}/dashboard`,
    tenantEmployees: (tenantId: TenantId) => `/tenant/${tenantId}/employees`,
    employeeDetail: (tenantId: TenantId, id: EmployeeId) =>
      `/tenant/${tenantId}/employees/${id}`,
    employeeCreate: (tenantId: TenantId) => `/tenant/${tenantId}/employees/new`,
    attendance: (tenantId: TenantId) => `/tenant/${tenantId}/attendance`,
  },
} as const;

// Utility to coerce route entries into react-router route objects if desired
export type RouteCatalog = typeof routes;
```

### 2) Typed hooks  
`src/routes/hooks.ts`

```ts
import { useParams, useLocation, useSearchParams } from 'react-router-dom';
import { AttendanceSearch, attendanceSearchSchema, KnownLocationState } from '.';
import { Navigate } from 'react-router-dom';
import { ReactNode } from 'react';

// Guard component for missing required params
export function RequireParams<T extends Record<string, string | undefined>>(
  params: T,
  required: (keyof T)[],
  fallback?: ReactNode
) {
  const missing = required.find((k) => !params[k]);
  if (missing) {
    // In production you may render an error boundary or redirect
    if (fallback) return <>{fallback}</>;
    return <Navigate to="/" replace />;
  }
  return null;
}

// Typed params (lightweight)
export function useTypedEmployeeParams() {
  const params = useParams<{ id: string }>();
  if (!params.id) {
    throw new Error('Missing required param: id');
  }
  return { id: params.id };
}

export function useTypedAttendanceParams() {
  const params = useParams<{ tenantId: string }>();
  if (!params.tenantId) {
    throw new Error('Missing required param: tenantId');
  }
  return { tenantId: params.tenantId };
}

// Typed location state
export function useTypedLocationState(): KnownLocationState {
  const { state } = useLocation();
  if (!state || typeof state !== 'object') return null;
  const s = state as any;
  if (s.fromDashboard === true) {
    return { fromDashboard: true, employeeName: s.employeeName };
  }
  return null;
}

// Typed search with validation + defaults
export function useTypedAttendanceSearch(): AttendanceSearch {
  const [searchParams] = useSearchParams();
  const raw: AttendanceSearchInput = {
    from: searchParams.get('from') || undefined,
    to: searchParams.get('to') || undefined,
    page: searchParams.get('page') || undefined,
    pageSize: searchParams.get('pageSize') || undefined,
    employeeId: searchParams.get('employeeId') || undefined,
  };

  const result = attendanceSearchSchema.safeParse(raw);
  const today = new Date().toISOString().slice(0, 10);

  return {
    from: result.success && result.data.from ? result.data.from : today,
    to: result.success && result.data.to ? result.data.to : today,
    page: result.success && typeof result.data.page === 'number' ? result.data.page : 1,
    pageSize: result.success && typeof result.data.pageSize === 'number' ? result.data.pageSize : 20,
    employeeId: result.success ? result.data.employeeId : undefined,
  };
}
```
