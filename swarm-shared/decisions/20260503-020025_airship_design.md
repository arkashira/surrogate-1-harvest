### 1. `mirror/cdn_ingest.py` (Python, CDN-first, deterministic sharding)

```python
#!/usr/bin/env python3
"""
mirror/cdn_ingest.py
Usage: python mirror/cdn_ingest.py <date> <upstream_repo>

Produces:
- mirror/batches/mirror-merged/<date>/surrogate-1-{i}.parquet
- mirror/train_filelist_<date>.json
"""
import json, hashlib, os, sys, requests, pyarrow as pa, pyarrow.parquet as pq, pandas as pd
from pathlib import Path
from typing import List, Dict
from huggingface_hub import list_repo_tree

SIBLINGS = [f"surrogate-1-{i}" for i in range(5)]

def shard_repo(slug: str) -> str:
    h = int(hashlib.sha256(slug.encode()).hexdigest(), 16)
    return SIBLINGS[h % len(SIBLINGS)]

def cdn_url(upstream_repo: str, path: str) -> str:
    return f"https://huggingface.co/datasets/{upstream_repo}/resolve/main/{path}"

def project_to_pr(raw_bytes: bytes, filename: str) -> Dict[str, str]:
    # Adapt to your actual raw format.
    try:
        data = json.loads(raw_bytes)
        return {"prompt": data.get("prompt", ""), "response": data.get("response", "")}
    except Exception:
        return {"prompt": raw_bytes.decode(errors="replace"), "response": ""}

def main(date: str, upstream_repo: str, out_root: str = "mirror/batches/mirror-merged"):
    out_root_p = Path(out_root) / date
    out_root_p.mkdir(parents=True, exist_ok=True)
    filelist_path = Path("mirror") / f"train_filelist_{date}.json"

    # 1) Snapshot file list (single API call per folder)
    print(f"Listing {upstream_repo}/{date}/ ...")
    try:
        items = list_repo_tree(repo=upstream_repo, path=date, recursive=False)
    except Exception as e:
        print(f"Error listing repo: {e}")
        sys.exit(1)

    file_paths = []
    for it in items:
        if it.type == "file":
            file_paths.append(f"{date}/{it.path.split('/')[-1]}")
        else:
            try:
                sub = list_repo_tree(repo=upstream_repo, path=it.path, recursive=False)
                for s in sub:
                    if s.type == "file":
                        file_paths.append(f"{it.path}/{s.path.split('/')[-1]}")
            except Exception:
                continue

    print(f"Found {len(file_paths)} files")
    filelist_path.write_text(json.dumps(file_paths, indent=2))

    # 2) Download via CDN + project + shard
    records = []
    for fpath in file_paths:
        url = cdn_url(upstream_repo, fpath)
        try:
            r = requests.get(url, timeout=30)
            r.raise_for_status()
        except Exception as e:
            print(f"Failed to download {fpath}: {e}")
            continue
        proj = project_to_pr(r.content, fpath)
        slug = f"{date}/{Path(fpath).stem}"
        records.append({"slug": slug, "prompt": proj["prompt"], "response": proj["response"]})

    if not records:
        print("No records to write.")
        return

    df = pd.DataFrame(records)
    table = pa.Table.from_pandas(df, preserve_index=False)

    # Write one parquet per shard group
    for shard in SIBLINGS:
        mask = [shard_repo(s) == shard for s in df["slug"]]
        if not any(mask):
            continue
        part = df[mask]
        out = out_root_p / f"{shard}.parquet"
        pq.write_table(pa.Table.from_pandas(part, preserve_index=False), str(out))
        print(f"Wrote {len(part)} rows -> {out}")

    print("CDN ingestion complete for", date)

if __name__ == "__main__":
    date_arg = sys.argv[1] if len(sys.argv) > 1 else "2026-05-03"
    repo_arg = sys.argv[2] if len(sys.argv) > 2 else "datasets/surrogate-1-raw"
    main(date_arg, repo_arg)
```

last_progress = time.time()

def reset_idle_timer():
    global last_progress

def idle_check():
    if time.time() - last_progress > IDLE_TIMEOUT_MIN * 60:
        print(f"No progress for {IDLE_TIMEOUT_MIN} min. Exiting to save quota.")
