# vanguard / ops

## 1. Diagnosis
- No persisted `(repo, dateFolder) → file-list` manifest: every training/data-selection run triggers authenticated `list_repo_tree` against HF API, burning quota and risking 429s.
- Training/data loader likely uses `load_dataset(streaming=True)` or repeated per-file API calls on heterogeneous repos, causing `pyarrow.CastError` from mixed schemas.
- No CDN-only data path: authenticated API calls used for file fetches during training instead of public CDN URLs, wasting rate-limit budget.
- No studio reuse guard: training script may recreate Lightning Studio instead of reusing running instances, burning 80+ hrs/mo quota.
- No idle-stop resilience: Lightning Studio idle timeout kills long-running training; no auto-restart on stopped state.

## 2. Proposed change
Create `/opt/axentx/vanguard/ops/persist_filelist.py` (single orchestration script, Mac-safe) that:
- Runs once per `(repo, dateFolder)` after rate-limit window clears
- Calls `list_repo_tree(path, recursive=False)` once, saves `{repo}/{dateFolder}/files.json`
- Emits a deterministic `train_manifest.json` containing public CDN URLs for each file (bypasses HF API auth during training)
- Prints Lightning Studio reuse + idle-resume commands for downstream training script

Also create `/opt/axentx/vanguard/train/train.py` patch that:
- Accepts `--manifest train_manifest.json`
- Uses `hf_hub_download` (or direct CDN `requests`/`wget`) per file listed in manifest; projects to `{prompt, response}` only
- Reuses running Lightning Studio or starts L40S in priority order; checks status before `.run()` and restarts if idle-killed

### /opt/axentx/vanguard/ops/persist_filelist.py
```python
#!/usr/bin/env python3
"""
Mac-safe orchestration: persist file-list + CDN manifest for a repo/dateFolder.
Run once per dataset version to avoid HF API rate-limits during training.
"""
import argparse
import json
import os
import sys
from datetime import datetime
from pathlib import Path

try:
    from huggingface_hub import list_repo_tree, hf_hub_download
except ImportError:
    print("Install: pip install huggingface_hub")
    sys.exit(1)

HF_REPO = "axentx/surrogate-1"  # adjust if needed
OUT_DIR = Path(__file__).parent.parent / "data" / "manifests"

def build_manifest(repo: str, date_folder: str, out_dir: Path):
    out_dir.mkdir(parents=True, exist_ok=True)
    cache_file = out_dir / f"{repo.replace('/', '_')}_{date_folder}_files.json"
    manifest_file = out_dir / f"{repo.replace('/', '_')}_{date_folder}_manifest.json"

    # 1) Single API call: list top-level folder (non-recursive)
    print(f"Listing {repo}/{date_folder} ...")
    try:
        tree = list_repo_tree(repo, path=date_folder, recursive=False)
    except Exception as e:
        print(f"HF API error (possibly rate-limited): {e}")
        print("Retry after window clears or use cached manifest if present.")
        if cache_file.exists():
            print(f"Using cached: {cache_file}")
            with open(cache_file) as f:
                tree = json.load(f)
        else:
            sys.exit(1)

    entries = [t for t in tree if t.get("type") == "file"]
    filenames = [t["path"].split("/")[-1] for t in entries]

    # 2) Persist raw file list (for reruns)
    with open(cache_file, "w") as f:
        json.dump(entries, f, indent=2)
    print(f"Saved file list: {cache_file}")

    # 3) Build CDN-only manifest (zero auth during training)
    cdn_base = f"https://huggingface.co/datasets/{repo}/resolve/main"
    manifest = {
        "repo": repo,
        "date_folder": date_folder,
        "created_at": datetime.utcnow().isoformat() + "Z",
        "files": [],
        "strategy": "cdn_bypass"
    }

    for entry in entries:
        rel = entry["path"]
        filename = rel.split("/")[-1]
        cdn_url = f"{cdn_base}/{rel}"
        # Optional local cache via hf_hub_download path (for validation)
        local_path = str(Path.home() / ".cache" / "huggingface" / "hub" / rel.replace("/", "--"))
        manifest["files"].append({
            "filename": filename,
            "cdn_url": cdn_url,
            "hf_path": rel,
            "size": entry.get("size"),
            "lfs": entry.get("lfs", {}),
            "_local_cache_hint": local_path
        })

    with open(manifest_file, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Saved CDN manifest: {manifest_file}")
    print("\nNext steps:")
    print(f"  python /opt/axentx/vanguard/train/train.py --manifest {manifest_file}")
    return manifest_file

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Persist HF file-list + CDN manifest")
    parser.add_argument("--repo", default=HF_REPO, help="HF repo (datasets/)")
    parser.add_argument("--date-folder", required=True, help="Folder in repo, e.g. batches/mirror-merged/2026-04-29")
    parser.add_argument("--out-dir", default=str(OUT_DIR), help="Output directory for manifests")
    args = parser.parse_args()
    build_manifest(args.repo, args.date_folder, Path(args.out_dir))
```
