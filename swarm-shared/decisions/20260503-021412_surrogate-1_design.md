# surrogate-1 / design

## Final Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID` and `SHARD_TOTAL` (16) from GitHub Actions matrix.
- Uses a pre-generated `file-list.json` (Mac-side, single `list_repo_tree` call) to deterministically assign 1/16 of files per shard (hash-slug → bucket).
- Downloads assigned files via **HF CDN bypass** (`https://huggingface.co/datasets/.../resolve/main/...`) — zero API calls during training, avoids 429/128-commit limits.
- Projects heterogeneous schemas to `{prompt,response}` only at parse time (avoids pyarrow CastError).
- Deduplicates via central `lib/dedup.py` md5 store (same as Space).
- Writes `batches/public-merged/{date}/shard{N}-{HHMMSS}.jsonl` with no extra metadata columns (filename carries attribution).
- Reuses existing `requirements.txt` (datasets, huggingface_hub, pyarrow, numpy).
- Keeps cron/workflow unchanged; only worker implementation changes.

def deterministic_shard(slug: str, total: int) -> int:
    """Map slug -> bucket in [0, total). Stable across runs."""
    digest = hashlib.sha256(slug.encode()).hexdigest()
    return int(digest, 16) % total

def load_file_list() -> list[str]:
    p = pathlib.Path("file-list.json")
    if not p.is_file():
        print("ERROR: file-list.json not found. Generate on Mac via:", file=sys.stderr)
        print("  python -c \"from huggingface_hub import list_repo_tree; ...\"", file=sys.stderr)
        sys.exit(1)
    with p.open() as f:
        data = json.load(f)
    # Accept either list of paths or tree objects with 'path'
    paths = []
    for item in data:
        if isinstance(item, str):
            paths.append(item)
        elif isinstance(item, dict) and "path" in item:
            paths.append(item["path"])
    return [p for p in paths if p.strip()]

def cdn_url(repo_path: str) -> str:
    return f"{BASE_CDN}/{repo_path.lstrip('/')}"

def download_cdn(url: str, timeout: int = 30) -> Optional[bytes]:
    """Download via CDN (no auth). Retry once on failure."""
    for attempt in range(2):
        try:
            resp = requests.get(url, timeout=timeout)
            resp.raise_for_status()
            return resp.content
        except Exception as exc:
            if attempt == 0:
                time.sleep(2)
                continue
            print(f"CDN download failed: {url} -> {exc}", file=sys.stderr)
            return None
    return None

def project_to_pair(raw: Dict[str, Any], file_path: str) -> Optional[Dict[str, str]]:
    """
    Heuristic projection for heterogeneous HF datasets.
    Returns {prompt, response} or None if unusable.
    """
    # Normalize keys
    low = {k.lower(): v for k, v in raw.items()}

    # Common patterns
    prompt_keys = {"prompt", "instruction", "question", "input", "user", "query"}
    response_keys = {"response", "completion", "answer", "output", "assistant", "system"}

    prompt = None
    response = None

    for pk in prompt_keys:
        if pk in low and isinstance(low[pk], str) and low[pk].strip():
            prompt = low[pk].strip()
            break
    for rk in response_keys:
        if rk in low and isinstance(low[rk], str) and low[rk].strip():
            response = low[rk].strip()
            break

    # Fallback: if exactly two text fields, treat as pair
    if prompt is None or response is None:
        text_fields = [v for v in low.values() if isinstance(v, str) and v.strip()]
        if len(text_fields) == 2:
            prompt, response = text_fields[0].strip(), text_fields[1].strip()

    if not prompt or not response:
        return None

    return {"prompt": prompt, "response": response}
