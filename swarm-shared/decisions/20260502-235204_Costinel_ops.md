**Principle**: “Sense + Signal — ไม่ Execute” (strictly no writes, no self-execution)

### 4) Frontend component (45 min)
Create `/opt/axentx/Costinel/src/components/TopHubSignalCard.vue`:

```vue
<template>
  <div class="top-hub-card">
    <div class="header">
      <span class="pill">Top Hub</span>
      <span class="label">{{ data.label }}</span>
      <span class="score">{{ (data.score * 100).toFixed(0) }}</span>
    </div>

    <svg v-if="data.trend.length" class="sparkline" viewBox="0 0 100 20">
      <polyline
        :points="points"
        fill="none"
        stroke="currentColor"
        stroke-width="1.5"
      />
    </svg>

    <ul class="signals">
      <li v-for="s in data.signals" :key="s.id" :class="s.severity">
        {{ s.title }}
      </li>
    </ul>

    <div class="footer">
      <small>Updated {{ ago(data.updatedAt) }}</small>
      <router-link :to="`/knowledge-rag/hubs/${data.hubId}`">Details →</router-link>
    </div>
  </div>
</template>

<script>
export default {
  name: 'TopHubSignalCard',
  data() {
    return { data: { hubId: '', label: '', score: 0, trend: [], signals: [], updatedAt: '' } };
  },
  computed: {
    points() {
      const vals = this.data.trend;
      const min = Math.min(...vals);
      const max = Math.max(...vals);
      const range = max - min || 1;
      return vals
        .map((v, i) => {
          const x = (i / (vals.length - 1)) * 100;
          const y = 20 - ((v - min) / range) * 20;
          return `${x},${y}`;
        })
        .join(' ');
    }
  },
  async mounted() {
    const res = await fetch('/api/hubs/top');
    this.data = await res.json();
  },
  methods: {
    ago(ts) {
      const d = new Date(ts);
      const diff = (Date.now() - d) / 60000;
      return diff < 60 ? `${Math.round(diff)}m ago` : `${Math.round(diff / 60)}h ago`;
    }
  }
};
</script>
```
