# surrogate-1 / ops

## Final Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16`, `DATE`, `HF_TOKEN`, `REPO_ID` (default: `axentx/surrogate-1-training-pairs`)
- Pre-lists target folder via **single** `list_repo_tree` call (non-recursive) for `public-raw/{DATE}/`, saves manifest JSON
- Deterministic shard assignment: `hash(slug) % SHARD_TOTAL == SHARD_ID`
- Downloads only assigned files via **CDN bypass** (`https://huggingface.co/datasets/.../resolve/main/...`) — **zero API/auth calls during data fetch**
- Projects each file to `{prompt, response}` at parse time (handles mixed schemas per surrogate-1 pattern)
- Deduplicates via central `lib/dedup.py` md5 store
- Writes output to `batches/public-merged/{DATE}/shard{SHARD_ID}-{HHMMSS}.jsonl`
- Exits 0 on success, non-zero on hard failure (GitHub Actions will retry)

### Why this is the highest-value incremental improvement
- Directly applies the **HF CDN bypass** and **manifest pre-list** patterns — eliminates 429s during training and reduces API pressure.
- Replaces brittle shell script with a testable, typed Python worker that handles schema heterogeneity and dedup consistently.
- Keeps runtime <2h: minimal refactor, reuses existing `lib/dedup.py`, no infra changes.

#!/usr/bin/env python3
"""
CDN-bypass ingestion worker for surrogate-1 public dataset shards.

Usage (via GitHub Actions matrix):
  SHARD_ID=0 SHARD_TOTAL=16 DATE=2026-05-03 \
  HF_TOKEN=hf_xxx \
  python bin/dataset-enrich.py

Env:
  SHARD_ID          int   [required]
  SHARD_TOTAL       int   default=16
  DATE              str   [required]  YYYY-MM-DD
  HF_TOKEN          str   [required]
  REPO_ID           str   default=axentx/surrogate-1-training-pairs
  RAW_PREFIX        str   default=public-raw
  OUT_PREFIX        str   default=batches/public-merged
"""

import os
import sys
import json
import hashlib
import datetime
import subprocess
from pathlib import Path
from typing import List, Dict, Any

import requests
from huggingface_hub import HfApi

# ── config --
SHARD_ID = int(os.environ["SHARD_ID"])
SHARD_TOTAL = int(os.environ.get("SHARD_TOTAL", 16))
DATE = os.environ["DATE"]
HF_TOKEN = os.environ["HF_TOKEN"]
REPO_ID = os.environ.get("REPO_ID", "axentx/surrogate-1-training-pairs")
RAW_PREFIX = os.environ.get("RAW_PREFIX", "public-raw")
OUT_PREFIX = os.environ.get("OUT_PREFIX", "batches/public-merged")
API = HfApi(token=HF_TOKEN)

# ── helpers --
def sha256_hex(s: str) -> str:
    return hashlib.sha256(s.encode()).hexdigest()

def belongs_to_shard(slug: str) -> bool:
    return int(sha256_hex(slug), 16) % SHARD_TOTAL == SHARD_ID

def list_date_files(date: str) -> List[str]:
    """Single API call: non-recursive tree for date folder."""
    folder = f"{RAW_PREFIX}/{date}"
    try:
        tree = API.list_repo_tree(repo_id=REPO_ID, path=folder, repo_type="dataset")
    except Exception as exc:
        # If folder missing -> nothing to do is OK
        print(f"No tree for {folder}: {exc}", file=sys.stderr)
        return []
    files = [item.rfilename for item in tree if item.type == "file"]
    return files

def cdn_url(path: str) -> str:
    return f"https://huggingface.co/datasets/{REPO_ID}/resolve/main/{path}"

def project_to_pair(raw_obj: Dict[str, Any]) -> Dict[str, str]:
    """
    Surrogate-1 pattern: heterogeneous schemas.
    Keep only prompt/response; everything else dropped.
    Accept common field variants.
    """
    prompt = (
        raw_obj.get("prompt")
        or raw_obj.get("instruction")
        or raw_obj.get("input")
        or raw_obj.get("question")
        or ""
    )
    response = (
        raw_obj.get("response")
        or raw_obj.get("output")
        or raw_obj.get("answer")
        or raw_obj.get("completion")
        or ""
    )
    return {"prompt": str(prompt).strip(), "response": str(response).strip()}

def run_cmd(cmd: List[str]) -> None:
    print(f"+ {' '.join(cmd)}")
    subprocess.run(cmd, check=True)

# ── dedup wrapper --
def is_duplicate(md5: str) -> bool:
    # lib/dedup.py exposes CLI: python lib/dedup.py check <md5>
    result = subprocess.run(
        [sys.executable, "lib/dedup.py", "check", md5],
        capture_output=True,
        text=True,
    )
    return result.returncode == 0

def mark_seen(md5: str) -> None:
    subprocess.run([sys.executable, "lib/dedup.py", "add", md5], check=False)

# ── main --
def main() -> None:
    if not all([HF_TOKEN, DATE, SHARD_ID is not None]):
        print("Missing required env", file=sys.stderr)
        sys.exit(1)

    ts = datetime.datetime.utcnow().strftime("%H%M%S")
    out_dir = Path(OUT_PREFIX) / DATE
    out_dir.mkdir(parents=True, exist_ok=True)
    out_path = out_dir / f"shard{SHARD_ID}-{ts}.jsonl"

    files = list_date_files(DATE)
    if not files:
        print(f"No files for {DATE}. Nothing to do.")
        sys.exit(0)

    assigned = [f for f in files if belongs_to_shard(f)]
    print(f"Total files: {len(files)} | Assigned to shard {SHARD_ID}: {len(assigned)}")

    written = 0
    skipped_dup = 0
    schema_warned = False

    with out_path.open("w", encoding="utf-8") as fout:
        for rpath in assigned:
            url = cdn_url(rpath)
            resp = requests.get(url, timeout=60)
            if resp.status_code != 200:
                print(f"CDN fetch failed {rpath}: {resp.status_code}", file=sys.stderr)
                continue

            # Try parquet first (common), fallback to jsonl
            import io
            try:
                import pyarrow.parquet as pq
                table = pq.read_table(io.BytesIO(resp.content))
                rows = table.to_pylist()
            except Exception:
                # assume jsonl
                rows = [json.loads(ln) for ln in resp.text.strip().splitlines() if ln.strip()]

            for row in rows:
                pair = project_to_pair(row)
                if not pair["prompt"] or not pair["response"]:
                    if not schema_warned:
                        print(f"Missing prompt/response in row from {rpath}", file=sys.stderr)
                        schema_warned = True
                    continue

                payload = json.dumps(pair, ensure_ascii=False, sort_keys=True)
                md5 = hashlib.md5(payload.encode()).hexdigest()
                if is_duplicate(md5):
                    skipped_dup += 1
                    continue

                fout.write(payload + "\n")
                mark_seen(md5)
                written += 1
