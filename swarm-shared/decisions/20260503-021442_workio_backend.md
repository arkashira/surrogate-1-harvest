# workio / backend

## 1. Diagnosis

- LINE webhook retries create duplicate punches because ingestion is not idempotent (no dedupe key or idempotency token check).
- Clock-in/out upsert is non-atomic (`findOne` → conditional `insert`/`update`) allowing races under concurrency.
- Tenant isolation is enforced at app layer but not at DB constraint level, risking cross-tenant data bleed on bugs.
- No transactional boundary for multi-step attendance flows (punch + daily-summary update) — partial updates possible on failure.
- Missing unique constraint prevents DB-level de-duplication for same tenant+employee+date+eventType within tolerance window.

## 2. Proposed change

File: `/opt/axentx/workio/server/src/services/attendanceService.ts`  
Scope: add idempotent, atomic punch ingestion with tenant-scoped uniqueness and single upsert path.

Also: `/opt/axentx/workio/server/src/db/schema.sql`  
Scope: add unique constraint and idempotency table/index to enforce exactly-once semantics at DB level.

## 3. Implementation

### 3.1 DB schema changes

```sql
-- /opt/axentx/workio/server/src/db/schema.sql
-- Idempotency keys for LINE webhook events (exact-once)
CREATE TABLE IF NOT EXISTS idempotency_keys (
  id          SERIAL PRIMARY KEY,
  tenant_id   INTEGER NOT NULL,
  key         TEXT    NOT NULL,
  payload_hash TEXT,
  created_at  TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE (tenant_id, key)
);
CREATE INDEX IF NOT EXISTS idx_idempotency_tenant_key ON idempotency_keys (tenant_id, key);

-- Attendance punches with tenant-scoped uniqueness per date+eventType
-- (allows at most one punch per type per employee per date;
-- use created_at for multiple punches if business requires)
ALTER TABLE attendance_punches
  ADD CONSTRAINT uq_tenant_employee_date_event
  UNIQUE (tenant_id, employee_id, punch_date, event_type);
```

### 3.2 Service: atomic upsert + idempotency

```ts
// /opt/axentx/workio/server/src/services/attendanceService.ts
import { pool } from '../db';

export interface PunchInput {
  tenantId: number;
  employeeId: number;
  eventType: 'clock_in' | 'clock_out';
  punchDate: string; // YYYY-MM-DD
  occurredAt: string; // ISO timestamp
  latitude?: number;
  longitude?: number;
  lineEventId: string; // used as idempotency key
}

/**
 * Idempotent, atomic punch upsert.
 * Uses a single SQL upsert (ON CONFLICT) and idempotency key to guarantee
 * exactly-once processing per LINE event per tenant.
 */
export async function upsertPunch(input: PunchInput) {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // 1) Idempotency check/insert (unique per tenant+lineEventId)
    const idemResult = await client.query(
      `INSERT INTO idempotency_keys (tenant_id, key, payload_hash)
       VALUES ($1, $2, $3)
       ON CONFLICT (tenant_id, key) DO UPDATE
       SET payload_hash = EXCLUDED.payload_hash
       RETURNING id, created_at`,
      [input.tenantId, input.lineEventId, 'sha256-placeholder']
    );

    const isNew = idemResult.rowCount === 1;
    if (!isNew) {
      // Already processed — return existing punch state
      const existing = await client.query(
        `SELECT id, event_type, occurred_at
         FROM attendance_punches
         WHERE tenant_id = $1 AND employee_id = $2 AND punch_date = $3`,
        [input.tenantId, input.employeeId, input.punchDate]
      );
      await client.query('COMMIT');
      return { created: false, punch: existing.rows[0] || null };
    }

    // 2) Atomic upsert for the punch (tenant-scoped uniqueness)
    const upsertResult = await client.query(
      `INSERT INTO attendance_punches
         (tenant_id, employee_id, punch_date, event_type, occurred_at, latitude, longitude)
       VALUES ($1, $2, $3, $4, $5, $6, $7)
       ON CONFLICT (tenant_id, employee_id, punch_date, event_type)
       DO UPDATE SET
         occurred_at = EXCLUDED.occurred_at,
         latitude    = EXCLUDED.latitude,
         longitude   = EXCLUDED.longitude,
         updated_at  = NOW()
       RETURNING id, event_type, occurred_at`,
      [
        input.tenantId,
        input.employeeId,
        input.punchDate,
        input.eventType,
        input.occurredAt,
        input.latitude ?? null,
        input.longitude ?? null,
      ]
    );

    await client.query('COMMIT');
    return { created: true, punch: upsertResult.rows[0] };
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  } finally {
    client.release();
  }
}
```

### 3.3 Webhook handler usage

```ts
// /opt/axentx/workio/server/src/routes/lineWebhook.ts
import { Router } from 'express';
import { upsertPunch } from '../services/attendanceService';

const router = Router();

router.post('/line', async (req, res) => {
  try {
    const { tenantId, employeeId, events } = req.body; // adapt to your LINE payload shape
    for (const ev of events) {
      // derive punchDate/occurredAt from ev.timestamp and user tz if needed
      await upsertPunch({
        tenantId,
        employeeId,
        eventType: ev.type === 'clock_in' ? 'clock_in' : 'clock_out',
        punchDate: new Date(ev.timestamp).toISOString().slice(0, 10),
        occurredAt: new Date(ev.timestamp).toISOString(),
        latitude: ev.location?.latitude,
        longitude: ev.location?.longitude,
        lineEventId: ev.id, // LINE event id for idempotency
      });
    }
    res.json({ ok: true });
  } catch (err) {
    console.error(err);
    res.status(500).json({ error: 'processing_failed' });
  }
});

export default router;
```
