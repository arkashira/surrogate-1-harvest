# workio / design

## 1. Diagnosis

- **No tenant cache on LINE webhook** — every webhook hits PostgreSQL to resolve `channelId → tenant`; slow under burst traffic, noisy under load, and fails fast if DB is cold.
- **Missing health/readiness endpoints** — no `/health` or `/ready` for k8s/PM2/liveness checks; deploy/rollback is blind.
- **No graceful shutdown** — in-flight LINE webhook requests can be dropped during restart/deploy, causing missed clock events.
- **Hardcoded config fallback risk** — `.env` missing keys can crash the process on boot instead of failing fast with clear diagnostics.
- **No request ID / correlation** — logs from LINE webhook → tenant lookup → clock event are hard to trace under load.

## 2. Proposed change

Add a **tenant service with TTL-backed LRU cache** (keyed by `channelId`) and wire it into the LINE webhook handler; add lightweight `/health` and `/ready` endpoints; add graceful shutdown and request-id middleware.  
Scope:
- `server/src/services/tenantService.ts` (new)
- `server/src/middleware/` (new: `requestId.ts`, `gracefulShutdown.ts`)
- `server/src/routes/health.ts` (new)
- `server/src/index.ts` (modify)
- `server/src/routes/lineWebhook.ts` (modify)

## 3. Implementation

### 3.1 tenantService.ts (TTL + LRU)

```ts
// server/src/services/tenantService.ts
import { LRUCache } from 'lru-cache';
import { Pool } from 'pg';

const pool = new Pool({
  connectionString: process.env.DATABASE_URL,
});

type Tenant = {
  id: string;
  name: string;
  channelId: string;
  config: Record<string, any>;
};

const tenantCache = new LRUCache<string, Tenant>({
  max: 500,
  ttl: 1000 * 60 * 5, // 5 minutes
  allowStale: false,
  updateAgeOnGet: true,
});

export async function getTenantByChannelId(channelId: string): Promise<Tenant | null> {
  const cached = tenantCache.get(channelId);
  if (cached) return cached;

  const client = await pool.connect();
  try {
    const res = await client.query<Tenant>(
      'SELECT id, name, channel_id AS "channelId", config FROM tenants WHERE channel_id = $1 AND deleted_at IS NULL LIMIT 1',
      [channelId]
    );
    const tenant = res.rows[0] || null;
    if (tenant) tenantCache.set(channelId, tenant);
    return tenant;
  } finally {
    client.release();
  }
}

export async function invalidateTenantCache(channelId: string): Promise<void> {
  tenantCache.delete(channelId);
}
```

### 3.2 requestId middleware

```ts
// server/src/middleware/requestId.ts
import { Request, Response, NextFunction } from 'express';
import crypto from 'crypto';

export function requestId(req: Request, _res: Response, next: NextFunction) {
  (req as any).requestId = req.headers['x-request-id'] || crypto.randomUUID();
  next();
}
```

### 3.3 gracefulShutdown utility

```ts
// server/src/middleware/gracefulShutdown.ts
let shuttingDown = false;
const connections = new Set<{ end: () => void }>();

export function trackRequest(socket: { on: (e: string, cb: () => void) => void }) {
  if (shuttingDown) return;
  const conn = socket as any;
  connections.add(conn);
  conn.on('close', () => connections.delete(conn));
}

export async function gracefulStop(signal: string, server: { close: (cb?: () => void) => void }, timeoutMs = 30_000) {
  if (shuttingDown) return;
  shuttingDown = true;
  console.log(`[${new Date().toISOString()}] ${signal} received — stopping accepting new requests`);

  // Stop accepting new connections
  server.close(() => {
    console.log(`[${new Date().toISOString()}] HTTP server closed`);
  });

  // Force close lingering connections after timeout
  const forceClose = setTimeout(() => {
    console.log(`[${new Date().toISOString()}] Force-closing ${connections.size} connections`);
    connections.forEach((conn) => conn.end?.());
    process.exit(0);
  }, timeoutMs);

  // Keep process alive until connections drain or timeout
  if (connections.size === 0) {
    clearTimeout(forceClose);
    process.exit(0);
  } else {
    const interval = setInterval(() => {
      if (connections.size === 0) {
        clearTimeout(forceClose);
        clearInterval(interval);
        process.exit(0);
      }
    }, 500);
  }
}
```

### 3.4 health routes

```ts
// server/src/routes/health.ts
import { Router } from 'express';
import { Pool } from 'pg';

const pool = new Pool({
  connectionString: process.env.DATABASE_URL,
});

const router = Router();

router.get('/health', (_req, res) => {
  res.json({ status: 'ok', uptime: process.uptime(), timestamp: new Date().toISOString() });
});

router.get('/ready', async (_req, res) => {
  try {
    await pool.query('SELECT 1');
    res.json({ status: 'ready', db: 'ok', timestamp: new Date().toISOString() });
  } catch (err) {
    res.status(503).json({ status: 'not_ready', db: 'error', error: String(err) });
  }
});

export default router;
```

### 3.5 wire into index.ts

```ts
// server/src/index.ts (excerpt — integrate into existing file)
import express from 'express';
import http from 'http';
import requestId from './middleware/requestId';
import { trackRequest, gracefulStop } from './middleware/gracefulShutdown';
import healthRoutes from './routes/health';
import lineWebhookRoutes from './routes/lineWebhook';

const app = express();
app.use(express.json());
app.use(requestId);

// Logging with requestId
app.use((req, res, next) => {
  const start = Date.now();
  res.on('finish', () => {
    console.log(
      `[${new Date().toISOString()}] ${(req as any).requestId} ${req.method} ${req.originalUrl} ${res.statusCode} ${Date.now() - start}ms`
    );
  });
  next();
});

app.use('/webhook/line', lineWebhookRoutes);
app.use('/health', healthRoutes);

const server = http.createServer(app);

// Graceful shutdown
process.on('SIGTERM', () => gracefulStop('SIGTERM', server));
process.on('SIGINT', () => gracefulStop('SIGINT', server));

// Track connections for graceful shutdown
server.on('connection', trackRequest);

const PORT = process.env.PORT || 3000;
server.listen(PORT, () => {
  console.log(`Server listening on port ${PORT}`);
});
```
