# Costinel / discovery

## Final Implementation Plan — CDN-First Top-Hub Signal Panel (<2h)

**Goal**: Add a resilient “Top Hub” signal panel to Costinel that shows the most-connected hub (e.g., “MOC”) and related docs using CDN-fetched artifacts (zero runtime HF API calls).  
**Principle**: Strict **Sense + Signal — no Execute**; CDN-only fetches; no runtime HF API; pre-listed file paths embedded at build/deploy time.

### Scope (incremental, <2h)
- Add `TopHubSignalPanel` component on Dashboard.
- Add `useTopHubSignal` hook + `lib/cdn.ts` to fetch pre-listed artifacts from CDN (no Authorization, bypass HF API rate limits).
- Single pre-list JSON file stored in repo (or generated by ops) containing hub metadata + related doc paths.
- Panel renders top hub name, short insight, and related doc links (opens in new tab).
- Graceful fallback UI when CDN fetch fails or data is stale.
- No mutations, no proposals, no execution — pure signal.

```ts
// src/lib/cdn.ts
const CDN_ROOT = 'https://huggingface.co/datasets/axentx/costinel-knowledge/resolve/main';

export async function fetchCdnJson<T>(path: string, init?: RequestInit): Promise<T> {
  const url = `${CDN_ROOT}/${path}`;
  const controller = new AbortController();
  const timeout = setTimeout(() => controller.abort(), 12_000);

  try {
    const res = await fetch(url, {
      ...init,
      signal: controller.signal,
      headers: {
        // No Authorization — public CDN bypass
        ...(init?.headers || {}),
      },
    });

    if (!res.ok) {
      throw new Error(`CDN fetch failed: ${res.status} ${res.statusText} (${url})`);
    }

    return (await res.json()) as T;
  } finally {
    clearTimeout(timeout);
  }
}
```

### 2) Hook (`src/hooks/useTopHubSignal.ts`)

```ts
// src/hooks/useTopHubSignal.ts
import { useEffect, useState, useCallback } from 'react';
import { fetchCdnJson } from '../lib/cdn';

type TopHub = {
  hub: string;
  rationale: string;
  score: number;
  updatedAt: string;
};

type RelatedDoc = {
  id: string;
  title: string;
  snippet: string;
  url: string;
  relevance: number;
};

type TopHubSnapshotMeta = {
  snapshot: string; // e.g., "2026-04-27"
  path: string;     // relative to CDN_ROOT
};

type TopHubSignal = {
  hub: TopHub | null;
  relatedDocs: RelatedDoc[];
  loading: boolean;
  error: string | null;
};

export function useTopHubSignal(): TopHubSignal {
  const [state, setState] = useState<TopHubSignal>({
    hub: null,
    relatedDocs: [],
    loading: true,
    error: null,
  });

  const load = useCallback(async () => {
    setState((s) => ({ ...s, loading: true, error: null }));
    try {
      // This index is embedded at build time (zero runtime HF API).
      const index = (await import('../static/top-hub-index.json')) as TopHubSnapshotMeta[];
      if (!index.length) {
        setState({ hub: null, relatedDocs: [], loading: false, error: null });
        return;
      }

      // Pick latest snapshot (lexicographic works for ISO dates)
      const latest = index.sort((a, b) => b.snapshot.localeCompare(a.snapshot))[0];

      const [topHub, relatedDocs] = await Promise.all([
        fetchCdnJson<TopHub>(`${latest.path}/top-hub.json`).catch(() => null),
        fetchCdnJson<RelatedDoc[]>(`${latest.path}/related-docs.json`).catch(() => [] as RelatedDoc[]),
      ]);

      setState({
        hub: topHub || null,
        relatedDocs: relatedDocs || [],
        loading: false,
        error: null,
      });
    } catch (err: any) {
      setState({ hub: null, relatedDocs: [], loading: false, error: err.message || 'Unknown error' });
    }
  }, []);

  useEffect(() => {
    load();
  }, [load]);

  return state;
}
```

### 3) Component (`src/components/TopHubSignalPanel.tsx`)

```tsx
// src/components/TopHubSignalPanel.tsx
import React from 'react';
import { useTopHubSignal } from '../hooks/useTopHubSignal';

export function TopHubSignalPanel() {
  const { hub, relatedDocs, loading, error } = useTopHubSignal();

  if (loading) {
    return (
      <section className="rounded-lg border bg-white p-4 shadow-sm">
        <p className="text-sm text-gray-500">Loading top hub signal…</p>
      </section>
    );
  }

  if (error) {
    return (
      <section className="rounded-lg border border-yellow-100 bg-yellow-50 p-4 shadow-sm">
        <p className="text-sm text-yellow-800">Unable to load top hub signal: {error}</p>
      </section>
    );
  }

  if (!hub) {
    return (
      <section className="rounded-lg border border-gray-100 bg-white p-4 shadow-sm">
        <p className="text-sm text-gray-400">No top hub signal available.</p>
      </section>
    );
  }

  return (
    <section className="rounded-lg border bg-white p-4 shadow-sm">
      <div className="mb-3 flex items-start justify-between">
        <div>
          <h3 className="text-base font-semibold text-gray-900">Top Hub: {hub.hub}</h3>
          <p className="text-sm text-gray-600">{hub.rationale}</p>
        </div>
        <span className="text-xs text-gray-400">
          Updated {new Date(hub.updatedAt).toLocaleDateString()}
        </span>
      </div>

      {relatedDocs.length > 0 && (
        <div>
          <h4 className="mb-2 text-sm font-medium text-gray-700">Related Docs</h4>
          <ul className="space-y-2" role="list">
            {relatedDocs.map((doc) => (
              <li key={doc.id} className="flex gap-3 text-sm">
                <span className="mt-0.5 flex h-2 w-2 shrink-0 rounded-full bg-blue-400" aria-hidden="true" />
                <div>
                  <a
                    href={doc.url}
                    target="_blank"
                    rel="noopener noreferrer"
                    className="font-medium text-blue-600 hover:underline"
                  >
                    {doc.title}
                  </a>
                  <p className="text-gray-600">{doc.snippet}</p>
                </div>
              </li>
            ))}
          </ul>
        </div>
      )}
    </section>
  );
}
```
