## Final Synthesis (Best Parts + Correctness + Actionability)

**Chosen approach:**  
CDN-only ingestion with deterministic sibling-repo sharding — eliminates HF API 429s by pre-listing once, downloading via public CDN URLs (no auth), projecting `{prompt,response}` at parse, and spreading commits across 5 sibling repos to bypass the 128/hr commit cap.

**Why this ships <2h and is correct:**  
- No schema migrations or infra changes.  
- Single ingestion script + tiny config.  
- Reuses existing `surrogate/` training pipeline (Lightning + CDN).  
- Deterministic sharding prevents collisions and hot spots.  
- CDN-only training fetches remove API calls during training (zero 429 risk after ingestion).

### 2. New ingestion script (45–60 min)
`surrogate/scripts/ingest_cdn_only.py`
```python
#!/usr/bin/env python3
"""
CDN-only HF ingestion with deterministic sibling sharding.
Pre-list once -> JSON -> Lightning training uses CDN-only fetches (zero API calls).
"""
import hashlib
import json
import os
import sys
from pathlib import Path

import pyarrow as pa
import pyarrow.parquet as pq
import requests
import yaml
from tqdm import tqdm

HF_API_BASE = "https://huggingface.co"
CDN_BASE = "https://huggingface.co/datasets"

def load_config() -> dict:
    cfg_path = Path(__file__).parent.parent / "config" / "ingest.yaml"
    with open(cfg_path) as f:
        return yaml.safe_load(f)

def list_date_folder(cfg: dict) -> list[str]:
    """
    Single API call to list one date folder (non-recursive).
    Returns relative paths like '2026-05-03/file1.jsonl'.
    """
    repo = cfg["hf_repo"]
    folder = cfg["date_folder"]
    url = f"{HF_API_BASE}/api/datasets/{repo}/tree/{folder}"
    resp = requests.get(url, timeout=30)
    if resp.status_code == 429:
        print("429 on list — wait 360s or run outside window", file=sys.stderr)
        raise RuntimeError("HF API rate-limited on list")
    resp.raise_for_status()
    entries = resp.json()
    # Keep only files (skip subfolders)
    files = [e["path"] for e in entries if e.get("type") == "file"]
    return files

def pick_sibling(slug: str, siblings: list[str]) -> str:
    """Deterministic sibling selection by hash(slug) % N."""
    digest = int(hashlib.sha256(slug.encode()).hexdigest(), 16)
    idx = digest % len(siblings)
    return siblings[idx]

def cdn_download(repo: str, file_path: str, out_path: Path) -> None:
    """Download via public CDN URL (no Authorization header)."""
    url = f"{CDN_BASE}/{repo}/resolve/main/{file_path}"
    out_path.parent.mkdir(parents=True, exist_ok=True)
    with requests.get(url, stream=True, timeout=60) as r:
        r.raise_for_status()
        with open(out_path, "wb") as f:
            for chunk in r.iter_content(chunk_size=8192):
                f.write(chunk)

def project_to_prompt_response(raw_path: Path) -> list[dict]:
    """
    Parse heterogeneous files and project to {prompt, response}.
    Supports JSONL with common field names.
    """
    rows = []
    text = raw_path.read_text(encoding="utf-8")
    for line in text.strip().splitlines():
        if not line.strip():
            continue
        try:
            obj = json.loads(line)
        except json.JSONDecodeError:
            continue

        # Flexible field mapping
        prompt = obj.get("prompt") or obj.get("input") or obj.get("question") or ""
        response = obj.get("response") or obj.get("output") or obj.get("answer") or ""
        if prompt and response:
            rows.append({"prompt": str(prompt), "response": str(response)})
    return rows

def upload_parquet(sibling_repo: str, date_folder: str, slug: str, rows: list[dict], cfg: dict) -> None:
    """
    Create parquet and upload via HF API (commits count toward 128/hr).
    Filename pattern: batches/mirror-merged/{date}/{slug}.parquet
    No extra metadata columns.
    """
    import io

    table = pa.Table.from_pylist(rows, schema=pa.schema([
        pa.field("prompt", pa.string()),
        pa.field("response", pa.string()),
    ]))
    buf = io.BytesIO()
    pq.write_table(table, buf)
    buf.seek(0)

    batch_dir = cfg["batch_dir"]
    parquet_path = f"{batch_dir}/{date_folder}/{slug}.parquet"

    # HF upload via requests (files/ API)
    url = f"{HF_API_BASE}/api/datasets/{sibling_repo}/upload"
    files = {"files": (parquet_path, buf, "application/octet-stream")}
    # Uses HF token from env: HF_TOKEN
    headers = {"Authorization": f"Bearer {os.getenv('HF_TOKEN')}"}
    resp = requests.post(url, headers=headers, files=files, timeout=120)
    if resp.status_code == 429:
        print("429 on upload — wait 360s", file=sys.stderr)
        raise RuntimeError("HF API rate-limited on upload")
    resp.raise_for_status()
    print(f"Uploaded {parquet_path} to {sibling_repo}")

def build_file_list_json(file_paths: list[str], out_path: Path) -> None:
    """Save file list for Lightning training (CDN-only)."""
    out_path.parent.mkdir(parents=True, exist_ok=True)
    with open(out_path, "w") as f:
        json.dump(file_paths, f, indent=2)
    print(f"File list saved to {out_path}")

def main() -> None:
    cfg = load_config()
    repo = cfg["hf_repo"]
    date_folder = cfg["date_folder"]
    siblings = cfg["sibling_repos"]

    # Step 1: pre-list once (single API call)
    print("Listing date folder (single API call)...")
    files = list_date_folder(cfg)
    file_list_path = Path("data/file_lists") / f"{date_folder}.json"
    build_file_list_json(files, file_list_path)

    # Step 2: process each file via CDN, project, shard
    print(f"Processing {len(files)} files via CDN...")
    for rel_path in tqdm(files, desc="Ingest"):
        try:
            raw_local = Path("data/raw") / rel_path
            cdn_download(repo, rel_path, raw_local)
            rows = project_to_prompt_response(raw_local)
            if not rows:
                continue

            slug = raw_local.stem  # use filename as slug
            sibling = pick_sibling(slug, siblings)
            upload_parquet(sibling, date_folder, slug, rows, cfg)

            # Clean raw to save space
            raw_local.unlink(missing_ok=True)
        except Exception as e:
            print(f"Failed {rel_path}: {e}", file=sys.stderr)
            continue

    print("CDN-only ingestion complete.")
    print("Next: Lightning training uses file_list JSON for CDN-only fetches (zero API calls).")

if __name__ == "__main__":
    main()
```
