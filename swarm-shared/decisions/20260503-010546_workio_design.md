# workio / design

## 1. Diagnosis

- Missing **database-level guard** for open punches: no partial unique index on `(user_id, date(clock_in_at))` where `clock_out_at IS NULL`, allowing duplicates under LINE webhook retries or client double-taps.
- Race-prone **find-then-insert** logic in clock-in/clock-out handlers (app-layer check + insert) instead of atomic upsert/locking.
- No **idempotency key** on webhook handler to deduplicate LINE event deliveries with same `webhookEventId`.
- Punch records can be created with **no tenant context** or weak tenant scoping, risking cross-tenant data bleed in multi-tenant queries.
- **Clock-in without location/GPS validation** can be bypassed; validation is likely best-effort in app layer rather than enforced at transaction boundary.

## 2. Proposed change

File scope:
- `server/src/db/schema.sql` — add partial unique index and tighten constraints.
- `server/src/db/migrations/` (or inline schema) — add idempotency table for LINE webhooks.
- `server/src/controllers/punchController.ts` — replace find-then-insert with atomic `INSERT ... ON CONFLICT DO NOTHING` + row-level locking for updates.
- `server/src/middleware/tenantMiddleware.ts` — enforce tenant_id presence and scoping on punch mutations.

Highest-value single deliverable (<2h): **atomic punch mutation with DB-level uniqueness + idempotency** (schema + controller).

## 3. Implementation

### 3.1 Schema changes (`server/src/db/schema.sql`)

```sql
-- Ensure tenant_id is present and not null on punches if multi-tenant
ALTER TABLE punches ALTER COLUMN tenant_id SET NOT NULL;

-- Partial unique index: at most one open punch per user per calendar day
CREATE UNIQUE INDEX IF NOT EXISTS idx_one_open_punch_per_user_per_day
  ON punches (user_id, date(clock_in_at))
  WHERE clock_out_at IS NULL;

-- Idempotency table for LINE webhook events
CREATE TABLE IF NOT EXISTS line_webhook_idempotency (
  webhook_event_id VARCHAR(255) PRIMARY KEY,
  tenant_id        UUID      NOT NULL,
  user_id          UUID      NOT NULL,
  punch_id         UUID      NOT NULL,
  created_at       TIMESTAMPTZ DEFAULT NOW()
);

-- Index for cleanup
CREATE INDEX IF NOT EXISTS idx_line_idemp_tenant_created
  ON line_webhook_idempotency (tenant_id, created_at);
```

### 3.2 Controller changes (`server/src/controllers/punchController.ts`)

Replace find-then-insert with atomic operations and idempotency check.

```ts
import { pool } from '../db';
import { Request, Response } from 'express';

// Helper to run in transaction
async function withTx<T>(fn: (client: any) => Promise<T>): Promise<T> {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');
    const result = await fn(client);
    await client.query('COMMIT');
    return result;
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  } finally {
    client.release();
  }
}

// Clock in (idempotent)
export async function clockIn(req: Request, res: Response) {
  const { userId, tenantId, lineEventId, latitude, longitude } = req.body;

  if (!tenantId || !userId) {
    return res.status(400).json({ error: 'tenantId and userId required' });
  }

  try {
    const result = await withTx(async (client) => {
      // Idempotency check
      if (lineEventId) {
        const idem = await client.query(
          'SELECT punch_id FROM line_webhook_idempotency WHERE webhook_event_id = $1',
          [lineEventId]
        );
        if (idem.rows.length > 0) {
          // Already processed — return existing punch
          const punch = await client.query(
            'SELECT * FROM punches WHERE id = $1',
            [idem.rows[0].punch_id]
          );
          return { existing: punch.rows[0] };
        }
      }

      // Try insert new open punch (conflict -> none inserted)
      const insertRes = await client.query(
        `INSERT INTO punches (user_id, tenant_id, clock_in_at, latitude, longitude)
         VALUES ($1, $2, NOW(), $3, $4)
         ON CONFLICT DO NOTHING
         RETURNING *`,
        [userId, tenantId, latitude, longitude]
      );

      let punch = insertRes.rows[0];

      // If conflict (race) or partial index violation avoided insert, fetch current open punch
      if (!punch) {
        const openRes = await client.query(
          `SELECT * FROM punches
           WHERE user_id = $1 AND tenant_id = $2 AND clock_out_at IS NULL
           ORDER BY clock_in_at DESC LIMIT 1 FOR UPDATE`,
          [userId, tenantId]
        );
        punch = openRes.rows[0];
      }

      // Record idempotency
      if (lineEventId && punch) {
        await client.query(
          `INSERT INTO line_webhook_idempotency (webhook_event_id, tenant_id, user_id, punch_id)
           VALUES ($1, $2, $3, $4)`,
          [lineEventId, tenantId, userId, punch.id]
        );
      }

      return { punch };
    });

    return res.json(result);
  } catch (err: any) {
    // Unique violation from partial index or other constraint
    if (err.code === '23505') {
      return res.status(409).json({ error: 'Already clocked in' });
    }
    console.error(err);
    return res.status(500).json({ error: 'Server error' });
  }
}

// Clock out (atomic update)
export async function clockOut(req: Request, res: Response) {
  const { userId, tenantId, lineEventId, latitude, longitude } = req.body;

  if (!tenantId || !userId) {
    return res.status(400).json({ error: 'tenantId and userId required' });
  }

  try {
    const result = await withTx(async (client) => {
      // Idempotency
      if (lineEventId) {
        const idem = await client.query(
          'SELECT punch_id FROM line_webhook_idempotency WHERE webhook_event_id = $1',
          [lineEventId]
        );
        if (idem.rows.length > 0) {
          const punch = await client.query('SELECT * FROM punches WHERE id = $1', [idem.rows[0].punch_id]);
          return { existing: punch.rows[0] };
        }
      }

      // Lock the open punch row
      const openRes = await client.query(
        `SELECT * FROM punches
         WHERE user_id = $1 AND tenant_id = $2 AND clock_out_at IS NULL
         ORDER BY clock_in_at DESC LIMIT 1 FOR UPDATE`,
        [userId, tenantId]
      );

      if (openRes.rows.length === 0) {
        throw new Error('No open punch to clock out');
      }

      const punch = openRes.rows[0];

      const updateRes = await client.query(
        `UPDATE punches
         SET clock_out_at = NOW(), latitude_out = $1, longitude_out = $2
         WHERE id = $3
         RETURNING *`,
        [latitude, longitude, punch.id]
      );

      if (lineEventId) {
        await client.query(
          `INSERT INTO line_webhook_idempotency (webhook_event_id, tenant_id, user_id, punch_id)
           VALUES ($1, $2, $3, $4)`,
          [lineEventId, tenantId, userId, updateRes.rows[0].id]
        );
      }

      return { punch: updateRes.rows[0] };
    });

    return res.json(result);
  } catch (err: any) {
    if (err.message === 'No open punch to clock out') {
      return res.status(404).json({ error: err.message });
    }
    console.error(err);
    return res.status(500).json({ error: 'Server error' });
  }
}
```
