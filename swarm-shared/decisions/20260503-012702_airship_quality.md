- **Embedding file list**: UI generates `file_list.json` next to training script; training script never calls HF API during epochs.

```python
# surrogate/training/hf_cdn_loader.py
# -*- coding: utf-8 -*-
"""
HF CDN-bypass loader.
- Single non-recursive API call to list one date folder.
- Persist file_list.json.
- Training uses CDN-only fetches (no API/auth/429 during training).
"""
import json
import time
from pathlib import Path
from typing import Dict, List

import requests
from huggingface_hub import HfApi

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"
api = HfApi()

def list_date_folder(repo: str, date_folder: str, out_dir: Path) -> Path:
    """
    List one date folder (non-recursive) and save file_list.json.
    Returns path to file_list.json.
    """
    out_dir.mkdir(parents=True, exist_ok=True)
    items = api.list_repo_tree(repo=repo, path=date_folder, recursive=False)
    files: List[Dict[str, object]] = []
    for it in items:
        if it.get("type") == "file":
            rel = it["path"]
            files.append({
                "repo": repo,
                "path": rel,
                "cdn_url": CDN_TEMPLATE.format(repo=repo, path=rel),
                "size": it.get("size"),
            })

    out_path = out_dir / "file_list.json"
    out_path.write_text(json.dumps(files, indent=2), encoding="utf-8")
    return out_path

def stream_cdn_files(file_list_path: Path):
    """
    Yield (path, content_bytes) for each file via CDN.
    Retries with backoff on 429/5xx.
    """
    files = json.loads(file_list_path.read_text(encoding="utf-8"))
    for f in files:
        url = f["cdn_url"]
        for attempt in range(5):
            try:
                resp = requests.get(url, timeout=30)
                if resp.status_code == 429:
                    wait = 360 if attempt == 0 else (2 ** attempt) * 60
                    time.sleep(wait)
                    continue
                resp.raise_for_status()
                yield f["path"], resp.content
                break
            except requests.RequestException:
                if attempt == 4:
                    raise
                time.sleep((2 ** attempt) * 5)

def project_prompt_response(content_bytes: bytes, filename: str) -> Dict[str, str]:
    """
    Project heterogeneous file content to {prompt, response}.
    Implement per-extension logic here; default is conservative.
    """
    text = content_bytes.decode("utf-8", errors="replace").strip()
    blocks = [b.strip() for b in text.split("\n\n") if b.strip()]
    if len(blocks) >= 2:
        return {"prompt": "\n\n".join(blocks[:-1]), "response": blocks[-1]}
    return {"prompt": text, "response": ""}
```

```python
# surrogate/training/studio_reuse.py
# -*- coding: utf-8 -*-
"""
Lightning Studio reuse utility.
"""
from lightning_sdk import Machine, Studio, Teamspace
from typing import Optional

def get_or_create(name: str, machine: str = "L40S") -> Studio:
    """
    Reuse a running studio if exists; otherwise create with machine fallback.
    If studio is stopped, restart it with requested machine.
    """
    studios = Teamspace.studios()
    for s in studios:
        if s.name == name:
            if s.status == "running":
                return s
            if s.status == "stopped":
                s.start(machine=Machine(machine))
                return s
            # For other states, return existing and let caller handle
            return s

    machines = [Machine.L40S, Machine.A10G, Machine.A10]
    for m in machines:
        try:
            studio = Studio(name=name, machine=m, create_ok=True)
            return studio
        except Exception:
            continue
    raise RuntimeError("Could not create studio on any available machine")
```
