### 1. Pre-list file paths (run once from Mac)
- Use `list_repo_tree(path, recursive=False)` per date folder in `datasets/{repo}`.
- Save to `filelist-{date}.json` committed to repo or passed to Lightning job.
- Do this after rate-limit window clears; only one API hit per folder.

```bash
# example run (mac orchestration only)
python scripts/list_hf_date_folder.py \
  --repo surrogate-dataset-mirror \
  --date 2026-05-03 \
  --out filelist-2026-05-03.json
```

```python
# scripts/list_hf_date_folder.py
import json
import argparse
from huggingface_hub import HfApi

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True)
    parser.add_argument("--date", required=True)  # corresponds to folder name
    parser.add_argument("--out", required=True)
    args = parser.parse_args()

    api = HfApi()
    # non-recursive per folder to avoid 100x pagination
    entries = api.list_repo_tree(repo_id=args.repo, path=args.date, recursive=False)
    files = [e.path for e in entries if e.type == "file" and e.path.endswith(".parquet")]
    with open(args.out, "w") as f:
        json.dump({"date": args.date, "files": files}, f, indent=2)
    print(f"Saved {len(files)} files to {args.out}")

if __name__ == "__main__":
    main()
```

### 2. CDN-only DataLoader (Lightning training script)
- Remove `load_dataset(streaming=True)`.
- Load file list from embedded JSON.
- Fetch via CDN URLs (`resolve/main/...`) with `datasets` or raw `pyarrow`/`pandas`.
- Project to `{prompt, response}` at parse time (ignore mixed schema cols).

```python
# surrogate/data.py
import json
import pyarrow.parquet as pq
import requests
from pathlib import Path
from torch.utils.data import IterableDataset

HF_DATASET_REPO = "datasets/surrogate-dataset-mirror"
CDN_BASE = "https://huggingface.co"

class CDNParquetDataset(IterableDataset):
    def __init__(self, filelist_path, shard_rank=0, shard_world_size=1):
        with open(filelist_path) as f:
            manifest = json.load(f)
        self.files = manifest["files"]
        # deterministic sharding across siblings or workers
        self.files = self.files[shard_rank::shard_world_size]

    def __iter__(self):
        for rel_path in self.files:
            cdn_url = f"{CDN_BASE}/{HF_DATASET_REPO}/resolve/main/{rel_path}"
            # stream download to temp file or memory
            r = requests.get(cdn_url, timeout=30)
            r.raise_for_status()
            table = pq.read_table(pq.ParquetFile(pq.BufferReader(r.content)))
            # project only required schema
            for batch in table.to_batches(columns=["prompt", "response"]):
                for i in range(batch.num_rows):
                    yield {
                        "prompt": batch["prompt"][i].as_py(),
                        "response": batch["response"][i].as_py(),
                    }
```

- In Lightning `Trainer`, pass `shard_rank=trainer.global_rank`, `shard_world_size=trainer.world_size`.

### 3. Deterministic sibling-repo sharding for writes
- When dataset-mirror produces enriched outputs, hash slug → pick sibling repo to spread HF commit cap (128/hr/repo).

```python
# surrogate/ingest/upload.py
import hashlib

SIBLING_REPOS = [
    "surrogate-dataset-mirror",
    "surrogate-dataset-mirror-sib1",
    "surrogate-dataset-mirror-sib2",
    "surrogate-dataset-mirror-sib3",
    "surrogate-dataset-mirror-sib4",
]

def pick_repo(slug: str) -> str:
    h = hashlib.sha256(slug.encode()).hexdigest()
    idx = int(h, 16) % len(SIBLING_REPOS)
    return SIBLING_REPOS[idx]
```

- Upload to `batches/mirror-merged/{date}/{slug}.parquet` with **only** `{prompt,response}` columns. No `source`/`ts` cols.

### 4. Lightning Studio reuse + idle handling
- Before `Studio(create_ok=True)`, list running studios and reuse.
- Wrap `.run()` with status check; restart if idle-stopped.

```python
# surrogate/train_launcher.py
from lightning import Studio, Teamspace, Machine
import time

def get_or_create_studio(name="surrogate-train"):
    for s in Teamspace.studios:
        if s.name == name and s.status == "Running":
            print(f"Reusing running studio: {s.id}")
            return s
    print("Creating new studio...")
    return Studio(
        name=name,
        machine=Machine.L40S,
        create_ok=True,
    )

def run_training_job(script_path, args):
    studio = get_or_create_studio()
    # ensure running
    if studio.status != "Running":
        studio.start(machine=Machine.L40S)
        # wait until running
        while studio.status != "Running":
            time.sleep(10)
            studio.refresh()
    # run training (non-blocking or blocking as preferred)
    job = studio.run(
        command=f"python {script_path} {' '.join(args)}",
        environment="surrogate-train",
    )
    return job
```
