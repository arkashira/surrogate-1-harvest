# vanguard / frontend

## Final Synthesis (Best Parts + Correctness + Actionability)

### Diagnosis (resolved)
- **No frontend entrypoint or mount** → prevents fast iteration and local validation.
- **No build/dev tooling** → no HMR, no type safety, no asset pipeline; changes require manual reloads.
- **Missing static file manifest** → browser re-enumerates repos per session, causing runtime HF API calls, quota burn, and 429 risk.
- **No clear CDN-bypass strategy in frontend** → every session can hit live HF endpoints instead of CDN.
- **No caching or precomputed paths** → repeated sessions pay enumeration cost.

**Resolution priority**: correctness first (avoid browser HF API calls), then concrete actionability (minimal scaffold that works immediately and can evolve).

---

### Proposed Change (single, coherent plan)
Create a minimal, production-ready frontend scaffold at `/opt/axentx/vanguard/` that:
- Provides a single-page app mount (`index.html` + `src/main.js` + `src/styles.css`).
- Embeds or loads a static file manifest to enable CDN-only dataset fetches (no HF API calls from browser).
- Uses a bundler-free, native ES module setup for fast iteration and easy deployment.
- Exposes a small client SDK (`src/api.js`) to request precomputed file lists from the backend and fetch dataset shards directly from the HF CDN.
- Includes a sample `manifest.json` and a minimal build/generation script (`scripts/generate-manifest.js`) to produce manifests and validate CDN URLs.

Scope (concrete files to create):
- `/opt/axentx/vanguard/index.html`
- `/opt/axentx/vanguard/src/main.js`
- `/opt/axentx/vanguard/src/api.js`
- `/opt/axentx/vanguard/src/styles.css`
- `/opt/axentx/vanguard/manifest.json` (sample; to be generated by build script)
- `/opt/axentx/vanguard/scripts/generate-manifest.js` (minimal generator + validation)

#### src/api.js
```javascript
// Lightweight client SDK for Vanguard frontend.
// Strategy: avoid HF API calls from browser; use CDN URLs and backend-provided manifests.

const CDN_ROOT = 'https://huggingface.co';

export async function fetchManifestFromBackend(repo, dateFolder = null) {
  // Backend should expose /api/manifest?repo=...&date=...
  const params = new URLSearchParams({ repo });
  if (dateFolder) params.set('date', dateFolder);

  const res = await fetch(`/api/manifest?${params}`);
  if (!res.ok) throw new Error(`Backend manifest fetch failed: ${res.status}`);
  return res.json(); // expected: { repo, date, files: ["path1.parquet", ...] }
}

export function buildCdnUrl(repo, filePath) {
  // repo expected like "datasets/owner/repo"
  if (!repo || !filePath) return '';
  const parts = repo.split('/').filter(Boolean);
  // normalize: datasets/owner/repo
  const repoPath = parts.join('/');
  return `${CDN_ROOT}/${repoPath}/resolve/main/${filePath}`;
}

export async function fetchFileHead(url) {
  // Lightweight HEAD to check availability without downloading body.
  try {
    const res = await fetch(url, { method: 'HEAD', mode: 'cors' });
    return { ok: res.ok, status: res.status, size: res.headers.get('content-length') };
  } catch (err) {
    return { ok: false, error: err.message };
  }
}
```

#### src/main.js
```javascript
import { fetchManifestFromBackend, buildCdnUrl, fetchFileHead } from './api.js';

const repoInput = document.getElementById('repoInput');
const loadBtn = document.getElementById('loadBtn');
const fileList = document.getElementById('fileList');
const statusEl = document.getElementById('status');
const refreshBtn = document.getElementById('refreshManifest');
const embeddedBtn = document.getElementById('useEmbedded');

let currentManifest = null;

function setStatus(text, type = 'info') {
  statusEl.textContent = text;
  statusEl.className = `status ${type}`;
}

function renderFiles(repo, files) {
  fileList.innerHTML = '';
  if (!files?.length) {
    fileList.innerHTML = '<p class="muted">No files available.</p>';
    return;
  }

  const ul = document.createElement('ul');
  ul.className = 'file-items';
  for (const f of files) {
    const li = document.createElement('li');
    li.className = 'file-item';

    const link = document.createElement('a');
    link.href = buildCdnUrl(repo, f);
    link.target = '_blank';
    link.rel = 'noopener';
    link.textContent = f;
    link.className = 'file-link';

    const check = document.createElement('button');
    check.textContent = 'Check';
    check.className = 'check-btn';
    check.onclick = async (e) => {
      e.preventDefault();
      check.disabled = true;
      check.textContent = 'Checking...';
      const url = buildCdnUrl(repo, f);
      const info = await fetchFileHead(url);
      check.textContent = info.ok ? 'OK' : 'Unavailable';
      check.disabled = false;
    };

    li.appendChild(link);
    li.appendChild(check);
    ul.appendChild(li);
  }
  fileList.appendChild(ul);
}

async function loadManifest(useBackend = true) {
  const repo = repoInput.value.trim();
  if (!repo) {
    setStatus('Enter a repo (e.g., datasets/owner/repo)', 'error');
    return;
  }

  setStatus('Loading...', 'info');
  fileList.innerHTML = '';

  try {
    if (useBackend) {
      currentManifest = await fetchManifestFromBackend(repo);
      setStatus('Loaded from backend', 'success');
    } else {
      // Try embedded manifest (to be generated at build time into /manifest.json)
      const res = await fetch('/manifest.json');
      if (!res.ok) throw new Error('No embedded manifest');
      currentManifest = await res.json();
      setStatus('Loaded embedded manifest', 'success');
    }

    if (currentManifest?.files) {
      renderFiles(repo, currentManifest.files);
    } else {
      setStatus('Manifest missing files', 'error');
    }
  } catch (err) {
    setStatus(err.message, 'error');
    console.error(err);
  }
}

loadBtn.onclick = () => loadManifest(true);
refreshBtn.onclick = () => loadManifest(true);
embeddedBtn.onclick = () => loadManifest
