### 1) scripts/generate_file_manifest.py
```python
#!/usr/bin/env python3
"""
Mac orchestration script: generate deterministic CDN file manifest for a date folder.
Run after HF API rate-limit window clears.
"""
import json, os, sys
from datetime import datetime
from huggingface_hub import HfApi

REPO = "axentx/surrogate-datasets"
DATE_FOLDER = os.getenv("DATE_FOLDER", datetime.utcnow().strftime("%Y-%m-%d"))
OUTPUT_DIR = os.path.join(os.path.dirname(__file__), "..", "manifests", DATE_FOLDER)
OUTPUT_PATH = os.path.join(OUTPUT_DIR, "file_manifest.json")

def main():
    api = HfApi()
    # Single API call; recursive=False keeps it small and fast
    entries = api.list_repo_tree(repo_id=REPO, path=DATE_FOLDER, recursive=False)
    files = [e.rfilename for e in entries if e.type == "file"]
    manifest = {
        "repo": REPO,
        "date_folder": DATE_FOLDER,
        "generated_at": datetime.utcnow().isoformat() + "Z",
        "files": sorted(files),
        "cdn_prefix": f"https://huggingface.co/datasets/{REPO}/resolve/main/{DATE_FOLDER}"
    }
    os.makedirs(OUTPUT_DIR, exist_ok=True)
    with open(OUTPUT_PATH, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Manifest written: {OUTPUT_PATH}")
    print(f"Files: {len(files)}")

if __name__ == "__main__":
    main()
```

### 2) surrogate/training/lightning_train.py
```python
#!/usr/bin/env python3
"""
Resilient Lightning training entrypoint.
- Uses embedded manifest for CDN-only data loading (zero HF API calls during training).
- Checks Studio status and restarts if idle-stopped.
"""
import json, os, sys
from pathlib import Path

import lightning as L
from lightning.fabric.plugins import LightningPlugin

# Surrogate imports (adjust as needed)
from surrogate.data.cdn_loader import CDNParquetDataModule
from surrogate.model.surrogate import SurrogateModel

MANIFEST_PATH = Path(__file__).parent.parent / "manifests" / "latest" / "file_manifest.json"

def ensure_studio_running(studio_name: str, machine=L.Machine.L40S):
    """Idempotent Studio reuse + restart on idle-stop."""
    teamspace = L.Teamspace()
    for s in teamspace.studios:
        if s.name == studio_name:
            if s.status == "running":
                print(f"Reusing running Studio: {studio_name}")
                return s
            else:
                print(f"Studio {studio_name} stopped (status={s.status}). Restarting...")
                s.start(machine=machine)
                return s
    print(f"Creating Studio: {studio_name}")
    return L.Studio.create(name=studio_name, machine=machine, create_ok=True)

def main():
    # Studio lifecycle
    studio = ensure_studio_running("surrogate-train-studio")

    # Load manifest (embedded at build or runtime)
    if not MANIFEST_PATH.exists():
        # fallback: expect manifest at runtime mount
        MANIFEST_PATH_ALT = Path("/manifests/latest/file_manifest.json")
        if MANIFEST_PATH_ALT.exists():
            manifest_path = MANIFEST_PATH_ALT
        else:
            raise FileNotFoundError(f"Manifest not found: {MANIFEST_PATH} or {MANIFEST_PATH_ALT}")
    else:
        manifest_path = MANIFEST_PATH

    with open(manifest_path) as f:
        manifest = json.load(f)

    # CDN-only DataModule (no HF API calls)
    dm = CDNParquetDataModule(
        cdn_prefix=manifest["cdn_prefix"],
        file_list=manifest["files"],
        batch_size=int(os.getenv("BATCH_SIZE", "8")),
        num_workers=int(os.getenv("NUM_WORKERS", "4")),
    )

    model = SurrogateModel.from_pretrained(os.getenv("MODEL_NAME", "surrogate-base"))

    trainer = L.Trainer(
        max_epochs=int(os.getenv("EPOCHS", "1")),
        devices=int(os.getenv("DEVICES", "1")),
        accelerator="gpu",
        precision="bf16-mixed",
        enable_checkpointing=True,
        default_root_dir=os.getenv("LIGHTNING_LOGS_DIR", "./logs"),
    )

    trainer.fit(model, dm)

if __name__ == "__main__":
    # Ensure bash when invoked from cron
    main()
```

```bash
python3 - <<PY
import lightning as L
studio = None
for s in L.Teamspace().studios:
    if s.name == "$STUDIO_NAME":
        studio = s
        break
if studio is not None:
    if studio.status != "running":
        print(f"Restarting stopped Studio: $STUDIO_NAME")
        studio.start(machine=L.Machine.$MACHINE)
    else:
        print(f"Studio $STUDIO_NAME already running.")
else:
    print(f"Creating Studio: $STUDIO_NAME")
```

import requests
from torch.utils.data import Dataset, DataLoader
import io
import os

class CDNParquetDataset(Dataset):
    def __init__(self, cdn_prefix, file_list, columns=("prompt", "response")):
        self.cdn_prefix = cdn_prefix.rstrip("/")
        self.files = file_list
        self.columns = columns

    def __len__(self):
        return len(self.files)

    def __getitem__(self, idx):
        url = f"{self.cdn_prefix}/{self.files[idx]}"

        resp = requests.get(url, timeout=30)
