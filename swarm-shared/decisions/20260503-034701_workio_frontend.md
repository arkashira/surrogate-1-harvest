# workio / frontend

## 1. Diagnosis
- No real-time subscription layer: UI relies on local React state; LINE webhook–driven changes (clock/leave/OT) are invisible until hard refresh → perceived latency and silent failures.
- Multi-tab/device divergence: actions performed in one tab do not propagate to others, causing inconsistent state and double submissions.
- Polling is absent or inefficient: if present, it is likely coarse or not deduplicated, wasting requests and battery.
- No optimistic UI for clock-in/out and leave/OT actions: users see lag and may retry, creating duplicates.
- Missing centralized store for live data: React state scattered across components makes it hard to add real-time updates consistently.

## 2. Proposed change
- Add a lightweight real-time subscription layer using Server-Sent Events (SSE) from the backend and a singleton `EventSource` in the frontend.
- Scope:
  - Frontend: create `src/lib/sse.ts` (singleton manager) and integrate into `src/App.tsx` (or layout) to subscribe to `/api/events` and dispatch normalized actions into a Zustand store.
  - Frontend: add `src/stores/useLiveStore.ts` (Zustand) to hold live entities (clocks, leaves, OT) and expose `upsert`/`remove` actions.
  - Frontend: convert critical UI slices (clock status, pending approvals, recent punches) to read from the live store instead of local component state.
  - Backend (minimal): ensure `GET /api/events` endpoint exists and streams `event: upsert`/`event: remove` with JSON payloads keyed by tenant+entity.

## 3. Implementation

### Frontend: SSE singleton (`src/lib/sse.ts`)
```ts
// src/lib/sse.ts
import { useLiveStore } from '@/stores/useLiveStore';

class SseClient {
  private source: EventSource | null = null;
  private reconnectDelay = 1000;
  private maxDelay = 30000;
  private tenantId: string | null = null;

  connect(tenantId: string) {
    if (this.source || this.tenantId === tenantId) return;
    this.tenantId = tenantId;

    const connect = () => {
      this.source = new EventSource(`/api/events?tenantId=${encodeURIComponent(tenantId)}`);

      this.source.addEventListener('upsert', (e) => {
        try {
          const payload = JSON.parse(e.data as string);
          useLiveStore.getState().upsert(payload);
        } catch (err) {
          console.error('[SSE] upsert parse error', err);
        }
      });

      this.source.addEventListener('remove', (e) => {
        try {
          const payload = JSON.parse(e.data as string);
          useLiveStore.getState().remove(payload);
        } catch (err) {
          console.error('[SSE] remove parse error', err);
        }
      });

      this.source.onerror = () => {
        this.source?.close();
        this.source = null;
        setTimeout(connect, this.reconnectDelay);
        this.reconnectDelay = Math.min(this.reconnectDelay * 2, this.maxDelay);
      };

      this.source.onopen = () => {
        this.reconnectDelay = 1000;
      };
    };

    connect();
  }

  disconnect() {
    this.source?.close();
    this.source = null;
    this.tenantId = null;
  }
}

export const sseClient = new SseClient();
```

### Frontend: Zustand live store (`src/stores/useLiveStore.ts`)
```ts
// src/stores/useLiveStore.ts
import { create } from 'zustand';

type Entity = { id: string; tenantId: string; [k: string]: any };

interface LiveState {
  clocks: Record<string, Entity>;
  leaves: Record<string, Entity>;
  otRequests: Record<string, Entity>;
  upsert: (payload: { type: 'clock' | 'leave' | 'ot'; data: Entity }) => void;
  remove: (payload: { type: 'clock' | 'leave' | 'ot'; id: string }) => void;
}

export const useLiveStore = create<LiveState>((set) => ({
  clocks: {},
  leaves: {},
  otRequests: {},
  upsert: (payload) =>
    set((state) => {
      const key = payload.type === 'clock' ? 'clocks' : payload.type === 'leave' ? 'leaves' : 'otRequests';
      return { [key]: { ...state[key], [payload.data.id]: payload.data } };
    }),
  remove: (payload) =>
    set((state) => {
      const key = payload.type === 'clock' ? 'clocks' : payload.type === 'leave' ? 'leaves' : 'otRequests';
      const { [payload.id]: _, ...rest } = state[key];
      return { [key]: rest };
    }),
}));
```

### Frontend: App integration (`src/App.tsx` or layout)
```tsx
// src/App.tsx (or your top layout)
import { useEffect } from 'react';
import { sseClient } from '@/lib/sse';
import { useTenant } from '@/hooks/useTenant'; // assume you have tenant context/hook

export function App() {
  const tenantId = useTenant(); // returns current tenant id

  useEffect(() => {
    if (tenantId) sseClient.connect(tenantId);
    return () => sseClient.disconnect();
  }, [tenantId]);

  return <YourRoutes />;
}
```

### Frontend: Optimistic UI example (clock button)
```tsx
// src/components/ClockButton.tsx
import { useLiveStore } from '@/stores/useLiveStore';
import { clockIn, clockOut } from '@/api/clock';

export function ClockButton() {
  const currentClock = useLiveStore((s) =>
    Object.values(s.clocks).find((c) => !c.clockedOutAt)
  );
  const upsert = useLiveStore((s) => s.upsert);

  const handleClock = async () => {
    const optimistic = { id: `temp-${Date.now()}`, tenantId: 'current', clockedInAt: new Date().toISOString(), clockedOutAt: null };
    upsert({ type: 'clock', data: optimistic });

    try {
      const result = currentClock ? await clockOut(currentClock.id) : await clockIn();
      // server returns canonical record; upsert will replace optimistic
      upsert({ type: 'clock', data: result });
    } catch (err) {
      // rollback: remove optimistic (or refetch)
      useLiveStore.getState().remove({ type: 'clock', id: optimistic.id });
    }
  };

  return (
    <button onClick={handleClock}>
      {currentClock ? 'Clock Out' : 'Clock In'}
    </button>
  );
}
```
