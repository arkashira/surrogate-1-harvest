# workio / quality

## Final synthesized implementation (best of both proposals)

**Core principles**
- **Idempotency by unique constraint + transactional upsert** (not advisory locks) — safest under concurrency and retries.
- **Commit-before-200** — DB transaction must fully commit before returning success to LINE.
- **Deterministic idempotency key** — tenant-scoped, derived from LINE event identifiers; optionally allow `X-Idempotency-Key` header for client-controlled keys.
- **Observability** — request ID, timings, and outcome metrics to detect retries, duplicates, and silent failures.
- **Realtime cross-tab/device sync** — publish after commit so all clients see consistent state immediately.

```sql
-- Stores raw LINE webhook events with idempotency guarantee
CREATE TABLE IF NOT EXISTS line_events (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id TEXT NOT NULL,
  external_id TEXT NOT NULL,               -- deterministic idempotency key
  type TEXT NOT NULL,                      -- 'clock' | 'leave' | 'ot'
  user_id TEXT NOT NULL,
  data JSONB NOT NULL,
  occurred_at TIMESTAMPTZ NOT NULL,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  CONSTRAINT line_events_tenant_external_uniq UNIQUE (tenant_id, external_id)
);

-- Optional: generic webhook ingest log for broader idempotency/auditing
CREATE TABLE IF NOT EXISTS webhook_ingest_log (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id TEXT NOT NULL,
  idempotency_key TEXT NOT NULL,
  event_type TEXT NOT NULL,
  external_id TEXT NOT NULL,
  processed_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  CONSTRAINT webhook_ingest_log_tenant_key_uniq UNIQUE (tenant_id, idempotency_key)
);
```

### 2) Webhook handler

File: `workio/server/src/routes/webhook/line.ts`

```ts
import express from 'express';
import { db } from '../../db';
import { and, eq } from 'drizzle-orm';
import { lineEvents, webhookIngestLog } from '../../db/schema';
import { broadcast } from '../../ws';
import { v4 as uuidv4 } from 'uuid';

const router = express.Router();

function deterministicIdempotencyKey(event: any, tenantId: string): string {
  // Prefer stable LINE identifiers; include tenant to avoid cross-tenant collisions
  return `line:${tenantId}:${event.source?.userId || ''}:${event.webhookEventId || ''}:${event.timestamp || ''}:${event.type || ''}:${event.mode || ''}`;
}

async function upsertLineEvent(
  tenantId: string,
  externalId: string,
  type: 'clock' | 'leave' | 'ot',
  userId: string,
  data: Record<string, unknown>,
  occurredAt: Date
) {
  return db.transaction(async (tx) => {
    // Idempotency check with row lock to prevent races under concurrent retries
    const existing = await tx
      .select()
      .from(lineEvents)
      .where(
        and(
          eq(lineEvents.tenantId, tenantId),
          eq(lineEvents.externalId, externalId)
        )
      )
      .for('update', { skipLocked: true })
      .execute();

    if (existing.length > 0) {
      return existing[0]; // idempotent no-op
    }

    const [row] = await tx
      .insert(lineEvents)
      .values({
        tenantId,
        externalId,
        type,
        userId,
        data,
        occurredAt,
        createdAt: new Date(),
      })
      .returning();

    // Record ingest for observability and broader idempotency
    await tx
      .insert(webhookIngestLog)
      .values({
        tenantId,
        idempotencyKey: externalId,
        eventType: type,
        externalId,
        processedAt: new Date(),
      })
      .onConflictDoNothing({ target: webhookIngestLog.idempotencyKey })
      .execute();

    return row;
  });
}

router.post('/line', express.raw({ type: 'application/json' }), async (req, res) => {
  const requestId = uuidv4();
  const started = Date.now();

  try {
    const body = JSON.parse(req.body.toString());
    const events = body.events || [];

    if (!Array.isArray(events) || events.length === 0) {
      return res.status(400).json({ requestId, error: 'invalid_payload' });
    }

    // Process sequentially to preserve ordering per webhook (simpler semantics)
    for (const ev of events) {
      const tenantId = ev.source?.tenantId || ev.destination?.[0] || 'default';
      const userId = ev.source?.userId;
      if (!userId) continue;

      // Map LINE event to internal type (customize to your domain)
      let type: 'clock' | 'leave' | 'ot' | null = null;
      if (ev.type === 'message' && ev.message?.text) {
        // heuristic example; replace with real mapping
        type = 'clock';
      } else if (ev.type === 'leave') {
        type = 'leave';
      } else if (ev.type === 'ot') {
        type = 'ot';
      }
      if (!type) continue;

      // Prefer provided idempotency key; otherwise deterministic key
      const externalId =
        req.header('X-Idempotency-Key') || deterministicIdempotencyKey(ev, tenantId);

      const occurredAt = ev.timestamp ? new Date(ev.timestamp) : new Date();

      // Transactional upsert + ingest log; commit guaranteed before returning 200
      await upsertLineEvent(tenantId, externalId, type, userId, ev, occurredAt);

      // Broadcast for realtime cross-tab/device sync
      broadcast(`tenant:${tenantId}:events`, {
        type: 'line_event',
        requestId,
        payload: { externalId, type, userId, occurredAt },
      });
    }

    // All DB work committed by upsertLineEvent transactions
    res.status(200).json({ requestId, ok: true });
  } catch (err) {
    console.error(`LINE webhook error requestId=${requestId}`, err);
    // Non-2xx causes LINE retry; idempotency prevents duplicates
    res.status(500).json({ requestId, error: 'server_error' });
  }
});

export default router;
```

```ts
// workio/src/lib/ws.ts
export function subscribeTenantEvents(tenantId: string, onMessage: (msg: any) => void) {
  const ws = new WebSocket(`wss://your-domain.com/ws?tenant=${tenantId}`);
  ws.onopen = () => console.debug('WS connected', tenantId);
  ws.onmessage = (e) => {
    try {
      onMessage(JSON.parse(e.data));
    } catch (err) {
      console.error('Invalid WS message', err);
    }
  };
  ws.onerror = (err) => console.error('WS error', err);
  return () => ws.close();
}

    }
  });
  return unsub;
}, [tenantId, queryClient]);
```

### 4) Verification & operational checks

- **Idempotency**: POST the same payload repeatedly (same `externalId`). Confirm only one row in `line_events` (unique constraint) and one entry in `webhook_ingest_log`.
- **Retry safety**: Force handler to return 500 after DB commit (simulate post-commit failure). LINE retries must not create duplicates.
- **Commit-before-200**: Ensure no background/async writes after 200. All writes must be inside the transaction returned to LINE.
