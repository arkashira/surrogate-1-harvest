#### 1) Add route (FastAPI example)
File: `app/api/v1/endpoints/cost_anomaly.py` (create or append)

```python
from fastapi import APIRouter, HTTPException
from app.services.knowledge_rag import query_top_hub
from datetime import datetime, timezone

router = APIRouter()

@router.get("/signal/top-hub", response_model=dict)
async def get_top_hub_signal() -> dict:
    """
    Sense + Signal: return today's top hub from knowledge graph
    as a cost-anomaly signal. No execution.
    """
    try:
        signal = query_top_hub(as_of=datetime.now(timezone.utc).date())
        return {
            "type": "cost-anomaly-signal",
            "category": "top-hub",
            "signal": signal,
            "meta": {
                "generated_at": datetime.now(timezone.utc).isoformat(),
                "policy": "sense+signal-no-execute"
            }
        }
    except Exception as exc:
        # Fail gracefully; return 503 so monitoring can alert
        raise HTTPException(status_code=503, detail=f"Unable to compute signal: {exc}") from exc
```

#### 2) Knowledge RAG service
File: `app/services/knowledge_rag.py`

```python
from datetime import date
from typing import Dict, List, Any
import logging

logger = logging.getLogger(__name__)

try:
    from app.libs.graph import query_top_hub as graph_top_hub
    GRAPH_AVAILABLE = True
except Exception:
    GRAPH_AVAILABLE = False
    logger.warning("Graph client unavailable; using fallback ranking.")

def query_top_hub(as_of: date) -> Dict[str, Any]:
    """
    Query top hub for the day.
    Returns:
      {
        "hub": "MOC",
        "score": 0.92,
        "signals": ["anomaly:spend-spike", "ri:low-coverage", "tag:orphaned"],
        "source": "knowledge-graph|fallback",
        "as_of": "YYYY-MM-DD"
      }
    """
    if GRAPH_AVAILABLE:
        try:
            result = graph_top_hub(as_of=as_of)
            if result and result.get("hub"):
                return {
                    "hub": result["hub"],
                    "score": result.get("score", 0.0),
                    "signals": result.get("signals", []),
                    "source": "knowledge-graph",
                    "as_of": str(as_of)
                }
        except Exception as exc:
            logger.warning("Graph query failed, falling back: %s", exc)

    # Fallback: lightweight local ranking (e.g., recent signals frequency)
    fallback = _fallback_top_hub(as_of)
    fallback["source"] = "fallback"
    fallback["as_of"] = str(as_of)
    return fallback

def _fallback_top_hub(as_of: date) -> Dict[str, Any]:
    # Minimal deterministic fallback: static priority list + date-based salt
    # Replace with real lightweight ranking if available (e.g., recent anomalies).
    hubs = ["MOC", "RI-COVERAGE", "TAG-ORPHAN", "SPIKE-DETECT", "FORECAST-DRIFT"]
    idx = (as_of.day + as_of.month * 7) % len(hubs)
    return {
        "hub": hubs[idx],
        "score": 0.5 + ((idx + 1) * 0.1),
        "signals": [
            "anomaly:spend-spike",
            "ri:low-coverage",
            "tag:orphaned"
        ],
    }
```

from fastapi.testclient import TestClient
from app.main import app
from unittest.mock import patch

client = TestClient(app)

def test_get_top_hub_signal():
    with patch("app.services.knowledge_rag.query_top_hub") as mock_q:
        mock_q.return_value = {
            "hub": "MOC",
            "score": 0.92,
            "signals": ["anomaly:spend-spike", "ri:low-coverage"],
            "source": "knowledge-graph",
            "as_of": "2026-05-02"

        resp = client.get("/api/v1/cost-anomaly/signal/top-hub")
        assert resp.status_code == 200
        data = resp.json()
        assert data["signal"]["hub"] == "MOC"
        assert data["category"] == "top-hub"
