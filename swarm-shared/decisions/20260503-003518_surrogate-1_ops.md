### snapshot_manifest.py
```python
#!/usr/bin/env python3
"""
Mac-side snapshot: list one date-partition of axentx/surrogate-1-training-pairs
and emit file_manifest.json for CDN-only training.

Usage:
    python snapshot_manifest.py --date 2026-04-29 --out file_manifest.json
"""
import argparse
import json
import os
import sys
from datetime import datetime

from huggingface_hub import HfApi, login

REPO_ID = "axentx/surrogate-1-training-pairs"

def main() -> None:
    parser = argparse.ArgumentParser(description="Create CDN file manifest for HF dataset partition")
    parser.add_argument("--date", required=True, help="Date partition (YYYY-MM-DD)")
    parser.add_argument("--out", default="file_manifest.json", help="Output JSON path")
    parser.add_argument("--token", default=os.getenv("HF_TOKEN"), help="HF token (env HF_TOKEN preferred)")
    args = parser.parse_args()

    if args.token:
        login(token=args.token)

    api = HfApi()
    prefix = f"{args.date}/"
    print(f"Listing repo tree: {REPO_ID} prefix={prefix}", file=sys.stderr)

    try:
        tree = api.list_repo_tree(repo_id=REPO_ID, path=prefix, recursive=True, repo_type="dataset")
    except Exception as e:
        print(f"HF API error (possible 429). Wait 360s and retry: {e}", file=sys.stderr)
        sys.exit(1)

    files = []
    total_bytes = 0
    for item in sorted(tree, key=lambda x: x.rfilename.lower()):
        if item.rfilename.endswith((".parquet", ".jsonl")):
            files.append({
                "path": item.rfilename,
                "size": getattr(item, "size", None),
                "etag": getattr(item, "etag", None),
            })
            if item.size:
                total_bytes += item.size

    manifest = {
        "date": args.date,
        "created_at": datetime.utcnow().isoformat() + "Z",
        "repo_id": REPO_ID,
        "prefix": prefix,
        "files": files,
        "count": len(files),
        "total_bytes": total_bytes,
    }

    with open(args.out, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2)

    print(f"Wrote {args.out} with {len(files)} files ({total_bytes} bytes)", file=sys.stderr)

if __name__ == "__main__":
    main()
```

#!/usr/bin/env python3
"""
Lightning training entrypoint that uses CDN-only fetches.
Reads file_manifest.json and streams files via hf_hub_download + pyarrow/jsonl.
Projects to {prompt, response} only at parse time.
"""
import json
import os
import sys
from pathlib import Path
from typing import Iterator, Dict, Any

import pyarrow.parquet as pq
from huggingface_hub import hf_hub_download

REPO_ID = "axentx/surrogate-1-training-pairs"
MANIFEST_PATH = os.getenv("MANIFEST_PATH", "file_manifest.json")
CACHE_DIR = os.getenv("HF_CACHE_DIR", "./.hf_cache")


def load_manifest(path: str) -> Dict[str, Any]:
    with open(path, "r", encoding="utf-8") as f:
        return json.load(f)


def stream_parquet_cdn(filepath: str) -> Iterator[Dict[str, str]]:
    local_path = hf_hub_download(
        repo_id=REPO_ID,
        filename=filepath,
        repo_type="dataset",
        cache_dir=CACHE_DIR,
        force_download=False,
    )
    pf = pq.ParquetFile(local_path)
    for batch in pf.iter_batches(batch_size=1024, columns=["prompt", "response"]):
        df = batch.to_pandas()
        for _, row in df.iterrows():
            # Defensive: ensure fields exist and are strings
            prompt = str(row.get("prompt", ""))
            response = str(row.get("response", ""))
            if prompt and response:
                yield {"prompt": prompt, "response": response}


def stream_jsonl_cdn(filepath: str) -> Iterator[Dict[str, str]]:
    local_path = hf_hub_download(
        repo_id=REPO_ID,
        filename=filepath,
        repo_type="dataset",
        cache_dir=CACHE_DIR,
        force_download=False,
    )
    with open(local_path, "r", encoding="utf-8") as f:
        for line in f:
            line = line.strip()
            if not line:
                continue
            try:
                obj = json.loads(line)
            except json.JSONDecodeError:
                continue
            # Project only prompt/response; ignore mixed schema extras
            prompt = str(obj.get("prompt", ""))
            response = str(obj.get("response", ""))
            if prompt and response:
                yield {"prompt": prompt, "response": response}


def stream_dataset(manifest_path: str) -> Iterator[Dict[str, str]]:
    manifest = load_manifest(manifest_path)
    files = manifest.get("files", [])
    if not files:
        print("No files in manifest", file=sys.stderr)
        return

    for entry in files:
        f = entry["path"]
        print(f"Streaming {f}", file=sys.stderr)
        if f.endswith(".parquet"):
            yield from stream_parquet_cdn(f)
        elif f.endswith(".jsonl"):
            yield from stream_jsonl_cdn(f)
        else:
            print(f"Skipping unsupported file: {f}", file=sys.stderr)


def build_dataloader(manifest_path: str, batch_size: int = 16):
    # Lightweight generator-based dataloader; replace with proper DataLoader as needed.
    def gen():
        batch = []
        for item in stream_dataset(manifest_path):
            batch.append(item)
            if len(batch) >= batch_size:
                yield batch
                batch = []
        if batch:
            yield batch

    return gen()


def main():
    if not Path(MANIFEST_PATH).exists():
        print(f"Manifest not found: {MANIFEST_PATH}. Run snapshot_manifest.py first.", file=sys.stderr)
        sys.exit(1)

    dataloader = build_dataloader(MANIFEST_PATH, batch_size=16
