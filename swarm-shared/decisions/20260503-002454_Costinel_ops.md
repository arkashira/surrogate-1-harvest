# Costinel / ops

## Implementation Plan — Costinel Top-Hub Signal Card (ops)

**Scope**: Production-ready, read-only ops card that surfaces the most-connected knowledge hub (e.g., "MOC") with contextual cost-governance signals. Uses CDN-bypass pattern for live cost data and embedded mock fallback for reliability. Ships in <2h.

```
Costinel (Next.js/React)

```

### Implementation Steps

#### 1. Create CDN-bypass cost fetcher (lib/cost-cdn.ts)
```typescript
// lib/cost-cdn.ts
export interface CostSnapshot {
  date: string;
  total: number;
  byService: Record<string, number>;
  byRegion: Record<string, number>;
  anomalies: Array<{
    service: string;
    delta: number;
    severity: 'low' | 'medium' | 'high';
  }>;
}

const CDN_BASE = 'https://huggingface.co/datasets/axentx/costinel-costdata/resolve/main';

export async function fetchDailyCost(date: string): Promise<CostSnapshot | null> {
  const path = `${date}/daily-summary.json`;
  try {
    const res = await fetch(`${CDN_BASE}/${path}`, {
      cache: 'no-store',
      headers: { Accept: 'application/json' },
    });
    if (!res.ok) return null;
    return await res.json();
  } catch {
    return null;
  }
}

export async function fetchCostTrend(days: number = 7): Promise<CostSnapshot[]> {
  const results: CostSnapshot[] = [];
  const now = new Date();
  
  for (let i = 0; i < days; i++) {
    const d = new Date(now);
    d.setDate(d.getDate() - i);
    const dateStr = d.toISOString().split('T')[0];
    const data = await fetchDailyCost(dateStr);
    if (data) results.unshift(data);
  }
  
  return results;
}
```

#### 2. Create TopHubSignalCard component (components/ops/TopHubSignalCard.tsx)
```typescript
// components/ops/TopHubSignalCard.tsx
'use client';

import { useEffect, useState } from 'react';
import { TrendingUp, AlertCircle, DollarSign, Network } from 'lucide-react';
import { fetchCostTrend } from '@/lib/cost-cdn';

interface HubInsight {
  id: string;
  label: string;
  connections: number;
  category: string;
  signals: Array<{
    type: 'cost-anomaly' | 'ri-opportunity' | 'governance';
    severity: 'low' | 'medium' | 'high';
    message: string;
    context: string;
  }>;
}

interface CostTrend {
  date: string;
  total: number;
}

export default function TopHubSignalCard() {
  const [hub, setHub] = useState<HubInsight | null>(null);
  const [trend, setTrend] = useState<CostTrend[]>([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    async function load() {
      try {
        // Fetch top hub from knowledge-rag (existing graph)
        const hubRes = await fetch('/api/ops/top-hub');
        if (hubRes.ok) {
          const data = await hubRes.json();
          setHub(data);
        }

        // Fetch cost trend via CDN-bypass
        const trendData = await fetchCostTrend(7);
        setTrend(trendData.map(d => ({ date: d.date, total: d.total })));
      } catch (e) {
        console.error('Failed to load hub signals:', e);
      } finally {
        setLoading(false);
      }
    }

    load();
  }, []);

  if (loading) {
    return (
      <div className="bg-white rounded-xl border border-gray-200 p-6 animate-pulse">
        <div className="h-6 bg-gray-200 rounded w-1/3 mb-4"></div>
        <div className="h-4 bg-gray-200 rounded w-1/2 mb-6"></div>
        <div className="space-y-3">
          <div className="h-16 bg-gray-100 rounded"></div>
          <div className="h-16 bg-gray-100 rounded"></div>
        </div>
      </div>
    );
  }

  if (!hub) {
    return (
      <div className="bg-white rounded-xl border border-gray-200 p-6">
        <div className="flex items-center gap-3 mb-4">
          <Network className="w-5 h-5 text-gray-400" />
          <h3 className="font-semibold text-gray-900">Top Hub Signals</h3>
        </div>
        <p className="text-sm text-gray-500">No hub insights available</p>
      </div>
    );
  }

  const highSeverityCount = hub.signals.filter(s => s.severity === 'high').length;
  const latestTrend = trend.length > 1 ? trend[trend.length - 1].total - trend[0].total : 0;
  const trendUp = latestTrend > 0;

  return (
    <div className="bg-white rounded-xl border border-gray-200 p-6 hover:shadow-md transition-shadow">
      <div className="flex items-start justify-between mb-4">
        <div className="flex items-center gap-3">
          <div className="w-10 h-10 rounded-lg bg-indigo-50 flex items-center justify-center">
            <Network className="w-5 h-5 text-indigo-600" />
          </div>
          <div>
            <h3 className="font-semibold text-gray-900">{hub.label}</h3>
            <p className="text-xs text-gray-500">{hub.connections} connections</p>
          </div>
        </div>
        {highSeverityCount > 0 && (
          <span className="px-2 py-1 text-xs font-medium bg-red-50 text-red-700 rounded-full">
            {highSeverityCount} high
          </span>
        )}
      </div>

      <div className="space-y-3 mb-4">
        {hub.signals.slice(0, 3).map((signal, i) => (
          <div key={i} className="flex items-start gap-2 p-3 bg-gray-50 rounded-lg">
            <AlertCircle className={`w-4 h-4 mt-0.5 flex-shrink-0 ${
              signal.severity === 'high' ? 'text-red-500' :
              signal.severity === 'medium' ? 'text-amber-500' : 'text-blue-500'
            }`} />
            <div className="flex-1 min-w-0">
              <p className="text-sm font-medium text-gray-900">{signal.message}</p>
              <p className="text-xs text-gray-500 mt-0.5">{signal.context}</p>
            </div>
          </div>
        ))}
      </div>

      {trend.length > 1 && (
        <div className="pt-4 border-t border-gray-100 flex items-center justify-between text-sm">
          <span className="text-gray-500">7-day trend</span>
          <div className="flex items-center gap-1 font-medium">
            <DollarSign className="w-4 h-4" />
            <span className={trendUp ? 'text-red-600' : 'text-green-600'}>
              {trendUp ? '+' : ''}{latestTrend.toLocaleString('en-US', { style: 'currency', currency: 'USD', maximumFractionDigits: 0 })}
            </span>
          </div>
        </div>
      )}
    </div>
  );
}
