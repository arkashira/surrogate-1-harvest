#### 1) `scripts/generate_file_list.py`

```python
#!/usr/bin/env python3
"""
Generate file_list.json for a given HF dataset repo and date folder.
Run on Mac (or any dev machine) after HF API rate-limit window clears.

Usage:
    python scripts/generate_file_list.py \
        --repo huggingface://datasets/axentx/surrogate-mirror \
        --date 2026-04-29 \
        --out surrogate/training/file_list.json
"""

import argparse
import json
import os
import sys

try:
    from huggingface_hub import list_repo_tree
except ImportError:
    print("ERROR: huggingface_hub required. Install with: pip install huggingface_hub")
    sys.exit(1)

CDN_BASE = "https://huggingface.co/datasets"

def build_file_list(repo_id: str, date_folder: str):
    # Normalize repo_id
    if repo_id.startswith("huggingface://datasets/"):
        repo_id = repo_id[len("huggingface://datasets/") :]

    entries = []

    # Top-level files
    try:
        tree = list_repo_tree(repo_id=repo_id, path="", recursive=False)
    except Exception as e:
        print(f"Failed to list repo root: {e}")
        sys.exit(1)

    for item in tree:
        if item.type == "file":
            entries.append({
                "repo": repo_id,
                "path": item.path,
                "cdn_url": f"{CDN_BASE}/{repo_id}/resolve/main/{item.path}"
            })

    # Date folder (non-recursive)
    date_path = date_folder.strip("/")
    try:
        date_tree = list_repo_tree(repo_id=repo_id, path=date_path, recursive=False)
    except Exception as e:
        print(f"Failed to list date folder '{date_path}': {e}")
        sys.exit(1)

    for item in date_tree:
        if item.type == "file":
            full_path = os.path.join(date_path, item.path)
            entries.append({
                "repo": repo_id,
                "path": full_path,
                "cdn_url": f"{CDN_BASE}/{repo_id}/resolve/main/{full_path}"
            })

    return entries

def main():
    parser = argparse.ArgumentParser(description="Generate CDN file list for surrogate training.")
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g., axentx/surrogate-mirror)")
    parser.add_argument("--date", required=True, help="Date folder (e.g., 2026-04-29)")
    parser.add_argument("--out", default="surrogate/training/file_list.json", help="Output JSON path")
    args = parser.parse_args()

    entries = build_file_list(args.repo, args.date)

    os.makedirs(os.path.dirname(os.path.abspath(args.out)), exist_ok=True)
    with open(args.out, "w") as f:
        json.dump(entries, f, indent=2)

    print(f"Wrote {len(entries)} files to {args.out}")

if __name__ == "__main__":
    main()
```

#### 2) `surrogate/training/train.py`

import json
import os
import sys
from pathlib import Path
from typing import List, Dict

import pyarrow.parquet as pq
import requests
from torch.utils.data import IterableDataset, DataLoader

_FILE_LIST_PATH = Path(__file__).parent / "file_list.json"

class CDNParquetIterable(IterableDataset):
    """
    Stream Parquet files via CDN URLs (no HF API/auth during training).
    Projects to {prompt, response} only at parse time.
    """
    def __init__(self, file_entries: List[Dict], max_retries: int = 3):
        self.file_entries = file_entries
        self.max_retries = max_retries

    def __iter__(self):
        for entry in self.file_entries:
            url = entry["cdn_url"]
            for attempt in range(self.max_retries):
                try:
                    resp = requests.get(url, timeout=30)
                    resp.raise_for_status()
                    tmp_path = "/tmp/temp.parquet"
                    with open(tmp_path, "wb") as f:
                        f.write(resp.content)
                    table = pq.read_table(tmp_path)
                    if "prompt" in table.column_names and "response" in table.column_names:
                        prompts = table["prompt"].to_pylist()
                        responses = table["response"].to_pylist()
                        for p, r in zip(prompts, responses):
                            if p is not None and r is not None:
                                yield {"prompt": p, "response": r}
                    # Done with this file
                    break
                except Exception as exc:
                    if attempt == self.max_retries - 1:
                        print(f"Failed to fetch {url}: {exc}", file=sys.stderr)
                    else:
                        continue

def make_dataloader(batch_size: int = 8, num_workers: int = 0):
    if not _FILE_LIST_PATH.is_file():
        raise FileNotFoundError(
            f"{_FILE_LIST_PATH} not found. Run scripts/generate_file_list.py first."
        )
    with open(_FILE_LIST_PATH) as f:
        entries = json.load(f)

    dataset = CDNParquetIterable(entries)
    return DataLoader(dataset, batch_size=batch_size, num_workers=num_workers)

#!/usr/bin/env python3
"""
Lightning Studio orchestration with idle-resilient retry and reuse.

Usage:
    python surrogate/orchestration/run_lightning.py \
        --script surrogate/training/train.py \
        --machine L40S \
        --cloud lightning-public-prod
"""

import argparse
import subprocess
import time
import sys

def run_with_retry(script_path: str, machine: str, cloud: str, max_retries: int = 3):
    attempt = 0
    while attempt < max_retries:
        attempt += 1
        print(f"Starting run (attempt {attempt}/{max_retries}) on {machine} in {cloud}")
        cmd = [
            "lightning", "run", "script", script_path,
            "--machine", machine,
            "--cloud", cloud,
            "--reuse",
        ]
        try:
            result = subprocess.run(cmd, check=True)
            if result.returncode == 0:
                print("Run completed successfully.")
                return 0
        except subprocess.CalledProcessError as e:
            print(f"Run failed (exit {e.returncode})")
            if attempt < max_retries:
                wait = 10 * attempt
                print(f"Retrying in {wait}s...")
                time.sleep(wait)
            else:
                print("Max retries exceeded.")
                return e.returncode
    return 1

def main():
    parser = argparse.ArgumentParser(description="Lightning Studio orchestration with idle-resilient retry and reuse.")
    parser.add_argument("--script", required=True, help="Training script path")
    parser.add_argument("--machine", default="L40S", help="Machine type (e.g., L40S, H200)")
    parser.add_argument("--cloud", default="lightning-public-prod", help="Cloud provider")
