# workio / ops

## Final Synthesis — One Correct, Actionable Plan

**Core principle:** enforce exactly-once semantics at the *business-entity* level (one open punch per user) and at the *request* level (idempotency key), with clear failure modes and observability.

---

### 1) Diagnosis (resolved)

- **Duplicate root causes**  
  1. No idempotency on `/punches` → retries/duplicates.  
  2. No DB constraint for “one open punch per user” → concurrent clock-ins.  
  3. Non-atomic state transitions (clock-in/clock-out) → races.  
  4. LINE webhook retries without de-duplication → phantom punches.  
  5. Missing request traceability → hard to debug.

- **Contradictions resolved**  
  - Prefer **DB-level uniqueness** over app-only checks (stronger correctness).  
  - Prefer **transactional upsert/close** over separate insert+update where possible (atomicity).  
  - Prefer **lightweight idempotency table** keyed by business intent (userId+transition) over opaque keys alone, but still accept `Idempotency-Key` header for client retries.  
  - LINE de-duplication should happen **before** business handler and be scoped to message id, not full event payload, to avoid false conflicts.

CREATE TABLE IF NOT EXISTS idempotency_keys (

### 3) Idempotency Middleware (reusable)

```ts
// workio/server/src/middleware/idempotency.ts
import { Request, Response, NextFunction } from 'express';
import crypto from 'crypto';
import pool from '../db';

export interface IdempotencyLocals {
  idemResolvedPunchId?: number;
}

export async function idempotency(
  req: Request,
  res: Response,
  next: NextFunction
) {
  const key = req.header('Idempotency-Key') || req.header('IdempotencyKey');
  if (!key) return next();

  const body = req.body || {};
  const hash = crypto.createHash('sha256').update(JSON.stringify(body)).digest('hex');
  const client = await pool.connect();

  try {
    await client.query('BEGIN');
    const { rows } = await client.query<{
      punch_id: number;
      payload_hash: string;
      operation: string;
    }>(
      `INSERT INTO idempotency_keys (key, user_id, operation, payload_hash)
       VALUES ($1, $2, $3, $4)
       ON CONFLICT (key) DO UPDATE
       SET payload_hash = EXCLUDED.payload_hash
       RETURNING punch_id, payload_hash, operation`,
      [key, body.userId, body._operation || 'unknown', hash]
    );

    const prev = rows[0];
    // If same hash and we already have a punch, skip business logic
    if (prev.payload_hash === hash && prev.punch_id) {
      (res.locals as IdempotencyLocals).idemResolvedPunchId = prev.punch_id;
      await client.query('COMMIT');
      client.release();
      return res.status(200).json({ punchId: prev.punch_id, idem: true });
    }

    // Different body under same key: reject to avoid confusion
    if (prev.payload_hash !== hash) {
      await client.query('ROLLBACK');
      client.release();
      return res.status(409).json({ error: 'Idempotency key used with different body' });
    }

    // New request: attach client and key for handler to finalize
    (res.locals as any)._idemClient = client;
    (res.locals as any)._idemKey = key;
    next();
  } catch (err) {
    await client.query('ROLLBACK').catch(() => {});
    client.release();
    next(err);
  }
}
```

### 4) LINE Webhook De-duplication

```ts
// workio/server/src/middleware/lineWebhook.ts
import { Request, Response, NextFunction } from 'express';
import crypto from 'crypto';
import pool from '../db';

export async function dedupeLineWebhook(
  req: Request,
  res: Response,
  next: NextFunction
) {
  const events = req.body?.events;
  if (!Array.isArray(events)) return next();

  const client = await pool.connect();
  try {
    await client.query('BEGIN');
    for (const ev of events) {
      const msgId = ev.message?.id;
      const userId = ev.source?.userId;
      if (!msgId || !userId) continue;

      const key = `line:${userId}:${msgId}`;
      const hash = crypto.createHash('sha256').update(JSON.stringify(ev)).digest('hex');
      await client.query(
        `INSERT INTO idempotency_keys (key, user_id, operation, payload_hash)
         VALUES ($1, $2, $3, $4)
         ON CONFLICT (key) DO NOTHING`,
        [key, userId, 'line_event', hash]
      );
    }
    await client.query('COMMIT');
    next();
  } catch (err) {
    await client.query('ROLLBACK').catch(() => {});
    next(err);
  } finally {
    client.release();
  }
}
```
