MANIFEST="${STAGING_DATE_DIR}/manifest.json"

mkdir -p "${STAGING_DATE_DIR}"

python3 "$(dirname "$0")/discover.py" \
  --repo "${REPO}" \
  --date-folder "${DATE_FOLDER}" \
  --out-dir "${STAGING_DATE_DIR}" \
  --manifest "${MANIFEST}"

echo "✅ Discover complete. Manifest: ${MANIFEST}"

### 3) Core engine (`/opt/axentx/airship/scripts/discover.py`)

```python
#!/usr/bin/env python3
"""
discover.py
- Lists a single date folder from an HF dataset repo (non-recursive)
- Downloads files via CDN (no auth)
- Projects to {prompt, response}
- Writes deterministic parquet (content-hash filename)
- Emits manifest.json with CDN-resolvable paths
"""
import argparse
import hashlib
import json
import os
import sys
import tempfile
from pathlib import Path
from typing import Any, Dict, List

import pyarrow as pa
import pyarrow.parquet as pq
import requests
from huggingface_hub import list_repo_tree

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def sha256_bytes(b: bytes) -> str:
    return hashlib.sha256(b).hexdigest()

def project_to_prompt_response(table: pa.Table) -> pa.Table:
    cols = []
    for name in ("prompt", "response"):
        if name in table.column_names:
            cols.append(table[name].cast(pa.string()))
        else:
            cols.append(pa.array([""] * table.num_rows, type=pa.string()))
    return pa.table({"prompt": cols[0], "response": cols[1]})

def safe_read_file(path: Path) -> pa.Table:
    suffix = path.suffix.lower()
    try:
        if suffix == ".parquet":
            return pq.read_table(path)
        elif suffix in (".jsonl", ".json"):
            return pa.json.read_json(str(path))
        elif suffix in (".csv", ".tsv"):
            parse_options = pa.csv.ParseOptions(delimiter="\t" if suffix == ".tsv" else ",")
            return pa.csv.read_csv(str(path), parse_options=parse_options)
        else:
            content = path.read_bytes()
            lines = content.splitlines()
            prompts, responses = [], []
            for line in lines:
                line = line.decode("utf-8", errors="replace").strip()
                if not line:
                    continue
                if "\t" in line:
                    p, r = line.split("\t", 1)
                    prompts.append(p)
                    responses.append(r)
                elif ":" in line:
                    p, r = line.split(":", 1)
                    prompts.append(p)
                    responses.append(r.strip())
                else:
                    prompts.append(line)
                    responses.append("")
            return pa.table({"prompt": prompts, "response": responses})
    except Exception as exc:
        raise RuntimeError(f"Failed to read {path}: {exc}") from exc

def run(repo: str, date_folder: str, out_dir: Path, manifest_path: Path) -> None:
    items = list_repo_tree(repo=repo, path=date_folder, recursive=False)
    file_entries = [it for it in items if it.type == "file"]

    manifest: List[Dict[str, Any]] = []
    rows_total = 0

    for entry in file_entries:
        cdn_url = CDN_TEMPLATE.format(repo=repo, path=entry.path)
        out_file = out_dir / entry.path
        out_file.parent.mkdir(parents=True, exist_ok=True)

        resp = requests.get(cdn_url, timeout=60)
        resp.raise_for_status()
        raw = resp.content

        with tempfile.NamedTemporaryFile(delete=False, suffix=out_file.suffix) as tmp:
            tmp.write(raw)
            tmp_path = Path(tmp.name)

        try:
            table = safe_read_file(tmp_path)
            projected = project_to_prompt_response(table)
            if projected.num_rows == 0:
                tmp_path.unlink(missing_ok=True)
                continue

            content_hash = sha256_bytes(raw)
            slug_file = f"{content_hash[:16]}.parquet"
            final_parquet = out_file.parent / slug_file
            pq.write_table(projected, final_parquet)

            manifest.append({
                "cdn_url": cdn_url,
                "local_path": str(final_parquet),
                "sha256": sha256_bytes(pq.read_table(final_parquet).to_pandas().to_parquet(index=False)),
                "row_count": projected.num_rows,
                "projected_cols": ["prompt", "response"],
            })
            rows_total += projected.num_rows
        finally:
            tmp_path.unlink(missing_ok=True)

    manifest_path.write_text(
        json.dumps(
            {
                "repo": repo,
                "date_folder": date_folder,
                "row_count_total": rows_total,
                "files": sorted(manifest, key=lambda x: x["local_path"]),
            },
            indent=2,
            sort_keys=True,
        )
    )

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Discover HF dataset files and stage as parquet.")
    parser.add_argument("--repo", required=True, help="HF dataset repo (owner/name)")
    parser.add_argument("--date-folder", required=True, help="Date folder in repo")
    parser.add_argument("--out-dir", required=True, help="Staging output directory")
    parser.add_argument("--manifest", required=True, help="Path to manifest.json")
    args = parser.parse_args()

    try:
        run(args.repo, args.date_folder, Path(args.out_dir), Path(args.manifest))
    except Exception as e:
        print(f"❌ Discover failed: {e}", file=sys.stderr)
        sys.exit(1)
```
