# workio / backend

## 1. Diagnosis
- Missing `X-Line-Signature` HMAC-SHA256 verification on `/webhook/line` allows spoofed/replayed events.
- No idempotency guard (deduplication key) enables duplicate clock-in/out or leave/OT records from retries.
- Body-parser JSON middleware mutates raw body before HMAC check; signature verification must use raw bytes.
- Tenant context is not enforced on webhook entry — events could be applied to the wrong tenant.
- No atomic upsert for clock events; race conditions between duplicate webhooks can create double entries.

## 2. Proposed change
File: `/opt/axentx/workio/server/src/routes/webhook/line.ts` (create if absent) or modify existing route handler for `POST /webhook/line`.  
Scope: add raw-body middleware, verify `X-Line-Signature`, enforce tenant-scoped idempotent upsert for clock events (and extendable to leave/OT).

```ts
// server/src/routes/webhook/line.ts
import express, { Request, Response, NextFunction } from 'express';
import crypto from 'crypto';
import { getConnection } from '../db';
import rawBody from 'raw-body';

const router = express.Router();
const CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';

function verifyLineSignature(raw: Buffer, signature: string): boolean {
  if (!signature || !CHANNEL_SECRET) return false;
  const expected = crypto
    .createHmac('sha256', CHANNEL_SECRET)
    .update(raw)
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

// Raw-body middleware for this route only
router.use(async (req: Request, res: Response, next: NextFunction) => {
  try {
    const buf = await rawBody(req);
    ;(req as any).rawBody = buf;
    // restore json parsing for downstream handlers
    req.body = JSON.parse(buf.toString('utf8'));
    next();
  } catch (err) {
    res.status(400).json({ error: 'Invalid body' });
  }
});

router.post('/webhook/line', async (req: Request, res: Response) => {
  const signature = req.get('X-Line-Signature') || '';
  const raw = (req as any).rawBody as Buffer;

  if (!verifyLineSignature(raw, signature)) {
    return res.status(401).json({ error: 'Invalid signature' });
  }

  const events = req.body.events;
  if (!Array.isArray(events)) {
    return res.status(400).json({ error: 'No events' });
  }

  const client = await getConnection().connect();
  try {
    await client.query('BEGIN');

    for (const ev of events) {
      if (ev.type !== 'message' || ev.message?.type !== 'text') continue;

      const userId = ev.source?.userId;
      const tenantId = extractTenantIdFromToken(ev); // implement: map LINE userId or channel to tenant
      const text = ev.message.text.trim();
      const ts = new Date(ev.timestamp);

      if (!userId || !tenantId) continue;

      // Idempotent clock event upsert by externalId (LINE event id)
      await client.query(
        `INSERT INTO clock_events (tenant_id, user_id, event_type, recorded_at, external_id, created_at)
         VALUES ($1, $2, $3, $4, $5, NOW())
         ON CONFLICT (tenant_id, external_id) DO UPDATE
           SET event_type = EXCLUDED.event_type,
               recorded_at = EXCLUDED.recorded_at
         WHERE clock_events.tenant_id = EXCLUDED.tenant_id`,
        [tenantId, userId, text.includes('ออก') ? 'out' : 'in', ts, ev.message.id]
      );

      // Optional: reply via LINE API (fire-and-forget)
      // await replyToLine(ev.replyToken, `บันทึกการ${text.includes('ออก') ? 'ออก' : 'เข้า'}งานแล้ว`);
    }

    await client.query('COMMIT');
    res.status(200).json({ ok: true });
  } catch (err) {
    await client.query('ROLLBACK');
    console.error('Webhook error:', err);
    res.status(500).json({ error: 'Processing failed' });
  } finally {
    client.release();
  }
});

function extractTenantIdFromToken(ev: any): string | null {
  // Placeholder: map by channelId or use a lookup table.
  // For MVP, use a single tenant from env or a static mapping.
  return process.env.DEFAULT_TENANT_ID || null;
}

export default router;
```

-- server/src/db/schema.sql  (add if not present)
CREATE TABLE IF NOT EXISTS clock_events (
  id BIGSERIAL PRIMARY KEY,
  tenant_id TEXT NOT NULL,
  user_id TEXT NOT NULL,
  event_type TEXT NOT NULL, -- 'in' | 'out'
  recorded_at TIMESTAMPTZ NOT NULL,
  external_id TEXT NOT NULL, -- LINE message id for idempotency
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  UNIQUE (tenant_id, external_id)
);
CREATE INDEX IF NOT EXISTS idx_clock_events_tenant_user ON clock_events (tenant_id, user_id, recorded_at);
