# airship / frontend

## Highest-Value Incremental Improvement (≤2h)

**What**: Add a CDN-first training slice to Surrogate that:
1. Lists one date folder from HF once (Mac orchestration) → saves `file-list.json`
2. Embeds that list in Lightning training so data loading uses **only CDN URLs** (zero HF API calls during train)
3. Reuses a running Lightning Studio (saves quota) and falls back to L40S if H200 unavailable
4. Projects mixed-schema files to `{prompt,response}` at parse time and stores attribution in filename pattern (`batches/mirror-merged/{date}/{slug}.parquet`)

**Why**: Directly fixes HF 429 and Lightning quota waste while remaining a small, shippable slice (<2h).

### 2) `scripts/build_file_list.py` (run on Mac)
```python
#!/usr/bin/env python3
"""
Mac-side helper: list a single date folder in HF dataset repo
and emit file_list.json for CDN-only training.

Usage:
  python scripts/build_file_list.py \
    --repo huggingface-username/airship-surrogate-mirror \
    --date 2026-04-29 \
    --out surrogate/train/file_list.json
"""

import argparse
import json
import os
import sys
import time
from pathlib import Path

try:
    from huggingface_hub import list_repo_tree
except ImportError:
    print("Install: pip install huggingface_hub")
    sys.exit(1)

HF_API_RATE_LIMIT_WAIT = 360  # seconds after 429

def list_date_folder(repo_id: str, date: str, retries: int = 3):
    """List files under {date}/ (non-recursive) with retry on 429."""
    prefix = f"{date}/"
    for attempt in range(1, retries + 1):
        try:
            tree = list_repo_tree(repo_id=repo_id, path=prefix, recursive=False)
            files = [
                f"{prefix}{item.path[len(prefix):]}"
                for item in tree
                if not item.path.endswith("/")
            ]
            return sorted(files)
        except Exception as exc:
            if hasattr(exc, "response") and getattr(exc.response, "status_code", None) == 429:
                wait = HF_API_RATE_LIMIT_WAIT
                print(f"[{attempt}] Rate limited 429 — waiting {wait}s", file=sys.stderr)
                time.sleep(wait)
                continue
            print(f"[{attempt}] Error listing repo: {exc}", file=sys.stderr)
            if attempt == retries:
                raise
            time.sleep(5)
    raise RuntimeError("Exhausted retries on HF API 429")

def main():
    parser = argparse.ArgumentParser(description="Build CDN file list for Surrogate training")
    parser.add_argument("--repo", required=True, help="HF dataset repo id")
    parser.add_argument("--date", required=True, help="Date folder (YYYY-MM-DD)")
    parser.add_argument("--out", default="surrogate/train/file_list.json", help="Output JSON path")
    args = parser.parse_args()

    files = list_date_folder(args.repo, args.date)
    out_path = Path(args.out)
    out_path.parent.mkdir(parents=True, exist_ok=True)
    out_path.write_text(json.dumps({"repo": args.repo, "date": args.date, "files": files}, indent=2))
    print(f"Wrote {len(files)} files to {out_path}")

if __name__ == "__main__":
    main()
```

Make executable:

### 3) `train/cdn_dataset.py` (CDN-only dataset)
```python
from pathlib import Path
from typing import Dict, List, Optional
import json
import random

import requests
from datasets import Dataset, DatasetDict
from transformers import AutoTokenizer

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def load_file_list(file_list_path: Path) -> Dict:
    return json.loads(file_list_path.read_text())

def cdn_fetch_text(repo: str, path: str, timeout: float = 30.0) -> str:
    url = CDN_TEMPLATE.format(repo=repo, path=path)
    resp = requests.get(url, timeout=timeout)
    resp.raise_for_status()
    return resp.text

def parse_mixed_file_to_qa(raw_text: str, filename: str) -> List[Dict[str, str]]:
    """
    Project mixed-schema file to {prompt, response}.
    Attribution moved to filename pattern:
      batches/mirror-merged/{date}/{slug}.parquet
    """
    # Minimal heuristic: split by common separators or JSONL lines.
    # Replace with your domain-specific parser as needed.
    examples = []
    for line in raw_text.strip().splitlines():
        line = line.strip()
        if not line:
            continue
        try:
            # Try JSONL first
            import json as _json
            obj = _json.loads(line)
            prompt = obj.get("prompt") or obj.get("input") or obj.get("question") or ""
            response = obj.get("response") or obj.get("output") or obj.get("answer") or ""
            if prompt and response:
                examples.append({"prompt": prompt, "response": response})
        except Exception:
            # Fallback: treat whole line as prompt, empty response (or customize)
            examples.append({"prompt": line, "response": ""})
    return examples

def build_cdn_dataset(file_list_path: Path, max_files: Optional[int] = None) -> Dataset:
    meta = load_file_list(file_list_path)
    repo = meta["repo"]
    files = meta["files"]
    if max_files:
        files = files[:max_files]

    rows = []
    for fpath in files:
        try:
            raw = cdn_fetch_text(repo, fpath)
            qas = parse_mixed_file_to_qa(raw, fpath)
            for qa in qas:
                rows.append({
                    "prompt": qa["prompt"],
                    "response": qa["response"],
                    "source_file": fpath,
                })
        except Exception as exc:
            # Log and skip bad files to avoid crashing training
            print(f"Skipping {fpath}: {exc}")
            continue

    return Dataset.from_list(rows)
```
