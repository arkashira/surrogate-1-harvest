### 1) `scripts/list_hf_files.py`

```python
#!/usr/bin/env python3
"""
Run once on Mac (after HF rate-limit window clears).
Lists ONE date folder (non-recursive) and saves train_files.json.

Usage:
  HF_REPO="elena/surrogate-mirror" python scripts/list_hf_files.py 2026-05-03
"""
import json
import os
import sys
from pathlib import Path

try:
    from huggingface_hub import list_repo_tree
except ImportError:
    print("pip install huggingface_hub")
    sys.exit(1)

HF_REPO = os.getenv("HF_REPO")
if not HF_REPO:
    print("Set HF_REPO env var")
    sys.exit(1)

date_slug = sys.argv[1] if len(sys.argv) > 1 else "2026-05-03"
folder = f"batches/mirror-merged/{date_slug}"

# Non-recursive to avoid pagination explosion
tree = list_repo_tree(repo_id=HF_REPO, path=folder, recursive=False)
files = sorted(
    f.rfilename for f in tree if f.rfilename.endswith(".parquet")
)

out = Path("surrogate/train_files.json")
out.parent.mkdir(parents=True, exist_ok=True)
out.write_text(
    json.dumps({"repo": HF_REPO, "date": date_slug, "files": files}, indent=2)
)
print(f"Wrote {len(files)} files -> {out}")
```

# -------------------------
# CDN-only dataset
# -------------------------
class CDNParquetDataset(Dataset):
    def __init__(self, files_json: str):
        manifest = json.loads(Path(files_json).read_text())
        self.repo = manifest["repo"]
        self.files = manifest["files"]

    def __len__(self):
        return len(self.files)

    def _fetch_parquet(self, file_path: str):
        url = f"{CDN_BASE}/{self.repo}/resolve/main/{file_path}"
        for attempt in range(1, MAX_RETRIES + 1):
            try:
                r = requests.get(url, timeout=30)
                r.raise_for_status()
                with tempfile.NamedTemporaryFile(delete=False, suffix=".parquet") as f:
                    f.write(r.content)
                    f.flush()
                    table = pq.read_table(f.name, columns=["prompt", "response"])
                os.unlink(f.name)
                return table
            except Exception as e:
                if attempt == MAX_RETRIES:
                    raise
                print(f"Retry {attempt}/{MAX_RETRIES} for {url}: {e}")
                time.sleep(2**attempt)

    def __getitem__(self, idx):
        table = self._fetch_parquet(self.files[idx])
        # Project to prompt/response only; ignore mixed schemas
        row = table.slice(0, 1).to_pydict()
        return {
            "prompt": row.get("prompt", [""])[0],
            "response": row.get("response", [""])[0],
        }

def get_or_create_studio(name: str = "surrogate-train"):
    ts = Teamspace()
    for s in ts.studios:

            print(f"Reusing running studio: {name}")
            return s
    from lightning.pytorch import Studio
    print(f"Creating studio: {name}")

        machine="L40S",
        create_ok=True,
    )

def ensure_running(studio):
    if studio.status != "Running":
        print(f"Studio stopped ({studio.status}). Restarting...")
        studio.start(machine="L40S")
        # Wait until running or timeout
        start = time.time()
        while studio.status != "Running":
            if time.time() - start > IDLE_RESTART_TIMEOUT:
                raise TimeoutError("Studio failed to restart within timeout")
            time.sleep(IDLE_POLL_INTERVAL)
            studio.refresh()
