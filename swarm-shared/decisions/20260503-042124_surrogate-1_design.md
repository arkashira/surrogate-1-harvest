# surrogate-1 / design

## Final Implementation Plan (≤2h)

**Highest-value improvement**: Replace fragile shell-based ingestion with a **manifest-driven, CDN-bypass Python worker** that eliminates HF API rate limits during training data loads and prevents mixed-schema CastErrors.

### Changes

1. **Add `bin/manifest.py`** — Mac-side script that lists one date folder via HF API (rate-limit window respected), writes `manifest.json` with CDN URLs and expected row counts.
2. **Replace `bin/dataset-enrich.sh` with `bin/worker.py`** — GitHub Actions runner script that:
   - Reads `manifest.json` and picks its 1/16 shard by `hash(slug) % 16`.
   - Downloads assigned files **directly from CDN** (`https://huggingface.co/datasets/.../resolve/main/...`) with no Authorization header.
   - Projects each file to `{prompt, response}` only at parse time (avoids pyarrow CastError on mixed schemas).
   - Dedups via central `lib/dedup.py` md5 store.
   - Outputs `batches/public-merged/<date>/shard<N>-<HHMMSS>.jsonl`.
3. **Update `.github/workflows/ingest.yml`** to:
   - Run `manifest.py` once in a setup step (or reuse cached manifest).
   - Pass `SHARD_ID` and `MANIFEST_PATH` to each matrix job.
   - Use `python bin/worker.py` instead of `bash bin/dataset-enrich.sh`.
4. **Remove shell ingestion logic** from `bin/dataset-enrich.sh` (keep only a thin wrapper that calls `python bin/worker.py` for backward compatibility if needed).

#### bin/manifest.py
```python
#!/usr/bin/env python3
"""
Generate manifest for one date folder.
Run on Mac (or in CI once per cycle) while HF API rate-limit window is clear.
"""
import json, os, sys, hashlib
from datetime import datetime, timezone
from huggingface_hub import HfApi

REPO = "datasets/axentx/surrogate-1-training-pairs"
DATE = os.getenv("INGEST_DATE", datetime.now(timezone.utc).strftime("%Y-%m-%d"))
OUT = os.getenv("MANIFEST_OUT", f"manifest-{DATE}.json")

def main() -> None:
    api = HfApi()
    prefix = f"{DATE}/"
    entries = list(api.list_repo_tree(REPO, path=prefix, recursive=False))

    files = []
    for e in entries:
        if not e.path.endswith((".parquet", ".jsonl", ".json")):
            continue
        slug = e.path.rsplit("/", 1)[-1].split(".")[0]
        files.append(
            {
                "path": e.path,
                "cdn_url": f"https://huggingface.co/{REPO}/resolve/main/{e.path}",
                "size": e.size or 0,
                "slug": slug,
                "shard": int(hashlib.md5(slug.encode()).hexdigest(), 16) % 16,
            }
        )

    manifest = {
        "date": DATE,
        "created_utc": datetime.now(timezone.utc).isoformat(),
        "repo": REPO,
        "files": files,
    }

    with open(OUT, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2)

    print(f"Wrote {len(files)} files -> {OUT}")

if __name__ == "__main__":
    main()
```

#### bin/worker.py
```python
#!/usr/bin/env python3
"""
GitHub Actions shard worker: CDN-only ingestion, schema projection, dedup.
"""
import json, os, sys, hashlib
from datetime import datetime
from pathlib import Path

import pyarrow as pa
import pyarrow.parquet as pq
import requests

sys.path.insert(0, str(Path(__file__).parent))
from lib.dedup import DedupStore  # noqa

MANIFEST = os.getenv("MANIFEST_PATH", "manifest.json")
SHARD_ID = int(os.getenv("SHARD_ID", "0"))
DATE = os.getenv("INGEST_DATE", "today")
OUT_DIR = Path(f"batches/public-merged/{DATE}")
OUT_DIR.mkdir(parents=True, exist_ok=True)

def project_to_pair(obj) -> dict | None:
    """Return {prompt, response} only; tolerate mixed schemas."""
    prompt = obj.get("prompt") or obj.get("input") or obj.get("question")
    response = obj.get("response") or obj.get("output") or obj.get("answer")
    if not prompt or not response:
        return None
    return {"prompt": str(prompt), "response": str(response)}

def hash_row(row: dict) -> str:
    return hashlib.md5(f"{row['prompt']}\n{row['response']}".encode()).hexdigest()

def stream_jsonl(url: str):
    with requests.get(url, timeout=60) as r:
        r.raise_for_status()
        for line in r.iter_lines(decode_unicode=True):
            if not line or line.isspace():
                continue
            yield json.loads(line)

def stream_parquet(url: str):
    with requests.get(url, timeout=120) as r:
        r.raise_for_status()
        with pa.BufferReader(r.content) as buf:
            table = pq.read_table(buf)
            cols = set(table.column_names)
            # Find candidate text columns
            prompt_col = next((c for c in ("prompt", "input", "question") if c in cols), None)
            response_col = next((c for c in ("response", "output", "answer") if c in cols), None)
            if not prompt_col or not response_col:
                return
            for i in range(table.num_rows):
                row = {
                    "prompt": str(table[prompt_col][i].as_py()),
                    "response": str(table[response_col][i].as_py()),
                }
                if row["prompt"] and row["response"]:
                    yield row

def main() -> None:
    with open(MANIFEST, encoding="utf-8") as f:
        manifest = json.load(f)

    shard_files = [f for f in manifest["files"] if f["shard"] == SHARD_ID]
    if not shard_files:
        print(f"No files for shard {SHARD_ID}")
        return

    stamp = datetime.now().strftime("%H%M%S")
    out_path = OUT_DIR / f"shard{SHARD_ID}-{stamp}.jsonl"
    dedup = DedupStore()

    written = 0
    with out_path.open("w", encoding="utf-8") as out_f:
        for meta in shard_files:
            url = meta["cdn_url"]
            try:
                if url.endswith(".jsonl"):
                    rows = stream_jsonl(url)
                elif url.endswith(".parquet"):
                    rows = stream_parquet(url)
                else:
                    continue

                for obj in rows:
                    pair = project_to_pair(obj) if isinstance(obj, dict) else None
                    if not pair:
                        continue
                    h = hash_row(pair)
                    if dedup.seen(h):
                        continue
                    dedup.add(h)
                    out_f.write(json.dumps(pair, ensure_ascii=False) + "\n")
                    written += 1
            except Exception as exc:
                print(f"Skip {url}: {exc}", file=sys.stderr)

    print(f"Shard {SHARD_ID}: wrote {written} rows -> {out_path}")

if __name__ == "__main__":
    main()
```
