## Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16`, `DATE`, `HF_TOKEN`
- Single `list_repo_tree(path=DATE, recursive=False)` → deterministic file list saved to `manifest-{DATE}.json`
- Deterministic shard assignment: `hash(slug) % SHARD_TOTAL == SHARD_ID`
- Downloads only assigned files via **HF CDN bypass** (`https://huggingface.co/datasets/.../resolve/main/...`) — zero API calls during stream
- Projects each file to `{prompt, response}` only at parse time (avoids pyarrow CastError on mixed schemas)
- Dedups via central `lib/dedup.py` md5 store
- Uploads to `batches/public-merged/{DATE}/shard{N}-{HHMMSS}.jsonl`
- Reuses existing HF Space if running; otherwise starts one (L40S fallback)
- Exits cleanly on HF 429 (wait 360s) and commit-cap backoff

### 1. Create `bin/dataset-enrich.py`

```python
#!/usr/bin/env python3
"""
CDN-bypass, manifest-driven ingestion worker for surrogate-1.
Usage:
  SHARD_ID=3 SHARD_TOTAL=16 DATE=2026-04-29 \
  HF_TOKEN=hf_xxx \
  python bin/dataset-enrich.py
"""
import os, sys, json, hashlib, time, datetime, subprocess, tempfile
from pathlib import Path

import requests
from huggingface_hub import HfApi, hf_hub_download, login

# ── config --
REPO_DATASET = "axentx/surrogate-1-training-pairs"
HF_TOKEN = os.getenv("HF_TOKEN")
SHARD_ID = int(os.getenv("SHARD_ID", "0"))
SHARD_TOTAL = int(os.getenv("SHARD_TOTAL", "16"))
DATE = os.getenv("DATE", datetime.date.today().isoformat())
API = HfApi(token=HF_TOKEN)

# ── helpers --
def deterministic_shard(key: str) -> int:
    return int(hashlib.sha256(key.encode()).hexdigest(), 16) % SHARD_TOTAL

def cdn_url(path: str) -> str:
    return f"https://huggingface.co/datasets/{REPO_DATASET}/resolve/main/{path}"

def backoff_429(attempt: int):
    wait = 360 * (attempt + 1)
    print(f"[rate-limit] waiting {wait}s", file=sys.stderr)
    time.sleep(wait)

# ── manifest --
def build_manifest() -> list[str]:
    """Single API call; returns filenames under DATE/ (non-recursive)."""
    manifest_path = Path(f"manifest-{DATE}.json")
    if manifest_path.exists():
        return json.loads(manifest_path.read_text())

    for attempt in range(5):
        try:
            tree = API.list_repo_tree(repo_id=REPO_DATASET, path=DATE, recursive=False)
            files = [item.rfilename for item in tree if item.type == "file"]
            manifest_path.write_text(json.dumps(files, indent=2))
            return files
        except Exception as e:
            if "429" in str(e):
                backoff_429(attempt)
                continue
            raise
    raise RuntimeError("Exhausted retries for list_repo_tree")

# ── projection --
def project_to_pair(raw_bytes: bytes, filename: str) -> dict | None:
    """
    Minimal schema projection: try common envelope keys,
    return {prompt, response} or None.
    """
    import pyarrow.parquet as pq
    with tempfile.NamedTemporaryFile(suffix=".parquet", delete=False) as f:
        f.write(raw_bytes)
        f.flush()
        try:
            tbl = pq.read_table(f.name, columns=["prompt", "response"])
        except Exception:
            # fallback: read all and pick first two string cols
            tbl = pq.read_table(f.name)
            cols = [c for c in tbl.column_names if tbl.schema.field(c).type in (str, "string")]
            if len(cols) < 2:
                return None
            tbl = tbl.select([cols[0], cols[1]]).rename_columns(["prompt", "response"])
        df = tbl.to_pandas()
    os.unlink(f.name)

    # take first row only (per-file single example for now)
    row = df.iloc[0]
    return {"prompt": str(row["prompt"]), "response": str(row["response"])}

# ── dedup --
def is_duplicate(md5: str) -> bool:
    from lib.dedup import DedupStore
    store = DedupStore()
    return store.exists(md5)

def register_md5(md5: str):
    from lib.dedup import DedupStore
    store = DedupStore()
    store.add(md5)

# ── upload --
def upload_shard(entries: list[dict], shard_id: int):
    ts = datetime.datetime.utcnow().strftime("%H%M%S")
    filename = f"shard{shard_id}-{ts}.jsonl"
    remote_path = f"batches/public-merged/{DATE}/{filename}"
    content = "\n".join(json.dumps(e, ensure_ascii=False) for e in entries)

    for attempt in range(5):
        try:
            API.upload_file(
                path_or_fileobj=content.encode(),
                path_in_repo=remote_path,
                repo_id=REPO_DATASET,
                commit_message=f"shard{shard_id} {DATE}",
            )
            print(f"[upload] {remote_path}")
            return
        except Exception as e:
            if "429" in str(e):
                backoff_429(attempt)
                continue
            if "128" in str(e) or "commit cap" in str(e).lower():
                print("[commit-cap] sleeping 60s", file=sys.stderr)
                time.sleep(60)
                continue
            raise

# ── main --
def main():
    if not HF_TOKEN:
        print("HF_TOKEN required", file=sys.stderr)
        sys.exit(1)
    login(token=HF_TOKEN)

    files = build_manifest()
    assigned = [f for f in files if deterministic_shard(f) == SHARD_ID]
    print(f"[shard{SHARD_ID}] assigned {len(assigned)} files")

    out = []
    for i, rel in enumerate(assigned):
        for attempt in range(3):
            try:
                raw = requests.get(cdn_url(rel), timeout=30).content
                pair = project_to_pair(raw, rel)
                if not pair:
                    continue
                md5 = hashlib.md5(json.dumps(pair, sort_keys=True).encode()).hexdigest()
                if is_duplicate(md5):
                    continue
                register_md5(md5)
                out.append(pair)
                break
            except Exception as e:
                print(f"[error] {rel}: {e}", file=sys.stderr)
                time.sleep(5 * (attempt + 1))

        if i % 50 == 0:
            print(f"[shard{SHARD_ID}] processed {i+1}/{len(assigned)}")

    if out:
        upload_shard(out, SHARD_ID)
    else:
        print("[shard] no new pairs")

if __name__ == "__main__":
    main()
```
