# workio / design

## 1) Diagnosis

- No idempotency key on `/punches` write path → repeated LINE webhook deliveries or frontend retries create duplicate punch records.
- Missing DB uniqueness constraint for “one open punch per user” → allows multiple concurrent clock-ins or overlapping open punches.
- No deduplication window or idempotency table → retries from LINE or network glitches produce side effects.
- Punch creation lacks tenant-scoped uniqueness (SuperAdmin/Admin boundary) → cross-tenant duplicates possible in shared DB.
- No audit trail for duplicate detection/rejection → operators can’t distinguish retry noise from real double-punch events.

## 2) Proposed change

- **File scope**: `workio/server/src/db/schema.sql` (add constraint + idempotency table), `workio/server/src/routes/punches.ts` (apply idempotency key on POST), `workio/server/src/middleware/idempotency.ts` (new middleware).
- **Line scope**: schema additions (2 tables), route handler update (~20 lines), middleware (~30 lines).

## 3) Implementation

### 3.1 DB schema — `workio/server/src/db/schema.sql`

Add idempotency table and open-punch uniqueness constraint.

```sql
-- Idempotency keys (tenant-scoped)
CREATE TABLE IF NOT EXISTS idempotency_keys (
  id              SERIAL PRIMARY KEY,
  tenant_id       INTEGER NOT NULL,
  key             TEXT NOT NULL,
  route           TEXT NOT NULL,
  request_hash    TEXT NOT NULL,
  response_status INTEGER NOT NULL,
  response_body   JSONB,
  created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  expires_at      TIMESTAMPTZ NOT NULL DEFAULT (NOW() + INTERVAL '24 hours'),
  CONSTRAINT uniq_tenant_key UNIQUE (tenant_id, key)
);

-- Ensure one open punch per user per tenant (clocked-in without clocked-out)
CREATE UNIQUE INDEX IF NOT EXISTS idx_one_open_punch_per_user
  ON punches (tenant_id, user_id)
  WHERE clocked_out_at IS NULL;
```

### 3.2 Idempotency middleware — `workio/server/src/middleware/idempotency.ts`

```ts
import { Request, Response, NextFunction } from 'express';
import { PoolClient } from 'pg';
import { getPool } from '../db';

const IDENTITY_ROUTES = new Set(['POST:/api/punches']);
const TTL_MS = 24 * 60 * 60 * 1000; // 24h

export async function idempotency(
  req: Request,
  res: Response,
  next: NextFunction
) {
  const method = req.method.toUpperCase();
  const routeKey = `${method}:${req.baseUrl || ''}${req.route?.path || req.path}`;
  if (!IDENTITY_ROUTES.has(routeKey)) return next();

  const key = req.header('Idempotency-Key') || req.body?.idempotencyKey;
  if (!key) {
    return res.status(400).json({ error: 'Idempotency-Key required' });
  }

  const tenantId = (req as any).tenant?.id;
  if (!tenantId) {
    return res.status(400).json({ error: 'Tenant context required' });
  }

  const requestHash = require('crypto')
    .createHash('sha256')
    .update(JSON.stringify(req.body))
    .digest('hex');

  const pool = getPool();
  const client: PoolClient = await pool.connect();
  try {
    await client.query('BEGIN');

    // Find existing
    const existing = await client.query(
      `SELECT response_status, response_body, created_at
         FROM idempotency_keys
         WHERE tenant_id = $1 AND key = $2
         FOR UPDATE`,
      [tenantId, key]
    );

    if (existing.rows.length > 0) {
      const row = existing.rows[0];
      // If same payload, replay; otherwise conflict
      if (row.request_hash === requestHash) {
        await client.query('ROLLBACK');
        client.release();
        return res
          .status(row.response_status)
          .json(row.response_body);
      }
      await client.query('ROLLBACK');
      client.release();
      return res.status(409).json({ error: 'Idempotency key collision' });
    }

    // Capture response
    const chunks: any[] = [];
    const originalSend = res.json;
    const originalStatus = res.status;
    let savedStatus = 200;

    (res as any).json = function (body: any) {
      chunks.push(body);
      return this;
    };
    res.status = function (code) {
      savedStatus = code;
      return this as any;
    } as any;

    // Store hook to persist after route handler
    res.on('finish', async () => {
      try {
        const responseBody = chunks.length ? chunks[0] : null;
        await client.query(
          `INSERT INTO idempotency_keys
             (tenant_id, key, route, request_hash, response_status, response_body, expires_at)
           VALUES ($1,$2,$3,$4,$5,$6,$7)`,
          [
            tenantId,
            key,
            routeKey,
            requestHash,
            savedStatus,
            responseBody,
            new Date(Date.now() + TTL_MS),
          ]
        );
        await client.query('COMMIT');
      } catch (e) {
        // best-effort
        await client.query('ROLLBACK').catch(() => {});
      } finally {
        client.release();
      }
    });

    // Attach client to request for route to use same tx if desired
    (req as any)._idemClient = client;
    next();
  } catch (err) {
    await client.query('ROLLBACK').catch(() => {});
    client.release();
    next(err);
  }
}
```

### 3.3 Route update — `workio/server/src/routes/punches.ts`

Apply middleware and enforce tenant/user uniqueness at app level (DB index is safety net).

```ts
import express from 'express';
import { getPool } from '../db';
import { idempotency } from '../middleware/idempotency';

const router = express.Router();

router.post('/', idempotency, async (req, res) => {
  const { userId, type, latitude, longitude, image } = req.body;
  const tenantId = (req as any).tenant.id;

  // Prevent double open punches at app level (fast feedback)
  const pool = getPool();
  const existing = await pool.query(
    `SELECT id FROM punches
      WHERE tenant_id = $1 AND user_id = $2 AND clocked_out_at IS NULL
      LIMIT 1`,
    [tenantId, userId]
  );

  if (type === 'clock_in' && existing.rows.length > 0) {
    return res.status(409).json({ error: 'User already clocked in' });
  }

  const result = await pool.query(
    `INSERT INTO punches (tenant_id, user_id, type, latitude, longitude, image, clocked_in_at)
     VALUES ($1,$2,$3,$4,$5,$6,NOW())
     RETURNING *`,
    [tenantId, userId, type, latitude, longitude, image]
  );

  res.status(201).json(result.rows[0]);
});

export default router;
```
