### 1. CDN Dataset Loader (`surrogate/cdn_dataset.py`)
Single source of truth combining both candidates’ strongest patterns: robust retry, column projection, and manifest caching.

```python
#!/usr/bin/env python3
"""
CDN-only dataset loader for Surrogate AI training.
Bypasses HF API auth/rate limits by using public CDN URLs.
"""
import json
import time
from pathlib import Path
from typing import Dict, List

import pyarrow as pa
import pyarrow.parquet as pq
import requests
from huggingface_hub import list_repo_tree

CDN_BASE = "https://huggingface.co/datasets"
REPO = "surrogate-datasets"
COLS_KEEP = {"prompt", "response"}

class CDNDatasetLoader:
    def __init__(self, cache_dir: Path = Path("surrogate/.cache")):
        self.cache_dir = Path(cache_dir)
        self.cache_dir.mkdir(parents=True, exist_ok=True)

    def _build_manifest(self, date: str) -> List[Dict[str, Any]]:
        """Single HF API call to list files; cache result."""
        cache_file = self.cache_dir / f"manifest_{date}.json"
        if cache_file.exists():
            return json.loads(cache_file.read_text())

        prefix = f"batches/mirror-merged/{date}"
        files = list_repo_tree(repo=REPO, path=prefix, recursive=False)
        manifest = [
            {"path": f.path, "size": f.size, "date": date}
            for f in files
            if f.path.endswith(".parquet")
        ]
        cache_file.write_text(json.dumps(manifest, indent=2))
        return manifest

    def _download_parquet(self, file_info: Dict[str, Any], retries: int = 3) -> pa.Table:
        """Download via CDN (no auth, no API rate limit)."""
        url = f"{CDN_BASE}/{REPO}/resolve/main/{file_info['path']}"
        for attempt in range(1, retries + 1):
            try:
                resp = requests.get(url, timeout=30)
                resp.raise_for_status()
                table = pq.read_table(pa.BufferReader(resp.content))
                # Project to required columns only
                available = set(table.column_names)
                keep = list(COLS_KEEP & available)
                return table.select(keep)
            except Exception as exc:
                if attempt == retries:
                    raise
                time.sleep(2**attempt)
        raise RuntimeError("Unreachable")

    def load_date(self, date: str) -> pa.Table:
        """Load all parquet files for a date via CDN and concatenate."""
        manifest = self._build_manifest(date)
        if not manifest:
            raise ValueError(f"No parquet files found for date={date}")

        tables = [self._download_parquet(f) for f in manifest]
        return pa.concat_tables(tables)
```

### 2. Lightning Studio Reuse + Idle Protection (`surrogate/lightning_utils.py`)
Idempotent studio reuse and safe run wrapper to prevent quota burn.

```python
#!/usr/bin/env python3
"""
Lightning Studio reuse and idle-aware execution.
"""
from lightning import Lightning, Teamspace, Machine
from typing import List

def get_or_create_studio(name: str, machine: Machine = Machine.L40S) -> Lightning:
    """Reuse existing running studio or create new one."""
    lightning = Lightning()
    teamspace = Teamspace()

    running: List[Lightning] = [
        s for s in teamspace.studios
        if s.name == name and getattr(s, "status", "").lower() == "running"
    ]
    if running:
        print(f"Reusing running studio: {name}")
        return running[0]

    print(f"Creating new studio: {name}")
    return lightning.studio(name=name, machine=machine, create_ok=True)

def safe_run_studio(studio: Lightning, script: str, args: List[str]):
    """Ensure studio is running, then execute script."""
    if getattr(studio, "status", "").lower() != "running":
        print(f"Studio stopped; restarting...")
        studio.start(machine=Machine.L40S)

    result = studio.run(script=script, arguments=args, sync=True)
    print(f"Run completed: {result}")
    return result
```

### 3. Training Entrypoint (`surrogate/run_training.py`)
Minimal orchestration that wires CDN loader and studio reuse together.

```python
#!/usr/bin/env python3
"""
Surrogate training launcher with CDN ingestion and studio reuse.
"""
import argparse
from pathlib import Path

from cdn_dataset import CDNDatasetLoader
from lightning_utils import get_or_create_studio, safe_run_studio

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--date", required=True, help="Dataset date (YYYY-MM-DD)")
    parser.add_argument("--epochs", type=int, default=3)
    parser.add_argument("--batch-size", type=int, default=16)
    args = parser.parse_args()

    # 1) Build/verify manifest and test CDN load (fast, local)
    loader = CDNDatasetLoader()
    table = loader.load_date(args.date)
    print(f"Loaded table: {table.num_rows} rows, cols={table.column_names}")

    # 2) Reuse/create Lightning Studio and run training
    studio = get_or_create_studio("surrogate-training-l40s")
    safe_run_studio(
        studio,
        script="train.py",
        args=[
            "--manifest-date", args.date,
            "--epochs", str(args.epochs),
            "--batch-size", str(args.batch_size),
        ],
    )

if __name__ == "__main__":
    main()
```
