# workio / design

## Final Synthesized Answer

### 1. Diagnosis (merged)
- **Non-idempotent LINE webhook ingestion**: duplicate `X-Line-Signature` retries create duplicate punches; no message-level deduplication or idempotency key.
- **Non-atomic punch upsert**: `findOne` → conditional `insert`/`update` across (tenant, employee, date, event) creates race windows under concurrency.
- **Missing tenant-isolation enforcement**: queries must consistently scope by `tenant_id`; missing tenant context at write path risks cross-tenant mutation.
- **No optimistic locking / state-transition safety**: concurrent updates can silently overwrite state changes.
- **No transactional boundaries**: leave/OT request + approval updates need atomic commits to prevent partial writes.
- **Missing unique constraint for natural punch key**: schema should enforce at DB level one active punch per employee per day per event.

// server/src/services/punchService.ts
import { PoolClient } from 'pg';
import { db } from '../db';

export type PunchEvent = 'clock_in' | 'clock_out';

interface CreatePunchOpts {
  tenantId: number;
  employeeId: number;
  event: PunchEvent;
  timestamp: Date;
  location?: { lat: number; lng: number };
  messageId?: string;        // LINE message id for idempotency
  handler?: string;          // e.g. 'line.clock'
}

export async function createPunchAtomic(
  opts: CreatePunchOpts,
  client?: PoolClient
): Promise<{ id: number; created: boolean }> {
  if (!opts.tenantId) {
    throw new Error('tenantId is required for tenant isolation');
  }

  const dbClient = client || db;
  const { tenantId, employeeId, event, timestamp, location, messageId, handler = 'line.clock' } = opts;
  const date = timestamp.toISOString().split('T')[0];

  // Idempotency check (only when messageId provided)
  if (messageId) {
    const exists = await dbClient.query(
      `SELECT 1 FROM webhook_idempotency WHERE tenant_id = $1 AND message_id = $2 AND handler = $3`,
      [tenantId, messageId, handler]
    );
    if (exists.rows.length > 0) {
      // Already processed; return current punch state for caller transparency
      const punch = await dbClient.query(
        `SELECT id FROM punches WHERE tenant_id = $1 AND employee_id = $2 AND date = $3 AND event = $4`,
        [tenantId, employeeId, date, event]
      );
      return { id: punch.rows[0]?.id ?? -1, created: false };
    }
  }

  // Atomic upsert using ON CONFLICT (tenant+employee+date+event)
  const result = await dbClient.query(
    `INSERT INTO punches (tenant_id, employee_id, event, timestamp, location, date)
     VALUES ($1, $2, $3, $4, $5, $6)
     ON CONFLICT (tenant_id, employee_id, date, event)
     DO UPDATE SET timestamp = EXCLUDED.timestamp, location = EXCLUDED.location, updated_at = NOW()
     RETURNING id, (xmax = 0) AS created`,
    [
      tenantId,
      employeeId,
      event,
      timestamp,
      location ? JSON.stringify(location) : null,
      date,
    ]
  );

  const punchId = result.rows[0].id;
  const created = result.rows[0].created === true || result.rows[0].created === 'true';

  // Record idempotency (if messageId provided)
  if (messageId) {
    try {
      await dbClient.query(
        `INSERT INTO webhook_idempotency (tenant_id, message_id, handler) VALUES ($1, $2, $3)`,
        [tenantId, messageId, handler]
      );
    } catch (e) {
      // Unique violation: race handled gracefully; continue
    }
  }

  return { id: punchId, created };
}

// Optimistic-locking update helper (state transitions)
export async function updatePunchWithVersion(
  tenantId: number,
  punchId: number,
  fields: Partial<{ event: PunchEvent; timestamp: Date; location: any }>,
  expectedUpdatedAt: Date,
  client?: PoolClient
) {
  const dbClient = client || db;
  const res = await dbClient.query(
    `UPDATE punches
     SET ${Object.keys(fields)
       .map((k, i) => `${k} = $${i + 1}`)
       .join(', ')}, updated_at = NOW()
     WHERE id = $${Object.keys(fields).length + 1} AND tenant_id = $${Object.keys(fields).length + 2} AND updated_at = $${
      Object.keys(fields).length + 3
    }
     RETURNING *`,
    [...Object.values(fields), punchId, tenantId, expectedUpdatedAt]
  );
  if (res.rowCount === 0) {
    throw new Error('Punch not found or already modified (optimistic lock failure)');
  }
  return res.rows[0];
}

// server/src/repositories/punchRepository.ts
import { db } from '../db';

export async function findPunchForUpdate(
  tenantId: number,
  employeeId: number,
  date: string,
  event: string,
  client = db
) {
  const res = await client.query(
    `SELECT * FROM punches WHERE tenant_id = $1 AND employee_id = $2 AND date = $3 AND event = $4 FOR UPDATE`,
    [tenantId, employeeId, date, event]
  );
  return res.rows[0] || null;
}

export async function getPunchByIdAndTenant(punchId: number, tenantId: number, client = db) {
  const res = await client.query(
    `SELECT * FROM punches WHERE id = $1 AND tenant_id = $2`,
    [punchId, tenantId]
  );
  return res.rows[0] || null;
}
