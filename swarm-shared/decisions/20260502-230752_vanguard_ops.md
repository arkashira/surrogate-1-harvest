"""
Vanguard ops orchestrator:
- Persists run state (date_window, repo, status, last_cursor, file_list_path) to ops/state/runs.json
- CDN-bypass: list_repo_tree once -> file-list.json -> Lightning training uses CDN URLs only
- Reuses running Lightning Studio; restarts if idle-stopped
- Idempotent: safe for cron/retry
"""

import json
import os

from datetime import datetime, timedelta, timezone

    import lightning as L
except ImportError:
    L = None

# ---- Configuration ----
STATE_DIR = Path(__file__).parent / "state"
STATE_FILE = STATE_DIR / "runs.json"
HF_REPO = os.getenv("HF_DATASET_REPO", "datasets/your-org/vanguard-ingest")
HF_TOKEN = os.getenv("HF_TOKEN", "")
LIGHTNING_TEAMSPACE = os.getenv("LIGHTNING_TEAMSPACE", "default")
RUN_NAME = os.getenv("VANGUARD_RUN_NAME", "daily-ingest")
DATE_WINDOW_DAYS = int(os.getenv("VANGUARD_DATE_WINDOW_DAYS", "1"))
# ----

STATE_DIR.mkdir(parents=True, exist_ok=True)

# ---- State helpers ----
def load_state() -> dict:
    if STATE_FILE.exists():
        with open(STATE_FILE) as f:
            return json.load(f)
    return {"runs": {}}

def save_state(state: dict) -> None:
    tmp = STATE_FILE.with_suffix(".tmp")
    with open(tmp, "w") as f:
        json.dump(state, f, indent=2)
    tmp.replace(STATE_FILE)

def get_run() -> dict:
    state = load_state()
    return state["runs"].setdefault(RUN_NAME, {
        "status": "idle",          # idle | listing | listed | training | done | failed
        "last_run_at": None,
        "date_window": None,       # {"start": ISO, "end": ISO}
        "hf_repo": HF_REPO,
        "file_list_path": None,
        "last_cursor": None,
        "lightning_studio_name": f"vanguard-{RUN_NAME}",
    })

def persist_run(update: dict) -> dict:
    state = load_state()
    run = state["runs"].setdefault(RUN_NAME, {})
    run.update(update)
    run["last_run_at"] = datetime.now(timezone.utc).isoformat()
    save_state(state)
    return run

# ---- HF helpers ----
def list_repo_tree_for_date(date_str: str) -> list:
    """
    Single API call to list top-level folder for a date (YYYY-MM-DD).
    Returns list of entries.
    """
    headers = {"Authorization": f"Bearer {HF_TOKEN}"} if HF_TOKEN else {}
    url = f"https://huggingface.co/api/datasets/{HF_REPO}/tree/{date_str}"
    resp = requests.get(url, headers=headers, timeout=30)
    if resp.status_code == 429:
        retry_after = int(resp.headers.get("Retry-After", 30))
        print(f"HF 429 rate-limited. Waiting {retry_after}s")
        time.sleep(retry_after)
        return list_repo_tree_for_date(date_str)
    resp.raise_for_status()
    return resp.json()

def build_file_list(date_str: str, out_path: Path) -> list:
    """
    List once and save file list for CDN-bypass training.
    Returns list of file entries.
    """
    entries = list_repo_tree_for_date(date_str)
    files = [e for e in entries if e.get("type") == "file"]
    out_path.parent.mkdir(parents=True, exist_ok=True)
    with open(out_path, "w") as f:
        json.dump(files, f, indent=2)
    print(f"Saved {len(files)} file entries to {out_path}")
    return files

def cdn_url_for_file(path: str) -> str:
    """
    CDN-bypass URL (no Authorization header required).
    """
    return f"https://huggingface.co/datasets/{HF_REPO}/resolve/main/{path}"

# ---- Lightning Studio helpers ----
def ensure_lightning_studio():
    if L is None:
        print("Lightning not installed; skipping studio reuse.")
        return None

    try:
        teamspace = L.Teamspace(name=LIGHTNING_TEAMSPACE, create_ok=True)
        run = get_run()
        studio_name = run["lightning_studio_name"]

        for s in teamspace.studios():
            if s.name == studio_name and s.status == "Running":
                print(f"Reusing running studio: {studio_name}")
                return s

        print(f"No running studio found with name {studio_name}. Creating...")
        machine = L.Machine.L40S
        studio = teamspace.create_studio(
            name=studio_name,
            machine=machine,
            create_ok=True,
        )
        return studio
    except Exception as e:
        print(f"Lightning studio setup failed (non-fatal for ingestion): {e}")
        return None
