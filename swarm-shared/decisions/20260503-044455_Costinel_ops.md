# Costinel / ops

## Implementation Plan: CDN-First Top-Hub Signal Panel (<2h)

**Highest-value improvement**: Embed a resilient “Top Hub” panel into Costinel’s dashboard that surfaces the most-connected hub (e.g., “MOC”) and related docs using **CDN-only fetches**, zero model compute, offline-first caching, and zero HF API calls during runtime.

### Why this ships fast (<2h)
- No backend changes; pure frontend + static JSON.
- CDN bypass avoids HF rate limits entirely.
- Reuses existing `knowledge-rag`/`top-hub` patterns.
- Works offline (local cache fallback).

### 2. Pre-generation script (run once on Mac)

`scripts/generate-top-hub.js`

```js
#!/usr/bin/env node
/**
 * Generate top-hub.json from HF dataset using CDN-only strategy.
 * Run on Mac (or CI) after rate-limit window clears.
 *
 * HF API calls: 1 list_repo_tree → CDN-only thereafter.
 */
import { HfApi } from "@huggingface/hub";
import fs from "fs";
import path from "path";

const HUB_DATASET = "axentx/knowledge-rag"; // adjust if different
const OUT_PATH = path.resolve("public/knowledge/top-hub.json");

async function main() {
  const api = new HfApi();

  // 1) Single API call: list top-level folder (non-recursive to avoid pagination)
  const tree = await api.listRepoTree({
    repo: HUB_DATASET,
    path: "top-hub",
    recursive: false,
  });

  // 2) Build CDN URLs (no auth, no API calls during runtime)
  const files = (tree || [])
    .filter((f) => f.type === "file" && f.path.endsWith(".json"))
    .map((f) => ({
      name: path.basename(f.path, ".json"),
      cdn: `https://huggingface.co/datasets/${HUB_DATASET}/resolve/main/${f.path}`,
    }));

  // 3) Fetch each file via CDN (high-rate, no auth)
  const hubs = [];
  for (const file of files) {
    const res = await fetch(file.cdn);
    if (!res.ok) continue;
    const data = await res.json();
    hubs.push({
      slug: file.name,
      title: data.title || file.name,
      summary: data.summary || "",
      connections: data.connections || 0,
      docs: (data.docs || []).slice(0, 5),
      updated: data.updated || null,
    });
  }

  // 4) Pick most-connected hub
  hubs.sort((a, b) => b.connections - a.connections);
  const top = hubs[0] || null;

  const payload = {
    generatedAt: new Date().toISOString(),
    sourceDataset: HUB_DATASET,
    topHub: top,
    allHubs: hubs,
  };

  fs.mkdirSync(path.dirname(OUT_PATH), { recursive: true });
  fs.writeFileSync(OUT_PATH, JSON.stringify(payload, null, 2));
  console.log("✅ top-hub.json generated:", OUT_PATH);
}

main().catch((err) => {
  console.error(err);
  process.exit(1);
});
```

Make executable and run once:

```bash
chmod +x scripts/generate-top-hub.js
node scripts/generate-top-hub.js
```

Commit `public/knowledge/top-hub.json`.

### 3. Frontend component

`src/components/TopHubPanel.tsx`

```tsx
import { useEffect, useState } from "react";
import "./TopHubPanel.css";

interface Doc {
  title: string;
  summary: string;
  url?: string;
}

interface Hub {
  slug: string;
  title: string;
  summary: string;
  connections: number;
  docs: Doc[];
  updated: string | null;
}

interface TopHubPayload {
  generatedAt: string;
  topHub: Hub | null;
  allHubs: Hub[];
}

export function TopHubPanel() {
  const [hub, setHub] = useState<Hub | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    let mounted = true;

    async function load() {
      try {
        // CDN-only fetch; falls back to bundled JSON if deployed statically
        const res = await fetch("/knowledge/top-hub.json", {
          cache: "no-cache",
        });
        if (!res.ok) throw new Error("Network response not ok");
        const data: TopHubPayload = await res.json();
        if (mounted && data.topHub) setHub(data.topHub);
      } catch (err) {
        if (mounted) setError(String(err));
      } finally {
        if (mounted) setLoading(false);
      }
    }

    load();
    return () => {
      mounted = false;
    };
  }, []);

  if (loading) return <div className="top-hub-panel loading">Loading insights…</div>;
  if (error || !hub) return null; // silent fail (graceful degradation)

  return (
    <div className="top-hub-panel" role="region" aria-label="Top hub insights">
      <header>
        <h3>{hub.title}</h3>
        <span className="badge">{hub.connections} connections</span>
      </header>
      <p className="summary">{hub.summary}</p>

      {hub.docs.length > 0 && (
        <section className="related-docs">
          <h4>Key docs</h4>
          <ul>
            {hub.docs.map((doc, i) => (
              <li key={i}>
                <strong>{doc.title}</strong>
                <p>{doc.summary}</p>
                {doc.url && (
                  <a href={doc.url} target="_blank" rel="noopener noreferrer">
                    Read more →
                  </a>
                )}
              </li>
            ))}
          </ul>
        </section>
      )}

      <footer className="meta">
        <small>Updated {hub.updated ? new Date(hub.updated).toLocaleDateString() : "—"}</small>
      </footer>
    </div>
  );
}
```
