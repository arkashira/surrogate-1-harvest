# vanguard / design

## Final Design-layer Fix (CDN-first, reproducible, 429-safe)

**Goal**: Eliminate runtime HF API calls during dataset selection, preview, and training; guarantee reproducible, CDN-only fetches with a single source of truth.

---

## 1. Diagnosis (merged)

- Frontend triggers runtime HF API calls (`list_repo_tree`, `load_dataset`) during dataset selection/preview → 429 risk and non-reproducible runs.
- No deterministic, content-addressed manifest mapping `{date}/{slug}` → file paths; training jobs re-enumerate and can’t guarantee CDN-only fetches.
- Backend ingestion/training can still hit API during data load (no pre-listed file manifest embedded in training scripts).
- Missing single source of truth for available datasets across frontend + training; each layer independently queries HF.
- No CDN-first strategy baked into design: training scripts should accept a static file list and fetch exclusively via `https://huggingface.co/datasets/{repo}/resolve/main/{path}` with zero Authorization.

---

## 2. Proposed change (merged)

Add a design-layer manifest generator and CDN-first loader contract:

- `vanguard/design/manifest.py` (new) — deterministic file-lister + manifest serializer.
- `vanguard/design/cdn_loader.py` (new) — CDN-only dataset loader that consumes a manifest.
- `vanguard/design/train_contract.md` (new) — short contract for training scripts (inputs, env, CDN-only rule).
- `vanguard/design/__init__.py` (create/update) — expose stable imports.

Scope: pure Python modules under `design/`; no runtime servers, no UI code. This is the design-layer fix that enables reproducible, CDN-only training and protects against 429s.

#!/usr/bin/env python3
"""
Deterministic HF manifest generator.
Produces content-addressed manifest: {date}/{slug} -> repo-relative path.
Rule: one API call per date folder (non-recursive), then CDN-only thereafter.
"""

from __future__ import annotations

import json
import hashlib
import os
from datetime import datetime, timezone
from pathlib import Path
from typing import Dict, List, Any

try:
    from huggingface_hub import list_repo_tree
except ImportError:
    list_repo_tree = None  # allow import without hf_hub in design layer

MANIFEST_NAME = "files.json"

def deterministic_slug(file_path: str) -> str:
    """Content-addressed slug derived from path (stable across runs)."""
    return hashlib.sha256(file_path.encode("utf-8")).hexdigest()[:16]

def build_manifest_for_date(
    repo_id: str,
    date_folder: str,
    revision: str = "main",
    extensions: tuple[str, ...] = (".parquet", ".jsonl", ".json"),
) -> Dict[str, Any]:
    """
    Single API call per date folder (non-recursive) to list immediate children.
    Returns manifest fragment keyed by deterministic slug.
    """
    if list_repo_tree is None:
        raise RuntimeError("huggingface_hub required to build manifest")

    # list_repo_tree(path, recursive=False) avoids heavy recursive pagination
    tree = list_repo_tree(repo_id=repo_id, path=date_folder, revision=revision, recursive=False)

    entries = []
    for item in tree:
        if item.rfilename.lower().endswith(extensions):
            full_path = f"{date_folder}/{item.rfilename}" if date_folder else item.rfilename
            slug = deterministic_slug(full_path)
            entries.append(
                {
                    "slug": slug,
                    "path": full_path,
                    "date": date_folder,
                    "cdn_url": f"https://huggingface.co/datasets/{repo_id}/resolve/main/{full_path}",
                }
            )

    # Deterministic ordering for stable manifests
    entries.sort(key=lambda e: (e["date"], e["path"]))

    manifest = {
        "repo_id": repo_id,
        "revision": revision,
        "generated_at_utc": datetime.now(timezone.utc).isoformat(),
        "date_folder": date_folder,
        "entries": entries,
    }
    return manifest

def save_manifest(manifest: Dict[str, Any], out_dir: Path) -> Path:
    out_dir = Path(out_dir)
    out_dir.mkdir(parents=True, exist_ok=True)
    out_path = out_dir / MANIFEST_NAME
    with open(out_path, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2, sort_keys=True)
    return out_path

def load_manifest(manifest_path: Path) -> Dict[str, Any]:
    with open(manifest_path, "r", encoding="utf-8") as f:
        return json.load(f)

if __name__ == "__main__":
    # Example usage (run from Mac orchestration only):
    # python -m vanguard.design.manifest
    import argparse

    parser = argparse.ArgumentParser(description="Build HF manifest for a date folder")
    parser.add_argument("--repo", required=True, help="HF dataset repo id")
    parser.add_argument("--date", required=True, help="Date folder (e.g. 2026-04-29)")
    parser.add_argument("--out", default=".", help="Output directory")
    parser.add_argument("--rev", default="main", help="Revision")
    args = parser.parse_args()

    manifest = build_manifest_for_date(args.repo, args.date, revision=args.rev)
    p = save_manifest(manifest, Path(args.out))
    print(f"Manifest saved to {p}")
```

#!/usr/bin/env python3
"""
CDN-first dataset loader contract.
Training scripts must accept a manifest file and use ONLY CDN URLs.
No Authorization headers; no HF API calls during data loading.
"""

from __future__ import annotations

import json
from pathlib import Path
from typing import Dict, Any, Iterator, List
import pyarrow.parquet as pq
import pyarrow as pa

class CDNLoader:
    """
    Loads files listed in a manifest via CDN URLs only.
    Usage:
        loader = CDNLoader(manifest_path)
        for record in loader.iter_parquet(projection=["prompt", "response"]):
            ...
    """

    def __init__(self, manifest_path: Path | str):
        self.manifest_path = Path(manifest_path)
        self.manifest = self._load()

    def _load(self) -> Dict[str, Any]:
        with open(self.manifest_path, "r", encoding="utf-8") as f:
            return json.load(f)

    @property
    def entries(self) -> List[Dict[str, Any]]:
        return self.manifest.get("entries", [])

    def iter_parquet(
        self,
        columns: List[str] | None = None,
        batch_size: int = 1000,
    ) -> Iterator[pa.Table]:
        """
        Stream parquet files via CDN URLs.
        columns: if provided, project only these columns at read time.
        """
        for ent in self.entries:
            url = ent["cdn_url"]
            if not url.endswith(".parquet"):
                continue
            try:
                table = pq.read_table(url, columns=columns)
                yield table
            except Exception as exc:
                # Log and continue to avoid breaking entire pipeline on single file
                # In production, route to structured logger.
                print(f"Failed to load {url}: {exc}")
                continue

    def file_urls(self) -> List[str]:
        return [e["cdn_url"] for e in self.entries if e["cdn_url"].endswith(".parquet")]
```
