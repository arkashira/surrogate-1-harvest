# airship / design

## Final Synthesized Plan (≤2h)

**Goal**: Eliminate HF API 429s during Surrogate training by switching to a **CDN-only ingestion pipeline** with **pre-listed file paths** and **deterministic sibling-repo sharding** for writes.  
**Impact**: Removes API-rate-limit failures during training; reduces Lightning quota churn by reusing running studios; stabilizes surrogate-1 ingestion.

## Code Snippets

### 1. `tools/ingest/list_repo_paths.py`
```python
#!/usr/bin/env python3
import json
import argparse
from pathlib import Path
from huggingface_hub import HfApi

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True, help="e.g. axentx/surrogate-mirror")
    parser.add_argument("--path", required=True, help="Folder path in repo, e.g. batches/mirror-merged/2026-05-01")
    parser.add_argument("--out", required=True, help="Output JSON file")
    args = parser.parse_args()

    api = HfApi()
    entries = api.list_repo_tree(repo_id=args.repo, path=args.path, recursive=False)
    files = sorted(e.rfilename for e in entries if e.rfilename.endswith((".jsonl", ".parquet")))

    manifest = {
        "repo": args.repo,
        "path": args.path,
        "files": files,
        "cdn_base": f"https://huggingface.co/datasets/{args.repo}/resolve/main",
    }
    Path(args.out).write_text(json.dumps(manifest, indent=2))
    print(f"Wrote {len(files)} files to {args.out}")

if __name__ == "__main__":
    main()
```

### 2. `lib/sharding.py`
```python
import hashlib
from typing import Sequence

SIBLING_REPOS = [
    "axentx/surrogate-mirror",
    "axentx/surrogate-mirror-s1",
    "axentx/surrogate-mirror-s2",
    "axentx/surrogate-mirror-s3",
    "axentx/surrogate-mirror-s4",
    "axentx/surrogate-mirror-s5",
]

def shard_repo_for_slug(slug: str, repos: Sequence[str] = SIBLING_REPOS) -> str:
    """Deterministic repo selection by slug hash."""
    digest = int(hashlib.sha256(slug.encode()).hexdigest(), 16)
    return repos[digest % len(repos)]
```

### 3. Dataset-mirror writer (project schema + sharding)
```python
import os
import pyarrow as pa
import pyarrow.parquet as pq
from lib.sharding import shard_repo_for_slug

def project_and_write(batch, date: str, slug: str, hf_api):
    # Keep only {prompt, response}; drop source/ts/other mixed cols
    table = pa.Table.from_pydict({
        "prompt": batch["prompt"],
        "response": batch["response"],
    })
    out_dir = f"batches/mirror-merged/{date}"
    os.makedirs(out_dir, exist_ok=True)
    out_path = f"{out_dir}/{slug}.parquet"
    pq.write_table(table, out_path)

    # Deterministic sharding for HF commit cap (128/hr/repo)
    target_repo = shard_repo_for_slug(slug)
    hf_api.upload_file(
        path_or_fileobj=out_path,
        path_in_repo=f"{out_dir}/{slug}.parquet",
        repo_id=target_repo,
    )
```

### 4. `training/loader_cdn.py`
```python
import json
import requests
import pyarrow.parquet as pq
from torch.utils.data import IterableDataset

class CDNParquetDataset(IterableDataset):
    def __init__(self, manifest_path="manifest.json"):
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.cdn_base = self.manifest["cdn_base"]
        self.files = self.manifest["files"]

    def __iter__(self):
        for f in self.files:
            url = f"{self.cdn_base}/{f}"
            # CDN download: no Authorization header, bypasses API rate-limit
            r = requests.get(url, timeout=30)
            r.raise_for_status()
            with open("/tmp/current.parquet", "wb") as tmp:
                tmp.write(r.content)
            table = pq.read_table("/tmp/current.parquet")
            for row in table.to_pydict():
                yield row["prompt"], row["response"]
```
