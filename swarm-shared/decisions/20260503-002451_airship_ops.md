**Replace any remaining recursive HF ingestion / `load_dataset(streaming=True)` in surrogate training with a manifest-driven, CDN-only iterable loader** — generated once on the Mac orchestrator and embedded in Lightning training to eliminate API rate limits and enforce strict `{prompt,response}` projection.

## Implementation plan (≤2h)

1. **Add manifest generator** (`scripts/build_hf_manifest.py`)  
   - Runs on Mac orchestrator after rate-limit window clears.  
   - Uses `list_repo_tree(path, recursive=False)` per date folder.  
   - Emits `manifest/{date}/files.json` with CDN URLs and metadata.  
   - Skips recursive listing; one API call per folder.

2. **Add CDN-only iterable dataset** (`surrogate/data/hf_cdn_dataset.py`)  
   - Accepts manifest path.  
   - Streams files via `https://huggingface.co/datasets/{repo}/resolve/main/{path}` (no auth).  
   - Projects to `{prompt, response}` only at parse time.  
   - Supports `IterableDataset` for Lightning.

3. **Update training entrypoint** (`surrogate/train.py`)  
   - Accept `--manifest` arg.  
   - Use `hf_cdn_dataset` instead of `load_dataset`.  
   - Reuse running Studio; restart if idle-stopped.

4. **Add small wrapper script** (`scripts/run_training.sh`)  
   - Proper Bash shebang, `chmod +x`.  
   - Sets `SHELL=/bin/bash`.  
   - Invokes training with manifest.

5. **Verify & commit**  
   - Quick smoke test with one date folder.  
   - Ensure no `load_dataset(streaming=True)` remains.

### 1) Manifest generator (Mac orchestrator)

```python
# scripts/build_hf_manifest.py
#!/usr/bin/env python3
import json, os, sys
from huggingface_hub import HfApi

API = HfApi()
REPO = sys.argv[1]                      # e.g. "axentx/surrogate-data"
OUT_DIR = sys.argv[2]                   # e.g. "manifest/2026-05-03"
FOLDER = sys.argv[3] if len(sys.argv) > 3 else ""  # optional subfolder

os.makedirs(OUT_DIR, exist_ok=True)

# Non-recursive per folder to avoid 100x pagination
items = API.list_repo_tree(repo_id=REPO, path=FOLDER, recursive=False)
files = []
for item in items:
    if item.type != "file":
        continue
    cdn_url = f"https://huggingface.co/datasets/{REPO}/resolve/main/{item.path}"
    files.append({
        "path": item.path,
        "cdn_url": cdn_url,
        "size": getattr(item, "size", None)
    })

out_path = os.path.join(OUT_DIR, "files.json")
with open(out_path, "w") as f:
    json.dump(files, f, indent=2)

print(f"Wrote {len(files)} files -> {out_path}")
```

Make executable:

```bash
chmod +x scripts/build_hf_manifest.py
```

### 2) CDN-only iterable dataset

```python
# surrogate/data/hf_cdn_dataset.py
import json, io, gzip
from typing import Optional, Dict, Any
from torch.utils.data import IterableDataset
import requests
import pyarrow as pa
import pyarrow.parquet as pq

class HFCDNDataset(IterableDataset):
    def __init__(self, manifest_path: str, repo: str, start: Optional[int] = None, end: Optional[int] = None):
        with open(manifest_path) as f:
            self.files = json.load(f)
        if start is not None or end is not None:
            self.files = self.files[start:end]
        self.repo = repo

    def _project_record(self, raw: Dict[str, Any]) -> Dict[str, str]:
        # Strict projection to {prompt, response} at parse time
        return {
            "prompt": str(raw.get("prompt", raw.get("input", ""))),
            "response": str(raw.get("response", raw.get("output", "")))
        }

    def _stream_file(self, url: str):
        resp = requests.get(url, timeout=30)
        resp.raise_for_status()
        content = io.BytesIO(resp.content)
        # Try parquet first (preferred), fallback to json/jsonl
        try:
            table = pq.read_table(content)
            for batch in table.to_batches(max_chunksize=1000):
                cols = batch.to_pydict()
                rows = [dict(zip(cols, t)) for t in zip(*cols.values())]
                for row in rows:
                    yield self._project_record(row)
            return
        except Exception:
            pass

        # Try json lines
        text = content.getvalue().decode("utf-8", errors="replace")
        lines = [ln.strip() for ln in text.splitlines() if ln.strip()]
        for ln in lines:
            try:
                row = json.loads(ln)
                yield self._project_record(row)
            except Exception:
                continue

    def __iter__(self):
        for entry in self.files:
            url = entry["cdn_url"]
            try:
                yield from self._stream_file(url)
            except Exception as exc:
                # Log and continue; don't kill entire epoch
                print(f"Skipping {url}: {exc}")
                continue
```

### 3) Training entrypoint update

```python
# surrogate/train.py
import argparse, os
from pathlib import Path
from torch.utils.data import DataLoader
from lightning.pytorch import Trainer
from lightning.pytorch.utilities import rank_zero_only
from surrogate.data.hf_cdn_dataset import HFCDNDataset
from surrogate.model import SurrogateModel  # existing

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--manifest", required=True, help="Path to manifest/files.json")
    parser.add_argument("--repo", default="axentx/surrogate-data")
    parser.add_argument("--batch-size", type=int, default=8)
    parser.add_argument("--max-steps", type=int, default=1000)
    args = parser.parse_args()

    dataset = HFCDNDataset(manifest_path=args.manifest, repo=args.repo)
    loader = DataLoader(dataset, batch_size=args.batch_size, num_workers=0)

    model = SurrogateModel()

    # Reuse running studio; restart if idle-stopped
    try:
        from lightning.pytorch import Studio
        from lightning.pytorch import Teamspace
        studios = Teamspace.studios
        running = [s for s in studios if s.name == "surrogate-train" and s.status == "Running"]
        if running:
            studio = running[0]
        else:
            studio = Studio.create(name="surrogate-train", machine="l40s", create_ok=True)
    except Exception:
        studio = None

    trainer = Trainer(
        max_steps=args.max_steps,
        accelerator="gpu" if torch.cuda.is_available() else "cpu",
        devices=1,
        enable_checkpointing=True,
    )

    @rank_zero_only
    def fit():
        trainer.fit(model, loader)

    if studio:
        studio.run(target=fit)
    else:
        fit()

if __name__ == "__main__":
    main()
```

```bash
python train.py --manifest "$MANIFEST_PATH" --repo "$REPO" --batch-size 8 --max-steps 2000
```
