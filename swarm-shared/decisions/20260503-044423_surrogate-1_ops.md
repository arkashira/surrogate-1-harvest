## Final Implementation Plan (≤2h)

**Highest-value improvement**: Replace fragile shell-based ingestion with a **manifest-driven, CDN-bypass Python worker** that eliminates HF API rate limits during training data loads and prevents mixed-schema CastErrors.

### Changes

1. **Add `bin/worker.py`**  
   - Single-file worker that:
     - Accepts `SHARD_ID`/`TOTAL_SHARDS` env vars (matrix).
     - Calls HF API **once** (after rate-limit window) to `list_repo_tree` for today’s folder; saves `manifest.json`.
     - Uses **CDN-only** downloads (`resolve/main/...`) for all file fetches (zero API calls during training).
     - Projects each file to `{prompt, response}` at parse time (avoids `load_dataset` mixed-schema CastError).
     - Deduplicates via `lib/dedup.py` (md5 store).
     - Writes `shard-<N>-<HHMMSS>.jsonl` to `batches/public-merged/<date>/`.

2. **Update `bin/dataset-enrich.sh`**  
   - Thin wrapper that sets `SHELL=/bin/bash`, validates env, then invokes `python3 bin/worker.py "$@"`.
   - Ensures executable bit and proper shebang hygiene (learned from opus/active-learning cron failures).

3. **Update `.github/workflows/ingest.yml`**  
   - Keep 16-shard matrix.
   - Add step before matrix to generate `manifest.json` on a **single runner** (or reuse cached from earlier run) and upload as artifact for other shards (optional optimization).
   - Ensure `HF_TOKEN` is available and `python -m pip install -r requirements.txt`.

4. **Update `requirements.txt`**  
   - Add `requests` if not present (for raw CDN fetches).

#!/usr/bin/env python3
"""
CDN-bypass ingestion worker for surrogate-1.
- SHARD_ID / TOTAL_SHARDS: matrix shard selection
- HF_TOKEN: write permission to axentx/surrogate-1-training-pairs
"""
import os
import sys
import json
import hashlib
import datetime
import pathlib
from typing import List, Dict, Any

import requests
from huggingface_hub import HfApi, login

REPO = "axentx/surrogate-1-training-pairs"
DATE = datetime.datetime.utcnow().strftime("%Y-%m-%d")
API = HfApi()

def hf_token() -> str:
    token = os.getenv("HF_TOKEN")
    if not token:
        print("HF_TOKEN missing", file=sys.stderr)
        sys.exit(1)
    return token

def today_folder() -> str:
    return f"batches/public-merged/{DATE}"

def list_today_files() -> List[str]:
    """Single API call to list files in today's folder (non-recursive)."""
    items = API.list_repo_tree(repo_id=REPO, path=today_folder(), recursive=False)
    names = []
    for it in items:
        name = it["path"] if isinstance(it, dict) else getattr(it, "path", str(it))
        if name and not name.endswith("/"):
            names.append(name)
    return names

def download_via_cdn(repo: str, path: str, dest: pathlib.Path) -> pathlib.Path:
    """CDN-only download (no Authorization header)."""
    url = f"https://huggingface.co/datasets/{repo}/resolve/main/{path}"
    dest.parent.mkdir(parents=True, exist_ok=True)
    r = requests.get(url, timeout=60)
    r.raise_for_status()
    dest.write_bytes(r.content)
    return dest

def project_to_pair(local_path: pathlib.Path) -> List[Dict[str, str]]:
    """
    Best-effort projection to {prompt,response} pairs.
    Supports .jsonl, .parquet, .json.
    """
    suffix = local_path.suffix.lower()
    pairs = []

    if suffix == ".parquet":
        import pyarrow.parquet as pq
        tbl = pq.read_table(local_path)
        cols = tbl.column_names
        prompt_col = next((c for c in ["prompt", "input", "question"] if c in cols), None)
        response_col = next((c for c in ["response", "output", "answer"] if c in cols), None)
        if prompt_col and response_col:
            pdf = tbl.to_pandas()
            for _, row in pdf.iterrows():
                pairs.append({"prompt": str(row[prompt_col]), "response": str(row[response_col])})
    else:
        # JSONL or JSON
        text = local_path.read_text(encoding="utf-8")
        if suffix == ".jsonl":
            lines = [ln.strip() for ln in text.splitlines() if ln.strip()]
            items = [json.loads(ln) for ln in lines]
        else:
            items = json.loads(text)
            if not isinstance(items, list):
                items = [items]

        for it in items:
            prompt = it.get("prompt") or it.get("input") or it.get("question")
            response = it.get("response") or it.get("output") or it.get("answer")
            if prompt and response:
                pairs.append({"prompt": str(prompt), "response": str(response)})
    return pairs

def md5_hex(data: str) -> str:
    return hashlib.md5(data.encode("utf-8")).hexdigest()

def dedup_and_upload(pairs: List[Dict[str, str]], shard_id: int) -> int:
    from lib.dedup import DedupStore  # local

    store = DedupStore()
    out_dir = pathlib.Path("out") / today_folder()
    out_dir.mkdir(parents=True, exist_ok=True)
    ts = datetime.datetime.utcnow().strftime("%H%M%S")
    out_path = out_dir / f"shard-{shard_id}-{ts}.jsonl"

    written = 0
    with out_path.open("w", encoding="utf-8") as f:
        for p in pairs:
            line = json.dumps(p, ensure_ascii=False)
            h = md5_hex(line)
            if not store.seen(h):
                store.add(h)
                f.write(line + "\n")
                written += 1

    # Upload via huggingface_hub (single commit per shard)
    if written > 0:
        API.upload_file(
            path_or_fileobj=str(out_path),
            path_in_repo=f"{today_folder()}/shard-{shard_id}-{ts}.jsonl",
            repo_id=REPO,
            repo_type="dataset",
            token=hf_token(),
        )
        print(f"Uploaded {written} pairs to {today_folder()}/shard-{shard_id}-{ts}.jsonl")
    else:
        print("No new pairs to upload")
    return written

def main() -> None:
    shard_id = int(os.getenv("SHARD_ID", "0"))
    total_shards = int(os.getenv("TOTAL_SHARDS", "16"))
    login(token=hf_token())

    # 1) List files once (single API call)
    files = list_today_files()
    if not files:
        print("No files for today")
        return

    # Optional: save manifest for reproducibility
    manifest_path = pathlib.Path("manifest.json")
    manifest_path.write_text(json.dumps({"date": DATE, "files": files}, indent=2))

    # 2) Select shard slice
    my_files = [f for i, f in enumerate(files) if hash(f) % total_shards == shard_id]
    print(f"Shard {shard_id}/{total_shards} processing {len(my_files)} files")

    # 3) CDN download + project + dedup + upload
    all_pairs = []
    for rel in my_files:
        tmp = pathlib.Path("tmp") / rel
        try:
            download_via_cdn(REPO, rel, tmp)
            pairs = project_to_pair(tmp)
            all_pairs.extend(pairs)
        except Exception as e:
            print(f"Failed {rel}: {e}", file=sys.stderr)
        finally:
            if tmp.exists():
                tmp.unlink(missing_ok=True)

    uploaded = dedup_and_upload(all_pairs, shard_id)
    print(f"Done. Uploaded {uploaded} new pairs")
