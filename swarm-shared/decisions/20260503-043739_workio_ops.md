# workio / ops

Below is the single, consolidated implementation that keeps the strongest, most actionable parts from both proposals and resolves contradictions in favor of correctness and production-grade reliability.

---

## 1. Diagnosis (consolidated)
- **No idempotency on LINE webhook ingestion** — LINE retries on 5xx/timeouts; duplicate clock-in/out, leave, or OT records can be created.
- **Missing `X-Line-Signature` verification** — allows spoofed events and attendance fraud.
- **No atomic upsert for clock events** — concurrent/retried webhooks can create duplicate punch rows (race on `INSERT`).
- **No replay protection window** — same event (`timestamp` + `source.userId` + `type`) can be re-ingested hours/days later.
- **No structured logging of delivery metadata** — makes debugging duplicates or signature failures difficult.

---

## 2. Proposed change (single scope)
File: `/opt/axentx/workio/server/src/routes/webhook/line.ts`  
Scope:
- Add `X-Line-Signature` HMAC-SHA256 verification middleware (reject invalid requests early).
- Add idempotency table `line_webhook_events` with a unique constraint on (`body_hash`, `source_user_id`, `event_type`, `event_timestamp`) to enforce replay protection.
- Wrap event handling in atomic DB transactions with per-event upsert for clock events and de-dup checks for leave/OT.
- Return `200` only after successful DB commit; return `400/401/500` appropriately so LINE retry semantics are honored.
- Capture and log raw body and key metadata for observability.

CREATE TABLE IF NOT EXISTS line_webhook_events (
  id              BIGSERIAL PRIMARY KEY,
  body_hash       TEXT NOT NULL,
  source_user_id  TEXT NOT NULL,
  event_type      TEXT NOT NULL,
  event_timestamp BIGINT NOT NULL,
  payload         JSONB NOT NULL,
  processed_at    TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  CONSTRAINT uq_line_event UNIQUE (body_hash, source_user_id, event_type, event_timestamp)

CREATE INDEX IF NOT EXISTS idx_line_webhook_events_lookup
  ON line_webhook_events (source_user_id, event_type, event_timestamp);

### 3.2 Webhook handler
`/opt/axentx/workio/server/src/routes/webhook/line.ts`
```ts
import express, { Request, Response, NextFunction } from 'express';
import crypto from 'crypto';
import { Pool } from 'pg';

const router = express.Router();
const pool = new Pool({ connectionString: process.env.DATABASE_URL });

const CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';

function verifyLineSignature(rawBody: string, signature: string): boolean {
  if (!CHANNEL_SECRET) return false;
  const expected = crypto
    .createHmac('sha256', CHANNEL_SECRET)
    .update(rawBody, 'utf8')
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

function hashBody(body: string): string {
  return crypto.createHash('sha256').update(body, 'utf8').digest('hex');
}

// Middleware to capture raw body for signature verification
function rawBodySaver(req: Request, _res: Response, buf: Buffer, encoding: BufferEncoding) {
  (req as any).rawBody = buf.toString(encoding || 'utf8');
}

router.post(
  '/line',
  express.json({ verify: rawBodySaver }),
  async (req: Request, res: Response, next: NextFunction) => {
    const signature = req.headers['x-line-signature'] as string;
    const rawBody = (req as any).rawBody as string;

    if (!signature || !verifyLineSignature(rawBody, signature)) {
      console.warn('[webhook/line] Invalid signature', {
        hasSignature: !!signature,
        bodyHash: hashBody(rawBody),
      });
      return res.status(401).send('Invalid signature');
    }

    const events = req.body.events;
    if (!Array.isArray(events) || events.length === 0) {
      return res.status(400).send('No events');
    }

    const client = await pool.connect();
    try {
      await client.query('BEGIN');

      for (const ev of events) {
        const { type, source, timestamp } = ev;
        const userId = source?.userId;
        const bodyHash = hashBody(rawBody);

        if (!userId || !type || typeof timestamp !== 'number') {
          console.warn('[webhook/line] Malformed event', ev);
          continue;
        }

        // Idempotency guard
        const idempotencyCheck = await client.query(
          `SELECT 1 FROM line_webhook_events
           WHERE body_hash = $1 AND source_user_id = $2 AND event_type = $3 AND event_timestamp = $4`,
          [bodyHash, userId, type, timestamp]
        );

        if (idempotencyCheck.rowCount && idempotencyCheck.rowCount > 0) {
          console.info('[webhook/line] Duplicate event skipped', { userId, type, timestamp });
          continue;
        }

        // Record ingestion
        await client.query(
          `INSERT INTO line_webhook_events (body_hash, source_user_id, event_type, event_timestamp, payload)
           VALUES ($1, $2, $3, $4, $5)`,
          [bodyHash, userId, type, timestamp, ev]
        );

        // Domain handling: clock in/out upsert (one punch per user per date)
        if (type === 'message' && ev.message?.text) {
          const text = ev.message.text.trim().toLowerCase();
          if (text === 'clock in' || text === 'clock out') {
            const isClockIn = text === 'clock in';
            const today = new Date(timestamp * 1000).toISOString().split('T')[0];

            await client.query(
              `INSERT INTO attendance (user_id, punch_date, clock_in_at, clock_out_at, updated_at)
               VALUES ($1, $2, $3, $4, NOW())
               ON CONFLICT (user_id, punch_date)
               DO UPDATE SET
                 clock_in_at = CASE WHEN $3 IS NOT NULL THEN COALESCE(EXCLUDED.clock_in_at, $3) ELSE attendance.clock_in_at END,
                 clock_out_at = CASE WHEN $4 IS NOT NULL THEN COALESCE(EXCLUDED.clock_out_at, $4) ELSE attendance.clock_out_at END,
                 updated_at = NOW()`,
              [userId, today, isClockIn ? new Date(timestamp * 1000) : null, isClockIn ? null : new Date(timestamp * 1000)]
            );
          }
        }

        // Extend here for leave/OT handling with similar upsert/de-dup logic
      }

      await client.query('COMMIT');
      res.status(200).send('Events processed');
    } catch (err) {
      await client.query('ROLLBACK');
      console.error('[webhook/line] Processing failed', err);
      res.status(500).send('Processing failed');
    } finally {
      client.release();
    }
  }
);

export default router;
```
