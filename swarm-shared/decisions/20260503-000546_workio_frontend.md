# workio / frontend

## 1. Diagnosis
- No optimistic UI on punch actions → perceived latency and ambiguous spinners when clocking in/out.
- No local queue/offline buffer → punches lost on flaky networks; no retry path.
- No client-side idempotency key → double-taps and LINE duplicate webhooks create duplicate punches.
- No clear rollback on failure → UI can drift from server state after network errors.
- Missing immediate feedback for GPS/validation failures → users retry blindly.

## 2. Proposed change
File: `workio/src/features/punch/usePunchQueue.ts` (new) + `workio/src/features/punch/PunchButton.tsx` (modify).  
Scope: add a lightweight, client-side queue with idempotency keys, optimistic UI, and exponential backoff retry; integrate into existing punch mutation calls. No schema or backend changes.

## 3. Implementation

### `workio/src/features/punch/usePunchQueue.ts`
```tsx
import { useCallback, useRef, useState } from 'react';
import { useMutation, useQueryClient } from '@tanstack/react-query';

type PunchType = 'IN' | 'OUT';
type PunchStatus = 'pending' | 'committing' | 'committed' | 'failed';

interface PunchItem {
  id: string; // client-side idempotency key (uuid)
  type: PunchType;
  tenantId: string;
  userId: string;
  location: { lat: number; lng: number };
  timestamp: string; // ISO
  attempts: number;
  status: PunchStatus;
  error?: string | null;
}

const MAX_ATTEMPTS = 5;
const BASE_DELAY = 1000; // ms

export function usePunchQueue() {
  const queryClient = useQueryClient();
  const queueRef = useRef<Map<string, PunchItem>>(new Map());
  const [items, setItems] = useState<PunchItem[]>([]);
  const processingRef = useRef<Set<string>>(new Set());

  const upsertItem = useCallback((item: PunchItem) => {
    queueRef.current.set(item.id, item);
    setItems(Array.from(queueRef.current.values()).sort((a, b) => a.timestamp.localeCompare(b.timestamp)));
  }, []);

  const removeItem = useCallback((id: string) => {
    queueRef.current.delete(id);
    processingRef.current.delete(id);
    setItems(Array.from(queueRef.current.values()));
  }, []);

  const updateStatus = useCallback((id: string, status: PunchStatus, error?: string | null) => {
    const existing = queueRef.current.get(id);
    if (!existing) return;
    const updated: PunchItem = {
      ...existing,
      status,
      error: error ?? null,
      attempts: status === 'committing' ? existing.attempts + 1 : existing.attempts,
    };
    queueRef.current.set(id, updated);
    setItems(Array.from(queueRef.current.values()));
  }, []);

  // Mutation that performs the actual API call
  const punchMutation = useMutation({
    mutationFn: async (item: PunchItem) => {
      const res = await fetch('/api/punches', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          idempotencyKey: item.id,
          type: item.type,
          tenantId: item.tenantId,
          userId: item.userId,
          location: item.location,
          timestamp: item.timestamp,
        }),
      });

      if (!res.ok) {
        const body = await res.json().catch(() => ({}));
        throw new Error(body?.message || `HTTP ${res.status}`);
      }
      return res.json();
    },
    onMutate: (item) => {
      updateStatus(item.id, 'committing');
    },
    onSuccess: (_data, item) => {
      updateStatus(item.id, 'committed');
      // optimistic invalidation so UI reflects latest state
      queryClient.invalidateQueries({ queryKey: ['punches', item.userId] });
      queryClient.invalidateQueries({ queryKey: ['today-punch', item.userId] });
      // remove after short delay so user sees success
      setTimeout(() => removeItem(item.id), 2500);
    },
    onError: (err: Error, item) => {
      updateStatus(item.id, 'failed', err.message);
      // schedule retry with exponential backoff
      scheduleRetry(item);
    },
  });

  const scheduleRetry = useCallback(
    (item: PunchItem) => {
      if (item.attempts >= MAX_ATTEMPTS) {
        updateStatus(item.id, 'failed', 'Max attempts reached');
        return;
      }
      processingRef.current.delete(item.id);
      const delay = BASE_DELAY * Math.pow(2, Math.min(item.attempts, 6));
      setTimeout(() => {
        // re-enqueue same item with updated attempts count
        const next: PunchItem = { ...item, status: 'pending' as const };
        upsertItem(next);
        processQueue();
      }, delay);
    },
    [upsertItem, updateStatus]
  );

  const enqueue = useCallback(
    (type: PunchType, tenantId: string, userId: string, location: { lat: number; lng: number }) => {
      const id = crypto.randomUUID();
      const item: PunchItem = {
        id,
        type,
        tenantId,
        userId,
        location,
        timestamp: new Date().toISOString(),
        attempts: 0,
        status: 'pending',
        error: null,
      };
      upsertItem(item);
      processQueue();
      return id;
    },
    [upsertItem]
  );

  const processQueue = useCallback(() => {
    const pending = Array.from(queueRef.current.values()).filter(
      (it) => it.status === 'pending' && !processingRef.current.has(it.id)
    );
    // serialize: one at a time to preserve order and reduce race risk
    const next = pending[0];
    if (!next) return;
    processingRef.current.add(next.id);
    punchMutation.mutate(next);
  }, [punchMutation]);

  const retryFailed = useCallback((id: string) => {
    const existing = queueRef.current.get(id);
    if (!existing) return;
    const reset: PunchItem = { ...existing, status: 'pending' as const, error: null };
    upsertItem(reset);
    processQueue();
  }, [upsertItem, processQueue]);

  const clearCompleted = useCallback(() => {
    for (const [id, item] of queueRef.current.entries()) {
      if (item.status === 'committed') queueRef.current.delete(id);
    }
    setItems(Array.from(queueRef.current.values()));
  }, []);

  return {
    items,
    enqueue,
    retryFailed,
    clearCompleted,
    processing: processingRef.current.size > 0,
  };
}
```

### `workio/src/features/punch/PunchButton.tsx` (modify)
```tsx
import { useState } from 'react';
import { usePunchQueue } from './usePunchQueue';
import { useCurrentUser } from '@/features/auth/useCurrentUser';
import { useGeolocation } from '@/hooks/useGeolocation';
import { Button } from '@/components/ui/button';
import { Loader2, Check, Clock, AlertCircle } from 'lucide-react';

export function PunchButton() {
  const { user } = useCurrentUser();
  const { location, error: locationError } = useGeolocation({ enableHighAccuracy: true });
  const { enqueue, items, processing, retryFailed } = usePunchQueue();
  const [localStatus, setLocalStatus] = useState<'idle' | 'optimistic'>('idle');

  if (!user) return null;

  const tenantId = user.tenantId;
  const userId = user.id;

  const handlePunch = (type: 'IN' | 'OUT') => {
    if (!location) {
      // immediate feedback for location failure
      return;
    }
    setLocalStatus('optimistic');
    enqueue(type, tenantId, userId, location);
    // optimistic local reset after short delay to allow queue to process
    setTimeout(() => setLocalStatus('idle'), 800);
  };
