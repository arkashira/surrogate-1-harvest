# airship / design

## Final Synthesis — One Correct, Actionable Plan

**Goal (unchanged):**  
Eliminate HF API 429s and `pyarrow.CastError`s during dataset loading, reduce Lightning quota burn, and ship frontend-only in ≤2 hours.

**Chosen approach (merged + resolved):**  
- **CDN-bypass dataset loader** (no `load_dataset`, no HF `/api/` calls, no auth headers).  
- **Strict schema projection** to `{prompt, response}` at parse time (eliminates `pyarrow.CastError` from mixed schemas).  
- **Lightning Studio reuse + idle-stop guard** (saves quota; prevents recreation).  
- **Frontend-only delivery** (React UI + lightweight launcher script; zero backend changes).

Contradictions resolved in favor of correctness + actionability:
- Use **pre-generated file-list JSON per date** (generated once on dev/Mac) to avoid repeated repo-tree API calls during training runs.  
- Use **direct CDN (`resolve/main/`) URLs** for fetches; do not rely on `streaming=True` or `load_dataset` on heterogeneous repos.  
- Keep **Studio lifecycle logic simple and defensive**: list → reuse if running; else create; ensure running before `.run()`; restart on idle-stop.  
- Keep **UI minimal**: date picker, preview (CDN), run training (Studio reuse). No schema inference in UI; projection happens in loader.

### 1. CDN-bypass loader (TypeScript — frontend utility)
`src/utils/hf-cdn-loader.ts`
```ts
export interface PromptResponseRecord {
  prompt: string;
  response: string;
}

const HF_CDN = "https://huggingface.co/datasets";

export async function listDateFiles(
  repo: string,
  dateFolder: string
): Promise<string[]> {
  // One-time call from dev machine; results saved to file_list.json.
  // In production UI, this JSON is bundled or fetched statically.
  const res = await fetch(
    `https://huggingface.co/api/datasets/${repo}/tree/${dateFolder}?recursive=1`
  );
  if (!res.ok) throw new Error("Failed to list files");
  const tree = await res.json();
  return tree
    .filter((f: any) => f.type === "file")
    .map((f: any) => `${dateFolder}/${f.path}`);
}

export async function fetchPromptResponseFile(
  repo: string,
  filePath: string
): Promise<PromptResponseRecord[]> {
  const url = `${HF_CDN}/${repo}/resolve/main/${encodeURIComponent(filePath)}`;
  const res = await fetch(url);
  if (!res.ok) throw new Error(`Failed to fetch ${filePath}`);
  const text = await res.text();
  const lines = text.trim().split("\n").filter(Boolean);
  return lines.map((line) => {
    const obj = JSON.parse(line);
    return {
      prompt: obj.prompt ?? obj.input ?? "",
      response: obj.response ?? obj.output ?? "",
    };
  });
}

export async function loadDatasetCdn(
  repo: string,
  dateFolder: string,
  fileList?: string[]
): Promise<PromptResponseRecord[]> {
  const files = fileList ?? (await listDateFiles(repo, dateFolder));
  const results: PromptResponseRecord[] = [];
  for (const f of files) {
    try {
      const recs = await fetchPromptResponseFile(repo, f);
      results.push(...recs);
    } catch (err) {
      console.warn(`Skip ${f}:`, err);
    }
  }
  return results;
}
```

### 3. Studio reuse + idle-stop guard (Python — launcher used by UI)
`surrogate/train/run.py`
```python
import time
import json
from pathlib import Path
from lightning import Studio, Machine, Teamspace

REPO = "your-org/surrogate-dataset"

def get_or_create_studio(name: str = "surrogate-train-l40s") -> Studio:
    for s in Teamspace.studios:
        if s.name == name and s.status == "Running":
            print(f"Reusing running studio: {s.name}")
            return s
    print(f"Creating studio: {name}")
    return Studio(name=name, machine=Machine.L40S, create_ok=True)

def ensure_running(studio: Studio) -> Studio:
    if studio.status != "Running":
        print(f"Studio stopped ({studio.status}). Restarting...")
        studio.start(machine=Machine.L40S)
        while studio.status != "Running":
            time.sleep(10)
            studio.refresh()
    return studio

def run_training_job(script: str, args: list[str]):
    studio = get_or_create_studio()
    studio = ensure_running(studio)
    run = studio.run(
        run_name="surrogate-cdn-train",
        command=f"python {script} {' '.join(args)}",
        local_dir="./outputs"
    )
    return run

# Lightweight loader used inside training script (if needed)
def load_dataset_cdn(date_folder: str):
    manifest_path = Path("file_list.json")
    if manifest_path.exists():
        manifest = json.loads(manifest_path.read_text())
        files = manifest.get(date_folder, [])
    else:
        raise FileNotFoundError("Run generate-file-list once to create file_list.json")

    import requests
    records = []
    for f in files:
        url = f"https://huggingface.co/datasets/{REPO}/resolve/main/{f}"
        resp = requests.get(url, timeout=30)
        resp.raise_for_status()
        for line in resp.text.strip().splitlines():
            if not line.strip():
                continue
            obj = json.loads(line)
            records.append({
                "prompt": obj.get("prompt", obj.get("input", "")),
                "response": obj.get("response", obj.get("output", ""))
            })
    return records
```
