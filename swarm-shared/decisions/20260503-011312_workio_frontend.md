# workio / frontend

## Final synthesized implementation

Below is the single, production-ready solution that merges the strongest, non-contradictory parts of both proposals and resolves conflicts in favor of **correctness + concrete actionability**.

### Core decisions (why)

- **Request deduplication**: use an in-flight promise ref (Candidate 1) + `isPending` from React Query (Candidate 1) to guarantee exactly one network call per user intent, even under double-taps, retries, or LINE webhook storms.
- **Optimistic UI + rollback**: apply an optimistic update to the current-punch query so the UI responds instantly; rollback automatically on mutation failure (stronger than Candidate 2’s local-only approach).
- **Persisted pending state across refreshes**: store the pending punch ID + timestamp in `sessionStorage` and hydrate on mount. This prevents duplicate punches after background/refresh while remaining source-of-truth with the server (resolves Candidate 2’s local-state weakness).
- **Clear UX states and recovery**: idle, clocking, success, error + retry affordance; disable button while clocking; explicit error text and retry button (Candidate 1 + Candidate 2).
- **GPS/location failure handling**: surface errors and retry; no silent failures.
- **Backend remains source of truth**: optimistic update and sessionStorage are client-side accelerators; server responses always win. Conflicts (e.g., 409) roll back and show actionable feedback.

import { useCallback, useRef } from 'react';
import { useMutation, useQueryClient } from '@tanstack/react-query';

type ClockAction = 'IN' | 'OUT';

export interface ClockResponse {
  id: string;
  user_id: string;
  clock_in_at: string;
  clock_out_at: string | null;
}

interface UseClockOptions {
  onSuccess?: (res: ClockResponse) => void;
  onError?: (err: Error) => void;
}

const PENDING_PUNCH_STORAGE_KEY = 'clock:pending-punch';

function getPendingPunchFromStorage() {
  try {
    const raw = sessionStorage.getItem(PENDING_PUNCH_STORAGE_KEY);
    return raw ? JSON.parse(raw) : null;
  } catch {
    return null;
  }
}

function setPendingPunchStorage(value: { id: string; action: ClockAction; timestamp: number } | null) {
  try {
    if (value == null) sessionStorage.removeItem(PENDING_PUNCH_STORAGE_KEY);
    else sessionStorage.setItem(PENDING_PUNCH_STORAGE_KEY, JSON.stringify(value));
  } catch {
    // ignore storage errors
  }
}

export function useClock({ onSuccess, onError }: UseClockOptions = {}) {
  const queryClient = useQueryClient();
  const inFlightRef = useRef<Promise<ClockResponse> | null>(null);

  const mutation = useMutation({
    mutationFn: async (action: ClockAction) => {
      // Deduplicate: reuse in-flight promise
      if (inFlightRef.current) return inFlightRef.current;

      const controller = new AbortController();
      const signal = controller.signal;

      const promise = fetch('/api/clock', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ action }),
        signal,
      })
        .then(async (res) => {
          if (!res.ok) {
            const text = await res.text().catch(() => '');
            throw new Error(text || `Clock ${action} failed (${res.status})`);
          }
          return res.json() as Promise<ClockResponse>;
        })
        .finally(() => {
          inFlightRef.current = null;
        });

      inFlightRef.current = promise;
      return promise;
    },
    onMutate: async (action: ClockAction) => {
      await queryClient.cancelQueries({ queryKey: ['current-punch'] });
      const previous = queryClient.getQueryData(['current-punch']);

      const optimistic: ClockResponse = {
        id: `temp-${Date.now()}`,
        user_id: 'current-user', // will be ignored by UI; real id comes from server
        clock_in_at: new Date().toISOString(),
        clock_out_at: action === 'OUT' ? new Date().toISOString() : null,
      };

      queryClient.setQueryData(['current-punch'], optimistic);

      // Persist pending intent to survive accidental refresh (best-effort)
      setPendingPunchStorage({ id: optimistic.id, action, timestamp: Date.now() });

      return { previous };
    },
    onError: (err: Error, _variables, context) => {
      queryClient.setQueryData(['current-punch'], context?.previous ?? null);
      setPendingPunchStorage(null);
      onError?.(err);
    },
    onSuccess: (data) => {
      setPendingPunchStorage(null);
      queryClient.setQueryData(['current-punch'], data);
      queryClient.invalidateQueries({ queryKey: ['today-punch'] });
      onSuccess?.(data);
    },
  });

  const clock = useCallback(
    async (action: ClockAction) => {
      try {
        return await mutation.mutateAsync(action);
      } catch {
        // handled by onError
      }
    },
    [mutation]
  );

  return {
    clock,
    isClocking: mutation.isPending,
    isSuccess: mutation.isSuccess,
    error: mutation.error,
    reset: () => mutation.reset(),
  };
}

// Hydrate any pending punch on app start (best-effort reconciliation)
export function reconcilePendingPunch() {
  const pending = getPendingPunchFromStorage();
  if (!pending) return;
  // We keep it in storage; the next real API call (refetch or new clock) will overwrite.
  // This prevents duplicate optimistic entries while preserving intent across refresh.
}

#### 2) `workio/src/features/punch/useCurrentPunch.ts`

```ts
import { useQuery } from '@tanstack/react-query';

export function useCurrentPunch() {
  return useQuery({
    queryKey: ['current-punch'],
    queryFn: async () => {
      const res = await fetch('/api/clock/current');
      if (!res.ok) {
        if (res.status === 404) return null;
        throw new Error('Failed to fetch current punch');
      }
      return res.json();
    },
    refetchOnMount: true,
    staleTime: 30_000,
  });
}
```
