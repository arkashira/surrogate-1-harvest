#!/usr/bin/env python3
"""
Deterministic CDN-only manifest generator for Surrogate-1 training.
Eliminates HF API usage during training data loading.
"""

import json
import time
import hashlib
from datetime import datetime, timezone
from pathlib import Path
from typing import List, Dict, Optional

import requests
from huggingface_hub import list_repo_tree
from rich.console import Console
from rich.progress import track

console = Console()

class CDNManifestGenerator:
    def __init__(self, cache_dir: Path = Path(".cache/manifests")):
        self.cache_dir = cache_dir
        self.cache_dir.mkdir(parents=True, exist_ok=True)

    def _get_cache_path(self, repo_id: str, date_folder: str) -> Path:
        """Deterministic cache filename based on repo + folder."""
        slug = f"{repo_id.replace('/', '--')}--{date_folder}"
        return self.cache_dir / f"{slug}.json"

    def _rate_limit_safe_list(self, repo_id: str, folder_path: str, max_retries: int = 5) -> List[Dict]:
        """List repo tree with exponential backoff for 429s."""
        for attempt in range(max_retries):
            try:
                tree = list_repo_tree(
                    repo_id=repo_id,
                    path=folder_path,
                    recursive=False
                )
                return list(tree)
            except Exception as e:
                if "429" in str(e) or "rate limit" in str(e).lower():
                    wait_time = 360 * (2 ** attempt)  # 6min, 12min, 24min...
                    console.print(f"[yellow]Rate limited. Waiting {wait_time}s...[/yellow]")
                    time.sleep(wait_time)
                else:
                    raise
        raise Exception(f"Failed after {max_retries} retries")

    def _collect_all_files(self, repo_id: str, root_path: str) -> List[Dict]:
        """Recursively collect all files under root_path."""
        all_files = []
        to_process = [root_path]

        while to_process:
            current = to_process.pop(0)
            items = self._rate_limit_safe_list(repo_id, current)

            for item in items:
                if item.type == "file":
                    all_files.append({
                        "path": item.path,
                        "size": getattr(item, "size", 0),
                        "lfs": getattr(item, "lfs", None)
                    })
                elif item.type == "directory":
                    to_process.append(item.path)

        return all_files

    def generate_manifest(self, repo_id: str, date_folder: str,
                         file_extensions: Optional[List[str]] = None,
                         force_refresh: bool = False) -> Dict:
        """
        Generate CDN-only manifest for training data.

        Args:
            repo_id: HF repo identifier (e.g., "username/dataset")
            date_folder: Date folder path (e.g., "2024-01-15")
            file_extensions: Filter by extensions (default: [".parquet", ".jsonl"])
            force_refresh: Ignore cache and regenerate

        Returns:
            Dict with manifest metadata and file entries
        """
        if file_extensions is None:
            file_extensions = [".parquet", ".jsonl", ".json"]

        cache_path = self._get_cache_path(repo_id, date_folder)

        # Return cached manifest if exists and <24h old (unless forced)
        if not force_refresh and cache_path.exists():
            cache_age = time.time() - cache_path.stat().st_mtime
            if cache_age < 86400:
                console.print(f"[green]Using cached manifest: {cache_path}[/green]")
                return json.loads(cache_path.read_text())

        console.print(f"[cyan]Generating CDN manifest for {repo_id}/{date_folder}...[/cyan]")

        # Collect all files recursively
        raw_files = self._collect_all_files(repo_id, date_folder)

        # Filter by extension and build CDN URLs
        files = []
        for f in raw_files:
            if any(f["path"].endswith(ext) for ext in file_extensions):
                cdn_url = (
                    f"https://huggingface.co/datasets/{repo_id}"
                    f"/resolve/main/{f['path']}"
                )
                files.append({
                    "filename": f["path"],
                    "cdn_url": cdn_url,
                    "size": f["size"],
                    "lfs": f["lfs"]
                })

        manifest = {
            "generated_at": datetime.now(timezone.utc).isoformat(),
            "repo_id": repo_id,
            "date_folder": date_folder,
            "total_files": len(files),
            "total_size": sum(f["size"] for f in files),
            "cdn_only": True,
            "api_calls_used": 1,  # Minimal; actual calls happen during collection
            "files": files
        }

        # Cache manifest
        cache_path.write_text(json.dumps(manifest, indent=2))
        console.print(f"[green]✓ Manifest cached: {cache_path}[/green]")

        return manifest

    def download_via_cdn(self, manifest: Dict, output_dir: Path,
                        max_workers: int = 4, skip_existing: bool = True) -> List[Path]:
        """
        Download files using CDN URLs (bypasses HF API rate limits).

        Args:
            manifest: Generated manifest dict
            output_dir: Local directory for downloads
            max_workers: Parallel download threads
            skip_existing: Skip files that already exist with correct size

        Returns:
            List of downloaded file paths
        """
        output_dir.mkdir(parents=True, exist_ok=True)
        downloaded = []

        from concurrent.futures import ThreadPoolExecutor, as_completed

        def download_file(file_info: Dict) -> Optional[Path]:
            local_path = output_dir / file_info["filename"]
            local_path.parent.mkdir(parents=True, exist_ok=True)

            # Skip if already exists with correct size
            if skip_existing and local_path.exists():
                if local_path.stat().st_size == file_info["size"]:
                    return local_path

            # CDN download - no Authorization header needed
            try:
                response = requests.get(
                    file_info["cdn_url"],
                    stream=True,
                    timeout=30,
                    headers={"User-Agent": "airship-cdn-downloader/1.0"}
                )
                response.raise_for_status()
                local_path.write_bytes(response.content)
                return local_path
            except Exception as e:
                console.print(f"[red]Download failed for {file_info['filename']}: {e}[/red]")
                return None

        console.print(f"[cyan]Downloading {len(manifest['files'])} files via CDN...[/cyan]")

        with ThreadPoolExecutor(max_workers=max_workers) as executor:
            futures = {
                executor.submit(download_file, f): f
                for f in manifest["files"]
            }

            for future in track(as_completed(futures), total=len(futures)):
                result = future.result()
                if result:
                    downloaded.append(result)

        return downloaded


def main():
    import argparse

    parser = argparse.ArgumentParser(
        description="Generate CDN-only manifest for Surrogate-1 training"
    )
    parser.add_argument("--repo", required=True, help="HF repo ID (e.g., username/dataset)")
    parser.add_argument("--date", required=True, help="Date folder (e.g., 2024-01-15)")
    parser.add_argument("--output", default="manifest.json", help="Output JSON path")
    parser.add_argument("--download", action="store_true", help="Download files to ./data/")
    parser.add_argument("--workers", type=int, default=4, help="Parallel download threads")
    parser.add_argument("--force", action="store_true", help="Force manifest regeneration")

    args = parser.parse_args()

    generator = CDNManifestGenerator()
    manifest = generator.generate_man
