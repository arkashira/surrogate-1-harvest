**Make surrogate training HF-rate-limit-proof and Lightning-idle-resilient** by:  
- Generating a **one-time CDN-only manifest** on the Mac (no per-worker HF API calls)

- Adding an **idle-resilient launcher** that auto-restarts the Lightning Studio if stopped, then runs training to completion  

**Why this ships in <2h**: small, focused changes (one generator script, one train.py patch, one launcher wrapper) that remove two major failure modes (HF 429, Lightning idle death) without touching infra or models.

### 1) `scripts/generate_manifest.py` (run on Mac)

```python
#!/usr/bin/env python3
"""
Generate CDN-only manifest for a dataset folder.
Usage:
  python generate_manifest.py \
    --repo huggingface_user/surrogate-dataset \
    --date 2026-04-29 \
    --out manifest.json
"""

import argparse
import json
import os
import time
from pathlib import Path
from typing import Dict, List
from huggingface_hub import HfApi

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def list_folder_safe(api: HfApi, repo: str, folder: str) -> List[str]:
    items = api.list_repo_tree(repo=repo, path=folder, recursive=False)
    files = []
    for item in items:
        if item.rfilename.endswith((".parquet", ".jsonl", ".json")):
            files.append(item.rfilename)
    return files

def build_manifest(repo: str, date_folder: str) -> Dict:
    api = HfApi()
    folder = f"batches/mirror-merged/{date_folder}"
    files = list_folder_safe(api, repo, folder)

    entries = []
    for f in files:
        entries.append({
            "repo": repo,
            "path": f,
            "cdn_url": CDN_TEMPLATE.format(repo=repo, path=f),
            "local_name": Path(f).stem
        })

    manifest = {
        "generated_at": time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime()),
        "repo": repo,
        "folder": folder,
        "file_count": len(entries),
        "files": entries
    }
    return manifest

def main() -> None:
    parser = argparse.ArgumentParser(description="Generate CDN manifest")
    parser.add_argument("--repo", required=True, help="HF dataset repo (user/name)")
    parser.add_argument("--date", required=True, help="Date folder (YYYY-MM-DD)")
    parser.add_argument("--out", default="manifest.json", help="Output JSON path")
    args = parser.parse_args()

    manifest = build_manifest(args.repo, args.date)
    out_path = Path(args.out)
    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"Wrote {out_path} with {manifest['file_count']} files")

if __name__ == "__main__":
    main()
```

### 2) Patch `train.py` (minimal, CDN-only loader)

```python
# Near top of train.py
import json
from pathlib import Path
from typing import Dict, Iterator

import pyarrow.parquet as pq
import requests
from tqdm import tqdm

MANIFEST_PATH = Path(__file__).parent / "manifest.json"

def load_manifest() -> Dict:
    if not MANIFEST_PATH.exists():
        raise FileNotFoundError(f"Manifest not found: {MANIFEST_PATH}")
    with open(MANIFEST_PATH) as f:
        return json.load(f)

def stream_parquet_from_cdn(url: str, local_cache: Path):
    """Download parquet via CDN (no auth) with resume support."""
    if local_cache.exists():
        return local_cache
    local_cache.parent.mkdir(parents=True, exist_ok=True)
    resp = requests.get(url, stream=True, timeout=60)
    resp.raise_for_status()
    with open(local_cache, "wb") as f:
        for chunk in resp.iter_content(chunk_size=8192):
            f.write(chunk)
    return local_cache

def build_cdn_dataset(manifest: Dict, cache_root: Path = Path(".cdn_cache")) -> Iterator[Dict[str, str]]:
    """Yield {prompt, response} rows from CDN files with zero HF API calls."""
    cache_root.mkdir(exist_ok=True)
    for entry in tqdm(manifest["files"], desc="Loading CDN files"):
        url = entry["cdn_url"]
        cache_file = cache_root / Path(entry["path"]).name
        try:
            local_path = stream_parquet_from_cdn(url, cache_file)
            table = pq.read_table(local_path, columns=["prompt", "response"])
            for i in range(table.num_rows):
                row = table.slice(i, 1).to_pydict()
                yield {"prompt": row["prompt"][0], "response": row["response"][0]}
        except Exception as exc:
            print(f"Skipping {url}: {exc}")
            continue
```

### 3) `scripts/lightning_launcher.py` (idle-resilient launcher)

#!/usr/bin/env python3
"""
Lightning idle-resilient launcher.
- Reuses running Studio if found
- Restarts if stopped before run
- Runs training with manifest baked in
"""

import time
import subprocess
from pathlib import Path
from lightning_sdk import Studio, Teamspace, Machine

MANIFEST_SRC = Path(__file__).parent.parent / "manifest.json"
TRAIN_SCRIPT = Path(__file__).parent.parent / "train.py"

def get_or_create_studio(name: str = "surrogate-train") -> Studio:
    team = Teamspace()
    running = [s for s in team.studios if s.name == name and s.status == "running"]
    if running:
        print(f"Reusing running studio: {running[0].id}")
        return running[0]

    print("No running studio found, creating...")
    studio = Studio(
        name=name,
        machine=Machine.L40S,
        cloud="lightning-public-prod",
        create_ok=True,
    )
    return studio

def ensure_running(studio: Studio) -> Studio:
    if studio.status != "running":
        print(f"Studio stopped ({studio.status}), restarting...")
        studio.start(machine=Machine.L40S)
        for _ in range(30):
            studio.refresh()
            if studio.status == "running":
                break
            time.sleep(10)
        else:
            raise RuntimeError("Studio failed to start")
    return studio

def run_training(studio: Studio) -> None:

        raise RuntimeError(f"Generate manifest first: {MANIFEST_SRC}")
