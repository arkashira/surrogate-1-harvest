# Costinel / ops

## Final Implementation Plan — Top-Hub Signal Panel (CDN-first, <2h)

### What we ship
- A **non-blocking Top‑Hub Signal Panel** mounted at the top of the Costinel dashboard.
- Defaults to hub **MOC** (configurable via `VITE_HUB_NAME`).
- Shows: hub title, short description, top 3 signals (anomaly/recommendation), and **last updated timestamp**.
- **CDN‑first data strategy**:
  - Pre‑list hub files once (Mac/Linux orchestration) and commit `hub-files.json` to the repo.
  - Runtime fetches **only** from public CDN:  
    `https://huggingface.co/datasets/AXENTX/Costinel-hub/resolve/main/{hub}/latest.json`  
    (and falls back to `{hub}/signals.json` if needed).
  - No Authorization header, no backend, no HF API calls during dashboard use → avoids rate limits.
- **Fail‑safe**: on any CDN/network error, the panel collapses to `null` and does not break the dashboard.

---

### Why this is highest‑value (<2h)
- Immediately applies the **#knowledge‑rag / #hub** pattern (MOC).
- Uses proven CDN‑bypass to avoid HF rate limits and backend changes.
- Non‑blocking: visible UX value without touching core cost pipelines.
- Reuses existing patterns (Lightning/Mac orchestration, CDN‑only runtime).
- Adds **last updated timestamp** (from Candidate 2) for operational trust.

#### 1) Hub file list (run once on orchestrator)
```bash
# /opt/axentx/Costinel/scripts/list-hub-files.sh
#!/usr/bin/env bash
set -euo pipefail

REPO="AXENTX/Costinel-hub"
OUT_DIR="src/config"
OUT_FILE="${OUT_DIR}/hub-files.json"
HUB_PATH="${HF_PATH:-moc}"

mkdir -p "${OUT_DIR}"

python3 - <<PY
import os, json
from huggingface_hub import list_repo_tree

repo = os.getenv("HF_REPO", "$REPO")
path = os.getenv("HF_PATH", "$HUB_PATH")

files = list_repo_tree(repo=repo, path=path, recursive=False, repo_type="dataset")
items = [{"path": f.path, "size": f.size} for f in files if f.type == "file"]
with open("$OUT_FILE", "w") as f:
    json.dump(items, f, indent=2)
print(f"Saved {len(items)} files to $OUT_FILE")
PY
```

Make executable and run:
```bash
chmod +x /opt/axentx/Costinel/scripts/list-hub-files.sh
HF_REPO=AXENTX/Costinel-hub HF_PATH=moc /opt/axentx/Costinel/scripts/list-hub-files.sh
```

Commit `src/config/hub-files.json`.

#### 3) Panel component (React, non‑blocking)
```tsx
// src/components/TopHubSignalPanel.tsx
import { useEffect, useState } from 'react';
import { HUB_CONFIG } from '../config/hub';

interface Signal {
  title: string;
  description: string;
  severity: 'high' | 'medium' | 'low';
  recommendation?: string;
}

interface HubManifest {
  title: string;
  description: string;
  signals: Signal[];
  lastUpdated?: string;
}

export function TopHubSignalPanel() {
  const [manifest, setManifest] = useState<HubManifest | null>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    const hub = HUB_CONFIG.name;
    // Primary CDN path (latest.json)
    const primary = `${HUB_CONFIG.cdnBase}/${HUB_CONFIG.repo}/resolve/main/${hub}/latest.json`;
    // Fallback path (signals.json)
    const fallback = `${HUB_CONFIG.cdnBase}/${HUB_CONFIG.repo}/resolve/main/${hub}/signals.json`;

    function tryFetch(url: string): Promise<HubManifest> {
      return fetch(url, { cache: 'no-store' }).then((res) => {
        if (!res.ok) throw new Error(`CDN fetch failed: ${res.status}`);
        return res.json();
      });
    }

    tryFetch(primary)
      .catch(() => tryFetch(fallback))
      .then((data: HubManifest) => {
        setManifest(data);
      })
      .catch((err) => {
        console.warn('Hub panel skipped:', err);
      })
      .finally(() => {
        setLoading(false);
      });
  }, []);

  if (loading || !manifest) return null;

  return (
    <aside style={styles.panel}>
      <div style={styles.header}>
        <h3 style={styles.title}>{manifest.title}</h3>
        <p style={styles.desc}>{manifest.description}</p>
        {manifest.lastUpdated && (
          <span style={styles.timestamp}>Last updated: {manifest.lastUpdated}</span>
        )}
      </div>

      <div style={styles.signals}>
        {manifest.signals.slice(0, 3).map((s, i) => (
          <div key={i} style={{ ...styles.signal, borderLeftColor: severityColor(s.severity) }}>
            <strong>{s.title}</strong>
            <p style={{ margin: 0, fontSize: 12 }}>{s.description}</p>
            {s.recommendation && (
              <span style={{ fontSize: 11, color: '#555' }}>{s.recommendation}</span>
            )}
          </div>
        ))}
      </div>

      <div style={styles.footer}>
        <a
          href={`${HUB_CONFIG.cdnBase}/${HUB_CONFIG.repo}/tree/main/${HUB_CONFIG.name}`}
          target="_blank"
          rel="noopener noreferrer"
          style={styles.link}
        >
          View details →
        </a>
      </div>
    </aside>
  );
}

function severityColor(s: string) {
  switch (s) {
    case 'high':
      return '#ef4444';
    case 'medium':
      return '#f59e0b';
    default:
      return '#10b981';
  }
}

const styles = {
  panel: {
    border: '1px solid #e5e7eb',
    borderRadius: 8,
    padding: 12,
    background: '#fff',
    marginBottom: 12,
    fontSize: 13,
  } as React.CSSProperties,
  header: { marginBottom: 8 } as React.CSSProperties,
  title: { margin: 0, fontSize: 14 } as React.CSSProperties,
  desc: { margin: 0, color: '#6b7280', fontSize: 12 } as React.CSSProperties,
  timestamp: { fontSize: 11, color: '#9ca3af' } as React.CSSProperties,
  signals: { display: 'flex', flexDirection: 'column', gap: 6 } as React.CSSProperties,
  signal: {
    borderLeft: '3px solid',
    paddingLeft: 8,
    display: 'flex',
    flexDirection: 'column',
    gap: 2,
  } as React.CSSProperties,
  footer: { marginTop: 8, textAlign: 'right' } as React.CSSProperties,
  link: { color: '#2563eb', textDecoration: 'none', fontSize: 12 } as React.CSSProperties,
};
```
