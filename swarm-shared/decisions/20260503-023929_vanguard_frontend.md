# vanguard / frontend

## 1. Diagnosis
- Frontend has no build/dev tooling or entrypoint — cannot ship or iterate on UI changes.
- Missing `index.html` and client mount point means zero fast feedback loop for frontend work.
- No asset pipeline or bundler configured; any JS/TS/CSS changes require manual reloads and lack module resolution.
- Frontend likely triggers authenticated HF API calls (`list_repo_tree`, dataset endpoints) at runtime, burning quota and risking 429s on user machines.
- No static file manifest embedded in the bundle → every session re-enumerates repos instead of using CDN-only file lists.

## 2. Proposed change
Create a minimal, production-capable frontend entrypoint and asset pipeline in `/opt/axentx/vanguard`:
- Add `index.html` (mount + CDN-friendly script tag)
- Add `src/main.js` (app bootstrap, mount, and safe HF CDN file-list usage)
- Add `vite.config.js` (dev server + build)
- Add `package.json` with scripts (`dev`, `build`, `preview`)
- Embed pre-listed file manifest support so training/data workflows can inject a JSON file list and the frontend uses only CDN URLs (no runtime HF API calls).

Scope: new files only; no changes to existing backend code.

const CDN_ROOT = 'https://huggingface.co/datasets';
const MANIFEST_PATH = '/manifest/files.json'; // to be injected by orchestration (e.g., {repo}/{date}/files.json)

const state = {
  files: [],        // [{ path, repo, date }]
  selected: null
};

const els = {
  loadBtn: document.getElementById('loadBtn'),
  clearBtn: document.getElementById('clearBtn'),
  fileList: document.getElementById('fileList'),
  preview: document.getElementById('preview')
};

function renderFiles() {
  els.fileList.innerHTML = '';
  if (!state.files.length) {
    els.fileList.innerHTML = '<span class="muted">No files loaded</span>';
    return;
  }

  const ul = document.createElement('ul');
  ul.style.listStyle = 'none';
  ul.style.padding = 0;
  ul.style.margin = 0;

  state.files.forEach((f, idx) => {
    const li = document.createElement('li');
    li.style.padding = '0.35rem 0';
    li.style.borderBottom = '1px solid #f0f0f0';
    li.style.cursor = 'pointer';
    li.textContent = f.path;
    li.title = `repo: ${f.repo || '(unknown)'}`;
    li.onclick = () => selectFile(f);
    ul.appendChild(li);
  });

  els.fileList.appendChild(ul);
}

function selectFile(f) {
  state.selected = f;
  const cdnUrl = `${CDN_ROOT}/${f.repo || 'datasets-org/placeholder'}/resolve/main/${f.path}`;
  els.preview.innerHTML = `
    <div><strong>Selected:</strong> ${f.path}</div>
    <div style="margin-top:0.5rem;"><strong>CDN URL:</strong> <a href="${cdnUrl}" target="_blank" rel="noopener">${cdnUrl}</a></div>
    <div class="muted" style="margin-top:0.5rem;">Note: This uses public CDN fetch — no Authorization header. Parquet files can be fetched and parsed client-side if needed.</div>
  `;
}

async function loadManifest() {
  els.loadBtn.disabled = true;
  els.loadBtn.textContent = 'Loading...';
  try {
    const res = await fetch(MANIFEST_PATH, { cache: 'no-cache' });
    if (!res.ok) throw new Error(`Manifest fetch failed: ${res.status}`);
    const payload = await res.json();
    // Expected shape: { repo: "owner/dataset", date: "2026-05-03", files: ["path/to/file.parquet", ...] }
    const repo = payload.repo || '';
    const date = payload.date || '';
    state.files = (payload.files || []).map(p => ({ path: p, repo, date }));
    renderFiles();
  } catch (err) {
    els.fileList.innerHTML = `<span style="color:#b91c1c;">Failed to load manifest: ${err.message}</span>`;
  } finally {
    els.loadBtn.disabled = false;
    els.loadBtn.textContent = 'Load files';
  }
}
