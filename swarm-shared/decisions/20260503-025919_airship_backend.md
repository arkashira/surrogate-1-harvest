### 1) HF CDN File-List Endpoint (Airship API)  
Expose a single endpoint that lists parquet files for a date folder and returns **direct CDN URLs**. This runs once per training cycle from the Mac orchestrator.

**File**: `/opt/axentx/airship/arkship/api/routes/training.py`

```python
from fastapi import APIRouter, HTTPException
from huggingface_hub import HfApi
from datetime import datetime, timezone
import os

router = APIRouter()
HF_REPO = os.getenv("HF_DATASET_REPO", "datasets/axentx/surrogate-mirror")
HF_API = HfApi()

@router.get("/training/file-list/{date_folder}")
async def get_training_file_list(date_folder: str):
    """
    List parquet files in a date folder (non-recursive) and return CDN URLs.
    Example: /training/file-list/2026-04-29
    """
    try:
        items = HF_API.list_repo_tree(
            repo_id=HF_REPO,
            path=date_folder,
            recursive=False
        )
        parquet_files = [
            f"https://huggingface.co/datasets/{HF_REPO}/resolve/main/{item['path']}"
            for item in items
            if isinstance(item, dict) and item.get("path", "").endswith(".parquet")
        ]
        return {
            "date_folder": date_folder,
            "count": len(parquet_files),
            "files": parquet_files,
            "generated_at": datetime.now(timezone.utc).isoformat()
        }
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))
```

**Register route** in `/opt/axentx/airship/arkship/api/main.py`:

### 2) CDN-Only Data Loading in Surrogate Training  
Modify the Lightning training script to accept a JSON file with CDN URLs and load via `pyarrow` row-group streaming to minimize memory and avoid HF API calls during training.

**File**: `/opt/axentx/airship/surrogate/train.py`

```python
import json
import pyarrow.parquet as pq
from torch.utils.data import Dataset, DataLoader
import lightning as L
from typing import List

class CDNParquetDataset(Dataset):
    def __init__(self, cdn_urls: List[str]):
        self.cdn_urls = cdn_urls
        self.row_groups = []  # list of (url, rg_index)
        for url in cdn_urls:
            pf = pq.ParquetFile(url)
            for rg in range(pf.metadata.num_row_groups):
                self.row_groups.append((url, rg))

    def __len__(self):
        return len(self.row_groups)

    def __getitem__(self, idx):
        url, rg = self.row_groups[idx]
        table = pq.read_table(url, columns=["prompt", "response"], use_threads=False, row_groups=[rg])
        row = table.slice(0, 1).to_pydict()
        return {
            "prompt": row["prompt"][0],
            "response": row["response"][0]
        }

class SurrogateDataModule(L.LightningDataModule):
    def __init__(self, file_list_path: str, batch_size: int = 8):
        super().__init__()
        self.file_list_path = file_list_path
        self.batch_size = batch_size

    def setup(self, stage=None):
        with open(self.file_list_path) as f:
            data = json.load(f)
        self.dataset = CDNParquetDataset(data["files"])

    def train_dataloader(self):
        return DataLoader(self.dataset, batch_size=self.batch_size, shuffle=True, num_workers=0)
```

### 3) Lightning Idle-Resilient Wrapper  
A lightweight wrapper that ensures the studio is running before `.run()` and restarts if idle-stopped.

**File**: `/opt/axentx/airship/surrogate/lightning_wrapper.py`

```python
import lightning as L
import time
import logging
import subprocess
import sys

log = logging.getLogger(__name__)

def get_or_create_studio(name: str, machine: L.Machine = L.Machine.L40S):
    teamspace = L.Teamspace()
    for s in teamspace.studios:
        if s.name == name and s.status == "running":
            log.info(f"Reusing running studio: {name}")
            return s
    log.info(f"Creating studio: {name}")
    return teamspace.create_studio(name=name, machine=machine, create_ok=True)

def resilient_run(studio, target_cmd: str, max_retries=3):
    """Run target_cmd on studio, restart if idle-stopped."""
    for attempt in range(max_retries):
        try:
            if studio.status != "running":
                log.warning(f"Studio {studio.name} not running (status={studio.status}). Restarting...")
                studio.start(machine=studio.machine)
                time.sleep(30)
            # Execute target command inside studio
            studio.run(target_cmd)
            return
        except Exception as e:
            log.error(f"Run failed (attempt {attempt+1}): {e}")
            if attempt == max_retries - 1:
                raise
            time.sleep(60)
            studio.start(machine=studio.machine)

if __name__ == "__main__":
    import argparse
    parser = argparse.ArgumentParser()
    parser.add_argument("--name", required=True)
    parser.add_argument("--machine", default="L40S")
    parser.add_argument("--target", required=True)
    args = parser.parse_args()

    machine = getattr(L.Machine, args.machine, L.Machine.L40S)
    studio = get_or_create_studio(args.name, machine=machine)
    resilient_run(studio, args.target)
```

### 4) Mac Orchestrator Script (One-Click Iteration)  
Fetches the CDN file list and launches training via the resilient wrapper.

**File**: `/opt/axentx/airship/scripts/run_training_cycle.sh`

```bash
#!/usr/bin/env bash
set -euo pipefail

AIRSHIP_API="http://localhost:8000"
DATE_FOLDER="${1:-$(date -u +%Y-%m-%d)}"
FILE_LIST="/tmp/surrogate_file_list_${DATE_FOLDER}.json"
SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
REPO_ROOT="$(cd "$SCRIPT_DIR/../.." && pwd)"

echo "== Airship Surrogate Training Cycle =="
echo "Date folder: $DATE_FOLDER"

# 1) Fetch CDN file list from Airship
echo "Fetching file list from Airship API..."
curl -sSf "${AIRSHIP_API}/api/v1/training/file-list/${DATE_FOLDER}" -o "$FILE_LIST"

if [[ ! -s "$FILE_LIST" ]]; then
  echo "ERROR: Empty file list"
  exit 1
fi

echo "File list saved to $FILE_LIST"
echo "Files: $(jq '.count' "$FILE_LIST")"

# 2) Launch resilient Lightning training
cd "$REPO_ROOT/surrogate"
python -m lightning_wrapper \
  --name "surrogate-training" \
  --machine L40S \
  --target "python train.py --file-list $FILE_LIST --epochs 1"

echo "== Training cycle complete =="
```
