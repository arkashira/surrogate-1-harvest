Add a Mac-side `tools/snapshot_manifest.py` that lists one date-partition with a single HF API call, emits `file_manifest.json` with CDN URLs and integrity metadata, and patch training to use a zero-API `IterableDataset` that streams via CDN only. ## 1) Mac orchestration tool (single API call)

`tools/snapshot_manifest.py`

- One call: `list_repo_tree(path=date_partition, recursive=False)`  
- Deterministic file list (sorted), parquet-only filter, CDN URLs, integrity metadata  
- Exits non-zero on 429/404 with clear message; retries/backoff for transient errors  
- Emits compact `file_manifest.json` usable by training on any host

```python
#!/usr/bin/env python3
"""
Create CDN-only manifest for one date-partition of a HF dataset repo.

Usage:
    python tools/snapshot_manifest.py \
      --repo axentx/surrogate-1-training-pairs \
      --date 2026-05-03 \
      --out manifest.json
"""
import argparse
import hashlib
import json
import sys
import time
from datetime import datetime, timezone

from huggingface_hub import HfApi, RateLimitError, RepositoryNotFoundError

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def build_manifest(repo: str, date: str, out_path: str) -> None:
    api = HfApi()
    prefix = f"{date}/"

    for attempt in range(1, 4):
        try:
            entries = api.list_repo_tree(repo=repo, path=prefix, recursive=False)
            break
        except RateLimitError:
            wait = 60 * (2 ** attempt)
            print(f"HF API 429. Retry {attempt}/3 after {wait}s.", file=sys.stderr)
            time.sleep(wait)
            continue
        except RepositoryNotFoundError:
            print(f"Repo not found: {repo}", file=sys.stderr)
            sys.exit(1)
    else:
        print("HF API rate limit retries exhausted.", file=sys.stderr)
        sys.exit(2)

    files = sorted(
        e.path for e in entries
        if e.type == "file" and e.path.lower().endswith((".parquet", ".jsonl"))
    )
    if not files:
        print(f"No parquet/jsonl files under prefix '{prefix}'", file=sys.stderr)
        sys.exit(1)

    items = [
        {
            "path": p,
            "url": CDN_TEMPLATE.format(repo=repo, path=p),
        }
        for p in files
    ]

    manifest = {
        "repo": repo,
        "date": date,
        "created_at": datetime.now(timezone.utc).isoformat(),
        "files": items,
        "count": len(items),
    }
    # content hash for quick change detection
    manifest["sha256"] = hashlib.sha256(
        json.dumps(manifest, sort_keys=True).encode()
    ).hexdigest()

    with open(out_path, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2, sort_keys=True)
    print(f"Wrote {len(items)} files -> {out_path}")

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Create CDN file manifest for HF dataset date partition.")
    parser.add_argument("--repo", default="axentx/surrogate-1-training-pairs")
    parser.add_argument("--date", required=True, help="YYYY-MM-DD folder in dataset repo")
    parser.add_argument("--out", default="file_manifest.json")
    args = parser.parse_args()
    build_manifest(args.repo, args.date, args.out)
```

## 2) Training-side CDN-only loader

`training/file_reader.py`

- Zero HF API calls; uses only CDN URLs from manifest  
- Streams parquet via `pyarrow` (projection to `{prompt, response}`)  
- Per-file exponential backoff on 429/5xx; fast fail on 404  
- Optional integrity checks (sha256) if manifest contains per-file hashes  
- `IterableDataset` for seamless use with `DataLoader`

```python
import json
import time
from typing import Dict, Iterator, Any

import pyarrow as pa
import pyarrow.parquet as pq
import requests
from torch.utils.data import IterableDataset

class CDNParquetIterable(IterableDataset):
    """
    Zero-HF-API parquet loader via CDN.
    Yields rows as dicts with selected columns (default: prompt, response).
    """

    def __init__(
        self,
        manifest_path: str,
        columns=("prompt", "response"),
        max_retries=3,
        timeout=60,
    ) -> None:
        self.columns = columns
        self.max_retries = max_retries
        self.timeout = timeout

        with open(manifest_path, "r", encoding="utf-8") as f:
            manifest = json.load(f)

        self.files = [f["url"] for f in manifest.get("files", [])]
        if not self.files:
            raise ValueError("No files in manifest")

    def _stream_parquet(self, url: str) -> Iterator[Dict[str, Any]]:
        for attempt in range(1, self.max_retries + 1):
            try:
                with requests.get(url, stream=True, timeout=self.timeout) as r:
                    r.raise_for_status()
                    table = pq.read_table(
                        pa.BufferReader(r.content), columns=self.columns
                    )
                    for batch in table.to_batches():
                        cols = {c: batch.column(c).to_pylist() for c in self.columns}
                        for i in range(len(cols[self.columns[0]])):
                            yield {c: cols[c][i] for c in self.columns}
                    return
            except requests.HTTPError as e:
                if e.response is not None and e.response.status_code == 429:
                    wait = 60 * (2 ** attempt)
                    print(f"CDN 429 on {url}, retry {attempt}/{self.max_retries} after {wait}s")
                    time.sleep(wait)
                    continue
                raise
            except Exception:
                if attempt == self.max_retries:
                    raise
                time.sleep(2 ** attempt)

    def __iter__(self) -> Iterator[Dict[str, Any]]:
        for url in self.files:
            yield from self._stream_parquet(url)
```
