**Value**: <2h implementation that operationalizes past patterns (business research + knowledge-rag + top-hub insight) into a single command that surfaces strategic context for the Arkship/Surrogate platform.

### 1. CLI Entrypoint (`/opt/axentx/airship/cli/discover.py`)

```python
#!/usr/bin/env python3
"""
airship discover
Zero-config discovery CLI that runs business research + knowledge-rag + top-hub insight.
"""
import json
import subprocess
import sys
from pathlib import Path
from typing import Optional

import typer
from neo4j import GraphDatabase

app = typer.Typer(help="Arkship discovery: business research + knowledge graph insights")

# --
# Config
# --
NEO4J_URI = "bolt://localhost:7687"
NEO4J_USER = "neo4j"
NEO4J_PASS = "password"  # override via env
PROJECT_ROOT = Path(__file__).parent.parent.parent

# --
# Business Research
# --
def business_research() -> dict:
    """Run granite-business-research.sh and capture structured output."""
    script = PROJECT_ROOT / "scripts" / "granite-business-research.sh"
    if not script.exists():
        return {"error": "granite-business-research.sh not found", "output": ""}

    try:
        result = subprocess.run(
            ["/bin/bash", str(script)],
            capture_output=True,
            text=True,
            cwd=PROJECT_ROOT,
            timeout=120,
        )
        return {
            "status": "ok" if result.returncode == 0 else "error",
            "stdout": result.stdout.strip(),
            "stderr": result.stderr.strip(),
            "returncode": result.returncode,
        }
    except Exception as e:
        return {"error": str(e), "output": ""}

# --
# Knowledge RAG (Neo4j)
# --
def _neo4j_session():
    driver = GraphDatabase.driver(
        NEO4J_URI,
        auth=(NEO4J_USER, sys.environ.get("NEO4J_PASS", NEO4J_PASS)),
    )
    return driver.session()

def knowledge_rag_query(limit: int = 10) -> list:
    """Query top connected hubs and related docs."""
    query = """
    MATCH (n:Document)
    WITH n, size((n)--()) AS degree
    ORDER BY degree DESC
    LIMIT $limit
    OPTIONAL MATCH (n)-[r]-(m)
    RETURN n.name AS name, labels(n) AS labels, degree,
           collect(DISTINCT {rel: type(r), target: m.name}) AS connections
    """
    try:
        with _neo4j_session() as session:
            records = session.run(query, limit=limit)
            return [dict(r) for r in records]
    except Exception as e:
        return [{"error": str(e)}]

def top_hub_insight() -> dict:
    """Review the most-connected hub (e.g., MOC) and 1-hop context."""
    query = """
    MATCH (n:Document)
    WITH n, size((n)--()) AS degree
    ORDER BY degree DESC
    LIMIT 1
    OPTIONAL MATCH (n)-[r]-(m)
    RETURN n {.*, degree} AS hub,
           collect(DISTINCT {rel: type(r), target: m.name, labels: labels(m)}) AS context
    """
    try:
        with _neo4j_session() as session:
            record = session.run(query).single()
            if record:
                return dict(record["hub"]), record["context"]
            return {}, []
    except Exception as e:
        return {"error": str(e)}, []

# --
# CLI Command
# --
@app.command()
def discover(
    format: str = typer.Option("table", "--format", help="Output format: table|json"),
    research: bool = typer.Option(True, "--research/--no-research", help="Run business research"),
    knowledge: bool = typer.Option(True, "--knowledge/--no-knowledge", help="Run knowledge-rag query"),
    hub: bool = typer.Option(True, "--hub/--no-hub", help="Show top-hub insight"),
):
    """
    Run zero-config discovery:
      - Business research (granite-business-research.sh)
      - Knowledge RAG (Neo4j top hubs)
      - Top-hub insight (most-connected doc)
    """
    results = {}

    if research:
        print("🔍 Running business research...")
        results["business_research"] = business_reresearch()

    if knowledge:
        print("🧠 Querying knowledge graph...")
        results["knowledge_rag"] = knowledge_rag_query()

    if hub:
        print("🏗️  Fetching top-hub insight...")
        hub_node, context = top_hub_insight()
        results["top_hub"] = {"hub": hub_node, "context": context}

    if format == "json":
        print(json.dumps(results, indent=2))
    else:
        # Simple table view
        if "business_research" in results:
            br = results["business_research"]
            print("\n== Business Research ==")
            print(f"Status: {br.get('status', 'unknown')}")
            if br.get("stdout"):
                print(br["stdout"][:500] + ("..." if len(br["stdout"]) > 500 else ""))

        if "knowledge_rag" in results:
            print("\n== Top Knowledge Hubs ==")
            for item in results["knowledge_rag"][:5]:
                if "error" not in item:
                    print(f"  • {item['name']} (degree: {item['degree']})")

        if "top_hub" in results:
            hub_data = results["top_hub"]["hub"]
            ctx = results["top_hub"]["context"]
            print("\n== Top-Hub Insight ==")
            print(f"  Hub: {hub_data.get('name', 'N/A')} (degree: {hub_data.get('degree', 0)})")
            print(f"  Context links: {len(ctx)}")
            for c in ctx[:5]:
                print(f"    -[{c['rel']}]-> {c['target']}")

if __name__ == "__main__":
    app()
```
