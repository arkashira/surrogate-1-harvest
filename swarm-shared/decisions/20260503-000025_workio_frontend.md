# workio / frontend

## Final Synthesized Implementation

### Diagnosis (merged)
- **No optimistic UI** → perceived latency and ambiguous spinners.
- **No local queue** → punches lost offline; no retry path.
- **No idempotency** → double-taps and duplicate webhooks create duplicates.
- **No rollback** → UI can remain in “success” when backend rejects.
- **No state machine** → invalid transitions (e.g., clock-in while already IN) and no persisted last-known/pending state across reloads.

### Scope
- Frontend-only changes.
- Single, focused hook + button component.
- Optional persistence for queue + last-known punch to survive reloads.
- Exponential-backoff retries with jitter, capped attempts, and clear user actions.

## 1. Hook: `useClockQueue`

File: `workio/src/features/clock/useClockQueue.ts`

```ts
// workio/src/features/clock/useClockQueue.ts
import { useState, useCallback, useRef, useEffect } from 'react';

type PunchType = 'IN' | 'OUT';
type Status = 'idle' | 'pending' | 'success' | 'error';

interface PendingPunch {
  id: string; // client-side idempotency key
  type: PunchType;
  timestamp: number;
  retries: number;
}

interface PunchResult {
  ok: boolean;
  error?: string;
}

const MAX_RETRIES = 3;
const STORAGE_KEY = 'clockQueue:v1';
const LAST_PUNCH_KEY = 'lastClockPunch:v1';

function loadPersisted<T>(key: string, fallback: T): T {
  try {
    const raw = localStorage.getItem(key);
    return raw ? JSON.parse(raw) : fallback;
  } catch {
    return fallback;
  }
}

function persist<T>(key: string, value: T) {
  try {
    localStorage.setItem(key, JSON.stringify(value));
  } catch {
    // ignore storage failures
  }
}

export function useClockQueue(
  punchFn: (type: PunchType, idempotencyKey: string) => Promise<PunchResult>,
  options?: { persistAcrossReloads?: boolean }
) {
  const persistEnabled = options?.persistAcrossReloads ?? true;

  const [status, setStatus] = useState<Status>('idle');
  const [lastError, setLastError] = useState<string | null>(null);
  const queueRef = useRef<PendingPunch[]>([]);
  const processingRef = useRef(false);
  const mountedRef = useRef(false);

  // Persisted last-known punch for UI decisions (e.g., show next expected action)
  const [lastPunch, setLastPunch] = useState<{ type: PunchType; timestamp: number } | null>(() => {
    if (!persistEnabled) return null;
    return loadPersisted<{ type: PunchType; timestamp: number } | null>(LAST_PUNCH_KEY, null);
  });

  // Hydrate queue from storage once on mount
  useEffect(() => {
    mountedRef.current = true;
    if (persistEnabled) {
      const saved = loadPersisted<PendingPunch[]>(STORAGE_KEY, []);
      queueRef.current = Array.isArray(saved) ? saved : [];
      if (queueRef.current.length > 0 && !processingRef.current) {
        processQueue();
      }
    }
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [persistEnabled]);

  function saveQueue() {
    if (persistEnabled) persist(STORAGE_KEY, queueRef.current);
  }

  function saveLastPunch(type: PunchType) {
    const record = { type, timestamp: Date.now() };
    setLastPunch(record);
    if (persistEnabled) persist(LAST_PUNCH_KEY, record);
  }

  const processQueue = useCallback(async () => {
    if (processingRef.current || queueRef.current.length === 0) return;
    processingRef.current = true;
    const punch = queueRef.current[0];

    setStatus('pending');
    setLastError(null);

    try {
      const result = await punchFn(punch.type, punch.id);
      if (result.ok) {
        queueRef.current.shift();
        saveQueue();
        saveLastPunch(punch.type);
        setStatus('success');
        setTimeout(() => setStatus('idle'), 2000);
      } else {
        punch.retries += 1;
        if (punch.retries >= MAX_RETRIES) {
          queueRef.current.shift();
          saveQueue();
          setStatus('error');
          setLastError(result.error || 'Failed to record punch');
        } else {
          saveQueue();
          const delay = Math.min(1000 * 2 ** punch.retries + Math.random() * 300, 10000);
          setTimeout(processQueue, delay);
          processingRef.current = false;
          return;
        }
      }
    } catch (err: any) {
      punch.retries += 1;
      if (punch.retries >= MAX_RETRIES) {
        queueRef.current.shift();
        saveQueue();
        setStatus('error');
        setLastError(err.message || 'Network error');
      } else {
        saveQueue();
        const delay = Math.min(1000 * 2 ** punch.retries + Math.random() * 300, 10000);
        setTimeout(processQueue, delay);
        processingRef.current = false;
        return;
      }
    }

    processingRef.current = false;
    if (queueRef.current.length > 0) processQueue();
  }, [punchFn, persistEnabled]);

  const enqueue = useCallback(
    (type: PunchType) => {
      const id = `${type}-${Date.now()}-${crypto.randomUUID()}`;
      const punch: PendingPunch = { id, type, timestamp: Date.now(), retries: 0 };
      queueRef.current.push(punch);
      saveQueue();
      setStatus('pending');
      processQueue();
    },
    [processQueue, persistEnabled]
  );

  const retryHead = useCallback(() => {
    if (queueRef.current.length === 0 && status === 'error') {
      // If nothing in queue but we're in error, re-enqueue last attempted action from lastPunch
      if (lastPunch) {
        enqueue(lastPunch.type === 'IN' ? 'OUT' : 'IN');
      }
    } else {
      processQueue();
    }
  }, [status, lastPunch, enqueue, processQueue]);

  const resetError = useCallback(() => {
    setStatus('idle');
    setLastError(null);
  }, []);

  const clearQueue = useCallback(() => {
    queueRef.current = [];
    saveQueue();
    setStatus('idle');
    setLastError(null);
  }, [persistEnabled]);

  return {
    status,
    lastError,
    lastPunch,
    enqueue,
    resetError,
    retryHead,
    clearQueue,
  };
}
```
