# workio / ops

## 1. Diagnosis

- **No idempotency enforcement on LINE webhook ingestion** — identical `X-Line-Signature` and body cause duplicate clock-in/out events under retries or double-taps.
- **Race-prone upsert in webhook handler** — read-then-write (`findOne` → update/insert) is non-atomic; concurrent replays corrupt punch state and balances.
- **Missing transactional boundary** — punch write and derived state (daily summary, balance) are not atomic; partial updates leave inconsistent reports.
- **No replay protection window** — legitimate retries after timeouts can be indistinguishable from duplicates after short TTLs.
- **No observability for dedupe decisions** — missing request-id and idempotency-key logging makes debugging duplicates and retries slow.

## 2. Proposed change

- **File:** `workio/server/src/routes/line/webhook.ts` (or equivalent handler module)
- **Scope:** Add idempotency table + middleware to enforce exactly-once processing per `X-Line-Signature` (or derived request hash) within a 10-minute window; wrap punch logic in a DB transaction; log decisions.

`server/src/middleware/lineIdempotency.ts`
```ts
import { Request, Response, NextFunction } from 'express';
import { pool } from '../db';
import crypto from 'crypto';

const IDEMPOTENCY_TTL_MS = 10 * 60 * 1000; // 10m

export function lineIdempotency() {
  return async (req: Request, res: Response, next: NextFunction) => {
    // Prefer client-supplied idempotency key; fallback to signature+body hash
    const idempotencyKey =
      req.headers['idempotency-key'] as string ||
      req.headers['x-line-signature'] as string ||
      crypto.createHash('sha256').update(JSON.stringify(req.body)).digest('hex');

    const requestHash = crypto
      .createHash('sha256')
      .update(JSON.stringify(req.body))
      .digest('hex');

    const client = await pool.connect();
    try {
      await client.query('BEGIN');

      // Lock row for update to prevent races
      const existing = await client.query(
        `SELECT handler_result, created_at
           FROM line_webhook_idempotency
          WHERE idempotency_key = $1
            FOR UPDATE`,
        [idempotencyKey]
      );

      if (existing.rowCount && existing.rowCount > 0) {
        const row = existing.rows[0];
        const age = Date.now() - new Date(row.created_at).getTime();

        // Within TTL: return cached result (idempotent replay)
        if (age < IDEMPOTENCY_TTL_MS) {
          await client.query('ROLLBACK');
          req.idempotency = { hit: true, key: idempotencyKey, cached: true };
          // If previous handler stored a response shape, replay it
          if (row.handler_result && row.handler_result.status) {
            return res.status(row.handler_result.status).json(row.handler_result.body || {});
          }
          return res.status(200).json({ ok: true, replayed: true });
        }

        // Expired: allow overwrite (treat as new logical request)
        await client.query(
          `DELETE FROM line_webhook_idempotency WHERE idempotency_key = $1`,
          [idempotencyKey]
        );
      }

      // Reserve key early to block concurrent replays
      await client.query(
        `INSERT INTO line_webhook_idempotency (idempotency_key, request_hash, handler_result, created_at)
         VALUES ($1, $2, $3, NOW())`,
        [idempotencyKey, requestHash, JSON.stringify({ status: 202, body: { processing: true } })]
      );

      await client.query('COMMIT');

      // Attach metadata and client for handler transaction reuse
      req.idempotency = { hit: false, key: idempotencyKey, client };
      req.idempotencyClient = client;
      next();
    } catch (err) {
      await client.query('ROLLBACK');
      client.release();
      next(err);
    }
  };
}
```

`server/src/routes/line/webhook.ts` (updated handler)
```ts
import { Router, Request, Response } from 'express';
import { lineIdempotency } from '../../middleware/lineIdempotency';
import { pool } from '../../db';

const router = Router();

router.post('/webhook/line', lineIdempotency(), async (req: Request, res: Response) => {
  const client = (req as any).idempotencyClient;
  const idempotencyKey = (req as any).idempotency?.key;

  if (!client || !idempotencyKey) {
    return res.status(500).json({ error: 'idempotency setup failed' });
  }

  try {
    const events = req.body.events || [];
    const results = [];

    for (const event of events) {
      // Process punch logic within same transaction
      const punchResult = await processPunchEvent(client, event);
      results.push(punchResult);
    }

    const responseBody = { ok: true, results };

    // Update idempotency record with final result
    await client.query(
      `UPDATE line_webhook_idempotency
          SET handler_result = $1
        WHERE idempotency_key = $2`,
      [JSON.stringify({ status: 200, body: responseBody }), idempotencyKey]
    );

    await client.query('COMMIT');
    res.status(200).json(responseBody);
  } catch (err) {
    await client.query('ROLLBACK');
    // Keep idempotency row with error for observability (short TTL)
    try {
      await client.query(
        `UPDATE line_webhook_idempotency
            SET handler_result = $1
          WHERE idempotency_key = $2`,
        [JSON.stringify({ status: 500, error: String(err) }), idempotencyKey]
      );
      await client.query('COMMIT');
    } catch (_) {
      // best-effort
    }
    res.status(500).json({ error: 'processing failed' });
  } finally {
    client.release();
  }
});

async function processPunchEvent(client: any, event: any) {
  // Example: validate, upsert punch, update daily summary in one tx
  const { userId, type, timestamp, location } = normalizeEvent(event);

  // Use deterministic punch idempotency (e.g., userId + date + type window)
  // and enforce business rules transactionally
  await client.query(
    `INSERT INTO punches (user_id, type, ts, location, source)
     VALUES ($1, $2, $3, $4, $5)
     ON CONFLICT (user_id, type, date_trunc('hour', ts))
     DO UPDATE SET ts = EXCLUDED.ts, location = EXCLUDED.location
     RETURNING *`,
    [userId, type, timestamp, location, 'line']
  );

  // Update daily totals in same tx
  await client.query(
    `UPDATE daily_summary
        SET total_hours = compute_daily_total($1, $2::date)
      WHERE user_id = $1 AND day = $2::date`,
    [userId, timestamp]
  );

  return { userId, type, ts: timestamp };
}
