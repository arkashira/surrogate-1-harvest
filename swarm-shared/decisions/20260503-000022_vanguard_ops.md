# vanguard / ops

## 1. Diagnosis
- No persisted HF file manifest per date-folder: every training run re-lists repo via authenticated API (wastes quota, risks 429).
- Training script still uses `load_dataset(streaming=True)` on heterogeneous repos → pyarrow CastError on mixed schemas.
- No CDN-only data path: training incurs API auth checks during data loading instead of using `resolve/main/` CDN bypass.
- Lightning Studio reuse not enforced: each run recreates studio and burns 80+ hr/mo quota.
- No idle-stop guard: Lightning idle timeout kills training mid-run.

## 2. Proposed change
Create `/opt/axentx/vanguard/ops/train_cdn_manifest.py` and update `/opt/axentx/vanguard/ops/launch_training.py` (or equivalent launcher) to:
- Single authenticated `list_repo_tree` per `(repo, dateFolder)` → write `manifests/{repo}__{dateFolder}.json`.
- Embed manifest in training script; use only CDN URLs (`https://huggingface.co/datasets/.../resolve/main/...`) for data fetches.
- Reuse running Lightning Studio; restart only if stopped.
- Project to `{prompt, response}` at parse time (avoid schema issues).

#!/usr/bin/env python3
"""
Generate and persist a CDN-only file manifest for a HF dataset repo/dateFolder.
Usage:
  python train_cdn_manifest.py <repo> <dateFolder> [--out-dir ./manifests]
"""
import json
import os
import sys
from datetime import datetime
from pathlib import Path

try:
    from huggingface_hub import HfApi
except ImportError:
    print("Install: pip install huggingface_hub")
    sys.exit(1)

HF_API = HfApi()

def list_datefolder_files(repo_id: str, date_folder: str, out_dir: Path):
    # Single API call: non-recursive per folder (avoids 100x pagination)
    entries = HF_API.list_repo_tree(repo_id=repo_id, path=date_folder, recursive=False)
    files = [e.rfilename for e in entries if e.type == "file"]

    manifest = {
        "repo_id": repo_id,
        "date_folder": date_folder,
        "generated_at_utc": datetime.utcnow().isoformat() + "Z",
        "files": sorted(files),
        "cdn_base": f"https://huggingface.co/datasets/{repo_id}/resolve/main",
        "file_urls": [f"{cdn_base}/{f}" for f in sorted(files)]
    }

    out_dir.mkdir(parents=True, exist_ok=True)
    out_path = out_dir / f"{repo_id.replace('/', '__')}__{date_folder}.json"
    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"Manifest written: {out_path}")
    return manifest

if __name__ == "__main__":
    if len(sys.argv) < 3:
        print("Usage: train_cdn_manifest.py <repo> <dateFolder> [--out-dir ./manifests]")
        sys.exit(1)

    repo = sys.argv[1]
    datef = sys.argv[2]
    out_d = Path("./manifests")
    for i, a in enumerate(sys.argv[3:]):
        if a == "--out-dir" and i + 4 <= len(sys.argv):
            out_d = Path(sys.argv[i + 4])
            break

    list_datefolder_files(repo, datef, out_d)
```

#!/usr/bin/env python3
"""
Launch Lightning training with CDN-only data and studio reuse.
"""
import json
import os
import sys
from pathlib import Path

try:
    import lightning as L
    from lightning.app import LightningFlow, LightningWork
except ImportError:
    print("Install: pip install lightning")
    sys.exit(1)

MANIFEST_DIR = Path(__file__).parent / "manifests"

def load_manifest(repo: str, date_folder: str):
    slug = repo.replace("/", "__")
    p = MANIFEST_DIR / f"{slug}__{date_folder}.json"
    if not p.is_file():
        raise FileNotFoundError(f"Manifest missing: {p}. Run train_cdn_manifest.py first.")
    return json.loads(p.read_text())

def find_running_studio(name: str):
    # Reuse running studio to save quota
    try:
        from lightning.app import Teamspace
        for s in Teamspace.studios:
            if getattr(s, "name", None) == name and getattr(s, "status", None) == "Running":
                return s
    except Exception:
        pass
    return None

class SurrogateTrainer(LightningFlow):
    def __init__(self, repo: str, date_folder: str, script_path: str):
        super().__init__()
        self.repo = repo
        self.date_folder = date_folder
        self.script_path = script_path
        self.manifest = load_manifest(repo, date_folder)
        self.studio = None

    def run(self):
        studio_name = f"vanguard-{self.repo.replace('/', '-')}-{self.date_folder}"
        existing = find_running_studio(studio_name)

        if existing:
            self.studio = existing
            print(f"Reusing running studio: {studio_name}")
        else:
            # Create once; prefer L40S if free tier, else fallback
            from lightning.app import LightningStudio, Machine
            machine = Machine.L40S  # free tier fallback; H200 requires paid cloud
            self.studio = LightningStudio(
                name=studio_name,
                script_path=self.script_path,
                machine=machine,
                create_ok=True,
            )

        # Guard against idle stop: if stopped, restart
        if getattr(self.studio, "status", None) == "stopped":
            print("Studio stopped. Restarting...")
            from lightning.app import Machine
            self.studio.start(machine=Machine.L40S)

        # Pass CDN-only file list to training via env var
        os.environ["VANGUARD_CDN_MANIFEST"] = json.dumps(self.manifest)
        # The training script should read this and fetch via CDN URLs only.

if __name__ == "__main__":
    # Example usage:
    # python launch_training.py my-dataset 2026-04-29 /opt/axentx/vanguard/train/train.py
    if len(sys.argv) < 4:
        print("Usage: launch_training.py <repo> <dateFolder> <train_script>")
        sys.exit(1)

    repo, datef, script = sys.argv[1], sys.argv[2], sys.argv[3]
    app = SurrogateTrainer(repo, datef, script)
    # In real usage, run via Lightning App; here we just trigger run once for ops check
    app.run()
```
