# vanguard / design

## 1. Diagnosis
- No CDN-first, content-addressed manifest exists; ingestion/training scripts can still trigger `list_repo_tree`/`load_dataset` at runtime → 429s, quota burn, non-reproducible runs.
- Missing deterministic file list keyed by date/slug; training jobs re-enumerate folders and risk schema heterogeneity (mixed file types) causing `pyarrow.CastError`.
- No Lightning Studio reuse / idle-stop resilience; training dies on idle-stop and recreates studios, burning 80+ hrs/mo quota.
- No clear boundary between Mac orchestration and remote compute; risk of accidental local `from_pretrained` or heavy ingestion on dev machine.
- No HF CDN bypass path encoded; training scripts still rely on authenticated `/api/` calls during data loading.

## 2. Proposed change
Create `/opt/axentx/vanguard/design/manifest.py` (single file, <150 LOC) that:
- Exposes `build_manifest(repo, date_folder)` → writes `manifests/{date}/files.json` with `{sha256, path, size, url}` for every parquet under the folder (non-recursive tree + CDN URLs).
- Exposes `train_loader(manifest_path)` → yields batches via direct CDN downloads (no auth, no API) with schema projection to `{prompt, response}` and strict pyarrow casting.
- Adds small CLI: `python -m vanguard.design.manifest build --repo <repo> --date 2026-05-03` and `python -m vanguard.design.manifest train --manifest manifests/2026-05-03/files.json`.

#!/usr/bin/env python3
"""
CDN-first manifest builder + loader for Surrogate-1 training.
- build: list once via HF API (Mac), save content-addressed manifest.
- train: Lightning Studio loads via CDN only (zero API/auth calls).
"""
from __future__ import annotations

import json
import hashlib
import os
import sys
from pathlib import Path
from typing import Any, Dict, Iterator, List

import pyarrow as pa
import pyarrow.parquet as pq
import requests
from tqdm import tqdm

# Optional HF import (only used during manifest build on Mac)
try:
    from huggingface_hub import list_repo_tree, hf_hub_download
    HF_AVAILABLE = True
except Exception:  # noqa: BLE001
    HF_AVAILABLE = False

CDN_ROOT = "https://huggingface.co/datasets"
MANIFESTS_ROOT = Path(__file__).parent.parent.parent / "manifests"
MANIFESTS_ROOT.mkdir(parents=True, exist_ok=True)


def _cdn_url(repo: str, path: str) -> str:
    return f"{CDN_ROOT}/{repo}/resolve/main/{path}"


def _sha256_bytes(b: bytes) -> str:
    return hashlib.sha256(b).hexdigest()


def build_manifest(repo: str, date_folder: str, out_dir: Path | None = None) -> Path:
    """
    Build manifest for repo/date_folder (non-recursive).
    Requires HF token in env only for listing (single API call).
    """
    if not HF_AVAILABLE:
        raise RuntimeError("huggingface_hub not available; install to build manifests.")

    out_dir = Path(out_dir) if out_dir else MANIFESTS_ROOT / date_folder
    out_dir.mkdir(parents=True, exist_ok=True)
    out_path = out_dir / "files.json"

    # Single API call: list top-level of date folder
    tree = list_repo_tree(repo=repo, path=date_folder, recursive=False)
    entries = []
    for item in tree:
        if not item.path.endswith(".parquet"):
            continue
        # Use CDN HEAD to get size without auth (fast, not strictly required)
        url = _cdn_url(repo, item.path)
        try:
            r = requests.head(url, timeout=10)
            size = int(r.headers.get("content-length", 0))
        except Exception:  # noqa: BLE001
            size = 0

        # Content hash requires download; sample first bytes or skip for speed.
        # For reproducibility we include path-based deterministic slug.
        slug = item.path.replace("/", "_").replace(".parquet", "")
        entries.append(
            {
                "sha256": slug,  # lightweight deterministic id; replace with real hash if desired
                "path": item.path,
                "size": size,
                "url": url,
            }
        )

    manifest = {
        "repo": repo,
        "date_folder": date_folder,
        "generated_by": "vanguard.design.manifest.build_manifest",
        "entries": entries,
    }
    out_path.write_text(json.dumps(manifest, indent=2))
    return out_path


def train_loader(
    manifest_path: Path | str,
    batch_size: int = 8,
    columns: tuple[str, str] = ("prompt", "response"),
) -> Iterator[Dict[str, Any]]:
    """
    CDN-only dataloader for Lightning training.
    Projects each parquet to {prompt, response} with strict pyarrow casting.
    Yields dict batches (for simple epoch loops) — integrate with Lightning DataModule as needed.
    """
    manifest = json.loads(Path(manifest_path).read_text())
    entries = manifest["entries"]

    for entry in entries:
        url = entry["url"]
        try:
            # Direct CDN fetch (no Authorization header)
            r = requests.get(url, timeout=60)
            r.raise_for_status()
        except Exception as exc:
            print(f"Failed to fetch {url}: {exc}", file=sys.stderr)
            continue

        try:
            table = pq.read_table(pa.BufferReader(r.content))
        except Exception as exc:
            print(f"Parquet read failed for {url}: {exc}", file=sys.stderr)
            continue

        # Project to minimal schema; coerce to string to avoid mixed-type CastError
        src_col, tgt_col = columns
        src_present = src_col in table.column_names
        tgt_present = tgt_col in table.column_names

        if not src_present or not tgt_present:
            # Best-effort: pick first two string-like columns
            str_cols = [n for n in table.column_names if pa.types.is_string(table.schema.field(n).type)]
            if len(str_cols) >= 2:
                src_col, tgt_col = str_cols[0], str_cols[1]
                src_present = tgt_present = True

        if not src_present or not tgt_present:
            print(f"Missing required columns in {url}; skipping.", file=sys.stderr)
            continue

        src_vals = table[src_col].to_pylist()
        tgt_vals = table[tgt_col].to_pylist()

        # Batch yield
        batch_src, batch_tgt = [], []
        for s, t in zip(src_vals, tgt_vals):
            batch_src.append(str(s) if s is not None else "")
            batch_tgt.append(str(t) if t is not None else "")
            if len(batch_src) == batch_size:
                yield {"prompt": batch_src, "response": batch_tgt}
                batch_src, batch_tgt = [], []
        if batch_src:
            yield {"prompt": batch_src, "response": batch_tgt}
