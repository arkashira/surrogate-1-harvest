# airship / design

## Final Consolidated Implementation (Best of Both Candidates)

**Highest-Value Improvement**: Resilient Surrogate-1 Training Pipeline that eliminates HF API 429s and Lightning idle-stop quota waste via CDN-only data loading, manifest-based orchestration, and studio reuse logic. Ships in ≤2 hours with single-file changes.

---

## Architecture Overview

1. **Pre-generated CDN Manifest** (Mac orchestration) → `batches/mirror-merged/{date}/manifest.json`
2. **Training script** reads manifest, downloads via CDN URLs only (`/resolve/main/...`) with retry/backoff
3. **Lightning launcher** checks for running studios, reuses running instances, auto-restarts on idle-stop

## Implementation

### 1. Manifest Generator (`scripts/generate_manifest.py`)

```python
#!/usr/bin/env python3
"""
Generate CDN-only manifest for Surrogate-1 training.
Usage: python generate_manifest.py --repo org/surrogate-data --date 2026-04-29
"""
import argparse
import json
import sys
from datetime import datetime
from pathlib import Path

try:
    from huggingface_hub import list_repo_tree
except ImportError:
    print("pip install huggingface_hub")
    sys.exit(1)

CDN_BASE = "https://huggingface.co/datasets"

def build_manifest(repo: str, date: str, out_dir: str = "batches/mirror-merged"):
    folder_path = f"{date}"
    print(f"Listing {repo}/{folder_path} (non-recursive)...")

    # Single API call, non-recursive to avoid pagination
    tree = list_repo_tree(repo=repo, path=folder_path, recursive=False)

    manifest = {
        "repo": repo,
        "date": date,
        "generated_at": datetime.utcnow().isoformat() + "Z",
        "files": []
    }

    for item in tree:
        if item.rfilename.endswith(".parquet"):
            cdn_url = f"{CDN_BASE}/{repo}/resolve/main/{folder_path}/{item.rfilename}"
            entry = {
                "filename": item.rfilename,
                "cdn_url": cdn_url,
                "size": getattr(item, "size", None),
                "lfs": getattr(item, "lfs", None)
            }
            manifest["files"].append(entry)

    out_path = Path(out_dir) / date
    out_path.mkdir(parents=True, exist_ok=True)
    manifest_file = out_path / "manifest.json"
    manifest_file.write_text(json.dumps(manifest, indent=2))
    print(f"Manifest written: {manifest_file} ({len(manifest['files'])} files)")
    return manifest_file

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Generate CDN manifest for training")
    parser.add_argument("--repo", default="org/surrogate-data", help="HF dataset repo")
    parser.add_argument("--date", required=True, help="Date folder (YYYY-MM-DD)")
    parser.add_argument("--out-dir", default="batches/mirror-merged", help="Output dir")
    args = parser.parse_args()

    build_manifest(args.repo, args.date, args.out_dir)
```

### 2. Training Script (`surrogate/train.py`)

```python
import json
import argparse
import requests
from pathlib import Path
from lightning import Lightning, Studio, Teamspace, Machine
import pyarrow.parquet as pq
from io import BytesIO
from tenacity import retry, stop_after_attempt, wait_exponential

@retry(stop=stop_after_attempt(5), wait=wait_exponential(multiplier=1, min=2, max=30))
def cdn_fetch(url: str) -> bytes:
    resp = requests.get(url, timeout=30)
    resp.raise_for_status()
    return resp.content

def load_from_manifest(manifest_path: str):
    manifest = json.loads(Path(manifest_path).read_text())
    for entry in manifest["files"]:
        data = cdn_fetch(entry["cdn_url"])
        table = pq.read_table(BytesIO(data))
        # Project to {prompt, response} only
        yield table.select(["prompt", "response"]).to_pylist()

def get_or_create_studio(name: str = "surrogate-training"):
    teamspace = Teamspace()
    for s in teamspace.studios:
        if s.name == name and s.status == "Running":
            print(f"Reusing running studio: {s.id}")
            return s
    print(f"Creating studio: {name}")
    return Studio(
        name=name,
        machine=Machine.L40S,
        create_ok=True
    )

def train(manifest_path: str):
    studio = get_or_create_studio()

    # Guard against idle-stop kill
    if studio.status != "Running":
        print(f"Studio stopped (status={studio.status}), restarting...")
        studio.start(machine=Machine.L40S)

    # Lightning run with CDN-only data
    studio.run(
        script="train_core.py",
        arguments={"manifest": manifest_path},
        sync_logs=True
    )

if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--manifest", required=True, help="Path to manifest.json")
    args = parser.parse_args()
    train(args.manifest)
```
