# Costinel / ops

## Final Implementation Plan — Top-Hub Signal Panel (CDN-first, <2h)

### What we ship
- A **non-blocking Top-Hub Signal Panel** mounted on the Costinel dashboard.
- Defaults to hub **MOC** (configurable via `HUB_NAME`).
- Shows: hub title, short description, top 5 signals (anomalies/recommendations), last refresh, and a “View full hub” link.
- **CDN-first data strategy**: pre-listed file paths embedded at build time; runtime fetches use `https://huggingface.co/datasets/.../resolve/main/...` (no API/auth, bypasses 429).
- Telemetry-aware: respects idle/stop (Lightning), never blocks dashboard render, graceful fallbacks.

## 2. File changes

### 2.1 CDN fetcher + file-list cache (`lib/hub-cdn.ts`)

```ts
// lib/hub-cdn.ts
import { HUB_DATA_REPO, HUB_NAME } from '$env/static/public';

const REPO = HUB_DATA_REPO || 'axentx/costinel-hubs';
const BASE = `https://huggingface.co/datasets/${REPO}/resolve/main`;

// Build-time embedded file list (generated by ops script)
// Example: import list from '$lib/buildtime/hub-file-list.json';
// For now, lightweight runtime fallback:
const KNOWN_HUB_FILES: Record<string, string> = {
  MOC: 'hubs/MOC/signals.json',
  // add others as needed
};

export type Signal = {
  id: string;
  title: string;
  description: string;
  severity: 'low' | 'medium' | 'high' | 'critical';
  recommendation: string;
  service?: string;
  region?: string;
};

export type HubData = {
  hub: string;
  title: string;
  description: string;
  signals: Signal[];
  generated_at: string;
};

export async function fetchHubData(hubName: string): Promise<HubData | null> {
  const filePath = KNOWN_HUB_FILES[hubName] || `${HUB_DATA_PATH}/${hubName}/signals.json`;
  const url = `${BASE}/${filePath}`;

  try {
    const res = await fetch(url, { cache: 'no-store' });
    if (!res.ok) throw new Error(`CDN fetch failed: ${res.status}`);
    return await res.json();
  } catch (err) {
    console.warn('[HubCDN]', err);
    return null;
  }
}
```

### 2.2 Panel component (`components/TopHubSignalPanel.tsx`)

```tsx
import { useEffect, useState } from 'react';
import { fetchHubData, type HubData, type Signal } from '$lib/hub-cdn';
import './TopHubSignalPanel.css';

const HUB_NAME = import.meta.env.VITE_HUB_NAME || 'MOC';

export default function TopHubSignalPanel() {
  const [data, setData] = useState<HubData | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    let mounted = true;
    setLoading(true);

    fetchHubData(HUB_NAME)
      .then((result) => {
        if (!mounted) return;
        if (result) {
          setData(result);
          setError(null);
        } else {
          throw new Error('No data returned');
        }
      })
      .catch((err) => {
        if (!mounted) return;
        setError(err.message);
      })
      .finally(() => {
        if (!mounted) return;
        setLoading(false);
      });

    return () => {
      mounted = false;
    };
  }, [HUB_NAME]);

  const severityColor = (s: Signal['severity']) => {
    switch (s) {
      case 'critical': return 'var(--severity-critical, #b91c1c)';
      case 'high': return 'var(--severity-high, #f59e0b)';
      case 'medium': return 'var(--severity-medium, #3b82f6)';
      default: return 'var(--severity-low, #10b981)';
    }
  };

  if (loading) {
    return (
      <div className="hub-panel loading">
        <div className="hub-header">
          <h3>{HUB_NAME}</h3>
        </div>
        <div className="hub-body skeleton">Loading signals...</div>
      </div>
    );
  }

  if (error || !data) {
    return (
      <div className="hub-panel error">
        <div className="hub-header">
          <h3>{HUB_NAME}</h3>
        </div>
        <div className="hub-body">
          <p className="error-msg">Unable to load hub signals. Using fallback insights.</p>
          <ul>
            <li>Check cloud cost anomalies in the Cost dashboard.</li>
            <li>Review RI coverage for top services.</li>
          </ul>
        </div>
      </div>
    );
  }

  const topSignals = data.signals.slice(0, 5);

  return (
    <div className="hub-panel">
      <div className="hub-header">
        <div>
          <h3>{data.title || HUB_NAME}</h3>
          <p className="hub-desc">{data.description}</p>
        </div>
        <div className="hub-meta">
          <small>Updated {new Date(data.generated_at).toLocaleString()}</small>
        </div>
      </div>

      <div className="hub-body">
        <ul className="signals-list">
          {topSignals.map((s) => (
            <li key={s.id} className="signal-item">
              <div className="signal-main">
                <span className="signal-badge" style={{ backgroundColor: severityColor(s.severity) }}>
                  {s.severity}
                </span>
                <div className="signal-text">
                  <div className="signal-title">{s.title}</div>
                  {(s.service || s.region) && (
                    <div className="signal-context">
                      {[s.service, s.region].filter(Boolean).join(' · ')}
                    </div>
                  )}
                  <div className="signal-desc">{s.description}</div>
                  <div className="signal-rec">Recommendation: {s.recommendation}</div>
                </div>
              </div>
            </li>
          ))}
        </ul>

        <div className="hub-actions">
          <a href={`/hubs/${HUB_NAME}`} className="btn-link">
            View full hub →
          </a>
        </div>
      </div>
    </div>
  );
}
```
