# workio / design

## 1. Diagnosis

- Missing **database-level uniqueness guard** for open punches: no constraint preventing multiple `clock_out_at IS NULL` rows per user per calendar day under concurrency or LINE retries.
- App-layer `find-then-insert` is race-prone; duplicate punches can occur when LINE webhook retries or users double-tap.
- No **idempotency key** on webhook handler to deduplicate repeated deliveries from LINE (same `timestamp` + `userId` can arrive multiple times).
- Punch timestamp handling uses JS `Date` without explicit timezone normalization; edge cases around midnight can place punches in wrong calendar day for tenant timezone.
- No **audit trail** for critical state changes (punch create/update) to support debugging and compliance.

## 2. Proposed change

- **Scope**: `server/src/db/schema.sql` + `server/src/db/migrations/` + `server/src/routes/punch.ts`
- Add partial unique index for open punches, idempotency table + key on webhook, timezone-aware calendar-day normalization, and minimal audit log for punches.

## 3. Implementation

### 3.2 Webhook handler with idempotency and safe upsert

File: `server/src/routes/punch.ts` (or wherever LINE webhook is handled)

```ts
import { Request, Response } from 'express';
import { pool } from '../db';
import { v4 as uuidv4 } from 'uuid';

// Helper: build idempotency key from LINE event
function buildIdempotencyKey(event: any): string {
  // Use userId + timestamp + type to detect duplicates across retries
  const userId = event.source?.userId;
  const ts = event.timestamp;
  const type = event.type;
  return `line:${userId}:${ts}:${type}`;
}

// POST /webhook/line (simplified)
export async function handleLineWebhook(req: Request, res: Response) {
  const body = req.body;
  const events = body.events || [];

  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    for (const ev of events) {
      const key = buildIdempotencyKey(ev);
      const userId = ev.source?.userId;

      // Idempotency check
      const idem = await client.query(
        'SELECT 1 FROM webhook_idempotency WHERE idempotency_key = $1',
        [key]
      );
      if (idem.rows.length > 0) {
        // Already processed — skip but continue processing other events
        continue;
      }

      // Record idempotency first (best-effort ordering)
      await client.query(
        'INSERT INTO webhook_idempotency (idempotency_key, event_type, user_id) VALUES ($1, $2, $3)',
        [key, ev.type, userId]
      );

      if (ev.type === 'message' && ev.message?.type === 'text') {
        const text = ev.message.text.trim().toLowerCase();

        if (text === 'clock in' || text === 'clock out') {
          const now = new Date(); // UTC
          const isClockIn = text === 'clock in';

          // Use DB-level upsert pattern with CTE to enforce one-open-per-day safely
          // If clock_in: insert only if no open punch exists for this user today (UTC day)
          // If clock_out: update the open punch for this user today
          if (isClockIn) {
            await client.query(`
              INSERT INTO punches (user_id, clock_in_at, clock_in_lat, clock_in_lng, clock_in_source)
              SELECT $1, $2, $3, $4, 'line'
              WHERE NOT EXISTS (
                SELECT 1 FROM punches
                WHERE user_id = $1
                  AND clock_out_at IS NULL
                  AND date_trunc('day', clock_in_at AT TIME ZONE 'UTC') = date_trunc('day', $2 AT TIME ZONE 'UTC')
              )
              RETURNING id
            `, [userId, now, null, null]);
          } else {
            // clock out: close the most recent open punch for this user
            const result = await client.query(`
              UPDATE punches
              SET clock_out_at = $1, clock_out_lat = $2, clock_out_lng = $3, clock_out_source = 'line'
              WHERE id = (
                SELECT id FROM punches
                WHERE user_id = $4
                  AND clock_out_at IS NULL
                ORDER BY clock_in_at DESC
                LIMIT 1
              )
              RETURNING id
            `, [now, null, null, userId]);

            if (result.rowCount === 0) {
              // No open punch to close — optionally log or notify user
              console.warn(`No open punch to close for user ${userId}`);
            }
          }
        }
      }
    }

    await client.query('COMMIT');
    res.status(200).json({ ok: true });
  } catch (err) {
    await client.query('ROLLBACK');
    console.error('Webhook processing failed', err);
    res.status(500).json({ error: 'processing_failed' });
  } finally {
    client.release();
  }
}
```
