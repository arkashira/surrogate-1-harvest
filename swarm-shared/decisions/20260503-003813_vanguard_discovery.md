## 2) `/opt/axentx/vanguard/scripts/build_manifest.py`

```python
#!/usr/bin/env python3
"""
Generate a CDN-only manifest for a repo+dateFolder.

Usage (recommended):
  python build_manifest.py \
    --repo datasets/username/repo \
    --date-folder mirror-merged/2026-04-29 \
    --out-dir /opt/axentx/vanguard/manifests

Also supports env vars:
  HF_REPO=datasets/username/repo DATE_FOLDER=mirror-merged/2026-04-29 python build_manifest.py

Outputs:
  {out_dir}/{repo_slug}/{dateFolder_slug}.json
  {out_dir}/{repo_slug}/{dateFolder_slug}_train_fragment.py
"""
import argparse
import hashlib
import json
import os
import sys
from datetime import datetime, timezone
from pathlib import Path

from huggingface_hub import HfApi

def parse_args():
    parser = argparse.ArgumentParser(description="Build CDN-only manifest for HF dataset repo.")
    parser.add_argument("--repo", help="HF dataset repo, e.g. datasets/username/repo")
    parser.add_argument("--date-folder", help="Folder inside repo, e.g. mirror-merged/2026-04-29")
    parser.add_argument("--out-dir", default="manifests", help="Base output directory")
    return parser.parse_args()

def main():
    args = parse_args()
    repo = args.repo or os.getenv("HF_REPO")
    date_folder = args.date_folder or os.getenv("DATE_FOLDER")

    if not repo or not date_folder:
        print("ERROR: --repo and --date-folder (or HF_REPO/DATE_FOLDER env) are required.", file=sys.stderr)
        sys.exit(1)

    api = HfApi()
    out_dir = Path(args.out_dir).expanduser().resolve()
    repo_slug = repo.replace("/", "_").replace("datasets_", "")
    date_slug = date_folder.replace("/", "_")
    out_dir = out_dir / repo_slug
    out_dir.mkdir(parents=True, exist_ok=True)

    manifest_path = out_dir / f"{date_slug}.json"
    frag_path = out_dir / f"{date_slug}_train_fragment.py"

    # Single non-recursive listing (paginated internally by SDK)
    tree = api.list_repo_tree(
        repo_id=repo,
        path=date_folder,
        repo_type="dataset",
        recursive=False,
    )

    files = []
    for f in tree:
        rn = f.rfilename
        if rn.endswith(".parquet"):
            files.append(
                {
                    "path": rn,
                    "size": getattr(f, "size", None),
                    "cdn_url": f"https://huggingface.co/datasets/{repo}/resolve/main/{rn}",
                }
            )

    if not files:
        print(f"WARNING: No parquet files found in {repo}/{date_folder}", file=sys.stderr)

    manifest = {
        "repo": repo,
        "date_folder": date_folder,
        "created_utc": datetime.now(timezone.utc).isoformat(),
        "files": sorted(files, key=lambda x: x["path"]),
        "cdn_base": f"https://huggingface.co/datasets/{repo}/resolve/main",
    }

    with open(manifest_path, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2, ensure_ascii=False)

    # Training fragment for zero-auth imports
    cdn_urls = [item["cdn_url"] for item in manifest["files"]]
    train_fragment = f"""# Auto-generated by build_manifest.py
# CDN-only file list (no HF API/auth required during training).
MANIFEST = {json.dumps(manifest, indent=2)}
DATASET_FILES = {json.dumps(cdn_urls, indent=2)}

# Example:
# from datasets import load_dataset
# ds = load_dataset("parquet", data_files=DATASET_FILES, split="train")
"""

    with open(frag_path, "w", encoding="utf-8") as f:
        f.write(train_fragment.strip() + "\n")

    print(f"Manifest: {manifest_path}")
    print(f"Train fragment: {frag_path}")
    print(f"Files: {len(files)}")

if __name__ == "__main__":
    main()
```

## 4) `/opt/axentx/vanguard/scripts/reuse_studio.py`

```python
#!/usr/bin/env python3
"""
Reuse an existing Lightning Studio or start an L40S if stopped.
Avoids duplicate studios.

Usage:
  STUDIO_NAME=surrogate-train python reuse_studio.py
"""
import os
import sys

try:
    from lightning_sdk import Studio, Teamspace, Machine
except ImportError as e:
    print(f"ERROR: lightning-sdk not available: {e}", file=sys.stderr)
    sys.exit(1)

STUDIO_NAME = os.getenv("STUDIO_NAME", "surrogate-train")

def main():
    teamspace = Teamspace()
    studio = None

    for s in teamspace.studios:
        if s.name == STUDIO_NAME:
            studio = s
            break

    if studio is None:
        print(f"Creating studio '{STUDIO_NAME}' (L40S)")
        studio = Studio.create(
            name=STUDIO_NAME,
            machine=Machine.L40S,
            create_ok=True,
        )
    else:
        print(f"Found studio '{STUDIO_NAME}' (status={studio.status})")

    if studio.status != "Running":
        print("Starting studio...")
        studio.start(machine=Machine.L40S)
        # Optionally poll or wait externally
    else:
        print("Studio already running — reuse it.")

    print(f"Studio URL: {studio.url}")
    return 0

if __name__ == "__main__":
    sys.exit(main())
```
