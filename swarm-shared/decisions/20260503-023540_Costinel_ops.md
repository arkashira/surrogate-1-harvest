const { HfApi } = require('@huggingface/hub');
const api = new HfApi();
(async () => {
  const tree = await api.listRepoTree({
    repo: 'datasets/AXENTX/Costinel',
    path: 'knowledge/hubs',
    recursive: false,
  });
  const files = tree.filter(f => f.type === 'file').map(f => f.path);
  require('fs').writeFileSync(
    'public/signals/top-hub-files.json',
    JSON.stringify({ files, updatedAt: new Date().toISOString() }, null, 2)
  );
  console.log('Saved top-hub file list');
})();
