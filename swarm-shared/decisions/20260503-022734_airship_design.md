# airship / design

## Final Synthesized Implementation (Best of Both Candidates)

**Goal**: Eliminate HF API 429s and Lightning quota waste during Surrogate training by implementing **CDN-first deterministic ingestion with embedded file lists and studio reuse**.

**Why this ships in <2h**: Single-file changes to the training pipeline + one orchestration wrapper. No infra changes, no new services.

---

## Implementation Plan

### 1. Create deterministic file list once (Mac orchestration)
- Run `list_repo_tree` for the target date folder (non-recursive)  
- Save flat list of file paths to `training/file_list_{date}.json`  
- Commit/embed this list in the training repo so Lightning fetches via CDN only  
- **Key improvement**: Use non-recursive listing per folder to avoid pagination explosion and ensure deterministic ordering

### 2. Update training script to use CDN-only fetches
- Load file list from embedded JSON  
- Use `hf_hub_download(repo_type="dataset", repo_id=repo, filename=path)` or raw CDN URLs  
- Project to `{prompt, response}` at parse time; ignore mixed schemas  
- **Key improvement**: No `load_dataset` with streaming, no recursive listing during training  
- **Key improvement**: Explicit error handling for Arrow/parquet parsing failures

### 3. Lightning Studio reuse wrapper
- Before `Studio(create_ok=True)`, list `Teamspace.studios`  
- If running studio with matching name exists, reuse it  
- If stopped, restart with `target.start(machine=Machine.L40S)`  
- **Key improvement**: Avoids recreating studios and burning quota  
- **Key improvement**: Proper status polling with timeout instead of blind retry

### 4. Deterministic sibling repo spread for writes (if writing enriched data)
- Hash slug → pick sibling repo index (0..4)  
- Spread commits across 5 repos = 640/hr aggregate cap  
- **Key improvement**: Deterministic selection ensures reproducible writes

## Code Snippets

### `scripts/pre_list_files.py` (run on Mac after rate-limit window)
```python
#!/usr/bin/env python3
"""
Pre-list HF dataset files for a date folder and emit JSON for CDN-only training.
Usage: python pre_list_files.py --repo <repo> --date 2026-04-29 --out file_list.json
"""
import argparse
import json
import os
from huggingface_hub import HfApi

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True, help="HF dataset repo (user/ds)")
    parser.add_argument("--date", required=True, help="Date folder (YYYY-MM-DD)")
    parser.add_argument("--out", default="file_list.json", help="Output JSON path")
    args = parser.parse_args()

    api = HfApi()
    # Non-recursive per top folder to avoid pagination explosion
    folders = [args.date]
    all_files = []

    for folder in folders:
        try:
            tree = api.list_repo_tree(repo_id=args.repo, repo_type="dataset", path=folder, recursive=False)
            for entry in tree:
                if entry.type == "file":
                    all_files.append(entry.path)
        except Exception as e:
            print(f"Warning listing {folder}: {e}")

    out_path = os.path.join(os.path.dirname(__file__), "..", "training", args.out)
    os.makedirs(os.path.dirname(out_path), exist_ok=True)
    with open(out_path, "w") as f:
        json.dump({"repo": args.repo, "date": args.date, "files": sorted(all_files)}, f, indent=2)
    print(f"Wrote {len(all_files)} files to {out_path}")

if __name__ == "__main__":
    main()
```

### `surrogate/train.py` (CDN-only data loader)
```python
import json
import os
from pathlib import Path
from typing import List, Dict
import pyarrow as pa
import pyarrow.parquet as pq
from huggingface_hub import hf_hub_download

# Embedded file list (generated by pre_list_files.py)
_FILE_LIST_PATH = Path(__file__).parent / "file_list_latest.json"

def load_file_list() -> Dict:
    if not _FILE_LIST_PATH.exists():
        raise FileNotFoundError(f"File list not found: {_FILE_LIST_PATH}. Run pre_list_files.py first.")
    with open(_FILE_LIST_PATH) as f:
        return json.load(f)

def cdn_fetch_one(repo: str, path: str, cache_dir: str = "./.hf_cache") -> str:
    """Download single file via CDN (no auth, no API rate limit)."""
    local = hf_hub_download(
        repo_id=repo,
        repo_type="dataset",
        filename=path,
        cache_dir=cache_dir,
        local_files_only=False,
    )
    return local

def stream_cdn_parquet(repo: str, files: List[str], batch_size: int = 512):
    """Yield {prompt, response} rows from CDN-downloaded parquet files."""
    for path in files:
        local_path = cdn_fetch_one(repo, path)
        try:
            # Project only needed columns; ignore mixed schemas
            table = pq.read_table(local_path, columns=["prompt", "response"])
            for batch in table.to_batches(max_chunksize=batch_size):
                df = batch.to_pydict()
                for prompt, response in zip(df["prompt"], df["response"]):
                    if prompt and response:
                        yield {"prompt": prompt, "response": response}
        except (pa.ArrowInvalid, KeyError, OSError) as e:
            print(f"Skipping {path}: {e}")
            continue

# Example usage in training loop
if __name__ == "__main__":
    cfg = load_file_list()
    repo = cfg["repo"]
    files = cfg["files"]
    for item in stream_cdn_parquet(repo, files):
        # train step
        pass
```

### `surrogate/lightning_studio.py` (Studio reuse)
```python
from lightning import Lightning, Teamspace, Studio, Machine
import time

def get_or_create_studio(name: str, machine: Machine = Machine.L40S) -> Studio:
    """
    Reuse running studio if exists; restart if stopped; create if missing.
    Avoids quota waste from repeated Studio(create_ok=True) calls.
    """
    teamspace = Teamspace()
    existing = [s for s in teamspace.studios if s.name == name]

    if existing:
        studio = existing[0]
        print(f"Found existing studio: {studio.name} ({studio.status})")
        if studio.status == "running":
            return studio
        if studio.status == "stopped":
            print("Restarting stopped studio...")
            studio.start(machine=machine)
            # Wait for running
            for _ in range(60):
                studio.refresh()
                if studio.status == "running":
                    return studio
                time.sleep(5)
            raise RuntimeError("Studio failed to start")
        # Unknown status — recreate
        studio.stop()
        time.sleep(5)

    print(f"Creating new studio: {name}")
    return Studio(
        name=name,
        machine=machine,
        create_ok=True,
    )

# Usage
if __name__ == "__main__":
    studio = get_or_create_studio("surrogate-train-l40s")
    # Run training via studio.run(...)
```

### `surrogate/sibling_repos.py` (optional: commit cap spread)
```python
import hashlib

SIBLINGS = [
    "user/surrogate-enriched-0",
    "user/surrogate-enriched-1",
    "user/surrogate-enriched-2",
    "user/surrogate-enriched-3",
    "user/surrogate-enriched-4",
]

def repo_for_slug(slug: str) -> str:
    """Deterministic sibling repo selection by slug hash."""
    digest = hashlib.sha256(slug.encode()).hexdigest()
    idx = int(digest, 16) % len(SIBLINGS)
    return SIBLINGS[idx]
```
