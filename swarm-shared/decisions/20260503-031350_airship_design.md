# airship / design

## Implementation Plan (≤2h)

**Highest-value incremental improvement:**  
Add a zero-API-training mode + Lightning auto-recovery so Surrogate training is rate-limit-proof and survives idle timeouts.

### Steps (concrete, 90min total)

1. **Pre-list CDN file manifest (15m)**  
   - One-time Mac script: `scripts/build-cdn-manifest.py`  
   - Calls `list_repo_tree(recursive=False)` for a date folder, writes `train/cdn-manifest.json` with `{"path":..., "cdn_url":...}`.  
   - Commit manifest into repo (or upload to `batches/mirror-merged/{date}/`).

2. **Training data loader: CDN-only (30m)**  
   - Replace `load_dataset(streaming=True)` with direct `requests.get(cdn_url, stream=True)` + `jsonlines`/`parquet` projection to `{prompt, response}`.  
   - No HuggingFace `datasets` API during training → zero auth/rate-limit exposure.

3. **Lightning auto-recovery wrapper (30m)**  
   - Add `train/run_surrogate.py` launcher that:  
     - Checks `Teamspace.studios` for existing running studio; reuses if found.  
     - If stopped, restarts with `target.start(machine=Machine.L40S)` (falls back to free-tier L40S).  
     - Runs training via `studio.run(command="python train/train.py --manifest cdn-manifest.json", wait=False)` and polls status; on stop → restart.

4. **CI/cron hygiene (15m)**  
   - Ensure any wrapper scripts have `#!/usr/bin/env bash`, `chmod +x`, and crontab sets `SHELL=/bin/bash`.

## Code snippets

### 1) Build CDN manifest (run once on Mac)

```python
# scripts/build-cdn-manifest.py
#!/usr/bin/env python3
import json, os
from huggingface_hub import list_repo_tree

REPO = "axentx/surrogate-dataset"
DATE = "2026-05-03"
OUT = os.path.join("train", "cdn-manifest.json")

files = list_repo_tree(repo_id=REPO, path=DATE, recursive=False)
manifest = []
for f in files:
    if f.rfilename.endswith((".jsonl", ".parquet")):
        manifest.append({
            "path": f.rfilename,
            "cdn_url": f"https://huggingface.co/datasets/{REPO}/resolve/main/{f.rfilename}"
        })

os.makedirs(os.path.dirname(OUT), exist_ok=True)
with open(OUT, "w") as fp:
    json.dump(manifest, fp, indent=2)
print(f"Wrote {len(manifest)} files to {OUT}")
```

### 2) CDN-only data loader (inside training)

```python
# train/data.py
import json, io, requests, pyarrow.parquet as pq
from typing import List, Dict

def load_cdn_examples(manifest_path: str, limit: int = None) -> List[Dict]:
    with open(manifest_path) as f:
        files = json.load(f)
    if limit:
        files = files[:limit]

    examples = []
    for entry in files:
        url = entry["cdn_url"]
        r = requests.get(url, stream=True, timeout=60)
        r.raise_for_status()

        if url.endswith(".jsonl"):
            import jsonlines
            # stream from bytes
            for obj in jsonlines.Reader(io.BytesIO(r.content)):
                examples.append({"prompt": obj["prompt"], "response": obj["response"]})
        elif url.endswith(".parquet"):
            table = pq.read_table(io.BytesIO(r.content))
            df = table.select(["prompt", "response"]).to_pandas()
            for _, row in df.iterrows():
                examples.append({"prompt": row["prompt"], "response": row["response"]})
    return examples
```

### 3) Lightning auto-recovery launcher

```python
# train/run_surrogate.py
#!/usr/bin/env python3
import time
from lightning_sdk import Teamspace, Studio, Machine

TEAMSPACE = "surrogate-team"
STUDIO_NAME = "surrogate-train"
MANIFEST = "train/cdn-manifest.json"
CMD = f"python train/train.py --manifest {MANIFEST}"

def ensure_running_studio() -> Studio:
    ts = Teamspace(name=TEAMSPACE)
    for s in ts.studios:
        if s.name == STUDIO_NAME:
            if s.status == "running":
                print(f"Reusing running studio {STUDIO_NAME}")
                return s
            else:
                print(f"Studio stopped; restarting...")
                s.start(machine=Machine.L40S)
                s.wait_until("running")
                return s
    # create if missing
    s = Studio.create(name=STUDIO_NAME, teamspace=TEAMSPACE, machine=Machine.L40S)
    s.wait_until("running")
    return s

def main():
    studio = ensure_running_studio()
    run = studio.run(command=CMD, wait=False)
    print(f"Started run {run.id}")
    # simple monitor: if studio stops, restart training
    while True:
        studio = ensure_running_studio()
        if studio.status != "running":
            print("Studio stopped unexpectedly; restarting training...")
            studio.start(machine=Machine.L40S)
            studio.wait_until("running")
            studio.run(command=CMD, wait=False)
        time.sleep(60)

if __name__ == "__main__":
    main()
```
