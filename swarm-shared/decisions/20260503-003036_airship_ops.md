```python
# scripts/build_file_manifest.py
#!/usr/bin/env python3
"""
Generate file_manifest.json for a single date folder in a HF dataset repo.

Usage:
  HF_REPO=datasets/your/repo
  HF_FOLDER=batches/mirror-merged/2026-05-03
  python scripts/build_file_manifest.py \
    --repo $HF_REPO \
    --folder $HF_FOLDER \
    --out file_manifest.json
"""
import argparse
import hashlib
import json
import os
import sys
from typing import Dict, List

try:
    from huggingface_hub import list_repo_tree
except ImportError:
    print("Install: pip install huggingface_hub")
    sys.exit(1)

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def build_manifest(repo: str, folder: str, out_path: str) -> None:
    # recursive=False keeps API calls minimal (one per folder)
    items = list_repo_tree(repo=repo, path=folder, recursive=False)
    files: List[Dict[str, object]] = []
    for item in items:
        if item.type != "file":
            continue
        path = item.path
        # Include sha256 when available; otherwise placeholder for local verification
        sha256 = getattr(item, "sha256", None) or hashlib.sha256(path.encode()).hexdigest()
        files.append(
            {
                "path": path,
                "cdn_url": CDN_TEMPLATE.format(repo=repo, path=path),
                "size": getattr(item, "size", 0),
                "sha256": sha256,
            }
        )

    manifest = {
        "repo": repo,
        "folder": folder,
        "generated_by": "scripts/build_file_manifest.py",
        "files": files,
    }

    os.makedirs(os.path.dirname(os.path.abspath(out_path)) if os.path.dirname(out_path) else ".", exist_ok=True)
    with open(out_path, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2, ensure_ascii=False)

    print(f"Wrote {len(files)} files to {out_path}")

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Build CDN-only file manifest.")
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g., datasets/your/repo)")
    parser.add_argument("--folder", required=True, help="Folder to list (non-recursive)")
    parser.add_argument("--out", default="file_manifest.json", help="Output JSON path")
    args = parser.parse_args()
    build_manifest(args.repo, args.folder, args.out)
```

Make executable:

```bash
chmod +x scripts/build_file_manifest.py
```

### 2) CDN-only iterable dataset (robust + schema-safe)

```python
# surrogate/data/cdn_dataset.py
import json
import logging
import time
from typing import Dict, Iterator, Optional

import requests
from tqdm import tqdm

logger = logging.getLogger(__name__)

class CDNIterableDataset:
    """
    Streaming dataset that reads files listed in a manifest via CDN (no HF API/auth).
    Yields {"prompt": str, "response": str, "source_file": str}.
    """

    def __init__(
        self,
        manifest_path: str,
        max_retries: int = 5,
        backoff_factor: float = 1.5,
        timeout: int = 30,
        max_files: Optional[int] = None,
        allowed_extensions: Optional[set] = None,
    ):
        self.manifest_path = manifest_path
        self.max_retries = max_retries
        self.backoff_factor = backoff_factor
        self.timeout = timeout
        self.max_files = max_files
        self.allowed_extensions = allowed_extensions or {".jsonl", ".json", ".txt", ".parquet"}
        self.manifest = self._load_manifest()

    def _load_manifest(self) -> Dict:
        with open(self.manifest_path, "r", encoding="utf-8") as f:
            return json.load(f)

    def _stream_file(self, url: str) -> Optional[str]:
        retries = 0
        while retries <= self.max_retries:
            try:
                resp = requests.get(url, timeout=self.timeout, headers={})
                resp.raise_for_status()
                return resp.text
            except Exception as exc:
                retries += 1
                if retries > self.max_retries:
                    logger.warning(f"Failed to fetch {url} after {self.max_retries} retries: {exc}")
                    return None
                sleep = self.backoff_factor ** retries
                logger.info(f"Retry {retries}/{self.max_retries} for {url} in {sleep:.1f}s: {exc}")
                time.sleep(sleep)
        return None

    def _parse_to_pair(self, text: str, path: str) -> Iterator[Dict[str, str]]:
        """
        Project raw file into {"prompt", "response"}.
        Supports:
          - JSONL lines with prompt/response (or input/output, question/answer).
          - Single JSON object with same fields.
          - Plain text with <prompt>...</prompt> <response>...</response> tags.
        Skips malformed or ambiguous entries.
        """
        import json as pyjson

        text = text.strip()
        if not text:
            return

        # Try JSONL first
        lines = [ln.strip() for ln in text.splitlines() if ln.strip()]
        for line in lines:
            try:
                obj = pyjson.loads(line)
                prompt = obj.get("prompt") or obj.get("input") or obj.get("question")
                response = obj.get("response") or obj.get("output") or obj.get("answer")
                if isinstance(prompt, str) and isinstance(response, str):
                    yield {"prompt": prompt.strip(), "response": response.strip(), "source_file": path}
                continue
            except Exception:
                pass

        # Try single JSON object
        try:
            obj = pyjson.loads(text)
            prompt = obj.get("prompt") or obj.get("input") or obj.get("question")
            response = obj.get("response") or obj.get("output") or obj.get("answer")
            if isinstance(prompt, str) and isinstance(response, str):
                yield {"prompt": prompt.strip(), "response": response.strip(), "source_file": path}
            return
        except Exception:
            pass

        # Fallback: simple tag extraction for lightly structured text
        import re
        prompt_match = re.search(r"<prompt>(.*?)</prompt>", text, re.DOTALL | re.IGNORECASE)
        response_match = re.search(r"<response>(.*?)</response>", text, re.DOTALL | re.IGNORECASE)
        if prompt_match and response_match:
            yield {
                "prompt": prompt_match.group(1).strip(),
                "response": response_match.group(1).strip(),
                "source_file": path,
            }

    def __iter__(self) -> Iterator[Dict[str, str]]:
        files = self.manifest.get("files", [])
        if self.max_files is not None:
            files = files[: self.max_files]

        for item in tqdm(files, desc="CDN dataset"):
            path = item["path"]
            # Skip unwanted file types early
            if not any(path.lower().endswith(ext) for ext in self.allowed_extensions):
                continue

            url = item["cdn_url"]
            text = self._stream_file(url)
            if text is None:
                continue
            for pair in self._parse_to_pair(text, path):
                yield pair
```
