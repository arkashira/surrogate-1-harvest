# surrogate-1 / design

## Final Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16`, `DATE`, `HF_TOKEN`
- Single `list_repo_tree(path=DATE, recursive=False)` → deterministic file list saved to `manifest-{DATE}.json` (enables reproducibility and re-runs without API calls)
- Deterministic shard assignment: `hash(slug) % SHARD_TOTAL == SHARD_ID`
- Downloads only assigned files via **HF CDN bypass** (`https://huggingface.co/datasets/.../resolve/main/...`) with no Authorization header during data fetch (uses token only for repo-tree listing and final upload)
- Projects each file to `{prompt, response}` at parse time (avoids `load_dataset(streaming=True)` mixed-schema `pyarrow.CastError`)
- Central md5 dedup via existing `lib/dedup.py`
- Output: `batches/public-merged/{DATE}/shard{N}-{HHMMSS}.jsonl`
- Commit via HF API (respect 128/hr cap; single file per shard per run)

### Why this is the highest-value incremental improvement
- Eliminates `load_dataset(streaming=True)` mixed-schema `pyarrow.CastError`
- Bypasses HF API 429 during training data load (CDN tier has much higher limits)
- Keeps RAM footprint low (streaming parse, no full dataset in memory)
- Reuses existing dedup store and folder structure
- Manifest file enables idempotent re-runs and auditability
- Fits in <2h: single-file replacement, no infra changes

#!/usr/bin/env python3
"""
CDN-bypass ingestion worker for surrogate-1-training-pairs.

Usage:
  HF_TOKEN=hf_xxx \
  SHARD_ID=0 SHARD_TOTAL=16 DATE=2026-04-29 \
  python bin/dataset-enrich.py
"""

import os
import sys
import json
import hashlib
import time
from datetime import datetime
from pathlib import Path

import requests
from huggingface_hub import list_repo_tree, hf_hub_download, HfApi

REPO = "axentx/surrogate-1-training-pairs"
BASE_CDN = f"https://huggingface.co/datasets/{REPO}/resolve/main"

# Accept env
SHARD_ID = int(os.getenv("SHARD_ID", 0))
SHARD_TOTAL = int(os.getenv("SHARD_TOTAL", 16))
DATE = os.getenv("DATE", datetime.utcnow().strftime("%Y-%m-%d"))
HF_TOKEN = os.getenv("HF_TOKEN")
if not HF_TOKEN:
    print("HF_TOKEN required", file=sys.stderr)
    sys.exit(1)

HEADERS = {"Authorization": f"Bearer {HF_TOKEN}"}
API = HfApi(token=HF_TOKEN)

# Local imports
sys.path.insert(0, str(Path(__file__).parent))
from lib.dedup import DedupStore  # noqa

def deterministic_shard(slug: str) -> int:
    return int(hashlib.md5(slug.encode()).hexdigest(), 16) % SHARD_TOTAL

def list_date_files(date_folder: str, manifest_path: Path):
    """Single API call: list top-level files in date folder and save manifest."""
    if manifest_path.exists():
        with manifest_path.open() as f:
            return json.load(f)
    items = list_repo_tree(repo_id=REPO, path=date_folder, recursive=False)
    files = sorted([it.path for it in items if it.type == "file"])
    with manifest_path.open("w") as f:
        json.dump(files, f, indent=2)
    return files

def stream_parse_cdn_file(cdn_url: str):
    """
    Yield {prompt, response} from one file.
    Supports .jsonl and .parquet via hf_hub_download fallback.
    """
    if cdn_url.endswith(".jsonl"):
        resp = requests.get(cdn_url, timeout=30)
        resp.raise_for_status()
        for line in resp.text.splitlines():
            line = line.strip()
            if not line:
                continue
            try:
                obj = json.loads(line)
            except json.JSONDecodeError:
                continue
            # Normalize to {prompt, response}
            prompt = obj.get("prompt") or obj.get("input") or obj.get("question")
            response = obj.get("response") or obj.get("output") or obj.get("answer")
            if prompt is None or response is None:
                continue
            yield {"prompt": str(prompt), "response": str(response)}
    else:
        # For parquet/other: download once and stream via pyarrow (small file)
        local_path = hf_hub_download(repo_id=REPO, filename=cdn_url.split(f"/{REPO}/resolve/main/")[-1], token=HF_TOKEN)
        try:
            import pyarrow.parquet as pq
            table = pq.read_table(local_path, columns=["prompt", "response"])
            for i in range(table.num_rows):
                row = table.slice(i, 1).to_pydict()
                prompt = row.get("prompt", [None])[0]
                response = row.get("response", [None])[0]
                if prompt is None or response is None:
                    continue
                yield {"prompt": str(prompt), "response": str(response)}
        except Exception:
            # Fallback: try jsonl-like projection from metadata if possible; else skip
            pass

def main():
    print(f"Shard {SHARD_ID}/{SHARD_TOTAL} | DATE={DATE}")

    dedup = DedupStore()
    out_dir = Path("batches/public-merged") / DATE
    out_dir.mkdir(parents=True, exist_ok=True)
    timestamp = datetime.utcnow().strftime("%H%M%S")
    out_path = out_dir / f"shard{SHARD_ID}-{timestamp}.jsonl"
    manifest_path = Path("manifests") / f"manifest-{DATE}.json"
    manifest_path.parent.mkdir(exist_ok=True)

    try:
        files = list_date_files(DATE, manifest_path)
    except Exception as e:
        print(f"Failed to list repo tree: {e}", file=sys.stderr)
        sys.exit(1)

    if not files:
        print("No files found for date", file=sys.stderr)
        return

    my_files = [f for f in files if deterministic_shard(f) == SHARD_ID]
    print(f"Processing {len(my_files)} files for this shard")

    written = 0
    skipped_dup = 0
    with out_path.open("w", encoding="utf-8") as fout:
        for file_rel in my_files:
            cdn_url = f"{BASE_CDN}/{file_rel}"
            try:
                for pair in stream_parse_cdn_file(cdn_url):
                    # Create canonical md5 for dedup
                    blob = json.dumps({"prompt": pair["prompt"], "response": pair["response"]}, sort_keys=True)
                    md5 = hashlib.md5(blob.encode()).hexdigest()
                    if dedup.exists(md5):
                        skipped_dup += 1
                        continue
                    dedup.add(md5)
                    fout.write(json.dumps(pair, ensure_ascii=False) + "\n")
                    written += 1
            except Exception as exc:
                print(f"Error processing {file_rel}: {exc}", file=sys.stderr)
                continue

    print(f"Written {written} pairs, skipped {skipped_dup} duplicates -> {out_path}")

    # Upload single file to dataset repo
    if written > 0:
        target_path = f"batches/public-merged/{DATE}/shard{SHARD_ID}-{timestamp}.jsonl"
        try:
            API.upload_file(
                path_or_fileobj=str(out_path),
                path_in_repo=target_path,
                repo_id=REPO,
                repo_type="dataset",
                commit_message=f"shard{SHARD_ID} {DATE} {timestamp}"
            )
            print(f"Uploaded {target_path}")
        except Exception as e:
            print(f"Upload failed: {e}", file=sys.stderr)
            sys.exit(1)
    else:
        print("No new pairs to upload")

if __name__ == "__main__":
    main()
