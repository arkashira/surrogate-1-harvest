# workio / frontend

## 1. Diagnosis

- Frontend has no client-side request deduplication for clock-in/out actions — rapid taps or retries on slow networks can fire multiple POSTs and create duplicate events despite backend idempotency.
- UI does not disable the clock-in/out button during inflight requests, so users can queue overlapping calls that race and confuse state (e.g., “clocked in” indicator flickers).
- No optimistic UI update or local pending state, so perceived latency is high and users retry.
- Missing short-term client-side lock (per-user, per-day) to suppress duplicates within a tolerance window (e.g., 5s) for accidental double-taps.
- No visual feedback distinguishing “processing” vs “failed” vs “success” for clock actions, increasing retry probability.

## 2. Proposed change

File scope (frontend):
- `workio/src/components/ClockControls.tsx` (or equivalent) — add inflight lock, optimistic state, and idempotency key header.
- `workio/src/api/clock.ts` — wrap fetch with client-side dedupe map and attach `Idempotency-Key` header.
- `workio/src/hooks/useClockAction.ts` — reusable hook that exposes `clockIn`/`clockOut` with loading/error states and 5s client lock.

## 3. Implementation

### 3.1 Add client-side dedupe + idempotency wrapper (`workio/src/api/clock.ts`)

```ts
// workio/src/api/clock.ts
import { getAuthHeader } from './auth';

const API_BASE = '/api';
const inflight = new Map<string, Promise<any>>();
const clientLock = new Map<string, number>(); // key -> timestamp

function clientLockKey(action: 'in' | 'out', userId: string, date: string) {
  return `clock:${action}:${userId}:${date}`;
}

function isLocked(key: string, windowMs = 5000) {
  const ts = clientLock.get(key);
  if (!ts) return false;
  return Date.now() - ts < windowMs;
}

function setLock(key: string) {
  clientLock.set(key, Date.now());
}

export async function clockAction(
  action: 'in' | 'out',
  { userId, location, note }: { userId: string; location?: GeolocationCoordinates; note?: string }
) {
  const today = new Date().toISOString().slice(0, 10); // YYYY-MM-DD
  const key = clientLockKey(action, userId, today);

  // Short client lock to suppress double-taps
  if (isLocked(key, 5000)) {
    return Promise.resolve({ duplicate: true, message: 'Action already processed' });
  }

  // Dedupe inflight identical requests
  const idempotencyKey = `${userId}:${today}:${action}:${Date.now()}`;
  const cacheKey = `${key}:${idempotencyKey}`;
  if (inflight.has(cacheKey)) {
    return inflight.get(cacheKey);
  }

  const payload = {
    userId,
    action,
    location,
    note,
    timestamp: new Date().toISOString(),
  };

  const request = fetch(`${API_BASE}/clock/${action}`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      ...getAuthHeader(),
      'Idempotency-Key': idempotencyKey,
    },
    body: JSON.stringify(payload),
  })
    .then(async (res) => {
      if (!res.ok) {
        const err = await res.json().catch(() => ({}));
        throw new Error(err.message || `Clock ${action} failed`);
      }
      return res.json();
    })
    .finally(() => {
      inflight.delete(cacheKey);
      // Keep lock for window to prevent immediate retry
      if (!clientLock.has(key)) setLock(key);
    });

  inflight.set(cacheKey, request);
  setLock(key);
  return request;
}
```

### 3.2 Reusable hook (`workio/src/hooks/useClockAction.ts`)

```ts
// workio/src/hooks/useClockAction.ts
import { useState, useCallback } from 'react';
import { clockAction } from '../api/clock';

export function useClockAction(userId: string) {
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const [lastResult, setLastResult] = useState<any>(null);

  const run = useCallback(
    async (action: 'in' | 'out', opts?: { location?: GeolocationCoordinates; note?: string }) => {
      if (loading) return;
      setLoading(true);
      setError(null);
      try {
        const result = await clockAction(action, { userId, ...opts });
        if (result?.duplicate) {
          // treat duplicate as success UX (already processed)
          setLastResult({ ...result, duplicate: true });
        } else {
          setLastResult(result);
        }
      } catch (err: any) {
        setError(err.message || 'Unknown error');
      } finally {
        setLoading(false);
      }
    },
    [userId, loading]
  );

  return {
    loading,
    error,
    lastResult,
    clockIn: (opts?: { location?: GeolocationCoordinates; note?: string }) => run('in', opts),
    clockOut: (opts?: { location?: GeolocationCoordinates; note?: string }) => run('out', opts),
  };
}
```

### 3.3 Update UI component (`workio/src/components/ClockControls.tsx`)

```tsx
// workio/src/components/ClockControls.tsx
import React, { useEffect } from 'react';
import { useClockAction } from '../hooks/useClockAction';

interface Props {
  userId: string;
  currentStatus?: 'in' | 'out' | null;
}

export function ClockControls({ userId, currentStatus }: Props) {
  const { clockIn, clockOut, loading, error, lastResult } = useClockAction(userId);

  const handleClockIn = async () => {
    // optional: acquire geolocation
    let location: GeolocationCoordinates | undefined;
    if (navigator.geolocation) {
      try {
        const pos = await new Promise<GeolocationPosition>((resolve, reject) =>
          navigator.geolocation.getCurrentPosition(resolve, reject, { enableHighAccuracy: true, timeout: 10000 })
        );
        location = pos.coords;
      } catch {
        // proceed without location if denied
      }
    }
    await clockIn({ location });
  };

  const handleClockOut = async () => {
    await clockOut({});
  };

  return (
    <div className="flex flex-col gap-3">
      <div className="flex gap-3">
        <button
          onClick={handleClockIn}
          disabled={loading || currentStatus === 'in'}
          className={`px-6 py-2 rounded font-medium ${
            loading || currentStatus === 'in'
              ? 'bg-gray-300 cursor-not-allowed'
              : 'bg-green-600 text-white hover:bg-green-700'
          }`}
        >
          {loading && lastResult?.action === 'in' ? 'กำลังลงเวลา...' : 'ลงเวลาทำงาน'}
        </button>

        <button
          onClick={handleClockOut}
          disabled={loading || currentStatus === 'out'}
          className={`px-6 py-2 rounded font-medium ${
            loading || currentStatus === 'out'
              ? 'bg-gray-300 cursor-not-allowed'
              : 'bg-red-600 text-white hover:bg-red-700'
          }`}
        >
          {loading && lastResult?.action === 'out' ? 'กำลังลงเวลาออก...' : 'ลงเวลาออก'}
        </button>
      </div>

      {error && <p className="text-red-600 text-sm">{error}</p>}
      {lastResult?.duplicate && (
        <p className="text-yellow-600 text-sm">รายการนี้ถูกบันทึกแล้ว</p>
      )}
    </div>
  );
}
```
