# workio / quality

## 1. Diagnosis

- Attendance submissions lack idempotency: retries (network blips, background-sync replay) can create duplicate clock-in/out records in PostgreSQL.
- No client-side durable queue with transactional safety: localStorage is ephemeral and can be cleared or corrupted; events can be lost on reload/storage pressure.
- No exponential backoff + jitter: naive retry loops risk thundering-herd bursts and wasted requests during outages.
- No cross-tab coordination: multiple tabs can enqueue and submit the same event concurrently, compounding duplicates.
- Backend does not enforce uniqueness for pending/out-of-order events (same employee + event_type + short window), allowing race-condition inserts.

## 2. Proposed change

- **Frontend**: `workio/src/lib/attendance/queue.ts` (new) + `workio/src/lib/attendance/useAttendanceQueue.ts` (new) + update `workio/src/lib/api/attendance.ts` and clock-in/out UI hooks.
- **Backend**: `workio/server/src/routes/attendance.ts` — add idempotency-key header check and unique constraint/index to prevent duplicates.

## 3. Implementation

```ts
// workio/src/lib/attendance/queue.ts
import { persist, createJSONStorage } from 'zustand/middleware';
import { create } from 'zustand';

export type AttendanceEvent = {
  id: string; // uuid v4 idempotency key
  type: 'clock_in' | 'clock_out';
  timestamp: string; // ISO
  location?: { lat: number; lng: number };
  retries: number;
  createdAt: number;
  lastAttemptAt?: number;
  status: 'pending' | 'sending' | 'sent' | 'failed';
};

type QueueState = {
  events: AttendanceEvent[];
  add: (e: Omit<AttendanceEvent, 'retries' | 'createdAt' | 'status'>) => void;
  update: (id: string, patch: Partial<AttendanceEvent>) => void;
  remove: (id: string) => void;
  nextPending: () => AttendanceEvent | undefined;
};

export const useAttendanceQueue = create<QueueState>()(
  persist(
    (set, get) => ({
      events: [],
      add: (e) =>
        set((s) => ({
          events: [
            ...s.events,
            { ...e, retries: 0, createdAt: Date.now(), status: 'pending' },
          ],
        })),
      update: (id, patch) =>
        set((s) => ({
          events: s.events.map((ev) => (ev.id === id ? { ...ev, ...patch } : ev)),
        })),
      remove: (id) =>
        set((s) => ({ events: s.events.filter((ev) => ev.id !== id) })),
      nextPending: () => get().events.find((ev) => ev.status === 'pending'),
    }),
    {
      name: 'workio-attendance-queue',
      storage: createJSONStorage(() => sessionStorage), // safer than localStorage; survives reloads, cleared on logout via session end
    }
  )
);
```

```ts
// workio/src/lib/attendance/useAttendanceQueueProcessor.ts
import { useEffect } from 'react';
import { useAttendanceQueue } from './queue';
import { submitAttendance } from '../api/attendance';

const MAX_RETRIES = 5;
const BASE_DELAY = 1_000;

export function useAttendanceQueueProcessor() {
  const { events, update, remove, nextPending } = useAttendanceQueue();

  useEffect(() => {
    let stopped = false;
    let lock = false;

    async function process() {
      if (lock) return;
      lock = true;

      try {
        const ev = nextPending();
        if (!ev) return;

        update(ev.id, { status: 'sending', lastAttemptAt: Date.now() });

        try {
          await submitAttendance({
            type: ev.type,
            timestamp: ev.timestamp,
            location: ev.location,
            idempotencyKey: ev.id,
          });
          if (!stopped) {
            remove(ev.id);
          }
        } catch (err: any) {
          if (!stopped) {
            const nextRetries = ev.retries + 1;
            if (nextRetries > MAX_RETRIES) {
              update(ev.id, { status: 'failed', retries: nextRetries });
            } else {
              // exponential backoff + jitter
              const delay = BASE_DELAY * 2 ** (nextRetries - 1) + Math.random() * 500;
              update(ev.id, { status: 'pending', retries: nextRetries });
              setTimeout(process, delay);
            }
          }
        }
      } finally {
        lock = false;
        // continue processing if more pending
        if (events.some((e) => e.status === 'pending')) {
          setTimeout(process, 200);
        }
      }
    }

    // start processing when online and pending items exist
    function schedule() {
      if (navigator.onLine && events.some((e) => e.status === 'pending')) {
        process();
      }
    }

    window.addEventListener('online', schedule);
    // cross-tab coordination via storage events
    window.addEventListener('storage', (e) => {
      if (e.key === 'workio-attendance-queue') {
        // re-evaluate queue state from storage
        if (events.some((e) => e.status === 'pending')) process();
      }
    });

    schedule();

    return () => {
      stopped = true;
      window.removeEventListener('online', schedule);
    };
  }, [events, nextPending, update, remove]);
}
```

```ts
// workio/src/lib/api/attendance.ts
export async function submitAttendance(payload: {
  type: 'clock_in' | 'clock_out';
  timestamp: string;
  location?: { lat: number; lng: number };
  idempotencyKey: string;
}) {
  const res = await fetch('/api/attendance', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(payload),
  });

  if (!res.ok) {
    const err = new Error('Attendance submission failed') as any;
    err.status = res.status;
    throw err;
  }
  return res.json();
}
```

```tsx
// workio/src/features/AttendanceButton.tsx (example usage)
import { useAttendanceQueue } from '../lib/attendance/queue';
import { v4 as uuidv4 } from 'uuid';
import { useAttendanceQueueProcessor } from '../lib/attendance/useAttendanceQueueProcessor';

export function AttendanceButton() {
  const add = useAttendanceQueue((s) => s.add);
  useAttendanceQueueProcessor();

  const handleClockIn = () => {
    add({
      id: uuidv4(),
      type: 'clock_in',
      timestamp: new Date().toISOString(),
      location: { lat: 13.7563, lng: 100.5018 }, // obtain from geolocation
    });
  };

  return <button onClick={handleClockIn}>Clock In</button>;
}
```

```sql
CREATE TABLE IF NOT EXISTS attendance (
  id SERIAL PRIMARY KEY,
  employee_id INTEGER NOT NULL REFERENCES employees(id),
  event_type VARCHAR(10) NOT NULL,
  event_time TIMESTAMPTZ NOT NULL,
  location_lat DOUBLE PRECISION,
  location_lng DOUBLE PRECISION,
  idempotency_key VARCHAR(255) NOT NULL,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  CONSTRAINT unique_idempotency UNIQUE (idempotency_key)
);
CREATE INDEX IF NOT EXISTS idx_attendance_employee_time ON attendance (employee_id, event_time);
```
