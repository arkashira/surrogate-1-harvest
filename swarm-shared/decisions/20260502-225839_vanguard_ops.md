# vanguard / ops

## 1. Diagnosis

- Hash router changes do not restore scroll position → disorienting jumps to top on every back/forward navigation.
- Dataset detail and list views lose filter/pagination/sort state on reload or shared links because they are not encoded in the URL.
- No loading or error UI during dataset fetch → blank screen or frozen UI on slow/unreliable networks.
- Missing deterministic route-to-state binding causes ephemeral UI state to be discarded on navigation.
- No canonical URL representation for dataset views prevents shareable, bookmarkable links with exact context.

## 2. Proposed change

Target: `/opt/axentx/vanguard/src/router.js` (or equivalent router entry) and `/opt/axentx/vanguard/src/pages/DatasetList.js` and `/opt/axentx/vanguard/src/pages/DatasetDetail.js`.

Scope:
- Add scroll restoration middleware to hash router (persist/restore `window.scrollY` on `hashchange`).
- Encode filter/pagination/sort into hash query params (e.g., `#/datasets?page=2&sort=name&filter=active`).
- Parse hash query on route entry and hydrate UI state from URL.
- Add lightweight loading/error boundary UI for dataset fetches.

## 3. Implementation

### src/router.js
```js
// src/router.js
(function () {
  'use strict';

  // Utilities
  function parseHashQuery() {
    const hash = window.location.hash || '#/';
    const idx = hash.indexOf('?');
    const query = idx > -1 ? hash.slice(idx + 1) : '';
    const params = new URLSearchParams(query);
    const state = {};
    for (const [k, v] of params.entries()) {
      state[k] = decodeURIComponent(v);
    }
    return { path: hash.split('?')[0].replace('#', '') || '/', state };
  }

  function buildHash(path, state) {
    const qs = new URLSearchParams();
    for (const [k, v] of Object.entries(state || {})) {
      if (v != null && v !== '') qs.set(k, encodeURIComponent(v));
    }
    const query = qs.toString();
    return `#${path}${query ? '?' + query : ''}`;
  }

  // Scroll restoration
  let lastY = 0;
  let isRestoring = false;

  function saveScroll() {
    lastY = window.scrollY || window.pageYOffset;
  }

  function restoreScroll() {
    isRestoring = true;
    window.scrollTo(0, lastY);
    // small timeout to allow DOM paint before releasing lock
    setTimeout(() => { isRestoring = false; }, 50);
  }

  // Router state
  const routes = {};
  let currentPath = null;
  let currentState = {};

  function navigate(path, state = {}, replace = false) {
    currentPath = path;
    currentState = state;
    const href = buildHash(path, state);
    if (replace) {
      window.location.replace(href);
    } else {
      window.location.hash = href;
    }
    // Save scroll before navigation renders
    saveScroll();
  }

  function onHashChange() {
    const { path, state } = parseHashQuery();
    currentPath = path;
    currentState = state;

    // Notify route handlers
    const handler = routes[path] || routes['*'];
    if (handler) handler(state);

    // Restore scroll after route render (defer to allow DOM update)
    requestAnimationFrame(() => {
      restoreScroll();
    });
  }

  // Public API
  window.Router = {
    route(path, handler) {
      routes[path] = handler;
    },
    navigate,
    getState() {
      return { path: currentPath, state: currentState };
    },
    parseHashQuery,
    buildHash,
    init() {
      window.addEventListener('hashchange', onHashChange);
      // capture scroll before user-initiated navigation (best-effort)
      window.addEventListener('scroll', saveScroll, { passive: true });
      // initial
      onHashChange();
    }
  };

  // Auto-init on DOM ready
  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', () => Router.init());
  } else {
    Router.init();
  }
})();
```

### src/loading.js
```js
// src/loading.js — tiny loading/error overlay
(function () {
  'use strict';
  let container = null;

  function ensureContainer() {
    if (!container) {
      container = document.createElement('div');
      container.id = 'vanguard-loading';
      container.style.cssText = `
        position:fixed;top:0;left:0;right:0;bottom:0;
        background:rgba(255,255,255,0.85);
        display:flex;align-items:center;justify-content:center;
        z-index:9999;opacity:0;pointer-events:none;
        transition:opacity .2s ease;
      `;
      container.innerHTML = `
        <div style="text-align:center">
          <div style="width:40px;height:40px;border:3px solid #e2e8f0;border-top-color:#2563eb;border-radius:50%;margin:0 auto 8px;animation:spin .6s linear infinite"></div>
          <div id="vanguard-loading-text" style="color:#64748b;font-size:13px">Loading...</div>
        </div>
        <style>@keyframes spin{to{transform:rotate(360deg)}}</style>
      `;
      document.body.appendChild(container);
    }
  }

  window.LoadingUI = {
    show(text) {
      ensureContainer();
      container.style.opacity = '1';
      container.style.pointerEvents = 'auto';
      const t = container.querySelector('#vanguard-loading-text');
      if (t && text) t.textContent = text;
    },
    hide() {
      if (!container) return;
      container.style.opacity = '0';
      container.style.pointerEvents = 'none';
    },
    error(message) {
      ensureContainer();
      container.innerHTML = `
        <div style="text-align:center;padding:24px;background:#fff;border-radius:8px;box-shadow:0 4px 24px rgba(0,0,0,.08)">
          <div style="color:#dc2626;font-weight:600;margin-bottom:8px">Unable to load</div>
          <div style="color:#64748b;font-size:14px">${message || 'Please try again.'}</div>
          <button onclick="location.reload()" style="margin-top:12px;padding:6px 14px;background:#2563eb;color:#fff;border:none;border-radius:6px;cursor:pointer">Reload</button>
        </div>
      `;
      container.style.opacity = '1';
      container.style.pointerEvents = 'auto';
    }
  };
})();
```
