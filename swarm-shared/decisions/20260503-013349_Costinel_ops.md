# Costinel / ops

## Final Implementation Plan — Top-hub Signal Panel (Costinel dashboard)

**Core philosophy**: *Sense + Signal — do not Execute*.  
**Scope**: Frontend-only, read-only panel that surfaces the most-connected hub and actionable proposals.  
**Timebox**: <2h  
**Stack**: React + TypeScript + Tailwind (resilient to missing graph data; graceful fallback).

---

### 1) Types — `src/types/knowledge-rag.ts`

```ts
export interface KnowledgeHub {
  id: string;
  label: string;
  description?: string;
  type?: 'hub' | 'proposal' | 'doc';
  connections: number;
  lastUpdated: string; // ISO
  tags?: string[];
}

export interface SignalProposal {
  id: string;
  title: string;
  summary: string;
  hubId: string;
  action?: string;            // human action (e.g., "Review RI coverage")
  priority: 'high' | 'medium' | 'low';
  impact?: 'high' | 'medium' | 'low';
  effort?: 'low' | 'medium' | 'high';
  evidenceLinks?: string[];
  createdAt?: string;
}

export interface TopHubPayload {
  topHub: KnowledgeHub | null;
  proposals: SignalProposal[];
  generatedAt?: string;
}
```

### 2) Lightweight RAG client — `src/lib/knowledge-rag.ts`

```ts
import { TopHubPayload, KnowledgeHub, SignalProposal } from '../types/knowledge-rag';

const API_BASE = '/api/knowledge-rag'; // proxy to backend when available

const MOCK_PAYLOAD: TopHubPayload = {
  topHub: {
    id: 'MOC',
    label: 'MOC',
    description: 'Most-connected operational hub',
    type: 'hub',
    connections: 128,
    lastUpdated: new Date().toISOString(),
    tags: ['ops', 'core', 'production'],
  },
  proposals: [
    {
      id: 'p-001',
      title: 'Review Reserved Instance coverage for MOC workloads',
      summary: 'Detected 34% RI underutilization in us-east-1/a over last 30d.',
      hubId: 'MOC',
      action: 'Open cost proposal',
      priority: 'high',
      impact: 'high',
      effort: 'medium',
      evidenceLinks: ['/dashboard/cost/ri-coverage'],
      createdAt: new Date().toISOString(),
    },
    {
      id: 'p-002',
      title: 'Schedule idle resource cleanup for dev clusters',
      summary: '12 idle node pools identified across sandbox accounts.',
      hubId: 'MOC',
      action: 'Create cleanup ticket',
      priority: 'medium',
      impact: 'medium',
      effort: 'low',
      evidenceLinks: ['/dashboard/resources/idle'],
      createdAt: new Date().toISOString(),
    },
  ],
  generatedAt: new Date().toISOString(),
};

async function fetchTopHub(): Promise<TopHubPayload> {
  try {
    const res = await fetch(`${API_BASE}/top-hub`, { cache: 'no-store' });
    if (!res.ok) throw new Error(`HTTP ${res.status}`);
    return await res.json();
  } catch {
    // Graceful fallback so dashboard never breaks
    return MOCK_PAYLOAD;
  }
}

export const knowledgeRag = {
  fetchTopHub,
};
```

### 3) Panel component — `src/components/TopHubSignalPanel.tsx`

```tsx
import { useEffect, useState } from 'react';
import { knowledgeRag } from '../lib/knowledge-rag';
import type { TopHubPayload, SignalProposal } from '../types/knowledge-rag';

const priorityColors = {
  high: 'bg-red-50 border-red-200 text-red-800',
  medium: 'bg-amber-50 border-amber-200 text-amber-800',
  low: 'bg-slate-50 border-slate-200 text-slate-800',
};

export default function TopHubSignalPanel() {
  const [data, setData] = useState<TopHubPayload | null>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    knowledgeRag.fetchTopHub().then((res) => {
      setData(res);
      setLoading(false);
    });
  }, []);

  if (loading) {
    return (
      <section className="rounded-xl border border-slate-200 bg-white p-5 shadow-sm">
        <div className="h-6 w-32 animate-pulse rounded bg-slate-200" />
        <div className="mt-4 h-20 animate-pulse rounded bg-slate-100" />
      </section>
    );
  }

  if (!data?.topHub) return null;

  return (
    <section className="rounded-xl border border-slate-200 bg-white p-5 shadow-sm">
      <div className="flex items-start justify-between gap-4">
        <div>
          <h3 className="text-base font-semibold text-slate-900">Top Hub</h3>
          <p className="text-2xl font-bold text-blue-600">{data.topHub.label}</p>
          <p className="text-sm text-slate-500">{data.topHub.description}</p>
          <p className="mt-1 text-xs text-slate-400">
            {data.topHub.connections} connections · Updated{' '}
            {new Date(data.topHub.lastUpdated).toLocaleDateString()}
          </p>
        </div>
      </div>

      <div className="mt-4">
        <h4 className="mb-2 text-sm font-semibold text-slate-700">Actionable Signals</h4>
        <div className="flex flex-col gap-2">
          {data.proposals.map((p: SignalProposal) => (
            <div
              key={p.id}
              className={`rounded-lg border p-3 text-sm ${priorityColors[p.priority]}`}
            >
              <div className="flex items-start justify-between gap-2">
                <div>
                  <p className="font-medium">{p.title}</p>
                  <p className="mt-0.5">{p.summary}</p>
                  {p.action && (
                    <span className="mt-1 inline-block text-xs underline decoration-dotted underline-offset-2">
                      {p.action}
                    </span>
                  )}
                </div>
                <span className="whitespace-nowrap rounded-full px-2 py-0.5 text-xs font-medium capitalize">
                  {p.priority}
                </span>
              </div>
            </div>
          ))}
        </div>
      </div>
    </section>
  );
}
```
