# Costinel / ops

## Final Implementation Plan — Costinel “Top-Hub Signal” Card (frontend-only)

**Scope**: Pure frontend change (~2h).  
Assumes endpoint `/api/v1/sense/top-hub-signal` exists.  
Goal: show the most-connected hub and 3 actionable docs with robust UX and zero backend work.

---

### Endpoint contract (canonical)

Use the shape that balances both proposals and is most practical for UI + future extensibility:

```json
GET /api/v1/sense/top-hub-signal
200 OK
{
  "hub": { "id": "MOC", "label": "MOC", "connections": 142 },
  "score": 0.92,
  "docs": [
    { "id": "doc-1", "title": "RI Coverage Playbook", "url": "/docs/ri-coverage", "snippet": "How to right-size RIs..." },
    { "id": "doc-2", "title": "Anomaly Triage SOP", "url": "/docs/anomaly-triage", "snippet": "Steps to validate cost spikes..." },
    { "id": "doc-3", "title": "Multi-Account Tagging Guide", "url": "/docs/tagging-guide", "snippet": "Required tags for chargeback..." }
  ]
}
```

- `hub.id` and `hub.label` for display and routing.  
- `hub.connections` for context.  
- `score` optional (used for ranking/confidence).  
- `docs[].id` for keys; `url` for navigation; `title` + `snippet` for content.

### Component: `src/components/dashboard/TopHubSignalCard.tsx`

Uses SWR for caching + polling, existing design tokens, graceful states, keyboard accessibility, and safe navigation.

```tsx
import useSWR from 'swr';
import { ExternalLink } from 'lucide-react';
import { cn } from '@/lib/utils';

interface RelatedDoc {
  id: string;
  title: string;
  url: string;
  snippet: string;
}

interface HubInfo {
  id: string;
  label: string;
  connections: number;
}

interface TopHubSignal {
  hub: HubInfo;
  score?: number;
  docs: RelatedDoc[];
}

const fetcher = (url: string) => fetch(url, { credentials: 'same-origin' }).then((r) => {
  if (!r.ok) throw new Error(`HTTP ${r.status}`);
  return r.json();
});

export function TopHubSignalCard() {
  const { data, error, isLoading } = useSWR<TopHubSignal>(
    '/api/v1/sense/top-hub-signal',
    fetcher,
    {
      refreshInterval: 300_000, // 5m
      revalidateOnFocus: false,
      dedupingInterval: 30_000,
      shouldRetryOnError: true,
      errorRetryCount: 2,
    }
  );

  // Loading state
  if (isLoading) {
    return (
      <div className="rounded-xl border bg-card p-6 shadow-sm">
        <div className="flex items-center gap-3">
          <div className="h-10 w-10 animate-pulse rounded-full bg-muted" />
          <div className="space-y-2">
            <div className="h-4 w-28 animate-pulse rounded bg-muted" />
            <div className="h-3 w-16 animate-pulse rounded bg-muted" />
          </div>
        </div>
        <div className="mt-4 space-y-2">
          {Array.from({ length: 3 }).map((_, i) => (
            <div key={i} className="h-12 animate-pulse rounded-lg border bg-muted/40" />
          ))}
        </div>
      </div>
    );
  }

  // Error state
  if (error || !data) {
    return (
      <div className="rounded-xl border bg-card p-6 shadow-sm">
        <div className="flex items-center gap-2 text-sm text-muted-foreground">
          <span className="inline-flex h-4 w-4 items-center justify-center">
            <svg
              className="h-4 w-4 text-muted-foreground/60"
              fill="none"
              viewBox="0 0 24 24"
              stroke="currentColor"
            >
              <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M12 9v2m0 4h.01m-6.938 4h13.856c1.54 0 2.502-1.667 1.732-3L13.732 4c-.77-1.333-2.694-1.333-3.464 0L3.34 16c-.77 1.333.192 3 1.732 3z" />
            </svg>
          </span>
          Unable to load top-hub signal
        </div>
      </div>
    );
  }

  const { hub, docs } = data;

  return (
    <article className="rounded-xl border bg-card p-6 shadow-sm">
      <header className="mb-4 flex items-start justify-between">
        <div>
          <p className="text-xs font-medium uppercase tracking-wide text-muted-foreground">
            Top-Hub Signal
          </p>
          <div className="mt-1 flex items-baseline gap-2">
            <span className="inline-flex items-center rounded-full bg-primary/10 px-3 py-1 text-lg font-semibold text-primary">
              {hub.label}
            </span>
            <span className="text-sm text-muted-foreground">
              {hub.connections} connections
            </span>
          </div>
        </div>
        <ExternalLink className="h-4 w-4 text-muted-foreground/40" aria-hidden="true" />
      </header>

      <div className="space-y-2">
        {docs.map((doc) => (
          <a
            key={doc.id}
            href={doc.url}
            target="_blank"
            rel="noopener noreferrer"
            className={cn(
              'group block rounded-lg border p-3 transition-colors hover:bg-accent focus-visible:bg-accent',
              'focus-visible:outline-none focus-visible:ring-2 focus-visible:ring-ring focus-visible:ring-offset-2'
            )}
          >
            <p className="text-sm font-medium text-foreground group-hover:underline">
              {doc.title}
            </p>
            <p className="mt-0.5 text-xs text-muted-foreground line-clamp-2">
              {doc.snippet}
            </p>
          </a>
        ))}
      </div>
    </article>
  );
}
```
