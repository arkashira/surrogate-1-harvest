# vanguard / design

## Final Synthesized Answer

### 1. Diagnosis (Consolidated)
- **No manifest persistence**: every training run performs authenticated `list_repo_tree`, burning HF API quota and risking 429s.
- **Streaming/heterogeneous loads**: `load_dataset(streaming=True)` or repeated per-file loads on mixed-schema repos cause `pyarrow.CastError`.
- **No CDN-only path**: training continues to route through HF `/api/` instead of using `https://huggingface.co/datasets/{repo}/resolve/main/`.
- **Repo write limits**: all enriched writes target one repo, risking 128 commits/hour limit.
- **Studio reuse missing**: each run recreates studio, burning 80hr/mo quota and risking idle-stop training death.

### 2. Proposed Change
Add a lightweight **manifest + CDN loader** layer to `vanguard` that:
- Persists `(repo, dateFolder) → [file-paths]` JSON manifest after a single authenticated `list_repo_tree` call.
- Exposes a `DataResolver` that emits CDN URLs for Lightning training scripts (zero API calls during training).
- Projects mixed-schema files to `{prompt, response}` at parse time only.
- Adds repo-sharding for enriched writes (hash-slug → sibling repo).
- Adds Studio reuse guard before `.run()`.

Scope: new module `vanguard/data/cdn_resolver.py` + small patch to orchestrator/train launcher.

```python
# /opt/axentx/vanguard/data/cdn_resolver.py
from __future__ import annotations
import json, hashlib, os
from pathlib import Path
from typing import List, Tuple
import requests

HF_CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"
MANIFEST_DIR = Path(os.getenv("VANGUARD_MANIFEST_DIR", ".manifests"))
MANIFEST_DIR.mkdir(exist_ok=True)

# -- repo sharding --
SIBLING_REPOS = [
    "axentx/vanguard-enriched",
    "axentx/vanguard-enriched-1",
    "axentx/vanguard-enriched-2",
    "axentx/vanguard-enriched-3",
    "axentx/vanguard-enriched-4",
]

def pick_repo(slug: str) -> str:
    """Deterministic repo by hash(slug) % N."""
    idx = int(hashlib.sha256(slug.encode()).hexdigest(), 16) % len(SIBLING_REPOS)
    return SIBLING_REPOS[idx]

# -- manifest persistence --
def manifest_path(repo: str, date_folder: str) -> Path:
    safe = repo.replace("/", "_")
    return MANIFEST_DIR / f"{safe}__{date_folder}.json"

def save_manifest(repo: str, date_folder: str, files: List[str]) -> None:
    p = manifest_path(repo, date_folder)
    p.write_text(json.dumps({repo: {date_folder: files}}, indent=2))

def load_manifest(repo: str, date_folder: str) -> List[str] | None:
    p = manifest_path(repo, date_folder)
    if not p.exists():
        return None
    data = json.loads(p.read_text())
    return data.get(repo, {}).get(date_folder)

# -- CDN resolver --
def build_cdn_urls(repo: str, date_folder: str, files: List[str]) -> List[str]:
    return [HF_CDN_TEMPLATE.format(repo=repo, path=f"{date_folder}/{f}") for f in files]

class DataResolver:
    """
    Usage:
      resolver = DataResolver(repo, date_folder)
      urls = resolver.cdn_urls()          # Lightning training uses these
      enriched_repo = resolver.enriched_repo(slug)
    """
    def __init__(self, repo: str, date_folder: str):
        self.repo = repo
        self.date_folder = date_folder
        self._files: List[str] | None = None

    def files(self, hf_token: str | None = None) -> List[str]:
        if self._files is not None:
            return self._files
        cached = load_manifest(self.repo, self.date_folder)
        if cached is not None:
            self._files = cached
            return self._files

        # single authenticated list (Mac orchestrator)
        headers = {"Authorization": f"Bearer {hf_token}"} if hf_token else {}
        url = f"https://huggingface.co/api/datasets/{self.repo}/tree/{self.date_folder}"
        resp = requests.get(url, headers=headers, timeout=30)
        resp.raise_for_status()
        items = resp.json()
        # non-recursive by default; flatten only files in this folder
        files = [it["path"].split("/", 1)[-1] for it in items if it["type"] == "file"]
        save_manifest(self.repo, self.date_folder, files)
        self._files = files
        return self._files

    def cdn_urls(self, hf_token: str | None = None) -> List[str]:
        return build_cdn_urls(self.repo, self.date_folder, self.files(hf_token))

    @staticmethod
    def enriched_repo(slug: str) -> str:
        return pick_repo(slug)
```

```python
# /opt/axentx/vanguard/orchestrator/train_launcher.py  (or wherever)
from vanguard.data.cdn_resolver import DataResolver
from lightning import Studio, Teamspace, Machine
import lightning as L
import json
from pathlib import Path

def launch_training(repo: str, date_folder: str, hf_token: str):
    resolver = DataResolver(repo, date_folder)
    cdn_urls = resolver.cdn_urls(hf_token=hf_token)
    # embed cdn_urls into train.py args/env so Lightning job uses CDN-only fetches
    Path("train_filelist.json").write_text(json.dumps(cdn_urls))

    # Studio reuse
    teamspace = Teamspace()
    studio = None
    for s in teamspace.studios:
        if s.name == "vanguard-train" and s.status == "Running":
            studio = s
            break

    if studio is None:
        studio = Studio(
            name="vanguard-train",
            script="train.py",
            machine=Machine.L40S,
            create_ok=True,
        )
    else:
        # restart if stopped (idle-stop kills training)
        if studio.status != "Running":
            studio.start(machine=Machine.L40S)

    # Pass CDN filelist via env/args
    studio.run(
        args=["--filelist", "train_filelist.json"],
        env={"HF_DATASETS_OFFLINE": "1", "HF_HUB_OFFLINE": "1"},
    )
```

```python
# train.py  (snippet)
import json, requests
from torch.utils.data import IterableDataset

class CDNParquetDataset(IterableDataset):
    def __init__(self, filelist_path):
        with open(filelist_path) as f:
            self.urls = json.load(f)

    def __iter__(self):
        for url in self.urls:
            # stream parquet from CDN (no auth, no API)
            resp = requests.get(url, stream=True, timeout=30)
            resp.raise_for_status()
            yield from self._parse_parquet(resp.content)

    def _parse_parquet(self, buf):
        import pyarrow.parquet as pq
        table = pq.read_table(pq.ParquetFile(buf))
        # keep only expected fields; coerce types if needed
        prompt_col = table["prompt"] if "prompt" in table.column_names else None
        response_col = table["response"] if "response" in table.column_names else None
        if prompt_col is None or response_col is None:
            return
        for i in range(len(table)):
            yield {
                "prompt": str(prompt_col[i].as_py()),
                "response": str(response_col[i].as_py()),
            }
```
