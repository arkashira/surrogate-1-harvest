# surrogate-1 / ops

## Final unified implementation plan (highest correctness + concrete actionability)

**Core improvement (merged from both candidates):**  
Deterministic pre-flight file listing + CDN-only ingestion.  
- Single `list_repo_tree` snapshot per date folder → `file-list.json`.  
- All shard workers and training use that snapshot; **zero HF API calls during data loading**.  
- CDN fetches via `https://huggingface.co/datasets/.../resolve/main/...` bypass auth checks and 429s from `/api/`.  
- Deterministic shard assignment by `hash(path) % N` and schema projection to `{prompt,response}` only.

### 1) `bin/list-snapshot.sh`
```bash
#!/usr/bin/env bash
set -euo pipefail

REPO="${HF_DATASET_REPO:-axentx/surrogate-1-training-pairs}"
DATE="${1:-$(date +%Y-%m-%d)}"
OUTFILE="${2:-file-list.json}"

# Requires huggingface_hub; install via pip if missing
python3 - "$REPO" "$DATE" "$OUTFILE" <<'PY'
import json, os, sys
from huggingface_hub import HfApi

repo, date, outfile = sys.argv[1], sys.argv[2], sys.argv[3]
api = HfApi()
# Single non-recursive call per date folder
tree = api.list_repo_tree(repo=repo, path=date, recursive=False)
files = [f.rfilename for f in tree if f.rfilename.endswith(('.jsonl', '.parquet', '.csv'))]
with open(outfile, "w") as f:
    json.dump({"date": date, "files": files, "snapshot_ts": time.time()}, f, indent=2)
print(f"Snapshot written: {len(files)} files -> {outfile}")
PY
```

### 2) `bin/dataset-enrich.sh` (updated core loop)
```bash
#!/usr/bin/env bash
set -euo pipefail
export SHELL=/bin/bash

HF_DATASET_REPO="${HF_DATASET_REPO:-axentx/surrogate-1-training-pairs}"
SHARD_ID="${SHARD_ID:-0}"
TOTAL_SHARDS="${TOTAL_SHARDS:-16}"
FILE_LIST="${FILE_LIST:-file-list.json}"
DATE=$(date +%Y-%m-%d)
OUTDIR="batches/public-merged/${DATE}"
mkdir -p "$OUTDIR"
TS=$(date +%H%M%S)
OUTFILE="${OUTDIR}/shard${SHARD_ID}-${TS}.jsonl"

python3 - "$SHARD_ID" "$TOTAL_SHARDS" "$FILE_LIST" "$OUTFILE" <<'PY'
import json, hashlib, os, sys, time, requests
from pathlib import Path

SHARD_ID = int(sys.argv[1])
TOTAL_SHARDS = int(sys.argv[2])
FILE_LIST = sys.argv[3]
OUTFILE = sys.argv[4]

REPO = os.getenv("HF_DATASET_REPO", "axentx/surrogate-1-training-pairs")
HF_TOKEN = os.getenv("HF_TOKEN", "")
HEADERS = {"Authorization": f"Bearer {HF_TOKEN}"} if HF_TOKEN else {}

with open(FILE_LIST) as f:
    manifest = json.load(f)

files = manifest["files"]
date = manifest["date"]

def shard_for(path: str) -> int:
    return int(hashlib.md5(path.encode()).hexdigest(), 16) % TOTAL_SHARDS

def cdn_url(path: str) -> str:
    return f"https://huggingface.co/datasets/{REPO}/resolve/main/{date}/{path}"

def fetch_cdn(path: str, retries: int = 5) -> bytes:
    url = cdn_url(path)
    for attempt in range(retries):
        try:
            r = requests.get(url, headers=HEADERS, timeout=30)
            if r.status_code == 429:
                wait = 360
                print(f"429 on CDN, waiting {wait}s", file=sys.stderr)
                time.sleep(wait)
                continue
            r.raise_for_status()
            return r.content
        except Exception as e:
            if attempt == retries - 1:
                raise
            time.sleep(2 ** attempt)
    raise RuntimeError("unreachable")

def parse_parquet(content: bytes):
    import pyarrow.parquet as pq
    import io
    table = pq.read_table(io.BytesIO(content))
    df = table.to_pandas()
    # Project to {prompt,response} only
    for _, row in df.iterrows():
        yield {"prompt": str(row.get("prompt", "")), "response": str(row.get("response", ""))}

def parse_jsonl(content: bytes):
    import io, json
    for line in io.BytesIO(content):
        if not line.strip():
            continue
        obj = json.loads(line)
        yield {"prompt": str(obj.get("prompt", "")), "response": str(obj.get("response", ""))}

def parse_csv(content: bytes):
    import csv, io
    reader = csv.DictReader(io.StringIO(content.decode()))
    for row in reader:
        yield {"prompt": str(row.get("prompt", "")), "response": str(row.get("response", ""))}

written = 0
with open(OUTFILE, "w") as out:
    for f in files:
        if shard_for(f) != SHARD_ID:
            continue
        try:
            data = fetch_cdn(f)
            if f.endswith(".parquet"):
                rows = parse_parquet(data)
            elif f.endswith(".jsonl"):
                rows = parse_jsonl(data)
            elif f.endswith(".csv"):
                rows = parse_csv(data)
            else:
                print(f"skip unknown: {f}", file=sys.stderr)
                continue
            for row in rows:
                out.write(json.dumps(row, ensure_ascii=False) + "\n")
                written += 1
        except Exception as e:
            print(f"error processing {f}: {e}", file=sys.stderr)

print(f"shard {SHARD_ID} wrote {written} rows -> {OUTFILE}")
PY
```

Make executable:
```bash
chmod +x bin/dataset-enrich.sh
```
