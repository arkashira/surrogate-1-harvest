### 2. Signature verification middleware

```typescript
// server/src/middleware/lineSignature.ts
import crypto from 'crypto';

const LINE_CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';

export function verifyLineSignature(rawBody: string, signature: string): boolean {
  if (!LINE_CHANNEL_SECRET) {
    // Fail closed in prod; allow override via explicit env opt-in for dev
    if (process.env.NODE_ENV === 'production') return false;
    return true;
  }
  const expected = crypto
    .createHmac('sha256', LINE_CHANNEL_SECRET)
    .update(rawBody, 'utf8')
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}
```

### 3. Idempotent webhook route with outbound retry/backoff

```typescript
// server/src/routes/webhook/line.ts
import { Router, Request, Response } from 'express';
import { verifyLineSignature } from '../middleware/lineSignature';
import { pool } from '../../db';
import { sendLineMessageWithRetry } from '../../lib/lineNotify';

const router = Router();

// Record processed LINE event; returns true if new, false if duplicate.
async function recordEvent(
  eventId: string,
  tenantId: string,
  userId: string,
  eventType: string
): Promise<boolean> {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');
    const res = await client.query(
      `INSERT INTO line_event_log (event_id, tenant_id, user_id, event_type, processed_at)
       VALUES ($1, $2, $3, $4, NOW())
       ON CONFLICT (event_id) DO NOTHING
       RETURNING event_id`,
      [eventId, tenantId, userId, eventType]
    );
    if (res.rowCount === 0) {
      await client.query('ROLLBACK');
      return false;
    }
    await client.query('COMMIT');
    return true;
  } catch (err) {
    await client.query('ROLLBACK');
    // Treat conflict/network errors as duplicate to avoid retries creating duplicates
    console.warn('[line-webhook] recordEvent conflict/error', eventId, err);
    return false;
  } finally {
    client.release();
  }
}

// Atomic upsert for clock events (uses schema unique constraint)
async function upsertClockEvent(
  tenantId: string,
  userId: string,
  punchedAt: Date,
  punchType: 'IN' | 'OUT'
) {
  await pool.query(
    `INSERT INTO clock_events (tenant_id, user_id, punched_at, punch_type)
     VALUES ($1, $2, $3, $4)
     ON CONFLICT (user_id, punch_type, DATE_TRUNC('minute', punched_at))
     DO UPDATE SET punched_at = EXCLUDED.punched_at`,
    [tenantId, userId, punchedAt, punchType]
  );
}

// Example domain handlers (extend for leave/OT with same idempotency pattern)
async function handleClockIn(tenantId: string, userId: string, at: Date) {
  await upsertClockEvent(tenantId, userId, at, 'IN');
  // Notify user (with retry/backoff)
  await sendLineMessageWithRetry(userId, `✅ Clocked in at ${at.toISOString()}`);
}

async function handleClockOut(tenantId: string, userId: string, at: Date) {
  await upsertClockEvent(tenantId, userId, at, 'OUT');
  await sendLineMessageWithRetry(userId, `✅ Clocked out at ${at.toISOString()}`);
}

router.post('/line', async (req: Request, res: Response) => {
  const signature = req.headers['x-line-signature'] as string;
  const rawBody = (req as any).rawBody || JSON.stringify(req.body);

  // 1) Verify signature
  if (!verifyLineSignature(rawBody, signature)) {
    console.warn('[line-webhook] invalid signature');
    return res.status(401).send('Invalid signature');
  }

  const events = req.body.events;
  if (!Array.isArray(events) || events.length === 0) {
    return res.status(200).send('OK');
  }

  // 2) Process events idempotently; never throw to avoid LINE retries
  for (const ev of events) {
    const eventId = ev.source?.userId
      ? `${ev.source.userId}:${ev.timestamp}:${ev.type}`
      : `unknown:${ev.timestamp}:${ev.type}`;

    const tenantId = ev.source?.groupId || ev.source?.userId || 'default';
    const userId = ev.source?.userId || 'unknown';
    const eventType = ev.type;

    const ok = await recordEvent(eventId, tenantId, userId, eventType);
    if (!ok) {
      continue; // duplicate — skip processing
    }

    try {
      if (ev.type === 'message' && ev.message?.text) {
        const text = ev.message.text.trim().toLowerCase();
        const now = new Date();
        if (text === 'clock in' || text === 'เข้างาน') {
          await handleClockIn(tenantId, userId, now);
        } else if (text === 'clock out' || text === 'เลิกงาน') {
          await handleClockOut(tenantId, userId, now);
        }
      }
      // Add leave/OT handlers here using the same idempotency pattern
    } catch (err) {
      console.error('[line-webhook] processing error', eventId, err);
      // Continue processing other events; return 200 for batch
    }
  }

  return res.status(200).send('OK');
});

export default router;
```
