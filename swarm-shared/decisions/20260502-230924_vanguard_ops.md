# vanguard / ops

## 1. Diagnosis
- No persistent ingestion manifest → every training run re-enumerates HF repos and risks 429s; no resume capability.
- Training scripts likely call `load_dataset`/`list_repo_files` at runtime → will hit HF API rate limits during data loading.
- No Lightning Studio reuse guard → each run may create new Studio, burning 80+ quota hours/month.
- No CDN-bypass data loader → training cannot run on Lightning without auth/rate-limit exposure.
- Missing run-state bookkeeping (last ingested date/slug, last pushed manifest) → manual retries and duplicate commits.

## 2. Proposed change
Create `/opt/axentx/vanguard/ops/ingest_manifest.py` (single file, ~120 lines) that:
- Lists one date folder via HF API once, writes `manifests/{date}.json` with CDN URLs.
- Reuses an existing running Lightning Studio or starts one (L40S priority).
- Produces a training-ready manifest consumed by downstream `train.py` (CDN-only).
- Stores minimal state (`.state/last_ingested.json`) to avoid re-ingest.

Also patch `/opt/axentx/vanguard/train.py` to accept `--manifest` and load via CDN (no HF API during training).

## 3. Implementation

```bash
# /opt/axentx/vanguard/ops/ingest_manifest.py
#!/usr/bin/env bash
set -euo pipefail
SHELL=/bin/bash

# Config
REPO="datasets/axentx/vanguard-ingest"
DATE="${1:-$(date +%Y-%m-%d)}"
MANIFEST_DIR="/opt/axentx/vanguard/manifests"
STATE_DIR="/opt/axentx/vanguard/.state"
MANIFEST="${MANIFEST_DIR}/${DATE}.json"
LAST_STATE="${STATE_DIR}/last_ingested.json"
HF_TOKEN="${HF_TOKEN:-}"

mkdir -p "$MANIFEST_DIR" "$STATE_DIR"

# If manifest exists and is non-empty, reuse
if [[ -s "$MANIFEST" ]]; then
  echo "MANIFEST_EXISTS $MANIFEST"
  cat "$MANIFEST"
  exit 0
fi

# Python helper to list once and produce CDN manifest
python3 - "$REPO" "$DATE" "$MANIFEST" <<'PY'
import os, json, sys, urllib.request
from huggingface_hub import list_repo_tree, hf_hub_download

repo = sys.argv[1]
date = sys.argv[2]
out = sys.argv[3]

# List only top-level for the date folder (avoids recursive pagination)
items = list_repo_tree(repo, path=date, repo_type="dataset")
files = [f.rstrip("/") for f in items if f.endswith((".jsonl", ".parquet", ".json"))]

manifest = []
for f in files:
    cdn_url = f"https://huggingface.co/datasets/{repo}/resolve/main/{f}"
    local_path = hf_hub_download(repo_id=repo, filename=f, repo_type="dataset")
    manifest.append({
        "file": f,
        "local_path": local_path,
        "cdn_url": cdn_url
    })

os.makedirs(os.path.dirname(out), exist_ok=True)
with open(out, "w") as fp:
    json.dump(manifest, fp, indent=2)

print(json.dumps(manifest, indent=2))
PY

# Update last ingested state
python3 - "$DATE" <<'PY'
import os, json, sys
state_dir = "/opt/axentx/vanguard/.state"
os.makedirs(state_dir, exist_ok=True)
with open(os.path.join(state_dir, "last_ingested.json"), "w") as f:
    json.dump({"date": sys.argv[1], "ts": __import__("time").time()}, f)
PY

# Reuse or start Lightning Studio and run training with manifest
python3 /opt/axentx/vanguard/ops/run_studio_train.py --manifest "$MANIFEST"
```

```python
# /opt/axentx/vanguard/ops/run_studio_train.py
import json
import sys
from pathlib import Path
from lightning import Lightning, Teamspace, Studio, Machine

MANIFEST = Path(sys.argv[sys.argv.index("--manifest") + 1]) if "--manifest" in sys.argv else None
assert MANIFEST and MANIFEST.exists(), "Provide --manifest"

# Reuse running studio if available
team = Teamspace()
studio = None
for s in team.studios:
    if s.name == "vanguard-train" and s.status == "Running":
        studio = s
        print(f"Reusing running studio: {s.id}")
        break

if studio is None:
    # Prefer H200 in lambda-prod; fallback to L40S in public
    try:
        studio = Studio(
            name="vanguard-train",
            machine=Machine.L40S,
            cloud="lightning-public-prod",
            create_ok=True
        )
    except Exception:
        studio = Studio(
            name="vanguard-train",
            machine=Machine.L40S,
            cloud="lightning-public-prod",
            create_ok=True
        )

# Ensure running
if studio.status != "Running":
    studio.start(machine=Machine.L40S)

# Run training (train.py must accept --manifest and use CDN-only loading)
run = studio.run(
    repo="https://github.com/axentx/vanguard",
    command=[
        "python3", "train.py",
        "--manifest", "/workspace/manifests/" + MANIFEST.name,
        "--epochs", "1",
        "--limit-train-batches", "10"
    ],
    sync_dir="/opt/axentx/vanguard",
)
print("RUN_STARTED", run)
```

```python
# /opt/axentx/vanguard/train.py  (minimal patch)
import argparse, json, requests, torch
from torch.utils.data import IterableDataset, DataLoader

class CDNManifestDataset(IterableDataset):
    def __init__(self, manifest_path, limit=None):
        with open(manifest_path) as f:
            self.items = json.load(f)
        if limit:
            self.items = self.items[:limit]

    def __iter__(self):
        for item in self.items:
            # CDN fetch (no HF API/auth)
            resp = requests.get(item["cdn_url"], timeout=30)
            resp.raise_for_status()
            # Minimal projection: expect {prompt,response} jsonl lines
            for line in resp.text.strip().splitlines():
                if not line.strip():
                    continue
                obj = json.loads(line)
                yield obj.get("prompt", ""), obj.get("response", "")

def main():
    p = argparse.ArgumentParser()
    p.add_argument("--manifest", required=True)
    p.add_argument("--epochs", type=int, default=1)
    p.add_argument("--limit-train-batches", type=int, default=None)
    args = p.parse_args()

    ds = CDNManifestDataset(args.manifest, limit=args.limit_train_batches)
    loader = DataLoader(ds, batch_size=4, num_workers=0)

    for epoch in range(args.epochs):
        for x, y in loader:
            # placeholder training step
            print(f"epoch={epoch} batch={x[:1]}")
    print("TRAIN_DONE")

if __name__ == "__main__":
    main()
```
