## Final Decision — Ship: Frontend-only “Top Hub Signal Panel” (CDN-first)

**Scope**: Add a dashboard panel to Costinel that surfaces the most-connected hub (e.g., MOC) and related docs using CDN-only JSON (no Authorization header).  
**Timebox**: <2h

- 2026-04-27 top-hub doc insight (#knowledge-rag #graph #hub)

### 2) TopHubSignalPanel component

```tsx
// src/components/TopHubSignalPanel.tsx
import { useEffect, useState } from "react";

type Hub = {
  id: string;
  label: string;
  description?: string;
  importance?: number;
  connections?: number;
  lastUpdated?: string;
};

type Doc = {
  id: string;
  title: string;
  summary: string;
  url: string;
};

type TopHubData = {
  hub: Hub;
  docs: Doc[];
};

const CDN_URL =
  "https://huggingface.co/datasets/axentx/costinel-knowledge/resolve/main/top-hub.json";
const LOCAL_URL = "/data/top-hub.json";

export default function TopHubSignalPanel() {
  const [data, setData] = useState<TopHubData | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    async function fetchData(url: string, isFallback = false) {
      try {
        const res = await fetch(url, { cache: "no-store" });
        if (!res.ok) throw new Error(`HTTP ${res.status}`);
        const json = (await res.json()) as TopHubData;
        setData(json);
        setError(null);
      } catch (err) {
        if (!isFallback) {
          // fallback to local
          await fetchData(LOCAL_URL, true);
        } else {
          setError("Unable to load top-hub insights.");
        }
      } finally {
        setLoading(false);
      }
    }

    fetchData(CDN_URL);
  }, []);

  if (loading) {
    return (
      <div className="rounded-lg border bg-white p-4 shadow-sm">
        <p className="text-sm text-gray-500">Loading top hub insights…</p>
      </div>
    );
  }

  if (error || !data) {
    return (
      <div className="rounded-lg border bg-white p-4 shadow-sm">
        <p className="text-sm text-red-600">{error}</p>
      </div>
    );
  }

  return (
    <div className="rounded-lg border bg-white p-4 shadow-sm">
      <h3 className="mb-3 text-sm font-semibold text-gray-900">Top Hub Signal</h3>

      <div className="mb-4 rounded-md bg-blue-50 p-3">
        <p className="text-xs font-medium text-blue-800">{data.hub.label}</p>
        <p className="text-lg font-bold text-blue-900">
          {data.hub.connections?.toLocaleString() ?? "—"}
        </p>
        <p className="text-xs text-blue-700">connections</p>
        {data.hub.description && (
          <p className="mt-1 text-xs text-blue-800/80">{data.hub.description}</p>
        )}
      </div>

      <ul className="space-y-2">
        {data.docs.map((doc) => (
          <li key={doc.id}>
            <a
              href={doc.url}
              target="_blank"
              rel="noopener noreferrer"
              className="block rounded border p-2 text-sm hover:bg-gray-50"
            >
              <p className="font-medium text-gray-900">{doc.title}</p>
              <p className="text-gray-600">{doc.summary}</p>
            </a>
          </li>
        ))}
      </ul>
    </div>
  );
}
```
