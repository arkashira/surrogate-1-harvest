# Costinel / discovery

## Final Implementation Plan — Top-Hub Signal Panel (Costinel Dashboard)

**Scope & Value**  
- Frontend-only, read-only React panel that surfaces the highest-signal / most-connected hub (default “MOC”) and its actionable proposals from the knowledge graph.  
- CDN-first data: a single JSON file (`/data/hubs/moc/signals.json`) is generated by backend/CI and served via CDN; frontend fetches it client-side with zero backend compute and no auth/rate-limit concerns.  
- Ships in <2h: one new component tree + route mount + CDN asset + tests.

```ts
export type ProposalSeverity = 'critical' | 'high' | 'medium' | 'low';

export interface Proposal {
  id: string;

  link?: string;
  kpi?: HubKPI;
```

```tsx
// src/components/TopHubSignalPanel/SignalCard.tsx
import React from 'react';
import type { Proposal } from '../../types/knowledge-graph';

const severityColor = (s: Proposal['severity']) => {
  switch (s) {
    case 'critical':
      return 'text-red-700 bg-red-50 border-red-200';
    case 'high':
      return 'text-orange-700 bg-orange-50 border-orange-200';
    case 'medium':
      return 'text-amber-700 bg-amber-50 border-amber-200';
    default:
      return 'text-blue-700 bg-blue-50 border-blue-200';
  }
};

interface SignalCardProps {
  proposal: Proposal;
}

export const SignalCard: React.FC<SignalCardProps> = ({ proposal }) => (
  <div
    className={`rounded-lg border-l-4 p-3 ${severityColor(proposal.severity)}`}
    role="article"
    aria-label={`Proposal: ${proposal.title} (${proposal.severity})`}
  >
    <div className="flex items-start justify-between gap-2">
      <div>
        <p className="text-sm font-semibold text-gray-900">{proposal.title}</p>
        <p className="text-xs text-gray-600">{proposal.description}</p>
        {proposal.impact && (
          <p className="mt-1 text-xs font-medium text-gray-700">
            Est. impact: {proposal.impact}
          </p>
        )}
      </div>
      <span
        className={`whitespace-nowrap rounded-full px-2 py-0.5 text-xs font-medium ${
          proposal.severity === 'critical'
            ? 'bg-red-100 text-red-700'
            : proposal.severity === 'high'
            ? 'bg-orange-100 text-orange-700'
            : proposal.severity === 'medium'
            ? 'bg-amber-100 text-amber-700'
            : 'bg-blue-100 text-blue-700'
        }`}
      >
        {proposal.severity}
      </span>
    </div>
  </div>
);
```
