### 1) Pre-list file paths once (Mac orchestration) — hardened

- One-shot script runs **after** rate-limit window clears.  
- Uses `list_repo_tree` (non-recursive) to avoid pagination and API quota burn.  
- Emits deterministic, sorted manifest.  
- Manifest is committed to the training container/volume so training uses **zero Authorization headers**.

`scripts/build_hf_manifest.py`
```python
#!/usr/bin/env python3
"""
Usage:
  python scripts/build_hf_manifest.py \
    --repo datasets/axentx/surrogate-mirror \
    --date 2026-05-03 \
    --out surrogate/file_manifest.json
"""
import argparse
import json
import os
from huggingface_hub import HfApi

def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True)
    parser.add_argument("--date", required=True)
    parser.add_argument("--out", required=True)
    args = parser.parse_args()

    api = HfApi()
    folder = f"batches/mirror-merged/{args.date}"
    entries = api.list_repo_tree(repo_id=args.repo, path=folder, recursive=False)

    files = sorted(
        e.rfilename
        for e in entries
        if e.rfilename.lower().endswith(".parquet")
    )

    os.makedirs(os.path.dirname(args.out), exist_ok=True)
    with open(args.out, "w") as f:
        json.dump(files, f, indent=2)

    print(f"Wrote {len(files)} files to {args.out}")

if __name__ == "__main__":
    main()
```

### 2) CDN-only PyTorch IterableDataset

- Manifest baked into container or mounted via volume.  
- Fetch via CDN URLs (`resolve/main/...`) — no Authorization header required.  
- Stream rows from Parquet without materializing full files in RAM.  
- Project only `prompt`/`response`; skip malformed rows.

`surrogate/data/hf_cdn_dataset.py`
```python
from pathlib import Path
from typing import Iterator, Dict, Any
import json
import pyarrow.parquet as pq
import pyarrow as pa
import torch
from torch.utils.data import IterableDataset
import requests

HF_REPO = "datasets/axentx/surrogate-mirror"
CDN_ROOT = f"https://huggingface.co/datasets/{HF_REPO}/resolve/main"

class HFCDNDataset(IterableDataset):
    """
    Lightning training uses ONLY CDN fetches.
    Manifest is baked into the binary at launch (or mounted via volume).
    """

    def __init__(self, manifest_path: str | Path, max_files: int | None = None):
        manifest_path = Path(manifest_path)
        with manifest_path.open() as f:
            self.files: list[str] = json.load(f)

        if max_files:
            self.files = self.files[:max_files]

        if not self.files:
            raise ValueError(f"No files found in {manifest_path}")

    def _stream_rows(self) -> Iterator[Dict[str, Any]]:
        for rpath in self.files:
            url = f"{CDN_ROOT}/{rpath}"
            resp = requests.get(url, timeout=30)
            resp.raise_for_status()

            with pa.BufferReader(resp.content) as reader:
                table = pq.read_table(reader)

            if "prompt" not in table.column_names or "response" not in table.column_names:
                continue

            prompts = table["prompt"].to_pylist()
            responses = table["response"].to_pylist()

            for p, r in zip(prompts, responses):
                if isinstance(p, str) and isinstance(r, str) and p.strip() and r.strip():
                    yield {"prompt": p.strip(), "response": r.strip()}

    def __iter__(self) -> Iterator[Dict[str, Any]]:
        yield from self._stream_rows()
```

### 3) Lightning idle-resilient runner

- Reuse running studio if available; otherwise create/start one.  
- If studio is stopped (idle timeout), restart it and retry training.  
- Training progress is preserved by Lightning + Studio checkpointing; loop retries until completion.

`surrogate/train/run_surrogate.py`
```python
#!/usr/bin/env python3
import time
from pathlib import Path

import lightning as L
from lightning.fabric.plugins import TorchTensorBoardLogger

from surrogate.model import SurrogateModel
from surrogate.data.hf_cdn_dataset import HFCDNDataset
from surrogate.train.config import TRAIN_CFG

MANIFEST = Path(__file__).parent.parent / "file_manifest.json"

def get_or_create_studio() -> L.studio.Studio:
    teamspace = L.Teamspace()
    for s in teamspace.studios:
        if s.name == TRAIN_CFG.studio_name and s.status == "running":
            print(f"Reusing running studio: {s.name}")
            return s

    print(f"Creating studio: {TRAIN_CFG.studio_name}")
    return L.studio.Studio(
        name=TRAIN_CFG.studio_name,
        create_ok=True,
        machine=TRAIN_CFG.machine,
    )

def train_loop(studio: L.studio.Studio) -> None:
    dataset = HFCDNDataset(manifest_path=MANIFEST, max_files=TRAIN_CFG.max_files)

    model = SurrogateModel(cfg=TRAIN_CFG.model)
    trainer = L.Trainer(
        max_epochs=TRAIN_CFG.epochs,
        logger=TorchTensorBoardLogger(save_dir="lightning_logs"),
        accelerator="gpu",
        devices=1,
        precision="bf16-mixed",
    )

    studio.run(trainer.fit, model=model, train_dataloaders=dataset)

def main() -> None:
    while True:
        studio = get_or_create_studio()

        if studio.status != "running":
            print(f"Studio {studio.name} is {studio.status}. Restarting...")
            studio.start(machine=TRAIN_CFG.machine)
            while studio.status != "running":
                time.sleep(10)
                studio.refresh()

        try:
            train_loop(studio)
            print("Training completed.")
            break
        except Exception as exc:
            print(f"Training failed: {exc}")
            time.sleep(30)

if __name__ == "__main__":
    main()
```

  - Use `studio.run(...)` with explicit `model` and `train_dataloaders` (fixes Candidate 1 trainer/model wiring).
