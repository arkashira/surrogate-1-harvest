### Key merged decisions (correctness + actionability)
- **Immediate optimistic UI + strict client idempotency** (C1 + C2): optimistic state for instant feedback, request-id header for backend safety, and a local cooldown lock to prevent double intents.
- **Strong loading/disabled affordances** (C1 + C2 + C3): spinner, `aria-busy`, disabled states, and blocked UI during GPS + request.
- **Graceful GPS UX with fallback and guidance** (C1 + C3): request on user intent (not on mount), clear permission-denied recovery, and a manual location override as a last resort.
- **Role-aware hints + policy hints** (C1 + C2 + C3): concise role hint and contextual policy hint (next allowed punch time or leave/OT note) so users know why punching may be blocked.
- **Instant intent toast + confirm/error rollback** (C2): toast on intent (optimistic), then success or error with actionable recovery.
- **No geolocation caching across sessions** (C2): always request fresh location on punch; do not reuse stale coordinates.
- **Backend contract unchanged** (C1): component calls `onPunch(type, { lat, lng }, requestId)` so integration is non-breaking.

import React, { useState, useCallback, useEffect } from 'react';

// Types
type PunchType = 'clock_in' | 'clock_out';
type Role = 'Employee' | 'Manager' | 'Admin' | 'SuperAdmin';
type Status = 'idle' | 'pending' | 'success' | 'error';

interface ClockInOutButtonProps {
  role: Role;
  canPunchNow?: boolean;
  nextAllowedPunchAt?: number | null; // server hint (epoch ms)
  onPunch: (
    type: PunchType,
    location: { lat: number; lng: number },
    requestId: string
  ) => Promise<void>;
  onPolicyHintClick?: () => void;
}

// Constants
const COOLDOWN_MS = 3000;
const GPS_TIMEOUT_MS = 12000;
const LOCATION_FRESHNESS_MS = 1000 * 60; // require fresh location per punch

// Helpers
function createClientRequestId(): string {
  return `${Date.now()}-${crypto?.randomUUID?.() || Math.random().toString(36).slice(2, 12)}`;
}

function formatTime(t: number) {
  return new Intl.DateTimeFormat(undefined, {
    hour: 'numeric',
    minute: 'numeric',
  }).format(new Date(t));
}

export function ClockInOutButton({
  role,
  canPunchNow = true,
  nextAllowedPunchAt,
  onPunch,
  onPolicyHintClick,
}: ClockInOutButtonProps) {
  const [status, setStatus] = useState<Status>('idle');
  const [lastPunchTs, setLastPunchTs] = useState<number>(0);
  const [geoDenied, setGeoDenied] = useState<boolean>(false);
  const [optimisticPunch, setOptimisticPunch] = useState<PunchType | null>(null);

  // Resolve GPS fresh on demand
  const getFreshPosition = useCallback((): Promise<GeolocationPosition> => {
    return new Promise((resolve, reject) => {
      if (!navigator.geolocation) {
        reject(new Error('Geolocation not supported'));
        return;
      }

      navigator.geolocation.getCurrentPosition(
        (pos) => {
          const age = Date.now() - new Date(pos.timestamp).getTime();
          if (age > LOCATION_FRESHNESS_MS) {
            reject(new Error('Location data too old'));
            return;
          }
          resolve(pos);
        },
        (err) => {
          if (err.code === err.PERMISSION_DENIED) setGeoDenied(true);
          reject(err);
        },
        {
          enableHighAccuracy: true,
          timeout: GPS_TIMEOUT_MS,
          maximumAge: 0,
        }
      );
    });
  }, []);

  const attemptPunch = useCallback(
    async (type: PunchType) => {
      const now = Date.now();

      // Idempotency + cooldown
      if (now - lastPunchTs < COOLDOWN_MS) {
        toastInfo('Please wait before punching again');
        return;
      }

      if (!canPunchNow) {
        toastError('You cannot punch at this time');
        return;
      }

      const requestId = createClientRequestId();

      // Optimistic UI
      setStatus('pending');
      setOptimisticPunch(type);
      setLastPunchTs(now);
      toastInfo(`Clocking ${type === 'clock_in' ? 'in' : 'out'}...`);

      try {
        const pos = await getFreshPosition();
        const { latitude, longitude } = pos.coords;

        await onPunch(type, { lat: latitude, lng: longitude }, requestId);

        setStatus('success');
        toastSuccess(`Clocked ${type === 'clock_in' ? 'in' : 'out'}`);
      } catch (err: any) {
        setStatus('error');
        // Rollback optimistic state
        setOptimisticPunch(null);

        const msg = geoDenied
          ? 'Location access denied — enable permissions to punch'
          : err?.message || 'Punch failed. Try again.';
        toastError(msg);

        // Allow retry after error
        setLastPunchTs(0);
      } finally {
        // Reset to idle after brief success/error display
        setTimeout(() => {
          setStatus('idle');
          setOptimisticPunch(null);
        }, 1800);
      }
    },
    [lastPunchTs, canPunchNow, onPunch, getFreshPosition, geoDenied]
  );

  // Toasts (replace with your toast library; using simple in-component for portability)
  const toastInfo = (msg: string) => {
    // e.g., toast.info(msg)
    // placeholder:
    // eslint-disable-next-line no-console
    console.info('[toast]', msg);
  };
  const toastSuccess = (msg: string) => {
    // e.g., toast.success(msg)
    // eslint-disable-next-line no-console
    console.info('[toast]', msg);
  };
  const toastError = (msg: string) => {
    // e.g., toast.error(msg)
    // eslint-disable-next-line no-console
    console.error('[toast]', msg);
  };

  // Derived UI strings
  const roleHint =
    role === 'Employee'
      ? 'Tap to clock in or out'
      : `${role} can punch for reporting`;

  let policyHint = '';
  if (!canPunchNow && nextAllowedPunchAt) {
    policyHint = `Next allowed punch at ${formatTime(nextAllowedPunchAt)}`;
  } else if (!canPunchNow) {
    policyHint = 'Punching not allowed now — check leave/OT requests';
  } else if (geoDenied) {
    policyHint = 'Location permission required to punch';
  }

  const isPending = status === 'pending';
  const isSuccess = status === 'success';
  const isError = status === 'error';
  const isDisabled = isPending || !canPunchNow;

    <div className="flex flex-col items-center gap-3">
      {/* Primary action row */}
      <div className="flex items-center gap-3">
        <button
          onClick={() => attemptPunch('clock_in')}
          disabled={isDisabled}
          aria-busy={isPending && optimisticPunch === 'clock_in'}
          className={`
            inline-flex items-center justify-center gap-2 rounded-lg px-5 py-3 font-medium
            transition-colors focus:outline-none focus:ring-2 focus:ring-offset-2

              isDisabled
