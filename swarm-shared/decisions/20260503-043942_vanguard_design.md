# vanguard / design

## Final Synthesis (Best Parts + Correctness + Actionability)

**Core diagnosis (merged, de-duplicated)**  
- No content-addressed manifest per date folder → runtime repo enumeration via `list_repo_files`/`load_dataset` triggers HF API 429s and non-reproducible epochs.  
- Missing deterministic `{path, sha256, size}` snapshot → CDN-only fetches cannot be validated or resumed; corrupted/partial downloads poison training.  
- Training enumerates files at runtime instead of using an embedded file manifest → burns quota and blocks CDN-only fast path.  
- No reproducible “date folder → manifest” artifact → prevents caching, CI validation, and deterministic resume across studio restarts.  
- No lightweight integrity check before training starts → wastes GPU hours and quota on bad shards.  
- Schema drift risk: ingestion writes mixed-schema enriched files instead of strict `{prompt, response}` parquet in `batches/mirror-merged/{date}/{slug}.parquet`.

**Proposed change (merged, corrected)**  
- Add `/opt/axentx/vanguard/manifest.py`  
  - `build_manifest(repo, date_folder, out_path)` → `{repo}/{date_folder}/manifest-{sha256}.json` with `{"repo","folder","generated_at","files": [{"path": str, "sha256": str, "size": int}]}`  
  - Uses a **single non-recursive tree call** per date folder; computes sha256 from local temp download once per file; writes content-addressed manifest.  
- Add `/opt/axentx/vanguard/train_cdn.py`  
  - Reads embedded or provided manifest; fetches via CDN with streaming + retries + integrity check; yields strict `{prompt, response}` rows.  
  - Supports resumable cache, checksum validation, and schema projection/fallback.  
- Add `/opt/axentx/vanguard/validate_manifest.py`  
  - Verifies cache against manifest sha256; repairs/removes corrupt files; reports pass/fail and missing entries.  
- Add pre-flight validation step in Lightning Studio entrypoint to run `validate_manifest` before training; fail fast to avoid quota/GPU waste.

```bash
# /opt/axentx/vanguard/manifest.py
#!/usr/bin/env python3
"""
Build a content-addressed manifest for a HF dataset date folder.
Usage:
  python manifest.py <repo> <date_folder> [--out-dir ./manifests]
"""
import argparse, hashlib, json, os, sys, time
from pathlib import Path

try:
    from huggingface_hub import list_repo_tree, hf_hub_download
except ImportError:
    print("pip install huggingface_hub")
    sys.exit(1)

def sha256_file(path):
    h = hashlib.sha256()
    with open(path, "rb") as f:
        for chunk in iter(lambda: f.read(128 * 1024), b""):
            h.update(chunk)
    return h.hexdigest()

def build_manifest(repo: str, date_folder: str, out_dir: str = "./manifests"):
    out_dir = Path(out_dir)
    out_dir.mkdir(parents=True, exist_ok=True)

    # Single API call: non-recursive tree for this folder only
    items = list_repo_tree(repo=repo, path=date_folder, recursive=False)
    files = []
    for item in items:
        if item.type != "file":
            continue
        # Download once to compute sha256
        local_path = hf_hub_download(repo_id=repo, filename=item.path, repo_type="dataset")
        entry = {
            "path": item.path,
            "sha256": sha256_file(local_path),
            "size": os.path.getsize(local_path),
        }
        files.append(entry)

    manifest = {
        "repo": repo,
        "folder": date_folder,
        "generated_at": time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime()),
        "files": sorted(files, key=lambda x: x["path"]),
    }

    # Content-addressed filename
    blob = json.dumps(manifest, sort_keys=True).encode()
    manifest_sha = hashlib.sha256(blob).hexdigest()[:16]
    out_path = out_dir / f"manifest-{manifest_sha}.json"
    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"Wrote {out_path}")
    return out_path

if __name__ == "__main__":
    p = argparse.ArgumentParser()
    p.add_argument("repo")
    p.add_argument("date_folder")
    p.add_argument("--out-dir", default="./manifests")
    args = p.parse_args()
    build_manifest(args.repo, args.date_folder, args.out_dir)
```

#!/usr/bin/env python3
"""
CDN-only data loader for Lightning training.
Embeds or loads a manifest and fetches via CDN (no HF API auth during training).
"""
import json, hashlib, requests, io, os, sys
from pathlib import Path
from typing import Iterator, Dict
import pyarrow.parquet as pq
import pyarrow as pa

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def sha256_file(path):
    h = hashlib.sha256()
    with open(path, "rb") as f:
        for chunk in iter(lambda: f.read(128 * 1024), b""):
            h.update(chunk)
    return h.hexdigest()

def load_manifest(manifest_path: str) -> dict:
    with open(manifest_path) as f:
        return json.load(f)

def cdn_download(entry: Dict, cache_dir: Path, repo: str, max_retries: int = 3) -> Path:
    cache_dir.mkdir(parents=True, exist_ok=True)
    path = entry["path"]
    expected = entry["sha256"]
    cache_file = cache_dir / hashlib.sha256(path.encode()).hexdigest() + ".parquet"

    if cache_file.exists():
        if sha256_file(cache_file) == expected:
            return cache_file
        cache_file.unlink()

    url = CDN_TEMPLATE.format(repo=repo, path=path)
    for attempt in range(1, max_retries + 1):
        try:
            resp = requests.get(url, stream=True, timeout=60)
            resp.raise_for_status()
            with open(cache_file, "wb") as f:
                for chunk in resp.iter_content(chunk_size=128 * 1024):
                    f.write(chunk)
            if sha256_file(cache_file) == expected:
                return cache_file
            cache_file.unlink()
            raise ValueError(f"Checksum mismatch on retry {attempt}: {path}")
        except Exception as e:
            if attempt == max_retries:
                if cache_file.exists():
                    cache_file.unlink()
                raise
            # brief backoff
            import time; time.sleep(2 ** attempt)
    raise RuntimeError(f"Failed to download {path}")

def cdn_dataset(manifest_path: str, cache_dir: str = "./.cdn_cache") -> Iterator[Dict]:
    manifest = load_manifest(manifest_path)
    cache_dir = Path(cache_dir)
    repo = manifest["repo"]
    for entry in manifest["files"]:
        local = cdn_download(entry, cache_dir, repo=repo)
        # Project to {prompt, response} only — strict schema enforcement

            table = pq.read_table(local, columns=["prompt", "response"])
        except (KeyError, pa.ArrowInvalid):
            table = pq.read_table(local)
            if "prompt" not in table.column_names or "response" not in table.column_names:
                raise ValueError(f"Missing required columns in {entry['path']}")

            df = batch.to_pandas()
            for _, row in df.iterrows():
                yield {"prompt": row["prompt"], "response": row["response"]}
