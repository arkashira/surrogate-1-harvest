# workio / frontend

## Final synthesized solution

### Diagnosis (merged)
- No optimistic update / rollback → ambiguous spinners and silent failures when LINE/webhook calls fail.
- No local pending queue → punches are lost on offline/poor connectivity and users have no feedback or retry path.
- No idempotency handling → double-taps or retries can create duplicate punches.
- No clear visual states (pending/syncing/synced/failed) → users can’t tell whether a punch was accepted.
- No persisted retry or reconnect handling → failed punches require manual retry and are not retried in background.

### Single concrete change
Create/replace `src/features/clock/ClockButton.tsx` and add `src/lib/queue.ts`.  
Scope: optimistic UI + persisted local queue + idempotency + background retry-on-reconnect + clear state visuals.

### 1) Lightweight persisted queue (`src/lib/queue.ts`)

```ts
// src/lib/queue.ts
import { v4 as uuidv4 } from 'uuid';

export type Punch = {
  id: string;
  type: 'IN' | 'OUT';
  timestamp: number;
  lat: number;
  lng: number;
  attempts: number;
  createdAt: number;
  lastAttemptAt?: number;
  failed?: boolean;
};

const STORAGE_KEY = 'workio:pending-punches';
const MAX_ATTEMPTS = 5;

class PunchQueue {
  private queue: Punch[] = [];

  constructor() {
    this.load();
  }

  private load() {
    try {
      const raw = localStorage.getItem(STORAGE_KEY);
      this.queue = raw ? JSON.parse(raw) : [];
    } catch {
      this.queue = [];
    }
  }

  private persist() {
    try {
      localStorage.setItem(STORAGE_KEY, JSON.stringify(this.queue));
    } catch {
      // ignore quota/incognito storage failures
    }
  }

  enqueue(punch: Omit<Punch, 'id' | 'attempts' | 'createdAt' | 'failed'>) {
    const item: Punch = {
      ...punch,
      id: uuidv4(),
      attempts: 0,
      createdAt: Date.now(),
      failed: false,
    };
    this.queue.push(item);
    this.persist();
    return item.id;
  }

  peek() {
    return this.queue[0] || null;
  }

  markAttempt(id: string) {
    const item = this.queue.find((q) => q.id === id);
    if (item) {
      item.attempts += 1;
      item.lastAttemptAt = Date.now();
      this.persist();
    }
  }

  remove(id: string) {
    this.queue = this.queue.filter((q) => q.id !== id);
    this.persist();
  }

  markFailed(id: string) {
    const item = this.queue.find((q) => q.id === id);
    if (item) {
      item.failed = true;
      this.persist();
    }
  }

  shouldDrop(id: string) {
    const item = this.queue.find((q) => q.id === id);
    return item ? item.attempts >= MAX_ATTEMPTS : false;
  }

  getAll() {
    return [...this.queue];
  }

  // allow external drain trigger
  size() {
    return this.queue.length;
  }
}

export const punchQueue = new PunchQueue();
```

### 2) ClockButton with optimistic UI and queue integration (`src/features/clock/ClockButton.tsx`)

```tsx
// src/features/clock/ClockButton.tsx
import React, { useEffect, useState, useCallback } from 'react';
import { punchQueue, type Punch } from '../../lib/queue';
import { postPunch } from '../../lib/api';

type Status = 'idle' | 'pending' | 'syncing' | 'synced' | 'failed';

export function ClockButton() {
  const [status, setStatus] = useState<Status>('idle');
  const [lastPunch, setLastPunch] = useState<{ type: 'IN' | 'OUT'; ts: number } | null>(null);
  const [queueCount, setQueueCount] = useState(punchQueue.size());

  // reflect queue size changes (lightweight polling; can be replaced with EventTarget)
  useEffect(() => {
    const update = () => setQueueCount(punchQueue.size());
    const id = setInterval(update, 1500);
    update();
    return () => clearInterval(id);
  }, []);

  // background retry + reconnect handling
  useEffect(() => {
    let mounted = true;

    const processQueue = async () => {
      while (mounted) {
        const next = punchQueue.peek();
        if (!next) break;
        if (!navigator.onLine) break;

        punchQueue.markAttempt(next.id);
        try {
          await postPunch({
            type: next.type,
            timestamp: next.timestamp,
            lat: next.lat,
            lng: next.lng,
            // Idempotency key prevents duplicates if same punch is retried
            idempotencyKey: `punch:${next.id}`,
          });
          punchQueue.remove(next.id);
          setLastPunch({ type: next.type, ts: next.timestamp });
          setStatus('synced');
          setTimeout(() => setStatus('idle'), 1800);
        } catch (err) {
          if (punchQueue.shouldDrop(next.id)) {
            punchQueue.remove(next.id);
          } else {
            punchQueue.markFailed(next.id);
          }
          // stop loop on failure; reconnect or next online event will retry
          break;
        }
      }
    };

    const onOnline = () => processQueue();
    window.addEventListener('online', onOnline);
    // initial drain
    processQueue();

    return () => {
      mounted = false;
      window.removeEventListener('online', onOnline);
    };
  }, []);

  const handlePunch = useCallback(async (type: 'IN' | 'OUT') => {
    if (!navigator.geolocation) {
      alert('Geolocation not supported');
      return;
    }

    setStatus('pending');

    navigator.geolocation.getCurrentPosition(
      (pos) => {
        const punchData = {
          type,
          timestamp: Date.now(),
          lat: pos.coords.latitude,
          lng: pos.coords.longitude,
        };

        // Optimistic UI
        setLastPunch({ type, ts: punchData.timestamp });
        setStatus('syncing');

        // Persist for guaranteed delivery
        punchQueue.enqueue(punchData);
        // Background effect will attempt delivery immediately
      },
      () => {
        alert('Location required to clock in/out');
        setStatus('failed');
        setTimeout(() => setStatus('idle'), 2000);
      },
      { enableHighAccuracy: true, timeout: 10000, maximumAge: 30000 }
    );
  }, []);

  const nextAction = lastPunch?.type === 'IN' && status !== 'pending' ? 'OUT' : 'IN';
  const isWorking = lastPunch?.type === 'IN' && status !== 'OUT';

  return (
    <div className="flex flex-col items-center gap-3">
      <button
        onClick={() => handlePunch(nextAction)}
        disabled={status === 'pending' || status === 'syncing'}
        className={`
          px-8 py-4 rounded-xl text-lg font-semibold text-white shadow-lg transition
          ${status === 'pending' || status === 'syncing'

            : isWorking
            ? 'bg-red-500 hover:bg-red-600'
            : 'bg-green-600 hover:bg-green-700'
          }
        `}
      >
        {status === 'pending' && 'Getting location...'}
        {status === 'syncing' && 'Sending...'}
        {status === 'idle' && (isWorking ? 'Clock OUT' : 'Clock IN')}
        {status === 'failed' && 'Location failed — retry'}
