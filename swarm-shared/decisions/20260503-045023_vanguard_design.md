```python
#!/usr/bin/env python3
"""
Generate a content-addressed manifest for one date folder.
Usage:
  python manifest.py --repo <datasets/repo> --date 2026-05-03 --out manifest.json
"""

import argparse
import hashlib
import json
import os
import sys
from typing import List, Dict

from huggingface_hub import list_repo_tree, hf_hub_download

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"


def sha256_of_local(path: str) -> str:
    h = hashlib.sha256()
    with open(path, "rb") as f:
        for chunk in iter(lambda: f.read(128 * 1024), b""):
            h.update(chunk)
    return h.hexdigest()


def build_manifest(repo: str, date: str, out_path: str) -> List[Dict]:
    """
    Build manifest for repo/batches/mirror-merged/{date}/
    Uses HF API once (list_repo_tree non-recursive) then CDN URLs only.
    """
    prefix = f"batches/mirror-merged/{date}/"
    items = list_repo_tree(repo=repo, path=prefix, recursive=False)

    files = []
    for item in items:
        if hasattr(item, "path"):
            p = item.path
            if p.endswith("/"):
                continue
            files.append(p)
        elif isinstance(item, str):
            files.append(item)
        else:
            raise ValueError(f"Unexpected tree item: {item}")

    if not files:
        print(f"No files found under {prefix}", file=sys.stderr)
        sys.exit(1)

    manifest = []
    for fpath in sorted(files):
        local_path = hf_hub_download(repo_id=repo, filename=fpath, repo_type="dataset")
        digest = sha256_of_local(local_path)
        size = os.path.getsize(local_path)

        manifest.append(
            {
                "repo": repo,
                "path": fpath,
                "url": CDN_TEMPLATE.format(repo=repo, path=fpath),
                "sha256": digest,
                "size": size,
            }
        )
        os.unlink(local_path)

    os.makedirs(os.path.dirname(os.path.abspath(out_path)), exist_ok=True)
    with open(out_path, "w") as f:
        json.dump(manifest, f, indent=2)

    print(f"Wrote {len(manifest)} entries to {out_path}")
    return manifest


if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Generate CDN manifest for a date folder.")
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g. org/data)")
    parser.add_argument("--date", required=True, help="Date folder (YYYY-MM-DD)")
    parser.add_argument("--out", default="manifest.json", help="Output JSON path")
    args = parser.parse_args()

    build_manifest(args.repo, args.date, args.out)
```

Lightning training shim that uses CDN-only fetches (zero HF API calls during training).

- Restarts studio if stopped by idle timeout before run.
- Deterministic shard selection + idle-stop resilience.

import time

from lightning.pytorch import Trainer

    STUDIO_AVAILABLE = True

    STUDIO_AVAILABLE = False
    Studio = Teamspace = Machine = None


def load_manifest(manifest_path: str):
    with open(manifest_path) as f:
        return json.load(f)


def get_or_create_studio(name: str, machine: str = "L40S"):
    """Reuse running studio; restart if stopped. Avoids quota burn."""
    if not STUDIO_AVAILABLE:
        print("Studio unavailable — skipping studio reuse.")
        return None

    teamspace = Teamspace()
    for s in teamspace.studios:
        if s.name == name:
            if s.status == "running":
                print(f"Reusing running studio: {name}")
                return s
            print(f"Studio {name} exists but stopped ({s.status}). Restarting...")
            s.start(machine=Machine.L40S)
            return s

    print(f"Creating studio: {name}")
    return Studio(name=name, machine=Machine.L40S, create_ok=True)


class CDNParquetIterable(IterableDataset):
    """Stream parquet from CDN; deterministic shard order from manifest."""

    def __init__(self, entries, max_samples_per_shard=None):
        self.entries = entries
        self.max_samples_per_shard = max_samples_per_shard

    def __iter__(self):
        for entry in self.entries:
            url = entry["url"]
            resp = requests.get(url, timeout=30)
            resp.raise_for_status()
            tmp = "/tmp/temp.parquet"
            with open(tmp, "wb") as f:
                f.write(resp.content)
            table = pq.read_table(tmp)
            cols = table.column_names

            has_prompt = "prompt" in cols
            has_response = "response" in cols
            if has_prompt and has_response:
                prompts = table["prompt"].to_pylist()
                responses = table["response"].to_pylist()
                limit = self.max_samples_per_shard
                for i, (p, r) in enumerate(zip(prompts, responses)):
                    if limit is not None and i >= limit:
                        break
                    # Replace with real tokenization in production
                    yield {"input_ids": torch.tensor([0]), "labels": torch.tensor([0])}
            else:
                # fallback dummy
                limit = self.max_samples_per_shard or 10
                for i in range(limit):
                    yield {"input_ids": torch.tensor([0]), "labels": torch.tensor([0])}
            os.unlink(tmp)


class SimpleDataModule:
