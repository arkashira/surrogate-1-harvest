## Final Implementation Plan (≤2h)

**Highest-value improvement**: Replace fragile shell-based ingestion with a **manifest-driven, CDN-bypass Python worker** that eliminates HF API rate limits during training data loads and guarantees deterministic shard assignment.

### Concrete steps (120 min budget)

1. **Create `bin/dataset-enrich.py`** (60 min)
   - Deterministic shard assignment via `hash(slug) % SHARD_TOTAL`
   - Single `list_repo_tree` call → save `file-list-<date>.json` (Mac orchestration)
   - Worker loads manifest, downloads only its shard via CDN URLs (`resolve/main/...`)
   - Schema projection to `{prompt, response}` only at parse time
   - Central dedup via existing `lib/dedup.py` (SQLite md5 store)
   - Output: `batches/public-merged/<date>/shard<N>-<HHMMSS>.jsonl`

2. **Update `.github/workflows/ingest.yml`** (20 min)
   - Switch from `bash dataset-enrich.sh` to `python bin/dataset-enrich.py`
   - Pass matrix `SHARD_ID`/`SHARD_TOTAL`, optional `DATE_FOLDER`
   - Keep `HF_TOKEN` for push only (not for CDN reads)

3. **Add retry/backoff + 429 handling** (15 min)
   - Wait 360s on 429 (HF API), exponential backoff on CDN 5xx
   - Skip already-downloaded files via local cache dir

4. **Cleanup & test** (25 min)
   - `chmod +x bin/dataset-enrich.py`
   - Dry-run locally with mock manifest

def deterministic_shard(slug: str) -> int:
    return int(hashlib.md5(slug.encode()).hexdigest(), 16) % SHARD_TOTAL

def get_or_create_manifest() -> List[str]:
    """Single API call to list root folder for DATE_FOLDER; cache manifest."""
    if MANIFEST_PATH.exists():
        log.info("Loading cached manifest %s", MANIFEST_PATH)
        return json.loads(MANIFEST_PATH.read_text())

    log.info("Listing repo tree for %s (single API call)", DATE_FOLDER)
    items = list_repo_tree(repo_id=REPO_ID, path=DATE_FOLDER, recursive=False)
    files = [it.rfilename for it in items if it.type == "file"]
    MANIFEST_PATH.parent.mkdir(parents=True, exist_ok=True)
    MANIFEST_PATH.write_text(json.dumps(files, indent=2))
    log.info("Saved manifest with %d files", len(files))
    return files

def cdn_download(url: str, dest: Path, max_retries: int = 5) -> bool:
    """Download via CDN (no auth/rate-limit for public datasets)."""
    dest.parent.mkdir(parents=True, exist_ok=True)
    if dest.exists():
        return True

    backoff = 1
    for attempt in range(max_retries):
        try:
            resp = requests.get(url, timeout=30)
            if resp.status_code == 429:
                wait = 360
                log.warning("HF CDN 429, waiting %ds", wait)
                time.sleep(wait)
                continue
            resp.raise_for_status()
            dest.write_bytes(resp.content)
            return True
        except Exception as exc:
            log.warning("Download attempt %d/%d failed: %s", attempt + 1, max_retries, exc)
            time.sleep(backoff)
            backoff = min(backoff * 2, 60)
    return False

def normalize_record(raw: Dict[str, Any], source_file: str) -> Dict[str, Any] | None:
    """
    Project heterogeneous schemas to {prompt, response}.
    Attribution moved to filename pattern; no extra metadata cols.
    """
    # Common surrogate-1 schema variants observed:
    # 1) {"prompt": "...", "response": "..."}
    # 2) {"input": "...", "output": "..."}
    # 3) {"instruction": "...", "completion": "..."}
    # 4) {"messages": [{"role": "user", "content": "..."}, {"role": "assistant", "content": "..."}]}

    prompt = raw.get("prompt") or raw.get("input") or raw.get("instruction")
    response = raw.get("response") or raw.get("output") or raw.get("completion")

    if isinstance(raw.get("messages"), list):
        # simple extract last assistant / first user
        msgs = raw["messages"]
        user_msgs = [m for m in msgs if isinstance(m, dict) and m.get("role") == "user"]
        assistant_msgs = [m for m in msgs if isinstance(m, dict) and m.get("role") == "assistant"]
        prompt = user_msgs[0].get("content") if user_msgs else None
        response = assistant_msgs[-1].get("content") if assistant_msgs else None

    if not isinstance(prompt, str) or not isinstance(response, str):
        log.debug("Skipping malformed record in %s: missing text fields", source_file)
        return None

    text = (prompt.strip() + "\n\n" + response.strip()).strip()
    md5 = hashlib.md5(text.encode()).hexdigest()
    if not dedup.add(md5):
        return None

    return {"prompt": prompt.strip(), "response": response.strip()}
