### 1) Pre-list files once (single API call)

`scripts/pre_list_files.py`
```python
#!/usr/bin/env python3
"""
Generate filelist JSON for a date folder in a dataset repo.
Usage:
    python3 scripts/pre_list_files.py \
        --repo datasets/your-org/surrogate-mirror \
        --date 2025-01-01 \
        --outfile filelist_20250101.json
"""
import argparse
import json
import sys

from huggingface_hub import list_repo_tree


def main() -> None:
    parser = argparse.ArgumentParser(description="List parquet files in a repo/date folder.")
    parser.add_argument("--repo", required=True, help="Repo id (e.g. datasets/your-org/surrogate-mirror)")
    parser.add_argument("--date", required=True, help="Date folder (YYYY-MM-DD)")
    parser.add_argument("--outfile", required=True, help="Output JSON path")
    args = parser.parse_args()

    date_dir = f"batches/mirror-merged/{args.date}"
    try:
        files = list_repo_tree(args.repo, path=date_dir, repo_type="dataset", recursive=False)
    except Exception as exc:
        print(f"Failed to list repo tree: {exc}", file=sys.stderr)
        sys.exit(1)

    items = sorted(f.rfilename for f in files if f.rfilename.endswith(".parquet"))
    with open(args.outfile, "w", encoding="utf-8") as f:
        json.dump(items, f, indent=2)
    print(f"Listed {len(items)} files -> {args.outfile}")


if __name__ == "__main__":
    main()
```

### 2) CDN-only bulk download (no auth/rate-limit)

`scripts/hf_cdn_download.py`
```python
import os
import requests
from concurrent.futures import ThreadPoolExecutor, as_completed
from tqdm import tqdm

HF_DATASETS_BASE = "https://huggingface.co/datasets"


def cdn_download(repo: str, filepath: str, dest: str, chunk_size: int = 8192) -> str:
    url = f"{HF_DATASETS_BASE}/{repo}/resolve/main/{filepath}"
    resp = requests.get(url, stream=True, timeout=30)
    resp.raise_for_status()

    total = int(resp.headers.get("content-length", 0))
    os.makedirs(os.path.dirname(dest), exist_ok=True)

    with open(dest, "wb") as f, tqdm(
        total=total, unit="B", unit_scale=True, desc=os.path.basename(filepath), leave=False
    ) as pbar:
        for chunk in resp.iter_content(chunk_size=chunk_size):
            if chunk:
                f.write(chunk)
                pbar.update(len(chunk))
    return dest


def bulk_cdn_download(
    repo: str,
    filepaths: list[str],
    dest_root: str = "data/cdn",
    max_workers: int = 8,
) -> list[str]:
    dests: list[str] = []
    with ThreadPoolExecutor(max_workers=max_workers) as ex:
        futures = {}
        for fp in filepaths:
            dest = os.path.join(dest_root, fp)
            if os.path.exists(dest):
                dests.append(dest)
                continue
            futures[ex.submit(cdn_download, repo, fp, dest)] = dest

        for future in as_completed(futures):
            try:
                dests.append(future.result())
            except Exception as exc:
                print(f"Download failed for {futures[future]}: {exc}")
                raise
    return dests
```

### 3) Deterministic sibling-repo sharding (bypass 128/hr cap)

`scripts/surrogate_train.py`
```python
import hashlib
import os
from pathlib import Path

from huggingface_hub import upload_file

SIBLING_REPOS = [
    "datasets/your-org/surrogate-mirror",
    "datasets/your-org/surrogate-mirror-s1",
    "datasets/your-org/surrogate-mirror-s2",
    "datasets/your-org/surrogate-mirror-s3",
    "datasets/your-org/surrogate-mirror-s4",
]


def pick_sibling(slug: str) -> str:
    h = int(hashlib.sha256(slug.encode()).hexdigest(), 16)
    return SIBLING_REPOS[h % len(SIBLING_REPOS)]


def upload_sharded(local_path: str, slug: str, date: str) -> None:
    repo = pick_sibling(slug)
    path_in_repo = f"batches/mirror-merged/{date}/{slug}.parquet"
    upload_file(
        path_or_fileobj=local_path,
        path_in_repo=path_in_repo,
        repo_id=repo,
        repo_type="dataset",
        commit_message=f"add {slug}",
    )
```

### 4) Lightning Studio lifecycle guard (reuse + idle-stop recovery)

`scripts/lightning_studio_guard.py`
```python
import time
from typing import Optional

from lightning import Studio


class StudioGuard:
    def __init__(self, studio_name: str, machine: str = "Lightning-L40S"):
        self.studio_name = studio_name
        self.machine = machine

    def get_running_studio(self) -> Optional[Studio]:
        # Note: Teamspace.studios requires proper Lightning SDK/runtime context.
        # If unavailable, fallback to Studio(name) lookup.
        try:
            from lightning import Teamspace

            for s in Teamspace.studios:
                if s.name == self.studio_name and s.status == "Running":
                    return s
        except Exception:
            pass

        # Fallback: try direct Studio lookup
        try:
            s = Studio(name=self.studio_name)
            if s.status == "Running":
                return s
        except Exception:
            return None
        return None

    def ensure_running(self) -> Studio:
        studio = self.get_running_studio()
        if studio:
            return studio

        return Studio(
            name=self.studio_name,
            machine=self.machine,
            create_ok=True,
            start_now=True,
        )

    def run_script(self, script_path: str, args: list[str] | None = None) -> None:
        studio = self.ensure_running()
        if studio.status != "Running":
            studio.start(machine=self.machine)
            time.sleep(30)  # allow startup

        studio.run(
            script_path,
            arguments=args or [],
            enable_pip_install=True,
        )
```
