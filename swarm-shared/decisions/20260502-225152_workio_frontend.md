# workio / frontend

## 1. Diagnosis

- Attendance events are stored ephemerally (likely `localStorage`) and submitted unreliably, causing duplicates, lost data, and poor offline UX.
- No idempotency keys or deduplication → retries create double punches.
- No durable retry/backoff state; offline events vanish on tab close or storage pressure.
- Cross-tab operations unsafe — multiple tabs can race and corrupt queue state.
- No automatic flush on reconnect or periodic retry; user must manually retry.

## 2. Proposed change

**Scope**: `src/lib/queue/` (new) + `src/hooks/useAttendanceQueue.ts` + integrate into existing clock-in/out UI components.  
**Target**: Replace fragile `localStorage` attendance buffer with an IndexedDB-backed, idempotent, cross-tab-safe queue with exponential backoff + jitter and online-triggered + periodic flush.

### 3.2 Create queue primitives

`src/lib/queue/db.ts`
```ts
import { openDB, DBSchema, IDBPDatabase } from 'idb';

interface AttendanceQueueDB extends DBSchema {
  events: {
    key: string; // idempotency key (uuid)
    value: {
      idempotencyKey: string;
      type: 'clock-in' | 'clock-out' | 'leave' | 'ot';
      payload: Record<string, unknown>;
      attempts: number;
      lastAttemptAt?: number;
      nextRetryAt?: number;
      createdAt: number;
      completedAt?: number;
      error?: string;
    };
    indexes: { 'by-nextRetryAt': number; 'by-completed': number };
  };
}

let dbPromise: Promise<IDBPDatabase<AttendanceQueueDB>> | null = null;

function getDB() {
  if (!dbPromise) {
    dbPromise = openDB<AttendanceQueueDB>('workio-attendance-queue', 1, {
      upgrade(db) {
        const store = db.createObjectStore('events', { keyPath: 'idempotencyKey' });
        store.createIndex('by-nextRetryAt', 'nextRetryAt', { unique: false });
        store.createIndex('by-completed', 'completedAt', { unique: false });
      },
    });
  }
  return dbPromise;
}

export async function addEvent(
  type: AttendanceQueueDB['events']['value']['type'],
  payload: Record<string, unknown>
) {
  const db = await getDB();
  const idempotencyKey = crypto.randomUUID();
  await db.add('events', {
    idempotencyKey,
    type,
    payload,
    attempts: 0,
    createdAt: Date.now(),
  });
  return idempotencyKey;
}

export async function peekPending(limit = 10): Promise<AttendanceQueueDB['events']['value'][]> {
  const db = await getDB();
  const tx = db.transaction('events', 'readonly');
  const store = tx.store;
  const now = Date.now();
  // items with no nextRetryAt or nextRetryAt <= now, not completed, ordered by nextRetryAt
  const index = store.index('by-nextRetryAt');
  const all = await index.getAll();
  const pending = all
    .filter((e) => !e.completedAt && (!e.nextRetryAt || e.nextRetryAt <= now))
    .sort((a, b) => (a.nextRetryAt || 0) - (b.nextRetryAt || 0))
    .slice(0, limit);
  return pending;
}

export async function markAttempt(idempotencyKey: string, error?: string) {
  const db = await getDB();
  const tx = db.transaction('events', 'readwrite');
  const store = tx.store;
  const item = await store.get(idempotencyKey);
  if (!item) return;
  item.attempts += 1;
  item.lastAttemptAt = Date.now();
  if (error) {
    item.error = error;
    // exponential backoff with jitter: 2^attempts * 1s ± 30%
    const base = Math.pow(2, item.attempts) * 1000;
    const jitter = base * 0.3 * (Math.random() * 2 - 1);
    item.nextRetryAt = Date.now() + Math.max(5000, base + jitter);
  } else {
    item.completedAt = Date.now();
    item.nextRetryAt = undefined;
  }
  await store.put(item);
  await tx.done;
}

export async function deleteCompleted() {
  const db = await getDB();
  const tx = db.transaction('events', 'readwrite');
  const store = tx.store;
  const index = store.index('by-completed');
  const completed = await index.getAll(IDBKeyRange.lowerBound(0));
  for (const c of completed) {
    await store.delete(c.idempotencyKey);
  }
  await tx.done;
}

// Cross-tab broadcast helper
export function broadcastQueueUpdate() {
  try {
    window.dispatchEvent(new CustomEvent('workio:queue-update'));
  } catch (e) {
    // ignore
  }
}
```

### 3.3 Hook to manage flush and retries

`src/hooks/useAttendanceQueue.ts`
```ts
import { useEffect, useRef } from 'react';
import { peekPending, markAttempt, deleteCompleted, broadcastQueueUpdate } from '../lib/queue/db';

const ENDPOINT = '/api/attendance/events'; // adjust to your backend route

async function flushQueue() {
  const pending = await peekPending(20);
  if (pending.length === 0) return;

  for (const item of pending) {
    try {
      const res = await fetch(ENDPOINT, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          idempotencyKey: item.idempotencyKey,
          type: item.type,
          payload: item.payload,
        }),
      });

      if (!res.ok) {
        const text = await res.text().catch(() => '');
        await markAttempt(item.idempotencyKey, `HTTP ${res.status}: ${text}`);
        continue;
      }

      await markAttempt(item.idempotencyKey);
    } catch (err: any) {
      await markAttempt(item.idempotencyKey, err?.message || String(err));
    }
  }

  // cleanup completed periodically
  await deleteCompleted();
  broadcastQueueUpdate();
}

export function useAttendanceQueue() {
  const intervalRef = useRef<NodeJS.Timeout>();

  useEffect(() => {
    // flush immediately when online
    function onOnline() {
      flushQueue();
    }

    // periodic retry every 30s while active
    intervalRef.current = setInterval(() => {
      if (navigator.onLine) flushQueue();
    }, 30_000);

    window.addEventListener('online', onOnline);
    window.addEventListener('workio:queue-update', flushQueue);

    // initial flush
    if (navigator.onLine) flushQueue();

    return () => {
      window.removeEventListener('online', onOnline);
      window.removeEventListener('workio:queue-update', flushQueue);
      if (intervalRef.current) clearInterval(intervalRef.current);
    };
  }, []);
}
```

### 3.4 Integrate into clock UI

In your clock-in/out component (example):

```tsx
import { addEvent } from '../lib/queue/db';
import { useAttendanceQueue } from '../hooks/useAttendanceQueue';

function ClockButton() {
  useAttendanceQueue(); // starts background flush

  const handleClockIn = async () => {
    const payload = {
      timestamp: Date.now(),
      location: await getLocation(), // your GPS helper
      // ...
    };
    await addEvent('clock-in', payload);
    // optimistic UI update if desired
  };

  return <button onClick={handleClockIn}>Clock In</button>;
}
```
