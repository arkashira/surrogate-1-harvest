# vanguard / design

## Final synthesized answer

### 1. Diagnosis (merged, de-duplicated)
- **Frontend/CLI still triggers authenticated `list_repo_tree` on every data-source selection or training launch**, burning HF API quota (1000/5min) and causing 429s.
- **No persisted `(repo, dateFolder) → file-list` manifest exists**, forcing re-enumeration and preventing CDN-only training workflows.
- **Training script lacks deterministic repo selection, idle-studio recovery, and schema hygiene**, risking quota waste, lost runs when Lightning idle-stop kills the session, and `pyarrow.CastError`/schema mismatches on heterogeneous repos.
- **No reuse guard for Lightning Studio**; each launch may create new studios and waste 80hr/mo quota.
- **Mac/CLI orchestration may still attempt local `from_pretrained()` or heavy compute** instead of delegating to Lightning/Kaggle/Cerebras.

### 2. Core principles for correctness + actionability
- **Single source of truth**: one authenticated tree-listing per `(repo, dateFolder, revision)`; cache result locally and reuse everywhere.
- **CDN-only data loading in training**: zero authenticated HF API calls during training; use `https://huggingface.co/datasets/.../resolve/main/...`.
- **Schema-safe, per-file loading**: avoid `load_dataset(streaming=True)` on heterogeneous repos; download per file and project to `{prompt, response}`.
- **Deterministic repo/revision selection** and **Lightning Studio reuse with idle recovery** to protect quota and prevent lost runs.
- **Lightweight, portable verification** (CLI + optional pytest) so Mac orchestrator and CI can validate behavior without heavy compute.

```python
# /opt/axentx/vanguard/manifest.py
from __future__ import annotations

import json
import os
import time
import hashlib
from pathlib import Path
from typing import Dict, List, Optional, Tuple

from huggingface_hub import HfApi, hf_hub_download, hf_hub_url

CACHE_DIR = Path(__file__).parent / "manifests"
CACHE_DIR.mkdir(exist_ok=True, parents=True)

_api = HfApi()


def _cache_path(repo: str, date_folder: str, revision: str) -> Path:
    slug = repo.replace("/", "--")
    rev_hash = hashlib.sha256(revision.encode()).hexdigest()[:12]
    return CACHE_DIR / f"{slug}__{date_folder.replace('/', '_')}__{rev_hash}.json"


def list_and_cache(
    repo: str,
    date_folder: str,
    revision: str = "main",
    *,
    bust_cache: bool = False,
) -> str:
    """
    Single authenticated list_repo_tree call per (repo, date_folder, revision).
    Returns local path to cached JSON manifest.

    Notes:
    - Uses path=date_folder and recursive=False to avoid enumerating huge repos.
    - If date_folder lists directories (common), this returns directory entries;
      callers may need to recurse or request deeper paths as required.
    """
    cp = _cache_path(repo, date_folder, revision)
    if cp.exists() and not bust_cache:
        return str(cp)

    # list only the requested folder (non-recursive)
    tree = _api.list_repo_tree(repo=repo, revision=revision, path=date_folder, recursive=False)
    entries = [{"path": f.path, "type": f.type} for f in tree]

    manifest = {
        "repo": repo,
        "date_folder": date_folder,
        "revision": revision,
        "cached_at": time.time(),
        "entries": entries,
    }
    cp.write_text(json.dumps(manifest, indent=2))
    return str(cp)


def cdn_urls(manifest_path: str, *, files_only: bool = True) -> List[str]:
    manifest = json.loads(Path(manifest_path).read_text())
    base = f"https://huggingface.co/datasets/{manifest['repo']}/resolve/main"
    urls = []
    for e in manifest["entries"]:
        if files_only and e["type"] != "file":
            continue
        urls.append(f"{base}/{e['path']}")
    return urls


def file_urls_by_subpath(manifest_path: str) -> Dict[str, str]:
    """Return {subpath: cdn_url} for files only."""
    manifest = json.loads(Path(manifest_path).read_text())
    base = f"https://huggingface.co/datasets/{manifest['repo']}/resolve/main"
    out = {}
    for e in manifest["entries"]:
        if e["type"] == "file":
            out[e["path"]] = f"{base}/{e['path']}"
    return out


def safe_project_to_prompt_response(
    repo: str,
    file_path: str,
    revision: str = "main",
    *,
    required_prompt: bool = True,
) -> List[Dict[str, str]]:
    """
    Download one file and project to [{prompt, response}] records.
    Avoids load_dataset(streaming=True) on heterogeneous schemas.
    """
    local_path = hf_hub_download(repo_id=repo, filename=file_path, revision=revision)

    # Lightweight readers; extend per dataset if needed
    if local_path.endswith(".jsonl"):
        import pandas as pd
        df = pd.read_json(local_path, lines=True)
    elif local_path.endswith(".json"):
        import pandas as pd
        df = pd.read_json(local_path)
    elif local_path.endswith(".parquet"):
        import pandas as pd
        df = pd.read_parquet(local_path)
    elif local_path.endswith(".csv"):
        import pandas as pd
        df = pd.read_csv(local_path)
    else:
        raise ValueError(f"Unsupported file type: {local_path}")

    # Normalize column names
    col_map = {c: c.lower().strip().replace(" ", "_") for c in df.columns}
    df = df.rename(columns=col_map)

    prompt_col = next((c for c in df.columns if "prompt" in c), None)
    response_col = next((c for c in df.columns if any(k in c for k in ("response", "completion", "answer"))), None)

    if required_prompt and (not prompt_col or not response_col):
        raise KeyError(
            f"Missing prompt/response in {file_path}. "
            f"Available columns: {list(df.columns)}. "
            f"prompt_col={prompt_col}, response_col={response_col}"
        )

    records = []
    for _, row in df.iterrows():
        rec = {"prompt": row[prompt_col] if prompt_col else "", "response": row[response_col] if response_col else ""}
        records.append(rec)
    return records
```

# /opt/axentx/vanguard/train.py
import os
import sys
from pathlib import Path

# Add project root to path if needed
sys.path.insert(0, str(Path(__file__).parent))

from manifest import list_and_cache
from lightning import Studio, Teamspace, Machine  # type: ignore

HF_REPO = os.getenv("HF_REPO", "myorg/surrogate-1-mirror")
DATE_FOLDER = os.getenv("DATE_FOLDER", "batches/mirror-merged/2026-04-29")
REVISION = os.getenv("REVISION", "main")
STUDIO_SUFFIX = os.getenv("STUDIO_SUFFIX", "train")

# 1) Manifest: single API call from orchestrator (Mac/CLI), reused in training
manifest_path = list_and_cache(HF_REPO, DATE_FOLDER, revision=REVISION)
print(f"Manifest: {manifest_path}")

# 2) Studio reuse + idle-recovery guard
team = Teamspace()
studio_name = f"vanguard-{STUDIO_SUFFIX}"
running = None
for s in team.studios:
    if s.name == studio_name and s.status == "running":
        running = s
        break

if running:
    studio = running
    print(f"Reusing running studio: {studio_name}")
else:
    # Prefer free-tier-compatible machine; allow override for prod
    preferred = os.getenv("STUDIO_MACHINE", "L40S")
    studio = Studio(
