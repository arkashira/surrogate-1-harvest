## Implementation Plan (≤2h)

**Highest-value improvement**: Add deterministic pre-flight file listing + CDN-only ingestion to eliminate HF API 429s during training and make shard workers resilient.

### What we’ll change
1. Add `bin/list-files.sh` — single Mac/Lightning call to `list_repo_tree` for a date folder, emit `file-list.json`.
2. Update `bin/dataset-enrich.sh` to accept an optional `FILE_LIST` path; if provided, iterate CDN URLs only (no `load_dataset`/`datasets` API calls).
3. Add `lib/cdn_stream.py` — lightweight CDN fetcher with retries, zero HF auth, projects to `{prompt,response}` on the fly.
4. Update GitHub Actions matrix to generate `file-list.json` once (in a setup job) and pass it to all 16 shards so every shard uses the same deterministic snapshot.
5. Keep existing `lib/dedup.py` unchanged (central md5 store still authoritative).

### Why this wins
- Eliminates HF API pagination + 429 risk during ingestion (CDN tier has much higher limits).
- Training script can reuse the same `file-list.json` for CDN-only data loading (zero API calls during training).
- No schema surprises: each file downloaded individually and projected to `{prompt,response}` at parse time (avoids pyarrow CastError on mixed schemas).
- Fits <2h: small focused scripts + workflow tweak.

#### 1) `bin/list-files.sh`
```bash
#!/usr/bin/env bash
# list-files.sh
# Usage: HF_TOKEN=... ./list-files.sh <date> [output.json]
#
# Calls HF API once (list_repo_tree) for the given date folder and emits
# a deterministic JSON list of file paths under that folder.
#
# Requires: jq, curl, HF_TOKEN

set -euo pipefail
REPO="datasets/axentx/surrogate-1-training-pairs"
HF_TOKEN="${HF_TOKEN:-}"
DATE="${1:-$(date +%Y-%m-%d)}"
OUT="${2:-file-list.json}"

if [ -z "$HF_TOKEN" ]; then
  echo "ERROR: HF_TOKEN is required" >&2
  exit 1
fi

# Single API call: non-recursive tree for the date folder
# HF API: GET /api/datasets/{repo}/tree/{revision}
# We use main branch and target path = batches/public-merged/<DATE>
URL="https://huggingface.co/api/datasets/${REPO}/tree/main/batches/public-merged/${DATE}"

echo "Fetching tree for ${DATE}..."
curl -sSL -H "Authorization: Bearer ${HF_TOKEN}" "${URL}" \
  | jq -c '[ .[] | select(.type=="file") | .path ]' > "${OUT}.tmp"

# Deterministic sort
jq -c 'sort' "${OUT}.tmp" > "${OUT}"
rm -f "${OUT}.tmp"

echo "Wrote $(jq length "${OUT}") files to ${OUT}"
```

#### 2) `lib/cdn_stream.py`
```python
#!/usr/bin/env python3
"""
cdn_stream.py

CDN-only fetcher for surrogate-1 training pairs.
- No Authorization header (bypasses HF API rate limits).
- Streams one file at a time and yields {prompt, response} dicts.
- Lightweight retry/backoff.
"""
import json
import sys
import time
from pathlib import Path
from typing import Iterator, Dict, Any

import requests
import pyarrow.parquet as pq
import pyarrow as pa
import numpy as np
from io import BytesIO

CDN_ROOT = "https://huggingface.co/datasets/axentx/surrogate-1-training-pairs/resolve/main"

HEADERS = {
    # No Authorization header -> CDN tier, not API tier
    "User-Agent": "axentx-surrogate-1-cdn/1.0"
}

MAX_RETRIES = 5
BACKOFF = 2.0


def cdn_url(path: str) -> str:
    return f"{CDN_ROOT}/{path}"


def fetch_with_retry(url: str) -> bytes:
    for attempt in range(1, MAX_RETRIES + 1):
        try:
            resp = requests.get(url, headers=HEADERS, timeout=30)
            if resp.status_code == 404:
                # Missing file is not transient
                raise FileNotFoundError(f"CDN 404: {url}")
            resp.raise_for_status()
            return resp.content
        except Exception as exc:
            if attempt == MAX_RETRIES:
                raise
            sleep = BACKOFF ** attempt
            print(f"[cdn] retry {attempt}/{MAX_RETRIES} for {url}: {exc}, sleeping {sleep}s", file=sys.stderr)
            time.sleep(sleep)
    raise RuntimeError("unreachable")


def project_to_pair(obj: Dict[str, Any]) -> Dict[str, str]:
    """
    Project raw object to {prompt, response}.
    Tolerates common field names and nulls.
    """
    prompt = obj.get("prompt") or obj.get("input") or obj.get("question") or ""
    response = obj.get("response") or obj.get("output") or obj.get("answer") or ""
    # Ensure strings
    return {"prompt": str(prompt) if prompt is not None else "",
            "response": str(response) if response is not None else ""}


def stream_parquet_cdn(path: str) -> Iterator[Dict[str, str]]:
    url = cdn_url(path)
    data = fetch_with_retry(url)
    try:
        table = pq.read_table(BytesIO(data))
    except Exception as exc:
        # If parquet fails, skip file (schema mismatch)
        print(f"[cdn] failed to read parquet {path}: {exc}", file=sys.stderr)
        return

    # Convert to dicts and project
    cols = table.to_pydict()
    n = len(next(iter(cols.values()), []))
    for i in range(n):
        row = {k: cols[k][i] for k in cols}
        yield project_to_pair(row)


def stream_jsonl_cdn(path: str) -> Iterator[Dict[str, str]]:
    url = cdn_url(path)
    data = fetch_with_retry(url)
    for line in data.splitlines():
        line = line.strip()
        if not line:
            continue
        try:
            obj = json.loads(line)
        except Exception:
            continue
        yield project_to_pair(obj)


def stream_file_cdn(path: str) -> Iterator[Dict[str, str]]:
    if path.endswith(".parquet"):
        yield from stream_parquet_cdn(path)
    elif path.endswith(".jsonl"):
        yield from stream_jsonl_cdn(path)
    else:
        # fallback: try jsonl, then parquet
        try:
            yield from stream_jsonl_cdn(path)
        except Exception:
            yield from stream_parquet_cdn(path)


def stream_from_file_list(file_list_path: str) -> Iterator[Dict[str, str]]:
    with open(file_list_path, "r") as f:
        paths = json.load(f)
    for p in paths:
        count = 0
        try:
            for pair in stream_file_cdn(p):
                if pair["prompt"] or pair["response"]:
                    yield pair
                    count += 1
        except Exception as exc:
            print(f"[cdn] failed file {p}: {exc}", file=sys.stderr)
        if count:
            print(f"[cdn] {p} -> {count} pairs", file=sys.stderr)


if __name__ == "__main__":
    if len(sys.argv) != 2:
        print("Usage: cdn_stream.py <file-list.json>")
        sys.exit(1)
    for item in stream_from_file_list(sys.argv[1]):
        print(json.dumps(item, ensure_ascii=False))
```
