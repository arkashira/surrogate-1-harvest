# workio / frontend

## Final Synthesized Implementation

I’ve merged the strongest, most compatible parts of both proposals and resolved contradictions in favor of **correctness, robustness, and concrete actionability**.

Key decisions:
- Use **stable idempotency keys** (not timestamp-based) per user+date+punchType to prevent duplicates across retries while allowing legitimate same-day in/out sequences.
- Keep **optimistic UI** for perceived performance, but **revert on failure** and provide actionable retry.
- Add **disabled/loading state** on buttons to prevent double-taps and races.
- Add **local last-punch cache** (localStorage) for instant display on open, with server as source of truth.
- Keep implementation **contained and minimal** (~100 lines) and easy to wire into existing pages.

// Client-side helpers for punch idempotency and last-punch cache

const IDEMPOTENCY_PREFIX = "workio:punch:inflight:";
const LAST_PUNCH_KEY = "workio:punch:last";

// One stable inflight key per user+date+punchType to absorb retries/dupes
// while allowing same-day clock-in/clock-out pairs.
export function inflightKey(userId: string, punchType: "CLOCK_IN" | "CLOCK_OUT"): string {
  const today = new Date().toISOString().split("T")[0];
  return `${IDEMPOTENCY_PREFIX}${userId}:${today}:${punchType}`;
}

export function markInflight(key: string): void {
  try {
    sessionStorage.setItem(key, "inflight");
  } catch {
    // ignore (incognito)
  }
}

export function isInflight(key: string): boolean {
  try {
    return sessionStorage.getItem(key) === "inflight";
  } catch {
    return false;
  }
}

export function clearInflight(key: string): void {
  try {
    sessionStorage.removeItem(key);
  } catch {
    // ignore
  }
}

// Last punch cache (for instant display on open)
export interface CachedPunch {
  type: "CLOCK_IN" | "CLOCK_OUT";
  time: string; // ISO
}

export function setLastPunchCache(punch: CachedPunch | null): void {
  try {
    if (punch) {
      localStorage.setItem(LAST_PUNCH_KEY, JSON.stringify(punch));
    } else {
      localStorage.removeItem(LAST_PUNCH_KEY);
    }
  } catch {
    // ignore
  }
}

export function getLastPunchCache(): CachedPunch | null {
  try {
    const raw = localStorage.getItem(LAST_PUNCH_KEY);
    return raw ? JSON.parse(raw) : null;
  } catch {
    return null;
  }
}
