### 1) Manifest generator (run from Mac orchestration host)

File: `/opt/axentx/airship/surrogate/scripts/generate_cdn_manifest.py`

```python
#!/usr/bin/env python3
"""
Generate a deterministic CDN manifest for one date folder in a HF dataset repo.
Usage:
    HF_REPO=datasets/your/repo HF_DATE_FOLDER=mirror-merged/2026-05-03 \
    python generate_cdn_manifest.py --out manifest.json

Output:
    {
      "repo": "datasets/your/repo",
      "date_folder": "mirror-merged/2026-05-03",
      "files": [
        "mirror-merged/2026-05-03/slug1.parquet",
        "mirror-merged/2026-05-03/slug2.parquet",
        ...
      ],
      "generated_at": "2026-05-03T04:27:00Z"
    }
"""

import argparse
import json
import os
import sys
from datetime import datetime, timezone

from huggingface_hub import HfApi

def build_parser() -> argparse.ArgumentParser:
    p = argparse.ArgumentParser(description="Generate CDN manifest for one HF dataset date folder.")
    p.add_argument("--repo", default=os.getenv("HF_REPO"), help="HF dataset repo (e.g. datasets/your/repo)")
    p.add_argument(
        "--date-folder",
        default=os.getenv("HF_DATE_FOLDER"),
        help="Subfolder inside repo (e.g. mirror-merged/2026-05-03)",
    )
    p.add_argument("--out", default="manifest.json", help="Output JSON path")
    return p

def main() -> None:
    args = build_parser().parse_args()
    if not args.repo or not args.date_folder:
        print("ERROR: provide --repo and --date-folder (or HF_REPO/HF_DATE_FOLDER env).", file=sys.stderr)
        sys.exit(1)

    api = HfApi()
    # Non-recursive per folder to avoid pagination explosion and API 429.
    entries = api.list_repo_tree(repo_id=args.repo, path=args.date_folder, recursive=False)

    files = sorted(
        e.rfilename
        for e in entries
        if e.rfilename.startswith(args.date_folder) and e.rfilename.endswith(".parquet")
    )

    manifest = {
        "repo": args.repo,
        "date_folder": args.date_folder.rstrip("/"),
        "files": files,
        "generated_at": datetime.now(timezone.utc).isoformat(),
    }

    with open(args.out, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2, sort_keys=True)

    print(f"Wrote {len(files)} files to {args.out}")

if __name__ == "__main__":
    main()
```

### 2) Update training data loader to use CDN-only fetches with column projection

File: `/opt/axentx/airship/surrogate/train.py` (add/modify)

```python
import json
import os
from pathlib import Path
from typing import List

import pyarrow.parquet as pq
import requests
from torch.utils.data import IterableDataset

CDN_BASE = "https://huggingface.co/datasets"

class CDNParquetIterable(IterableDataset):
    """
    Zero HF API calls during training.
    Reads parquet files directly from HF CDN using a pre-generated manifest.
    Projects each row to {prompt, response} at parse time.
    """

    def __init__(self, manifest_path: str, start_index: int = 0):
        manifest_path = Path(manifest_path)
        if not manifest_path.is_file():
            raise FileNotFoundError(f"Manifest not found: {manifest_path}")

        with manifest_path.open("r", encoding="utf-8") as f:
            manifest = json.load(f)

        self.repo = manifest["repo"]
        self.files: List[str] = manifest["files"]
        if not self.files:
            raise ValueError("No parquet files found in manifest.")

        # deterministic shard ordering
        self.files.sort()
        self.start_index = start_index % len(self.files)

    def _cdn_url(self, filepath: str) -> str:
        # CDN fetch; no Authorization header required for public datasets.
        return f"{CDN_BASE}/{self.repo}/resolve/main/{filepath}"

    def _stream_file(self, filepath: str):
        url = self._cdn_url(filepath)
        # stream to avoid large memory spikes; project only {prompt,response}
        with requests.get(url, stream=True, timeout=60) as r:
            r.raise_for_status()
            # Use raw stream with pyarrow for memory-efficient projection
            with pq.ParquetFile(pq.ParquetFile(r.raw).metadata) as pf:
                cols = set(pf.schema.names)
                wanted = [c for c in ("prompt", "response") if c in cols]
                if not wanted:
                    # fallback: read all and project
                    batch = pq.read_table(r.raw).to_pylist()
                else:
                    batch = pq.read_table(r.raw, columns=wanted).to_pylist()

                for row in batch:
                    # normalize to {prompt, response}
                    prompt = row.get("prompt") or row.get("instruction") or ""
                    response = row.get("response") or row.get("output") or ""
                    if prompt and response:
                        yield {"prompt": prompt, "response": response}

    def __iter__(self):
        n = len(self.files)
        idx = self.start_index
        while True:
            filepath = self.files[idx % n]
            yield from self._stream_file(filepath)
            idx += 1
```

Usage in training loop (example snippet):
```python
manifest_path = os.getenv("CDN_MANIFEST", "manifest.json")
train_dataset = CDNParquetIterable(manifest_path=manifest_path)
```

"""

            name=STUDIO_NAME,
            machine=Machine(MACHINE_TYPE),
            create_ok=True,
        )
    else:
        studio = running
