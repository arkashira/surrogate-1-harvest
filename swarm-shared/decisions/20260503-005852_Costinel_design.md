# Costinel / design

## Implementation Plan — Costinel “Top-Hub Signal” Card (≤2h, frontend-only)

### Highest-value incremental improvement
Add a **Top-Hub Signal card** to the dashboard that surfaces the most-connected knowledge-rag hub (e.g., “MOC”) with actionable signals and graceful fallback when graph data is unavailable. This delivers immediate governance value (Sense + Signal) without backend changes.

### Concrete steps (≤2h)
1. **Locate dashboard layout** — find main dashboard component (likely `pages/Dashboard.tsx` or `components/Dashboard/`).
2. **Create reusable card** — `components/Costinel/TopHubSignalCard.tsx`.
3. **Static top-hub data** — embed curated top-hub + signals (from knowledge-rag patterns) as fallback JSON; optionally attempt lightweight fetch of `/data/knowledge-rag/top-hubs.json` if present.
4. **Render signals** — show hub name, short insight, recommended actions, and audit trail note.
5. **Graceful fallback** — if no graph data, show “MOC” (from pattern) with default signals and subtle “offline” badge.
6. **Styling** — reuse existing design tokens and dashboard card styles.
7. **Integrate** — add card to dashboard grid.

```tsx
import React from 'react';
import { Icon } from '@/components/Icon';
import { Badge } from '@/components/Badge';
import { Card } from '@/components/Card';

type Signal = {
  id: string;
  title: string;
  description: string;
  action: string;
  impact: 'high' | 'medium' | 'low';
};

type TopHub = {
  slug: string;
  name: string;
  summary: string;
  signals: Signal[];
  updatedAt?: string;
};

const DEFAULT_TOP_HUB: TopHub = {
  slug: 'MOC',
  name: 'MOC (Minutes of Conversation)',
  summary:
    'Most-connected hub in knowledge-rag graph. Central context for cost governance decisions and cross-team alignment.',
  signals: [
    {
      id: 'S1',
      title: 'Standardize cost-review notes',
      description: 'Adopt MOC template for all cost-review meetings to improve auditability.',
      action: 'Apply MOC template to next 3 reviews',
      impact: 'high',
    },
    {
      id: 'S2',
      title: 'Link anomalies to MOC entries',
      description: 'Ensure every detected cost anomaly references a MOC entry for traceability.',
      action: 'Backfill last 10 anomalies with MOC links',
      impact: 'medium',
    },
    {
      id: 'S3',
      title: 'Tag proposals with hub context',
      description: 'Include MOC slug in proposal metadata to enable graph-based impact analysis.',
      action: 'Update proposal form to include hub tag',
      impact: 'medium',
    },
  ],
  updatedAt: '2026-04-27',
};

const impactColor = (impact: Signal['impact']) => {
  switch (impact) {
    case 'high':
      return 'text-red-600 bg-red-50';
    case 'medium':
      return 'text-amber-600 bg-amber-50';
    case 'low':
      return 'text-green-600 bg-green-50';
  }
};

export const TopHubSignalCard: React.FC<{ hub?: TopHub; isLive?: boolean }> = ({
  hub = DEFAULT_TOP_HUB,
  isLive = false,
}) => {
  return (
    <Card className="flex flex-col h-full">
      <div className="flex items-start justify-between mb-3">
        <div className="flex items-center gap-2">
          <Icon name="graph" className="w-5 h-5 text-blue-600" />
          <h3 className="font-semibold text-gray-900">{hub.name}</h3>
          {!isLive && <Badge variant="neutral">offline</Badge>}
        </div>
        {hub.updatedAt && (
          <span className="text-xs text-gray-500">Updated {hub.updatedAt}</span>
        )}
      </div>

      <p className="text-sm text-gray-600 mb-4">{hub.summary}</p>

      <div className="flex-1 space-y-3">
        {hub.signals.map((s) => (
          <div key={s.id} className="p-2 rounded border border-gray-100 bg-gray-50/50">
            <div className="flex items-start justify-between gap-2">
              <div className="min-w-0 flex-1">
                <div className="flex items-center gap-2">
                  <span className="font-medium text-sm text-gray-900">{s.title}</span>
                  <span className={`text-xs px-1.5 py-0.5 rounded ${impactColor(s.impact)}`}>
                    {s.impact}
                  </span>
                </div>
                <p className="text-xs text-gray-600 mt-0.5">{s.description}</p>
              </div>
            </div>
            <p className="text-xs text-blue-700 mt-2">Action: {s.action}</p>
            <p className="text-xs text-gray-400 mt-1">
              Note: Sense + Signal — no execution. Log in change management for approval.
            </p>
          </div>
        ))}
      </div>

      <div className="mt-4 pt-3 border-t border-gray-100 text-xs text-gray-500">
        Audit trail: all signals map to knowledge-rag graph nodes. Use hub slug ({hub.slug}) when
        creating proposals.
      </div>
    </Card>
  );
};
```

#### 2) Attempt lightweight fetch (optional) — dashboard component
```tsx
import { useEffect, useState } from 'react';
import { TopHubSignalCard } from '@/components/Costinel/TopHubSignalCard';

export default function Dashboard() {
  const [topHub, setTopHub] = useState<any>(null);
  const [isLive, setIsLive] = useState(false);

  useEffect(() => {
    // Lightweight attempt to fetch precomputed top-hub JSON from public CDN path.
    // If unavailable, fallback to DEFAULT_TOP_HUB (MOC) — graceful degradation.
    fetch('/data/knowledge-rag/top-hubs.json')
      .then((r) => {
        if (!r.ok) throw new Error('Not found');
        return r.json();
      })
      .then((data) => {
        setTopHub(data);
        setIsLive(true);
      })
      .catch(() => {
        // silent fallback to default
        setTopHub(null);
        setIsLive(false);
      });
  }, []);

  return (
    <div className="grid grid-cols-1 lg:grid-cols-3 gap-6">
      {/* Other dashboard cards ... */}
      <div className="lg:col-span-1">
        <TopHubSignalCard hub={topHub} isLive={isLive} />
      </div>
    </div>
  );
}
```
