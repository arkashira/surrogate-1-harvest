# workio / backend

## 1. Diagnosis

- **Tenant cache missing on `/webhook/line`** — every LINE event resolves `channelId → tenant` via DB; under retries/spikes this causes DB thrash and increases 429/timeout risk.
- **No request-scoped correlation IDs** — webhook → service → DB spans lack trace IDs; incidents are hard to reconstruct and slow to resolve.
- **No idempotency for clock-in/out webhook events** — LINE retries on 5xx can create duplicate attendance records for the same user within short windows.
- **Missing circuit-breaker/retry budget for LINE API calls** — outbound calls to LINE (e.g., reply/notify) have no backoff or failure isolation; cascading failures can stall webhook processing.
- **No structured logging with context** — logs are text-only and don’t include tenantId/channelId/requestId; debugging multi-tenant incidents is slow.

## 2. Proposed change

File: `workio/server/src/routes/webhook/line.ts` (or equivalent route handler)  
Scope: add lightweight in-memory tenant cache + request-scoped correlation ID middleware + idempotency guard for duplicate events + structured logging. Target: <2h, minimal deps, no schema changes.

```ts
// /opt/axentx/workio/server/src/routes/webhook/line.ts
import express, { Request, Response, NextFunction } from 'express';
import crypto from 'crypto';
import pino from 'pino';
import { LRUCache } from 'lru-cache';

const logger = pino({
  transport: {
    target: 'pino-pretty',
    options: { colorize: true }
  }
});

// Lightweight tenant cache (TTL 5m, max 10k entries)
const tenantCache = new LRUCache<string, { tenantId: string; updatedAt: number }>({
  max: 10_000,
  ttl: 1000 * 60 * 5
});

// Simple in-memory idempotency store for webhook events (TTL 10m)
const idempotencyCache = new LRUCache<string, boolean>({
  max: 50_000,
  ttl: 1000 * 60 * 10
});

const router = express.Router();

// Middleware: correlation ID + structured logging
router.use((req: Request, _res: Response, next: NextFunction) => {
  const requestId = req.headers['x-request-id'] as string || crypto.randomUUID();
  (req as any).requestId = requestId;
  (req as any).log = logger.child({ requestId, route: '/webhook/line' });
  next();
});

// Helpers
async function resolveTenantId(channelId: string, log: any): Promise<string | null> {
  const cached = tenantCache.get(channelId);
  if (cached) {
    log.debug({ channelId }, 'tenant cache hit');
    return cached.tenantId;
  }

  // Replace with your actual DB client import/query
  const db = (global as any).db || require('../../db').default;
  const row = await db.oneOrNone('SELECT id FROM tenants WHERE line_channel_id = $1', [channelId]);
  if (row) {
    tenantCache.set(channelId, { tenantId: row.id, updatedAt: Date.now() });
    log.debug({ channelId, tenantId: row.id }, 'tenant cache miss -> stored');
    return row.id;
  }

  log.warn({ channelId }, 'tenant not found');
  return null;
}

function isDuplicate(eventKey: string): boolean {
  const exists = idempotencyCache.has(eventKey);
  if (!exists) idempotencyCache.set(eventKey, true);
  return exists;
}

// LINE webhook handler
router.post('/', async (req: Request, res: Response) => {
  const log = (req as any).log;
  const body = req.body;

  try {
    // Basic validation
    if (!body?.events || !Array.isArray(body.events)) {
      log.warn({ body }, 'invalid webhook payload');
      return res.status(400).json({ error: 'invalid payload' });
    }

    for (const event of body.events) {
      const channelId = event.source?.userId ? null : (event.source?.groupId || event.source?.roomId || '');
      // LINE webhook top-level destination is more reliable for channel mapping
      const destination = body.destination; // channelId (bot's channel)
      const eventType = event.type;
      const userId = event.source?.userId || '';
      const timestamp = event.timestamp || Date.now();

      // Idempotency key: destination + event type + user + timestamp rounded to 1s + msgId if present
      const msgId = event.message?.id || '';
      const idempotencyKey = crypto
        .createHash('sha256')
        .update(`${destination}:${eventType}:${userId}:${Math.floor(timestamp / 1000)}:${msgId}`)
        .digest('hex');

      if (isDuplicate(idempotencyKey)) {
        log.info({ idempotencyKey }, 'duplicate event skipped');
        continue;
      }

      const tenantId = await resolveTenantId(destination, log);
      if (!tenantId) {
        log.warn({ destination }, 'no tenant for destination - skipping event');
        continue;
      }

      // Attach context for downstream handlers
      (req as any).tenantId = tenantId;

      // Handle clock-in/out (example)
      if (eventType === 'message' && event.message?.type === 'text') {
        const text = event.message.text.trim().toLowerCase();
        if (text === 'clock in' || text === 'clock out') {
          const db = (global as any).db || require('../../db').default;
          const action = text === 'clock in' ? 'in' : 'out';

          // Prevent duplicate attendance within 60s for same user/tenant (extra safety)
          const recent = await db.oneOrNone(
            `SELECT id FROM attendances
             WHERE tenant_id = $1 AND user_id = $2 AND action = $3
             AND created_at > NOW() - INTERVAL '60 seconds'
             ORDER BY created_at DESC LIMIT 1`,
            [tenantId, userId, action]
          );

          if (!recent) {
            await db.none(
              `INSERT INTO attendances (tenant_id, user_id, action, created_at)
               VALUES ($1, $2, $3, NOW())`,
              [tenantId, userId, action]
            );
            log.info({ tenantId, userId, action }, 'attendance recorded');
          } else {
            log.info({ tenantId, userId, action }, 'duplicate within 60s skipped');
          }
        }
      }
    }

    // Reply 200 quickly; LINE retries on non-2xx
    res.status(200).json({ ok: true });
  } catch (err) {
    log.error({ err }, 'webhook processing error');
    res.status(500).json({ error: 'internal' });
  }
});

export default router;
```
