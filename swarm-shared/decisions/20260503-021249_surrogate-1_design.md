# surrogate-1 / design

## Final Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID` and `SHARD_TOTAL` (16) from GitHub Actions matrix.
- Uses a pre-generated `file-list.json` (Mac-side, one `list_repo_tree` call per date folder) to avoid recursive HF API pagination and 429s.
- Deterministically assigns files via `hash(slug) % SHARD_TOTAL` (not positional modulo) to tolerate manifest ordering changes and keep shard assignments stable across reruns.
- Downloads only assigned shard files via HF CDN (`resolve/main/...`) — zero Authorization header, bypasses `/api/` rate limits.
- Projects heterogeneous files to `{prompt, response}` at parse time (avoids pyarrow CastError from mixed schemas).
- Deduplicates via central `lib/dedup.py` md5 store.
- Writes deterministic output: `batches/public-merged/{date}/shard{N}-{HHMMSS}.jsonl`.
- Reuses existing Lightning Studio pattern for any local dev (not needed here — workers run in GitHub Actions).
- Adds proper Bash shebang and executable bit for any wrapper scripts; sets `SHELL=/bin/bash` in cron if used.

### 1) New worker: `bin/dataset-enrich.py`

```python
#!/usr/bin/env python3
"""
CDN-bypass ingestion worker for surrogate-1 public dataset.
Usage:
  SHARD_ID=0 SHARD_TOTAL=16 python bin/dataset-enrich.py \
    --repo axentx/surrogate-1-training-pairs \
    --date 2026-05-03 \
    --file-list file-list.json \
    --out-dir batches/public-merged
"""

import argparse
import hashlib
import json
import os
import sys
import time
from datetime import datetime
from pathlib import Path
from typing import Dict, Any

import requests
from huggingface_hub import hf_hub_download, HfApi

# Local dedup store
sys.path.insert(0, str(Path(__file__).parent))
from lib.dedup import DedupStore  # noqa

HF_API = HfApi()
CDN_DOWNLOAD_BASE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

# Minimal schema projectors per known source
def project_to_pair(raw: Dict[str, Any], filename: str) -> Dict[str, str] | None:
    """
    Return {"prompt": ..., "response": ...} or None if unparseable.
    Keep this small and defensive to avoid schema CastErrors.
    """
    # Heuristic: look for common key names
    prompt_keys = {"prompt", "instruction", "input", "question", "user"}
    response_keys = {"response", "output", "answer", "assistant", "completion"}

    # Try direct keys first
    for pk in prompt_keys:
        if pk in raw:
            for rk in response_keys:
                if rk in raw:
                    return {"prompt": str(raw[pk]), "response": str(raw[rk])}

    # Fallback: if JSON has exactly two string fields, treat as pair
    str_fields = {k: v for k, v in raw.items() if isinstance(v, str)}
    if len(str_fields) == 2:
        vals = list(str_fields.values())
        return {"prompt": vals[0], "response": vals[1]}

    # Unparseable for our training pair
    return None


def download_via_cdn(repo: str, path: str, retries: int = 3, backoff: int = 5) -> bytes:
    url = CDN_DOWNLOAD_BASE.format(repo=repo, path=path)
    for attempt in range(1, retries + 1):
        try:
            resp = requests.get(url, timeout=30)
            resp.raise_for_status()
            return resp.content
        except Exception as exc:
            if attempt == retries:
                raise
            time.sleep(backoff * attempt)
    raise RuntimeError("unreachable")


def load_jsonl_lines(content: bytes):
    for line in content.decode("utf-8").splitlines():
        line = line.strip()
        if not line:
            continue
        try:
            yield json.loads(line)
        except Exception:
            continue


def shard_for(path: str, shard_total: int) -> int:
    """Deterministic shard assignment stable across manifest ordering changes."""
    slug = Path(path).name
    return hash(slug) % shard_total


def main():
    parser = argparse.ArgumentParser(description="CDN-bypass shard worker")
    parser.add_argument("--repo", default="axentx/surrogate-1-training-pairs")
    parser.add_argument("--date", required=True, help="Date folder, e.g. 2026-05-03")
    parser.add_argument("--file-list", required=True, help="Path to file-list.json")
    parser.add_argument("--out-dir", default="batches/public-merged")
    args = parser.parse_args()

    shard_id = int(os.getenv("SHARD_ID", "0"))
    shard_total = int(os.getenv("SHARD_TOTAL", "16"))
    hf_token = os.getenv("HF_TOKEN")
    if not hf_token:
        print("ERROR: HF_TOKEN required for dedup store push", file=sys.stderr)
        sys.exit(1)

    # Load manifest produced by Mac-side list_repo_tree(date, recursive=False)
    with open(args.file_list) as f:
        manifest = json.load(f)  # expect list of paths under date folder

    # Filter to date folder and shard assignment
    date_prefix = f"{args.date}/"
    files = [p for p in manifest if p.startswith(date_prefix)]
    files.sort()
    if not files:
        print(f"No files for date {args.date} in manifest", file=sys.stderr)
        sys.exit(0)

    # Deterministic shard assignment by content hash (stable across reruns)
    assigned = [f for f in files if shard_for(f, shard_total) == shard_id]
    print(f"Shard {shard_id}/{shard_total} assigned {len(assigned)} files")

    dedup = DedupStore()
    out_dir = Path(args.out_dir) / args.date
    out_dir.mkdir(parents=True, exist_ok=True)
    ts = datetime.utcnow().strftime("%H%M%S")
    out_path = out_dir / f"shard{shard_id}-{ts}.jsonl"

    written = 0
    skipped_dup = 0
    skipped_unparse = 0

    with out_path.open("w", buffering=1) as out_f:
        for rel_path in assigned:
            try:
                # CDN download (no auth header)
                raw_bytes = download_via_cdn(args.repo, rel_path)
            except Exception as exc:
                print(f"WARN: failed to download {rel_path}: {exc}", file=sys.stderr)
                continue

            for raw_obj in load_jsonl_lines(raw_bytes):
                pair = project_to_pair(raw_obj, rel_path)
                if not pair:
                    skipped_unparse += 1
                    continue

                # Deterministic content hash for dedup
                payload = json.dumps(pair, sort_keys=True, separators=(",", ":"))
                md5 = hashlib.md5(payload.encode("utf-8")).hexdigest()
                if dedup.exists(md5):
                    skipped_dup += 1
                    continue

                # Mark and write
                dedup.add(md5)
                out_f.write(json.dumps(pair, ensure_ascii=False) + "\n")
                written += 1

    # Push dedup store updates (central SQLite on HF Space)
    try:
        dedup.flush()
    except Exception as exc:
        print(f"WARN: failed to flush dedup store: {exc}", file=sys.stderr)

    print(f"Shard {shard_id} done: written={written}, dup={skipped_dup}, unparse={skipped_unparse}")
    print(f"Output: {out_path}")


if __name__ == "__main__":
    main()
```

```bash
chmod +x bin/dataset-enrich.py
```
