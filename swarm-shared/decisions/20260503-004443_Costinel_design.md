# Costinel / design

**Final Implementation Plan — Costinel “Top-Hub Signal” Card (read-only)**  
*Sense + Signal pattern; ≤2h; frontend-only; deterministic; production-ready.*

---

## 1) Highest-value incremental improvement
Ship a **single reusable React card** that:
- Reads a lightweight knowledge‑RAG graph snapshot (JSON) from `/public/data/knowledge-rag.json`.
- **Deterministically selects the top hub by degree** (most connections).
- Renders the hub label + **up to 3 directly connected signals** (title + short context + optional link).
- Zero backend changes; graceful fallback; mobile-responsive; no execution controls.

---

### B) Reusable card component
Path: `src/components/TopHubSignalCard/TopHubSignalCard.tsx`

```tsx
import { useEffect, useState } from 'react';
import './TopHubSignalCard.css';

interface Node {
  id: string;
  label: string;
  type: 'hub' | 'signal';
  context?: string;
  href?: string;
}

interface Edge {
  source: string;
  target: string;
}

interface KnowledgeGraph {
  nodes: Node[];
  edges: Edge[];
}

export default function TopHubSignalCard() {
  const [topHub, setTopHub] = useState<Node | null>(null);
  const [signals, setSignals] = useState<Node[]>([]);

  useEffect(() => {
    fetch('/data/knowledge-rag.json')
      .then((r) => r.json())
      .then((graph: KnowledgeGraph) => {
        const degree: Record<string, number> = {};
        graph.edges.forEach((e) => {
          degree[e.source] = (degree[e.source] || 0) + 1;
          degree[e.target] = (degree[e.target] || 0) + 1;
        });

        const hubCandidates = graph.nodes.filter((n) => n.type === 'hub');
        const hubNode = hubCandidates.reduce(
          (best, cur) => ((degree[cur.id] || 0) > (degree[best.id] || 0) ? cur : best),
          hubCandidates[0]
        );

        const connectedSignals = graph.edges
          .filter((e) => e.source === hubNode.id || e.target === hubNode.id)
          .map((e) => (e.source === hubNode.id ? e.target : e.source))
          .map((id) => graph.nodes.find((n) => n.id === id))
          .filter((n): n is Node => Boolean(n) && n.type === 'signal')
          .slice(0, 3);

        setTopHub(hubNode);
        setSignals(connectedSignals);
      })
      .catch(() => {
        // Graceful fallback
        setTopHub({ id: 'MOC', label: 'MOC', type: 'hub' });
        setSignals([
          { id: 'RI-Coverage', label: 'RI Coverage', type: 'signal', context: 'Optimize committed-use coverage' },
          { id: 'Anomaly-Detection', label: 'Anomaly Detection', type: 'signal', context: 'Detect cost anomalies' },
          { id: 'Forecast-Accuracy', label: 'Forecast Accuracy', type: 'signal', context: 'Improve forecast reliability' }
        ]);
      });
  }, []);

  if (!topHub) return null;

  return (
    <div className="top-hub-card" role="region" aria-label="Top hub signal">
      <div className="hub-header">
        <span className="hub-badge">Top Hub</span>
        <h3 className="hub-name">{topHub.label}</h3>
        <p className="hub-sub">Sense + Signal — ไม่ Execute</p>
      </div>

      <div className="signals-list">
        {signals.map((s) => (
          <a
            key={s.id}
            className="signal-item"
            href={s.href}
            target={s.href ? '_blank' : undefined}
            rel={s.href ? 'noopener noreferrer' : undefined}
            tabIndex={s.href ? 0 : -1}
          >
            <span className="signal-dot" />
            <div className="signal-content">
              <div className="signal-label">{s.label}</div>
              {s.context && <div className="signal-context">{s.context}</div>}
            </div>
          </a>
        ))}
      </div>
    </div>
  );
}
```
