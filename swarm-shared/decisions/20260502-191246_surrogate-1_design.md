# surrogate-1 / design

## Implementation Plan (≤2h)

**Highest-value incremental improvement**: Add deterministic pre-flight file listing + CDN-only ingestion path to eliminate HF API rate limits during training and make shard workers resilient to 429s.

### What we’ll change
- Add `bin/list-files.py` (run once from Mac) that calls `list_repo_tree` for today’s folder, saves `file-list.json`.
- Update `bin/dataset-enrich.sh` to accept an optional file-list JSON; if present, workers stream from CDN URLs only (no `load_dataset`/`datasets` API calls during iteration).
- Add fallback: if CDN fetch fails or list missing, fall back to current `datasets` streaming path (preserves behavior).
- Add small Python helper `lib/cdn_reader.py` to stream-parse JSONL/Parquet from CDN URLs with schema projection to `{prompt, response}` and md5 dedup via `lib/dedup.py`.
- Update workflow to generate the file-list once per workflow run (single lightweight job) and pass it to the 16 shard matrix jobs via `needs` + `outputs`.

### Why this is highest value
- Eliminates repeated `list_repo_files`/`load_dataset` API calls that trigger 429s.
- CDN downloads bypass auth rate limits and are much higher quota.
- Keeps shard isolation and deterministic hashing intact.
- Fits in <2h: ~30 min for scripts, ~30 min for workflow wiring, ~60 min for testing.

### 1) `bin/list-files.py`
```python
#!/usr/bin/env python3
"""
Generate deterministic file list for a date folder.
Usage:
  HF_TOKEN=<token> python bin/list-files.py \
    --repo axentx/surrogate-1-training-pairs \
    --date 2026-05-02 \
    --out file-list.json
"""
import argparse
import json
import os
import sys
from huggingface_hub import HfApi

def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True)
    parser.add_argument("--date", required=True, help="YYYY-MM-DD folder under public-merged/")
    parser.add_argument("--out", required=True)
    args = parser.parse_args()

    token = os.environ.get("HF_TOKEN")
    if not token:
        print("HF_TOKEN required", file=sys.stderr)
        sys.exit(1)

    api = HfApi(token=token)
    # List only immediate children of the date folder (non-recursive)
    folder_path = f"public-merged/{args.date}"
    entries = api.list_repo_tree(
        repo_id=args.repo,
        path=folder_path,
        repo_type="dataset",
        recursive=False,
    )

    files = []
    for e in entries:
        if e.type == "file":
            # CDN URL (no auth, bypasses API rate limits)
            cdn_url = f"https://huggingface.co/datasets/{args.repo}/resolve/main/{e.path}"
            files.append({"path": e.path, "cdn_url": cdn_url})

    # Deterministic ordering
    files.sort(key=lambda x: x["path"])

    out = {
        "repo": args.repo,
        "date": args.date,
        "folder": folder_path,
        "files": files,
    }

    with open(args.out, "w", encoding="utf-8") as f:
        json.dump(out, f, indent=2)

    print(f"Wrote {len(files)} files to {args.out}")

if __name__ == "__main__":
    main()
```

Make executable:
```bash
chmod +x bin/list-files.py
```

### 2) `lib/cdn_reader.py`
```python
import json
import logging
import os
import pyarrow.parquet as pq
import requests
from typing import Generator, Tuple

from .dedup import is_duplicate, mark_seen

logger = logging.getLogger(__name__)

def _project_record(rec: dict) -> Tuple[str, str]:
    """Return (prompt, response) from a record; best-effort schema projection."""
    prompt = rec.get("prompt") or rec.get("input") or rec.get("question") or ""
    response = rec.get("response") or rec.get("output") or rec.get("answer") or ""
    return str(prompt), str(response)

def _md5_for_pair(prompt: str, response: str) -> str:
    import hashlib
    return hashlib.md5(f"{prompt}\0{response}".encode("utf-8")).hexdigest()

def stream_cdn_jsonl(cdn_url: str, batch_size: int = 1000) -> Generator[dict, None, None]:
    """Stream JSONL from CDN URL line-by-line."""
    resp = requests.get(cdn_url, timeout=30, stream=True)
    resp.raise_for_status()
    for line in resp.iter_lines(decode_unicode=True):
        if not line or line.isspace():
            continue
        try:
            yield json.loads(line)
        except json.JSONDecodeError:
            logger.warning("Invalid JSON line in %s: %s", cdn_url, line[:120])

def stream_cdn_parquet(cdn_url: str, columns=("prompt", "response", "input", "output", "question", "answer")) -> Generator[dict, None, None]:
    """Stream row groups from parquet via CDN URL."""
    # Use pyarrow to read remote file (HTTP range requests)
    pf = pq.ParquetFile(cdn_url)
    for batch in pf.iter_batches(batch_size=10_000, columns=columns):
        df = batch.to_pandas()
        for rec in df.to_dict(orient="records"):
            yield rec

def process_file(cdn_url: str, dedup_db_path: str) -> Generator[dict, None, None]:
    """
    Stream file from CDN, project to {prompt, response}, dedup, yield accepted pairs.
    """
    ext = os.path.splitext(cdn_url)[1].lower()
    if ext == ".jsonl":
        streamer = stream_cdn_jsonl(cdn_url)
    elif ext == ".parquet":
        streamer = stream_cdn_parquet(cdn_url)
    else:
        logger.error("Unsupported extension: %s", cdn_url)
        return

    accepted = 0
    skipped_dup = 0
    for rec in streamer:
        prompt, response = _project_record(rec)
        if not prompt.strip() or not response.strip():
            continue
        md5 = _md5_for_pair(prompt, response)
        if is_duplicate(md5, dedup_db_path):
            skipped_dup += 1
            continue
        mark_seen(md5, dedup_db_path)
        accepted += 1
        yield {"prompt": prompt, "response": response, "md5": md5}

    logger.info("CDN file processed: %s accepted=%d dup=%d", cdn_url, accepted, skipped_dup)
```
