# surrogate-1 / design

## Final Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16` (matrix) and optional `DATE_FOLDER` (defaults to today `YYYY-MM-DD`)
- Uses **one API call per cron tick** (`list_repo_tree(path=DATE_FOLDER, recursive=False)`) to produce a deterministic manifest; worker loads this manifest and operates entirely via **HF CDN direct fetches** (`https://huggingface.co/datasets/.../resolve/main/...`) with **no Authorization header** to bypass `/api/` rate limits
- Projects each file to `{prompt, response}` at parse time (avoids pyarrow CastError from mixed schemas)
- Dedups via central md5 store (`lib/dedup.py`) and writes `batches/public-merged/{DATE_FOLDER}/shard{N}-{HHMMSS}.jsonl`
- Exits non-zero on unrecoverable errors (GitHub Actions fails fast)

"""
surrogate-1 CDN-bypass ingestion worker.

Usage (GitHub Actions matrix):
  SHARD_ID=0 SHARD_TOTAL=16 \
  HF_TOKEN=hf_xxx \
  DATE_FOLDER=2026-05-03 \
  python bin/dataset-enrich.py

  SHARD_ID          - 0..15
  SHARD_TOTAL       - total shards (default 16)

SHARD_TOTAL = int(os.getenv("SHARD_TOTAL", "16"))
DATE_FOLDER = os.getenv("DATE_FOLDER", datetime.date.today().isoformat())
RUN_ID = os.getenv("RUN_ID", datetime.datetime.utcnow().strftime("%Y%m%d%H%M%S"))

OUTPUT_PATH = f"batches/public-merged/{DATE_FOLDER}/shard{SHARD_ID}-{RUN_ID}.jsonl"

# ---------- helpers ----------
def build_file_manifest(date_folder: str) -> List[str]:
    """
    Single API call: list top-level files in date_folder.
    Returns list of file paths relative to repo root.
    """
    try:
        items = HF_API.list_repo_tree(
            repo_id=DATASET_REPO,
            path=date_folder,
            recursive=False,
        )
    except Exception as exc:
        raise RuntimeError(f"list_repo_tree failed for {date_folder}: {exc}") from exc

    paths: List[str] = []
    for item in items:
        p = item.get("path") if isinstance(item, dict) else getattr(item, "path", None)
        if p and not p.endswith("/"):
            paths.append(p)

    if not paths:
        print(f"WARNING: no files found in {date_folder}")
    return sorted(paths)

def deterministic_shard(path: str, total: int) -> int:
    """Consistent shard assignment by path hash."""
    h = int(hashlib.sha256(path.encode()).hexdigest(), 16)
    return h % total

def cdn_download_url(path: str) -> str:
    """
    HF CDN URL that bypasses /api/ auth checks.
    Public datasets do not require Authorization header for CDN.
    """
    return f"https://huggingface.co/datasets/{DATASET_REPO}/resolve/main/{path}"

def safe_parquet_to_pairs(path: str) -> List[Dict[str, str]]:
    """
    Download parquet via CDN and project to {prompt, response}.
    Handles mixed schemas by selecting known columns.
    """
    url = cdn_download_url(path)

    tmp_path = tempfile.NamedTemporaryFile(suffix=".parquet", delete=False).name
    try:
        with httpx.stream("GET", url, timeout=60.0) as resp:
            resp.raise_for_status()
            with open(tmp_path, "wb") as f:
                for chunk in resp.iter_bytes(chunk_size=8192):
                    f.write(chunk)

        pf = pq.ParquetFile(tmp_path)
        schema = pf.schema.names

        prompt_col = None
        response_col = None
        for c in schema:
            low = c.strip().lower()
            if "prompt" in low:
                prompt_col = c
            if "response" in low or "completion" in low or "answer" in low:
                response_col = c

        if prompt_col is None or response_col is None:
            text_cols = [c for c in schema if pa.types.is_string(pf.schema.field(c).type)]
            if len(text_cols) >= 2:
                prompt_col, response_col = text_cols[0], text_cols[1]
            else:
                raise ValueError(f"Cannot find prompt/response columns in {path}: {schema}")

        table = pf.read(columns=[prompt_col, response_col])
        prompts = table.column(prompt_col).to_pylist()
        responses = table.column(response_col).to_pylist()

        pairs = []
        for p, r in zip(prompts, responses):
            if isinstance(p, str) and isinstance(r, str):
                pairs.append({"prompt": p.strip(), "response": r.strip()})
        return pairs
    finally:
        try:
            os.unlink(tmp_path)
        except Exception:
            pass

def build_output_rows(pairs: List[Dict[str, str]]) -> List[str]:
    """Produce JSONL lines."""
    rows = []
    for pair in pairs:
        rows.append(json.dumps(pair, ensure_ascii=False))
    return rows

# ---------- main ----------

    # 1) manifest (single API call)
    print("Building manifest...")
    all_files = build_file_manifest(DATE_FOLDER)
    eligible = [p for p in all_files if deterministic_shard(p, SHARD_TOTAL) == SHARD_ID]
    print(f"Eligible files for shard {SHARD_ID}: {len(eligible)}")

    # 2) dedup store
    dedup = DedupStore()

    # 3) process
    out_rows: List[str] = []
    total_pairs = 0
    for path in eligible:
        print(f"Processing {path}...")
        try:
            pairs = safe_parquet_to_pairs(path)
