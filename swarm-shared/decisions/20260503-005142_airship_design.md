## 2) Mac orchestration — `file_list.py`
Run once per date window (after rate-limit reset). Produces deterministic sibling mapping and avoids repeated API calls.

```python
#!/usr/bin/env python3
"""
Generate file-list.json for Surrogate-1 training.
Run on Mac (or any orchestration host) after rate-limit window clears.
"""
import json, os, datetime, hashlib
from huggingface_hub import list_repo_tree

REPO = "org/surrogate-1"
DATE = datetime.date.today().isoformat()        # e.g. 2026-05-03
FOLDER = f"batches/mirror-merged/{DATE}"
OUT = "file-list.json"
SIBLINGS = 5

# Single non-recursive API call
tree = list_repo_tree(REPO, path=FOLDER, recursive=False)
files = [
    {"repo": REPO, "path": node.path}
    for node in tree
    if node.type == "file" and node.path.lower().endswith(".parquet")
]

# Deterministic 5-sibling sharding (commit-cap avoidance)
for f in files:
    slug = os.path.splitext(os.path.basename(f["path"]))[0]
    idx = int(hashlib.sha256(slug.encode()).hexdigest(), 16) % SIBLINGS
    f["repo"] = f"org/surrogate-1-sibling-{idx}"

manifest = {"date": DATE, "folder": FOLDER, "files": files}
with open(OUT, "w", encoding="utf-8") as fp:
    json.dump(manifest, fp, indent=2)

print(f"Wrote {len(files)} files to {OUT}")
```

## 3) CDN loader — `cdn_loader.py`
Zero HF API usage; robust schema projection to `{prompt, response}` and avoids `pyarrow.CastError`.

```python
import pyarrow.parquet as pq
import requests, io
from typing import List, Dict

HF_CDN = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def cdn_fetch(repo: str, path: str, timeout: int = 30) -> bytes:
    url = HF_CDN.format(repo=repo, path=path)
    resp = requests.get(url, timeout=timeout)
    resp.raise_for_status()
    return resp.content

def project_to_chat(buf: bytes) -> List[Dict[str, str]]:
    # Try strict schema first (avoids unnecessary full reads)
    try:
        table = pq.read_table(io.BytesIO(buf), columns=["prompt", "response"])
        df = table.to_pandas()
        return df[["prompt", "response"]].to_dict(orient="records")
    except (KeyError, ValueError, OSError):
        pass

    # Fallback: tolerant single-text-column handling
    try:
        table = pq.read_table(io.BytesIO(buf))
        col = next((c for c in table.column_names if "text" in c.lower()), table.column_names[0])
        texts = table[col].to_pylist()
    except Exception:
        # Last resort: return empty pairs to keep pipeline alive
        return [{"prompt": "", "response": ""}]

    pairs = []
    sep = "### Response:"
    for t in texts:
        s = str(t or "")
        if sep in s:
            p, r = s.split(sep, 1)
            pairs.append({"prompt": p.strip(), "response": r.strip()})
        else:
            pairs.append({"prompt": "", "response": s})
    return pairs

def load_manifest(manifest_path: str):
    with open(manifest_path, encoding="utf-8") as f:
        cfg = json.load(f)
    for fspec in cfg.get("files", []):
        buf = cdn_fetch(fspec["repo"], fspec["path"])
        yield from project_to_chat(buf)
```

## 4) Lightning training script — `train_cdn.py`
Zero HF API calls during training; plug-and-play with Surrogate-1 model.

```python
#!/usr/bin/env python3
"""
Lightning Studio entrypoint for Surrogate-1 training (CDN-only).
"""
import json, os
import lightning as L
from torch.utils.data import Dataset, DataLoader
from cdn_loader import load_manifest

MANIFEST = os.getenv("MANIFEST_PATH", "file-list.json")

class ChatDataset(Dataset):
    def __init__(self, manifest_path: str):
        self.records = list(load_manifest(manifest_path))

    def __len__(self) -> int:
        return len(self.records)

    def __getitem__(self, idx: int):
        r = self.records[idx]
        return {"prompt": r["prompt"], "response": r["response"]}

class SurrogateDataModule(L.LightningDataModule):
    def __init__(self, manifest_path: str, batch_size: int = 8):
        super().__init__()
        self.manifest_path = manifest_path
        self.batch_size = batch_size

    def setup(self, stage=None):
        self.ds = ChatDataset(self.manifest_path)

    def train_dataloader(self):
        return DataLoader(
            self.ds,
            batch_size=self.batch_size,
            shuffle=True,
            num_workers=4,
            pin_memory=True,
        )

if __name__ == "__main__":
    from lightning.pytorch import Trainer
    from lightning.pytorch.callbacks import ModelCheckpoint

    # TODO: import and initialize Surrogate-1 model
    # model = SurrogateModel(...)

    dm = SurrogateDataModule(MANIFEST)
    trainer = Trainer(
        max_epochs=1,
        accelerator="gpu",
        devices=1,
        callbacks=[ModelCheckpoint(monitor="train_loss", save_top_k=1)],
        limit_train_batches=10,  # quick smoke test
    )
    trainer.fit(model=None, datamodule=dm)  # replace `None` with model when ready
```
