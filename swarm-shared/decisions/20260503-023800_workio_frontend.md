# workio / frontend

## 1. Diagnosis

- **No optimistic UI for clock in/out**: users see no immediate feedback on punch actions; perceived latency increases double-tap risk.
- **Missing client-side idempotency key plumbing**: frontend does not generate or attach `X-Idempotency-Key` (or equivalent) for punch requests, so retries/retries from LINE or double-taps can race.
- **No transition guards while request is inflight**: button remains enabled and accepts duplicate submissions during network flight.
- **Geofence check is synchronous/blocking and not pre-validated**: UX blocks on GPS before allowing action; no early “out of zone” prevention.
- **No offline queue for punches**: if network fails, action is lost instead of queued locally and retried.

## 2. Proposed change

File: `workio/src/features/punch/PunchButton.tsx` (or equivalent component handling clock in/out).  
Scope: add optimistic state, idempotency key generation, inflight guard, geofence pre-check, and lightweight offline queue (IndexedDB/localStorage) with retry.

```tsx
// workio/src/features/punch/PunchButton.tsx
import { useState, useCallback, useRef } from 'react';
import { useLocation } from './useLocation';
import { queuePunch, PunchPayload } from './punchQueue';

function generateIdempotencyKey(): string {
  return `${crypto.randomUUID()}`;
}

export function PunchButton() {
  const [status, setStatus] = useState<'idle' | 'pending' | 'success' | 'error'>('idle');
  const inflightRef = useRef(false);
  const { getCurrentPosition, isInsideGeofence } = useLocation();

  const punch = useCallback(async () => {
    if (inflightRef.current) return;
    inflightRef.current = true;
    setStatus('pending');

    try {
      // 1) pre-check geofence (fast, cached or fresh)
      const pos = await getCurrentPosition({ timeout: 8000 });
      if (!isInsideGeofence(pos)) {
        throw new Error('OUTSIDE_GEOFENCE');
      }

      const payload: PunchPayload = {
        type: 'PUNCH',
        idempotencyKey: generateIdempotencyKey(),
        timestamp: Date.now(),
        location: pos,
      };

      // 2) optimistic update (UI)
      setStatus('success');

      // 3) queue for delivery (offline-first)
      await queuePunch(payload);
    } catch (err: any) {
      setStatus('error');
      console.error(err);
      // keep button enabled for retry on error
    } finally {
      inflightRef.current = false;
      // reset UI state after brief success display
      if (status === 'success') {
        setTimeout(() => setStatus('idle'), 1500);
      }
    }
  }, [getCurrentPosition, isInsideGeofence]);

  return (
    <button
      onClick={punch}
      disabled={status === 'pending'}
      className={`
        px-6 py-3 rounded-lg font-medium transition
        ${status === 'idle' && 'bg-blue-600 text-white hover:bg-blue-700'}
        ${status === 'pending' && 'bg-gray-400 text-white cursor-wait'}
        ${status === 'success' && 'bg-green-600 text-white'}
        ${status === 'error' && 'bg-red-600 text-white'}
      `}
    >
      {status === 'idle' && 'ลงเวลา'}
      {status === 'pending' && 'กำลังบันทึก...'}
      {status === 'success' && 'บันทึกแล้ว'}
      {status === 'error' && 'ลองใหม่'}
    </button>
  );
}
```

```ts
// workio/src/features/punch/punchQueue.ts
import localForage from 'localforage';

const QUEUE_KEY = 'punch_queue_v1';

export interface PunchPayload {
  type: 'PUNCH';
  idempotencyKey: string;
  timestamp: number;
  location: GeolocationCoordinates;
}

const store = localForage.createInstance({
  name: 'workio_punch_queue',
});

export async function queuePunch(payload: PunchPayload) {
  const queue: PunchPayload[] = (await store.getItem(QUEUE_KEY)) || [];
  queue.push(payload);
  await store.setItem(QUEUE_KEY, queue);
  // best-effort immediate flush
  flushQueue().catch(() => {/* handled by retry loop */});
}

export async function flushQueue(): Promise<void> {
  const queue: PunchPayload[] = (await store.getItem(QUEUE_KEY)) || [];
  if (queue.length === 0) return;

  const remaining: PunchPayload[] = [];
  for (const item of queue) {
    try {
      const res = await fetch('/api/punch', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'X-Idempotency-Key': item.idempotencyKey,
        },
        body: JSON.stringify(item),
      });

      if (!res.ok) {
        // keep item for retry
        remaining.push(item);
        // stop on 4xx (except 409/429) to avoid hammering invalid payloads
        if (res.status >= 400 && res.status < 500 && res.status !== 409 && res.status !== 429) {
          // discard malformed requests after client-side validation
          continue;
        }
      }
    } catch {
      // network error — keep item
      remaining.push(item);
    }
  }

  await store.setItem(QUEUE_KEY, remaining);
}

// periodic retry (called from app bootstrap)
export function startQueuePolling(intervalMs = 30000) {
  setInterval(() => flushQueue().catch(() => {}), intervalMs);
  // flush on visibility change
  document.addEventListener('visibilitychange', () => {
    if (!document.hidden) flushQueue().catch(() => {});
  });
}
```

```ts
// workio/src/features/punch/useLocation.ts
export function useLocation() {
  const getCurrentPosition = (options?: PositionOptions): Promise<GeolocationCoordinates> => {
    return new Promise((resolve, reject) => {
      if (!navigator.geolocation) {
        reject(new Error('GEOLOCATION_UNSUPPORTED'));
        return;
      }
      navigator.geolocation.getCurrentPosition(
        (pos) => resolve(pos.coords),
        (err) => reject(err),
        { enableHighAccuracy: true, timeout: 10000, maximumAge: 30000, ...options }
      );
    });
  };

  // simple circular geofence; replace with your tenant polygon/radius logic
  const isInsideGeofence = (coords: GeolocationCoordinates): boolean => {
    const center = { lat: 13.7563, lng: 100.5018 }; // example: Bangkok
    const R = 6371e3;
    const φ1 = (center.lat * Math.PI) / 180;
    const φ2 = (coords.latitude * Math.PI) / 180;
    const Δφ = ((coords.latitude - center.lat) * Math.PI) / 180;
    const Δλ = ((coords.longitude - center.lng) * Math.PI) / 180;
    const a = Math.sin(Δφ / 2) ** 2 + Math.cos(φ1) * Math.cos(φ2) * Math.sin(Δλ / 2) ** 2;
    const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
    const distance = R * c;
    return distance <= 500; // 500m radius
  };

  return { getCurrentPosition, isInsideGeofence };
}
```
