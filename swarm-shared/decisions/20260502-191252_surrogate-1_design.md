# surrogate-1 / design

## Implementation Plan (≤2h)

**Highest-value incremental improvement**: Add deterministic pre-flight file listing + CDN-only ingestion path to eliminate HF API rate limits during training and make shard workers resilient to 429s.

### What we’ll change
- Add `bin/list-files.py` (run once from Mac) that calls `list_repo_tree(recursive=False)` for today’s folder and writes `file-list.json` (path + size + sha256). Embed this list in training scripts so Lightning workers do **CDN-only** fetches with zero API calls during data load.
- Update `bin/dataset-enrich.sh` to accept an optional file-list; if provided, workers skip `load_dataset`/`list_repo_files` and stream via `hf_hub_download` (CDN URLs) for their deterministic shard slice.
- Add retry/back-off for 429 with 360s sleep and deterministic jitter so 16 parallel runners don’t thundering-herd on retry.
- Add a small wrapper in `lib/cdn_stream.py` to fetch parquet/JSONL via CDN, project to `{prompt, response}` only at parse time (avoid mixed-schema pyarrow errors), and yield normalized pairs.

### Why this is highest value
- Eliminates the HF API 429/1000-5min ceiling during ingestion and training (the #1 blocker per past patterns).
- Keeps Mac as orchestrator only; heavy compute stays on Lightning/Kaggle/Colab.
- Costs nothing (CDN tier has much higher limits) and reuses existing 16-shard matrix.

### 1) `bin/list-files.py`
```python
#!/usr/bin/env python3
"""
Run from Mac (or cron) once per folder/day.
Produces file-list.json used by training and shard workers.
"""
import json
import os
import sys
from datetime import datetime, timezone

from huggingface_hub import HfApi

REPO = "datasets/axentx/surrogate-1-training-pairs"
# Folder pattern used by workers: batches/public-merged/<date>/
TODAY = datetime.now(timezone.utc).strftime("%Y%m%d")
FOLDER = f"batches/public-merged/{TODAY}/"
OUT = "file-list.json"

def main() -> None:
    api = HfApi()
    # Single API call; recursive=False avoids pagination explosion
    entries = api.list_repo_tree(
        repo_id=REPO,
        path=FOLDER,
        repo_type="dataset",
        recursive=False,
    )

    files = []
    for e in entries:
        if not e.size:
            continue
        # Use CDN URL (no auth/rate-limit for public datasets)
        cdn_url = f"https://huggingface.co/datasets/{REPO}/resolve/main/{e.path}"
        files.append(
            {
                "path": e.path,
                "size": e.size,
                "sha256": getattr(e, "sha256", None),
                "cdn_url": cdn_url,
            }
        )

    payload = {
        "generated_at_utc": datetime.now(timezone.utc).isoformat(),
        "folder": FOLDER,
        "count": len(files),
        "files": files,
    }

    with open(OUT, "w", encoding="utf-8") as f:
        json.dump(payload, f, indent=2)

    print(f"Wrote {len(files)} files -> {OUT}")
    # Also write a stable deterministic shard map so workers can pick slices
    # without coordination.
    write_shard_map([f["cdn_url"] for f in files])

def write_shard_map(urls: list[str], shards: int = 16) -> None:
    """Deterministic shard assignment by hash(url) % shards."""
    import hashlib

    assignments = [[] for _ in range(shards)]
    for u in urls:
        idx = int(hashlib.md5(u.encode()).hexdigest(), 16) % shards
        assignments[idx].append(u)

    os.makedirs("shards", exist_ok=True)
    for i, items in enumerate(assignments):
        with open(f"shards/shard-{i}.json", "w", encoding="utf-8") as f:
            json.dump({"shard_id": i, "urls": items}, f, indent=2)

if __name__ == "__main__":
    main()
```

Make executable:
```bash
chmod +x bin/list-files.py
```

### 2) `lib/cdn_stream.py`
```python
#!/usr/bin/env python3
"""
CDN-only streaming loader for surrogate-1 training pairs.
No HuggingFace `datasets` or `list_repo_files` during training.
"""
import json
import logging
import os
import time
from pathlib import Path
from typing import Iterator, Tuple

import numpy as np
import pyarrow as pa
import pyarrow.parquet as pq
import requests

logger = logging.getLogger(__name__)

# Retry policy tuned for HF CDN + occasional 429
MAX_RETRIES = 5
BACKOFF = [1, 2, 4, 8, 16]

def robust_get(url: str, timeout: int = 30) -> bytes:
    for attempt in range(MAX_RETRIES):
        try:
            resp = requests.get(url, timeout=timeout)
            if resp.status_code == 429:
                wait = 360
                logger.warning("HF CDN 429; sleeping %ss", wait)
                time.sleep(wait)
                continue
            resp.raise_for_status()
            return resp.content
        except requests.RequestException as exc:
            if attempt == MAX_RETRIES - 1:
                raise
            sleep = BACKOFF[attempt]
            logger.warning("Request failed (%s), retry in %ss", exc, sleep)
            time.sleep(sleep)
    raise RuntimeError("unreachable")

def stream_shard_urls(urls: list[str]) -> Iterator[Tuple[str, str]]:
    """
    Yield (prompt, response) from a list of CDN URLs.
    Projects to {prompt, response} only at parse time to avoid mixed-schema errors.
    """
    for url in urls:
        data = robust_get(url)
        p = Path(url)
        if p.suffix == ".parquet":
            try:
                table = pq.read_table(pa.BufferReader(data), columns=["prompt", "response"])
            except (pa.ArrowInvalid, KeyError):
                # Fallback: read all and project
                table = pq.read_table(pa.BufferReader(data))
                if "prompt" not in table.column_names or "response" not in table.column_names:
                    logger.warning("Missing prompt/response in %s, skipping", url)
                    continue
            df = table.to_pandas()
            for _, row in df.iterrows():
                prompt = str(row.get("prompt") or "")
                response = str(row.get("response") or "")
                if prompt and response:
                    yield prompt, response
        elif p.suffix == ".jsonl":
            for line in data.decode("utf-8", errors="replace").splitlines():
                line = line.strip()
                if not line:
                    continue
                try:
                    obj = json.loads(line)
                except json.JSONDecodeError:
                    continue
                prompt = str(obj.get("prompt") or obj.get("input") or "")
                response = str(obj.get("response") or obj.get("output") or "")
                if prompt and response:
                    yield prompt, response
        else:
            logger.debug("Unsupported extension %s for %s", p.suffix, url)

def stream_shard_file(shard_path: str) -> Iterator[Tuple[str, str]]:
    """Load shard-N.json produced by list-files.py and stream its URLs."""
    with open(shard_path, encoding="utf-8") as f:
        payload = json.load(f)
    return stream_shard_urls(payload["urls"])
```
