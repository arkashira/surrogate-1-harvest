# workio / quality

## Final Synthesis — Correct, Actionable Solution

**Core diagnosis (accepted from both candidates)**
- No server-side idempotency → retries/double-taps create duplicates.
- Read-then-insert race condition on `(employee_id, punch_type, date)`.
- No database uniqueness guard → app logic can be bypassed.
- No durable retry/deduplication for LINE/webhook or client network failures.

**Chosen approach**
- Enforce uniqueness at the database level (single source of truth).
- Add lightweight server-side idempotency with `Idempotency-Key` to make retries safe and deterministic.
- Keep implementation minimal, atomic, and observable.

## 1) Database changes (schema.sql)

```sql
-- Prevent duplicate punches for same employee/type/date
CREATE UNIQUE INDEX IF NOT EXISTS idx_attendance_unique_punch
ON attendance (employee_id, punch_type, DATE(punched_at));

-- Lightweight idempotency ledger (kept minimal)
CREATE TABLE IF NOT EXISTS idempotency_keys (
  key TEXT PRIMARY KEY,
  request_hash TEXT NOT NULL,
  response_status INTEGER NOT NULL,
  response_body TEXT NOT NULL,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
```

## 2) Server implementation (attendance.ts)

```ts
// workio/server/src/routes/attendance.ts
import { Router, Request, Response } from 'express';
import { db } from '../db';
import crypto from 'crypto';

const router = Router();

function hashPayload(body: unknown): string {
  return crypto.createHash('sha256').update(JSON.stringify(body)).digest('hex');
}

router.post('/punch', async (req: Request, res: Response) => {
  const idempotencyKey = req.header('Idempotency-Key');
  const { employee_id, punch_type, latitude, longitude } = req.body;

  if (!idempotencyKey) {
    return res.status(400).json({ error: 'Idempotency-Key header required' });
  }
  if (!employee_id || !punch_type) {
    return res.status(400).json({ error: 'employee_id and punch_type required' });
  }

  const client = await db.connect();
  try {
    await client.query('BEGIN');

    // Idempotency check (exact replay)
    const idem = await client.query(
      'SELECT response_status, response_body FROM idempotency_keys WHERE key = $1',
      [idempotencyKey]
    );
    if (idem.rows.length > 0) {
      await client.query('ROLLBACK');
      return res
        .status(idem.rows[0].response_status)
        .json(JSON.parse(idem.rows[0].response_body));
    }

    const punchedAt = new Date();
    const date = punchedAt.toISOString().split('T')[0];

    // Try insert; on unique violation, fetch existing punch for this date/type
    try {
      const insertResult = await client.query(
        `INSERT INTO attendance (employee_id, punch_type, punched_at, latitude, longitude)
         VALUES ($1, $2, $3, $4, $5)
         RETURNING *`,
        [employee_id, punch_type, punchedAt, latitude, longitude]
      );

      const responseBody = { punch: insertResult.rows[0], created: true };
      await client.query(
        `INSERT INTO idempotency_keys (key, request_hash, response_status, response_body)
         VALUES ($1, $2, $3, $4)`,
        [idempotencyKey, hashPayload(req.body), 201, JSON.stringify(responseBody)]
      );

      await client.query('COMMIT');
      return res.status(201).json(responseBody);
    } catch (err: any) {
      if (err.code === '23505') {
        const existing = await client.query(
          `SELECT * FROM attendance
           WHERE employee_id = $1 AND punch_type = $2 AND DATE(punched_at) = $3
           ORDER BY punched_at DESC LIMIT 1`,
          [employee_id, punch_type, date]
        );

        const responseBody = {
          punch: existing.rows[0] || null,
          created: false,
          duplicate: true
        };
        await client.query(
          `INSERT INTO idempotency_keys (key, request_hash, response_status, response_body)
           VALUES ($1, $2, $3, $4)`,
          [idempotencyKey, hashPayload(req.body), 200, JSON.stringify(responseBody)]
        );

        await client.query('COMMIT');
        return res.status(200).json(responseBody);
      }
      throw err;
    }
  } catch (err) {
    await client.query('ROLLBACK');
    console.error('Attendance punch error:', err);
    return res.status(500).json({ error: 'Internal server error' });
  } finally {
    client.release();
  }
});

export default router;
```

Key points:
- All checks and writes are inside a transaction to avoid races.
- Unique index is the authoritative guard; idempotency key ensures safe retries.
- On `23505` (unique violation), we return the existing punch rather than erroring.

## 3) Client changes (frontend/service)

```ts
// workio/src/lib/services/attendance.ts
export async function punchWithIdempotency(
  employeeId: string,
  punchType: 'in' | 'out',
  location?: { lat: number; lng: number }
) {
  const idempotencyKey = crypto.randomUUID();
  const payload = { employee_id: employeeId, punch_type: punchType, ...location };

  try {
    const res = await fetch('/attendance/punch', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Idempotency-Key': idempotencyKey
      },
      body: JSON.stringify(payload)
    });

    if (!res.ok) throw new Error(`HTTP ${res.status}`);
    return await res.json();
  } catch (err) {
    // Enqueue for retry with same idempotencyKey (use existing durable queue)

  }
}
```

- Use the same `idempotencyKey` across retries for exactly-once semantics.
- Durable queue (IndexedDB) preserves keys across page reloads/network loss.
