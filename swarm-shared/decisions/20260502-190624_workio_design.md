# workio / design

**Final synthesized plan (best of both proposals, corrected + actionable)**

---

## 1. Diagnosis (merged, de-duplicated)

- **No tenant cache on LINE webhook**  
  Every webhook performs `channelId → tenant` DB lookup; under burst traffic this amplifies DB load, increases latency, and raises error rates.

- **Signature verification likely happens after DB hit**  
  Wasted DB cycles on unauthenticated requests; must verify LINE signature **before** tenant resolution.

- **Missing health/readiness endpoints**  
  No `/health` or `/ready`; deployments risk traffic sent to unhealthy or not-ready pods.

- **No graceful shutdown handling**  
  In-flight LINE webhook requests can be dropped during restart/deploy, causing missed clock events.

- **No request-id / trace-id**  
  Hard to correlate LINE deliveries with logs when debugging clock-in/out failures.

- **No retry/backoff or circuit-breaker on tenant lookup**  
  Transient DB or pool exhaustion cascades into total webhook failure; LINE may drop events on immediate 5xx.

---

## 2. Scope of change

- Add early LINE signature verification middleware.
- Add in-memory LRU tenant cache with short TTL and circuit-breaker fallback.
- Add request-id middleware and propagate to logs + DB.
- Add `/health` and `/ready` endpoints.
- Add graceful shutdown handling.
- Make webhook handler fast and resilient (return 200 quickly; persist events reliably).

File targets:
- `server/src/middleware/line-signature.ts`
- `server/src/middleware/request-id.ts`
- `server/src/utils/tenant-cache.ts`
- `server/src/routes/line-webhook.ts`
- `server/src/routes/health.ts`
- `server/src/index.ts` (or main entry)

### 3.2 Middleware: LINE signature verification

`server/src/middleware/line-signature.ts`

```ts
import crypto from 'crypto';
import { Request, Response, NextFunction } from 'express';

export function lineSignature(secret: string) {
  return (req: Request, res: Response, next: NextFunction) => {
    const sig = req.headers['x-line-signature'] as string | undefined;
    if (!sig) return res.status(401).json({ error: 'Missing signature' });

    // Use raw body if available; otherwise serialize parsed JSON deterministically
    const body = (req as any).rawBody || JSON.stringify(req.body || {});
    const expected = crypto
      .createHmac('sha256', secret)
      .update(body, 'utf8')
      .digest('base64');

    if (!crypto.timingSafeEqual(Buffer.from(sig), Buffer.from(expected))) {
      return res.status(401).json({ error: 'Invalid signature' });
    }
    next();
  };
}
```

> **Note**: If you need strict raw-body handling, configure body-parser to buffer raw body for this route only and attach `req.rawBody`.

### 3.3 Tenant cache with circuit-breaker

`server/src/utils/tenant-cache.ts`

```ts
import LRU from 'lru-cache';
import { Pool } from 'pg';
import CircuitBreaker from 'opossum';

const cache = new LRU<string, { tenantId: string }>({
  max: 500,
  ttl: 1000 * 60 * 5, // 5m
});

async function fetchTenantFromDb(pool: Pool, channelId: string): Promise<string | null> {
  const { rows } = await pool.query(
    'SELECT id FROM tenants WHERE line_channel_id = $1 LIMIT 1',
    [channelId]
  );
  return rows[0]?.id ?? null;
}

export function createTenantResolver(pool: Pool) {
  const breaker = new CircuitBreaker(
    (channelId: string) => fetchTenantFromDb(pool, channelId),
    {
      timeout: 3000,
      errorThresholdPercentage: 50,
      resetTimeout: 30000,
    }
  );

  // Fallback: return null so webhook can continue (or reject) depending on policy
  breaker.fallback(() => null);

  return async function resolveTenantByChannel(channelId: string): Promise<string | null> {
    const cached = cache.get(channelId);
    if (cached) return cached.tenantId;

    const tenantId = await breaker.fire(channelId);
    if (tenantId) {
      cache.set(channelId, { tenantId });
    }
    return tenantId;
  };
}
```

### 3.4 Request-id middleware

`server/src/middleware/request-id.ts`

```ts
import { Request, Response, NextFunction } from 'express';
import { v4 as uuidv4 } from 'uuid';

declare global {
  namespace Express {
    interface Request {
      id?: string;
    }
  }
}

export function requestId() {
  return (req: Request, res: Response, next: NextFunction) => {
    req.id = req.headers['x-request-id'] as string || uuidv4();
    res.setHeader('X-Request-Id', req.id);
    next();
  };
}
```

### 3.5 LINE webhook route

`server/src/routes/line-webhook.ts`

```ts
import { Router } from 'express';
import { pool } from '../db';
import { lineSignature } from '../middleware/line-signature';
import { createTenantResolver } from '../utils/tenant-cache';

const router = Router();
const resolveTenant = createTenantResolver(pool);

// Early signature verification
router.use(lineSignature(process.env.LINE_CHANNEL_SECRET!));

router.post('/webhook/line', async (req, res) => {
  const requestId = (req as any).id || 'unknown';
  const start = Date.now();

  try {
    const events = req.body.events || [];
    const channelId = req.body.destination as string | undefined;

    if (!channelId) {
      return res.status(400).json({ error: 'Missing channel' });
    }

    const tenantId = await resolveTenant(channelId);
    if (!tenantId) {
      return res.status(404).json({ error: 'Tenant not found' });
    }

    // Persist events (minimal example)
    for (const ev of events) {
      await pool.query(
        `INSERT INTO clock_events (tenant_id, user_id, event_type, timestamp, raw_event, request_id)
         VALUES ($1,$2,$3,$4,$5,$6)`,
        [tenantId, ev.source?.userId, ev.type, new Date(ev.timestamp), JSON.stringify(ev), requestId]
      );
    }

    const elapsed = Date.now() - start;
    console.log(`[${requestId}] processed ${events.length} events in ${elapsed}ms`);
    return res.status(200).json({ ok: true });
  } catch (err) {
    console.error(`[${requestId}] webhook error`, err);
    // Return 200 to avoid LINE redelivery only if you can safely ignore; otherwise 500
    return res.status(500).json({ error: 'Internal server error' });
  }
});

export default router;
```

### 3.6 Health endpoints

`server/src/routes/health.ts`

```ts
import { Router } from 'express';
import { pool } from '../db';

const router = Router();

router.get('/health', (_req, res) => {
  res.status(200).json({ status: 'ok', uptime: process.uptime() });
});

router.get('/ready', async (_req, res) => {
  try {
    await pool.query('SELECT 1');
    res.status(200).json({ status: 'ready' });
  } catch (err) {
    res.status(503).json({ status: 'not ready', db: 'down' });
  }
});

export default router;
```
