# workio / design

## Final synthesized implementation (correct + actionable)

### 1. Diagnosis (resolved)

- **Duplicate root cause**: LINE retries + client double-taps + no request-level deduplication + missing storage-level uniqueness.
- **Concurrency root cause**: app-level “check-then-upsert” races; must be resolved at DB level with a unique constraint and atomic upsert.
- **UI/UX root cause**: optimistic updates without rollback or idempotency token produce unrecoverable divergence.
- **Auditability root cause**: no immutable log for corrections or compliance.

### 2. Changes (minimal, high-leverage)

**Files to change**
- DB/schema + migration: add uniqueness + idempotency + audit tables.
- Backend: atomic upsert service + idempotency middleware + LINE webhook handler.
- Frontend: idempotency key generation + rollback on conflict.

**Net-new files**
- `server/src/db/migrations/20260504_punch_idempotency.sql`
- `server/src/middleware/idempotency.ts`
- `server/src/services/punch.ts` (atomic upsert + audit)
- `src/components/PunchButton.tsx` (idempotency + rollback)

### 3. DB schema + migration (storage-level correctness)

```sql
-- server/src/db/migrations/20260504_punch_idempotency.sql
BEGIN;

-- One active punch per employee per day per type.
-- Soft-delete via is_active allows corrections without breaking constraint.
ALTER TABLE punches
  ADD CONSTRAINT uq_employee_date_type_active
  UNIQUE (employee_id, punch_date, punch_type)
  WHERE (is_active IS TRUE);

-- Idempotency table (7d retention recommended; index for expiry cleanup)
CREATE TABLE IF NOT EXISTS punch_idempotency (
  idempotency_key TEXT PRIMARY KEY,
  employee_id     INTEGER NOT NULL REFERENCES employees(id) ON DELETE CASCADE,
  punch_type      TEXT    NOT NULL,
  punch_date      DATE    NOT NULL,
  created_at      TIMESTAMPTZ DEFAULT NOW()
);
CREATE INDEX idx_idempotency_expiry ON punch_idempotency (created_at);

-- Immutable audit log for compliance and reconciliation
CREATE TABLE punch_audit (
  id              BIGSERIAL PRIMARY KEY,
  punch_id        INTEGER,
  employee_id     INTEGER NOT NULL,
  punch_type      TEXT    NOT NULL,
  punch_date      DATE    NOT NULL,
  clocked_at      TIMESTAMPTZ NOT NULL,
  latitude        DOUBLE PRECISION,
  longitude       DOUBLE PRECISION,
  action          TEXT NOT NULL, -- 'create','revoke','reconcile'
  reason          TEXT,
  performed_by    TEXT,
  idempotency_key TEXT,
  created_at      TIMESTAMPTZ DEFAULT NOW()
);

COMMIT;
```

### 4. Idempotency middleware (API boundary safety)

```ts
// server/src/middleware/idempotency.ts
import { Request, Response, NextFunction } from 'express';
import { db } from '../db';

export async function withPunchIdempotency(
  req: Request,
  res: Response,
  next: NextFunction
) {
  const key = req.header('Idempotency-Key') || req.body.idempotencyKey;
  if (!key) {
    return res.status(400).json({ error: 'Idempotency-Key is required' });
  }

  const exists = await db.oneOrNone(
    'SELECT 1 FROM punch_idempotency WHERE idempotency_key = $1',
    [key]
  );
  if (exists) {
    return res.status(409).json({ error: 'Duplicate request', idempotencyKey: key });
  }

  (req as any).idempotencyKey = key;
  next();
}
```

### 5. Atomic punch service (concurrency + audit)

```ts
// server/src/services/punch.ts
import { db } from '../db';

export interface PunchInput {
  employeeId: number;
  punchType: 'clock_in' | 'clock_out';
  clockedAt: Date;
  latitude?: number;
  longitude?: number;
  idempotencyKey?: string;
}

export async function recordPunch(input: PunchInput) {
  const client = await db.connect();
  try {
    await client.query('BEGIN');

    const punchDate = new Date(input.clockedAt).toISOString().split('T')[0];

    // Revoke any active same-type for the day (enforces single active punch)
    await client.query(
      `UPDATE punches
       SET is_active = FALSE, updated_at = NOW()
       WHERE employee_id = $1
         AND punch_date = $2
         AND punch_type = $3
         AND (is_active IS TRUE)`,
      [input.employeeId, punchDate, input.punchType]
    );

    // Insert new active punch
    const result = await client.query(
      `INSERT INTO punches (employee_id, punch_type, punch_date, clocked_at, latitude, longitude, is_active)
       VALUES ($1, $2, $3, $4, $5, $6, TRUE)
       RETURNING id`,
      [input.employeeId, input.punchType, punchDate, input.clockedAt, input.latitude, input.longitude]
    );

    const punchId = result.rows[0].id;

    // Record idempotency if provided (prevents retries from re-entering)
    if (input.idempotencyKey) {
      await client.query(
        `INSERT INTO punch_idempotency (idempotency_key, employee_id, punch_type, punch_date)
         VALUES ($1, $2, $3, $4)`,
        [input.idempotencyKey, input.employeeId, input.punchType, punchDate]
      );
    }

    // Immutable audit
    await client.query(
      `INSERT INTO punch_audit (punch_id, employee_id, punch_type, punch_date, clocked_at, latitude, longitude, action, idempotency_key)
       VALUES ($1, $2, $3, $4, $5, $6, $7, 'create', $8)`,
      [punchId, input.employeeId, input.punchType, punchDate, input.clockedAt, input.latitude, input.longitude, input.idempotencyKey]
    );

    await client.query('COMMIT');
    return { punchId, punchDate, punchType: input.punchType };
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  } finally {
    client.release();
  }
}
```

### 6. LINE webhook handler (fast, idempotent)

```ts
// server/src/routes/line.ts (excerpt)
import { Router } from 'express';
import { recordPunch } from '../services/punch';
import { withPunchIdempotency } from '../middleware/idempotency';

const router = Router();

router.post('/webhook/line', withPunchIdempotency, async (req, res) => {
  try {
    // existing LINE signature validation here
    const { employeeId, punchType, latitude, longitude } = normalizeLineEvent(req.body);
    const now = new Date();

    const result = await recordPunch({
      employeeId,
      punchType,
      clockedAt: now,
      latitude,
      longitude,
      idempotencyKey: (req as any).idempotencyKey,
    });

    // quick 200 to LINE to prevent retries
    res.json({ ok: true, punch: result });
  } catch (err: any) {
    // uniqueness violation or duplicate idempotency
    if (err.code === '23505' || err.message?.includes('Duplicate')) {
      return res.status(409).json({ error: 'Duplicate request' });
    }
    console.error(err);
    res.status(500).json({ error: 'Server error' });
  }
});
```
