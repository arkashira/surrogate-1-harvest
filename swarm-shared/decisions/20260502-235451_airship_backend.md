### 1. HF Manifest Route (surrogate/api/routes/hf_manifest.py)
```python
from fastapi import APIRouter, HTTPException, Depends, Query
from huggingface_hub import HfApi, list_repo_tree
from datetime import datetime, timezone
import hashlib
import json
from typing import Optional
import os

router = APIRouter(prefix="/api/hf-manifest", tags=["hf-manifest"])
hf_api = HfApi()

# Simple in-memory cache (1 hour TTL)
_MANIFEST_CACHE = {}

def get_cache_key(repo: str, path: str) -> str:
    return f"{repo}:{path or ''}"

@router.get("/")
async def get_hf_manifest(
    repo: str = Query(..., description="HF dataset repo (e.g., 'username/dataset')"),
    path: str = Query("", description="Subfolder path within repo"),
    nocache: bool = Query(False, description="Bypass cache")
):
    cache_key = get_cache_key(repo, path)
    
    if not nocache and cache_key in _MANIFEST_CACHE:
        cached = _MANIFEST_CACHE[cache_key]
        if datetime.now(timezone.utc).timestamp() - cached["generated_at"] < 3600:
            return cached["data"]
    
    try:
        # Use list_repo_tree per folder (avoids recursive pagination)
        tree = list_repo_tree(repo=repo, path=path or "", recursive=False)
        
        files = []
        for item in tree:
            if item.type == "file":
                # CDN URL (bypasses API rate limits)
                cdn_url = f"https://huggingface.co/datasets/{repo}/resolve/main/{item.path}"
                
                # Generate deterministic hash from path (for caching)
                sha256 = hashlib.sha256(item.path.encode()).hexdigest()[:16]
                
                files.append({
                    "name": os.path.basename(item.path),
                    "path": item.path,
                    "cdn_url": cdn_url,
                    "size": getattr(item, "size", 0),
                    "sha256": sha256,
                    "lfs": getattr(item, "lfs", False)
                })
        
        result = {
            "repo": repo,
            "path": path or "/",
            "generated_at": datetime.now(timezone.utc).isoformat(),
            "file_count": len(files),
            "files": sorted(files, key=lambda x: x["path"]),
            "cdn_bypass": True,
            "note": "CDN URLs bypass HF API rate limits. Use in Lightning training with data_files parameter."
        }
        
        _MANIFEST_CACHE[cache_key] = {
            "data": result,
            "generated_at": datetime.now(timezone.utc).timestamp()
        }
        
        return result
        
    except Exception as e:
        raise HTTPException(status_code=500, detail=f"HF manifest generation failed: {str(e)}")
```

### 2. CLI Wrapper (surrogate/scripts/generate_hf_manifest.sh)
```bash
#!/usr/bin/env bash
# Deterministic HF CDN manifest generator for Surrogate-1 training
# Usage: ./generate_hf_manifest.sh <repo> <path> <date> [output_dir]

set -euo pipefail
SHELL=/bin/bash

REPO="${1:-}"
PATH_IN_REPO="${2:-}"
DATE="${3:-$(date +%Y%m%d)}"
OUTPUT_DIR="${4:-batches/mirror-merged/$DATE}"

if [[ -z "$REPO" ]]; then
    echo "Usage: $0 <repo> <path> <date> [output_dir]" >&2
    exit 1
fi

# Ensure executable
chmod +x "$0"

# Generate slug for deterministic naming
SLUG=$(echo "$REPO" | tr '/' '-' | tr -cd '[:alnum:]-')
OUTPUT_FILE="$OUTPUT_DIR/${SLUG}.json"

mkdir -p "$OUTPUT_DIR"

echo "Generating HF CDN manifest for $REPO (path: $PATH_IN_REPO)..."

# Call surrogate API (running on port 8001)
curl -s "http://localhost:8001/api/hf-manifest?repo=$REPO&path=$PATH_IN_REPO&nocache=false" \
    -o "$OUTPUT_FILE"

if [[ $? -eq 0 && -s "$OUTPUT_FILE" ]]; then
    echo "✓ Manifest saved to $OUTPUT_FILE"
    echo "Files: $(jq '.file_count' "$OUTPUT_FILE")"
else
    echo "✗ Failed to generate manifest" >&2
    exit 1
fi
```

def load_cdn_manifest(manifest_path: str):
    """Load deterministic CDN manifest for Lightning training."""
    with open(manifest_path) as f:

    # Extract CDN URLs for HF load_dataset

    return {
        "data_files": cdn_files,  # Pass directly to load_dataset
        "repo": manifest["repo"],
        "file_count": manifest["file_count"]
