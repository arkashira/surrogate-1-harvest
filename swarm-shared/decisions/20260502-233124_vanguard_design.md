# vanguard / design

## Final Synthesized Design (Correct + Actionable)

**Core principle:**  
- **Mac = orchestrator only** (no model loading).  
- **Lightning Studio = single reused long-lived run** (no recreate cycles).  
- **Training = CDN-only data** (zero HF API/auth during epochs).  
- **Writes = deterministic sibling rotation** to bypass 128 commit/hr/repo cap.  
- **Schema = projection at parse time** to avoid `pyarrow.CastError`.

### 1. Orchestrator (run on Mac)

```python
#!/usr/bin/env python3
"""
Orchestrator (run on Mac).
- One API call per date folder → manifest with CDN URLs.
- Reuse/create Lightning Studio.
- Deterministic sibling repo for writes.
"""
import json, hashlib, os, sys
from pathlib import Path
from typing import List, Dict

from huggingface_hub import HfApi, list_repo_tree
from lightning import Teamspace, Studio, Machine

HF_REPO = os.getenv("HF_REPO", "axentx/surrogate-1")
HF_SIBLINGS = int(os.getenv("HF_SIBLINGS", "5"))     # 5 siblings => 640 writes/hr total
DATE_FOLDER = os.getenv("DATE_FOLDER", "2026-04-29")
MANIFEST_DIR = Path(__file__).parent.parent / "manifests" / DATE_FOLDER
HF_TOKEN = os.getenv("HF_TOKEN", "")

if not HF_TOKEN:
    sys.exit("HF_TOKEN required")

api = HfApi(token=HF_TOKEN)

# ---- 1) Manifest ----
def list_date_files(repo: str, date_folder: str) -> List[str]:
    tree = list_repo_tree(repo=repo, path=date_folder, recursive=False, token=HF_TOKEN)
    return sorted(f.rfilename for f in tree if f.type == "file")

def write_manifest(files: List[str], date_folder: str) -> Path:
    MANIFEST_DIR.mkdir(parents=True, exist_ok=True)
    manifest_path = MANIFEST_DIR / "files.json"
    payload = {
        "repo": HF_REPO,
        "date_folder": date_folder,
        "files": files,
        "cdn_prefix": f"https://huggingface.co/datasets/{HF_REPO}/resolve/main/{date_folder}"
    }
    manifest_path.write_text(json.dumps(payload, indent=2))
    return manifest_path

# ---- 2) Sibling rotation ----
def pick_sibling_repo(slug: str) -> str:
    h = hashlib.sha256(slug.encode()).hexdigest()
    idx = int(h, 16) % HF_SIBLINGS
    if idx == 0:
        return HF_REPO
    return f"{HF_REPO}-s{idx}"

# ---- 3) Studio reuse ----
def get_or_create_studio(name: str, machine: Machine = Machine.L40S) -> Studio:
    teamspace = Teamspace()
    for s in teamspace.studios:
        if s.name == name:
            if s.status == "running":
                print(f"Reusing running studio: {name}")
                return s
            print(f"Studio {name} status={s.status}. Restarting on {machine}.")
            s.start(machine=machine)
            return s
    print(f"Creating studio {name} on {machine}")
    return Studio(name=name, machine=machine, create_ok=True)

# ---- 4) CDN config for training ----
def build_cdn_config(manifest_path: Path) -> Dict:
    data = json.loads(manifest_path.read_text())
    files = data["files"]
    cdn_urls = [f"{data['cdn_prefix']}/{f}" for f in files]
    return {
        "cdn_urls": cdn_urls,
        "repo": data["repo"],
        "date_folder": data["date_folder"],
        "note": "Use these URLs directly (fsspec/requests). No HF API/auth during training."
    }

if __name__ == "__main__":
    # Generate manifest once per date folder
    files = list_date_files(HF_REPO, DATE_FOLDER)
    manifest_path = write_manifest(files, DATE_FOLDER)
    print(f"Manifest: {manifest_path}")

    cfg = build_cdn_config(manifest_path)
    cfg_path = MANIFEST_DIR / "cdn_config.json"
    cfg_path.write_text(json.dumps(cfg, indent=2))
    print(f"CDN config: {cfg_path}")

    # Reuse/create Studio
    studio = get_or_create_studio("vanguard-surrogate-train", machine=Machine.L40S)
    print(f"Studio URL: {studio.url}")

    # Example launch (uncomment to run)
    # sibling = pick_sibling_repo(f"mirror-merged-{DATE_FOLDER}")
    # studio.run(
    #     target="train.py",
    #     arguments=[
    #         "--manifest", str(cfg_path),
    #         "--output-repo", sibling
    #     ]
    # )
```

### 2. Training script (runs in Lightning Studio)

`/opt/axentx/vanguard/train.py`

```python
import argparse
import io
import json
from pathlib import Path

import pandas as pd
import requests
from datasets import Dataset

def load_cdn_files(cdn_config_path: str) -> Dataset:
    cfg = json.loads(Path(cdn_config_path).read_text())
    cdn_urls = cfg["cdn_urls"]

    def gen():
        for url in cdn_urls:
            r = requests.get(url, timeout=30)
            r.raise_for_status()
            # Assume parquet; adapt if jsonl/txt
            df = pd.read_parquet(io.BytesIO(r.content))
            for _, row in df.iterrows():
                # Projection at parse time → avoids pyarrow.CastError on mixed schemas
                prompt = row.get("prompt") or row.get("text") or row.get("input") or ""
                response = row.get("response") or row.get("output") or row.get("completion") or ""
                if prompt and response:
                    yield {"prompt": str(prompt), "response": str(response)}
    return Dataset.from_generator(gen)

if __name__ == "__main__":
    p = argparse.ArgumentParser()
    p.add_argument("--manifest", required=True)
    p.add_argument("--output-repo", required=True)
    args = p.parse_args()

    ds = load_cdn_files(args.manifest)
    print(f"Loaded {len(ds)} examples via CDN")
    # Continue training here (no HF API/auth during data loading)
```
