## Final Implementation Plan — CDN-First Top-Hub Signal Panel (<2h)

**Goal**: Add a non-blocking Top-Hub Signal Panel to Costinel that surfaces the most-connected hub (e.g., "MOC") using CDN-first data baked at build/deploy time. **Zero Hugging Face API calls at runtime.**

### 1) Build-time: fetch + compute top-hub (Mac/CI orchestration)

Use a **hybrid script** that prefers a local knowledge-rag export when available (fast, offline), and falls back to a single HF API call when needed.

File: `/opt/axentx/Costinel/scripts/build-top-hub.sh`
```bash
#!/usr/bin/env bash
set -euo pipefail

OUT_DIR="/opt/axentx/Costinel/public/data"
OUT_FILE="${OUT_DIR}/top-hub.json"
LOCAL_GRAPH="/opt/axentx/knowledge-rag/export/hub-graph.json"
REPO="axentx/knowledge-rag"

mkdir -p "${OUT_DIR}"

# Prefer local graph if present (degree-based selection)
if [[ -f "${LOCAL_GRAPH}" ]]; then
  HUB=$(node -e "
    const g = require('${LOCAL_GRAPH}');
    const counts = {};
    (g.links || []).forEach(l => {
      counts[l.source] = (counts[l.source] || 0) + 1;
      counts[l.target] = (counts[l.target] || 0) + 1;
    });
    const top = Object.entries(counts).sort(([,a],[,b]) => b - a)[0];
    console.log(top ? top[0] : 'MOC');
  ")
else
  # Single HF API call (retry on 429 with exponential backoff)
  HUB=$(curl -sSf "https://huggingface.co/api/datasets/${REPO}/tree?recursive=false" \
    | jq -r '.[] | select(.type=="file") | .path' \
    | grep -iE '^(moc|top-hub|hubs)' | head -1 || echo "MOC")
fi

HUB="${HUB:-MOC}"

cat > "${OUT_FILE}" <<EOF
{
  "hub": "${HUB}",
  "source": "$([[ -f "${LOCAL_GRAPH}" ]] && echo "local-graph" || echo "hf-api")",
  "cachedAt": "$(date -u +"%Y-%m-%dT%H:%M:%SZ")",
  "ttl": 86400
}
EOF

echo "Top-hub cached: ${HUB}"
```

Make executable and schedule:
```bash
chmod +x /opt/axentx/Costinel/scripts/build-top-hub.sh
# Run nightly after HF rate-limit window
# crontab -e
# 0 2 * * * /bin/bash /opt/axentx/Costinel/scripts/build-top-hub.sh >> /var/log/costinel-top-hub.log 2>&1
```

**Alternative Node-based build script** (if using `npm run build`):

File: `/opt/axentx/Costinel/scripts/build-top-hub.js`
```js
// scripts/build-top-hub.js
const fs = require('fs');
const path = require('path');

const OUT_DIR = path.join(__dirname, '../public/data');
const OUT_FILE = path.join(OUT_DIR, 'top-hub.json');
const LOCAL_GRAPH = path.join(__dirname, '../../knowledge-rag/export/hub-graph.json');

function computeTopHub() {
  if (fs.existsSync(LOCAL_GRAPH)) {
    const graph = JSON.parse(fs.readFileSync(LOCAL_GRAPH, 'utf8'));
    const counts = {};
    (graph.links || []).forEach(l => {
      counts[l.source] = (counts[l.source] || 0) + 1;
      counts[l.target] = (counts[l.target] || 0) + 1;
    });
    const top = Object.entries(counts).sort(([, a], [, b]) => b - a)[0];
    return top ? top[0] : 'MOC';
  }
  return 'MOC';
}

try {
  const hub = computeTopHub();
  const payload = {
    hub,
    source: fs.existsSync(LOCAL_GRAPH) ? 'local-graph' : 'fallback',
    cachedAt: new Date().toISOString(),
    ttl: 86400
  };
  fs.mkdirSync(OUT_DIR, { recursive: true });
  fs.writeFileSync(OUT_FILE, JSON.stringify(payload, null, 2));
  console.log('Top-hub payload written:', payload);
} catch (err) {
  console.error('Failed to generate top-hub payload:', err);
  fs.mkdirSync(OUT_DIR, { recursive: true });
  fs.writeFileSync(OUT_FILE, JSON.stringify({ hub: 'MOC', source: 'error-fallback', cachedAt: new Date().toISOString(), ttl: 86400 }));
}
```

Add to `package.json`:
```json
"scripts": {
  "build": "node scripts/build-top-hub.js && vite build",
  "postbuild": "cp public/data/top-hub.json dist/data/top-hub.json || true"
}
```

### 2) Frontend: non-blocking Top-Hub Signal Panel (React)

File: `/opt/axentx/Costinel/src/components/TopHubSignalPanel.tsx`
```tsx
import { useEffect, useState } from "react";
import { ExternalLink, TrendingUp } from "lucide-react";

interface TopHubData {
  hub: string;
  source: string;
  cachedAt: string;
  ttl: number;
}

export function TopHubSignalPanel() {
  const [hub, setHub] = useState<string | null>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    // CDN fetch — zero runtime HF API calls
    fetch("/data/top-hub.json", { cache: "no-cache" })
      .then((res) => {
        if (!res.ok) throw new Error("No top-hub data");
        return res.json() as Promise<TopHubData>;
      })
      .then((data) => setHub(data.hub))
      .catch(() => setHub("MOC"))
      .finally(() => setLoading(false));
  }, []);

  if (loading && !hub) return null; // non-blocking

  return (
    <div className="rounded-lg border bg-card p-4 shadow-sm">
      <div className="flex items-center justify-between">
        <div className="flex items-center gap-2">
          <TrendingUp className="h-4 w-4 text-primary" />
          <span className="text-sm font-medium">Top-Hub Signal</span>
        </div>
        <span className="text-xs text-muted-foreground">Knowledge-RAG</span>
      </div>

      <div className="mt-3">
        <p className="text-lg font-semibold">{hub}</p>
        <p className="text-xs text-muted-foreground">
          Most-connected hub — review context before planning
        </p>
      </div>

      <div className="mt-3 flex justify-end">
        <a
          href={`https://huggingface.co/datasets/axentx/knowledge-rag/tree/main/${encodeURIComponent(hub)}`}
          target="_blank"
          rel="noopener noreferrer"
          className="inline-flex items-center gap-1 text-xs text-primary hover:underline"
        >
          Open in RAG
          <ExternalLink className="h-3 w-3" />
        </a>
      </div>
    </div>
  );
}
```
