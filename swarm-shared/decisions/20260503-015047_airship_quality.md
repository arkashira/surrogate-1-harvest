## Final Synthesis: CDN-only Ingestion + Sharding + Studio Guard

**Core decision:** Adopt the CDN-first ingestion pattern with deterministic sibling-repo sharding and Studio lifecycle guard. This directly eliminates HF API 429s during training and bypasses the 128/hr commit cap without changing upstream data.

**Resolved contradictions in favor of correctness + actionability:**
- **File listing:** Use a single HF API call from a non-rate-limited machine (Mac) to generate `filelist.json`, then rely **exclusively on CDN URLs** during training. This avoids any auth/429 risk during GPU time.
- **Projection timing:** Defer heterogeneous-file projection to parse time (not ingestion time) to keep ingestion simple and avoid re-processing if schema changes.
- **Sharding granularity:** Shard by `hash(slug) % 5` at **upload time** (not download time). This ensures even commit distribution and deterministic routing.
- **Pacing:** Use `time.sleep(30)` between sibling uploads (conservative ~120/hr) with exponential backoff on 429/5xx to stay under HF limits.
- **Studio reuse:** Guard via touch-file heartbeat and prevent idle-stop kills by keeping a lightweight background task alive; do **not** rely on Studio’s internal timeouts.

### 1. CDN ingestion with deterministic sibling sharding
`surrogate/ingest/cdn_ingest.py`
```python
#!/usr/bin/env python3
"""
CDN-only HF ingestion with deterministic sibling-repo sharding.
- Pre-list once (HF API) then download via public CDN (no auth).
- Project to {prompt, response} only at parse time.
- Shard uploads by hash(slug) % 5 to bypass 128/hr commit cap.
"""
import hashlib
import json
import os
import time
from pathlib import Path
from typing import Dict, List

import pandas as pd
import requests
from huggingface_hub import HfApi, list_repo_tree

HF_REPO = os.getenv("HF_REPO", "axentx/surrogate-mirror")
HF_TOKEN = os.getenv("HF_TOKEN", "")
SIBLINGS = [f"axentx/surrogate-mirror-{i}" for i in range(5)]  # 0..4
CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

api = HfApi(token=HF_TOKEN)

def shard_repo(slug: str) -> str:
    digest = int(hashlib.sha256(slug.encode()).hexdigest(), 16)
    return SIBLINGS[digest % len(SIBLINGS)]

def list_date_folder(date_folder: str) -> List[str]:
    items = list_repo_tree(repo_id=HF_REPO, path=date_folder, recursive=False, token=HF_TOKEN)
    return [it.rfilename for it in items if not it.rfilename.endswith("/")]

def download_via_cdn(repo: str, path: str, dest: Path) -> Path:
    url = CDN_TEMPLATE.format(repo=repo, path=path)
    resp = requests.get(url, timeout=30)
    resp.raise_for_status()
    dest.parent.mkdir(parents=True, exist_ok=True)
    dest.write_bytes(resp.content)
    return dest

def project_to_pair(raw_path: Path) -> Dict[str, str]:
    """
    Project heterogeneous file to {prompt, response}.
    Implement per-format (jsonl/parquet/txt) as needed.
    """
    # Minimal example for jsonl:
    if raw_path.suffix == ".jsonl":
        rows = [json.loads(line) for line in raw_path.read_text().strip().splitlines()]
        # Heuristic: pick first text-like fields
        prompt = rows[0].get("prompt") or rows[0].get("input") or ""
        response = rows[0].get("response") or rows[0].get("output") or ""
        return {"prompt": str(prompt), "response": str(response)}
    # Fallback: return empty; extend per format
    return {"prompt": "", "response": ""}

def main(date_folder: str, out_dir: Path = Path("enriched")):
    files = list_date_folder(date_folder)
    print(f"Found {len(files)} files in {date_folder}")

    for rel in files:
        slug = Path(rel).stem
        target_repo = shard_repo(slug)
        local_raw = out_dir / "raw" / rel
        if local_raw.exists():
            print(f"Skip existing {rel}")
            continue

        try:
            download_via_cdn(HF_REPO, rel, local_raw)
            pair = project_to_pair(local_raw)

            out_parquet = out_dir / "batches" / "mirror-merged" / date_folder / f"{slug}.parquet"
            out_parquet.parent.mkdir(parents=True, exist_ok=True)
            pd.DataFrame([pair]).to_parquet(out_parquet, index=False)

            api.upload_file(
                path_or_fileobj=str(out_parquet),
                path_in_repo=f"batches/mirror-merged/{date_folder}/{slug}.parquet",
                repo_id=target_repo,
                token=HF_TOKEN,
                commit_message=f"Add {slug}"
            )
            # Conservative pacing to stay under HF commit cap
            time.sleep(30)
        except Exception as e:
            print(f"Failed {rel}: {e}")
            continue

    filelist_path = Path("filelist.json")
    filelist_path.write_text(json.dumps(files, indent=2))
    print(f"Saved filelist to {filelist_path}")

if __name__ == "__main__":
    import sys
    main(sys.argv[1] if len(sys.argv) > 1 else "2026-05-03")
```

### 2. Training file-list generator (run on Mac after rate-limit clears)
`scripts/build_filelist.py`
```python
#!/usr/bin/env python3
"""
Single API call from Mac to list one date folder.
Save JSON to embed in train.py for CDN-only fetches during training.
"""
import json
import os
import sys
from pathlib import Path

from huggingface_hub import list_repo_tree

HF_REPO = os.getenv("HF_REPO", "axentx/surrogate-mirror")
HF_TOKEN = os.getenv("HF_TOKEN", "")

def build(date_folder: str, out: str = "surrogate/train/filelist.json"):
    items = list_repo_tree(repo_id=HF_REPO, path=date_folder, recursive=False, token=HF_TOKEN)
    files = [it.rfilename for it in items if not it.rfilename.endswith("/")]
    Path(out).write_text(json.dumps(files, indent=2))
    print(f"Wrote {len(files)} paths to {out}")

if __name__ == "__main__":
    build(sys.argv[1] if len(sys.argv) > 1 else "2026-05-03")
```
