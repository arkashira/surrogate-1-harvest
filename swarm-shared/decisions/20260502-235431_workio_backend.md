# workio / backend

## Final Synthesized Solution

### 1. Diagnosis (merged)
- **Non-idempotent LINE webhook**: retries on 5xx/timeouts cause duplicate punches and invalid open/close state.
- **Non-atomic punch transitions**: `SELECT → decide → INSERT/UPDATE` interleaves under concurrency/retries, producing dangling opens or double-closes.
- **No deduplication**: missing idempotency key and no stored request/delivery ID to correlate retries.
- **Race-prone state logic**: row-level locking scope is either missing or too narrow; no transactional boundary around full punch transition.
- **Missing audit trail**: no correlation of LINE delivery/webhook event IDs with processed punches.

### 2. Target
- **File**: `/opt/axentx/workio/server/src/routes/webhook/line.ts` (or equivalent handler).
- **Scope**: make handler idempotent and atomic; add idempotency table, row-level locking, transactional upsert-safe close logic, and audit fields.

### 3. Implementation

```ts
// server/src/routes/webhook/line.ts
import { Router, Request, Response } from 'express';
import { pool } from '../../db';
import { verifyLineSignature } from '../../utils/line';

const router = Router();

/**
 * Idempotent LINE clock-in/out handler.
 * Guarantees:
 * - Exactly-once processing per LINE event (idempotency key = stable event identifier)
 * - Atomic punch transition via SELECT ... FOR UPDATE + transaction
 * - No double-close / no dangling open punches
 * - Audit trail: stores LINE delivery/webhook event IDs and request IDs
 */
router.post('/line', verifyLineSignature, async (req: Request, res: Response) => {
  const events = req.body.events;
  if (!events || !Array.isArray(events)) {
    return res.status(400).json({ error: 'invalid_payload' });
  }

  const client = await pool.connect();
  try {
    for (const event of events) {
      if (event.type !== 'message' || event.message?.type !== 'text') continue;

      const userId = event.source?.userId;
      const text = event.message.text?.trim().toLowerCase();
      const timestamp = event.timestamp;
      const lineDeliveryId = event.deliveryId || null;
      const lineWebhookEventId = event.webhookEventId || null;

      // Stable idempotency key: prefer explicit IDs; fallback to deterministic composite
      const eventId = lineDeliveryId || lineWebhookEventId || `${timestamp}-${userId}-${text}`;

      if (!userId) continue;

      const action = text === 'ออก' || text === 'out' ? 'OUT' : 'IN';
      const now = new Date(timestamp);

      await client.query('BEGIN');
      try {
        // 1) Idempotency check (unique constraint protects race between check/insert)
        const idem = await client.query(
          `SELECT 1 FROM line_webhook_idempotency
           WHERE event_id = $1 AND user_id = $2`,
          [eventId, userId]
        );

        if (idem.rowCount && idem.rowCount > 0) {
          await client.query('ROLLBACK');
          continue; // already processed
        }

        // 2) Lock current open punch for this user to serialize transitions
        const openPunch = await client.query(
          `SELECT id, user_id, tenant_id, clock_in, clock_out
           FROM punches
           WHERE user_id = $1 AND clock_out IS NULL
           ORDER BY clock_in DESC
           LIMIT 1
           FOR UPDATE`,
          [userId]
        );

        if (action === 'IN') {
          // If there's an open punch, auto-close it (safety) then create new IN
          if (openPunch.rowCount && openPunch.rowCount > 0) {
            await client.query(
              `UPDATE punches
               SET clock_out = $1, updated_at = NOW()
               WHERE id = $2`,
              [now, openPunch.rows[0].id]
            );
          }

          await client.query(
            `INSERT INTO punches (user_id, tenant_id, clock_in, clock_out, created_at, updated_at, request_id, line_delivery_id, line_webhook_event_id)
             SELECT $1, COALESCE((SELECT active_tenant_id FROM users WHERE id = $1), 1), $2, NULL, NOW(), NOW(), $3, $4, $5`,
            [userId, now, eventId, lineDeliveryId, lineWebhookEventId]
          );
        } else {
          // OUT
          if (openPunch.rowCount && openPunch.rowCount > 0) {
            await client.query(
              `UPDATE punches
               SET clock_out = $1, updated_at = NOW(),
                   request_id = COALESCE(request_id, $2),
                   line_delivery_id = COALESCE(line_delivery_id, $3),
                   line_webhook_event_id = COALESCE(line_webhook_event_id, $4)
               WHERE id = $5`,
              [now, eventId, lineDeliveryId, lineWebhookEventId, openPunch.rows[0].id]
            );
          } else {
            // No open punch: create paired record with clock_in = clock_out (graceful handling)
            await client.query(
              `INSERT INTO punches (user_id, tenant_id, clock_in, clock_out, created_at, updated_at, request_id, line_delivery_id, line_webhook_event_id)
               SELECT $1, COALESCE((SELECT active_tenant_id FROM users WHERE id = $1), 1), $2, $2, NOW(), NOW(), $3, $4, $5`,
              [userId, now, eventId, lineDeliveryId, lineWebhookEventId]
            );
          }
        }

        // 3) Record idempotency (unique constraint prevents race insert)
        await client.query(
          `INSERT INTO line_webhook_idempotency (event_id, user_id, action, line_delivery_id, line_webhook_event_id, processed_at)
           VALUES ($1, $2, $3, $4, $5, NOW())`,
          [eventId, userId, action, lineDeliveryId, lineWebhookEventId]
        );

        await client.query('COMMIT');
      } catch (err) {
        await client.query('ROLLBACK');
        console.error('Failed to process LINE punch event', { eventId, userId, action, err });
        // continue processing other events
      }
    }

    res.status(200).json({ ok: true });
  } catch (err) {
    console.error('LINE webhook handler fatal error', err);
    res.status(500).json({ error: 'internal_error' });
  } finally {
    client.release();
  }
});

export default router;
```

-- server/src/db/schema.sql  (add idempotency table + audit columns)
CREATE TABLE IF NOT EXISTS line_webhook_idempotency (
  id BIGSERIAL PRIMARY KEY,
  event_id TEXT NOT NULL,
  user_id TEXT NOT NULL,
  action TEXT NOT NULL,
  line_delivery_id TEXT NULL,
  line_webhook_event_id TEXT NULL,
  processed_at TIMESTAMPTZ NOT NULL,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  UNIQUE (event_id, user_id)
);

CREATE INDEX IF NOT EXISTS idx_line_idempotency_event_user ON line_webhook_idempotency (event_id, user_id);
CREATE INDEX IF NOT EXISTS idx_line_idempotency_user_action ON line_webhook_idempotency (user_id, action);
CREATE INDEX IF NOT EXISTS idx_punches_user_open ON punches (user_id) WHERE clock_out IS NULL;

-- Add audit columns to punches if not present
ALTER TABLE punches
  ADD COLUMN IF NOT EXISTS request_id TEXT NULL,
  ADD COLUMN IF NOT EXISTS line_delivery_id TEXT NULL,
  ADD COLUMN IF NOT EXISTS line_webhook_event_id TEXT NULL;

CREATE INDEX IF NOT EXISTS idx_punches_request_id ON punches (request_id) WHERE request_id IS NOT NULL;
CREATE INDEX IF NOT EXISTS idx_punches_line_delivery_id ON punches (line_delivery_id) WHERE line_delivery_id IS NOT NULL;
