## 2. Manifest generation (single HF API call)

```python
# /opt/axentx/vanguard/ops/generate_manifest.py
#!/usr/bin/env python3
"""
Generate a content-addressed manifest for a date folder.

Usage:
  python3 generate_manifest.py <repo> <date> [out_dir]

Example:
  python3 generate_manifest.py axentx/mirror-merged 2026-05-03
"""

import json
import os
import sys
import time
import hashlib
from pathlib import Path
from typing import List, Dict

from huggingface_hub import HfApi, hf_hub_download, list_repo_tree
from tqdm import tqdm

def build_manifest(repo: str, date: str, out_dir: str) -> Path:
    api = HfApi()
    manifest_path = Path(out_dir) / "manifest.json"
    manifest_path.parent.mkdir(parents=True, exist_ok=True)

    # One API call: list files in the date folder (non-recursive to keep scope tight).
    entries = list_repo_tree(repo_id=repo, path=date, repo_type="dataset", recursive=False)
    items: List[Dict] = []

    for e in tqdm(entries, desc=f"Processing {date}"):
        if e.type != "file":
            continue

        file_path = f"{date}/{e.path}"
        # Single cached download per file (local HF cache).
        local_path = hf_hub_download(repo_id=repo, filename=file_path, repo_type="dataset")

        sha256 = hashlib.sha256()
        with open(local_path, "rb") as f:
            for chunk in iter(lambda: f.read(8192), b""):
                sha256.update(chunk)

        size = os.path.getsize(local_path)
        items.append({
            "file_path": file_path,
            "sha256": sha256.hexdigest(),
            "size": size,
            "cdn_url": f"https://huggingface.co/datasets/{repo}/resolve/main/{file_path}"
        })

    manifest = {
        "repo": repo,
        "date": date,
        "generated_at": time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime()),
        "items": items
    }

    with open(manifest_path, "w") as f:
        json.dump(manifest, f, indent=2)

    print(f"Wrote {len(items)} entries to {manifest_path}")
    return manifest_path

if __name__ == "__main__":
    if len(sys.argv) < 3:
        print("Usage: generate_manifest.py <repo> <date> [out_dir]")
        sys.exit(1)

    repo = sys.argv[1]
    date = sys.argv[2]
    out_dir = sys.argv[3] if len(sys.argv) > 3 else f"../batches/mirror-merged/{date}"
    build_manifest(repo, date, out_dir)
```

Key properties:
- Exactly one `list_repo_tree` call per run.
- Deterministic manifest; safe to commit or embed in training configs.
- CDN URLs recorded for direct fetches; training will avoid HF API entirely.

## 3. Training launcher (CDN-only, manifest-driven)

```python
# /opt/axentx/vanguard/train_launcher.py
"""
Launch training with a manifest and CDN-only data fetches.

Usage:
  python3 train_launcher.py --manifest <path> --repo <repo> [--script train.py] [--args ...]
"""

import argparse
import json
import os
import subprocess
from pathlib import Path
from typing import Iterator, Dict, Any

from huggingface_hub import hf_hub_download

def load_manifest(manifest_path: str) -> Dict[str, Any]:
    with open(manifest_path) as f:
        return json.load(f)

def cdn_data_iter(manifest: Dict[str, Any], use_cdn_direct: bool = True) -> Iterator[Path]:
    """
    Yield local file paths for training. Uses hf_hub_download (cached) or direct CDN.
    Zero HF API calls during iteration when using direct CDN or cached files.
    """
    repo = manifest["repo"]
    for item in manifest["items"]:
        fp = item["file_path"]
        if use_cdn_direct:
            # Direct CDN fetch (no auth/API). Replace with your loader (parquet/jsonl/etc).
            # Example: wget/curl or requests to item["cdn_url"]
            # Here we still use hf_hub_download for cache consistency and simplicity.
            local = hf_hub_download(repo_id=repo, filename=fp, repo_type="dataset")
        else:
            local = hf_hub_download(repo_id=repo, filename=fp, repo_type="dataset")
        yield Path(local)

def build_data_args(manifest_path: str, repo: str) -> list:
    """
    Return training CLI arguments that embed the manifest and avoid runtime HF API calls.
    Customize to your training script flags.
    """
    return [
        "--manifest", manifest_path,
        "--repo", repo,
        "--data-loader-mode", "cdn-only",
        "--deterministic"
    ]

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--manifest", required=True)
    parser.add_argument("--repo", required=True)
    parser.add_argument("--script", default="train.py")
    parser.add_argument("--out-dir", default="outputs")
    args, passthrough = parser.parse_known_args()

    manifest = load_manifest(args.manifest)
    data_args = build_data_args(args.manifest, args.repo)

    cmd = [
        "python3", args.script,
        *data_args,
        *passthrough
    ]

    print("Running:", " ".join(cmd))
    subprocess.run(cmd, check=True)

if __name__ == "__main__":
    main()
```

In your `train.py`, read `--manifest` and use `cdn_data_iter` (or equivalent) to materialize files. No `list_repo_tree`/`load_dataset` during training.

## 4. Lightning Studio reuse + idle-stop handling

```python
# /opt/axentx/vanguard/ops/lightning_launcher.py
"""
Lightweight guardrails to reuse or restart a Lightning Studio.
"""

import time
from lightning import Lightning, Teamspace, Machine, Venv

def get_or_create_studio(
    name: str,
    machine=Machine.L40S,
    requirements=None
) -> Lightning:
    ls = Lightning()
    studios = Teamspace.studios()

    # Reuse running studio
    for s in studios:
        if s.name == name and s.status == "Running":
            print(f"Reusing running studio: {name}")
            return s

    # Restart stopped studio
    for s in studios:
        if s.name == name:
            print(f"Starting stopped studio: {name}")
            s.start(machine=machine)
            # Wait until running
            for _ in range(30):
                if s.status == "Running":
                    return s
                time.sleep(10)
            raise
