## Implementation Plan (≤2h)

**Highest-value incremental improvement**: Add deterministic pre-flight file listing + CDN-only ingestion path to eliminate HF API rate limits during training and make shard workers resilient to 429s.

### What we’ll change
- Add `bin/list-files.py` (run once from Mac) → produces `file-list-<date>.json` with CDN URLs.
- Update `bin/dataset-enrich.sh` to accept an optional file-list JSON; if provided, workers fetch via CDN (no HF API) and skip `load_dataset` for heterogeneous schemas.
- Add lightweight `lib/cdn_download.py` helper with retry/backoff and `hf_hub_download` fallback for private files.
- Update README with usage and the “pre-list once, embed in training” pattern.

### Why this is highest value
- Eliminates 429s during ingestion/training (CDN bypass).
- Removes recursive `list_repo_files` calls that trigger pagination/rate limits.
- Keeps shard workers independent and deterministic (same file list → no collisions).
- Fits <2h: ~60 lines of Python + small shell changes + docs.

### 1) `bin/list-files.py`
```python
#!/usr/bin/env python3
"""
Pre-flight file lister for surrogate-1 dataset.
Run from Mac (or any machine with huggingface_hub) after rate-limit window clears.

Outputs: file-list-<date>.json
Each entry:
{
  "path": "raw/2026-04-27/part-00000.parquet",
  "cdn_url": "https://huggingface.co/datasets/axentx/surrogate-1-training-pairs/resolve/main/raw/2026-04-27/part-00000.parquet",
  "size": 123456,
  "sha256": null
}
"""

import argparse
import datetime
import json
import os
import sys

from huggingface_hub import HfApi

REPO_ID = "axentx/surrogate-1-training-pairs"

def list_date_folder(date_folder: str, recursive: bool = False):
    api = HfApi()
    entries = api.list_repo_tree(repo_id=REPO_ID, path=date_folder, recursive=recursive)
    out = []
    for e in entries:
        if e.type != "file":
            continue
        path = e.path
        out.append({
            "path": path,
            "cdn_url": f"https://huggingface.co/datasets/{REPO_ID}/resolve/main/{path}",
            "size": e.size if hasattr(e, "size") else None,
            "sha256": None
        })
    return out

def main():
    parser = argparse.ArgumentParser(description="List dataset files for CDN ingestion.")
    parser.add_argument("--date-folder", default=None, help="Folder path in dataset (e.g. raw/2026-04-27)")
    parser.add_argument("--out-dir", default=".", help="Output directory for JSON file list")
    args = parser.parse_args()

    if args.date_folder is None:
        # default: yesterday's folder in raw/
        yesterday = (datetime.datetime.utcnow() - datetime.timedelta(days=1)).strftime("%Y-%m-%d")
        args.date_folder = f"raw/{yesterday}"

    print(f"Listing {REPO_ID}/{args.date_folder} (recursive=False)...", file=sys.stderr)
    files = list_date_folder(args.date_folder, recursive=False)
    if not files:
        print(f"WARNING: no files found under {args.date_folder}", file=sys.stderr)

    os.makedirs(args.out_dir, exist_ok=True)
    date_tag = datetime.datetime.utcnow().strftime("%Y%m%d-%H%M%S")
    out_path = os.path.join(args.out_dir, f"file-list-{date_tag}.json")
    with open(out_path, "w") as f:
        json.dump(files, f, indent=2)
    print(out_path)

if __name__ == "__main__":
    main()
```

### 2) `lib/cdn_download.py`
```python
import os
import time
import logging
import requests
from pathlib import Path
from typing import Optional

logger = logging.getLogger(__name__)

def cdn_download(url: str, out_path: Path, max_retries: int = 5, backoff: float = 1.0) -> Path:
    """
    Download a public dataset file via CDN (no auth).
    Retries with exponential backoff on 429/5xx.
    """
    out_path = Path(out_path)
    out_path.parent.mkdir(parents=True, exist_ok=True)

    for attempt in range(1, max_retries + 1):
        try:
            with requests.get(url, stream=True, timeout=60) as r:
                r.raise_for_status()
                with open(out_path, "wb") as f:
                    for chunk in r.iter_content(chunk_size=8192):
                        f.write(chunk)
            logger.info(f"CDN downloaded {url} -> {out_path}")
            return out_path
        except requests.HTTPError as e:
            if e.response is not None and e.response.status_code == 429:
                wait = backoff * (2 ** (attempt - 1))
                logger.warning(f"429 on {url}, retry {attempt}/{max_retries} in {wait}s")
                time.sleep(wait)
                continue
            raise
        except (requests.RequestException, OSError) as e:
            if attempt == max_retries:
                raise
            wait = backoff * (2 ** (attempt - 1))
            logger.warning(f"Error downloading {url}: {e}, retry {attempt}/{max_retries} in {wait}s")
            time.sleep(wait)

    raise RuntimeError(f"Failed to download {url} after {max_retries} attempts")
```

### 3) Update `bin/dataset-enrich.sh` (minimal diff)
Add optional file-list mode and CDN-first fetch. Keep existing behavior as fallback.

```bash
#!/usr/bin/env bash
# dataset-enrich.sh
# Usage:
#   ./dataset-enrich.sh                    # existing behavior (HF API)
#   ./dataset-enrich.sh --file-list file-list-20260502-190649.json  # CDN-only mode

set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "${BASH[0]}")" && pwd)"
REPO_ROOT="$(cd "$SCRIPT_DIR/.." && pwd)"
LIB_DIR="$REPO_ROOT/lib"

export PYTHONPATH="$LIB_DIR:$PYTHONPATH"

FILE_LIST=""
if [[ "${1:-}" == "--file-list" ]]; then
  FILE_LIST="${2:-}"
  if [[ ! -f "$FILE_LIST" ]]; then
    echo "ERROR: file list not found: $FILE_LIST" >&2
    exit 1
  fi
  echo "INFO: running in CDN mode with file list: $FILE_LIST"
fi

# Existing worker parameters (SHARD_ID, N_SHARDS) expected from matrix
: "${SHARD_ID:=0}"
: "${N_SHARDS:=16}"

if [[ -n "$FILE_LIST" ]]; then
  # CDN mode: deterministic list, zero HF API calls during ingestion
  python3 "$LIB_DIR/worker_cdn.py" \
    --file-list "$FILE_LIST" \
    --shard-id "$SHARD_ID" \
    --n-shards "$N_SHARDS" \
    --out-dir "batches/public-merged/$(date -u +%Y-%m-%d)"
else
  # Legacy mode: uses HF API (existing behavior)
  python3 "$LIB_DIR/worker.py" \
    --shard-id "$SHARD_ID" \
    --n-shards "$N_SHARDS" \
    --out-dir "batches/public-merged/$(date -u +%Y-%m-%d)"
fi
```
