# Costinel / design

## Final Implementation Plan  
CDN-first, zero-runtime-HF-API Top-Hub Signal Panel (deliverable < 2 h)

---

### 1. Architecture (single source of truth)

| Phase | Action | Runtime cost |
|-------|--------|--------------|
| **Build / CI (Mac/CI only)** | 1) `list_repo_tree` (non-recursive) for `top-hub/`  <br>2) Pick latest `top-hub/{DATE}/panel.json`  <br>3) Fetch via CDN (`resolve/main/...`)  <br>4) Write `public/data/top-hub-latest.json` into repo/Docker image | Zero HF API at runtime |
| **Deploy** | Baked file lands in Docker image at `/app/public/data/top-hub-latest.json` (or served via CDN) | Static asset |
| **Runtime (dashboard)** | `fetch('/data/top-hub-latest.json')` (ISR or no-store) → render. No HF API, no auth, no 429. | Zero HF API |

### 2. Concrete file changes

#### 2.1 Build script (Node, run on Mac/CI)  
`scripts/build-top-hub-panel.js`
```js
#!/usr/bin/env node
/**
 * Build-time: bake top-hub panel from CDN into static asset.
 * Run: HUGGING_FACE_REPO=AXENTX/Costinel-Knowledge node scripts/build-top-hub-panel.js
 */
import { writeFileSync, mkdirSync, existsSync } from 'fs';
import { join, dirname } from 'path';
import { fileURLToPath } from 'url';
import { execSync } from 'child_process';
import fetch from 'node-fetch';

const __dirname = dirname(fileURLToPath(import.meta.url));
const REPO = process.env.HUGGING_FACE_REPO || 'AXENTX/Costinel-Knowledge';
const OUT_DIR = join(__dirname, '..', 'public', 'data');
const OUT_FILE = join(OUT_DIR, 'top-hub-latest.json');
const DATE = new Date().toISOString().slice(0, 10); // YYYY-MM-DD

function safeExec(cmd) {
  try {
    return execSync(cmd, { env: { ...process.env, HF_REPO: REPO }, stdio: 'pipe', encoding: 'utf8' }).trim();
  } catch {
    return null;
  }
}

async function run() {
  try {
    mkdirSync(OUT_DIR, { recursive: true });

    // 1) Try to list top-hub/ to pick latest file; fallback to DATE path
    let latestFile = `top-hub/${DATE}/panel.json`;
    const treeOut = safeExec(
      `python -c "from huggingface_hub import list_repo_tree; import json, os; repo=os.getenv('HF_REPO','${REPO}'); tree=list_repo_tree(repo, path='top-hub', recursive=False); files=[f.rfilename for f in tree if f.rfilename.endswith('.json')]; print(json.dumps(files))"`
    );

    if (treeOut) {
      try {
        const files = JSON.parse(treeOut);
        if (Array.isArray(files) && files.length > 0) {
          files.sort().reverse();
          latestFile = files[0];
        }
      } catch {
        // ignore malformed; keep DATE fallback
      }
    }

    // 2) CDN fetch (no auth)
    const cdnUrl = `https://huggingface.co/datasets/${REPO}/resolve/main/${latestFile}`;
    const res = await fetch(cdnUrl, { headers: { Accept: 'application/json' } });

    let payload;
    if (res.ok) {
      payload = await res.json();
    } else {
      // Fallback stub
      payload = {
        hub: 'MOC',
        score: 0.91,
        summary: 'Top hub (cached fallback)',
        details: [],
        updated: new Date().toISOString(),
        source: 'cdn-fallback'
      };
    }

    const enriched = {
      hub: String(payload.hub || 'MOC'),
      score: Number(payload.score ?? 0),
      summary: String(payload.summary || 'No summary'),
      details: Array.isArray(payload.details) ? payload.details.map(String) : [],
      updated: payload.updated || new Date().toISOString(),
      source: String(payload.source || 'cdn'),
      _path: latestFile
    };

    writeFileSync(OUT_FILE, JSON.stringify(enriched, null, 2));
    console.log('Top-hub panel baked to:', OUT_FILE);
  } catch (err) {
    console.error('Failed to build top-hub panel:', err);
    const fallback = {
      hub: 'MOC',
      score: 0,
      summary: 'Unavailable',
      details: [],
      updated: new Date().toISOString(),
      source: 'error'
    };
    writeFileSync(OUT_FILE, JSON.stringify(fallback, null, 2));
  }
}

run();
```

Make executable and schedule (if desired):
