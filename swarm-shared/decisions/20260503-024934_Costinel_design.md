# Costinel / design

## Final Implementation Plan — Top-Hub Signal Panel (CDN-first, <2h, telemetry-aware, production-ready)

### What we ship
- A **non-blocking Top-Hub Signal Panel** mounted on the Costinel dashboard.
- Defaults to hub **MOC** (configurable via `VITE_HUB_NAME`).
- Shows: hub title, description, connection strength, top 5 signals (title + severity + description + action), and last-updated timestamp.
- **CDN-first data fetch** via public HuggingFace dataset URL (bypasses API rate limits).
- **Telemetry-aware**: respects `window.axentxTelemetry` opt-out; emits minimal, privacy-safe events.
- **Zero backend changes** and **zero runtime API calls during render**; pure frontend addition deployable via existing CI/CD.
- Graceful degradation if CDN fetch fails.

#### 1) `src/components/TopHubSignalPanel.tsx` (new)
```tsx
import { useEffect, useState } from "react";

const HUB_NAME = import.meta.env.VITE_HUB_NAME || "MOC";
const HUB_DATA_URL = `https://huggingface.co/datasets/AXENTX/Costinel/resolve/main/hubs/${HUB_NAME}.json`;

type Signal = {
  id: string;
  title: string;
  severity: "low" | "medium" | "high";
  description: string;
  action?: string;
};

type HubData = {
  hub: string;
  title: string;
  description: string;
  connectionStrength: number;
  signals: Signal[];
  updatedAt: string;
};

function sendTelemetry(event: string, payload?: Record<string, unknown>) {
  if (typeof window === "undefined" || !window.axentxTelemetry) return;
  try {
    window.axentxTelemetry(event, payload);
  } catch {
    // best-effort
  }
}

export default function TopHubSignalPanel() {
  const [data, setData] = useState<HubData | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    let mounted = true;
    setLoading(true);

    fetch(HUB_DATA_URL, { cache: "no-cache" })
      .then((res) => {
        if (!res.ok) throw new Error(`HTTP ${res.status}`);
        return res.json();
      })
      .then((json: HubData) => {
        if (!mounted) return;
        setData(json);
        sendTelemetry("top_hub_loaded", { hub: json.hub, signalCount: json.signals.length });
      })
      .catch((err) => {
        if (!mounted) return;
        setError(err.message);
        sendTelemetry("top_hub_error", { hub: HUB_NAME, error: err.message });
      })
      .finally(() => {
        if (mounted) setLoading(false);
      });

    return () => {
      mounted = false;
    };
  }, []);

  if (loading) {
    return (
      <div className="rounded-lg border bg-card/50 p-4">
        <div className="h-5 w-32 animate-pulse rounded bg-muted" />
        <div className="mt-3 space-y-2">
          {[...Array(3)].map((_, i) => (
            <div key={i} className="h-4 w-full animate-pulse rounded bg-muted/50" />
          ))}
        </div>
      </div>
    );
  }

  if (error || !data) {
    return (
      <div className="rounded-lg border bg-card/50 p-4 text-sm text-muted-foreground">
        Unable to load hub signals. Using local governance rules.
      </div>
    );
  }

  return (
    <div className="rounded-lg border bg-card/50 p-4">
      <div className="flex items-start justify-between">
        <div>
          <h3 className="font-semibold">{data.title}</h3>
          <p className="text-xs text-muted-foreground">{data.description}</p>
        </div>
        <span className="text-xs text-muted-foreground">
          Updated {new Date(data.updatedAt).toLocaleDateString()}
        </span>
      </div>

      <div className="mt-3 flex items-center gap-2">
        <span className="text-xs text-muted-foreground">Connection strength</span>
        <div className="h-1.5 w-20 rounded-full bg-muted">
          <div
            className="h-1.5 rounded-full bg-primary transition-all"
            style={{ width: `${Math.min(100, Math.max(0, data.connectionStrength * 100))}%` }}
          />
        </div>
        <span className="text-xs font-medium">{Math.round(data.connectionStrength * 100)}%</span>
      </div>

      <div className="mt-4 space-y-2">
        {data.signals.slice(0, 5).map((s) => (
          <div key={s.id} className="rounded-md border bg-background/50 p-3 text-sm">
            <div className="flex items-start justify-between gap-2">
              <span className="font-medium">{s.title}</span>
              <span
                className={`shrink-0 rounded px-1 text-[10px] font-semibold uppercase
                  ${s.severity === "high" ? "bg-destructive/10 text-destructive" : ""}
                  ${s.severity === "medium" ? "bg-warning/10 text-warning" : ""}
                  ${s.severity === "low" ? "bg-muted text-muted-foreground" : ""}`}
              >
                {s.severity}
              </span>
            </div>
            <p className="mt-1 text-xs text-muted-foreground">{s.description}</p>
            {s.action && (
              <p className="mt-2 text-xs underline decoration-dotted underline-offset-2">
                {s.action}
              </p>
            )}
          </div>
        ))}
      </div>
    </div>
  );
}
```
