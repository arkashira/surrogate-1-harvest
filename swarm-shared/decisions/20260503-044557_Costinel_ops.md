### 2) Generate hub-index.json (once, on Mac orchestration host)

- Run after rate-limit window clears; commit result so CDN can serve.
- Use `recursive=False` to avoid pagination/timeouts.
- Deterministic sort: `MOC*` first, then alphabetical.

`scripts/generate_hub_index.py`
```python
#!/usr/bin/env python3
import json
import os
import sys
from datetime import datetime, timezone

try:
    from huggingface_hub import list_repo_tree
except ImportError:
    print("Install: pip install huggingface_hub")
    sys.exit(1)

def main():
    repo = os.getenv("HF_REPO", "AXENTX/knowledge-rag")
    path = os.getenv("HF_PATH", "top-hubs")
    out = os.getenv("HF_OUT", "public/hub-index.json")

    tree = list_repo_tree(repo=repo, path=path, recursive=False)
    files = [
        {
            "name": os.path.basename(f.path.rstrip("/")),
            "cdn_url": f"https://huggingface.co/datasets/{repo}/resolve/main/{f.path}"
        }
        for f in tree
        if f.type == "file"
    ]

    # Deterministic priority: MOC first, then alphabetical
    files.sort(key=lambda x: (not x["name"].lower().startswith("moc"), x["name"].lower()))

    os.makedirs(os.path.dirname(out), exist_ok=True)
    payload = {
        "generated_at": datetime.now(timezone.utc).isoformat(),
        "repo": repo,
        "path": path,
        "files": files
    }

    with open(out, "w", encoding="utf-8") as fp:
        json.dump(payload, fp, indent=2, ensure_ascii=False)

    print(f"Wrote {len(files)} files to {out}")

if __name__ == "__main__":
    main()
```

Run (manual/cron):
```bash
chmod +x scripts/generate_hub_index.py
# Manual
HF_REPO=AXENTX/knowledge-rag HF_PATH=top-hubs HF_OUT=public/hub-index.json ./scripts/generate_hub_index.py

# Cron (example)
SHELL=/bin/bash
0 3 * * * cd /opt/axentx/Costinel && HF_REPO=AXENTX/knowledge-rag HF_PATH=top-hubs HF_OUT=public/hub-index.json ./scripts/generate_hub_index.py >> logs/hub-index.log 2>&1
```

### 3) TopHubPanel component (CDN-only, resilient)

`components/TopHubPanel.tsx`
```tsx
import { useEffect, useState } from "react";
import "./TopHubPanel.css";

type HubFile = { name: string; cdn_url: string };
type Manifest = { generated_at: string; repo: string; path: string; files: HubFile[] };

export default function TopHubPanel() {
  const [manifest, setManifest] = useState<Manifest | null>(null);
  const [activeFile, setActiveFile] = useState<HubFile | null>(null);
  const [content, setContent] = useState<string>("");
  const [loading, setLoading] = useState<boolean>(false);
  const [error, setError] = useState<string | null>(null);

  // Load manifest (CDN)
  useEffect(() => {
    fetch("/hub-index.json", { cache: "no-store" })
      .then((r) => {
        if (!r.ok) throw new Error("Manifest unavailable");
        return r.json();
      })
      .then(setManifest)
      .catch((e) => setError(e.message));
  }, []);

  // Pick active file: MOC first, else first
  useEffect(() => {
    if (!manifest?.files?.length) return;
    const preferred = manifest.files.find((f) => f.name.toLowerCase().startsWith("moc")) || manifest.files[0];
    setActiveFile(preferred);
  }, [manifest]);

  // Fetch active file content (CDN)
  useEffect(() => {
    if (!activeFile?.cdn_url) return;
    setLoading(true);
    setError(null);
    fetch(activeFile.cdn_url, { cache: "max-age=3600" })
      .then((r) => {
        if (!r.ok) throw new Error(`Failed to fetch ${activeFile.name}`);
        return r.text();
      })
      .then(setContent)
      .catch((e) => {
        setError(e.message);
        setContent("");
      })
      .finally(() => setLoading(false));
  }, [activeFile]);

  if (error && !manifest) {
    return (
      <div className="top-hub-panel error">
        <strong>Top Hub</strong>
        <p className="muted">Insights unavailable.</p>
      </div>
    );
  }

  return (
    <div className="top-hub-panel">
      <div className="top-hub-header">
        <strong>Top Hub</strong>
        <span className="hub-name">{activeFile?.name.replace(/\.[^/.]+$/, "") || "—"}</span>
      </div>

      <div className="hub-files" role="tablist" aria-label="Hub files">
        {manifest?.files.map((f) => (
          <button
            key={f.name}
            role="tab"
            aria-selected={activeFile?.name === f.name}
            className={`hub-file ${activeFile?.name === f.name ? "active" : ""}`}
            onClick={() => setActiveFile(f)}
            title={f.name}
          >
            {f.name.replace(/\.[^/.]+$/, "")}
          </button>
        ))}
      </div>

      <div className="hub-content">
        {loading && <p className="muted">Loading insights…</p>}
        {error && <p className="muted error-msg">{error}</p>}
        {!loading && !error && <pre className="hub-text">{content || "No content"}</pre>}
      </div>

      <div className="hub-footer">
        <small className="muted">
          Updated {manifest?.generated_at ? new Date(manifest.generated_at).toLocaleDateString() : "—"}
        </small>
      </div>
    </div>
  );
}
```
