## Code Snippets

### 1. `snapshot_manifest.py` (Mac orchestration)

```python
#!/usr/bin/env python3
"""
Generate deterministic file manifest for a date-partition
of axentx/surrogate-1-training-pairs to enable CDN-only fetches.

Usage:
    python snapshot_manifest.py --date 2026-04-29 --out file_manifest.json
"""

import argparse
import json
import os
import sys
from datetime import datetime

from huggingface_hub import HfApi

REPO_ID = "axentx/surrogate-1-training-pairs"

def main() -> None:
    parser = argparse.ArgumentParser(description="Create CDN file manifest")
    parser.add_argument("--date", required=True, help="Date partition (YYYY-MM-DD)")
    parser.add_argument("--out", default="file_manifest.json", help="Output JSON path")
    parser.add_argument("--repo", default=REPO_ID, help="HF dataset repo")
    parser.add_argument("--token", default=os.getenv("HF_TOKEN"), help="HF token (optional for public reads)")
    args = parser.parse_args()

    api = HfApi(token=args.token)

    # List top-level objects in the date folder (non-recursive)
    # Avoids recursive pagination and heavy list_repo_files calls
    prefix = f"{args.date}/"
    try:
        items = api.list_repo_tree(
            repo_id=args.repo,
            path=prefix,
            recursive=False,
        )
    except Exception as exc:
        print(f"ERROR listing repo tree at {prefix!r}: {exc}", file=sys.stderr)
        sys.exit(1)

    # Keep only file-like entries with useful extensions
    allowed_ext = {".parquet", ".jsonl", ".json", ".csv", ".tsv"}
    files = sorted(
        item.rfilename for item in items
        if not item.rfilename.endswith("/") and any(item.rfilename.endswith(ext) for ext in allowed_ext)
    )

    if not files:
        print(f"WARN: no files found under {prefix!r}", file=sys.stderr)

    manifest = {
        "repo": args.repo,
        "date": args.date,
        "created_at": datetime.utcnow().isoformat() + "Z",
        "files": files,
        "count": len(files),
    }

    out_path = os.path.abspath(args.out)
    os.makedirs(os.path.dirname(out_path) or ".", exist_ok=True)
    with open(out_path, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2, sort_keys=True)

    print(f"Manifest written to {out_path} ({len(files)} files)")

if __name__ == "__main__":
    main()
```

Make executable:

```bash
chmod +x snapshot_manifest.py
```

### 2. Training loader patch (CDN-only)

Add to your Lightning training script (e.g., `train.py`):

```python
import json
import os
from pathlib import Path
from typing import List

from datasets import load_dataset
from huggingface_hub import HfApi

HF_REPO = "axentx/surrogate-1-training-pairs"

def load_file_manifest(manifest_path: str) -> List[str]:
    with open(manifest_path, "r", encoding="utf-8") as f:
        manifest = json.load(f)
    return manifest["files"]

def build_cdn_urls(files: List[str], repo: str = HF_REPO) -> List[str]:
    return [f"https://huggingface.co/datasets/{repo}/resolve/main/{f}" for f in files]

def get_train_dataset(manifest_path: str = None, date: str = None):
    """
    Load dataset using CDN URLs from manifest.
    If manifest missing and date provided, optionally generate via HF API once.
    """
    if manifest_path and Path(manifest_path).exists():
        files = load_file_manifest(manifest_path)
    elif date:
        # One-time fallback: list via API (use sparingly)
        api = HfApi()
        prefix = f"{date}/"
        items = api.list_repo_tree(HF_REPO, path=prefix, recursive=False)
        files = sorted(
            item.rfilename for item in items
            if not item.rfilename.endswith("/") and item.rfilename.endswith(".parquet")
        )
    else:
        raise ValueError("Provide manifest_path or date")

    urls = build_cdn_urls(files)
    # Use data_files to point to remote URLs; datasets will stream via CDN
    ds = load_dataset("parquet", data_files={"train": urls}, streaming=False)
    return ds["train"]
```

Usage in Lightning `LightningDataModule` or training script:

```python
manifest = "data/file_manifest.json"
train_data = get_train_dataset(manifest_path=manifest)
```

### 3. GitHub Actions snippet (optional cache)

```yaml
- name: Generate manifest (if missing)
  run: |
    if [ ! -f data/file_manifest.json ]; then
      python snapshot_manifest.py --date ${{ env.DATE }} --out data/file_manifest.json
    fi
  env:
    HF_TOKEN: ${{ secrets.HF_TOKEN }}
    DATE: "2026-04-29"

- name: Cache manifest
  uses: actions/cache@v4
  with:
    path: data/file_manifest.json
    key: manifest-${{ env.DATE }}
```
