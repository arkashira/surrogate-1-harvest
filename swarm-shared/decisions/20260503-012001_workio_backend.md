# workio / backend

## Final Synthesis (Correct + Actionable)

**Scope**: `workio/server/src/routes/webhook/line.ts`, `workio/server/src/services/punch.service.ts`, and DB (`schema.sql`).  
**Goal**: enforce at-most-one-open-punch and idempotent LINE webhook handling in <2h with production-grade correctness.

---

### 1. Diagnosis (resolved)

- **Idempotency gap on LINE webhook** → duplicate deliveries (LINE retries) can create duplicate punches/leave/OT entries and double notifications.
- **Race on open punch** (`find-then-insert/update`) → multiple concurrent requests can create multiple open punches per employee.
- **No transactional boundary for punch + notification** → partial failure can persist a punch without notification (or vice-versa).
- **Missing optimistic locking on leave/OT** → lost updates when managers concurrently approve.
- **No request deduplication log** → no cheap way to detect/replay of LINE events or client double-taps.

### 2. DB changes (`schema.sql`)

```sql
-- Optimistic locking for leave/OT
ALTER TABLE leave_requests ADD COLUMN IF NOT EXISTS version INTEGER NOT NULL DEFAULT 1;
ALTER TABLE ot_requests  ADD COLUMN IF NOT EXISTS version INTEGER NOT NULL DEFAULT 1;

-- Idempotency key for punches (tenant-scoped)
ALTER TABLE punches ADD COLUMN IF NOT EXISTS idempotency_key UUID;
CREATE UNIQUE INDEX IF NOT EXISTS uq_punches_tenant_idempotency
  ON punches (tenant_id, idempotency_key)
  WHERE idempotency_key IS NOT NULL;

-- Partial unique constraint: at-most-one-open-punch per employee
CREATE UNIQUE INDEX IF NOT EXISTS uq_punches_open_per_employee
  ON punches (employee_id, tenant_id)
  WHERE status = 'IN';

-- Covering index for fast open-punch lookup
CREATE INDEX IF NOT EXISTS ix_punches_open_lookup
  ON punches (employee_id, tenant_id, status);
```

### 3. Punch service (`punch.service.ts`)

```ts
import { Pool, PoolClient } from 'pg';

export class PunchService {
  constructor(private pool: Pool) {}

  async createPunch(payload: {
    tenantId: string;
    employeeId: string;
    type: 'IN' | 'OUT';
    timestamp: Date;
    location?: { lat: number; lng: number };
    idempotencyKey: string;
  }) {
    const client = await this.pool.connect();
    try {
      await client.query('BEGIN');

      // 1) Idempotent insert: if same idempotency_key exists, return existing row
      const insertText = `
        INSERT INTO punches (tenant_id, employee_id, type, timestamp, location, status, idempotency_key)
        VALUES ($1, $2, $3, $4, $5, $6, $7)
        ON CONFLICT (tenant_id, idempotency_key) DO UPDATE
          SET updated_at = NOW()
        RETURNING *;
      `;
      const insertRes = await client.query(insertText, [
        payload.tenantId,
        payload.employeeId,
        payload.type,
        payload.timestamp,
        payload.location ? JSON.stringify(payload.location) : null,
        payload.type, // status mirrors type (IN/OUT)
        payload.idempotencyKey,
      ]);

      // 2) If OUT, close the current open punch (safe: partial unique prevents double-IN)
      if (payload.type === 'OUT') {
        const closeText = `
          UPDATE punches
          SET type = 'OUT', status = 'OUT', timestamp = $1, updated_at = NOW()
          WHERE employee_id = $2 AND tenant_id = $3 AND status = 'IN'
          RETURNING *;
        `;
        const closeRes = await client.query(closeText, [
          payload.timestamp,
          payload.employeeId,
          payload.tenantId,
        ]);
        await client.query('COMMIT');
        return { punch: closeRes.rows[0] || insertRes.rows[0], created: closeRes.rowCount > 0 };
      }

      await client.query('COMMIT');
      return { punch: insertRes.rows[0], created: true };
    } catch (err: any) {
      await client.query('ROLLBACK');

      // If partial unique violation (duplicate open punch), fetch existing open punch
      if (err.code === '23505' && err.constraint === 'uq_punches_open_per_employee') {
        const existing = await client.query(
          'SELECT * FROM punches WHERE employee_id = $1 AND tenant_id = $2 AND status = $3',
          [payload.employeeId, payload.tenantId, 'IN']
        );
        return { punch: existing.rows[0], created: false };
      }
      throw err;
    } finally {
      client.release();
    }
  }

  // Optional: transactional update for leave/OT with optimistic locking
  async updateLeaveRequest(id: string, tenantId: string, fields: any, expectedVersion: number) {
    const client = await this.pool.connect();
    try {
      await client.query('BEGIN');
      const updateText = `
        UPDATE leave_requests
        SET ${Object.keys(fields)
          .map((k, i) => `${k} = $${i + 1}`)
          .join(', ')}, version = version + 1
        WHERE id = $${Object.keys(fields).length + 1} AND tenant_id = $${
        Object.keys(fields).length + 2
      } AND version = $${Object.keys(fields).length + 3}
        RETURNING *;
      `;
      const values = [...Object.values(fields), id, tenantId, expectedVersion];
      const res = await client.query(updateText, values);
      if (res.rowCount === 0) {
        await client.query('ROLLBACK');
        throw new Error('Concurrent modification detected (version mismatch)');
      }
      await client.query('COMMIT');
      return res.rows[0];
    } catch (err) {
      await client.query('ROLLBACK');
      throw err;
    } finally {
      client.release();
    }
  }
}
```
