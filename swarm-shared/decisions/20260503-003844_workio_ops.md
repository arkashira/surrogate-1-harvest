# workio / ops

## 1. Diagnosis

- No idempotency guard on `/webhook/line` allows LINE retries (network blips, 5xx, slow client) to create duplicate punches for the same user+day.
- Missing DB-level constraint enforcing “at most one open punch per user” (`clock_out_at IS NULL`) enables race conditions under concurrency.
- No de-duplication key in webhook handler; LINE `webhookId` or message `id` is not used to prevent re-processing.
- Clock-in/out writes do not run in serializable or advisory-lock scope, so concurrent requests can both see “no open punch” and insert two rows.
- Observability gap: no structured logging of idempotency decisions and no alerting on duplicate punch attempts.

## 2. Proposed change

- **File**: `workio/server/src/routes/webhook/line.ts` (or equivalent route handler for LINE webhook).
- **Scope**: add idempotency check using LINE `webhookId` (or message `id`) + DB unique constraint/index on `user_id, date, clock_out_at` (partial index where `clock_out_at IS NULL`) to enforce single open punch.
- **Add**: lightweight idempotency table or Redis key for webhook-level dedup; fallback to DB unique constraint for punch state.

## 3. Implementation

```sql
-- workio/server/src/db/schema.sql
-- Ensure one open punch per user (partial unique index)
CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_one_open_per_user
ON punches (user_id)
WHERE clock_out_at IS NULL;

-- Idempotency table for webhook events (fast, small)
CREATE TABLE IF NOT EXISTS webhook_idempotency (
  idempotency_key TEXT PRIMARY KEY,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
```

```typescript
// workio/server/src/routes/webhook/line.ts
import express from 'express';
import { db } from '../db';
import { logger } from '../lib/logger';

const router = express.Router();

async function recordIdempotency(key: string): Promise<boolean> {
  try {
    await db.query(
      'INSERT INTO webhook_idempotency (idempotency_key) VALUES ($1) ON CONFLICT DO NOTHING',
      [key]
    );
    // If row count === 1, this is first time seeing key
    return true;
  } catch {
    return false;
  }
}

router.post('/line', async (req, res) => {
  const body = req.body;
  const events = body?.events;
  if (!Array.isArray(events) || events.length === 0) {
    return res.status(400).json({ error: 'invalid payload' });
  }

  // Use LINE webhookId if provided, else fallback to first message id
  const webhookId = body.webhookId;
  const idempotencyKey = webhookId
    ? `line:webhook:${webhookId}`
    : `line:msg:${events[0].source?.userId}:${events[0].message?.id}`;

  const isNew = await recordIdempotency(idempotencyKey);
  if (!isNew) {
    logger.info({ idempotencyKey }, 'duplicate webhook skipped');
    return res.status(200).json({ ok: true });
  }

  for (const event of events) {
    if (event.type !== 'message' || !event.message?.text) continue;
    const userId = event.source?.userId;
    const text = event.message.text.trim().toLowerCase();

    if (!userId) continue;

    try {
      if (text === 'clock in' || text === 'เข้างาน') {
        // Rely on partial unique index to prevent two open punches
        await db.query(
          `INSERT INTO punches (user_id, clock_in_at, clock_out_at, location)
           VALUES ($1, NOW(), NULL, $2)`,
          [userId, event.message.text] // replace with GPS if available
        );
        logger.info({ userId }, 'clock in recorded');
      } else if (text === 'clock out' || text === 'เลิกงาน') {
        // Atomically close the open punch
        const result = await db.query(
          `UPDATE punches
           SET clock_out_at = NOW()
           WHERE user_id = $1 AND clock_out_at IS NULL
           RETURNING id`,
          [userId]
        );
        if (result.rowCount === 0) {
          logger.warn({ userId }, 'no open punch to clock out');
        } else {
          logger.info({ userId }, 'clock out recorded');
        }
      }
    } catch (err: any) {
      // Unique violation from partial index => duplicate open punch
      if (err.code === '23505') {
        logger.warn({ userId, err: err.message }, 'duplicate open punch prevented');
      } else {
        logger.error({ userId, err }, 'punch handling failed');
      }
    }
  }

  return res.status(200).json({ ok: true });
});

export default router;
```
