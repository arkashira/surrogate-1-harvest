### 1) CDN-bypass dataset loader
`surrogate/training/cdn_dataset.py`
```python
"""
HF CDN-bypass dataset loader.
- Avoids HF API (429) and pyarrow schema issues.
- Downloads via public CDN URLs (no auth).
- Projects to {prompt, response} only.

Usage:
    filelist = ["2026-04-29/file1.parquet", ...]  # from list_repo_tree JSON
    ds = CDNParquetDataset(filelist, repo="myorg/surrogate-data")
    for row in ds:
        prompt, response = row["prompt"], row["response"]
"""

from __future__ import annotations

import json
import tempfile
from concurrent.futures import ThreadPoolExecutor, as_completed
from pathlib import Path
from typing import Dict, Iterable, List, Optional

import pyarrow.parquet as pq
import requests
from tqdm import tqdm

CDN_ROOT = "https://huggingface.co/datasets"


class CDNParquetDataset:
    """Iterable over selected columns from parquet files via CDN (no HF API)."""

    def __init__(
        self,
        file_paths: List[str],
        repo: str,
        columns: Iterable[str] = ("prompt", "response"),
        max_workers: int = 8,
        cache_dir: Optional[str] = None,
    ) -> None:
        self.repo = repo.rstrip("/")
        self.file_paths = [p.lstrip("/") for p in file_paths]
        self.columns = list(columns)
        self.max_workers = max_workers
        self.cache_dir = Path(cache_dir or tempfile.mkdtemp(prefix="cdn_cache_"))
        self.cache_dir.mkdir(parents=True, exist_ok=True)

    @staticmethod
    def _cdn_url(repo: str, path: str) -> str:
        return f"{CDN_ROOT}/{repo}/resolve/main/{path}"

    def _download_one(self, path: str) -> Path:
        out = self.cache_dir / path.replace("/", "_")
        if out.exists():
            return out
        url = self._cdn_url(self.repo, path)
        out.parent.mkdir(parents=True, exist_ok=True)
        resp = requests.get(url, timeout=60)
        resp.raise_for_status()
        out.write_bytes(resp.content)
        return out

    def _load_parquet(self, path: Path):
        try:
            table = pq.read_table(path, columns=self.columns)
        except Exception as exc:
            raise RuntimeError(f"Failed to read {path}: {exc}") from exc
        return table.to_pylist()

    def __iter__(self):
        # Download batch
        with ThreadPoolExecutor(max_workers=self.max_workers) as ex:
            future_to_path = {ex.submit(self._download_one, p): p for p in self.file_paths}
            for future in tqdm(as_completed(future_to_path), total=len(future_to_path), desc="Downloading via CDN"):
                path = future_to_path[future]
                try:
                    local_path = future.result()
                except Exception as exc:
                    print(f"Download failed for {path}: {exc}")
                    continue

                try:
                    rows = self._load_parquet(local_path)
                except Exception as exc:
                    print(f"Parse failed for {local_path}: {exc}")
                    continue

                for row in rows:
                    # Ensure required fields exist; missing -> None
                    yield {c: row.get(c) for c in self.columns}

    def to_list(self) -> List[Dict[str, str]]:
        return list(self.__iter__())


def load_file_list(json_path: str) -> List[str]:
    """Load pre-listed file paths JSON produced by Mac-side list_repo_tree."""
    with open(json_path) as f:
        data = json.load(f)
    if isinstance(data, dict) and "files" in data:
        return [f["path"] if isinstance(f, dict) else f for f in data["files"]]
    if isinstance(data, list):
        return [f["path"] if isinstance(f, dict) else f for f in data]
    raise ValueError(f"Unexpected file list format in {json_path}")
```

### 2) Lightning Studio reuse helper
`surrogate/training/lightning_utils.py`
```python
"""
Lightning Studio reuse utilities to avoid quota burn.
"""

from __future__ import annotations

import time
from typing import Optional

try:
    from lightning_sdk import Studio, Teamspace, Machine
except ImportError:
    Studio = Teamspace = Machine = None  # type: ignore


def get_or_create_studio(
    name: str,
    machine: str = "L40S",
    cloud: str = "lightning-public-prod",
    idle_timeout_min: int = 30,
) -> Optional[Studio]:
    """
    Reuse a running studio if exists; otherwise create one.
    Returns None if Lightning SDK unavailable or quota exhausted.
    """
    if Studio is None:
        print("Lightning SDK not installed. Skipping studio reuse.")
        return None

    try:
        teamspace = Teamspace(name=".")  # default/current teamspace
        for s in teamspace.studios:
            if s.name == name and s.status == "Running":
                print(f"Reusing running studio: {name}")
                return s

        print(f"No running studio '{name}' found. Creating...")
        # Map machine names safely
        try:
            machine_obj = Machine(machine, cloud=cloud)
        except Exception:
            # Fallback to default
            machine_obj = Machine("L40S", cloud="lightning-public-prod")

        studio = Studio(
            name=name,
            machine=machine_obj,
            create_ok=True,
            idle_timeout_min=idle_timeout_min,
        )
        print(f"Created studio: {name} on {machine} ({cloud})")
        return studio
    except Exception as exc:
        print(f"Studio reuse/create failed: {exc}")
        return None


def ensure_studio_running(studio: Studio, machine: str = "L40S", cloud: str = "lightning-public-prod") -> bool:
    """
    Check studio status; restart if stopped (idle-timeout kills training).
    Returns True if running/usable.
    """
    if studio is None:
        return False
    try:
        studio.refresh()
        if studio.status != "Running":
            print(f"Studio stopped ({studio.status}). Restarting...")
            studio.start(machine=Machine(machine, cloud=cloud))
            # Poll briefly
            for _ in range(10):
                time.sleep(6)
                studio.refresh()
                if studio.status == "Running":
                    return True
            return False
        return True
    except Exception as exc:
        print(f"Failed to ensure studio running: {exc}")
        return False
```
