# Costinel / design

## Implementation Plan — Top-Hub Signal Panel (Costinel Dashboard)

**Scope & Value**  
- Frontend-only, read-only React panel that surfaces the most-connected hub (default “MOC”) and its actionable proposals from the knowledge graph.  
- CDN-first data delivery to eliminate HF API rate limits during dashboard loads.  
- Ships in <2h: static JSON file + React component + route + tests.

### 3) Types (src/types/knowledgeGraph.ts)

```ts
export interface SignalImpact {
  estimated_monthly_savings_usd: number;
  confidence: number;
}

export interface SignalAction {
  label: string;
  href: string;
}

export interface SignalProposal {
  id: string;
  title: string;
  type: 'reserved_instance' | 'anomaly' | 'rightsize' | 'policy' | string;
  impact: SignalImpact;
  actions: SignalAction[];
  tags: string[];
}

export interface TopHubSignal {
  hub: string;
  title: string;
  summary: string;
  proposals: SignalProposal[];
  generated_at: string;
  ttl_minutes: number;
}
```

### 4) Component (src/components/TopHubSignalPanel/TopHubSignalPanel.tsx)

```tsx
import React, { useEffect, useState } from 'react';
import { TopHubSignal } from '../../types/knowledgeGraph';
import './TopHubSignalPanel.css';

const CDN_URL = '/data/top-hub-signal.json';

const TopHubSignalPanel: React.FC = () => {
  const [signal, setSignal] = useState<TopHubSignal | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    const fetchSignal = async () => {
      try {
        const res = await fetch(CDN_URL, { cache: 'no-cache' });
        if (!res.ok) throw new Error(`Failed to load top-hub signal: ${res.status}`);
        const data: TopHubSignal = await res.json();
        setSignal(data);
      } catch (err) {
        setError(err instanceof Error ? err.message : String(err));
      } finally {
        setLoading(false);
      }
    };

    fetchSignal();
  }, []);

  if (loading) return <div className="top-hub-panel loading">Loading signals…</div>;
  if (error) return <div className="top-hub-panel error">{error}</div>;
  if (!signal) return null;

  return (
    <section className="top-hub-panel" aria-label={`Top hub: ${signal.hub}`}>
      <header className="top-hub-header">
        <h2 className="top-hub-title">{signal.title}</h2>
        <p className="top-hub-summary">{signal.summary}</p>
      </header>

      <div className="top-hub-proposals" role="list">
        {signal.proposals.map((p) => (
          <article key={p.id} className="proposal-card" role="listitem">
            <div className="proposal-header">
              <span className="proposal-type">{p.type}</span>
              <span className="proposal-confidence">{Math.round(p.impact.confidence * 100)}%</span>
            </div>
            <h3 className="proposal-title">{p.title}</h3>
            <p className="proposal-savings">
              Save ~${p.impact.estimated_monthly_savings_usd.toLocaleString()}/mo
            </p>
            <div className="proposal-tags">
              {p.tags.map((t) => (
                <span key={t} className="tag">
                  {t}
                </span>
              ))}
            </div>
            <div className="proposal-actions">
              {p.actions.map((a) => (
                <a key={a.href} href={a.href} className="proposal-action">
                  {a.label}
                </a>
              ))}
            </div>
          </article>
        ))}
      </div>

      <footer className="top-hub-footer">
        <small>
          Generated {new Date(signal.generated_at).toLocaleString()} — TTL {signal.ttl_minutes}m
        </small>
      </footer>
    </section>
  );
};

export default TopHubSignalPanel;
```
