## 1) CDN-only manifest generator (Mac orchestration)
- **One-time per date-folder**: non-recursive `list_repo_tree` for `batches/mirror-merged/{date}`.  
- **Output**: `manifests/{date}.json` committed to repo (or kept local) with:
  - `repo`, `date`, `cdn_prefix`, `files[]`
- **Why**: Removes all HF API calls during dataload → zero 429 risk.

```python
# scripts/gen_cdn_manifest.py
#!/usr/bin/env python3
import json, os, argparse
from datetime import datetime
from huggingface_hub import HfApi

def main(date_folder: str, repo: str, out_dir: str = "manifests"):
    api = HfApi()
    tree = api.list_repo_tree(repo_id=repo, path=f"batches/mirror-merged/{date_folder}", recursive=False)
    files = sorted(item.rfilename for item in tree if item.rfilename.endswith(".parquet"))
    manifest = {
        "date": date_folder,
        "repo": repo,
        "cdn_prefix": f"https://huggingface.co/datasets/{repo}/resolve/main/batches/mirror-merged/{date_folder}/",
        "files": files
    }
    os.makedirs(out_dir, exist_ok=True)
    out_path = os.path.join(out_dir, f"{date_folder}.json")
    with open(out_path, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Manifest written to {out_path}")

if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--date", default=datetime.utcnow().strftime("%Y-%m-%d"))
    parser.add_argument("--repo", default="axentx/surrogate-mirror")
    parser.add_argument("--out-dir", default="manifests")
    args = parser.parse_args()
    main(args.date, args.repo, args.out_dir)
```

## 2) CDN-only dataset loader (Lightning training)
- Replace `load_dataset(streaming=True)` with `IterableDataset`.
- Fetch parquet via CDN URL, project to `{prompt, response}` only.
- **Retry with exponential backoff** for CDN fetches (correctness + robustness).
- **No HF API calls during iteration**.

```python
# surrogate/train.py (excerpt)
import io, json, time, random, requests, pyarrow.parquet as pq
from torch.utils.data import IterableDataset

class CDNParquetDataset(IterableDataset):
    def __init__(self, manifest_path, project_columns=("prompt", "response"), max_retries=5):
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.prefix = self.manifest["cdn_prefix"]
        self.files = self.manifest["files"]
        self.project_columns = project_columns
        self.max_retries = max_retries

    def _stream_parquet(self, url):
        for attempt in range(1, self.max_retries + 1):
            try:
                resp = requests.get(url, timeout=60)
                resp.raise_for_status()
                table = pq.read_table(io.BytesIO(resp.content), columns=self.project_columns)
                return table.to_pylist()
            except Exception as e:
                if attempt == self.max_retries:
                    raise
                sleep_sec = (2 ** attempt) + random.uniform(0, 1)
                print(f"CDN fetch failed (attempt {attempt}): {e}. Retry in {sleep_sec:.1f}s")
                time.sleep(sleep_sec)

    def __iter__(self):
        for fname in self.files:
            url = f"{self.prefix}{fname}"
            try:
                rows = self._stream_parquet(url)
                for row in rows:
                    yield row
            except Exception as e:
                print(f"Failed {url}: {e}")
                continue
```

## 3) Lightning Studio reuse + idle-stop guard
- **Resolve contradiction**: Candidate 1 used `Lightning.Studio(...).run()`; Candidate 2 used `Teamspace.studios` + `target.start(...)`.  
- **Correct + actionable**: Use `Teamspace` to find by name; if running, reuse; if stopped, restart with `Machine.L40S`. Keep small lock file to avoid concurrent runs.

```python
# surrogate/lightning_launcher.py
import os, time, fcntl
from lightning import Lightning, Teamspace, Machine

LOCK_PATH = os.path.expanduser("~/.lightning_surrogate_training.lock")

def get_or_create_studio(name: str):
    ts = Teamspace()
    for s in ts.studios:
        if s.name == name:
            if s.status == "Running":
                print(f"Reusing running studio: {name}")
                return s
            print(f"Studio {name} is stopped; restarting...")
            s.start(machine=Machine.L40S)
            while s.status != "Running":
                time.sleep(10)
                s.refresh()
            return s
    print(f"Creating new studio: {name}")
    return Lightning.Studio(name=name, machine=Machine.L40S, create_ok=True)

def run_training_with_guard(studio_name="surrogate-train", script="train.py", max_retries=3):
    # simple file lock to prevent concurrent runs
    with open(LOCK_PATH, "w") as lock_f:
        try:
            fcntl.flock(lock_f, fcntl.LOCK_EX | fcntl.LOCK_NB)
        except BlockingIOError:
            print("Another training run is active. Exiting.")
            return

        studio = get_or_create_studio(studio_name)
        for attempt in range(1, max_retries + 1):
            try:
                run = studio.run(script, arguments=[], wait=False)
                print(f"Started run {run.id}")
                return run
            except Exception as e:
                print(f"Run attempt {attempt} failed: {e}")
                if attempt == max_retries:
                    raise
                time.sleep(30)
                studio.refresh()
                if studio.status != "Running":
                    studio.start(machine=Machine.L40S)
```

## 4) Wrapper script with bash hygiene
- Shebang, `set -euo pipefail`, executable bit, cron-safe.

```bash
#!/usr/bin/env bash
# scripts/run_surrogate_training.sh
set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
cd "${SCRIPT_DIR}/.."

# Generate manifest if missing (Mac side)
MANIFEST="manifests/$(date -u +%Y-%m-%d).json"
if [[ ! -f "${MANIFEST}" ]]; then
  python3 scripts/gen_cdn_manifest.py
fi

# Launch Lightning training with studio reuse
python3 surrogate/lightning_launcher.py
```
