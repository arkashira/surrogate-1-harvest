# Costinel / design

## Costinel — Design Increment (≤2h)

**Highest-value incremental improvement:**  
Ship a deterministic discovery module that produces machine-readable manifests (`manifests/sources.json`, `manifests/discovery.json`) and a top-hub insight file (`insights/top-hub.md`) so onboarding, training, and governance scripts can run without manual wiring.

This enables:
- Reproducible environment bootstraps (CI, local, training)
- Governance scripts to validate sources before collection
- Knowledge-rag pipelines to reference the top hub (MOC-style) for context
- Zero-touch onboarding for new cloud accounts

---

## Implementation Plan

1. **Create deterministic schema** for sources and discovery outputs (typed, versioned).
2. **Implement discovery CLI** (`scripts/discover.py`) that:
   - Reads environment + `.env` for cloud cred hints
   - Detects available connectors (aws, gcp, azure)
   - Emits normalized `manifests/sources.json`
   - Emits `manifests/discovery.json` with inferred resources, owners, tags, collection cadence
   - Exits 0 if valid; non-zero + stderr if missing required inputs
3. **Add top-hub insight generator** (`scripts/top_hub.py`) that:
   - Reads `manifests/discovery.json`
   - Computes highest-connected entity (by tag/owner/count)
   - Writes `insights/top-hub.md` (human + frontmatter)
4. **Add lightweight validation** (`scripts/validate_manifests.py`) for CI.
5. **Wire into Makefile** targets: `make discover`, `make validate`, `make insights`.

Estimated time: ~90 minutes.

### 2) `scripts/discover.py`

```python
#!/usr/bin/env python3
"""
Deterministic discovery for Costinel.
Produces:
  - manifests/sources.json
  - manifests/discovery.json
"""
import json
import os
import sys
import hashlib
from datetime import datetime, timezone
from pathlib import Path
from typing import Dict, List, Any

MANIFESTS_DIR = Path(__file__).parent.parent / "manifests"
INSIGHTS_DIR = Path(__file__).parent.parent / "insights"
MANIFESTS_DIR.mkdir(exist_ok=True)
INSIGHTS_DIR.mkdir(exist_ok=True)

VERSION = "1.0.0"

def slug(text: str) -> str:
    return text.lower().replace(" ", "-").replace("_", "-")

def detect_aws() -> Dict[str, Any]:
    access = os.getenv("AWS_ACCESS_KEY_ID") or os.getenv("AWS_ACCESS_KEY")
    if not access:
        return {"available": False}
    return {
        "available": True,
        "provider": "aws",
        "auth_method": "keys",
        "regions": [r.strip() for r in os.getenv("AWS_REGIONS", "us-east-1,eu-west-1").split(",")],
        "accounts": [os.getenv("AWS_ACCOUNT_ID", "unknown")],
        "collection": {
            "cadence": os.getenv("AWS_CADENCE", "daily"),
            "method": "cost-explorer+usage"
        }
    }

def detect_gcp() -> Dict[str, Any]:
    cred = os.getenv("GOOGLE_APPLICATION_CREDENTIALS") or os.getenv("GCP_CREDENTIALS_JSON")
    if not cred and not os.path.exists(os.path.expanduser("~/.config/gcloud/application_default_credentials.json")):
        return {"available": False}
    return {
        "available": True,
        "provider": "gcp",
        "auth_method": "service-account" if cred else "adc",
        "projects": [os.getenv("GCP_PROJECT", "unknown")],
        "collection": {
            "cadence": os.getenv("GCP_CADENCE", "daily"),
            "method": "billing-export"
        }
    }

def detect_azure() -> Dict[str, Any]:
    client = os.getenv("AZURE_CLIENT_ID")
    tenant = os.getenv("AZURE_TENANT_ID")
    if not client or not tenant:
        return {"available": False}
    return {
        "available": True,
        "provider": "azure",
        "auth_method": "sp",
        "subscriptions": [os.getenv("AZURE_SUBSCRIPTION_ID", "unknown")],
        "collection": {
            "cadence": os.getenv("AZURE_CADENCE", "daily"),
            "method": "cost-management-api"
        }
    }

def build_sources(aws: Dict[str, Any], gcp: Dict[str, Any], azure: Dict[str, Any]) -> Dict[str, Any]:
    sources = []
    for det in (aws, gcp, azure):
        if not det.get("available"):
            continue
        p = det["provider"]
        source = {
            "id": f"src-{p}-{slug(det.get('accounts', det.get('projects', det.get('subscriptions', ['unknown'])))[0])}",
            "provider": p,
            "auth": {
                "method": det["auth_method"],
                "env_hint": True
            },
            "scope": {
                "accounts": det.get("accounts", []),
                "projects": det.get("projects", []),
                "subscriptions": det.get("subscriptions", []),
                "regions": det.get("regions", [])
            },
            "collection": det["collection"],
            "tags": {
                "managed_by": "costinel",
                "discovered_at": datetime.now(timezone.utc).isoformat()
            }
        }
        sources.append(source)
    return {
        "version": VERSION,
        "generated_at": datetime.now(timezone.utc).isoformat(),
        "sources": sources
    }

def build_discovery(sources_doc: Dict[str, Any]) -> Dict[str, Any]:
    discovered = []
    for src in sources_doc["sources"]:
        p = src["provider"]
        scope = src["scope"]
        accounts = scope.get("accounts", scope.get("projects", scope.get("subscriptions", ["unknown"])))
        regions = scope.get("regions", ["global"])
        for acc in accounts:
            for r in regions:
                discovered.append({
                    "resource_type": "cloud-account",
                    "provider": p,
                    "id": f"{p}://{acc}/{r}",
                    "account": acc,
                    "region": r,
                    "owner": os.getenv("OWNER", "platform"),
                    "tags": {
                        "source_id": src["id"],
                        "cadence": src["collection"]["cadence"]
                    },
                    "collection": src["collection"]
                })

    # Add synthetic services inferred from providers
    services = []
    seen = set()
    for d in discovered:
        svc_key = (d["provider"], d["resource_type"])
        if svc_key in seen:
            continue
        seen.add(svc_key)
        services.append({
            "service": f"{d['provider']}-{d['resource_type']}",
            "provider": d["provider"],
            "resource_type": d["resource_type"],
            "count": sum(1 for x in discovered if x["provider"] == d["provider"] and x["resource_type"] == d["resource_type"])
        })

    return {
        "version": VERSION,
        "generated_at": datetime.now(timezone.utc).isoformat(),
        "discovered": discovered,
        "services": services,
        "summary": {
            "total_accounts": len(set(d["account"] for d in discovered)),
            "total_regions": len(set(d["region"] for d in discovered if d["region"] != "global")),
            "providers": list(set(d["provider"] for d in discovered))
        }
    }

def write_json(path: Path, data: Dict[str, Any]) -> None:
    with open(path, "w", encoding="utf-8") as f:
        json.dump(data, f, indent=2, ensure_ascii=False)
        f.write("\n")
