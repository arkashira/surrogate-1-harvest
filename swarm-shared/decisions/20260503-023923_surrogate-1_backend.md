# surrogate-1 / backend

## Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16`, `DATE`, `HF_TOKEN`
- Single `list_repo_tree` call (from Mac orchestrator) → `files.json` committed to repo; worker loads `files.json` and processes only its deterministic shard via **CDN direct download** (no HF API calls during training)
- Projects each file to `{prompt, response}` only at parse time; writes normalized JSONL to `batches/public-merged/{DATE}/shard{N}-{HHMMSS}.jsonl`
- Uses central `lib/dedup.py` SQLite store for cross-run dedup (md5 hash)
- Includes retry/backoff for CDN 429, commits via `huggingface_hub` only after successful shard build

### 1) Create `bin/dataset-enrich.py`

```python
#!/usr/bin/env python3
"""
surrogate-1 CDN-bypass ingestion worker.

Usage:
  HF_TOKEN=hf_xxx \
  SHARD_ID=0 SHARD_TOTAL=16 DATE=2026-05-03 \
  python bin/dataset-enrich.py

Expects files.json (generated by Mac orchestrator) in repo root:
  [{"path": "folder/file.parquet", "sha": "...", "size": 123}, ...]
"""
import os
import sys
import json
import hashlib
import time
import logging
from pathlib import Path
from typing import List, Dict, Any

import requests
import pyarrow.parquet as pq
import pyarrow as pa
from huggingface_hub import hf_hub_download, upload_file

logging.basicConfig(level=logging.INFO, format="%(asctime)s %(levelname)s %(message)s")
log = logging.getLogger("surrogate-1-ingest")

# ----------
# Config
# ----------
REPO_ID = "axentx/surrogate-1-training-pairs"
BASE_CDN = f"https://huggingface.co/datasets/{REPO_ID}/resolve/main"
CHUNK_SIZE = 8 * 1024 * 1024  # 8 MB
MAX_RETRIES = 5
RETRY_BACKOFF = 30  # seconds (HF 429 -> wait 360s; we use smaller for CDN)

# ----------
# Dedup store
# ----------
sys.path.insert(0, str(Path(__file__).parent.parent))
from lib.dedup import DedupStore  # noqa: E402

dedup = DedupStore()

# ----------
# Helpers
# ----------
def deterministic_shard(key: str, total: int) -> int:
    """Map key to shard by md5."""
    digest = hashlib.md5(key.encode()).hexdigest()
    return int(digest, 16) % total

def load_manifest(manifest_path: Path) -> List[Dict[str, Any]]:
    if not manifest_path.exists():
        log.error("Manifest not found: %s", manifest_path)
        sys.exit(1)
    with manifest_path.open() as f:
        return json.load(f)

def cdn_download(url: str, dest: Path) -> bool:
    for attempt in range(1, MAX_RETRIES + 1):
        try:
            with requests.get(url, stream=True, timeout=60) as r:
                r.raise_for_status()
                with dest.open("wb") as f:
                    for chunk in r.iter_content(chunk_size=CHUNK_SIZE):
                        f.write(chunk)
            return True
        except Exception as exc:
            wait = RETRY_BACKOFF * (2 ** (attempt - 1))
            log.warning("Download failed (attempt %s/%s): %s — retrying in %ss", attempt, MAX_RETRIES, exc, wait)
            time.sleep(wait)
    return False

def parse_parquet_to_pairs(path: Path):
    """Yield {prompt, response} dicts from parquet; schema-agnostic."""
    try:
        pf = pq.ParquetFile(path)
        # Read in small row groups to limit memory
        for batch in pf.iter_batches(batch_size=5000):
            table = pa.Table.from_batches([batch])
            cols = {c: table.column(c).to_pylist() for c in table.column_names}
            # Heuristic projection: look for prompt/response keys (case-insensitive)
            prompt_keys = [k for k in cols if "prompt" in k.lower()]
            response_keys = [k for k in cols if "response" in k.lower()]
            if not prompt_keys or not response_keys:
                # Fallback: first two text-like columns
                text_cols = [k for k in cols if isinstance(cols[k][0], str)]
                if len(text_cols) >= 2:
                    prompt_keys, response_keys = [text_cols[0]], [text_cols[1]]
                else:
                    continue
            pkey, rkey = prompt_keys[0], response_keys[0]
            for i in range(len(cols[pkey])):
                p = cols[pkey][i]
                r = cols[rkey][i]
                if isinstance(p, str) and isinstance(r, str) and p.strip() and r.strip():
                    yield {"prompt": p.strip(), "response": r.strip()}
    except Exception as exc:
        log.error("Failed to parse %s: %s", path, exc)

def build_output_path(date_folder: str, shard_id: int) -> str:
    ts = time.strftime("%H%M%S")
    filename = f"shard{shard_id}-{ts}.jsonl"
    return f"batches/public-merged/{date_folder}/{filename}"

# ----------
# Main
# ----------
def main():
    shard_id = int(os.getenv("SHARD_ID", "0"))
    shard_total = int(os.getenv("SHARD_TOTAL", "16"))
    date_folder = os.getenv("DATE", time.strftime("%Y-%m-%d"))
    hf_token = os.getenv("HF_TOKEN")
    if not hf_token:
        log.error("HF_TOKEN required")
        sys.exit(1)

    manifest = load_manifest(Path("files.json"))
    log.info("Loaded %s files; processing shard %s/%s", len(manifest), shard_id, shard_total)

    # Select shard files
    shard_files = [
        item for item in manifest
        if deterministic_shard(item["path"], shard_total) == shard_id
    ]
    log.info("Shard %s has %s files", shard_id, len(shard_files))

    out_lines = []
    temp_dir = Path(".tmp")
    temp_dir.mkdir(exist_ok=True)

    for item in shard_files:
        url = f"{BASE_CDN}/{item['path']}"
        temp_file = temp_dir / Path(item["path"]).name

        if not cdn_download(url, temp_file):
            log.error("Failed to download %s after retries", item["path"])
            continue

        count = 0
        for pair in parse_parquet_to_pairs(temp_file):
            # Dedup by content hash
            payload = f"{pair['prompt']}\n{pair['response']}"
            md5 = hashlib.md5(payload.encode()).hexdigest()
            if dedup.exists(md5):
                continue
            dedup.add(md5)
            out_lines.append(json.dumps(pair, ensure_ascii=False))
            count += 1

        temp_file.unlink(missing_ok=True)
        log.info("Processed %s -> %s pairs", item["path"], count)

    if not out_lines:
        log.info("No new pairs for shard %s", shard_id)
        return

    out_path = build_output_path(date_folder, shard_id)
    tmp_out = temp_dir / "out.jsonl"
    tmp_out.write_text("\n".join(out_lines) + "\n", encoding="utf-8")

    # Upload
    log.info("Uploading %s (%s lines) to %s", tmp_out.name, len(out_lines), out_path)
    upload_file(
        path_or_fileobj=str(tmp_out),
        path_in_repo=out_path,
        repo_id=REPO_ID,
        repo_type="dataset",
        token=hf_token,
    )
    log.info("Upload complete")
    tmp_out.unlink(missing_ok=True)

if __name__ == "__main__":
    main()
```
