# Costinel / design

## Final Implementation Plan — Top-Hub Signal Panel (Costinel Dashboard)

**Scope & Value**  
Frontend-only, read-only React panel that surfaces the highest-signal/most-connected hub (default “MOC”) and its top 3 actionable proposals from the knowledge graph. CDN-first data path (no backend changes), ships in <2h.

### 3) Hook: `useTopHub.ts`

```tsx
// src/hooks/useTopHub.ts
import { useEffect, useState } from 'react';

export interface Proposal {
  id: string;
  title: string;
  rationale: string;
  impact: 'high' | 'medium' | 'low';
  tags: string[];
  updatedAt: string;
}

export interface Hub {
  id: string;
  name: string;
  label?: string;
  description: string;
  tags: string[];
  updatedAt: string;
}

export interface TopHubData {
  hub: Hub;
  proposals: Proposal[];
}

const DEFAULT_DATA: TopHubData = {
  hub: {
    id: 'MOC',
    name: 'MOC',
    label: 'MOC',
    description: 'Multi-cloud observability & cost guardrails — detect drift, enforce policy, and surface savings without execution.',
    tags: ['#knowledge-rag', '#graph', '#hub'],
    updatedAt: new Date().toISOString(),
  },
  proposals: [],
};

const CDN_URL = '/data/knowledge-graph/top-hub.json';

export default function useTopHub() {
  const [data, setData] = useState<TopHubData>(DEFAULT_DATA);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    let mounted = true;
    setLoading(true);
    fetch(CDN_URL, { cache: 'no-cache' })
      .then((res) => {
        if (!res.ok) throw new Error(`Failed to load hub data: ${res.status}`);
        return res.json();
      })
      .then((json) => {
        if (mounted) {
          setData(json);
          setError(null);
        }
      })
      .catch((err) => {
        if (mounted) {
          console.warn(err);
          setError(err.message);
          // keep DEFAULT_DATA (graceful fallback)
        }
      })
      .finally(() => {
        if (mounted) setLoading(false);
      });

    return () => {
      mounted = false;
    };
  }, []);

  return { data, loading, error };
}
```

### 4) Drawer: `ProposalDrawer.tsx`

```tsx
// src/components/TopHubSignalPanel/ProposalDrawer.tsx
import React from 'react';
import { Proposal } from '../../hooks/useTopHub';
import './TopHubSignalPanel.module.css';

interface ProposalDrawerProps {
  proposal: Proposal | null;
  onClose: () => void;
}

export default function ProposalDrawer({ proposal, onClose }: ProposalDrawerProps) {
  if (!proposal) return null;

  return (
    <div className="drawer-backdrop" onClick={onClose} role="presentation">
      <aside className="drawer" onClick={(e) => e.stopPropagation()} role="dialog" aria-modal="true" aria-label="Proposal details">
        <header className="drawer-header">
          <h3>{proposal.title}</h3>
          <button
            className="drawer-close"
            onClick={onClose}
            aria-label="Close proposal details"
          >
            ×
          </button>
        </header>
        <div className="drawer-body">
          <p className="rationale">{proposal.rationale}</p>
          <div className="meta">
            <span className={`impact impact-${proposal.impact}`}>{proposal.impact}</span>
            <div className="tags">
              {proposal.tags.map((t) => (
                <span key={t} className="tag">
                  {t}
                </span>
              ))}
            </div>
          </div>
        </div>
      </aside>
    </div>
  );
}
```
