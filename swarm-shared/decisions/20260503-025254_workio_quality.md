# workio / quality

## 1. Diagnosis

- **No idempotency key enforcement on punch endpoints** — LINE retries + client retries can create duplicate clock-in/out records within seconds; no `idempotency_key` or `request_id` column/index to dedupe.
- **Race condition between clock-in/out checks** — concurrent requests for same user can pass `is_clocked_in` check before either insert commits, creating double punches.
- **Missing structured request tracing** — no correlation ID propagated from LINE webhook through to DB row; hard to link duplicates to originating request in logs.
- **No unique constraint to enforce one active punch per user** — application-level check (`is_clocked_in`) is not a reliable concurrency guard.
- **Frontend optimistic UI can diverge** — retry logic without idempotency can cause duplicate requests that server accepts.

## 2. Proposed change

- **File scope**: `workio/server/src/routes/punch.ts` (punch clock-in/out handlers) + `workio/server/src/db/schema.sql` (add idempotency + active punch constraint) + `workio/server/src/middleware/request-id.ts` (correlation ID middleware).
- **Change type**: Add server-side idempotency + unique constraint + request tracing to eliminate duplicates under retries/races.

## 3. Implementation

### 3.1 DB schema changes (`workio/server/src/db/schema.sql`)

```sql
-- Add idempotency table for punch requests
CREATE TABLE IF NOT EXISTS punch_idempotency (
  idempotency_key TEXT NOT NULL,
  tenant_id INTEGER NOT NULL,
  user_id INTEGER NOT NULL,
  punch_id INTEGER NOT NULL,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  PRIMARY KEY (idempotency_key, tenant_id)
);

-- Ensure one active punch per user (nullable clocked_out_at)
CREATE UNIQUE INDEX IF NOT EXISTS idx_one_active_punch_per_user
ON punches (user_id, tenant_id)
WHERE clocked_out_at IS NULL;

-- Add request_id column to punches for tracing
ALTER TABLE punches
ADD COLUMN IF NOT EXISTS request_id TEXT;
```

### 3.2 Middleware: request-id (`workio/server/src/middleware/request-id.ts`)

```ts
import { Request, Response, NextFunction } from 'express';
import { randomUUID } from 'crypto';

export function requestIdMiddleware(req: Request, _res: Response, next: NextFunction) {
  // Prefer upstream trace id (LINE may send X-Line-Request-Id or similar)
  const headerId = req.headers['x-request-id'] || req.headers['x-line-request-id'];
  req.id = (headerId as string) || randomUUID();
  next();
}

declare global {
  namespace Express {
    interface Request {
      id: string;
    }
  }
}
```

Register in `workio/server/src/app.ts` (or main server file) before routes:

```ts
import { requestIdMiddleware } from './middleware/request-id';
app.use(requestIdMiddleware);
```

router.post('/clock-in', async (req: Request, res: Response) => {
  const { tenant_id, user_id } = req.body;
  const idempotencyKey = req.headers['idempotency-key'] as string || randomUUID();
  const requestId = req.id;

  if (!tenant_id || !user_id) {
    return res.status(400).json({ error: 'tenant_id and user_id required' });
  }

  try {
    // Check idempotency first
    const idem = await pool.query(
      'SELECT punch_id FROM punch_idempotency WHERE idempotency_key = $1 AND tenant_id = $2',
      [idempotencyKey, tenant_id]
    );
    if (idem.rows.length > 0) {
      const punch = await pool.query('SELECT * FROM punches WHERE id = $1', [idem.rows[0].punch_id]);
      return res.status(200).json({ duplicate: true, punch: punch.rows[0], request_id: requestId });
    }

    // Create punch (unique index will block double active punches)
    const insertPunch = await pool.query(
      `INSERT INTO punches (tenant_id, user_id, clocked_in_at, clocked_out_at, request_id)
       VALUES ($1, $2, NOW(), NULL, $3)
       RETURNING *`,
      [tenant_id, user_id, requestId]
    );

    const punch = insertPunch.rows[0];

    // Record idempotency
    await pool.query(
      'INSERT INTO punch_idempotency (idempotency_key, tenant_id, user_id, punch_id) VALUES ($1, $2, $3, $4)',
      [idempotencyKey, tenant_id, user_id, punch.id]
    );

    res.status(201).json({ punch, request_id: requestId });
  } catch (err: any) {
    // Unique violation: one active punch already exists
    if (err.code === '23505' && err.constraint === 'idx_one_active_punch_per_user') {
      const existing = await pool.query(
        'SELECT * FROM punches WHERE user_id = $1 AND tenant_id = $2 AND clocked_out_at IS NULL',
        [user_id, tenant_id]
      );
      return res.status(409).json({
        error: 'already_clocked_in',
        punch: existing.rows[0],
        request_id: requestId,
      });
    }
    console.error('Clock-in error', { err, request_id: requestId });
    res.status(500).json({ error: 'internal_error', request_id: requestId });
  }
});
