# surrogate-1 / ops

## Final Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16`, `DATE`, `HF_TOKEN` via env
- Single `list_repo_tree(path, recursive=False)` per date folder → deterministic shard slice
- Saves file manifest to `manifest/{DATE}/shard{SHARD_ID}.json` (for audit/repro)
- Downloads only assigned files via **HF CDN bypass** (`resolve/main/...`, no Authorization header)
- Projects each file to `{prompt, response}` at parse time (avoids pyarrow CastError on mixed schemas)
- Produces `batches/public-merged/{DATE}/shard{N}-{HHMMSS}.jsonl`
- Uses central dedup via `lib/dedup.py` (md5 store) to avoid duplicates
- Exits non-zero on unrecoverable errors; logs summary for GitHub Actions

### Why this is the highest-value incremental improvement
- Fixes the core risk in the README: 16 shards × 1.5–2 GB each → OOM on Space. This worker runs in GitHub Actions (7 GB isolated) and uses CDN-only fetches (no API rate limits during training).
- Aligns with Lightning Studio quota strategy: reuse running Studio for training iterations; zero API calls during data load via pre-saved manifest.

### 1) New worker: `bin/dataset-enrich.py`

```python
#!/usr/bin/env python3
"""
CDN-bypass ingestion worker for surrogate-1-training-pairs.
Usage (via env):
  SHARD_ID=0 SHARD_TOTAL=16 DATE=2026-05-03 HF_TOKEN=hf_xxx python bin/dataset-enrich.py
"""
import os
import sys
import json
import hashlib
import time
from datetime import datetime, timezone
from pathlib import Path

import requests
from huggingface_hub import HfApi

REPO = "axentx/surrogate-1-training-pairs"
API = HfApi()
DATE_FMT = "%Y-%m-%d"
TS_FMT = "%Y%m%d-%H%M%S"

# ---------- config ----------
SHARD_ID = int(os.getenv("SHARD_ID", "0"))
SHARD_TOTAL = int(os.getenv("SHARD_TOTAL", "16"))
DATE = os.getenv("DATE", datetime.now(timezone.utc).strftime(DATE_FMT))
HF_TOKEN = os.getenv("HF_TOKEN")
if not HF_TOKEN:
    print("HF_TOKEN required", file=sys.stderr)
    sys.exit(1)

HEADERS = {"Authorization": f"Bearer {HF_TOKEN}"}
OUT_DIR = Path("batches/public-merged") / DATE
OUT_DIR.mkdir(parents=True, exist_ok=True)
TS = datetime.now(timezone.utc).strftime(TS_FMT)
OUT_FILE = OUT_DIR / f"shard{SHARD_ID}-{TS}.jsonl"

# ---------- dedup ----------
DEDUP_DB = Path("lib/dedup.py")
if DEDUP_DB.exists():
    # lightweight import of dedup helper
    import importlib.util
    spec = importlib.util.spec_from_file_location("dedup", str(DEDUP_DB))
    dedup_mod = importlib.util.module_from_spec(spec)
    spec.loader.exec_module(dedup_mod)
    seen = dedup_mod.DedupStore()
else:
    class DummyDedup:
        def __init__(self): self.s = set()
        def exists(self, h): return h in self.s
        def add(self, h): self.s.add(h)
    seen = DummyDedup()

# ---------- helpers ----------
def shard_for(slug: str) -> int:
    return int(hashlib.md5(slug.encode()).hexdigest(), 16) % SHARD_TOTAL

def list_date_files(date: str):
    """Single API call: list top-level folder for date."""
    try:
        tree = API.list_repo_tree(repo_id=REPO, path=date, recursive=False)
        return [t.path for t in tree if t.type == "file"]
    except Exception as e:
        print(f"list_repo_tree failed: {e}", file=sys.stderr)
        return []

def cdn_url(path: str) -> str:
    return f"https://huggingface.co/datasets/{REPO}/resolve/main/{path}"

def parse_file_cdn(path: str):
    """Download via CDN and project to {prompt,response} only."""
    url = cdn_url(path)
    r = requests.get(url, timeout=60)
    r.raise_for_status()
    # Try parquet first (common), fallback to jsonl
    import io
    try:
        import pyarrow.parquet as pq
        tbl = pq.read_table(io.BytesIO(r.content))
        df = tbl.to_pandas()
    except Exception:
        # try jsonl lines
        lines = r.text.strip().splitlines()
        import json as _json
        df = [_json.loads(l) for l in lines if l.strip()]
        # normalize list-of-dict to single dict stream
        if isinstance(df, list):
            pass
        else:
            df = [df]

    # Project to prompt/response regardless of schema
    for row in (df if isinstance(df, list) else df.to_dict(orient="records")):
        prompt = row.get("prompt") or row.get("input") or row.get("text") or ""
        response = row.get("response") or row.get("output") or row.get("completion") or ""
        if not prompt or not response:
            continue
        md5 = hashlib.md5((prompt + response).encode()).hexdigest()
        if seen.exists(md5):
            continue
        seen.add(md5)
        yield {"prompt": prompt, "response": response, "source": path}

# ---------- main ----------
def main():
    files = list_date_files(DATE)
    if not files:
        print(f"No files found for {DATE}", file=sys.stderr)
        sys.exit(1)

    manifest_dir = Path("manifest") / DATE
    manifest_dir.mkdir(parents=True, exist_ok=True)
    manifest_path = manifest_dir / f"shard{SHARD_ID}.json"
    manifest_path.write_text(json.dumps(files, indent=2))
    print(f"Manifest saved: {manifest_path} ({len(files)} files)")

    my_files = [f for f in files if shard_for(f) == SHARD_ID]
    print(f"Shard {SHARD_ID}/{SHARD_TOTAL} -> {len(my_files)} files")

    written = 0
    with OUT_FILE.open("w", buffering=1) as out:
        for path in my_files:
            try:
                for rec in parse_file_cdn(path):
                    out.write(json.dumps(rec, ensure_ascii=False) + "\n")
                    written += 1
            except Exception as e:
                print(f"Skip {path}: {e}", file=sys.stderr)

    # persist dedup state if module supports it
    if hasattr(seen, "persist"):
        seen.persist()

    print(f"Done: {written} records -> {OUT_FILE}")

if __name__ == "__main__":
    main()
```

Make executable:

```bash
chmod +x bin/dataset-enrich.py
```
