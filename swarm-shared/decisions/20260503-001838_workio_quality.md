# workio / quality

## Final consolidated solution (best of both proposals)

### Diagnosis (merged)
- No client-side idempotency key → double-taps and LINE webhook replays create duplicates.
- Button has no optimistic state or debounce → users can spam and queue multiple requests.
- No local validation of state transitions → invalid requests and confusing UX.
- No persisted “last punch” cache → reloads or offline/online flips cause stale UI and invalid transitions.
- Missing visual feedback during submission → users retry and amplify duplicates.
- No backend deduplication → duplicates can still land if frontend fixes are bypassed.

### Scope
- One frontend component (`PunchButton.tsx`) + one backend route (`punch.ts`) + one small DB migration.
- End-to-end <2 hours.

### Frontend: `PunchButton.tsx`
```tsx
// src/components/PunchButton.tsx
import { useState, useEffect, useCallback } from 'react';

const PUNCH_COOLDOWN_MS = 3000;
const IDEMPOTENCY_TTL_MS = 24 * 60 * 60 * 1000; // 24h

type PunchType = 'IN' | 'OUT';

interface PunchResponse {
  id: string;
  type: PunchType;
  duplicate?: boolean;
}

function getStoredPunch(): { type: PunchType; at: string } | null {
  try {
    const raw = sessionStorage.getItem('lastPunch');
    return raw ? JSON.parse(raw) : null;
  } catch {
    return null;
  }
}

function setStoredPunch(type: PunchType) {
  try {
    sessionStorage.setItem(
      'lastPunch',
      JSON.stringify({ type, at: new Date().toISOString() })
    );
  } catch {}
}

function makeIdempotencyKey(): string {
  const date = new Date().toISOString().slice(0, 10); // YYYY-MM-DD
  return `punch-${date}-${crypto.randomUUID()}`;
}

export function PunchButton() {
  const [punchType, setPunchType] = useState<PunchType>('IN');
  const [isSubmitting, setIsSubmitting] = useState(false);
  const [cooldown, setCooldown] = useState(false);

  // Initialize from server + persisted cache
  useEffect(() => {
    async function load() {
      try {
        const res = await fetch('/api/punch/status');
        if (!res.ok) throw new Error('Failed to load status');
        const data = await res.json();
        if (data.lastType) setPunchType(data.lastType === 'in' ? 'OUT' : 'IN');
      } catch {
        // fallback to persisted cache
        const stored = getStoredPunch();
        if (stored) {
          setPunchType(stored.type === 'IN' ? 'OUT' : 'IN');
        }
      }
    }
    load();
  }, []);

  const submitPunch = useCallback(
    async (idempotencyKey: string) => {
      const res = await fetch('/api/punch', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ idempotencyKey }),
      });
      if (!res.ok) throw new Error('Punch failed');
      return (await res.json()) as PunchResponse;
    },
    []
  );

  const handleClick = useCallback(async () => {
    if (isSubmitting || cooldown) return;

    // Local transition validation
    const willBe: PunchType = punchType === 'IN' ? 'IN' : 'OUT';
    if (punchType === 'OUT' && willBe === 'OUT') {
      // This branch is defensive; UI should prevent it.
      return;
    }

    setIsSubmitting(true);
    const key = makeIdempotencyKey();

    try {
      const data = await submitPunch(key);
      const nextType: PunchType = data.type === 'in' ? 'OUT' : 'IN';
      setPunchType(nextType);
      setStoredPunch(data.type === 'in' ? 'IN' : 'OUT');
    } catch (err) {
      console.error(err);
    } finally {
      setIsSubmitting(false);
      setCooldown(true);
      setTimeout(() => setCooldown(false), PUNCH_COOLDOWN_MS);
    }
  }, [isSubmitting, cooldown, punchType, submitPunch]);

  const label = punchType === 'IN' ? 'ลงเวลาเข้างาน' : 'ลงเวลาออกงาน';
  const isNextIn = punchType === 'IN';

  return (
    <button
      onClick={handleClick}
      disabled={isSubmitting || cooldown}
      className={`px-6 py-3 rounded-lg font-medium transition-all ${
        isSubmitting || cooldown
          ? 'bg-gray-400 cursor-not-allowed'
          : isNextIn
          ? 'bg-green-600 hover:bg-green-700 text-white'
          : 'bg-red-600 hover:bg-red-700 text-white'
      }`}
    >
      {isSubmitting ? 'กำลังบันทึก...' : label}
    </button>
  );
}
```
