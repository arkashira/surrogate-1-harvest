## 1. Diagnosis (merged)
- **Missing idempotency on LINE webhook ingestion** — LINE retries on 5xx/timeouts; duplicate `webhookEventId` causes double punches/leave/OT.
- **Race conditions on punch state** — `find-then-insert` / `find-then-update` allows multiple open punches or lost/duplicate clock-outs under concurrency.
- **No durable deduplication for webhook events** — transient DB/network errors allow LINE to resend the same event with no server-side record.
- **No enforcement of business invariants at DB level** — DB allows multiple open punches per user/tenant; constraints/indexes should surface bugs early.
- **Insufficient observability and retry strategy** — errors can be lost; retries should be bounded and observable.

---

## 2. Proposed changes (merged)
- Add a `webhook_events` table with unique `webhook_event_id` for durable idempotency.
- Add a partial unique index to enforce **one open punch per (user, tenant)**.
- Convert punch create/close to **atomic, transactional upsert/update** inside a single DB transaction.
- Implement idempotency check **before** processing and record the event atomically with the punch change.
- Return clear, safe responses to LINE (200 for duplicates/success; 5xx only for persistent failures) and add structured logging/metrics.
- Do **not** implement application-level retry loops inside the webhook handler (avoid double-processing); rely on idempotency + safe retries by LINE where appropriate.

## 3. Implementation (single, actionable version)

### 3.1 DB schema (`server/src/db/schema.sql`)

```sql
-- Idempotency table for LINE webhook events
CREATE TABLE IF NOT EXISTS webhook_events (
  id              BIGSERIAL PRIMARY KEY,
  webhook_event_id VARCHAR(255) NOT NULL,
  tenant_id       BIGINT NOT NULL,
  user_id         BIGINT NOT NULL,
  event_type      VARCHAR(50) NOT NULL,
  payload         JSONB,
  created_at      TIMESTAMPTZ DEFAULT NOW(),
  CONSTRAINT uniq_webhook_event_id UNIQUE (webhook_event_id)
);

-- Ensure one open punch per (user, tenant)
-- Assumes punches has columns: id, user_id, tenant_id, status ('open'/'closed'), clock_in, clock_out, location, created_at, updated_at
CREATE UNIQUE INDEX IF NOT EXISTS idx_one_open_punch
ON punches (user_id, tenant_id)
WHERE status = 'open';
```

### 3.2 Idempotent, atomic webhook handler (`server/src/controllers/lineWebhook.ts`)

```typescript
// server/src/controllers/lineWebhook.ts
import { Request, Response } from 'express';
import { pool } from '../db';
import { logger } from '../utils/logger';

type ClockEvent = 'clock_in' | 'clock_out';

async function processClockEvent(
  tenantId: number,
  userId: number,
  webhookEventId: string,
  eventType: ClockEvent,
  payload: any
) {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // 1) Idempotency check
    const dup = await client.query(
      `SELECT 1 FROM webhook_events WHERE webhook_event_id = $1`,
      [webhookEventId]
    );
    if (dup.rows.length > 0) {
      await client.query('ROLLBACK');
      return { alreadyProcessed: true };
    }

    // 2) Record webhook event atomically
    await client.query(
      `INSERT INTO webhook_events (webhook_event_id, tenant_id, user_id, event_type, payload)
       VALUES ($1,$2,$3,$4,$5)`,
      [webhookEventId, tenantId, userId, eventType, payload]
    );

    const now = new Date();

    if (eventType === 'clock_in') {
      // 3a) Atomic insert or no-op if already open
      const res = await client.query(
        `INSERT INTO punches (user_id, tenant_id, clock_in, status, location, created_at, updated_at)
         VALUES ($1,$2,$3,'open',$4,$5,$5)
         ON CONFLICT (user_id, tenant_id) WHERE status = 'open'
         DO UPDATE SET updated_at = EXCLUDED.updated_at
         RETURNING *`,
        [userId, tenantId, now, payload?.location || null]
      );
      await client.query('COMMIT');
      return { punch: res.rows[0], new: res.rows[0].clock_in.getTime() === now.getTime() };
    } else {
      // 3b) Atomic close of open punch
      const res = await client.query(
        `UPDATE punches
         SET clock_out = $1, status = 'closed', updated_at = $2
         WHERE user_id = $3 AND tenant_id = $4 AND status = 'open'
         RETURNING *`,
        [now, now, userId, tenantId]
      );
      if (res.rows.length === 0) {
        await client.query('ROLLBACK');
        return { error: 'no_open_punch_to_close' };
      }
      await client.query('COMMIT');
      return { punch: res.rows[0], closed: true };
    }
  } catch (err) {
    await client.query('ROLLBACK');
    logger.error('Failed to process clock event', {
      tenantId,
      userId,
      webhookEventId,
      eventType,
      error: err instanceof Error ? err.message : err
    });
    throw err;
  } finally {
    client.release();
  }
}

// Webhook endpoint
export async function lineWebhookHandler(req: Request, res: Response) {
  const body = req.body;

  // Basic validation
  if (!body || !Array.isArray(body.events) || body.events.length === 0) {
    return res.status(400).json({ error: 'invalid_payload' });
  }

  const results: any[] = [];
  let hasError = false;

  for (const ev of body.events) {
    const webhookEventId = ev.webhookEventId;
    const eventType = ev.type; // e.g., 'clock_in'/'clock_out'
    const userId = ev.source?.userId;
    const tenantId = ev.source?.tenantId;

    if (!webhookEventId || !eventType || !userId || !tenantId) {
      logger.warn('Skipping malformed event', { ev });
      continue;
    }

    try {
      const result = await processClockEvent(
        Number(tenantId),
        Number(userId),
        webhookEventId,
        eventType,
        ev
      );
      results.push({ webhookEventId, result });
    } catch (err) {
      hasError = true;
      results.push({ webhookEventId, error: 'processing_failed' });
      // Do not throw here; continue processing other events, but mark overall failure
    }
  }

  if (hasError) {
    // Return 500 so LINE may retry unprocessed events (idempotency protects duplicates)
    logger.warn('Webhook partially failed', { results });
    return res.status(500).json({ results });
  }

  return res.status(200).json({ results });
}
```
