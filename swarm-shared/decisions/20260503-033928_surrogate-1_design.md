# surrogate-1 / design

## Final Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16` (matrix) and optional `DATE_FOLDER` (defaults to today `YYYY-MM-DD`)
- Uses **manifest-first strategy**: single `list_repo_tree` call for the date folder → saves `manifest.json` → shard workers stream exclusively via CDN (`resolve/main/`) with **zero HF API calls** during data load
- Downloads via HF CDN with **no Authorization header** → completely bypasses `/api/` rate limits
- Projects heterogeneous schemas to `{prompt, response}` only at parse time (avoids pyarrow CastError); skips files that cannot be projected
- Dedups via central md5 store (`lib/dedup.py`) and writes deterministic shard output:
  ```
  batches/public-merged/<date>/shard<N>-<HHMMSS>.jsonl
  ```
- Retries with exponential backoff on 429 (wait 360s) and 5xx; skips already-processed files via dedup + manifest mtime check
- Logs structured JSON for GitHub Actions matrix visibility

def download_with_retry(url: str, max_retries: int = 5) -> bytes:
    backoff = 1
    for attempt in range(max_retries):
        try:
            resp = requests.get(url, timeout=30)
            if resp.status_code == 429:
                wait = 360
                log.warning(json.dumps({"event": "rate_limit_429", "wait": wait}))
                time.sleep(wait)
                continue
            resp.raise_for_status()
            return resp.content
        except Exception as exc:
            if attempt == max_retries - 1:
                log.error(json.dumps({"event": "download_failed", "url": url, "error": str(exc)}))
                raise
            sleep_time = backoff * (2 ** attempt)
            log.warning(json.dumps({"event": "download_retry", "attempt": attempt, "sleep": sleep_time, "error": str(exc)}))
            time.sleep(sleep_time)
    raise RuntimeError("unreachable")

def extract_pair(raw: bytes, filename: str) -> Optional[Dict[str, str]]:
    """
    Project heterogeneous schemas to {prompt, response}.
    Supports: jsonl, json, parquet (via pyarrow), txt (single prompt).
    Returns None if projection fails.
    """
    try:
        import pyarrow.parquet as pq
        from io import BytesIO
    except Exception:
        pq = None  # type: ignore

    suffix = Path(filename).suffix.lower()
    try:
        if suffix == ".parquet" and pq is not None:
            table = pq.read_table(BytesIO(raw))
            cols = table.column_names
            if not cols:
                return None
            prompt_col = next((c for c in cols if "prompt" in c.lower()), cols[0])
            response_col = next((c for c in cols if "response" in c.lower() or "output" in c.lower()), cols[-1])
            return {
                "prompt": str(table[prompt_col][0].as_py()),
                "response": str(table[response_col][0].as_py()),
            }

        if suffix == ".jsonl":
            lines = raw.decode("utf-8").strip().split("\n")
            if not lines or not lines[0].strip():
                return None
            obj = json.loads(lines[0])
            prompt = obj.get("prompt") or obj.get("instruction") or list(obj.values())[0]
            response = obj.get("response") or obj.get("output") or list(obj.values())[-1]
            return {"prompt": str(prompt), "response": str(response)}

        if suffix == ".json":
            obj = json.loads(raw)
            if isinstance(obj, list):
                if not obj:
                    return None
                obj = obj[0]
            prompt = obj.get("prompt") or obj.get("instruction") or list(obj.values())[0]
            response = obj.get("response") or obj.get("output") or list(obj.values())[-1]
            return {"prompt": str(prompt), "response": str(response)}

        # fallback: treat as single prompt
        text = raw.decode("utf-8", errors="replace").strip()
        if not text:
            return None
        return {"prompt": text, "response": ""}
    except Exception as exc:
        log.debug(json.dumps({"event": "projection_failed", "file": filename, "error": str(exc)}))
        return None
