## Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16` (matrix) and optional `DATE_FOLDER` (defaults to today `YYYY-MM-DD`)
- Uses **manifest-first strategy**: single `list_repo_tree` call from Mac (outside runner) to produce `manifest/<DATE_FOLDER>.json` containing all file paths; runners load this manifest and deterministically shard by `hash(slug) % SHARD_TOTAL`
- Downloads via **HF CDN bypass** (`https://huggingface.co/datasets/.../resolve/main/...`) with no Authorization header to avoid API rate limits during training
- Projects heterogeneous schemas to `{prompt, response}` only at parse time (avoids `pyarrow.CastError`)
- Deduplicates via central `lib/dedup.py` md5 store (unchanged)
- Writes `batches/public-merged/<DATE_FOLDER>/shard<N>-<HHMMSS>.jsonl` with deterministic naming to prevent collisions
- Adds retry/backoff for 429 (wait 360s) and commit-cap spreading across sibling repos if needed
- Keeps orchestration on Mac (Lightning SDK launcher, HF API) and heavy compute on remote (Lightning Studio)

def deterministic_shard(key: str, total: int) -> int:
    return int(hashlib.sha256(key.encode("utf-8")).hexdigest(), 16) % total

def load_or_create_manifest(date_folder: str) -> List[str]:
    """Load manifest if present; otherwise create via single API call."""
    manifest_file = Path(MANIFEST_PATH)
    if manifest_file.exists():
        log.info("Loading existing manifest: %s", manifest_file)
        return json.loads(manifest_file.read_text())

    log.info("Creating manifest for %s via list_repo_tree (non-recursive per folder)", date_folder)
    if api is None:
        raise RuntimeError("HF_TOKEN required to create manifest")

    # List top-level date folder non-recursive, then walk subfolders minimally
    entries = list_repo_tree(repo_id=HF_REPO, path=date_folder, recursive=False, token=HF_TOKEN)
    file_paths: List[str] = []

    # If entries are files directly under date_folder, include them
    for entry in entries:
        if entry.type == "file":
            file_paths.append(f"{date_folder}/{entry.path.split('/')[-1]}")
        else:
            # One-level deeper (avoid recursive explosion)
            subentries = list_repo_tree(repo_id=HF_REPO, path=entry.path, recursive=False, token=HF_TOKEN)
            for sub in subentries:
                if sub.type == "file":
                    file_paths.append(sub.path)

    manifest_file.parent.mkdir(parents=True, exist_ok=True)
    manifest_file.write_text(json.dumps(file_paths, indent=2))
    log.info("Manifest saved with %d files", len(file_paths))
    return file_paths

def cdn_download(url: str, dest: Path, max_retries: int = MAX_RETRIES) -> bool:
    for attempt in range(1, max_retries + 1):
        try:
            resp = requests.get(url, timeout=60, stream=True)
            if resp.status_code == 429:
                log.warning("Rate limited (429). Waiting %ds before retry %d/%d", RATE_LIMIT_WAIT, attempt, max_retries)
                time.sleep(RATE_LIMIT_WAIT)
                continue
            resp.raise_for_status()
            dest.parent.mkdir(parents=True, exist_ok=True)
            with open(dest, "wb") as f:
                for chunk in resp.iter_content(chunk_size=8192):
                    f.write(chunk)
            return True
        except Exception as exc:
            wait = RETRY_BACKOFF * (2 ** (attempt - 1))
            log.warning("Download failed (attempt %d/%d): %s. Retrying in %ds", attempt, max_retries, exc, wait)
            time.sleep(wait)
    return False

def parse_file_to_pairs(local_path: Path) -> List[Dict[str, str]]:
    """
    Project heterogeneous file schemas to {prompt, response} only.
    Supports JSONL/JSON/parquet via hf_hub_download + pyarrow projection when possible.
    Falls back to CDN download + line-by-line JSONL parsing.
    """
    pairs = []
    suffix = local_path.suffix.lower()

    try:
        if suffix == ".jsonl":
            import json as jsonlib
            for line in local_path.read_text().splitlines():
                line = line.strip()
                if not line:
                    continue
                obj = jsonlib.loads(line)
                prompt = obj.get("prompt") or obj.get("input") or obj.get("text") or ""
                response = obj.get("response") or obj.get("output") or obj.get("completion") or ""
                if prompt and response:
                    pairs.append({"prompt": str(prompt), "response": str(response)})
        elif suffix == ".json":
            import json as jsonlib
            data = jsonlib.loads(local_path.read_text())
            if isinstance(data, list):
                for obj in data:
                    prompt = obj.get("prompt") or obj.get("input") or obj.get("text") or ""
                    response = obj.get("response") or obj.get("output") or obj.get("completion") or ""
                    if prompt and response:
                        pairs.append({"prompt": str(prompt), "response": str(response)})
            elif isinstance(data, dict):
                prompt = data.get("prompt") or data.get("input") or data.get("text") or ""
                response = data.get("response") or data.get("output") or data.get("completion") or ""
                if prompt and response:
                    pairs.append({"prompt": str(prompt), "response": str(response)})
        elif suffix in (".parquet", ".arrow"):
            # Avoid streaming heterogeneous schemas; project minimally via pyarrow if available
            try:
                import pyarrow.parquet as pq
                table = pq.read_table(local_path, columns=["prompt", "response"])
                pdf = table.to_pandas()
                for _, row in pdf.iterrows():
                    prompt = row.get("prompt") or ""
                    response = row.get("response") or ""
                    if prompt and response:
                        pairs.append({"prompt": str(prompt), "response": str(response)})
            except Exception:
                log.warning("Parquet projection failed for %s, skipping", local_path)
        else:
            log.warning("Unsupported file type %s for %s", suffix, local_path)
    except Exception as exc:
        log.warning("Parse error for %s: %s", local_path, exc)

    return pairs
