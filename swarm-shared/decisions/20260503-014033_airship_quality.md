## Final Synthesized Implementation (Best of Both Candidates)

### Core Strategy
**Eliminate HF API 429s during Surrogate training** by switching to **CDN-only ingestion** with a **pre-listed file manifest**. This bypasses authentication checks (which trigger rate limits) and preserves Lightning quota via studio reuse.

**Why this wins:**
- CDN downloads (`resolve/main/`) have significantly higher rate limits than API endpoints
- Pre-listing files once avoids repeated `list_repo_files` pagination (100x+ requests)
- Lightning Studio reuse prevents quota burn on environment setup
- Concrete, <2h implementation with immediate ROI

import json, sys, traceback
from huggingface_hub import list_repo_tree

repo = sys.argv[1]
folder = sys.argv[2]
outfile = sys.argv[3]

try:
    # Non-recursive to avoid pagination explosion
    tree = list_repo_tree(repo=repo, path=folder, recursive=False)
    files = [f.rfilename for f in tree 
             if f.type == "file" and f.rfilename.endswith(".parquet")]
    
    manifest = {
        "repo": repo,
        "base_url": f"https://huggingface.co/datasets/{repo}/resolve/main",
        "date_folder": folder,
        "files": sorted(files),
        "generated_at": __import__('datetime').datetime.utcnow().isoformat()
    }
    
    with open(outfile, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"✅ Wrote {len(files)} files to {outfile}")
    
except Exception as e:
    print(f"❌ Failed: {e}")
    traceback.print_exc()
    sys.exit(1)
PY

#### 2. CDN-Only Data Loader (Lightning Training)
*Best from Candidate 1: Arrow streaming for memory efficiency, Candidate 2: Cleaner schema*

```python
# /opt/axentx/airship/surrogate/data/cdn_parquet_dataset.py
import json
import pyarrow.parquet as pq
import requests
import io
from typing import Iterator, Dict
from torch.utils.data import IterableDataset

class CDNParquetDataset(IterableDataset):
    """Loads Parquet files from HF CDN (no API auth) using pre-generated manifest."""
    
    def __init__(self, manifest_path: str, columns=("prompt", "response")):
        with open(manifest_path) as f:
            manifest = json.load(f)
        
        self.base_url = manifest["base_url"]
        self.files = [f"{manifest['base_url']}/{f}" for f in manifest["files"]]
        self.columns = columns

    def _stream_parquet(self, url: str) -> Iterator[Dict]:
        """Stream parquet via CDN (no Authorization header)."""
        resp = requests.get(url, stream=True, timeout=30)
        resp.raise_for_status()
        
        # Use Arrow for zero-copy column projection
        with pq.ParquetFile(io.BytesIO(resp.content)) as pf:
            for batch in pf.iter_batches(columns=self.columns, batch_size=1000):
                yield batch.to_pydict()

    def __iter__(self) -> Iterator[Dict]:
        for url in self.files:
            yield from self._stream_parquet(url)
```

#### 3. Lightning Studio Reuse (Quota Preservation)
*Best from Candidate 1: Studio reuse logic, Candidate 2: Cleaner integration*

```python
# /opt/axentx/airship/surrogate/launch_studio.py
from lightning_sdk import Teamspace, Studio, Machine
import time
import json
from pathlib import Path

TEAMSPACE = "surrogate-team"
STUDIO_NAME = "surrogate-train-l40s"
MANIFEST_PATH = Path(__file__).parent / "file_manifest.json"

def ensure_studio_running() -> Studio:
    """Reuse existing studio or create new L40S instance."""
    team = Teamspace(TEAMSPACE)
    running = [s for s in team.studios 
               if s.name == STUDIO_NAME and s.status == "Running"]
    
    if running:
        studio = running[0]
        print(f"♻️  Reusing running studio: {studio.id}")
        return studio
    
    studio = Studio.create(
        teamspace=TEAMSPACE,
        name=STUDIO_NAME,
        machine=Machine.L40S,
    )
    print(f"🚀 Created studio: {studio.id}")
    return studio

def launch_training():
    studio = ensure_studio_running()
    
    # Wait for studio to be ready
    while studio.status != "Running":
        time.sleep(10)
        studio.refresh()
    
    # Verify manifest exists
    if not MANIFEST_PATH.exists():
        raise FileNotFoundError(f"Manifest missing: {MANIFEST_PATH}")
    
    # Copy manifest to studio workspace
    studio.upload_file(
        local_path=str(MANIFEST_PATH),
        remote_path="file_manifest.json"
    )
    
    # Launch training with zero HF API calls during data load
    run = studio.run(
        command="python train.py --manifest file_manifest.json",
        sync=False,
    )
    
    print(f"📊 Training started: {run.url}")
    return run

if __name__ == "__main__":
    launch_training()
```
