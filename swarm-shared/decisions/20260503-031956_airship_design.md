# airship / design

## Implementation Plan — airship/surrogate (≤2h)

**Highest-value incremental improvement:**  
Make Surrogate training HF-rate-limit-proof and Lightning-idle-resilient by embedding a CDN-only file list and adding auto-recovery for idle timeouts.

### Why this ships value in <2h
- Eliminates 429 failures during training (uses CDN bypass).
- Survives Lightning idle stops without manual intervention.
- No schema or model changes — infra/data-loading only.
- Reuses existing patterns (Lightning Studio reuse, CDN bypass, pre-list file paths).

### Code snippets

#### 1) CDN manifest builder (run from Mac)
`scripts/build_cdn_manifest.py`
```python
#!/usr/bin/env python3
"""
Build CDN-only manifest for a dataset repo/date folder.
Run from Mac after rate-limit window clears.
"""
import json
import os
import sys
from datetime import datetime, timedelta
from pathlib import Path

from huggingface_hub import HfApi

HF_REPO = os.getenv("HF_DATASET_REPO", "myorg/surrogate-mirror")
DATE_FOLDER = os.getenv("DATE_FOLDER", (datetime.utcnow() - timedelta(days=1)).strftime("%Y%m%d"))
OUT_DIR = Path("manifests")
OUT_DIR.mkdir(exist_ok=True)

api = HfApi()

def build_manifest():
    # Non-recursive per folder to avoid pagination explosion
    entries = api.list_repo_tree(repo_id=HF_REPO, path=DATE_FOLDER, recursive=False)
    files = []
    for entry in entries:
        if entry.rfilename.endswith(".parquet"):
            files.append({
                "repo": HF_REPO,
                "path": entry.rfilename,
                "url": f"https://huggingface.co/datasets/{HF_REPO}/resolve/main/{entry.rfilename}"
            })

    out_path = OUT_DIR / f"cdn_files_{DATE_FOLDER}.json"
    out_path.write_text(json.dumps(files, indent=2))
    print(f"Wrote {len(files)} files to {out_path}")
    return out_path

if __name__ == "__main__":
    build_manifest()
```

Make executable:
```bash
chmod +x scripts/build_cdn_manifest.py
```

import json
from pathlib import Path
from typing import List, Dict

MANIFEST_DIR = Path(__file__).parent.parent.parent / "manifests"

def latest_manifest() -> List[Dict]:
    files = sorted(MANIFEST_DIR.glob("cdn_files_*.json"))
    if not files:
        return []
    latest = files[-1]
    return json.loads(latest.read_text())

#### 3) CDN-only dataset loader
`surrogate/training/dataset.py`
```python
import io
import time
from typing import Iterator, Dict

import pyarrow.parquet as pq
import requests
from datasets import Dataset

from .config import latest_manifest

CDN_TIMEOUT = 30
MAX_RETRIES = 5
BACKOFF = 360  # seconds on 429

def cdn_parquet_loader() -> Dataset:
    items = []
    for meta in latest_manifest():
        url = meta["url"]
        for attempt in range(1, MAX_RETRIES + 1):
            try:
                resp = requests.get(url, timeout=CDN_TIMEOUT)
                resp.raise_for_status()
                table = pq.read_table(io.BytesIO(resp.content))
                # Project to {prompt, response} only at parse time
                df = table.select(["prompt", "response"]).to_pandas()
                items.extend(df.to_dict(orient="records"))
                break
            except Exception as exc:
                if attempt == MAX_RETRIES:
                    raise
                if getattr(exc, "response", None) and exc.response.status_code == 429:
                    time.sleep(BACKOFF)
                    continue
                time.sleep(2 ** attempt)
    return Dataset.from_list(items)
```

#### 4) Lightning runner with auto-recovery
`surrogate/training/lightning_runner.py`
```python
import time
from lightning import Lightning, Teamspace, Machine, Studio

from .train import train_fn  # your existing training entrypoint

LIGHTNING_MACHINE = Machine.L40S  # fallback; override via env
STUDIO_NAME = os.getenv("SURROGATE_STUDIO_NAME", "surrogate-train")

def ensure_running_studio() -> Studio:
    team = Teamspace()
    for s in team.studios:
        if s.name == STUDIO_NAME and s.status == "Running":
            print(f"Reusing running studio: {s.name}")
            return s

    print(f"Creating studio: {STUDIO_NAME}")
    return Lightning.Studio(
        name=STUDIO_NAME,
        create_ok=True,
        target=LIGHTNING_MACHINE,
    )

def run_with_recovery():
    max_checks = 3
    for i in range(max_checks):
        studio = ensure_running_studio()
        if studio.status != "Running":
            print(f"Studio not running ({studio.status}), restarting...")
            studio.target.start(machine=LIGHTNING_MACHINE)
            time.sleep(60)  # allow startup
            continue

        try:
            studio.run(train_fn)
            print("Training completed.")
            return
        except Exception as exc:
            print(f"Training run failed (attempt {i+1}/{max_checks}): {exc}")
            time.sleep(120)

    print("Max recovery attempts reached.")

if __name__ == "__main__":
    run_with_recovery()
```
