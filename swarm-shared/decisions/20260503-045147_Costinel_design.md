# Costinel / design

## Implementation Plan — Costinel Top Hub Signal Panel (CDN-first)

**Scope**: Frontend-only addition to Costinel dashboard.  
**Effort**: ~60–90 minutes.  
**Mechanism**: CDN JSON fetch (no auth, no backend) with local fallback and client cache.  
**Goal**: Show the most-connected hub (e.g., "MOC") and top 3 related signals from the knowledge-rag graph, surfaced as actionable cards on the dashboard.

### 3) Code snippets

#### `src/lib/cdn.ts`
```ts
const TOP_HUB_CDN =
  'https://huggingface.co/datasets/axentx/knowledge-rag/resolve/main/top-hub/latest.json';
const FALLBACK_PATH = '/data/top-hub-fallback.json';
const CACHE_KEY = 'costinel-top-hub-cache';
const CACHE_TTL_MS = 5 * 60 * 1000; // 5m

export interface TopHubSignal {
  id: string;
  title: string;
  severity: 'low' | 'medium' | 'high';
  impact: string;
  action: string;
  link: string;
}

export interface TopHubPayload {
  hub: string;
  updatedAt: string;
  signals: TopHubSignal[];
}

async function fetchWithTimeout(url: string, timeoutMs = 6000): Promise<Response> {
  const controller = new AbortController();
  const id = setTimeout(() => controller.abort(), timeoutMs);
  try {
    return await fetch(url, { signal: controller.signal, cache: 'no-store' });
  } finally {
    clearTimeout(id);
  }
}

async function loadFromCDN(): Promise<TopHubPayload | null> {
  try {
    const res = await fetchWithTimeout(TOP_HUB_CDN);
    if (!res.ok) throw new Error(`CDN ${res.status}`);
    return (await res.json()) as TopHubPayload;
  } catch {
    return null;
  }
}

async function loadFromFallback(): Promise<TopHubPayload | null> {
  try {
    const res = await fetch(FALLBACK_PATH, { cache: 'no-store' });
    if (!res.ok) throw new Error('Fallback missing');
    return (await res.json()) as TopHubPayload;
  } catch {
    return null;
  }
}

function getCached(): TopHubPayload | null {
  try {
    const raw = localStorage.getItem(CACHE_KEY);
    if (!raw) return null;
    const { ts, payload } = JSON.parse(raw);
    if (Date.now() - ts > CACHE_TTL_MS) return null;
    return payload as TopHubPayload;
  } catch {
    return null;
  }
}

function setCached(payload: TopHubPayload) {
  try {
    localStorage.setItem(
      CACHE_KEY,
      JSON.stringify({ ts: Date.now(), payload })
    );
  } catch {
    // ignore storage errors
  }
}

export async function fetchTopHub(): Promise<TopHubPayload | null> {
  const cached = getCached();
  if (cached) return cached;

  const cdn = await loadFromCDN();
  if (cdn) {
    setCached(cdn);
    return cdn;
  }

  const fallback = await loadFromFallback();
  if (fallback) {
    setCached(fallback);
    return fallback;
  }

  return null;
}
```
