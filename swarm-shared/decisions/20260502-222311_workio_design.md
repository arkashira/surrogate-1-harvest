# workio / design

## Final synthesized implementation

### Core diagnosis (merged)
- **DB thrashing on webhook**: LINE webhook resolves `channelId → tenant` on every request with no caching; retries/backoff multiply DB load.
- **No observability**: Missing request/trace IDs and structured logs make retries and failures hard to correlate.
- **Missing readiness/liveness**: No `/health` or `/ready` endpoints for k8s/PM2; deployments can route traffic before dependencies are usable.
- **Signature verification gaps**: Existence unclear; needs explicit, fast failure path and instrumentation.
- **Payload/timeout risk**: Accepting unbounded JSON can exhaust event loops under burst.

#### 2) Structured logger + request ID
`workio/server/src/middleware/requestLogger.ts`
```ts
import { Request, Response, NextFunction } from 'express';
import { v4 as uuidv4 } from 'uuid';
import winston from 'winston';

export const logger = winston.createLogger({
  level: process.env.LOG_LEVEL || 'info',
  format: winston.format.combine(
    winston.format.timestamp(),
    winston.format.json()
  ),
  transports: [new winston.transports.Console()]
});

export function requestLogger(req: Request, res: Response, next: NextFunction) {
  const requestId = (req.headers['x-request-id'] as string) || uuidv4();
  (req as any).requestId = requestId;

  const start = Date.now();
  res.on('finish', () => {
    logger.info({
      requestId,
      method: req.method,
      url: req.originalUrl,
      status: res.statusCode,
      durationMs: Date.now() - start,
      ip: req.ip,
      userAgent: req.get('User-Agent')
    });
  });

  next();
}

export function getRequestId(req: Request): string {
  return (req as any).requestId || 'unknown';
}
```

import LRU from 'lru-cache';

export interface TenantInfo {
  id: string;
  name: string;
  channelId: string;
}

export const tenantCache = new LRU<string, TenantInfo>({
  max: 500,
  ttl: 10 * 60 * 1000, // 10m
  allowStale: false,
  updateAgeOnGet: true
});

#### 4) LINE signature verification (constant-time)
`workio/server/src/middleware/lineSignature.ts`
```ts
import crypto from 'crypto';

export function verifyLineSignature(
  channelSecret: string,
  body: string,
  signature: string
): boolean {
  if (!channelSecret || !signature) return false;
  const expected = crypto
    .createHmac('sha256', channelSecret)
    .update(body, 'utf8')
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}
```

#### 5) Health/readiness endpoints
`workio/server/src/routes/health.ts`
```ts
import { Router, Request, Response } from 'express';
import { tenantCache } from '../cache/tenantCache';
import db from '../db';

const router = Router();

router.get('/health', (req: Request, res: Response) => {
  // Liveness: process is alive
  res.json({ status: 'ok', uptime: process.uptime() });
});

router.get('/ready', async (req: Request, res: Response) => {
  // Readiness: dependencies usable
  try {
    await db.query('SELECT 1');
    const cacheOk = tenantCache.max > 0;
    res.json({ status: 'ready', db: 'up', cache: cacheOk ? 'up' : 'degraded' });
  } catch (err: any) {
    res.status(503).json({ status: 'not_ready', db: 'down', error: String(err?.message || err) });
  }
});

export default router;
```

#### 6) Updated LINE webhook (cached tenant resolution + signature + observability)
`workio/server/src/routes/lineWebhook.ts`
```ts
import { Router, Request, Response } from 'express';
import { verifyLineSignature } from '../middleware/lineSignature';
import { tenantCache } from '../cache/tenantCache';
import { getRequestId } from '../middleware/requestLogger';
import { logger } from '../middleware/requestLogger';
import db from '../db';

const router = Router();

async function resolveTenant(channelId: string) {
  const cached = tenantCache.get(channelId);
  if (cached) return cached;

  const row = await db.query(
    'SELECT id, name, channel_id FROM tenants WHERE channel_id = $1 LIMIT 1',
    [channelId]
  );
  if (!row.rows.length) return null;

  const tenant = {
    id: row.rows[0].id,
    name: row.rows[0].name,
    channelId: row.rows[0].channel_id
  };
  tenantCache.set(channelId, tenant);
  return tenant;
}

router.post('/line', async (req: Request, res: Response) => {
  const requestId = getRequestId(req);
  const channelSecret = process.env.LINE_CHANNEL_SECRET || '';
  const signature = (req.headers['x-line-signature'] as string) || '';
  // Use raw body if available; fallback to stringified JSON (ensure middleware provides consistent body)
  const bodyRaw = (req as any).rawBody || JSON.stringify(req.body || {});

  if (!channelSecret) {
    logger.warn({ requestId, msg: 'LINE_CHANNEL_SECRET missing' });
    return res.status(500).json({ error: 'server config' });
  }

  if (!verifyLineSignature(channelSecret, bodyRaw, signature)) {
    logger.warn({ requestId, msg: 'invalid LINE signature' });
    return res.status(401).json({ error: 'invalid signature' });
  }

  const events = req.body.events || [];
  if (!events.length) return res.json({ message: 'no events' });

  const channelId = events[0]?.source?.groupId || events[0]?.source?.roomId || events[0]?.source?.userId;
  if (!channelId) {
    logger.warn({ requestId, msg: 'no channelId in events' });
    return res.status(400).json({ error: 'no channelId' });
  }

  const tenant = await resolveTenant(channelId);
  if (!tenant) {
    logger.warn({ requestId, channelId, msg: 'tenant not found' });
    return res.status(404).json({ error: 'tenant not found' });
  }

  // TODO: enqueue or process events scoped to tenant.id
  logger.info({ requestId, tenantId: tenant.id, events: events.length, channelId });

  // Respond quickly; process async if needed
  res.json({ message: 'accepted', requestId });
});

export default router;
```

#### 7) Wire up routes and middleware
`workio/server/src/index.ts` (or app.ts)
```ts
import express from 'express';
import requestLogger from './middleware/requestLogger';
import lineWebhook from './routes/lineWebhook';
import health from './routes/health';

const app = express();

// Limit payload size for webhook endpoints to avoid event-loop pressure
app.use(express.json({ limit: '256kb', verify: (req: any, res, buf) => { (req as any).rawBody = buf.toString(); } }));
app.use(requestLogger);

app.use('/webhook/line', lineWebhook);
app.use('/health', health);
app.use('/ready', health);

const PORT = process.env.PORT || 3000;
app.listen(PORT, () => {
  console.log(`Workio server listening on ${PORT}`);
});
```
