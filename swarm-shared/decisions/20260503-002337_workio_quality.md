### 1) DB schema and migration (enforce correctness at storage)

File: `workio/server/src/db/migrations/2026050401_punch_constraints.sql`

```sql
-- 1) Idempotency keys (24h TTL)
CREATE TABLE IF NOT EXISTS idempotency_keys (
  key          TEXT PRIMARY KEY,
  payload_sha  TEXT NOT NULL,
  response     JSONB NOT NULL,
  created_at   TIMESTAMPTZ DEFAULT NOW()
);
CREATE INDEX IF NOT EXISTS idx_idempotency_created ON idempotency_keys (created_at);

-- 2) De-duplicate existing punches (keep earliest per user/date/type)
--    Use company TZ to determine date (Asia/Bangkok)
DELETE FROM punches
WHERE ctid NOT IN (
  SELECT MIN(ctid)
  FROM punches
  GROUP BY user_id, DATE(timestamp AT TIME ZONE 'Asia/Bangkok'), type
);

-- 3) Add computed date column (stored) for robust constraint/index
ALTER TABLE punches
  ADD COLUMN IF NOT EXISTS punch_date DATE
  GENERATED ALWAYS AS (DATE(timestamp AT TIME ZONE 'Asia/Bangkok')) STORED;

-- 4) One punch per user per date per type (in OR out)
CREATE UNIQUE INDEX IF NOT EXISTS uq_user_punch_date_type
ON punches (user_id, punch_date, type);

-- 5) Optional but recommended: index for fast latest-punch lookup
CREATE INDEX IF NOT EXISTS idx_punches_user_ts ON punches (user_id, timestamp DESC);
```

**Why this wins**:
- Uniqueness is enforced at DB level (prevents races and replay).
- Idempotency table is separate and lightweight.
- `punch_date` is company-TZ based (resolves midnight edge cases).
- Existing duplicates are cleaned before constraint is added.

### 2) Idempotency middleware (API boundary)

File: `workio/server/src/middleware/idempotency.ts`

```ts
import { Request, Response, NextFunction } from 'express';
import { pool } from '../db';

const TTL_MS = 24 * 60 * 60 * 1000;

export async function idempotencyMiddleware(req: Request, res: Response, next: NextFunction) {
  if (req.method !== 'POST') return next();
  const key = req.header('Idempotency-Key');
  if (!key) return next();

  const crypto = require('crypto');
  const payloadSha = crypto.createHash('sha256').update(JSON.stringify(req.body)).digest('hex');

  const { rows } = await pool.query(
    `SELECT response, payload_sha, created_at FROM idempotency_keys WHERE key = $1`,
    [key]
  );

  if (rows.length > 0) {
    const row = rows[0];
    // Same payload: replay stored response
    if (row.payload_sha === payloadSha) {
      return res.status(200).json(row.response);
    }
    // Different payload: conflict (caller should use new key)
    return res.status(409).json({ error: 'Idempotency key used with different payload' });
  }

  // Capture response to store
  const originalJson = res.json.bind(res);
  res.json = (body: any) => {
    // Best-effort async store; don't block response
    pool.query(
      `INSERT INTO idempotency_keys(key, payload_sha, response, created_at)
       VALUES ($1,$2,$3,NOW())
       ON CONFLICT (key) DO UPDATE
       SET payload_sha = EXCLUDED.payload_sha,
           response = EXCLUDED.response,
           created_at = NOW()`,
      [key, payloadSha, JSON.stringify(body)]
    ).catch(() => {});
    return originalJson(body);
  };

  next();
}
```

**Why this wins**:
- Enforces idempotency at the edge for all POSTs.
- Detects payload mismatch (409) instead of silently replaying wrong result.
- Non-blocking store keeps latency low.

### 3) Punch route (clock in/out) with business rules

File: `workio/server/src/routes/punch.ts`

```ts
import { Router } from 'express';
import { pool } from '../db';
import { idempotencyMiddleware } from '../middleware/idempotency';

const router = Router();
router.use(idempotencyMiddleware);

const COMPANY_TZ = 'Asia/Bangkok';

async function latestPunch(userId: number) {
  const { rows } = await pool.query(
    `SELECT type, timestamp FROM punches
     WHERE user_id = $1 ORDER BY timestamp DESC LIMIT 1`,
    [userId]
  );
  return rows[0] || null;
}

router.post('/in', async (req, res) => {
  const { userId } = req.body;
  if (!userId) return res.status(400).json({ error: 'userId required' });

  const last = await latestPunch(userId);
  const now = new Date();
  // Prevent double clock-in on same calendar day (company TZ)
  if (last && last.type === 'in' && new Date(last.timestamp).toDateString() === now.toDateString()) {
    return res.status(409).json({ error: 'Already clocked in today' });
  }

  const { rows } = await pool.query(
    `INSERT INTO punches(user_id, type, timestamp)
     VALUES ($1, 'in', NOW() AT TIME ZONE $2)
     RETURNING id, timestamp`,
    [userId, COMPANY_TZ]
  );

  res.json({ punch: rows[0], status: 'in' });
});

router.post('/out', async (req, res) => {
  const { userId } = req.body;
  if (!userId) return res.status(400).json({ error: 'userId required' });

  const last = await latestPunch(userId);
  if (!last || last.type !== 'in') {
    return res.status(409).json({ error: 'No active clock-in to clock out' });
  }

  const { rows } = await pool.query(
    `INSERT INTO punches(user_id, type, timestamp)
     VALUES ($1, 'out', NOW() AT TIME ZONE $2)
     RETURNING id, timestamp`,
    [userId, COMPANY_TZ]
  );

  res.json({ punch: rows[0], status: 'out' });
});

export default router;
```

**Why this wins**:
- Uses DB-level uniqueness as ultimate safeguard.
- Business rule checks (no double-in same day; must be in to out) provide clear user errors.
- Company-TZ timestamps normalize dates across user locales.
