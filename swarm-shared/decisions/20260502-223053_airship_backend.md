# airship / backend

## Implementation Plan — `airship discover` (backend)

### Scope (highest-value, <2h)
Ship a single cron-safe orchestrator `/opt/axentx/airship/bin/airship-discover` that:
1. Runs `granite-business-research.sh` (if present) and tags outputs `#business-research #knowledge-rag #graph`
2. Executes `knowledge-rag` to query top hub (e.g., "MOC") and emit tagged insights `#knowledge-rag #graph #hub`
3. Produces a CDN-only training manifest (HF dataset file list) with **one** HF API call per run (bypasses rate limits during training)
4. Writes machine-readable outputs to `/opt/axentx/airship/out/` with timestamps and idempotent filenames
5. Logs verbosely to stdout/stderr (cron capture) and exits non-zero on hard failures

### Architecture
- Single Bash entrypoint with `#!/usr/bin/env bash` and `set -euo pipefail`
- HF API used only once: `list_repo_tree` for one date folder → saved JSON
- Training scripts use CDN URLs (`https://huggingface.co/datasets/.../resolve/main/...`) — zero auth/API calls during data loading
- Reuse running Lightning Studio if available (quota-safe)
- All heavy compute offloaded to Lightning/Kaggle/Colab — Mac runs orchestration only

## Implementation

```bash
#!/usr/bin/env bash
# /opt/axentx/airship/bin/airship-discover
# Cron-safe orchestrator: business research + knowledge-rag + CDN manifest
# Usage: airship discover
#        (or run directly via cron: 0 2 * * * /opt/axentx/airship/bin/airship-discover >>/var/log/airship-discover.log 2>&1)

set -euo pipefail
IFS=$'\n\t'

# ---- Config ----
SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
PROJECT_ROOT="$(cd "${SCRIPT_DIR}/.." && pwd)"
OUT_DIR="${PROJECT_ROOT}/out"
MANIFEST_DIR="${PROJECT_ROOT}/manifests"
HF_REPO="${HF_DATASET_REPO:-axentx/surrogate-training}"  # override via env
DATE_LABEL="$(date -u +%Y-%m-%d)"
TIMESTAMP="$(date -u +%Y%m%dT%H%M%SZ)"
LOG_PREFIX="airship-discover[${TIMESTAMP}]"

# ---- Helpers ----
log() { echo "${LOG_PREFIX} $*" >&2; }
fail() { log "ERROR: $*"; exit 1; }

ensure_tools() {
  local tool
  for tool in jq curl; do
    command -v "${tool}" >/dev/null 2>&1 || fail "missing required tool: ${tool}"
  done
}

# ---- Main ----
main() {
  log "starting airship discover (backend orchestrator)"
  ensure_tools

  mkdir -p "${OUT_DIR}" "${MANIFEST_DIR}"

  # 1) Business research
  if [[ -x "${PROJECT_ROOT}/granite-business-research.sh" ]]; then
    log "running granite-business-research.sh"
    "${PROJECT_ROOT}/granite-business-research.sh" \
      > "${OUT_DIR}/business-research-${TIMESTAMP}.jsonl" \
      2> "${OUT_DIR}/business-research-${TIMESTAMP}.stderr" || true
    # Tag outputs
    sed -i.bak 's/$/ #business-research #knowledge-rag #graph/' "${OUT_DIR}/business-research-${TIMESTAMP}.jsonl" 2>/dev/null || \
      sed -i '' 's/$/ #business-research #knowledge-rag #graph/' "${OUT_DIR}/business-research-${TIMESTAMP}.jsonl"
    log "business research complete -> ${OUT_DIR}/business-research-${TIMESTAMP}.jsonl"
  else
    log "granite-business-research.sh not found or not executable; skipping"
  fi

  # 2) Knowledge RAG — top hub insight (e.g., MOC)
  if command -v knowledge-rag >/dev/null 2>&1; then
    log "querying knowledge-rag for top hub (MOC)"
    knowledge-rag query --top-hub MOC \
      > "${OUT_DIR}/top-hub-moc-${TIMESTAMP}.jsonl" \
      2> "${OUT_DIR}/top-hub-moc-${TIMESTAMP}.stderr" || true
    sed -i.bak 's/$/ #knowledge-rag #graph #hub/' "${OUT_DIR}/top-hub-moc-${TIMESTAMP}.jsonl" 2>/dev/null || \
      sed -i '' 's/$/ #knowledge-rag #graph #hub/' "${OUT_DIR}/top-hub-moc-${TIMESTAMP}.jsonl"
    log "top-hub insights complete -> ${OUT_DIR}/top-hub-moc-${TIMESTAMP}.jsonl"
  else
    log "knowledge-rag not available; skipping hub query"
  fi

  # 3) CDN-only training manifest (single HF API call)
  log "generating CDN-only training manifest for ${HF_REPO}/${DATE_LABEL}"
  local tree_json="${OUT_DIR}/hf-tree-${DATE_LABEL}-${TIMESTAMP}.json"
  local manifest_file="${MANIFEST_DIR}/training-manifest-${DATE_LABEL}-${TIMESTAMP}.json"

  # One API call: list top-level folder for the date (non-recursive to avoid pagination explosion)
  if [[ -n "${HF_TOKEN:-}" ]]; then
    curl -s -H "Authorization: Bearer ${HF_TOKEN}" \
      "https://huggingface.co/api/datasets/${HF_REPO}/tree?path=${DATE_LABEL}&recursive=false" \
      > "${tree_json}" || fail "HF API request failed"
  else
    curl -s \
      "https://huggingface.co/api/datasets/${HF_REPO}/tree?path=${DATE_LABEL}&recursive=false" \
      > "${tree_json}" || fail "HF API request failed"
  fi

  # Build manifest: CDN URLs only (no auth required during training)
  jq -r --arg repo "${HF_REPO}" --arg date "${DATE_LABEL}" '
    [.[] | select(.type=="file") | {
      path: .path,
      cdn_url: ("https://huggingface.co/datasets/" + $repo + "/resolve/main/" + $date + "/" + .path),
      size: .size
    }]' "${tree_json}" > "${manifest_file}" || fail "failed to build manifest"

  # Record metadata
  jq -n \
    --arg repo "${HF_REPO}" \
    --arg date "${DATE_LABEL}" \
    --arg ts "${TIMESTAMP}" \
    --arg manifest "${manifest_file}" \
    '{
      dataset_repo: $repo,
      folder_date: $date,
      generated_at: $ts,
      manifest: $manifest,
      note: "CDN-only manifest; training should use cdn_url fields to bypass HF API rate limits",
      strategy: "single HF API call per run; CDN downloads during training"
    }' > "${OUT_DIR}/manifest-meta-${TIMESTAMP}.json"

  local file_count
  file_count=$(jq 'length' "${manifest_file}")
  log "manifest generated: ${file_count} files -> ${manifest_file}"
  log "manifest meta -> ${OUT_DIR}/manifest-meta-${TIMESTAMP}.json"

  # 4) Optional: reuse running Lightning Studio (quota-safe)
  if command -v python3 >/dev/null 2>&1; then
    log "checking for running Lightning Studio to reuse (quota-safe)"
    python3 -c "
import lightning as L
try:
    teamspace = L.Teamspace()
    studios = teamspace.studios
    running = [s for s in studios if s.name == 'surrogate-training' and s.status == 'Running']
    if running:
        print('${LOG_PREFIX} found running studio, reuse recommended')
    else:
        print('${LOG_PREFIX} no running surrogate-training studio found')
except Exception as e:
    print('${LOG_PREFIX} studio check skipped:', e)
" 2>/dev/null || true
  fi

  log "airship discover completed successfully"
}

main "$@"
```
