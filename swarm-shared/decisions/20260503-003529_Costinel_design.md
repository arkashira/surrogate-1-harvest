# Costinel / design

**Final Implementation — Costinel “Top-Hub Signal” Card (read-only)**

---

### Why this scope (resolved)
- **Highest-value, incremental**: surfaces the most-connected hub + 3 actionable signals immediately after a market-analysis/knowledge-rag run.
- **Frontend-only, zero backend/ops changes**: stable, fast, reversible.
- **Static local JSON as source** (swap to API later without UI changes).
- **Timeboxed <2h**: focused on component + data contract + one dashboard mount.

### Component (read-only card)
`src/components/cards/TopHubSignalCard.tsx`
```tsx
import React from "react";
import topHubData from "../../data/top-hub.json";

type Signal = (typeof topHubData.signals)[number];

const severityColors = {
  high: "bg-red-50 border-red-200 text-red-800",
  medium: "bg-amber-50 border-amber-200 text-amber-800",
  low: "bg-slate-50 border-slate-200 text-slate-800",
} as const;

export const TopHubSignalCard: React.FC = () => {
  const { hub, signals, generatedAt } = topHubData;

  return (
    <section
      className="rounded-lg border bg-white p-5 shadow-sm"
      aria-label={`Top hub: ${hub.label}`}
    >
      <header className="mb-4">
        <div className="flex items-center justify-between">
          <h3 className="text-lg font-semibold text-slate-900">Top Hub</h3>
          <span className="text-xs text-slate-500">
            {new Date(generatedAt).toLocaleDateString()}
          </span>
        </div>
        <p className="text-sm text-slate-600">{hub.description}</p>
        <div className="mt-2 flex items-center gap-2">
          <span className="inline-flex items-center rounded-full bg-indigo-50 px-2.5 py-0.5 text-xs font-medium text-indigo-700">
            {hub.label}
          </span>
          <span className="text-xs text-slate-500">{hub.connections} connections</span>
        </div>
      </header>

      <div className="space-y-3">
        <h4 className="text-sm font-medium text-slate-800">Top signals</h4>
        {signals.map((signal: Signal) => (
          <article
            key={signal.id}
            className={`rounded-md border px-3 py-2 ${severityColors[signal.severity]}`}
          >
            <div className="flex items-start justify-between gap-2">
              <div>
                <p className="text-sm font-medium">{signal.title}</p>
                <p className="text-xs">{signal.summary}</p>
              </div>
              <span className="whitespace-nowrap text-xs font-medium capitalize">
                {signal.severity}
              </span>
            </div>
            <div className="mt-2">
              <button
                type="button"
                className="text-xs font-medium text-indigo-700 hover:underline focus:outline-none focus:ring-2 focus:ring-indigo-300"
                onClick={() => alert(`Action: ${signal.action}`)}
              >
                {signal.action}
              </button>
            </div>
          </article>
        ))}
      </div>
    </section>
  );
};
```
