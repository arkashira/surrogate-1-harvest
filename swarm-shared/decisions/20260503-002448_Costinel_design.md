# Costinel / design

## Implementation Plan — Costinel “Top-Hub Signal” Card (read-only)

**Scope**: Frontend-only card (React/TypeScript) that renders the most-connected hub from the existing knowledge-rag graph with contextual insights.  
**Timebox**: <2h  
**Constraints**: zero backend changes; uses existing graph data (assumed available via static JSON or in-memory import).  
**Pattern applied**: top-hub doc insight (#knowledge-rag #graph #hub)

### 3) Component implementation

`src/components/CostinelTopHubCard.tsx`

```tsx
import React from 'react';
import { TopHub } from '../types/knowledgeRag';
import graph from '../data/knowledgeRagGraph.json';
import styles from './CostinelTopHubCard.module.css';

function findTopHub(): TopHub | null {
  if (!Array.isArray(graph?.hubs) || graph.hubs.length === 0) return null;
  // most connections wins
  return graph.hubs.reduce((best, cur) =>
    (cur.connections ?? 0) > (best?.connections ?? 0) ? cur : best
  , null as TopHub | null);
}

const CostinelTopHubCard: React.FC = () => {
  const hub = findTopHub();

  if (!hub) {
    return (
      <div className={styles.card}>
        <div className={styles.header}>
          <h3 className={styles.title}>Top Hub</h3>
        </div>
        <div className={styles.body}>
          <p className={styles.empty}>No hub data available.</p>
        </div>
      </div>
    );
  }

  return (
    <div className={styles.card}>
      <div className={styles.header}>
        <h3 className={styles.title}>Top Hub</h3>
        <span className={styles.badge}>{hub.type ?? 'hub'}</span>
      </div>

      <div className={styles.body}>
        <div className={styles.hubMeta}>
          <div className={styles.hubLabel}>{hub.label}</div>
          <div className={styles.hubConnections}>
            {hub.connections ?? 0} connections
          </div>
        </div>

        {Array.isArray(hub.topInsights) && hub.topInsights.length > 0 && (
          <div className={styles.insights}>
            <h4 className={styles.insightsTitle}>Key insights</h4>
            <ul className={styles.insightsList}>
              {hub.topInsights.map((text, idx) => (
                <li key={idx} className={styles.insightItem}>
                  <span className={styles.bullet} />
                  {text}
                </li>
              ))}
            </ul>
          </div>
        )}

        <div className={styles.footer}>
          <small className={styles.updated}>
            Updated {new Date(hub.lastUpdated).toLocaleString()}
          </small>
        </div>
      </div>
    </div>
  );
};

export default CostinelTopHubCard;
```
