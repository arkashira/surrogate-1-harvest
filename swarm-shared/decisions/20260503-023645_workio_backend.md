# workio / backend

## Final Synthesis — Single Correct + Actionable Answer

**Root cause**: duplicate punches occur because the system relies on application-level checks instead of database-level guarantees, and it does not persist or enforce idempotency keys from LINE webhook retries.

**Single source of truth**: the database must enforce one active punch per `(employee_id, punch_date, punch_type)` and deduplicate by `line_message_id`.

---

### 1. Schema changes (apply once)

File: `workio/server/src/db/schema.sql`

```sql
-- Add idempotency column if missing
ALTER TABLE punches
  ADD COLUMN IF NOT EXISTS line_message_id VARCHAR(255) NULL;

-- One active punch per employee per day per type
CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_unique_active
  ON punches (employee_id, punch_date, punch_type)
  WHERE deleted_at IS NULL;

-- Idempotency: one LINE message maps to one punch
CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_line_message_id
  ON punches (line_message_id)
  WHERE line_message_id IS NOT NULL AND deleted_at IS NULL;
```

### 2. Route implementation (atomic + idempotent)

File: `workio/server/src/routes/punch.ts`

```ts
import { Request, Response } from 'express';
import { pool } from '../db';

export async function handlePunch(req: Request, res: Response) {
  const { employee_id, type } = req.body;
  const line_message_id = req.body.line_message_id || (req.headers['x-line-message-id'] as string);

  if (!employee_id || !type) {
    return res.status(400).json({ error: 'employee_id and type required' });
  }

  const now = new Date();
  const punch_date = now.toISOString().split('T')[0]; // YYYY-MM-DD
  const clock_in_time = type === 'in' ? now : null;

  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // 1) Try insert with natural-key constraint first (fast path)
    const insertSQL = `
      INSERT INTO punches (employee_id, punch_date, punch_type, clock_in_time, line_message_id, created_at)
      VALUES ($1, $2, $3, $4, $5, NOW())
      ON CONFLICT (employee_id, punch_date, punch_type)
      WHERE deleted_at IS NULL
      DO NOTHING
      RETURNING *
    `;
    const insertRes = await client.query(insertSQL, [
      employee_id,
      punch_date,
      type,
      clock_in_time,
      line_message_id || null,
    ]);

    if (insertRes.rowCount > 0) {
      await client.query('COMMIT');
      return res.json({ punch: insertRes.rows[0], status: 'created' });
    }

    // 2) Conflict on natural key: fetch existing by natural key
    const existingByNatural = await client.query(
      'SELECT * FROM punches WHERE employee_id = $1 AND punch_date = $2 AND punch_type = $3 AND deleted_at IS NULL',
      [employee_id, punch_date, type]
    );

    // If we have a line_message_id, ensure it matches the existing row (strict idempotency)
    if (line_message_id) {
      const byMsg = await client.query(
        'SELECT * FROM punches WHERE line_message_id = $1 AND deleted_at IS NULL',
        [line_message_id]
      );

      // If line_message_id maps to a different punch, reject (collision/integrity)
      if (byMsg.rowCount > 0) {
        const msgPunch = byMsg.rows[0];
        const naturalPunch = existingByNatural.rows[0];

        // If same row, treat as duplicate; otherwise integrity violation
        if (naturalPunch && msgPunch.id === naturalPunch.id) {
          await client.query('COMMIT');
          return res.json({ punch: naturalPunch, status: 'duplicate' });
        }

        await client.query('ROLLBACK');
        return res.status(409).json({
          error: 'Idempotency key conflict',
          detail: 'line_message_id already used for a different punch',
        });
      }

      // 3) No row with this line_message_id, but natural key exists.
      //    Attach line_message_id to the existing punch (idempotent binding) if allowed.
      if (existingByNatural.rowCount > 0) {
        const updateSQL = `
          UPDATE punches
          SET line_message_id = $1
          WHERE id = $2
            AND line_message_id IS NULL
          RETURNING *
        `;
        const updateRes = await client.query(updateSQL, [line_message_id, existingByNatural.rows[0].id]);
        await client.query('COMMIT');

        if (updateRes.rowCount > 0) {
          return res.json({ punch: updateRes.rows[0], status: 'idempotency_bound' });
        }

        // Another request bound it concurrently — return current state
        const refreshed = await client.query(
          'SELECT * FROM punches WHERE id = $1',
          [existingByNatural.rows[0].id]
        );
        return res.json({ punch: refreshed.rows[0], status: 'duplicate' });
      }
    }

    // 4) Natural key conflict but no row found (edge race) — try line_message_id unique insert
    if (line_message_id) {
      const msgInsert = await client.query(
        `INSERT INTO punches (employee_id, punch_date, punch_type, clock_in_time, line_message_id, created_at)
         VALUES ($1, $2, $3, $4, $5, NOW())
         ON CONFLICT (line_message_id)
         WHERE line_message_id IS NOT NULL AND deleted_at IS NULL
         DO NOTHING
         RETURNING *`,
        [employee_id, punch_date, type, clock_in_time, line_message_id]
      );

      if (msgInsert.rowCount > 0) {
        await client.query('COMMIT');
        return res.json({ punch: msgInsert.rows[0], status: 'created' });
      }

      // Still duplicate — fetch and return
      const final = await client.query(
        'SELECT * FROM punches WHERE line_message_id = $1 AND deleted_at IS NULL',
        [line_message_id]
      );
      await client.query('COMMIT');
      return res.json({ punch: final.rows[0], status: 'duplicate' });
    }

    // Fallback: return existing natural punch
    await client.query('COMMIT');
    return res.json({
      punch: existingByNatural.rows[0] || null,
      status: existingByNatural.rowCount > 0 ? 'duplicate' : 'not_found',
    });
  } catch (err) {
    await client.query('ROLLBACK');
    console.error('Punch handling error:', err);
    return res.status(500).json({ error: 'Internal server error' });
  } finally {
    client.release();
  }
}
```

### 3. Required caller behavior (LINE webhook)

- Extract `message.id` (or webhook event ID) from LINE and provide it as `line_message_id` in the punch request body or header `x-line-message-id`.
- Do not retry with a different `line_message_id` for the same event.
