# Costinel / design

## Final Synthesis — Top-Hub Signal Panel (CDN-first, <2h)

**Goal:** Add a non-blocking Top-Hub Signal Panel that surfaces the most-connected hub with 3–5 actionable cost-governance signals, uses CDN-first data with graceful fallback, and never blocks dashboard render.

---

### Chosen approach (merged + resolved)

- **Data source:** CDN-first JSON (HF CDN or repo `public/data/`) with **local fallback file** committed to the repo for reliability and offline dev.
- **Content:** Include both **hub metadata** (name, score, rationale) **and 3–5 actionable signals** per Candidate 2, because a panel that only says “MOC” is less useful than one that says “MOC — do these 3 things now.”
- **Caching strategy:** Memory (5m) + `localStorage` (1h). Background refresh after initial render so UI is instant.
- **UI behavior:** Client-only render for CDN fetch; SSR-safe skeleton; never blocks dashboard. Stale badge when using fallback or stale cache.
- **Ops:** Optional cron-friendly script to regenerate the JSON from the knowledge graph.

---

### File layout (minimal, 2h scope)

```
public/data/hubs/top.json            # committed fallback + CDN target
src/services/topHubSignal.ts         # fetch + cache logic
src/components/TopHubSignalPanel.tsx # non-blocking UI

```

### 2) Service layer (final)

// src/services/topHubSignal.ts
const CDN_URL =
  'https://huggingface.co/datasets/axentx/costinel-knowledge/resolve/main/hubs/top.json';
const LOCAL_JSON_PATH = '/data/hubs/top.json'; // fallback from public/
const LOCAL_KEY = 'costinel:topHub:cache';
const MEMO_TTL_MS = 5 * 60 * 1000; // 5m
const STALE_TTL_MS = 60 * 60 * 1000; // 1h

let memo: { data: TopHubInsight; ts: number } | null = null;

export interface TopHubSignal {
  id: string;
  title: string;
  severity: 'low' | 'medium' | 'high';
  description: string;
  action: string;
  context?: string;
  docsHref?: string;
}

export interface TopHubInsight {
  hub: string;
  score: number;
  rationale: string;
  updated_at: string;
  signals: TopHubSignal[];
  relatedDocs: Array<{ title: string; href: string }>;
  generatedAt: string;
}

async function fetchFromCDN(): Promise<TopHubInsight> {
  const res = await fetch(CDN_URL, { cache: 'no-store' });
  if (!res.ok) throw new Error(`CDN fetch failed: ${res.status}`);
  return (await res.json()) as TopHubInsight;
}

async function fetchLocalFallback(): Promise<TopHubInsight> {
  // In SSR/build, this resolves to public/ file; in dev it's available.
  const res = await fetch(LOCAL_JSON_PATH, { cache: 'default' });
  if (!res.ok) throw new Error(`Local fallback missing: ${res.status}`);
  return (await res.json()) as TopHubInsight;
}

export async function getTopHubInsights(): Promise<TopHubInsight | null> {
  // Memory cache (fast)
  if (memo && Date.now() - memo.ts < MEMO_TTL_MS) {
    return memo.data;
  }

  // Try localStorage stale cache
  try {
    const raw = localStorage.getItem(LOCAL_KEY);
    if (raw) {
      const parsed = JSON.parse(raw);
      if (Date.now() - parsed.ts < STALE_TTL_MS) {
        memo = { data: parsed.data, ts: parsed.ts };
        return parsed.data;
      }
    }
  } catch (e) {
    // ignore
  }

  // Try CDN first
  try {
    const data = await fetchFromCDN();
    const payload = { data, ts: Date.now() };
    memo = payload;
    try {
      localStorage.setItem(LOCAL_KEY, JSON.stringify(payload));
    } catch (e) {
      // ignore storage limits
    }
    return data;
  } catch (err) {
    // Fallback to local committed JSON
    try {
      const data = await fetchLocalFallback();
      const payload = { data, ts: Date.now() };
      memo = payload;
      try {
        localStorage.setItem(LOCAL_KEY, JSON.stringify(payload));
      } catch (e) {
        // ignore
      }
      return data;
    } catch (e2) {
      // Ultimate fallback: minimal hardcoded object
      const fallback: TopHubInsight = {
        hub: 'MOC',
        score: 0.94,
        rationale: 'Most-connected node across cost, ops, and compliance signals (fallback)',
        updated_at: new Date().toISOString(),
        signals: [
          {
            id: 'fallback-1',
            title: 'Check MOC cost drivers',
            severity: 'medium',
            description: 'Review recent cost anomalies for MOC-related resources.',
            action: 'Open cost explorer for MOC',
            context: 'Fallback guidance',
            docsHref: '/docs/hubs/moc'
          }
        ],
        relatedDocs: [
          { title: 'MOC Overview', href: '/docs/hubs/moc' },
          { title: 'Cost Impact Analysis', href: '/docs/analysis/cost-moc' }
        ],
        generatedAt: new Date().toISOString()
      };
      memo = { data: fallback, ts: Date.now() };
      return fallback;
    }
  }
}
