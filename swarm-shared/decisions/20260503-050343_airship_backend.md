## Implementation Plan (≤2h)

### 1. Add manifest generation script (Mac orchestration)
- Single API call to `list_repo_tree` for a date folder → save JSON of `{path, size, sha256?}`  
- Embed JSON into surrogate training repo as `data/manifest-{date}.json`  
- Mark as code-generated; commit alongside training script changes.

### 2. Update training data loader
- Replace `load_dataset(streaming=True, repo_type="dataset", ...)`  
- Use `datasets.load_dataset` **only for local files** or read parquet via `pandas.read_parquet` + CDN URLs.  
- Build `IterableDataset` that streams from CDN URLs (`https://huggingface.co/datasets/{repo}/resolve/main/{path}`) with `aiohttp` or `fsspec` + `pyarrow` projection to `{prompt, response}`.  
- No `list_repo_files` or recursive API calls during training.

### 3. Lightning Studio integration
- Reuse running studio if present (`Teamspace.studios` check).  
- Pass manifest path as script arg or env var.  
- Ensure `target.run()` only after checking studio status; restart if idle-stopped.

### 4. Fallback & limits
- If CDN fetch fails (404/403), skip file and log warning (do not retry in loop).  
- Respect `max_connections` and `backoff` for CDN.

### `scripts/generate_manifest.py` (run on Mac)
```python
#!/usr/bin/env python3
"""
Generate CDN-only manifest for a date folder in a HF dataset repo.
Run from Mac after rate-limit window clears.
"""
import json
import os
import sys
from datetime import datetime

from huggingface_hub import HfApi

HF_REPO = os.getenv("HF_DATASET_REPO", "my-org/surrogate-data")
DATE_FOLDER = os.getenv("DATE_FOLDER", datetime.utcnow().strftime("%Y-%m-%d"))
OUT_PATH = os.getenv("OUT_MANIFEST", f"data/manifest-{DATE_FOLDER}.json")

def main() -> None:
    api = HfApi()
    # Single non-recursive call per folder (avoids pagination)
    entries = api.list_repo_tree(
        repo_id=HF_REPO,
        path=DATE_FOLDER,
        repo_type="dataset",
        recursive=False,
    )
    files = [
        {
            "path": e.path,
            "size": getattr(e, "size", None),
            "lfs": getattr(e, "lfs", None),
        }
        for e in entries
        if e.type == "file" and e.path.lower().endswith(".parquet")
    ]
    os.makedirs(os.path.dirname(OUT_PATH), exist_ok=True)
    with open(OUT_PATH, "w") as f:
        json.dump({"repo": HF_REPO, "date": DATE_FOLDER, "files": files}, f, indent=2)
    print(f"Wrote {len(files)} files to {OUT_PATH}")

if __name__ == "__main__":
    main()
```

### `surrogate/train/data.py` (CDN-only iterable dataset)
```python
from __future__ import annotations

import io
import json
import logging
from pathlib import Path
from typing import Iterator, Optional

import aiohttp
import pandas as pd
import pyarrow as pa
import pyarrow.parquet as pq
from datasets import IterableDataset
from huggingface_hub import hf_hub_download

logger = logging.getLogger(__name__)

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

class CDNParquetIterable(IterableDataset):
    """
    Streams rows from parquet files via CDN URLs (zero HF API calls during training).
    Manifest is a local JSON produced by generate_manifest.py.
    """

    def __init__(
        self,
        manifest_path: str | Path,
        repo: Optional[str] = None,
        max_concurrent: int = 8,
        columns: tuple[str, str] = ("prompt", "response"),
    ):
        super().__init__()
        self.manifest_path = Path(manifest_path)
        with open(self.manifest_path) as f:
            self.manifest = json.load(f)
        self.repo = repo or self.manifest.get("repo")
        self.files = [f["path"] for f in self.manifest.get("files", [])]
        self.columns = columns
        self.max_concurrent = max_concurrent
        if not self.repo:
            raise ValueError("repo must be provided or present in manifest")

    def _stream_file(self, path: str) -> Iterator[dict]:
        url = CDN_TEMPLATE.format(repo=self.repo, path=path)
        try:
            # Use hf_hub_download as robust CDN fetch (caches, no auth for public)
            local_path = hf_hub_download(
                repo_id=self.repo,
                filename=path,
                repo_type="dataset",
                cache_dir=None,  # use default HF cache
                force_download=False,
            )
            table = pq.read_table(local_path, columns=self.columns)
        except Exception as e:
            # Fallback: direct CDN read via pandas (avoids API)
            logger.warning("hf_hub_download failed for %s (%s), trying pandas CDN read", path, e)
            df = pd.read_parquet(url, columns=self.columns)
            table = pa.Table.from_pandas(df)

        for batch in table.to_batches(max_chunksize=1024):
            cols = {c: batch.column(c).to_pylist() for c in self.columns}
            for i in range(len(cols[self.columns[0]])):
                yield {c: cols[c][i] for c in self.columns}

    def __iter__(self) -> Iterator[dict]:
        for path in self.files:
            yield from self._stream_file(path)
```

### `surrogate/train/train.py` (Lightning entrypoint)
```python
#!/usr/bin/env python3
import argparse
import logging
import os

import lightning as L
from lightning.pytorch.callbacks import ModelCheckpoint
from torch.utils.data import DataLoader

from surrogate.train.data import CDNParquetIterable
from surrogate.train.model import SurrogateModel  # your model

logger = logging.getLogger(__name__)

def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("--manifest", default=os.getenv("MANIFEST_PATH", "data/manifest-latest.json"))
    parser.add_argument("--max-steps", type=int, default=1000)
    parser.add_argument("--batch-size", type=int, default=16)
    parser.add_argument("--num-workers", type=int, default=4)
    args = parser.parse_args()

    # Reuse running studio if available (saves quota)
    try:
        from lightning.pytorch import Studio
        from lightning.pytorch.cli import LightningCLI
        studios = Studio.list()
        running = [s for s in studios if s.name == os.getenv("STUDIO_NAME", "surrogate-train") and s.status == "running"]
        if running:
            logger.info("Reusing running studio: %s", running[0].name)
    except Exception:
        logger.debug("Studio reuse check skipped", exc_info=True)

    dataset = CDNParquetIterable(manifest_path=args.manifest)
    loader = DataLoader(
        dataset,
        batch_size=args.batch_size,
        num_workers=args.num_workers,
        pin_memory=True,
        # IterableDataset handles streaming; avoid shuffle that buffers entire epoch
    )

    model = SurrogateModel()
    checkpoint = ModelCheckpoint(monitor="loss", mode="min", save_top_k=1)
    trainer = L.Trainer(
        max_steps=args.max_steps,
        callbacks=[checkpoint],
        log_every_n_steps=10,
        accelerator="gpu" if L.accelerators.find_usable_cuda_devices() else "cpu",
        devices=1,
    )

    trainer.fit(model, loader)

if __name__ == "__main__":
    main()
```
