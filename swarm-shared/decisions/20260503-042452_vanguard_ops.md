# vanguard / ops

## 1. Diagnosis

- No content-addressed manifest (file list + SHA256) exists → CDN-only fetching impossible; training scripts cannot avoid HF API rate limits during data loading.
- Training/data-loading scripts likely use `load_dataset(streaming=True)` or recursive `list_repo_files` on heterogeneous repos → triggers HF API 429s and non-reproducible epochs.
- No deterministic repo-to-sibling mapping for HF ingestion → commit cap (128/hr/repo) blocks bulk writes when mirroring large datasets.
- No Lightning Studio reuse guard → each run risks creating new studios and burning 80hr/mo quota; idle-stop kills training if not handled.
- No CDN-bypass downloader for public dataset files → every file fetch hits `/api/` auth-check limits instead of CDN tier.

## 2. Proposed change

Add a small, ops-focused CLI tool that:
- Creates a content-addressed manifest for a given HF dataset repo + date folder (single `list_repo_tree` call, saved as JSON with SHA256 per file).
- Generates a deterministic repo-to-sibling mapping (hash-slug → sibling index) for commit-cap scaling.
- Provides a CDN-only download helper (raw `https://huggingface.co/datasets/.../resolve/main/...`) for files listed in the manifest.
- Includes a Lightning Studio reuse/idle-check guard to prevent quota waste.

Scope: new file `/opt/axentx/vanguard/ops/hf_cdn_manifest.py` + optional companion script `/opt/axentx/vanguard/ops/run_hf_cdn_manifest.sh`.

# /opt/axentx/vanguard/ops/hf_cdn_manifest.py
#!/usr/bin/env python3
"""
Ops helper: produce CDN-fetch manifest for an HF dataset repo folder
and deterministic sibling mapping to avoid HF commit caps and API 429s.

Usage:
  python3 hf_cdn_manifest.py \
    --repo <repo> \
    --folder <date-folder> \
    --out-manifest <path.json> \
    --sibling-count 5
"""

import argparse
import hashlib
import json
import os
import sys
import time
from typing import Dict, List, Optional

try:
    from huggingface_hub import HfApi, hf_hub_download
except ImportError:
    print("Install: pip install huggingface_hub")
    sys.exit(1)

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def list_folder_files(repo: str, folder: str) -> List[Dict]:
    """Single non-recursive tree call for a folder; returns name+sha256."""
    api = HfApi()
    tree = api.list_repo_tree(repo=repo, path=folder, recursive=False)
    files = []
    for entry in tree:
        if entry.type != "file":
            continue
        # entry may not include sha256; fetch via hf_hub_download metadata if needed
        files.append({
            "path": entry.path,
            "size": getattr(entry, "size", None),
            "lfs": getattr(entry, "lfs", None),
        })
    return files

def sha256_file(path: str) -> str:
    h = hashlib.sha256()
    with open(path, "rb") as f:
        for chunk in iter(lambda: f.read(8192), b""):
            h.update(chunk)
    return h.hexdigest()

def build_manifest(repo: str, folder: str, out_path: str, max_files: int = 50_000) -> None:
    files = list_folder_files(repo, folder)
    if not files:
        print(f"No files found in {repo}/{folder}")
        return

    manifest = {
        "repo": repo,
        "folder": folder,
        "created_unix": int(time.time()),
        "files": [],
    }

    for idx, f in enumerate(files[:max_files]):
        cdn_url = CDN_TEMPLATE.format(repo=repo, path=f["path"])
        item = {
            "idx": idx,
            "path": f["path"],
            "cdn_url": cdn_url,
            "size": f["size"],
        }
        manifest["files"].append(item)

    # Optional: local sha256 if files already cached via hf_hub_download
    # (skip by default to avoid downloads; populate lazily during ingestion)
    manifest["sha256_provided"] = False

    os.makedirs(os.path.dirname(os.path.abspath(out_path)), exist_ok=True)
    with open(out_path, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Manifest written: {out_path} ({len(manifest['files'])} files)")

def sibling_for_slug(slug: str, sibling_count: int) -> int:
    """Deterministic repo-to-sibling index to spread HF commit writes."""
    if sibling_count <= 1:
        return 0
    digest = hashlib.sha256(slug.encode()).digest()
    return int.from_bytes(digest[:4], "little") % sibling_count

def cdn_download_file(cdn_url: str, dest_path: str, retries: int = 3, backoff: int = 5) -> bool:
    """CDN-only fetch (no Authorization header) to bypass /api/ rate limits."""
    import requests
    os.makedirs(os.path.dirname(os.path.abspath(dest_path)), exist_ok=True)
    for attempt in range(1, retries + 1):
        try:
            # Do NOT send Authorization header; use CDN public URL
            resp = requests.get(cdn_url, timeout=60, stream=True)
            resp.raise_for_status()
            with open(dest_path, "wb") as f:
                for chunk in resp.iter_content(chunk_size=8192):
                    f.write(chunk)
            return True
        except Exception as exc:
            if attempt == retries:
                print(f"Failed to download {cdn_url}: {exc}")
                return False
            print(f"Retry {attempt}/{retries} for {cdn_url} in {backoff}s...")
            time.sleep(backoff * attempt)
    return False

def lightning_studio_reuse(
    name: str,
    target_machine: str = "L40S",
    max_idle_restarts: int = 3,
):
    """
    Reuse a running Lightning Studio if present; otherwise create one.
    Checks status to avoid quota waste and handles idle-stop restarts.
    """
    try:
        from lightning import Studio, Machine, Teamspace
    except ImportError:
        print("lightning not installed; skipping Lightning Studio helpers.")
        return None

    teamspace = Teamspace()
    running = None
    for s in teamspace.studios:
        if s.name == name:
            running = s
            break

    if running is not None:
        # If stopped, restart (idle-stop kills training)
        if running.status != "running":
            print(f"Studio {name} is {running.status}; restarting...")
            machine = Machine[target_machine] if hasattr(Machine, target_machine) else Machine.L40S
            running.start(machine=machine)
        else:
            print(f"Reusing running studio: {name}")
        return running

    # Not found: create
    print(f"Creating studio: {name}")
    machine = Machine[target_machine] if hasattr(Machine, target_machine) else Machine.L40S
    return Studio(
        name=name,
        machine=machine,
        create_ok=True,
    )

def main() -> None:
    parser = argparse.ArgumentParser(description="Build CDN manifest for HF dataset folder.")
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g., 'username/dataset')")
    parser.add_argument("--folder", required=True, help="Folder path in repo (e.g., 'batches/mirror-merged/2026-04-29')")
    parser.add_argument("--out-manifest", required=True, help="Output manifest JSON path")
    parser.add_argument("--sibling-count", type=int, default=5, help="Number of HF sibling repos for commit-cap scaling")
    parser.add_argument("--cdn-download-to", help="If provided, download all files to this directory (CDN-only)")
    args = parser.parse_args()

    build_manifest(args.repo, args.folder, args.out_manifest)
