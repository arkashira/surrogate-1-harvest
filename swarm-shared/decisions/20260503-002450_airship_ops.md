### 1) Mac-side manifest generator
`scripts/build_hf_manifest.py`
```python
#!/usr/bin/env python3
"""
Generate CDN-only manifest for HuggingFace dataset folders.
Usage:
  python scripts/build_hf_manifest.py \
    --repo datasets/axentx/surrogate-mirror \
    --date 2026-05-03 \
    --out manifest/2026-05-03/filelist.json
"""
import argparse
import json
import os
import time
from pathlib import Path
from typing import List, Dict

from huggingface_hub import HfApi, hf_hub_download

API = HfApi()

def list_date_folder(repo: str, date: str) -> List[Dict]:
    # Single API call per folder (non-recursive)
    try:
        tree = API.list_repo_tree(repo=repo, path=date, recursive=False)
    except Exception as e:
        # If repo is actually a dataset repo, prefix may need "datasets/"
        if not repo.startswith("datasets/"):
            return list_date_folder(f"datasets/{repo}", date)
        raise
    items = []
    for entry in tree:
        if entry.type == "file":
            items.append({
                "repo": repo,
                "path": f"{date}/{entry.path.split('/')[-1]}",
                "url": f"https://huggingface.co/datasets/{repo}/resolve/main/{date}/{entry.path.split('/')[-1]}",
                "size": getattr(entry, "size", None),
                "lfs": getattr(entry, "lfs", None)
            })
    return items

def build_manifest(repo: str, date: str, out_path: Path):
    out_path.parent.mkdir(parents=True, exist_ok=True)
    items = list_date_folder(repo, date)
    manifest = {
        "repo": repo,
        "date": date,
        "generated_at": time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime()),
        "files": items
    }
    with open(out_path, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Wrote {len(items)} files to {out_path}")

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Build CDN manifest for HF dataset folder")
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g. axentx/surrogate-mirror)")
    parser.add_argument("--date", required=True, help="Date folder (e.g. 2026-05-03)")
    parser.add_argument("--out", required=True, help="Output JSON path")
    args = parser.parse_args()
    build_manifest(args.repo, args.date, Path(args.out))
```

### 2) CDN-only iterable dataset
`surrogate/data/hf_cdn_dataset.py`
```python
#!/usr/bin/env python3
import json
import logging
from typing import Dict, Iterator, Optional

import requests
from torch.utils.data import IterableDataset

logger = logging.getLogger(__name__)

class HFCDNDataset(IterableDataset):
    """
    Stream files from HF CDN using a manifest.
    Manifest format:
    {
      "repo": "...",
      "date": "...",
      "files": [
        {"repo": "...", "path": "...", "url": "...", ...}
      ]
    }
    """
    def __init__(self, manifest_path: str, start: int = 0, end: Optional[int] = None):
        super().__init__()
        self.manifest_path = manifest_path
        self.start = start
        self.end = end
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.files = self.manifest["files"]
        if self.end is None:
            self.end = len(self.files)
        self.files = self.files[self.start:self.end]

    def _project_record(self, raw: Dict) -> Dict:
        # Keep only prompt/response; drop heterogeneous schema fields.
        # Adjust projection to match your surrogate schema.
        return {
            "prompt": raw.get("prompt") or raw.get("input") or raw.get("text") or "",
            "response": raw.get("response") or raw.get("output") or raw.get("completion") or "",
        }

    def _stream_file(self, entry: Dict) -> Iterator[Dict]:
        url = entry["url"]
        try:
            # CDN fetch — no Authorization header (bypasses API rate limits)
            resp = requests.get(url, timeout=30, stream=False)
            resp.raise_for_status()
            # Assume newline-delimited JSON or single JSON array/object.
            # Adjust parser to match your file format (parquet/jsonl/json).
            # For parquet, you'd need pyarrow; for JSONL:
            text = resp.text
            for line in text.strip().splitlines():
                if not line.strip():
                    continue
                try:
                    obj = json.loads(line)
                    yield self._project_record(obj)
                except json.JSONDecodeError:
                    logger.warning("Invalid JSON line in %s", url)
                    continue
        except Exception as exc:
            logger.error("Failed to fetch %s: %s", url, exc)

    def __iter__(self) -> Iterator[Dict]:
        for entry in self.files:
            yield from self._stream_file(entry)
```

            logger.info("Reusing running studio: %s", s.name)

    return Studio(name=name, create_ok=True).start(machine=machine)
