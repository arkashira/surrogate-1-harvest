# workio / design

## 1. Diagnosis

- **Missing HMAC-SHA256 signature verification** — `/webhook/line` accepts unverified payloads, enabling spoofing and replay.
- **No idempotency/replay protection** — duplicate LINE events (retries, user double-taps) can create duplicate clock-in/out records.
- **Loose payload schema** — downstream handlers rely on implicit shape; malformed events can throw 500s and break LINE retry flow.
- **Tenant resolution fragile** — tenant is inferred from channel ID or webhook domain; no explicit tenant-scoped validation means cross-tenant collisions if channels are misconfigured.
- **No structured audit trail for webhook ingestion** — failures are opaque; retries amplify load without observability.

## 2. Proposed change

**File**: `/opt/axentx/workio/server/src/routes/webhook/line.ts`  
**Scope**: Add a single middleware layer that (a) verifies `X-Line-Signature`, (b) validates payload shape with Zod, (c) deduplicates by `X-Line-Delivery`/event `timestamp`+`source.userId`+`type`, (d) attaches resolved tenantId, and (e) returns `200` fast (so LINE stops retrying) while delegating business handling to a queue/job.

```ts
// /opt/axentx/workio/server/src/routes/webhook/line.ts
import { Router, Request, Response, NextFunction } from 'express';
import crypto from 'crypto';
import { z } from 'zod';
import { getRawBody } from '../../middleware/rawBody'; // tiny helper below
import { db } from '../../db'; // your db client
import { logger } from '../../logger';

const router = Router();

// ── Helpers ──────────────────────────────────────────────────────────────
const CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';
if (!CHANNEL_SECRET) {
  logger.warn('LINE_CHANNEL_SECRET missing; webhook verification disabled');
}

function verifyLineSignature(rawBody: Buffer, signature: string): boolean {
  if (!CHANNEL_SECRET) return false;
  const expected = crypto
    .createHmac('sha256', CHANNEL_SECRET)
    .update(rawBody)
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

// ── Schema ───────────────────────────────────────────────────────────────
const LineWebhookSchema = z.object({
  destination: z.string(),
  events: z.array(
    z.object({
      type: z.string(),
      mode: z.enum(['active', 'standby']),
      timestamp: z.number(),
      source: z.object({
        type: z.string(),
        userId: z.string().optional(),
        groupId: z.string().optional(),
        roomId: z.string().optional(),
      }),
      webhookEventId: z.string(),
      deliveryContext: z.object({ isRedelivery: z.boolean() }),
      // Minimal required fields; extend per event type as needed
      message: z.object({ type: z.string(), id: z.string() }).optional(),
      postback: z.object({ data: z.string() }).optional(),
    })
  ),
});

type LineWebhook = z.infer<typeof LineWebhookSchema>;

// ── Idempotency check ───────────────────────────────────────────────────
async function isDuplicateEvent(event: LineWebhook['events'][number], tenantId: string): Promise<boolean> {
  const key = `${tenantId}:${event.webhookEventId}`;
  // Use Redis or DB unique constraint; here we use a simple DB table:
  // CREATE TABLE line_webhook_events (id TEXT PRIMARY KEY, tenant_id TEXT, created_at TIMESTAMP DEFAULT NOW());
  const exists = await db
    .selectFrom('line_webhook_events')
    .select('id')
    .where('id', '=', key)
    .executeTakeFirst();
  return !!exists;
}

async function markEventSeen(event: LineWebhook['events'][number], tenantId: string): Promise<void> {
  const key = `${tenantId}:${event.webhookEventId}`;
  await db
    .insertInto('line_webhook_events')
    .values({ id: key, tenant_id: tenantId })
    .onConflict((oc) => oc.column('id').doNothing())
    .execute();
}

// ── Tenant resolution ───────────────────────────────────────────────────
function resolveTenantId(req: Request): string | null {
  // Strategy 1: explicit header (preferred)
  const tenantHeader = req.headers['x-workio-tenant-id'];
  if (typeof tenantHeader === 'string' && tenantHeader) return tenantHeader;

  // Strategy 2: map destination(channel) -> tenant via DB
  // (Implement if you store LINE channel <-> tenant mapping)
  return null;
}

// ── Middleware: verify + validate + idempotency ─────────────────────────
router.post('/line', getRawBody, async (req: Request & { rawBody?: Buffer }, res: Response, next: NextFunction) => {
  try {
    const signature = req.headers['x-line-signature'] as string;
    const rawBody = req.rawBody;

    if (!rawBody || !verifyLineSignature(rawBody, signature)) {
      logger.warn({ signaturePresent: !!signature }, 'Invalid LINE signature');
      return res.status(401).json({ error: 'Invalid signature' });
    }

    const payload = LineWebhookSchema.parse(JSON.parse(rawBody.toString()));
    const tenantId = resolveTenantId(req);
    if (!tenantId) {
      logger.warn({ destination: payload.destination }, 'Tenant resolution failed');
      return res.status(400).json({ error: 'Tenant resolution failed' });
    }

    // Process events sequentially; fast-fail duplicates but accept rest
    for (const event of payload.events) {
      if (await isDuplicateEvent(event, tenantId)) {
        logger.debug({ webhookEventId: event.webhookEventId }, 'Duplicate event skipped');
        continue;
      }

      await markEventSeen(event, tenantId);

      // Delegate to job/queue to avoid blocking webhook response
      // Example: await queue.add('line-event', { event, tenantId });
      // For now, minimal handling:
      if (event.type === 'message' && event.message?.type === 'text') {
        // Placeholder: clock-in/out command parsing
        logger.info({ tenantId, userId: event.source.userId, text: event.message.id }, 'LINE text event queued');
      }
    }

    // Always 200 so LINE stops retrying
    return res.status(200).json({ ok: true });
  } catch (err) {
    logger.error({ err }, 'LINE webhook processing failed');
    return res.status(400).json({ error: 'Invalid payload' });
  }
});

export default router;
```

```ts
// /opt/axentx/workio/server/src/middleware/rawBody.ts
import type { Request, Response, NextFunction } from 'express';

export function getRawBody(req: Request, res: Response, next: NextFunction) {
  const chunks: Buffer[] = [];
  req.on('data', (chunk) => chunks.push(chunk));
  req.on('end', () => {
    (req as any).rawBody = Buffer.concat(chunks);
    next();
  });
  req.on('error', next);
}
```
