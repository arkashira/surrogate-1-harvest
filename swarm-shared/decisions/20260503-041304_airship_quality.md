import json, os
from huggingface_hub import list_repo_tree

repo = "axentx/surrogate-datasets"

os.makedirs("config", exist_ok=True)
with open("config/file_list.json", "w") as f:
    json.dump(out, f, indent=2)

### 2. CDN-only DataLoader (Lightning training) — 45 min
Replace `load_dataset(streaming=True)` with direct CDN downloads. Zero API calls during training.

```python
# /opt/axentx/airship/surrogate/train.py  (excerpt)
import json, io, os
import pyarrow.parquet as pq
import requests
import torch
from torch.utils.data import Dataset, DataLoader

HF_CDN = "https://huggingface.co/datasets"

class CDNParquetDataset(Dataset):
    def __init__(self, file_list_path, repo, max_files=None):
        with open(file_list_path) as f:
            cfg = json.load(f)
        self.repo = cfg["repo"]
        self.files = cfg["files"]
        if max_files:
            self.files = self.files[:max_files]

    def __len__(self):
        return len(self.files)

    def __getitem__(self, idx):
        path = self.files[idx]
        url = f"{HF_CDN}/{self.repo}/resolve/main/{path}"
        resp = requests.get(url, timeout=30)
        resp.raise_for_status()
        buf = io.BytesIO(resp.content)
        table = pq.read_table(buf)
        # project to {prompt, response} only
        return {
            "prompt": table["prompt"].to_pylist()[0],
            "response": table["response"].to_pylist()[0],
        }

def build_dataloader(file_list_path, batch_size=8, max_files=None):
    repo = "axentx/surrogate-datasets"
    ds = CDNParquetDataset(file_list_path, repo, max_files=max_files)
    return DataLoader(ds, batch_size=batch_size, shuffle=True, num_workers=4)
```

### 3. Lightning idle-resilient launcher — 30 min
Reuse running Studio, restart if idle-stopped, fallback to available cloud.

```python
# /opt/axentx/airship/surrogate/lightning_launcher.py
#!/usr/bin/env python3
import lightning as L
from lightning.pytorch.cli import LightningCLI

def launch_training():
    from lightning.pytorch import Trainer
    from lightning.pytorch.callbacks import ModelCheckpoint

    # Reuse running studio if exists
    teamspace = L.Teamspace()
    studio_name = "surrogate-training"
    studio = None

    for s in teamspace.studios:
        if s.name == studio_name and s.status == "running":
            studio = s
            print(f"Reusing running studio: {studio_name}")
            break

    if not studio:
        print(f"Creating studio: {studio_name}")
        # Priority: H200 (lambda-prod) → L40S (public-prod)
        clouds = [
            ("lightning-lambda-prod", "H200"),
            ("lightning-public-prod", "L40S"),
        ]
        for cloud, machine in clouds:
            try:
                studio = L.Studio(
                    name=studio_name,
                    cloud=cloud,
                    machine=machine,
                    create_ok=True,
                )
                print(f"Launched on {cloud}/{machine}")
                break
            except Exception as e:
                print(f"{cloud}/{machine} failed: {e}")
                continue
        else:
            raise RuntimeError("No available cloud/machine")

    # Idle-resilient run
    max_retries = 3
    for attempt in range(max_retries):
        try:
            if studio.status != "running":
                print(f"Studio stopped, restarting (attempt {attempt+1})...")
                studio.start(machine=studio.machine)

            # Run training via CLI
            cli = LightningCLI(
                run=False,
                save_config_kwargs={"overwrite": True},
            )
            trainer = cli.trainer
            trainer.fit(cli.model, cli.datamodule)
            break
        except Exception as e:
            print(f"Training failed (attempt {attempt+1}): {e}")
            if attempt == max_retries - 1:
                raise
            studio.stop()
            import time
            time.sleep(60)

if __name__ == "__main__":
    launch_training()
```
