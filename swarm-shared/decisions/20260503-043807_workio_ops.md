# workio / ops

## Final Consolidated Implementation

### 1. Diagnosis (merged, de-duplicated)
- **Missing `X-Line-Signature` verification** — allows spoofed events and attendance fraud.
- **No idempotency / replay protection** — LINE retries on 5xx/timeouts create duplicate clock/leave/OT records; same event can be replayed hours later.
- **No atomic upsert for clock events** — concurrent retries can create multiple rows per user per day.
- **No structured observability** — missing request/response metadata, idempotency decisions, and outbound notification outcomes, making retries, failures, and fraud hard to audit.
- **No reliable outbound LINE notification retry** — silent drops on transient failures degrade user experience.

### 2. Proposed change (single scope)
- **File scope**: `/opt/axentx/workio/server/src/routes/webhook/line.ts` (handler) plus supporting middleware/services.
- **Add**:
  - `X-Line-Signature` HMAC-SHA256 verification middleware.
  - Idempotency/replay protection using a DB-backed idempotency table with TTL and unique key per event.
  - Transactional upsert for clock events (one row per user per day).
  - Structured logging for every webhook receipt, idempotency decision, processing outcome, and outbound notification status.
  - Exponential-backoff LINE notify helper with retry on network/idempotent errors and 429.
- **No schema changes** beyond one new table and one index.

#### server/src/middleware/lineSignature.ts
```ts
import { Request, Response, NextFunction } from 'express';
import crypto from 'crypto';

const LINE_CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';

export function verifyLineSignature(req: Request, res: Response, next: NextFunction) {
  const signature = req.get('X-Line-Signature');
  if (!signature) return res.status(401).json({ error: 'Missing X-Line-Signature' });

  // Use raw body for accurate HMAC; ensure bodyParser raw middleware runs before this for this route
  const body = typeof req.rawBody === 'string' ? req.rawBody : JSON.stringify(req.body);
  const expected = crypto
    .createHmac('sha256', LINE_CHANNEL_SECRET)
    .update(body, 'utf8')
    .digest('base64');

  if (signature !== expected) {
    return res.status(401).json({ error: 'Invalid signature' });
  }
  next();
}
```

#### server/src/services/idempotency.ts
```ts
import { Pool } from 'pg';
import logger from './logger';

const pool = new Pool({ connectionString: process.env.DATABASE_URL });

export async function ensureIdempotency(key: string, ttlSeconds = 600): Promise<boolean> {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');
    const res = await client.query(
      `INSERT INTO idempotency_keys(key, created_at) VALUES ($1, NOW()) ON CONFLICT (key) DO NOTHING RETURNING key`,
      [key]
    );
    await client.query('COMMIT');
    const isNew = res.rowCount > 0;
    if (!isNew) {
      logger.debug({ key }, 'Duplicate idempotency key skipped');
    }
    // Best-effort TTL cleanup (non-blocking)
    client.query('DELETE FROM idempotency_keys WHERE created_at < NOW() - ($1 || \' seconds\')::interval', [ttlSeconds.toString()]).catch(() => {});
    return isNew;
  } catch (e) {
    await client.query('ROLLBACK').catch(() => {});
    logger.error({ key, err: e instanceof Error ? e.message : String(e) }, 'Idempotency check failed');
    return false;
  } finally {
    client.release();
  }
}
```

#### server/src/routes/webhook/line.ts
```ts
import { Router, Request } from 'express';
import { verifyLineSignature } from '../../middleware/lineSignature';
import { ensureIdempotency } from '../../services/idempotency';
import { replyLine } from '../../services/lineNotify';
import { upsertClockEvent } from '../../services/clock';
import logger from '../../services/logger';

const router = Router();
router.use(verifyLineSignature);

router.post('/line', async (req: Request & { rawBody?: string }, res) => {
  const events = req.body.events || [];
  const receivedAt = Date.now();
  logger.info({ eventCount: events.length, receivedAt }, 'LINE webhook received');

  for (const ev of events) {
    const dk = ev.source?.type === 'user' ? ev.source.userId : ev.source?.groupId || ev.source?.roomId || 'unknown';
    const idKey = `line:${dk}:${ev.timestamp}:${ev.deliveryContext || 'once'}:${ev.type}:${ev.message?.id || ev.postback?.data || ''}`;
    const ok = await ensureIdempotency(idKey, 600);
    if (!ok) {
      logger.debug({ idKey }, 'Skipping duplicate event');
      continue;
    }

    try {
      if (ev.type === 'message' && ev.message?.type === 'text') {
        const text = ev.message.text.trim().toLowerCase();
        const userId = ev.source.userId;
        if (!userId) continue;
        if (text === 'clock in' || text === 'clock out') {
          await upsertClockEvent(userId, text === 'clock in' ? 'in' : 'out', ev.timestamp);
          await replyLine(ev.replyToken, { type: 'text', text: `Clock ${text} recorded` });
        }
      }
    } catch (err: any) {
      logger.error({ idKey, err: err?.message }, 'Webhook processing error');
      // Return 200 to stop LINE retries after idempotency acceptance; processing failure logged for investigation
    }
  }
  res.status(200).json({ ok: true });
});

export default router;
```
