# workio / design

## 1. Diagnosis

- Missing idempotency key on punch records → LINE webhook redeliveries and concurrent deliveries create duplicate clock‑in/out rows.
- Non‑atomic read‑then‑insert for punch state → races produce multiple active punches per employee.
- No transactional boundary around punch state transition (clock‑in/out) and employee status update.
- No unique constraint to prevent duplicate active punches at DB level.
- Webhook handler does not return early when duplicate event id is detected.

## 2. Proposed change

- **File:** `workio/server/src/services/lineWebhookService.ts` (or equivalent webhook handler)  
- **Scope:** add idempotency check + transactional upsert for punch records; add unique constraint on (`employee_id`, `date`, `type`) where `type` ∈ ('clock_in','clock_out') and `active = true` to enforce at most one active punch of each type per day.

## 3. Implementation

### 3.1 DB migration (add constraint + idempotency table)

```sql
-- server/src/db/migrations/20260503_add_punch_constraints.sql
BEGIN;

-- Idempotency keys for LINE webhook events
CREATE TABLE IF NOT EXISTS webhook_deliveries (
  idempotency_key TEXT PRIMARY KEY,
  event_id        TEXT NOT NULL,
  processed_at    TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Ensure at most one active clock_in per employee per day
CREATE UNIQUE INDEX IF NOT EXISTS idx_one_active_clock_in_per_day
  ON punches (employee_id, date, type)
  WHERE active = true AND type = 'clock_in';

-- Ensure at most one active clock_out per employee per day
CREATE UNIQUE INDEX IF NOT EXISTS idx_one_active_clock_out_per_day
  ON punches (employee_id, date, type)
  WHERE active = true AND type = 'clock_out';

COMMIT;
```

### 3.2 Webhook handler (idempotent, transactional)

```typescript
// server/src/services/lineWebhookService.ts
import { pool } from '../db/client.js';

export async function handleLineWebhook(body: any, signature: string): Promise<{ status: number }> {
  const { events } = body;
  if (!Array.isArray(events) || events.length === 0) return { status: 200 };

  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    for (const ev of events) {
      const idempotencyKey = `line:${ev.source.userId}:${ev.type}:${ev.message?.id || ev.postback?.data || ev.timestamp}`;
      // Fast path: already processed
      const exists = await client.query(
        `SELECT 1 FROM webhook_deliveries WHERE idempotency_key = $1`,
        [idempotencyKey]
      );
      if (exists.rows.length > 0) continue;

      // Record idempotency first (best-effort ordering)
      await client.query(
        `INSERT INTO webhook_deliveries (idempotency_key, event_id) VALUES ($1, $2)`,
        [idempotencyKey, ev.type]
      );

      if (ev.type === 'message' && ev.message.type === 'text') {
        const text = ev.message.text.trim().toLowerCase();
        const employeeId = await getOrCreateEmployeeByLineUid(client, ev.source.userId);
        const today = new Date().toISOString().slice(0, 10); // YYYY-MM-DD

        if (text === 'clock in' || text === 'เข้างาน') {
          // End any dangling active clock_out for same day (safety)
          await client.query(
            `UPDATE punches SET active = false WHERE employee_id = $1 AND date = $2 AND type = 'clock_out' AND active = true`,
            [employeeId, today]
          );

          // Try insert; unique index prevents duplicate active clock_in
          try {
            await client.query(
              `INSERT INTO punches (employee_id, date, type, timestamp, location, active)
               VALUES ($1, $2, 'clock_in', NOW(), $3, true)`,
              [employeeId, today, ev.source.userId] // location placeholder; replace with GPS if available
            );
          } catch (err: any) {
            if (err.code === '23505') {
              // duplicate active clock_in — ignore
              continue;
            }
            throw err;
          }
        } else if (text === 'clock out' || text === 'เลิกงาน') {
          // Ensure there's an active clock_in to pair
          const lastIn = await client.query(
            `SELECT id FROM punches
             WHERE employee_id = $1 AND date = $2 AND type = 'clock_in' AND active = true
             ORDER BY timestamp DESC LIMIT 1`,
            [employeeId, today]
          );

          if (lastIn.rows.length === 0) continue; // require clock_in first

          // Try insert clock_out; unique index prevents duplicate active clock_out
          try {
            await client.query(
              `INSERT INTO punches (employee_id, date, type, timestamp, location, active)
               VALUES ($1, $2, 'clock_out', NOW(), $3, true)`,
              [employeeId, today, ev.source.userId]
            );
          } catch (err: any) {
            if (err.code === '23505') {
              // duplicate active clock_out — ignore
              continue;
            }
            throw err;
          }

          // Deactivate the paired clock_in (state transition)
          await client.query(
            `UPDATE punches SET active = false WHERE id = $1`,
            [lastIn.rows[0].id]
          );
        }
      }
    }

    await client.query('COMMIT');
    return { status: 200 };
  } catch (err) {
    await client.query('ROLLBACK');
    console.error('Webhook processing failed', err);
    return { status: 500 };
  } finally {
    client.release();
  }
}

async function getOrCreateEmployeeByLineUid(client: any, lineUid: string): Promise<number> {
  const row = await client.query(
    `SELECT id FROM employees WHERE line_user_id = $1`,
    [lineUid]
  );
  if (row.rows.length > 0) return row.rows[0].id;

  const insert = await client.query(
    `INSERT INTO employees (line_user_id, name, status) VALUES ($1, $2, 'active') RETURNING id`,
    [lineUid, `Employee (${lineUid.slice(0, 6)}...)`]
  );
  return insert.rows[0].id;
}
```

### 3.3 Expose handler in route

```typescript
// server/src/routes/lineWebhookRoute.ts
import express from 'express';
import { handleLineWebhook } from '../services/lineWebhookService.js';

const router = express.Router();

router.post('/webhook/line', express.raw({ type: 'application/json' }), (req, res) => {
  const signature = req.headers['x-line-signature'] as string;
  const body = JSON.parse(req.body.toString());
  handleLineWebhook(body, signature).then(({ status }) => res.sendStatus(status));
});

export default router;
```
