### 1. Deterministic CDN Manifest Generation (Mac Orchestration)
**Action:** Generate a single `manifest.json` containing only parquet files under `batches/mirror-merged/{date}/` with full CDN URLs.

**Key Insight:** Use recursive listing to capture nested shards (contradiction resolved: recursive=True ensures completeness for unknown folder depths, while date-scoping prevents pagination explosion).

```python
#!/usr/bin/env python3
"""
Generate CDN-only manifest for a date folder.
Usage: python manifest_generator.py --repo org/surrogate-data --date 2026-05-03 --out manifest.json
"""
import argparse
import json
from huggingface_hub import HfApi

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True, help="HF dataset repo (org/name)")
    parser.add_argument("--date", required=True, help="Date folder (YYYY-MM-DD)")
    parser.add_argument("--out", default="manifest.json")
    args = parser.parse_args()

    api = HfApi()
    prefix = f"batches/mirror-merged/{args.date}/"
    # Use recursive=True to capture all nested parquet shards
    files = api.list_repo_tree(repo_id=args.repo, path=prefix, recursive=True)

    entries = []
    for f in files:
        if f.path.lower().endswith(".parquet"):
            entries.append({
                "path": f.path,
                "cdn_url": f"https://huggingface.co/datasets/{args.repo}/resolve/main/{f.path}"
            })

    manifest = {
        "repo": args.repo,
        "date": args.date,
        "entries": entries
    }

    with open(args.out, "w") as fp:
        json.dump(manifest, fp, indent=2)
    print(f"Wrote {len(entries)} entries to {args.out}")

if __name__ == "__main__":
    main()
```

### 2. Lightning-Resilient Training Launcher (Studio Reuse + Auto-Restart)
**Action:** Implement a launcher that reuses existing studios and automatically restarts on idle timeout, passing the manifest path via job config.

**Key Insight:** Combine status polling with async job monitoring to handle idle-timeout recovery without manual intervention (contradiction resolved: sync=False allows external monitoring while maintaining retry logic).

```python
#!/usr/bin/env python3
"""
Lightning Studio launcher with reuse + idle-resilient restart.
"""
import os
import time
from lightning_sdk import Studio, Teamspace, Machine

TEAMSPACE = "default"
STUDIO_NAME = os.getenv("STUDIO_NAME", "surrogate-train")
MANIFEST_PATH = os.getenv("MANIFEST_PATH", "manifest.json")

def get_or_create_studio():
    team = Teamspace(name=TEAMSPACE)
    for s in team.studios:
        if s.name == STUDIO_NAME:
            print(f"Found existing studio: {s.name} ({s.status})")
            return s
    print("Creating new studio...")
    return Studio.create(
        name=STUDIO_NAME,
        teamspace=TEAMSPACE,
        machine=Machine.L40S,
        create_ok=True
    )

def run_training_job(script_path="train.py", max_retries=3):
    studio = get_or_create_studio()
    retry = 0
    while retry < max_retries:
        if studio.status != "Running":
            print(f"Studio not running ({studio.status}). Restarting...")
            studio.start(machine=Machine.L40S)
            # wait until running
            while studio.status != "Running":
                time.sleep(10)
                studio.refresh()

        print(f"Running training job (attempt {retry+1})...")
        job = studio.run(
            command=[
                "python", script_path,
                "--manifest", MANIFEST_PATH
            ],
            sync=False  # run async; monitor separately
        )
        # Monitor loop with timeout handling
        while True:
            job.refresh()
            if job.status in ("succeeded", "failed", "cancelled"):
                break
            time.sleep(30)

        if job.status == "succeeded":
            print("Training succeeded.")
            return job
        else:
            print(f"Job ended with status {job.status}. Will retry...")
            retry += 1
            time.sleep(60)

    raise RuntimeError("Max retries exceeded")

if __name__ == "__main__":
    run_training_job()
```

### 3. CDN-Only Dataset Loader (Zero HF API Calls)
**Action:** Replace `load_dataset(streaming=True)` with a custom `IterableDataset` that streams parquet shards directly from CDN URLs via `fsspec`.

**Key Insight:** Use `pyarrow.parquet.ParquetFile` for streaming reads without full downloads, with worker-aware sharding and graceful error handling for corrupt shards.

```python
import json
import io
import pyarrow.parquet as pq
import fsspec
import torch
from torch.utils.data import IterableDataset

class CDNParquetIterable(IterableDataset):
    def __init__(self, manifest_path, buffer_size=1024*1024):
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.urls = [e["cdn_url"] for e in self.manifest["entries"]]
        self.buffer_size = buffer_size

    def __iter__(self):
        worker_info = torch.utils.data.get_worker_info()
        if worker_info is None:
            urls = self.urls
        else:
            per_worker = len(self.urls) // worker_info.num_workers
            urls = self.urls[worker_info.id * per_worker : (worker_info.id + 1) * per_worker]

        for url in urls:
            try:
                with fsspec.open(url, "rb") as f:
                    # stream parquet without full download; project columns at parse time
                    with pq.ParquetFile(f) as pf:
                        for batch in pf.iter_batches(batch_size=512, columns=["prompt", "response"]):
                            df = batch.to_pydict()
                            for prompt, response in zip(df["prompt"], df["response"]):
                                yield {"prompt": prompt, "response": response}
            except Exception as exc:
                # skip corrupt shards; log and continue
                print(f"Skipping {url}: {exc}")
                continue
```
