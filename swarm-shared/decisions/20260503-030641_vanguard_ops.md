# vanguard / ops

## 1. Diagnosis
- No CDN-first manifest: training/ingestion still rely on recursive `list_repo_tree`/`load_dataset` and risk HF API 429s.
- No static file list: each training run re-enumerates repos, consuming quota and producing non-deterministic slices.
- No integrity verification: parquet corruption during long surrogate ingestion is silent and unrecoverable.
- No idempotent Lightning Studio reuse: scripts recreate studios instead of reusing running ones, burning 80+ hrs/mo quota.
- No Mac→remote separation guard: orchestration may accidentally trigger local `from_pretrained` on Mac instead of delegating to Lightning/Kaggle/Cerebras.

## 2. Proposed change
Create `/opt/axentx/vanguard/ops/cdn_manifest.py` (single source of truth) and update training launcher to use it:
- Single non-recursive `list_repo_tree` per date folder → write `manifest-{date}.json` with CDN URLs and sha256.
- Training script loads manifest and fetches via CDN-only (zero API calls).
- Add studio reuse + idle restart guard.
- Add Mac remote-execution guard.

```bash
# /opt/axentx/vanguard/ops/cdn_manifest.py
#!/usr/bin/env python3
"""
Generate and verify CDN-first manifest for HF dataset folders.
Usage:
  python cdn_manifest.py --repo datasets/my-mirror --date 2026-04-29 --out manifest-2026-04-29.json
"""
import argparse
import hashlib
import json
import os
import sys
from datetime import datetime
from pathlib import Path
from typing import Dict, List

import requests
from huggingface_hub import HfApi, hf_hub_download

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def list_folder_files(repo: str, folder: str) -> List[str]:
    """Non-recursive listing to avoid pagination/429."""
    api = HfApi()
    items = api.list_repo_tree(repo=repo, path=folder, recursive=False)
    files = [it.rfilename for it in items if it.type == "file"]
    return files

def sha256_file(path: Path) -> str:
    h = hashlib.sha256()
    with open(path, "rb") as f:
        for chunk in iter(lambda: f.read(8192), b""):
            h.update(chunk)
    return h.hexdigest()

def build_manifest(repo: str, date_folder: str, out_path: Path) -> Dict:
    files = list_folder_files(repo, date_folder)
    entries = []
    for f in files:
        path = Path(date_folder) / f
        url = CDN_TEMPLATE.format(repo=repo, path=str(path).replace("\\", "/"))
        # lightweight HEAD to get size; skip if CDN unreachable
        try:
            r = requests.head(url, timeout=10)
            size = int(r.headers.get("content-length", 0))
        except Exception:
            size = 0
        entries.append(
            {
                "file": str(path),
                "url": url,
                "size": size,
                "sha256": None,  # filled after optional local download
            }
        )
    manifest = {
        "repo": repo,
        "date_folder": date_folder,
        "generated_at": datetime.utcnow().isoformat() + "Z",
        "entries": entries,
    }
    out_path.write_text(json.dumps(manifest, indent=2))
    return manifest

def verify_and_patch(manifest_path: Path, cache_dir: Path) -> None:
    """Download missing files and patch sha256."""
    manifest = json.loads(manifest_path.read_text())
    cache_dir.mkdir(parents=True, exist_ok=True)
    for e in manifest["entries"]:
        local = cache_dir / Path(e["file"]).name
        if not local.exists():
            print(f"Downloading {e['file']}...")
            hf_hub_download(
                repo_id=manifest["repo"],
                filename=e["file"],
                cache_dir=str(cache_dir),
                force_download=False,
            )
        e["sha256"] = sha256_file(local)
        e["local_path"] = str(local)
    manifest_path.write_text(json.dumps(manifest, indent=2))

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="CDN-first manifest generator")
    parser.add_argument("--repo", required=True, help="HF dataset repo")
    parser.add_argument("--date", required=True, help="Date folder (e.g. 2026-04-29)")
    parser.add_argument("--out", default="manifest.json", help="Output manifest path")
    parser.add_argument("--verify", action="store_true", help="Verify and patch sha256")
    parser.add_argument("--cache", default=".cache", help="Local cache directory")
    args = parser.parse_args()

    out = Path(args.out)
    manifest = build_manifest(args.repo, args.date, out)
    print(f"Wrote {len(manifest['entries'])} entries to {out}")

    if args.verify:
        verify_and_patch(out, Path(args.cache))
        print("Verified and patched sha256.")
```

```python
#!/usr/bin/env python3
"""
Lightning training launcher with CDN manifest and studio reuse.
Mac-safe: only orchestrates remote jobs; never runs model.from_pretrained() locally.
"""
import json
import os
import sys
from pathlib import Path

# Mac guard
if sys.platform == "darwin":
    print("Mac detected — running orchestration only (no local training).")

from lightning import LightningWork, LightningFlow, LightningApp, Machine
from lightning.app import LightningStudio

# Prevent accidental local heavy compute
assert "RUN_LOCAL_TRAINING" not in os.environ, "Set RUN_LOCAL_TRAINING only on remote runners."

class SurrogateTrainer(LightningWork):
    def __init__(self, manifest_path: str, **kwargs):
        super().__init__(**kwargs)
        self.manifest_path = manifest_path
        self._studio = None

    def run(self):
        # Studio reuse
        from lightning.app import Teamspace
        for s in Teamspace.studios:
            if s.name == "surrogate-trainer" and s.status == "Running":
                self._studio = s
                print(f"Reusing running studio: {s.name}")
                break
        if self._studio is None:
            self._studio = LightningStudio(
                name="surrogate-trainer",
                script=str(Path(__file__).parent / "train.py"),
                machine=Machine.L40S,  # free-tier fallback; sweeps handled externally
                create_ok=True,
            )

        # Guard against idle stop
        if self._studio.status != "Running":
            print("Studio stopped — restarting...")
            self._studio.start(machine=Machine.L40S)

        # Pass manifest to training via env or CLI
        self._studio.run(
            arguments=[
                "--manifest",
                self.manifest_path,
                "--no-api",
            ]
        )

class TrainFlow(LightningFlow):
    def __init__(self, manifest_path: str):
        super().__init__()
        self.trainer = SurrogateTrainer(manifest_path)

    def run(self):
        self.trainer.run()

if __name__ == "__main__":
    # Example: python train_launcher.py --manifest manifest-2026-04-29.json
    import argparse
    parser = argparse.ArgumentParser()
    parser.add_argument("--manifest", required=True)
    args = parser.parse_args()
    app = LightningApp(TrainFlow(args.manifest))
```
