### 1. scripts/list_hf_files.py
```python
#!/usr/bin/env python3
"""
Mac-side one-shot: list a date folder in a HF dataset repo
and emit file list as JSON for CDN-only training.

Usage:
    python list_hf_files.py \
        --repo "axentx/surrogate-mirror" \
        --path "batches/mirror-merged/2026-05-03" \
        --out "filelist_2026-05-03.json"
"""

import argparse
import json
import os
import sys
from typing import List

try:
    from huggingface_hub import HfApi
except ImportError:
    print("Install: pip install huggingface_hub")
    sys.exit(1)

def list_date_folder(repo: str, path: str) -> List[str]:
    api = HfApi()
    # non-recursive per folder to avoid 100x pagination and rate limits
    items = api.list_repo_tree(repo=repo, path=path, recursive=False)
    files = [
        item.rfilename
        for item in items
        if getattr(item, "type", None) == "file" and item.rfilename.endswith(".parquet")
    ]
    return sorted(files)

def main() -> None:
    parser = argparse.ArgumentParser(description="List HF dataset folder for CDN training.")
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g. axentx/surrogate-mirror)")
    parser.add_argument("--path", required=True, help="Folder path in repo (e.g. batches/mirror-merged/2026-05-03)")
    parser.add_argument("--out", required=True, help="Output JSON file path")
    args = parser.parse_args()

    print(f"Listing {args.repo}@{args.path} ...")
    files = list_date_folder(args.repo, args.path)
    payload = {
        "repo": args.repo,
        "path": args.path,
        "files": files,
        "count": len(files),
    }

    os.makedirs(os.path.dirname(os.path.abspath(args.out)), exist_ok=True)
    with open(args.out, "w") as f:
        json.dump(payload, f, indent=2)
    print(f"Wrote {len(files)} files to {args.out}")

if __name__ == "__main__":
    main()
```

### 2. surrogate/cdn_loader.py
```python
"""
CDN-only parquet loader for Surrogate training.
Uses direct CDN URLs to bypass HF API rate limits.
Expects file list JSON produced by scripts/list_hf_files.py.
"""

import json
import logging
import os
import tempfile
from typing import Dict, Iterator, List

import pandas as pd
import pyarrow.parquet as pq
import requests
from tqdm import tqdm

logger = logging.getLogger(__name__)

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

class CDNParquetLoader:
    def __init__(self, filelist_path: str, repo: str = None, path_prefix: str = ""):
        with open(filelist_path) as f:
            manifest = json.load(f)
        self.repo = repo or manifest["repo"]
        self.path_prefix = path_prefix or manifest.get("path", "")
        self.files: List[str] = manifest["files"]

    def _cdn_url(self, rel_path: str) -> str:
        full = os.path.join(self.path_prefix, rel_path).lstrip("/")
        return CDN_TEMPLATE.format(repo=self.repo, path=full)

    def _download_parquet(self, url: str) -> str:
        # small temp file; for larger workloads consider streaming to disk
        with tempfile.NamedTemporaryFile(suffix=".parquet", delete=False) as tmp:
            resp = requests.get(url, timeout=60)
            resp.raise_for_status()
            tmp.write(resp.content)
            return tmp.name

    def iter_rows(self, max_files: int = None) -> Iterator[Dict[str, str]]:
        limit = len(self.files) if max_files is None else min(max_files, len(self.files))
        for fname in tqdm(self.files[:limit], desc="CDN parquet"):
            url = self._cdn_url(fname)
            local_path = None
            try:
                local_path = self._download_parquet(url)
                # project only prompt/response at parse time (mixed-schema safe)
                pf = pq.ParquetFile(local_path)
                for batch in pf.iter_batches(batch_size=64, columns=["prompt", "response"]):
                    df = batch.to_pandas()
                    for _, row in df.iterrows():
                        prompt = str(row.get("prompt", ""))
                        response = str(row.get("response", ""))
                        if prompt.strip() and response.strip():
                            yield {"prompt": prompt, "response": response}
            except Exception as exc:
                logger.warning("Failed to process %s: %s", url, exc)
            finally:
                if local_path and os.path.exists(local_path):
                    os.unlink(local_path)
```

#!/usr/bin/env python3
"""
Surrogate training entrypoint (Lightning Studio compatible).
- Uses CDN-only loader (no HF API during data load).
- Resilient to Lightning idle timeouts: checks studio status and restarts if stopped.
"""

import argparse
import logging
import os
import time
from pathlib import Path

import lightning as L
import torch
from torch.utils.data import DataLoader, IterableDataset

from cdn_loader import CDNParquetLoader

logging.basicConfig(level=logging.INFO)
log = logging.getLogger(__name__)

# ---- Dataset ----
class ParquetIterableDataset(IterableDataset):
    def __init__(self, loader: CDNParquetLoader, max_files: int = None):
        self.loader = loader
        self.max_files = max_files

    def __iter__(self):
        yield from self.loader.iter_rows(max_files=self.max_files)

# ---- Model stub ----
class SurrogateModel(torch.nn.Module):
    def __init__(self, vocab_size=32000, d_model=1024):
        super().__init__()
        self.emb = torch.nn.Embedding(vocab_size, d_model)
        self.out = torch.nn.Linear(d_model, vocab_size)

    def forward(self, x):
        return self.out(self.emb(x))

# ---- Lightning idle-resilient runner ----
def ensure_running_studio(target: L.studio.Studio, machine: L.cloud.L40S) -> L.studio.Studio:
    """Reuse or restart a Lightning Studio if stopped (idle-timeout resilient)."""
    teamspace = L.Teamspace()
    for s in teamspace.studios:
        if s.name == target.name and s.status == "running":
            log.info("Reusing running studio: %s", s.name)
            return s
    log.info("Starting studio: %s", target.name)
    target.start(machine=machine)
    return target
