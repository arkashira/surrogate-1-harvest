# workio / design

## 1. Diagnosis

- **No database-level uniqueness guard against concurrent/open punches**: current logic uses find-then-insert/update in app code, which is unsafe under LINE webhook retries and client double-taps; duplicates can occur.
- **Missing partial index for “open punch” queries**: queries filtering `clock_out_at IS NULL` perform full scans as tenant/employee counts grow; no index exists.
- **Clock-in/clock-out state logic is implicit in app code**: no single source-of-truth constraint to enforce “at most one open punch per user per calendar day”.
- **No idempotency key on webhook handler**: LINE retries with same event payload can create duplicate punches because requests are not deduplicated.
- **Schema lacks created-at precision for ordering**: `created_at` defaults to `now()` but without time-zone clarity and no monotonic ordering guarantee across app/db clocks; complicates “latest punch” resolution.

## 2. Proposed change

File scope:
- `workio/server/src/db/schema.sql` — add constraint + partial index + idempotency table.
- `workio/server/src/db/punch.ts` (or equivalent DAO) — add idempotency check and safe upsert.
- `workio/server/src/routes/line/webhook.ts` — attach idempotency key from LINE event id and short-circuit duplicates.

## 3. Implementation

### 3.1 Schema changes (`workio/server/src/db/schema.sql`)

```sql
-- 1) Ensure at most one open punch per user per calendar day (UTC)
CREATE UNIQUE INDEX CONCURRENTLY IF NOT EXISTS idx_punches_one_open_per_day
  ON punches (tenant_id, user_id, date(clock_in_at))
  WHERE clock_out_at IS NULL;

-- 2) Fast lookup for open punches
CREATE INDEX CONCURRENTLY IF NOT EXISTS idx_punches_open
  ON punches (tenant_id, user_id, clock_in_at)
  WHERE clock_out_at IS NULL;

-- 3) Idempotency table for LINE webhook events (lightweight)
CREATE TABLE IF NOT EXISTS line_webhook_idempotency (
  idempotency_key TEXT PRIMARY KEY,
  event_type      TEXT NOT NULL,
  punch_id        BIGINT,
  created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Optional: cleanup old keys (older than 7 days) via cron or TTL job
-- CREATE INDEX IF NOT EXISTS idx_line_idempotency_created ON line_webhook_idempotency (created_at);
```

Apply migration safely:


### 3.2 Punch DAO safe upsert (`workio/server/src/db/punch.ts`)

```ts
import { Pool } from 'pg';

const pool = new Pool({ connectionString: process.env.DATABASE_URL });

export async function clockInWithIdempotency(
  tenantId: string,
  userId: string,
  clockInAt: Date,
  location: { lat: number; lng: number },
  idempotencyKey: string
) {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // Short-circuit duplicate event
    const dup = await client.query(
      `SELECT 1 FROM line_webhook_idempotency WHERE idempotency_key = $1`,
      [idempotencyKey]
    );
    if (dup.rows.length > 0) {
      await client.query('ROLLBACK');
      return { existing: true, punchId: dup.rows[0].punch_id };
    }

    // Try to find today's open punch
    const openPunch = await client.query(
      `SELECT id FROM punches
       WHERE tenant_id = $1 AND user_id = $2 AND clock_out_at IS NULL
       ORDER BY clock_in_at DESC LIMIT 1 FOR UPDATE`,
      [tenantId, userId]
    );

    let punchId: number;
    if (openPunch.rows.length > 0) {
      // If already open, treat as duplicate (or optionally update location)
      punchId = openPunch.rows[0].id;
    } else {
      const res = await client.query(
        `INSERT INTO punches (tenant_id, user_id, clock_in_at, clock_in_lat, clock_in_lng, clock_out_at)
         VALUES ($1, $2, $3, $4, $5, NULL)
         RETURNING id`,
        [tenantId, userId, clockInAt, location.lat, location.lng]
      );
      punchId = res.rows[0].id;
    }

    // Record idempotency
    await client.query(
      `INSERT INTO line_webhook_idempotency (idempotency_key, event_type, punch_id)
       VALUES ($1, $2, $3)`,
      [idempotencyKey, 'clockIn', punchId]
    );

    await client.query('COMMIT');
    return { existing: false, punchId };
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  } finally {
    client.release();
  }
}

export async function clockOut(
  tenantId: string,
  userId: string,
  clockOutAt: Date,
  location: { lat: number; lng: number }
) {
  const res = await pool.query(
    `UPDATE punches
     SET clock_out_at = $1, clock_out_lat = $2, clock_out_lng = $3
     WHERE id = (
       SELECT id FROM punches
       WHERE tenant_id = $4 AND user_id = $5 AND clock_out_at IS NULL
       ORDER BY clock_in_at DESC LIMIT 1
     )
     RETURNING id`,
    [clockOutAt, location.lat, location.lng, tenantId, userId]
  );
  return res.rows[0]?.id || null;
}
```

### 3.3 Webhook handler (`workio/server/src/routes/line/webhook.ts`)

```ts
import express from 'express';
import { clockInWithIdempotency, clockOut } from '../../db/punch';

const router = express.Router();

router.post('/line/webhook', express.raw({ type: 'application/json' }), async (req, res) => {
  try {
    const body = JSON.parse(req.body.toString());
    const event = body.events?.[0];
    if (!event) return res.sendStatus(200);

    // Use LINE event id as idempotency key
    const idempotencyKey = event.source.userId + ':' + event.timestamp + ':' + event.type;

    // Minimal mapping — adapt to your tenant resolution (e.g. by userId or channel secret)
    const tenantId = process.env.DEFAULT_TENANT_ID;
    const userId = event.source.userId;

    if (event.type === 'message' && event.message.type === 'text') {
      const text = event.message.text.toLowerCase();
      const now = new Date();
      const location = { lat: 13.7563, lng: 100.5018 }; // Replace with GPS from event or user profile

      if (text.includes('in') || text.includes('clock')) {
        await clockInWithIdempotency(tenantId, userId, now, location, idempotencyKey);
      } else if (text.includes('out') || text.includes('clock out')) {
        await clockOut(tenantId, userId, now, location);
      }
    }

    res.sendStatus(200);
  } catch (err) {
    console.error('Webhook error', err);
    res.status(500).send('Internal Server Error');
  }
});

export default router;
```
