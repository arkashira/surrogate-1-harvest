# workio / backend

## 1. Diagnosis

- **Non-idempotent webhook ingestion**: LINE retries on 5xx/timeouts can re-deliver the same event and create duplicate clock/leave/OT records.
- **No unique constraint to prevent duplicates**: concurrent/retried deliveries can double-insert events for the same user/day/type.
- **No idempotency key in request path**: webhook handler relies on implicit state checks instead of explicit idempotency tokens or unique constraints.
- **Race condition between existence check and insert**: `SELECT → INSERT` pattern allows duplicates under concurrency/retries.
- **No transactional upsert for clock events**: clock in/out and leave/OT requests should be atomic upserts keyed on `(tenant_id, user_id, event_date, event_type)`.

## 2. Proposed change

- **File**: `server/src/routes/webhook/line.ts` (or equivalent webhook handler)
- **Scope**: add idempotent upsert for clock events and leave/OT requests using a unique constraint + `ON CONFLICT DO NOTHING` (or update) in a single transaction; return 200 only after commit.
- **DB**: add unique constraint on clock_events `(tenant_id, user_id, event_date, event_type)` and on leave_requests `(tenant_id, user_id, request_date, type, status)` where appropriate.

## 3. Implementation

### 3.1 DB migration (run once)

```sql
-- server/src/db/migrations/20260503_add_unique_constraints.sql

-- Clock events: one record per tenant/user/date/type (clock_in/clock_out)
ALTER TABLE clock_events
  ADD CONSTRAINT uq_clock_events_tenant_user_date_type
  UNIQUE (tenant_id, user_id, event_date, event_type);

-- Leave requests: prevent duplicate requests for same day/type while open
ALTER TABLE leave_requests
  ADD CONSTRAINT uq_leave_requests_tenant_user_date_type_status
  UNIQUE (tenant_id, user_id, request_date, type, status)
  WHERE status IN ('pending', 'approved');
```

### 3.2 Webhook handler (idempotent upsert)

```ts
// server/src/routes/webhook/line.ts
import { Request, Response } from 'express';
import { pool } from '../../db';
import { v4 as uuidv4 } from 'uuid';

export async function handleLineWebhook(req: Request, res: Response) {
  const { events } = req.body;
  const client = await pool.connect();

  try {
    await client.query('BEGIN');

    for (const ev of events) {
      if (ev.type !== 'message' && ev.type !== 'postback') continue;

      const tenantId = process.env.DEFAULT_TENANT_ID; // or resolve from source
      const userId = ev.source.userId;
      const now = new Date();
      const today = now.toISOString().split('T')[0];

      // Example: clock in/out via postback data or message text
      const action = ev.postback?.data || ev.message?.text;
      const isClockIn = action === 'clock_in' || /(clock.?in|เข้า)/i.test(action || '');
      const isClockOut = action === 'clock_out' || /(clock.?out|ออก)/i.test(action || '');

      if (isClockIn || isClockOut) {
        const eventType = isClockIn ? 'clock_in' : 'clock_out';

        // Idempotent upsert: insert if not exists; if exists, keep original (or optionally update metadata)
        const upsertQuery = `
          INSERT INTO clock_events (id, tenant_id, user_id, event_date, event_type, timestamp, location, raw_event, created_at)
          VALUES ($1, $2, $3, $4, $5, $6, $7, $8, $9)
          ON CONFLICT (tenant_id, user_id, event_date, event_type)
          DO UPDATE SET timestamp = EXCLUDED.timestamp, location = EXCLUDED.location, raw_event = EXCLUDED.raw_event
          WHERE clock_events.tenant_id = EXCLUDED.tenant_id
            AND clock_events.user_id = EXCLUDED.user_id
            AND clock_events.event_date = EXCLUDED.event_date
            AND clock_events.event_type = EXCLUDED.event_type
          RETURNING id;
        `;

        await client.query(upsertQuery, [
          uuidv4(),
          tenantId,
          userId,
          today,
          eventType,
          now,
          ev.location || null,
          JSON.stringify(ev),
          now,
        ]);
      }

      // Leave/OT requests (simplified)
      if (ev.type === 'postback' && ev.postback?.data?.startsWith('leave:')) {
        const [, type, date] = ev.postback.data.split(':');
        const leaveQuery = `
          INSERT INTO leave_requests (id, tenant_id, user_id, request_date, type, status, created_at)
          VALUES ($1, $2, $3, $4, $5, $6, $7)
          ON CONFLICT (tenant_id, user_id, request_date, type, status)
          DO NOTHING
          RETURNING id;
        `;
        await client.query(leaveQuery, [
          uuidv4(),
          tenantId,
          userId,
          date,
          type,
          'pending',
          now,
        ]);
      }
    }

    await client.query('COMMIT');
    res.status(200).json({ ok: true });
  } catch (err) {
    await client.query('ROLLBACK');
    console.error('Webhook processing failed', err);
    // Return 200 to prevent LINE retries only if error is non-retryable; otherwise 5xx to allow retry.
    // Conservative: return 500 so LINE retries transient failures.
    res.status(500).json({ error: 'processing_failed' });
  } finally {
    client.release();
  }
}
```
