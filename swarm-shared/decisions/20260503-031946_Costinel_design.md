# Costinel / design

## Implementation Plan — Top-Hub Signal Panel (CDN-first, <2h)

### Scope (single highest-value deliverable)
Add a **non-blocking Top-Hub Signal Panel** to the Costinel dashboard that:
- Detects and surfaces the most-connected hub (default `MOC`) from a lightweight hub-graph index
- Shows 3 contextual insights from knowledge-rag
- Uses CDN-first data fetching (bypasses HF API rate limits)
- Zero runtime API calls during dashboard render

### Architecture (fits existing patterns)
- **Data source**: `hub-index.json` + `hub-insights.json` stored in repo (committed via CI)
- **Runtime**: CDN fetch via `/resolve/main/` URLs (no auth, no rate limit)
- **Fallback**: Local defaults if CDN fails (graceful degradation)
- **No execution**: Pure signal/recommendation (Sense + Signal)

```tsx
import { useEffect, useState } from 'react';

interface HubGraph {
  [key: string]: { connections: number; degree: number; tags: string[] };
}

interface HubIndex {
  generatedAt: string;
  topHub: string;
  topHubScore: number;
  graph: HubGraph;
  lastUpdated: string;
}

interface Insight {
  id: string;
  title: string;
  summary: string;
  severity: 'high' | 'medium' | 'low';
  action: string;
}

interface HubInsights {
  hub: string;
  insights: Insight[];
  context: string;
}

const CDN_BASE = 'https://huggingface.co/datasets/AXENTX/Costinel/resolve/main';

const severityColor = {
  high: 'bg-red-50 border-red-200 text-red-800',
  medium: 'bg-amber-50 border-amber-200 text-amber-800',
  low: 'bg-blue-50 border-blue-200 text-blue-800',
};

export default function TopHubSignalPanel() {
  const [hub, setHub] = useState<string | null>(null);
  const [insights, setInsights] = useState<Insight[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    let mounted = true;

    async function fetchCDN<T>(path: string): Promise<T | null> {
      try {
        const res = await fetch(`${CDN_BASE}/${path}`, { cache: 'no-store' });
        if (!res.ok) throw new Error(`CDN fetch failed: ${res.status}`);
        return (await res.json()) as T;
      } catch (err) {
        console.warn(`CDN fetch error for ${path}:`, err);
        return null;
      }
    }

    async function load() {
      setLoading(true);
      // CDN-first: no Authorization header, bypasses HF API rate limits
      const [indexData, insightsData] = await Promise.all([
        fetchCDN<HubIndex>('src/data/hub-index.json'),
        fetchCDN<HubInsights>('src/data/hub-insights.json'),
      ]);

      if (!mounted) return;

      if (indexData && insightsData) {
        setHub(indexData.topHub);
        // Ensure alignment
        if (insightsData.hub === indexData.topHub) {
          setInsights(insightsData.insights.slice(0, 3));
        } else {
          // Fallback: use index topHub with empty insights
          setInsights([]);
        }
        setError(null);
      } else {
        // Graceful fallback to defaults
        setHub('MOC');
        setInsights([
          {
            id: 'fallback-001',
            title: 'Cost governance hub (MOC)',
            summary: 'Unable to load live hub data. Showing default top hub.',
            severity: 'medium',
            action: 'Check data pipeline status',
          },
        ]);
        setError('Using fallback data');
      }
      setLoading(false);
    }

    load();
    return () => {
      mounted = false;
    };
  }, []);

  if (loading) {
    return (
      <div className="rounded-lg border bg-white p-4 shadow-sm">
        <div className="flex items-center gap-3">
          <div className="h-10 w-10 animate-pulse rounded-full bg-gray-200" />
          <div className="space-y-2">
            <div className="h-4 w-32 animate-pulse rounded bg-gray-200" />
            <div className="h-3 w-24 animate-pulse rounded bg-gray-200" />
          </div>
        </div>
      </div>
    );
  }

  return (
    <div className="rounded-lg border bg-white p-4 shadow-sm">
      <div className="mb-3 flex items-start justify-between">
        <div className="flex items-center gap-3">
          <div className="flex h-10 w-10 items-center justify-center rounded-full bg-indigo-100 text-indigo-700">
            <svg
              className="h-5 w-5"
              fill="none"
              stroke="currentColor"
              viewBox="0 0 24 24"
            >
              <path
                strokeLinecap="round"
                strokeLinejoin="round"
                strokeWidth={2}
                d="M13 10V3L4 14h7v7l9-11h-7z"
              />
            </svg>
          </div>
          <div>
            <h3 className="font-semibold text-gray-900">Top Hub Signal</h3>
            <
