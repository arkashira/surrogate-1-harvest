# vanguard / design

## Final Consolidated Solution

### Diagnosis (merged)
- **No persisted manifest**: every training run re-enumerates via authenticated HF API, burning quota and risking 429.
- **Recursive enumeration / pagination**: heavy API use and exposure to mixed-schema files.
- **Schema/type hazards**: `load_dataset(streaming=True)` on heterogeneous repos triggers `pyarrow.CastError` and parse failures.
- **Studio inefficiency**: repeated studio creation wastes quota (~80 hr/mo) instead of reusing running sessions.
- **Auth-dependent data path**: training relies on authenticated `/api/` calls instead of public CDN URLs, increasing rate-limit pressure.

### Core Principles (resolve contradictions in favor of correctness + actionability)
1. **Correctness**: avoid mixed-schema casts; project to stable `{prompt,response}`; skip corrupt files rather than fail training.
2. **Actionability**: single non-recursive `list_repo_tree` per `(repo,dateFolder)`; persist manifest; use CDN URLs; reuse running Lightning Studio; run on L40S in `lightning-public-prod` when needed.
3. **Robustness**: timeouts, retries, and graceful skip for unavailable/corrupt files; minimal dependencies; clear CLI/env contract.

#### 1) Manifest generator (non-recursive, CDN-only, cached)
`/opt/axentx/vanguard/manifest.py`
```python
#!/usr/bin/env python3
"""
Generate and cache repo+dateFolder file manifests to avoid HF API enumeration
during training. Uses non-recursive tree + CDN-only paths.
"""
from __future__ import annotations

import json
import os
import sys
import time
from pathlib import Path
from typing import List

try:
    from huggingface_hub import HfApi, TreeEntry
except ImportError:
    print("huggingface_hub not installed; install via `pip install huggingface_hub`")
    sys.exit(1)

HF_TOKEN = os.getenv("HF_TOKEN", "")
MANIFEST_ROOT = Path(__file__).parent / "manifests"
MANIFEST_ROOT.mkdir(exist_ok=True, parents=True)

def _cdn_urls(repo: str, file_paths: List[str]) -> List[str]:
    return [f"https://huggingface.co/datasets/{repo}/resolve/main/{p}" for p in file_paths]

def repo_date_manifest(repo: str, date_folder: str, force: bool = False) -> List[str]:
    """
    Return list of file paths for repo/date_folder (non-recursive).
    Cached to manifests/{repo}/{date_folder}.json
    """
    cache_dir = MANIFEST_ROOT / repo
    cache_dir.mkdir(exist_ok=True, parents=True)
    cache_path = cache_dir / f"{date_folder}.json"

    if not force and cache_path.exists():
        try:
            cached = json.loads(cache_path.read_text())
            if isinstance(cached, list) and all(isinstance(x, str) for x in cached):
                return cached
        except Exception:
            pass  # regenerate on corrupt cache

    api = HfApi(token=HF_TOKEN)
    # Non-recursive: one API call, no deep pagination across subfolders
    tree: List[TreeEntry] = list(
        api.list_repo_tree(repo=repo, path=date_folder, recursive=False, token=HF_TOKEN)
    )
    files = sorted(f.rfilename for f in tree if f.type == "file")

    cache_path.write_text(json.dumps(files, indent=2))
    return files

if __name__ == "__main__":
    # CLI: manifest.py <repo> <date_folder> [--force]
    if len(sys.argv) < 3:
        print("Usage: manifest.py <repo> <date_folder> [--force]")
        sys.exit(1)

    r = sys.argv[1]
    d = sys.argv[2]
    force = "--force" in sys.argv
    files = repo_date_manifest(r, d, force=force)
    print(json.dumps({"repo": r, "date_folder": d, "files": files, "urls": _cdn_urls(r, files)}, indent=2))
```

#!/usr/bin/env python3
"""
Surrogate-1 training entrypoint (Lightning Studio).
Uses CDN-only data loading; expects manifest at manifests/{repo}/{date}.json
"""
import json
import os
import sys
import time
import warnings
from pathlib import Path

import pyarrow as pa
import pyarrow.parquet as pq
import requests
import torch
from torch.utils.data import IterableDataset, DataLoader

try:
    from lightning.pytorch import Trainer, LightningModule
except ImportError:
    print("lightning not installed; install via `pip install lightning`")
    sys.exit(1)

MANIFEST_PATH = Path(__file__).parent / "manifests"

class CDNParquetIterable(IterableDataset):
    """
    Stream parquet files via CDN URLs. Projects to {prompt, response} only.
    Avoids load_dataset(streaming=True) and mixed-schema CastError.
    """

    def __init__(self, urls, max_retries: int = 3, timeout: float = 30.0):
        self.urls = urls
        self.max_retries = max_retries
        self.timeout = timeout

    def _fetch_table(self, url: str):
        for attempt in range(1, self.max_retries + 1):
            try:
                resp = requests.get(url, timeout=self.timeout)
                resp.raise_for_status()
                return pq.read_table(pa.BufferReader(resp.content))
            except Exception as exc:
                if attempt == self.max_retries:
                    warnings.warn(f"Failed to fetch {url} after {attempt} attempts: {exc}")
                    return None
                time.sleep(1.0 * attempt)
        return None

    def __iter__(self):
        for url in self.urls:
            table = self._fetch_table(url)
            if table is None:
                continue

            # Project to expected schema; ignore extra columns
            if "prompt" not in table.column_names or "response" not in table.column_names:
                continue

            prompts = table["prompt"].to_pylist()
            responses = table["response"].to_pylist()
            for p, r in zip(prompts, responses):
                # Basic validity filtering
                if isinstance(p, str) and isinstance(r, str) and p.strip() and r.strip():
                    yield {"prompt": p, "response": r}

def load_dataset_from_manifest(repo: str, date_folder: str):
    manifest_file = MANIFEST_PATH / repo / f"{date_folder}.json"
    if not manifest_file.exists():
        raise FileNotFoundError(f"Manifest missing: {manifest_file}. Run manifest.py first.")
    files = json.loads(manifest_file.read_text())
    urls = [f"https://huggingface.co/datasets/{repo}/resolve/main/{p}" for p in files]
    return CDNParquetIterable(urls)
