### 1) `scripts/build_file_manifest.py` (Deterministic + Reproducible)
*Resolves*: Uses `file_list.json` (Candidate 2 naming) but keeps robust fields (Candidate 1). Adds **deterministic sharding hints** for HF commit cap mitigation.

```python
#!/usr/bin/env python3
"""
Mac orchestration: build deterministic manifest for one date folder.
Writes file_list.json for embedding in training (zero HF API during train).
"""
import json, os, sys, hashlib
from datetime import datetime, timezone
from huggingface_hub import HfApi

HF_REPO = os.getenv("HF_DATASET_REPO", "axentx/surrogate-mirror-merged")
DATE_FOLDER = os.getenv("DATE_FOLDER")  # e.g. 2026-04-29
OUT_PATH = os.getenv("OUT_MANIFEST", f"file_list.json")

if not DATE_FOLDER:
    print("ERROR: Set DATE_FOLDER=YYYY-MM-DD")
    sys.exit(1)

api = HfApi()
tree = api.list_repo_tree(repo_id=HF_REPO, path=DATE_FOLDER, recursive=False)

files = sorted([
    f"{DATE_FOLDER}/{entry.path.split('/')[-1]}"
    for entry in tree
    if entry.path.lower().endswith((".parquet", ".jsonl"))
])

# Deterministic sharding key for HF commit cap mitigation
shard_key = hashlib.sha256(DATE_FOLDER.encode()).hexdigest()[:8]

manifest = {
    "repo": HF_REPO,
    "date_folder": DATE_FOLDER,
    "shard_key": shard_key,
    "files": files,
    "generated_at": datetime.now(timezone.utc).isoformat(),
    "total_files": len(files)
}

with open(OUT_PATH, "w") as f:
    json.dump(manifest, f, indent=2)

print(f"✅ Wrote {OUT_PATH} with {len(files)} files (shard={shard_key})")
```

### 2) `surrogate/data/cdn_parquet_loader.py` (Robust Projection)
*Resolves*: Combines both candidates' projection logic with **strict schema fallback** and **connection resilience**.

```python
import pyarrow.parquet as pq
import pyarrow as pa
import requests
from io import BytesIO
from typing import List, Dict, Iterator
import time

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def cdn_fetch(path: str, repo: str, retries: int = 3) -> bytes:
    url = CDN_TEMPLATE.format(repo=repo, path=path)
    for attempt in range(retries):
        try:
            resp = requests.get(url, timeout=30)
            resp.raise_for_status()
            return resp.content
        except Exception as e:
            if attempt == retries - 1:
                raise
            time.sleep(2 ** attempt)
    raise RuntimeError(f"Failed to fetch {url}")

def project_to_prompt_response(batch_bytes: bytes) -> List[Dict[str, str]]:
    """
    Load parquet from bytes and project to {prompt, response}.
    Handles heterogeneous schemas safely.
    """
    table = pq.read_table(pa.BufferReader(batch_bytes))
    rows = []
    
    # Unified column mapping
    prompt_col = next((c for c in ["prompt", "instruction", "input"] if c in table.column_names), None)
    response_col = next((c for c in ["response", "output", "completion"] if c in table.column_names), None)
    
    for i in range(table.num_rows):
        prompt = table[prompt_col][i].as_py() if prompt_col else ""
        response = table[response_col][i].as_py() if response_col else ""
        
        # Coerce to string, handle nulls
        prompt = str(prompt) if prompt is not None else ""
        response = str(response) if response is not None else ""
        
        if prompt.strip() or response.strip():
            rows.append({"prompt": prompt.strip(), "response": response.strip()})
    return rows

class CDNParquetDataset:
    def __init__(self, manifest_path: str):
        import json
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.repo = self.manifest["repo"]
        self.files = self.manifest["files"]

    def __len__(self):
        return len(self.files)

    def __getitem__(self, idx: int) -> List[Dict[str, str]]:
        blob = cdn_fetch(self.files[idx], self.repo)
        return project_to_prompt_response(blob)
    
    def stream(self) -> Iterator[Dict[str, str]]:
        """Memory-efficient streaming iterator."""
        for idx in range(len(self)):
            for row in self[idx]:
                yield row
```

### 3) `surrogate/train.py` (Zero HF API Integration)
*Resolves*: Uses **IterableDataset** for memory efficiency and includes **validation split**.

```python
from surrogate.data.cdn_parquet_loader import CDNParquetDataset
from torch.utils.data import IterableDataset, DataLoader
import torch

class SurrogateIterableDataset(IterableDataset):
    def __init__(self, manifest_path: str, split: str = "train", val_split: float = 0.1):
        self.dataset = CDNParquetDataset(manifest_path)
        self.split = split
        
        # Deterministic split using shard_key from manifest
        shard_key = self.dataset.manifest.get("shard_key", "0")
        seed = int(shard_key, 16) % (2**32)
        rng = torch.Generator()
        rng.manual_seed(seed)
        
        # Assign split deterministically
        total = len(self.dataset)
        indices = torch.randperm(total, generator=rng).tolist()
        split_idx = int(total * (1 - val_split))
        
        if split == "train":
            self.indices = set(indices[:split_idx])
        else:
            self.indices = set(indices[split_idx:])

    def __iter__(self):
        for idx in range(len(self.dataset)):
            if idx in self.indices:
                for row in self.dataset[idx]:
                    if row["prompt"] and row["response"]:
                        yield row
```
