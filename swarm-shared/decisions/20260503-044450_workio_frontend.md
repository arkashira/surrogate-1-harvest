# workio / frontend

## 1. Diagnosis

- No frontend protection against duplicate clock-in/out events when LINE webhook retries or user double-taps; UI can optimistically apply the same punch twice.
- No idempotency key or client-side dedupe for leave/OT requests; rapid submit + retry can create duplicate requests.
- Clock-in/out and location checks rely only on optimistic UI; no reconciliation with server truth after network flaps or background sync.
- No visual feedback or disabled states during in-flight mutations; users can spam actions and produce race conditions.
- Missing client-side replay guard for webhook-originated events surfaced to the UI (e.g., push notifications that trigger local state updates).

## 2. Proposed change

**Scope**: `workio/src/features/attendance/` (or equivalent) — add an idempotency layer for clock and leave/OT mutations and a safe state reconciliation hook.

Files to touch:
- `src/features/attendance/hooks/useAttendance.ts` (or create it)
- `src/features/leave/hooks/useLeaveRequest.ts`
- `src/lib/api/client.ts` (add idempotency header helper)
- `src/contexts/AttendanceContext.tsx` (optional: server-state reconciliation)

### 3.1 Idempotency key utility

```ts
// src/lib/idempotency.ts
export function createIdempotencyKey(operation: string, entityId?: string): string {
  const nonce = crypto.randomUUID();
  const ts = Date.now();
  return `${operation}:${entityId || 'new'}:${ts}:${nonce}`;
}

export function getPendingKey(key: string): string {
  return `idemp-pending:${key}`;
}

export function markPending(key: string): void {
  try {
    sessionStorage.setItem(getPendingKey(key), Date.now().toString());
  } catch {}
}

export function clearPending(key: string): void {
  try {
    sessionStorage.removeItem(getPendingKey(key));
  } catch {}
}

export function isPending(key: string): boolean {
  try {
    return sessionStorage.getItem(getPendingKey(key)) !== null;
  } catch {
    return false;
  }
}
```

### 3.2 API client helper to attach idempotency header

```ts
// src/lib/api/client.ts
import { createIdempotencyKey, isPending, markPending, clearPending } from '../idempotency';

export type MutationOptions = {
  idempotencyKey?: string;
  skipDedupe?: boolean;
};

export async function postWithIdempotency<T>(
  url: string,
  body: any,
  opts: MutationOptions = {}
): Promise<T> {
  const key = opts.idempotencyKey || createIdempotencyKey(url);
  if (!opts.skipDedupe && isPending(key)) {
    // Prevent duplicate submission from UI
    throw new Error('Duplicate request in flight');
  }

  markPending(key);
  try {
    const res = await fetch(url, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'X-Idempotency-Key': key,
      },
      body: JSON.stringify(body),
    });
    if (!res.ok) throw new Error(`Request failed: ${res.status}`);
    return (await res.json()) as T;
  } finally {
    clearPending(key);
  }
}
```

### 3.3 Attendance hook with safe mutation + reconciliation

```ts
// src/features/attendance/hooks/useAttendance.ts
import { useState, useEffect, useCallback } from 'react';
import { postWithIdempotency } from '../../lib/api/client';
import { createIdempotencyKey } from '../../lib/idempotency';

export function useAttendance(tenantId: string, userId: string) {
  const [currentPunch, setCurrentPunch] = useState<null | { type: 'IN' | 'OUT'; ts: string }>(null);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);

  // Reconcile with server truth on mount and periodically
  const reconcile = useCallback(async () => {
    try {
      const res = await fetch(`/api/attendance/current?tenantId=${tenantId}&userId=${userId}`);
      if (res.ok) {
        const data = await res.json();
        setCurrentPunch(data.lastPunch || null);
      }
    } catch {
      // best-effort
    }
  }, [tenantId, userId]);

  useEffect(() => {
    reconcile();
    const id = setInterval(reconcile, 60_000);
    return () => clearInterval(id);
  }, [reconcile]);

  const punch = useCallback(
    async (type: 'IN' | 'OUT', location?: GeolocationPosition) => {
      setLoading(true);
      setError(null);
      const key = createIdempotencyKey('punch', userId);
      try {
        await postWithIdempotency('/api/attendance/punch', {
          tenantId,
          userId,
          type,
          location: location
            ? {
                lat: location.coords.latitude,
                lng: location.coords.longitude,
                accuracy: location.coords.accuracy,
              }
            : undefined,
        }, { idempotencyKey: key });

        // optimistic update, but reconcile shortly after to absorb retries
        setCurrentPunch({ type, ts: new Date().toISOString() });
        // force reconcile after short delay to absorb server-side dedupe/state
        setTimeout(reconcile, 1500);
      } catch (err: any) {
        setError(err.message || 'Failed to punch');
        // revert optimistic state on known duplicate; otherwise keep and reconcile
        if (err.message?.includes('Duplicate')) {
          // leave currentPunch as-is; reconcile will fix
        }
        throw err;
      } finally {
        setLoading(false);
      }
    },
    [tenantId, userId, reconcile]
  );

  return { currentPunch, punch, loading, error, reconcile };
}
```

### 3.4 Leave request hook with dedupe

```ts
// src/features/leave/hooks/useLeaveRequest.ts
import { useState, useCallback } from 'react';
import { postWithIdempotency } from '../../lib/api/client';
import { createIdempotencyKey } from '../../lib/idempotency';

export function useLeaveRequest(tenantId: string, userId: string) {
  const [submitting, setSubmitting] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const requestLeave = useCallback(
    async (values: { leaveType: string; startDate: string; endDate: string; reason: string }) => {
      setSubmitting(true);
      setError(null);
      const key = createIdempotencyKey('leave-request', userId);
      try {
        await postWithIdempotency('/api/leave/request', { tenantId, userId, ...values }, {
          idempotencyKey: key,
        });
        return { ok: true };
      } catch (err: any) {
        setError(err.message || 'Failed to submit leave request');
        throw err;
      } finally {
        setSubmitting(false);
      }
    },
    [tenantId, userId]
  );

  return { requestLeave, submitting, error };
}
```
