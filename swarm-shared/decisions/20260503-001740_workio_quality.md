# workio / quality

## Final Synthesis (Best Parts + Correctness + Actionable)

**Core diagnosis (merged):**  
- No idempotency key on punch submissions → duplicates from double-taps, retries, and LINE webhook replays.  
- No optimistic UI / debounce → users spam and queue concurrent requests.  
- No client-side state-transition validation → invalid records reach the server.  
- No server-side unique guard → duplicates can still be persisted.  
- No visual feedback during submit → perceived failures and retries.

**Chosen approach (merged + resolved contradictions):**  
- **Frontend:** optimistic UI, client-side transition guard, and per-attempt idempotency key (in-memory + persisted to request).  
- **Backend:** idempotency-key deduplication **plus** a data-level unique constraint as the source-of-truth backstop.  
- **Database:** add `idempotency_key` column + unique constraint on `(user_id, date, type)` to guarantee exactly-one punch per user per day per type.

### 1) Database (source-of-truth backstop)

`workio/server/src/db/schema.sql`
```sql
-- Idempotency column
ALTER TABLE punches ADD COLUMN IF NOT EXISTS idempotency_key TEXT;
CREATE INDEX IF NOT EXISTS idx_punches_idempotency_key ON punches(idempotency_key);

-- Guarantee at most one punch per user per calendar day per type
-- (keeps data clean even if application logic is bypassed)
CREATE UNIQUE INDEX IF NOT EXISTS idx_user_date_type_unique
ON punches (user_id, DATE(timestamp), type)
WHERE type IN ('in', 'out');
```

### 2) Backend route (idempotency + constraint handling)

`workio/server/src/routes/punch.ts`
```ts
import express from 'express';
import { db } from '../db';

const router = express.Router();

router.post('/', async (req, res) => {
  const { type, idempotencyKey } = req.body;
  const userId = (req as any).user?.id;

  if (!userId || !type || !idempotencyKey) {
    return res.status(400).json({ error: 'Missing required fields' });
  }

  const client = await db.pool.connect();
  try {
    await client.query('BEGIN');

    // 1) Idempotency check: same key -> return existing
    const idem = await client.query(
      'SELECT id, user_id, timestamp, type FROM punches WHERE idempotency_key = $1',
      [idempotencyKey]
    );

    if (idem.rows.length > 0) {
      await client.query('COMMIT');
      return res.json({ punch: idem.rows[0], duplicate: true });
    }

    // 2) Insert (unique constraint is backstop)
    const result = await client.query(
      `INSERT INTO punches (user_id, type, timestamp, idempotency_key)
       VALUES ($1, $2, NOW(), $3)
       RETURNING id, user_id, timestamp, type`,
      [userId, type, idempotencyKey]
    );

    await client.query('COMMIT');
    return res.json({ punch: result.rows[0], duplicate: false });
  } catch (err: any) {
    await client.query('ROLLBACK');

    // 3) Unique constraint violation: return existing punch for this user/date/type
    if (err.code === '23505') {
      const existing = await client.query(
        `SELECT id, user_id, timestamp, type
         FROM punches
         WHERE user_id = $1 AND DATE(timestamp) = CURRENT_DATE AND type = $2
         ORDER BY timestamp DESC LIMIT 1`,
        [userId, type]
      );
      return res.json({ punch: existing.rows[0] || null, duplicate: true });
    }

    return res.status(500).json({ error: 'Internal server error' });
  } finally {
    client.release();
  }
});

export default router;
```

### 3) Frontend hook (idempotency + optimistic UI + transition guard)

`workio/src/lib/punch.ts`
```ts
export function generateIdempotencyKey(): string {
  return `${Date.now()}-${Math.random().toString(36).slice(2, 11)}`;
}

export function canTransition(
  lastType: 'in' | 'out' | null,
  nextType: 'in' | 'out'
): boolean {
  // Valid transitions: null -> in, in -> out, out -> in
  if (nextType === 'in') return lastType === 'out' || lastType === null;
  if (nextType === 'out') return lastType === 'in';
  return false;
}
```

`workio/src/hooks/usePunch.ts`
```ts
import { useState, useCallback } from 'react';
import { generateIdempotencyKey, canTransition } from '../lib/punch';

type PunchState = 'idle' | 'submitting' | 'success' | 'error';
type PunchType = 'in' | 'out';

export function usePunch(lastKnownType: PunchType | null) {
  const [state, setState] = useState<PunchState>('idle');

  const punch = useCallback(
    async (type: PunchType) => {
      if (state === 'submitting') return;
      if (!canTransition(lastKnownType, type)) {
        setState('error');
        return;
      }

      const idempotencyKey = generateIdempotencyKey();
      setState('submitting');

      try {
        const res = await fetch('/api/punch', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ type, idempotencyKey }),
        });

        if (!res.ok) throw new Error('Punch failed');
        setState('success');
      } catch {
        setState('error');
      }
    },
    [state, lastKnownType]
  );

  return { state, punch };
}
```

### 4) Frontend component (optimistic UI + disabled states)

`workio/src/components/PunchButton.tsx`
```tsx
import React from 'react';
import { usePunch } from '../hooks/usePunch';

type PunchButtonProps = {
  lastKnownType: 'in' | 'out' | null;
};

export function PunchButton({ lastKnownType }: PunchButtonProps) {
  const { state, punch } = usePunch(lastKnownType);
  const isSubmitting = state === 'submitting';

  return (
    <div className="flex gap-3">
      <button
        onClick={() => punch('in')}
        disabled={isSubmitting || !canTransition(lastKnownType, 'in')}
        className={`px-5 py-2.5 rounded font-medium transition ${
          isSubmitting || !canTransition(lastKnownType, 'in')
            ? 'bg-gray-300 text-gray-500 cursor-not-allowed'
            : 'bg-green-600 hover:bg-green-700 text-white'
        }`}
      >
        {isSubmitting ? 'Clocking in...' : 'Clock In'}
      </button>

      <button
        onClick={() => punch('out')}
        disabled={isSubmitting || !canTransition(lastKnownType, 'out')}
        className={`px-5 py-2.5 rounded font-medium transition ${
          isSubmitting || !canTransition(lastKnownType, 'out')
            ? 'bg-gray-300 text-gray-500 cursor-not-allowed'
            : 'bg-red-600 hover:bg-red-700 text-white'
        }`}
      >
        {isSubmitting ? 'Clocking out...' : 'Clock Out'}
      </button>
    </div>
  );
}

function canTransition(lastType: 'in' | 'out' | null, nextType: 'in' | 'out') {
  if (nextType === 'in') return lastType === 'out' || lastType === null;
  if (nextType === 'out') return lastType === 'in';
  return false;
}
```
