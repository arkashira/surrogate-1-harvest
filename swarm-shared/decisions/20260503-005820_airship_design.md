## Goal
Implement a **manifest-driven, CDN-only dataset loader** for Surrogate-1 training that:
- Pre-lists the target date folder **once** (Mac orchestration) → saves `dataset_manifest.json`
- Embeds the manifest in `train.py`
- Lightning Studio training fetches **only via CDN** (`resolve/main/...`) → **zero HF API calls during data load**
- Projects heterogeneous repo files to `{prompt, response}` at parse time (avoids `pyarrow.CastError`)
- Reuses running Studio to preserve quota
- Survives idle-stop by checking status before `.run()`

### 1) `scripts/list_hf_date_folder.py`

```python
#!/usr/bin/env python3
"""
List files in a date folder of a Hugging Face dataset repo.
Outputs a JSON file with paths usable for CDN fetches.

Usage:
  python scripts/list_hf_date_folder.py \
    --repo surrogate-1-mirror \
    --date 2026-04-29 \
    --out dataset_manifest.json
"""

import argparse
import json
import os
import time
from typing import List

from huggingface_hub import HfApi

HF_API_RATE_LIMIT_RETRY = 360  # seconds (per pattern)

def list_date_folder(repo_id: str, date: str, max_retries: int = 3) -> List[str]:
    api = HfApi()
    folder_path = f"batches/mirror-merged/{date}"

    for attempt in range(1, max_retries + 1):
        try:
            # recursive=False avoids heavy pagination; we only want direct files in date folder
            items = list_repo_tree(repo_id=repo_id, path=folder_path, recursive=False)
            # Keep only files (not subdirs) and use POSIX paths
            files = [it.rfilename for it in items if it.type == "file"]
            return files
        except Exception as exc:
            if attempt == max_retries:
                raise
            # If 429-like, wait and retry
            print(f"Attempt {attempt}/{max_retries} failed: {exc}. Retrying in {HF_API_RATE_LIMIT_RETRY}s")
            time.sleep(HF_API_RATE_LIMIT_RETRY)

    return []

def main() -> None:
    parser = argparse.ArgumentParser(description="List HF dataset date folder for CDN loader")
    parser.add_argument("--repo", required=True, help="HF dataset repo id (e.g. surrogate-1-mirror)")
    parser.add_argument("--date", required=True, help="Date folder (YYYY-MM-DD)")
    parser.add_argument("--out", default="dataset_manifest.json", help="Output JSON path")
    args = parser.parse_args()

    files = list_date_folder(args.repo, args.date)
    manifest = {
        "repo_id": args.repo,
        "date": args.date,
        "files": sorted(files),
        "cdn_prefix": f"https://huggingface.co/datasets/{args.repo}/resolve/main"
    }

    os.makedirs(os.path.dirname(os.path.abspath(args.out)) or ".", exist_ok=True)
    with open(args.out, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Wrote {len(files)} files to {args.out}")

if __name__ == "__main__":
    main()
```

### 2) `train.py` (excerpt: CDN-only loader + projection)

```python
import json
import logging
from pathlib import Path
from typing import Dict, Iterator, Any

import pyarrow as pa
import pyarrow.parquet as pq
import requests
from torch.utils.data import IterableDataset

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger("surrogate-train")

MANIFEST_PATH = Path(__file__).parent / "dataset_manifest.json"


class CDNParquetDataset(IterableDataset):
    """
    CDN-only dataset loader:
    - Uses pre-listed manifest (zero HF API calls during training)
    - Downloads parquet files via public CDN URLs
    - Projects each row to {prompt, response} (avoids mixed-schema CastError)
    """

    def __init__(self, manifest_path: Path = MANIFEST_PATH):
        super().__init__()
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.cdn_prefix = self.manifest["cdn_prefix"]
        self.files = [f for f in self.manifest["files"] if f.endswith(".parquet")]
        if not self.files:
            raise ValueError("No parquet files found in manifest")

    def _fetch_parquet(self, rel_path: str) -> pa.Table:
        url = f"{self.cdn_prefix}/{rel_path}"
        logger.info("Fetching CDN: %s", url)
        resp = requests.get(url, timeout=120)
        resp.raise_for_status()
        return pq.read_table(pa.BufferReader(resp.content))

    def _project_row(self, row: Dict[str, Any]) -> Dict[str, str]:
        # Keep only prompt/response; ignore mixed schema extras
        return {
            "prompt": str(row.get("prompt", row.get("input", ""))),
            "response": str(row.get("response", row.get("output", ""))),
        }

    def __iter__(self) -> Iterator[Dict[str, str]]:
        for rel_path in self.files:
            table = self._fetch_parquet(rel_path)
            for batch in table.to_batches():
                for row in batch.to_pylist():
                    yield self._project_row(row)
```
