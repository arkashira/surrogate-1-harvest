# workio / quality

## Final synthesized implementation

**Chosen approach**: merge the strongest, most concrete pieces from both proposals into one minimal, production-ready solution. Keep the same file paths and API contract Candidate 1 proposed (`/api/clock` with `{ type, userId, latitude?, longitude?, idempotencyKey }`).

### Key decisions (resolve contradictions)
- **Idempotency**: use a stable client-side key per user+date+action (`userId-date-type`) **plus** a short random suffix to avoid cross-tab/retry collisions, but rely on server unique constraint (`user_id, date, type`) as the source of truth. Treat 409 as “duplicate”.
- **Geolocation**: require location for clock-in/out, but degrade gracefully with a clear UX (allow action with warning or block with explanation) instead of silent failure. Use a timeout and clear errors.
- **UI state**: single pending/success/duplicate/error state with disabled lock during flight and auto-reset. Do not allow multiple in-flight requests.
- **Feedback**: inline button state + accessible live region + toast for visibility.
- **Scope**: one hook (`useClock`) + one component (`ClockButton`). No extra libraries required beyond axios (or swap to fetch).

### 1) Hook: `workio/src/hooks/useClock.ts`

```ts
// workio/src/hooks/useClock.ts
import { useState, useCallback, useRef } from 'react';
import axios from 'axios';

type ClockType = 'IN' | 'OUT';
type Status = 'idle' | 'pending' | 'success' | 'error' | 'duplicate';

export interface ClockResult {
  status: Status;
  message: string;
  data?: any;
  locationMissing?: boolean;
}

const LOCATION_TIMEOUT = 8000;

function makeIdempotencyKey(userId: string, type: ClockType) {
  const today = new Date().toISOString().slice(0, 10); // YYYY-MM-DD
  const random = Math.random().toString(36).slice(2, 8);
  return `${userId}-${today}-${type}-${random}`;
}

function getLocation(timeoutMs = LOCATION_TIMEOUT): Promise<{ latitude: number; longitude: number } | null> {
  return new Promise((resolve) => {
    if (!navigator.geolocation) {
      resolve(null);
      return;
    }

    let timedOut = false;
    const timer = setTimeout(() => {
      timedOut = true;
      resolve(null);
    }, timeoutMs);

    navigator.geolocation.getCurrentPosition(
      (pos) => {
        if (timedOut) return;
        clearTimeout(timer);
        resolve({
          latitude: pos.coords.latitude,
          longitude: pos.coords.longitude,
        });
      },
      () => {
        if (timedOut) return;
        clearTimeout(timer);
        resolve(null);
      },
      { enableHighAccuracy: true, timeout: timeoutMs, maximumAge: 30000 }
    );
  });
}

export function useClock() {
  const [status, setStatus] = useState<Status>('idle');
  const [lastResult, setLastResult] = useState<ClockResult | null>(null);
  const inflightRef = useRef(false);

  const clock = useCallback(
    async (type: ClockType, userId: string, requireLocation = false): Promise<ClockResult> => {
      if (inflightRef.current || status === 'pending') {
        // Prevent queueing multiple requests
        const out: ClockResult = { status: 'pending', message: 'Request already in progress' };
        setStatus('pending');
        setLastResult(out);
        return out;
      }

      inflightRef.current = true;
      setStatus('pending');

      const idempotencyKey = makeIdempotencyKey(userId, type);
      const location = await getLocation();

      if (requireLocation && !location) {
        inflightRef.current = false;
        const out: ClockResult = {
          status: 'error',
          message: 'Location unavailable. Enable GPS to clock in/out.',
          locationMissing: true,
        };
        setStatus('error');
        setLastResult(out);
        return out;
      }

      try {
        const res = await axios.post('/api/clock', {
          type,
          userId,
          ...(location && { latitude: location.latitude, longitude: location.longitude }),
          idempotencyKey,
        });

        // Server may indicate duplicate explicitly
        if (res.data?.duplicate) {
          const out: ClockResult = { status: 'duplicate', message: 'Already recorded for today', data: res.data };
          setStatus('duplicate');
          setLastResult(out);
          inflightRef.current = false;
          return out;
        }

        const out: ClockResult = { status: 'success', message: `Clocked ${type}`, data: res.data };
        setStatus('success');
        setLastResult(out);
        inflightRef.current = false;
        return out;
      } catch (err: any) {
        const isConflict = err?.response?.status === 409;
        const out: ClockResult = {
          status: isConflict ? 'duplicate' : 'error',
          message: isConflict ? 'Already recorded' : err?.message || 'Request failed',
          data: err?.response?.data,
        };
        setStatus(out.status);
        setLastResult(out);
        inflightRef.current = false;
        return out;
      }
    },
    [status]
  );

  const reset = useCallback(() => {
    setStatus('idle');
    setLastResult(null);
  }, []);

  return { clock, status, lastResult, reset };
}
```

### 2) Component: `workio/src/features/clock/ClockButton.tsx`

```tsx
// workio/src/features/clock/ClockButton.tsx
import React, { useEffect } from 'react';
import { useClock } from '../../hooks/useClock';
import './ClockButton.css';

interface Props {
  userId: string;
  type: 'IN' | 'OUT';
  requireLocation?: boolean;
  autoResetMs?: number;
}

export const ClockButton: React.FC<Props> = ({
  userId,
  type,
  requireLocation = false,
  autoResetMs = 4000,
}) => {
  const { clock, status, lastResult, reset } = useClock();

  useEffect(() => {
    if (status === 'success' || status === 'duplicate' || status === 'error') {
      const timer = setTimeout(reset, autoResetMs);
      return () => clearTimeout(timer);
    }
  }, [status, reset, autoResetMs]);

  const handleClick = async () => {
    await clock(type, userId, requireLocation);
  };

  const isPending = status === 'pending';
  const isSuccess = status === 'success';
  const isDuplicate = status === 'duplicate';
  const isError = status === 'error';

  const label = (() => {
    if (isPending) return '⏳ Processing...';
    if (isSuccess) return '✅ Clocked';
    if (isDuplicate) return 'ℹ️ Already recorded';
    if (isError) return '❌ Failed';
    return `Clock ${type}`;
  })();

  return (
    <div className="clock-button-wrapper">
      <button
        className={`clock-button ${isPending ? 'pending' : ''} ${isSuccess ? 'success' : ''} ${
          isDuplicate ? 'duplicate' : ''
        } ${isError ? 'error' : ''}`}
        onClick={handleClick}
        disabled={isPending}
        aria-busy={isPending}
        aria-live="polite"
      >
        {label}
      </button>

      {lastResult?.message && !isPending && (
        <div className="clock-feedback" role="status" aria-live="polite">
          {lastResult.message}
        </div>
      )}
    </div>
  );
};
```
