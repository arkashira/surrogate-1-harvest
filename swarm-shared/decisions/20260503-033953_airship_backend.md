### 3. Add Lightning idle-resilient runner

- Shebang `#!/usr/bin/env bash`

### `scripts/generate-cdn-manifest.sh`
```bash
#!/usr/bin/env bash
set -euo pipefail

REPO="axentx/surrogate-training"
DATE=$(date +%Y-%m-%d)
OUT_DIR="surrogate/training-manifests"
OUT_FILE="${OUT_DIR}/manifest-${DATE}.json"

mkdir -p "${OUT_DIR}"

# Single API call: list one date folder (non-recursive)
python3 - <<PY
import os, json, sys
from huggingface_hub import list_repo_tree

REPO = os.getenv("REPO")
DATE = os.getenv("DATE")

items = list_repo_tree(REPO, path=DATE, recursive=False)
manifest = []
for item in items:
    if item.path.endswith(".parquet"):
        manifest.append({
            "repo": REPO,
            "path": item.path,
            "sha": item.commit_id or "main",
            "size": item.size or 0
        })

out_file = os.getenv("OUT_FILE")
with open(out_file, "w") as f:
    json.dump(manifest, f, indent=2)
print(f"Wrote {len(manifest)} files to {out_file}")
PY
```

### `surrogate/train.py` (CDN-only loader snippet)
```python
import json
import pyarrow.parquet as pq
import requests
from io import BytesIO
from pathlib import Path

def load_parquet_via_cdn(manifest_path: str):
    """Load parquet files via CDN only (zero HF API calls)."""
    with open(manifest_path) as f:
        files = json.load(f)

    rows = []
    for fmeta in files:
        url = f"https://huggingface.co/datasets/{fmeta['repo']}/resolve/main/{fmeta['path']}"
```

### `surrogate/lightning_runner.py` (idle-resilient)
```python
from lightning.pai import Studio, Machine
import time

def run_with_idle_resilience(studio_name: str, script: str, machine=Machine.L40S):
    studio = Studio(name=studio_name, create_ok=True)
    max_retries = 3

    for attempt in range(max_retries):
        # Check status before run
        studio.refresh()
        if studio.status != "running":
            print(f"Studio stopped, restarting on {machine}...")
            studio.start(machine=machine)
            studio.wait_for_running()

        try:
            run = studio.run(script, wait=True)
            if run.status == "success":
                return run
        except Exception as e:
            print(f"Run failed (attempt {attempt+1}): {e}")
            if attempt == max_retries - 1:
                raise
            time.sleep(60)
    return None
```

### `scripts/surrogate-train-wrapper.sh`
```bash
#!/usr/bin/env bash
set -euo pipefail
SHELL=/bin/bash

MANIFEST_DATE=$(date +%Y-%m-%d)
MANIFEST_FILE="surrogate/training-manifests/manifest-${MANIFEST_DATE}.json"

# Generate manifest if missing
if [[ ! -f "${MANIFEST_FILE}" ]]; then
    bash scripts/generate-cdn-manifest.sh
fi

# Run resilient training
python3 surrogate/lightning_runner.py \
    --studio-name surrogate-training \
    --script "surrogate/train.py --manifest ${MANIFEST_FILE}"
```
