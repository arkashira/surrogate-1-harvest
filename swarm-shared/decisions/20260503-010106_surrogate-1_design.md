# surrogate-1 / design

## Final Implementation Plan (≤2h)

**Highest-value change**: Eliminate HF API rate-limit failures during ingestion by switching to CDN-bypass ingestion with a pre-computed manifest. This preserves 16-shard deterministic parallelism and central dedup while removing all API calls from the worker hot path.

**Total time**: ~90 min (20 min manifest generator, 45 min worker refactor, 25 min test/validate).

### 1) Manifest generator (run once per cron, before workers start)

File: `bin/gen-manifest.py`

```python
#!/usr/bin/env python3
"""
Generate manifest-<date>.json for CDN-bypass ingestion.
Run from Mac/CI after rate-limit window clears.

Outputs: manifests/manifest-2026-04-29.json
  {
    "date": "2026-04-29",
    "files": [
      {"path": "2026-04-29/part-00000.parquet", "size": 12345, "etag": "...", "shard_hint": 3},
      ...
    ]
  }
"""

import hashlib
import json
import os
import sys
from datetime import datetime, timezone

from huggingface_hub import HfApi

REPO_ID = "axentx/surrogate-1-training-pairs"
MANIFEST_DIR = "manifests"

def main(date_str: str = None):
    if date_str is None:
        date_str = datetime.now(timezone.utc).strftime("%Y-%m-%d")

    api = HfApi()
    entries = api.list_repo_tree(
        repo_id=REPO_ID,
        path=date_str,
        repo_type="dataset",
    )

    files = []
    for e in entries:
        if not e.rfilename.endswith((".parquet", ".jsonl")):
            continue
        shard_hint = int(hashlib.sha256(e.rfilename.encode()).hexdigest(), 16) % 16
        files.append({
            "path": e.rfilename,
            "size": getattr(e, "size", 0),
            "etag": getattr(e, "etag", ""),
            "shard_hint": shard_hint,
        })

    # deterministic ordering
    files.sort(key=lambda x: x["path"])

    manifest = {"date": date_str, "files": files}
    os.makedirs(MANIFEST_DIR, exist_ok=True)
    out_path = os.path.join(MANIFEST_DIR, f"manifest-{date_str}.json")
    with open(out_path, "w") as f:
        json.dump(manifest, f, indent=2)

    print(f"Wrote {len(files)} files to {out_path}")

if __name__ == "__main__":
    main(*sys.argv[1:])
```


### 2) Refactored worker: CDN-only ingestion with manifest

File: `bin/dataset-enrich.sh`

#!/usr/bin/env bash
# surrogate-1 dataset-enrich worker (CDN-bypass mode)
#
# Usage:
#   SHARD_ID=0 NUM_SHARDS=16 ./dataset-enrich.sh [manifest.json]
#
# Behavior:
# - Reads manifest produced by bin/gen-manifest.py
# - Each shard deterministically selects entries by pre-computed shard_hint
# - Downloads via CDN (no HF API auth/rate-limit during data load)
# - Projects to {prompt,response}, dedups via lib/dedup.py
# - Outputs: batches/public-merged/<date>/shard<N>-<HHMMSS>.jsonl

set -euo pipefail

REPO="axentx/surrogate-1-training-pairs"
BASE_URL="https://huggingface.co/datasets/${REPO}/resolve/main"
MANIFEST_PATH="${1:-manifests/manifest-$(date -u +%Y-%m-%d).json}"

SHARD_ID="${SHARD_ID:-0}"
NUM_SHARDS="${NUM_SHARDS:-16}"
DATE="$(date -u +%Y-%m-%d)"
TS="$(date -u +%H%M%S)"
OUT_DIR="batches/public-merged/${DATE}"
OUT_FILE="${OUT_DIR}/shard${SHARD_ID}-${TS}.jsonl"

mkdir -p "${OUT_DIR}"

log() {
  echo "[$(date -u +%Y-%m-%dT%H:%M:%SZ)] shard=${SHARD_ID} $*"
}

# ---- validate ----
if [[ ! -f "${MANIFEST_PATH}" ]]; then
  log "ERROR: ${MANIFEST_PATH} not found. Run bin/gen-manifest.py first."
  exit 1
fi

# Select entries by pre-computed shard_hint
mapfile -t SHARD_FILES < <(python3 -c "
import json, sys
with open('${MANIFEST_PATH}') as f:
    data = json.load(f)
for entry in data['files']:

")

fi

cleanup() {
  rm -rf "${TMP_DIR}"
}

cat > "${PROJECT_PY}" <<'PYEOF'
import pyarrow.parquet as pq
import json
import os
import sys

def extract_text(val):
    if val is None:
        return ""
    if isinstance(val, str):
        return val.strip()
    if isinstance(val, (list, tuple)):
        return " ".join(str(v) for v in val if v is not None).strip()
    return str(val).strip()

def project_file(path, out_f):
    ext = os.path.splitext(path)[1].lower()
    try:
        if ext == ".parquet":
            tbl = pq.read_table(path)
            cols = tbl.column_names
            prompt_col = next((c for c in cols if "prompt" in c.lower()), cols[0] if cols else None)
            resp_col = next((c for c in cols if "response" in c.lower() or "completion" in c.lower()), cols[1] if len(cols) > 1 else (cols[0] if cols else None))

            for batch in tbl.to_batches(max_chunksize=8192):
                for i in range(batch.num_rows):
                    prompt = extract_text(batch[prompt_col][i].as_py() if prompt_col else None)
                    response = extract_text(batch[resp_col][i].as_py() if resp_col else None)
                    if prompt and response:
                        out_f.write(json.dumps({"prompt": prompt, "response": response}, ensure_ascii=False) + "\n")

        elif ext == ".jsonl":
            with open(path, "r", encoding="utf-8", errors="replace") as f:
                for line in f:
                    line = line.strip()
                    if not line:
                        continue
                    try:
                        obj = json.loads(line)
                    except Exception:
                        continue
                    if isinstance(obj, dict):
                        prompt = extract_text(obj.get("prompt") or obj.get("input") or list(obj.values())[0] if obj else None)
                        response = extract_text(obj.get("response") or obj.get("output") or list(obj.values())[1] if len(obj) > 1 else None)
                        if prompt and response:
                            out_f.write(json.dumps({"prompt": prompt, "response": response}, ensure_ascii=False) + "\n")
    except Exception as e:
        sys.stderr.write(f"WARN: failed to project {path}: {e}\n")

if __name__ == "__main__":
    _, src, dst = sys.argv
    with open(dst, "w", encoding="utf-8") as out_f:
        project_file(src, out_f)
PYEOF
