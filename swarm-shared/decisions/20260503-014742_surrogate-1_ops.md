"""
Manifest-driven, CDN-bypass ingestion worker for surrogate-1.

Usage:
  SHARD_ID=0 python bin/dataset-enrich.py \
    --repo axentx/surrogate-1-training-pairs \
    --date 2026-05-03 \
    --manifest manifests/2026-05-03-tree.json \
    --out-dir batches/public-merged

Behavior:
- Loads manifest (local file or URL).
- Selects files assigned to SHARD_ID by hash(slug) % 16.
- Downloads via HF CDN (no Authorization header; bypasses API rate limits).
- Projects to {prompt, response} per known schema.
- Deduplicates via lib.dedup (central md5 store).
- Writes shard-N-<ts>.jsonl.

sys.path.insert(0, str(Path(__file__).parent.parent))
from lib.dedup import DedupStore  # type: ignore

HF_CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"
RETRY_WAIT = 360  # seconds after 429
SHARD_COUNT = 16

def slug_hash(slug: str) -> int:
    return int(hashlib.md5(slug.encode("utf-8")).hexdigest(), 16)

def assign_shard(slug: str, n_shards: int = SHARD_COUNT) -> int:
    return slug_hash(slug) % n_shards

def load_manifest(path_or_url: str) -> List[Dict[str, Any]]:
    if path_or_url.startswith("http"):
        resp = requests.get(path_or_url, timeout=60)
        resp.raise_for_status()
        return resp.json()
    with open(path_or_url, "r", encoding="utf-8") as f:
        return json.load(f)

def cdn_download(url: str, timeout: int = 120) -> bytes:
    resp = requests.get(url, timeout=timeout)
    if resp.status_code == 429:
        wait = RETRY_WAIT
        print(f"Rate limited 429; waiting {wait}s", file=sys.stderr)
        time.sleep(wait)
        resp = requests.get(url, timeout=timeout)
    resp.raise_for_status()
    return resp.content

def extract_pairs_from_parquet(data: bytes, slug: str) -> List[Dict[str, str]]:
    table = pq.read_table(pa.BufferReader(data))
    df = table.to_pandas()
    pairs = []
    for _, row in df.iterrows():
        prompt = row.get("prompt") or row.get("input") or row.get("question") or ""
        response = row.get("response") or row.get("output") or row.get("answer") or row.get("completion") or ""
        if not isinstance(prompt, str):
            prompt = str(prompt)
        if not isinstance(response, str):
            response = str(response)
        prompt = prompt.strip()
        response = response.strip()
        if prompt and response:
            pairs.append({"prompt": prompt, "response": response, "source": slug})
    return pairs

def extract_pairs_from_jsonl(data: bytes, slug: str) -> List[Dict[str, str]]:
    pairs = []
    for line in data.decode("utf-8", errors="replace").splitlines():
        line = line.strip()
        if not line:
            continue
        try:
            obj = json.loads(line)
        except json.JSONDecodeError:
            continue
        prompt = obj.get("prompt") or obj.get("input") or obj.get("question") or ""
        response = obj.get("response") or obj.get("output") or obj.get("answer") or obj.get("completion") or ""
        if not isinstance(prompt, str):
            prompt = str(prompt)
        if not isinstance(response, str):
            response = str(response)
        prompt = prompt.strip()
        response = response.strip()
        if prompt and response:
            pairs.append({"prompt": prompt, "response": response, "source": slug})
    return pairs

def normalize_file_record(rec: Dict[str, Any]) -> Dict[str, Any]:
    # expected manifest record: {"path": "...", "slug": "..."} or {"path": "..."}
    path = rec.get("path") or rec.get("file") or ""
    slug = rec.get("slug") or rec.get("source") or path
    return {"path": path, "slug": slug}
