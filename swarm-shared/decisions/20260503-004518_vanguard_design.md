# vanguard / design

## Final Synthesis (Best Parts + Correctness + Actionability)

**Diagnosis (Consensus)**
- No persisted `(repo, dateFolder)` file manifest → every run re-enumerates via authenticated HF API → quota burn + 429 risk.
- Runtime enumeration + heterogeneous schema → `pyarrow.CastError` and rate-limit amplification.
- Training fetches through HF API auth path instead of raw CDN → misses CDN bypass.
- Lightning Studio lifecycle is recreate-on-each-run → wastes quota, no idempotent reuse or idle-stop resilience.
- Non-deterministic runtime enumeration → non-reproducible, API-dependent runs.

**Proposed Change (Consensus)**
- Add a single CLI command (run on Mac orchestrator) to snapshot repo+dateFolder into `manifests/{repo}__{date}.json` using one `list_repo_tree` call.
- Embed manifest path in training so Lightning performs **CDN-only** fetches (`https://huggingface.co/datasets/.../resolve/main/...`) with zero authenticated API calls during data load.
- Add idempotent Lightning Studio reuse + idle-stop resilience in launcher.

### 1) Manifest generator (new)
`/opt/axentx/vanguard/training/manifest.py`

```python
#!/usr/bin/env python3
"""
Generate deterministic file manifest for repo+dateFolder.
Run on Mac orchestrator after HF API rate-limit window clears.
"""
import json
import sys
from datetime import datetime, timezone
from pathlib import Path

try:
    from huggingface_hub import HfApi
except ImportError:
    print("Install: pip install huggingface_hub")
    sys.exit(1)

API = HfApi()
MANIFEST_DIR = Path(__file__).parent / "manifests"
MANIFEST_DIR.mkdir(exist_ok=True, parents=True)

def snapshot_repo_date(repo_id: str, date_folder: str, out_dir: Path = MANIFEST_DIR) -> Path:
    """
    Single non-recursive tree call for date_folder.
    Returns manifest path.
    """
    tree = API.list_repo_tree(repo_id=repo_id, path=date_folder, recursive=False)
    files = []
    for entry in tree:
        if entry.path.endswith("/"):
            continue
        # entry.path is relative to repo root; date_folder is subpath under dataset/
        filename = entry.path.split("/")[-1]
        rel_path = f"{date_folder}/{filename}"
        files.append(
            {
                "repo_id": repo_id,
                "path": rel_path,
                "url": f"https://huggingface.co/datasets/{repo_id}/resolve/main/{rel_path}",
                "size": getattr(entry, "size", None),
            }
        )

    slug = repo_id.replace("/", "_")
    manifest_path = out_dir / f"{slug}__{date_folder}.json"
    manifest_path.write_text(
        json.dumps(
            {
                "created": datetime.now(timezone.utc).isoformat(),
                "repo_id": repo_id,
                "date_folder": date_folder,
                "files": files,
            },
            indent=2,
        )
    )
    print(f"Manifest written: {manifest_path} ({len(files)} files)")
    return manifest_path

if __name__ == "__main__":
    if len(sys.argv) != 3:
        print("Usage: python manifest.py <repo_id> <date_folder>")
        sys.exit(1)
    snapshot_repo_date(sys.argv[1], sys.argv[2])
```

```python
#!/usr/bin/env python3
"""
Lightning training entrypoint.
Expects manifest JSON via --manifest.
Performs CDN-only fetches (no HF API auth during training).
"""
import argparse
import json
from pathlib import Path
from typing import List, Dict

import torch
from torch.utils.data import Dataset, DataLoader
import lightning as L

class CDNTextDataset(Dataset):
    def __init__(self, manifest_path: Path, max_files: int = -1):
        manifest = json.loads(manifest_path.read_text())
        self.files: List[Dict] = manifest["files"]
        if max_files > 0:
            self.files = self.files[:max_files]

    def __len__(self) -> int:
        return len(self.files)

    def __getitem__(self, idx: int) -> Dict:
        import requests
        entry = self.files[idx]
        # CDN fetch — no Authorization header (bypasses API rate limits)
        resp = requests.get(entry["url"], timeout=30)
        resp.raise_for_status()
        # Placeholder: project to {prompt,response} here based on file format
        text = resp.text
        return {"text": text, "url": entry["url"]}

class SurrogateDataModule(L.LightningDataModule):
    def __init__(self, manifest_path: Path, batch_size: int = 8, max_files: int = -1):
        super().__init__()
        self.manifest_path = manifest_path
        self.batch_size = batch_size
        self.max_files = max_files

    def setup(self, stage=None):
        self.ds = CDNTextDataset(self.manifest_path, max_files=self.max_files)

    def train_dataloader(self):
        return DataLoader(self.ds, batch_size=self.batch_size, shuffle=True, num_workers=4)

class SurrogateModel(L.LightningModule):
    def __init__(self, lr: float = 1e-4):
        super().__init__()
        self.lr = lr
        self.net = torch.nn.Linear(1024, 1024)  # placeholder

    def training_step(self, batch, batch_idx):
        # Replace with real model/tokenizer logic
        loss = torch.tensor(0.0, requires_grad=True)
        self.log("train_loss", loss)
        return loss

    def configure_optimizers(self):
        return torch.optim.AdamW(self.parameters(), lr=self.lr)

def run_training(
    manifest_path: Path,
    studio_name: str = "vanguard-surrogate",
    machine: str = "L40S",
    max_files: int = -1,
    epochs: int = 1,
):
    """
    Idempotent Lightning Studio reuse + training launch.
    """
    try:
        from lightning.pytorch import Trainer
        from lightning.pytorch.loggers import CSVLogger
        from lightning import Teamspace, Studio, Machine
    except ImportError:
        print("Install: pip install lightning")
        return

    # Reuse running studio if exists
    running = None
    for s in Teamspace.studios:
        if s.name == studio_name and s.status == "Running":
            running = s
            print(f"Reusing running studio: {s.name}")
            break

    if running is None:
        print(f"Starting new studio: {studio_name}")
        running = Studio.create(
            name=studio_name,
            machine=Machine(machine),
            create_ok=True,
        )

    # Ensure studio is running before .run()
    if running.status != "Running":
        print(f"Studio stopped; restarting on {machine}")
        running.start(machine=Machine(machine))

    data = SurrogateDataModule(manifest_path=manifest_path, max_files=max_files)
    model = SurrogateModel()
    logger = CSVLogger(save_dir="lightning_logs")
    trainer = Trainer(max_epochs=epochs, logger=logger, devices=1)
    trainer.fit(model, datamodule=data)
    print("Training complete.")

if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--manifest", type=Path, required=True, help="Path to manifest JSON")
    parser.add_argument("--studio", default="vanguard-surrogate")
    parser.add_argument("--machine", default="L40S")
    parser.add_argument("--max-files", type=int, default=-1)
    parser.add_argument("--epochs", type=int, default=1)
    args = parser.parse_args()
    run_training(
        manifest_path=args.manifest,
        studio_name=args.studio,
        machine=args.machine,
        max_files=args.max_files,
        epochs=args.epochs,
    )
```
