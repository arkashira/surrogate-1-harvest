# workio / ops

## 1. Diagnosis
- **No idempotency guard on LINE webhook ingestion** — LINE retries on 5xx/timeouts; duplicate clock-in/out or leave/OT records are created on replay.
- **Missing `X-Line-Signature` verification** — webhook accepts spoofed events; enables attendance fraud and phantom punches.
- **No replay-safe upsert for attendance** — race between duplicate events can create two rows for same employee/day/shift instead of one canonical record.
- **No structured logging/audit trail for webhook events** — hard to debug duplicates, verify signature failures, or reconstruct timeline.
- **No circuit-breaker/back-pressure for LINE API calls** — outbound notification bursts can hit rate limits and block critical messages.

## 2. Proposed change
- **File:** `/opt/axentx/workio/server/src/routes/line-webhook.ts` (or equivalent webhook handler)
- **Scope:** Add signature verification, idempotency key store (Redis or DB), and upsert for attendance events; add structured logging and safe LINE notify wrapper.

// server/src/middleware/line-signature.ts
import crypto from 'crypto';

const LINE_CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';

export function verifyLineSignature(rawBody: string, signature: string): boolean {
  if (!LINE_CHANNEL_SECRET) {
    // In dev, allow bypass with warning (log only)
    console.warn('LINE_CHANNEL_SECRET missing — signature check bypassed');
    return true;
  }
  const expected = crypto
    .createHmac('sha256', LINE_CHANNEL_SECRET)
    .update(rawBody)
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

// server/src/utils/idempotency.ts
import { Pool } from 'pg';

const pool = new Pool({ connectionString: process.env.DATABASE_URL });

export async function ensureIdempotent(key: string, payload: any): Promise<{ ok: boolean; existing?: any }> {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');
    // Use ON CONFLICT to guarantee single insert across workers
    const res = await client.query(
      `INSERT INTO webhook_idempotency (idempotency_key, payload, created_at)
       VALUES ($1, $2, NOW())
       ON CONFLICT (idempotency_key) DO UPDATE
       SET last_seen_at = NOW()
       WHERE webhook_idempotency.idempotency_key = $1
       RETURNING payload`,
      [key, JSON.stringify(payload)]
    );
    await client.query('COMMIT');
    // If row existed before upsert, it's a replay
    const existed = res.rowCount && res.rows[0] && res.rows[0].payload;
    return { ok: true, existing: existed ? JSON.parse(res.rows[0].payload) : undefined };
  } catch (err) {
    await client.query('ROLLBACK');
    // If unique violation still slips through, treat as replay
    if ((err as any)?.code === '23505') return { ok: true, existing: payload };
    throw err;
  } finally {
    client.release();
  }
}

// server/src/routes/line-webhook.ts  (patch)
import { Router } from 'express';
import { verifyLineSignature } from '../middleware/line-signature';
import { ensureIdempotent } from '../utils/idempotency';
import { logWebhookEvent } from '../utils/logger';
import { safeLineNotify } from '../utils/line-notify';

const router = Router();

router.post('/webhook/line', async (req, res) => {
  const rawBody = JSON.stringify(req.body);
  const signature = req.headers['x-line-signature'] as string;

  if (!verifyLineSignature(rawBody, signature)) {
    logWebhookEvent({ type: 'invalid-signature', signature, body: req.body });
    return res.status(401).json({ error: 'Invalid signature' });
  }

  const events = req.body?.events || [];
  if (!Array.isArray(events) || events.length === 0) {
    return res.status(200).json({ ok: true });
  }

  // Idempotency key per batch (or per event if preferred)
  const idempotencyKey = `line-webhook:${req.body.destination || 'unknown'}:${req.body.timestamp || Date.now()}`;
  const { existing } = await ensureIdempotent(idempotencyKey, req.body);
  if (existing) {
    logWebhookEvent({ type: 'replay-detected', idempotencyKey });
    return res.status(200).json({ ok: true }); // LINE expects 2xx
  }

  for (const ev of events) {
    try {
      await handleLineEvent(ev); // your existing handler (clock-in/out/leave/ot)
    } catch (err) {
      logWebhookEvent({ type: 'event-handler-error', event: ev, error: String(err) });
      // Do not return 5xx for single event failure; continue processing others
    }
  }

  // Best-effort LINE notification (non-blocking)
  safeLineNotify('System', 'Webhook processed').catch((e) =>
    logWebhookEvent({ type: 'notify-error', error: String(e) })
  );

  return res.status(200).json({ ok: true });
});

// server/src/utils/line-notify.ts
import axios from 'axios';

const CHANNEL_TOKEN = process.env.LINE_CHANNEL_ACCESS_TOKEN || '';
const MAX_RETRIES = 2;
const RETRY_DELAY = 1000;

export async function safeLineNotify(to: string, text: string) {
  if (!CHANNEL_TOKEN) return;
  for (let i = 0; i <= MAX_RETRIES; i++) {
    try {
      await axios.post(
        'https://api.line.me/v2/bot/message/push',
        { to, messages: [{ type: 'text', text }] },
        { headers: { Authorization: `Bearer ${CHANNEL_TOKEN}` }, timeout: 8000 }
      );
      return;
    } catch (err: any) {
      if (i === MAX_RETRIES) throw err;
      if (err.response?.status === 429) {
        // Respect Retry-After if present
        const wait = Number(err.response.headers['retry-after'] || RETRY_DELAY) * 1000;
        await new Promise((r) => setTimeout(r, wait));
      } else {
        await new Promise((r) => setTimeout(r, RETRY_DELAY));
      }
    }
  }
}
