## 3. Implementation

### 3.1. CDN file list generator (orchestration)
Run on Mac/CI **before** training. One API call; no pagination.

`scripts/update_file_list.py`
```python
import json
import os
from pathlib import Path
from huggingface_hub import HfApi

HF_REPO = "datasets/axentx/surrogate-mirror"
FOLDER = os.getenv("HF_FOLDER", "batches/mirror-merged/20260503")  # override per run
OUT_PATH = Path("surrogate/training/file_list_latest.json")

api = HfApi(token=os.getenv("HF_TOKEN"))
tree = api.list_repo_tree(repo_id=HF_REPO, path=FOLDER, recursive=False)
files = sorted(entry.path for entry in tree if entry.path.endswith(".parquet"))

OUT_PATH.parent.mkdir(parents=True, exist_ok=True)
with OUT_PATH.open("w") as f:
    json.dump(
        {
            "date": FOLDER.rstrip("/").split("/")[-1],
            "files": files,
            "base_url": f"https://huggingface.co/{HF_REPO}/resolve/main",
        },
        f,
        indent=2,
    )
print(f"Saved {len(files)} files -> {OUT_PATH}")
```

### 3.2. Lightning Studio reuse + auto-restart
Resolves quota waste; guards every `.run()` against idle-stop.

`surrogate/launch_studio.py`
```python
import os
import time
from huggingface_hub import HfApi
from lightning_sdk import Client, Teamspace, Machine

LIGHTNING_API_KEY = os.getenv("LIGHTNING_API_KEY")
HF_TOKEN = os.getenv("HF_TOKEN")
LIGHTNING_TEAM = os.getenv("LIGHTNING_TEAM", "default")
STUDIO_NAME = "surrogate-train-l40s"
MACHINE = Machine.L40S  # free-tier fallback; H200 available in lambda-prod if needed

client = Client(api_key=LIGHTNING_API_KEY)
team = Teamspace(client, name=LIGHTNING_TEAM)

def get_or_create_studio() -> Teamspace:
    for s in team.studios:
        if s.name == STUDIO_NAME:
            if s.status == "running":
                print(f"Reusing running studio: {STUDIO_NAME}")
                return s
            print(f"Restarting stopped studio: {STUDIO_NAME}")
            s.start(machine=MACHINE)
            return s
    print(f"Creating new studio: {STUDIO_NAME} on {MACHINE}")
    return team.studios.create(name=STUDIO_NAME, machine=MACHINE)

def run_training(script_path: str = "train.py"):
    studio = get_or_create_studio()
    # Guard against idle-timeout kill
    if studio.status != "running":
        studio.start(machine=MACHINE)
        time.sleep(15)
    run = studio.runs.create(
        script=script_path,
        requirements=["torch", "pyarrow", "requests", "pandas"],
        env={"HF_TOKEN": HF_TOKEN} if HF_TOKEN else None,
    )
    print(f"Started run: {run.id}")
    return run

if __name__ == "__main__":
    run_training()
```

### 3.3. CDN-only training loader
Zero HF API calls during training; minimal dependencies.

`surrogate/train.py`
```python
import json
import os
import io
import requests
import pyarrow.parquet as pq
import pandas as pd
import torch
from torch.utils.data import Dataset, DataLoader
from pathlib import Path
from typing import List

HF_REPO = "datasets/axentx/surrogate-mirror"
FILE_LIST_PATH = Path(__file__).parent / "training" / "file_list_latest.json"
CDN_ROOT = f"https://huggingface.co/{HF_REPO}/resolve/main"

class CDNParquetDataset(Dataset):
    def __init__(self, files: List[str], base_url: str):
        self.files = files
        self.base_url = base_url

    def __len__(self) -> int:
        return len(self.files)

    def __getitem__(self, idx: int):
        rel = self.files[idx]
        url = f"{self.base_url}/{rel}"
        r = requests.get(url, timeout=30)
        r.raise_for_status()
        table = pq.read_table(io.BytesIO(r.content))
        df = table.select(["prompt", "response"]).to_pandas()
        # Return first row for this file; extend as needed
        row = df.iloc[0]
        return {
            "prompt": str(row["prompt"]),
            "response": str(row["response"]),
        }

def build_dataset() -> Dataset:
    if not FILE_LIST_PATH.exists():
        raise FileNotFoundError(f"Missing {FILE_LIST_PATH}. Run update_file_list.py first.")
    with FILE_LIST_PATH.open() as f:
        meta = json.load(f)
    return CDNParquetDataset(files=meta["files"], base_url=meta["base_url"])

def main():
    dataset = build_dataset()
    loader = DataLoader(dataset, batch_size=8, shuffle=True, num_workers=0)
    for batch in loader:
        # Replace with actual surrogate training step
        print({k: v[0][:60] + "..." for k, v in batch.items()})
        break

if __name__ == "__main__":
    main()
```
