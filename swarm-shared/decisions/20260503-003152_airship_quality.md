#!/usr/bin/env python3
"""
CDN-only streaming dataset for surrogate training.
Bypasses HF API rate limits by using public CDN URLs.
"""
import json
import time
import logging
from pathlib import Path
from typing import Optional, Dict, Any, Iterator

import torch
from torch.utils.data import IterableDataset
import requests
from requests.adapters import HTTPAdapter, Retry

logger = logging.getLogger(__name__)

HF_CDN_BASE = "https://huggingface.co/datasets"


class CdnStreamingDataset(IterableDataset):
    """
    Stream dataset files directly from HF CDN without API calls.
    
    Manifest format:
    {
      "repo": "owner/dataset",
      "files": [
        {"path": "train/part-00001.parquet", "size": 12345, "etag": "abc"},
        ...
      ]
    }
    """
    
    def __init__(
        self,
        manifest_path: str,
        repo: Optional[str] = None,
        max_retries: int = 5,
        backoff_factor: float = 1.0,
        projection_fields: tuple = ("prompt", "response"),
    ):
        self.manifest_path = Path(manifest_path)
        self.repo = repo
        self.projection_fields = projection_fields
        
        with open(self.manifest_path) as f:
            self.manifest = json.load(f)
        
        if not self.repo:
            self.repo = self.manifest.get("repo")
        assert self.repo, "repo must be provided or in manifest"
        
        self.files = self.manifest.get("files", [])
        logger.info(f"Loaded {len(self.files)} files from manifest")
        
        # Session with retry for CDN flakiness
        self.session = requests.Session()
        retries = Retry(
            total=max_retries,
            backoff_factor=backoff_factor,
            status_forcelist=[429, 500, 502, 503, 504],
        )
        self.session.mount("https://", HTTPAdapter(max_retries=retries))
    
    def _download_file(self, file_path: str) -> bytes:
        """Download single file from CDN (no auth header)."""
        url = f"{HF_CDN_BASE}/{self.repo}/resolve/main/{file_path}"
        resp = self.session.get(url, timeout=30)
        resp.raise_for_status()
        return resp.content
    
    def _parse_projection(self, raw_bytes: bytes) -> Iterator[Dict[str, Any]]:
        """
        Parse file bytes and project only required fields.
        Supports: .jsonl, .parquet (via pyarrow if available)
        """
        # Try parquet first (common for HF datasets)
        try:
            import pyarrow.parquet as pq
            import pyarrow as pa
            import io
            
            table = pq.read_table(io.BytesIO(raw_bytes))
            # Project only needed columns to avoid mixed-schema issues
            available = set(table.column_names)
            to_project = [f for f in self.projection_fields if f in available]
            
            if not to_project:
                logger.warning(f"No projection fields {self.projection_fields} in {table.column_names}")
                return
            
            projected = table.select(to_project)
            for i in range(projected.num_rows):
                row = {col: projected[col][i].as_py() for col in to_project}
                # Skip rows with None/empty prompts
                if row.get("prompt"):
                    yield row
            return
        except ImportError:
            pass
        except Exception as e:
            logger.debug(f"Parquet parse failed, trying jsonl: {e}")
        
        # Fallback: JSONL lines
        try:
            for line in raw_bytes.decode("utf-8").strip().split("\n"):
                if not line.strip():
                    continue
                data = json.loads(line)
                row = {k: data.get(k) for k in self.projection_fields if k in data}
                if row.get("prompt"):
                    yield row
        except Exception as e:
            logger.warning(f"Failed to parse {file_path}: {e}")
    
    def __iter__(self) -> Iterator[Dict[str, Any]]:
        worker_info = torch.utils.data.get_worker_info()
        
        if worker_info is None:
            # Single-process: iterate all files
            file_iter = enumerate(self.files)
        else:
            # Multi-worker: shard files across workers
            per_worker = len(self.files) // worker_info.num_workers
            start = worker_info.id * per_worker
            end = start + per_worker if worker_info.id < worker_info.num_workers - 1 else len(self.files)
            file_iter = enumerate(self.files[start:end], start=start)
        
        for idx, file_info in file_iter:
            file_path = file_info["path"] if isinstance(file_info, dict) else file_info
            try:
                raw = self._download_file(file_path)
                for item in self._parse_projection(raw):
                    yield item
            except Exception as e:
                logger.error(f"Failed to process {file_path}: {e}")
                continue
