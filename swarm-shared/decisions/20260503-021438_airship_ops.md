### 1. Deterministic file lister (`surrogate/ops/list_hf_files.py`)

```python
#!/usr/bin/env python3
"""
List HF dataset files for a given date folder (non-recursive per subfolder)
and emit a JSON file for Lightning training (CDN-only paths).
Run from Mac after rate-limit window clears.
"""
import json
import os
import sys
from datetime import datetime

from huggingface_hub import HfApi

REPO = "axentx/surrogate-mirror"  # adjust
DATE_FOLDER = os.getenv("DATE_FOLDER", datetime.utcnow().strftime("%Y-%m-%d"))
OUT_PATH = os.getenv("OUT_PATH", f"file_lists/{DATE_FOLDER}.json")

def main() -> None:
    api = HfApi()
    # Top-level non-recursive to avoid pagination explosion
    entries = api.list_repo_tree(repo_id=REPO, path=DATE_FOLDER, recursive=False)
    files = []
    for e in entries:
        if e.type == "file":
            files.append(f"{DATE_FOLDER}/{e.path}")
        elif e.type == "folder":
            # one level deeper, still non-recursive
            sub = api.list_repo_tree(repo_id=REPO, path=f"{DATE_FOLDER}/{e.path}", recursive=False)
            for se in sub:
                if se.type == "file":
                    files.append(f"{DATE_FOLDER}/{e.path}/{se.path}")

    os.makedirs(os.path.dirname(OUT_PATH), exist_ok=True)
    with open(OUT_PATH, "w") as f:
        json.dumps({"repo": REPO, "date": DATE_FOLDER, "files": files}, f, indent=2)
    print(f"Wrote {len(files)} files to {OUT_PATH}")

if __name__ == "__main__":
    main()
```

### 2. CDN-only DataLoader (`surrogate/training/cdn_dataset.py`)

```python
import json
import os
from typing import Dict, List

import pyarrow as pa
import pyarrow.parquet as pq
import requests
from torch.utils.data import IterableDataset

HF_DATASETS_BASE = "https://huggingface.co/datasets"

class CDNParquetDataset(IterableDataset):
    """
    Zero HF API calls during training.
    file_list_json must contain: {"repo": "...", "files": ["date/file.parquet", ...]}
    Projects each file to {prompt, response} on the fly.
    """
    def __init__(self, file_list_json: str, repo: str | None = None):
        with open(file_list_json) as f:
            meta = json.load(f)
        self.repo = repo or meta["repo"]
        self.files: List[str] = meta["files"]
        self._ix = 0

    def _stream_file(self, path: str):
        url = f"{HF_DATASETS_BASE}/{self.repo}/resolve/main/{path}"
        # CDN download; no Authorization header -> bypasses /api/ rate limits
        with requests.get(url, timeout=30, stream=True) as r:
            r.raise_for_status()
            with open("/tmp/tmp.parquet", "wb") as f:
                for chunk in r.iter_content(chunk_size=8192):
                    f.write(chunk)
            table = pq.read_table("/tmp/tmp.parquet")
            # Project only {prompt, response}; attribution via filename pattern
            for batch in table.to_batches():
                df = batch.to_pandas()
                for _, row in df.iterrows():
                    yield {"prompt": row.get("prompt", ""), "response": row.get("response", "")}

    def __iter__(self):
        for path in self.files:
            yield from self._stream_file(path)
```

### 3. Lightning Studio reuse + idle guard (`surrogate/training/lightning_launcher.py`)

```python
#!/usr/bin/env python3
import os
import time
from lightning_sdk import Studio, Teamspace, Machine

TEAMSPACE = os.getenv("LIGHTNING_TEAMSPACE", "default")
STUDIO_NAME = os.getenv("LIGHTNING_STUDIO", "surrogate-train")
MACHINE = Machine.L40S  # fallback; H200 requires lightning-lambda-prod
MAX_RETRIES = 3

def get_or_create_studio() -> Studio:
    teamspace = Teamspace(TEAMSPACE)
    # Reuse running studio to save quota
    for s in teamspace.studios:
        if s.name == STUDIO_NAME and s.status == "running":
            print(f"Reusing running studio: {s.name}")
            return s
    # Create if not running
    print(f"Creating studio: {STUDIO_NAME}")
    return Studio(
        name=STUDIO_NAME,
        teamspace=TEAMSPACE,
        machine=MACHINE,
        create_ok=True,
    )

def run_training(script: str, args: list[str] | None = None) -> None:
    studio = get_or_create_studio()
    for attempt in range(1, MAX_RETRIES + 1):
        if studio.status != "running":
            print(f"Studio stopped (idle-timeout?). Restarting... (attempt {attempt})")
            studio.start(machine=MACHINE)
            time.sleep(30)
        try:
            run = studio.run(script, arguments=args or [], wait=False)
            print(f"Started run {run.id}")
            run.wait_until_finished()
            if run.status == "failed":
                raise RuntimeError(f"Run failed: {run.id}")
            break
        except Exception as e:
            print(f"Run failed: {e}")
            if attempt == MAX_RETRIES:
                raise
            time.sleep(60 * attempt)

if __name__ == "__main__":
    run_training("train.py", ["--data", "file_lists/latest.json"])
```

### 4. Arkship idle-stop resilience hook (`arkship/ops/surrogate_guard.py`)

```python
#!/usr/bin/env python3
"""
Guard for Arkship-triggered Surrogate training runs.
Checks studio status and restarts on idle-stop before each .run().
"""
import subprocess
import sys

def guard_and_run(script: str, *args) -> None:
    cmd = [sys.executable, "surrogate/training/lightning_launcher.py", script, *args]
    subprocess.run(cmd, check=True)

if __name__ == "__main__":
    guard_and_run(*sys.argv[1:])
```
