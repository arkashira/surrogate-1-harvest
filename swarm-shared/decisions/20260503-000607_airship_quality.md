## Highest-Value Incremental Improvement (<2h)

**Goal**: Eliminate HF API 429s during Surrogate-1 training by replacing `load_dataset(streaming=True)` with a CDN-only `IterableDataset` that reads from a pre-generated `file_list.json`.

**Why this ships fast**:
- Single-file change (`surrogate/data/cdn_dataset.py`) + 10-line script to generate `file_list.json`.
- No infra changes, no retraining code changes (drop-in replacement for current dataset loader).
- Removes recursive `list_repo_files` and streaming dataset calls that trigger 429s.
- Uses HuggingFace CDN (`/resolve/main/`) which has separate (much higher) rate limits and requires no auth.

## Implementation Plan

### 1. Add CDN-only IterableDataset
Create `surrogate/data/cdn_dataset.py`:

```python
# surrogate/data/cdn_dataset.py
import json
import io
import pyarrow as pa
import pyarrow.parquet as pq
from typing import Dict, Optional, Iterator
from torch.utils.data import IterableDataset
import requests
from tqdm import tqdm

HF_DATASETS_BASE = "https://huggingface.co/datasets"

class CDNParquetIterableDataset(IterableDataset):
    """
    CDN-only dataset loader for HuggingFace repos.
    Reads file list from file_list.json and streams Parquet files
    directly from HF CDN (no API calls, no auth, no 429s).

    file_list.json format:
    [
      {"repo": "owner/repo", "path": "batches/mirror-merged/2026-04-29/s1.parquet"},
      ...
    ]
    """

    def __init__(
        self,
        file_list_path: str,
        repo_column: str = "repo",
        path_column: str = "path",
        columns: Optional[list] = None,
        buffer_size: int = 8 * 1024 * 1024,
    ):
        with open(file_list_path) as f:
            self.files = json.load(f)
        self.repo_column = repo_column
        self.path_column = path_column
        self.columns = columns
        self.buffer_size = buffer_size

    def _stream_parquet(self, repo: str, path: str) -> Iterator[Dict]:
        url = f"{HF_DATASETS_BASE}/{repo}/resolve/main/{path}"
        resp = requests.get(url, stream=True, timeout=30)
        resp.raise_for_status()

        # Stream into memory buffer and read with pyarrow
        buf = io.BytesIO()
        for chunk in resp.iter_content(chunk_size=self.buffer_size):
            buf.write(chunk)
        buf.seek(0)

        table = pq.read_table(buf, columns=self.columns)
        for batch in table.to_batches(max_chunksize=1024):
            for row in batch.to_pylist():
                yield row

    def __iter__(self) -> Iterator[Dict]:
        for item in self.files:
            repo = item[self.repo_column]
            path = item[self.path_column]
            try:
                yield from self._stream_parquet(repo, path)
            except Exception as e:
                # Log and skip corrupted/unavailable files
                print(f"Skipping {repo}/{path}: {e}")
                continue


class Surrogate1Dataset(IterableDataset):
    """
    Surrogate-1 specific dataset: projects raw repo files to {prompt, response}.
    Expects file_list.json with repo/path entries pointing to parquet files
    that contain at least 'prompt' and 'response' columns.
    """

    def __init__(self, file_list_path: str):
        self.base = CDNParquetIterableDataset(
            file_list_path=file_list_path,
            columns=["prompt", "response"],
        )

    def __iter__(self):
        for row in self.base:
            # Ensure minimal contract for training
            if row.get("prompt") and row.get("response"):
                yield {
                    "prompt": row["prompt"],
                    "response": row["response"],
                }
```

### 2. Generate `file_list.json` (one-time Mac/Launcher script)

Create `surrogate/scripts/generate_file_list.py`:

```python
#!/usr/bin/env python3
"""
Generate file_list.json for CDN-only loading.
Run from Mac (or any machine with HF token) once per date folder.
Respects HF API limits: uses list_repo_tree per folder, not recursive repo scan.
"""

import os
import json
import argparse
from huggingface_hub import HfApi

HF_DATASET_REPO = os.getenv("HF_DATASET_REPO", "axentx/surrogate-mirror")
DATE_FOLDER = os.getenv("DATE_FOLDER", "2026-04-29")  # e.g. batches/mirror-merged/2026-04-29

def main(output_path: str):
    api = HfApi()
    # Single API call per folder (non-recursive)
    entries = api.list_repo_tree(
        repo_id=HF_DATASET_REPO,
        path=DATE_FOLDER,
        repo_type="dataset",
        recursive=False,
    )

    files = []
    for e in entries:
        if e.path.endswith(".parquet"):
            files.append({
                "repo": HF_DATASET_REPO,
                "path": e.path,
            })

    out = {
        "generated_from": HF_DATASET_REPO,
        "date_folder": DATE_FOLDER,
        "files": files,
    }

    os.makedirs(os.path.dirname(output_path), exist_ok=True)
    with open(output_path, "w") as f:
        json.dump(out, f, indent=2)

    print(f"Wrote {len(files)} files to {output_path}")

if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--output", default="surrogate/data/file_list.json")
    args = parser.parse_args()
    main(args.output)
```

Make executable:

```bash
chmod +x surrogate/scripts/generate_file_list.py
```

### 3. Update training launcher to use CDN dataset

In your Lightning training script (or wherever dataset is instantiated), replace:

```python
# OLD (triggers HF API + 429s)
from datasets import load_dataset
ds = load_dataset("axentx/surrogate-mirror", streaming=True)
```

with:

```python
# NEW (CDN-only, zero API calls during training)
from surrogate.data.cdn_dataset import Surrogate1Dataset
ds = Surrogate1Dataset("surrogate/data/file_list.json")
```

If using PyTorch DataLoader:

```python
from torch.utils.data import DataLoader

loader = DataLoader(
    ds,
    batch_size=8,
    num_workers=2,
    pin_memory=True,
)
```

### 4. Workflow for daily runs (fits existing patterns)

1. **Mac (or launcher)**: After `dataset-mirror` writes new date folder, run once:
   ```bash
   python surrogate/scripts/generate_file_list.py --output surrogate/data/file_list.json
   ```
   (This is the only HF API interaction; runs after rate-limit window clears if needed.)

2. **Lightning Studio**: Reuse running studio or restart if idle-killed:
   ```python
   from lightning import Studio, Teamspace, Machine

   # Reuse existing studio (saves quota)
   studio = None
   for s in Teamspace.studios:
       if s.name == "surrogate-train-l40s" and s.status == "running":
           studio = s
           break

   if studio is None or studio.status != "running":
       studio = Studio(
           name="surrogate-train-l40s",
           machine=Machine.L40S,  # or fallback to free-tier L40S
           cloud="lightning-public-prod",
       )

   studio.run(
       "train.py",  # uses CDN dataset
       requirements=["torch", "pyarrow", "requests", "tqdm"],
   )
   ```
