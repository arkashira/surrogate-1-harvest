# Costinel / frontend

## Final Implementation Plan — Top Hub Signal Panel (CDN-first, resilient, frontend-only)

**Scope**: Add a lightweight, resilient Top Hub signal card to the Costinel dashboard that surfaces the most-connected hub (e.g., "MOC") with contextual insights and actionable links.  
**Effort**: ~60–90 minutes (frontend only).  
**Mechanism**:
- Mac orchestration writes `hubFileList.json` and a fallback snapshot into `/public/data/` once per day (no runtime HF API calls).
- Frontend loads the file list from `/public/data/`, picks the latest hub file, and fetches its content via CDN (`/resolve/main/...`) with no Authorization header.
- Graceful fallback to `public/data/hubs/top-hub-latest.json` if CDN fails.
- Vue component with refresh, last-updated timestamp, copy-to-clipboard, and local caching for resilience.

#### `src/utils/cdnFetch.js`

```js
// Lightweight CDN fetcher with exponential backoff and jitter.
// Uses no Authorization header to bypass HF API rate limits.
const DEFAULT_RETRIES = 3;
const BASE_DELAY = 400; // ms

function jitter() {
  return Math.random() * 200 + 50;
}

export async function cdnFetch(path, retries = DEFAULT_RETRIES) {
  const url = `https://huggingface.co/datasets/AXENTX/Costinel/resolve/main/${encodeURIComponent(path)}`;
  for (let attempt = 0; attempt <= retries; attempt++) {
    try {
      const res = await fetch(url, { cache: 'no-store' });
      if (!res.ok) throw new Error(`CDN ${res.status}`);
      return await res.text();
    } catch (err) {
      if (attempt === retries) throw err;
      const delay = (BASE_DELAY * 2 ** attempt) + jitter();
      await new Promise((r) => setTimeout(r, delay));
    }
  }
}
```

#### `src/composables/useHubInsights.js`

```js
import { ref } from 'vue';
import { cdnFetch } from '@/utils/cdnFetch';

export function useHubInsights() {
  const topHub = ref(null);
  const relatedDocs = ref([]);
  const loading = ref(false);
  const error = ref(null);
  const lastUpdated = ref(null);

  async function load(useFallbackOnFailure = true) {
    loading.value = true;
    error.value = null;
    try {
      // 1) Load file list (embedded in /public/data)
      const listRes = await fetch('/data/hubFileList.json', { cache: 'no-store' });
      if (!listRes.ok) throw new Error('Failed to load hub file list');
      const fileList = await listRes.json(); // { files: [{ path, hub, date, slug }], dateFolder }

      if (!fileList.files?.length) throw new Error('No hub files listed');

      // 2) Pick most recent entry
      const latest = fileList.files.sort((a, b) => b.date.localeCompare(a.date))[0];

      // 3) Try CDN fetch for latest hub content
      let content;
      try {
        content = await cdnFetch(latest.path);
      } catch (cdnErr) {
        if (!useFallbackOnFailure) throw cdnErr;
        // Fallback to local snapshot
        const fb = await fetch('/data/hubs/top-hub-latest.json', { cache: 'no-store' });
        if (!fb.ok) throw new Error('CDN failed and fallback unavailable');
        content = await fb.text();
      }

      // Expect NDJSON lines: { hub, degree, summary, related: [...] }
      const lines = content.trim().split('\n').map((l) => JSON.parse(l));
      const best = lines.sort((a, b) => b.degree - a.degree)[0];

      topHub.value = {
        name: best.hub,
        degree: best.degree,
        summary: best.summary,
        slug: latest.slug,
        date: latest.date,
      };
      relatedDocs.value = best.related || [];
      lastUpdated.value = fileList.generatedAt || new Date().toISOString();
    } catch (e) {
      error.value = e.message || 'Failed to load hub insights';
    } finally {
      loading.value = false;
    }
  }

  return { topHub, relatedDocs, loading, error, lastUpdated, load };
}
```
