## Implementation Plan

### 1. CDN-first deterministic file list (Mac orchestration)
- Single `list_repo_tree` call per date folder → save `file_list.json`
- Embed `file_list.json` in Lightning training script
- Training uses only CDN URLs (`resolve/main/...`) — zero API calls during data load

### 2. Lightning Studio reuse + idle-stop guard
- Before `Studio(create_ok=True)`, list `Teamspace.studios` and reuse running ones
- Wrap `.run()` with status check; if stopped, restart with `target.start(machine=Machine.L40S)`
- Prevents 80hr/mo quota waste from recreation + idle-timeout deaths

### 3. Arkship UI idle-stop resilience
- Add polling heartbeat to Arkship frontend (or lightweight backend watchdog)
- If Surrogate training idle-stops, auto-restart via Lightning SDK (same guard as #2)
- Keeps end-to-end pipeline alive without manual intervention

### `scripts/ingest_cdn_filelist.py` (Mac orchestration)
```python
#!/usr/bin/env python3
"""
Generate deterministic CDN file list for Surrogate training.
Run from Mac after HF rate-limit window clears.
"""
import json
import os
from huggingface_hub import HfApi

API = HfApi()
REPO = "axentx/surrogate-dataset"
DATE_FOLDER = "2026-05-03"  # parameterized in prod
OUT_FILE = "file_list.json"

def build_cdn_file_list():
    # Single API call — recursive=False to avoid pagination explosion
    tree = API.list_repo_tree(REPO, path=DATE_FOLDER, recursive=False)
    files = []
    for item in tree:
        if item.path.endswith((".parquet", ".jsonl")):
            cdn_url = f"https://huggingface.co/datasets/{REPO}/resolve/main/{item.path}"
            files.append({
                "path": item.path,
                "cdn_url": cdn_url,
                "size": getattr(item, "size", None)
            })
    with open(OUT_FILE, "w") as f:
        json.dump(files, f, indent=2)
    print(f"Wrote {len(files)} files to {OUT_FILE}")

if __name__ == "__main__":
    build_cdn_filelist()
```

### `surrogate/train.py` (Lightning Studio — CDN-only loader)
```python
import json
import pyarrow.parquet as pq
import requests
from pathlib import Path
from torch.utils.data import IterableDataset

class CDNParquetDataset(IterableDataset):
    def __init__(self, file_list_path="file_list.json"):
        with open(file_list_path) as f:
            self.files = json.load(f)
        self.idx = 0

    def __iter__(self):
        for entry in self.files:
            # CDN fetch — no HF API, no auth, no 429
            resp = requests.get(entry["cdn_url"], stream=True)
            resp.raise_for_status()
            with open("/tmp/current.parquet", "wb") as tmp:
                for chunk in resp.iter_content(chunk_size=8192):
                    tmp.write(chunk)
            table = pq.read_table("/tmp/current.parquet")
            # Project to {prompt, response} only
            for batch in table.to_batches():
                for row in zip(batch["prompt"].to_pylist(), batch["response"].to_pylist()):
                    yield {"prompt": row[0], "response": row[1]}
```

### `scripts/lightning_studio_guard.py` (Reuse + idle-stop resilience)
```python
#!/usr/bin/env python3
import time
from lightning_sdk import Studio, Teamspace, Machine

TEAMSPACE = "axentx"
STUDIO_NAME = "surrogate-train-l40s"
MACHINE = Machine.L40S

def get_or_create_studio():
    team = Teamspace(TEAMSPACE)
    running = [s for s in team.studios if s.name == STUDIO_NAME and s.status == "running"]
    if running:
        print(f"Reusing running studio: {STUDIO_NAME}")
        return running[0]
    existing = [s for s in team.studios if s.name == STUDIO_NAME]
    if existing:
        studio = existing[0]
        if studio.status == "stopped":
            print(f"Restarting stopped studio: {STUDIO_NAME}")
            studio.start(machine=MACHINE)
            return studio
    print(f"Creating new studio: {STUDIO_NAME}")
    return Studio.create(
        teamspace=TEAMSPACE,
        name=STUDIO_NAME,
        machine=MACHINE,
        create_ok=True
    )

def run_with_guard(train_script, args):
    studio = get_or_create_studio()
    max_retries = 3
    for attempt in range(max_retries):
        if studio.status != "running":
            print(f"Studio not running (status={studio.status}), restarting...")
            studio.start(machine=MACHINE)
            time.sleep(30)
        try:
            run = studio.run(
                command=f"python {train_script} {' '.join(args)}",
                sync=True  # or False + poll for long runs
            )
            if run.status == "succeeded":
                print("Training succeeded")
                return run
            else:
                print(f"Run failed: {run.status}, logs: {run.logs_url}")
        except Exception as e:
            print(f"Attempt {attempt+1} failed: {e}")
            time.sleep(60)
    raise RuntimeError("Training failed after retries")

if __name__ == "__main__":
    import sys
    run_with_guard("train.py", sys.argv[1:])
```

### Arkship UI resilience (lightweight watchdog)
Add to Arkship backend (`arkship/services/surrogate_watchdog.py`):
```python
import threading
import time
from scripts.lightning_studio_guard import get_or_create_studio, run_with_guard

def surrogate_idle_watchdog(interval=300):
    """Ensure Surrogate training studio is alive; restart if stopped."""
    while True:
        try:
            studio = get_or_create_studio()
            if studio.status != "running":
                print(f"Watchdog: studio stopped ({studio.status}), restarting...")
                studio.start(machine=Machine.L40S)
        except Exception as e:
            print(f"Watchdog error: {e}")
        time.sleep(interval)

def start_watchdog():
    t = threading.Thread(target=surrogate_idle_watchdog, daemon=True)
    t.start()
```

Wire into Arkship app startup (e.g., `arkship/__init__.py` or main.py):
```python
# If Surrogate AI is configured, start watchdog
if config.SURROGATE_ENABLED:
    from arkship.services.surrogate_watchdog import start_watchdog
    start_watchdog()
```
