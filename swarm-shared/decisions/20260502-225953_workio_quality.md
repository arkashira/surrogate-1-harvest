# workio / quality

## 1) Diagnosis

- **No idempotency key enforcement at DB layer**: clock-in/out endpoints accept duplicate requests for the same `(employee_id, punch_type, date)` when clients retry on flaky mobile networks → double punches and payroll drift.
- **Race condition on concurrent punches**: two near-simultaneous requests can both pass “last punch” checks and insert duplicate/out-of-order punches.
- **Missing immutable audit trail**: updates/deletes possible on punch records; compliance requires append-only, tamper-evident history.
- **Webhook replay vulnerability**: LINE may redeliver webhook events; without idempotency keys these create duplicate punches.
- **Soft-delete / update surface**: current schema allows `UPDATE`/`DELETE` on punches; increases risk of data drift and complicates audit.

## 2) Proposed change

- **File scope**: `workio/server/src/db/schema.sql` (add constraints + audit table), `workio/server/src/routes/punch.ts` (add idempotency + transaction), `workio/server/src/middleware/idempotency.ts` (new).
- **Change type**: enforce exactly-once punches at DB layer with unique constraint + idempotency key table, add immutable punch_audit table, wrap punch creation in transaction with `ON CONFLICT DO NOTHING`.

## 3) Implementation

### 3.1 DB schema changes (`workio/server/src/db/schema.sql`)

```sql
-- Idempotency keys for punch/webhook requests
CREATE TABLE IF NOT EXISTS punch_idempotency (
  idempotency_key TEXT PRIMARY KEY,
  employee_id     INTEGER NOT NULL,
  punch_type      TEXT NOT NULL CHECK (punch_type IN ('clock_in', 'clock_out')),
  punch_date      DATE NOT NULL,
  punch_record_id INTEGER NOT NULL, -- references punches(id)
  created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Immutable audit trail for punches (append-only)
CREATE TABLE IF NOT EXISTS punch_audit (
  audit_id      BIGSERIAL PRIMARY KEY,
  punch_id      INTEGER NOT NULL,
  employee_id   INTEGER NOT NULL,
  punch_type    TEXT NOT NULL,
  punched_at    TIMESTAMPTZ NOT NULL,
  latitude      NUMERIC,
  longitude     NUMERIC,
  image_url     TEXT,
  metadata      JSONB,
  operation     TEXT NOT NULL, -- 'INSERT','UPDATE','DELETE'
  changed_by    TEXT,
  changed_at    TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  tx_id         TEXT -- optional: LINE webhook/delivery id for replay tracking
);

-- Prevent duplicate punches per employee per punch_type per calendar date
-- (one clock_in and one clock_out per day)
CREATE UNIQUE INDEX IF NOT EXISTS uq_employee_punch_type_date
  ON punches (employee_id, punch_type, DATE(punched_at));

-- Audit trigger (append-only)
CREATE OR REPLACE FUNCTION fn_punch_audit()
RETURNS TRIGGER AS $$
BEGIN
  IF (TG_OP = 'INSERT') THEN
    INSERT INTO punch_audit(punch_id, employee_id, punch_type, punched_at, latitude, longitude, image_url, metadata, operation, tx_id)
    VALUES (NEW.id, NEW.employee_id, NEW.punch_type, NEW.punched_at, NEW.latitude, NEW.longitude, NEW.image_url, NEW.metadata, 'INSERT', NEW.tx_id);
    RETURN NEW;
  ELSIF (TG_OP = 'UPDATE') THEN
    INSERT INTO punch_audit(punch_id, employee_id, punch_type, punched_at, latitude, longitude, image_url, metadata, operation, changed_by, tx_id)
    VALUES (NEW.id, NEW.employee_id, NEW.punch_type, NEW.punched_at, NEW.latitude, NEW.longitude, NEW.image_url, NEW.metadata, 'UPDATE', NEW.updated_by, NEW.tx_id);
    RETURN NEW;
  ELSIF (TG_OP = 'DELETE') THEN
    INSERT INTO punch_audit(punch_id, employee_id, punch_type, punched_at, latitude, longitude, image_url, metadata, operation, changed_by, tx_id)
    VALUES (OLD.id, OLD.employee_id, OLD.punch_type, OLD.punched_at, OLD.latitude, OLD.longitude, OLD.image_url, OLD.metadata, 'DELETE', OLD.updated_by, OLD.tx_id);
    RETURN OLD;
  END IF;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_punch_audit
AFTER INSERT OR UPDATE OR DELETE ON punches
FOR EACH ROW EXECUTE FUNCTION fn_punch_audit();
```

### 3.2 Idempotency middleware (`workio/server/src/middleware/idempotency.ts`)

```ts
import { Request, Response, NextFunction } from 'express';
import { pool } from '../db';

export async function withIdempotency(req: Request, res: Response, next: NextFunction) {
  const key = req.headers['idempotency-key'] as string;
  if (!key) {
    return res.status(400).json({ error: 'Missing idempotency-key header' });
  }

  // Try to claim idempotency key early (short TTL app-side; DB enforces uniqueness)
  const client = await pool.connect();
  try {
    await client.query('BEGIN');
    // If key exists, fetch previous response metadata and replay
    const exists = await client.query(
      'SELECT punch_record_id FROM punch_idempotency WHERE idempotency_key = $1',
      [key]
    );
    if (exists.rows.length > 0) {
      // Re-send the created punch (caller should make GET or we store minimal response)
      const punchId = exists.rows[0].punch_record_id;
      const punch = await client.query('SELECT * FROM punches WHERE id = $1', [punchId]);
      await client.query('ROLLBACK');
      client.release();
      return res.status(200).json(punch.rows[0]);
    }

    // Attach connection and key to request for later commit
    (req as any).idempotencyKey = key;
    (req as any).idempotencyClient = client;
    next();
  } catch (err) {
    await client.query('ROLLBACK');
    client.release();
    next(err);
  }
}
```
