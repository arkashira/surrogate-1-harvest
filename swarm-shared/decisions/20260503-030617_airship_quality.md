### 1) Pre-cache file list once (10–15 min)  
- Run from Mac (or non-GPU orchestrator) **after** HF rate-limit window is clear.  
- Target one date folder:  
  `datasets/my-org/surrogate-mirror/main/batches/mirror-merged/2026-05-03/`  
- Save `training/file_list.json` (or repo-root) so training uses only CDN fetches.

`scripts/prefetch_file_list.py`
```python
#!/usr/bin/env python3
"""
Pre-cache parquet file list for a single date folder to avoid HF API calls during training.
Usage:
  HF_REPO=datasets/my-org/surrogate-mirror \
  DATE_FOLDER=batches/mirror-merged/2026-05-03 \
  python scripts/prefetch_file_list.py
"""
import os
import json
from huggingface_hub import list_repo_tree

HF_REPO = os.getenv("HF_REPO")
DATE_FOLDER = os.getenv("DATE_FOLDER")
OUT_PATH = os.getenv("OUT_PATH", "training/file_list.json")

if not HF_REPO or not DATE_FOLDER:
    raise SystemExit("HF_REPO and DATE_FOLDER required")

# recursive=False keeps pagination small and avoids 429 on large repos
tree = list_repo_tree(repo_id=HF_REPO, path=DATE_FOLDER, recursive=False)

# Keep only parquet files (or whatever you train on)
files = sorted(
    f.rfilename
    for f in tree
    if f.rfilename.endswith(".parquet")
)

os.makedirs(os.path.dirname(OUT_PATH), exist_ok=True)
with open(OUT_PATH, "w") as f:
    json.dump({"date_folder": DATE_FOLDER, "files": files}, f, indent=2)

print(f"Wrote {len(files)} files to {OUT_PATH}")
```

Run once and commit `training/file_list.json`:
```bash
cd /opt/axentx/airship

DATE_FOLDER=batches/mirror-merged/2026-05-03 \
python scripts/prefetch_file_list.py
```

- Replace `load_dataset(streaming=True)` with direct CDN downloads.  
- Project to `{prompt, response}` only at parse time.  
- Use file list from step 1; **no `list_repo_*` calls during training**.  
- Add robust retry/backoff and timeouts for CDN downloads.

`surrogate/data/train_loader.py`

```python
import json
import time
import requests
import pyarrow.parquet as pq
from pathlib import Path
from huggingface_hub import hf_hub_download

HF_REPO = "datasets/my-org/surrogate-mirror"
CDN_ROOT = f"https://huggingface.co/datasets/{HF_REPO}/resolve/main"

_MAX_RETRIES = 5
_BACKOFF_FACTOR = 1.5
_TIMEOUT = 30

def load_file_list(path="training/file_list.json"):
    with open(path) as f:
        return json.load(f)["files"]

def _cdn_get_with_retry(url, dest_path, max_retries=_MAX_RETRIES):
    for attempt in range(1, max_retries + 1):
        try:
            resp = requests.get(url, timeout=_TIMEOUT, stream=True)
            resp.raise_for_status()
            with open(dest_path, "wb") as f:
                for chunk in resp.iter_content(chunk_size=8192):
                    f.write(chunk)
            return dest_path
        except Exception as exc:
            if attempt == max_retries:
                raise
            sleep_sec = _BACKOFF_FACTOR ** attempt
            time.sleep(sleep_sec)

def cdn_parquet_loader(file_list, cache_dir=".cache_hf"):
    """
    Yield {"prompt": ..., "response": ...} from parquet files via CDN.
    Uses hf_hub_download (CDN-backed) with retries/timeouts.
    """
    Path(cache_dir).mkdir(parents=True, exist_ok=True)

    for rel_path in file_list:
        local_path = hf_hub_download(
            repo_id=HF_REPO,
            filename=rel_path,
            cache_dir=cache_dir,
            force_download=False,
            # hf_hub_download already uses CDN; add timeout via env or wrapper if needed
        )
        # If you prefer zero huggingface_hub at train time, replace above with:
        # url = f"{CDN_ROOT}/{rel_path}"
        # local_path = Path(cache_dir) / rel_path.replace("/", "_")
        # _cdn_get_with_retry(url, local_path)

        table = pq.read_table(local_path, columns=["prompt", "response"])
        for batch in table.to_batches(max_chunksize=1024):
            chunk = batch.slice(0, batch.num_rows).to_pydict()
            for prompt, response in zip(chunk["prompt"], chunk["response"]):
                yield {"prompt": prompt, "response": response}
```

### 3) Lightning Studio reuse + idle-stop resilience (30–45 min)  
- Reuse a running studio instead of creating new ones.  
- Before each run, check status; if stopped, restart on `L40S` (or available free-tier GPU).  
- Ensure training entrypoint uses the CDN-only loader from step 2.

`scripts/run_training.py`
```python
#!/usr/bin/env python3
"""
Run Surrogate training with studio reuse + idle-stop resilience.
"""
import time
import lightning as L
from surrogate.train import train  # your training entrypoint

STUDIO_NAME = "surrogate-train-studio"
MACHINE = L.cloud.Machine.L40S  # free tier falls back to L40S; H200 requires paid account

def get_or_create_studio():
    teamspace = L.Teamspace()
    for s in teamspace.studios:
        if s.name == STUDIO_NAME:
            return s
    return L.Studio(
        name=STUDIO_NAME,
        machine=MACHINE,
        create_ok=True,
    )

def main():
    studio = get_or_create_studio()

    # If stopped, restart
    if studio.status != "running":
        print(f"Studio {STUDIO_NAME} is {studio.status}. Restarting...")
        studio.start(machine=MACHINE)
        # Wait until running
        for _ in range(60):
            studio.refresh()
            if studio.status == "running":
                break
            time.sleep(10)
        else:
            raise RuntimeError("Studio failed to start")

    print("Starting training iteration...")
    train()  # ensure train() uses CDN-only loader from step 2
    print("Training iteration complete.")

if __name__ == "__main__":
    main()
```

If you use Lightning `Work`/`Flow` APIs, adapt to check `studio.status` before `run()` and restart if needed.
