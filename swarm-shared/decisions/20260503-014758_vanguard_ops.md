# vanguard / ops

## 1. Diagnosis

- No persisted file-list cache exists; every training/data-selection run triggers authenticated `list_repo_tree` against HF API, burning quota and risking 429s.
- Training scripts likely use `load_dataset(streaming=True)` or repeated per-file API calls instead of CDN-only fetches, violating the HF CDN bypass pattern.
- No deterministic repo-sibling routing for commits; ingestion writes risk hitting the 128/hr/repo HF commit cap.
- No studio reuse guard before `Studio(create_ok=True)`; idle-stop kills training and wastes Lightning quota.
- Mac may still run local model loading or heavy compute instead of delegating to Lightning/Kaggle/Cerebras per the Mac=CLI rule.

## 2. Proposed change

Create `/opt/axentx/vanguard/ops/hf_file_cache.py` and wire it into the training launcher:
- Single function: `cache_repo_date_snapshot(repo, date_folder, out_path)` → uses one authenticated `list_repo_tree` call, saves `{repo}/{date_folder}/files.json`.
- Training script imports the JSON and builds CDN URLs (`https://huggingface.co/datasets/{repo}/resolve/main/{path}`) for zero-API data loading.
- Add sibling-aware commit helper: `pick_sibling_repo(slug, n=5)` → deterministic modulo hash to spread writes.
- Add studio reuse guard: `get_or_start_studio(name, machine)` → reuse Running studios, restart only if stopped.
- Guard at top-level launcher: refuse `model.from_pretrained` on non-Lightning contexts when on Mac.

## 3. Implementation

# /opt/axentx/vanguard/ops/hf_file_cache.py
#!/usr/bin/env python3
"""
HF CDN-only file-list cache and helpers for vanguard ops.
- Pre-list once per (repo, date_folder) and persist to JSON.
- Build CDN URLs for training to avoid HF API during data load.
- Deterministic sibling repo routing to avoid 128/hr commit cap.
- Studio reuse guard to save Lightning quota.
"""
from __future__ import annotations

import hashlib
import json
import os
import sys
from pathlib import Path
from typing import Dict, List, Optional

import requests

try:
    from huggingface_hub import HfApi, list_repo_tree
    from lightning import Fabric, LightningWork, LightningFlow, Studio
except ImportError as e:
    print(f"Missing dependency: {e}", file=sys.stderr)
    sys.exit(1)

HF_API = HfApi()
CACHE_DIR = Path(__file__).parent.parent / ".cache" / "hf_file_lists"
CACHE_DIR.mkdir(parents=True, exist_ok=True)

# ---------- CDN file-list cache ----------
def _cache_path(repo: str, date_folder: str) -> Path:
    safe_repo = repo.replace("/", "_")
    return CACHE_DIR / f"{safe_repo}__{date_folder}.json"

def cache_repo_date_snapshot(repo: str, date_folder: str, token: Optional[str] = None) -> List[str]:
    """
    Authenticated single call to list_repo_tree for one date folder.
    Returns list of file paths and persists to JSON.
    """
    cache_file = _cache_path(repo, date_folder)
    if cache_file.exists():
        return json.loads(cache_file.read_text())

    headers = {"Authorization": f"Bearer {token}"} if token else {}
    # list_repo_tree supports recursive=False to avoid paginating huge trees
    items = list(list_repo_tree(repo=repo, path=date_folder, recursive=False, token=token))
    paths = [it.rfilename for it in items if it.rfilename and not it.rfilename.endswith("/")]

    payload = {
        "repo": repo,
        "date_folder": date_folder,
        "paths": paths,
    }
    cache_file.write_text(json.dumps(payload, indent=2))
    return paths

def load_cached_paths(repo: str, date_folder: str) -> Optional[List[str]]:
    cache_file = _cache_path(repo, date_folder)
    if not cache_file.exists():
        return None
    data = json.loads(cache_file.read_text())
    return data.get("paths", [])

def build_cdn_url(repo: str, filepath: str) -> str:
    return f"https://huggingface.co/datasets/{repo}/resolve/main/{filepath}"

# ---------- Sibling repo routing ----------
def pick_sibling_repo(base_slug: str, n: int = 5, suffix_template: str = "-sib{}") -> str:
    """
    Deterministically pick one of N sibling repos to spread HF commit load.
    Example: base_slug='org/dataset-mirror' -> 'org/dataset-mirror-sib2'
    """
    digest = hashlib.sha256(base_slug.encode()).digest()
    idx = int.from_bytes(digest[:4], "big") % n
    if idx == 0:
        return base_slug
    return f"{base_slug}{suffix_template.format(idx)}"

# ---------- Lightning Studio reuse ----------
def get_or_start_studio(
    name: str,
    machine: str = "CPU-small",
    create_ok: bool = True,
) -> Studio:
    """
    Reuse a Running studio; restart only if stopped.
    """
    from lightning import Teamspace

    for s in Teamspace.studios:
        if s.name == name:
            if s.status == "running":
                return s
            if s.status in ("stopped", "failed"):
                s.start(machine=machine)
                return s
            # other transient states: wait or raise
            raise RuntimeError(f"Studio {name} in state {s.status}")

    if not create_ok:
        raise RuntimeError(f"Studio {name} not found and create_ok=False")
    return Studio(name=name, machine=machine, create_ok=True)

# ---------- Mac/CLI guard ----------
def ensure_not_local_model_load_on_mac():
    """
    Prevent expensive local model loading on orchestration machines.
    """
    if sys.platform == "darwin":
        # Lightweight check: if someone imports and calls from_pretrained in this process,
        # raise early in orchestration context. Training should go to Lightning.
        import warnings
        warnings.warn(
            "Running on macOS. Do not run model.from_pretrained() locally. "
            "Use Lightning Studio or remote APIs (Cerebras/Groq/Kaggle).",
            RuntimeWarning,
        )

# ---------- CLI helper ----------
def main():
    import argparse
    parser = argparse.ArgumentParser(description="Cache HF file list for CDN-only training.")
    parser.add_argument("--repo", required=True, help="HF dataset repo (user/repo)")
    parser.add_argument("--date", required=True, help="Date folder inside repo (e.g. 2026-04-29)")
    parser.add_argument("--token", default=os.getenv("HF_TOKEN"), help="HF token (optional if public)")
    parser.add_argument("--print-urls", action="store_true", help="Print CDN URLs")
    args = parser.parse_args()

    paths = cache_repo_date_snapshot(args.repo, args.date, token=args.token)
    if args.print_urls:
        for p in paths:
            print(build_cdn_url(args.repo, p))
    else:
        print(json.dumps({"repo": args.repo, "date": args.date, "count": len(paths), "paths": paths}, indent=2))

if __name__ == "__main__":
    main()
