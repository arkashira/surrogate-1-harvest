# airship / ops

## Highest-Value Incremental Improvement (≤2h)

**Goal**: Harden `airship discover` into a deterministic, CDN-only, zero-HF-API-runtime orchestrator that eliminates HF API rate limits and PyArrow schema errors while producing reproducible file lists for downstream training.

**Why this wins**:
- Removes HF API calls during training (bypasses 429/1000-5min and 128/hr commit caps).
- Eliminates PyArrow `CastError` from heterogeneous repo schemas by projecting only `{prompt,response}` at parse time.
- Enables reproducible Lightning Studio runs via embedded file lists (no API calls during data load).
- Fits within 2h: focused change to CLI + training script + small orchestration wrapper.

---

## Implementation Plan

1. **Add `airship discover` CLI command** (single entrypoint)
   - Accepts repo, date folder, and output JSON path.
   - Uses `list_repo_tree(path, recursive=False)` per subfolder (non-recursive to avoid pagination blowup).
   - Emits `{"date":"...","repo":"...","files":["path1","path2",...],"sha":"...","ts":"..."}`
   - Writes to `file-lists/` for traceability.

2. **Embed file list in training script**
   - Train script reads the JSON and downloads via CDN URLs only (`https://huggingface.co/datasets/{repo}/resolve/main/{path}`).
   - No `load_dataset` or `list_repo_files` during training.
   - Projection to `{prompt,response}` happens at parse time per file.

3. **Lightning Studio orchestration wrapper**
   - Reuse running studio if available (`Teamspace.studios` check).
   - Pass file-list path as argument to training script.
   - On idle stop, restart with `target.start(machine=Machine.L40S)` (or fallback to public tier).

4. **Cron/scheduler hygiene**
   - Ensure any wrapper scripts have `#!/usr/bin/env bash`, are `chmod +x`, and invoked via `bash <script> "$@"`.
   - Set `SHELL=/bin/bash` in crontab.

5. **Schema hardening for ingestion**
   - Move attribution to filename pattern: `batches/mirror-merged/{date}/{slug}.parquet`.
   - Do not add `source` or `ts` columns to parquet schema.

### 1. `airship/cli/discover.py`

```python
#!/usr/bin/env python3
import argparse
import json
import os
from datetime import datetime
from pathlib import Path

from huggingface_hub import HfApi, list_repo_tree

def discover(repo: str, date_folder: str, output_path: str):
    api = HfApi()
    base_path = f"{date_folder}"
    entries = list_repo_tree(repo=repo, path=base_path, recursive=False)

    files = []
    for entry in entries:
        # Only include files (skip subfolders if any)
        if entry.type == "file":
            files.append(entry.path)
        elif entry.type == "directory":
            # non-recursive per pattern; skip or one-level expand if needed
            sub = list_repo_tree(repo=repo, path=entry.path, recursive=False)
            for se in sub:
                if se.type == "file":
                    files.append(se.path)

    payload = {
        "repo": repo,
        "date": date_folder,
        "files": sorted(files),
        "sha": None,  # optionally resolve via api.repo_info(repo).sha
        "ts": datetime.utcnow().isoformat() + "Z",
    }

    out = Path(output_path)
    out.parent.mkdir(parents=True, exist_ok=True)
    out.write_text(json.dumps(payload, indent=2))
    print(f"Discovered {len(files)} files -> {out}")

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Deterministic CDN-only file discovery")
    parser.add_argument("--repo", required=True, help="HF dataset repo (user/repo)")
    parser.add_argument("--date", required=True, help="Date folder (e.g., 2026-04-29)")
    parser.add_argument("--output", required=True, help="Output JSON path")
    args = parser.parse_args()
    discover(args.repo, args.date, args.output)
```

### 2. `surrogate/train.py` (CDN-only loader)

```python
#!/usr/bin/env python3
import json
import requests
import pyarrow as pa
import pyarrow.parquet as pq
from io import BytesIO
from typing import List, Dict

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def load_file_list(file_list_path: str) -> Dict:
    with open(file_list_path) as f:
        return json.load(f)

def cdn_download(repo: str, path: str) -> bytes:
    url = CDN_TEMPLATE.format(repo=repo, path=path)
    resp = requests.get(url, timeout=30)
    resp.raise_for_status()
    return resp.content

def project_to_prompt_response(table: pa.Table) -> pa.Table:
    # Keep only prompt/response; tolerate missing cols by filling null
    cols = set(table.column_names)
    prompt_col = "prompt" if "prompt" in cols else None
    response_col = "response" if "response" in cols else None

    arrays = {}
    if prompt_col:
        arrays["prompt"] = table[prompt_col]
    else:
        arrays["prompt"] = pa.array([None] * len(table), type=pa.string())

    if response_col:
        arrays["response"] = table[response_col]
    else:
        arrays["response"] = pa.array([None] * len(table), type=pa.string())

    return pa.table(arrays)

def build_dataset(file_list_path: str):
    meta = load_file_list(file_list_path)
    repo = meta["repo"]
    rows = []

    for path in meta["files"]:
        try:
            raw = cdn_download(repo, path)
            # Try parquet first (common for mirror); fallback to jsonl if needed
            try:
                table = pq.read_table(BytesIO(raw))
            except Exception:
                # fallback: interpret as line-delimited json
                import pyarrow.json as pj
                table = pj.read_json(BytesIO(raw))
            projected = project_to_prompt_response(table)
            rows.append(projected)
        except Exception as exc:
            print(f"Skip {path}: {exc}")
            continue

    if not rows:
        raise ValueError("No valid files loaded")
    return pa.concat_tables(rows).combine_chunks()

if __name__ == "__main__":
    import sys
    if len(sys.argv) != 2:
        print("Usage: train.py <file-list.json>")
        sys.exit(1)
    dataset = build_dataset(sys.argv[1])
    print(f"Built dataset with {dataset.num_rows} rows")
    # Continue training using dataset.to_batches() or convert to HF Dataset in-memory
```
