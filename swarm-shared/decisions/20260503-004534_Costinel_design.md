**Scope**: Frontend-only card that identifies the most-connected hub from the knowledge-rag graph and surfaces the top hub with 3 contextual signals.  
**Timebox**: ≤2h.  
**Pattern**: Sense + Signal (read-only; no execution).

---

### 1) Highest-value incremental improvement
Ship a reusable, zero-backend card that:
- Reads a lightweight JSON graph snapshot (or in-memory map) produced by knowledge-rag.
- Picks the node with highest degree (most connections) as “Top Hub”.
- Renders a compact card: hub label, short description, and 3 contextual signals (neighbor highlights).
- Uses existing design tokens and fits the current dashboard surface.

Why this now: it directly applies “top-hub doc insight” and “knowledge-rag” patterns, requires no infra changes, and delivers immediate visibility for governance workflows.

#### `src/lib/topHubSignal.ts`
```ts
export interface RagNode {
  id: string;
  label: string;
  description: string;
  type: string;
}

export interface RagEdge {
  source: string;
  target: string;
  weight?: number;
}

export interface RagGraph {
  nodes: RagNode[];
  edges: RagEdge[];
}

export interface Signal {
  label: string;
  description: string;
}

export interface TopHubResult {
  hub: RagNode | null;
  signals: Signal[];
}

export function getTopHubWithSignals(graph: RagGraph): TopHubResult {
  if (!graph?.nodes?.length) return { hub: null, signals: [] };

  const degree: Record<string, number> = {};
  for (const node of graph.nodes) {
    degree[node.id] = 0;
  }
  for (const edge of graph.edges) {
    degree[edge.source] = (degree[edge.source] || 0) + (edge.weight || 1);
    degree[edge.target] = (degree[edge.target] || 0) + (edge.weight || 1);
  }

  const topId = Object.entries(degree).sort((a, b) => b[1] - a[1])[0]?.[0] || null;
  const hub = topId ? graph.nodes.find((n) => n.id === topId) || null : null;

  // Collect neighbor nodes as signals (unique, exclude hub)
  const neighborIds = new Set<string>();
  for (const edge of graph.edges) {
    if (edge.source === topId && edge.target !== topId) neighborIds.add(edge.target);
    if (edge.target === topId && edge.source !== topId) neighborIds.add(edge.source);
  }

  const signals: Signal[] = [];
  for (const nid of neighborIds) {
    if (signals.length >= 3) break;
    const node = graph.nodes.find((n) => n.id === nid);
    if (node) signals.push({ label: node.label, description: node.description });
  }

  return { hub, signals };
}
```

#### `src/components/TopHubSignalCard.tsx`
```tsx
import React from "react";
import { getTopHubWithSignals, type RagGraph } from "../lib/topHubSignal";
import graphData from "../data/knowledge-rag-graph.json";

export const TopHubSignalCard: React.FC<{ graph?: RagGraph }> = ({ graph = graphData as RagGraph }) => {
  const { hub, signals } = getTopHubWithSignals(graph);

  if (!hub) return null;

  return (
    <article className="rounded-lg border border-gray-200 bg-white p-4 shadow-sm dark:border-gray-700 dark:bg-gray-800" aria-label="Top hub signal">
      <div className="mb-3">
        <h2 className="text-base font-semibold text-gray-900 dark:text-gray-100">{hub.label}</h2>
        <p className="text-sm text-gray-600 dark:text-gray-400">{hub.description}</p>
      </div>

      {signals.length > 0 && (
        <ul className="space-y-2" aria-label="Contextual signals">
          {signals.map((s, i) => (
            <li key={i} className="flex items-start gap-2 text-sm">
              <span className="mt-1 flex h-2 w-2 shrink-0 rounded-full bg-blue-500 dark:bg-blue-400" aria-hidden="true" />
              <div>
                <span className="font-medium text-gray-900 dark:text-gray-100">{s.label}</span>
                <p className="text-gray-600 dark:text-gray-400">{s.description}</p>
              </div>
            </li>
          ))}
        </ul>
      )}

      <footer className="mt-3 text-xs text-gray-400 dark:text-gray-500">Sense + Signal — ไม่ Execute</footer>
    </article>
  );
};
```
