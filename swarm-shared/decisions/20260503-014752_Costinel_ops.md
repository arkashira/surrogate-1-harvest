# Costinel / ops

## Implementation Plan — Top-Hub Signal Panel (Costinel Dashboard)

**Scope & value**  
- Frontend-only, read-only panel that surfaces the most-connected hub (default “MOC”) and its actionable proposals from the knowledge graph.  
- Resilient to missing backend: falls back to local seed data so the panel always renders.  
- Zero runtime dependencies on HF API / training infra — safe for ops dashboards.  
- Estimated effort: ~90 min (build + test + polish).

### 2) Seed knowledge graph (MOC hub)

`src/lib/knowledgeGraph.ts`

```ts
// Minimal, typed contract for TopHub insights
export interface Proposal {
  id: string;
  title: string;
  summary: string;
  impact: 'high' | 'medium' | 'low';
  tags: string[];
  href?: string;
}

export interface HubInsight {
  hub: string;
  description: string;
  proposals: Proposal[];
  updatedAt: string;
}

// Local seed (MOC = most-connected hub)
const SEED_MOC: HubInsight = {
  hub: 'MOC',
  description:
    'Multi-org cost governance hub — central signals for cross-account RI coverage, orphaned resources, and budget anomalies.',
  updatedAt: new Date().toISOString(),
  proposals: [
    {
      id: 'MOC-001',
      title: 'Increase RI coverage for prod us-east-1',
      summary: 'Current coverage 42% → 75% target; estimated 34% YoY savings.',
      impact: 'high',
      tags: ['RI', 'AWS', 'prod'],
    },
    {
      id: 'MOC-002',
      title: 'Orphaned EBS snapshot cleanup',
      summary: '31 snapshots >90d unused across 4 accounts; reclaim ~$1.2k/mo.',
      impact: 'medium',
      tags: ['storage', 'cleanup'],
    },
    {
      id: 'MOC-003',
      title: 'Anomaly: weekend dev-cluster spend spike',
      summary: 'Sat–Sun spend +220% vs weekday baseline; verify scheduled jobs.',
      impact: 'high',
      tags: ['anomaly', 'dev', 'governance'],
    },
  ],
};

export async function fetchTopHubInsights(): Promise<HubInsight> {
  // Best-effort remote fetch; fallback to seed to keep panel resilient.
  try {
    const res = await fetch('/api/knowledge-graph/top-hub', {
      method: 'GET',
      headers: { 'Content-Type': 'application/json' },
      // short timeout to avoid blocking dashboard
      signal: AbortSignal.timeout(2500),
    });

    if (!res.ok) throw new Error(`HTTP ${res.status}`);
    const json = (await res.json()) as HubInsight;
    if (!json?.hub || !Array.isArray(json.proposals)) throw new Error('Invalid payload');
    return json;
  } catch {
    return SEED_MOC;
  }
}
```

### 3) Hook to load insights (SSR-friendly)

`src/hooks/useTopHubInsights.ts`

```ts
import { useQuery } from '@tanstack/react-query';
import { fetchTopHubInsights, type HubInsight } from '@/lib/knowledgeGraph';

export function useTopHubInsights() {
  return useQuery<HubInsight>({
    queryKey: ['top-hub-insights'],
    queryFn: fetchTopHubInsights,
    staleTime: 5 * 60 * 1000, // 5m
    retry: 1,
    refetchOnWindowFocus: false,
  });
}
```

(If not using react-query, a simple `useEffect` + state is fine — keep it lightweight.)

### 4) Panel component

`src/components/TopHubSignalPanel.tsx`

```tsx
import { useTopHubInsights } from '@/hooks/useTopHubInsights';
import { Proposal } from '@/lib/knowledgeGraph';

const impactColors = {
  high: 'bg-red-50 border-red-200 text-red-800',
  medium: 'bg-amber-50 border-amber-200 text-amber-800',
  low: 'bg-emerald-50 border-emerald-200 text-emerald-800',
} as const;

function ProposalCard({ p }: { p: Proposal }) {
  return (
    <article className="rounded-lg border bg-white p-3 shadow-sm">
      <div className="flex items-start justify-between gap-2">
        <div>
          <h4 className="text-sm font-semibold text-gray-900">{p.title}</h4>
          <p className="mt-1 text-xs text-gray-600">{p.summary}</p>
        </div>
        <span className={`whitespace-nowrap rounded-full px-2 py-0.5 text-xs font-medium ${impactColors[p.impact]}`}>
          {p.impact}
        </span>
      </div>
      <div className="mt-2 flex flex-wrap gap-1">
        {p.tags.map((t) => (
          <span key={t} className="rounded bg-gray-100 px-1.5 py-0.5 text-[10px] uppercase text-gray-500">
            {t}
          </span>
        ))}
      </div>
    </article>
  );
}

export default function TopHubSignalPanel() {
  const { data, isLoading, isError } = useTopHubInsights();

  if (isLoading) {
    return (
      <section className="rounded-xl border bg-gray-50 p-4">
        <div className="h-5 w-32 animate-pulse rounded bg-gray-200" />
        <div className="mt-4 space-y-3">
          {Array(3)
            .fill(0)
            .map((_, i) => (
              <div key={i} className="h-16 animate-pulse rounded bg-white shadow-sm" />
            ))}
        </div>
      </section>
    );
  }

  if (isError || !data) {
    return (
      <section className="rounded-xl border bg-gray-50 p-4 text-center text-sm text-gray-500">
        Unable to load top-hub signals.
      </section>
    );
  }

  return (
    <section className="rounded-xl border bg-white p-4 shadow-sm">
      <header className="mb-3">
        <div className="flex items-center gap-2">
          <h3 className="text-base font-semibold text-gray-900">Top Hub: {data.hub}</h3>
          <span className="rounded bg-blue-50 px-2 py-0.5 text-[10px] font-medium uppercase text-blue-700">
            most-connected
          </span>
        </div>
        <p className="mt-1 text-xs text-gray-500">{data.description}</p>
      </header>

      <div className="space-y-2">
        {data.proposals.map((p) => (
          <ProposalCard key={p.id} p={p} />
        ))}
      </div>

      <footer className="mt-3 text-right">
        <span className="text-[10px] uppercase text-gray-400">Updated {new Date(data.updatedAt).toLocaleString()}</span>
      </footer>
    </section>
  );
}
```
