## 1) Mac orchestration script (list once → train many, zero API during training)

`scripts/list_hf_files.py`

```python
#!/usr/bin/env python3
"""
List files in a date folder of a HuggingFace dataset repo.
Run from Mac after rate-limit window clears. Saves file list JSON
for Lightning training (CDN-only, zero API calls during training).

Usage:
  python scripts/list_hf_files.py \
    --repo datasets/your-org/surrogate-mirror \
    --date 2026-04-29 \
    --out surrogate/file_list_2026-04-29.json
"""
import argparse
import json
import os
import sys
from typing import List

try:
    from huggingface_hub import HfApi
except ImportError:
    print("Install: pip install huggingface_hub")
    sys.exit(1)

API = HfApi()

def list_date_folder(repo_id: str, date: str) -> List[str]:
    """
    List files in repo at path=f'{date}/' (non-recursive by folder).
    Returns relative paths like '2026-04-29/shard-0001.parquet'.
    """
    base = date.strip("/")
    out = []

    try:
        items = API.list_repo_tree(repo_id=repo_id, path=base, recursive=False)
    except Exception as e:
        print(f"Error listing {repo_id}/{base}: {e}")
        return []

    for item in items:
        if not item.path.startswith(base):
            continue
        if item.type == "file":
            out.append(item.path)
        elif item.type == "dir":
            try:
                subitems = API.list_repo_tree(repo_id=repo_id, path=item.path, recursive=False)
                for si in subitems:
                    if si.type == "file":
                        out.append(si.path)
            except Exception as e:
                print(f"Error listing {repo_id}/{item.path}: {e}")

    out.sort()
    return out

def main() -> None:
    parser = argparse.ArgumentParser(description="List HF dataset files for a date folder.")
    parser.add_argument("--repo", required=True, help="HF dataset repo id (e.g., datasets/your-org/surrogate-mirror)")
    parser.add_argument("--date", required=True, help="Date folder (e.g., 2026-04-29)")
    parser.add_argument("--out", required=True, help="Output JSON path")
    args = parser.parse_args()

    repo_id = args.repo.replace("datasets/", "")
    files = list_date_folder(repo_id, args.date)

    payload = {
        "repo_id": repo_id,
        "date": args.date,
        "files": files,
        "count": len(files),
        "note": "Use CDN URLs during training: https://huggingface.co/datasets/{repo_id}/resolve/main/{path}"
    }

    os.makedirs(os.path.dirname(os.path.abspath(args.out)), exist_ok=True)
    with open(args.out, "w") as f:
        json.dump(payload, f, indent=2)

    print(f"Wrote {len(files)} files to {args.out}")

if __name__ == "__main__":
    main()
```

Make executable:

```bash
chmod +x scripts/list_hf_files.py
```

## 2) CDN-only parquet loader (zero HF API/auth during training)

`surrogate/data.py`

```python
import json
from typing import List, Optional, Dict
import pyarrow.parquet as pq
import requests
from torch.utils.data import Dataset

HF_CDN_TEMPLATE = "https://huggingface.co/datasets/{repo_id}/resolve/main/{path}"

def load_parquet_cdn_only(repo_id: str, file_path: str, columns: Optional[List[str]] = None):
    """
    Download a single parquet file via CDN and return a pyarrow Table.
    No Authorization header -> bypasses /api/ rate limits.
    """
    url = HF_CDN_TEMPLATE.format(repo_id=repo_id, path=file_path)
    resp = requests.get(url, timeout=60)
    resp.raise_for_status()
    return pq.read_table(pq.ParquetFile(pq.ParquetBuffer(resp.content)), columns=columns)

class CDNParquetDataset(Dataset):
    """
    Dataset that loads parquet shards via CDN.
    Expects file_list JSON produced by scripts/list_hf_files.py.
    Projects each row to {prompt, response} at parse time.
    """
    def __init__(self, repo_id: str, file_list_json: str, max_rows: Optional[int] = None):
        super().__init__()
        self.repo_id = repo_id
        with open(file_list_json) as f:
            payload = json.load(f)
        self.files = payload["files"]
        if not self.files:
            raise ValueError("No files found in file_list_json")
        self.max_rows = max_rows
        self._length = None

    def __len__(self):
        if self._length is not None:
            return self._length
        try:
            tbl = load_parquet_cdn_only(self.repo_id, self.files[0], columns=[])
            self._length = tbl.num_rows * len(self.files)
            if self.max_rows:
                self._length = min(self._length, self.max_rows)
        except Exception:
            self._length = 0
        return self._length

    def __getitem__(self, idx):
        shard_idx = 0
        rows_acc = 0
        while shard_idx < len(self.files):
            tbl = load_parquet_cdn_only(
                self.repo_id,
                self.files[shard_idx],
                columns=["prompt", "response"]
            )
            n = tbl.num_rows
            if idx < rows_acc + n:
                row = tbl.slice(idx - rows_acc, 1).to_pydict()
                return {
                    "prompt": row.get("prompt", [""])[0],
                    "response": row.get("response", [""])[0],
                }
            rows_acc += n
            shard_idx += 1
        raise IndexError("idx out of range")
```
