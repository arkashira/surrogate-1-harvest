# Costinel / ops

## Final Implementation Plan — CDN-First Top-Hub Signal Panel (<2h)

**Goal**: Add a resilient, read-only “Top Hub” signal panel to the Costinel Dashboard that surfaces the most-connected hub and related docs using CDN-fetched artifacts (zero runtime HF API calls, no auth, no mutations).

**Principle**: Strict **Sense + Signal (ไม่ Execute)** — graceful degradation, CDN-only, no build-breaking changes.

---

### 1) Artifact contract (published by ops/knowledge-rag)

**Path**  
`https://huggingface.co/datasets/axentx/costinel-knowledge/resolve/main/top-hub/latest.json`

**Shape** (canonical, normalized)
```json
{
  "hub": "MOC",
  "score": 0.94,
  "description": "Mission Operations Center — highest betweenness centrality in cost-governance graph",
  "related": [
    { "title": "Cost governance playbook", "url": "https://huggingface.co/datasets/axentx/costinel-knowledge/resolve/main/docs/playbook.md" },
    { "title": "RI coverage analysis", "url": "https://huggingface.co/datasets/axentx/costinel-knowledge/resolve/main/docs/ri-coverage.md" }
  ],
  "updatedAt": "2026-05-03T04:31:40.000Z"
}
```

**Rules**
- `hub` (string, required), `score` (number 0–1), `updatedAt` (ISO).
- `related` is optional; each item must have `title` and `url`.
- If the CDN 404s or payload is malformed, the panel renders nothing (silent degradation).

### 3) Code

#### `lib/cdn.ts`
```ts
// lib/cdn.ts
export interface TopHubPayload {
  hub: string;
  score: number;
  description: string;
  related: Array<{ title: string; url: string }>;
  updatedAt: string;
}

const TOP_HUB_URL =
  'https://huggingface.co/datasets/axentx/costinel-knowledge/resolve/main/top-hub/latest.json';

const FALLBACK_TTL = 5 * 60 * 1000; // 5 minutes

type CacheEntry<T> = { data: T; ts: number };
const memoryCache = new Map<string, CacheEntry<TopHubPayload | null>>();

export async function fetchTopHub(opts?: { fresh?: boolean }): Promise<TopHubPayload | null> {
  const cacheKey = TOP_HUB_URL;
  const cached = memoryCache.get(cacheKey);

  if (!opts?.fresh && cached && Date.now() - cached.ts < FALLBACK_TTL) {
    return cached.data;
  }

  try {
    const res = await fetch(TOP_HUB_URL, {
      cache: 'no-store',
      headers: { Accept: 'application/json' },
    });

    if (!res.ok) {
      // CDN may 404 if not published yet — degrade gracefully
      console.warn('[CDN] top-hub unavailable', res.status);
      memoryCache.set(cacheKey, { data: null, ts: Date.now() });
      return null;
    }

    const data = (await res.json()) as TopHubPayload;

    // Basic validation
    if (!data?.hub || typeof data.score !== 'number') {
      console.warn('[CDN] top-hub malformed', data);
      memoryCache.set(cacheKey, { data: null, ts: Date.now() });
      return null;
    }

    memoryCache.set(cacheKey, { data, ts: Date.now() });
    return data;
  } catch (err) {
    console.error('[CDN] top-hub fetch failed', err);
    memoryCache.set(cacheKey, { data: null, ts: Date.now() });
    return null;
  }
}
```

#### `hooks/useTopHubSignal.ts`
```ts
// hooks/useTopHubSignal.ts
import { useEffect, useState } from 'react';
import { fetchTopHub, TopHubPayload } from '@/lib/cdn';

export function useTopHubSignal(pollMs = 300_000) {
  const [data, setData] = useState<TopHubPayload | null>(null);
  const [loading, setLoading] = useState(true);

  const load = async (opts?: { fresh?: boolean }) => {
    setLoading(true);
    const result = await fetchTopHub(opts);
    setData(result);
    setLoading(false);
  };

  useEffect(() => {
    load({ fresh: true });
    const id = setInterval(() => load(), pollMs);
    return () => clearInterval(id);
  }, [pollMs]);

  return { data, loading };
}
```

#### `components/TopHubSignalPanel.tsx`
```tsx
// components/TopHubSignalPanel.tsx
import { useTopHubSignal } from '@/hooks/useTopHubSignal';
import { ExternalLink } from 'lucide-react';

export function TopHubSignalPanel() {
  const { data, loading } = useTopHubSignal();

  if (loading) {
    return (
      <div className="rounded-lg border bg-card p-4">
        <div className="h-5 w-32 animate-pulse rounded bg-muted" />
        <div className="mt-2 h-4 w-24 animate-pulse rounded bg-muted" />
      </div>
    );
  }

  if (!data) {
    // Silent degradation — no signal to show
    return null;
  }

  return (
    <div className="rounded-lg border bg-card p-4 shadow-sm">
      <div className="flex items-start justify-between">
        <div>
          <h3 className="text-sm font-semibold text-foreground">Top Hub</h3>
          <p className="mt-1 text-2xl font-bold tracking-tight">{data.hub}</p>
          <p className="mt-1 text-xs text-muted-foreground">
            Score {Math.round(data.score * 100)}%
          </p>
          {data.description && (
            <p className="mt-2 text-sm text-muted-foreground">{data.description}</p>
          )}
        </div>
      </div>

      {data.related && data.related.length > 0 && (
        <div className="mt-3">
          <p className="text-xs font-medium text-muted-foreground">Related</p>
          <ul className="mt-1 space-y-1">
            {data.related.map((item) => (
              <li key={item.url}>
                <a
                  href={item.url}
                  target="_blank"
                  rel="noopener noreferrer"
                  className="flex items-center gap-1 text-xs text-primary hover:underline"
                >
                  <ExternalLink className="h-3 w-3" />
                  {item.title}
                </a>
              </li>
            ))}
          </ul>
        </div>
      )}

      <p className="mt-3 text-[10px] text-muted-foreground/60">
        Updated {new Date(data.updatedAt).toLocaleString()}
      </p>
    </div>
  );
}
```
