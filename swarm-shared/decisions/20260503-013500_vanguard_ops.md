# vanguard / ops

### Final Consolidated Plan (Ops-focused)

**Core diagnosis**  
- Authenticated `list_repo_tree` runs on every load → burns HF API quota (1000/5 min) → 429s.  
- No persisted `(repo, dateFolder) → file-list` manifest → repeated enumeration and redundant API calls.  
- Training uses `load_dataset(streaming=True)` or similar against mixed-schema repos → pagination, schema/pyarrow issues, and more API traffic.  
- Lightning Studio is not reused → jobs recreate sessions and waste quota; idle timeouts kill runs.  
- Local/mac scripts may bypass Lightning and train directly → local compute waste and schema issues.

**Goal**  
Eliminate authenticated tree/list calls during training, remove 429s, and minimize quota waste by:  
1) building a one-shot file manifest,  
2) training via CDN-only fetches using that manifest,  
3) reusing a running Lightning Studio.

```python
#!/usr/bin/env python3
"""
One-shot manifest builder.
Usage:
  SHELL=/bin/bash python3 ops/build_file_manifest.py \
    --repo axentx/surrogate-corpus \
    --date-folder 2026-05-03
"""
import argparse
import json
import sys
from datetime import datetime
from pathlib import Path

try:
    from huggingface_hub import list_repo_tree
except ImportError:
    print("Install: pip install huggingface_hub")
    sys.exit(1)

def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g. axentx/surrogate-corpus)")
    parser.add_argument("--date-folder", required=True, help="Date folder inside repo (e.g. 2026-05-03)")
    args = parser.parse_args()

    # Single non-recursive call (avoids pagination explosion)
    tree = list_repo_tree(repo_id=args.repo, path=args.date_folder, recursive=False)
    files = [item.rfilename for item in tree if not item.rfilename.endswith("/")]

    manifest = {
        "repo": args.repo,
        "date_folder": args.date_folder,
        "created_at": datetime.utcnow().isoformat() + "Z",
        "files": sorted(files),
        "note": "CDN-only; use https://huggingface.co/datasets/{repo}/resolve/main/{date_folder}/{file}"
    }

    out_dir = Path(__file__).parent.parent / "data" / "manifests"
    out_dir.mkdir(parents=True, exist_ok=True)
    out_path = out_dir / f"{args.repo.replace('/', '__')}__{args.date_folder}.json"
    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"Manifest written: {out_path}")

if __name__ == "__main__":
    main()
```

```python
"""
Surrogate training with CDN-only fetches.
Usage:
  python surrogate_train.py --manifest vanguard/data/manifests/axentx__surrogate-corpus__2026-05-03.json
"""
import argparse
import json
import sys
from pathlib import Path

try:
    import lightning as L
    import torch
    from torch.utils.data import IterableDataset, DataLoader
    import requests
except ImportError as e:
    print(f"Missing dep: {e}")
    sys.exit(1)

HF_CDN = "https://huggingface.co/datasets"

class CDNTextDataset(IterableDataset):
    def __init__(self, manifest_path, max_files=None):
        manifest_path = Path(manifest_path)
        with manifest_path.open() as f:
            self.manifest = json.load(f)
        self.repo = self.manifest["repo"]
        self.date_folder = self.manifest["date_folder"]
        self.files = self.manifest["files"]
        if max_files:
            self.files = self.files[:max_files]

    def _stream_files(self):
        for fn in self.files:
            url = f"{HF_CDN}/{self.repo}/resolve/main/{self.date_folder}/{fn}"
            resp = requests.get(url, timeout=30)
            resp.raise_for_status()
            text = resp.content.decode("utf-8", errors="replace").strip()
            if text:
                # Lightweight projection: treat whole file as one prompt/response pair
                # Replace with your actual parsing logic if format differs.
                yield {"prompt": text, "response": ""}

    def __iter__(self):
        yield from self._stream_files()

class SimpleModel(torch.nn.Module):
    def __init__(self, vocab_size=32000, d_model=512):
        super().__init__()
        self.embed = torch.nn.Embedding(vocab_size, d_model)
        self.out = torch.nn.Linear(d_model, vocab_size)

    def forward(self, x):
        return self.out(self.embed(x))

class SurrogateTask(L.LightningModule):
    def __init__(self, manifest_path, lr=1e-3, max_files=None):
        super().__init__()
        self.save_hyperparameters()
        self.model = SimpleModel()
        self.dataset = CDNTextDataset(manifest_path, max_files=max_files)

    def train_dataloader(self):
        return DataLoader(self.dataset, batch_size=1)

    def training_step(self, batch, batch_idx):
        # Minimal step: dummy loss to keep training loop alive.
        # Replace with real tokenization + LM objective.
        loss = torch.tensor(0.0, requires_grad=True)
        self.log("train_loss", loss)
        return loss

    def configure_optimizers(self):
        return torch.optim.Adam(self.parameters(), lr=self.hparams.lr)

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--manifest", required=True, help="Path to manifest JSON")
    parser.add_argument("--max-files", type=int, default=None, help="Limit files for quick tests")
    parser.add_argument("--lr", type=float, default=1e-3)
    args = parser.parse_args()

    trainer = L.Trainer(max_epochs=1, accelerator="cpu", devices=1)
```
