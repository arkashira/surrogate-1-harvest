# workio / backend

## Final Synthesis (Correct + Actionable)

### 1. Root-cause summary (merged)
- **Non-atomic upsert**: check-then-insert allows duplicates under retries/races.
- **No DB uniqueness guard**: no constraint on `(employee_id, punch_date, punch_type)`.
- **No idempotency key**: LINE retries and client double-taps are indistinguishable.
- **Missing state-transition guard**: clock-out allowed without prior clock-in; no minimum interval enforcement.
- **Location/shift rules applied outside transaction**: validation can be bypassed by races.

### 2. Concrete changes (prioritized)

#### 2.2 Attendance controller — atomic, state-safe, location-aware

import { Request, Response } from 'express';
import { pool } from '../db';
import { validatePunchLocation } from '../utils/location';

type PunchType = 'clock_in' | 'clock_out';

export async function recordPunch(req: Request, res: Response) {
  const { employee_id, punch_type, latitude, longitude } = req.body;

  if (!employee_id || !['clock_in', 'clock_out'].includes(punch_type)) {
    return res.status(400).json({ error: 'Invalid payload' });
  }

  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    const today = new Date().toISOString().split('T')[0]; // YYYY-MM-DD

    // Location validation (must pass within same transaction context)
    const locationOk = await validatePunchLocation(latitude, longitude);
    if (!locationOk) {
      await client.query('ROLLBACK');
      return res.status(400).json({ error: 'Location not allowed' });
    }

    // State transition guard + atomic upsert in one roundtrip
    // - Prevent duplicate punch (uniq constraint will block)
    // - Prevent clock_out without prior clock_in today
    // - Enforce minimum interval (e.g., 10 minutes) between same-type punches
    const upsertQuery = `
      WITH last_punch AS (
        SELECT punch_type AS last_type, created_at AS last_at
        FROM punches
        WHERE employee_id = $1 AND punch_date = $2
        ORDER BY created_at DESC
        LIMIT 1
      ),
      can_insert AS (
        SELECT
          CASE
            WHEN $3 = 'clock_out' AND (last_type IS NULL OR last_type <> 'clock_in') THEN false
            WHEN $3 = 'clock_in'  AND last_type = 'clock_in'
              AND (NOW() - last_at) < INTERVAL '10 minutes' THEN false
            ELSE true
          END AS ok
      )
      INSERT INTO punches (employee_id, punch_date, punch_type, latitude, longitude, created_at)
      SELECT $1, $2, $3, $4, $5, NOW()
      FROM can_insert
      WHERE ok
      ON CONFLICT (employee_id, punch_date, punch_type) DO NOTHING
      RETURNING *;
    `;
    const result = await client.query(upsertQuery, [employee_id, today, punch_type, latitude, longitude]);

    if (result.rowCount === 0) {
      // Fetch existing to return stable payload
      const existing = await client.query(
        'SELECT * FROM punches WHERE employee_id = $1 AND punch_date = $2 AND punch_type = $3',
        [employee_id, today, punch_type]
      );
      await client.query('COMMIT');
      return res.json({ punch: existing.rows[0], duplicate: true });
    }

    await client.query('COMMIT');
    return res.json({ punch: result.rows[0], duplicate: false });
  } catch (err: any) {
    await client.query('ROLLBACK').catch(() => {});
    // Unique violation can still occur under extreme races; treat as duplicate
    if (err.code === '23505') {
      const existing = await pool.query(
        'SELECT * FROM punches WHERE employee_id = $1 AND punch_date = $2 AND punch_type = $3',
        [employee_id, todayFromReq(req), punch_type]
      );
      return res.json({ punch: existing.rows[0], duplicate: true });
    }
    console.error('recordPunch error:', err);
    return res.status(500).json({ error: 'Internal server error' });
  } finally {
    client.release();
  }
}

function todayFromReq(req: Request): string {
  return new Date().toISOString().split('T')[0];
}
