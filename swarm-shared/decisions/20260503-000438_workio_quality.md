# workio / quality

## Final synthesized answer

### Diagnosis (merged)
- Duplicate requests within the same minute (retries, double-taps, LINE webhook replays) create multiple punches for the same tenant/user/day.
- No server-side state machine: invalid sequences allowed (clock-in twice, clock-out without clock-in).
- No idempotency layer at the API boundary: retries and webhook replays create duplicate rows and corrupt daily totals.
- Frontend lacks optimistic UI and local pending queue: ambiguous spinners and lost punches on offline/network flaps.
- No rollback/compensation: partial failures in webhook processing leave inconsistent side effects.

### Single focused change
Add server-side idempotency + state-machine enforcement for punches in one cohesive change.

**Scope**
- `workio/server/src/services/punchService.ts` (create/update)
- `workio/server/src/routes/punchRoutes.ts` (or equivalent)
- `workio/server/src/middleware/idempotency.ts` (create)
- Add `idempotency_key` column to `punches` table (nullable, indexed).

// workio/server/src/services/punchService.ts
import { Pool } from 'pg';
import { getPool } from './db';

const pool: Pool = getPool();

export type PunchType = 'CLOCK_IN' | 'CLOCK_OUT';

export interface Punch {
  id: string;
  tenant_id: string;
  user_id: string;
  type: PunchType;
  timestamp: Date;
  latitude?: number;
  longitude?: number;
  idempotency_key?: string;
  created_at: Date;
}

// State rules
function canClockIn(lastPunch: Punch | null): boolean {
  return !lastPunch || lastPunch.type === 'CLOCK_OUT';
}

function canClockOut(lastPunch: Punch | null): boolean {
  return !!lastPunch && lastPunch.type === 'CLOCK_IN';
}

/**
 * Ensure one valid punch per (tenant,user,type) within a 60-second dedup window
 * and enforce state transitions. Uses idempotency key when provided.
 */
export async function ensureUniquePunchForWindow(
  tenantId: string,
  userId: string,
  type: PunchType,
  timestamp: Date,
  options?: { lat?: number; lng?: number; idempotencyKey?: string }
): Promise<{ punch: Punch; created: boolean }> {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    const dateStr = timestamp.toISOString().split('T')[0];
    const windowStart = new Date(timestamp.getTime() - 60_000);
    const windowEnd = new Date(timestamp.getTime() + 60_000);

    // 1) Idempotency key check (preferred)
    if (options?.idempotencyKey) {
      const idem = await client.query<Punch>(
        `SELECT * FROM punches
         WHERE tenant_id = $1 AND user_id = $2 AND idempotency_key = $3
         LIMIT 1`,
        [tenantId, userId, options.idempotencyKey]
      );
      if (idem.rows[0]) {
        await client.query('ROLLBACK');
        return { punch: idem.rows[0], created: false };
      }
    }

    // 2) Dedup window check (same type within 60s)
    const dup = await client.query<Punch>(
      `SELECT * FROM punches
       WHERE tenant_id = $1 AND user_id = $2 AND type = $3
         AND timestamp BETWEEN $4 AND $5
       ORDER BY timestamp DESC
       LIMIT 1`,
      [tenantId, userId, type, windowStart, windowEnd]
    );
    if (dup.rows[0]) {
      await client.query('ROLLBACK');
      return { punch: dup.rows[0], created: false };
    }

    // 3) State machine: last punch
    const last = await client.query<Punch>(
      `SELECT * FROM punches
       WHERE tenant_id = $1 AND user_id = $2
       ORDER BY timestamp DESC
       LIMIT 1`,
      [tenantId, userId]
    );
    const lastPunch: Punch | null = last.rows[0] || null;

    if (type === 'CLOCK_IN' && !canClockIn(lastPunch)) {
      await client.query('ROLLBACK');
      throw new Error('Cannot clock in: previous punch is not a clock-out.');
    }
    if (type === 'CLOCK_OUT' && !canClockOut(lastPunch)) {
      await client.query('ROLLBACK');
      throw new Error('Cannot clock out: no active clock-in.');
    }

    // 4) Insert
    const id = `p_${Date.now()}_${Math.random().toString(36).slice(2, 9)}`;
    const insert = await client.query<Punch>(
      `INSERT INTO punches (id, tenant_id, user_id, type, timestamp, latitude, longitude, idempotency_key)
       VALUES ($1, $2, $3, $4, $5, $6, $7, $8) RETURNING *`,
      [
        id,
        tenantId,
        userId,
        type,
        timestamp,
        options?.lat ?? null,
        options?.lng ?? null,
        options?.idempotencyKey ?? null,
      ]
    );

    await client.query('COMMIT');
    return { punch: insert.rows[0], created: true };
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  } finally {
    client.release();
  }
}

// workio/server/src/middleware/idempotency.ts
import { Request, Response, NextFunction } from 'express';

export function idempotencyKeyFromLine(req: Request, _res: Response, next: NextFunction) {
  // Deterministic key for LINE webhook dedup: X-Line-Signature + route + body length
  const sig = req.headers['x-line-signature'] as string | undefined;
  if (sig) {
    const body = JSON.stringify(req.body || {});
    const key = `line_${Buffer.from(sig + req.path + body.length).toString('base64url').slice(0, 64)}`;
    req.idempotencyKey = key;
  }
  next();
}

export function acceptIdempotencyKey(req: Request, _res: Response, next: NextFunction) {
  const header = req.headers['idempotency-key'] as string | undefined;
  if (header) req.idempotencyKey = header;
  next();
}

// Types
declare module 'express-serve-static-core' {
  interface Request {
    idempotencyKey?: string;
  }
}

// workio/server/src/routes/punchRoutes.ts
import { Router } from 'express';
import { ensureUniquePunchForWindow } from '../services/punchService';
import { acceptIdempotencyKey, idempotencyKeyFromLine } from '../middleware/idempotency';

const router = Router();

router.post('/clock-in', acceptIdempotencyKey, async (req, res) => {
  try {
    const { tenantId, userId, timestamp, lat, lng } = req.body;
    const { punch, created } = await ensureUniquePunchForWindow(
      tenantId,
      userId,
      'CLOCK_IN',
      new Date(timestamp || Date.now()),
      { lat, lng, idempotencyKey: req.idempotencyKey }
    );
    res.status(created ? 201 : 200).json({ punch, duplicate: !created });
  } catch (err: any) {
    res.status(400).json({ error: err.message });
  }
});
