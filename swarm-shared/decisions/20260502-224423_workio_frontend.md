# workio / frontend

## Final Synthesis (adopted)

This merges the strongest, most actionable parts of both proposals and resolves contradictions in favor of correctness and concrete actionability.

- **Local-first queue + optimistic UI** for clock-in/out and leave/OT requests (adopt scope from Candidate 1; naming/structure refined).
- **Idempotency keys** on all state-changing requests (Candidate 1 implementation, tightened).
- **Client-side validation before submit** (Candidate 2) to avoid avoidable roundtrips: required fields for leave; GPS/location pre-check for clock-in.
- **Explicit sync status + inline retry affordances** (Candidate 2) so users can see pending/failed and retry.
- **Atomic state updates + rollback** (Candidate 1) with clear error boundaries and persistence for critical drafts.
- **Offline-first retry with exponential backoff + reconnect replay** (both proposals; unified).

```ts
export function createIdempotencyKey(): string {
  // monotonic-ish + random to avoid collisions across tabs/clients
  return `${Date.now()}-${crypto.randomUUID?.() || Math.random().toString(36).slice(2)}`;
}

export async function postWithIdempotency<T>(
  url: string,
  body: unknown,
  idempotencyKey?: string,
  signal?: AbortSignal
): Promise<T> {
  const headers: Record<string, string> = {
    'Content-Type': 'application/json',
  };
  if (idempotencyKey) headers['Idempotency-Key'] = idempotencyKey;

  const res = await fetch(url, {
    method: 'POST',
    headers,
    body: JSON.stringify(body),
    credentials: 'include',
    signal,
  });

  if (!res.ok) {
    const text = await res.text().catch(() => '');
    const err = new Error(`HTTP ${res.status}: ${text || res.statusText}`);
    // surface status for classification (4xx vs 5xx)
    (err as any).status = res.status;
    throw err;
  }

  return res.json();
}
```

### src/hooks/useActionQueue.ts
Robust, minimal queue with persistence, idempotency, and retry/backoff.

```ts
import { useRef, useEffect, useCallback, useState } from 'react';

type Action<T = unknown> = {
  id: string;
  fn: (signal?: AbortSignal) => Promise<T>;
  optimistic?: () => void;
  rollback?: () => void;
  onSuccess?: (result: T) => void;
  onError?: (err: unknown) => void;
  retries: number;
  maxRetries?: number;
};

export function useActionQueue({ persistKey }: { persistKey?: string } = {}) {
  const queueRef = useRef<Map<string, Action>>(new Map());
  const processingRef = useRef<Set<string>>(new Set());
  const [pendingIds, setPendingIds] = useState<string[]>([]);

  // optional persistence for critical drafts
  useEffect(() => {
    if (!persistKey) return;
    try {
      const raw = localStorage.getItem(persistKey);
      if (raw) {
        const items = JSON.parse(raw) as Array<{ id: string } & Partial<Action>>;
        // only restore metadata; callers must rehydrate handlers
      }
    } catch {
      // ignore storage errors
    }
  }, [persistKey]);

  const savePersist = useCallback(() => {
    if (!persistKey) return;
    try {
      const items = Array.from(queueRef.current.values()).map(({ id }) => ({ id }));
      localStorage.setItem(persistKey, JSON.stringify(items));
    } catch {
      // ignore storage errors
    }
  }, [persistKey]);

  const enqueue = useCallback(<T,>(action: Omit<Action<T>, 'id' | 'retries'>) => {
    const id = crypto.randomUUID?.() || Math.random().toString(36).slice(2);
    const a: Action<T> = { ...action, id, retries: action.maxRetries ?? 3 };
    queueRef.current.set(id, a as Action);
    if (action.optimistic) {
      try { action.optimistic(); } catch { /* optimistic errors shouldn't block queue */ }
    }
    setPendingIds((prev) => [...prev, id]);
    savePersist();
    run(id);
    return id;
  }, [savePersist]);

  const run = useCallback(async (id: string) => {
    const action = queueRef.current.get(id);
    if (!action || processingRef.current.has(id)) return;
    processingRef.current.add(id);

    const controller = new AbortController();
    const timeout = setTimeout(() => controller.abort(), 30_000);

    try {
      const result = await action.fn(controller.signal);
      action.onSuccess?.(result);
      queueRef.current.delete(id);
      setPendingIds((prev) => prev.filter((x) => x !== id));
      savePersist();
    } catch (err) {
      // classify: 4xx -> don't retry; 5xx/network -> retry
      const status = (err as any)?.status;
      const isClientError = typeof status === 'number' && status >= 400 && status < 500;

      if (!isClientError && action.retries > 0) {
        action.retries -= 1;
        const delay = (4 - action.retries) * 1_500; // 1.5s, 3s, 4.5s
        setTimeout(() => run(id), delay);
      } else {
        action.onError?.(err);
        try { action.rollback?.(); } catch { /* ignore rollback errors */ }
        queueRef.current.delete(id);
        setPendingIds((prev) => prev.filter((x) => x !== id));
        savePersist();
      }
    } finally {
      clearTimeout(timeout);
      processingRef.current.delete(id);
    }
  }, [savePersist]);

  const isPending = useCallback((id?: string) => {
    if (!id) return false;
    return queueRef.current.has(id) || processingRef.current.has(id);
  }, []);

  const retry = useCallback((id: string) => {
    if (queueRef.current.has(id)) run(id);
  }, [run]);

  // retry all on reconnect
  useEffect(() => {
    const onOnline = () => {
      for (const id of queueRef.current.keys()) run(id);
    };
    window.addEventListener('online', onOnline);
    return () => window.removeEventListener('online', onOnline);
  }, [run]);

  return { enqueue, isPending, pendingIds, retry };
}
```

### src/hooks/useLocationValidator.ts
Lightweight GPS pre-check to avoid avoidable API failures.

```ts
export async function validateLocation(
  options: PositionOptions = { enableHighAccuracy: true, timeout: 10_000, maximumAge: 60_000 }
): Promise<{ lat: number; lng: number }> {
  return new Promise((resolve, reject) => {
    if (!navigator.geolocation) {
      reject(new Error('Geolocation not supported'));
      return;
    }
    navigator.geolocation.getCurrentPosition(
      (pos) => resolve({ lat: pos.coords.latitude, lng: pos.coords.longitude }),
      (err) => reject(err),
      options
    );
  });
}
```
