# workio / ops

## 1) Diagnosis
- No idempotency layer on `/punches` write path → repeated LINE webhook deliveries or frontend retries create duplicate punch records.
- Missing DB uniqueness constraint for “one open punch per user” → allows multiple concurrent clock-ins or overlapping open punches.
- No row-level locking or atomic upsert in punch creation → race conditions under load (two near-simultaneous clock-ins from same user).
- Punch close (clock-out) path lacks validation against already-closed punches → can double-close or overwrite.
- No operational metric/alert for duplicate punch rate or constraint violation spikes.

## 2) Proposed change
- **File scope**: `workio/server/src/db/schema.sql` (add constraint + idempotency table), `workio/server/src/controllers/punchController.ts` (add idempotency key + atomic upsert), `workio/server/src/routes/punchRoutes.ts` (ensure idempotency header/body accepted).
- **Change type**: Add partial unique index for open punches, add idempotency table with unique key, convert punch creation to atomic upsert using `ON CONFLICT DO NOTHING` + row locking for updates.

## 3) Implementation

### Step 1 — DB schema changes (`workio/server/src/db/schema.sql`)
Add idempotency table and partial unique index for open punches.

```sql
-- Idempotency keys for punch writes (supports retries & LINE webhook replays)
CREATE TABLE IF NOT EXISTS punch_idempotency (
  idempotency_key TEXT PRIMARY KEY,
  punch_id        INTEGER NOT NULL,
  created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- One open punch per user (status = 'in')
CREATE UNIQUE INDEX IF NOT EXISTS idx_one_open_punch_per_user
ON punches (user_id)
WHERE status = 'in';
```

### Step 2 — Punch controller (`workio/server/src/controllers/punchController.ts`)
Use atomic upsert and idempotency check.

```ts
import { Request, Response } from 'express';
import { pool } from '../db';

export async function createPunch(req: Request, res: Response) {
  const { userId, type = 'in', location, note } = req.body;
  const idempotencyKey = req.headers['idempotency-key'] as string || req.body.idempotencyKey;

  if (!userId) {
    return res.status(400).json({ error: 'userId is required' });
  }
  if (!idempotencyKey) {
    return res.status(400).json({ error: 'idempotency-key header or idempotencyKey body is required' });
  }

  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // Check idempotency first
    const idem = await client.query(
      'SELECT punch_id FROM punch_idempotency WHERE idempotency_key = $1',
      [idempotencyKey]
    );
    if (idem.rows.length > 0) {
      await client.query('COMMIT');
      const punch = await client.query('SELECT * FROM punches WHERE id = $1', [idem.rows[0].punch_id]);
      return res.status(200).json({ punch: punch.rows[0], replayed: true });
    }

    // Atomic upsert: try insert, if unique violation on open punch then return existing open punch
    const insertText = `
      INSERT INTO punches (user_id, type, status, location, note, created_at)
      VALUES ($1, $2, 'in', $3, $4, NOW())
      RETURNING *
    `;
    let result;
    try {
      result = await client.query(insertText, [userId, type, location, note]);
    } catch (err: any) {
      // If unique violation on open punch, fetch existing open punch
      if (err.code === '23505') {
        const existing = await client.query(
          'SELECT * FROM punches WHERE user_id = $1 AND status = $2 ORDER BY created_at DESC LIMIT 1',
          [userId, 'in']
        );
        if (existing.rows.length > 0) {
          result = existing;
        } else {
          throw err;
        }
      } else {
        throw err;
      }
    }

    const punch = result.rows[0];

    // Record idempotency
    await client.query(
      'INSERT INTO punch_idempotency (idempotency_key, punch_id) VALUES ($1, $2)',
      [idempotencyKey, punch.id]
    );

    await client.query('COMMIT');
    return res.status(201).json({ punch, replayed: false });
  } catch (err) {
    await client.query('ROLLBACK');
    console.error('createPunch error', err);
    return res.status(500).json({ error: 'Internal server error' });
  } finally {
    client.release();
  }
}

export async function closePunch(req: Request, res: Response) {
  const { userId } = req.body;
  const idempotencyKey = req.headers['idempotency-key'] as string || req.body.idempotencyKey;

  if (!userId) {
    return res.status(400).json({ error: 'userId is required' });
  }
  if (!idempotencyKey) {
    return res.status(400).json({ error: 'idempotency-key header or idempotencyKey body is required' });
  }

  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // Idempotency check
    const idem = await client.query(
      'SELECT punch_id FROM punch_idempotency WHERE idempotency_key = $1',
      [idempotencyKey]
    );
    if (idem.rows.length > 0) {
      await client.query('COMMIT');
      const punch = await client.query('SELECT * FROM punches WHERE id = $1', [idem.rows[0].punch_id]);
      return res.status(200).json({ punch: punch.rows[0], replayed: true });
    }

    // Lock open punch row for update
    const open = await client.query(
      'SELECT * FROM punches WHERE user_id = $1 AND status = $2 ORDER BY created_at DESC LIMIT 1 FOR UPDATE',
      [userId, 'in']
    );

    if (open.rows.length === 0) {
      await client.query('ROLLBACK');
      return res.status(404).json({ error: 'No open punch to close' });
    }

    const punch = open.rows[0];
    if (punch.status === 'out') {
      await client.query('ROLLBACK');
      return res.status(400).json({ error: 'Punch already closed' });
    }

    const updated = await client.query(
      'UPDATE punches SET status = $1, type = $2, updated_at = NOW() WHERE id = $3 RETURNING *',
      ['out', req.body.type || 'out', punch.id]
    );

    await client.query(
      'INSERT INTO punch_idempotency (idempotency_key, punch_id) VALUES ($1, $2)',
      [idempotencyKey, updated.rows[0].id]
    );

    await client.query('COMMIT');
    return res.status(200).json({ punch: updated.rows[0], replayed: false });
  } catch (err) {
    await client.query('ROLLBACK');
    console.error('closePunch error', err);
    return res.status(500).json({ error: 'Internal server error' });
  } finally {
    client.release();
  }
}
```

### Step 3 — Routes (`workio/server/src/routes/punchRoutes.ts`)
Accept idempotency key header/body.

```ts
import express from 'express';
import { createPunch, closePunch } from '../controllers/punchController';

const router = express.Router();

router.post('/punches', createPunch);
router.post('/punches/close', closePunch);

export default router;
```
