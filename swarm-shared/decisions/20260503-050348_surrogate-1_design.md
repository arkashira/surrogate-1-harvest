# surrogate-1 / design

## Implementation Plan (≤2h)

**Highest-value improvement**: Replace fragile shell-based ingestion with a **manifest-first, CDN-only Python worker** that eliminates HF API rate limits during training data loads and prevents mixed-schema `CastError`.

### Changes

1. Add `bin/worker.py` — deterministic shard worker that:
   - Uses a single `list_repo_tree` call (per date folder) from the runner, embeds file list in a `manifest.json`
   - Downloads via HF CDN (`resolve/main/...`) with no Authorization header → bypasses API rate limits
   - Projects each file to `{prompt, response}` only at parse time (avoids `pyarrow.CastError` on heterogeneous schemas)
   - Produces `shard-<N>-<HHMMSS>.jsonl` in `batches/public-merged/<date>/`
   - Writes lightweight attribution into filename only (no extra columns)

2. Update `bin/dataset-enrich.sh` → thin wrapper that:
   - Accepts `SHARD_ID`, `TOTAL_SHARDS`, `DATE`, `MANIFEST` (JSON file)
   - Invokes `python3 bin/worker.py` with those env vars
   - Keeps executable + Bash shebang for cron/workflow compatibility

3. Update `.github/workflows/ingest.yml` → matrix job:
   - Step 1 (single): generate `manifest.json` for today’s folder via `list_repo_tree` (only 1 API call)
   - Step 2 (16-shard matrix): pass manifest + shard id to each runner; each runs `dataset-enrich.sh`

4. Keep `lib/dedup.py` unchanged (central md5 store) — workers import and use it.

session = requests.Session()

CDN_ROOT = f"https://huggingface.co/datasets/{HF_REPO}/resolve/main"

def cdn_url(path: str) -> str:
    return f"{CDN_ROOT}/{path}"

def safe_get(url: str, retries: int = 3) -> bytes:
    for attempt in range(retries):
        resp = session.get(url, timeout=30)
        if resp.status_code == 200:
            return resp.content
        if resp.status_code == 429:
            wait = 30 * (attempt + 1)
            time.sleep(wait)
            continue
        resp.raise_for_status()
    raise RuntimeError(f"Failed to fetch {url}")

def extract_pair(obj: Dict[str, Any]) -> Dict[str, str]:
    """
    Project heterogeneous schemas to {prompt,response}.
    Accepts common variants without loading full nested schema.
    """
    prompt = obj.get("prompt") or obj.get("input") or obj.get("question") or ""
    response = obj.get("response") or obj.get("output") or obj.get("answer") or obj.get("completion") or ""
    # coerce to string
    prompt = "" if prompt is None else str(prompt).strip()
    response = "" if response is None else str(response).strip()
    return {"prompt": prompt, "response": response}

def hash_content(content: bytes) -> str:
    return hashlib.md5(content).hexdigest()

def process_parquet(content: bytes) -> Generator[Dict[str, str], None, None]:
    try:
        tbl = pq.read_table(pa.BufferReader(content))
    except Exception as e:
        # If schema mismatch across row groups/files, try column projection
        try:
            tbl = pq.read_table(pa.BufferReader(content), columns=["prompt", "response"])
        except Exception:
            # Last resort: convert to pandas and project
            import pandas as pd
            df = pd.read_parquet(pa.BufferReader(content))
            for _, row in df.iterrows():
                yield extract_pair(row.to_dict())
            return
    # Convert to list of dicts (memory-friendly for small shards in CI)
    df = tbl.to_pandas()
    for _, row in df.iterrows():
        yield extract_pair(row.to_dict())

def process_jsonl(content: bytes) -> Generator[Dict[str, str], None, None]:
    for line in content.decode("utf-8", errors="ignore").splitlines():
        line = line.strip()
        if not line:
            continue        try:
            obj = json.loads(line)
            yield extract_pair(obj)
        except Exception:
            continue

def process_file(path: str) -> int:
    """Download via CDN, dedup, return count of accepted pairs."""
    url = cdn_url(path)
    raw = safe_get(url)
    digest = hash_content(raw)
    if dedup.seen(digest):
        return 0

    ext = Path(path).suffix.lower()
    pairs = []
    if ext == ".parquet":
        pairs = list(process_parquet(raw))
    elif ext == ".jsonl":
        pairs = list(process_jsonl(raw))
    else:
        # Skip unsupported
        return 0

    accepted = 0
    for p in pairs:
        if not p["prompt"] or not p["response"]:
            continue
        # Optional: lightweight content dedup by normalized text hash
        norm = hashlib.md5((p["prompt"] + "\n" + p["response"]).encode()).hexdigest()
        if dedup.seen(norm):
            continue
        dedup.add(norm)
        accepted += 1
        yield p

    if accepted > 0:
        dedup.add(digest)
    return accepted
