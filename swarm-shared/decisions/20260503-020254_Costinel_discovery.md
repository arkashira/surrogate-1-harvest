# Costinel / discovery

## Final Implementation Plan — Top-Hub Signal Panel (Costinel Dashboard)

**Scope & Value**  
- Frontend-only, read-only React panel that surfaces the highest-signal / most-connected hub (default “MOC”) and its actionable cost-saving proposals from the knowledge graph.  
- CDN-first data delivery to eliminate HF API rate limits and ensure fast, reliable dashboard loads.  
- ~90 min implementation + 30 min smoke test = <2 h total.

## 3) Hook: CDN-first fetch with stale-while-revalidate (`src/hooks/useTopHubSignals.ts`)

```ts
import { useEffect, useState, useCallback } from 'react';
import { TopHubPayload } from '../types/knowledgeGraph';

const CDN_URL = '/data/knowledge-graph/top-hub-moc.json';
const STALE_TIME = 5 * 60 * 1000; // 5m

export function useTopHubSignals() {
  const [data, setData] = useState<TopHubPayload | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  const fetchFromCDN = useCallback(async (): Promise<TopHubPayload> => {
    const res = await fetch(CDN_URL, { cache: 'no-store' });
    if (!res.ok) throw new Error(`CDN fetch failed: ${res.status}`);
    return res.json();
  }, []);

  const load = useCallback(async () => {
    try {
      setLoading(true);
      const payload = await fetchFromCDN();
      setData(payload);
      setError(null);
    } catch (err) {
      setError(err instanceof Error ? err.message : 'Unknown error');
    } finally {
      setLoading(false);
    }
  }, [fetchFromCDN]);

  useEffect(() => {
    load();
    const id = setInterval(load, STALE_TIME);
    return () => clearInterval(id);
  }, [load]);

  return { data, loading, error, refetch: load };
}
```

## 4) Component: `src/components/TopHubSignalPanel/TopHubSignalPanel.tsx`

```tsx
import React, { useState } from 'react';
import { useTopHubSignals } from '../../hooks/useTopHubSignals';
import { TopHubPayload } from '../../types/knowledgeGraph';
import styles from './TopHubSignalPanel.module.css';

const impactColor = (v: TopHubPayload['proposals'][0]['impact']) =>
  ({ high: '#ef4444', medium: '#f59e0b', low: '#10b981' }[v] || '#6b7280');

export const TopHubSignalPanel: React.FC = () => {
  const { data, loading, error, refetch } = useTopHubSignals();
  const [expandedProposalId, setExpandedProposalId] = useState<string | null>(null);

  if (loading) return <div className={styles.panel}>Loading signals…</div>;
  if (error) return <div className={styles.panel}>Unable to load signals.</div>;
  if (!data) return null;

  return (
    <div className={styles.panel}>
      <header className={styles.header}>
        <h3 className={styles.title}>Top Hub: {data.hub.label}</h3>
        <span className={styles.badge}>{data.hub.tags.join(', ')}</span>
      </header>
      <p className={styles.desc}>{data.hub.description}</p>

      <section className={styles.proposals}>
        {data.proposals.map((p) => {
          const expanded = expandedProposalId === p.id;
          return (
            <article key={p.id} className={styles.card}>
              <div className={styles.cardHeader}>
                <span className={styles.impact} style={{ backgroundColor: impactColor(p.impact) }}>
                  {p.impact}
                </span>
                <span className={styles.savings}>
                  {p.savings.currency} {p.savings.amount.toLocaleString()} / {p.savings.period}
                </span>
              </div>
              <h4 className={styles.cardTitle}>{p.title}</h4>
              <p className={styles.summary}>{p.summary}</p>

              {expanded && (
                <>
                  <ul className={styles.actions}>
                    {p.actions.map((a, i) => (
                      <li key={i}>{a}</li>
                    ))}
                  </ul>
                  {p.references.length > 0 && (
                    <div className={styles.refs}>
                      {p.references.map((r) => (
                        <a key={r.href} href={r.href} target="_blank" rel="noopener noreferrer">
                          {r.title}
                        </a>
                      ))}
                    </div>
                  )}
                </>
              )}

              <div className={styles.cardFooter}>
                <button
                  className={styles.viewMore}
                  onClick={() =>
                    setExpandedProposalId(expanded ? null : p.id)
                  }
                >
                  {expanded ? 'View less' : 'View more'}
                </button>
              </div>
            </article>
          );
        })}
      </section>

      <footer className={styles.footer}>
        <small>Updated {new Date(data.generatedAt).toLocaleString()}</small>
        <button className={styles.refresh} onClick={() => refetch()}>
          Refresh
        </button>
      </footer>
    </div>
  );
};
```
