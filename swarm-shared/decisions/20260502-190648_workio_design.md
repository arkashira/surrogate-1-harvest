# workio / design

## 1. Diagnosis

- **No tenant cache on LINE webhook** — every webhook hits PostgreSQL to resolve `channelId → tenant`; slow under burst traffic, noisy under load, and fails fast if DB is cold.
- **Missing health/readiness endpoints** — no `/health` or `/ready` for k8s/PM2/liveness checks; deployments risk traffic to unhealthy pods.
- **No graceful shutdown** — in-flight LINE webhook requests can be cut mid-processing during deploy/restart; can lose clock-in/out events.
- **Hardcoded tenant resolution in route handlers** — repeated `SELECT tenant_id FROM channels WHERE channel_id=$1` scattered across webhook and API routes; inconsistent and hard to instrument.
- **No request ID / traceability** — LINE events and API calls lack correlation IDs; debugging multi-step flows (clock-in → notify → dashboard) is noisy.

## 2. Proposed change

Add a lightweight in-memory tenant cache with TTL + stale-while-revalidate, plus `/health` and `/ready` endpoints, and wire a request-id middleware for traceability. Scope:

- `server/src/middleware/tenant-cache.ts` (new)
- `server/src/middleware/request-id.ts` (new)
- `server/src/server.ts` (add middleware and health routes)
- `server/src/routes/webhook/line.ts` (use tenant cache)
- `server/src/routes/api/tenants.ts` (optional: expose cache metrics)

## 3. Implementation

### 3.1 Tenant cache (TTL 5m, stale-while-revalidate 60m)

```ts
// server/src/middleware/tenant-cache.ts
import { Pool } from 'pg';

type TenantEntry = {
  tenantId: string;
  channelId: string;
  cachedAt: number;
  value: string; // tenantId
};

const TTL_MS = 5 * 60 * 1000;       // 5m fresh
const STALE_TTL_MS = 60 * 60 * 1000; // 1h stale allowed

export class TenantCache {
  private cache = new Map<string, TenantEntry>();
  private pool: Pool;

  constructor(pool: Pool) {
    this.pool = pool;
  }

  async resolve(channelId: string): Promise<string | null> {
    const now = Date.now();
    const existing = this.cache.get(channelId);

    // Fresh hit
    if (existing && now - existing.cachedAt < TTL_MS) {
      return existing.value;
    }

    // Stale hit: return stale while revalidating async
    if (existing && now - existing.cachedAt < STALE_TTL_MS) {
      this._refresh(channelId).catch(() => {});
      return existing.value;
    }

    // Miss / expired stale: fetch sync
    return this._refresh(channelId);
  }

  private async _refresh(channelId: string): Promise<string | null> {
    try {
      const res = await this.pool.query(
        `SELECT tenant_id FROM channels WHERE channel_id = $1 LIMIT 1`,
        [channelId]
      );
      const tenantId = res.rows[0]?.tenant_id ?? null;
      const entry: TenantEntry = {
        tenantId: channelId,
        channelId,
        cachedAt: Date.now(),
        value: tenantId,
      };
      if (tenantId) {
        this.cache.set(channelId, entry);
      } else {
        this.cache.delete(channelId);
      }
      return tenantId;
    } catch {
      // On DB error, keep stale if exists; otherwise fail closed (null)
      const stale = this.cache.get(channelId);
      if (stale && Date.now() - stale.cachedAt < STALE_TTL_MS) return stale.value;
      return null;
    }
  }

  invalidate(channelId: string) {
    this.cache.delete(channelId);
  }

  clear() {
    this.cache.clear();
  }

  metrics() {
    return { size: this.cache.size };
  }
}
```

### 3.2 Request ID middleware

```ts
// server/src/middleware/request-id.ts
import { Request, Response, NextFunction } from 'express';

export function requestId(req: Request, _res: Response, next: NextFunction) {
  const id = req.headers['x-request-id'] || crypto.randomUUID();
  ;(req as any).requestId = id;
  next();
}
```

### 3.3 Health / ready endpoints + graceful shutdown

```ts
// server/src/server.ts  (excerpt)
import express from 'express';
import { Pool } from 'pg';
import { requestId } from './middleware/request-id';
import { TenantCache } from './middlerate/tenant-cache';
import lineWebhook from './routes/webhook/line';

const app = express();
const pool = new Pool({ connectionString: process.env.DATABASE_URL });
const tenantCache = new TenantCache(pool);

app.use(express.json());
app.use(requestId);

// Attach tenantCache to app locals for routes
app.locals.tenantCache = tenantCache;

// Health & readiness
app.get('/health', (_req, res) => res.json({ status: 'ok', uptime: process.uptime() }));
app.get('/ready', async (_req, res) => {
  try {
    await pool.query('SELECT 1');
    res.json({ status: 'ready' });
  } catch (e) {
    res.status(503).json({ status: 'not ready', db: 'down' });
  }
});

// Routes
app.use('/webhook/line', lineWebhook);

const server = app.listen(process.env.PORT || 3000, () => {
  console.log('Workio API running');
});

// Graceful shutdown
let shuttingDown = false;
async function shutdown() {
  if (shuttingDown) return;
  shuttingDown = true;
  console.log('Shutting down gracefully...');
  server.close(async () => {
    await pool.end();
    tenantCache.clear();
    process.exit(0);
  });
  // Force exit after 10s
  setTimeout(() => process.exit(1), 10_000).unref();
}
process.on('SIGTERM', shutdown);
process.on('SIGINT', shutdown);
```

### 3.4 Use tenant cache in LINE webhook

```ts
// server/src/routes/webhook/line.ts
import { Router } from 'express';

const router = Router();

router.post('/', async (req, res) => {
  const tenantCache = req.app.locals.tenantCache;
  const events = req.body.events || [];

  // Fast fail if no events
  if (!events.length) return res.status(400).end();

  // Assume channelId is in source or can be derived from headers (LINE channel)
  // Example: extract from webhook event source.channelId or use configured channel mapping
  const channelId = events[0]?.source?.channelId || (req as any).headers?.['x-line-channel-id'];
  if (!channelId) return res.status(400).json({ error: 'missing_channel' });

  const tenantId = await tenantCache.resolve(channelId);
  if (!tenantId) {
    console.warn(`[${(req as any).requestId}] Unknown channel ${channelId}`);
    return res.status(404).json({ error: 'tenant_not_found' });
  }

  // Attach tenantId for downstream handlers
  (req as any).tenantId = tenantId;

  // Process events (clock-in/out, leave, etc.)
  for (const ev of events) {
    // ... existing logic using tenantId
  }

  res.status(200).end();
});

export default router;
```
