# Costinel / design

## Final Implementation Plan — Top-Hub Signal Panel (Costinel Dashboard)

**Scope & Value**  
- Frontend-only, read-only widget that surfaces the most-connected hub (default **MOC**) and its actionable proposals from the knowledge graph.  
- Ships in **<2h** as a resilient, self-contained component that can be dropped into the existing dashboard layout.  
- Zero backend changes; uses static graph snapshots served via CDN (HF-compatible, no auth headers at runtime).

---

### 1. Design Decisions (resolved)

- **Hub source**: Default to `MOC`; fallback to first item in `hubs.json` if missing.  
- **Data shape**:  
  ```json
  {
    "hubKey": "MOC",
    "hubTitle": "MOC — Multi-Org Cost governance",
    "connectionCount": 42,
    "updatedAt": "2026-05-03T08:12:00Z",
    "proposals": [
      {
        "id": "MOC-001",
        "title": "Centralize tag enforcement policy",
        "signal": "Tag sprawl increasing cost allocation errors by ~12% QoQ",
        "impact": "high",
        "effort": "medium",
        "owner": "cloud-governance@axentx.com",
        "due": "2026-06-15",
        "tags": ["governance", "tags", "cost"]
      }
    ]
  }
  ```
- **Delivery**: Static JSON at build time or runtime fetch from CDN (`/data/knowledge-graph/2026-05-03/top-hub-MOC.json`). No Authorization header.  
- **UI**: Card list with signal text, impact/effort badges, tag filters, and a “View in Knowledge Graph” link. No write operations.  
- **Error resilience**: If graph data fails to load, show empty state with “No signals” and a non-blocking toast/console warning. Do not crash.

### 2. File Changes (concrete)

#### `src/components/TopHubSignalPanel.tsx`
```tsx
import { useEffect, useState } from 'react';
import './TopHubSignalPanel.css';

interface Proposal {
  id: string;
  title: string;
  signal: string;
  impact: 'high' | 'medium' | 'low';
  effort: 'low' | 'medium' | 'high';
  owner?: string;
  due?: string;
  tags: string[];
}

interface HubData {
  hubKey: string;
  hubTitle: string;
  connectionCount: number;
  updatedAt: string;
  proposals: Proposal[];
}

const DEFAULT_HUB: HubData = {
  hubKey: 'MOC',
  hubTitle: 'MOC — Multi-Org Cost governance',
  connectionCount: 0,
  updatedAt: new Date().toISOString(),
  proposals: []
};

const CDN_PATH = '/data/knowledge-graph/2026-05-03/top-hub-MOC.json';

export function TopHubSignalPanel() {
  const [data, setData] = useState<HubData>(DEFAULT_HUB);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);
  const [filter, setFilter] = useState<string>('all');

  useEffect(() => {
    fetch(CDN_PATH, { method: 'GET', cache: 'no-cache' })
      .then((res) => {
        if (!res.ok) throw new Error(`Failed to load hub data: ${res.status}`);
        return res.json() as Promise<HubData>;
      })
      .then((json) => {
        setData(json);
        setError(null);
      })
      .catch((err) => {
        console.warn(err);
        setError(err.message);
        setData(DEFAULT_HUB);
      })
      .finally(() => setLoading(false));
  }, []);

  const proposals = data.proposals.filter((p) =>
    filter === 'all' ? true : p.tags.includes(filter)
  );

  const allTags = Array.from(new Set(data.proposals.flatMap((p) => p.tags)));

  if (loading) {
    return (
      <div className="top-hub-panel loading">
        <div className="panel-header">
          <h3>{data.hubTitle}</h3>
          <span className="badge">Loading signals…</span>
        </div>
        <div className="panel-body skeleton-list" />
      </div>
    );
  }

  return (
    <div className="top-hub-panel">
      <div className="panel-header">
        <div>
          <h3>{data.hubTitle}</h3>
          <p className="subtitle">{data.connectionCount} connections</p>
        </div>
        <div className="header-meta">
          <span className="badge">{data.proposals.length} signals</span>
          <a
            href={`/knowledge-graph?hub=${data.hubKey}`}
            className="view-link"
            target="_blank"
            rel="noopener noreferrer"
          >
            View in Knowledge Graph
          </a>
        </div>
      </div>

      {error && <div className="panel-note error">⚠️ {error}</div>}

      <div className="tag-filters">
        <button
          className={filter === 'all' ? 'active' : ''}
          onClick={() => setFilter('all')}
        >
          All
        </button>
        {allTags.map((t) => (
          <button
            key={t}
            className={filter === t ? 'active' : ''}
            onClick={() => setFilter(t)}
          >
            {t}
          </button>
        ))}
      </div>

      <div className="panel-body">
        {proposals.length === 0 ? (
          <div className="empty-state">No actionable signals for this hub.</div>
        ) : (
          <ul className="proposal-list">
            {proposals.map((p) => (
              <li key={p.id} className="proposal-card">
                <div className="proposal-title">{p.title}</div>
                <div className="proposal-signal">{p.signal}</div>
                <div className="proposal-meta">
                  <span className={`impact ${p.impact}`}>{p.impact}</span>
                  <span className={`effort ${p.effort}`}>{p.effort}</span>
                  {p.owner && <span className="owner">{p.owner}</span>}
                  {p.due && <span className="due">Due {p.due}</span>}
                </div>
                <div className="proposal-tags">
                  {p.tags.map((t) => (
                    <span key={t} className="tag">
                      {t}
                    </span>
                  ))}
                </div>
              </li>
            ))}
          </ul>
        )}
      </div>
    </div>
  );
}
```
