- Single function: `list_date_folder_via_api()` (Mac orchestration only) → writes `file_list.json`
- Training script uses only CDN URLs (`resolve/main/...`) with `requests`/`aiohttp` + `pyarrow` projection to `{prompt, response}`

### 2) Lightning Studio reuse guard

- If stopped, restart with `target.start(machine=Machine.L40S)` (avoid idle-timeout deaths)

### 3) Wire into surrogate training endpoint
- Expose `/training/start` that:
  1. Ensures file list exists (or creates it once per date folder)
  2. Reuses/creates Studio
  3. Runs training script with CDN-only file list embedded

### 4) Remove HF API calls from training hot path
- No `list_repo_files`, no `load_dataset(streaming=True)` on mixed-schema repos
- Attribution via filename pattern: `batches/mirror-merged/{date}/{slug}.parquet`

# surrogate/backend/cdn_dataset.py
import json
import os
from pathlib import Path
from typing import List, Dict
import requests
import pyarrow.parquet as pq
from io import BytesIO

HF_DATASETS_REPO = "datasets/your-org/your-repo"  # configurable
CDN_BASE = f"https://huggingface.co/datasets/{HF_DATASETS_REPO}/resolve/main"

def list_date_folder_via_api(date_folder: str, token: str) -> List[str]:
    """
    One-off Mac orchestration call (outside training) to list files for a date folder.
    Save result to file_list.json to embed in training.
    Uses HF API minimally (once per date folder) to avoid 429s.
    """
    from huggingface_hub import HfApi
    api = HfApi(token=token)
    # Non-recursive to avoid pagination explosion
    tree = api.list_repo_tree(repo_id=HF_DATASETS_REPO, path=date_folder, recursive=False)
    files = [item.rfilename for item in tree if item.type == "file"]
    out_path = Path("file_list.json")
    out_path.write_text(json.dumps({"date_folder": date_folder, "files": files}, indent=2))
    return files

def load_parquet_from_cdn(cdn_path: str, columns=("prompt", "response")) -> pq.Table:
    """
    Download single parquet via CDN (no auth/rate-limit) and project columns.
    Avoids pyarrow CastError from mixed-schema files by selecting only known columns.
    """
    resp = requests.get(cdn_path, timeout=30)
    resp.raise_for_status()
    buf = BytesIO(resp.content)
    # Read only required columns; ignore extra/mismatched fields
    try:
        table = pq.read_table(buf, columns=columns)
    except (ValueError, KeyError, OSError):
        # Fallback: read metadata and intersect available columns
        pf = pq.ParquetFile(buf)
        available = set(pf.schema.names)
        use_cols = [c for c in columns if c in available]
        if not use_cols:
            raise ValueError(f"No usable columns in {cdn_path}. Available: {available}")
        table = pq.read_table(buf, columns=use_cols)
    return table

def stream_cdn_dataset(file_list: List[str], batch_size: int = 512):
    """
    Generator yielding {prompt, response} rows from CDN-hosted parquet files.
    Zero HF API calls during training.
    """
    batch = []
    for rel in file_list:
        cdn_url = f"{CDN_BASE}/{rel}"
        try:
            table = load_parquet_from_cdn(cdn_url, columns=("prompt", "response"))
        except Exception as exc:
            # Skip malformed files; log and continue
            print(f"Skipping {rel}: {exc}")
            continue
        df = table.to_pandas()
        for _, row in df.iterrows():
            batch.append({"prompt": row["prompt"], "response": row["response"]})
            if len(batch) >= batch_size:
                yield batch
                batch = []
    if batch:
        yield batch
```

# surrogate/backend/lightning_studio.py
import time
from lightning_sdk import Teamspace, Studio, Machine

TEAMSPACE = "your-teamspace"
STUDIO_NAME = "surrogate-training"
MACHINE = Machine.L40S  # fallback; H200 requires lightning-lambda-prod

def get_or_create_study() -> Studio:
    ts = Teamspace(TEAMSPACE)
    # Reuse running studio to save quota
    for s in ts.studios:
        if s.name == STUDIO_NAME and s.status == "Running":
            print(f"Reusing running studio: {s.id}")
            return s
    # Create if none running
    studio = Studio.create(
        teamspace=TEAMSPACE,
        name=STUDIO_NAME,
        machine=MACHINE,
        create_ok=True,
    )
    print(f"Created studio: {studio.id}")
    return studio

def ensure_running(studio: Studio, timeout: int = 300) -> Studio:
    """Ensure studio is running; restart if stopped (idle-timeout safe)."""
    start = time.time()
    while time.time() - start < timeout:
        studio.refresh()
        if studio.status == "Running":
            return studio
        if studio.status == "Stopped":
            print("Studio stopped; restarting...")
            studio.start(machine=MACHINE)
        time.sleep(10)
    raise RuntimeError("Studio failed to reach Running state")
```

    """
    Start training for a date folder using CDN-bypass dataset loader.
    """
    # 1) Ensure file list exists (idempotent; run once per date folder)
    list_path = Path("file_list.json")
    if not list_path.exists():
        # Requires HF token from env; this runs rarely (orchestration step)
        token = os.getenv("HF_TOKEN")
        files = list_date_folder_via_api(date_folder, token)
