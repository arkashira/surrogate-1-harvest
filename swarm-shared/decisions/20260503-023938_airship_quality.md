# tools/build_cdn_filelist.py
import json
import os
from huggingface_hub import HfApi

def build_filelist(repo_id: str, date_folder: str, out_path: str):
    api = HfApi()
    tree = api.list_repo_tree(repo_id=repo_id, path=date_folder, recursive=False)
    files = [
        f.rfilename for f in tree
        if f.rfilename.endswith((".jsonl", ".parquet", ".json"))
    ]
    manifest = {
        "repo_id": repo_id,
        "date_folder": date_folder,
        "files": files,
        "cdn_prefix": f"https://huggingface.co/datasets/{repo_id}/resolve/main/{date_folder}"
    }
    os.makedirs(os.path.dirname(out_path), exist_ok=True)
    with open(out_path, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Wrote {len(files)} files to {out_path}")

if __name__ == "__main__":
    build_filelist(
        repo_id="axentx/surrogate-mirror",
        date_folder="2026-04-29",
        out_path="training/filelist-2026-04-29.json"
    )

### 2) CDN-only dataset loader (robust)
```python
# training/cdn_dataset.py
import json
import logging
import requests
from torch.utils.data import IterableDataset

logger = logging.getLogger(__name__)

class CDNTextDataset(IterableDataset):
    def __init__(self, manifest_path: str, max_retries: int = 3, timeout: int = 30):
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.prefix = self.manifest["cdn_prefix"]
        self.files = self.manifest["files"]
        self.max_retries = max_retries
        self.timeout = timeout

    def _fetch_file(self, filename: str):
        url = f"{self.prefix}/{filename}"
        for attempt in range(1, self.max_retries + 1):
            try:
                resp = requests.get(url, timeout=self.timeout)
                resp.raise_for_status()
                return resp.content
            except Exception as exc:
                logger.warning(
                    f"Fetch {url} failed (attempt {attempt}/{self.max_retries}): {exc}"
                )
                if attempt == self.max_retries:
                    raise
        return None

    def _parse_to_record(self, content: bytes, filename: str):
        try:
            if filename.endswith(".jsonl"):
                lines = [ln.strip() for ln in content.decode("utf-8").splitlines() if ln.strip()]
                for ln in lines:
                    obj = json.loads(ln)
                    yield {"prompt": obj.get("prompt", ""), "response": obj.get("response", "")}
            else:
                obj = json.loads(content.decode("utf-8"))
                if isinstance(obj, list):
                    for item in obj:
                        yield {"prompt": item.get("prompt", ""), "response": item.get("response", "")}
                else:
                    yield {"prompt": obj.get("prompt", ""), "response": obj.get("response", "")}
        except Exception as exc:
            logger.error(f"Parse error in {filename}: {exc}")

    def __iter__(self):
        for filename in self.files:
            try:
                content = self._fetch_file(filename)
                if content is None:
                    continue
                yield from self._parse_to_record(content, filename)
            except Exception as exc:
                logger.error(f"Skipping {filename}: {exc}")
                continue
```

import lightning

def get_or_create_studio(name: str, machine=None):
    teamspace = lightning.Teamspace()
    for studio in teamspace.studios:
        if studio.name == name and studio.status == "Running":
            return studio
    for studio in teamspace.studios:
        if studio.name == name and studio.status == "Stopped":
            studio.start(machine=machine or lightning.Machine.L40S)
            return studio
    return teamspace.create_studio(name)
