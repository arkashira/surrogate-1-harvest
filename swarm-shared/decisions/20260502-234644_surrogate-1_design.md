# surrogate-1 / design

## Implementation Plan (≤2h)

**Highest-value improvement**: Add a pre-flight snapshot generator (`bin/snapshot.sh`) that lists all dataset files once, embeds the file manifest into training, and enables CDN-only downloads during parallel shard processing. This eliminates HF API rate-limit pressure during ingestion and guarantees stable training data availability.

### Why this now
- Removes the 429/1000 req/5min HF API bottleneck during parallel shard runs
- Enables Lightning training to use CDN-only fetches (zero API calls during data load)
- Fits cleanly into existing 16-shard matrix workflow with no behavioral changes to output
- <2h implementation: single script + small Python helper + workflow integration

## Concrete Implementation

### 1. `bin/snapshot.sh` — pre-flight manifest generator
```bash
#!/usr/bin/env bash
# bin/snapshot.sh
# Lists public dataset files for a date folder and emits file manifest JSON.
# Usage: bin/snapshot.sh <date-folder> [output.json]
# Example: bin/snapshot.sh 2026-05-02 batches/public-merged/2026-05-02/files.json

set -euo pipefail

REPO="axentx/surrogate-1-training-pairs"
DATE_FOLDER="${1:-}"
OUTFILE="${2:-}"

if [[ -z "$DATE_FOLDER" ]]; then
  echo "Usage: $0 <date-folder> [output.json]" >&2
  exit 1
fi

if [[ -z "$OUTFILE" ]]; then
  OUTFILE="batches/public-merged/${DATE_FOLDER}/files.json"
fi

mkdir -p "$(dirname "$OUTFILE")"

# Single API call: non-recursive tree for the date folder
# Avoids list_repo_files recursive pagination and rate limits
python3 - "$REPO" "$DATE_FOLDER" "$OUTFILE" <<'PY'
import json
import os
import sys
from huggingface_hub import list_repo_tree

repo_id = sys.argv[1]
path = sys.argv[2]
outfile = sys.argv[3]

# Non-recursive: one API call for the folder
tree = list_repo_tree(repo_id=repo_id, path=path, recursive=False)
files = [item.rfilename for item in tree if item.type == "file"]

# Stable ordering for deterministic sharding
files.sort()

manifest = {
    "repo": repo_id,
    "folder": path,
    "files": files,
    "count": len(files)
}

os.makedirs(os.path.dirname(outfile), exist_ok=True)
with open(outfile, "w") as f:
    json.dump(manifest, f, indent=2)

print(f"Snapshot: {len(files)} files -> {outfile}")
PY

echo "Snapshot written to $OUTFILE"
```

Make executable:
```bash
chmod +x bin/snapshot.sh
```

### 2. `bin/lib/manifest.py` — lightweight helper to read manifest and produce shard slices
```python
# bin/lib/manifest.py
import json
import hashlib
import os
import sys
from typing import List

def read_manifest(path: str) -> dict:
    with open(path) as f:
        return json.load(f)

def shard_files(files: List[str], shard_id: int, total_shards: int = 16) -> List[str]:
    """Deterministic shard assignment by filename hash."""
    shard_files = []
    for f in files:
        # Stable hash -> bucket
        h = int(hashlib.md5(f.encode()).hexdigest(), 16)
        if (h % total_shards) == shard_id:
            shard_files.append(f)
    return shard_files

def cdn_url(repo: str, filepath: str) -> str:
    """CDN bypass URL (no Authorization header)."""
    return f"https://huggingface.co/datasets/{repo}/resolve/main/{filepath}"

if __name__ == "__main__":
    # CLI helper: prints CDN URLs for this shard's files
    manifest_path = sys.argv[1]
    shard_id = int(sys.argv[2])
    total_shards = int(sys.argv[3]) if len(sys.argv) > 3 else 16

    manifest = read_manifest(manifest_path)
    files = shard_files(manifest["files"], shard_id, total_shards)
    for f in files:
        print(cdn_url(manifest["repo"], f))
```

Create `bin/lib/fetch_shard.py` (minimal):
```python
#!/usr/bin/env python3
# bin/lib/fetch_shard.py
import json
import hashlib
import subprocess
import sys
import tempfile
import pyarrow.parquet as pq
import pyarrow as pa
import os

def shard_files(files, shard_id, total_shards):
    out = []
    for f in files:
        h = int(hashlib.md5(f.encode()).hexdigest(), 16)
        if (h % total_shards) == shard_id:
            out.append(f)
    return out

def main():
    manifest_path = sys.argv[1]
    shard_id = int(sys.argv[2])
    total_shards = int(sys.argv[3]) if len(sys.argv) > 3 else 16

    with open(manifest_path) as f:
        manifest = json.load(f)

    files = shard_files(manifest["files"], shard_id, total_shards)
    repo = manifest["repo"]

    for fn in files:
        url = f"https://huggingface.co/datasets/{repo}/resolve/main/{fn}"
        # Stream download to temp file
        with tempfile.NamedTemporaryFile(suffix=".parquet", delete=False) as tmp:
            tmp_path = tmp.name
        try:
            subprocess.run(["curl", "-L", "-o", tmp_path, url], check=True)
            # Project to {prompt, response} only at parse time
            # (handles mixed schemas safely)
            table = pq.read_table(tmp_path, columns=["prompt", "response"])
            # Emit normalized JSONL lines to stdout
            for batch in table.to_batches():
                df = batch.to_pandas()
                for _, row in df.iterrows():
                    print(json.dumps({"prompt": row["prompt"], "response": row["response"]}))
        finally:
            if os.path.exists(tmp_path):
                os.unlink(tmp_path)

if __name__ == "__main__":
    main()
```

Make helpers executable:
```bash
chmod +x bin/lib/fetch_shard.py bin/lib/manifest.py
```
