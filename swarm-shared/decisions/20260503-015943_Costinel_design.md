# Costinel / design

## Final Implementation Plan — Top-Hub Signal Panel (Costinel Dashboard)

**Scope & Value**  
- Frontend-only, read-only React panel that surfaces the most-connected hub (default “MOC”) and its actionable proposals from the knowledge graph.  
- CDN-first data delivery to eliminate API rate limits and ensure fast dashboard loads.  
- Ships in <2h: one new component + one JSON contract + route mount.

### 3) Component — `src/components/TopHubSignalPanel/TopHubSignalPanel.tsx`

Key decisions:
- Use `public/` path for CDN-friendly fetch in prod (`/data/...`).  
- Show severity, confidence, coverage, and primary actions.  
- Provide clear empty/error states and a refresh affordance.  
- Keep accessibility and keyboard-friendly action links.

```tsx
import React, { useEffect, useState, useCallback } from "react";
import styles from "./TopHubSignalPanel.module.css";

type Action = { label: string; href: string };
type Proposal = {
  id: string;
  title: string;
  type: string;
  service: string;
  region: string;
  severity: "low" | "medium" | "high";
  currentSpendUSD: number;
  estimatedSavingsUSD: number;
  confidence: number;
  coveragePct: number;
  window: string;
  tags: string[];
  actions: Action[];
};

type TopHubData = {
  hub: string;
  title: string;
  description: string;
  lastUpdated: string;
  metrics: {
    connectedProposals: number;
    estimatedMonthlySavingsUSD: number;
    coveragePercent: number;
    anomaliesOpen: number;
  };
  proposals: Proposal[];
};

const DATA_URL =
  process.env.NODE_ENV === "production"
    ? "/data/knowledge-graph/top-hub-MOC.json"
    : "/data/knowledge-graph/top-hub-MOC.json";

const fmtUSD = (v: number) =>
  new Intl.NumberFormat("en-US", {
    style: "currency",
    currency: "USD",
    maximumFractionDigits: 0,
  }).format(v);

const severityClass = (s: Proposal["severity"]) => ({
  high: styles.severityHigh,
  medium: styles.severityMedium,
  low: styles.severityLow,
}[s]);

  const [data, setData] = useState<TopHubData | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  const load = useCallback(async () => {
    setLoading(true);
    setError(null);
    try {
      const res = await fetch(DATA_URL, { cache: "no-cache" });
      if (!res.ok) throw new Error(`Failed to load top-hub data (${res.status})`);
      const json = (await res.json()) as TopHubData;
      setData(json);
    } catch (err: any) {
      console.error(err);
      setError(err.message || "Unable to load signals.");
    } finally {
      setLoading(false);
    }
  }, []);

  useEffect(() => {
    load();
  }, [load]);

  if (loading) {
    return <div className={styles.panel}>Loading signals…</div>;
  }

  if (error || !data) {
    return (
      <div className={styles.panel}>
        <p className={styles.error}>{error || "Unable to load signals."}</p>
        <button className={styles.retryBtn} onClick={load}>
          Retry
        </button>
      </div>
    );
