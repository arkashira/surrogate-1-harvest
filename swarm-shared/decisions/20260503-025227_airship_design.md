### **1) Pre-cache CDN Manifest (Mac orchestrator)**  
Run once per dataset snapshot (or after rate-limit window clears).  

**File**: `surrogate/training/build_manifest.py`  
```python
#!/usr/bin/env python3
"""
Generate CDN manifest for a HuggingFace dataset folder.
Usage:
    python build_manifest.py --repo <repo> --path <folder> --out manifest.json
"""
import argparse
import json
from huggingface_hub import HfApi

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g., org/dataset)")
    parser.add_argument("--path", required=True, help="Folder path in repo")
    parser.add_argument("--out", default="manifest.json", help="Output JSON file")
    args = parser.parse_args()

    api = HfApi()
    entries = api.list_repo_tree(repo_id=args.repo, path=args.path, recursive=False)

    files = []
    for entry in entries:
        if entry.type == "file":
            cdn_url = f"https://huggingface.co/datasets/{args.repo}/resolve/main/{entry.path}"
            files.append({
                "path": entry.path,
                "cdn_url": cdn_url,
                "size": getattr(entry, "size", None)
            })

    manifest = {
        "repo": args.repo,
        "folder": args.path,
        "files": files
    }

    with open(args.out, "w") as f:
        json.dump(manifest, f, indent=2)

    print(f"Wrote {len(files)} files to {args.out}")

if __name__ == "__main__":
    main()
```

**Setup**:  
```bash
chmod +x surrogate/training/build_manifest.py
```

**Action**:  
- Run before training:  
  ```bash
  python surrogate/training/build_manifest.py --repo org/dataset --path folder --out manifest.json
  ```
- Embed `manifest.json` into the training container or pass its path to `train.py`.

### **2) Training Script: CDN Fetches + Studio Reuse + Idle Resilience**  
**File**: `surrogate/training/train.py`  

#### **Key Changes**  
1. **CDN-only dataset loading** (no HF API during training).  
2. **Lightning Studio reuse** (avoid recreation).  
3. **Idle timeout handling** (auto-restart stopped studios).  

#### **Code**  
```python
#!/usr/bin/env python3
"""
Surrogate training with CDN-only dataset fetches and Lightning Studio reuse.
"""
import json
import os
import time
from pathlib import Path
from typing import List, Dict

import lightning as L
import torch
from torch.utils.data import IterableDataset, DataLoader
from transformers import AutoTokenizer, AutoModelForCausalLM

# ---- CDN Dataset (no HF API during training) ----
class CDNParquetDataset(IterableDataset):
    def __init__(self, manifest_path: str, columns=("prompt", "response")):
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.columns = columns

    def _stream_parquet_from_cdn(self, cdn_url: str):
        # Stream parquet via CDN URL (use pyarrow or direct download)
        import pyarrow.parquet as pq
        import requests
        resp = requests.get(cdn_url, stream=True)
        resp.raise_for_status()
        with open("temp.parquet", "wb") as f:
            for chunk in resp.iter_content(chunk_size=8192):
                f.write(chunk)
        table = pq.read_table("temp.parquet", columns=self.columns)
        os.remove("temp.parquet")
        return table

    def __iter__(self):
        for file_info in self.manifest["files"]:
            if not file_info["path"].endswith(".parquet"):
                continue
            table = self._stream_parquet_from_cdn(file_info["cdn_url"])
            for batch in table.to_batches():
                cols = {c: batch.column(c).to_pylist() for c in self.columns}
                for i in range(len(cols[self.columns[0]])):
                    yield {c: cols[c][i] for c in self.columns}

# ---- Lightning Studio Reuse + Idle Resilience ----
def get_or_create_studio(studio_name: str, machine: str = "L40S"):
    client = L.LightningClient()
    studios = client.list_studios()
    for studio in studios:
        if studio.name == studio_name and studio.status == "Running":
            return studio.id
    # No running studio found; create or restart
    for studio in studios:
        if studio.name == studio_name and studio.status == "Stopped":
            client.start_studio(studio.id, machine=machine)
            return studio.id
    # Create new if none exist
    return client.create_studio(studio_name, machine=machine).id

def ensure_studio_running(studio_id: str, machine: str = "L40S"):
    client = L.LightningClient()
    if client.get_studio(studio_id).status != "Running":
        client.start_studio(studio_id, machine=machine)

# ---- Training Logic ----
def train():
    studio_name = "SurrogateStudio"
    studio_id = get_or_create_studio(studio_name)
    ensure_studio_running(studio_id)

    # Example training setup
    dataset = CDNParquetDataset("manifest.json")
    loader = DataLoader(dataset, batch_size=8)

    model = AutoModelForCausalLM.from_pretrained("gpt2")
    optimizer = torch.optim.Adam(model.parameters(), lr=1e-4)

    for epoch in range(3):
        ensure_studio_running(studio_id)  # Check before each epoch
        for batch in loader:
            inputs = batch["prompt"]
            labels = batch["response"]
            outputs = model(inputs, labels=labels)
            loss = outputs.loss
            loss.backward()
            optimizer.step()
            optimizer.zero_grad()
        print(f"Epoch {epoch} done")

if __name__ == "__main__":
    train()
```
