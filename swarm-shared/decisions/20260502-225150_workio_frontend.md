# workio / frontend

## 1. Diagnosis

- Attendance events are currently queued in `localStorage` (per past decisions) which is lossy under storage pressure, not transactional, and vulnerable to cross-tab overwrite and race conditions.
- No idempotency key on events → retries can create duplicate clock-in/out records on the backend.
- Retry/backoff is naive or missing; failures during extended offline periods are not persisted durably and can be lost on page reload.
- No cross-tab coordination → multiple tabs can emit conflicting events or double-flush the same payload.
- No visibility into queue health (size, stuck items, last flush) for debugging or support.

## 2. Proposed change

- **Scope**: Frontend only — `src/lib/queue/` (new) + integrate into existing attendance hook/component (likely `src/hooks/useAttendance.ts` or `src/services/attendance.ts`).
- **Files**:
  - `src/lib/queue/idbQueue.ts` — IndexedDB queue with idempotency keys, durable state, exponential backoff + jitter, cross-tab safe ops (BroadcastChannel).
  - `src/lib/queue/types.ts` — types for queue item, status, attempt.
  - `src/services/attendanceQueue.ts` — thin adapter that enqueues clock-in/out with idempotency key and exposes flush/retry.
  - Wire into existing attendance submit path (replace localStorage put with `attendanceQueue.enqueue`).
- **Lines**: new files + ~30–50 lines integration in attendance submit handler.

```ts
export type QueueStatus = 'pending' | 'processing' | 'failed' | 'completed';

export interface QueueItem<T = unknown> {
  id: string;               // internal queue id (uuid)
  idempotencyKey: string;   // business idempotency (e.g. `clock-in:${employeeId}:${date}:${sessionId}`)
  payload: T;
  endpoint: string;         // e.g. '/api/attendance/clock-in'
  method?: 'POST' | 'PUT' | 'PATCH';
  headers?: Record<string, string>;
  attempts: number;
  maxAttempts?: number;     // default 10
  nextRetryAt?: number;     // epoch ms
  createdAt: number;
  updatedAt: number;
  status: QueueStatus;
  lastError?: string;
}
```

### `src/lib/queue/idbQueue.ts`
```ts
import { openDB, DBSchema, IDBPDatabase } from 'idb';
import { v4 as uuidv4 } from 'uuid';

interface QueueDB extends DBSchema {
  items: {
    key: string;
    value: QueueItem;
    indexes: { 'by-status': QueueStatus; 'by-nextRetry': number };
  };
  meta: {
    key: string;
    value: { key: string; value: unknown };
  };
}

const DB_NAME = 'workio_queue';
const STORE = 'items';
const META_STORE = 'meta';

function getDb(): Promise<IDBPDatabase<QueueDB>> {
  return openDB<QueueDB>(DB_NAME, 1, {
    upgrade(db) {
      const store = db.createObjectStore(STORE, { keyPath: 'id' });
      store.createIndex('by-status', 'status', { unique: false });
      store.createIndex('by-nextRetry', 'nextRetryAt', { unique: false });
      db.createObjectStore(META_STORE, { keyPath: 'key' });
    },
  });
}

// Cross-tab coordination
const CHANNEL = new BroadcastChannel('workio_queue_channel');

export async function enqueue<T>(item: Omit<QueueItem<T>, 'id' | 'attempts' | 'createdAt' | 'updatedAt' | 'status'>): Promise<string> {
  const id = uuidv4();
  const now = Date.now();
  const queueItem: QueueItem<T> = {
    ...item,
    id,
    attempts: 0,
    createdAt: now,
    updatedAt: now,
    status: 'pending',
    maxAttempts: item.maxAttempts ?? 10,
  };
  const db = await getDb();
  await db.add(STORE, queueItem);
  CHANNEL.postMessage({ type: 'enqueue', id });
  return id;
}

export async function dequeue(batchSize = 10): Promise<QueueItem[]> {
  const db = await getDb();
  const tx = db.transaction(STORE, 'readwrite');
  const store = tx.objectStore(STORE);
  const idx = store.index('by-status');
  const now = Date.now();

  // Select pending items where nextRetryAt <= now (or undefined)
  const pending = await idx.getAll(IDBKeyRange.only('pending'));
  const ready = pending.filter((p) => !p.nextRetryAt || p.nextRetryAt <= now).slice(0, batchSize);

  for (const item of ready) {
    item.status = 'processing';
    item.attempts += 1;
    item.updatedAt = Date.now();
    await store.put(item);
  }
  await tx.done;
  return ready;
}

export async function markCompleted(id: string): Promise<void> {
  const db = await getDb();
  const tx = db.transaction(STORE, 'readwrite');
  const store = tx.objectStore(STORE);
  const item = await store.get(id);
  if (item) {
    item.status = 'completed';
    item.updatedAt = Date.now();
    await store.put(item);
  }
  await tx.done;
  CHANNEL.postMessage({ type: 'completed', id });
}

export async function markFailed(id: string, error: string): Promise<void> {
  const db = await getDb();
  const tx = db.transaction(STORE, 'readwrite');
  const store = tx.objectStore(STORE);
  const item = await store.get(id);
  if (!item) return;

  item.lastError = error;
  item.updatedAt = Date.now();

  if (item.attempts >= (item.maxAttempts ?? 10)) {
    item.status = 'failed';
  } else {
    item.status = 'pending';
    // exponential backoff with jitter: 2^attempts * 1s ± 30%
    const base = Math.pow(2, item.attempts) * 1000;
    const jitter = base * 0.3 * (Math.random() * 2 - 1);
    item.nextRetryAt = Date.now() + Math.max(5000, base + jitter);
  }

  await store.put(item);
  await tx.done;
  CHANNEL.postMessage({ type: 'failed', id });
}

export async function peekQueue(): Promise<QueueItem[]> {
  const db = await getDb();
  const tx = db.transaction(STORE, 'readonly');
  const store = tx.objectStore(STORE);
  return store.getAll();
}

// Flush worker: caller should run periodically and on online event
export async function flushQueue(
  fetchImpl: typeof fetch,
  options?: { batchSize?: number; signal?: AbortSignal }
): Promise<{ processed: number; errors: number }> {
  const batchSize = options?.batchSize ?? 10;
  const items = await dequeue(batchSize);
  let processed = 0;
  let errors = 0;

  for (const item of items) {
    if (options?.signal?.aborted) break;
    try {
      const res = await fetchImpl(item.endpoint, {
        method: item.method ?? 'POST',
        headers: { 'Content-Type': 'application/json', ...item.headers },
        body: JSON.stringify(item.payload),
        credentials: 'include',
      });

      if (!res.ok) {
        const text = await res.text().catch(() => '');
        throw new Error(`HTTP ${res.status}: ${text}`);
      }

      await markCompleted(item.id);
      processed += 1;
    } catch (err: any) {
      await markFailed(item.id, err?.message ?? String(err));
      errors += 1;
    }
  }
  return { processed, errors };
}
```
