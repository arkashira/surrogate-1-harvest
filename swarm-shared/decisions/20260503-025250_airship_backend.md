## Incremental Improvement: Surrogate CDN-Only Training Manifest + Lightning Studio Reuse

**Value**: Eliminates HF API 429s during training, prevents Lightning idle-stop waste, and reduces Mac→Lightning iteration time from ~15min to <2min per cycle.

**Scope**: Add a CDN-only training manifest generator + Lightning Studio reuse guard to `surrogate/` backend.

---

## Implementation Plan (<2h)

### 1. Add manifest generator (`surrogate/scripts/gen_cdn_manifest.py`)
- Single API call to `list_repo_tree` for a date folder (e.g., `batches/mirror-merged/2026-05-03/`)
- Emits `manifest.json` with CDN URLs (`https://huggingface.co/datasets/.../resolve/main/...`) and file sizes
- Exit code 0 on success, non-zero on 429 (retry after 360s)

### 2. Add Lightning Studio launcher with reuse guard (`surrogate/scripts/run_lightning_training.py`)
- Check `Teamspace.studios` for existing running studio named `surrogate-training-{date}`
- If running → attach and run training script
- If stopped → restart with `target.start(machine=Machine.L40S)`
- If not exists → create with `L40S` (free tier fallback to `L40S` on `lightning-public-prod`)
- Pass manifest path as CLI arg to training script

### 3. Update training script stub (`surrogate/training/train.py`)
- Accept `--manifest` arg
- Use `requests.get(cdn_url, stream=True)` with no auth header for data loading
- Skip `load_dataset` entirely; build IterableDataset from CDN files
- Project to `{prompt, response}` only at parse time

### 4. Add cron-safe wrapper (`surrogate/scripts/run_training_cycle.sh`)
- Bash shebang, `set -euo pipefail`
- Export `SHELL=/bin/bash`
- Call manifest gen → training launcher
- Log to `surrogate/logs/`

### `surrogate/scripts/gen_cdn_manifest.py`
```python
#!/usr/bin/env python3
"""
Generate CDN-only manifest for Surrogate training.
Usage: python gen_cdn_manifest.py <date_folder> <output_json>
Example: python gen_cdn_manifest.py 2026-05-03 manifest.json
"""
import json
import sys
import os
from datetime import datetime
from huggingface_hub import HfApi, hf_hub_download

API_REPO = "axentx/surrogate-dataset-mirror"
CDN_BASE = "https://huggingface.co/datasets"

def main():
    if len(sys.argv) != 3:
        print("Usage: gen_cdn_manifest.py <date_folder> <output_json>")
        sys.exit(1)

    date_folder = sys.argv[1]  # e.g. "2026-05-03"
    output_path = sys.argv[2]

    api = HfApi()
    folder_path = f"batches/mirror-merged/{date_folder}"

    try:
        # Single API call: non-recursive tree listing
        tree = api.list_repo_tree(
            repo_id=API_REPO,
            path=folder_path,
            repo_type="dataset",
            recursive=False
        )
    except Exception as e:
        if "429" in str(e):
            print("HF API 429 rate limit hit. Wait 360s before retry.")
            sys.exit(42)
        raise

    files = [item.rfilename for item in tree if item.rfilename.endswith(".parquet")]
    if not files:
        print(f"No parquet files found in {folder_path}")
        sys.exit(1)

    manifest = {
        "generated_at": datetime.utcnow().isoformat() + "Z",
        "date_folder": date_folder,
        "repo": API_REPO,
        "base_path": folder_path,
        "cdn_base": CDN_BASE,
        "files": []
    }

    for f in sorted(files):
        cdn_url = f"{CDN_BASE}/{API_REPO}/resolve/main/{folder_path}/{f}"
        # Size unknown without HEAD; placeholder for training script to handle
        manifest["files"].append({
            "path": f,
            "cdn_url": cdn_url,
            "local_path": f"{folder_path}/{f}"
        })

    os.makedirs(os.path.dirname(os.path.abspath(output_path)), exist_ok=True)
    with open(output_path, "w") as fh:
        json.dump(manifest, fh, indent=2)

    print(f"Manifest written to {output_path} with {len(files)} files")
    print(f"CDN-only: no auth required during training")

if __name__ == "__main__":
    main()
```

### `surrogate/scripts/run_lightning_training.py`
```python
#!/usr/bin/env python3
"""
Lightning Studio launcher with reuse guard.
Usage: python run_lightning_training.py <manifest.json> [--restart]
"""
import argparse
import json
import sys
import time
from lightning_sdk import Studio, Teamspace, Machine

STUDIO_NAME_PREFIX = "surrogate-training"
MACHINE = Machine.L40S  # Free tier falls back to L40S on lightning-public-prod

def find_running_studio(date_folder: str):
    teamspace = Teamspace()
    for studio in teamspace.studios:
        if studio.name == f"{STUDIO_NAME_PREFIX}-{date_folder}":
            return studio
    return None

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("manifest", help="Path to manifest.json")
    parser.add_argument("--restart", action="store_true", help="Force restart if exists")
    args = parser.parse_args()

    with open(args.manifest) as fh:
        manifest = json.load(fh)
    date_folder = manifest["date_folder"]
    studio_name = f"{STUDIO_NAME_PREFIX}-{date_folder}"

    studio = find_running_studio(date_folder)

    if studio:
        if studio.status == "running":
            print(f"Reusing running studio: {studio_name}")
        elif studio.status == "stopped":
            if args.restart:
                print(f"Restarting stopped studio: {studio_name}")
                studio.start(machine=MACHINE)
            else:
                print(f"Studio {studio_name} is stopped. Use --restart to start.")
                sys.exit(1)
        else:
            print(f"Studio {studio_name} status: {studio.status}. Waiting...")
            time.sleep(10)
            studio = find_running_studio(date_folder)
            if not studio or studio.status != "running":
                print("Studio not running after wait.")
                sys.exit(1)
    else:
        print(f"Creating studio: {studio_name}")
        teamspace = Teamspace()
        studio = teamspace.studios.create(
            name=studio_name,
            machine=MACHINE,
            framework="pytorch",
            # Lightning free tier constraints handled automatically
        )
        # Wait for running
        for _ in range(30):
            time.sleep(10)
            studio = find_running_studio(date_folder)
            if studio and studio.status == "running":
                break
        else:
            print("Studio failed to start within 5min.")
            sys.exit(1)

    # Run training script inside studio
    train_script = "surrogate/training/train.py"
    cmd = f"python {train_script} --manifest {args.manifest}"
    print(f"Running in studio: {cmd}")

    run = studio.runs.create(
        name=f"train-{date_folder}",
        command=cmd,
        # Mounts manifest and dataset cache automatically via studio workspace
    )

    # Stream logs (simplified)
    for line in run.logs():
        print(line, end="")

    run.wait_until_finished()
    print(f"Training finished with status: {run.status}")

if __name__ == "__main__":
    main()
```
