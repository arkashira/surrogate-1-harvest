# airship / backend

## Implementation Plan — `airship discover` (backend)

**Scope**: ≤2h  
**Goal**: Deterministic CLI command `airship discover` that emits a CDN-only file manifest for a given `repo_id` and `date_folder`, plus HTTP endpoint `/api/discover` (GET) for frontend integration.

### What ships
1. CLI: `airship discover <repo_id> <date_folder> [--out FILE]`
   - Single `list_repo_tree(path, recursive=False)` call for the date folder.
   - Outputs JSON manifest of `{repo_id, date_folder, files[], cdn_base, generated_at}`.
   - Validates date folder format (`YYYY-MM-DD`).
   - Respects HF rate limits; falls back to cached manifest if 429 and retries after 360s.

2. HTTP: `GET /api/discover?repo_id=...&date_folder=...`
   - Returns same JSON manifest.
   - Optional `?cached_only=true` to skip live API calls.
   - Returns 429 with `Retry-After` when rate-limited.

3. Internal module: `airship/backend/services/hf_cdn.py`
   - `build_cdn_manifest(repo_id, date_folder, use_api=True)`
   - Uses HF Hub `list_repo_tree` (non-recursive) for one folder.
   - Builds CDN URLs: `https://huggingface.co/datasets/{repo_id}/resolve/main/{date_folder}/{file}`
   - Caches results locally (`.cache/hf_manifests/`) keyed by `repo_id/date_folder.json`.

4. Tests: minimal unit tests for CLI + service (can run offline with fixtures).

### Why this is highest-value (<2h)
- Enables surrogate training pipeline to use CDN-only fetches (bypasses HF API rate limits during training).
- Fits surrogate-1 lessons: single API call from orchestration layer (Mac), embed file list in training script, Lightning does CDN-only data loads.
- Small, deterministic surface; no schema changes, no DB migrations.

```python
from __future__ import annotations

import json
import os
import time
from datetime import datetime, timezone
from pathlib import Path
from typing import Dict, List, Optional

import httpx
from huggingface_hub import HfApi, hf_hub_url, list_repo_tree
from huggingface_hub.utils import HFValidationError, RepositoryNotFoundError

CACHE_DIR = Path(__file__).resolve().parents[3] / ".cache" / "hf_manifests"
CACHE_DIR.mkdir(parents=True, exist_ok=True)

HF_API_RATE_LIMIT_RETRY = 360  # seconds (per lessons)


def _cache_path(repo_id: str, date_folder: str) -> Path:
    safe = repo_id.replace("/", "--")
    return CACHE_DIR / f"{safe}__{date_folder}.json"


def _is_date_folder(date_folder: str) -> bool:
    try:
        datetime.strptime(date_folder, "%Y-%m-%d")
        return True
    except ValueError:
        return False


def build_cdn_manifest(
    repo_id: str,
    date_folder: str,
    use_api: bool = True,
    timeout: float = 30.0,
) -> Dict:
    """
    Build a CDN-only manifest for repo_id/date_folder.

    If use_api=True, calls list_repo_tree once (non-recursive) for the folder.
    If API returns 429, raises RuntimeError with retry-after suggestion.
    Always produces CDN URLs:
      https://huggingface.co/datasets/{repo_id}/resolve/main/{date_folder}/{file}
    """
    if not repo_id or not date_folder:
        raise ValueError("repo_id and date_folder are required")
    if not _is_date_folder(date_folder):
        raise ValueError("date_folder must be YYYY-MM-DD")

    files: List[str] = []

    if use_api:
        try:
            tree = list_repo_tree(
                repo_id=repo_id,
                path=date_folder,
                recursive=False,
            )
            files = [item.path.split("/", 1)[-1] for item in tree if item.type == "file"]
        except Exception as exc:
            # Preserve HF rate-limit behavior: if 429-like, raise with retry hint
            msg = str(exc).lower()
            if "429" in msg or "rate limit" in msg:
                raise RuntimeError(
                    f"HF API rate limited for {repo_id}/{date_folder}. "
                    f"Wait ~{HF_API_RATE_LIMIT_RETRY}s before retry."
                ) from exc
            # Other errors (e.g., folder missing) -> return empty file list but still cache
            files = []

    manifest = {
        "repo_id": repo_id,
        "date_folder": date_folder,
        "generated_at": datetime.now(timezone.utc).isoformat(),
        "cdn_base": f"https://huggingface.co/datasets/{repo_id}/resolve/main",
        "files": sorted(files),
        "file_urls": [
            f"https://huggingface.co/datasets/{repo_id}/resolve/main/{date_folder}/{f}"
            for f in sorted(files)
        ],
        "note": "CDN-only URLs; no Authorization required for downloads",
    }

    # Cache best-effort
    try:
        _cache_path(repo_id, date_folder).write_text(json.dumps(manifest, indent=2))
    except Exception:
        pass

    return manifest


def load_cached_manifest(repo_id: str, date_folder: str) -> Optional[Dict]:
    p = _cache_path(repo_id, date_folder)
    if not p.is_file():
        return None
    try:
        return json.loads(p.read_text())
    except Exception:
        return None
```

### 2) Add CLI command

`/opt/axentx/airship/airship/backend/cli/discover.py`
```python
#!/usr/bin/env python3
import argparse
import json
import sys
from pathlib import Path

# Ensure package root is importable
sys.path.insert(0, str(Path(__file__).resolve().parents[2]))

from airship.backend.services.hf_cdn import build_cdn_manifest, load_cached_manifest


def main() -> None:
    parser = argparse.ArgumentParser(
        description="Discover CDN file manifest for a HuggingFace dataset folder."
    )
    parser.add_argument("repo_id", help="HF dataset repo id (e.g., 'org/dataset')")
    parser.add_argument("date_folder", help="Date folder (YYYY-MM-DD)")
    parser.add_argument(
        "--out",
        default=None,
        help="Output file (default: stdout)",
    )
    parser.add_argument(
        "--cached-only",
        action="store_true",
        help="Only use cached manifest; do not call HF API",
    )
    args = parser.parse_args()

    use_api = not args.cached_only
    cached = load_cached_manifest(args.repo_id, args.date_folder)

    if args.cached_only and cached is None:
        print("No cached manifest found and --cached-only set.", file=sys.stderr)
        sys.exit(1)

    try:
        manifest = (
            cached
            if args.cached_only
            else build_cdn_manifest(args.repo_id, args.date_folder, use_api=use_api)
        )
    except RuntimeError as exc:
        # Likely rate-limited
        print(f"Error: {exc}", file=sys.stderr)
        # If cached exists, offer it
        if cached:
            print("Falling back to cached manifest.", file=sys.stderr)
            manifest = cached
        else:
            sys.exit(1)

    out = json.dumps(manifest, indent=2)
    if args.out:
        Path(args.out).write_text(out)
        print(f"Manifest written to {args.out}")
    else:
        print(out)


if __name__ == "__main__":
    main()
```
