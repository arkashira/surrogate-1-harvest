# vanguard / design

## Final synthesized solution (best parts, contradictions resolved)

**Core diagnosis (agreed across candidates)**
- No persisted `(repo, dateFolder) → file-list` manifest causes repeated authenticated `list_repo_tree` and quota/429 risk.
- Training uses `load_dataset`/streaming or per-file loads on heterogeneous schemas → `pyarrow.CastError` and wasted ingestion.
- No CDN-only fetch path during training → authenticated API calls continue while training, increasing quota pressure.
- Missing schema projection: raw files with mixed schemas written to `enriched/` instead of projecting to `{prompt, response}` only, breaking surrogate-1 schema discipline.
- Orchestration/training coupling: Mac-side orchestration and Lightning-side training are not cleanly separated, risking quota exhaustion on dev machines.

**Resolved design choices (correctness + actionability)**
- Keep manifest generation **offline/Mac-side** (single `list_repo_tree` per folder) and **training Lightning-side** strictly **CDN-only** (no auth, no HF API calls).
- Use **public CDN URLs** (`https://huggingface.co/datasets/<repo>/resolve/main/<path>`) for all training fetches.
- Add **schema projection** that streams each parquet (or jsonl) from CDN and yields only `{prompt, response}`; tolerate missing columns and skip unreadable files.
- Keep manifest format minimal and deterministic: `{"repo": "...", "date_folder": "...", "files": [...]}`.
- Provide **clear verification steps** and **failure isolation** (skip bad files; log in production).

## 1. Manifest generator (Mac/orchestrator side)

`/opt/axentx/vanguard/scripts/generate_file_manifest.py`

```python
#!/usr/bin/env python3
"""
Generate (repo, dateFolder) → file-list manifest for CDN-only training.
Run once per folder (or on schedule) to avoid repeated HF API list calls.
"""
from __future__ import annotations

import argparse
import json
from pathlib import Path

from huggingface_hub import HfApi

CDN_ROOT = "https://huggingface.co/datasets"

def build_file_manifest(repo: str, date_folder: str, out_dir: Path) -> Path:
    api = HfApi()
    # Non-recursive list for the folder (paginated; cheap)
    tree = api.list_repo_tree(repo_id=repo, path=date_folder, recursive=False)
    files = sorted(item.path for item in tree if item.type == "file")

    out_dir = out_dir / repo.replace("/", "_")
    out_dir.mkdir(parents=True, exist_ok=True)
    out_path = out_dir / f"{date_folder}.json"
    manifest = {"repo": repo, "date_folder": date_folder, "files": files}
    out_path.write_text(json.dumps(manifest, indent=2))
    return out_path

def main() -> None:
    parser = argparse.ArgumentParser(description="Generate CDN file manifest for a repo/date folder.")
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g., user/dataset)")
    parser.add_argument("--date-folder", required=True, help="Date folder in dataset (e.g., 2026-05-01)")
    parser.add_argument("--out-dir", default="manifests", help="Output directory for manifests")
    args = parser.parse_args()

    out_path = build_file_manifest(args.repo, args.date_folder, Path(args.out_dir))
    print(f"Manifest written: {out_path}")

if __name__ == "__main__":
    main()
```

## 2. CDN-first dataset + schema projection (Lightning/training side)

`/opt/axentx/vanguard/data/cdn_dataset.py`

```python
"""
CDN-only dataset for surrogate-1 training.
Uses public CDN URLs to bypass HF API auth and rate limits during training.
"""
from __future__ import annotations

import json
from pathlib import Path
from typing import Dict, Iterator, List, Optional, Tuple

import pyarrow as pa
import pyarrow.parquet as pq
import requests
from torch.utils.data import IterableDataset

CDN_ROOT = "https://huggingface.co/datasets"

class CDNParquetReader:
    """Stream a parquet file from CDN without auth and project columns."""

    def __init__(self, repo: str, cdn_path: str) -> None:
        self.url = f"{CDN_ROOT}/{repo}/resolve/main/{cdn_path}"
        self.repo = repo

    def read_project(
        self,
        columns: Tuple[str, ...] = ("prompt", "response"),
        max_retries: int = 2,
    ) -> List[Dict[str, str]]:
        """Download parquet from CDN and project to requested columns."""
        resp = requests.get(self.url, timeout=30, retries=max_retries)
        resp.raise_for_status()

        # Read parquet from buffer
        table = pq.read_table(pq.ParquetFile(pq.ParquetReader(pa.BufferReader(resp.content))))
        existing = [c for c in columns if c in table.column_names]
        if not existing:
            raise ValueError(
                f"None of {columns} found in {self.url}. Available: {table.column_names}"
            )

        # Select and normalize
        proj = table.select(existing).to_pylist()
        for row in proj:
            for c in columns:
                # Coerce missing to empty string; ensure str for surrogate-1 discipline
                val = row.get(c, "")
                row[c] = "" if val is None else str(val)
        return proj

class CDNDataSet(IterableDataset):
    """
    Lightning-friendly dataset that reads a manifest and streams parquet files via CDN.
    Skips unreadable files to avoid crashing training on corrupt data.
    """

    def __init__(
        self,
        manifest_path: str | Path,
        columns: Tuple[str, ...] = ("prompt", "response"),
        max_retries: int = 2,
    ) -> None:
        manifest_path = Path(manifest_path)
        self.meta = json.loads(manifest_path.read_text())
        self.files: List[str] = self.meta["files"]
        self.repo: str = self.meta["repo"]
        self.columns = columns
        self.max_retries = max_retries

    def _iter_file(self, fpath: str) -> Iterator[Dict[str, str]]:
        reader = CDNParquetReader(self.repo, fpath)
        try:
            rows = reader.read_project(self.columns, max_retries=self.max_retries)
            yield from rows
        except Exception:
            # Skip corrupt/unreadable files; in production log with file path and error.
            return

    def __iter__(self) -> Iterator[Dict[str, str]]:
        for fpath in self.files:
            yield from self._iter_file(fpath)
```

Optional schema projection utility (for preprocessing/validation):

`/opt/axentx/vanguard/data/project_schema.py`

```python
"""
Project raw parquet/jsonl (from CDN) to surrogate-1 schema {prompt, response}.
Can be used in preprocessing pipelines or validation.
"""
from __future__ import annotations

import argparse
import json
from pathlib import Path

from vanguard.data.cdn_dataset import CDNParquetReader

def project_file(repo: str, cdn_path: str, out_path: Path) -> None:
    reader = CDNParquetReader(repo, cdn_path)
    rows = reader.read_project(columns=("prompt", "response"))
    out_path.write_text(json.dumps(rows, indent=2))
    print(f"Projected {len(rows)} rows -> {out_path}")

def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True)
    parser.add_argument("--cdn-path", required=True)
    parser.add_argument("--out", required=True)
    args = parser.parse_args()
    project_file(args.repo, args.cdn_path, Path(args.out))

if __name__ == "__main__":
    main()
```
