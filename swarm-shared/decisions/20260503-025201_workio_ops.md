# workio / ops

## 1. Diagnosis

- No server-side idempotency key enforcement on `/webhook/line` → LINE retries on slow 2xx/timeouts create duplicate punches.
- Clock-in/out handler does optimistic concurrency or dedupe check; races between duplicate webhooks can insert two punches for same user within seconds.
- No structured logging / tracing for webhook deliveries (status, duration, idempotency key) → hard to debug duplicates in prod.
- No operational alerting on abnormal punch volume (spikes, duplicates per user/hour) → incidents noticed late.
- Missing graceful handling of out-of-order GPS/latency edge cases (clock-in after clock-out within same minute) causing invalid state.

## 2. Proposed change

File: `/opt/axentx/workio/server/src/routes/webhook/line.ts` (or equivalent route handler for LINE webhook).  
Scope: add idempotency table + middleware, enforce idempotency key from LINE `X-Line-Signature` (or generate from event+user), and add duplicate suppression with upsert. Add structured logs and a lightweight metric for duplicate attempts.

`server/src/middleware/idempotency.ts`
```ts
import { Request, Response, NextFunction } from 'express';
import { pool } from '../db';
import crypto from 'crypto';

const IDENTITY_TTL_MS = 5 * 60 * 1000; // 5m

function hashPayload(body: any): string {
  return crypto.createHash('sha256').update(JSON.stringify(body)).digest('hex');
}

export async function lineIdempotency(
  req: Request,
  res: Response,
  next: NextFunction
) {
  // LINE webhook: use X-Line-Signature as primary component + user-level stable key
  const lineSig = req.get('X-Line-Signature') || '';
  const body = req.body;
  const events = body.events || [];

  // If no events, skip (but allow through)
  if (!events.length) return next();

  // Build per-event idempotency keys; attach to request for handler
  const keys: string[] = [];
  for (const ev of events) {
    const userId = ev.source?.userId || ev.destination || 'unknown';
    const eventType = ev.type || 'unknown';
    const payloadHash = hashPayload(ev);
    // Deterministic key: userId + eventType + signature + payloadHash
    const key = crypto
      .createHash('sha256')
      .update(`${userId}:${eventType}:${lineSig}:${payloadHash}`)
      .digest('hex');
    keys.push(key);
  }

  (req as any)._idemKeys = keys;
  (req as any)._idemValues = events.map((ev: any, i: number) => ({
    key: keys[i],
    userId: ev.source?.userId || ev.destination || null,
    eventType: ev.type || 'unknown',
    payloadHash: hashPayload(ev),
  }));

  next();
}

export async function recordIdempotentResult(
  req: Request,
  res: Response,
  next: NextFunction
) {
  const keys: string[] = (req as any)._idemKeys || [];
  const values: any[] = (req as any)._idemValues || [];
  const status = res.statusCode;
  const body = JSON.stringify(res.locals._responseBody || res.body || {});

  if (!keys.length) return next();

  try {
    const client = await pool.connect();
    try {
      for (let i = 0; i < keys.length; i++) {
        const v = values[i];
        await client.query(
          `INSERT INTO webhook_idempotency (idempotency_key, user_id, event_type, payload_hash, processed_at, response_status, response_body)
           VALUES ($1,$2,$3,$4,NOW(),$5,$6)
           ON CONFLICT (idempotency_key) DO UPDATE
           SET processed_at = EXCLUDED.processed_at,
               response_status = EXCLUDED.response_status,
               response_body = EXCLUDED.response_body`,
          [v.key, v.userId, v.eventType, v.payloadHash, status, body]
        );
      }
    } finally {
      client.release();
    }
  } catch (err) {
    // non-fatal: log but don't fail response
    console.error('[idempotency] record failed', err);
  }
  next();
}

export async function dedupeProcessed(
  req: Request,
  res: Response,
  next: NextFunction
) {
  const values: any[] = (req as any)._idemValues || [];
  if (!values.length) return next();

  const client = await pool.connect();
  try {
    // Find already-processed keys within TTL
    const keys = values.map((v) => v.key);
    const { rows } = await client.query(
      `SELECT idempotency_key, response_status, response_body FROM webhook_idempotency
       WHERE idempotency_key = ANY($1) AND processed_at IS NOT NULL AND created_at > NOW() - ($2 || ' ms')::INTERVAL`,
      [keys, IDENTITY_TTL_MS]
    );

    const processedSet = new Set(rows.map((r) => r.idempotency_key));
    if (processedSet.size > 0) {
      // Short-circuit: return 200 for duplicates to suppress LINE retries
      res.status(200).json({ duplicate: true, suppressed: Array.from(processedSet) });
      return;
    }
  } catch (err) {
    // If dedupe fails, continue processing (safe fallback)
    console.warn('[idempotency] dedupe check failed', err);
  } finally {
    client.release();
  }
  next();
}
```

`server/src/routes/webhook/line.ts` (apply middleware + handler changes)
```ts
import express from 'express';
import { lineIdempotency, dedupeProcessed, recordIdempotentResult } from '../../middleware/idempotency';
import { handleLineEvent } from '../../services/line';

const router = express.Router();

router.post(
  '/line',
  lineIdempotency,
  dedupeProcessed,
  async (req, res, next) => {
    try {
      const events = req.body.events || [];
      // Process events sequentially to reduce race window per user
      for (const ev of events) {
        await handleLineEvent(ev);
      }
      res.locals._responseBody = { ok: true };
      res.status(200).json({ ok: true });
    } catch (err) {
      next(err);
    }
  },
  recordIdempotentResult
);

export default router;
```

export async function handleLineEvent(event: any) {
  if (event.type !== 'message' && event.type !== 'postback') return;

  const userId = event.source?.userId;
  if (!userId) return;

  const isClockIn = payload.includes('in') || payload.includes('clockin');
  const isClockOut = payload.includes('out') || payload.includes('clockout');

  if (!isClockIn && !isClockOut) return;

  const client = await pool.connect();
  try {
