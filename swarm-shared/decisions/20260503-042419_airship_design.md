### 1) Manifest generator (`scripts/build_cdn_manifest.py`)
```python
#!/usr/bin/env python3
"""
Generate a CDN-only manifest for a HuggingFace dataset repo folder.
Run from dev machine (Mac) to avoid HF API 429s during training.
"""
import argparse
import hashlib
import json
import sys
from datetime import datetime, timezone
from pathlib import Path

from huggingface_hub import HfApi

CDN_BASE = "https://huggingface.co/datasets"

def list_folder(api, repo_id: str, path: str):
    # recursive=False keeps list_repo_tree cheap; recurse manually
    items = api.list_repo_tree(repo_id=repo_id, path=path, recursive=False)
    files = [i for i in items if i.type == "file"]
    folders = [i for i in items if i.type == "dir"]
    for f in folders:
        files.extend(list_folder(api, repo_id, f.path))
    return files

def build_manifest(repo_id: str, base: str, out_file: Path):
    api = HfApi()
    files = list_folder(api, repo_id, base)
    paths = sorted(f.rfilename for f in files)

    manifest = {
        "repo": repo_id,
        "base": base,
        "paths": paths,
        "count": len(paths),
        "created": datetime.now(timezone.utc).isoformat(),
    }
    manifest_bytes = json.dumps(manifest, sort_keys=True).encode()
    manifest["sha256"] = hashlib.sha256(manifest_bytes).hexdigest()

    out_file.parent.mkdir(parents=True, exist_ok=True)
    out_file.write_text(json.dumps(manifest, indent=2, sort_keys=True))
    print(f"Wrote {out_file} ({len(paths)} files)")

def main() -> None:
    parser = argparse.ArgumentParser(description="Generate CDN manifest for HF dataset folder.")
    parser.add_argument("--repo", default="axentx/surrogate-dataset", help="HF dataset repo")
    parser.add_argument("--folder", required=True, help="Folder path in repo (e.g. batches/mirror-merged/2026-04-29)")
    parser.add_argument("--out", default="manifests/cdn-manifest.json", help="Output JSON path")
    args = parser.parse_args()

    build_manifest(args.repo, args.folder, Path(args.out))

if __name__ == "__main__":
    main()
```

### 2) CDN dataset loader (`surrogate/data.py`)
```python
import json
import time
import random
import requests
from pathlib import Path
from torch.utils.data import IterableDataset

CDN_BASE = "https://huggingface.co/datasets"

class CDNDataset(IterableDataset):
    def __init__(self, manifest_path, project_fn, max_retries=5, backoff_base=1.0):
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.repo = self.manifest["repo"]
        self.paths = self.manifest["paths"]
        self.project_fn = project_fn  # fn(raw_bytes) -> {prompt, response}
        self.max_retries = max_retries
        self.backoff_base = backoff_base

    def _download(self, rel_path):
        url = f"{CDN_BASE}/{self.repo}/resolve/main/{rel_path}"
        for attempt in range(1, self.max_retries + 1):
            try:
                resp = requests.get(url, timeout=30)
                resp.raise_for_status()
                return resp.content
            except Exception as exc:
                if attempt == self.max_retries:
                    raise
                jitter = random.uniform(0.5, 1.5)
                delay = (self.backoff_base * (2 ** (attempt - 1))) * jitter
                time.sleep(delay)

    def __iter__(self):
        for p in self.paths:
            raw = self._download(p)
            yield self.project_fn(raw)
```

"""

Keeps quota usage minimal and prevents idle-stop training loss.

# Add project root to path for surrogate imports
PROJECT_ROOT = Path(__file__).parent.parent
sys.path.insert(0, str(PROJECT_ROOT))

from lightning import Studio, Teamspace, Machine

STUDIO_NAME = os.getenv("STUDIO_NAME", "surrogate-training")
MACHINE = os.getenv("MACHINE", "L40S")
MANIFEST = os.getenv("MANIFEST", "manifests/cdn-manifest-2026-04-29.json")

def get_studio():
    for s in Teamspace.studios:
        if s.name == STUDIO_NAME:
            return s
    return None

def ensure_running_studio():
    studio = get_studio()
    if studio is None:
        print(f"Creating Studio {STUDIO_NAME}")
        studio = Studio.create(name=STUDIO_NAME, machine=Machine.L40S, create_ok=True)
    else:
        print(f"Found Studio {STUDIO_NAME} (status={studio.status})")

        print("Studio not running; starting...")

            studio.refresh()
            if studio.status == "running":
                break
            time.sleep(10)
        else:
            raise RuntimeError("Studio failed to start")
    return studio
