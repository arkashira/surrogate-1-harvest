# airship / ops

## Final Implementation (merged + reconciled)

**Scope** (unchanged, <2h):  
Zero-runtime-dependency, CDN-safe `airship discover` orchestrator (Bash-first) that deterministically emits `status.json` + `manifest.json` for local/CI use and CDN upload. Frontends fetch via CDN URLs.

**Why this scope** (merged):  
- Fits <2h: pure Bash + jq (no Python/runtime deps).  
- Highest-value ops improvement: deterministic discovery for CI/CD without container/ML overhead.  
- Enables CDN-bypass pattern: static manifests + file lists usable by frontends and training pipelines without API auth.  
- Aligns with “Mac=CLI rule + heavy compute on remote” (orchestration only on Mac/CI).

### 1) Create orchestrator script

`bin/airship-discover`
```bash
#!/usr/bin/env bash
# airship-discover
# Deterministic discovery + manifest generator (zero runtime deps except jq).
#
# Usage:
#   ./bin/airship-discover [--out-dir <dir>] [--cdn-base <url>]
#
# Outputs:
#   <out-dir>/status.json   - runtime state snapshot (ports, health, git, uptime)
#   <out-dir>/manifest.json - deterministic file manifest with CDN-ready paths
#
# Notes:
#   - Uses only POSIX tools + jq. No Python.
#   - Deterministic ordering via `sort`.
#   - Emits stable hashes (sha256) for cacheable assets.
#   - CDN URLs constructed without Authorization (CDN bypass pattern).
#   - Health checks are best-effort (non-fatal failures).

set -euo pipefail

REPO_ROOT="$(cd "$(dirname "$0")/.." && pwd)"
OUT_DIR="${2:-${AIRSHIP_OUT_DIR:-${REPO_ROOT}/dist/discover}}"
CDN_BASE="${3:-${AIRSHIP_CDN_BASE:-}}"

mkdir -p "$OUT_DIR"

# Helper: stable timestamp (ISO 8601, UTC) for this run
TS="$(date -u +"%Y-%m-%dT%H:%M:%SZ")"

# ---- helpers ----
safe_hostname() { hostname 2>/dev/null || echo "unknown-host"; }
safe_whoami() { whoami 2>/dev/null || echo "unknown-user"; }
safe_uname_s() { uname -s 2>/dev/null || echo "unknown-os"; }
safe_uname_m() { uname -m 2>/dev/null || echo "unknown-arch"; }

git_metadata() {
  local repo_root="$1"
  local branch commit dirty remote url
  branch=$(git -C "$repo_root" rev-parse --abbrev-ref HEAD 2>/dev/null || echo "unknown-branch")
  commit=$(git -C "$repo_root" rev-parse HEAD 2>/dev/null || echo "unknown-commit")
  dirty=$(git -C "$repo_root" status --porcelain 2>/dev/null | head -c1 | wc -c | tr -d ' ')
  remote=$(git -C "$repo_root" remote get-url origin 2>/dev/null || echo "")
  url=$(git -C "$repo_root" config --get remote.origin.url 2>/dev/null || echo "$remote")
  jq -n \
    --arg branch "$branch" \
    --arg commit "$commit" \
    --arg dirty "$dirty" \
    --arg remote "$url" \
    '{
      branch: $branch,
      commit: $commit,
      dirty: ($dirty | tonumber > 0),
      remote: (if $remote == "" then null else $remote end)
    }'
}

uptime_seconds() {
  if [ -f /proc/uptime ]; then
    awk '{printf "%.0f",$1}' /proc/uptime 2>/dev/null || echo "0"
  elif command -v sysctl >/dev/null 2>&1; then
    sysctl -n kern.boottime 2>/dev/null | awk -F'[ ,]' '{now=systime(); boot=$4; printf "%.0f", now-boot}' || echo "0"
  else
    echo "0"
  fi
}

# Non-fatal health probe for a service
probe_health() {
  local name="$1"
  local port="$2"
  local path="${3:-/health}"
  local timeout=2
  local result="unreachable"
  local code=""
  local body=""

  if command -v curl >/dev/null 2>&1; then
    body=$(curl -sSf --max-time "$timeout" "http://localhost:${port}${path}" 2>/dev/null || true)
    code=$(curl -sSf -o /dev/null -w "%{http_code}" --max-time "$timeout" "http://localhost:${port}${path}" 2>/dev/null || true)
  elif command -v wget >/dev/null 2>&1; then
    body=$(wget -qO- --timeout="$timeout" --tries=1 "http://localhost:${port}${path}" 2>/dev/null || true)
    code=$(wget -qO- --server-response --timeout="$timeout" --tries=1 "http://localhost:${port}${path}" 2>&1 | awk '/^  HTTP/{print $2}' | head -1 || true)
  fi

  if [ -n "$code" ] && [ "$code" -ge 200 ] && [ "$code" -lt 400 ] 2>/dev/null; then
    result="healthy"
  elif [ -n "$code" ] && [ "$code" -ge 400 ] 2>/dev/null; then
    result="unhealthy"
  fi

  jq -n \
    --arg name "$name" \
    --arg port "$port" \
    --arg path "$path" \
    --arg status "$result" \
    --arg code "${code:-}" \
    --arg body "$body" \
    '{
      name: $name,
      port: ($port | tonumber),
      path: $path,
      status: $status,
      code: (if $code == "" then null else ($code | tonumber) end),
      body: (if $body == "" then null else $body end)
    }'
}

# ---- status.json ----
# Deterministic runtime state snapshot (ports, health, git, uptime).
status_json() {
  local arkship_health surrogate_health
  arkship_health=$(probe_health "arkship" 8000 "/health")
  surrogate_health=$(probe_health "surrogate" 8001 "/health")

  jq -n \
    --arg ts "$TS" \
    --arg repo_root "$REPO_ROOT" \
    --arg hostname "$(safe_hostname)" \
    --arg user "$(safe_whoami)" \
    --arg os "$(safe_uname_s)" \
    --arg arch "$(safe_uname_m)" \
    --arg uptime "$(uptime_seconds)" \
    --argjson arkship "$arkship_health" \
    --argjson surrogate "$surrogate_health" \
    --argjson git "$(git_metadata "$REPO_ROOT")" \
    '{
      generated_at: $ts,
      repo_root: $repo_root,
      host: $hostname,
      user: $user,
      os: $os,
      arch: $arch,
      uptime_seconds: ($uptime | tonumber),
      services: [
        {
          name: "arkship",
          port: 8000,
          ui_path: "arkship",
          status: "defined",
          docker_compose: "arkship/docker-compose.yml",
          health: $arkship
        },
        {
          name: "surrogate",
          port: 8001,
          status: "defined",
          docker_compose: "surrogate/docker-compose.yml",
          health: $surrogate
        }
      ],
      git: $git,
      checks: {
        docker_compose_present: ([
          "arkship/docker-compose.yml",
          "surrogate/docker-compose.yml"
        ] | map(test("^(\\.\\/|${REPO_ROOT}/).*")) | any),
        bin_airship_discover_present: (($REPO_ROOT + "/bin/airship-discover") | test(".*"))
      }
    }'
}
