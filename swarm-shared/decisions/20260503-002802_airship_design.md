### 1. Manifest Generator (run on Mac orchestrator)
`scripts/generate_manifest.py`
```python
#!/usr/bin/env python3
"""
Generate a date-scoped manifest for CDN-only dataset loading.
Run on Mac orchestrator after rate-limit window clears.
Uses single non-recursive call per folder.
"""
import json, os, sys
from datetime import datetime, timezone
from pathlib import Path
from huggingface_hub import HfApi

HF_REPO = os.getenv("HF_DATASET_REPO", "axentx/surrogate-mirror")
DATE_FOLDER = os.getenv("DATE_FOLDER", datetime.now(timezone.utc).strftime("%Y-%m-%d"))
OUT_MANIFEST = os.getenv("MANIFEST_PATH", f"surrogate/manifests/{DATE_FOLDER}.json")

def main() -> None:
    api = HfApi()
    # Single non-recursive call per folder (avoids recursive list_repo_files)
    files = api.list_repo_tree(repo_id=HF_REPO, path=DATE_FOLDER, recursive=False)
    entries = []
    for f in files:
        if f.path.endswith(".parquet"):
            entries.append({
                "path": f.path,
                "url": f"https://huggingface.co/datasets/{HF_REPO}/resolve/main/{f.path}",
                "size": getattr(f, "size", None)
            })

    manifest = {
        "repo": HF_REPO,
        "date": DATE_FOLDER,
        "generated_at": datetime.now(timezone.utc).isoformat(),
        "entries": entries
    }

    Path(OUT_MANIFEST).parent.mkdir(parents=True, exist_ok=True)
    with open(OUT_MANIFEST, "w") as fp:
        json.dump(manifest, fp, indent=2)
    print(f"Wrote {len(entries)} entries to {OUT_MANIFEST}")

if __name__ == "__main__":
    main()
```

### 2. CDN-Only Iterable Dataset (Robust)
`surrogate/data/cdn_dataset.py`
```python
import json, io, os, logging
from typing import Iterator, Dict, Any

import pyarrow.parquet as pq
import requests
from torch.utils.data import IterableDataset

logger = logging.getLogger(__name__)

class CDNParquetIterable(IterableDataset):
    """
    Zero HF API calls during training.
    Reads parquet files directly from HF CDN using a pre-generated manifest.
    Projects only {prompt, response} at parse time to handle mixed schemas.
    Skips malformed files gracefully.
    """
    def __init__(self, manifest_path: str, columns=("prompt", "response")):
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.entries = self.manifest["entries"]
        self.columns = columns

    def _stream_file(self, entry: Dict[str, Any]) -> Iterator[Dict[str, Any]]:
        url = entry["url"]
        try:
            # CDN download: no Authorization header, bypasses /api/ rate limits
            resp = requests.get(url, timeout=60)
            resp.raise_for_status()
            table = pq.read_table(io.BytesIO(resp.content), columns=self.columns)
            for batch in table.to_batches():
                cols = {c: batch.column(c).to_pylist() for c in self.columns}
                for i in range(len(cols[self.columns[0]])):
                    yield {c: cols[c][i] for c in self.columns}
        except Exception as e:
            logger.warning(f"Skipping malformed file {url}: {e}")
            return

    def __iter__(self) -> Iterator[Dict[str, Any]]:
        for entry in self.entries:
            yield from self._stream_file(entry)
```

### 3. Patch Training Script with CLI
`surrogate/train.py`
```python
import argparse
import os
import torch
from surrogate.data.cdn_dataset import CDNParquetIterable

def parse_args():
    parser = argparse.ArgumentParser()
    parser.add_argument("--manifest", default=os.getenv("MANIFEST_PATH", "surrogate/manifests/latest.json"))
    parser.add_argument("--steps", type=int, default=None)
    return parser.parse_args()

def main():
    args = parse_args()
    
    # Use CDN dataset instead of HF streaming
    train_dataset = CDNParquetIterable(args.manifest, columns=("prompt", "response"))
    
    train_loader = torch.utils.data.DataLoader(
        train_dataset,
        batch_size=8,
        num_workers=0,  # recommended for streaming IO
        pin_memory=True
    )
    
    # Training loop (example)
    for step, batch in enumerate(train_loader):
        if args.steps and step >= args.steps:
            break
        # Your training step here
        print(f"Step {step}: batch keys {batch.keys()}")

if __name__ == "__main__":
    main()
```

### 4. Studio Reuse + Smoke Test
`scripts/reuse_studio.py`
```python
#!/usr/bin/env python3
"""
Reuse running Lightning Studio to save quota.
Auto-restarts if idle-stop killed training.
"""
from lightning_sdk import Teamspace, Studio, Machine
import time

team = Teamspace.current()
studio_name = "surrogate-train-l40s"
running = [s for s in team.studios if s.name == studio_name and s.status == "running"]

if running:
    studio = running[0]
    print(f"Reusing running studio: {studio.id}")
else:
    studio = Studio.create(
        name=studio_name,
        machine=Machine.L40S,
        container="pytorch/pytorch:latest",
        teamspace=team.name,
        start_now=True
    )
    print(f"Created studio: {studio.id}")

# Quick smoke: one train step with manifest
job = studio.run(
    command="python surrogate/train.py --steps 100 --manifest surrogate/manifests/latest.json",
    wait=True
)
print(f"Smoke job finished: {job.status}")
```
