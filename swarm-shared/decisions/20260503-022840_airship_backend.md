import json
import os
from huggingface_hub import list_repo_tree

REPO = "axentx/surrogate-mirror"
DATE_FOLDER = "2026-05-03"
OUT = "file_list.json"

def main():
    items = list_repo_tree(REPO, path=DATE_FOLDER, recursive=False)
    files = [f.rfilename for f in items if f.type == "file"]
    with open(OUT, "w") as f:
        json.dumps(files, f, indent=2)
    print(f"Wrote {len(files)} files to {OUT}")

if __name__ == "__main__":
    main()

# surrogate/train/train.py
import json
import os
import torch
from torch.utils.data import IterableDataset, DataLoader
from huggingface_hub import hf_hub_download

REPO = "axentx/surrogate-mirror"
CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

class CDNParquetDataset(IterableDataset):
    def __init__(self, file_list_path: str, cache_dir: str = ".cache_train"):
        with open(file_list_path) as f:
            self.files = json.load(f)
        self.cache_dir = cache_dir
        os.makedirs(cache_dir, exist_ok=True)

    def __iter__(self):
        for rel in self.files:
            local = hf_hub_download(repo_id=REPO, filename=rel, cache_dir=self.cache_dir)
            # stream rows from parquet without full load
            import pyarrow.parquet as pq
            table = pq.read_table(local, columns=["prompt", "response"])
            for row in table.to_pylist():
                # tokenize/collate handled downstream
                yield row["prompt"], row["response"]

def get_dataloader(file_list_path: str, batch_size: int = 8, num_workers: int = 4):
    dataset = CDNParquetDataset(file_list_path)
    return DataLoader(dataset, batch_size=batch_size, num_workers=num_workers)

### D) Lightning launcher: reuse studio + idle restart
```python
# surrogate/orchestration/lightning_launcher.py
import lightning as L
from lightning import Machine

TEAMSPACE = L.Teamspace()
STUDIO_NAME = "surrogate-train-l40s"

def get_or_create_studio():
    for s in TEAMSPACE.studios:
        if s.name == STUDIO_NAME and s.status == "running":
            print(f"Reusing running studio: {s.name}")
            return s
    print(f"Creating studio: {STUDIO_NAME}")
    return L.Studio(
        name=STUDIO_NAME,
        machine=Machine.L40S,
        create_ok=True,
    )

def run_training(script: str, args: list[str] = None):
    studio = get_or_create_studio()
    if studio.status != "running":
        print("Studio stopped; restarting...")
        studio.start(machine=Machine.L40S)
    # Use target.run to execute training script inside studio
    job = studio.run(
        target=script,
        args=args or [],
        detach=False,
    )
    print(f"Training finished: {job}")
    return job

if __name__ == "__main__":
    run_training("train.py", ["--file-list", "file_list.json", "--epochs", "1"])
```
