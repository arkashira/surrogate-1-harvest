# airship / ops

Below is the **single, merged implementation** that keeps every high-leverage idea from both proposals while eliminating duplication and hardening for production use.

It is organized for **immediate execution** on your Mac orchestrator and Lightning environment.

---

## 1. Core idea (why this wins)
- **Eliminates HF API 429s**: data loading uses **CDN-only** URLs (`resolve/main/...`) with no Authorization header.
- **Saves Lightning quota**: reuse running Studios instead of recreate; idle-timeout safe.
- **Schema-safe**: projects to `{prompt, response}` at parse time.
- **Cron-safe**: proper shebang, env, and executable bits.
- **≤2h**: ~90 min implementation + 30 min verification.

## 2. Implementation

### 2.1. CDN-bypass file lister (Mac orchestration)
Single paginated call per date folder; outputs JSON consumed by training.

```bash
# /opt/axentx/airship/scripts/list_hf_date_folder.sh
#!/usr/bin/env bash
set -euo pipefail

REPO="${1:-datasets/your-org/surrogate-mirror}"
DATE="${2:-$(date +%Y-%m-%d)}"
OUT="${3:-./file-list-${DATE}.json}"

python3 - "$REPO" "$DATE" "$OUT" <<'PY'
import json, os, sys
from huggingface_hub import HfApi

repo_id, date_folder = sys.argv[1], sys.argv[2]
api = HfApi()
# Non-recursive: one folder = one paginated call (cheap)
files = api.list_repo_tree(repo_id, path=date_folder, recursive=False)
file_list = [f.rfilename for f in files if f.rfilename.endswith('.parquet')]
with open(sys.argv[3], 'w') as f:
    json.dump({"date": date_folder, "files": file_list, "repo": repo_id}, f, indent=2)
print(f"Listed {len(file_list)} files -> {sys.argv[3]}")
PY

chmod +x "$0"
```

### 2.2. Lightning training script (CDN-only ingestion)
No HF API calls during training; minimal model stub included.

```python
# /opt/axentx/airship/surrogate/train_cdn.py
import json, io, os, sys
import pyarrow.parquet as pq
import requests
import torch
from torch.utils.data import Dataset, DataLoader
from lightning import LightningModule, Trainer
from lightning.pytorch.cli import LightningCLI

class CDNParquetDataset(Dataset):
    def __init__(self, file_list_path):
        with open(file_list_path) as f:
            cfg = json.load(f)
        self.base = f"https://huggingface.co/datasets/{cfg['repo']}/resolve/main/{cfg['date']}"
        self.files = cfg["files"]
        self._cache = {}

    def __len__(self):
        return len(self.files)

    def __getitem__(self, idx):
        fname = self.files[idx]
        if fname in self._cache:
            return self._cache[fname]

        url = f"{self.base}/{fname}"
        resp = requests.get(url, timeout=30)
        resp.raise_for_status()
        table = pq.read_table(io.BytesIO(resp.content))

        # Schema-safe projection
        def _col_or_empty(key):
            if key in table.column_names:
                return table[key].to_pylist()
            return [""] * len(table)

        batch = {
            "prompt": _col_or_empty("prompt"),
            "response": _col_or_empty("response"),
        }
        self._cache[fname] = batch
        return batch

class SurrogateDataModule:
    def __init__(self, file_list):
        self.ds = CDNParquetDataset(file_list)

    def train_dataloader(self):
        return DataLoader(self.ds, batch_size=8, shuffle=True, num_workers=4)

# Minimal model stub (replace with your Surrogate model)
class SurrogateModel(LightningModule):
    def __init__(self):
        super().__init__()
        # Your model definition here
        self.layer = torch.nn.Linear(1, 1)

    def training_step(self, batch, batch_idx):
        # Your training logic
        return sum(len(p) for p in batch["prompt"]) * 0.0  # dummy

    def configure_optimizers(self):
        return torch.optim.Adam(self.parameters(), lr=1e-4)

if __name__ == "__main__":
    cli = LightningCLI(
        SurrogateModel,
        SurrogateDataModule,
        run=False,
        save_config_callback=None,
    )
```

### 2.3. Studio reuse launcher (idle-timeout safe)
Reuses running Studio; restarts gracefully if stopped.

```python
# /opt/axentx/airship/surrogate/launch_studio.py
#!/usr/bin/env python3
import os, sys
from lightning import Studio, Teamspace, Machine

def get_or_create_studio(name="surrogate-training", machine=Machine.L40S):
    studios = Teamspace.studios
    for s in studios:
        if s.name == name:
            if s.status == "running":
                print(f"Reusing running studio: {s.id}")
                return s
            else:
                print(f"Restarting stopped studio: {s.id}")
                s.start(machine=machine)
                return s
    print(f"Creating new studio: {name}")
    return Studio(
        name=name,
        machine=machine,
        create_ok=True,
    )

if __name__ == "__main__":
    machine = Machine.L40S
    studio = get_or_create_studio(machine=machine)

    file_list = "./file-list-2026-05-03.json"  # generated by list_hf_date_folder.sh
    job = studio.run(
        [
            "python3", "train_cdn.py",
            "--data.init", f"SurrogateDataModule('{file_list}')",
            "--trainer.max_epochs", "1",
            "--trainer.devices", "1",
        ],
        wait_until="running",
    )
    print(f"Job started: {job}")
```
