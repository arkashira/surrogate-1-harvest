# workio / backend

## 1. Diagnosis

- **Missing idempotency on LINE webhook ingestion**: duplicate `X-Line-Signature` replays (network retries) can double-apply clock-in/out events and leave/OT transitions with no de-dup guard.
- **No strict signature verification timing**: verification should occur before any side-effect (DB write, notification) and must include replay-window checks to prevent delayed replays.
- **Race-prone attendance punch logic**: concurrent webhook deliveries for the same user can produce two “in” records when the second arrives before the first commits (no row-level lock or unique constraint per day/open session).
- **Silent notification failures**: LINE notify/reply calls are fire-and-forget; transient 429/5xx are not retried with backoff, so users don’t receive confirmations or error messages.
- **Missing structured audit trail for compliance**: punches and leave/OT transitions lack immutable event rows (who/what/when) which are required for labor audits and debugging replay issues.

## 2. Proposed change

- **File**: `server/src/routes/line/webhook.ts` (or equivalent route handling `POST /webhook/line`)
- **Scope**: add idempotency + signature verification middleware, unique constraint/index on attendance punches (`user_id, date, session_type`), retry-with-backoff for outbound LINE replies, and append-only event log table.

-- Idempotency key for webhook events (store raw sig or hash)
CREATE TABLE IF NOT EXISTS line_webhook_events (
  id              BIGSERIAL PRIMARY KEY,
  idempotency_key TEXT NOT NULL UNIQUE,   -- X-Line-Signature or derived hash
  body_hash       TEXT NOT NULL,          -- sha256(raw_body) for strict replay check
  received_at     TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  processed_at    TIMESTAMPTZ,
  event_type      TEXT,
  payload         JSONB
);

-- Prevent duplicate open sessions per user per day
CREATE UNIQUE INDEX IF NOT EXISTS idx_attendance_open_session
ON attendance (user_id, date, session_type)
WHERE session_type = 'in' AND clock_out_at IS NULL;

-- Immutable event log for punches/leave/OT transitions
CREATE TABLE IF NOT EXISTS workio_events (
  id          BIGSERIAL PRIMARY KEY,
  entity_type TEXT NOT NULL,     -- 'attendance','leave','ot'
  entity_id   BIGINT NOT NULL,
  action      TEXT NOT NULL,     -- 'clock_in','clock_out','request','approve',...
  actor_id    BIGINT,            -- user who caused it (or null for system)
  payload     JSONB,
  created_at  TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

COMMIT;

### 3.2 Signature + idempotency middleware

```ts
// server/src/middleware/lineSignature.ts
import crypto from 'crypto';
import { Request, Response, NextFunction } from 'express';
import { pool } from '../db';

const CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET!;

function verifySignature(rawBody: string, signature: string): boolean {
  const expected = crypto
    .createHmac('sha256', CHANNEL_SECRET)
    .update(rawBody, 'utf8')
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

export async function lineWebhookGuard(
  req: Request,
  res: Response,
  next: NextFunction
) {
  const signature = req.get('X-Line-Signature');
  const rawBody = req.rawBody as string;

  if (!signature || !rawBody) {
    return res.status(400).json({ error: 'Missing signature or body' });
  }

  if (!verifySignature(rawBody, signature)) {
    return res.status(401).json({ error: 'Invalid signature' });
  }

  const bodyHash = crypto.createHash('sha256').update(rawBody, 'utf8').digest('hex');
  const idempotencyKey = signature; // or hash of (signature+bodyHash)

  // Reject replays within window (e.g., 5min)
  const { rows } = await pool.query(
    `SELECT 1 FROM line_webhook_events
     WHERE idempotency_key = $1 OR body_hash = $2
     AND received_at > NOW() - INTERVAL '5 minutes'`,
    [idempotencyKey, bodyHash]
  );

  if (rows.length > 0) {
    // Idempotent success: treat as already processed
    return res.status(200).json({ ok: true, reason: 'duplicate' });
  }

  // Store receipt immediately (before side-effects)
  await pool.query(
    `INSERT INTO line_webhook_events (idempotency_key, body_hash, event_type, payload)
     VALUES ($1, $2, $3, $4)`,
    [idempotencyKey, bodyHash, 'webhook', req.body]
  );

  // Attach for downstream handlers
  (req as any).lineIdempotencyKey = idempotencyKey;
  next();
}
```

### 3.3 Safe attendance handler with lock + event log

```ts
// server/src/routes/line/webhook.ts (excerpt)
import { pool } from '../../db';
import { lineWebhookGuard } from '../../middleware/lineSignature';
import { retryLineReply } from '../../lib/lineRetry';

async function handleClockIn(userId: number, timestamp: Date) {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // Lock any open session for this user to prevent races
    const { rows } = await client.query(
      `SELECT id FROM attendance
       WHERE user_id = $1 AND session_type = 'in' AND clock_out_at IS NULL
       FOR UPDATE SKIP LOCKED`,
      [userId]
    );

    if (rows.length > 0) {
      // Already clocked in — idempotent no-op
      await client.query('COMMIT');
      return { ok: true, message: 'Already clocked in' };
    }

    const date = new Date().toISOString().split('T')[0];
    await client.query(
      `INSERT INTO attendance (user_id, date, session_type, clock_in_at)
       VALUES ($1, $2, 'in', $3)`,
      [userId, date, timestamp]
    );

    await client.query(
      `INSERT INTO workio_events (entity_type, entity_id, action, actor_id, payload)
       VALUES ('attendance', currval('attendance_id_seq'), 'clock_in', $1, $2)`,
      [userId, { timestamp }]
    );

    await client.query('COMMIT');
    return { ok: true };
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  } finally {
    client.release();
  }
}

// Route
router.post(
  '/webhook/line',
  // capture raw body before json parser (ensure rawBody populated)
  lineWebhookGuard,
  async (req, res) => {
    try {
      const events = req.body.events || [];
      for (const ev of events) {
        if (ev.type === 'message' && ev.message.type === 'text') {
          const text = ev.message.text.toLowerCase();
          const userId = ev.source.userId;

          if (text.includes('in') || text.includes('clock in')) {
            await handleClockIn(userId, new Date());
            await retryLineReply(ev.replyToken, 'คุณได้ลงเวลาทำงานเรียบร้อยแล้ว');
          } else if (text.includes('out') || text.includes('clock out')) {
            // similar safe handler for clock-out
          }
        }
      }
      res.status(200).json({ ok: true });
    } catch (err) {
      console.error('Webhook processing failed', err);
      res.status(500).json({ error: 'Processing failed' });
    }
  }
);
```
