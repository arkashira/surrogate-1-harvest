# airship / backend

## Final Implementation Plan — `airship discover` (backend)

### Goal
Ship a single, zero-config, cron-safe orchestrator (`airship discover`) that:
1. Runs `granite-business-research.sh` (if present) and tags outputs `#business-research #knowledge-rag #graph`.
2. Executes `knowledge-rag` to query top hub and related docs (tag `#knowledge-rag #graph #hub`).
3. Produces a **CDN-only training manifest** (JSON) for Surrogate training (bypasses HF API rate limits).
4. Is idempotent, logs to `logs/discover/`, and exits non-zero only on fatal errors (cron-safe).

## File: `/opt/axentx/airship/bin/airship`

```bash
#!/usr/bin/env bash
# airship — orchestration CLI (backend)
# Cron-safe, zero-config entrypoint.
set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
PROJECT_ROOT="$(cd "$SCRIPT_DIR/.." && pwd)"
export AIRSHIP_ROOT="$PROJECT_ROOT"
export SHELL=/bin/bash

log() {
  echo "[$(date -u +'%Y-%m-%dT%H:%M:%SZ')] [airship] $*" >&2
}

fail() {
  log "ERROR: $*"
  exit 1
}

cmd_discover() {
  log "Starting airship discover (backend orchestrator)"
  python3 "$PROJECT_ROOT/airship/discover.py" "$@"
  local rc=$?
  if [[ $rc -eq 0 ]]; then
    log "airship discover completed successfully"
  else
    log "airship discover exited with code $rc"
  fi
  return $rc
}

case "${1:-}" in
  discover)
    shift
    cmd_discover "$@"
    ;;
  *)
    echo "Usage: $0 {discover}" >&2
    exit 1
    ;;
esac
```

Make executable:
```bash
chmod +x /opt/axentx/airship/bin/airship
```

#!/usr/bin/env python3
"""
airship discover — backend orchestrator
- Runs granite-business-research.sh (if present) and tags outputs.
- Executes knowledge-rag and reviews top hub (MOC) for contextual insights.
- Produces CDN-only training manifest (JSON) for Surrogate training.

import sys

# ----------
# Config
# ----------
PROJECT_ROOT = Path(os.environ.get("AIRSHIP_ROOT", Path(__file__).parent.parent)).resolve()
RESEARCH_SCRIPT = PROJECT_ROOT / "airship" / "granite-business-research.sh"
KNOWLEDGE_RAG_SCRIPT = PROJECT_ROOT / "scripts" / "knowledge-rag"
OUTPUT_DIR = PROJECT_ROOT / "outputs" / "discover"
LOG_DIR = PROJECT_ROOT / "logs" / "discover"
MANIFEST_DIR = PROJECT_ROOT / "manifests"
HF_DATASET_REPO = os.environ.get("HF_DATASET_REPO", "datasets/your-org/your-dataset")
HF_DATE_FOLDER = os.environ.get("HF_DATE_FOLDER", datetime.now(timezone.utc).strftime("%Y-%m-%d"))

# ----------
# Logging
# ----------
def log(msg: str, level: str = "INFO") -> None:
    ts = datetime.now(timezone.utc).strftime("%Y-%m-%dT%H:%M:%SZ")
    line = f"[{ts}] [{level}] {msg}"
    print(line, file=sys.stderr, flush=True)
    # Also append to log file
    LOG_DIR.mkdir(parents=True, exist_ok=True)
    log_file = LOG_DIR / f"discover-{datetime.now(timezone.utc).strftime('%Y%m%d')}.log"
    with log_file.open("a") as f:
        f.write(line + "\n")

def fail(msg: str, exc: Optional[Exception] = None) -> None:
    log(f"FAILED: {msg}", "ERROR")
    if exc:
        traceback.print_exc(file=sys.stderr)
    sys.exit(1)

# ----------
# Helpers
# ----------
def ensure_dir(p: Path) -> Path:
    p.mkdir(parents=True, exist_ok=True)
    return p

def run_cmd(cmd: List[str], cwd: Optional[Path] = None, env: Optional[Dict[str, str]] = None) -> str:
    log(f"Running: {' '.join(cmd)}")
    result = subprocess.run(
        cmd,
        cwd=cwd or PROJECT_ROOT,
        env={**os.environ, **(env or {})},
        capture_output=True,
        text=True,
    )
    if result.returncode != 0:
        log(f"STDOUT: {result.stdout}", "DEBUG")
        log(f"STDERR: {result.stderr}", "DEBUG")
        fail(f"Command failed (rc={result.returncode}): {' '.join(cmd)}")
    if result.stdout.strip():
        log(f"STDOUT: {result.stdout.strip()}", "DEBUG")
    return result.stdout.strip()

def tag_outputs(base_dir: Path, tags: List[str]) -> None:
    """Write a .tags file alongside outputs for traceability."""
    tag_file = base_dir / ".tags"
    existing = []
    if tag_file.exists():
        try:
            existing = json.loads(tag_file.read_text().strip())
        except Exception:
            pass
    merged = list(dict.fromkeys(existing + tags))
    tag_file.write_text(json.dumps(merged, indent=2) + "\n")
    log(f"Tagged outputs at {base_dir} with {tags}")

# ----------
# Steps
# ----------
def run_granite_business_research() -> None:
    """Run granite-business-research.sh if present; tag outputs."""
    if not RESEARCH_SCRIPT.exists():
        log(f"Research script not found: {RESEARCH_SCRIPT}. Skipping.")
        return

    if not os.access(str(RESEARCH_SCRIPT), os.X_OK):
        log(f"Making research script executable: {RESEARCH_SCRIPT}")
        RESEARCH_SCRIPT.chmod(0o755)

    # Run via bash for cron-safety
    run_cmd(["/bin/bash", str(RESEARCH_SCRIPT)])

    # Tag outputs
    ts = datetime.now(timezone.utc).strftime("%Y%m%dT%H%M%SZ")
    research_out = OUTPUT_DIR / f"business-research-{ts}"
    ensure_dir(research_out)
    tag_outputs(research_out, ["#business-research", "#knowledge-rag", "#graph"])
    log("Granite business research completed and tagged.")

def run_knowledge_rag() -> None:
    """Execute knowledge-rag to query top hub and related docs."""
    if not KNOWLEDGE_RAG_SCRIPT.exists():
        log(f"knowledge-rag not found: {KNOWLEDGE_RAG_SCRIPT}. Skipping.")
        return

    if not os.access(str(KNOWLEDGE_RAG_SCRIPT), os.X_OK):
        KNOWLEDGE_RAG_SCRIPT.chmod(0o755)

    run_cmd(["/bin/bash", str(KNOWLEDGE_RAG_SCRIPT)])

    ts = datetime.now(timezone.utc).strftime("%Y%m%dT%H%M%SZ")
    rag_out = OUTPUT_DIR / f"knowledge-rag-{ts}"
    ensure_dir(rag_out)
    tag_outputs(rag_out, ["#knowledge-rag", "#graph", "#hub"])
    log("knowledge-rag completed and tagged.")
