# workio / quality

## 1. Diagnosis

- No end-to-end idempotency for clock-in/out: retries or double-taps can create duplicate punches with different timestamps, corrupting attendance records.
- Missing optimistic UI on punch actions: UI waits for round-trip before reflecting state, causing perceived latency and encouraging repeated taps.
- Punch state is list-local only: after a successful punch, the UI doesn’t update the current session status (e.g., “Clocked in @ 09:03”) consistently across views.
- No client-side in-flight lock: concurrent clicks or background retries can issue multiple requests before the first resolves.
- No graceful rollback on server rejection: failed punches leave the UI in an indeterminate state (spinner hangs or disappears without clear error).

## 2. Proposed change

File: `workio/src/components/PunchButton.tsx`  
Scope: add idempotency key generation, optimistic UI, in-flight lock, and rollback on failure. Keep API contract unchanged (POST `/api/punches` with `{ type, latitude, longitude }`). Expose minimal props: `onPunchSuccess` optional callback.

## 3. Implementation

// workio/src/components/PunchButton.tsx
import React, { useState, useCallback, useRef } from 'react';
import { useMutation, useQueryClient } from '@tanstack/react-query';
import { useToast } from '@/hooks/use-toast';

type PunchType = 'CLOCK_IN' | 'CLOCK_OUT';

interface PunchPayload {
  type: PunchType;
  latitude: number;
  longitude: number;
}

interface PunchButtonProps {
  onPunchSuccess?: (punch: any) => void;
}

export function PunchButton({ onPunchSuccess }: PunchButtonProps) {
  const queryClient = useQueryClient();
  const { toast } = useToast();
  const inFlightRef = useRef(false);
  const idempotencyKeyRef = useRef<string | null>(null);

  const punchMutation = useMutation({
    mutationFn: async (payload: PunchPayload) => {
      const res = await fetch('/api/punches', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          ...payload,
          // send idempotency key so server can dedupe
          idempotencyKey: idempotencyKeyRef.current,
        }),
      });

      if (!res.ok) {
        const err = await res.json().catch(() => ({}));
        throw new Error(err.message || 'Punch failed');
      }

      return res.json();
    },
    onMutate: async (newPunch) => {
      // generate idempotency key once per user intent
      idempotencyKeyRef.current = `${Date.now()}-${Math.random().toString(36).slice(2, 9)}`;

      // cancel outgoing refetches for punches list/current session so we stay optimistic
      await queryClient.cancelQueries({ queryKey: ['punches'] });
      await queryClient.cancelQueries({ queryKey: ['currentPunch'] });

      // snapshot previous value
      const previousPunches = queryClient.getQueryData(['punches']);
      const previousCurrent = queryClient.getQueryData(['currentPunch']);

      // optimistic punch item
      const optimisticPunch = {
        id: `optimistic-${idempotencyKeyRef.current}`,
        type: newPunch.type,
        timestamp: new Date().toISOString(),
        latitude: newPunch.latitude,
        longitude: newPunch.longitude,
        _optimistic: true,
      };

      // optimistically update list
      queryClient.setQueryData(['punches'], (old: any[] = []) => [
        optimisticPunch,
        ...old,
      ]);

      // optimistically set current punch
      queryClient.setQueryData(['currentPunch'], optimisticPunch);

      return { previousPunches, previousCurrent };
    },
    onError: (err, _newPunch, context) => {
      // rollback
      if (context?.previousPunches) {
        queryClient.setQueryData(['punches'], context.previousPunches);
      }
      if (context?.previousCurrent) {
        queryClient.setQueryData(['currentPunch'], context.previousCurrent);
      } else {
        queryClient.removeQueries({ queryKey: ['currentPunch'] });
      }

      toast({
        title: 'ไม่สามารถลงเวลาได้',
        description: err instanceof Error ? err.message : 'เกิดข้อผิดพลาด',
        variant: 'destructive',
      });
    },
    onSettled: () => {
      // always refetch after settle to sync with server
      queryClient.invalidateQueries({ queryKey: ['punches'] });
      queryClient.invalidateQueries({ queryKey: ['currentPunch'] });
      inFlightRef.current = false;
      idempotencyKeyRef.current = null;
    },
  });

  const handlePunch = useCallback(async () => {
    if (inFlightRef.current) return;
    inFlightRef.current = true;

    if (!navigator.geolocation) {
      toast({
        title: 'ไม่พบ GPS',
        description: 'โปรดเปิดใช้งานตำแหน่งเพื่อลงเวลา',
        variant: 'destructive',
      });
      inFlightRef.current = false;
      return;
    }

    navigator.geolocation.getCurrentPosition(
      async (pos) => {
        const { latitude, longitude } = pos.coords;
        const type: PunchType =
          queryClient.getQueryData(['currentPunch'])?.type === 'CLOCK_IN'
            ? 'CLOCK_OUT'
            : 'CLOCK_IN';

        punchMutation.mutate({ type, latitude, longitude });
      },
      (err) => {
        toast({
          title: 'ไม่สามารถดึงตำแหน่ง',
          description: 'โปรดอนุญาตการเข้าถึงตำแหน่ง',
          variant: 'destructive',
        });
        inFlightRef.current = false;
      },
      { enableHighAccuracy: true, timeout: 10000, maximumAge: 30000 }
    );
  }, [punchMutation, queryClient, toast]);

  const current = queryClient.getQueryData(['currentPunch']);
  const isClockedOn = current?.type === 'CLOCK_IN';
  const isPending = punchMutation.isPending || inFlightRef.current;

  return (
    <button
      onClick={handlePunch}
      disabled={isPending}
      className={`
        inline-flex items-center justify-center rounded-full px-8 py-4 text-lg font-semibold
        shadow-lg transition-all active:scale-95 disabled:opacity-60 disabled:cursor-not-allowed
        ${
          isClockedOn
            ? 'bg-red-600 hover:bg-red-700 text-white'
            : 'bg-emerald-600 hover:bg-emerald-700 text-white'
        }
      `}
    >
      {isPending ? (
        <span className="flex items-center gap-2">
          <svg className="h-5 w-5 animate-spin" viewBox="0 0 24 24">
            <circle
              className="opacity-25"
              cx="12"
              cy="12"
              r="10"
              stroke="currentColor"
              strokeWidth="4"
            />
            <path
              className="opacity-75"
              fill="currentColor"
              d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4z"
            />
          </svg>
          กำลังลงเวลา...
        </span>
      ) : (
        <span className="flex items-center gap-2">
          {isClockedOn ? (
            <>
              <svg
                className="h-5 w-5"
                fill="none"
                stroke="currentColor"
                viewBox="0 0 24 24"
              >
                <path
                  strokeLinecap="round"
                  strokeLinejoin="round"
                  strokeWidth={2}
                  d="M18.364 5.636l-9.9 9.9-4.243-4.243"
                />
