### 1. Export HF file list (Mac orchestration side)

```python
# scripts/export_hf_filelist.py
#!/usr/bin/env python3
"""
Export a deterministic list of HF dataset files for a given date folder.
Run from Mac after rate-limit window clears. Embed this JSON in training.
"""
import json
import os
from datetime import datetime
from huggingface_hub import HfApi

HF_REPO = "datasets/axentx/surrogate-mirror"
DATE_FOLDER = datetime.utcnow().strftime("%Y-%m-%d")  # or pass via CLI
OUTPUT = "filelist.json"

def main():
    api = HfApi()
    entries = api.list_repo_tree(
        repo_id=HF_REPO,
        path=DATE_FOLDER,
        repo_type="dataset",
        recursive=False,
    )

    files = sorted([e.rfilename for e in entries if e.type == "file"])

    filelist = [
        {
            "filename": f,
            "cdn_url": f"https://huggingface.co/datasets/{HF_REPO}/resolve/main/{f}",
        }
        for f in files
    ]

    out_path = os.path.join(os.path.dirname(__file__), "..", OUTPUT)
    with open(out_path, "w") as fp:
        json.dump(filelist, fp, indent=2)

    print(f"Exported {len(filelist)} files to {out_path}")

if __name__ == "__main__":
    main()
```

### 2. Surrogate training: CDN-only data loader

```python
# surrogate/train.py  (excerpt)
import json
import os
import pyarrow.parquet as pq
import requests
from torch.utils.data import IterableDataset

class CDNParquetDataset(IterableDataset):
    def __init__(self, filelist_path="filelist.json", columns=("prompt", "response")):
        with open(filelist_path) as f:
            self.files = json.load(f)
        self.columns = columns

    def _stream_parquet(self, cdn_url):
        resp = requests.get(cdn_url, timeout=60, stream=True)
        resp.raise_for_status()
        with pq.ParquetFile(resp.content) as pf:
            for batch in pf.iter_batches(columns=self.columns, batch_size=512):
                yield from batch.to_pylist()

    def __iter__(self):
        for item in self.files:
            yield from self._stream_parquet(item["cdn_url"])
```

### 3. Lightning Studio guard + reuse + idle-stop resilience

```python
# surrogate/lightning_launcher.py
import time
from lightning_sdk import Teamspace, Studio, Machine

TEAMSPACE = "axentx"
STUDIO_NAME = "surrogate-trainer"
MACHINE = Machine.L40S

def get_or_create_studio():
    teamspace = Teamspace(TEAMSPACE)
    for s in teamspace.studios:
        if s.name == STUDIO_NAME and s.status == "running":
            print(f"Reusing running studio: {s.name}")
            return s

    studio = teamspace.studios.create(
        name=STUDIO_NAME,
        machine=MACHINE,
        create_ok=True,
    )
    print(f"Created studio: {studio.name}")
    return studio

def wait_for_ready(studio, timeout=300):
    elapsed = 0
    while elapsed < timeout:
        studio.refresh()
        if studio.status == "running":
            return True
        time.sleep(5)
        elapsed += 5
    return False

def run_training_job(target_script="train.py", args=None):
    args = args or []
    studio = get_or_create_studio()

    if studio.status != "running":
        print(f"Studio stopped ({studio.status}). Restarting...")
        studio.start(machine=MACHINE)
        if not wait_for_ready(studio):
            raise RuntimeError("Studio failed to start")

    studio.refresh()
    if studio.status != "running":
        studio.start(machine=MACHINE)
        wait_for_ready(studio)

    run = studio.run(
        command=f"python {target_script} {' '.join(args)}",
        environment={"SHELL": "/bin/bash"},
    )
    print(f"Started run {run.id}")
    return run

if __name__ == "__main__":
    run_training_job(target_script="train.py", args=["--filelist", "filelist.json"])
```

### 4. Arkship UI: training status + auto-reconnect resilience + error display

```vue
<!-- arkship/ui/src/components/TrainingStatus.vue -->
<template>
  <div class="training-status">
    <div :class="['status-badge', statusClass]">{{ statusText }}</div>
    <small v-if="lastError" class="error-msg">{{ lastError }}</small>
  </div>
</template>

<script>
export default {
  data() {
    return {
      status: "idle", // idle | running | stopped | error
      lastError: null,
      pollInterval: null,
    };
  },
  computed: {
    statusClass() {
      return {
        idle: "bg-gray",
        running: "bg-green",
        stopped: "bg-yellow",
        error: "bg-red",
      }[this.status];
    },
    statusText() {
      return {
        idle: "Idle",
        running: "Training",
        stopped: "Stopped (auto-restart)",
        error: "Error",
      }[this.status];
    },
  },
  mounted() {
    this.pollStatus();
    this.pollInterval = setInterval(this.pollStatus, 15000);
  },
  beforeDestroy() {
    clearInterval(this.pollInterval);
  },
  methods: {
    async pollStatus() {
      try {
        const res = await fetch("/api/v1/surrogate/training/status");
        if (!res.ok) throw new Error(`HTTP ${res.status}`);
        const data = await res.json();
        this.status = data.status || "idle";
        this.lastError = data.error || null;
      } catch (err) {
        this.status = "error";
        this.lastError = err.message;
      }
    },
  },
};
</script>
