```js
/**
 * HF CDN-bypass utilities
 * Avoids /api/ rate limits by using /resolve/main/ CDN endpoints
 */

export const buildCdnUrl = (repo, path) => {
  const cleanRepo = repo.replace(/^datasets\//, '');
  return `https://huggingface.co/datasets/${cleanRepo}/resolve/main/${path}`;
};

export const fetchViaCdn = async (url, options = {}) => {
  const res = await fetch(url, {
    ...options,
    headers: { ...options.headers } // No Authorization
  });

  if (res.status === 429) {
    const retryAfter = 360;
    console.warn(`HF CDN 429, waiting ${retryAfter}s`);
    await new Promise(r => setTimeout(r, retryAfter * 1000));
    return fetchViaCdn(url, options);
  }

  if (!res.ok) throw new Error(`CDN fetch failed: ${res.status} ${res.statusText}`);
  return res;
};

export const listRepoFilesCached = async (repo, folder = '') => {
  const cacheKey = `hf-tree-${repo}-${folder}`;
  const cached = localStorage.getItem(cacheKey);
  if (cached) return JSON.parse(cached);

  const res = await fetch(`/api/hf/tree?repo=${encodeURIComponent(repo)}&folder=${encodeURIComponent(folder)}`);
  if (!res.ok) throw new Error('Failed to list repo files');

  const tree = await res.json();
  localStorage.setItem(cacheKey, JSON.stringify(tree));
  return tree;
};
```

### `src/utils/parquet-projection.js`
```js
/**
 * Safe parquet reader that projects only {prompt, response}
 * Avoids pyarrow.CastError from mixed-schema files
 */

export const projectParquetToQA = async (arrayBuffer) => {
  try {
    const arrow = await import('apache-arrow');
    const table = arrow.Table.from(new Uint8Array(arrayBuffer));

    const prompts = [];
    const responses = [];

    const promptCol = table.schema.fields.find(f =>
      ['prompt', 'instruction', 'input', 'question'].includes(f.name.toLowerCase())
    );
    const responseCol = table.schema.fields.find(f =>
      ['response', 'output', 'answer', 'completion'].includes(f.name.toLowerCase())
    );

    if (!promptCol || !responseCol) {
      throw new Error('Required prompt/response columns not found');
    }

    for (let i = 0; i < table.numRows; i++) {
      const row = table.get(i);
      prompts.push(row[promptCol.name]);
      responses.push(row[responseCol.name]);
    }

    return { prompts, responses };
  } catch (err) {
    console.warn('Parquet projection failed, trying JSON fallback:', err);
    return projectJsonToQA(arrayBuffer);
  }
};

const projectJsonToQA = (buffer) => {
  const text = new TextDecoder().decode(buffer);
  const lines = text.trim().split('\n');
  const prompts = [];
  const responses = [];

  for (const line of lines) {
    try {
      const obj = JSON.parse(line);
      prompts.push(obj.prompt || obj.instruction || obj.input || obj.question);
      responses.push(obj.response || obj.output || obj.answer || obj.completion);
    } catch {
      // Skip malformed lines
    }
  }

  return { prompts, responses };
};
```

### `src/composables/useLightningStudio.js`
```js
import * as Lightning from 'lightningai';

export const useLightningStudio = () => {
  const getOrCreateStudio = async (name, machine = 'L40S') => {
    const studios = await Lightning.Teamspace.studios();
    const existing = studios.find(s => s.name === name && s.status === 'Running');
    if (existing) {
      console.log(`Reusing running studio: ${name}`);
      return existing;
    }

    const stopped = studios.find(s => s.name === name && s.status === 'Stopped');
    if (stopped) {
      console.log(`Restarting stopped studio: ${name}`);
      await stopped.start({ machine });
      return stopped;
    }

    console.log(`Creating new studio: ${name}`);
    return await Lightning.Studio.create({ name, machine, createOk: true });
  };

  const ensureStudioRunning = async (studio) => {
    if (studio.status !== 'Running') {
      console.log(`Studio ${studio.name} not running, restarting...`);
      await studio.start({ machine: studio.machine });
    }
    return studio;
  };

  return { getOrCreateStudio, ensureStudioRunning };
};
```
