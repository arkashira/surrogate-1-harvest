# workio / design

## Final Synthesis — One Correct, Actionable Solution

**Core principle:**  
Accept LINE webhooks only when authenticated, deduplicated, and applied atomically. Return `200` **after** all side-effects are durably committed; otherwise return `4xx/5xx` so LINE retries safely.

```ts
import { Router, Request, Response, NextFunction } from 'express';
import crypto from 'crypto';
import { pool } from '../../db';
import { redis } from '../../lib/redis'; // optional; can be null

const router = Router();
const LINE_CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';
if (!LINE_CHANNEL_SECRET) {
  throw new Error('LINE_CHANNEL_SECRET is required');
}
const DEDUPE_TTL_SECONDS = 300; // 5 min sliding window

/* -- Signature verification -- */
function verifyLineSignature(rawBody: string, signature: string): boolean {
  if (!signature || !rawBody) return false;
  const expected = crypto
    .createHmac('sha256', LINE_CHANNEL_SECRET)
    .update(rawBody, 'utf8')
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

/* -- Deduplication -- */
async function isDuplicate(eventId: string): Promise<boolean> {
  // Fast path: Redis SET NX with expiry
  if (redis) {
    const ok = await redis.set(`line:evt:${eventId}`, '1', { PX: DEDUPE_TTL_SECONDS * 1000, NX });
    return !ok;
  }

  // Fallback: DB unique constraint
  try {
    await pool.query(
      `INSERT INTO line_event_dedupe (event_id, expires_at)
       VALUES ($1, NOW() + ($2 || ' seconds')::interval)`,
      [eventId, DEDUPE_TTL_SECONDS]
    );
    return false;
  } catch (err: any) {
    if (err.code === '23505') return true; // unique_violation
    throw err;
  }
}

/* -- Idempotent clock transition -- */
async function applyClockEvent(
  tx: any,
  employeeId: string,
  eventType: 'clock_in' | 'clock_out',
  occurredAt: Date
): Promise<void> {
  // Lock employee row to serialize concurrent webhooks
  const emp = await tx.query(
    `SELECT id, last_clock_state, last_clock_at
     FROM employees
     WHERE id = $1
     FOR UPDATE`,
    [employeeId]
  );
  if (emp.rows.length === 0) {
    throw new Error('Employee not found');
  }

  const row = emp.rows[0];

  // Idempotency: ignore duplicate within tolerance (same state, ~same minute)
  if (
    row.last_clock_state === eventType &&
    row.last_clock_at &&
    Math.abs(row.last_clock_at.getTime() - occurredAt.getTime()) < 60_000
  ) {
    return;
  }

  // Insert attendance record
  await tx.query(
    `INSERT INTO attendance_records (employee_id, event_type, recorded_at, line_event_id)
     VALUES ($1, $2, $3, $4)`,
    [employeeId, eventType, occurredAt, null] // line_event_id can be populated if stored
  );

  // Update employee state
  await tx.query(
    `UPDATE employees
     SET last_clock_state = $1, last_clock_at = $2
     WHERE id = $3`,
    [eventType, occurredAt, employeeId]
  );
}

/* -- Webhook handler -- */
router.post('/line', async (req: Request, res: Response, next: NextFunction) => {
  const signature = req.headers['x-line-signature'] as string;
  const rawBody = typeof req.body === 'string' ? req.body : JSON.stringify(req.body);

  // 1) Authenticate
  if (!verifyLineSignature(rawBody, signature)) {
    return res.status(401).json({ error: 'Invalid signature' });
  }

  // 2) Parse events
  let payload: { events?: any[] };
  try {
    payload = JSON.parse(rawBody);
  } catch {
    return res.status(400).json({ error: 'Invalid JSON' });
  }
  const events = Array.isArray(payload.events) ? payload.events : [];

  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    for (const ev of events) {
      // Basic shape checks
      if (!ev || typeof ev !== 'object') continue;
      if (!ev.id || typeof ev.id !== 'string') continue;
      if (!ev.source?.userId || typeof ev.source.userId !== 'string') continue;
      if (!ev.timestamp || typeof ev.timestamp !== 'number') continue;

      // 3) Deduplicate by LINE event id
      if (await isDuplicate(ev.id)) {
        continue;
      }

      const userId = ev.source.userId;
      const occurredAt = new Date(ev.timestamp);

      // Resolve employee id from user id
      const emp = await client.query(
        `SELECT id FROM employees WHERE user_id = $1`,
        [userId]
      );
      if (emp.rows.length === 0) {
        // Optionally log unknown user but continue processing other events
        continue;
      }
      const employeeId = emp.rows[0].id;

      // 4) Map to clock action
      // Support postback data or text message
      const data = (ev.postback?.data || ev.message?.text || '').toString().trim().toLowerCase();

      if (data === 'clock_in' || data === 'clock_out') {
        await applyClockEvent(
          client,
          employeeId,
          data === 'clock_in' ? 'clock_in' : 'clock_out',
          occurredAt
        );
      }
    }

    // 5) Commit only after all side-effects succeed
    await client.query('COMMIT');
    res.status(200).json({ ok: true });
  } catch (err) {
    await client.query('ROLLBACK').catch(() => {});
    next(err);
  } finally {
    client.release();
  }
});

export default router;
```

### 3. Database prerequisites (run once)

```sql
-- Deduplication table (used if Redis unavailable)
CREATE TABLE IF NOT EXISTS line_event_dedupe (
  event_id TEXT PRIMARY KEY,
  expires_at TIMESTAMPTZ NOT NULL
);
CREATE INDEX IF NOT EXISTS idx_line_event_dedupe_expires
  ON line_event_dedupe (expires_at);

-- Ensure employees table supports state tracking
ALTER TABLE employees
  ADD COLUMN IF NOT EXISTS last_clock_state TEXT,
  ADD COLUMN IF NOT EXISTS last_clock_at TIMESTAMPTZ;

-- Attendance table (if not present)
CREATE TABLE IF NOT EXISTS attendance_records (
  id SERIAL PRIMARY KEY,
  employee_id INTEGER NOT NULL REFERENCES employees(id),
  event_type TEXT NOT NULL,
  recorded_at TIMESTAMPTZ NOT NULL,
  line_event_id TEXT,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
```
