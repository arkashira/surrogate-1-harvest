# workio / backend

## Final consolidated solution

**Diagnosis (merged, highest-confidence)**  
- No idempotency key allows LINE webhook redeliveries and concurrent deliveries to create duplicate punch rows.  
- Non-atomic read-then-insert/update causes race conditions that can produce two active punches for the same user/date.  
- Missing storage-layer guard (unique constraint/index) permits invalid double-clock-in or double-clock-out states.  
- No replay guard or deduplication window for LINE events; application logic alone is insufficient under concurrency.

**Schema changes** (`/opt/axentx/workio/server/src/db/schema.sql`)  
```sql
-- Idempotency table for LINE webhook deliveries (global, cross-handler)
CREATE TABLE IF NOT EXISTS line_deliveries (
  line_delivery_id VARCHAR(255) PRIMARY KEY,
  created_at       TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Keep only one active punch per user per date per type (active = clock_out_at IS NULL)
CREATE UNIQUE INDEX IF NOT EXISTS idx_one_active_punch_per_day
  ON punches (user_id, date, type)
  WHERE clock_out_at IS NULL;

-- Optional safety: ensure clock_out_at >= clock_in_at when present
ALTER TABLE punches
  ADD CONSTRAINT chk_punches_clock_out_after_in
  CHECK (clock_out_at IS NULL OR clock_out_at >= clock_in_at);
```

**Webhook handler** (`/opt/axentx/workio/server/src/routes/line.ts`)  
```ts
import { Router } from 'express';
import crypto from 'crypto';
import { pool } from '../db';

const router = Router();

function deriveLineDeliveryId(body: any, signature: string): string | null {
  const firstEvent = body.events?.[0];
  if (firstEvent?.deliveryId) return `line:${firstEvent.deliveryId}`;
  if (signature && body) {
    return `line:sha256:${crypto.createHash('sha256')
      .update(signature + JSON.stringify(body))
      .digest('hex')}`;
  }
  return null;
}

router.post('/webhook/line', async (req, res) => {
  const lineSignature = req.get('X-Line-Signature') || '';
  const body = req.body;
  const deliveryId = deriveLineDeliveryId(body, lineSignature);

  if (!deliveryId) {
    return res.status(400).json({ error: 'missing_delivery_id' });
  }

  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // Global idempotency check (fast fail for replays)
    const seen = await client.query(
      `SELECT 1 FROM line_deliveries WHERE line_delivery_id = $1`,
      [deliveryId]
    );
    if (seen.rows.length > 0) {
      await client.query('COMMIT');
      return res.status(200).json({ ok: true });
    }

    for (const ev of body.events || []) {
      if (ev.type !== 'message' || ev.message?.type !== 'text') continue;

      const userId = ev.source?.userId;
      const text = ev.message.text?.trim().toLowerCase();
      if (!userId || !['clock in', 'clock out'].includes(text)) continue;

      const today = new Date().toISOString().split('T')[0]; // YYYY-MM-DD
      const now = new Date();

      if (text === 'clock in') {
        // Atomic upsert: insert only if no active punch exists for today
        await client.query(
          `INSERT INTO punches (user_id, date, type, clock_in_at, created_at)
           VALUES ($1, $2, 'work', $3, NOW())
           ON CONFLICT (user_id, date, type)
           DO UPDATE SET clock_in_at = EXCLUDED.clock_in_at
           WHERE punches.clock_out_at IS NULL`,
          [userId, today, now]
        );
      } else if (text === 'clock out') {
        // Atomic transition: close the active punch for today
        await client.query(
          `UPDATE punches
           SET clock_out_at = $1
           WHERE user_id = $2 AND date = $3 AND type = 'work' AND clock_out_at IS NULL`,
          [now, userId, today]
        );
      }
    }

    // Record idempotency key after processing events
    await client.query(
      `INSERT INTO line_deliveries (line_delivery_id) VALUES ($1) ON CONFLICT DO NOTHING`,
      [deliveryId]
    );

    await client.query('COMMIT');
    res.status(200).json({ ok: true });
  } catch (err) {
    await client.query('ROLLBACK');
    console.error('LINE webhook error:', err);
    res.status(500).json({ error: 'internal' });
  } finally {
    client.release();
  }
});

export default router;
```
