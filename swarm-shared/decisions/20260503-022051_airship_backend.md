### 2) hf_ingest.py — deterministic file list + CDN URLs
```python
#!/usr/bin/env python3
"""
hf_ingest.py
- Lists repo tree ONCE (date folder)
- Produces file_list.json with CDN URLs
- Avoids load_dataset(streaming=True) for mixed-schema repos
"""
import json, os, hashlib
from pathlib import Path
from huggingface_hub import HfApi, hf_hub_download

HF_REPO = os.getenv("HF_DATASET_REPO", "axentx/surrogate-data")
DATE_FOLDER = os.getenv("INGEST_DATE", "data/train")  # e.g. data/train or batches/2026-04-29
OUT_DIR = Path(os.getenv("OUT_DIR", "surrogate/file_lists"))
OUT_DIR.mkdir(parents=True, exist_ok=True)

def deterministic_repo_for(slug: str, n_siblings: int = 5) -> str:
    """Hash slug -> sibling repo index to spread HF commit cap."""
    idx = int(hashlib.sha256(slug.encode()).hexdigest(), 16) % n_siblings
    return f"{HF_REPO}-sibling-{idx}" if idx > 0 else HF_REPO

def build_file_list():
    api = HfApi()
    # Single API call: non-recursive per folder to avoid pagination explosion
    entries = api.list_repo_tree(repo_id=HF_REPO, path=DATE_FOLDER, recursive=False)
    files = [e.rfilename for e in entries if e.rfilename.endswith((".parquet", ".jsonl", ".json"))]

    file_list = []
    for f in files:
        cdn_url = f"https://huggingface.co/datasets/{HF_REPO}/resolve/main/{f}"
        file_list.append({
            "repo": HF_REPO,
            "path": f,
            "cdn_url": cdn_url,
            "local_path": hf_hub_download(repo_id=HF_REPO, filename=f)  # optional: download once for validation
        })

    out_path = OUT_DIR / f"file_list_{DATE_FOLDER.replace('/', '_')}.json"
    out_path.write_text(json.dumps(file_list, indent=2))
    print(f"Wrote {len(file_list)} files -> {out_path}")
    return out_path

if __name__ == "__main__":
    build_file_list()
```

from datasets import Dataset

FILE_LIST = Path(os.getenv("FILE_LIST", "surrogate/file_lists/file_list_data_train.json"))
assert FILE_LIST.exists(), f"Missing {FILE_LIST}. Run hf_ingest.py first."

def load_cdn_or_local(file_infos):
    rows = []
    for info in file_infos:
        path = info["local_path"]
        if os.path.exists(path):
            df = pq.read_table(path).to_pandas()
        else:
            # fallback: use CDN URL with streaming (no auth, CDN tier)
            resp = requests.get(info["cdn_url"], timeout=30)
            resp.raise_for_status()
            df = pq.read_table(io.BytesIO(resp.content)).to_pandas()

        # Project to {prompt, response} only; ignore mixed schema cols
        for _, r in df.iterrows():
            prompt = r.get("prompt") or r.get("input") or r.get("text") or ""
            response = r.get("response") or r.get("output") or r.get("completion") or ""
            if prompt and response:
                rows.append({"prompt": str(prompt), "response": str(response)})
    return Dataset.from_pandas(pd.DataFrame(rows))

#!/usr/bin/env python3
"""
lightning_launcher.py
- Reuse running Studio; restart if stopped (idle-timeout resilience)
- Uses L40S on lightning-public-prod (free tier) or H200 on lightning-lambda-prod (paid)
"""
import os, time
from lightning_sdk import Studio, Teamspace, Machine

STUDIO_NAME = os.getenv("STUDIO_NAME", "surrogate-trainer")
MACHINE = Machine.L40S  # free tier fallback; switch to Machine.H200 if on lightning-lambda-prod

def get_or_create_studio():
    team = Teamspace()
    running = [s for s in team.studios if s.name == STUDIO_NAME and s.status == "Running"]
    if running:
        print(f"Reusing running studio: {STUDIO_NAME}")
        return running[0]

    stopped = [s for s in team.studios if s.name == STUDIO_NAME and s.status == "Stopped"]
    if stopped:
        s = stopped[0]
        print(f"Restarting stopped studio: {STUDIO_NAME}")
        s.start(machine=MACHINE)
        return s

    print(f"Creating new studio: {STUDIO_NAME}")
    return Studio.create(
        name=STUDIO_NAME,
        machine=MACHINE,
        # optional: set idle_timeout_minutes if supported by SDK
    )

def launch_training():
    studio = get_or_create_studio()
    # Run training script inside studio (adjust command to your entrypoint)
    job = studio.run(
        command="python surrogate/train.py",
        environment="lightning",
        # optional: specify compute/memory if needed
    )
    print(f"Launched training job: {job.id}")
    return job

if __name__ == "__main__":
    launch_training()
