# workio / backend

## Final Consolidated Implementation Plan

### Diagnosis (merged)
- **Root cause**: LINE webhook handler is not idempotent; retries on 5xx/timeouts create duplicate clock/leave/OT events.
- **Contributing factors**:
  - No transactional upsert or unique constraint to deduplicate retries.
  - Race conditions between concurrent retries can double-insert rows.
  - Mutations commit only after returning 200, but without an idempotency key retries create duplicates instead of no-ops.
  - Frontend relies on local state/polling; webhook-driven changes are invisible until refresh, causing divergence across tabs/devices.
  - Missing server-side idempotency increases support load and complicates audits.

### Scope
- **Primary file**: `workio/server/src/routes/webhook/line.ts` (webhook handler).
- **Supporting files**:
  - DB migration for unique constraint and `idempotency_key` column.
  - Repository helper for transactional upsert (`attendanceRepository.ts`).
  - Lightweight in-process pub/sub (`server/src/lib/events.ts`) for real-time push to frontend (SSE/WebSocket).

#### 2. Idempotent Upsert Helper (repository)
```ts
// server/src/repositories/attendanceRepository.ts
import { pool } from '../db/client';

export async function upsertAttendanceEvent(
  userId: string,
  eventType: 'clock_in' | 'clock_out' | 'leave_request' | 'ot_request',
  eventTimestamp: Date,
  payload: Record<string, any>,
  idempotencyKey: string | null
) {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // Fast path: idempotency key lookup
    if (idempotencyKey) {
      const existing = await client.query(
        `SELECT id FROM attendance_events
         WHERE user_id = $1 AND event_type = $2 AND event_timestamp = $3 AND idempotency_key = $4`,
        [userId, eventType, eventTimestamp, idempotencyKey]
      );
      if (existing.rows.length > 0) {
        await client.query('COMMIT');
        return existing.rows[0];
      }
    }

    const result = await client.query(
      `INSERT INTO attendance_events
         (user_id, event_type, event_timestamp, payload, idempotency_key, created_at)
       VALUES ($1, $2, $3, $4, $5, NOW())
       RETURNING *`,
      [userId, eventType, eventTimestamp, JSON.stringify(payload), idempotencyKey]
    );

    await client.query('COMMIT');
    return result.rows[0];
  } catch (err: any) {
    await client.query('ROLLBACK');
    // Race-safe fallback on unique violation
    if (idempotencyKey && err.code === '23505') {
      const existing = await client.query(
        `SELECT id FROM attendance_events
         WHERE user_id = $1 AND event_type = $2 AND event_timestamp = $3 AND idempotency_key = $4`,
        [userId, eventType, eventTimestamp, idempotencyKey]
      );
      if (existing.rows.length > 0) return existing.rows[0];
    }
    throw err;
  } finally {
    client.release();
  }
}
```

#### 3. Lightweight Pub/Sub for Real-Time Push (in-process)
```ts
// server/src/lib/events.ts
type Listener = (payload: any) => void;

class EventBus {
  private listeners = new Map<string, Set<Listener>>();

  on(channel: string, listener: Listener) {
    if (!this.listeners.has(channel)) this.listeners.set(channel, new Set());
    this.listeners.get(channel)!.add(listener);
    return () => this.off(channel, listener);
  }

  off(channel: string, listener: Listener) {
    this.listeners.get(channel)?.delete(listener);
  }

  emit(channel: string, payload: any) {
    for (const listener of this.listeners.get(channel) || []) {
      try { listener(payload); } catch (e) { console.error(e); }
    }
  }
}

export const eventBus = new EventBus();
```

#### 4. Webhook Handler Update (idempotent + broadcast)
```ts
// server/src/routes/webhook/line.ts
import { Router } from 'express';
import { upsertAttendanceEvent } from '../../repositories/attendanceRepository';
import { eventBus } from '../../lib/events';
import crypto from 'crypto';

const router = Router();

function deriveIdempotencyKey(body: any, headers: any): string | null {
  if (headers['x-idempotency-key']) return headers['x-idempotency-key'];

  const delivery = body?.deliveryContext?.[0]?.deliveryId;
  const timestamp = body?.timestamp;
  const userId = body?.events?.[0]?.source?.userId;
  if (delivery && timestamp && userId) {
    return crypto.createHash('sha256').update(`${delivery}:${timestamp}:${userId}`).digest('hex');
  }
  return null;
}

router.post('/line', async (req, res) => {
  try {
    const body = req.body;
    const idempotencyKey = deriveIdempotencyKey(body, req.headers);

    const events = body.events || [];
    for (const ev of events) {
      if (ev.type === 'message' && ev.message?.type === 'text') {
        const text = ev.message.text.toLowerCase();
        const userId = ev.source.userId;
        const ts = new Date(ev.timestamp);

        let record = null;
        if (text.includes('เข้างาน') || text.includes('clock in')) {
          record = await upsertAttendanceEvent(userId, 'clock_in', ts, { line: ev }, idempotencyKey);
        } else if (text.includes('เลิกงาน') || text.includes('clock out')) {
          record = await upsertAttendanceEvent(userId, 'clock_out', ts, { line: ev }, idempotencyKey);
        }
        // Extend for leave/OT similarly

        // Broadcast real-time update to subscribed frontends
        if (record) {
          eventBus.emit(`attendance:${userId}`, {
            type: record.event_type,
            timestamp: record.event_timestamp,
            payload: record.payload,
          });
        }
      }
    }

    res.status(200).json({ ok: true });
  } catch (err) {
    console.error('LINE webhook error:', err);
    res.status(500).json({ error: 'internal_error' });
  }
});

export default router;
```

#### 5. Frontend Integration (SSE example)
- Expose an SSE endpoint (or WebSocket) that wraps `eventBus` for browser subscriptions:
```ts
// server/src/routes/events.ts
import { Router } from 'express';
import { eventBus } from '../lib/events';

const router = Router();

router.get('/stream/:userId', (req, res) => {
  const { userId } = req.params;
  res.setHeader('Content-Type', 'text/event-stream');
  res.setHeader('Cache-Control', 'no-cache');
  res.setHeader('Connection', 'keep-alive');

  const onEvent = (payload: any) => {
    res.write(`data: ${JSON.stringify(payload)}\n\n`);
  };

  const remove = eventBus.on(`attendance:${userId}`, onEvent);
  req.on('close', () => remove());
});

export default router;
```
- Frontend connects to `/events/stream/:userId` and updates local state immediately on `message`, eliminating polling and cross-tab divergence.
