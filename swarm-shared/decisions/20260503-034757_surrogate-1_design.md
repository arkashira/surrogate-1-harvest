# surrogate-1 / design

## Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16` (matrix) and optional `DATE_FOLDER` (defaults to today `YYYY-MM-DD`).
- Uses a **single HF API call** from the runner (before the 5min window closes) to list one date folder via `list_repo_tree(recursive=False)` and saves `file-list.json`.
- Embeds that list in the runner; worker loads it and **downloads only assigned shard files via CDN** (`https://huggingface.co/datasets/.../resolve/main/...`) with **no Authorization header** → bypasses `/api/` rate limits.
- Projects each file to `{prompt, response}` at parse time (avoids `load_dataset(streaming=True)` on mixed schemas).
- Deduplicates via central md5 store (`lib/dedup.py`) and uploads shard output to:
  ```
  batches/public-merged/<date>/shard<SHARD_ID>-<HHMMSS>.jsonl
  ```
- Adds retry/backoff for CDN 429s and HF API 429s (360s wait), and falls back to `hf_hub_download` per file if CDN fails.

"""
CDN-bypass ingestion worker for surrogate-1 public dataset.
Usage (GitHub Actions matrix):
  SHARD_ID=0 SHARD_TOTAL=16 python bin/dataset-enrich.py
"""

import os
import sys
import json
import time
import hashlib
import datetime
from pathlib import Path

import requests
import pyarrow.parquet as pq
import pyarrow as pa
from huggingface_hub import HfApi, hf_hub_download

HF_REPO = "datasets/axentx/surrogate-1-training-pairs"
HF_TOKEN = os.getenv("HF_TOKEN")
SHARD_ID = int(os.getenv("SHARD_ID", "0"))
SHARD_TOTAL = int(os.getenv("SHARD_TOTAL", "16"))
DATE_FOLDER = os.getenv("DATE_FOLDER", datetime.date.today().isoformat())

API = HfApi(token=HF_TOKEN)
BASE_CDN = f"https://huggingface.co/datasets/{HF_REPO}/resolve/main"
WORKDIR = Path(__file__).parent.parent
MANIFEST = WORKDIR / "file-list.json"
OUT_DIR = WORKDIR / "batches" / "public-merged" / DATE_FOLDER
OUT_DIR.mkdir(parents=True, exist_ok=True)

HEADERS = {"Authorization": f"Bearer {HF_TOKEN}"} if HF_TOKEN else {}

def hf_api_list_once():
    """Single API call to list one date folder; save manifest."""
    try:
        items = API.list_repo_tree(
            repo_id=HF_REPO,
            path=DATE_FOLDER,
            repo_type="dataset",
            recursive=False,
        )
    except Exception as e:
        print(f"HF API list failed: {e}", file=sys.stderr)
        return []
    files = [it.rfilename for it in items if it.type == "file"]
    manifest_data = {"date": DATE_FOLDER, "files": files}
    MANIFEST.write_text(json.dumps(manifest_data))
    return files

def load_manifest():
    if MANIFEST.exists():
        data = json.loads(MANIFEST.read_text())
        if data.get("date") == DATE_FOLDER:
            return data["files"]
    return hf_api_list_once()

def shard_filter(paths):
    """Deterministic shard assignment by slug hash."""
    assigned = []
    for p in paths:
        slug = Path(p).stem
        h = int(hashlib.md5(slug.encode()).hexdigest(), 16)
        if h % SHARD_TOTAL == SHARD_ID:
            assigned.append(p)
    return assigned

def download_cdn(path):
    url = f"{BASE_CDN}/{path}"
    for attempt in range(5):
        try:
            r = requests.get(url, timeout=60)
            if r.status_code == 429:
                wait = int(r.headers.get("retry-after", 60))
                print(f"CDN 429, wait {wait}s", file=sys.stderr)
                time.sleep(wait)
                continue
            r.raise_for_status()
            return r.content
        except Exception as e:
            print(f"CDN fetch failed ({path}): {e}", file=sys.stderr)
            time.sleep(2 ** attempt)
    return None

def download_fallback(path):
    try:
        local = hf_hub_download(
            repo_id=HF_REPO,
            filename=path,
            repo_type="dataset",
        )
        return Path(local).read_bytes()
    except Exception as e:
        print(f"hf_hub_download failed ({path}): {e}", file=sys.stderr)
        return None

def project_to_pair(raw_bytes, path):
    """Project heterogeneous file to {prompt, response} only."""
    try:
        tbl = pq.read_table(pa.BufferReader(raw_bytes))
    except Exception:
        # fallback: try to read as JSON lines
        import io, json as _json
        lines = raw_bytes.decode().strip().splitlines()
        rows = [_json.loads(l) for l in lines if l.strip()]
        tbl = pa.Table.from_pylist(rows)

    # Normalize column names
    cols = {c.lower(): c for c in tbl.column_names}
    prompt_col = None
    response_col = None
    for k in ("prompt", "instruction", "input", "question"):
        if k in cols:
            prompt_col = cols[k]
            break
    for k in ("response", "output", "answer", "completion"):
        if k in cols:
            response_col = cols[k]
            break

    if prompt_col is None or response_col is None:
        # best-effort: pick first text-like pair
        text_cols = [c for c in tbl.column_names if pa.types.is_string(tbl.schema.field(c).type)]
        if len(text_cols) >= 2:
            prompt_col, response_col = text_cols[0], text_cols[1]
        else:
            raise ValueError(f"Cannot find prompt/response in {path}")

    pairs = []
    for i in range(tbl.num_rows):
        row = tbl.slice(i, 1).to_pylist()[0]
        pairs.append({
            "prompt": str(row[prompt_col]).strip(),
            "response": str(row[response_col]).strip(),
        })
    return pairs
