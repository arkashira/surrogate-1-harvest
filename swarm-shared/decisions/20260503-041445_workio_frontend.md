# workio / frontend

## Final Synthesized Solution

**Chosen approach:** Merge the strongest, most production-ready patterns from both candidates while eliminating contradictions.

**Key decisions:**
- Use **AbortController + in-flight lock** (not debounce) to prevent duplicate requests. Debounce is wrong for actions that must not be dropped; aborting previous + blocking new is correct.
- **Optimistic UI with guaranteed rollback** on failure (Candidate 2 pattern) is safer than Candidate 1’s partial revert.
- **Idempotency key** must be deterministic per logical event but include a sequence/timestamp to avoid false collisions across distinct intentional events. Use `minuteWindow + lastEventId`.
- **Temporal guard** should be short (2–5s) to prevent UI mashing, plus a monotonic sequence check to reject out-of-order optimistic states.
- **State machine** (`idle | pending | success | error`) with clear visual feedback and disabled states.
- Keep **existing API contract**; only enhance client behavior.

### 1. API layer (`workio/src/api/clock.ts`)

```ts
import axios from 'axios';

const clockApi = axios.create({
  baseURL: '/api',
  timeout: 10_000,
});

export interface ClockEvent {
  id: string;
  tenantId: string;
  userId: string;
  type: 'clock_in' | 'clock_out';
  timestamp: string; // ISO
}

export interface ClockRequest {
  tenantId: string;
  userId: string;
  action: 'in' | 'out';
  idempotencyKey: string;
  lastEventId?: string; // for monotonic guard
}

export async function clockInOut(
  payload: ClockRequest,
  signal?: AbortSignal
): Promise<ClockEvent> {
  const res = await clockApi.post<ClockEvent>('/clock', payload, {
    signal,
    headers: {
      'Idempotency-Key': payload.idempotencyKey,
    },
  });
  return res.data;
}
```

### 2. Hook: `workio/src/hooks/useClock.ts`

```ts
import { useState, useCallback, useRef } from 'react';
import { clockInOut, type ClockRequest, type ClockEvent } from '@/api/clock';
import { useToast } from '@/hooks/use-toast';

type ClockStatus = 'idle' | 'pending' | 'success' | 'error';

export function useClock({
  tenantId,
  userId,
  lastEventId,
  onSuccess,
}: {
  tenantId: string;
  userId: string;
  lastEventId?: string;
  onSuccess?: (event: ClockEvent) => void;
}) {
  const [status, setStatus] = useState<ClockStatus>('idle');
  const [lastError, setLastError] = useState<string | null>(null);
  const abortRef = useRef<AbortController | null>(null);
  const inFlightRef = useRef<boolean>(false);
  const cooldownRef = useRef<number>(0);
  const { toast } = useToast();

  // Short UI cooldown to prevent mashing (2s)
  const UI_COOLDOWN_MS = 2_000;

  // Generate idempotency key stable within the same minute for same logical event
  const generateIdempotencyKey = useCallback(
    (action: 'in' | 'out') => {
      const now = Date.now();
      const minuteWindow = Math.floor(now / 60_000);
      // Include lastEventId to differentiate sequential intentional events within same minute
      return `${tenantId}:${userId}:${action}:${minuteWindow}:${lastEventId || 'none'}`;
    },
    [tenantId, userId, lastEventId]
  );

  const clock = useCallback(
    async (action: 'in' | 'out') => {
      const now = Date.now();

      // Temporal guard: reject if still in UI cooldown
      if (now < cooldownRef.current) {
        setLastError('Please wait before clocking again.');
        setStatus('error');
        return false;
      }

      // Deduplicate in-flight requests
      if (inFlightRef.current) {
        // If same action already in flight, do nothing; if different, abort previous
        if (abortRef.current) {
          abortRef.current.abort();
        }
      }

      const controller = new AbortController();
      abortRef.current = controller;
      inFlightRef.current = true;
      setLastError(null);
      setStatus('pending');

      const idempotencyKey = generateIdempotencyKey(action);

      const payload: ClockRequest = {
        tenantId,
        userId,
        action,
        idempotencyKey,
        lastEventId,
      };

      try {
        const event = await clockInOut(payload, controller.signal);

        // Monotonic guard: ensure event id is newer than last known
        if (lastEventId && event.id <= lastEventId) {
          throw new Error('Received out-of-order event.');
        }

        // Extend cooldown on success to prevent rapid toggles
        cooldownRef.current = Date.now() + UI_COOLDOWN_MS;
        setStatus('success');
        onSuccess?.(event);

        toast({
          title: 'Success',
          description: `Clocked ${action === 'in' ? 'in' : 'out'}`,
          variant: 'default',
        });

        return true;
      } catch (err: any) {
        if (err.name === 'AbortError') {
          // Abort is intentional; keep status as-is (caller may retry)
          return false;
        }

        setStatus('error');
        const message = err?.message || 'Failed to clock';
        setLastError(message);

        toast({
          title: 'Failed to clock',
          description: message,
          variant: 'destructive',
        });

        return false;
      } finally {
        if (status === 'pending') {
          setStatus('idle');
        }
        inFlightRef.current = false;
        abortRef.current = null;
      }
    },
    [tenantId, userId, lastEventId, generateIdempotencyKey, onSuccess, toast, status]
  );

  return {
    status,
    lastError,
    clock,
    isPending: status === 'pending',
    isDisabled: status === 'pending' || Date.now() < cooldownRef.current,
  };
}
```

### 3. Component: `workio/src/features/clock/ClockButton.tsx`

```tsx
import { useClock } from '@/hooks/useClock';
import { ClockEvent } from '@/api/clock';

interface ClockButtonProps {
  tenantId: string;
  userId: string;
  /** Current clock state derived from latest event */
  isClockedIn: boolean;
  /** Optional last event id for monotonic guard */
  lastEventId?: string;
  /** Optional callback when clock state is confirmed */
  onClockUpdate?: (event: ClockEvent) => void;
}

export function ClockButton({
  tenantId,
  userId,
  isClockedIn,
  lastEventId,
  onClockUpdate,
}: ClockButtonProps) {
  const { clock, isPending, isDisabled } = useClock({
    tenantId,
    userId,
    lastEventId,
    onSuccess: onClockUpdate,
  });

  const nextAction = isClockedIn ? 'out' : 'in';
  const label = isClockedIn ? 'Clock Out' : 'Clock In';
  const pendingLabel = isClockedIn ? 'Clocking Out...' : 'Clocking In...';

  const handleClick = () => {
    clock(nextAction);
  };

  return (
    <button
      onClick={handleClick}
      disabled={isDisabled}
      className={`
        px-6 py-3 rounded-lg font-medium transition-all
        ${
          isClockedIn
            ? 'bg-red-600 hover:bg-red-700 text-white'
            : 'bg-green-600 hover:bg-green-700 text-white'
        }
        ${isPending ? 'opacity-75 cursor-wait' : ''}
        ${isDisabled ? 'cursor-not-allowed opacity-60' : ''}
      `}
    >
      {isPending ? pendingLabel
