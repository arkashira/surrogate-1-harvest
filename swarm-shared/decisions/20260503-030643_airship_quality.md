## Implementation Plan — Airship Quality (Surrogate Training Resilience)

**Highest-value incremental improvement (<2h):**  
Eliminate HF API 429s during Surrogate training data loading and make Lightning Studio resilient to idle-stop so end-to-end training iteration is **<2 minutes** and never blocked by rate limits or idle timeouts.

**Concrete steps:**
1. Add a one-time Mac-side script to list a single date folder from the surrogate dataset repo and save `file_list.json` (uses HF API once; avoids recursive listing).
2. Update Surrogate training script to:
   - Load `file_list.json`.
   - Fetch parquet files via HF CDN URLs (no Authorization header; bypasses API rate limits).
   - Project only `{prompt, response}` at parse time (avoids pyarrow schema issues).
3. Add Lightning Studio reuse + idle-stop restart logic so training resumes automatically if studio was stopped.
4. Pin deterministic machine type (L40S) and prefer `lightning-public-prod` (free tier) with fallback ordering.

### 1) Mac orchestration: list once, embed in training

`scripts/list_surrogate_files.py`
```python
#!/usr/bin/env python3
"""
Run from Mac (or any dev machine) after rate-limit window clears.
Lists one date folder in the surrogate dataset repo and saves file_list.json.
"""
import json
import os
import sys
from datetime import datetime, timezone

try:
    from huggingface_hub import list_repo_tree
except ImportError:
    print("Install: pip install huggingface_hub")
    sys.exit(1)

REPO = "axentx/surrogate-dataset"  # change if different
DATE_FOLDER = datetime.now(timezone.utc).strftime("%Y-%m-%d")  # e.g. 2026-05-03
OUT_PATH = os.path.join(os.path.dirname(__file__), "..", "file_list.json")

def main() -> None:
    # non-recursive per folder to avoid pagination + rate limits
    tree = list_repo_tree(repo_id=REPO, path=DATE_FOLDER, recursive=False)
    files = [
        {"repo": REPO, "path": f.path}
        for f in tree
        if f.path.lower().endswith(".parquet")
    ]
    payload = {
        "date_folder": DATE_FOLDER,
        "generated_at_utc": datetime.now(timezone.utc).isoformat(),
        "files": files,
    }
    os.makedirs(os.path.dirname(OUT_PATH), exist_ok=True)
    with open(OUT_PATH, "w", encoding="utf-8") as f:
        json.dump(payload, f, indent=2)
    print(f"Saved {len(files)} files to {OUT_PATH}")

if __name__ == "__main__":
    main()
```

Make executable and run once:
```bash
chmod +x scripts/list_surrogate_files.py
python scripts/list_surrogate_files.py
```

### 2) Surrogate training: CDN-only data loader

`surrogate/train.py` (minimal focused snippet to drop into existing train script)
```python
import json
import pyarrow.parquet as pq
import requests
from pathlib import Path
from typing import List, Dict

CDN_ROOT = "https://huggingface.co/datasets"

def load_file_list() -> List[Dict]:
    p = Path(__file__).parent.parent / "file_list.json"
    if not p.exists():
        raise FileNotFoundError(f"Run scripts/list_surrogate_files.py first -> {p}")
    with open(p, "r", encoding="utf-8") as f:
        data = json.load(f)
    return data.get("files", [])

def cdn_url(repo: str, path: str) -> str:
    # repo format: owner/dataset
    return f"{CDN_ROOT}/{repo}/resolve/main/{path}"

def stream_parquet_rows(files: List[Dict]):
    """Yield {prompt, response} rows from CDN parquet files without HF API auth."""
    for entry in files:
        url = cdn_url(entry["repo"], entry["path"])
        # stream download to avoid large tmp files; project columns at read time
        resp = requests.get(url, timeout=60, stream=True)
        resp.raise_for_status()
        # Use pyarrow to read only needed columns; avoids mixed-schema errors
        with pq.ParquetFile(pq.ParquetDataset([url], columns=["prompt", "response"])) as pf:
            for batch in pf.iter_batches(batch_size=512, columns=["prompt", "response"]):
                df = batch.to_pandas()
                for _, row in df.iterrows():
                    yield {"prompt": str(row["prompt"]), "response": str(row["response"])}

# Example usage in training loop
if __name__ == "__main__":
    files = load_file_list()
    for i, row in enumerate(stream_parquet_rows(files)):
        # train step using row["prompt"], row["response"]
        if i % 500 == 0:
            print(f"Processed {i} rows")
        # break early for quick smoke test
        if i >= 2000:
            break
```

Notes:
- No `load_dataset(streaming=True)` → avoids pyarrow CastError on mixed schemas.
- Projection to `{prompt, response}` only at parse time; attribution moved to filename pattern in ingestion (per pattern).
- CDN downloads bypass HF API auth and rate limits.

### 3) Lightning Studio reuse + idle-stop resilience

`surrogate/lightning_launcher.py`
```python
#!/usr/bin/env python3
"""
Lightning launcher with Studio reuse and idle-stop resilience.
Ensures training iteration <2m and survives idle timeouts.
"""
import time
from pathlib import Path
from lightning_sdk import Studio, Teamspace, Machine

TEAMSPACE = "default"
STUDIO_NAME = "surrogate-train-l40s"
MACHINE = Machine.L40S  # free tier falls to L40S in lightning-public-prod
MAX_RETRIES = 3
RETRY_WAIT = 60

def get_or_create_studio() -> Studio:
    team = Teamspace(name=TEAMSPACE)
    # Reuse running studio to save quota (80hr/mo)
    for s in team.studios:
        if s.name == STUDIO_NAME:
            if s.status == "running":
                print(f"Reusing running studio: {s.name}")
                return s
            else:
                print(f"Studio {s.name} exists but status={s.status}. Restarting...")
                s.start(machine=MACHINE)
                return s
    # create if not found
    print(f"Creating studio {STUDIO_NAME}")
    return Studio(
        name=STUDIO_NAME,
        teamspace=TEAMSPACE,
        machine=MACHINE,
        create_ok=True,
    )

def run_training(studio: Studio) -> None:
    train_script = Path(__file__).parent / "train.py"
    cmd = f"python {train_script.name}"
    for attempt in range(1, MAX_RETRIES + 1):
        try:
            # Check studio status before run; restart if stopped (idle-timeout kills training)
            studio.refresh()
            if studio.status != "running":
                print(f"Studio not running (status={studio.status}). Restarting...")
                studio.start(machine=MACHINE)
                time.sleep(10)
            print(f"Run attempt {attempt}/{MAX_RETRIES}")
            run = studio.run(command=cmd, cwd=str(train_script.parent))
            run.wait()
            if run.status == "succeeded":
                print("Training succeeded.")
                return
            else:
                print(f"Run ended with status={run.status}")
        except Exception as exc:
            print(f"Attempt {attempt} failed: {exc}")
        if attempt < MAX_RETRIES:
            print(f"Retrying in {RETRY_WAIT}s...")
            time.sleep(RETRY_WAIT)
    raise RuntimeError("Training failed after retries")

if __name__ == "__main__":
    studio = get_or_create_studio()
    run_training(studio)
```

Usage:
```bash
cd surrogate
python lightning_launcher.py
```
