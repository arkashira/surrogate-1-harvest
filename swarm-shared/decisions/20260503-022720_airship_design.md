### 1. Deterministic file-list generation (run once per day from Mac)

`/opt/axentx/airship/scripts/generate_file_list.py`

```python
#!/usr/bin/env python3
"""
Generate deterministic CDN file list for HF dataset ingestion.
Run after HF rate-limit window clears.
"""
import argparse
import json
from datetime import date
from huggingface_hub import HfApi

def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", default="datasets/surrogate-training")
    parser.add_argument("--date", default=str(date.today()), help="YYYY-MM-DD folder under batches/mirror-merged/")
    parser.add_argument("--out", required=True, help="Path to write JSON manifest")
    args = parser.parse_args()

    api = HfApi()
    folder = f"batches/mirror-merged/{args.date}"
    # One non-recursive API call
    tree = api.list_repo_tree(repo_id=args.repo, path=folder, recursive=False)
    files = sorted(
        node.rfilename
        for node in tree
        if node.rfilename.endswith((".parquet", ".jsonl"))
    )

    payload = {
        "dataset_repo": args.repo,
        "date": args.date,
        "folder": folder,
        "files": files,
        "total": len(files),
        "cdn_prefix": f"https://huggingface.co/datasets/{args.repo}/resolve/main"
    }

    out_path = Path(args.out)
    out_path.parent.mkdir(parents=True, exist_ok=True)
    out_path.write_text(json.dumps(payload, indent=2))
    print(f"Wrote {len(files)} files to {out_path}")

if __name__ == "__main__":
    main()
```

### 2. CDN-only dataset loader (zero HF API calls)

`/opt/axentx/airship/surrogate/training/data_loader.py`

```python
import json
import pyarrow.parquet as pq
import requests
from io import BytesIO
from pathlib import Path
from typing import Iterator, Dict, Any

class CDNParquetDataset:
    """Deterministic CDN-first loader; no HF API usage."""
    def __init__(self, manifest_path: str | Path) -> None:
        manifest = json.loads(Path(manifest_path).read_text())
        self.cdn_prefix = manifest["cdn_prefix"]
        self.folder = manifest["folder"]
        self.files = manifest["files"]

    def _fetch_table(self, fname: str) -> pq.Table:
        url = f"{self.cdn_prefix}/{self.folder}/{fname}"
        resp = requests.get(url, timeout=30)
        resp.raise_for_status()
        return pq.read_table(pq.ParquetFile(pq.BufferReader(resp.content)))

    def __iter__(self) -> Iterator[Dict[str, Any]]:
        for fname in self.files:
            table = self._fetch_table(fname)
            # Project to {prompt, response} at parse time
            yield from table.select(["prompt", "response"]).to_pylist()
```

### 3. Training module with Studio reuse

`/opt/axentx/airship/surrogate/training/train.py`

```python
import torch
from pathlib import Path
from lightning import Fabric, LightningModule, Teamspace, Studio, Machine
from lightning.pytorch.utilities import rank_zero_only

from data_loader import CDNParquetDataset

class SurrogateModel(LightningModule):
    def train_dataloader(self):
        manifest = Path(__file__).parent / "file_manifest.json"
        dataset = CDNParquetDataset(manifest)
        return torch.utils.data.DataLoader(dataset, batch_size=32, num_workers=0)

    def training_step(self, batch, batch_idx):
        # Minimal stub; replace with real forward+loss
        return torch.tensor(0.0, requires_grad=True)

    def configure_optimizers(self):
        return torch.optim.Adam(self.parameters(), lr=1e-4)

@rank_zero_only
def launch_studio() -> Studio:
    for s in Teamspace().studios():
        if s.name == "surrogate-train" and s.status == "running":
            print(f"Reusing running studio: {s.name}")
            return s
    studio = Studio(name="surrogate-train", machine=Machine.L40S, create_ok=True)
    return studio

if __name__ == "__main__":
    studio = launch_studio()
    if studio.status != "running":
        studio.start(machine=Machine.L40S)

    fabric = Fabric(devices=1, accelerator="gpu", precision="bf16-mixed")
    model = SurrogateModel()
    trainer = fabric.Trainer(max_epochs=1)
    trainer.fit(model)
```

### 5. Cron-safe wrapper

`/opt/axentx/airship/scripts/train_wrapper.sh`

```bash
#!/usr/bin/env bash
SHELL=/bin/bash
set -euo pipefail

cd /opt/axentx/airship

MANIFEST="surrogate/training/file_manifest.json"
# Regenerate once per day if missing or older than 24h
if [[ ! -f "$MANIFEST" ]] || [[ $(find "$MANIFEST" -mmin +1440) ]]; then
    python scripts/generate_file_list.py --out "$MANIFEST"
fi

exec python surrogate/training/train.py "$@"
```
