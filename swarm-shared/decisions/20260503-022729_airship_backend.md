### 1. CDN Manifest Generator (`scripts/generate_cdn_manifest.py`)
*Run on Mac (or orchestration host) after rate-limit window clears.*

```python
#!/usr/bin/env python3
"""
Generate CDN-only file manifest for Surrogate training.
Single API call per folder → deterministic JSON for zero-API-load during training.
"""
import json
import os
from datetime import datetime, timezone
from pathlib import Path

from huggingface_hub import HfApi

HF_REPO = os.getenv("HF_DATASET_REPO", "datasets/axentx/surrogate-mirror")
OUTPUT_DIR = Path(__file__).parent.parent / "surrogate" / "training" / "manifests"
OUTPUT_DIR.mkdir(parents=True, exist_ok=True)

def generate_manifest(date_folder: str = None) -> Path:
    """List parquet files in date folder (non-recursive) and save CDN manifest."""
    if date_folder is None:
        date_folder = datetime.utcnow().strftime("%Y-%m-%d")
    
    api = HfApi()
    # Single API call — non-recursive to avoid pagination explosion
    tree = api.list_repo_tree(
        repo_id=HF_REPO,
        path=date_folder,
        repo_type="dataset",
        recursive=False
    )
    
    # Filter to parquet files only (skip dirs, non-parquet)
    files = [
        {
            "repo": HF_REPO,
            "filename": item.rfilename,
            "cdn_url": (
                f"https://huggingface.co/datasets/{HF_REPO}"
                f"/resolve/main/{item.rfilename}"
            ),
            "size": getattr(item, "size", None)
        }
        for item in tree
        if hasattr(item, "rfilename") and item.rfilename.endswith(".parquet")
    ]
    
    manifest = {
        "repo": HF_REPO,
        "folder": date_folder,
        "generated_utc": datetime.now(timezone.utc).isoformat(),
        "files": files,
        "count": len(files)
    }
    
    output_path = OUTPUT_DIR / f"file_manifest_{date_folder}.json"
    output_path.write_text(json.dumps(manifest, indent=2))
    print(f"✅ Wrote {len(files)} files → {output_path}")
    return output_path

if __name__ == "__main__":
    import sys
    date_arg = sys.argv[1] if len(sys.argv) > 1 else None
    generate_manifest(date_arg)
```

**Make executable**:  
```bash
chmod +x scripts/generate_cdn_manifest.py
```

import json
from pathlib import Path
from typing import Optional

import torch
from lightning import LightningModule, Trainer
from lightning.pytorch.utilities import LightningEnum

try:
    from lightning import Studio, Teamspace, Machine
    LIGHTNING_AVAILABLE = True
except ImportError:
    LIGHTNING_AVAILABLE = False
    # Mock classes for local dev without lightning installed
    class Studio: pass
    class Teamspace: studios = []
    class Machine: L40S = "L40S"

MANIFEST_DIR = Path(__file__).parent / "manifests"

class CDNParquetDataset(torch.utils.data.Dataset):
    """
    Lightning training uses CDN-only fetches.
    Zero HF API calls during data load.
    """
    def __init__(self, manifest_path: Path):
        manifest = json.loads(manifest_path.read_text())
        self.urls = [f["cdn_url"] for f in manifest["files"]]
        self._cache = {}

    def __len__(self):
        return len(self.urls)

    def __getitem__(self, idx):
        import pyarrow.parquet as pq
        url = self.urls[idx]
        if url not in self._cache:
            # Arrow can read directly from HTTPS (CDN)
            table = pq.read_table(url, columns=["prompt", "response"])
            self._cache[url] = table.to_pydict()
        return self._cache[url]

    def configure_optimizers(self):

def launch_studio(
    manifest_path: Path,
    studio_name: str = "surrogate-l40s-training",
    machine: str = "L40S",
    resume_ok: bool = True
) -> Optional[Studio]:
    """
    Reuse running studio to save quota. Auto-restart if idle-stop killed.
    Returns Studio object if launched, None if LIGHTNING not available.
    """
    if not LIGHTNING_AVAILABLE:
        print("⚠️  Lightning SDK not available. Running locally instead.")
        # Local fallback for development
        module = SurrogateTrainingModule(manifest_path)
        trainer = Trainer(max_epochs=1, accelerator="cpu")
        trainer.fit(module)
        return None

    # Reuse running studio to save quota
    target = None
    if resume_ok:
        for s in Teamspace.studios:
            if s.name == studio_name and s.status == "Running":
                print(f"♻️  Reusing running studio: {studio_name}")
                target = s
                break
    
    if target is None:
        target = Studio(
            name=studio_name,
            machine=Machine.L40S,
            create_ok=True,
        )

    # Guard against idle-stop death
    if target.status != "Running":
        print(f"🔄 Starting studio: {studio_name}")
        target.start(machine=Machine.L40S)

    target.run(
        entry_point="python",
        arguments=[
            "-m", "surrogate.training.train",
            "--manifest", str(manifest_path)
        ],
        requirements=["pyarrow", "lightning", "torch", "huggingface-hub"],
    )
    return target

if __name__ == "__main__":
    import argparse
    parser = argparse.ArgumentParser()
    parser.add_argument("--manifest", type=Path, help="Path to CDN manifest JSON")
    args = parser.parse_args()
    
    if args.manifest:
        manifest_path = args.manifest
    else:
        # Auto-select latest manifest
        manifests = sorted(MANIFEST_DIR.glob("file_manifest_*.json"))
        if not manifests:
            raise FileNotFoundError(f"No manifests found in {MANIFEST_DIR}")
        manifest_path = manifests[-1]
    
    print(f"🚀 Launching training with: {manifest_path}")
    launch_studio(manifest_path)
