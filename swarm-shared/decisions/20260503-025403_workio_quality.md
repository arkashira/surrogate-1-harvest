## 1. DB changes (single migration)

```sql
-- Idempotency ledger for punch events (24h TTL)
CREATE TABLE IF NOT EXISTS punch_idempotency (
  tenant_id      TEXT NOT NULL,
  idempotency_key TEXT NOT NULL,
  punch_id       TEXT NOT NULL,
  created_at     TIMESTAMPTZ DEFAULT now(),
  PRIMARY KEY (tenant_id, idempotency_key)
);

CREATE INDEX IF NOT EXISTS idx_punch_idempotency_created_at
  ON punch_idempotency (created_at);

-- At most one open punch per tenant/user
-- Assumes punches has: id, tenant_id, user_id, status ('in'|'out'), clock_in_time, clock_out_time
CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_open_per_tenant_user
  ON punches (tenant_id, user_id)
  WHERE status = 'in';
```

Why:
- `punch_idempotency` gives safe, repeatable deduplication for LINE retries and client retries.
- Partial unique index enforces business rule at DB level (strongest correctness guarantee).

```ts
import { Request, Response, NextFunction } from 'express';
import { v4 as uuidv4 } from 'uuid';

export function requestIdMiddleware(req: Request, res: Response, next: NextFunction) {
  const requestId = (req.headers['x-request-id'] as string) || uuidv4();
  (req as any).requestId = requestId;
  res.setHeader('X-Request-Id', requestId);
  next();
}

export function logStructured(event: string, data: Record<string, any>) {
  console.log(JSON.stringify({ event, timestamp: new Date().toISOString(), ...data }));
}
```

Wire `requestIdMiddleware` early in the app (before routes).

## 3. Idempotency service (transaction-safe)

`server/src/services/idempotency.ts`
```ts
import { PoolClient } from 'pg';
import { logStructured } from '../middleware/requestId';

export async function ensureIdempotentPunch(
  client: PoolClient,
  tenantId: string,
  idempotencyKey: string,
  createPunch: () => Promise<{ punchId: string }>
): Promise<{ punchId: string; isDuplicate: boolean }> {
  // 1) Try existing within same transaction
  const existing = await client.query(
    'SELECT punch_id FROM punch_idempotency WHERE tenant_id = $1 AND idempotency_key = $2',
    [tenantId, idempotencyKey]
  );

  if (existing.rows.length > 0) {
    logStructured('punch_idempotency_hit', { tenantId, idempotencyKey, punchId: existing.rows[0].punch_id });
    return { punchId: existing.rows[0].punch_id, isDuplicate: true };
  }

  // 2) Create punch (will fail on unique open-punch violation if race)
  const result = await createPunch();

  // 3) Record idempotency; handle races by re-fetching stored punch_id
  try {
    await client.query(
      'INSERT INTO punch_idempotency (tenant_id, idempotency_key, punch_id) VALUES ($1, $2, $3)',
      [tenantId, idempencyKey, result.punchId]
    );
  } catch (err: any) {
    if (err.code === '23505') {
      const dup = await client.query(
        'SELECT punch_id FROM punch_idempotency WHERE tenant_id = $1 AND idempotency_key = $2',
        [tenantId, idempotencyKey]
      );
      if (dup.rows.length > 0) {
        logStructured('punch_idempotency_race', { tenantId, idempotencyKey, punchId: dup.rows[0].punch_id });
        return { punchId: dup.rows[0].punch_id, isDuplicate: true };
      }
    }
    throw err;
  }

  logStructured('punch_created', { tenantId, idempotencyKey, punchId: result.punchId });
  return { punchId: result.punchId, isDuplicate: false };
}
```

Notes:
- Runs inside the same DB transaction as punch creation → no split-brain.
- Handles unique violation races safely by re-selecting the stored punch_id.

function lineEventToIdempotencyKey(event: any, tenantId: string): string {
  const mode = event.mode || 'unknown';
  const delivery = event.deliveryContext?.isUnreliable ? 'unreliable' : 'reliable';
  const ts = event.timestamp || Date.now();
  const userId = event.source?.userId || 'unknown';
  // Include event type to avoid collisions across different event intents
  const eventType = event.type || 'unknown';
  return `line:${tenantId}:${eventType}:${mode}:${delivery}:${ts}:${userId}`;
}
