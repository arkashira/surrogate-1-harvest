# workio / frontend

## 1. Diagnosis

- **No resilient offline-first punch queue**: frontend can lose punches on flaky networks or LINE webhook delays; users retry and create duplicates.
- **Missing idempotency key plumbing**: `usePunchQueue` exists in shared decisions but is not wired into the actual punch UI; optimistic state can diverge from server truth.
- **Race between rapid IN/OUT**: nothing prevents double-IN or double-OUT at UI + client state level before server constraint rejects it.
- **No local persistence for pending punches**: if user closes tab before confirmation, the punch is lost (bad UX for clocking).
- **No visual feedback for queued vs confirmed punches**: users don’t know whether their clock-in was accepted or still pending.

## 2. Proposed change

- **Scope**: `workio/src/components/PunchButton.tsx` (or equivalent main clock-in/out UI) + `workio/src/hooks/usePunchQueue.ts` (create if missing) + minimal `workio/src/api/punch.ts`.
- **Goal**: wire an idempotent, persisted, retry-safe punch queue into the existing punch UI so every IN/OUT is locally queued with a UUID idempotency key and retried until server ack; UI reflects queue state.

## 3. Implementation

### 3.1 Create `src/api/punch.ts`

```ts
// src/api/punch.ts
import axios from 'axios';

const api = axios.create({ baseURL: '/api' });

export type PunchType = 'IN' | 'OUT';

export interface PunchRequest {
  type: PunchType;
  latitude?: number;
  longitude?: number;
  idempotencyKey: string;
}

export async function submitPunch(payload: PunchRequest) {
  return api.post<{ id: string; type: PunchType; ts: string }>('/punches', payload, {
    headers: { 'Idempotency-Key': payload.idempotencyKey },
  });
}
```

### 3.2 Create `src/hooks/usePunchQueue.ts`

```tsx
// src/hooks/usePunchQueue.ts
import { useCallback, useEffect, useRef, useState } from 'react';
import { v4 as uuidv4 } from 'uuid';
import { submitPunch, PunchRequest } from '../api/punch';

export type PunchType = 'IN' | 'OUT';

export type PendingPunch = {
  id: string;
  type: PunchType;
  latitude?: number;
  longitude?: number;
  createdAt: number;
  attempts: number;
  lastError?: string | null;
};

type QueueState = 'idle' | 'sending' | 'error' | 'success';

export function usePunchQueue() {
  const [queue, setQueue] = useState<PendingPunch[]>([]);
  const [status, setStatus] = useState<QueueState>('idle');
  const processingRef = useRef(false);
  const maxAttempts = 5;

  // Load persisted queue once
  useEffect(() => {
    try {
      const raw = localStorage.getItem('workio:punchQueue');
      if (raw) setQueue(JSON.parse(raw));
    } catch {
      // ignore corrupt storage
    }
  }, []);

  // Persist queue changes
  useEffect(() => {
    try {
      localStorage.setItem('workio:punchQueue', JSON.stringify(queue));
    } catch {
      // ignore write failures
    }
  }, [queue]);

  const enqueue = useCallback((type: PunchType, location?: { latitude: number; longitude: number }) => {
    const item: PendingPunch = {
      id: uuidv4(),
      type,
      latitude: location?.latitude,
      longitude: location?.longitude,
      createdAt: Date.now(),
      attempts: 0,
      lastError: null,
    };
    setQueue((q) => [item, ...q]);
  }, []);

  const processQueue = useCallback(async () => {
    if (processingRef.current || queue.length === 0) return;
    processingRef.current = true;
    setStatus('sending');

    const item = queue[0];
    const payload: PunchRequest = {
      type: item.type,
      latitude: item.latitude,
      longitude: item.longitude,
      idempotencyKey: item.id,
    };

    try {
      await submitPunch(payload);
      setQueue((q) => q.filter((x) => x.id !== item.id));
      setStatus('success');
    } catch (err: any) {
      const nextAttempts = item.attempts + 1;
      if (nextAttempts >= maxAttempts) {
        setQueue((q) =>
          q.map((x) => (x.id === item.id ? { ...x, attempts: nextAttempts, lastError: err?.message || 'Unknown' } : x))
        );
        setStatus('error');
      } else {
        // keep at front, increment attempts
        setQueue((q) =>
          q.map((x) => (x.id === item.id ? { ...x, attempts: nextAttempts, lastError: err?.message || null } : x))
        );
        // exponential backoff retry
        const delay = Math.min(1000 * 2 ** nextAttempts, 30000);
        setTimeout(processQueue, delay);
      }
    } finally {
      processingRef.current = false;
      // continue processing next item
      if (queue.length > 0 && queue[0].id === item.id && !queue.find((x) => x.id === item.id)) {
        // item removed -> process next
        setTimeout(processQueue, 200);
      }
    }
  }, [queue]);

  // Kick processor when queue changes
  useEffect(() => {
    processQueue();
  }, [processQueue]);

  const retryAll = useCallback(() => {
    setQueue((q) => q.map((x) => ({ ...x, attempts: 0, lastError: null })));
  }, []);

  return {
    queue,
    status,
    enqueue,
    retryAll,
    processQueue,
  };
}
```

### 3.3 Update punch UI component (example)

```tsx
// src/components/PunchButton.tsx
import { useState } from 'react';
import { usePunchQueue } from '../hooks/usePunchQueue';

export function PunchButton() {
  const { enqueue, queue, status } = usePunchQueue();
  const [locError, setLocError] = useState<string | null>(null);

  const handlePunch = (type: 'IN' | 'OUT') => {
    setLocError(null);
    if (!navigator.geolocation) {
      // fallback: submit without location (server may reject; queue will retry)
      enqueue(type);
      return;
    }
    navigator.geolocation.getCurrentPosition(
      (pos) => {
        enqueue(type, {
          latitude: pos.coords.latitude,
          longitude: pos.coords.longitude,
        });
      },
      (err) => {
        // still enqueue without location so user isn't blocked
        setLocError('Location unavailable — punch queued without GPS');
        enqueue(type);
      },
      { timeout: 10000 }
    );
  };

  const pendingCount = queue.filter((x) => x.attempts < 5).length;
  const failedCount = queue.filter((x) => x.attempts >= 5).length;

  return (
    <div className="p-4 space-y-3">
      <div className="flex gap-3">
        <button
          onClick={() => handlePunch('IN')}
          className="px-6 py-3 bg-green-600 text-white rounded-lg font-medium hover:bg-green-700"
        >
          Clock IN
        </button>
        <button
          onClick={() => handlePunch('OUT')}
          className="px-6 py-3 bg-red-600 text-white rounded-lg font-medium hover:bg-red-700"
        >
          Clock OUT
        </button>
      </div>

      {locError && <p className="text-sm text-yellow-700">{locError}</p>}

      <div className="flex gap-4 text-sm text-gray-600">
        {pendingCount > 0 && <span>Queued: {pendingCount}</span
