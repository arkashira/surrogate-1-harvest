## 1) One-shot Mac orchestration (run once per date folder)

`scripts/list_hf_date_folder.py`

```python
#!/usr/bin/env python3
"""
List a single date folder in a HF dataset repo and emit file-list.json.
Run from Mac after rate-limit window clears.

Usage:
  python scripts/list_hf_date_folder.py \
    --repo datasets/your-org/surrogate-mirror \
    --date 2026-04-29 \
    --out surrogate/data/file-list.json
"""

import argparse
import json
import os
import sys

try:
    from huggingface_hub import HfApi
except ImportError:
    print("Install huggingface_hub: pip install huggingface_hub", file=sys.stderr)
    sys.exit(1)

def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g. datasets/your-org/surrogate-mirror)")
    parser.add_argument("--date", required=True, help="Date folder (e.g. 2026-04-29)")
    parser.add_argument("--out", required=True, help="Output JSON path")
    parser.add_argument("--max-files", type=int, default=None, help="Optional cap for quick tests")
    args = parser.parse_args()

    api = HfApi()
    repo_id = args.repo.replace("datasets/", "")
    folder_path = f"batches/mirror-merged/{args.date}"

    # recursive=False keeps API calls minimal (one per folder).
    items = api.list_repo_tree(repo_id=repo_id, path=folder_path, recursive=False, repo_type="dataset")

    files = []
    for item in items:
        if item.type == "file":
            # Use CDN URL (no auth, bypasses /api/ rate limits during training).
            cdn_url = f"https://huggingface.co/datasets/{repo_id}/resolve/main/{item.path}"
            files.append({"path": item.path, "cdn_url": cdn_url})
        if args.max_files is not None and len(files) >= args.max_files:
            break

    payload = {
        "repo": repo_id,
        "date": args.date,
        "folder": folder_path,
        "files": files,
        "generated_by": "list_hf_date_folder.py",
    }

    os.makedirs(os.path.dirname(os.path.abspath(args.out)), exist_ok=True)
    with open(args.out, "w") as f:
        json.dump(payload, f, indent=2)

    print(f"Wrote {len(files)} files to {args.out}")

if __name__ == "__main__":
    main()
```

## 2) CDN-only dataset loader with schema projection + robust caching

`surrogate/data/cdn_dataset.py`

```python
"""
CDN-only dataset loader for Surrogate-1 training.
- Uses pre-listed file-list.json (zero HF API calls during training).
- Downloads via CDN (no Authorization header) to bypass rate limits.
- Projects each file to {prompt, response} at parse time to avoid pyarrow.CastError.
- Streams rows (no full materialization) and skips malformed files.
"""

import json
import os
import sys
from typing import Dict, Iterator, Any

import pyarrow.parquet as pq
import requests
from tqdm import tqdm

class CDNDataset:
    def __init__(self, file_list_path: str, cache_dir: str = ".cache/cdn", timeout: int = 60):
        with open(file_list_path) as f:
            self.manifest = json.load(f)
        self.cache_dir = cache_dir
        self.timeout = timeout
        os.makedirs(self.cache_dir, exist_ok=True)

    def _cached_path(self, cdn_url: str) -> str:
        # Deterministic filename from last two path components.
        suffix = cdn_url.rsplit("/", 2)[-2] + "_" + cdn_url.rsplit("/", 1)[-1]
        safe = suffix.replace("/", "_").replace("=", "_")
        return os.path.join(self.cache_dir, safe)

    def _download(self, cdn_url: str) -> str:
        out_path = self._cached_path(cdn_url)
        if os.path.exists(out_path):
            return out_path

        resp = requests.get(cdn_url, timeout=self.timeout, stream=True)
        resp.raise_for_status()
        with open(out_path, "wb") as f:
            for chunk in resp.iter_content(chunk_size=8192):
                f.write(chunk)
        return out_path

    def _project_row(self, row: Dict[str, Any]) -> Dict[str, str]:
        # Heuristic projection to {prompt, response}.
        # Adjust keys to match your schema variants.
        prompt_keys = {"prompt", "input", "question", "instruction", "user"}
        response_keys = {"response", "output", "answer", "assistant", "completion"}

        prompt = None
        response = None

        for k in row:
            if k in prompt_keys and row[k] is not None:
                prompt = str(row[k])
            if k in response_keys and row[k] is not None:
                response = str(row[k])

        # Fallbacks: try to find any text-like field.
        if prompt is None or response is None:
            texts = [v for v in row.values() if isinstance(v, str) and len(v) > 10]
            if prompt is None and texts:
                prompt = texts[0]
            if response is None and len(texts) > 1:
                response = texts[1]

        return {"prompt": prompt or "", "response": response or ""}

    def __iter__(self) -> Iterator[Dict[str, str]]:
        files = self.manifest.get("files", [])
        if not files:
            print("No files in manifest; nothing to load.", file=sys.stderr)
            return

        for item in tqdm(files, desc="Loading CDN files"):
            cdn_url = item["cdn_url"]
            try:
                local_path = self._download(cdn_url)
                table = pq.read_table(local_path)
                for batch in table.to_batches(max_chunksize=1024):
                    for row in batch.to_pylist():
                        projected = self._project_row(row)
                        if projected["prompt"] and projected["response"]:
                            yield projected
            except Exception as e:
                # Log and skip malformed files instead of crashing training.
                print(f"Skipping {cdn_url}: {e}", file=sys.stderr)
                continue

    def to_list(self) -> list:
        # Avoid in production (materializes everything). Prefer streaming via __iter__.
        return list(self)
```
