### 1) `airship/cli/discover.py` — pre-list + CDN file list generation

```python
#!/usr/bin/env python3
import json
import os
from datetime import datetime
from pathlib import Path
from typing import List

from huggingface_hub import HfApi, list_repo_tree

HF_REPO = os.getenv("HF_DATASET_REPO", "axentx/surrogate-mirror")
DATE_FOLDER = datetime.utcnow().strftime("%Y-%m-%d")
OUTPUT_DIR = Path("airship/discover/outputs")
OUTPUT_DIR.mkdir(parents=True, exist_ok=True)

def preflight_list_files() -> List[str]:
    """
    Single API call to list files for one date folder.
    Avoids recursive listing on big repos; paginates minimally.
    """
    api = HfApi()
    items = list_repo_tree(
        repo_id=HF_REPO,
        path=DATE_FOLDER,
        repo_type="dataset",
        recursive=False,
    )
    files = [it.rfilename for it in items if it.type == "file"]
    return files

def build_cdn_file_list() -> None:
    files = preflight_list_files()
    payload = {
        "date": DATE_FOLDER,
        "repo": HF_REPO,
        "files": files,
        "cdn_prefix": f"https://huggingface.co/datasets/{HF_REPO}/resolve/main/{DATE_FOLDER}",
        "generated_at": datetime.utcnow().isoformat() + "Z",
    }
    out_path = OUTPUT_DIR / "file_list.json"
    out_path.write_text(json.dumps(payload, indent=2))
    print(f"CDN file list written to {out_path}")

if __name__ == "__main__":
    build_cdn_file_list()
```

### 2) `airship/train/cdn_dataset.py` — CDN-only streaming loader (zero HF API during training)

```python
import json
import logging
from pathlib import Path
from typing import Iterator, Dict, Any

import pyarrow as pa
import pyarrow.parquet as pq
import requests
from torch.utils.data import IterableDataset

logger = logging.getLogger(__name__)

class CDNParquetDataset(IterableDataset):
    """
    Loads Parquet shards exclusively via CDN URLs.
    No Hugging Face Hub API calls during data loading.
    """

    def __init__(self, file_list_path: Path, columns=("prompt", "response")):
        file_list = json.loads(file_list_path.read_text())
        self.shards = [
            f"{file_list['cdn_prefix']}/{f}"
            for f in file_list["files"]
            if f.endswith(".parquet")
        ]
        self.columns = columns
        if not self.shards:
            raise ValueError("No parquet shards found in CDN file list")

    def _stream_shard(self, url: str) -> Iterator[Dict[str, Any]]:
        # CDN download: no Authorization header -> bypasses API rate limits
        resp = requests.get(url, timeout=60)
        resp.raise_for_status()
        with pa.BufferReader(resp.content) as reader:
            table = pq.read_table(reader, columns=self.columns)
            for batch in table.to_batches(max_chunksize=1024):
                df = batch.to_pydict()
                rows = [
                    {k: df[k][i] for k in self.columns}
                    for i in range(batch.num_rows)
                ]
                yield from rows

    def __iter__(self) -> Iterator[Dict[str, Any]]:
        for shard in self.shards:
            logger.info("Streaming CDN shard: %s", shard)
            yield from self._stream_shard(shard)
```

### 3) `airship/lightning/reuse.py` — Studio reuse + idle-restart guard

```python
from lightning_sdk import Teamspace, Studio, Machine
import time
import logging

logger = logging.getLogger(__name__)

def get_or_start_studio(
    name: str,
    machine: Machine = Machine.L40S,
    max_retries: int = 3,
) -> Studio:
    """
    Reuse running studio if available; restart if stopped.
    Avoids recreating studios and burning quota.
    """
    teamspace = Teamspace()
    for studio in teamspace.studios:
        if studio.name == name:
            if studio.status == "running":
                logger.info("Reusing running studio: %s", name)
                return studio
            elif studio.status == "stopped":
                logger.info("Studio stopped. Restarting: %s", name)
                studio.start(machine=machine)
                # Wait until running
                for _ in range(max_retries * 10):
                    studio.refresh()
                    if studio.status == "running":
                        return studio
                    time.sleep(6)
                raise RuntimeError(f"Studio {name} failed to start")
    logger.info("Creating new studio: %s", name)
    return Studio.create(name=name, machine=machine, create_ok=True)
```

### 4) Schema projection helper (used by dataset-mirror → enriched/)

```python
def project_to_prompt_response(raw: Dict[str, Any]) -> Dict[str, str]:
    """
    Keep only prompt/response. Move attribution to filename pattern.
    Do NOT add source/ts columns to avoid mixed schema in parquet.
    """
    return {
        "prompt": str(raw.get("prompt", raw.get("instruction", ""))),
        "response": str(raw.get("response", raw.get("output", ""))),
    }
```
