### 3) CDN fetch utility (zero-auth, timeout + cache + stale-while-revalidate)

`src/lib/cdn.ts`
```ts
const HUB_INDEX_URL =
  "https://huggingface.co/datasets/axentx/costinel-top-hub/resolve/main/index.json";

const DEFAULT_TIMEOUT = 4500;
const CACHE_TTL_MS = 5 * 60 * 1000; // 5m

export interface TopHubRelatedDoc {
  id: string;
  label: string;
  snippet: string;
  url: string;
  hubRefs: string[];
}

export interface TopHubPayload {
  generatedAt: string;
  topHub: {
    id: string;
    label: string;
    description: string;
    docCount: number;
    url: string;
  };
  related: TopHubRelatedDoc[];
}

async function fetchWithTimeout(url: string, timeoutMs: number): Promise<Response> {
  const controller = new AbortController();
  const id = setTimeout(() => controller.abort(), timeoutMs);
  try {
    return await fetch(url, {
      method: "GET",
      headers: {
        Accept: "application/json",
      },
      signal: controller.signal,
      cache: "no-store",
    });
  } finally {
    clearTimeout(id);
  }
}

let memoryCache: { payload: TopHubPayload; ts: number } | null = null;

export async function fetchTopHubIndex(): Promise<TopHubPayload | null> {
  const now = Date.now();
  // Serve fresh in-memory cache if valid
  if (memoryCache && now - memoryCache.ts < CACHE_TTL_MS) {
    return memoryCache.payload;
  }

  try {
    const res = await fetchWithTimeout(HUB_INDEX_URL, DEFAULT_TIMEOUT);
    if (!res.ok) throw new Error(`CDN fetch failed: ${res.status}`);
    const json = (await res.json()) as TopHubPayload;
    // Minimal shape validation
    if (!json?.topHub?.id || !Array.isArray(json.related)) {
      throw new Error("Invalid payload shape");
    }
    memoryCache = { payload: json, ts: now };
    return json;
  } catch (err) {
    console.warn("[TopHub] CDN fetch failed, falling back to cached or empty", err);
    // Best-effort: return stale cache if present
    if (memoryCache) return memoryCache.payload;
    return null;
  }
}
```

**Key decisions**:
- No Authorization header to bypass HF API rate limits.
- 4.5s timeout prevents UI jank.
- In-memory 5m TTL reduces CDN hits while keeping data fresh.
- Stale-while-revalidate behavior: return stale cache on failure for graceful degradation.
