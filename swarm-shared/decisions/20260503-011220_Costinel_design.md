# Costinel / design

## Final decision
**Implement a backend-only `/api/v1/sense/top-hub-signal` endpoint (FastAPI)** that:

- Runs `granite-business-research.sh` **once per day** (idempotent, non-blocking, non-fatal).
- Queries knowledge-RAG for the top-connected hub (MOC) and related actionable docs.
- Returns a compact, stable JSON payload the frontend can render immediately.
- Uses safe subprocess execution, short timeouts, and graceful degradation.

This is the highest-value improvement (<2h), directly applies the top-hub doc insight and business-research + knowledge-RAG patterns, and unifies the strongest, safest choices from both candidates.

# app/api/sense.py
import json
import logging
import subprocess
import os
from datetime import datetime, timezone
from pathlib import Path
from typing import Any, Dict, List
from fastapi import APIRouter, HTTPException

router = APIRouter(prefix="/api/v1/sense", tags=["sense"])
log = logging.getLogger("costinel.sense")

# Paths (adjust to repo root)
REPO_ROOT = Path(__file__).resolve().parent.parent.parent
RESEARCH_SCRIPT = REPO_ROOT / "scripts" / "granite-business-research.sh"
RAG_QUERY_SCRIPT = REPO_ROOT / "scripts" / "knowledge-rag-query.sh"
STATE_DIR = REPO_ROOT / ".state"
STATE_DIR.mkdir(exist_ok=True, parents=True)

def _ran_today(flag_file: Path) -> bool:
    if not flag_file.exists():
        return False
    mtime = datetime.fromtimestamp(flag_file.stat().st_mtime, tz=timezone.utc)
    return mtime.date() == datetime.now(tz=timezone.utc).date()

def _run_granite_research() -> None:
    """Run research script once per UTC day if present; non-fatal on failure."""
    flag = STATE_DIR / "granite-business-research.done"
    if _ran_today(flag):
        log.info("granite-business-research already run today; skipping.")
        return

    if not RESEARCH_SCRIPT.exists():
        log.warning("granite-business-research.sh not found at %s; skipping.", RESEARCH_SCRIPT)
        return

    log.info("Running granite-business-research.sh")
    try:
        result = subprocess.run(
            [str(RESEARCH_SCRIPT)],
            check=True,
            timeout=120,
            capture_output=True,
            text=True,
            cwd=str(REPO_ROOT),
            env={**os.environ, "PATH": "/usr/bin:/bin"},
        )
        flag.touch()
        log.info("granite-business-research completed: %s", result.stdout.strip()[:200])
    except subprocess.CalledProcessError as exc:
        log.error("granite-business-research failed (exit %s): %s", exc.returncode, exc.stderr.strip()[:500])
        # non-fatal; continue to query RAG
    except Exception as exc:
        log.exception("Unexpected error running research script: %s", exc)

def _query_top_hub() -> Dict[str, Any]:
    """
    Query knowledge-RAG for top-connected hub (MOC) and related docs.
    Prefer JSON output from script; fallback to text summary.
    """
    default_payload = {
        "hub": "MOC",
        "summary": "Most-connected operational hub (MOC) — review governance policies and recent anomalies.",
        "relatedDocs": [],
    }

    if RAG_QUERY_SCRIPT.exists():
        try:
            out = subprocess.run(
                [str(RAG_QUERY_SCRIPT), "top-hub", "MOC"],
                check=True,
                timeout=60,
                capture_output=True,
                text=True,
                cwd=str(REPO_ROOT),
            ).stdout.strip()
            if out:
                try:
                    parsed = json.loads(out)
                    # Normalize keys expected by endpoint
                    return {
                        "hub": parsed.get("hub", parsed.get("name", "MOC")),
                        "summary": parsed.get("summary", parsed.get("insights", "")),
                        "relatedDocs": _normalize_docs(parsed.get("relatedDocs", parsed.get("docs", []))),
                    }
                except json.JSONDecodeError:
                    # treat raw text as summary
                    return {
                        "hub": "MOC",
                        "summary": out,
                        "relatedDocs": [],
                    }
        except subprocess.CalledProcessError:
            log.warning("knowledge-rag-query.sh failed; returning minimal payload.")
        except Exception as exc:
            log.exception("Unexpected error querying RAG: %s", exc)

    return default_payload

def _normalize_docs(docs: Any) -> List[Dict[str, str]]:
    if not docs:
        return []
    normalized = []
    for d in docs:
        if isinstance(d, dict):
            normalized.append({
                "title": d.get("title", d.get("name", "Untitled")),
                "snippet": d.get("snippet", d.get("summary", "")),
                "path": d.get("path", ""),
            })
        elif isinstance(d, str):
            normalized.append({"title": d, "snippet": "", "path": d})
    return normalized

@router.get("/top-hub-signal")
def top_hub_signal() -> Dict[str, Any]:
    """
    Run business research (if needed) and return top-hub signal from knowledge-RAG.
    """
    try:
        _run_granite_research()
        payload = _query_top_hub()

        result = {
            "hub": payload.get("hub", "MOC"),
            "summary": payload.get("summary", ""),
            "relatedDocs": payload.get("relatedDocs", []) or [],
            "generatedAt": datetime.now(tz=timezone.utc).isoformat(),
        }
        return result
    except Exception as exc:
        log.exception("Failed to produce top-hub signal")
        raise HTTPException(status_code=500, detail="Failed to produce top-hub signal") from exc
