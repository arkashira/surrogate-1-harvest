# workio / quality

## 1) Diagnosis

- No server-side idempotency enforcement on `/punch` → repeated LINE webhook deliveries or frontend retries create duplicate clock-in/out records.
- Missing DB uniqueness constraint for “one open punch per user” → allows multiple concurrent clock-ins or overlapping records (no `UNIQUE (user_id, clock_out_time IS NULL)`).
- Punch API accepts stale timestamps without monotonicity checks → clock-in can be after clock-out or vice-versa on retry/race.
- Frontend lacks optimistic locking / debounce on punch button → rapid double-taps queue multiple requests before first response.
- No audit trail for punch mutations (who/when/why) → hard to debug duplicates or reconcile reports.

## 2) Proposed change

- **Files**:  
  - `server/src/db/schema.sql` — add partial unique index for open punches.  
  - `server/src/db/migrations/` — new migration for idempotency table + index.  
  - `server/src/routes/punch.ts` — add idempotency key header check and monotonic timestamp validation.  
  - `server/src/middleware/idempotency.ts` — new middleware to enforce idempotency (UPSERT into `punch_idempotency`).  
  - `frontend/src/components/PunchButton.tsx` — add debounce + optimistic state + disabled-while-pending.

### Idempotency middleware

```ts
// server/src/middleware/idempotency.ts
import { Request, Response, NextFunction } from 'express';
import { db } from '../db';

export async function punchIdempotency(
  req: Request,
  res: Response,
  next: NextFunction
) {
  const key = req.headers['idempotency-key'] as string;
  if (!key) return res.status(400).json({ error: 'idempotency-key required' });

  const client = await db.connect();
  try {
    await client.query('BEGIN');

    // Upsert idempotency record; if exists, return existing punch
    const exists = await client.query(
      `SELECT punch_id FROM punch_idempotency WHERE idempotency_key = $1 FOR UPDATE`,
      [key]
    );

    if (exists.rows.length > 0) {
      const punch = await client.query(
        `SELECT * FROM punches WHERE id = $1`,
        [exists.rows[0].punch_id]
      );
      await client.query('COMMIT');
      return res.json(punch.rows[0]);
    }

    // Attach key + client to request for later insert
    (req as any).idempotencyKey = key;
    (req as any).idempotencyClient = client;
    next();
  } catch (err) {
    await client.query('ROLLBACK');
    client.release();
    next(err);
  }
}
```

### Punch route with monotonicity + idempotency commit

```ts
// server/src/routes/punch.ts (excerpt)
import { Router } from 'express';
import { db } from '../db';
import { punchIdempotency } from '../middleware/idempotency';

const router = Router();

router.post('/', punchIdempotency, async (req, res) => {
  const { user_id, type } = req.body; // type: 'in' | 'out'
  const client = (req as any).idempotencyClient;
  const key = (req as any).idempotencyKey;

  try {
    const now = new Date();

    if (type === 'in') {
      // Prevent double open punch (DB unique index will raise; handle gracefully)
      const result = await client.query(
        `INSERT INTO punches (user_id, clock_in_time)
         VALUES ($1, $2)
         RETURNING *`,
        [user_id, now]
      );

      await client.query(
        `INSERT INTO punch_idempotency (idempotency_key, user_id, punch_id)
         VALUES ($1, $2, $3)`,
        [key, user_id, result.rows[0].id]
      );

      await client.query('COMMIT');
      client.release();
      return res.status(201).json(result.rows[0]);
    }

    if (type === 'out') {
      // Close latest open punch
      const closeResult = await client.query(
        `UPDATE punches
         SET clock_out_time = $1
         WHERE id = (
           SELECT id FROM punches
           WHERE user_id = $2 AND clock_out_time IS NULL
           ORDER BY clock_in_time DESC
           LIMIT 1
         )
         RETURNING *`,
        [now, user_id]
      );

      if (closeResult.rowCount === 0) {
        await client.query('ROLLBACK');
        client.release();
        return res.status(400).json({ error: 'no open punch to close' });
      }

      await client.query(
        `INSERT INTO punch_idempotency (idempotency_key, user_id, punch_id)
         VALUES ($1, $2, $3)`,
        [key, user_id, closeResult.rows[0].id]
      );

      await client.query('COMMIT');
      client.release();
      return res.json(closeResult.rows[0]);
    }

    await client.query('ROLLBACK');
    client.release();
    return res.status(400).json({ error: 'invalid type' });
  } catch (err: any) {
    await client.query('ROLLBACK');
    client.release();

    // Unique violation from double clock-in or idempotency race
    if (err.code === '23505') {
      return res.status(409).json({ error: 'duplicate or conflict' });
    }

    next(err);
  }
});
```

### Frontend debounce + optimistic state

```tsx
// frontend/src/components/PunchButton.tsx
import { useState, useCallback } from 'react';
import axios from 'axios';

export function PunchButton({ userId, type }: { userId: number; type: 'in' | 'out' }) {
  const [pending, setPending] = useState(false);
  const [lastPunch, setLastPunch] = useState<any>(null);

  const punch = useCallback(async () => {
    if (pending) return;
    setPending(true);
    const key = `${userId}-${type}-${Date.now()}`;

    try {
      const res = await axios.post('/punch', { user_id: userId, type }, {
        headers: { 'idempotency-key': key }
      });
      setLastPunch(res.data);
    } catch (err: any) {
      if (err.response?.status !== 409) {
        console.error(err);
      }
    } finally {
      setPending(false);
    }
  }, [userId, type, pending]);

  return (
    <button
      onClick={punch}
      disabled={pending}
      className={`px-4 py-2 rounded font-medium transition ${
        pending
          ? 'bg-gray-300 cursor-not-allowed'
          : type === 'in'
          ? 'bg-green-600 text-white hover:bg-green-700'
          : 'bg-red-600 text-white hover:bg-red-700'
      }`}
    >
      {pending ? 'กำลังบันทึก...' : type === 'in' ? 'Clock In' : 'Clock Out'}
    </button>
  );
}
```
