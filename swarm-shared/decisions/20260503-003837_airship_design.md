## Implementation Plan — Manifest-Driven CDN-Only Dataset Loader

**Highest-value improvement**: Replace `load_dataset`/`list_repo_files` usage in surrogate training with a manifest-driven, CDN-only iterable loader to eliminate HF API rate limits and mixed-schema ingestion failures.

### Scope (≤2h)
- Add a manifest generator (run once on Mac) that lists parquet files for a date folder via `list_repo_tree` and saves `file_manifest.json`.
- Add a CDN-only iterable dataset class that streams parquet files directly from `https://huggingface.co/datasets/{repo}/resolve/main/...` with zero API calls during training.
- Project only `{prompt, response}` at parse time; ignore extra columns.
- Drop `source`/`ts` cols; keep attribution via filename pattern `batches/mirror-merged/{date}/{slug}.parquet`.
- Ensure Lightning Studio reuse and graceful idle-stop handling.

### 1) Manifest generator (run on Mac)

`scripts/build_file_manifest.py`
```python
#!/usr/bin/env python3
"""
Generate file_manifest.json for a date folder in a HF dataset repo.
Run from Mac after rate-limit window clears.
"""
import json
import os
import sys
from pathlib import Path
from huggingface_hub import HfApi

HF_REPO = os.getenv("HF_DATASET_REPO", "axentx/surrogate-mirror")
DATE_FOLDER = os.getenv("DATE_FOLDER", "2026-05-03")  # e.g. batches/mirror-merged/2026-05-03
OUT_PATH = Path(os.getenv("OUT_MANIFEST", "file_manifest.json"))

def main() -> None:
    api = HfApi()
    prefix = f"batches/mirror-merged/{DATE_FOLDER}/"
    # recursive=False keeps API calls minimal; list once.
    entries = api.list_repo_tree(
        repo_id=HF_REPO,
        path=prefix,
        repo_type="dataset",
        recursive=False,
    )
    files = [e.rfilename for e in entries if e.rfilename.endswith(".parquet")]
    if not files:
        print(f"No parquet files found under {prefix}")
        sys.exit(1)

    manifest = {
        "repo_id": HF_REPO,
        "date_folder": DATE_FOLDER,
        "files": sorted(files),
        "cdn_base": f"https://huggingface.co/datasets/{HF_REPO}/resolve/main",
    }
    OUT_PATH.write_text(json.dumps(manifest, indent=2))
    print(f"Wrote {len(files)} files to {OUT_PATH}")

if __name__ == "__main__":
    main()
```

Make executable:
```bash
chmod +x scripts/build_file_manifest.py
```

Usage:
```bash
HF_DATASET_REPO=axentx/surrogate-mirror DATE_FOLDER=batches/mirror-merged/2026-05-03 \
  python scripts/build_file_manifest.py
```

### 2) CDN-only iterable dataset

`surrogate/data/cdn_parquet_dataset.py`
```python
#!/usr/bin/env python3
"""
Iterable dataset that reads parquet files directly from HF CDN (no API/auth).
Projects only {prompt, response} at parse time.
"""
import pyarrow as pa
import pyarrow.parquet as pq
import requests
from torch.utils.data import IterableDataset
from typing import Iterator, Dict, Any
import io
import logging

logger = logging.getLogger(__name__)

class CDNParquetIterable(IterableDataset):
    """
    Args:
        manifest_path: JSON produced by build_file_manifest.py
        columns: tuple of column names to keep (default: ("prompt", "response"))
    """

    def __init__(self, manifest_path: str, columns: tuple = ("prompt", "response")):
        super().__init__()
        import json
        manifest = json.loads(open(manifest_path).read())
        self.cdn_base = manifest["cdn_base"].rstrip("/")
        self.files = manifest["files"]
        self.columns = columns

    def _stream_file(self, rfilename: str) -> Iterator[Dict[str, Any]]:
        url = f"{self.cdn_base}/{rfilename}"
        # CDN downloads bypass HF API rate limits; no Authorization header.
        resp = requests.get(url, timeout=60)
        resp.raise_for_status()
        table = pq.read_table(io.BytesIO(resp.content), columns=self.columns)
        # Convert to dict batches and yield rows
        for batch in table.to_batches():
            cols = {c: batch.column(c).to_pylist() for c in self.columns}
            n = len(next(iter(cols.values())))
            for i in range(n):
                yield {c: cols[c][i] for c in self.columns}

    def __iter__(self) -> Iterator[Dict[str, Any]]:
        for rfilename in self.files:
            try:
                yield from self._stream_file(rfilename)
            except Exception as exc:
                logger.warning("Failed to stream %s: %s", rfilename, exc)
                continue
```
