```python
import requests
import json

def get_file_list(repo, path):
    """Fetch a list of files from the HF CDN."""
    url = f"https://huggingface.co/datasets/{repo}/resolve/main/{path}"
    response = requests.get(url)
    if response.status_code == 200:
        return json.loads(response.content)
    else:
        return []

def download_file(repo, path, filename):
    """Download a file from the HF CDN."""
    url = f"https://huggingface.co/datasets/{repo}/resolve/main/{path}/{filename}"
    response = requests.get(url)
    if response.status_code == 200:
        with open(filename, "wb") as f:
            f.write(response.content)
    else:
        print(f"Failed to download {filename}")

def hf_cdn_bypass(repo, path):
    """Bypass the HF CDN API rate limit by downloading files directly."""
    file_list = get_file_list(repo, path)
    for file in file_list:
        download_file(repo, path, file)
```
