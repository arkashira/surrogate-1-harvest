# Costinel / design

## Final Implementation Plan — Top Hub Signal Panel (CDN-first)

**Scope**: Frontend-only addition to Costinel dashboard.  
**Effort**: ~60–90 minutes.  
**Mechanism**: CDN JSON fetch (no auth, no backend) with local fallback, retry/backoff, and graceful degradation.

### 2) CDN helper (retry + timeout)

`src/config/cdn.ts`
```ts
const CDN_BASE = 'https://huggingface.co/datasets/axentx/knowledge-rag/resolve/main';
const TIMEOUT_MS = 4000;
const RETRIES = 2;

export async function fetchCdnJson<T>(path: string): Promise<T | null> {
  const url = `${CDN_BASE}/${path.replace(/^\/+/, '')}`;
  for (let i = 0; i <= RETRIES; i++) {
    try {
      const controller = new AbortController();
      const timer = setTimeout(() => controller.abort(), TIMEOUT_MS);
      const res = await fetch(url, { signal: controller.signal, cache: 'no-store' });
      clearTimeout(timer);
      if (!res.ok) throw new Error(`HTTP ${res.status}`);
      return (await res.json()) as T;
    } catch (err) {
      if (i === RETRIES) return null;
      // small backoff
      await new Promise((r) => setTimeout(r, 300 * (i + 1)));
    }
  }
  return null;
}
```

### 3) TopHubSignalPanel component

`src/components/TopHubSignalPanel.tsx`
```tsx
import { useEffect, useState } from 'react';
import { fetchCdnJson } from '../config/cdn';
import './TopHubSignalPanel.css';

type HubPayload = {
  hub: string;
  score: number;
  insight: string;
  updatedAt: string;
  source: string;
};

const LOCAL_FALLBACK: HubPayload = {
  hub: 'MOC',
  score: 0.94,
  insight:
    'MOC is the most-connected hub (2026-04-27). Prioritize MOC-linked cost anomalies for fastest signal propagation.',
  updatedAt: new Date().toISOString(),
  source: 'local-fallback',
};

export default function TopHubSignalPanel() {
  const [data, setData] = useState<HubPayload>(LOCAL_FALLBACK);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    let mounted = true;
    fetchCdnJson<HubPayload>('top-hub/latest.json').then((res) => {
      if (mounted && res) {
        setData(res);
      }
      setLoading(false);
    });
    return () => {
      mounted = false;
    };
  }, []);

  return (
    <div className="top-hub-panel" title={`Updated ${data.updatedAt}`}>
      <div className="top-hub-header">
        <span className="top-hub-badge">Top Hub</span>
        <span className="top-hub-name">{data.hub}</span>
        <span className="top-hub-score">
          {typeof data.score === 'number' && `${Math.round(data.score * 100)}%`}
        </span>
      </div>
      <p className="top-hub-insight">
        {loading ? 'Loading signal…' : data.insight}
      </p>
    </div>
  );
}
```
