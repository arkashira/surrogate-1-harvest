### scripts/list_hf_files.py
```python
#!/usr/bin/env python3
"""
List parquet files in a single date folder (non-recursive) for a HuggingFace dataset.
Run once from Mac after rate-limit window clears.

Usage:
    python scripts/list_hf_files.py \
        --repo org/surrogate-data \
        --date-folder 2026-05-03 \
        --out surrogate/file_list.json
"""

import argparse
import json
import os
import sys

try:
    from huggingface_hub import HfApi
except ImportError:
    print("Install: pip install huggingface_hub")
    sys.exit(1)

HF_TOKEN = os.getenv("HF_TOKEN", "")  # optional; public repos don't require auth for tree

def main() -> None:
    parser = argparse.ArgumentParser(description="List HF dataset files (non-recursive).")
    parser.add_argument("--repo", required=True, help="HF dataset repo, e.g. org/surrogate-data")
    parser.add_argument("--date-folder", required=True, help="Folder path in repo, e.g. 2026-05-03")
    parser.add_argument("--out", default="surrogate/file_list.json", help="Output JSON path")
    args = parser.parse_args()

    api = HfApi(token=HF_TOKEN or None)
    # Non-recursive to avoid pagination explosion
    entries = api.list_repo_tree(repo_id=args.repo, path=args.date_folder, recursive=False)

    files = []
    for entry in entries:
        if entry.rfilename.endswith(".parquet"):
            files.append(entry.rfilename)

    payload = {
        "repo": args.repo,
        "date_folder": args.date_folder,
        "files": sorted(files),
    }

    os.makedirs(os.path.dirname(os.path.abspath(args.out)), exist_ok=True)
    with open(args.out, "w", encoding="utf-8") as f:
        json.dump(payload, f, indent=2)

    print(f"Wrote {len(files)} parquet files to {args.out}")

if __name__ == "__main__":
    main()
```

CDN_BASE = "https://huggingface.co/datasets"
FILE_LIST_PATH = Path(__file__).parent.parent / "surrogate" / "file_list.json"

def load_file_list() -> Dict:
    if not FILE_LIST_PATH.exists():
        raise FileNotFoundError(f"Missing {FILE_LIST_PATH}. Run scripts/list_hf_files.py first.")
    with open(FILE_LIST_PATH, encoding="utf-8") as f:
        return json.load(f)

def cdn_url(repo: str, path: str) -> str:
    return f"{CDN_BASE}/{repo}/resolve/main/{path}"

def stream_parquet_cdn(url: str):
    # CDN public URLs require no Authorization header
    resp = requests.get(url, timeout=60)
    resp.raise_for_status()
    return pq.read_table(pq.ParquetFile(pq.BufferReader(resp.content)))

def build_dataset(repo: str, files: List[str], max_files: int = None):
    rows = []
    for i, path in enumerate(files):
        if max_files is not None and i >= max_files:
            break
        url = cdn_url(repo, path)
        tbl = stream_parquet_cdn(url)
        df = tbl.to_pandas()
        # Project to {prompt, response} only (schema-safe)
        if "prompt" not in df.columns or "response" not in df.columns:
            # Best-effort fallback: pick first text-like pair
            text_cols = [c for c in df.columns if df[c].dtype == "object"]
            if len(text_cols) >= 2:
                df = df.rename(columns={text_cols[0]: "prompt", text_cols[1]: "response"})
            else:
                continue
        rows.append(df[["prompt", "response"]])
    if not rows:
        raise ValueError("No valid prompt/response data found.")
    return pd.concat(rows, ignore_index=True)
