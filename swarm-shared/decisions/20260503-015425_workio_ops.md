# workio / ops

## Final synthesized solution (correct + actionable)

### 1. Diagnosis (merged, de-duplicated)
- **No idempotency enforcement** — LINE retries with identical payload/signature can create duplicate punch rows.
- **Non-atomic upsert** — `findOne → update/insert` races under concurrent retries or multi-instance deployments.
- **Missing DB-level guardrails** — no unique constraint/index to prevent duplicate punches or replayed webhook processing.
- **No replay-protection window** — identical requests within the LINE retry window are treated as new punches.
- **Observability gap** — no structured logs/metrics to detect duplicates or idempotency hits.

### 2. Scope (minimal, high-leverage)
- One migration: idempotency table + unique constraint on punches.
- One handler change: verify signature, enforce idempotency, atomic upsert.
- One small service: idempotency check + atomic punch upsert.
- Add structured logging/metrics for duplicates and processing outcomes.

#### 3.2 Idempotency service
```ts
// workio/server/src/services/idempotency.ts
import { Pool } from 'pg';
import crypto from 'crypto';

const pool = new Pool({ connectionString: process.env.DATABASE_URL });

export async function ensureIdempotent(
  key: string,
  payload: string,
  resultType: string,
  resultRef?: string
): Promise<{ isNew: boolean; existingRef?: string }> {
  const payloadHash = crypto.createHash('sha256').update(payload).digest('hex');
  const client = await pool.connect();
  try {
    await client.query('BEGIN');
    const exists = await client.query(
      `SELECT payload_hash, processed_at, result_ref FROM line_webhook_idempotency WHERE idempotency_key = $1 FOR UPDATE`,
      [key]
    );

    if (exists.rows.length > 0) {
      const row = exists.rows[0];
      if (row.processed_at && row.payload_hash === payloadHash) {
        // already processed same payload — safe to skip
        await client.query('COMMIT');
        return { isNew: false, existingRef: row.result_ref || undefined };
      }
      // same key, different payload: reject (possible replay attack or key collision)
      await client.query('ROLLBACK');
      throw new Error('Idempotency key collision with different payload');
    }

    await client.query(
      `INSERT INTO line_webhook_idempotency (idempotency_key, payload_hash, processed_at, result_type, result_ref)
       VALUES ($1, $2, NOW(), $3, $4)`,
      [key, payloadHash, resultType, resultRef || null]
    );
    await client.query('COMMIT');
    return { isNew: true };
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  } finally {
    client.release();
  }
}
```

#### 3.3 Atomic punch upsert
```ts
// workio/server/src/services/punch.ts
import { Pool } from 'pg';
const pool = new Pool({ connectionString: process.env.DATABASE_URL });

export async function upsertPunch(params: {
  userId: number;
  tenantId: number;
  punchType: 'IN' | 'OUT';
  latitude?: number;
  longitude?: number;
  lineUserId: string;
}): Promise<{ id: number; existed: boolean }> {
  const { userId, tenantId, punchType, latitude, longitude, lineUserId } = params;
  const client = await pool.connect();
  try {
    await client.query('BEGIN');
    // Use the unique index semantics: one active punch per user/tenant/date/type.
    // Try to find today's record of same type to make this idempotent.
    const existing = await client.query(
      `SELECT id FROM punches
       WHERE user_id = $1 AND tenant_id = $2 AND punch_type = $3
         AND DATE(created_at) = CURRENT_DATE
         AND deleted_at IS NULL
       ORDER BY created_at DESC LIMIT 1 FOR UPDATE`,
      [userId, tenantId, punchType]
    );

    if (existing.rows.length > 0) {
      await client.query('COMMIT');
      return { id: existing.rows[0].id, existed: true };
    }

    const res = await client.query(
      `INSERT INTO punches (user_id, tenant_id, punch_type, latitude, longitude, line_user_id)
       VALUES ($1, $2, $3, $4, $5, $6)
       RETURNING id`,
      [userId, tenantId, punchType, latitude, longitude, lineUserId]
    );
    await client.query('COMMIT');
    return { id: res.rows[0].id, existed: false };
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  } finally {
    client.release();
  }
}
```

import crypto from 'crypto';

  const signature = req.headers['x-line-signature'] as string;
  const body = JSON.stringify(req.body);
  const channelSecret = process.env.LINE_CHANNEL_SECRET!;
  const expected = crypto
    .createHmac('sha256', channelSecret)
    .update(body)
    .digest('base64');

  if (signature !== expected) {
    return res.status(401).send('Invalid signature');
  }

      // Deterministic, replay-safe idempotency key
      const idempotencyKey = `line:${ev.source.userId}:${ev.timestamp}:${ev.message.id}`;
      const idRes = await ensureIdempotent(idempotencyKey, body, 'line-webhook', ev.source.userId);

          msg: 'idempotent_skip',
