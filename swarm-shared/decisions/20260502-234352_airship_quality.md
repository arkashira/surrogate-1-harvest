## Implementation Plan — `airship discover` (CDN-only manifest)

**Goal (≤2h):**  
Add a deterministic CLI command `airship discover` that produces a CDN-only file manifest for a given `repo_id` and `date_folder`.  
- Uses **one** `list_repo_tree` call (non-recursive) from the Mac orchestrator.  
- Saves a JSON manifest that Lightning training consumes via CDN-only fetches (zero HF API calls during training).  
- Avoids `load_dataset(streaming=True)` and heterogeneous schemas.  
- Reuses running Lightning Studio when possible.

#### `airship/lib/hf_cdn.py`
```python
from __future__ import annotations

import json
import time
from datetime import datetime, timezone
from pathlib import Path
from typing import Dict, List

import requests
from hugginggingface import HfApi  # huggingface-hf

CDN_ROOT = "https://huggingface.co/datasets"
API_RATE_LIMIT_RESET = 360  # seconds after 429

def build_cdn_url(repo_id: str, file_path: str) -> str:
    # repo_id may include namespace; file_path must not start with /
    clean_path = file_path.lstrip("/")
    return f"{CDN_ROOT}/{repo_id}/resolve/main/{clean_path}"

def list_date_folder(
    repo_id: str,
    date_folder: str,
    token: str | None = None,
) -> List[Dict]:
    """
    Single non-recursive tree call for one date folder.
    Returns list of dicts:
      {"repo_id": ..., "path": ..., "cdn_url": ..., "size": ...}
    """
    api = HfApi(token=token)
    folder = date_folder.rstrip("/")
    attempt = 0

    while True:
        try:
            items = api.list_repo_tree(
                repo_id=repo_id,
                path=folder,
                repo_type="dataset",
                recursive=False,
            )
            break
        except requests.HTTPError as exc:
            if exc.response.status_code == 429:
                wait = API_RATE_LIMIT_RESET
                print(f"[cdn] rate-limited 429 — waiting {wait}s")
                time.sleep(wait)
                attempt += 1
                if attempt > 3:
                    raise
                continue
            raise

    out = []
    for item in items:
        if item.type != "file":
            continue
        out.append(
            {
                "repo_id": repo_id,
                "path": f"{folder}/{item.path[len(folder):].lstrip('/')}",
                "cdn_url": build_cdn_url(repo_id, f"{folder}/{item.path[len(folder):].lstrip('/')}"),
                "size": getattr(item, "size", None),
            }
        )
    return out

def save_manifest(entries: List[Dict], out_path: Path) -> Path:
    manifest = {
        "generated_at": datetime.now(timezone.utc).isoformat(),
        "repo_id": entries[0]["repo_id"] if entries else None,
        "count": len(entries),
        "strategy": "cdn-only",
        "files": entries,
    }
    out_path.parent.mkdir(parents=True, exist_ok=True)
    out_path.write_text(json.dumps(manifest, indent=2))
    return out_path
```

#### `airship/lib/lightning_reuse.py`
```python
from lightning import Studio, Teamspace
from typing import Optional

def reuse_or_create_studio(
    name: str,
    machine: str = "L40S",
    idle_timeout_minutes: int = 15,
) -> Studio:
    """
    Reuse a running Studio if present; otherwise create one.
    Avoids recreating to save Lightning quota.
    """
    teamspace = Teamspace()
    for s in teamspace.studios:
        if s.name == name and s.status == "running":
            print(f"[lightning] reusing running studio: {name}")
            return s

    print(f"[lightning] creating studio: {name} on {machine}")
    return Studio(
        name=name,
        machine=machine,
        idle_timeout_minutes=idle_timeout_minutes,
        create_ok=True,
    )

def ensure_running(studio: Studio, machine: str = "L40S") -> Studio:
    """
    If studio stopped, restart it. Lightning idle stop kills training.
    """
    if studio.status != "running":
        print(f"[lightning] studio not running ({studio.status}) — restarting")
        studio.start(machine=machine)
    return studio
```

#### `airship/commands/discover_cmd.py`
```python
from pathlib import Path

from airship.lib.hf_cdn import list_date_folder, save_manifest

def run_discover(
    repo_id: str,
    date_folder: str,
    out_file: Path,
    hf_token: str | None = None,
) -> Path:
    """
    CLI action: produce deterministic CDN manifest.
    """
    print(f"[discover] repo={repo_id} folder={date_folder}")
    entries = list_date_folder(repo_id=repo_id, date_folder=date_folder, token=hf_token)
    out = save_manifest(entries, out_file)
    print(f"[discover] manifest written: {out} ({len(entries)} files)")
    return out
```

#### `airship/cli/discover.py`
```python
import argparse
from pathlib import Path

from airship.commands.discover_cmd import run_discover

def main() -> None:
    parser = argparse.ArgumentParser(description="Generate CDN-only manifest for HF dataset folder")
    parser.add_argument("--repo-id", required=True, help="HF dataset repo id (e.g., org/repo)")
    parser.add_argument("--date-folder", required=True, help="Folder path in repo (e.g., batches/mirror-merged/2026-05-02)")
    parser.add_argument("--out", type=Path, required=True, help="Output JSON manifest path")
    parser.add_argument("--hf-token", default=None, help="HF token (optional; public CDN bypasses auth)")

    args = parser.parse_args()
    run_discover(repo_id=args.repo_id, date_folder=args.date_folder, out_file=args.out, hf_token=args.hf_token)

if __name__ == "__main__":
    main()
```

  --date-folder batches/mirror-merged/2026-05-02 \

### 4) Training script expectations (minimal)

`train.py` should:
- Load the JSON manifest.
- Iterate `files` and download via `cdn_url` using `requests` or `torch.hub`/`wget` (no HF API).
- Parse each file into `{prompt, response}` at load time (do not rely on `load_dataset` for heterogeneous schemas).
- Upload outputs to HF using sibling repos + hash-slug deterministic routing to respect 128/hr commit cap.

---

### 5) Acceptance criteria

- `airship discover` produces valid JSON with `cdn_url` for every file in the target folder.
- No `load_dataset(streaming=True)` or schema-dependent dataset loading in training path.
- Lightning training uses only CDN URLs (zero HF API calls during data loading).
- Running Studio is reused when available; idle-stop handled by `ensure_running`.
