# airship / design

## Final Synthesized Plan (Correct + Actionable)

**Core improvement** (ships in <2h):  
Turn `airship discover` into a deterministic, **CDN-only orchestrator** that:

1. **Eliminates HF API rate limits** by using `https://huggingface.co/datasets/{repo}/resolve/main/...` URLs (no auth, no `/api/`).
2. **Avoids PyArrow schema errors** by never calling `load_dataset(streaming=True)` on heterogeneous repos; instead download individual files and project `{prompt,response}` at parse time.
3. **Produces reproducible file lists** via a single `list_repo_tree` snapshot saved to JSON, embedded in downstream training so Lightning Studio runs zero API calls during data loading.

### 1. `airship/discover/cdn_orchestrator.py`
```python
from __future__ import annotations

import json
import datetime as dt
from pathlib import Path
from typing import List, Dict

from huggingface_hub import list_repo_tree

CDN_ROOT = "https://huggingface.co/datasets"

def deterministic_file_list(repo_id: str, date_folder: str, out_dir: Path) -> Path:
    """
    Single API call to snapshot tree for one date folder.
    Returns path to saved manifest JSON.
    """
    tree = list_repo_tree(repo_id=repo_id, path=date_folder, recursive=False)
    files = [
        {
            "path": str(node.path),
            "cdn_url": f"{CDN_ROOT}/{repo_id}/resolve/main/{node.path}",
        }
        for node in tree
        if node.type == "file"
    ]

    manifest = {
        "repo": repo_id,
        "date_folder": date_folder,
        "snapshot_ts": dt.datetime.utcnow().isoformat() + "Z",
        "files": files,
    }
    out_dir.mkdir(parents=True, exist_ok=True)
    slug = repo_id.replace("/", "-")
    manifest_path = out_dir / f"{slug}-{date_folder.replace('/', '-')}.json"
    manifest_path.write_text(json.dumps(manifest, indent=2))
    return manifest_path
```

### 2. `airship/discover/safe_parser.py`
```python
from __future__ import annotations

import json
import gzip
import io
import hashlib
from pathlib import Path
from typing import Generator, Tuple

import httpx
import pyarrow.parquet as pq

CDN_ROOT = "https://huggingface.co/datasets"

def download_file(url: str, timeout: int = 30) -> bytes:
    with httpx.Client(timeout=timeout, follow_redirects=True) as client:
        r = client.get(url)
        r.raise_for_status()
        return r.content

def sha256_bytes(data: bytes) -> str:
    return hashlib.sha256(data).hexdigest()

def _guess_text(obj, keys) -> str:
    for k in keys:
        val = obj.get(k)
        if val is not None:
            return str(val)
    return ""

def project_parquet(content: bytes) -> Generator[Tuple[str, str], None, None]:
    tbl = pq.read_table(io.BytesIO(content))
    prompt_col = next((c for c in ["prompt", "instruction", "input"] if c in tbl.column_names), None)
    response_col = next((c for c in ["response", "output", "completion"] if c in tbl.column_names), None)
    if prompt_col is None or response_col is None:
        return
    for i in range(tbl.num_rows):
        prompt = str(tbl[prompt_col][i].as_py()).strip()
        response = str(tbl[response_col][i].as_py()).strip()
        if prompt and response:
            yield prompt, response

def project_jsonl(content: bytes) -> Generator[Tuple[str, str], None, None]:
    stream = io.BytesIO(content)
    if content[:2] == b"\x1f\x8b":
        stream = gzip.GzipFile(fileobj=stream)
    for line in stream:
        obj = json.loads(line)
        prompt = _guess_text(obj, ["prompt", "instruction", "input"]).strip()
        response = _guess_text(obj, ["response", "output", "completion"]).strip()
        if prompt and response:
            yield prompt, response

def parse_file(url: str) -> Generator[Tuple[str, str], None, None]:
    data = download_file(url)
    if url.endswith(".parquet"):
        yield from project_parquet(data)
    elif url.endswith(".jsonl") or url.endswith(".json"):
        yield from project_jsonl(data)
    # skip unknown extensions
```

### 3. `airship/discover/export_manifest.py`
```python
from __future__ import annotations

import json
from pathlib import Path
from typing import Dict, Any

from .cdn_orchestrator import deterministic_file_list
from .safe_parser import download_file, sha256_bytes

def create_enriched_manifest(repo_id: str, date_folder: str, out_root: Path) -> Path:
    manifest_path = deterministic_file_list(repo_id, date_folder, out_root)
    manifest = json.loads(manifest_path.read_text())

    enriched = []
    for f in manifest["files"]:
        url = f["cdn_url"]
        try:
            data = download_file(url)
            f["sha256"] = sha256_bytes(data)
            f["size"] = len(data)
            enriched.append(f)
        except Exception as exc:
            f["error"] = str(exc)
            enriched.append(f)

    slug = repo_id.replace("/", "-")
    enriched_path = out_root / "mirror-merged" / date_folder.replace("/", "-") / f"{slug}.json"
    enriched_path.parent.mkdir(parents=True, exist_ok=True)
    enriched_path.write_text(json.dumps({"repo": repo_id, "date": date_folder, "files": enriched}, indent=2))
    return enriched_path
```

### 4. `airship/cli.py` (excerpt)

```python
import argparse
import json
import os
from pathlib import Path

from airship.discover.export_manifest import create_enriched_manifest

def discover(args):
    manifest = create_enriched_manifest(args.repo, args.date, Path("batches"))
    manifest_data = json.loads(manifest.read_text())
    # Lightning Studio friendly: zero API calls during training
    os.environ["AIRSHIP_MANIFEST_PATH"] = str(manifest.absolute())
    print(f"MANIFEST={manifest.absolute()}")
    print(f"FILES={len(manifest_data['files'])}")
    print("Use manifest with CDN URLs in train.py (no HF API during data load).")
```
