# vanguard / design

## 1. Diagnosis
- No persistent file-list cache for HF repos → training scripts repeatedly call `list_repo_tree`/`load_dataset` and hit 429 rate limits.
- Lightning Studio lifecycle not reused → quota burned on recreation; idle-stop deaths silently kill training.
- Training scripts use `load_dataset(streaming=True)` on heterogeneous repos → pyarrow `CastError` on mixed schemas.
- Data ingestion writes mixed-schema files to `enriched/` with extra metadata columns → downstream training schema drift.
- No CDN-bypass strategy during training → every epoch re-authenticates against `/api/` and consumes rate-limit budget.

## 2. Proposed change
Create `/opt/axentx/vanguard/training/file_index.py` + update `/opt/axentx/vanguard/training/train.py` (or create if absent) to:
- Single API call to `list_repo_tree` (per date folder) → save `file_index.json`.
- Embed file list in training script; use HF CDN URLs (`resolve/main/...`) for zero-auth, zero-rate-limit data fetches.
- Add Lightning Studio reuse + idle-stop guard before `.run()`.
- Project to `{prompt,response}` only; write parquet to `batches/mirror-merged/{date}/{slug}.parquet` (no extra cols).

```bash
# /opt/axentx/vanguard/training/file_index.py
#!/usr/bin/env python3
"""
Generate and cache HF repo file index for a date folder.
Usage:
  python file_index.py --repo datasets/mycorp/surrogate-1 --date 2026-04-29 --out file_index.json
"""
import argparse
import json
import os
import time
from pathlib import Path
from huggingface_hub import HfApi, hf_hub_download

HF_TOKEN = os.getenv("HF_TOKEN", "")
api = HfApi(token=HF_TOKEN)

def build_file_index(repo_id: str, date_folder: str, out_path: Path):
    # Single API call: non-recursive per folder to minimize pagination/429 risk
    entries = api.list_repo_tree(repo_id=repo_id, path=date_folder, recursive=False)
    files = [e.path for e in entries if e.type == "file"]

    index = {
        "repo_id": repo_id,
        "date_folder": date_folder,
        "generated_at": time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime()),
        "files": sorted(files),
        "cdn_base": f"https://huggingface.co/datasets/{repo_id}/resolve/main",
    }
    out_path.parent.mkdir(parents=True, exist_ok=True)
    out_path.write_text(json.dumps(index, indent=2))
    print(f"Wrote {len(files)} files -> {out_path}")
    return index

def download_via_cdn(index, file_rel_path, local_dir):
    local_dir = Path(local_dir)
    local_path = local_dir / file_rel_path
    local_path.parent.mkdir(parents=True, exist_ok=True)
    # CDN download bypasses /api/ auth checks and rate limits
    url = f"{index['cdn_base']}/{file_rel_path}"
    # Use huggingface_hub util (it will stream from CDN without auth overhead when public)
    hf_hub_download(
        repo_id=index["repo_id"],
        filename=file_rel_path,
        repo_type="dataset",
        cache_dir=local_dir,
        force_download=False,
        local_files_only=False,
    )
    # Return cached location
    from huggingface_hub import snapshot_download
    cached = snapshot_download(
        repo_id=index["repo_id"],
        repo_type="dataset",
        allow_patterns=file_rel_path,
        cache_dir=local_dir,
    )
    # Find exact file
    import glob
    found = glob.glob(f"{cached}/**/{file_rel_path}", recursive=True)
    return found[0] if found else None

if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True)
    parser.add_argument("--date", required=True)
    parser.add_argument("--out", default="file_index.json")
    args = parser.parse_args()
    build_file_index(args.repo, args.date, Path(args.out))
```

class CDNTextDataset(Dataset):
    def __init__(self, index_path, repo_id, max_files=None):
        with open(index_path) as f:
            self.index = json.load(f)
        self.files = self.index["files"]
        if max_files:
            self.files = self.files[:max_files]
        self.repo_id = repo_id
        self.tokenizer = AutoTokenizer.from_pretrained("meta-llama/Meta-Llama-3-8B-Instruct", use_fast=True)
        if self.tokenizer.pad_token is None:
            self.tokenizer.pad_token = self.tokenizer.eos_token
        self._cache_dir = Path(".cache_hf")

    def __len__(self):
        return len(self.files)

    def _load_parquet_projected(self, file_rel):
        # Lightweight: read only prompt/response from parquet; ignore extra cols
        import pyarrow.parquet as pq
        import pandas as pd
        from huggingface_hub import hf_hub_download
        # Download via CDN-friendly util (cached)
        local = hf_hub_download(repo_id=self.repo_id, filename=file_rel,
                                repo_type="dataset", cache_dir=self._cache_dir)
        df = pq.read_table(local).to_pandas()
        # Project to expected schema
        if "prompt" not in df.columns or "response" not in df.columns:
            # fallback: try to infer
            text_col = next((c for c in df.columns if "text" in c.lower()), None)
            if text_col:
                # naive split; adapt to your format
                df["prompt"] = df[text_col].apply(lambda x: x.split("### Response:")[0].strip())
                df["response"] = df[text_col].apply(lambda x: x.split("### Response:")[1].strip() if "### Response:" in x else "")
            else:
                raise ValueError(f"Cannot find prompt/response in {file_rel}: {df.columns.tolist()}")
        return df[["prompt", "response"]].dropna()

    def __getitem__(self, idx):
        rel = self.files[idx % len(self.files)]
        df = self._load_parquet_projected(rel)
        # Simple row selection (deterministic)
        row = df.iloc[idx % len(df)]
        prompt = row["prompt"]
        response = row["response"]
        text = f"### Prompt: {prompt}\n### Response: {response}"
        enc = self.tokenizer(text, truncation=True, max_length=2048, padding="max_length", return_tensors="pt")
        return {
            "input_ids": enc["input_ids"].squeeze(0),
            "attention_mask": enc["attention_mask"].squeeze(0),
            "labels": enc["input_ids"].squeeze(0).clone(),
        }
