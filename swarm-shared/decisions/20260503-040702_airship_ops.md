### 1. Mac orchestration script (≤10 min) — one-time per date folder

`scripts/list_hf_date_folder.py`

```python
#!/usr/bin/env python3
"""
Run from Mac (or any dev machine) after HF rate-limit window clears.
Lists a single date folder in a HF dataset repo and saves file manifest.
Training will use CDN-only URLs (no auth, no API calls).
"""
import json
import os
from huggingface_hub import HfApi

HF_REPO = os.getenv("HF_REPO", "my-org/surrogate-mirror-merged")
DATE_FOLDER = os.getenv("DATE_FOLDER", "2026-04-29")  # e.g. batches/mirror-merged/2026-04-29
OUT_FILE = os.getenv("OUT_FILE", "training_file_manifest.json")

def main() -> None:
    api = HfApi()
    # Non-recursive per folder to avoid pagination explosion
    entries = api.list_repo_tree(
        repo_id=HF_REPO,
        path=f"batches/mirror-merged/{DATE_FOLDER}",
        repo_type="dataset",
        recursive=False,
    )
    files = [e.path for e in entries if e.type == "file" and e.path.endswith(".parquet")]
    # Build CDN URLs (no auth, bypasses /api/ rate limits)
    cdn_urls = [
        f"https://huggingface.co/datasets/{HF_REPO}/resolve/main/{path}"
        for path in sorted(files)
    ]
    manifest = {
        "repo": HF_REPO,
        "date_folder": DATE_FOLDER,
        "files": files,
        "cdn_urls": cdn_urls,
    }
    with open(OUT_FILE, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Wrote {len(cdn_urls)} CDN URLs to {OUT_FILE}")

if __name__ == "__main__":
    main()
```

Make executable and run once per date folder:

```bash
chmod +x scripts/list_hf_date_folder.py
python3 scripts/list_hf_date_folder.py
```

### 2. CDN-only parquet loader (training/loader.py)

Uses the manifest produced above. Lightning training will perform **zero HF API calls** during data loading.

```python
# training/loader.py
import json
import pyarrow.parquet as pq
import requests
from io import BytesIO
from typing import List, Dict

def load_cdn_parquet_shard(cdn_url: str, columns=("prompt", "response")) -> List[Dict]:
    """Download single parquet via CDN and project to {prompt, response}."""
    resp = requests.get(cdn_url, timeout=60)
    resp.raise_for_status()
    table = pq.read_table(BytesIO(resp.content), columns=columns)
    return table.to_pylist()

def build_cdn_dataloader(manifest_path: str, batch_size: int = 64):
    with open(manifest_path) as f:
        manifest = json.load(f)

    cdn_urls = manifest["cdn_urls"]
    if not cdn_urls:
        raise ValueError("No CDN URLs in manifest")

    # Simple concatenation for demo; replace with streaming/generator for large data
    records = []
    for url in cdn_urls:
        records.extend(load_cdn_parquet_shard(url))

    # Convert to torch dataset
    import torch
    from torch.utils.data import Dataset, DataLoader

    class PromptResponseDataset(Dataset):
        def __init__(self, records):
            self.records = records

        def __len__(self):
            return len(self.records)

        def __getitem__(self, idx):
            r = self.records[idx]
            return {"prompt": r["prompt"], "response": r["response"]}

    dataset = PromptResponseDataset(records)
    return DataLoader(dataset, batch_size=batch_size, shuffle=True)
```

### 3. Lightning idle-resilient runner (training/run.py)

Reuses running studios and restarts if idle-timeout killed the job.

`training/run.py`

```python
#!/usr/bin/env python3
"""
Lightning training launcher with idle-resilience and studio reuse.
"""
import os
import time
from lightning import LightningWork, LightningFlow, LightningApp, Machine
from lightning.app import BuildConfig

# Replace with your actual training function that uses CDN loader
def train_step():
    from training.loader import build_cdn_dataloader
    loader = build_cdn_dataloader("training_file_manifest.json", batch_size=32)
    print(f"Loaded {len(loader.dataset)} records via CDN")
    # Insert your actual training loop here
    for i, batch in enumerate(loader):
        if i >= 2:  # demo only
            break
        print(f"Batch {i}: prompts={len(batch['prompt'])}")
    print("Training epoch complete (demo).")

class SurrogateTrainer(LightningWork):
    def __init__(self, studio_name: str = "surrogate-trainer", machine: Machine = None):
        super().__init__()
        self.studio_name = studio_name
        self.machine = machine or Machine.L40S  # fallback if H200 unavailable
        self._studio = None

    def run(self):
        from lightning import Teamspace

        # Reuse running studio if available
        for s in Teamspace.studios:
            if s.name == self.studio_name and s.status == "running":
                self._studio = s
                print(f"Reusing running studio: {self.studio_name}")
                break

        if self._studio is None:
            from lightning import Studio
            self._studio = Studio(
                name=self.studio_name,
                machine=self.machine,
                create_ok=True,
            )
            print(f"Created new studio: {self.studio_name}")

        # Ensure studio is running before executing
        if self._studio.status != "running":
            print(f"Studio stopped (idle-timeout?). Restarting on {self.machine}...")
            self._studio.start(machine=self.machine)
            # Wait until running
            while self._studio.status != "running":
                time.sleep(10)

        # Execute training inside studio context
        # Note: LightningWork.run() executes on the orchestration side.
        # For actual training code inside the Studio, use .run() or execute via SSH/notebook.
        # This demo runs training locally in this process (for <2h scope).
        train_step()

class SurrogateFlow(LightningFlow):
    def __init__(self):
        super().__init__()
        self.trainer = SurrogateTrainer()

    def run(self):
        self.trainer.run()

app = LightningApp(SurrogateFlow())
```
