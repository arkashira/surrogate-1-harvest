// workio/src/features/clock/ClockButton.tsx
import { useState, useEffect, useCallback, useRef } from 'react';
import { useMutation, useQueryClient } from '@tanstack/react-query';
import { CheckIn, CheckOut } from '../../api/clock';
import { useLocationWatch } from '../../hooks/useLocationWatch';
import { Button } from '../../components/ui/button';
import { Loader2, MapPin, AlertCircle } from 'lucide-react';

const IDEMPOTENCY_STORAGE_KEY = 'clock_last_idempotency_key';

function generateIdempotencyKey(): string {
  return `${Date.now()}-${crypto.randomUUID?.() || Math.random().toString(36).slice(2)}`;
}

export function ClockButton() {
  const queryClient = useQueryClient();
  const [idempotencyKey, setIdempotencyKey] = useState<string | null>(null);
  const mutationAttemptRef = useRef<string | null>(null);

  const {
    location,
    error: locationError,
    status: locationStatus,
  } = useLocationWatch({
    enableHighAccuracy: true,
    maximumAge: 30_000,
    timeout: 10_000,
  });

  // Restore last idempotency key to avoid duplicates across remounts/retries
  useEffect(() => {
    try {
      const stored = localStorage.getItem(IDEMPOTENCY_STORAGE_KEY);
      if (stored) {
        setIdempotencyKey(stored);
        mutationAttemptRef.current = stored;
      }
    } catch {}
  }, []);

  const clockMutation = useMutation({
    mutationFn: async (type: 'in' | 'out') => {
      if (!location) throw new Error('Location unavailable');
      const key = generateIdempotencyKey();
      setIdempotencyKey(key);
      mutationAttemptRef.current = key;
      try {
        localStorage.setItem(IDEMPOTENCY_STORAGE_KEY, key);
      } catch {}
      return type === 'in'
        ? CheckIn({ idempotencyKey: key, lat: location.latitude, lng: location.longitude })
        : CheckOut({ idempotencyKey: key, lat: location.latitude, lng: location.longitude });
    },
    onMutate: async (type) => {
      // Cancel outgoing refetches so optimistic update isn't overwritten
      await queryClient.cancelQueries({ queryKey: ['clock', 'current'] });

      // Snapshot previous value for rollback
      const previous = queryClient.getQueryData(['clock', 'current']);

      // Optimistic update
      queryClient.setQueryData(['clock', 'current'], {
        type: type === 'in' ? 'in' : 'out',
        timestamp: new Date().toISOString(),
        optimistic: true,
        idempotencyKey: mutationAttemptRef.current,
      });

      // Return context with previous for onError
      return { previous };
    },
    onSuccess: (data) => {
      // Replace optimistic with authoritative record
      queryClient.setQueryData(['clock', 'current'], {
        ...data,
        optimistic: false,
      });
      queryClient.invalidateQueries({ queryKey: ['clock', 'history'] });
      try {
        localStorage.removeItem(IDEMPOTENCY_STORAGE_KEY);
        setIdempotencyKey(null);
        mutationAttemptRef.current = null;
      } catch {}
    },
    onError: (_err, _variables, context) => {
      // Rollback to previous state if we have it
      if (context?.previous) {
        queryClient.setQueryData(['clock', 'current'], context.previous);
      } else {
        // If no previous, clear optimistic so user can retry cleanly
        queryClient.setQueryData(['clock', 'current'], (old: any) =>
          old?.optimistic ? null : old
        );
      }
      queryClient.invalidateQueries({ queryKey: ['clock', 'history'] });
      try {
        localStorage.removeItem(IDEMPOTENCY_STORAGE_KEY);
        setIdempotencyKey(null);
        mutationAttemptRef.current = null;
      } catch {}
    },
    onSettled: () => {
      // Ensure fresh data after settle
      queryClient.invalidateQueries({ queryKey: ['clock', 'current'] });
    },
  });

  const handleClock = useCallback(() => {
    if (locationStatus === 'denied') {
      // Actionable: open browser settings hint or show modal
      alert(
        'Location permission is required to clock in/out. Please enable location access in your browser settings and reload.'
      );
      return;
    }
    if (!location) {
      // Should not happen if button disabled correctly, but guard anyway
      alert('Location not available. Wait for verification or check device GPS.');
      return;
    }
    const current = queryClient.getQueryData(['clock', 'current']) as any;
    const nextType = current?.type === 'in' ? 'out' : 'in';
    clockMutation.mutate(nextType);
  }, [location, locationStatus, clockMutation, queryClient]);

  const current = queryClient.getQueryData(['clock', 'current']) as any;
  const isCheckedIn = current?.type === 'in' && !current?.optimistic;
  const isPending = clockMutation.isPending || current?.optimistic === true;

  const showLocationError = locationStatus === 'denied' || Boolean(locationError);
  const isVerifying = locationStatus === 'loading' || !location;
  const disabled = isVerifying || showLocationError || isPending;

  return (
    <div className="flex flex-col items-center gap-3">
      <Button
        size="lg"
        className={`
          w-full max-w-xs
          ${isCheckedIn ? 'bg-red-500 hover:bg-red-600' : 'bg-green-500 hover:bg-green-600'}
          ${disabled ? 'opacity-70 cursor-not-allowed' : ''}
        `}
        onClick={handleClock}
        disabled={disabled}
        aria-busy={isPending}
      >
        {isPending ? (
          <>
            <Loader2 className="mr-2 h-4 w-4 animate-spin" />
            {isCheckedIn ? 'Clocking out...' : 'Clocking in...'}
          </>
        ) : isCheckedIn ? (
          <>
            <MapPin className="mr-2 h-4 w-4" />
            Clock out
          </>
        ) : (
          <>
            <MapPin className="mr-2 h-4 w-4" />
            Clock in
          </>
        )}
      </Button>

      {showLocationError && (
        <div className="flex items-center gap-2 text-xs text-red-500">
          <AlertCircle className="h-3 w-3" />
          Location permission required — enable GPS or allow browser prompt.
        </div>
      )}

      {locationStatus === 'loading' && location && (
        <div className="text-xs text-muted-foreground">Verifying location...</div>
      )}
    </div>
  );
}
