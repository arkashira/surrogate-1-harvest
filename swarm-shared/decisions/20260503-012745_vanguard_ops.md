# vanguard / ops

## 1. Diagnosis

- Frontend still triggers authenticated `list_repo_tree` on page/training load, burning HF API quota (1000/5min) and causing 429s.
- No persisted `(repo, dateFolder) → file-list` artifact; each session re-enumerates folders via API instead of using a cached JSON manifest.
- Training/data-loading paths likely use authenticated `/api/` endpoints instead of public CDN URLs (`https://huggingface.co/datasets/{repo}/resolve/main/{path}`), which bypass auth and rate limits.
- No guard to reuse existing Lightning Studio; training jobs likely recreate studios and waste 80hr/mo quota.
- No check for Lightning idle-stop before `.run()`; stopped studios cause silent training failures.

## 2. Proposed change

Create `/opt/axentx/vanguard/ops/preflight_training.py` (single orchestration script) that:
- Accepts `repo`, `date_folder`
- If cached `file-list-{date_folder}.json` missing or stale, performs **one** `list_repo_tree` call (non-recursive) and saves it
- Generates a small `train_config.json` with CDN-only file URLs and repo identity
- Reuses a running Lightning Studio if available; otherwise starts one (L40S priority, fallback to public tier)
- Verifies studio is running before handing off to training

Also add `/opt/axentx/vanguard/ops/cdn_train_loader.py` snippet to be copied into training code to ensure zero API calls during data loading.

### /opt/axentx/vanguard/ops/preflight_training.py
```python
#!/usr/bin/env python3
"""
Preflight for Surrogate-1 training (ops).
- Persists repo+date -> file-list (non-recursive) to avoid repeated HF API calls
- Emits train_config.json with CDN-only URLs
- Reuses or starts a Lightning Studio (L40S -> public fallback)
"""
import json, os, sys, time
from datetime import datetime, timezone
from pathlib import Path

try:
    import lightning as L
    from lightning.fabric.accelerators import L40S
except Exception as e:
    print("Lightning SDK unavailable; install via `pip install lightning` if needed.")
    L = None

try:
    from huggingface_hub import list_repo_tree, HfApi
except Exception as e:
    print("huggingface_hub unavailable; install via `pip install huggingface_hub`.")
    sys.exit(1)

HF_REPO = os.getenv("HF_REPO", "datasets/axentx/surrogate-1")
DATE_FOLDER = os.getenv("DATE_FOLDER", datetime.now(timezone.utc).strftime("%Y-%m-%d"))
CACHE_DIR = Path(__file__).parent / ".cache"
CACHE_DIR.mkdir(exist_ok=True)

def cached_file_list(repo: str, date_folder: str, max_age_seconds: int = 3600):
    cache_path = CACHE_DIR / f"file-list-{date_folder}.json"
    now = time.time()

    if cache_path.exists():
        mtime = cache_path.stat().st_mtime
        if (now - mtime) < max_age_seconds:
            print(f"[cache] using existing {cache_path}")
            return json.loads(cache_path.read_text())

    print(f"[api] listing {repo}/{date_folder} (non-recursive)...")
    items = list_repo_tree(repo=repo, path=date_folder, recursive=False)
    files = [p.rstrip("/").split("/")[-1] for p in items if p.strip("/")]
    payload = {"repo": repo, "date_folder": date_folder, "files": files, "listed_at": now}
    cache_path.write_text(json.dumps(payload, indent=2))
    print(f"[cache] saved {len(files)} entries to {cache_path}")
    return payload

def build_train_config(payload):
    repo = payload["repo"]
    date = payload["date_folder"]
    files = payload["files"]
    cdn_urls = [
        f"https://huggingface.co/datasets/{repo}/resolve/main/{date}/{f}"
        for f in files
    ]
    cfg = {
        "dataset_repo": repo,
        "date_folder": date,
        "file_count": len(files),
        "cdn_file_urls": cdn_urls,
        "created_at": datetime.now(timezone.utc).isoformat(),
        "notes": "Use these URLs directly (no auth) during training to bypass HF API rate limits."
    }
    out = CACHE_DIR / "train_config.json"
    out.write_text(json.dumps(cfg, indent=2))
    print(f"[config] written {out}")
    return cfg

def ensure_lightning_studio(name="vanguard-surrogate1-train", machine="L40S", fallback_machine="L40"):
    if L is None:
        print("[lightning] SDK unavailable; skipping studio checks.")
        return None

    teamspace = L.Teamspace
    for s in teamspace.studios():
        if s.name == name and s.status == "running":
            print(f"[lightning] reusing running studio: {s.name} ({s.id})")
            return s

    print(f"[lightning] no running studio '{name}'; starting one...")
    try:
        studio = L.Studio(
            name=name,
            machine=eval(f"L.Machine.{machine}"),
            create_ok=True,
        )
        print(f"[lightning] started studio: {studio.name} ({studio.id})")
        return studio
    except Exception as e:
        print(f"[lightning] failed with {machine}: {e}; trying fallback {fallback_machine}")
        try:
            studio = L.Studio(
                name=name,
                machine=eval(f"L.Machine.{fallback_machine}"),
                create_ok=True,
            )
            return studio
        except Exception as e2:
            print(f"[lightning] fallback failed: {e2}")
            return None

def main():
    payload = cached_file_list(HF_REPO, DATE_FOLDER)
    if not payload["files"]:
        print("[error] no files found; check repo/date_folder.")
        sys.exit(1)

    cfg = build_train_config(payload)
    studio = ensure_lightning_studio()

    if studio:
        # Verify studio is running before handing off
        if studio.status != "running":
            print(f"[lightning] studio not running (status={studio.status}); restarting...")
            studio.start(machine=L.Machine.L40S)
            # wait briefly
            for _ in range(10):
                studio.refresh()
                if studio.status == "running":
                    break
                time.sleep(6)

        print(f"[ready] studio={studio.name} status={studio.status}")
        print(f"[ready] run training with config: {CACHE_DIR / 'train_config.json'}")
    else:
        print("[warning] no studio available; run training manually with config.")

    # Print concise instructions for next step
    print("\n--- Next step (example) ---")
    print("python train.py --config .cache/train_config.json")

if __name__ == "__main__":
    main()
```

# Include this in your Lightning training script or data module.
# It ensures zero authenticated HF API calls during data loading.
import json
from pathlib import Path
from torch.utils.data import IterableDataset
import requests

class CDNParquetIterable(IterableDataset):
    def __init__(self, config_path):
        cfg = json.loads(Path(config_path).read_text())
        self.urls = cfg["cdn_file_urls"]

    def __iter__(self):
        for url in self.urls:
            # stream download from CDN (no auth)
            with requests.get(url, stream=True, timeout=60) as r:
                r.raise_for_status()
                # project to {prompt, response} here; example:
                # table = pq.read_table(io.BytesIO(r.content))
                # for row in table.to_pylist():
                #     yield {"prompt": row["prompt"], "response": row["response"]}
                yield {"url": url, "size": len(r.content)}
