#### A) Hardened orchestrator (`scripts/airship-discover`)

```bash
#!/usr/bin/env bash
# airship-discover — hardened orchestrator
# Usage: bash scripts/airship-discover <date-folder>
set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
REPO_ROOT="$(cd "${SCRIPT_DIR}/../" && pwd)"
DATE_FOLDER="${1:-$(date +%Y-%m-%d)}"
MANIFEST="${REPO_ROOT}/file-manifest.json"
HF_REPO="datasets/your-org/surrogate-ingest"
HF_BASE_CDN="https://huggingface.co/datasets/${HF_REPO}/resolve/main"

log() {
  echo "[$(date -u +"%Y-%m-%dT%H:%M:%SZ")] $*"
}

retry_cdn() {
  local url="$1"
  local out="$2"
  local max_retries=5
  local attempt=0
  local wait_sec=10
  local max_wait=360

  while (( attempt < max_retries )); do
    if curl -fsSL --retry 3 --retry-delay 2 -o "${out}" "${url}"; then
      return 0
    fi
    local rc=$?
    log "Download failed (attempt=$((attempt+1))/$max_retries): ${url} -> rc=${rc}"

    # Detect HTTP 429 via response header when possible; conservative backoff
    log "Backing off ${wait_sec}s"
    sleep "${wait_sec}"

    wait_sec=$(( wait_sec * 2 ))
    if (( wait_sec > max_wait )); then
      wait_sec=${max_wait}
    fi
    attempt=$(( attempt + 1 ))
  done
  log "ERROR: Failed to download after ${max_retries} attempts: ${url}"
  return 1
}

build_manifest() {
  log "Building file manifest for ${DATE_FOLDER}"
  python3 "${REPO_ROOT}/scripts/build-file-manifest.py" \
    --repo "${HF_REPO}" \
    --date-folder "${DATE_FOLDER}" \
    --out "${MANIFEST}"
}

download_manifest_files() {
  if [[ ! -f "${MANIFEST}" ]]; then
    log "ERROR: Manifest not found: ${MANIFEST}"
    log "Run build_manifest first (from runner with HF API access) or commit file-manifest.json."
    return 1
  fi

  local out_dir="${REPO_ROOT}/data/cdn-ingest/${DATE_FOLDER}"
  mkdir -p "${out_dir}"

  local files
  # Use null-delimited parsing for robustness with special chars
  files=$(jq -r '.files[]' "${MANIFEST}")
  local total
  total=$(jq '.files | length' "${MANIFEST}")
  local count=0

  for rel in ${files}; do
    count=$((count + 1))
    local url="${HF_BASE_CDN}/${rel}"
    local dest="${out_dir}/$(basename "${rel}")"
    log "(${count}/${total}) CDN fetch: ${url}"
    retry_cdn "${url}" "${dest}"
  done

  log "Downloaded ${count} files to ${out_dir}"
  echo "${out_dir}"
}

main() {
  log "Start airship-discover for ${DATE_FOLDER}"
  if [[ ! -f "${MANIFEST}" ]]; then
    log "Manifest missing — attempting one-time build (requires HF API access)."
    build_manifest || true
  fi
  download_manifest_files
  log "Done airship-discover for ${DATE_FOLDER}"
}

main "$@"
```

#### B) One-shot manifest builder (`scripts/build-file-manifest.py`)

```python
#!/usr/bin/env python3
"""
build-file-manifest.py
Run from a runner (with HF token) to list files for a date folder
and emit file-manifest.json for CDN-only training.
"""
import argparse
import json
import os
import sys

try:
    from huggingface_hub import list_repo_tree
except ImportError:
    print("ERROR: huggingface_hub not installed. Install via: pip install huggingface_hub", file=sys.stderr)
    sys.exit(1)

def main() -> None:
    parser = argparse.ArgumentParser(description="Build CDN file manifest for a date folder")
    parser.add_argument("--repo", required=True, help="HF repo (e.g., datasets/your-org/surrogate-ingest)")
    parser.add_argument("--date-folder", required=True, help="Date folder (e.g., 2026-04-29)")
    parser.add_argument("--out", required=True, help="Output JSON path")
    args = parser.parse_args()

    try:
        # Try recursive listing under date folder to capture nested files
        tree = list_repo_tree(repo_id=args.repo, path=args.date_folder, recursive=True)
    except Exception as e:
        print(f"ERROR listing repo tree at {args.date_folder}: {e}", file=sys.stderr)
        sys.exit(1)

    entries = tree if isinstance(tree, list) else tree.get("entries", [])
    # rfilename is relative to repo root; keep paths as stored
    files = [e.rfilename for e in entries]

    manifest = {
        "repo": args.repo,
        "date_folder": args.date_folder,
        "files": sorted(files)
    }

    os.makedirs(os.path.dirname(os.path.abspath(args.out)), exist_ok=True)
    with open(args.out, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2)
    print(f"Wrote {len(files)} files to {args.out}")

if __name__ == "__main__":
    main()
```
