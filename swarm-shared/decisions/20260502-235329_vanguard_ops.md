## Final Synthesized Implementation

### Diagnosis (merged, contradictions resolved)
- **Repeated HF API calls**: both frontend and backend re-list on every training launch; this burns quota and risks 429. Fix: generate one durable manifest per `dateFolder` and reuse it.
- **No CDN-only fetch guarantee**: training workers must use `resolve/main/...` URLs from the manifest, never call HF APIs during training.
- **Cloud/size selection**: sweep clouds and sizes in priority order (H200 first, then L40S; `lightning-lambda-prod` then `lightning-public-prod`) to maximize chance of success without quota waste.
- **Idle-guard + reuse**: detect and restart stopped studios; reuse running studios instead of creating duplicates.
- **Deterministic sharding for HF writes**: use a stable 5-shard selector to respect HF commit caps.
- **Backend/frontend consistency**: provide both TS and Python implementations so ingestion and training prep can run server-side without frontend involvement.

```ts
import { writeFileSync, mkdirSync, existsSync } from "fs";
import { resolve } from "path";

const HF_REPO = "datasets/axentx/surrogate-1";
const MANIFEST_DIR = resolve(process.cwd(), "manifests");

export interface FileEntry {
  path: string;
  type: "file" | "directory";
  size?: number;
  lfs?: { size?: number; pointer?: boolean };
}

export interface Manifest {
  dateFolder: string;
  repo: string;
  files: FileEntry[];
  generatedAt: string;
  cdnPrefix: string;
}

export async function listRepoTreeOnce(
  path: string,
  recursive: boolean = false
): Promise<Array<{ path: string; type: "file" | "directory"; size?: number }>> {
  // Lightweight adapter: replace with real HF client call in your runtime.
  // Must perform exactly ONE non-recursive tree request per dateFolder.
  const res = await fetch(
    `https://huggingface.co/api/datasets/${HF_REPO}/tree?path=${encodeURIComponent(
      path
    )}&recursive=${recursive ? "true" : "false"}`,
    { headers: { Accept: "application/json" } }
  );
  if (!res.ok) throw new Error(`HF tree failed: ${res.status}`);
  return res.json();
}

export async function generateManifest(dateFolder: string): Promise<Manifest> {
  if (!existsSync(MANIFEST_DIR)) mkdirSync(MANIFEST_DIR, { recursive: true });

  // Single non-recursive call per dateFolder (critical for quota)
  const tree = await listRepoTreeOnce(dateFolder, false);
  const files: FileEntry[] = tree
    .filter((n) => n.type === "file")
    .map((n) => ({ path: n.path, type: "file", size: n.size }));

  const manifest: Manifest = {
    dateFolder,
    repo: HF_REPO,
    files,
    generatedAt: new Date().toISOString(),
    cdnPrefix: `https://huggingface.co/datasets/${HF_REPO}/resolve/main`,
  };

  const outPath = resolve(MANIFEST_DIR, `${dateFolder}.json`);
  writeFileSync(outPath, JSON.stringify(manifest, null, 2));
  return manifest;
}

// Deterministic 5-shard selector for HF writes (respects commit caps)
export function pickShardRepo(slug: string): string {
  const hash = Array.from(slug).reduce(
    (a, c) => ((a << 5) - a + c.charCodeAt(0)) | 0,
    0
  );
  const idx = Math.abs(hash) % 5; // 0..4
  return `datasets/axentx/surrogate-1-shard-${idx}`;
}
```

### File: `/opt/axentx/vanguard/src/features/training/TrainingLauncher.tsx`
```tsx
import React, { useEffect, useState } from "react";
import { generateManifest, pickShardRepo } from "../../ops/hf-manifest";
import { Lightning, Studio, Machine } from "@lightning-ai/sdk";

interface Props {
  dateFolder: string;
  scriptPath: string;
}

export const TrainingLauncher: React.FC<Props> = ({ dateFolder, scriptPath }) => {
  const [status, setStatus] = useState("idle");
  const [manifestUrl, setManifestUrl] = useState<string | null>(null);

  useEffect(() => {
    // Generate once per dateFolder and cache locally
    generateManifest(dateFolder)
      .then(() => setManifestUrl(`/manifests/${dateFolder}.json`))
      .catch((e) => setStatus(`error: ${e.message}`));
  }, [dateFolder]);

  const launch = async () => {
    setStatus("preparing");

    // Reuse running studio if available
    const studios = await Lightning.teamspace?.studios?.() ?? [];
    let studio = studios.find(
      (s) => s.name === `vanguard-train-${dateFolder}` && s.status === "Running"
    );

    if (!studio) {
      // Priority sweep: H200 (lambda) -> L40S (lambda) -> fallback
      const clouds = ["lightning-lambda-prod", "lightning-public-prod"];
      const sizes = ["H200", "L40S"];
      let machine: Machine | null = null;

      for (const cloud of clouds) {
        for (const size of sizes) {
          try {
            machine = Machine.create({ cloud, size });
            break;
          } catch {
            continue;
          }
        }
        if (machine) break;
      }

      if (!machine) machine = Machine.L40S;

      studio = await Studio.create({
        name: `vanguard-train-${dateFolder}`,
        machine,
        scriptPath,
        env: {
          HF_MANIFEST_PATH: `/manifests/${dateFolder}.json`,
          HF_REPO_SHARD: pickShardRepo(dateFolder),
          // Training script must use only CDN URLs from manifest
        },
        createOk: true,
      });
    }

    // Idle-guard: restart if stopped before run
    if (studio.status === "Stopped") {
      await studio.start({ machine: studio.machine || Machine.L40S });
    }

    const run = await studio.run({
      arguments: [manifestUrl || ""],
    });

    setStatus(`running:${run.id}`);
  };

  return (
    <div>
      <button onClick={launch} disabled={!manifestUrl || status.startsWith("running")}>
        {status === "idle" ? "Launch Training" : status}
      </button>
      {manifestUrl && <pre>Manifest: {manifestUrl}</pre>}
    </div>
  );
};
```
