# vanguard / design

# Final Design (merged + resolved)

## 1. Diagnosis (merged)
- **No persisted manifest**: every training run performs authenticated `list_repo_tree` calls, burning HF API quota and risking 429s.
- **Data loader fragility**: training uses `load_dataset(...)` or per-file streaming across heterogeneous repos, causing `pyarrow` schema mismatches and slow/fragile ingestion.
- **No CDN-only path**: authenticated API calls during training are unnecessary for public datasets and expose the pipeline to rate limits.
- **HF write cap exposure**: unsharded writes to a single repo can hit the 128 commits/hr limit and block ingestion.
- **Lightning Studio churn**: jobs may recreate studios or fail on idle-stop instead of deterministically reusing running instances, wasting quota and time.

## 2. Proposed change (merged)
Add a small, self-contained design artifact and reference implementation under `/opt/axentx/vanguard/design/`:

- `hf_cdn_manifest.py` — CLI utility to produce `(repo, dateFolder) → file-list` JSON manifest using a single authenticated `list_repo_tree` call; training scripts embed this manifest and fetch via CDN URLs only.
- `lightning_studio_reuse.py` — helper to find, start, or create a Lightning Studio deterministically and run training in it, avoiding quota waste and idle-stop churn.
- `hf_write_shard.py` — deterministic repo selection for HF writes (hash-slug → sibling repo) to spread commits and avoid the 128/hr cap.
- `README.md` — concise design notes tying these together and the CDN-bypass strategy.

Scope: add new files only (no changes to existing code). Total implementation <2h.

### `/opt/axentx/vanguard/design/hf_cdn_manifest.py`

```python
#!/usr/bin/env python3
"""
Produce (repo, dateFolder) -> file-list manifest for CDN-only training.

Usage:
  python hf_cdn_manifest.py \
    --repo datasets/owner/repo \
    --date-folder 2026-05-03 \
    --out manifest.json \
    --token $HF_TOKEN

Training scripts should use the manifest to build CDN URLs:
  https://huggingface.co/datasets/{repo}/resolve/main/{path}
(no authenticated API calls during training).
"""

import argparse
import json
import os
import sys
from typing import Dict, List

from huggingface_hub import HfApi

CDN_BASE = "https://huggingface.co/datasets"

def build_manifest(repo: str, date_folder: str, token: str) -> Dict:
    api = HfApi(token=token)
    # Non-recursive top-level of date folder to minimize paginated calls
    tree = api.list_repo_tree(
        repo_id=repo,
        path=date_folder,
        recursive=False,
    )
    files = [
        f.rfilename
        for f in tree
        if f.rfilename.startswith(date_folder) and not f.rfilename.endswith("/")
    ]
    manifest = {
        "repo": repo,
        "date_folder": date_folder,
        "files": sorted(files),
        "cdn_base": CDN_BASE,
        "paths": [f"{CDN_BASE}/{repo}/resolve/main/{f}" for f in sorted(files)],
    }
    return manifest

def main() -> None:
    parser = argparse.ArgumentParser(description="Build CDN manifest for HF dataset folder.")
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g., datasets/owner/repo)")
    parser.add_argument("--date-folder", required=True, help="Folder in repo (e.g., 2026-05-03)")
    parser.add_argument("--out", default="manifest.json", help="Output JSON path")
    parser.add_argument("--token", default=os.getenv("HF_TOKEN"), help="HF token (env HF_TOKEN)")
    args = parser.parse_args()

    if not args.token:
        print("ERROR: HF token required (set HF_TOKEN or --token)", file=sys.stderr)
        sys.exit(1)

    manifest = build_manifest(args.repo, args.date_folder, args.token)
    with open(args.out, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2)
    print(f"Wrote {args.out} with {len(manifest['files'])} files")

if __name__ == "__main__":
    main()
```

```python
#!/usr/bin/env python3
"""
Reuse a running Lightning Studio when possible to save quota.

Usage:
  python lightning_studio_reuse.py \
    --name surrogate-1-train \
    --machine L40S \
    --script train.py \
    --dependencies requirements.txt \
    --run -- run-arg1 --lr 0.001

Behavior:
- If a studio with `name` is Running -> reuse it.
- If stopped -> start it with `machine`.
- If missing -> create it.
"""

import argparse
import json
import subprocess
import sys
from typing import List, Optional

try:
    from lightning_sdk import Studio, Teamspace
except ImportError:
    print("ERROR: lightning-sdk required (pip install lightning)", file=sys.stderr)
    sys.exit(1)

def find_running_studio(name: str) -> Optional[Studio]:
    for s in Teamspace.studios:
        if s.name == name and s.status == "Running":
            return s
    return None

def ensure_studio(name: str, machine: str, script: str, deps: str) -> Studio:
    existing = find_running_studio(name)
    if existing:
        print(f"Reusing running studio: {name}")
        return existing

    # Try to find stopped studio and restart
    for s in Teamspace.studios:
        if s.name == name:
            print(f"Starting stopped studio: {name}")
            s.start(machine=machine)
            return s

    print(f"Creating new studio: {name}")
    return Studio(
        name=name,
        script=script,
        dependencies=deps,
        machine=machine,
        create_ok=True,
    )

def run_in_studio(studio: Studio, script: str, args: List[str]) -> None:
    # Lightweight run; adapt to your training launcher
    cmd = [sys.executable, script] + args
    print(f"Running in studio {studio.name}: {' '.join(cmd)}")
    studio.run(command=" ".join(cmd))

def main() -> None:
    parser = argparse.ArgumentParser(description="Lightning Studio reuse helper.")
    parser.add_argument("--name", default="surrogate-1-train", help="Studio name")
    parser.add_argument("--machine", default="L40S", help="Machine type")
    parser.add_argument("--script", default="train.py", help="Training script")
    parser.add_argument("--dependencies", default="requirements.txt", help="Deps file")
    parser.add_argument("--run", action="store_true", help="Run training after ensure")
    parser.add_argument("run_args", nargs="*", help="Args to pass to training script")
    args = parser.parse_args()

    studio = ensure_studio(args.name, args.machine, args.script, args.dependencies)
    if args.run:
        run_in_studio(studio, args.script, args.run_args)

if __name__ == "__main__":
    main()
```
