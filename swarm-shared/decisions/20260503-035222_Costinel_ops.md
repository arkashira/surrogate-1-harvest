# Costinel / ops

## Final Implementation Plan  
**Goal**: Add a non-blocking, CDN-first Top-Hub Signal Panel to Costinel that surfaces the most-connected hub (e.g., "MOC") with **zero HuggingFace API calls at runtime**.

---

### Architecture (CDN-first, zero-runtime-HF)
- **Build-time** (Mac/CI): single `list_repo_tree` call to HF REST API → deterministic JSON baked into repo / Docker image.  
- **Runtime** (Costinel): static fetch from CDN (`/data/top-hub.json`) → React component renders signal chip/card.  
- **No HF auth / no rate-limits at runtime**.  
- **Non-blocking**: lazy-load + graceful fallback (render nothing if unavailable).

---

### 1) Build script — `scripts/build-top-hub.js`
```js
#!/usr/bin/env node
/**
 * Build-time script: fetch top-hub once and bake into public assets.
 * Uses HF CDN bypass: list_repo_tree once (respect rate limits), then CDN-only.
 * Output: public/data/top-hub.json
 *
 * Usage: node scripts/build-top-hub.js [owner/repo] [tree_path]
 * Example: node scripts/build-top-hub.js axentx/knowledge-rag graph
 */

import fs from 'fs';
import path from 'path';
import { fileURLToPath } from 'url';

const __filename = fileURLToPath(import.meta.url);
const __dirname = path.dirname(__filename);
const repoRoot = path.resolve(__dirname, '..', '..');

const REPO = process.argv[2] || 'axentx/knowledge-rag';
const TREE_PATH = process.argv[3] || 'graph';
const OUT_DIR = path.join(repoRoot, 'public', 'data');
const OUT_FILE = path.join(OUT_DIR, 'top-hub.json');

async function fetchJSON(url) {
  const res = await fetch(url, { headers: { Accept: 'application/json' } });
  if (!res.ok) throw new Error(`HTTP ${res.status} ${res.statusText} for ${url}`);
  return res.json();
}

function pickTopHub(tree) {
  // Prefer explicit MOC; otherwise largest JSON/Parquet file in tree (non-recursive)
  const candidates = Array.isArray(tree) ? tree : [];
  const moc = candidates.find((t) => t.type === 'file' && /moc/i.test(t.path));
  if (moc) return { hub: 'MOC', size: moc.size || 0, path: moc.path };

  const files = candidates.filter(
    (t) =>
      t.type === 'file' &&
      (t.path.endsWith('.json') || t.path.endsWith('.parquet'))
  );
  if (files.length === 0) return null;

  const largest = files.reduce((a, b) => ((b.size || 0) > (a.size || 0) ? b : a));
  const name = path.basename(largest.path).replace(/\.(json|parquet)$/i, '');
  return { hub: name || 'MOC', size: largest.size || 0, path: largest.path };
}

async function main() {
  try {
    const treeUrl = `https://huggingface.co/api/datasets/${REPO}/tree/${TREE_PATH}`;
    console.log(`Fetching tree: ${treeUrl}`);
    const tree = await fetchJSON(treeUrl);

    const top = pickTopHub(tree);
    const result = {
      hub: top?.hub || 'MOC',
      size: top?.size || 0,
      path: top?.path || `graph/moc.json`,
      generated_at: new Date().toISOString(),
      repo: REPO,
      tree_path: TREE_PATH,
      source: top ? 'cdn-tree' : 'fallback',
    };

    fs.mkdirSync(OUT_DIR, { recursive: true });
    fs.writeFileSync(OUT_FILE, JSON.stringify(result, null, 2) + '\n');
    console.log(`Wrote ${path.relative(repoRoot, OUT_FILE)}`);
    console.log(JSON.stringify(result));
  } catch (err) {
    // Graceful fallback: produce minimal valid file so runtime never fails
    const fallback = {
      hub: 'MOC',
      size: 0,
      path: 'graph/moc.json',
      generated_at: new Date().toISOString(),
      repo: REPO,
      tree_path: TREE_PATH,
      source: 'fallback',
    };
    fs.mkdirSync(OUT_DIR, { recursive: true });
    fs.writeFileSync(OUT_FILE, JSON.stringify(fallback, null, 2) + '\n');
    console.warn('Build failed, using fallback:', err.message);
    console.log(`Wrote fallback ${path.relative(repoRoot, OUT_FILE)}`);
  }
}

main();
```
