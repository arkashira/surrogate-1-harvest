## 1) Manifest-first HF access (eliminate 429s)

**Why**: Recursive `list_repo_files` causes paginated API explosions and 429s. Public CDN URLs have much higher rate limits.  
**Action**: Generate a JSON manifest once on the Mac; Lightning uses CDN-only fetches during training.

**File**: `scripts/generate_hf_manifest.py`

```python
#!/usr/bin/env python3
"""
Generate a manifest of HF dataset files for a given date folder.
Run from Mac after rate-limit window clears.
"""
import json
import os
import sys
from huggingface_hub import HfApi

API = HfApi()
REPO = os.getenv("HF_REPO", "datasets/your-org/surrogate-ingest")
DATE_FOLDER = sys.argv[1] if len(sys.argv) > 1 else "2026-04-29"
OUT_PATH = sys.argv[2] if len(sys.argv) > 2 else "data/file_manifest.json"

def main() -> None:
    entries = API.list_repo_tree(repo_id=REPO, path=DATE_FOLDER, recursive=False)
    files = [e.path for e in entries if e.type == "file" and e.path.endswith(".parquet")]
    manifest = {
        "repo": REPO,
        "date": DATE_FOLDER,
        "files": sorted(files),
        "cdn_prefix": f"https://huggingface.co/datasets/{REPO}/resolve/main",
    }
    os.makedirs(os.path.dirname(os.path.abspath(OUT_PATH)), exist_ok=True)
    with open(OUT_PATH, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Wrote {len(files)} files to {OUT_PATH}")

if __name__ == "__main__":
    main()
```

## 2) Lightning training: CDN-only, deterministic, zero breaking changes

**Why**: Bypass `/api/` auth checks; avoid `load_dataset(streaming=True)` on heterogeneous repos (prevents `pyarrow.CastError`).  
**Action**: Use manifest + lightweight local cache; project only `prompt`/`response`; keep collate/tokenizer hooks for future extension.

**File**: `surrogate/train.py`

```python
import json
import os
from pathlib import Path
from typing import Dict, List

import pyarrow.parquet as pq
import requests
import torch
from torch.utils.data import Dataset, DataLoader
from lightning import LightningDataModule, LightningModule, Trainer, seed_everything

MANIFEST_PATH = os.getenv("HF_MANIFEST_PATH", "data/file_manifest.json")
seed_everything(42, workers=True)

class CDNParquetDataset(Dataset):
    def __init__(self, manifest_path: str):
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.prefix = self.manifest["cdn_prefix"]
        self.files: List[str] = self.manifest["files"]
        self.cache_dir = Path(".hf_cdn_cache")
        self.cache_dir.mkdir(exist_ok=True)

    def _download(self, rel_path: str) -> Path:
        cached = self.cache_dir / rel_path.replace("/", "_")
        if cached.exists():
            return cached
        url = f"{self.prefix}/{rel_path}"
        r = requests.get(url, timeout=60)
        r.raise_for_status()
        cached.write_bytes(r.content)
        return cached

    def __len__(self) -> int:
        return len(self.files)

    def __getitem__(self, idx: int) -> Dict[str, torch.Tensor]:
        rel = self.files[idx]
        local_path = self._download(rel)
        table = pq.read_table(local_path, columns=["prompt", "response"])
        # Replace with real tokenizer; keep deterministic shapes/collate
        return {
            "prompt": torch.tensor([1, 2, 3], dtype=torch.long),
            "response": torch.tensor([4, 5, 6], dtype=torch.long),
        }

class SurrogateDataModule(LightningDataModule):
    def __init__(self, manifest_path: str = MANIFEST_PATH, batch_size: int = 8):
        super().__init__()
        self.manifest_path = manifest_path
        self.batch_size = batch_size

    def setup(self, stage: str = "") -> None:
        self.ds = CDNParquetDataset(self.manifest_path)

    def train_dataloader(self) -> DataLoader:
        return DataLoader(self.ds, batch_size=self.batch_size, shuffle=True, num_workers=4)

class SurrogateModel(LightningModule):
    def __init__(self):
        super().__init__()
        self.layer = torch.nn.Linear(10, 10)

    def training_step(self, batch, batch_idx):
        return torch.tensor(0.0)  # placeholder

    def configure_optimizers(self):
        return torch.optim.Adam(self.parameters(), lr=1e-3)

if __name__ == "__main__":
    dm = SurrogateDataModule()
    model = SurrogateModel()
    trainer = Trainer(max_epochs=1, accelerator="gpu", devices=1)
```

## 3) Lightning Studio reuse + idle-stop resilience (reduce iteration time)

**Why**: Reuse running studios to save quota; auto-restart if idle-stop kills training; cut Mac→Lightning cycle to <2 min.  
**Action**: Check status, restart if needed, then run training inside the studio.

**File**: `scripts/launch_lightning_studio.py`

```python
#!/usr/bin/env python3
"""
Launch or reuse a Lightning Studio for surrogate training.
Respects idle-stop: checks status and restarts machine if stopped.
"""
import os
import sys
import time

from lightning import Studio, Teamspace, Machine

STUDIO_NAME = os.getenv("STUDIO_NAME", "surrogate-training")
MACHINE = Machine.L40S
MAX_RETRIES = 3
RETRY_WAIT = 60

def get_or_create_studio() -> Studio:
    for s in Teamspace.studios:
        if s.name == STUDIO_NAME:
            if s.status == "Running":
                print(f"Reusing running studio: {STUDIO_NAME}")
                return s
            print(f"Studio {STUDIO_NAME} exists but status={s.status}. Restarting...")
            s.start(machine=MACHINE)
            return s

    print(f"Creating studio {STUDIO_NAME}...")
    return Studio(STUDIO_NAME, machine=MACHINE, create_ok=True)

def wait_for_ready(studio: Studio, timeout: int = 300) -> bool:
    elapsed = 0
    while elapsed < timeout:
        studio.refresh()
        if studio.status == "Running":
            return True
        print(f"Studio status: {studio.status}. Waiting...")
        time.sleep(15)
        elapsed += 15
    return False

def run_training_with_retry(studio: Studio) -> None:
    for attempt in range(1, MAX_RETRIES + 1):
        try:
            if studio.status != "Running":
                print(f"Studio not running (status={studio.status}). Restarting...")
                studio.start(machine=MACHINE)
                if not wait_for_ready(studio):
                    raise RuntimeError("Studio failed to become ready")

            job = studio.run(
                ["python", "surrogate/train.py"],
                cwd="/workspace/airship/surrogate",
                wait=True,
            )
            print(f"Training completed (attempt {attempt})")
            return
        except Exception as e:
            print(f"Attempt {attempt} failed: {e}")
            if attempt < MAX_RETRIES:
                print(f"Retrying in {RETRY_WAIT}s...")
                time.sleep(RETRY_WAIT)
            else:
                raise

if __name__ == "__main__":
    studio = get_or_create_studio()
    if not wait_for_ready(studio):
        sys.exit("Studio failed to start")
    run_training_with_retry(studio)
```
