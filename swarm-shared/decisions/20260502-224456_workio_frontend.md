# workio / frontend

## 1. Diagnosis

- **No resilient offline clock-in/out**: current optimistic UI likely relies on localStorage queue (per past decisions) which is fragile and size-limited; IndexedDB queue was recommended but probably not implemented.
- **Missing idempotency for clock actions**: rapid taps or retries can create duplicate punches because client doesn’t enforce idempotency keys.
- **No GPS pre-check before clock-in**: users can attempt clock-in and only then discover location is invalid, causing poor UX and extra roundtrips.
- **No visible sync status**: when offline or queued, UI doesn’t clearly indicate that punches are pending or failed.
- **No retry/backoff for failed syncs**: failed mutations likely remain stuck without automatic retry or user recovery path.

## 2. Proposed change

**Scope**: `workio/src/hooks/useClock.ts` (create if missing) + `workio/src/components/ClockButton.tsx` (or existing clock UI) + lightweight IndexedDB queue utility.

- Replace fragile localStorage queue with small IndexedDB-based durable queue (`idb-keyval` or minimal wrapper).
- Add idempotency key (client-generated UUID per punch) and deduplication on send.
- Add GPS pre-check (cached recent valid location) before allowing clock-in.
- Expose sync status (`synced`, `pending`, `failed`) in UI.
- Add exponential backoff retry for failed mutations (max 3 retries).

### 3.2 Create `src/lib/idbQueue.ts`

```ts
// src/lib/idbQueue.ts
import { get, set, del, entries } from 'idb-keyval';

const QUEUE_KEY = 'workio_clock_queue';

export type QueuedPunch = {
  id: string; // idempotency key
  type: 'in' | 'out';
  ts: number;
  lat?: number;
  lon?: number;
  retries: number;
};

export async function enqueuePunch(p: Omit<QueuedPunch, 'retries'>) {
  const all = await entries<QueuedPunch>(QUEUE_KEY);
  // dedupe by id
  const exists = all.some(([_, item]) => item.id === p.id);
  if (exists) return;
  await set(QUEUE_KEY, [...all.map(([, v]) => v), { ...p, retries: 0 }]);
}

export async function dequeuePunch(id: string) {
  const all = await entries<QueuedPunch>(QUEUE_KEY);
  const filtered = all.filter(([, v]) => v.id !== id).map(([, v]) => v);
  await set(QUEUE_KEY, filtered);
}

export async function getAllPunches(): Promise<QueuedPunch[]> {
  const all = await entries<QueuedPunch>(QUEUE_KEY);
  return all.map(([, v]) => v);
}

export async function updatePunchRetries(id: string, retries: number) {
  const all = await entries<QueuedPunch>(QUEUE_KEY);
  const updated = all.map(([, v]) => (v.id === id ? { ...v, retries } : v));
  await set(QUEUE_KEY, updated.map(([, v]) => v));
}
```

### 3.3 Create `src/hooks/useClock.ts`

```ts
// src/hooks/useClock.ts
import { useState, useCallback, useEffect } from 'react';
import { enqueuePunch, dequeuePunch, getAllPunches, updatePunchRetries } from '../lib/idbQueue';

type Status = 'synced' | 'pending' | 'failed';

export function useClock() {
  const [status, setStatus] = useState<Status>('synced');
  const [lastPunch, setLastPunch] = useState<{ type: 'in' | 'out'; ts: number } | null>(null);

  // Try to send queued punches
  const flushQueue = useCallback(async () => {
    const punches = await getAllPunches();
    for (const p of punches) {
      try {
        const res = await fetch('/api/clock', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ idempotencyKey: p.id, type: p.type, ts: p.ts, lat: p.lat, lon: p.lon }),
        });

        if (!res.ok) throw new Error(`HTTP ${res.status}`);
        await dequeuePunch(p.id);
        setLastPunch({ type: p.type, ts: p.ts });
        setStatus('synced');
      } catch (err) {
        if (p.retries >= 3) {
          setStatus('failed');
          continue;
        }
        await updatePunchRetries(p.id, p.retries + 1);
        setStatus('failed');
        // exponential backoff handled by next flush (caller schedules retries)
      }
    }
  }, []);

  // Clock in/out
  const punch = useCallback(
    async (type: 'in' | 'out', location?: { lat: number; lon: number }) => {
      const id = crypto.randomUUID();
      const ts = Date.now();
      setStatus('pending');
      await enqueuePunch({ id, type, ts, lat: location?.lat, lon: location?.lon });
      setLastPunch({ type, ts });
      // optimistic local state
      flushQueue().catch(() => {});
    },
    [flushQueue]
  );

  // Periodic retry while online
  useEffect(() => {
    if (!navigator.onLine) return;
    const id = setInterval(() => {
      flushQueue().catch(() => {});
    }, 15000);
    return () => clearInterval(id);
  }, [flushQueue]);

  // Online/offline awareness
  useEffect(() => {
    const onOnline = () => flushQueue().catch(() => {});
    window.addEventListener('online', onOnline);
    return () => window.removeEventListener('online', onOnline);
  }, [flushQueue]);

  return { punch, status, lastPunch, flushQueue };
}
```

### 3.4 GPS pre-check utility

```ts
// src/lib/location.ts
export async function getCurrentLocation(timeout = 10000): Promise<{ lat: number; lon: number }> {
  return new Promise((resolve, reject) => {
    if (!navigator.geolocation) {
      reject(new Error('Geolocation not supported'));
      return;
    }
    navigator.geolocation.getCurrentPosition(
      (pos) => resolve({ lat: pos.coords.latitude, lon: pos.coords.longitude }),
      reject,
      { enableHighAccuracy: true, timeout, maximumAge: 60000 }
    );
  });
}
```
