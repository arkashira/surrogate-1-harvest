# workio / ops

## 1. Diagnosis

- **No idempotency key storage** — LINE webhook `X-Line-Signature` is verified but not persisted; retries with same signature still produce duplicate punch rows.
- **Non-atomic upsert path** — current flow does `findOne → if exists update else insert`; concurrent/retried webhooks can interleave and create duplicate or conflicting attendance rows.
- **Missing server-side clock-in/out guard** — no DB constraint or unique index to prevent multiple open `clock_in` rows per user per day; allows impossible state (multiple active punches).
- **No retry-safe webhook handler** — 2xx is returned before transaction commit; if process crashes after response but before commit, LINE may retry and create duplicates.
- **No operational observability** — missing structured request/response logs and metrics for webhook delivery, duplicates, and DB constraint violations.

## 2. Proposed change

- **File scope**: `workio/server/src/routes/webhook/line.ts` (or equivalent) and `workio/server/src/db/schema.sql`
- **Goal**: add idempotency table + unique constraint + atomic upsert in a single transaction so retries and concurrency cannot corrupt attendance state.

## 3. Implementation

### 3.1 DB schema change (add idempotency key + enforce one open punch)

```sql
-- workio/server/src/db/schema.sql
-- Add idempotency table for LINE webhooks
CREATE TABLE IF NOT EXISTS line_webhook_idempotency (
  idempotency_key VARCHAR(255) PRIMARY KEY,
  payload_hash VARCHAR(64) NOT NULL,
  processed_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Ensure one active punch per user (null clock_out means active)
-- If you already have a unique partial index, keep it; otherwise add:
CREATE UNIQUE INDEX IF NOT EXISTS idx_attendance_one_open_punch
ON attendance (user_id, DATE(clock_in))
WHERE clock_out IS NULL;
```

### 3.2 Webhook handler with atomic upsert + idempotency

```ts
// workio/server/src/routes/webhook/line.ts
import { Router } from 'express';
import { pool } from '../../db';
import crypto from 'crypto';

const router = Router();

function verifySignature(body: string, signature: string, channelSecret: string): boolean {
  const expected = crypto
    .createHmac('sha256', channelSecret)
    .update(body)
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

router.post('/line', async (req, res) => {
  const signature = req.headers['x-line-signature'] as string;
  const channelSecret = process.env.LINE_CHANNEL_SECRET!;
  const rawBody = JSON.stringify(req.body);
  const payloadHash = crypto.createHash('sha256').update(rawBody).digest('hex');
  const idempotencyKey = signature; // LINE signature is stable for same payload+secret

  if (!verifySignature(rawBody, signature, channelSecret)) {
    return res.status(401).json({ error: 'Invalid signature' });
  }

  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // Idempotency check
    const idem = await client.query(
      `SELECT 1 FROM line_webhook_idempotency WHERE idempotency_key = $1 AND payload_hash = $2`,
      [idempotencyKey, payloadHash]
    );
    if (idem.rows.length > 0) {
      await client.query('ROLLBACK');
      return res.status(200).json({ ok: true, reason: 'already_processed' });
    }

    // Parse LINE event (support single event for simplicity)
    const event = req.body.events?.[0];
    if (!event || event.type !== 'message' || event.message.type !== 'text') {
      await client.query('ROLLBACK');
      return res.status(400).json({ error: 'Unsupported event' });
    }

    const userId = event.source.userId; // LINE user ID -> map to tenant user internally
    const text = event.message.text.trim().toLowerCase();
    const isClockIn = text.includes('in') || text.includes('clock') || text.includes('เข้า');
    const isClockOut = text.includes('out') || text.includes('leave') || text.includes('ออก');

    if (!isClockIn && !isClockOut) {
      await client.query('ROLLBACK');
      return res.status(400).json({ error: 'Unrecognized command' });
    }

    // Resolve internal user (example: map LINE userId to workio user)
    const userResult = await client.query(
      `SELECT id, tenant_id FROM users WHERE line_user_id = $1 LIMIT 1`,
      [userId]
    );
    if (userResult.rows.length === 0) {
      await client.query('ROLLBACK');
      return res.status(404).json({ error: 'User not found' });
    }
    const user = userResult.rows[0];

    const now = new Date();

    if (isClockIn) {
      // Atomic upsert: insert new punch if no open punch exists for today
      const upsert = await client.query(
        `INSERT INTO attendance (user_id, tenant_id, clock_in, clock_out, created_at, updated_at)
         SELECT $1, $2, $3, NULL, $3, $3
         WHERE NOT EXISTS (
           SELECT 1 FROM attendance
           WHERE user_id = $1 AND DATE(clock_in) = DATE($3) AND clock_out IS NULL
         )
         RETURNING id`,
        [user.id, user.tenant_id, now]
      );

      if (upsert.rows.length === 0) {
        // Already clocked in — optionally update last activity or ignore
        await client.query('ROLLBACK');
        return res.status(200).json({ ok: true, reason: 'already_clocked_in' });
      }
    } else {
      // Clock out: update the open punch for the day
      const result = await client.query(
        `UPDATE attendance
         SET clock_out = $1, updated_at = $1
         WHERE user_id = $2 AND DATE(clock_in) = DATE($1) AND clock_out IS NULL
         RETURNING id`,
        [now, user.id]
      );

      if (result.rows.length === 0) {
        await client.query('ROLLBACK');
        return res.status(400).json({ error: 'No active punch to clock out' });
      }
    }

    // Record idempotency after successful processing
    await client.query(
      `INSERT INTO line_webhook_idempotency (idempotency_key, payload_hash)
       VALUES ($1, $2)`,
      [idempotencyKey, payloadHash]
    );

    await client.query('COMMIT');
    return res.status(200).json({ ok: true });
  } catch (err) {
    await client.query('ROLLBACK');
    console.error('LINE webhook error', err);
    return res.status(500).json({ error: 'Internal server error' });
  } finally {
    client.release();
  }
});

export default router;
```
