```tsx
// workio/src/components/PunchClockButton.tsx
import React, { useState, useCallback, useRef } from 'react';
import { v4 as uuidv4 } from 'uuid';

type PunchType = 'IN' | 'OUT';

interface PunchClockButtonProps {
  /** Current expected punch direction based on last known server state */
  currentPunchType: PunchType;
  /** Optional success callback with canonical server record */
  onPunchSuccess?: (punch: { punchId: string; type: PunchType; timestamp: string }) => void;
  /** Optional className for layout */
  className?: string;
}

const API_ENDPOINT = '/api/punch';

export function PunchClockButton({
  currentPunchType,
  onPunchSuccess,
  className = '',
}: PunchClockButtonProps) {
  const [status, setStatus] = useState<'idle' | 'pending' | 'success' | 'error'>('idle');
  const [lastPunch, setLastPunch] = useState<{
    type: PunchType;
    id: string;
    ts: string;
  } | null>(null);

  // Stable idempotency key per user intent; ref avoids stale closures in abort/retry logic.
  const idempotencyKeyRef = useRef<string | null>(null);
  const abortControllerRef = useRef<AbortController | null>(null);

  const punch = useCallback(async (type: PunchType) => {
    // Prevent concurrent submissions for the same intent.
    if (status === 'pending') return;

    const key = uuidv4();
    idempotencyKeyRef.current = key;
    const controller = new AbortController();
    abortControllerRef.current = controller;

    const optimistic = {
      type,
      id: `optimistic-${key}`,
      ts: new Date().toISOString(),
    };

    setStatus('pending');
    setLastPunch(optimistic);

    try {
      const res = await fetch(API_ENDPOINT, {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'Idempotency-Key': key,
        },
        body: JSON.stringify({
          type,
          idempotencyKey: key,
          timestamp: optimistic.ts,
        }),
        signal: controller.signal,
      });

      if (!res.ok) {
        // Distinguish idempotent replay (e.g., 409) from server errors.
        if (res.status === 409) {
          // Accept the existing punch as authoritative.
          const data = await res.json().catch(() => ({}));
          const accepted = {
            type,
            id: data.punchId || `duplicate-${key}`,
            ts: data.timestamp || new Date().toISOString(),
          };
          setLastPunch(accepted);
          setStatus('success');
          onPunchSuccess?.({ punchId: accepted.id, type: accepted.type, timestamp: accepted.ts });
          return;
        }
        throw new Error(`HTTP ${res.status}`);
      }

      const data = await res.json();
      const accepted = {
        type,
        id: data.punchId,
        ts: data.timestamp,
      };
      setLastPunch(accepted);
      setStatus('success');
      onPunchSuccess?.({ punchId: accepted.id, type: accepted.type, timestamp: accepted.ts });
    } catch (err) {
      if ((err as Error).name === 'AbortError') {
        // Abort is intentional; keep current state.
        return;
      }

      // Rollback optimistic state and surface error.
      setLastPunch(null);
      setStatus('error');
      // Replace console.error with your toast/notification system if preferred.
      console.error('Punch failed', err);
    } finally {
      // Cleanup and reset after cooldown to allow next action.
      if (abortControllerRef.current === controller) {
        abortControllerRef.current = null;
      }
      if (idempotencyKeyRef.current === key) {
        idempotencyKeyRef.current = null;
      }

      // Brief success visibility then return to idle.
      setTimeout(() => {
        setStatus((prev) => (prev === 'success' ? 'idle' : prev));
      }, 1800);
    }
  }, [status, onPunchSuccess]);

  const isPending = status === 'pending';
  const label = isPending
    ? `${currentPunchType === 'IN' ? 'Clocking in...' : 'Clocking out...'}`
    : `${currentPunchType === 'IN' ? 'Clock In' : 'Clock Out'}`;

  return (
    <div className={`flex flex-col items-center gap-3 ${className}`}>
      <button
        onClick={() => punch(currentPunchType)}
        disabled={isPending}
        className={`
          px-8 py-3 rounded-lg font-semibold text-white transition-all
          ${isPending
            ? 'bg-gray-400 cursor-not-allowed'
            : currentPunchType === 'IN'
            ? 'bg-green-600 hover:bg-green-700'
            : 'bg-red-600 hover:bg-red-700'
          }
          ${status === 'success' ? 'ring-2 ring-green-500 ring-opacity-50' : ''}
          ${status === 'error' ? 'ring-2 ring-red-500 ring-opacity-50' : ''}
        `}
      >
        {label}
      </button>

      {status === 'error' && (
        <p className="text-sm text-red-600">Failed to punch. Please retry.</p>
      )}

      {lastPunch && !isPending && status !== 'error' && (
        <p className="text-sm text-gray-500">
          Last: {lastPunch.type} at {new Date(lastPunch.ts).toLocaleTimeString()}
        </p>
      )}
    </div>
  );
}
```
