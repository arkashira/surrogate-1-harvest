### 1) Manifest generator (Mac orchestrator)

File: `surrogate/scripts/generate_manifest.py`

```python
#!/usr/bin/env python3
"""
Generate file_manifest.json for a HuggingFace dataset repo
using list_repo_tree (non-recursive per folder) to avoid 429.
Includes only files we actually need (.parquet, .jsonl) with
SHA256 and size for integrity/validation.
"""
import os
import json
import hashlib
import argparse
from huggingface_hub import HfApi

HF_REPO = os.getenv("HF_DATASET_REPO", "myorg/surrogate-dataset")
FOLDER  = os.getenv("HF_DATASET_FOLDER", "batches/mirror-merged")
OUTPUT  = os.getenv("MANIFEST_OUT", "surrogate/training/file_manifest.json")

def _sha256_of_url(url: str) -> str | None:
    try:
        import requests
        resp = requests.get(url, timeout=30)
        resp.raise_for_status()
        return hashlib.sha256(resp.content).hexdigest()
    except Exception:
        return None

def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", default=HF_REPO)
    parser.add_argument("--folder", default=FOLDER)
    parser.add_argument("--out", default=OUTPUT)
    parser.add_argument("--include-sha", action="store_true", default=True)
    args = parser.parse_args()

    api = HfApi()
    entries = api.list_repo_tree(repo_id=args.repo, path=args.folder, recursive=False)

    files = []
    for e in entries:
        if not e.path.endswith((".parquet", ".jsonl")):
            continue
        cdn_url = f"https://huggingface.co/datasets/{args.repo}/resolve/main/{e.path}"
        item = {
            "path": e.path,
            "cdn_url": cdn_url,
            "size": getattr(e, "size", None),
        }
        if args.include_sha:
            item["sha256"] = _sha256_of_url(cdn_url)
        files.append(item)

    os.makedirs(os.path.dirname(os.path.abspath(args.out)), exist_ok=True)
    manifest = {"repo": args.repo, "folder": args.folder, "files": files}
    with open(args.out, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2, ensure_ascii=False)

    print(f"Wrote {len(files)} files to {args.out}")

if __name__ == "__main__":
    main()
```

Make executable:

```bash

```

Usage (Mac):

```bash

HF_DATASET_REPO=myorg/surrogate-dataset \
```

### 2) CDN-only iterable dataset loader

File: `surrogate/training/cdn_dataset.py`

```python
import json
import pyarrow.parquet as pq
import pyarrow as pa
import aiohttp
import asyncio
import os
from typing import Iterator, Dict, Any, AsyncIterator

MANIFEST_PATH = os.getenv(
    "MANIFEST_PATH",
    os.path.join(os.path.dirname(__file__), "file_manifest.json")
)

class CDNIterableDataset:
    """
    Lightweight iterable dataset that streams parquet/jsonl via CDN URLs.
    No HF API calls during training. Projects only {prompt, response}.
    """

    def __init__(self, manifest_path: str = MANIFEST_PATH):
        with open(manifest_path, encoding="utf-8") as f:
            self.manifest = json.load(f)
        self.files = [f["cdn_url"] for f in self.manifest["files"]]

    def _stream_parquet(self, url: str) -> Iterator[Dict[str, Any]]:
        # pyarrow supports HTTP(S) for single files; suitable for small/medium parquet.
        table = pq.read_table(url, columns=["prompt", "response"])
        for row in table.to_pylist():
            yield {
                "prompt": row.get("prompt") or "",
                "response": row.get("response") or ""
            }

    async def _fetch_jsonl(self, session: aiohttp.ClientSession, url: str) -> list:
        async with session.get(url, timeout=aiohttp.ClientTimeout(total=60)) as resp:
            resp.raise_for_status()
            text = await resp.text()
            out = []
            for line in text.strip().split("\n"):
                if not line.strip():
                    continue
                obj = json.loads(line)
                out.append({
                    "prompt": obj.get("prompt") or "",
                    "response": obj.get("response") or ""
                })
            return out

    async def _stream_jsonl_async(self, urls: list) -> AsyncIterator[Dict[str, Any]]:
        async with aiohttp.ClientSession() as session:
            for url in urls:
                items = await self._fetch_jsonl(session, url)
                for item in items:
                    yield item

    def __iter__(self) -> Iterator[Dict[str, Any]]:
        parquet_urls = [u for u in self.files if u.endswith(".parquet")]
        jsonl_urls = [u for u in self.files if u.endswith(".jsonl")]

        for url in parquet_urls:
            yield from self._stream_parquet(url)

        if jsonl_urls:
            # Simple sync fallback for jsonl; prefer async in prod pipelines.
            import requests
            for url in jsonl_urls:
                resp = requests.get(url, timeout=60)
                resp.raise_for_status()
                for line in resp.text.strip().split("\n"):
                    if not line.strip():
                        continue
                    obj = json.loads(line)
                    yield {
                        "prompt": obj.get("prompt") or "",
                        "response": obj.get("response") or ""
                    }

    async def aiter(self) -> AsyncIterator[Dict[str, Any]]:
        parquet_urls = [u for u in self.files if u.endswith(".parquet")]
        jsonl_urls = [u for u in self.files if u.endswith(".jsonl")]

        for url in parquet_urls:
            # Keep sync for parquet to avoid extra complexity; can be async if needed.
            for row in self._stream_parquet(url):
                yield row

        async for row in self._stream_jsonl_async(jsonl_urls):
            yield row
```

```python
import torch
from torch.utils.data import IterableDataset
from cdn_dataset import CDNIterableDataset

class TorchIterable(IterableDataset):
    def __init__(self):
        self.cdn = CDNIterableDataset()

    def __iter__(self):
        for item in self.cdn:
            yield item
```
