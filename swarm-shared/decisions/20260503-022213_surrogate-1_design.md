# surrogate-1 / design

## Final Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16`, `DATE`, `HF_TOKEN`, `REPO_ID` (default: `axentx/surrogate-1-training-pairs`)
- Single API call from runner to list one date-folder via `list_repo_tree(recursive=False)` → save `file-list.json`
- Deterministic shard assignment: `hash(slug) % SHARD_TOTAL == SHARD_ID`
- Downloads only assigned files via **HF CDN bypass** (`resolve/main/...`) with zero Authorization header during data fetch (uses token only for repo list + final upload)
- Projects heterogeneous schemas to `{prompt, response}` at parse time (avoids pyarrow CastError)
- Dedup via central `lib/dedup.py` md5 store (unchanged)
- Outputs `batches/public-merged/{DATE}/shard{N}-{HHMMSS}.jsonl`
- Returns exit 0 only if all assigned files processed; logs skipped/retry counts

### Why this ships in <2h
- Reuses existing patterns: HF rate-limit bypass, schema projection, dedup store, shard-by-hash
- One new Python file + tiny update to workflow to pass `DATE`
- No infra changes; keeps GitHub Actions 16-shard matrix

#!/usr/bin/env python3
"""
CDN-bypass ingestion worker for surrogate-1 public-dataset shards.

Environment:
  SHARD_ID        (required) 0..15
  SHARD_TOTAL     (default: 16)
  DATE            (required) YYYY-MM-DD
  HF_TOKEN        (required)
  REPO_ID         (default: axentx/surrogate-1-training-pairs)
  API_BASE        (optional) override for tests
"""

import os
import sys
import json
import hashlib
import time
import logging
from datetime import datetime
from pathlib import Path
from typing import List, Dict, Any

import requests
from huggingface_hub import HfApi, hf_hub_download

logging.basicConfig(level=logging.INFO, format="%(asctime)s %(levelname)s %(message)s")
log = logging.getLogger("dataset-enrich")

# ---- config ----
SHARD_ID = int(os.environ.get("SHARD_ID", -1))
SHARD_TOTAL = int(os.environ.get("SHARD_TOTAL", 16))
DATE = os.environ.get("DATE", "")
HF_TOKEN = os.environ.get("HF_TOKEN", "")
REPO_ID = os.environ.get("REPO_ID", "axentx/surrogate-1-training-pairs")
API_BASE = os.environ.get("API_BASE", "https://huggingface.co")

if SHARD_ID < 0 or SHARD_ID >= SHARD_TOTAL:
    log.error("Invalid SHARD_ID or SHARD_TOTAL")
    sys.exit(1)
if not DATE:
    log.error("DATE is required (YYYY-MM-DD)")
    sys.exit(1)

HEADERS = {"Authorization": f"Bearer {HF_TOKEN}"} if HF_TOKEN else {}
API = HfApi(token=HF_TOKEN)

# ---- dedup (central store) ----
sys.path.insert(0, str(Path(__file__).parent.parent))
from lib.dedup import DedupStore  # noqa: E402

dedup = DedupStore()

# ---- helpers ----
def list_date_files(date_folder: str) -> List[str]:
    """Single API call: list files in date folder (non-recursive)."""
    try:
        tree = API.list_repo_tree(repo_id=REPO_ID, path=date_folder, recursive=False)
        files = [item.path for item in tree if getattr(item, "type", None) == "file"]
        log.info("Listed %d files in %s", len(files), date_folder)
        return files
    except Exception as e:
        log.exception("Failed to list repo tree")
        raise RuntimeError(f"list_repo_tree failed: {e}")

def slug_from_path(path: str) -> str:
    """Stable slug for sharding (repo-relative path without extension)."""
    p = Path(path)
    stem = p.stem or p.name
    return f"{REPO_ID}/{p.parent}/{stem}".rstrip("/")

def shard_for(slug: str) -> int:
    """Deterministic shard assignment."""
    h = int(hashlib.sha256(slug.encode("utf-8")).hexdigest(), 16)
    return h % SHARD_TOTAL

def cdn_download_url(repo_id: str, path: str) -> str:
    """Public CDN URL (no auth header required for fetch)."""
    return f"{API_BASE}/datasets/{repo_id}/resolve/main/{path}"

def parse_file_to_pairs(path: str, content: bytes) -> List[Dict[str, str]]:
    """
    Project heterogeneous file schemas to {prompt, response}.
    Supports:
      - JSONL lines with prompt/response fields (case-insensitive)
      - JSON objects with prompt/response
      - Parquet bytes (via pyarrow) projected to prompt/response columns
    """
    import pyarrow.parquet as pq
    import io
    import json as jsonlib

    pairs = []

    def normalize_fields(obj: Dict[str, Any]) -> Dict[str, str]:
        prompt_key = None
        response_key = None
        for k in obj.keys():
            if k.lower() in ("prompt", "instruction", "question", "input"):
                prompt_key = k
            if k.lower() in ("response", "completion", "answer", "output"):
                response_key = k
        prompt = str(obj.get(prompt_key or "prompt" or "")).strip()
        response = str(obj.get(response_key or "response" or "")).strip()
        return {"prompt": prompt, "response": response}

    # Try JSONL first
    text = None
    try:
        text = content.decode("utf-8")
    except UnicodeDecodeError:
        pass

    if text is not None:
        for line in text.strip().splitlines():
            line = line.strip()
            if not line:
                continue
            try:
                obj = jsonlib.loads(line)
                pairs.append(normalize_fields(obj))
            except jsonlib.JSONDecodeError:
                continue
        if pairs:
            return pairs

    # Try JSON
    if text is not None:
        try:
            obj = jsonlib.loads(text)
            if isinstance(obj, list):
                for item in obj:
                    if isinstance(item, dict):
                        pairs.append(normalize_fields(item))
            elif isinstance(obj, dict):
                pairs.append(normalize_fields(obj))
            if pairs:
                return pairs
        except jsonlib.JSONDecodeError:
            pass

    # Try Parquet
    try:
        table = pq.read_table(io.BytesIO(content))
        df = table.to_pandas()
        prompt_col = None
        response_col = None
        for c in df.columns:
            if c.lower() in ("prompt", "instruction", "question", "input"):
                prompt_col = c
            if c.lower() in ("response", "completion", "answer", "output"):
                response_col = c
        if prompt_col is None or response_col is None:
            str_cols = [c for c in df.columns if df[c].dtype == "object"]
            if len(str_cols) >= 2:
                prompt_col, response_col = str_cols[0], str_cols[1]
            else:
                return []
        for _, row in df.iterrows():
            pairs.append(
                {
                    "prompt": str(row.get(prompt_col, "")).strip(),
                    "response": str(row.get(response_col, "")).strip(),
                }
            )
        return pairs
    except Exception:
        pass

    return []

def process_file(path: str) -> int:
    """Download via CDN, parse, dedup, return count of new pairs."""
    url = cdn_download_url(REPO_ID, path)
    try:
        resp = requests.get(url, timeout=60)
        resp.raise_for_status()
    except Exception as e:
        log.warning("CDN download failed for %s: %s", path, e)
        return 0

    content = resp.content
    pairs = parse_file_to_pairs(path, content)
    if not pairs:
        log.warning("No pairs extracted from %s", path)
