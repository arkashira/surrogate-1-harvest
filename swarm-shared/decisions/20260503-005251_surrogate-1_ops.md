# surrogate-1 / ops

**Final consolidated implementation (≤2h)**

**Highest-value change**  
Add a Mac-side `tools/snapshot_manifest.py` that lists one date-partition with **one** HF API call, emits `file_manifest.json` with CDN URLs, and a training loader that uses **CDN-only** fetches (zero HF API calls during training). This eliminates 429 rate-limit risk and quota burn while keeping Mac as orchestrator and Lightning for compute.

## Steps (1h 30m total)

1. **Create tools/snapshot_manifest.py** (20m)  
   - Accept `repo`, `date_partition` (e.g. `2026-04-29`), optional `out`.  
   - Call `list_repo_tree(path=date_partition, recursive=True)` once (recursive=True to capture nested files).  
   - Emit `file_manifest.json`:  
     ```json
     {
       "repo": "...",
       "date": "...",
       "files": [
         {"path": "...", "cdn_url": "...", "size": ..., "sha": "..."}
       ]
     }
     ```
   - Include `sha`/`size` when available.

2. **Create training loader module `surrogate_1/data/cdn_loader.py`** (30m)  
   - Accept `file_manifest.json`.  
   - Use `urllib.request` or `requests` with streaming to fetch each parquet file from CDN URL.  
   - Read via `pyarrow.parquet` in-memory; project only `{prompt, response}`.  
   - Yield dicts; optionally dedup via local md5 cache.  
   - No `datasets.load_dataset` and no HF API calls inside training loop.

3. **Update training script `train.py`** (20m)  
   - Add CLI arg `--manifest` pointing to `file_manifest.json`.  
   - Replace dataset loading with `cdn_loader.iter_samples(manifest)`.

### tools/snapshot_manifest.py
```python
#!/usr/bin/env python3
"""
Create a CDN-only file manifest for a date partition in a HuggingFace dataset repo.
Usage:
    python snapshot_manifest.py --repo axentx/surrogate-1-training-pairs \
        --date 2026-04-29 --out file_manifest.json
"""
import argparse
import json
import os
import sys
from typing import List, Dict

from huggingface_hub import HfApi

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def build_manifest(repo: str, date_partition: str, out_path: str) -> None:
    api = HfApi(token=os.getenv("HF_TOKEN"))  # optional for public repos, required for private
    # Single API call: list tree for this partition (recursive=True to capture nested files)
    tree = api.list_repo_tree(repo=repo, path=date_partition, recursive=True)

    files: List[Dict] = []
    for entry in tree:
        if entry.type != "file":
            continue
        # Keep parquet/jsonl; skip others
        if not entry.path.lower().endswith((".parquet", ".jsonl")):
            continue
        cdn_url = CDN_TEMPLATE.format(repo=repo, path=entry.path)
        files.append({
            "path": entry.path,
            "cdn_url": cdn_url,
            "size": getattr(entry, "size", None),
            "sha": getattr(entry, "sha", None),
        })

    manifest = {
        "repo": repo,
        "date": date_partition,
        "generated_by": "snapshot_manifest.py",
        "files": files,
    }

    with open(out_path, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2)

    print(f"Wrote {len(files)} files to {out_path}")

def main() -> None:
    parser = argparse.ArgumentParser(description="Create CDN manifest for a date partition.")
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g. axentx/surrogate-1-training-pairs)")
    parser.add_argument("--date", required=True, help="Date partition folder (e.g. 2026-04-29)")
    parser.add_argument("--out", default="file_manifest.json", help="Output JSON path")
    args = parser.parse_args()

    try:
        build_manifest(args.repo, args.date, args.out)
    except Exception as e:
        print(f"Error: {e}", file=sys.stderr)
        sys.exit(1)

if __name__ == "__main__":
    main()
```

### surrogate_1/data/cdn_loader.py
```python
import json
import pyarrow.parquet as pq
import pyarrow as pa
import urllib.request
from typing import Dict, Iterator, Optional
import io

def iter_parquet_from_cdn(cdn_url: str, columns: Optional[list] = None) -> Iterator[Dict]:
    """Stream a parquet file from CDN and yield rows as dicts (projected to columns)."""
    req = urllib.request.Request(cdn_url, headers={"User-Agent": "surrogate-1-train"})
    with urllib.request.urlopen(req) as resp:
        data = resp.read()
    table = pq.read_table(io.BytesIO(data), columns=columns)
    cols = table.column_names
    for i in range(table.num_rows):
        row = {col: table[col][i].as_py() for col in cols}
        yield row

def iter_samples(manifest_path: str, prompt_col: str = "prompt", response_col: str = "response") -> Iterator[Dict]:
    """
    Iterate training samples from a manifest.
    Projects each file to {prompt_col, response_col} at parse time.
    """
    with open(manifest_path, "r", encoding="utf-8") as f:
        manifest = json.load(f)

    for entry in manifest.get("files", []):
        cdn_url = entry["cdn_url"]
        try:
            for row in iter_parquet_from_cdn(cdn_url, columns=[prompt_col, response_col]):
                prompt = row.get(prompt_col)
                response = row.get(response_col)
                if prompt is None or response is None:
                    continue
                yield {
                    "prompt": prompt,
                    "response": response,
                    "_source_file": entry["path"],
                }
        except Exception as exc:
            # Log and continue; don't crash training on one bad file
            print(f"Skipping {cdn_url}: {exc}")
            continue
```
