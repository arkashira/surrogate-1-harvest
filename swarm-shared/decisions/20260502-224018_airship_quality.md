# airship / quality

## Final Implementation — Airship Quality: Cron-Safe Discover + CDN-Only Manifest + Static Status

### Core deliverable (highest-value, <2h)
A single **`airship discover` orchestrator** that:
1. Runs market research + knowledge-RAG and emits tagged insights (JSON).
2. Produces a **CDN-only HF training manifest** (one API call) so training loads data with zero API calls and bypasses 429s.
3. Exposes a **cron-safe, static-first status dashboard** showing Arkship (3000/8000) and Surrogate (8001) health with pass/fail indicators and latency.

This directly applies the surrogate-1 pipeline and HF CDN bypass patterns and gives operators immediate, actionable visibility.

## 1. Orchestrator: `/opt/axentx/airship/bin/airship`

#!/usr/bin/env bash
# airship — orchestrator for discovery, manifest generation, and status
# Cron-safe: invoke via `bash /opt/axentx/airship/bin/airship discover`
set -euo pipefail
SHELL=/bin/bash

ROOT="$(cd "$(dirname "${BASH_SOURCE[0]}")/.." && pwd)"
MANIFEST_DIR="${ROOT}/manifests"
STATUS_DIR="${ROOT}/status"
INSIGHTS_DIR="${ROOT}/insights"
PUBLIC_DIR="${ROOT}/public"

mkdir -p "${MANIFEST_DIR}" "${STATUS_DIR}" "${INSIGHTS_DIR}" "${PUBLIC_DIR}"

log() { echo "[$(date -Iseconds)] $*" >&2; }

# ---- Helpers ----
check_http() {
  local url="$1"
  if curl -fsS --max-time 5 "$url" >/dev/null 2>&1; then
    return 0
  else
    return 1
  fi
}

latency_ms() {
  local url="$1"
  # Use curl time_total in seconds, convert to ms integer
  local ms
  ms=$(curl -fsS --max-time 5 -o /dev/null -w '%{time_total}' "$url" 2>/dev/null || echo "0")
  printf "%.0f" "$(echo "$ms * 1000" | bc 2>/dev/null || echo "0")"
}

# ---- 1) Market research + knowledge-RAG ----
run_research() {
  log "Running market research + knowledge-RAG..."
  local raw="${INSIGHTS_DIR}/market-rag.raw.json"
  local hub_out="${INSIGHTS_DIR}/hub-moc.json"
  local out="${INSIGHTS_DIR}/market-rag.json"

  # Produce minimal tagged structure if scripts missing
  if [[ -x "${ROOT}/scripts/granite-business-research.sh" ]]; then
    "${ROOT}/scripts/granite-business-research.sh" > "${raw}"
  else
    cat > "${raw}" <<'EOF'
{"insights":[],"tags":["#business-research","#knowledge-rag","#graph"],"hub_review":"MOC","generated_at":"'"$(date -Iseconds)"'"}
EOF
  fi

  if [[ -x "${ROOT}/scripts/knowledge-rag.sh" ]]; then
    "${ROOT}/scripts/knowledge-rag.sh" --hub MOC --output "${hub_out}"
    jq -s '.[0] * {hub_insight: .[1]}' "${raw}" "${hub_out}" > "${out}"
  else
    cp "${raw}" "${out}"
  fi
  log "Insights written to ${out}"
}

# ---- 2) HF CDN-only manifest ----
# Single API call; intended to run after rate-limit window clears.
# Training should use the cdn_url fields (no Authorization header).
generate_hf_manifest() {
  local repo="${1:-datasets/your-org/surrogate-data}"
  local folder="${2:-batches/mirror-merged}"
  local out="${MANIFEST_DIR}/hf-cdn-manifest.json"

  log "Generating HF CDN manifest for ${repo}/${folder}..."

  python3 - <<PY > "${out}"
import json
import os
import sys
from datetime import datetime, timezone

try:
    from huggingface_hub import list_repo_tree
except ImportError:
    sys.stderr.write("error: huggingface_hub not installed\\n")
    sys.exit(1)

REPO = os.getenv("HF_REPO", "${repo}")
FOLDER = os.getenv("HF_FOLDER", "${folder}")

# Single API call (run once per cron window)
items = list_repo_tree(REPO, path=FOLDER, recursive=False)

files = []
for it in items:
    if it.type != "file":
        continue
    cdn_url = f"https://huggingface.co/datasets/{REPO}/resolve/main/{it.path}"
    files.append({
        "path": it.path,
        "cdn_url": cdn_url,
        "size": getattr(it, "size", None)
    })

manifest = {
    "repo": REPO,
    "folder": FOLDER,
    "generated_at": datetime.now(timezone.utc).isoformat(),
    "strategy": "cdn-only",
    "note": "Use these URLs in training; zero API calls during data loading.",
    "files": files
}

sys.stdout.write(json.dumps(manifest, indent=2))
PY

  log "HF CDN manifest written to ${out}"
}
