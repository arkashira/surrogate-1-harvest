# workio / frontend

## 1. Diagnosis

- Clock-in/out UI has no network resilience: failed LINE webhook calls (timeout/5xx) show no feedback and leave UI state ambiguous (spinner forever or silent fail).
- No optimistic UI + rollback for punch actions, so retries create duplicate requests and race conditions (matches backend non-idempotency pattern).
- Missing local timestamp and GPS snapshot preview before submit, causing user uncertainty and extra round-trips.
- No offline queue: if user is underground/low-signal, punch is lost instead of being persisted locally and retried.
- Punch list doesn’t reflect in-flight state, so users can double-tap and compound duplicates.

## 2. Proposed change

**Scope:** `workio/src/components/PunchClock.tsx` (create if absent) and `workio/src/hooks/usePunch.ts` — single component + hook, <2h.  
Add optimistic UI, idempotent client key, offline queue (IndexedDB), and feedback states (toast + button state).

```ts
// src/hooks/usePunch.ts
import { create } from 'zustand';
import { persist } from 'zustand/middleware';
import { openDB, DBSchema, IDBPDatabase } from 'idb';
import toast from 'react-hot-toast';

type PunchType = 'IN' | 'OUT';
type PunchStatus = 'idle' | 'pending' | 'synced' | 'failed';

interface Punch {
  id: string;          // client-generated uuid
  type: PunchType;
  ts: number;          // local timestamp ms
  lat?: number;
  lng?: number;
  status: PunchStatus;
  attempts: number;
}

interface PunchState {
  current: Punch | null;
  queue: Punch[];
  setCurrent: (p: Punch | null) => void;
  addToQueue: (p: Punch) => void;
  updateStatus: (id: string, status: PunchStatus) => void;
  removeFromQueue: (id: string) => void;
}

export const usePunchStore = create<PunchState>()(
  persist(
    (set) => ({
      current: null,
      queue: [],
      setCurrent: (p) => set({ current: p }),
      addToQueue: (p) => set((s) => ({ queue: [...s.queue, p] })),
      updateStatus: (id, status) =>
        set((s) => ({
          queue: s.queue.map((q) => (q.id === id ? { ...q, status, attempts: q.attempts + 1 } : q)),
        })),
      removeFromQueue: (id) => set((s) => ({ queue: s.queue.filter((q) => q.id !== id) })),
    }),
    { name: 'workio-punch-store' }
  )
);

const DB_NAME = 'workio-punch-db';
const STORE_NAME = 'punch-queue';

interface PunchDB extends DBSchema {
  [STORE_NAME]: {
    key: string;
    value: Punch;
    indexes: { status: PunchStatus };
  };
}

export async function getDB() {
  return openDB<PunchDB>(DB_NAME, 1, {
    upgrade(db) {
      const store = db.createObjectStore(STORE_NAME, { keyPath: 'id' });
      store.createIndex('status', 'status', { unique: false });
    },
  });
}

export async function pushPunch(punch: Punch) {
  const db = await getDB();
  await db.put(STORE_NAME, punch);
}

export async function popPunches(): Promise<Punch[]> {
  const db = await getDB();
  return db.getAllFromIndex(STORE_NAME, 'status', 'failed');
}

export async function markSynced(id: string) {
  const db = await getDB();
  await db.delete(STORE_NAME, id);
}

export async function sendPunch(punch: Punch): Promise<boolean> {
  try {
    const res = await fetch('/api/webhook/line/punch', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        userId: punch.id, // client idempotency key; backend should dedupe by id or userId+ts window
        type: punch.type,
        ts: punch.ts,
        location: punch.lat && punch.lng ? { lat: punch.lat, lng: punch.lng } : undefined,
      }),
    });
    if (!res.ok) throw new Error(`HTTP ${res.status}`);
    return true;
  } catch (err) {
    console.warn('Punch send failed', err);
    return false;
  }
}

export function usePunchActions() {
  const { current, setCurrent, addToQueue, updateStatus, removeFromQueue } = usePunchStore();

  const attemptPunch = async (type: PunchType) => {
    // optimistic local state
    const punch: Punch = {
      id: crypto.randomUUID(),
      type,
      ts: Date.now(),
      status: 'pending',
      attempts: 0,
    };

    // try geolocation (best-effort)
    try {
      const pos = await new Promise<GeolocationPosition>((resolve, reject) =>
        navigator.geolocation.getCurrentPosition(resolve, reject, { timeout: 5000, enableHighAccuracy: true })
      );
      punch.lat = pos.coords.latitude;
      punch.lng = pos.coords.longitude;
    } catch {
      // continue without location
    }

    setCurrent(punch);
    addToQueue(punch);
    await pushPunch(punch);
    toast.loading('กำลังส่งข้อมูล...', { id: punch.id });

    const ok = await sendPunch(punch);
    if (ok) {
      updateStatus(punch.id, 'synced');
      removeFromQueue(punch.id);
      await markSynced(punch.id);
      toast.success(type === 'IN' ? 'เข้างานแล้ว' : 'ออกงานแล้ว', { id: punch.id });
      setCurrent(null);
    } else {
      updateStatus(punch.id, 'failed');
      toast.error('ไม่สามารถส่งข้อมูลได้ ระบบจะลองใหม่อัตโนมัติ', { id: punch.id });
      setCurrent(null);
    }
  };

  return { current, attemptPunch };
}
```
