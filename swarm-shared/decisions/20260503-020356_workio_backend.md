# workio / backend

## Final Synthesis (Best Parts + Correctness + Actionability)

### Diagnosis (merged)
- **No idempotency enforcement** — LINE retries identical deliveries, causing duplicate punches/leave/OT records.
- **Race-prone non-atomic upsert** — `findOne` → conditional `update`/`insert` loses under concurrent retries.
- **Missing DB-level uniqueness constraint** — no composite unique index prevents duplicates at the persistence layer.
- **No persisted idempotency key** — retries cannot be detected reliably; signature is delivery-level, not event-level.
- **Handler treats constraint violations as errors** — returns 5xx to LINE, causing repeated retries instead of graceful idempotent success.

### Goal
Make LINE webhook ingestion **atomic, idempotent, and safe under retries/concurrency** without changing downstream notification logic.

#### 2. Idempotent handler with atomic upsert and graceful conflict handling

```ts
// workio/server/src/controllers/lineWebhook.ts
import { Request, Response, NextFunction } from 'express';
import crypto from 'crypto';
import { pool } from '../db';

// Derive stable event-level idempotency key
function getLineEventId(body: any): string | null {
  if (!body.events || !Array.isArray(body.events)) return null;
  const first = body.events[0];
  if (first.id) return first.id;
  return crypto.createHash('sha256').update(JSON.stringify(first)).digest('hex');
}

// Optional: signature verification middleware
export function verifyLineSignature(req: Request, res: Response, next: NextFunction) {
  const signature = req.headers['x-line-signature'] as string;
  const body = JSON.stringify(req.body);
  const channelSecret = process.env.LINE_CHANNEL_SECRET;
  const expected = crypto.createHmac('sha256', channelSecret).update(body).digest('base64');
  if (signature !== expected) return res.status(401).json({ error: 'Invalid signature' });
  next();
}

export async function handleLineWebhook(req: Request, res: Response) {
  const lineEventId = getLineEventId(req.body);
  if (!lineEventId) return res.status(400).json({ error: 'Missing LINE event ID' });

  const event = req.body.events[0];
  const userId = event.source?.userId;
  const type = event.type;
  const lineTimestamp = event.timestamp ? new Date(event.timestamp) : new Date();
  const payload = event.postback?.data || event.message?.text || null;

  try {
    // Atomic insert; on conflict treat as successful idempotent apply
    const result = await pool.query(
      `INSERT INTO punches (line_event_id, user_id, type, payload, created_at, line_timestamp)
       VALUES ($1, $2, $3, $4, NOW(), $5)
       ON CONFLICT (line_event_id) DO NOTHING
       RETURNING id, created_at`,
      [lineEventId, userId, type, payload, lineTimestamp]
    );

    const isNew = result.rowCount && result.rowCount > 0;

    // Run side-effects only for new records (non-blocking)
    if (isNew) {
      // Example: send confirmation via LINE API (fire-and-forget or queue)
      // await sendLineReply(userId, 'Clock recorded').catch(() => {/* non-fatal */});
    }

    // Always 200 for LINE — treat constraint conflict as success
    return res.status(200).json({ ok: true, isNew });
  } catch (err: any) {
    // If it's a uniqueness violation (should be caught by ON CONFLICT, but be defensive),
    // treat as idempotent success. Otherwise 500.
    if (err.code === '23505') { // PostgreSQL unique_violation
      return res.status(200).json({ ok: true, isNew: false });
    }
    console.error('LINE webhook handler error:', err);
    return res.status(500).json({ error: 'Internal server error' });
  }
}
```
