# workio / frontend

## Final Synthesis (Best-of + Conflict Resolution)

**Chosen approach**: IndexedDB queue with idempotency keys, exponential backoff + jitter, cross-tab coordination, and automatic flush on online/visibility/interval.  
**Why**: Durable (IndexedDB), safe (idempotency), resilient (backoff + jitter), and correct under multi-tab use (BroadcastChannel + atomic queue updates).

**Conflicts resolved in favor of correctness + actionability**:
- **Storage**: IndexedDB over localStorage (durability, size, transactional safety).
- **Idempotency**: client-generated UUID + server-side idempotency key; 409 treated as success (drop item).
- **Backoff**: exponential with jitter (not fixed) to reduce thundering herd and battery drain.
- **Cross-tab**: BroadcastChannel for flush signaling + single-writer flush lock (`flushing`) to avoid races.
- **Flush triggers**: online, visibilitychange (foreground), periodic interval while online, and explicit signal after enqueue.
- **Retry limit**: finite (5) with drop to avoid unbounded growth; surfaced to UI via pending count.

```ts
import { openDB, DBSchema, IDBPDatabase } from 'idb';

interface AttendanceDB extends DBSchema {
  queue: {
    key: string; // idempotencyKey
    value: {
      idempotencyKey: string;
      action: 'clock_in' | 'clock_out';
      timestamp: number;
      payload: {
        location?: { lat: number; lng: number };
        note?: string;
      };
      retries: number;
      nextRetryAt: number;
      createdAt: number;
    };
  };
  meta: {
    key: string;
    value: { lastFlushAt: number };
  };
}

const DB_NAME = 'workio-attendance';
const QUEUE_STORE = 'queue';
const META_STORE = 'meta';
const MAX_RETRIES = 5;
const BASE_DELAY = 2000;
const JITTER = 0.5;
const FLUSH_INTERVAL = 30_000;
const CROSS_TAB_CHANNEL = 'workio-attendance-channel';

function randomJitter() {
  return 1 + (Math.random() * 2 - 1) * JITTER;
}

function nextBackoff(retries: number) {
  return Math.min(300_000, BASE_DELAY * Math.pow(2, retries)) * randomJitter();
}

class AttendanceQueue {
  private dbPromise: Promise<IDBPDatabase<AttendanceDB>>;
  private channel = new BroadcastChannel(CROSS_TAB_CHANNEL);
  private flushTimer: number | null = null;
  private flushing = false;

  constructor() {
    this.dbPromise = openDB<AttendanceDB>(DB_NAME, 1, {
      upgrade(db) {
        db.createObjectStore(QUEUE_STORE, { keyPath: 'key' });
        db.createObjectStore(META_STORE);
      },
    });

    this.channel.onmessage = (ev) => {
      if (ev.data === 'flush') this.tryFlush();
    };

    window.addEventListener('online', () => this.tryFlush());
    document.addEventListener('visibilitychange', () => {
      if (!document.hidden) this.tryFlush();
    });
  }

  private async withDB<T>(fn: (db: IDBPDatabase<AttendanceDB>) => Promise<T>) {
    const db = await this.dbPromise;
    return fn(db);
  }

  async enqueue(
    action: 'clock_in' | 'clock_out',
    payload: { location?: { lat: number; lng: number }; note?: string } = {}
  ) {
    const idempotencyKey = crypto.randomUUID();
    const now = Date.now();
    await this.withDB(async (db) => {
      await db.add(QUEUE_STORE, {
        key: idempotencyKey,
        value: {
          idempotencyKey,
          action,
          timestamp: now,
          payload,
          retries: 0,
          nextRetryAt: now,
          createdAt: now,
        },
      });
    });
    this.signalFlush();
    return idempotencyKey;
  }

  private signalFlush() {
    this.channel.postMessage('flush');
    this.tryFlush();
  }

  async tryFlush() {
    if (this.flushing || !navigator.onLine) return;
    this.flushing = true;
    try {
      await this.flushOnce();
    } finally {
      this.flushing = false;
    }
  }

  private async flushOnce() {
    const items = await this.withDB(async (db) => {
      const tx = db.transaction(QUEUE_STORE, 'readonly');
      const store = tx.store;
      const all = await store.getAll();
      const now = Date.now();
      return all.filter((it) => it.value.nextRetryAt <= now).sort((a, b) => a.value.nextRetryAt - b.value.nextRetryAt);
    });

    for (const item of items) {
      try {
        await this.submit(item);
        await this.withDB(async (db) => db.delete(QUEUE_STORE, item.key));
      } catch (err: any) {
        if (err?.status === 409) {
          // idempotency conflict — already processed; safe to drop
          await this.withDB(async (db) => db.delete(QUEUE_STORE, item.key));
          continue;
        }
        const nextRetries = item.value.retries + 1;
        if (nextRetries > MAX_RETRIES) {
          await this.withDB(async (db) => db.delete(QUEUE_STORE, item.key));
          continue;
        }
        await this.withDB(async (db) =>
          db.put(QUEUE_STORE, {
            ...item,
            value: {
              ...item.value,
              retries: nextRetries,
              nextRetryAt: Date.now() + nextBackoff(nextRetries),
            },
          })
        );
      }
    }
  }

  private async submit(item: { value: AttendanceDB['queue']['value'] }) {
    const res = await fetch('/api/attendance/clock', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        action: item.value.action,
        timestamp: item.value.timestamp,
        location: item.value.payload.location,
        note: item.value.payload.note,
        idempotencyKey: item.value.idempotencyKey,
      }),
    });

    if (!res.ok) {
      const err: any = new Error(`HTTP ${res.status}`);
      err.status = res.status;
      throw err;
    }
  }

  async startPeriodic() {
    if (this.flushTimer !== null) return;
    this.flushTimer = window.setInterval(() => {
      if (navigator.onLine) this.tryFlush();
    }, FLUSH_INTERVAL);
  }

  async stopPeriodic() {
    if (this.flushTimer !== null) {
      clearInterval(this.flushTimer);
      this.flushTimer = null;
    }
  }

  async getPendingCount() {
    return this.withDB(async (db) => db.count(QUEUE_STORE));
  }
}

export const attendanceQueue = new AttendanceQueue();
```
