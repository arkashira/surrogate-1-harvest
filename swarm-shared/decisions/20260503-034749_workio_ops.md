# workio / ops

## 1. Diagnosis
- LINE webhook handler is non-idempotent: retries on 5xx/timeouts replay clock/leave/OT events and corrupt state (duplicate punches, double approvals).
- Missing `X-Line-Signature` verification: requests can be spoofed and processed without cryptographic proof, allowing forged punches/leave.
- No real-time state propagation: frontend relies on local React state/polling; LINE webhook updates are invisible until hard refresh → silent data drift and perceived latency across tabs/devices.
- No audit trail for critical mutations: clock/leave/OT changes lack immutable log (who/when/why) for ops debugging and compliance.
- Ops observability gap: no structured request/response logging or metrics on webhook latency/errors; failures are opaque.

## 2. Proposed change
File: `/opt/axentx/workio/server/src/routes/webhook/line.ts` (or create if absent)  
Scope: add idempotency key + `X-Line-Signature` verification + structured logging + Server-Sent Events (SSE) broadcast for real-time propagation + minimal audit insert. Target: single handler function covering clock/leave/OT events.

// /opt/axentx/workio/server/src/routes/webhook/line.ts
import crypto from 'crypto';
import express, { Request, Response, NextFunction } from 'express';
import { v4 as uuidv4 } from 'uuid';
import { getDb } from '../../db';
import { broadcastEvent } from '../../lib/sse';
import { logger } from '../../lib/logger';

const router = express.Router();
const IDEMPOTENCY_TTL_MS = 10 * 60 * 1000; // 10m

function verifyLineSignature(rawBody: Buffer, signature: string): boolean {
  const channelSecret = process.env.LINE_CHANNEL_SECRET || '';
  if (!channelSecret) return false;
  const expected = crypto
    .createHmac('sha256', channelSecret)
    .update(rawBody)
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

async function isDuplicate(idempotencyKey: string): Promise<boolean> {
  const db = getDb();
  const row = await db.get<{ created_at: string }>(
    `SELECT created_at FROM webhook_idempotency WHERE key = ?`,
    [idempotencyKey]
  );
  if (!row) return false;
  const age = Date.now() - new Date(row.created_at).getTime();
  return age < IDEMPOTENCY_TTL_MS;
}

async function markProcessed(idempotencyKey: string, payload: unknown) {
  const db = getDb();
  await db.run(
    `INSERT INTO webhook_idempotency (key, payload, created_at) VALUES (?, ?, datetime('now'))`,
    [idempotencyKey, JSON.stringify(payload)]
  );
  // prune old keys (best-effort)
  db.run(`DELETE FROM webhook_idempotency WHERE created_at < datetime('now', '-1 hour')`);
}

async function recordAudit(tenantId: string | number, actor: string, action: string, target: string, details: unknown) {
  const db = getDb();
  await db.run(
    `INSERT INTO audit_log (tenant_id, actor, action, target, details, created_at)
     VALUES (?, ?, ?, ?, ?, datetime('now'))`,
    [tenantId, actor, action, target, JSON.stringify(details)]
  );
}

router.post('/line', async (req: Request, res: Response, next: NextFunction) => {
  const signature = req.headers['x-line-signature'] as string;
  const rawBody = req.body as Buffer;
  const requestId = `line-webhook-${uuidv4()}`;

  // structured request start
  logger.info({ requestId, event: 'line_webhook_received', path: req.path, signaturePresent: !!signature });

  // 1) Signature verification
  if (!verifyLineSignature(rawBody, signature)) {
    logger.warn({ requestId, event: 'invalid_signature' });
    return res.status(401).json({ error: 'Invalid signature' });
  }

  let events;
  try {
    events = JSON.parse(rawBody.toString('utf8')).events;
    if (!Array.isArray(events) || events.length === 0) {
      logger.info({ requestId, event: 'no_events' });
      return res.status(200).json({ ok: true });
    }
  } catch (err) {
    logger.error({ requestId, event: 'parse_error', error: err });
    return res.status(400).json({ error: 'Invalid payload' });
  }

  // 2) Idempotency per webhook delivery (use delivery id if available, else first event id)
  const deliveryId = (req.headers['x-line-delivery-id'] as string) || events[0]?.source?.userId || uuidv4();
  const idempotencyKey = `line:${deliveryId}`;
  if (await isDuplicate(idempotencyKey)) {
    logger.info({ requestId, event: 'duplicate_delivery', idempotencyKey });
    return res.status(200).json({ ok: true });
  }

  const db = getDb();
  let processed = false;

  try {
    await db.run('BEGIN');

    for (const ev of events) {
      const { type, source, message, postback } = ev;
      const userId = source?.userId;
      const tenantId = null; // resolve tenant from userId via mapping table (simplified)
      const actor = `line:${userId}`;

      if (type === 'message' && message?.type === 'text') {
        const text = (message.text || '').trim().toLowerCase();

        // Clock in/out
        if (text === 'clock in' || text === 'clock out') {
          const action = text === 'clock in' ? 'clock_in' : 'clock_out';
          // simplified: upsert daily punch
          await db.run(
            `INSERT INTO punches (tenant_id, user_id, action, ts, line_event_id, created_at)
             VALUES (?, ?, ?, datetime('now'), ?, datetime('now'))
             ON CONFLICT(tenant_id, user_id, date(ts)) DO UPDATE SET action=excluded.action, ts=datetime('now'), line_event_id=excluded.line_event_id`,
            [tenantId, userId, action, ev.source?.type + ':' + ev.source?.userId]
          );

          await recordAudit(tenantId, actor, action, `punch:${userId}`, { ev });
          processed = true;
        }

        // Leave/OT requests (simplified)
        if (text.startsWith('leave') || text.startsWith('ot')) {
          const action = text.startsWith('leave') ? 'leave_request' : 'ot_request';
          await db.run(
            `INSERT INTO requests (tenant_id, user_id, type, status, payload, created_at)
             VALUES (?, ?, ?, 'pending', ?, datetime('now'))`,
            [tenantId, userId, action, JSON.stringify({ text, ev })]
          );
          await recordAudit(tenantId, actor, action, `request:${userId}`, { text, ev });
          processed = true;
        }
      }

      if (type === 'postback') {
        // handle approve/reject postbacks
        const data = postback?.data || '';
        // simplified: parse and update request
        // ...
        processed = true;
      }
    }

    await markProcessed(idempotencyKey, { events });
    await db.run('COMMIT');

    // 3) Broadcast real-time update for ops/frontend
    broadcastEvent({ type: 'line_webhook_processed', requestId, processed });

    logger.info({ requestId, event: 'processed', processed, idempotencyKey });
    res.status(200).json({ ok: true, requestId });
  } catch (err) {
    await db.run('ROLLBACK');
    logger.error({ requestId, event: 'processing_error', error: err });
    next(err);
  }
});

export default router;
