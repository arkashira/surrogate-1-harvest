### 1. One-time Mac orchestration script (≤10 min)

`scripts/list_hf_date_folder.py`

```python
#!/usr/bin/env python3
"""
Run from Mac (or any dev machine) after rate-limit window clears.
Produces file_list.json for Lightning training to use CDN-only fetches.
"""
import json
import os
from datetime import datetime, timezone
from huggingface_hub import HfApi

REPO = "axentx/surrogate-training-data"
DATE_FOLDER = datetime.now(timezone.utc).strftime("%Y-%m-%d")
OUT_PATH = "file_list.json"

api = HfApi()
# Single API call: recursive=False avoids pagination explosion
tree = api.list_repo_tree(
    repo_id=REPO,
    path=DATE_FOLDER,
    recursive=False,
)

files = [
    f.rfilename
    for f in tree
    if f.rfilename.endswith(".parquet")
]

payload = {
    "repo": REPO,
    "date_folder": DATE_FOLDER,
    "files": files,
    "generated_utc": datetime.now(timezone.utc).isoformat(),
    "cdn_base": f"https://huggingface.co/datasets/{REPO}/resolve/main",
}

os.makedirs(os.path.dirname(OUT_PATH), exist_ok=True)
with open(OUT_PATH, "w") as f:
    json.dump(payload, f, indent=2)

print(f"Wrote {len(files)} files to {OUT_PATH}")
```

import json
import pyarrow.parquet as pq
import torch
from torch.utils.data import Dataset, DataLoader
from lightning import LightningDataModule
import requests
from io import BytesIO
import os

class CDNParquetDataset(Dataset):
    """Zero HF API calls during training — uses public CDN URLs only."""
    def __init__(self, file_list_path, max_rows_per_file=50_000):
        with open(file_list_path) as f:
            cfg = json.load(f)

        self.base = cfg["cdn_base"]
        self.files = cfg["files"]
        self.max_rows = max_rows_per_file
        self._build_index()

    def _build_index(self):
        """Build lightweight index without loading full files."""
        self.index = []  # (file_idx, row_offset)
        for f_idx, fname in enumerate(self.files):
            url = f"{self.base}/{fname}"
            # Lightweight request: only HEAD to get row count via metadata
            # Parquet metadata is at end of file; small range request works
            try:
                r = requests.head(url, timeout=10)
                r.raise_for_status()
                # If HEAD doesn't expose rows, do a tiny range request for footer
                # For simplicity, assume ~N rows; or use pyarrow.parquet.read_metadata
                # via streaming range request (CDN supports Range headers).
                meta = pq.read_metadata(BytesIO(requests.get(url, timeout=30).content[-1024*1024:]))
                n_rows = meta.num_rows
            except Exception:
                # Fallback: skip problematic file
                continue

            for offset in range(0, n_rows, self.max_rows):
                self.index.append((f_idx, offset))

    def __len__(self):
        return len(self.index)

    def __getitem__(self, idx):
        f_idx, offset = self.index[idx]
        fname = self.files[f_idx]
        url = f"{self.base}/{fname}"

        # Range request for just the needed row group(s)
        # Simplified: download whole file slice (small); optimize with row-group ranges if needed
        resp = requests.get(url, timeout=30)
        resp.raise_for_status()
        table = pq.read_table(BytesIO(resp.content))
        # Project to {prompt, response} only
        chunk = table.slice(offset, self.max_rows).to_pydict()
        # Return single sample (or batch) — adapt to your format
        prompt = chunk.get("prompt", [""])[0]
        response = chunk.get("response", [""])[0]
        return {"prompt": prompt, "response": response}


class CDNParquetDataModule(LightningDataModule):
    def __init__(self, file_list_path, batch_size=8, num_workers=4):
        super().__init__()
        self.file_list_path = file_list_path
        self.batch_size = batch_size
        self.num_workers = num_workers

    def setup(self, stage=None):
        self.ds = CDNParquetDataset(self.file_list_path)

    def train_dataloader(self):
        return DataLoader(
            self.ds,
            batch_size=self.batch_size,
            num_workers=self.num_workers,
            shuffle=True,
            pin_memory=True,
        )
