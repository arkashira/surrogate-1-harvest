# airship / backend

## Implementation Plan (≤2h)

**Highest-value incremental improvement**:  
Ship a cron-safe `airship discover` orchestrator that:
1. Produces a **CDN-only training manifest** (bypasses HF API rate limits during training)
2. Runs **knowledge-rag + top-hub insight** for business context
3. Is idempotent, logs to structured files, and can run nightly without human intervention

**Why this now**:  
- Uses the CDN bypass insight (HF CDN URLs don’t count against API rate limits)
- Embeds the file-list once so Lightning training does zero API calls during data load
- Integrates knowledge-rag + top-hub review per past patterns
- Safe for cron (proper shebang, executable, no interactive deps)

## 1) Create library: `/opt/axentx/airship/lib/airship-discover.sh`

```bash
#!/usr/bin/env bash
# airship-discover.sh — library used by bin/airship-discover
# Cron-safe, idempotent, CDN-only manifest generation + knowledge-rag insights

set -euo pipefail
IFS=$'\n\t'

# Config
AXENTX_ROOT="${AXENTX_ROOT:-/opt/axentx/airship}"
MANIFEST_DIR="${AXENTX_ROOT}/manifests"
LOG_DIR="${AXENTX_ROOT}/logs"
TIMESTAMP="$(date -u +%Y%m%dT%H%M%SZ)"
MANIFEST_FILE="${MANIFEST_DIR}/hf-cdn-manifest-${TIMESTAMP}.json"
LATEST_MANIFEST="${MANIFEST_DIR}/hf-cdn-manifest-latest.json"
LOG_FILE="${LOG_DIR}/airship-discover-${TIMESTAMP}.log"

# Ensure dirs
mkdir -p "${MANIFEST_DIR}" "${LOG_DIR}"
exec > >(tee -a "${LOG_FILE}") 2>&1

log() {
  echo "[$(date -u +%Y-%m-%dT%H:%M:%SZ)] $*"
}

# Requires: curl, jq
# Optional but recommended: HF_TOKEN for private repos (not needed for public CDN)
# Usage: generate_hf_cdn_manifest <owner>/<dataset_repo> <date_folder?>
generate_hf_cdn_manifest() {
  local repo="$1"
  local folder="${2:-}"  # optional subfolder like "batches/2026-04-29"
  local api_base="https://huggingface.co/api/datasets/${repo}/tree"
  local tree_path="${folder}"
  local items

  log "INFO: Listing repo tree: ${repo} (path: ${tree_path:-root})"

  # Single API call (non-recursive) to avoid pagination/rate limits
  if [ -n "${tree_path}" ]; then
    items=$(curl -sSfL --retry 3 --retry-delay 5 "${api_base}/${tree_path}?recursive=false")
  else
    items=$(curl -sSfL --retry 3 --retry-delay 5 "${api_base}?recursive=false")
  fi

  # Build CDN-only manifest (no auth required at training time)
  echo "$items" | jq -r '
    map(select(.type == "file")) |
    map({
      repo: "'"${repo}"'",
      path: .path,
      size: .size,
      cdn_url: ("https://huggingface.co/datasets/" + "'"${repo}"'" + "/resolve/main/" + .path)
    })
  ' > "${MANIFEST_FILE}.tmp"

  mv "${MANIFEST_FILE}.tmp" "${MANIFEST_FILE}"
  cp "${MANIFEST_FILE}" "${LATEST_MANIFEST}"
  log "INFO: Manifest written: ${MANIFEST_FILE}"
  log "INFO: Latest symlink: ${LATEST_MANIFEST}"
}

# Usage: run_knowledge_rag [--top-hub]
run_knowledge_rag() {
  local top_hub=0
  for arg in "$@"; do
    case "$arg" in
      --top-hub) top_hub=1 ;;
    esac
  done

  log "INFO: Running knowledge-rag pipeline..."
  # Placeholder: call your knowledge-rag CLI or script here.
  # Example (adjust to your actual command):
  #   knowledge-rag --query "top hub and related docs" --output "${MANIFEST_DIR}/rag-insights-${TIMESTAMP}.json"
  #
  # If you have a script at scripts/knowledge-rag.sh, invoke it:
  if [[ -x "${AXENTX_ROOT}/scripts/knowledge-rag.sh" ]]; then
    "${AXENTX_ROOT}/scripts/knowledge-rag.sh" --output "${MANIFEST_DIR}/rag-insights-${TIMESTAMP}.json"
    log "INFO: knowledge-rag completed"
  else
    log "WARN: knowledge-rag script not found or not executable; skipping"
  fi

  if [[ $top_hub -eq 1 ]]; then
    log "INFO: Extracting top-hub insight (MOC or highest-degree node)..."
    # Placeholder: query Neo4j or graph store for most-connected hub.
    # Example:
    #   python3 -c "from neo4j import GraphDatabase; ...; print(most_connected)"
    # Save to file for later consumption.
    if [[ -x "${AXENTX_ROOT}/scripts/top-hub-insight.sh" ]]; then
      "${AXENTX_ROOT}/scripts/top-hub-insight.sh" > "${MANIFEST_DIR}/top-hub-${TIMESTAMP}.txt"
      log "INFO: Top-hub insight saved"
    else
      log "INFO: No top-hub script; skipping"
    fi
  fi
}

# Main orchestration used by bin/airship-discover
main() {
  log "INFO: Starting airship discover (CDN manifest + RAG insights)"

  # Example defaults — override via env or CLI args in bin/airship-discover
  local dataset_repo="${DISCOVER_DATASET_REPO:-example/dataset}"
  local date_folder="${DISCOVER_DATE_FOLDER:-}"
  local do_rag="${DISCOVER_RAG:-1}"
  local do_top_hub="${DISCOVER_TOP_HUB:-1}"

  generate_hf_cdn_manifest "${dataset_repo}" "${date_folder}"

  if [[ "${do_rag}" == "1" ]]; then
    if [[ "${do_top_hub}" == "1" ]]; then
      run_knowledge_rag --top-hub
    else
      run_knowledge_rag
    fi
  fi

  log "INFO: airship discover completed successfully"
}

# If executed directly (for testing), run main
if [[ "${BASH_SOURCE[0]}" == "${0}" ]]; then
  main "$@"
fi
```

## 2) Create executable entrypoint: `/opt/axentx/airship/bin/airship-discover`

```bash
#!/usr/bin/env bash
# airship-discover — cron-safe orchestrator
# Usage (cron):
#   0 2 * * * SHELL=/bin/bash /opt/axentx/airship/bin/airship-discover >> /opt/axentx/airship/logs/cron.log 2>&1

set -euo pipefail
IFS=$'\n\t'

AXENTX_ROOT="${AXENTX_ROOT:-/opt/axentx/airship}"
export AXENTX_ROOT

# Optional overrides via environment (recommended for cron)
#   DISCOVER_DATASET_REPO=owner/dataset
#   DISCOVER_DATE_FOLDER=batches/2026-04-29
#   DISCOVER_RAG=1
#   DISCOVER_TOP_HUB=1

exec "${AXENTX_ROOT}/lib/airship-discover.sh" "$@"
```
