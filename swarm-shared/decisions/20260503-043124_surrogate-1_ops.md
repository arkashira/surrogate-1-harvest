## Implementation Plan (≤2h)

**Highest-value improvement**: Replace fragile shell-based ingestion with a **manifest-driven, CDN-bypass Python worker** that eliminates HF API rate limits during training data loads and prevents mixed-schema CastErrors.

### Changes

1. **Add `bin/worker.py`** — deterministic shard worker that:
   - Uses `list_repo_tree(path, recursive=False)` once per date folder (single API call) to build manifest.
   - Persists manifest to `manifest-<date>.json` so Lightning training can do **CDN-only fetches** with zero API calls.
   - Downloads via raw CDN URLs (`https://huggingface.co/datasets/.../resolve/main/...`) — bypasses `/api/` auth rate limits.
   - Projects each file to `{prompt, response}` at parse time (avoids `load_dataset` mixed-schema CastError).
   - Deduplicates via central `lib/dedup.py` md5 store.
   - Writes `batches/public-merged/<date>/shard<N>-<HHMMSS>.jsonl`.

2. **Update `bin/dataset-enrich.sh`** — thin wrapper that:
   - Exports `PYTHONUNBUFFERED=1`, `SHELL=/bin/bash`.
   - Invokes `python3 bin/worker.py "$@"` (no `streaming=True`, no `load_dataset` on heterogeneous repos).

3. **Update `.github/workflows/ingest.yml`** — ensure matrix uses `bash` to launch wrapper and sets `env.SHELL=/bin/bash`.

4. **Add `requirements.txt` entries** if missing: `requests`, `tqdm`, `python-slugify`.

#!/usr/bin/env python3
"""
CDN-bypass shard worker for surrogate-1 public dataset ingestion.
- Builds manifest once per date folder via list_repo_tree (single API call).
- Downloads via raw CDN URLs (no Authorization header) to bypass HF API rate limits.
- Projects heterogeneous files to {prompt, response} at parse time.
- Deduplicates via lib.dedup and writes shard-N JSONL.
"""
import os
import sys
import json
import hashlib
import datetime
import subprocess
from pathlib import Path
from typing import List, Dict, Any

import requests
from huggingface_hub import HfApi, hf_hub_download

# -- config --
REPO_ID = os.getenv("HF_DATASET_REPO", "axentx/surrogate-1-training-pairs")
HF_TOKEN = os.getenv("HF_TOKEN")
SHARD_ID = int(os.getenv("SHARD_ID", "0"))
SHARD_TOTAL = int(os.getenv("SHARD_TOTAL", "16"))
DATE_FOLDER = os.getenv("DATE_FOLDER", datetime.datetime.utcnow().strftime("%Y-%m-%d"))
BASE_DIR = Path(__file__).parent.parent
MANIFEST_DIR = BASE_DIR / "manifests"
OUTPUT_DIR = BASE_DIR / "batches" / "public-merged" / DATE_FOLDER
DEDUPS_DB = BASE_DIR / "lib" / "dedup.py"  # central md5 store (imported)

# -- init --
MANIFEST_DIR.mkdir(parents=True, exist_ok=True)
OUTPUT_DIR.mkdir(parents=True, exist_ok=True)

api = HfApi(token=HF_TOKEN)
session = requests.Session()

# -- dedup bridge --
# Import central dedup store; fallback to local sqlite if not available.
try:
    sys.path.insert(0, str(BASE_DIR / "lib"))
    from dedup import is_duplicate, record_hash
except Exception:
    # Minimal fallback: local sqlite in batches dir
    import sqlite3
    db_path = OUTPUT_DIR.parent / "dedup.db"
    db_path.parent.mkdir(parents=True, exist_ok=True)
    conn = sqlite3.connect(str(db_path))
    conn.execute("CREATE TABLE IF NOT EXISTS hashes (md5 TEXT PRIMARY KEY)")
    def is_duplicate(md5: str) -> bool:
        return conn.execute("SELECT 1 FROM hashes WHERE md5=?", (md5,)).fetchone() is not None
    def record_hash(md5: str) -> None:
        conn.execute("INSERT OR IGNORE INTO hashes (md5) VALUES (?)", (md5,))
        conn.commit()

# -- helpers --
def slug_hash(s: str) -> int:
    return int(hashlib.md5(s.encode()).hexdigest(), 16)

def belongs_to_shard(slug: str) -> bool:
    return slug_hash(slug) % SHARD_TOTAL == SHARD_ID

def build_manifest(date_folder: str) -> List[str]:
    """Single API call: list top-level files in date folder (non-recursive)."""
    manifest_path = MANIFEST_DIR / f"manifest-{date_folder}.json"
    if manifest_path.exists():
        return json.loads(manifest_path.read_text())

    items = api.list_repo_tree(repo_id=REPO_ID, path=date_folder, recursive=False)
    files = [it.rfilename for it in items if it.type == "file"]
    # Keep only files likely to contain training pairs (parquet/jsonl/csv)
    allowed = {".parquet", ".jsonl", ".json", ".csv", ".tsv"}
    files = [f for f in files if any(f.endswith(ext) for ext in allowed)]
    manifest_path.write_text(json.dumps(files, indent=2))
    return files

def cdn_url(path: str) -> str:
    return f"https://huggingface.co/datasets/{REPO_ID}/resolve/main/{path}"

def download_via_cdn(path: str, dest: Path) -> Path:
    url = cdn_url(path)
    resp = session.get(url, timeout=60)
    resp.raise_for_status()
    dest.write_bytes(resp.content)
    return dest

def project_to_pair(record: Dict[str, Any]) -> Dict[str, str] | None:
    """Project heterogeneous schema to {prompt, response}. Return None if invalid."""
    # Common field names seen in public datasets
    prompt_keys = {"prompt", "input", "question", "instruction", "text", "content"}
    response_keys = {"response", "output", "answer", "completion", "target"}

    # Normalize keys to lowercase
    rec = {k.lower(): v for k, v in record.items()}
    prompt = None
    response = None

    for k in prompt_keys:
        if k in rec and isinstance(rec[k], str) and rec[k].strip():
            prompt = rec[k].strip()
            break
    for k in response_keys:
        if k in rec and isinstance(rec[k], str) and rec[k].strip():
            response = rec[k].strip()
            break

    # Fallback: if only one text-like field, split by common separators
    if not prompt or not response:
        text = rec.get("text") or rec.get("content")
        if isinstance(text, str):
            parts = [p.strip() for p in text.split("###") if p.strip()]
            if len(parts) >= 2:
                prompt, response = parts[0], parts[1]
            elif len(parts) == 1 and "Q:" in parts[0] and "A:" in parts[0]:
                qa = parts[0]
                try:
                    prompt = qa.split("A:")[0].replace("Q:", "").strip()
                    response = qa.split("A:")[1].strip()
                except Exception:
                    pass

    if not prompt or not response:
        return None
    return {"prompt": prompt, "response": response}

def process_parquet(path: Path) -> List[Dict[str, str]]:
    """Stream parquet and project rows; avoids load_dataset on mixed schemas."""
    try:
        import pyarrow.parquet as pq
        table = pq.read_table(path)
        rows = table.to_pylist()
        pairs = []
        for row in rows:
            pair = project_to_pair(row)
            if pair:
                pairs.append(pair)
        return pairs
    except Exception as e:
        print(f"Parquet parse error {path}: {e}", file=sys.stderr)
        return []

def process_jsonl(path: Path) -> List[Dict[str, str]]:
    pairs = []
    for line in path.read_text().splitlines():
        line = line.strip()
        if not line:
            continue
        try:
            row = json.loads(line)
            pair = project_to_pair(row)
            if pair:
                pairs.append(pair)
        except Exception:
            continue
    return pairs
