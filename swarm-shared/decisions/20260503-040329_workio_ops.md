# workio / ops

## Final Synthesized Implementation Plan

**Core philosophy:** Combine **observability** (Candidate 1) with **resilience** (Candidate 2). We get immediate ops visibility *and* prevent data loss during LINE outages.

---

### 1. Unified Diagnosis (Accepted)
All issues from both candidates are valid and non-contradictory:
- **Observability gap**: No structured logs, correlation IDs, or metrics.
- **Resilience gap**: No circuit breaker/retry for LINE API, no idempotency/locking for cron jobs.
- **Ops gap**: No health/readiness checks or lightweight dashboard.

---

### 2. Final Architecture (Best of Both)

```
server/src/
├── middleware/
│   ├── observability.ts    ← logs, request-id, metrics, ring buffer (C1)
│   ├── health.ts           ← liveness/readiness + DB pool checks (C2)
│   └── circuitBreaker.ts   ← LINE API retry/backoff/429 handling (C2)
├── routes/
│   ├── ops.ts              ← /ops/health, /ready, /metrics (C1)
│   └── line.ts             ← webhook handler w/ metrics + circuitBreaker (C1+C2)
└── utils/
    └── locks.ts            ← advisory locks for cron/queue (C2)
```

import { Request, Response, NextFunction } from 'express';
import { v4 as uuidv4 } from 'uuid';

const LOG_LEVEL = (process.env.LOG_LEVEL || 'info').toLowerCase();
const LEVELS = { error: 0, warn: 1, info: 2, debug: 3 } as const;
type Level = keyof typeof LEVELS;

function shouldLog(level: Level) {
  return LEVELS[level] <= (LEVELS[LOG_LEVEL as Level] || 2);
}

declare global {
  namespace Express {
    interface Request { id: string; }
  }
}

export function logger(req: Request, res: Response, next: NextFunction) {
  const requestId = req.headers['x-request-id'] as string || uuidv4();
  req.id = requestId;
  res.setHeader('X-Request-Id', requestId);

  const start = Date.now();
  res.on('finish', () => {
    if (shouldLog('info')) {
      console.log(JSON.stringify({
        t: new Date().toISOString(),
        level: 'info' as const,
        requestId,
        method: req.method,
        path: req.path,
        status: res.statusCode,
        ms: Date.now() - start,
        ip: req.ip,
        ua: req.get('User-Agent') || ''
      }));
    }
  });
  next();
}

// server/src/middleware/circuitBreaker.ts
type State = 'closed' | 'open' | 'half-open';

export class CircuitBreaker {
  private state: State = 'closed';
  private failures = 0;
  private lastFailure = 0;
  private nextAttempt = 0;
  
  constructor(
    private threshold = 5,      // failures to open
    private timeout = 30000,    // 30s open state
    private halfOpenTimeout = 10000 // 10s before retry
  ) {}

  async call<T>(fn: () => Promise<T>): Promise<T> {
    if (this.state === 'open') {
      if (Date.now() > this.nextAttempt) {
        this.state = 'half-open';
      } else {
        throw new Error('Circuit breaker is OPEN');
      }
    }

    try {
      const result = await fn();
      this.onSuccess();
      return result;
    } catch (err: any) {
      this.onFailure(err);
      throw err;
    }
  }

  private onSuccess() {
    this.failures = 0;
    this.state = 'closed';
  }

  private onFailure(err: any) {
    this.failures++;
    this.lastFailure = Date.now();
    
    if (this.failures >= this.threshold) {
      this.state = 'open';
      this.nextAttempt = Date.now() + this.timeout;
    }
  }

  getState() {
    return { state: this.state, failures: this.failures };
  }
}
