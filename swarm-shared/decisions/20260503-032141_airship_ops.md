### 1) Generate CDN file list (run once per date folder)

```python
# scripts/generate_file_list.py
#!/usr/bin/env python3
import argparse, json, os
from huggingface_hub import list_repo_tree

def main() -> None:
    p = argparse.ArgumentParser()
    p.add_argument("--repo", required=True, help="datasets/your-org/surrogate-mirror")
    p.add_argument("--date", required=True, help="YYYY-MM-DD")
    p.add_argument("--out", default="surrogate/train_file_list.json")
    a = p.parse_args()

    folder = a.date
    entries = list_repo_tree(repo_id=a.repo, path=folder, recursive=False)

    files = []
    for e in entries:
        if e.type == "file":
            cdn = f"https://huggingface.co/datasets/{a.repo}/resolve/main/{folder}/{e.path}"
            files.append(cdn)

    out = os.path.join(os.path.dirname(__file__), "..", a.out.lstrip("/"))
    os.makedirs(os.path.dirname(out), exist_ok=True)
    with open(out, "w") as f:
        json.dump({"date": a.date, "files": sorted(files)}, f, indent=2)
    print(f"Wrote {len(files)} files to {out}")

if __name__ == "__main__":
    main()
```

### 2) Update `surrogate/train.py` for CDN + idle recovery

```python
# surrogate/train.py
import argparse, json, os, time
from typing import Iterator
import torch
from datasets import IterableDataset
from torch.utils.data import DataLoader
from transformers import AutoTokenizer, AutoModelForCausalLM, Trainer, TrainingArguments

class CDNIterableDataset(IterableDataset):
    def __init__(self, file_urls: list[str], tokenizer, max_length: int = 2048):
        self.file_urls = file_urls
        self.tokenizer = tokenizer
        self.max_length = max_length

    def _stream_file(self, url: str) -> Iterator[tuple[str, str]]:
        import requests
        with requests.get(url, stream=True, timeout=60) as r:
            r.raise_for_status()
            for line in r.iter_lines(decode_unicode=True):
                if not line:
                    continue
                try:
                    obj = json.loads(line)
                    p, r = obj.get("prompt"), obj.get("response")
                    if p is not None and r is not None:
                        yield p, r
                except Exception:
                    continue

    def _tokenize_pair(self, prompt: str, response: str) -> dict:
        text = f"{prompt}\n\n{response}"
        enc = self.tokenizer(text, truncation=True, max_length=self.max_length, padding="max_length")
        enc["labels"] = enc["input_ids"].copy()
        return enc

    def __iter__(self):
        wi = torch.utils.data.get_worker_info()
        urls = (self.file_urls if wi is None else
                self.file_urls[wi.id * (len(self.file_urls)//wi.num_workers) : (wi.id+1) * (len(self.file_urls)//wi.num_workers)])
        for url in urls:
            for prompt, response in self._stream_file(url):
                yield self._tokenize_pair(prompt, response)

def build_dataset(tokenizer, file_list_path: str):
    with open(file_list_path) as f:
        manifest = json.load(f)
    urls = manifest["files"]
    if not urls:
        raise ValueError(f"No files in {file_list_path}")
    return CDNIterableDataset(urls, tokenizer)

def train(output_dir: str, file_list: str, resume_from_checkpoint: str = None):
    model_name = os.getenv("MODEL_NAME", "Qwen/Qwen2.5-0.5B-Instruct")
    tokenizer = AutoTokenizer.from_pretrained(model_name)
    model = AutoModelForCausalLM.from_pretrained(model_name)

    train_dataset = build_dataset(tokenizer, file_list)

    training_args = TrainingArguments(
        output_dir=output_dir,
        per_device_train_batch_size=4,
        gradient_accumulation_steps=8,
        num_train_epochs=1,
        learning_rate=2e-5,
        logging_steps=10,
        save_steps=500,
        save_total_limit=3,
        fp16=torch.cuda.is_available(),
        dataloader_num_workers=2,
        remove_unused_columns=False,
    )

    trainer = Trainer(model=model, args=training_args, train_dataset=train_dataset)
    trainer.train(resume_from_checkpoint=resume_from_checkpoint)
    trainer.save_model(output_dir)

if __name__ == "__main__":
    ap = argparse.ArgumentParser()
    ap.add_argument("--output-dir", default="./surrogate-output")
    ap.add_argument("--file-list", default="train_file_list.json")
    ap.add_argument("--resume-from-checkpoint", default=None)
    a = ap.parse_args()
    train(output_dir=a.output_dir, file_list=a.file_list, resume_from_checkpoint=a.resume_from_checkpoint)
```

**Key behaviors:**
- **Zero HF API calls during training** (CDN direct download).
- **Lightning idle-safe**: script is restartable; Trainer auto-resumes from latest checkpoint when re-run.
- **Schema**: one JSONL object per line with `{"prompt": "...", "response": "..."}`.

### 3) Lightning launcher with studio reuse + idle recovery

# scripts/launch_training.py
#!/usr/bin/env python3
import os, time, subprocess, argparse

try:
    from lightning_sdk import Client
except ImportError:
    print("Missing lightning-sdk; install with: pip install lightning")
    exit(1)

def launch(
    file_list: str,
    output_dir: str = "./surrogate-output",
    studio_name: str = "surrogate-train",
    cloud: str = "aws",
    machine: str = "cpu.small",
    max_retries: int = 10,
    retry_delay: int = 60,
):
    client = Client()
    cmd = [
        "python", "surrogate/train.py",
        "--file-list", file_list,
        "--output-dir", output_dir,
    ]

    # Resume existing studio if present
    try:
        studio = client.studios(name=studio_name)
        print(f"Resuming existing studio: {studio_name}")
    except Exception:
        print(f"Creating studio: {studio_name}")
        studio = client.studios.create(
            name=studio_name,
            cloud=cloud,
            machine=machine,
        )

    # Run with retry on idle-stop
    for attempt in range(1, max_retries + 1):
        try:
            print(f"Run attempt {attempt}/{max_retries}")
            run = studio.runners.create(name=f"train-{int(time.time())}")
            run.execute(command=" ".join(cmd), wait=True)
            print("Training completed.")
            return
        except Exception as e:
            if "idle" in str(e).lower() or "stopped" in str(e).lower():
                print(f"Idle/stop detected (attempt {attempt}). Restarting...")
                time.sleep(retry_delay)
                continue
            else:
                print(f"Non-idle error: {e}")
                raise

    print("Max retries reached; studio remains alive for manual resume.")

if __name__ == "__main__":
    ap = argparse.ArgumentParser()
    ap.add_argument("--file-list", required=True)
