### 1) CDN manifest generator (run once per ingestion batch)

```python
# scripts/generate_cdn_manifest.py
#!/usr/bin/env python3
"""
Generate a CDN-only manifest for a date folder in a HuggingFace dataset repo.
Run from Mac (or any machine) once per ingestion batch.

Usage:
  python generate_cdn_manifest.py \
    --repo-id axentx/surrogate-dataset \
    --date-folder 2026-05-03 \
    --out-file cdn_manifest.json
"""
import argparse
import json
import time
from pathlib import Path

try:
    from huggingface_hub import HfApi
except ImportError:
    raise SystemExit("pip install huggingface_hub")

HF_API_RATE_LIMIT_RETRY = 360  # seconds (per pattern)


def list_date_folder(repo_id: str, date_folder: str, retries: int = 3):
    api = HfApi()
    for attempt in range(1, retries + 1):
        try:
            # Non-recursive per-folder listing (avoids heavy pagination)
            items = api.list_repo_tree(
                repo_id=repo_id,
                path=date_folder,
                repo_type="dataset",
                recursive=False,
            )
            # Keep only parquet files
            files = [p.rfilename for p in items if p.rfilename.endswith(".parquet")]
            return sorted(files)
        except Exception as exc:
            wait = HF_API_RATE_LIMIT_RETRY if "429" in str(exc) else (2 ** attempt)
            print(f"[attempt {attempt}/{retries}] list_repo_tree failed: {exc}. retry in {wait}s")
            time.sleep(wait)
    raise RuntimeError(f"Failed to list {repo_id}/{date_folder} after {retries} attempts")


def build_manifest(repo_id: str, date_folder: str):
    files = list_date_folder(repo_id, date_folder)
    base_url = f"https://huggingface.co/datasets/{repo_id}/resolve/main"
    manifest = {
        "repo_id": repo_id,
        "date_folder": date_folder,
        "generated_at": time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime()),
        "files": [
            {
                "path": p,
                "cdn_url": f"{base_url}/{p}",
            }
            for p in files
        ]
    }
    return manifest


def main():
    parser = argparse.ArgumentParser(description="Generate CDN manifest for dataset date folder")
    parser.add_argument("--repo-id", required=True, help="HF dataset repo id")
    parser.add_argument("--date-folder", required=True, help="Date folder path in dataset (e.g. 2026-05-03)")
    parser.add_argument("--out-file", default="cdn_manifest.json", help="Output JSON path")
    args = parser.parse_args()

    manifest = build_manifest(args.repo_id, args.date_folder)
    out_path = Path(args.out_file)
    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"Wrote manifest with {len(manifest['files'])} files to {out_path}")


if __name__ == "__main__":
    main()
```

### 2) Training script patch (`surrogate/train.py`)

Key changes:
- Accept `--cdn-manifest` (path to `cdn_manifest.json`). If provided, use CDN URLs only (no HF API during training).
- Reuse a Running Studio; if stopped, restart once.
- Guard against idle-stop by checking status before `.run()` and restarting if needed.

class CDNParquetDataset(IterableDataset):
    """
    Streams rows from parquet files listed in a CDN manifest.
    Projects to {prompt, response} only at parse time.
    """
    def __init__(self, manifest_path: str, columns: List[str] = ("prompt", "response")):
        self.manifest_path = manifest_path
        self.columns = columns
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.file_urls = [item["cdn_url"] for item in self.manifest["files"]]

    def _stream_file(self, url: str) -> Iterable[Dict[str, str]]:
        resp = requests.get(url, timeout=60)
        resp.raise_for_status()
        table = pq.read_table(pq.ParquetFile(pq.ParquetReader(pq.BufferReader(resp.content))))
        missing = [c for c in self.columns if c not in table.column_names]
        if missing:
            raise ValueError(f"Missing columns {missing} in {url}")
        df = table.select(self.columns).to_pandas()
        for _, row in df.iterrows():
            yield {k: row[k] for k in self.columns}

    def __iter__(self):
        for url in self.file_urls:
            yield from self._stream_file(url)

# ------------------------------
# Studio reuse + idle-stop guard
# ------------------------------
def get_or_create_studio(name: str, machine: str = "L40S"):
    """
    Reuse a Running Studio if exists; otherwise create.
    If Studio exists but is stopped, restart it (idle-stop guard).
    """
    if Studio is None:
        print("Lightning Studio not available; skipping Studio orchestration.")
        return None

    teamspace = Teamspace()
    for s in teamspace.studios:
        if s.name == name:
            if s.status == "Running":
                print(f"Reusing running Studio: {name}")
                return s
            else:
                print(f"Studio {name} exists but status={s.status}. Restarting...")
                s.start(machine=machine)
                return s

    print(f"Creating Studio: {name}")
    return Studio(name=name, machine=machine)


def run_training_with_studio_guard(studio, train_fn):
    """
    Guard against idle-stop: check status before run and restart if stopped.
    """
    if studio is None:
        return train_fn()

    if studio.status != "Running":
        print(f"Studio not running (status={studio.status}). Restarting before training.")
        studio.start(machine=getattr(studio, "machine", "L40S"))

    return studio.run(train_fn)
