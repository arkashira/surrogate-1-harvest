## Final Synthesized Implementation (Best Parts Combined)

**Goal**: Eliminate HF API 429s and Lightning quota waste during Surrogate training by implementing **CDN-first deterministic ingestion + Lightning Studio reuse**.

**Why this ships in <2h**: Single-file change to training orchestration + one small utility. No infra changes, no schema migrations, no new services.

---

## Implementation Plan

### 1. Create `scripts/build_file_manifest.py` (Mac orchestration only)
- Runs after rate-limit window clears
- Uses `list_repo_tree` per date folder (non-recursive, single call)
- Outputs `data/manifest-{date}.json` with CDN URLs
- Embeds into training container at build time
- Validates CDN accessibility (spot-check)

### 2. Modify `surrogate/training/dataloader.py`
- Accept `--manifest` arg (path to JSON)
- Data loader uses `requests.get(cdn_url)` with retry/backoff
- Zero HF API calls during training
- Projects to `{prompt, response}` only

### 3. Add `surrogate/training/orchestrator.py`
- Reuse running Lightning Studio if found
- Auto-restart if stopped (idle timeout)
- Deterministic repo selection via hash(slug) % 5 siblings

### `scripts/build_file_manifest.py`
```python
#!/usr/bin/env python3
"""
Build CDN-only file manifest for Surrogate training.
Run from Mac after HF rate-limit window clears.
"""
import json, hashlib, os, time
from datetime import datetime, timezone
from huggingface_hub import HfApi

HF_REPO = os.getenv("HF_DATASET_REPO", "axentx/surrogate-mirror")
DATE_FOLDER = os.getenv("DATE_FOLDER", datetime.now(timezone.utc).strftime("%Y-%m-%d"))
OUTPUT_PATH = os.getenv("MANIFEST_OUT", f"data/manifest-{DATE_FOLDER}.json")
SIBLINGS = 5

def pick_repo(slug: str) -> str:
    h = int(hashlib.md5(slug.encode()).hexdigest(), 16)
    idx = h % SIBLINGS
    return f"axentx/surrogate-datasets-{idx}"

def validate_cdn(url: str, timeout: int = 5) -> bool:
    try:
        r = requests.head(url, timeout=timeout)
        return r.status_code == 200
    except Exception:
        return False

def main() -> None:
    api = HfApi()
    prefix = f"batches/mirror-merged/{DATE_FOLDER}/"

    # Single non-recursive call per folder (avoids pagination)
    items = api.list_repo_tree(
        repo_id=HF_REPO,
        path=prefix,
        repo_type="dataset",
        recursive=False,
    )

    # Only parquet files (ignore _SUCCESS, _committed, etc)
    files = []
    for item in items:
        if not item.path.endswith(".parquet"):
            continue
        slug = os.path.splitext(os.path.basename(item.path))[0]
        target_repo = pick_repo(slug)
        cdn_url = f"https://huggingface.co/datasets/{target_repo}/resolve/main/{item.path}"
        
        # Spot-check CDN accessibility
        if not validate_cdn(cdn_url):
            print(f"WARNING: CDN inaccessible for {item.path}")
        
        files.append({
            "cdn_url": cdn_url,
            "path": item.path,
            "slug": slug,
            "repo": target_repo,
            "size": getattr(item, "size", None),
        })

    manifest = {
        "dataset_repo": HF_REPO,
        "date_folder": DATE_FOLDER,
        "generated_at_utc": datetime.now(timezone.utc).isoformat(),
        "files": sorted(files, key=lambda x: x["path"]),
        "total_files": len(files),
    }

    os.makedirs(os.path.dirname(os.path.abspath(OUTPUT_PATH)), exist_ok=True)
    with open(OUTPUT_PATH, "w") as f:
        json.dump(manifest, f, indent=2)

    print(f"Wrote {len(files)} files to {OUTPUT_PATH}")

if __name__ == "__main__":
    main()
```

### `surrogate/training/dataloader.py`
```python
import json, io, requests, time
import pyarrow.parquet as pq
from torch.utils.data import Dataset

class CDNParquetDataset(Dataset):
    def __init__(self, manifest_path, max_retries=3):
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.files = self.manifest["files"]
        self.max_retries = max_retries

    def __len__(self):
        return len(self.files)

    def __getitem__(self, idx):
        entry = self.files[idx]
        for attempt in range(self.max_retries):
            try:
                r = requests.get(entry["cdn_url"], timeout=30)
                r.raise_for_status()
                table = pq.read_table(io.BytesIO(r.content))
                # Project to {prompt, response} only
                return {
                    "prompt": table["prompt"].to_pylist(),
                    "response": table["response"].to_pylist()
                }
            except Exception as e:
                if attempt == self.max_retries - 1:
                    raise
                time.sleep(2 ** attempt)
```

### `surrogate/training/orchestrator.py`
```python
from lightning import Lightning, Teamspace, Machine

def get_or_create_studio(name: str):
    """Reuse running studio or create new L40S."""
    ls = Lightning()
    studios = Teamspace.studios()
    
    for s in studios:
        if s.name == name and s.status == "running":
            print(f"Reusing running studio: {name}")
            return s
    
    print(f"Creating new studio: {name}")
    return ls.studio(
        name=name,
        machine=Machine.L40S,
        create_ok=True
    )

def safe_run(studio, script, args):
    """Run script, restart if studio stopped by idle timeout."""
    if studio.status != "running":
        print(f"Studio stopped, restarting...")
        studio.start(machine=Machine.L40S)
    studio.run(script, arguments=args)
```
