# vanguard / design

## Final synthesized answer

### Diagnosis (merged, corrected)
- **No persisted manifest** causes every training run to re-enumerate via authenticated HF API → quota burn and 429 risk.
- **Recursive/tree or streaming enumeration at runtime** triggers many paginated API calls and can hit schema/parsing errors.
- **Training recomputes file lists instead of using static CDN URLs** → prevents true CDN bypass and wastes quota during long jobs.
- **Lightning Studio is not reused** → duplicate studios burn quota (80+ hrs/mo) and increase cold-start latency.
- **No idle-stop resilience or deterministic write sharding** → idle timeouts kill training; large mirror/writes risk HF’s 128/hr/repo commit cap.

### Core change (single, concrete plan)
1. **Generate once, persist, and embed a CDN-only manifest** per `(repo, dateFolder)` using a single non-recursive `list_repo_tree` call.  
2. **Train using only CDN URLs** (`https://huggingface.co/datasets/{repo}/resolve/main/{path}`) with zero authenticated HF API calls during data loading.  
3. **Reuse an existing Running Lightning Studio**; create only if none running.  
4. **Add deterministic repo sharding for writes** to respect HF commit limits.  
5. **Add idle-resilient launcher logic** (restart on L40S if stopped by idle timeout).

```python
#!/usr/bin/env python3
"""
Generate a CDN-only file manifest for a repo + dateFolder.
Run once per folder (e.g. from Mac orchestration node) after rate-limit window.
"""
import json
import os
import sys
from pathlib import Path
from typing import List, Dict

try:
    from huggingface_hub import HfApi
except ImportError:
    print("Install: pip install huggingface_hub")
    sys.exit(1)

API = HfApi()
REPO = os.getenv("HF_REPO", "vanguard/datasets")
DATE_FOLDER = os.getenv("DATE_FOLDER", "2026-04-29")
OUT_DIR = Path(__file__).parent.parent / "manifests"
OUT_PATH = OUT_DIR / f"{DATE_FOLDER}.json"


def list_date_files(repo: str, date_folder: str) -> List[Dict[str, object]]:
    """Single non-recursive call; returns CDN-ready entries."""
    entries = []
    tree = API.list_repo_tree(repo=repo, path=date_folder, recursive=False)
    for node in tree:
        if node.type != "file":
            continue
        cdn_url = (
            f"https://huggingface.co/datasets/{repo}/resolve/main/"
            f"{date_folder}/{node.path}"
        )
        entries.append({
            "repo": repo,
            "path": f"{date_folder}/{node.path}",
            "cdn_url": cdn_url,
            "size": getattr(node, "size", None),
        })
    return entries


def pick_shard_repo(slug: str, n_shards: int = 5) -> str:
    """Deterministic repo sharding to bypass HF 128/hr commit cap."""
    h = hash(slug) % (2**32)
    return f"vanguard-mirror-{h % n_shards}"


def main() -> None:
    OUT_DIR.mkdir(parents=True, exist_ok=True)
    files = list_date_files(REPO, DATE_FOLDER)
    manifest = {
        "repo": REPO,
        "date_folder": DATE_FOLDER,
        "generated_by": "pipeline/manifest.py",
        "files": files,
    }
    OUT_PATH.write_text(json.dumps(manifest, indent=2))
    print(f"Manifest written: {OUT_PATH} ({len(files)} files)")


if __name__ == "__main__":
    main()
```

```python
def load_manifest(manifest_path: Path = MANIFEST_PATH) -> Dict[str, Any]:
    if not manifest_path.exists():
        raise FileNotFoundError(
            f"Manifest missing: {manifest_path}. Run pipeline/manifest.py first."

# ---- CDN-only dataset ----
class CDNIterableDataset(IterableDataset):
    """Yield parsed examples from CDN URLs listed in manifest."""

    def __init__(self, manifest_path: Path = MANIFEST_PATH):
        manifest = load_manifest(manifest_path)
        self.items = manifest.get("files", [])
        if not self.items:
            raise ValueError("No files in manifest.")

    def _fetch(self, cdn_url: str) -> bytes:
        resp = requests.get(cdn_url, timeout=30)
        resp.raise_for_status()
        return resp.content

    def _parse(self, raw: bytes) -> Iterable[Dict[str, Any]]:
        """
        Implement parsing for your format (JSONL, JSON, etc.).
        Must yield dicts with at least {prompt, response} or equivalent.
        """
        # Example for JSONL:
        for line in raw.decode("utf-8").strip().splitlines():
            line = line.strip()
            if not line:
                continue
            obj = json.loads(line)
            # Keep only required fields; drop extra columns.
            yield {
                "prompt": obj["prompt"],
                "response": obj["response"],
            }

    def __iter__(self):
        for item in self.items:
            raw = self._fetch(item["cdn_url"])
            yield from self._parse(raw)


def make_dataloader(manifest_path: Path = MANIFEST_PATH, **dl_kwargs) -> DataLoader:
```
