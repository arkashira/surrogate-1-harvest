## Highest-Value Incremental Improvement
**Make Surrogate training HF-rate-limit-proof and Lightning-idle-resilient** by:
- Generating a one-time CDN-only file list on the Mac (bypasses `/api/` auth checks)
- Embedding the file list in `train.py` so Lightning workers do **zero API calls** during data loading
- Adding studio-reuse + idle-restart logic to survive Lightning 30m idle timeouts
- Projecting to `{prompt,response}` only before upload (no mixed-schema writes)

**Estimated effort:** <2h (single-file change + one orchestration script)

---

## Implementation Plan

### 1. Mac orchestration script (run once per date folder)
- Uses `list_repo_tree(path, recursive=False)` per folder (no recursive paginated 100x calls)
- Writes `file_list.json` with CDN URLs (`resolve/main/...`)
- Embeds file list into `train.py` via template or direct injection

### 2. `surrogate/training/train.py` changes
- Replace `load_dataset(streaming=True)` with CDN-only streaming via `IterableWrapper` + `fsspec` HTTPFileSystem
- Deterministic shard selection: `hash(slug) % N` for sibling repo spread (if needed)
- Project to `{prompt, response}` only at parse time; drop `source`, `ts`, extra cols
- Studio reuse: list running studios before `Studio(create_ok=True)`
- Idle resilience: check `studio.status` before `.run()`; restart with `Machine.L40S` if stopped

### 3. Deployment
- No infra changes; keep existing `docker-compose.microservices.yml`
- Training triggered via Lightning SDK from Mac (orchestration only)

### `scripts/build_cdn_filelist.py` (run on Mac)
```python
#!/usr/bin/env python3
"""
Generate CDN-only file list for a date folder.
Run on Mac (or any machine) after rate-limit window clears.
Embeds into train.py so Lightning workers do zero API calls.
"""
import json
import os
import hashlib
from huggingface_hub import HfApi

API_TOKEN = os.getenv("HF_TOKEN")  # optional for public reads
REPO_ID   = "your-org/surrogate-dataset"  # or datasets/...
DATE_PATH = "batches/mirror-merged/2026-05-03"  # example
OUTPUT_TRAIN_PY = "surrogate/training/train.py"
OUTPUT_JSON     = "surrogate/training/file_list.json"

def deterministic_shard(slug: str, n_shards: int = 5) -> int:
    return int(hashlib.sha256(slug.encode()).hexdigest(), 16) % n_shards

def main():
    api = HfApi(token=API_TOKEN)
    # Single non-recursive call per folder (avoids 100x pagination)
    items = api.list_repo_tree(REPO_ID, path=DATE_PATH, recursive=False)

    files = []
    for item in items:
        if item.rfilename.endswith(".parquet"):
            slug = item.rfilename.replace(".parquet", "")
            cdn_url = f"https://huggingface.co/datasets/{REPO_ID}/resolve/main/{item.rfilename}"
            files.append({
                "filename": item.rfilename,
                "cdn_url": cdn_url,
                "shard": deterministic_shard(slug, n_shards=5),
                "slug": slug
            })

    os.makedirs(os.path.dirname(OUTPUT_JSON), exist_ok=True)
    with open(OUTPUT_JSON, "w") as f:
        json.dump(files, f, indent=2)

    # Embed into train.py as a module-level constant
    with open(OUTPUT_TRAIN_PY, "r") as f:
        content = f.read()

    injection = f"""
# AUTO-GENERATED CDN FILE LIST (do not edit manually)
# Generated from {REPO_ID}/{DATE_PATH}
CDN_FILE_LIST = {json.dumps(files, indent=2)}
"""

    # Replace existing CDN_FILE_LIST block or append after imports
    if "CDN_FILE_LIST = " in content:
        # naive but sufficient for our controlled file
        lines = content.splitlines()
        out = []
        skip = False
        for line in lines:
            if line.strip().startswith("CDN_FILE_LIST = "):
                skip = True
                out.append(injection.strip())
            elif skip and line.strip() and not line.startswith(" "):
                skip = False
                out.append(line)
            elif not skip:
                out.append(line)
        content = "\n".join(out)
    else:
        # insert after last import
        parts = content.splitlines()
        for i, line in enumerate(parts):
            if line.startswith("import ") or line.startswith("from "):
                continue
            else:
                # insert before this line
                parts.insert(i, injection.strip())
                break
        else:
            parts.append(injection.strip())
        content = "\n".join(parts)

    with open(OUTPUT_TRAIN_PY, "w") as f:
        f.write(content)

    print(f"Wrote {len(files)} files to {OUTPUT_JSON} and injected into {OUTPUT_TRAIN_PY}")

if __name__ == "__main__":
    main()
```

from lightning.pytorch.studio import Studio, Machine, Teamspace

def get_or_create_studio(name: str = "surrogate-train", machine: Machine = Machine.L40S):
    team = Teamspace.current()
    for s in team.studios:
        if s.name == name:
            if s.status == "Running":
                print(f"Reusing running studio: {name}")
                return s
            else:
                print(f"Restarting stopped studio: {name}")
                s.start(machine=machine)
                return s
    print(f"Creating new studio: {name}")
    return Studio(name=name, create_ok=True, machine=machine)

# ---- CDN-only streaming dataset ----
class CDNParquetStream(IterableDataset):
    def __init__(self, file_list, max_files=None):
        super().__init__()
        self.files = file_list[:max_files] if max_files else file_list

    def __iter__(self):
        worker_info = torch.utils.data.get_worker_info()
        if worker_info is None:
            iter_files = self.files
        else:
            # shard by worker id
            per_worker = len(self.files) // worker_info.num_workers
            iter_files = self.files[worker_info.id * per_worker : (worker_info.id + 1) * per_worker]

        for meta in iter_files:
            url = meta["cdn_url"]
            try:
                with fsspec.open(url, "rb", anon=True) as f:
                    # Lightweight: read only required columns if possible
                    # For simplicity we stream whole parquet; for huge files use pyarrow.dataset
                    import pyarrow.parquet as pq
                    table = pq.read_table(f, columns=["prompt", "response"])
                    for row in table.to_pylist():
                        # Ensure minimal schema
                        yield {
                            "prompt": row.get("prompt", ""),
                            "response": row.get("response", "")
                        }
            except Exception as e:
                print(f"Skipping {url}: {e}")
                continue
