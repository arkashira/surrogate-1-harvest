# vanguard / design

## Final synthesized solution (single, actionable change)

### 1. Diagnosis (merged + resolved)

- **CDN-bypass ingestion path missing**: training scripts hit HF API during data loading → 429s. Fix: list once, persist manifest, train via CDN URLs only.
- **No durable ingestion run state**: retries manual, no persisted file list/metadata. Fix: single-run manifest + `latest` symlink; training reads local JSON.
- **Dataset list state not URL-encoded**: reloads/shared links lose page/sort/filter/selection. Fix: encode into hash (URLSearchParams) and restore on load/hashchange.
- **Hash-router scroll not restored**: back/forward jumps to top. Fix: per-route scroll restoration keyed by location key + scroll-to-anchor/previous viewport.
- **Mac ingestion orchestration missing/risky**: local scripts may run model code. Fix: provide lightweight CLI orchestrator that only lists and writes manifests; never loads data into model process.

Contradictions resolved in favor of correctness + actionability:
- Use **hash-based URL state** (works with existing hash router) rather than search params if router config requires it; either is fine provided encoding/decoding is consistent and robust.
- **Scroll restoration**: implement a small, framework-agnostic manager that restores per-location viewport (scrollY) and anchor targets; avoid heavy dependencies.
- **Ingestion orchestrator**: single Python script, Mac-friendly, retries with exponential backoff and 429 wait, writes JSON + latest symlink; training uses CDN-only URLs.

#### 2.1 CDN-bypass ingestion orchestrator (Mac/Lightning control plane)

File: `/opt/axentx/vanguard/src/orchestration/ingest_cdn.py`

```python
#!/usr/bin/env python3
"""
CDN-bypass ingestion orchestrator (Mac/Lightning control plane).
- Lists HF repo tree once (after rate-limit window) and saves file manifest.
- Produces train_manifest.json for Lightning training (CDN-only fetches).
- Never loads datasets via HF API during training.
"""
import json
import os
import sys
import time
from datetime import datetime, timezone
from pathlib import Path
from typing import List, Dict

try:
    from huggingface_hub import HfApi
except ImportError:
    print("Install: pip install huggingface_hub")
    sys.exit(1)

HF_REPO = os.getenv("HF_DATASET_REPO", "datasets/company/vanguard-ingest")
DATE_FOLDER = os.getenv("INGEST_DATE_FOLDER", datetime.now(timezone.utc).strftime("%Y-%m-%d"))
OUTPUT_DIR = Path(os.getenv("INGEST_OUTDIR", "manifests"))
OUTPUT_DIR.mkdir(parents=True, exist_ok=True)

API = HfApi()

def list_date_files(repo: str, date_folder: str, max_retries: int = 3) -> List[str]:
    """Single API call to list files in one date folder (non-recursive)."""
    for attempt in range(max_retries):
        try:
            items = API.list_repo_tree(repo=repo, path=date_folder, recursive=False)
            files = []
            for it in items:
                name = it["path"] if isinstance(it, dict) else getattr(it, "path", str(it))
                if name.endswith((".parquet", ".jsonl", ".csv")):
                    files.append(name)
            return files
        except Exception as exc:
            wait = 360 if "429" in str(exc) else (2**attempt)
            print(f"Attempt {attempt+1}/{max_retries} failed: {exc}. Wait {wait}s", file=sys.stderr)
            time.sleep(wait)
    raise RuntimeError("Failed to list repo tree after retries")

def build_manifest(file_paths: List[str], repo: str, date_folder: str) -> List[Dict]:
    """Build CDN-only manifest: {cdn_url, slug, date_folder}."""
    manifest = []
    for fp in file_paths:
        slug = Path(fp).stem
        cdn_url = f"https://huggingface.co/datasets/{repo}/resolve/main/{fp}"
        manifest.append({
            "cdn_url": cdn_url,
            "hf_path": fp,
            "slug": slug,
            "date_folder": date_folder,
        })
    return manifest

def main() -> None:
    print(f"Listing {HF_REPO}/{DATE_FOLDER} ...")
    files = list_date_files(HF_REPO, DATE_FOLDER)
    if not files:
        print("No parquet/jsonl/csv files found in date folder.", file=sys.stderr)
        sys.exit(1)

    manifest = build_manifest(files, HF_REPO, DATE_FOLDER)
    ts = datetime.now(timezone.utc).strftime("%Y%m%dT%H%M%SZ")
    out_path = OUTPUT_DIR / f"train_manifest_{DATE_FOLDER}_{ts}.json"
    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"Manifest written: {out_path}")
    print(f"Total files: {len(manifest)}")

    # Latest symlink for training script
    latest = OUTPUT_DIR / "train_manifest_latest.json"
    latest.write_text(json.dumps(manifest, indent=2))

if __name__ == "__main__":
    main()
```

function parseHashState(hash = "") {
  const params = new URLSearchParams(hash.replace(/^#/, ""));
  return {
    page: Math.max(1, parseInt(params.get("page") || "1", 10)),
    pageSize: Math.max(1, parseInt(params.get("pageSize") || "50", 10)),
    sort: params.get("sort") || "createdAt",
    dir: params.get("dir") || "desc",
    filter_status: params.get("filter_status") || "",
    selection: (params.get("selection") || "").split(",").filter(Boolean),
  };
}

function serializeHashState(state) {
  const p = new URLSearchParams();
  if (state.page > 1) p.set("page", String(state.page));
  if (state.pageSize !== 50) p.set("pageSize", String(state.pageSize));
  if (state.sort !== "createdAt") p.set("sort", state.sort);
  if (state.dir !== "desc") p.set("dir", state.dir);
  if (state.filter_status) p.set("filter_status", state.filter_status);
  if (state.selection?.length) p.set("selection", state.selection.join(","));
  return p.toString() ? `#${p.toString()}` : "";
}
