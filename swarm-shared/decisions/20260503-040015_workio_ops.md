### 1. DB schema (idempotency + safety)

```sql
-- server/src/db/schema.sql

-- 1) Generic webhook_events for reliable dedupe and audit
CREATE TABLE IF NOT EXISTS webhook_events (
  id               UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id        UUID NOT NULL,
  user_id          UUID NOT NULL,
  event_type       TEXT NOT NULL,            -- clock_in, clock_out, leave_request, ot_request, etc.
  line_delivery_id TEXT NOT NULL,            -- from X-Line-Delivery-Tag or generated client-side
  payload          JSONB NOT NULL,
  occurred_at      TIMESTAMPTZ NOT NULL,
  created_at       TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- Dedupe at ingestion (same LINE delivery should never insert twice)
CREATE UNIQUE INDEX IF NOT EXISTS uq_webhook_line_delivery
  ON webhook_events (line_delivery_id);

-- Domain-level dedupe (same user+type+time should not create duplicate domain records)
CREATE UNIQUE INDEX IF NOT EXISTS uq_webhook_user_event_occurred
  ON webhook_events (user_id, event_type, occurred_at);

-- 2) Domain tables: add external_id + unique guards where appropriate
-- Clock events
ALTER TABLE clock_records
  ADD COLUMN IF NOT EXISTS external_id TEXT;
CREATE UNIQUE INDEX IF NOT EXISTS uq_clock_user_occurred
  ON clock_records (user_id, occurred_at);

-- Leave requests
ALTER TABLE leave_requests
  ADD COLUMN IF NOT EXISTS external_id TEXT;
CREATE UNIQUE INDEX IF NOT EXISTS uq_leave_external
  ON leave_requests (external_id)
  WHERE external_id IS NOT NULL;

-- OT requests
ALTER TABLE ot_requests
  ADD COLUMN IF NOT EXISTS external_id TEXT;
CREATE UNIQUE INDEX IF NOT EXISTS uq_ot_external
  ON ot_requests (external_id)
  WHERE external_id IS NOT NULL;
```

Notes:  
- `webhook_events.line_delivery_id` is the primary ingestion dedupe key (fast, precise).  
- Domain-level unique guards are a safety net and support idempotent upserts when projecting.  
- Use `external_id` populated from LINE payload (or generated) so retries don’t create duplicates.

### 2. Idempotent webhook handler (commit-before-200)

```typescript
// server/src/routes/webhook/line.ts
import { Router, Request, Response } from 'express';
import { pool } from '../../db';
import { v4 as uuidv4 } from 'uuid';

const router = Router();

router.post('/line', async (req: Request, res: Response) => {
  const deliveryId = req.header('X-Line-Delivery-Tag') || uuidv4();
  const body = req.body;

  if (!body.events || !Array.isArray(body.events)) {
    return res.status(400).json({ error: 'invalid_payload' });
  }

  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    for (const event of body.events) {
      const { type, source, timestamp, ...rest } = event;
      const occurredAt = new Date(timestamp).toISOString();
      const userId = source?.userId; // adapt to your user mapping
      const tenantId = '00000000-0000-0000-0000-000000000000'; // replace with real tenant resolution

      // 1) Idempotent insert into webhook_events (dedupe by delivery)
      await client.query(
        `INSERT INTO webhook_events (tenant_id, user_id, event_type, line_delivery_id, payload, occurred_at)
         VALUES ($1, $2, $3, $4, $5, $6)
         ON CONFLICT (line_delivery_id) DO NOTHING`,
        [tenantId, userId, type, deliveryId, { ...rest, raw_event: event }, occurredAt]
      );

      // 2) Project to domain tables idempotently
      // Example for clock events; adapt for leave/ot similarly
      if (type === 'clock_in' || type === 'clock_out') {
        await client.query(
          `INSERT INTO clock_records (user_id, action, occurred_at, external_id)
           VALUES ($1, $2, $3, $4)
           ON CONFLICT (user_id, occurred_at) DO UPDATE
           SET action = EXCLUDED.action, external_id = EXCLUDED.external_id`,
          [userId, type, occurredAt, deliveryId]
        );
      }

      if (type === 'leave_request') {
        await client.query(
          `INSERT INTO leave_requests (user_id, status, requested_at, external_id, details)
           VALUES ($1, $2, $3, $4, $5)
           ON CONFLICT (external_id) DO UPDATE
           SET status = EXCLUDED.status, details = EXCLUDED.details`,
          [userId, 'pending', occurredAt, deliveryId, { ...rest, occurred_at: occurredAt }]
        );
      }

      if (type === 'ot_request') {
        await client.query(
          `INSERT INTO ot_requests (user_id, status, requested_at, external_id, details)
           VALUES ($1, $2, $3, $4, $5)
           ON CONFLICT (external_id) DO UPDATE
           SET status = EXCLUDED.status, details = EXCLUDED.details`,
          [userId, 'pending', occurredAt, deliveryId, { ...rest, occurred_at: occurredAt }]
        );
      }
    }

    await client.query('COMMIT');
    // Return 200 only after commit (so LINE retries only on true failure)
    return res.status(200).json({ ok: true });
  } catch (err) {
    await client.query('ROLLBACK');
    console.error('Webhook processing failed', { deliveryId, err });
    // 5xx causes LINE retry; avoid 2xx on partial writes
    return res.status(500).json({ error: 'processing_failed' });
  } finally {
    client.release();
  }
});

export default router;
```
