# vanguard / ops

## Final consolidated solution

**Highest-leverage fix**: Persist a single CDN-only manifest per `(repo, dateFolder)` once (outside training), then train exclusively from CDN URLs. This eliminates repeated HF API enumeration, removes 429 risk during training, and makes epochs start fast and deterministic.

**Secondary fix**: In Lightning launchers, reuse a running Studio when present; if stopped, restart it before `.run()` so idle-stop doesn’t silently kill jobs.

### 1. Manifest generator (single source of truth)

`/opt/axentx/vanguard/ops/persist_manifest.py`

- One authenticated, non-recursive tree call per `(repo, dateFolder)`.
- Idempotent: overwrites same dateFolder.
- Stores CDN URLs and sizes; no auth needed at training time.
- Deterministic filename safe for scripting.

```python
#!/usr/bin/env python3
"""
Persist a local CDN-only manifest for (repo, dateFolder) to avoid HF API calls during training.

Usage:
  HF_TOKEN=hf_... persist_manifest.py <repo> <dateFolder> [out_dir]

Examples:
  persist_manifest.py mycorp/surrogate-1 2026-04-29
  persist_manifest.py datasets/mycorp/surrogate-1 2026-04-29 manifests
"""
import json
import os
import sys
from pathlib import Path

import requests

HF_API_BASE = "https://huggingface.co/api"

def list_repo_tree(repo: str, path: str = "") -> list:
    """Single non-recursive tree call for a folder."""
    url = f"{HF_API_BASE}/datasets/{repo}/tree"
    params = {"path": path, "recursive": "false"}
    token = os.environ.get("HF_TOKEN")
    headers = {"Authorization": f"Bearer {token}"} if token else {}
    resp = requests.get(url, params=params, headers=headers, timeout=30)
    if resp.status_code == 429:
        print("HF API 429 — wait 360s before retry", file=sys.stderr)
    resp.raise_for_status()
    return resp.json()

def build_manifest(repo: str, date_folder: str, out_dir: Path):
    items = list_repo_tree(repo, path=date_folder)
    files = []
    for it in items:
        if it.get("type") != "file":
            continue
        files.append({
            "path": it["path"],
            "size": it.get("size"),
            "cdn_url": f"https://huggingface.co/datasets/{repo}/resolve/main/{it['path']}"
        })

    manifest = {
        "repo": repo,
        "dateFolder": date_folder,
        "generatedBy": "persist_manifest",
        "cdn_prefix": f"https://huggingface.co/datasets/{repo}/resolve/main/{date_folder}/",
        "files": files,
    }

    out_dir.mkdir(parents=True, exist_ok=True)
    out_path = out_dir / f"manifest-{repo.replace('/', '-')}-{date_folder}.json"
    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"Manifest written to {out_path}")
    return out_path

def main():
    if len(sys.argv) < 3:
        print("Usage: persist_manifest.py <repo> <dateFolder> [out_dir]")
        sys.exit(1)

    repo = sys.argv[1].replace("huggingface.co/datasets/", "").strip("/")
    date_folder = sys.argv[2].strip("/")
    out_dir = Path(sys.argv[3] if len(sys.argv) > 3 else Path(__file__).parent / "manifests")

    build_manifest(repo, date_folder, out_dir)

if __name__ == "__main__":
    main()
```

### 2. Manifest loader for training scripts

`/opt/axentx/vanguard/ops/load_manifest.py`

- Loads local manifest; raises clear error if missing.
- Returns CDN URLs and file metadata for DataLoader/IterableDataset.

```python
import json
from pathlib import Path
from typing import Dict, List

MANIFESTS_DIR = Path(__file__).parent / "manifests"

def load_manifest(repo: str, date_folder: str, manifests_dir: Path = MANIFESTS_DIR) -> Dict:
    manifest_path = manifests_dir / f"manifest-{repo.replace('/', '-')}-{date_folder}.json"
    if not manifest_path.exists():
        raise FileNotFoundError(
            f"Manifest missing: {manifest_path}. "
            f"Run persist_manifest.py {repo} {date_folder} first."
        )
    return json.loads(manifest_path.read_text())

def build_cdn_datalist(repo: str, date_folder: str, manifests_dir: Path = MANIFESTS_DIR) -> List[str]:
    manifest = load_manifest(repo, date_folder, manifests_dir)
    return [f["cdn_url"] for f in manifest["files"]]

def build_cdn_records(repo: str, date_folder: str, manifests_dir: Path = MANIFESTS_DIR):
    manifest = load_manifest(repo, date_folder, manifests_dir)
    return manifest["files"]
```

### 3. Lightning Studio reuse / restart guard

Add this helper where you create/run studios (e.g., launcher or train script). It:

- Reuses a running Studio with the same name.
- Restarts a stopped Studio before `.run()`.
- Avoids quota waste from repeated creation.

```python
from lightning import Teamspace, Studio, Machine

def get_or_restart_studio(name: str, machine: Machine):
    for s in Teamspace.studios():
        if s.name == name:
            if s.status == "Running":
                print(f"Reusing running studio: {name}")
                return s
            else:
                print(f"Studio {name} is {s.status}. Restarting...")
                s.stop()
                s.start()
                return s
    print(f"Creating studio: {name}")
    return Studio(name=name, machine=machine, create_ok=True)
```

```python
import requests

class CDNIterableDataset(IterableDataset):
    def __init__(self, urls):
        self.urls = urls

    def __iter__(self):
        for url in self.urls:
            # stream from CDN (no auth)
            with requests.get(url, stream=True, timeout=30) as r:
                r.raise_for_status()
                yield r.content  # or parse file as needed

def main():

    date_folder = "2026-04-29"

    loader = DataLoader(dataset, batch_size=None, num_workers=4)

if __name__ == "__main__":
    main()
```
