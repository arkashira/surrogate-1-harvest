# vanguard / ops

## Final Synthesis (single, correct, actionable)

**Root cause**: training and data selection hit the HF API on every run (list + streaming) instead of using a persisted CDN-only snapshot. This burns quota (429s) and adds schema/streaming risk.

**Goal**: pre-list once → persist `file_list.json` → train using CDN URLs only; make Lightning Studio reuse/restart automatic and robust.

#### `/opt/axentx/vanguard/ops/prepare_training.py`
```python
#!/usr/bin/env python3
"""
Pre-list a single HF repo date folder and emit file_list.json for CDN-only training.
Run from orchestration host (or CI) after rate-limit window is clear.
"""
import json
import os
import sys
from datetime import datetime, timezone

from huggingface_hub import HfApi

HF_REPO = os.getenv("HF_REPO", "datasets/axentx/surrogate-1")
DATE_FOLDER = os.getenv("DATE_FOLDER", datetime.now(timezone.utc).strftime("%Y-%m-%d"))
OUT_PATH = os.getenv("OUT_PATH", "ops/file_list.json")

def main() -> None:
    api = HfApi()
    folder = DATE_FOLDER.rstrip("/")

    print(f"Listing {HF_REPO}/{folder} (non-recursive)...")
    try:
        tree = list(api.list_repo_tree(repo_id=HF_REPO, path=folder, recursive=False))
    except Exception as exc:
        print(f"HF API error: {exc}")
        sys.exit(1)

    files = [
        {
            "repo": HF_REPO,
            "path": node.path,
            "cdn_url": f"https://huggingface.co/datasets/{HF_REPO}/resolve/main/{node.path}",
        }
        for node in tree
        if node.type == "file"
    ]

    payload = {
        "repo": HF_REPO,
        "date_folder": folder,
        "generated_utc": datetime.now(timezone.utc).isoformat(),
        "files": files,
    }

    os.makedirs(os.path.dirname(os.path.abspath(OUT_PATH)), exist_ok=True)
    with open(OUT_PATH, "w", encoding="utf-8") as f:
        json.dump(payload, f, indent=2)

    print(f"Wrote {len(files)} files to {OUT_PATH}")

if __name__ == "__main__":
    main()
```

```python
#!/usr/bin/env python3
"""
Lightning training entrypoint that uses CDN-only fetches.
Expects ops/file_list.json produced by prepare_training.py.
"""
import json
import os
from pathlib import Path
from typing import Dict, List

import lightning as L
import pyarrow.parquet as pq
import torch
from torch.utils.data import DataLoader, Dataset

HF_REPO = os.getenv("HF_REPO", "datasets/axentx/surrogate-1")
FILE_LIST_PATH = os.getenv("FILE_LIST_PATH", "ops/file_list.json")

class CDNParquetDataset(Dataset):
    def __init__(self, file_list_path: str):
        with open(file_list_path, "r", encoding="utf-8") as f:
            manifest = json.load(f)
        self.files: List[Dict] = [f for f in manifest["files"] if f["path"].endswith(".parquet")]

    def __len__(self) -> int:
        return len(self.files)

    def __getitem__(self, idx: int) -> Dict[str, torch.Tensor]:
        info = self.files[idx]
        # CDN fetch (public file, no auth)
        local_path = str(Path("cache") / Path(info["path"]).name)
        os.makedirs(os.path.dirname(local_path) if os.path.dirname(local_path) else "cache", exist_ok=True)

        if not os.path.exists(local_path):
            # Use raw CDN URL to avoid API calls
            url = info["cdn_url"]
            # Simple download; replace with requests/urllib3 if needed.
            import urllib.request
            urllib.request.urlretrieve(url, local_path)

        # Read only required columns to avoid schema mismatches
        table = pq.read_table(local_path, columns=["prompt", "response"])
        # Minimal projection to tensors; adapt to your tokenizer/format.
        # Here we return raw columns; downstream collate/tokenize as needed.
        return {
            "prompt": table.column("prompt").to_pylist(),
            "response": table.column("response").to_pylist(),
        }

class SurrogateDataModule(L.LightningDataModule):
    def __init__(self, file_list_path: str, batch_size: int = 8):
        super().__init__()
        self.file_list_path = file_list_path
        self.batch_size = batch_size

    def setup(self, stage: str = "") -> None:
        self.ds = CDNParquetDataset(self.file_list_path)

    def train_dataloader(self) -> DataLoader:
        return DataLoader(self.ds, batch_size=self.batch_size, shuffle=True, num_workers=4)

class SurrogateTrainer(L.LightningModule):
    def __init__(self):
        super().__init__()
        self.model = torch.nn.Linear(1, 1)  # placeholder

    def training_step(self, batch, batch_idx):
        # Replace with real training logic
        loss = torch.tensor(0.0, requires_grad=True)
        self.log("train_loss", loss)
        return loss

    def configure_optimizers(self):
        return torch.optim.Adam(self.parameters(), lr=1e-3)

if __name__ == "__main__":
    dm = SurrogateDataModule(file_list_path=FILE_LIST_PATH)
    model = SurrogateTrainer()
    trainer = L.Trainer(max_epochs=1, devices=1)
    trainer.fit(model, dm)
```

#### `/opt/axentx/vanguard/ops/launch_training.sh`
```bash
#!/usr/bin/env bash
# Wrapper: prepare file list → reuse/restart Lightning Studio → run training.
set -euo pipefail
export SHELL=/bin/bash

cd /opt/axentx/vanguard

# 1) Prepare file list (once per date folder; refresh if older than 12h)
if [[ ! -f ops/file_list.json ]] || [[ $(find ops/file_list.json -mmin +720 2>/dev/null || true) ]]; then
  echo "Preparing file list..."
  python3 ops/prepare_training.py
else
  echo "Using existing file_list.json"
fi

# 2) Studio reuse + restart on stopped/idle
python3 - <<'PY'
import lightning as L
import sys

from ops.train_surrogate import SurrogateDataModule, SurrogateTrainer

team = L.Teamspace()
studio_name = "vanguard-surrogate-train"

running = None
for s in team.studios:
    if s.name == studio_name:
        running = s
        break

if running is None:
    print("Creating studio...")
    studio = L.Studio(
        name=studio_name,
        cloud="lightning-public-prod",
        machine="L40S",
        create_ok=True,
    )
elif running.status == "running":
    print(f"Reusing running studio: {studio_name}")
    studio = running
else:
    print(f"Studio stopped; restarting: {studio_name}")
    running.start()
    studio = running

print("Launching training...")
dm = SurrogateDataModule(file_list_path="ops/file_list.json")
model = SurrogateTrainer()
trainer = L.Trainer(max_epochs=1, devices=1)
trainer.fit(model, dm)
PY
```

Make it executable:
