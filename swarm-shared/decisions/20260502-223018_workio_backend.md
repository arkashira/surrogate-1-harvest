# workio / backend

Below is the **single, merged, production-ready implementation** that keeps the strongest, non-contradictory parts of both proposals and prioritizes **correctness + concrete actionability**.

Key resolution choices:
- **Fail-fast signature verification** (C1) kept before any cache/DB I/O.
- **Structured logging + correlation ID** unified into lightweight middleware (C2 pattern) so every span (webhook → service → DB) is traceable.
- **Tenant LRU cache** lives in a small, testable module (C2 pattern) and is used by the webhook (C1 behavior) with a 5m TTL and small max.
- **Idempotency guard** added (C2) to prevent duplicate attendance on LINE retries, using `X-Line-Delivery`/event id + user + type as the idempotency key with a 24h TTL.
- **Graceful degradation**: attendance write wrapped in try/catch; failures are logged and acknowledged as 200 (to stop LINE retries) while pushing to a dead-letter/alert path so events aren’t silently lost.

export const correlationMiddleware = (req: Request, res: Response, next: NextFunction) => {
  const correlationId = (req.headers['x-correlation-id'] as string) || crypto.randomUUID();
  req.id = correlationId;
  res.setHeader('x-correlation-id', correlationId);
  next();

5. Tenant cache module (testable, explicit interface)

`workio/server/src/cache/tenantCache.ts`
```ts
import LRU from 'lru-cache';

type TenantInfo = { tenantId: string };

const tenantCache = new LRU<string, TenantInfo>({
  max: 500,
  ttl: 1000 * 60 * 5, // 5 minutes
});

export function getCachedTenant(channelId: string): string | undefined {
  return tenantCache.get(channelId)?.tenantId;
}

export function setCachedTenant(channelId: string, tenantId: string): void {
  tenantCache.set(channelId, { tenantId });
}
```

// Key: `${channelId}:${userId}:${eventType}:${deliveryId}`
const idempotencyCache = new LRU<string, boolean>({
  max: 10_000,
  ttl: 1000 * 60 * 60 * 24, // 24 hours (LINE retries window)
});

export function isDuplicate(key: string): boolean {
  return idempotencyCache.has(key);
}

export function markProcessed(key: string): void {
  idempotencyCache.set(key, true);
}

7. LINE webhook route (fail-fast, cache-first, idempotent)

`workio/server/src/routes/webhook/line.ts`
```ts
import express, { Request, Response } from 'express';
import crypto from 'crypto';
import { logger } from '../utils/logger';
import { getCachedTenant, setCachedTenant } from '../cache/tenantCache';
import { getTenantByChannelId } from '../services/tenantService';
import { isDuplicate, markProcessed } from '../cache/idempotencyCache';
import { recordAttendance } from '../services/attendanceService';

const router = express.Router();

function verifyLineSignature(rawBody: Buffer, signature: string, secret: string): boolean {
  if (!signature || !secret) return false;
  const expected = crypto.createHmac('sha256', secret).update(rawBody).digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

router.post(
  '/webhook/line',
  express.raw({ type: 'application/json' }),
  async (req: Request, res: Response) => {
    const requestId = req.id || crypto.randomUUID();
    const channelId = process.env.LINE_CHANNEL_ID || '';
    const channelSecret = process.env.LINE_CHANNEL_SECRET || '';
    const signature = (req.headers['x-line-signature'] as string) || '';

    const logMeta = { requestId, channelId };

    // 1) Fail fast: signature
    if (!verifyLineSignature(req.body, signature, channelSecret)) {
      logger.warn({ ...logMeta, msg: 'Invalid LINE signature' });
      return res.status(401).send('Unauthorized');
    }

    // 2) Parse payload
    let event;
    try {
      event = JSON.parse(req.body.toString());
    } catch (err) {
      logger.warn({ ...logMeta, msg: 'Invalid JSON payload', err });
      return res.status(400).send('Bad Request');
    }

    // 3) Resolve tenant (cache-first)
    let tenantId = getCachedTenant(channelId);
    if (!tenantId) {
      try {
        const tenant = await getTenantByChannelId(channelId);
        if (!tenant) {
          logger.warn({ ...logMeta, msg: 'Tenant not found for channel' });
          return res.status(404).send('Tenant not found');
        }
        tenantId = tenant.id;
        setCachedTenant(channelId, tenantId);
        logger.info({ ...logMeta, tenantId, msg: 'Tenant resolved and cached' });
      } catch (err) {
        logger.error({ ...logMeta, msg: 'Failed to resolve tenant', err });
        return res.status(500).send('Internal Server Error');
      }
    }

    // 4) Idempotency guard for attendance events
    const deliveryId = (event as any)?.deliveryId || (event as any)?.events?.[0]?.source?.userId
      ? `${channelId}:${(event as any).events?.[0]?.source?.userId}:${(event as any).events?.[0]?.type}:${deliveryIdFallback(event)}`
      : null;

    function deliveryIdFallback(e: any): string {
      // Best-effort stable id for dedupe: event-level or top-level
      return e.events?.[0]?.eventId || e.deliveryId || crypto.createHash('sha256').update(JSON.stringify(e)).digest('hex').slice(0, 16);
    }

      logger.info({ ...logMeta, tenantId, deliveryId, msg: 'Duplicate webhook ignored (idempotency)' });
      return res.status(200).json({ status: 'duplicate' });
    }

      channelId,

      deliveryId,
