# vanguard / design

# Final Synthesis & Actionable Plan

## 1. Diagnosis (merged, de-duplicated)
- **No persisted manifest**: every training run performs authenticated `list_repo_tree`, burning HF API quota and risking 429s.
- **Heterogeneous schemas**: training/data loader uses `load_dataset` or per-file streaming across repos, causing `pyarrow.CastError` on schema mismatch.
- **No CDN-only fetch path**: authenticated API calls during training amplify rate-limit exposure instead of bypassing via CDN.
- **HF commit cap exposure**: no deterministic repo selection for HF’s 128 commits/hr/repo limit; ingestion stalls when a single repo hits the cap.
- **Lightning Studio waste**: no reuse/idle-restart guard; iterations recreate studios and burn 80+ hrs/mo quota.
- **Training robustness gaps**: missing progressive backoff, per-file schema isolation, checkpointing, and validation.

## 2. Proposed Change (merged + hardened)
Create **two canonical files** under `/opt/axentx/vanguard/design/` and wire the launcher:

1. `/opt/axentx/vanguard/design/manifest.py`  
   - One-time authenticated `list_repo_tree` (with recursion control and pagination).  
   - Deterministic sibling-repo selector to spread HF commit cap.  
   - Manifest schema: `{repo, path, cdn_url, etag?, size?}`.

2. `/opt/axentx/vanguard/design/train.py`  
   - CDN-only dataset with schema-isolated, best-effort projection to `{prompt, response}`.  
   - Per-file schema isolation (via `datasets` for non-JSONL) to prevent `pyarrow.CastError`.  
   - Lightning Studio reuse + idle-restart guard.  
   - Training loop with checkpointing, validation split, and HF commit cap–aware artifact writes.

Launcher changes:
- Add `--manifest` and `--reuse-studio` flags.
- Default to CDN-only mode; authenticated calls only during manifest build.

```bash
# /opt/axentx/vanguard/design/manifest.py
import json
import hashlib
from pathlib import Path
from typing import List, Dict
from huggingface_hub import HfApi

HF_API = HfApi()

def _hash_slug(slug: str) -> int:
    return int(hashlib.sha256(slug.encode()).hexdigest(), 16)

def build_manifest(
    repo: str,
    folder: str,
    out_path: str,
    recursive: bool = True,
    max_per_page: int = 1000,
) -> Path:
    """
    One-time authenticated tree listing.
    Saves: [{"repo": repo, "path": path, "cdn_url": url}, ...]
    """
    entries: List[Dict[str, str]] = []
    cursor = None
    while True:
        tree = HF_API.list_repo_tree(
            repo=repo,
            path=folder,
            recursive=recursive,
            max_per_page=max_per_page,
            cursor=cursor,
        )
        for item in tree:
            if item.type == "file":
                entries.append({
                    "repo": repo,
                    "path": item.path,
                    "cdn_url": f"https://huggingface.co/datasets/{repo}/resolve/main/{item.path}"
                })
        cursor = getattr(tree, "cursor", None)
        if not cursor:
            break

    out = Path(out_path)
    out.parent.mkdir(parents=True, exist_ok=True)
    with open(out, "w") as f:
        json.dump(entries, f, indent=2)
    return out

def pick_sibling_repo(base_repo: str, slug: str, n_siblings: int = 5) -> str:
    """Deterministic sibling repo to spread HF commit cap (128/hr/repo)."""
    idx = _hash_slug(slug) % n_siblings
    return f"{base_repo}-{idx}" if idx > 0 else base_repo
```

class CDNTextDataset(IterableDataset):
    """
    CDN-only dataset with schema isolation and best-effort projection.
    Avoids authenticated HF API calls during training.
    """
    def __init__(self, manifest_path: str, tokenizer, max_length: int = 2048):
        super().__init__()
        with open(manifest_path) as f:
            self.files = [e["cdn_url"] for e in json.load(f)]
        self.tokenizer = tokenizer
        self.max_length = max_length

    @staticmethod
    def _project_record(raw: dict) -> dict:
        prompt = raw.get("prompt") or raw.get("input") or raw.get("text") or ""
        response = raw.get("response") or raw.get("output") or raw.get("completion") or ""
        return {"prompt": str(prompt), "response": str(response)}

    def _tokenize_pair(self, pair: dict):
        text = f"{pair['prompt']}\n\n{pair['response']}"
        enc = self.tokenizer(
            text,
            truncation=True,
            max_length=self.max_length,
            padding="max_length",
        )
        return {k: torch.tensor(v, dtype=torch.long) for k, v in enc.items()}

    def _stream_jsonl(self, url: str):
        r = requests.get(url, timeout=30)
        r.raise_for_status()
        for line in r.text.splitlines():
            line = line.strip()
            if not line:
                continue
            try:
                raw = json.loads(line)
                pair = self._project_record(raw)
                if pair["prompt"] and pair["response"]:
                    yield self._tokenize_pair(pair)
            except Exception:
                continue

    def _stream_via_datasets(self, url: str):
        # Use HF datasets for parquet/json to avoid pyarrow.CastError across files
        try:
            ds = load_dataset("json", data_files=url, split="train", streaming=True)
            for raw in ds:
                pair = self._project_record(raw)
                if pair["prompt"] and pair["response"]:
                    yield self._tokenize_pair(pair)
        except Exception:
            pass

    def __iter__(self):
        for url in self.files:
            try:
                if url.endswith(".jsonl"):
                    yield from self._stream_jsonl(url)
                else:
                    yield from self._stream_via_datasets(url)
            except Exception:
                continue

def get_lightning_studio(name: str, machine=L.Machine.L40S):
    """Reuse running studio; restart if idle-stopped."""
    ts = Teamspace.current()
    for s in ts.studios:
        if s.name == name:
            if s.status == "running":
                return s
            s.start(machine=machine)
            return s
    return L.Studio(
        name=name,
        machine=machine,
        dependencies=["torch", "transformers", "datasets", "requests"],
        create_ok=True,
    )
