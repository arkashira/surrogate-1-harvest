# workio / frontend

## 1. Diagnosis
- No optimistic UI for clock-in/out → users tap multiple times waiting for LINE round-trip, creating duplicate intents and visual jank.
- No local pending state or request dedupe key → duplicate submissions can race and corrupt daily totals even if backend is idempotent.
- No visual feedback on GPS acquisition or LINE send progress → users unsure if action registered.
- Clock-in/out button does not reflect current day state (already clocked in/out) after page refresh or re-mount → stale UI.
- No lightweight retry/backoff for transient LINE failures → silent drop on flaky networks.

## 2. Proposed change
File: `workio/src/components/AttendanceButton.tsx` (create) and integrate into main dashboard view.  
Scope: single component + one hook (`useAttendance`) encapsulating optimistic state, request dedupe, and retry. Minimal footprint (<2h).

## 3. Implementation

### `workio/src/hooks/useAttendance.ts`
```ts
import { useState, useCallback, useRef } from 'react';

type AttendanceState = 'in' | 'out' | null;
type Status = 'idle' | 'pending' | 'success' | 'error';

const RETRY_DELAYS = [1000, 3000, 5000]; // ms

export function useAttendance() {
  const [state, setState] = useState<AttendanceState>(null);
  const [status, setStatus] = useState<Status>('idle');
  const [error, setError] = useState<string | null>(null);
  const inflightRef = useRef<string | null>(null); // dedupe key: YYYY-MM-DD-HH-mm-ss
  const retryCountRef = useRef(0);

  // Fetch current day state on mount
  const fetchToday = useCallback(async () => {
    try {
      const res = await fetch('/api/attendance/today');
      if (!res.ok) throw new Error('Failed to fetch today');
      const data = await res.json();
      setState(data.clockedIn ? 'in' : 'out');
    } catch {
      // fail silently; keep previous state
    }
  }, []);

  const send = useCallback(
    async (target: AttendanceState) => {
      const now = new Date();
      const dedupeKey = now.toISOString().slice(0, 19).replace(/[-:T]/g, ''); // YYYYMMDDHHmmss
      if (inflightRef.current === dedupeKey) return; // dedupe rapid double-taps
      inflightRef.current = dedupeKey;
      setStatus('pending');
      setError(null);

      // Optimistic update
      setState(target);

      const attempt = async (retryIdx = 0): Promise<void> => {
        try {
          const res = await fetch('/api/attendance/clock', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ action: target === 'in' ? 'clock_in' : 'clock_out', dedupeKey }),
          });

          if (!res.ok) {
            const msg = await res.text().catch(() => `Server error ${res.status}`);
            throw new Error(msg || 'Request failed');
          }

          retryCountRef.current = 0;
          inflightRef.current = null;
          setStatus('success');
          // Small UX confirmation window then settle
          setTimeout(() => setStatus('idle'), 1500);
        } catch (err: any) {
          if (retryIdx < RETRY_DELAYS.length) {
            setTimeout(() => attempt(retryIdx + 1), RETRY_DELAYS[retryIdx]);
            return;
          }
          // Exhausted retries — revert optimistic state and show error
          setState((prev) => (prev === target ? (target === 'in' ? 'out' : 'in') : prev));
          setError(err.message || 'Network error');
          setStatus('error');
          retryCountRef.current = 0;
          inflightRef.current = null;
        }
      };

      await attempt();
    },
    []
  );

  return { state, status, error, fetchToday, send };
}
```

### `workio/src/components/AttendanceButton.tsx`
```tsx
import React, { useEffect } from 'react';
import { useAttendance } from '../hooks/useAttendance';

export function AttendanceButton() {
  const { state, status, error, fetchToday, send } = useAttendance();

  useEffect(() => {
    fetchToday();
  }, [fetchToday]);

  const handleClick = () => {
    if (status === 'pending') return;
    send(state === 'in' ? 'out' : 'in');
  };

  const isPending = status === 'pending';
  const isError = status === 'error';

  return (
    <div className="flex flex-col items-center gap-3">
      <button
        onClick={handleClick}
        disabled={isPending}
        className={`
          px-8 py-4 rounded-xl font-semibold text-lg shadow-lg

            </svg>

          </>
        )}
      </button>
