# workio / quality

## 1. Diagnosis

- No resilient offline-first clock-in/out flow: optimistic UI exists in plans but not implemented; users lose punches on flaky mobile networks.
- No idempotency for LINE webhook/clock events: duplicate punches possible on retries (LINE may redeliver webhooks).
- No client-side validation/sync status: UI can’t distinguish “pending” vs “synced” vs “failed” for punches, causing confusion.
- No durable offline queue: localStorage-only drafts are fragile and size-limited; IndexedDB not used.
- No GPS pre-check before attempting punch: users can attempt clock-in then fail on location, wasting time and battery.

## 2. Proposed change

File: `workio/src/hooks/usePunch.ts` (new) + `workio/src/components/ClockButton.tsx` (modify) + `workio/src/lib/offlineQueue.ts` (new)  
Scope: implement optimistic UI + idempotent mutation + IndexedDB offline queue + GPS pre-check for clock-in/out.

### 3.2 Create offline queue (IndexedDB)

`src/lib/offlineQueue.ts`

```ts
import { openDB, DBSchema, IDBPDatabase } from 'idb';

interface PunchQueueItem {
  id: string; // idempotency key
  type: 'clock-in' | 'clock-out';
  timestamp: number;
  coords?: { latitude: number; longitude: number };
  retries: number;
}

interface WorkioDB extends DBSchema {
  punches: {
    key: string;
    value: PunchQueueItem;
    indexes: { 'by-type': string };
  };
}

const DB_NAME = 'workio-offline';
const DB_VERSION = 1;

async function getDB(): Promise<IDBPDatabase<WorkioDB>> {
  return openDB<WorkioDB>(DB_NAME, DB_VERSION, {
    upgrade(db) {
      const store = db.createObjectStore('punches', { keyPath: 'id' });
      store.createIndex('by-type', 'type');
    },
  });
}

export async function enqueuePunch(item: Omit<PunchQueueItem, 'retries'>) {
  const db = await getDB();
  await db.add('punches', { ...item, retries: 0 });
}

export async function dequeuePunch(id: string) {
  const db = await getDB();
  await db.delete('punches', id);
}

export async function peekPendingPunches(): Promise<PunchQueueItem[]> {
  const db = await getDB();
  return db.getAll('punches');
}

export async function markRetry(id: string, retries: number) {
  const db = await getDB();
  const item = await db.get('punches', id);
  if (item) {
    item.retries = retries;
    await db.put('punches', item);
  }
}
```

### 3.3 Create resilient punch hook

`src/hooks/usePunch.ts`

```ts
import { useState, useCallback, useEffect } from 'react';
import { enqueuePunch, dequeuePunch, peekPendingPunches, markRetry } from '../lib/offlineQueue';

const MAX_RETRIES = 5;
const RETRY_BACKOFF = [1000, 3000, 10000, 30000, 60000]; // ms

type PunchStatus = 'idle' | 'pending' | 'synced' | 'failed';

export function usePunch() {
  const [status, setStatus] = useState<PunchStatus>('idle');
  const [lastPunch, setLastPunch] = useState<{ id: string; type: 'clock-in' | 'clock-out'; ts: number } | null>(null);

  // Process queue on mount + periodically
  useEffect(() => {
    let mounted = true;
    async function processQueue() {
      const items = await peekPendingPunches();
      for (const item of items) {
        if (!mounted) return;
        try {
          await sendPunchToServer(item);
          await dequeuePunch(item.id);
        } catch (err) {
          const nextRetries = item.retries + 1;
          if (nextRetries >= MAX_RETRIES) {
            // keep item for manual recovery; mark high retry count
            await markRetry(item.id, nextRetries);
          } else {
            await markRetry(item.id, nextRetries);
            const delay = RETRY_BACKOFF[Math.min(nextRetries, RETRY_BACKOFF.length - 1)];
            setTimeout(processQueue, delay);
            return; // stop processing this tick after failure
          }
        }
      }
    }

    processQueue();
    const interval = setInterval(processQueue, 30000);
    return () => {
      mounted = false;
      clearInterval(interval);
    };
  }, []);

  const sendPunchToServer = useCallback(async (item: { id: string; type: 'clock-in' | 'clock-out'; timestamp: number; coords?: { latitude: number; longitude: number } }) => {
    const res = await fetch('/api/punches', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        idempotencyKey: item.id,
        type: item.type,
        timestamp: item.timestamp,
        coords: item.coords,
      }),
    });
    if (!res.ok) throw new Error(`Punch failed: ${res.status}`);
    return res.json();
  }, []);

  const punch = useCallback(async (type: 'clock-in' | 'clock-out', coords?: { latitude: number; longitude: number }) => {
    const id = `${type}-${Date.now()}-${crypto.randomUUID()}`;
    const ts = Date.now();

    // optimistic update
    setStatus('pending');
    setLastPunch({ id, type, ts });

    // try immediate send
    try {
      await sendPunchToServer({ id, type, timestamp: ts, coords });
      setStatus('synced');
      // no queue entry to remove (wasn't queued)
    } catch (err) {
      // persist to offline queue for retry
      await enqueuePunch({ id, type, timestamp: ts, coords });
      setStatus('failed');
    }
  }, [sendPunchToServer]);

  return { status, lastPunch, punch };
}
```

### 3.4 Add GPS pre-check utility

`src/lib/location.ts`

```ts
export async function getCurrentLocation(timeoutMs = 10000): Promise<{ latitude: number; longitude: number }> {
  return new Promise((resolve, reject) => {
    if (!navigator.geolocation) {
      reject(new Error('Geolocation not supported'));
      return;
    }
    navigator.geolocation.getCurrentPosition(
      (pos) => resolve({ latitude: pos.coords.latitude, longitude: pos.coords.longitude }),
      (err) => reject(err),
      { enableHighAccuracy: true, timeout: timeoutMs, maximumAge: 30000 }
    );
  });
}
```
