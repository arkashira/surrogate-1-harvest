## 1) scripts/list_hf_files.py

```python
#!/usr/bin/env python3
"""
List files in a single date folder for a HuggingFace dataset repo.
Run from Mac (or any machine) when API limits are clear.

Usage:
  python list_hf_files.py \
    --repo "org/surrogate-dataset" \
    --date "2026-04-29" \
    --out "file_list.json"

Outputs:
  {
    "repo": "org/surrogate-dataset",
    "date": "2026-04-29",
    "files": [
      "batches/mirror-merged/2026-04-29/abc123.parquet",
      ...
    ]
  }
"""

import argparse
import json
import os
import sys
import time

try:
    from huggingface_hub import HfApi, HfFolder
except ImportError:
    print("Install: pip install huggingface_hub")
    sys.exit(1)

RATE_LIMIT_RETRY = 360  # seconds after 429

def list_date_files(repo_id: str, date_folder: str, out_path: str) -> None:
    api = HfApi()
    # Use non-recursive per-folder listing to avoid 100x pagination on big repos.
    try:
        items = api.list_repo_tree(
            repo_id=repo_id,
            path=date_folder,
            repo_type="dataset",
            recursive=False,
        )
    except Exception as e:
        if getattr(e, "response", None) and e.response.status_code == 429:
            print(f"Rate limited (429). Waiting {RATE_LIMIT_RETRY}s before retry.")
            time.sleep(RATE_LIMIT_RETRY)
            items = api.list_repo_tree(
                repo_id=repo_id,
                path=date_folder,
                repo_type="dataset",
                recursive=False,
            )
        else:
            raise

    files = sorted(
        f"{date_folder}/{item.path.split('/')[-1]}"
        for item in items
        if item.type == "file" and item.path.lower().endswith(".parquet")
    )

    payload = {
        "repo": repo_id,
        "date": date_folder,
        "files": files,
    }

    os.makedirs(os.path.dirname(os.path.abspath(out_path)), exist_ok=True)
    with open(out_path, "w") as f:
        json.dump(payload, f, indent=2)
    print(f"Wrote {len(files)} files to {out_path}")

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="List HF dataset files for one date folder.")
    parser.add_argument("--repo", required=True, help="HF dataset repo, e.g. org/surrogate-dataset")
    parser.add_argument("--date", required=True, help="Date folder, e.g. 2026-04-29")
    parser.add_argument("--out", default="file_list.json", help="Output JSON path")
    args = parser.parse_args()
    list_date_files(args.repo, args.date, args.out)
```

## 2) surrogate/train.py

```python
#!/usr/bin/env python3
"""
CDN-only parquet loader + Lightning idle-resilient surrogate training runner.

Key behaviors:
- Zero HuggingFace API calls during training (CDN-only downloads).
- Deterministic sibling repo selection for HF commit-cap scaling.
- Survives Lightning idle timeouts by checking studio status and restarting.
- Projects only {prompt, response}; attribution via filename pattern.
"""

import hashlib
import json
import os
import sys
import time
from pathlib import Path
from typing import Dict, Iterable, List

import pyarrow.parquet as pq
import requests
import torch
from torch.utils.data import IterableDataset, DataLoader

# Optional Lightning import (install if needed)
try:
    from lightning import Fabric, LightningModule, Trainer
    from lightning.fabric.plugins import LightningPlugin
    from lightning.pytorch.strategies import DDPStrategy
except ImportError:
    print("Install: pip install lightning")
    LIGHTNING_AVAILABLE = False
else:
    LIGHTNING_AVAILABLE = True

# ---------- Config ----------
HF_DATASET_REPO = os.getenv("HF_DATASET_REPO", "org/surrogate-dataset")
DATE_FOLDER = os.getenv("DATE_FOLDER", "2026-04-29")
FILE_LIST_PATH = os.getenv("FILE_LIST_PATH", "scripts/file_list.json")
SIBLING_REPOS = int(os.getenv("SIBLING_REPOS", "5"))  # 5 siblings => 640 commits/hr aggregate
CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"
BATCH_SIZE = int(os.getenv("BATCH_SIZE", "8"))
MAX_STEPS = int(os.getenv("MAX_STEPS", "1000"))
LIGHTNING_STUDIO_NAME = os.getenv("LIGHTNING_STUDIO_NAME", "surrogate-train")
IDLE_CHECK_INTERVAL = int(os.getenv("IDLE_CHECK_INTERVAL", "300"))  # seconds
# ----------

def pick_sibling_repo(slug: str) -> str:
    """Deterministic sibling repo selection based on slug hash."""
    h = int(hashlib.sha256(slug.encode()).hexdigest(), 16)
    idx = h % SIBLING_REPOS
    if idx == 0:
        return HF_DATASET_REPO
    return f"{HF_DATASET_REPO}-sibling{idx}"

def cdn_parquet_urls(file_list: List[str]) -> List[str]:
    """Convert repo-relative file paths into CDN URLs (no auth)."""
    urls = []
    for rel in file_list:
        repo = pick_sibling_repo(rel)
        urls.append(CDN_TEMPLATE.format(repo=repo, path=rel))
    return urls

class CDNParquetIterable(IterableDataset):
    """
    Stream parquet files via CDN URLs.
    Projects to {prompt, response} only.
    """

    def __init__(self, urls: List[str], shuffle: bool = True):
        self.urls = urls if not shuffle else urls.copy()
        self.shuffle = shuffle

    def __iter__(self):
        urls = self.urls
        if self.shuffle:
            urls = urls.copy()
            # Simple deterministic-ish shuffle per worker via torch seed
            g = torch.Generator()
            g.manual_seed(torch.initial_seed() % (2**32))
            idx = torch.randperm(len(urls), generator=g).tolist()
            urls = [urls[i] for i in idx]

        for url in urls:
            try:
                resp = requests.get(url, timeout=60)
                resp.raise_for_status()
            except Exception as exc:
                print(f"Failed to fetch {url}: {exc}", file=sys.stderr)
