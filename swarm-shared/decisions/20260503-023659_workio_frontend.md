# workio / frontend

## Final Synthesis & Actionable Implementation

**Chosen approach**: merge Candidate 1’s optimistic UI + idempotency plumbing with Candidate 2’s queue-first/offline-first behavior, geofence pre-check, and strict transition guards. Contradictions resolved in favor of **correctness + concrete actionability**.

---

## 1. Diagnosis (resolved)

- **Duplicate requests** → rapid clicks + retries before previous promise settles.  
  Fix: client-side dedupe key + optimistic disabled state + queue.
- **No idempotency propagation** → backend can’t reliably dedupe retries/LINE replays.  
  Fix: generate stable idempotency key per logical punch; include `X-Idempotency-Key` header; reuse LINE message ID when available.
- **Invalid transitions** → users attempt clock-in while already clocked-in.  
  Fix: derive expected next punch from server truth; block invalid actions in UI.
- **GPS blocks UI + flaky/slow acquisition** → synchronous getCurrentPosition on click causes timeouts.  
  Fix: best-effort cached location + background refresh; allow punch without GPS if policy permits; non-blocking acquisition.
- **Offline failures** → punches lost on network loss.  
  Fix: local-first “pending punches” queue with exponential backoff + replay.

```ts
export function makeIdempotencyKey(employeeId: string, type: 'clock_in' | 'clock_out', lineMsgId?: string) {
  // Prefer stable LINE message ID when present; otherwise time-based + random suffix.
  const stable = lineMsgId ? `line-${lineMsgId}` : `local-${employeeId}-${type}-${Date.now()}-${crypto.randomUUID()}`;
  return stable;
}
```

### 3.2 Geolocation utilities (non-blocking, cached)

```ts
// workio/src/features/punch/utils/geo.ts
type Coord = { latitude: number; longitude: number } | null;

let cached: Coord = null;
let lastFetch = 0;
const CACHE_TTL = 5 * 60 * 1000; // 5m

export async function getLocationCached(): Promise<Coord> {
  const now = Date.now();
  if (cached && now - lastFetch < CACHE_TTL) return cached;

  return new Promise((resolve) => {
    navigator.geolocation.getCurrentPosition(
      (pos) => {
        cached = { latitude: pos.coords.latitude, longitude: pos.coords.longitude };
        lastFetch = Date.now();
        resolve(cached);
      },
      () => {
        // Don't block punch on GPS failure; resolve null and let server/backend policy decide.
        resolve(null);
      },
      { enableHighAccuracy: true, timeout: 10000, maximumAge: 30000 }
    );
  });
}
```

```ts
import axios from 'axios';

const api = axios.create({ baseURL: '/api' });

export type PunchType = 'clock_in' | 'clock_out';

export interface PunchRequest {
  employeeId: string;
  type: PunchType;
  latitude?: number | null;
  longitude?: number | null;
  idempotencyKey: string;
  lineMessageId?: string;
}

export interface PunchResponse {
  id: string;
  employeeId: string;
  type: PunchType;
  timestamp: string;
  latitude?: number | null;
  longitude?: number | null;
}

export async function submitPunch(payload: PunchRequest): Promise<PunchResponse> {
  const res = await api.post<PunchResponse>('/punches', payload, {
    headers: { 'Idempotency-Key': payload.idempotencyKey },
  });
  return res.data;
}

export async function fetchLastPunch(employeeId: string): Promise<PunchResponse | null> {
  try {
    const res = await api.get<PunchResponse>(`/punches/last?employeeId=${encodeURIComponent(employeeId)}`);
    return res.data;
  } catch {
    return null;
  }
}
```

### 3.4 Pending queue (offline-first)

```ts
// workio/src/features/punch/store/pendingQueue.ts
import { submitPunch, type PunchRequest } from './api/punch';

const STORAGE_KEY = 'pending-punches-v1';

type QueueItem = {
  payload: PunchRequest;
  attempts: number;
  maxAttempts: number;
  addedAt: number;
};

class PendingQueue {
  private queue: QueueItem[] = [];
  private processing = false;
  private listeners: (() => void)[] = [];

  constructor() {
    this.load();
    window.addEventListener('online', () => this.process());
  }

  private load() {
    try {
      const raw = localStorage.getItem(STORAGE_KEY);
      this.queue = raw ? JSON.parse(raw) : [];
    } catch {
      this.queue = [];
    }
  }

  private save() {
    try {
      localStorage.setItem(STORAGE_KEY, JSON.stringify(this.queue));
    } catch {}
    this.listeners.forEach((l) => l());
  }

  subscribe(cb: () => void) {
    this.listeners.push(cb);
    return () => {
      this.listeners = this.listeners.filter((l) => l !== cb);
    };
  }

  add(payload: PunchRequest) {
    // Avoid adding obvious duplicates already in-flight in queue (same idempotencyKey)
    if (this.queue.some((q) => q.payload.idempotencyKey === payload.idempotencyKey)) return;
    this.queue.push({ payload, attempts: 0, maxAttempts: 5, addedAt: Date.now() });
    this.save();
    this.process();
  }

  private async process() {
    if (this.processing || this.queue.length === 0) return;
    if (!navigator.onLine) return;

    this.processing = true;
    const item = this.queue[0];

    try {
      await submitPunch(item.payload);
      // success: remove item
      this.queue.shift();
      this.save();
      // continue processing
      this.processing = false;
      this.process();
    } catch (err) {
      item.attempts += 1;
      if (item.attempts >= item.maxAttempts) {
        // too many failures: drop and notify via UI elsewhere
        this.queue.shift();
      } else {
        // exponential backoff before retry
        const delay = Math.min(1000 * 2 ** item.attempts, 30000);
        setTimeout(() => {
          this.processing = false;
          this.process();
        }, delay);
        return;
      }
      this.save();
      this.processing = false;
      this.process();
    }
  }

  getItems() {
    return this.queue.slice();
  }
}

export const pendingQueue = new PendingQueue();
```
