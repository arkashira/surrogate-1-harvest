## Final synthesized implementation (correct + actionable)

### 1. Diagnosis (resolved)
- **Missing idempotency** on `/webhook/line` → LINE retries create duplicate punches.
- **No partial unique constraint** for open punches → double-clock-in possible.
- **No transactional upsert / ON CONFLICT handling** → race conditions under load.
- **Missing signature + replay-window validation** → delayed/replayed requests accepted.
- **No observability/alerting** for duplicates or constraint violations.

**Resolution priority**: DB-level guarantees first (constraints + transactional upsert), then idempotency + signature + observability. Use DB-backed idempotency (no Redis) to keep ownership simple and strongly consistent.

#### B) Webhook handler update
File: `/opt/axentx/workio/server/src/routes/webhook/line.ts`

Key choices:
- Use **DB transaction** per batch of events with `BEGIN/COMMIT/ROLLBACK`.
- Use **`INSERT ... ON CONFLICT DO NOTHING`** for idempotency table, then proceed only if inserted (or already present).
- Use **`INSERT ... ON CONFLICT DO UPDATE`** (or upsert pattern) for punches where appropriate.
- Keep **partial unique index** as source-of-truth for open-punch invariant.
- Validate **signature + short replay window** (5 min) using `x-line-signature` and event timestamps.
- Log **requestId/idempotencyKey** and expose metrics counters (duplicates, constraint violations).

```ts
// /opt/axentx/workio/server/src/routes/webhook/line.ts
import { Router, Request, Response } from 'express';
import { Pool, PoolClient } from 'pg';
import crypto from 'crypto';

const router = Router();
const pool = new Pool({ connectionString: process.env.DATABASE_URL });

const LINE_SIGNATURE_HEADER = 'x-line-signature';
const MAX_REPLAY_WINDOW_MS = 5 * 60 * 1000; // 5 minutes

function verifyLineSignature(body: string, signature: string, channelSecret: string): boolean {
  if (!signature || !channelSecret) return false;
  const expected = crypto.createHmac('sha256', channelSecret).update(body, 'utf8').digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

function parsePunchType(text: string): 'in' | 'out' | null {
  const t = text.toLowerCase();
  if (t.includes('in') || t.includes('clock') || t.includes('มา')) return 'in';
  if (t.includes('out') || t.includes('leave') || t.includes('กลับ')) return 'out';
  return null;
}

async function upsertPunch(
  tx: PoolClient,
  userId: string,
  eventTime: Date,
  punchType: 'in' | 'out',
  idempotencyKey: string
): Promise<string> {
  // 1) Idempotency check (fast)
  const idem = await tx.query(
    `SELECT punch_id FROM line_webhook_idempotency WHERE idempotency_key = $1`,
    [idempotencyKey]
  );
  if (idem.rows.length > 0 && idem.rows[0].punch_id) {
    return idem.rows[0].punch_id;
  }

  // 2) Try insert with ON CONFLICT handling for open punch constraint
  if (punchType === 'in') {
    // Close any stale open punch defensively (same user) then insert new
    await tx.query(
      `UPDATE punches
       SET clock_out_at = $1, updated_at = NOW()
       WHERE user_id = $2 AND clock_out_at IS NULL`,
      [eventTime, userId]
    );

    const res = await tx.query(
      `INSERT INTO punches (user_id, clock_in_at, clock_out_at, idempotency_key, created_at, updated_at)
       VALUES ($1, $2, NULL, $3, NOW(), NOW())
       RETURNING id`,
      [userId, eventTime, idempotencyKey]
    );
    const punchId = res.rows[0].id;

    await tx.query(
      `INSERT INTO line_webhook_idempotency (idempotency_key, line_event_id, user_id, punch_id)
       VALUES ($1, $1, $2, $3)
       ON CONFLICT (idempotency_key) DO UPDATE SET punch_id = EXCLUDED.punch_id`,
      [idempotencyKey, userId, punchId]
    );
    return punchId;
  } else {
    // Clock out: close latest open punch
    const up = await tx.query(
      `UPDATE punches
       SET clock_out_at = $1, updated_at = NOW()
       WHERE user_id = $2 AND clock_out_at IS NULL
       RETURNING id`,
      [eventTime, userId]
    );

    if (up.rowCount > 0) {
      const punchId = up.rows[0].id;
      await tx.query(
        `INSERT INTO line_webhook_idempotency (idempotency_key, line_event_id, user_id, punch_id)
         VALUES ($1, $1, $2, $3)
         ON CONFLICT (idempotency_key) DO UPDATE SET punch_id = EXCLUDED.punch_id`,
        [idempotencyKey, userId, punchId]
      );
      return punchId;
    } else {
      // No open punch: create a closed punch record to preserve idempotency (defensive)
      const ins = await tx.query(
        `INSERT INTO punches (user_id, clock_in_at, clock_out_at, idempotency_key, created_at, updated_at)
         VALUES ($1, $1, $1, $2, NOW(), NOW())
         RETURNING id`,
        [userId, idempotencyKey]
      );
      const punchId = ins.rows[0].id;
      await tx.query(
        `INSERT INTO line_webhook_idempotency (idempotency_key, line_event_id, user_id, punch_id)
         VALUES ($1, $1, $2, $3)
         ON CONFLICT (idempotency_key) DO UPDATE SET punch_id = EXCLUDED.punch_id`,
        [idempotencyKey, userId, punchId]
      );
      return punchId;
    }
  }
}
