# workio / design

## 1. Diagnosis

- **Missing idempotency layer**: LINE webhook retries and client double-taps can deliver the same `webhookEventId` multiple times, causing duplicate clock-in/out or leave/OT transitions and corrupting daily totals.
- **No signature time-window enforcement**: Accepts stale or replayed `X-Line-Signature` beyond a safe validity window, increasing spoof/replay risk.
- **Non-transactional state updates**: Clock/leave/OT writes are not atomic with idempotency checks — partial failures can leave totals inconsistent.
- **No strict event deduplication key**: Relying on downstream uniqueness instead of a first-class `webhook_event_idempotency` table invites race conditions under concurrency.
- **Missing operational telemetry**: No lightweight audit trail for webhook ingest (received, signature valid, idempotent key, applied) to debug duplicates or replays.

## 2. Proposed change

- **File**: `workio/server/src/services/webhookService.ts` (create if absent)  
  **Lines**: new module; integrate into existing webhook handler route (`/webhook/line`).  
- **File**: `workio/server/src/db/schema.sql`  
  **Lines**: add `webhook_event_idempotency` table + index.  
- **File**: `workio/server/src/routes/lineWebhookRoute.ts` (or equivalent)  
  **Lines**: wrap handler with idempotency + signature time-window check before business logic.

## 3. Implementation

### 3.1 DB schema (idempotency table)

```sql
-- workio/server/src/db/schema.sql
-- Add near other metadata tables
CREATE TABLE IF NOT EXISTS webhook_event_idempotency (
  webhook_event_id      VARCHAR(255) PRIMARY KEY,
  received_at           TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  applied_at            TIMESTAMPTZ,
  event_type            VARCHAR(100) NOT NULL,
  source_user_id        VARCHAR(255),
  handled_payload_hash  BYTEA,
  status                VARCHAR(50) NOT NULL DEFAULT 'received' -- received, applied, skipped, rejected
);

-- Fast lookup by status and age (cleanup / replay checks)
CREATE INDEX IF NOT EXISTS idx_webhook_idempotency_status_received
  ON webhook_event_idempotency (status, received_at);
```

### 3.2 Idempotency + signature-time service

```ts
// workio/server/src/services/webhookService.ts
import crypto from 'crypto';
import { Pool } from 'pg';

const pool = new Pool({ connectionString: process.env.DATABASE_URL });
const SIGNATURE_TTL_MS = 5 * 60 * 1000; // 5 minutes

export interface LineWebhookBody {
  destination: string;
  events: Array<{
    type: string;
    mode: string;
    timestamp: number;
    source: { userId: string; type: string };
    webhookEventId: string;
    replyToken?: string;
    message?: any;
    postback?: any;
    things?: any;
  }>;
}

export async function verifyLineSignature(rawBody: string, signature: string): Promise<boolean> {
  if (!signature || !process.env.LINE_CHANNEL_SECRET) return false;
  const expected = crypto
    .createHmac('sha256', process.env.LINE_CHANNEL_SECRET)
    .update(rawBody)
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

export function isSignatureFresh(timestampMs: number): boolean {
  const now = Date.now();
  return Math.abs(now - timestampMs) <= SIGNATURE_TTL_MS;
}

export async function recordWebhookEvent(
  webhookEventId: string,
  eventType: string,
  sourceUserId: string,
  payloadHash: Buffer,
  status: 'received' | 'applied' | 'skipped' | 'rejected' = 'received'
) {
  await pool.query(
    `INSERT INTO webhook_event_idempotency
       (webhook_event_id, event_type, source_user_id, handled_payload_hash, status)
     VALUES ($1,$2,$3,$4,$5)
     ON CONFLICT (webhook_event_id) DO UPDATE
       SET status = EXCLUDED.status,
           applied_at = CASE WHEN EXCLUDED.status = 'applied' THEN NOW() ELSE applied_at END`,
    [webhookEventId, eventType, sourceUserId, payloadHash, status]
  );
}

export async function isDuplicateOrProcessed(webhookEventId: string): Promise<boolean> {
  const res = await pool.query(
    `SELECT status FROM webhook_event_idempotency WHERE webhook_event_id = $1`,
    [webhookEventId]
  );
  if (res.rowCount === 0) return false;
  // If previously applied or skipped, treat as duplicate
  return ['applied', 'skipped', 'rejected'].includes(res.rows[0].status);
}

export async function markApplied(webhookEventId: string) {
  await pool.query(
    `UPDATE webhook_event_idempotency
     SET status = 'applied', applied_at = NOW()
     WHERE webhook_event_id = $1`,
    [webhookEventId]
  );
}
```

### 3.3 Route integration (atomic tx + idempotency)

```ts
// workio/server/src/routes/lineWebhookRoute.ts
import express from 'express';
import { verifyLineSignature, isSignatureFresh, isDuplicateOrProcessed, recordWebhookEvent, markApplied } from '../services/webhookService';
import { Pool } from 'pg';

const router = express.Router();
const pool = new Pool({ connectionString: process.env.DATABASE_URL });

router.post('/webhook/line', express.raw({ type: 'application/json' }), async (req, res) => {
  const signature = req.headers['x-line-signature'] as string;
  const rawBody = req.body.toString('utf8');
  const okSig = await verifyLineSignature(rawBody, signature);
  if (!okSig) return res.status(401).send('Invalid signature');

  let body;
  try {
    body = JSON.parse(rawBody) as any;
  } catch {
    return res.status(400).send('Invalid JSON');
  }

  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    for (const ev of body.events || []) {
      const fresh = isSignatureFresh(ev.timestamp);
      if (!fresh) {
        await recordWebhookEvent(ev.webhookEventId, ev.type, ev.source?.userId, Buffer.from(''), 'rejected');
        continue; // skip stale events but record for audit
      }

      const duplicate = await isDuplicateOrProcessed(ev.webhookEventId);
      if (duplicate) {
        // idempotent success response
        continue;
      }

      // Record receipt first (prevents races on concurrent deliveries)
      const payloadHash = crypto.createHash('sha256').update(JSON.stringify(ev)).digest();
      await recordWebhookEvent(ev.webhookEventId, ev.type, ev.source?.userId, payloadHash, 'received');

      // Dispatch to domain handlers (clock/leave/OT) within same tx
      // Example: await handleClockEvent(client, ev);
      // For now we mark applied immediately after domain logic succeeds.
      // Replace with actual domain handlers:
      // await handleClockEvent(client, ev);

      await markApplied(ev.webhookEventId);
    }

    await client.query('COMMIT');
    res.status(200).send('OK');
  } catch (err) {
    await client.query('ROLLBACK');
    console.error('Webhook processing failed', err);
    res.status(500).send('Processing error');
  } finally {
    client.release();
  }
});

export default router;
```
