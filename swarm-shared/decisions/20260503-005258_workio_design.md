# workio / design

## Final consolidated implementation (strongest parts, resolved contradictions)

### 1. Diagnosis (merged)
- **No DB-level guardrails**: multiple open punches (`clock_out_at IS NULL`) per user per day are possible under concurrency/retries.
- **Race-prone upsert**: app-layer find-then-insert is not atomic; duplicate LINE webhooks can insert two open rows.
- **No idempotency**: LINE retries are not deduplicated; same event can be processed twice.
- **No lightweight probes**: ops lack `/readyz` and `/livez` to verify DB + external dependency health.
- **Missing defense-in-depth**: no time-window normalization (midnight edge cases) and no monotonicity check for clock-in while already open.

### 2. Design decisions (resolved)
- **Idempotency**: use a dedicated `idempotency_keys` table (Candidate 2) rather than a nullable column on `punches`. This keeps punches clean, supports any idempotent endpoint, and avoids nullable unique indexes that complicate queries.
- **Partial unique index**: use `DATE(clock_in_at)` (Candidate 2) instead of `DATE(created_at)` for correctness; business rule is “one open punch per calendar day” based on when the punch started.
- **Transactional upsert**: perform atomic “insert-if-not-exists” inside a DB transaction with proper isolation; rely on the partial unique index as the final guard.
- **Health endpoints**: add `/livez` (process) and `/readyz` (DB + migrations + optional external checks).
- **Time normalization**: normalize input times to UTC and use `DATE(clock_in_at AT TIME ZONE $timezone)` or UTC date boundaries; include monotonicity check before insert.

### 3. Implementation

#### 3.1 DB schema (`server/src/db/schema.sql`)
```sql
-- 1) Partial unique index: at most one open punch per user per calendar day (by clock_in_at)
CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_open_per_day
  ON punches (user_id, DATE(clock_in_at))
  WHERE clock_out_at IS NULL;

-- 2) Idempotency table for safe retries across endpoints
CREATE TABLE IF NOT EXISTS idempotency_keys (
  key         TEXT PRIMARY KEY,
  payload_hash TEXT NOT NULL,
  created_at  TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Optional: index for cleanup
CREATE INDEX IF NOT EXISTS idx_idempotency_keys_created_at
  ON idempotency_keys (created_at);
```

#### 3.2 Idempotency middleware (`server/src/middleware/idempotency.ts`)
```ts
import { Request, Response, NextFunction } from 'express';
import { pool } from '../db';
import crypto from 'crypto';

export function idempotencyMiddleware(req: Request, res: Response, next: NextFunction) {
  const key = req.headers['idempotency-key'] as string;
  if (!key) return next();

  // Compute stable payload hash (exclude idempotency header)
  const payload = JSON.stringify(req.body);
  const payloadHash = crypto.createHash('sha256').update(payload).digest('hex');

  req.idempotency = { key, payloadHash };
  next();
}

export async function enforceIdempotency(
  key: string,
  payloadHash: string
): Promise<{ duplicate: true; existing: any } | { duplicate: false }> {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // Check existing key
    const found = await client.query(
      `SELECT payload_hash FROM idempotency_keys WHERE key = $1 FOR UPDATE`,
      [key]
    );
    if (found.rows.length > 0) {
      // Optionally validate payload_hash matches to detect misuse
      await client.query('COMMIT');
      // Fetch associated punch if this was a punch operation (best-effort)
      return { duplicate: true, existing: null };
    }

    // Reserve idempotency key
    await client.query(
      `INSERT INTO idempotency_keys (key, payload_hash) VALUES ($1, $2)`,
      [key, payloadHash]
    );
    await client.query('COMMIT');
    return { duplicate: false };
  } catch (err: any) {
    await client.query('ROLLBACK');
    if (err.code === '23505') {
      return { duplicate: true, existing: null };
    }
    throw err;
  } finally {
    client.release();
  }
}
```

#### 3.3 Clock-in route with transactional upsert and monotonicity (`server/src/routes/punch.ts`)
```ts
import { Router } from 'express';
import { pool } from '../db';
import { enforceIdempotency } from '../middleware/idempotency';

const router = Router();

/**
 * POST /punch/clock-in
 * Body: { userId: number, location?: { lat: number, lng: number }, timezone?: string }
 * Headers: Idempotency-Key (recommended for LINE webhooks)
 */
router.post('/clock-in', async (req, res) => {
  const { userId, location, timezone = 'UTC' } = req.body;
  if (!userId) return res.status(400).json({ error: 'userId required' });

  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // Idempotency check (if header provided)
    const idKey = (req.headers['idempotency-key'] || '') as string;
    if (idKey) {
      const payload = JSON.stringify(req.body);
      const crypto = require('crypto');
      const payloadHash = crypto.createHash('sha256').update(payload).digest('hex');
      const idem = await enforceIdempotency(idKey, payloadHash);
      if (idem.duplicate) {
        // Try to return the existing open punch for clarity
        const existing = await client.query(
          `SELECT id, user_id, clock_in_at, clock_out_at, location, created_at
           FROM punches
           WHERE user_id = $1 AND clock_out_at IS NULL
           ORDER BY clock_in_at DESC
           LIMIT 1`,
          [userId]
        );
        await client.query('COMMIT');
        return res.json({ ok: true, existing: existing.rows[0] || null, deduplicated: true });
      }
    }

    // Monotonicity: ensure no open punch exists for this user
    const open = await client.query(
      `SELECT id, user_id, clock_in_at, clock_out_at, location, created_at
       FROM punches
       WHERE user_id = $1 AND clock_out_at IS NULL
       ORDER BY clock_in_at DESC
       LIMIT 1`,
      [userId]
    );
    if (open.rows.length > 0) {
      await client.query('COMMIT');
      return res.status(409).json({ error: 'Already clocked in', punch: open.rows[0] });
    }

    // Insert new clock-in (use UTC for storage; timezone used only for day boundary business rules)
    const result = await client.query(
      `INSERT INTO punches (user_id, clock_in_at, clock_out_at, location, created_at)
       VALUES ($1, NOW(), NULL, $2, NOW())
       RETURNING id, user_id, clock_in_at, clock_out_at, location, created_at`,
      [userId, location ? JSON.stringify(location) : null]
    );

    await client.query('COMMIT');
    res.json({ ok: true, punch: result.rows[0], deduplicated: false });
  } catch (err: any) {
    await client.query('ROLLBACK');
    // Unique violation from partial index: someone else inserted concurrently
    if (err.code === '23505') {
      const existing = await pool.query(
        `SELECT id, user_id, clock_in_at, clock_out_at, location, created_at
         FROM punches
         WHERE user_id = $1 AND clock_out_at IS NULL
         ORDER BY clock_in_at DESC
         LIMIT 1`,
        [userId]
      );
      return res.json({ ok: true, existing: existing.rows[0] || null, conflict: true });
    }
    console.error('Clock-in error', err);
    res.status(500).json({ error: 'Internal server error' });
  } finally {
    client.release();
  }
});
```
