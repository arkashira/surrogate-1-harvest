## Highest-Value Incremental Improvement (<2h)

**Goal**: Eliminate HF API rate-limit (429) risk during Surrogate-1 Lightning Studio training by switching from `load_dataset(streaming=True)` to CDN-only parquet loading via a pre-computed file manifest.

**Why this**:  
- Directly addresses the Surrogate-1 training pipeline HF rate-limit pattern.  
- Uses the HF CDN bypass insight (public `resolve/main/` URLs require no auth and have much higher limits).  
- Fits in <2h: single-file change + small manifest generation script.  
- Safe: keeps existing training loop; only changes data-loading path.

---

## Implementation Plan

1. **Add manifest generation script** (`scripts/build_file_manifest.py`)  
   - Runs on Mac (or any dev machine) after rate-limit window clears.  
   - Uses `list_repo_tree(path, recursive=False)` per date folder under `batches/mirror-merged/`.  
   - Emits `file_manifest.json` mapping `{date: [{"slug": "...", "cdn_url": "...", "local_path": "..."}]}`.  
   - Commits or copies manifest into repo for training.

2. **Create CDN dataset loader** (`surrogate/data/cdn_parquet_dataset.py`)  
   - Accepts `manifest_path` and optional `date_filter`.  
   - Builds list of CDN URLs.  
   - Streams downloads via `requests.get(url, timeout=60, stream=True)` + `pyarrow.parquet.ParquetFile` from buffer.  
   - Projects only `{prompt, response}` at parse time (avoids mixed-schema issues).  
   - Returns an iterable of dicts compatible with existing training loop.

3. **Patch training entrypoint** (`surrogate/train.py` or equivalent)  
   - Replace `load_dataset(streaming=True, ...)` with `CdnParquetDataset(manifest_path=...)`.  
   - Keep collator/tokenizer logic unchanged.  
   - Add fallback: if manifest missing, print clear error and exit (avoid silent HF API fallback).

4. **Lightning Studio reuse guard** (small safety)  
   - Before `.run()`, check studio status; if stopped, restart with `target.start(machine=Machine.L40S)`.

## Code Snippets

### 1. Manifest builder (`scripts/build_file_manifest.py`)

```python
#!/usr/bin/env python3
"""
Build CDN-only file manifest for Surrogate-1 training.
Run from dev machine (Mac) to avoid HF API rate limits during training.
"""
import json
import os
import sys
from datetime import datetime
from pathlib import Path

from huggingface_hub import HfApi

REPO_ID = os.getenv("HF_DATASET_REPO", "org/surrogate-mirror")
OUTPUT_PATH = Path(__file__).parent.parent / "surrogate" / "data" / "file_manifest.json"
DATE_PREFIX = "batches/mirror-merged"  # e.g. batches/mirror-merged/2026-04-29/

def main() -> None:
    api = HfApi()
    manifest = {}

    # List top-level date folders (non-recursive to minimize API calls)
    try:
        repo_tree = api.list_repo_tree(repo_id=REPO_ID, path=DATE_PREFIX, recursive=False)
    except Exception as e:
        print(f"Error listing repo tree: {e}", file=sys.stderr)
        sys.exit(1)

    for item in repo_tree:
        if item.type != "directory":
            continue
        date_folder = item.path  # e.g. batches/mirror-merged/2026-04-29
        date_key = Path(date_folder).name  # 2026-04-29

        # List parquet files in this date folder
        try:
            files = api.list_repo_tree(repo_id=REPO_ID, path=date_folder, recursive=False)
        except Exception as e:
            print(f"Error listing {date_folder}: {e}", file=sys.stderr)
            continue

        entries = []
        for f in files:
            if not f.path.lower().endswith(".parquet"):
                continue
            slug = Path(f.path).stem  # e.g. abc123
            cdn_url = f"https://huggingface.co/datasets/{REPO_ID}/resolve/main/{f.path}"
            entries.append(
                {
                    "slug": slug,
                    "cdn_url": cdn_url,
                    "hf_path": f.path,
                }
            )

        if entries:
            manifest[date_key] = entries

    OUTPUT_PATH.parent.mkdir(parents=True, exist_ok=True)
    with open(OUTPUT_PATH, "w") as fp:
        json.dump(manifest, fp, indent=2)

    print(f"Manifest written to {OUTPUT_PATH} with {len(manifest)} dates.")

if __name__ == "__main__":
    main()
```

### 2. CDN dataset loader (`surrogate/data/cdn_parquet_dataset.py`)

```python
from __future__ import annotations

import io
import json
import logging
from dataclasses import dataclass
from pathlib import Path
from typing import Dict, Iterable, List, Optional

import pyarrow.parquet as pq
import requests

logger = logging.getLogger(__name__)

@dataclass
class CdnParquetDataset:
    manifest_path: Path
    date_filter: Optional[str] = None  # e.g. "2026-04-29"
    timeout: int = 60
    chunk_size: int = 8192

    def __post_init__(self) -> None:
        with open(self.manifest_path) as f:
            self.manifest: Dict[str, List[dict]] = json.load(f)

        if self.date_filter:
            if self.date_filter not in self.manifest:
                raise ValueError(f"Date {self.date_filter} not found in manifest.")
            self._items = [
                entry for entry in self.manifest[self.date_filter]
            ]
        else:
            self._items = [
                entry for entries in self.manifest.values() for entry in entries
            ]

        if not self._items:
            raise ValueError("No parquet files found in manifest with current filters.")

    def __len__(self) -> int:
        return len(self._items)

    def _stream_parquet(self, url: str) -> bytes:
        resp = requests.get(url, timeout=self.timeout, stream=True)
        resp.raise_for_status()
        buf = io.BytesIO()
        for chunk in resp.iter_content(chunk_size=self.chunk_size):
            buf.write(chunk)
        buf.seek(0)
        return buf

    def __iter__(self) -> Iterable[dict]:
        for item in self._items:
            url = item["cdn_url"]
            slug = item["slug"]
            try:
                buf = self._stream_parquet(url)
                table = pq.read_table(buf)
                # Project only prompt/response to avoid mixed-schema issues
                cols = set(table.column_names)
                prompt_col = next((c for c in ("prompt", "Prompt") if c in cols), None)
                response_col = next((c for c in ("response", "Response") if c in cols), None)

                if prompt_col is None or response_col is None:
                    logger.warning("Missing prompt/response in %s; skipping.", slug)
                    continue

                prompts = table.column(prompt_col).to_pylist()
                responses = table.column(response_col).to_pylist()
                for p, r in zip(prompts, responses):
                    if p is None or r is None:
                        continue
                    yield {"prompt": p, "response": r, "slug": slug}
            except Exception as e:
                logger.error("Failed to load %s from CDN: %s", url, e)
                continue
```
