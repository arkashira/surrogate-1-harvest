#### 1) `scripts/list_hf_files.py`

```python
#!/usr/bin/env python3
"""
Run once (Mac) after HF rate-limit window clears.
Produces filelist.json for a single date folder (non-recursive).
"""
import json
import os
import sys
from huggingface_hub import HfApi

REPO = os.getenv("HF_DATASET_REPO", "my-org/surrogate-data")
FOLDER = sys.argv[1] if len(sys.argv) > 1 else "batches/mirror-merged/2026-05-03"
OUT = sys.argv[2] if len(sys.argv) > 2 else "surrogate/data/filelist.json"

api = HfApi()
# Single non-recursive call (avoids pagination on big repos)
tree = api.list_repo_tree(repo_id=REPO, path=FOLDER, recursive=False)
files = [item.rfilename for item in tree if item.type == "file"]

with open(OUT, "w") as f:
    json.dump({"repo": REPO, "folder": FOLDER, "files": files}, f, indent=2)

print(f"Wrote {len(files)} files to {OUT}")
```

#### 2) `surrogate/data/cdn_dataset.py`
```python
"""
CDN-only dataset loader.
Uses filelist.json produced by list_hf_files.py.
No HF API calls during training.
"""
import json
import os
from typing import Dict

from datasets import load_dataset


class CDNDataset:
    def __init__(self, filelist_path: str):
        with open(filelist_path) as f:
            meta = json.load(f)
        self.repo = meta["repo"]
        self.folder = meta["folder"]
        self.files = meta["files"]
        if not self.files:
            raise ValueError("No files in filelist")

    def load(self, split="train"):
        # Build a dummy split mapping to CDN URLs
        # Use data_files with explicit CDN URLs (bypasses HF API auth/rate limits)
        data_files = []
        for fname in self.files:
            url = f"https://huggingface.co/datasets/{self.repo}/resolve/main/{fname}"
            data_files.append(url)

        # Streaming avoids loading all into RAM; map projects schema at parse time
        ds = load_dataset(
            "json",
            name=split,
            data_files=data_files,
            split=split,
            streaming=True,
        )

        # Project to {prompt, response} only; keep attribution in filename if needed
        def _project(example):
            # Heuristic: prefer 'prompt'/'response'; fallback to common keys
            prompt = example.get("prompt") or example.get("input") or example.get("question") or ""
            response = example.get("response") or example.get("output") or example.get("answer") or ""
            return {"prompt": prompt, "response": response}

        ds = ds.map(_project, remove_columns=list(ds.features.keys()) if ds.features else None)
        return ds
```

```python
#!/usr/bin/env python3
"""
Lightning training entrypoint with:
- CDN-only dataset loader
- Idle-stop resilience (auto-restart)
- Deterministic sibling-repo write for uploads
"""
import hashlib
import os
import sys
import time

from lightning import Lightning, L40S, Machine, Teamspace
from surrogate.data.cdn_dataset import CDNDataset

# ---- Config ----
FILELIST = os.getenv("FILELIST", "surrogate/data/filelist.json")
SIBLING_REPOS = int(os.getenv("SIBLING_REPOS", "5"))  # aggregate commit cap
TARGET_REPO_TMPL = os.getenv("TARGET_REPO_TMPL", "my-org/surrogate-data-{ix}")

# ---- Helpers ----
def pick_sibling_repo(slug: str) -> str:
    """Deterministic repo selection by hash."""
    h = int(hashlib.sha256(slug.encode()).hexdigest(), 16)
    ix = h % SIBLING_REPOS
    return TARGET_REPO_TMPL.format(ix=ix)

def wait_for_studio(studio, timeout=300):
    waited = 0
    while studio.status != "Running" and waited < timeout:
        time.sleep(10)
        waited += 10
        studio.refresh()
    if studio.status != "Running":
        raise RuntimeError(f"Studio {studio.name} not running after {timeout}s")

# ---- Dataset ----
cdn = CDNDataset(FILELIST)
train_ds = cdn.load("train")

# ---- Lightning runner with idle-resilience ----
def run_training():
    lightning = Lightning()
    teamspace = Teamspace()

    studio_name = "surrogate-train-l40s"
    running = None
    for s in teamspace.studios:
        if s.name == studio_name and s.status == "Running":
            running = s
            break

    if running:
        print(f"Reusing running studio: {running.name}")
        studio = running
    else:
        print(f"Creating studio: {studio_name}")
        studio = lightning.Studio(
            name=studio_name,
            machine=Machine.L40S,
            create_ok=True,
        )

    # Ensure studio is running before .run()
    wait_for_studio(studio)

    # Example training script body (replace with real train loop)
    script = """
import json
import os
from surrogate.data.cdn_dataset import CDNDataset

cdn = CDNDataset(os.environ["FILELIST"])
ds = cdn.load("train")

count = 0
for ex in ds:
    # dummy train step
    count += 1
    if count >= 100:
        break

# Deterministic upload target
from surrogate.train import pick_sibling_repo
repo = pick_sibling_repo("mirror-merged/2026-05-03")
print(f"Selected repo: {repo}")
# hf_api.upload_file(..., repo_id=repo)
"""

    # Run with retry on idle-stop
    max_retries = 3
    for attempt in range(1, max_retries + 1):
        try:
            studio.run(
                project="surrogate-training",
                environment=os.environ,
                script=script,
                wait=True,
            )
            print("Training completed.")
            break
        except Exception as exc:
            print(f"Attempt {attempt} failed: {exc}")
            studio.refresh()
            if studio.status != "Running":
                print("Studio stopped (idle-timeout). Restarting...")
                studio.start(machine=Machine.L40S)
                wait_for_studio(studio)
                if attempt == max_retries:
                    raise
            else:
                raise

if __name__ == "__main__":
    run_training()
```
