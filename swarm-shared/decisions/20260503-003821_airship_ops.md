## Final Implementation Plan  
**Goal:** Replace `load_dataset`/`list_repo_files` in surrogate training with a manifest-driven, CDN-only iterable loader that eliminates HF API rate limits and mixed-schema ingestion failures while remaining RAM-safe and deterministic.

---

### 1) Highest-value change (≤2h)
- Add a **manifest builder** (run once on the Mac orchestrator after the rate-limit window clears) that produces `file_manifest.json` for a chosen date-folder.
- Add a **CDN-only `IterableDataset`** that:
  - Uses only public CDN URLs (`https://huggingface.co/datasets/{repo}/resolve/main/{path}`) — no Authorization header, no API calls during training.
  - Projects every record to `{prompt, response}` at parse time (ignores schema drift/extra columns).
  - Supports JSONL and JSON (object or array-of-objects) with tolerant per-record skipping.
  - Streams via PyTorch `IterableDataset` with per-worker deterministic sharding so training never materializes the full dataset in RAM.
- Update the surrogate training entrypoint to use this dataset and remove/disable any `load_dataset(streaming=True)` and `list_repo_files` calls.

---

### 2) Concrete steps
1. **Create manifest builder**  
   `scripts/build_manifest.py` → consumes HF repo + date folder → emits `surrogate/data/file_manifest.json`.

2. **Create CDN streaming dataset**  
   `surrogate/data/cdn_streaming_dataset.py` → `CdnStreamingDataset` with retries, timeouts, and per-worker sharding.

3. **Integrate into training**  
   Replace dataset construction in the surrogate training entrypoint with `CdnStreamingDataset(manifest_path=...)`. Remove/disable legacy HF API calls.

4. **Validation smoke test**  
   Run one training epoch on a small manifest subset; confirm no HF API calls (e.g., monitor traffic or logs) and correct `{prompt, response}` projection.

### 3) Code snippets

#### `scripts/build_manifest.py`
```python
#!/usr/bin/env python3
"""
Run on Mac after HF API rate-limit window clears.
Produces file_manifest.json for a given date folder.

Usage:
  python scripts/build_manifest.py \
    --repo "my-org/surrogate-data" \
    --date "2026-04-29" \
    --out "surrogate/data/file_manifest.json"
"""
import argparse
import json
import os
import sys
from typing import List

try:
    from huggingface_hub import list_repo_tree
except ImportError:
    print("ERROR: huggingface_hub required. pip install huggingface_hub")
    sys.exit(1)

def build_manifest(repo_id: str, date_folder: str, out_path: str) -> None:
    # List top-level items in the date folder (non-recursive) to reduce pagination pressure.
    # If deeper nesting is required, run recursive once and store full relative paths.
    items = list_repo_tree(repo_id=repo_id, path=date_folder, recursive=True)
    files: List[str] = sorted(f.rfilename for f in items if f.type == "file")

    manifest = {
        "repo_id": repo_id,
        "date_folder": date_folder,
        "files": files,
        "total_files": len(files),
    }

    os.makedirs(os.path.dirname(os.path.abspath(out_path)), exist_ok=True)
    with open(out_path, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2)

    print(f"Wrote manifest with {len(files)} files to {out_path}")

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Build CDN file manifest for surrogate training.")
    parser.add_argument("--repo", required=True, help="HF dataset repo id (e.g. org/repo)")
    parser.add_argument("--date", required=True, help="Date folder in dataset (e.g. 2026-04-29)")
    parser.add_argument("--out", default="surrogate/data/file_manifest.json", help="Output JSON path")
    args = parser.parse_args()
    build_manifest(args.repo, args.date, args.out)
```

#### `surrogate/data/cdn_streaming_dataset.py`
```python
"""
Manifest-driven, CDN-only streaming dataset for surrogate training.
- Zero HF API calls during training (bypasses rate limits).
- Projects each file to {prompt, response} at parse time.
- Supports JSONL and JSON (object or array-of-objects).
- Deterministic per-worker sharding for DataLoader workers.
"""
import json
import logging
from pathlib import Path
from typing import Dict, Iterator, List, Optional

import requests
import torch
from torch.utils.data import IterableDataset

logger = logging.getLogger(__name__)

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo_id}/resolve/main/{path}"

class CdnStreamingDataset(IterableDataset):
    def __init__(
        self,
        manifest_path: str,
        repo_id: Optional[str] = None,
        columns: Optional[List[str]] = None,
        max_retries: int = 3,
        timeout: int = 30,
    ):
        """
        Args:
            manifest_path: Path to file_manifest.json produced by scripts/build_manifest.py
            repo_id: Override repo id (defaults to manifest value)
            columns: Keys to select from each record; default ["prompt","response"]
            max_retries: Per-file HTTP retries
            timeout: HTTP timeout
        """
        self.manifest_path = Path(manifest_path)
        if not self.manifest_path.is_file():
            raise FileNotFoundError(f"Manifest not found: {manifest_path}")

        with open(self.manifest_path, "r", encoding="utf-8") as f:
            self.manifest = json.load(f)

        self.repo_id = repo_id or self.manifest.get("repo_id")
        if not self.repo_id:
            raise ValueError("repo_id must be provided or present in manifest")
        self.files: List[str] = self.manifest.get("files", [])
        self.columns = columns or ["prompt", "response"]
        self.max_retries = max_retries
        self.timeout = timeout

    def _fetch_file(self, rel_path: str) -> bytes:
        url = CDN_TEMPLATE.format(repo_id=self.repo_id, path=rel_path)
        for attempt in range(1, self.max_retries + 1):
            try:
                resp = requests.get(url, timeout=self.timeout)
                resp.raise_for_status()
                return resp.content
            except Exception as exc:
                if attempt == self.max_retries:
                    logger.error(f"Failed to fetch {url} after {attempt} attempts: {exc}")
                    raise
                logger.warning(f"Retry {attempt}/{self.max_retries} for {url}: {exc}")
        raise RuntimeError(f"Unreachable: failed to fetch {rel_path}")

    def _parse_records(self, content: bytes, rel_path: str) -> Iterator[Dict]:
        text = content.decode("utf-8")
        lines = [ln.strip() for ln in text.splitlines() if ln.strip()]

        # Try JSONL: every non-empty line must be a JSON object
        parsed: List[Dict] = []
        all_lines_are_objects = True
        for ln in lines:
            try:
                obj = json.loads(ln)
                if isinstance(obj, dict):
                    parsed.append(obj)
                else:
                    all_lines_are_objects = False
                    break
            except json.JSONDecodeError:
                all_lines_are_objects = False
                break

        if all_lines_are_objects and parsed:
            for obj in parsed:
                yield self._project(obj, rel_path)
            return

        # Fallback: treat entire file as single JSON (object or array)
        try:
            single = json.loads(text)
        except json.JSONDecodeError as exc:
            logger.error(f"Invalid JSON in {rel_path}: {exc}")
            return

        if isinstance(single, list):
            for obj in single:
                if isinstance(obj, dict):
                    yield self._project(obj, rel_path)
        elif isinstance(single, dict):
            yield self._project(single, rel_path)
        else:
            logger.warning(f"Unexpected JSON root type in {rel_path}: {type(single)}")

    def _project(self, obj: Dict, rel_path: str) -> Dict:
        # Keep only requested columns; missing keys become None
        projected = {k: obj.get(k) for k in self.columns}
        # Lightweight
