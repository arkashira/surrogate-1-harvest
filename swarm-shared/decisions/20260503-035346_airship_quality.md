### 1) scripts/list_hf_files.py

```python
#!/usr/bin/env python3
"""
Pre-list HF dataset files (non-recursive per folder) and emit file_list.json.
Run from Mac after HF API rate-limit window clears.
"""
import json
import os
import sys
from datetime import datetime, timezone
from pathlib import Path

import requests
from tqdm import tqdm

HF_REPO = "datasets/axentx/surrogate-mirror"  # adjust if needed
HF_API_BASE = "https://huggingface.co/api"
OUTPUT_FILE = Path(__file__).parent.parent / "surrogate" / "file_list.json"

# If repo is private, set HF_TOKEN in env. Public datasets don't require auth for CDN.
HF_TOKEN = os.getenv("HF_TOKEN")
HEADERS = {"Authorization": f"Bearer {HF_TOKEN}"} if HF_TOKEN else {}

def list_repo_tree(path: str = "") -> list[dict]:
    url = f"{HF_API_BASE}/datasets/{HF_REPO}/tree"
    params = {"recursive": "false"}
    if path:
        params["path"] = path
    resp = requests.get(url, headers=HEADERS, params=params, timeout=30)
    if resp.status_code == 429:
        print("HF API 429 — wait 360s before retry", file=sys.stderr)
        sys.exit(1)
    resp.raise_for_status()
    return resp.json()

def collect_files(date_folder: str) -> list[str]:
    """Return CDN paths for parquet files under date_folder."""
    entries = list_repo_tree(date_folder)
    paths = []
    for entry in entries:
        if entry.get("type") == "file" and entry.get("path", "").endswith(".parquet"):
            paths.append(entry["path"])
    return paths

def main() -> None:
    # Example: list top-level date folders (YYYY-MM-DD)
    top = list_repo_tree("")
    date_folders = [e["path"] for e in top if e.get("type") == "dir" and e["path"]]
    if not date_folders:
        # fallback: list root parquet files
        date_folders = [""]

    file_list = []
    for folder in tqdm(date_folders, desc="Scanning folders"):
        try:
            paths = collect_files(folder)
            file_list.extend(paths)
        except Exception as exc:
            print(f"Skip {folder}: {exc}", file=sys.stderr)

    out = {
        "repo": HF_REPO,
        "generated_utc": datetime.now(timezone.utc).isoformat(),
        "count": len(file_list),
        "paths": sorted(file_list),
    }

    OUTPUT_FILE.parent.mkdir(parents=True, exist_ok=True)
    OUTPUT_FILE.write_text(json.dumps(out, indent=2))
    print(f"Wrote {len(file_list)} paths to {OUTPUT_FILE}")

if __name__ == "__main__":
    main()
```

# ---------- Helpers ----------
def pick_sibling_repo(slug: str) -> str:
    """Deterministic sibling repo selection to spread writes (not used during CDN reads)."""
    digest = int(hashlib.sha256(slug.encode()).hexdigest(), 16)
    idx = digest % len(SIBLING_REPOS)
    return SIBLING_REPOS[idx]

def cdn_url(repo: str, path: str) -> str:
    return f"{CDN_BASE}/{repo}/resolve/main/{path}"

def download_parquet_cdn(repo: str, path: str, dest: Path) -> Optional[Path]:
    url = cdn_url(repo, path)
    try:
        resp = requests.get(url, timeout=60, stream=True)
        resp.raise_for_status()
        with open(dest, "wb") as f:
            for chunk in resp.iter_content(chunk_size=8192):
                f.write(chunk)
        return dest
    except Exception as exc:
        print(f"Failed to download {url}: {exc}")
        return None

# ---------- Dataset ----------
class SurrogateParquetDataset(Dataset):
    def __init__(self, file_list: List[str], repo: str, max_samples: Optional[int] = None):
        self.repo = repo
        self.paths = file_list
        if max_samples:
            self.paths = self.paths[:max_samples]

    def __len__(self) -> int:
        return len(self.paths)

    def __getitem__(self, idx: int) -> Dict[str, str]:
        rel = self.paths[idx]
        with tempfile.TemporaryDirectory() as td:
            local = Path(td) / Path(rel).name
            downloaded = download_parquet_cdn(self.repo, rel, local)
            if not downloaded:
                return {"prompt": "", "response": ""}
            try:
                table = pq.read_table(str(downloaded), columns=["prompt", "response"])
                df = table.to_pandas()
                row = df.iloc[0]
                return {"prompt": str(row.get("prompt", "")), "response": str(row.get("response", ""))}
            except Exception as exc:
                print(f"Parse error {rel}: {exc}")
                return {"prompt": "", "response": ""}
