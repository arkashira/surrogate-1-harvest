### 1. API layer with idempotency + retry + backoff

`workio/src/api/attendance.ts`

```ts
import { v4 as uuidv4 } from 'uuid';

export type PunchType = 'CLOCK_IN' | 'CLOCK_OUT';

export interface PunchRequest {
  type: PunchType;
  latitude?: number;
  longitude?: number;
  accuracy?: number;
}

const MAX_RETRIES = 3;
const BASE_DELAY = 800;

function wait(ms: number) {
  return new Promise((r) => setTimeout(r, ms));
}

export async function sendPunch(
  req: PunchRequest,
  idempotencyKey?: string
): Promise<{ ok: true; data: any } | { ok: false; error: string; retryable: boolean }> {
  const key = idempotencyKey || uuidv4();
  let lastError: Error | null = null;

  for (let attempt = 0; attempt <= MAX_RETRIES; attempt++) {
    try {
      const res = await fetch('/api/attendance/punch', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'Idempotency-Key': key,
        },
        body: JSON.stringify(req),
      });

      if (res.ok) {
        return { ok: true, data: await res.json() };
      }

      // Retry on 429 and 5xx
      if (res.status === 429 || (res.status >= 500 && res.status < 600)) {
        const retryAfter = Number(res.headers.get('Retry-After')) || (BASE_DELAY / 1000);
        const delay = (retryAfter * 1000) || (BASE_DELAY * 2 ** attempt) + Math.random() * 200;
        if (attempt < MAX_RETRIES) await wait(delay);
        continue;
      }

      // 4xx (non-429) -> client error, do not retry
      const body = await res.text().catch(() => '');
      return { ok: false, error: body || 'Request failed', retryable: false };
    } catch (err: any) {
      lastError = err;
      if (attempt < MAX_RETRIES) {
        const delay = (BASE_DELAY * 2 ** attempt) + Math.random() * 200;
        await wait(delay);
      }
    }
  }

  return { ok: false, error: lastError?.message || 'Unknown error', retryable: true };
}
```

### 2. Persistent punch queue (offline/weak-network safe)

`workio/src/features/attendance/usePunchQueue.ts`

```ts
import { useEffect, useCallback } from 'react';
import { sendPunch, PunchRequest } from '../../api/attendance';
import { v4 as uuidv4 } from 'uuid';

type QueuedPunch = {
  id: string;
  req: PunchRequest;
  idempotencyKey: string;
  attempts: number;
  createdAt: number;
};

const STORAGE_KEY = 'workio:pending-punches';

function loadQueue(): QueuedPunch[] {
  try {
    const raw = localStorage.getItem(STORAGE_KEY);
    return raw ? JSON.parse(raw) : [];
  } catch {
    return [];
  }
}

function saveQueue(queue: QueuedPunch[]) {
  try {
    localStorage.setItem(STORAGE_KEY, JSON.stringify(queue));
  } catch {
    // ignore storage failures
  }
}

export function usePunchQueue(onFlush?: () => void) {
  // Flush on mount + periodically
  useEffect(() => {
    const attemptFlush = () => void flushQueue();
    attemptFlush();
    const id = setInterval(attemptFlush, 15000);
    return () => clearInterval(id);
  }, []);

  const add = useCallback((req: PunchRequest) => {
    const item: QueuedPunch = {
      id: uuidv4(),
      req,
      idempotencyKey: uuidv4(),
      attempts: 0,
      createdAt: Date.now(),
    };
    const q = loadQueue();
    q.push(item);
    saveQueue(q);
  }, []);

  async function flushQueue() {
    let q = loadQueue();
    if (q.length === 0) return;

    const remaining: QueuedPunch[] = [];
    for (const item of q) {
      if (item.attempts >= 6) continue;

      const result = await sendPunch(item.req, item.idempotencyKey);
      if (result.ok) {
        continue; // success — drop
      }

      remaining.push({ ...item, attempts: item.attempts + 1 });
    }

    saveQueue(remaining);
    if (remaining.length !== q.length) onFlush?.();
  }

  return { add, flushQueue, pendingCount: loadQueue().length };
}
```
