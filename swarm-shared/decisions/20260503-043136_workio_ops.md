# workio / ops

## 1. Diagnosis
- **No idempotency guard on LINE webhook ingestion** — LINE retries on 5xx/timeouts; duplicate clock-in/out or leave/OT records are created on replay.
- **Missing `X-Line-Signature` verification** — webhook accepts spoofed events; enables attendance fraud and phantom clock-ins.
- **No structured logging / tracing for webhook ops** — failures are opaque; retries amplify duplicates and make incident triage slow.
- **No graceful handling of out-of-order clock events** — clock-out before clock-in or duplicate clock-in within tolerance creates invalid sessions.
- **No circuit-breaker / backpressure on downstream writes** — burst retries from LINE can overload DB or create contention during incidents.

## 2. Proposed change
File: `/opt/axentx/workio/server/src/routes/webhook/line.ts` (or create if missing).  
Scope: add `X-Line-Signature` verification, idempotency key (dedupe by `webhookEvent.id`), structured logging, and safe-clock session guard — all behind a small, testable middleware layer.

```ts
import express, { Request, Response, NextFunction } from 'express';
import crypto from 'crypto';
import { db } from '../../db';
import { logger } from '../../utils/logger';

const router = express.Router();

// Helpers
function verifyLineSignature(rawBody: Buffer, signature: string, channelSecret: string): boolean {
  const expected = crypto
    .createHmac('sha256', channelSecret)
    .update(rawBody)
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

// Idempotency store (in-memory for now; replace with Redis/DB for multi-instance)
const seenEvents = new Set<string>();

function lineWebhookGuard(channelSecret: string) {
  return (req: Request, res: Response, next: NextFunction) => {
    const signature = req.headers['x-line-signature'] as string;
    if (!signature) {
      logger.warn({ msg: 'line_webhook_missing_signature' });
      return res.status(401).json({ error: 'missing signature' });
    }

    const raw = req.body as Buffer;
    if (!verifyLineSignature(raw, signature, channelSecret)) {
      logger.warn({ msg: 'line_webhook_invalid_signature' });
      return res.status(401).json({ error: 'invalid signature' });
    }

    // Parse events after signature OK (body should be JSON; ensure raw buffer preserved)
    const payload = JSON.parse(raw.toString('utf8'));
    req.body = payload;

    // Idempotency: reject replays of already-processed events
    const eventIds = (payload.events || []).map((e: any) => e.id).filter(Boolean);
    const duplicates = eventIds.filter((id: string) => seenEvents.has(id));
    if (duplicates.length > 0) {
      logger.info({ msg: 'line_webhook_duplicate', ids: duplicates });
      // LINE expects 2xx to stop retries — accept but skip processing
      return res.status(200).json({ ok: true, skipped: duplicates });
    }

    // Mark seen (best-effort)
    eventIds.forEach((id: string) => seenEvents.add(id));

    // Attach metadata for handlers
    (req as any).lineEvents = payload.events || [];
    next();
  };
}

// Clock session guard: prevent invalid transitions
async function upsertClockSession(userId: string, type: 'in' | 'out', ts: Date, location?: string) {
  const latest = await db.query(
    `SELECT id, type, created_at FROM attendance_sessions
     WHERE user_id = $1 AND tenant_id = (SELECT tenant_id FROM users WHERE id = $1)
     ORDER BY created_at DESC LIMIT 1`,
    [userId]
  );

  const last = latest.rows[0];
  // If last is 'in' and this is 'in' within N minutes, ignore duplicate (idempotency already handled at event level)
  if (last && last.type === 'in' && type === 'in') {
    const diffMin = (ts.getTime() - new Date(last.created_at).getTime()) / 60000;
    if (diffMin < 5) return { action: 'ignored_duplicate_in', sessionId: last.id };
  }

  // If last is 'out' (or none) and this is 'out', ignore
  if (last && last.type === 'out' && type === 'out') {
    return { action: 'ignored_duplicate_out' };
  }

  // Create session
  const result = await db.query(
    `INSERT INTO attendance_sessions(user_id, type, clocked_at, location, tenant_id)
     VALUES ($1, $2, $3, $4, (SELECT tenant_id FROM users WHERE id = $1))
     RETURNING id`,
    [userId, type, ts, location]
  );

  return { action: 'created', sessionId: result.rows[0].id };
}

// Route handler
router.post(
  '/line',
  express.raw({ type: 'application/json' }), // preserve raw buffer for signature
  lineWebhookGuard(process.env.LINE_CHANNEL_SECRET || ''),
  async (req: Request & { lineEvents?: any[] }, res: Response) => {
    const events = req.lineEvents || [];
    const results = [];

    for (const ev of events) {
      try {
        const { type, source, message, timestamp } = ev;
        const userId = source?.userId;
        const ts = timestamp ? new Date(timestamp * 1000) : new Date();

        if (!userId) {
          logger.warn({ msg: 'line_event_no_user', ev });
          continue;
        }

        // Example: text command "in" / "out" (or parse payload from LIFF)
        if (type === 'message' && message?.type === 'text') {
          const text = message.text.trim().toLowerCase();
          if (text === 'in' || text === 'out') {
            const r = await upsertClockSession(userId, text === 'in' ? 'in' : 'out', ts);
            logger.info({ msg: 'clock_processed', userId, type: text, r });
            results.push(r);
          }
        }

        // TODO: handle leave/OT callback events, location stamps, etc.
      } catch (err) {
        logger.error({ msg: 'line_event_process_error', err, ev });
        // continue processing other events; do not throw to avoid blocking batch
      }
    }

    // Always 200 to stop LINE retries (idempotency handled above)
    res.status(200).json({ ok: true, processed: results.length });
  }
);

export default router;
```
