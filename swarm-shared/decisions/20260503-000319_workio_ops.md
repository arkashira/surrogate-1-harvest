## Final synthesized implementation (correct + actionable)

Use Candidate 1’s schema + constraint design and Candidate 2’s emphasis on client-provided idempotency and clear state-machine rules. The result is a small, safe, and race-proof change that can be deployed incrementally.

## 1. Diagnosis (resolved)
- Root cause: no server-side uniqueness, no idempotency, and no state machine → duplicates and invalid sequences (double-in/out-without-in).
- Fix priority:
  1. DB uniqueness to make duplicates impossible (source of truth).
  2. Server-side upsert + state machine to give clear errors and prevent invalid sequences.
  3. Client idempotency key to make retries safe across network/LINE retries.

## 2. Schema change (single source of truth)

File: `/opt/axentx/workio/server/src/db/schema.sql`

```sql
-- Ensure deterministic date column (if not present) and unique punch per tenant/user/date/type.
-- This prevents duplicates even under races and retries.

-- Add generated date column (if not already present)
ALTER TABLE punches
  ADD COLUMN IF NOT EXISTS punch_date DATE
  GENERATED ALWAYS AS (punched_at::date) STORED;

-- One 'in' and one 'out' per tenant/user/date (enforced by unique index)
CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_unique_tenant_user_date_type
  ON punches (tenant_id, user_id, punch_date, type)
  WHERE type IN ('in', 'out');

-- Fast lookup for state-machine checks
CREATE INDEX IF NOT EXISTS idx_punches_lookup_tenant_user_date_type
  ON punches (tenant_id, user_id, punch_date, type, punched_at DESC);
```

Notes:
- The partial unique index guarantees at DB level that duplicate `in` or duplicate `out` cannot exist for the same tenant/user/date.
- If you prefer no schema change, you can still use app-level upserts, but the index/constraint is strongly recommended to eliminate races.

## 3. Server-side handler (idempotent + state machine)

File: `/opt/axentx/workio/server/src/routes/punch.ts`

```ts
import { Request, Response } from 'express';
import { pool } from '../db';

export async function createPunch(req: Request, res: Response) {
  const { tenant_id, user_id, type, idempotency_key } = req.body; // type: 'in' | 'out'
  const punched_at = req.body.punched_at ? new Date(req.body.punched_at) : new Date();
  const punch_date = punched_at.toISOString().split('T')[0]; // YYYY-MM-DD

  if (!tenant_id || !user_id || !['in', 'out'].includes(type)) {
    return res.status(400).json({ error: 'Invalid payload' });
  }

  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // 1) Idempotency by natural key (fast path)
    const existing = await client.query(
      `SELECT id, punched_at, type
       FROM punches
       WHERE tenant_id = $1 AND user_id = $2 AND punch_date = $3 AND type = $4
       LIMIT 1`,
      [tenant_id, user_id, punch_date, type]
    );

    if (existing.rows.length > 0) {
      await client.query('COMMIT');
      return res.status(200).json({ punch: existing.rows[0], duplicate: true });
    }

    // 2) Optional: idempotency table for cross-retry safety (recommended)
    if (idempotency_key) {
      const idem = await client.query(
        `SELECT punch_id FROM punch_idempotency
         WHERE tenant_id = $1 AND idempotency_key = $2
         LIMIT 1`,
        [tenant_id, idempotency_key]
      );
      if (idem.rows.length > 0) {
        // Fetch and return the punch that was created for this idempotency key
        const p = await client.query(
          `SELECT id, punched_at, type FROM punches WHERE id = $1`,
          [idem.rows[0].punch_id]
        );
        await client.query('COMMIT');
        if (p.rows.length) {
          return res.status(200).json({ punch: p.rows[0], duplicate: true });
        }
      }
    }

    // 3) State-machine guards
    if (type === 'in') {
      const lastIn = await client.query(
        `SELECT id FROM punches
         WHERE tenant_id = $1 AND user_id = $2 AND punch_date = $3 AND type = 'in'
         ORDER BY punched_at DESC LIMIT 1`,
        [tenant_id, user_id, punch_date]
      );
      if (lastIn.rows.length > 0) {
        await client.query('ROLLBACK');
        return res.status(409).json({ error: 'Already clocked in for this date' });
      }
    } else {
      // type === 'out'
      const lastIn = await client.query(
        `SELECT id FROM punches
         WHERE tenant_id = $1 AND user_id = $2 AND punch_date = $3 AND type = 'in'
         ORDER BY punched_at DESC LIMIT 1`,
        [tenant_id, user_id, punch_date]
      );
      if (lastIn.rows.length === 0) {
        await client.query('ROLLBACK');
        return res.status(409).json({ error: 'No clock-in to clock out from' });
      }
      const lastOut = await client.query(
        `SELECT id FROM punches
         WHERE tenant_id = $1 AND user_id = $2 AND punch_date = $3 AND type = 'out'
         ORDER BY punched_at DESC LIMIT 1`,
        [tenant_id, user_id, punch_date]
      );
      if (lastOut.rows.length > 0) {
        await client.query('ROLLBACK');
        return res.status(409).json({ error: 'Already clocked out for this date' });
      }
    }

    // 4) Insert punch
    const insert = await client.query(
      `INSERT INTO punches (tenant_id, user_id, punched_at, type)
       VALUES ($1, $2, $3, $4)
       RETURNING id, tenant_id, user_id, punched_at, type`,
      [tenant_id, user_id, punched_at, type]
    );

    const punchId = insert.rows[0].id;

    // 5) Persist idempotency key if provided
    if (idempotency_key) {
      await client.query(
        `INSERT INTO punch_idempotency (tenant_id, idempotency_key, punch_id)
         VALUES ($1, $2, $3)`,
        [tenant_id, idempotency_key, punchId]
      );
    }

    await client.query('COMMIT');
    return res.status(201).json({ punch: insert.rows[0], duplicate: false });
  } catch (err: any) {
    await client.query('ROLLBACK');

    // Unique constraint violation (race) -> return existing
    if (err.code === '23505') {
      const existing = await client.query(
        `SELECT id, punched_at, type
         FROM punches
         WHERE tenant_id = $1 AND user_id = $2 AND punch_date = $3 AND type = $4
         LIMIT 1`,
        [tenant_id, user_id, punch_date, type]
      );
      if (existing.rows.length) {
        return res.status(200).json({ punch: existing.rows[0], duplicate: true });
      }
    }

    console.error('Punch error', err);
    return res.status(500).json({ error: 'Internal server error' });
  } finally {
    client.release();
  }
}
```

Optional idempotency table (run once):

```sql
CREATE TABLE IF NOT EXISTS punch_idempotency (
  tenant_id      BIGINT NOT NULL,
  idempotency_key TEXT NOT NULL,
  punch_id       BIGINT NOT NULL,
  created_at     TIMESTAMPTZ DEFAULT now(),
  PRIMARY KEY (tenant_id, idempotency_key)
);
```
