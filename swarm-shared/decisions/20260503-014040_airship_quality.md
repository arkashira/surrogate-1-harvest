### 1) Manifest Generator (run once per date folder)  
*Location*: `/opt/axentx/airship/surrogate/scripts/generate_manifest.py`  
*Why this version*: Uses `huggingface_hub` for reliability, filters to parquet, includes metadata, non-recursive single call (fast + avoids pagination cost), outputs to `data/` for clear separation.

```python
#!/usr/bin/env python3
"""
Generate CDN file manifest for a specific date folder.
Run on Mac orchestrator after rate-limit window clears.
"""
import json
import os
from datetime import datetime, timezone
from huggingface_hub import HfApi

HF_REPO = "axentx/surrogate-datasets"
DATE_FOLDER = datetime.now(timezone.utc).strftime("%Y-%m-%d")  # e.g., "2026-05-03"
OUTPUT_PATH = os.path.join(
    os.path.dirname(__file__), "..", "data", f"manifest-{DATE_FOLDER}.json"
)

def main() -> None:
    api = HfApi()
    # Single non-recursive call per folder (avoids pagination + rate pressure)
    files = api.list_repo_tree(
        repo_id=HF_REPO,
        path=DATE_FOLDER,
        recursive=False,
        repo_type="dataset",
    )
    # Keep only parquet and preserve relative path
    file_paths = [
        f"{DATE_FOLDER}/{f.rfilename}"
        for f in files
        if f.rfilename.endswith(".parquet")
    ]
    manifest = {
        "repo": HF_REPO,
        "date": DATE_FOLDER,
        "generated_at": datetime.now(timezone.utc).isoformat(),
        "files": file_paths,
        "total_files": len(file_paths),
    }
    os.makedirs(os.path.dirname(OUTPUT_PATH), exist_ok=True)
    with open(OUTPUT_PATH, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"✅ Manifest saved: {OUTPUT_PATH} ({len(file_paths)} files)")

if __name__ == "__main__":
    main()
```

### 2) CDN-Only Dataset Loader  
*Location*: `/opt/axentx/airship/surrogate/training/cdn_dataset.py`  
*Why this version*: Uses `pyarrow.parquet` for fast columnar reads, streams bytes via CDN (no auth, no API), projects schema to `{prompt, response}` for robustness, skips corrupt files, and keeps resource usage low.

```python
"""
CDN-only dataset loader for Surrogate training.
Zero HF API calls during training — bypasses 429 rate limits.
"""
import json
import pyarrow.parquet as pq
import requests
from pathlib import Path
from typing import Iterator, Dict, Any
import pyarrow as pa
import io

CDN_BASE = "https://huggingface.co/datasets"

class CDNDataset:
    def __init__(self, manifest_path: str):
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.repo = self.manifest["repo"]
        self.files = self.manifest["files"]

    def _download_file(self, file_path: str) -> bytes:
        """Download single file via CDN (no auth, no API rate limit)."""
        url = f"{CDN_BASE}/{self.repo}/resolve/main/{file_path}"
        resp = requests.get(url, timeout=30)
        resp.raise_for_status()
        return resp.content

    def _project_to_pair(self, table: pa.Table) -> Iterator[Dict[str, Any]]:
        """Project to {prompt, response} only — ignore mixed schema cols."""
        df = table.to_pandas()
        for _, row in df.iterrows():
            prompt = row.get("prompt") or row.get("instruction") or ""
            response = row.get("response") or row.get("output") or ""
            if prompt and response:
                yield {"prompt": str(prompt), "response": str(response)}

    def stream(self) -> Iterator[Dict[str, Any]]:
        """Stream examples via CDN with zero HF API usage."""
        for file_path in self.files:
            try:
                raw = self._download_file(file_path)
                table = pq.read_table(io.BytesIO(raw))
                yield from self._project_to_pair(table)
            except Exception as e:
                print(f"⚠️  Skipping {file_path}: {e}")
                continue

    def __len__(self) -> int:
        return self.manifest["total_files"]
```

### Key Design Decisions (Resolved Contradictions)
- **CDN-only vs hybrid**: Chose CDN-only during training to guarantee no 429s. Manifest generation uses a single API call on the orchestrator where rate limits are easier to manage.
- **Manifest scope**: Date-folder scoped, non-recursive, parquet-only — balances freshness, speed, and cost.
