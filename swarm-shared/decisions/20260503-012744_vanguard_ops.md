# vanguard / ops

## 1. Diagnosis

- Frontend still triggers authenticated `list_repo_tree` on page/training load, burning HF API quota (1000/5min) and causing 429s.
- No persisted `(repo, dateFolder)` → file-list mapping; every reload re-enumerates via API instead of embedding a static manifest.
- Training/data-loading paths likely use authenticated `/api/` endpoints instead of public CDN URLs (`resolve/main/...`), wasting rate-limit budget during data streaming.
- No guard to reuse existing Lightning Studio; training iterations likely recreate studios and burn 80hr/mo quota.
- Mac-side orchestration may still invoke model loading locally (`.from_pretrained`) instead of delegating training to Lightning Studio.

## 2. Proposed change

Create a lightweight ops manifest generator and CDN-aware loader for the surrogate-1 pipeline:

- Add: `/opt/axentx/vanguard/ops/generate_file_manifest.py`
- Add: `/opt/axentx/vanguard/ops/train_cdn_loader.py` (or patch existing train script)
- Update: any orchestration script that lists repo files to use one-time Mac-side enumeration and persist `manifest.json`.
- Update: Lightning launcher to reuse running studios and embed the manifest path.

Scope: single PR <2h — implement manifest generator + CDN loader stub and wire them into the training entrypoint.

# /opt/axentx/vanguard/ops/generate_file_manifest.py
#!/usr/bin/env bash
# Usage: generate_file_manifest.sh <repo> <date_folder> <out_json>
# Example: generate_file_manifest.sh axentx/surrogate-1 2026-04-29 ./manifest.json

set -euo pipefail
REPO="${1:-axentx/surrogate-1}"
DATEFOLDER="${2:-$(date +%Y-%m-%d)}"
OUT="${3:-./manifest.json}"

# Requires: huggingface_hub installed in Python env
python3 - "$REPO" "$DATEFOLDER" "$OUT" <<'PY'
import json
import os
import sys
from huggingface_hub import HfApi

def main(repo: str, date_folder: str, out_path: str):
    api = HfApi()
    # Single non-recursive call per date folder (avoids 100x pagination)
    entries = api.list_repo_tree(repo=repo, path=date_folder, recursive=False)
    files = []
    for e in entries:
        if not e.path.endswith("/"):  # skip subdirs
            files.append({
                "repo": repo,
                "path": e.path,
                "cdn_url": f"https://huggingface.co/datasets/{repo}/resolve/main/{e.path}"
            })
    manifest = {
        "repo": repo,
        "date_folder": date_folder,
        "generated_by": "generate_file_manifest",
        "files": files
    }
    os.makedirs(os.path.dirname(os.path.abspath(out_path)) or ".", exist_ok=True)
    with open(out_path, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Wrote {len(files)} files to {out_path}")

if __name__ == "__main__":
    main(sys.argv[1], sys.argv[2], sys.argv[3])
PY

# /opt/axentx/vanguard/ops/train_cdn_loader.py
import json
import os
from pathlib import Path
from typing import Iterator, Tuple

import requests

def load_manifest(manifest_path: str) -> dict:
    with open(manifest_path) as f:
        return json.load(f)

def cdn_stream(manifest_path: str, max_files: int = -1) -> Iterator[Tuple[str, bytes]]:
    """Yield (path, raw_bytes) using CDN URLs (no HF API auth)."""
    manifest = load_manifest(manifest_path)
    count = 0
    for fmeta in manifest["files"]:
        if max_files > 0 and count >= max_files:
            break
        resp = requests.get(fmeta["cdn_url"], timeout=30)
        resp.raise_for_status()
        yield fmeta["path"], resp.content
        count += 1

# /opt/axentx/vanguard/ops/lightning_launcher.py
import os
from lightning.pytorch import Trainer
from lightning.pytorch.studio import Studio, Teamspace
from pathlib import Path

def reuse_or_create_studio(name: str, machine: str = "L40S"):
    teamspace = Teamspace()
    for s in teamspace.studios:
        if s.name == name and s.status == "Running":
            print(f"Reusing running studio: {name}")
            return s
    print(f"Creating studio: {name}")
    return Studio(
        name=name,
        machine=machine,
        create_ok=True
    )

def run_training_with_manifest(
    script_path: str,
    manifest_path: str,
    studio_name: str = "vanguard-surrogate-train"
):
    studio = reuse_or_create_studio(studio_name)
    if studio.status != "Running":
        studio.start(machine="L40S")
    # Pass manifest as an arg/env to the training script so it uses CDN-only fetches
    studio.run(
        script=script_path,
        arguments=[f"--manifest={manifest_path}"],
        env={
            "MANIFEST_PATH": manifest_path,
            "SHELL": "/bin/bash"
        }
    )
