# workio / design

## Final Synthesis (adopted)

Use Candidate 1 as the primary implementation, strengthened by Candidate 2’s emphasis on testability, ordering, and separation of concerns.

Key decisions:
- **Signature verification is mandatory** — reject 401 immediately; prevents spoofing and stops retries on bad requests.
- **Idempotency is table-based with event_id + payload_hash** — protects against replay and race conditions; return 200 after commit so LINE stops retrying.
- **Attendance state machine is strict and atomic** — validate transitions (clock-in → clock-out) and enforce one clock-in per day; out-of-order clock-outs are rejected, not silently fixed.
- **Separate pure validation from side-effects** — `validateAttendanceTransition` is synchronous and testable; `processAttendanceEvent` orchestrates DB writes in a transaction.
- **Raw-body handling is scoped to the LINE route only** — avoids breaking other JSON routes.

CREATE TABLE IF NOT EXISTS line_webhook_events (
  event_id      TEXT PRIMARY KEY,

## 2. Route: LINE webhook (raw-body safe)

File: `/opt/axentx/workio/server/src/routes/webhook/line.ts`

```ts
import { Router, Request, Response, NextFunction } from 'express';
import crypto from 'crypto';
import { pool } from '../../db';
import {
  validateAttendanceTransition,
  processAttendanceEvent,
} from '../../services/attendanceService';

const router = Router();
const LINE_CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';

function verifyLineSignature(rawBody: Buffer, signature: string): boolean {
  if (!LINE_CHANNEL_SECRET) return false;
  const expected = crypto
    .createHmac('sha256', LINE_CHANNEL_SECRET)
    .update(rawBody)
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

function buildEventId(ev: any): string {
  // Prefer stable source+timestamp+type; fallback to content hash
  if (ev.source?.userId && ev.timestamp && ev.type) {
    return `${ev.source.userId}:${ev.timestamp}:${ev.type}`;
  }
  return crypto.createHash('sha256').update(JSON.stringify(ev)).digest('hex');
}

router.post(
  '/line',
  async (req: Request, res: Response, next: NextFunction) => {
    const signature = req.headers['x-line-signature'] as string;
    const rawBody = req.body as Buffer;

    if (!verifyLineSignature(rawBody, signature)) {
      return res.status(401).json({ error: 'Invalid signature' });
    }

    let events;
    try {
      events = JSON.parse(rawBody.toString('utf8')).events || [];
    } catch {
      return res.status(400).json({ error: 'Invalid JSON' });
    }

    const client = await pool.connect();
    try {
      await client.query('BEGIN');

      for (const ev of events) {
        const eventId = buildEventId(ev);
        const payloadHash = crypto
          .createHash('sha256')
          .update(JSON.stringify(ev))
          .digest('hex');

        const already = await client.query(
          `SELECT 1 FROM line_webhook_events WHERE event_id = $1`,
          [eventId]
        );
        if (already.rowCount > 0) {
          continue; // idempotent skip
        }

        if (ev.type === 'message' && ev.message?.type === 'text') {
          const validation = validateAttendanceTransition(ev);
          if (!validation.valid) {
            // Client error: don't retry this event, but commit idempotency record
            await client.query(
              `INSERT INTO line_webhook_events (event_id, payload_hash, processed_at)
               VALUES ($1, $2, NOW()) ON CONFLICT (event_id) DO NOTHING`,
              [eventId, payloadHash]
            );
            await client.query('COMMIT');
            client.release();
            return res.status(400).json({ error: validation.reason });
          }

          await processAttendanceEvent(client, ev);
        }

        await client.query(
          `INSERT INTO line_webhook_events (event_id, payload_hash, processed_at)
           VALUES ($1, $2, NOW()) ON CONFLICT (event_id) DO NOTHING`,
          [eventId, payloadHash]
        );
      }

      await client.query('COMMIT');
      return res.status(200).json({ ok: true });
    } catch (err) {
      await client.query('ROLLBACK');
      console.error('LINE webhook processing failed', err);
      // 4xx for client-side issues (no retry), 5xx for server issues (retry)
      const status = err instanceof Error && err.message.includes('validation') ? 400 : 500;
      return res.status(status).json({ error: 'Processing failed' });
    } finally {
      client.release();
    }
  }
);

export default router;
```

## 3. Service: attendance logic (atomic + testable)

File: `/opt/axentx/workio/server/src/services/attendanceService.ts`

```ts
import { PoolClient } from 'pg';

export type TransitionValidation = {
  valid: boolean;
  reason?: string;
};

export function validateAttendanceTransition(ev: any): TransitionValidation {
  const userId = ev.source?.userId;
  const text = (ev.message?.text || '').trim().toLowerCase();

  if (!userId) {
    return { valid: false, reason: 'Missing user' };
  }
  if (!['clock in', 'clock out', 'เข้างาน', 'เลิกงาน'].includes(text)) {
    return { valid: false, reason: 'Unrecognized command' };
  }
  return { valid: true };
}

export async function processAttendanceEvent(
  tx: PoolClient,
  ev: any
): Promise<void> {
  const userId = ev.source?.userId;
  const text = (ev.message?.text || '').trim().toLowerCase();
  const occurredAt = new Date(ev.timestamp);

  if (!userId) return;

  const emp = await tx.query(
    `SELECT id FROM employees WHERE line_user_id = $1`,
    [userId]
  );
  if (emp.rowCount === 0) return;
  const employeeId = emp.rows[0].id;

  const today = new Date().toISOString().split('T')[0];

  if (text === 'clock in' || text === 'เข้างาน') {
    const dup = await tx.query(
      `SELECT 1 FROM attendances
       WHERE employee_id = $1 AND date = $2 AND clock_in IS NOT NULL`,
      [employeeId, today]
    );
    if (dup.rowCount > 0) return;

    await tx.query(
      `INSERT INTO attendances (employee_id, date, clock_in, created_at)
       VALUES ($1, $2, $3, NOW())
       ON CONFLICT (employee_id, date)
       DO UPDATE SET clock_in = EXCLUDED.clock_in, updated_at = NOW()`,
      [employeeId, today, occurredAt]
    );
  } else if (text === 'clock out' || text === 'เลิกงาน') {
    const row = await tx.query(
      `SELECT id, clock_in, clock_out FROM attendances
       WHERE employee_id = $1 AND date = $2`,
      [employeeId, today]
    );
    if (row.rowCount === 0) return;

    const record = row.rows[0];
    if (!record.clock_in) return; // no clock-in to close
    if (record.clock_out) return; // already clocked out

    await tx.query(
      `UPDATE attendances
       SET clock_out = $1, updated_at = NOW()
       WHERE id = $2`,
      [occurredAt, record.id]
    );
