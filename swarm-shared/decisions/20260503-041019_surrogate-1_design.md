# surrogate-1 / design

## Final Implementation Plan (≤2h)

**Highest-value improvement**: Replace fragile shell-based ingestion with a **manifest-driven, CDN-bypass Python worker** that eliminates HF API rate limits during training data loads and prevents mixed-schema CastErrors.

### Changes

1. **Add `bin/generate_manifest.py`** — one-time Mac/CI step that calls `list_repo_tree` on today’s date folder and writes `manifest.json` (path, sha, size). This manifest is uploaded as a workflow artifact for all matrix jobs.

2. **Add `bin/worker.py`** — deterministic shard worker that:
   - Reads `manifest.json` and selects its 1/N slice via stable hash-based sharding.
   - Downloads files directly via raw CDN URLs (`https://huggingface.co/datasets/.../resolve/main/...`) — zero HF API calls during ingest.
   - Projects each file to `{prompt, response}` at parse time (avoids `load_dataset` mixed-schema CastError).
   - Streams output to `batches/public-merged/{date}/shard{N}-{HHMMSS}.jsonl`.
   - Uses central `lib/dedup.py` for cross-shard md5 dedup.

3. **Update `bin/dataset-enrich.sh`** → thin wrapper that:
   - Accepts `SHARD_ID`/`SHARD_TOTAL` env vars (from matrix).
   - Invokes `python bin/worker.py` with proper error handling.
   - Keeps Bash shebang and `set -euo pipefail`.

4. **Update `.github/workflows/ingest.yml`** to:
   - Add a **pre-step** (runs once) that generates `manifest.json` for today’s folder via a single HF API call (after rate-limit window) and uploads it as an artifact for all matrix jobs.
   - Pass `SHARD_ID`/`SHARD_TOTAL` to each matrix job.
   - Ensure `SHELL=/bin/bash` in any cron context.

5. **Keep requirements minimal**: `requests`, `pyarrow`, `numpy`, `huggingface_hub` (only for initial manifest + upload).

### Code snippets

#### `bin/generate_manifest.py`
```python
#!/usr/bin/env python3
"""
Generate manifest.json for today's folder in axentx/surrogate-1-training-pairs.
Usage: python bin/generate_manifest.py
"""
import json
import os
import datetime
from pathlib import Path

from huggingface_hub import HfApi

HF_DATASET = "axentx/surrogate-1-training-pairs"

def main() -> None:
    api = HfApi()
    today = datetime.datetime.utcnow().strftime("%Y-%m-%d")

    tree = api.list_repo_tree(
        repo_id=HF_DATASET,
        repo_type="dataset",
        path=today,
    )
    files = [
        {"path": f.path, "sha": f.lfs.get("sha256", ""), "size": f.size}
        for f in tree
        if f.type == "file"
    ]

    manifest = {"date": today, "files": files}
    out_path = Path("manifest.json")
    out_path.write_text(json.dumps(manifest, indent=2), encoding="utf-8")
    print(f"Wrote manifest with {len(files)} files to {out_path}")

if __name__ == "__main__":
    main()
```

#### `bin/worker.py`
```python
#!/usr/bin/env python3
"""
CDN-bypass ingest worker for surrogate-1 public dataset.
Usage: SHARD_ID=0 SHARD_TOTAL=16 python bin/worker.py
"""
import json
import os
import sys
import hashlib
import datetime
from pathlib import Path
from typing import Dict, Any

import requests
import pyarrow as pa
import pyarrow.parquet as pq

HF_DATASET = "axentx/surrogate-1-training-pairs"
BASE_CDN = f"https://huggingface.co/datasets/{HF_DATASET}/resolve/main"

# Local imports
sys.path.insert(0, str(Path(__file__).parent))
from lib.dedup import DedupStore  # noqa

def load_manifest(manifest_path: str) -> Dict[str, Any]:
    with open(manifest_path) as f:
        return json.load(f)

def project_to_pair(raw_bytes: bytes, file_path: str) -> Dict[str, str]:
    """
    Project heterogeneous file to {prompt, response} only.
    Supports: jsonl, parquet, json.
    """
    suffix = Path(file_path).suffix.lower()
    if suffix == ".parquet":
        tbl = pq.read_table(pa.BufferReader(raw_bytes))
        cols = tbl.column_names
        prompt_col = next((c for c in ("prompt", "instruction", "input") if c in cols), None)
        response_col = next((c for c in ("response", "output", "completion") if c in cols), None)
        if prompt_col is None or response_col is None:
            raise ValueError(f"Cannot map pair columns in {file_path}")
        # Stream row-by-row in production; simplified single-row example
        prompt = str(tbl[prompt_col][0].as_py())
        response = str(tbl[response_col][0].as_py())
        return {"prompt": prompt, "response": response}

    elif suffix == ".jsonl":
        lines = raw_bytes.decode("utf-8").strip().splitlines()
        if not lines:
            raise ValueError("Empty jsonl")
        obj = json.loads(lines[0])
        prompt = obj.get("prompt") or obj.get("instruction") or obj.get("input")
        response = obj.get("response") or obj.get("output") or obj.get("completion")
        if prompt is None or response is None:
            raise ValueError(f"Missing pair fields in {file_path}")
        return {"prompt": str(prompt), "response": str(response)}

    elif suffix == ".json":
        obj = json.loads(raw_bytes)
        if isinstance(obj, list):
            obj = obj[0]
        prompt = obj.get("prompt") or obj.get("instruction") or obj.get("input")
        response = obj.get("response") or obj.get("output") or obj.get("completion")
        if prompt is None or response is None:
            raise ValueError(f"Missing pair fields in {file_path}")
        return {"prompt": str(prompt), "response": str(response)}

    else:
        raise ValueError(f"Unsupported file type: {suffix}")

def cdn_download(url: str) -> bytes:
    # No Authorization header -> bypasses HF API rate limits
    resp = requests.get(url, timeout=60)
    resp.raise_for_status()
    return resp.content

def run_shard(manifest: Dict[str, Any], shard_id: int, shard_total: int) -> str:
    files = manifest.get("files", [])
    if not files:
        raise ValueError("Manifest contains no files")

    # Deterministic shard assignment by path hash
    shard_files = [
        f["path"]
        for f in files
        if (int(hashlib.md5(f["path"].encode()).hexdigest(), 16) % shard_total) == shard_id
    ]

    today = datetime.datetime.utcnow().strftime("%Y-%m-%d")
    ts = datetime.datetime.utcnow().strftime("%H%M%S")
    out_name = f"shard{shard_id}-{ts}.jsonl"
    out_dir = Path("batches") / "public-merged" / today
    out_dir.mkdir(parents=True, exist_ok=True)
    out_path = out_dir / out_name

    dedup = DedupStore()
    written = 0

    with out_path.open("w", encoding="utf-8") as out_f:
        for rel_path in shard_files:
            try:
                cdn_url = f"{BASE_CDN}/{rel_path}"
                raw = cdn_download(cdn_url)
                pair = project_to_pair(raw, rel_path)
                payload = json.dumps(pair, ensure_ascii=False)
                digest = hashlib.md5(payload.encode()).hexdigest()
                if dedup.is_duplicate(digest):
                    continue
                dedup.add(digest)
                out_f.write(payload + "\n")
                written += 1
            except Exception as exc:
                # Log and continue to avoid shard failure
                print(f"WARN: skip {rel_path}: {exc}", file=sys.stderr)
                continue

    print(f"Shard {shard_id} wrote {written} pairs to {out_path}")
    return str(out_path
