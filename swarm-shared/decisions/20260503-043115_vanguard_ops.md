- No deterministic CDN-only fetch path; scripts rely on `load_dataset`/`list_repo_files` which hit auth/rate limits.
- Missing reproducible snapshot prevents safe resume/restart and causes non-reproducible epochs.
- No lightweight orchestration wrapper to pre-list once and embed the file list into training (zero API calls during data load).
- No recovery from idle-stop studios causing lost progress when Lightning idle timeout kills jobs.

**2. Single proposed change**
Add an ops-first workflow under `/opt/axentx/vanguard/ops/` that produces a content-addressed manifest per date and a CDN-only Lightning training entrypoint, plus a resilient run wrapper. Scope: new files only; no changes to existing training logic yet. This enables the CDN-only path immediately and cuts HF API usage to a single pre-list per date.

`/opt/axentx/vanguard/ops/build_manifest.py`
```python
#!/usr/bin/env python3
"""
Generate a content-addressed manifest for one date folder.

Usage:
  python ops/build_manifest.py \
    --repo datasets/axentx/surrogate-1 \
    --date 2026-04-29 \
    --out manifests/2026-04-29/files.json
"""
import argparse
import hashlib
import json
import sys
from pathlib import Path

from huggingface_hub import list_repo_tree

CDN_ROOT = "https://huggingface.co/datasets"

def build_manifest(repo: str, date: str, out_path: Path) -> None:
    prefix = f"{date}/"
    entries = []

    # One API call: non-recursive list for the date folder
    tree = list_repo_tree(repo=repo, path=prefix, recursive=False)

    for item in tree:
        if item.type != "file":
            continue
        path = item.path  # e.g. 2026-04-29/slug.parquet
        url = f"{CDN_ROOT}/{repo}/resolve/main/{path}"
        # Prefer LFS sha256 when available; otherwise leave empty (can be filled later if needed)
        sha256 = getattr(item, "lfs", {}).get("sha256", "")
        entries.append({
            "path": path,
            "sha256": sha256,
            "size": item.size,
            "url": url,
        })

    out = {
        "repo": repo,
        "date": date,
        "created_by": "ops/build_manifest.py",
        "files": entries,
    }

    out_path.parent.mkdir(parents=True, exist_ok=True)
    with open(out_path, "w", encoding="utf-8") as f:
        json.dump(out, f, indent=2)
    print(f"Wrote {len(entries)} files -> {out_path}")

if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True)
    parser.add_argument("--date", required=True)
    parser.add_argument("--out", required=True, type=Path)
    args = parser.parse_args()
    build_manifest(args.repo, args.date, args.out)
```

#!/usr/bin/env python3
"""
Lightning Studio entrypoint (CDN-only).
Expects manifest at /data/manifest.json and outputs model artifacts.
"""
import json
from pathlib import Path

import pyarrow as pa
import pyarrow.parquet as pq
import requests
import torch
from torch.utils.data import IterableDataset, DataLoader
from lightning import LightningModule, Trainer

CDN_TIMEOUT = 30

class CDNParquetDataset(IterableDataset):
    def __init__(self, manifest_path, columns=("prompt", "response")):
        manifest_path = Path(manifest_path)
        with open(manifest_path) as f:
            manifest = json.load(f)
        self.files = [f["url"] for f in manifest["files"]]
        self.columns = columns

    def __iter__(self):
        for url in self.files:
            resp = requests.get(url, timeout=CDN_TIMEOUT)
            resp.raise_for_status()
            table = pq.read_table(pa.BufferReader(resp.content))
            table = table.select(self.columns)
            for batch in table.to_batches(max_chunksize=512):
                cols = {c: batch.column(c).to_pylist() for c in self.columns}
                for i in range(len(cols[self.columns[0]])):
                    yield {k: cols[k][i] for k in self.columns}

class SurrogateModel(LightningModule):
    def __init__(self):
        super().__init__()
        self.net = torch.nn.Linear(1024, 1024)  # placeholder

    def training_step(self, batch, batch_idx):
        # Minimal placeholder; replace with real model/tokenizer logic
        loss = torch.tensor(0.0, requires_grad=True)
        self.log("train_loss", loss)
        return loss

    def configure_optimizers(self):
        return torch.optim.Adam(self.parameters(), lr=1e-3)

if __name__ == "__main__":
    dataset = CDNParquetDataset("/data/manifest.json")
    loader = DataLoader(dataset, batch_size=8, num_workers=0)
    model = SurrogateModel()
    trainer = Trainer(max_epochs=1, accelerator="gpu", devices=1)
    trainer.fit(model, loader)

`/opt/axentx/vanguard/ops/run_training.sh`
```bash
#!/usr/bin/env bash
set -euo pipefail

# Resilient wrapper to handle idle-stop studios and resume training.
# Usage: ./ops/run_training.sh --date 2026-04-29

MANIFEST_DIR="/opt/axentx/vanguard/ops/manifests"
REPO="datasets/axentx/surrogate-1"
DATE=""

while [[ $# -gt 0 ]]; do
  case "$1" in
    --date) DATE="$2"; shift ;;
    *) echo "Unknown option: $1"; exit 1 ;;
  esac
  shift
done

if [[ -z "$DATE" ]]; then
  echo "Error: --date is required"
  exit 1
fi

MANIFEST="${MANIFEST_DIR}/${DATE}/files.json"

# 1) Build manifest if missing (single HF API call)
if [[ ! -f "$MANIFEST" ]]; then
  echo "Building manifest for ${DATE}..."
  python /opt/axentx/vanguard/ops/build_manifest.py \
    --repo "$REPO" \
    --date "$DATE" \
    --out "$MANIFEST"
fi

# 2) Studio lifecycle: reuse if running; otherwise create L40S studio.
#    On idle-stop, re-create and resume by re-launching this script.
STUDIO_NAME="vanguard-${DATE}"
if ! lightning studio list --name "$STUDIO_NAME" | grep -q "RUNNING"; then
  echo "Starting studio ${STUDIO_NAME} (L40S)..."
  lightning studio create --name "$STUDIO_NAME" --machine L40S --idle-timeout 3600
fi

# 3) Upload manifest + training script and run.
lightning studio cp "$MANIFEST" "${STUDIO_NAME}:/data/manifest.json"
lightning studio cp /opt/axentx/vanguard/ops/train_cdn.py "${STUDIO_NAME}:/data/train_cdn.py"
lightning studio run "${STUDIO_NAME}" -- python /data/train_cdn.py

# If studio was idle-stopped, this script will exit; an external orchestrator
# (or rerun) will re-create the studio and re-execute this wrapper to resume.
```
