# workio / quality

## Final Synthesized Solution

### Diagnosis (merged, de-duplicated)
- **No idempotency**: frontend retries and backend replays create duplicate punches for the same `(employee, punch_type, time_window)`.
- **No durable queue**: `localStorage` + naive `fetch` loses punches on network failure, page reload, or visibility/page-hide events.
- **No retry/backoff**: transient failures become permanent data loss.
- **No server-side guard**: missing unique constraint and idempotency-key enforcement allows duplicate inserts.
- **No observability**: missing metrics/logs to detect submission failure spikes.

```sql
ALTER TABLE attendance
  ADD COLUMN IF NOT EXISTS idempotency_key VARCHAR(255),
  ADD COLUMN IF NOT EXISTS created_at TIMESTAMPTZ DEFAULT NOW();

-- 5-minute uniqueness window per employee per punch_type (IN/OUT)
CREATE UNIQUE INDEX IF NOT EXISTS uq_employee_punch_window
  ON attendance (employee_id, date_trunc('hour', punch_time), (EXTRACT(minute FROM punch_time)::int / 5), punch_type)
  WHERE deleted_at IS NULL;

-- Idempotency ledger (cross-endpoint safety)
CREATE TABLE IF NOT EXISTS idempotency_keys (
  key         VARCHAR(255) PRIMARY KEY,
  response    JSONB NOT NULL,
  created_at  TIMESTAMPTZ DEFAULT NOW()
);
```

```ts
// server/src/middleware/idempotency.ts
import type { Request, Response, NextFunction } from 'express';
import { db } from '../db';

const TTL_MS = 24 * 60 * 60 * 1000; // 24h

export async function idempotencyMiddleware(req: Request, res: Response, next: NextFunction) {
  const key = req.header('Idempotency-Key');
  if (!key || req.method !== 'POST') return next();

  const existing = await db.query('SELECT response FROM idempotency_keys WHERE key = $1', [key]);
  if (existing.rowCount && existing.rowCount > 0) {
    return res.status(200).json(existing.rows[0].response);
  }

  // Monkey-patch res.json to store response on success
  const originalJson = res.json.bind(res);
  res.json = (body) => {
    if (res.statusCode >= 200 && res.statusCode < 300) {
      db.query(
        'INSERT INTO idempotency_keys(key, response) VALUES ($1, $2) ON CONFLICT DO NOTHING',
        [key, JSON.stringify(body)]
      ).catch(() => {});
    }
    return originalJson(body);
  };

  next();
}
```

```ts
// server/src/routes/attendance.ts
import express from 'express';
import { idempotencyMiddleware } from '../middleware/idempotency';
import { db } from '../db';

const router = express.Router();
router.use(idempotencyMiddleware);

router.post('/attendance/punch', async (req, res) => {
  const { employeeId, type, timestamp, location } = req.body;
  if (!employeeId || !type || !timestamp) return res.status(400).json({ error: 'Missing fields' });

  try {
    const result = await db.query(
      `INSERT INTO attendance(employee_id, punch_type, punch_time, location, idempotency_key)
       VALUES ($1, $2, $3, $4, $5)
       RETURNING id`,
      [employeeId, type, timestamp, location, req.header('Idempotency-Key') || null]
    );
    res.status(201).json({ id: result.rows[0].id });
  } catch (err: any) {
    // Unique violation from window constraint => treat as success (idempotent)
    if (err.code === '23505' && err.constraint === 'uq_employee_punch_window') {
      const existing = await db.query(
        'SELECT id FROM attendance WHERE employee_id = $1 AND punch_type = $2 AND punch_time = $3 LIMIT 1',
        [employeeId, type, timestamp]
      );
      return res.status(200).json({ id: existing.rows[0]?.id, duplicate: true });
    }
    res.status(500).json({ error: 'Server error' });
  }
});
```

#### 2) Frontend: durable queue + retry + observability

```ts
// src/lib/idbQueue.ts
import { openDB, DBSchema, IDBPDatabase } from 'idb';

interface AttendanceDB extends DBSchema {
  outbox: {
    key: string;
    value: {
      key: string;
      payload: { employeeId: string; type: 'IN' | 'OUT'; timestamp: string; location?: GeolocationCoordinates };
      attempts: number;
      createdAt: number;
    };
  };
}

const DB_NAME = 'workio-attendance';
const OUTBOX_STORE = 'outbox';

export async function getDB(): Promise<IDBPDatabase<AttendanceDB>> {
  return openDB<AttendanceDB>(DB_NAME, 1, {
    upgrade(db) {
      db.createObjectStore(OUTBOX_STORE);
    },
  });
}

export async function enqueueAttendance(
  payload: { employeeId: string; type: 'IN' | 'OUT'; timestamp: string; location?: GeolocationCoordinates }
) {
  const key = `${payload.employeeId}-${payload.type}-${payload.timestamp}`;
  const db = await getDB();
  await db.put(OUTBOX_STORE, { key, payload, attempts: 0, createdAt: Date.now() }, key);
  return key;
}

export async function* drainQueue(): AsyncGenerator<{ key: string; value: AttendanceDB['outbox']['value'] }> {
  const db = await getDB();
  const tx = db.transaction(OUTBOX_STORE, 'readonly');
  const store = tx.objectStore(OUTBOX_STORE);
  const keys = await store.getAllKeys();
  for (const key of keys) {
    const value = await store.get(key);
    if (value) yield { key, value };
  }
}

export async function removeFromQueue(key: string) {
  const db = await getDB();
  await db.delete(OUTBOX_STORE, key);
}

export async function updateAttempts(key: string, attempts: number) {
  const db = await getDB();
  const value = await db.get(OUTBOX_STORE, key);
  if (value) await db.put(OUTBOX_STORE, { ...value, attempts }, key);
}
```

```ts
// src/features/attendance/api/attendanceApi.ts
import axios from 'axios';
import { removeFromQueue, updateAttempts } from '../../lib/idbQueue';
import { captureMetric } from '../../lib/metrics';

const api = axios.create({ baseURL: '/api', timeout: 10000 });

export async function sendPunch(payload: { employeeId: string; type: 'IN' | 'OUT'; timestamp: string; location?: GeolocationCoordinates }, idempotencyKey: string) {
  try {
    const res = await api.post('/attendance/punch', payload, {
      headers: { 'Idempotency-Key': idempotencyKey },
    });
    await removeFromQueue(idempotencyKey);
    captureMetric('attendance.submit.success', 1, { type: payload.type });
    return { ok: true, duplicate: res.data.duplicate === true };
  } catch (err: any) {
    const entry = await (await import('../../lib/idbQueue')).getDB()
      .then((db) => db.get('outbox', idempotencyKey));
    if (entry) await updateAttempts(idempotencyKey, entry.attempts + 1);
    captureMetric('attendance.submit.failure', 1, { type: payload.type, status: err?.response?.status || 'network' });
    throw err;
  }
}
```
