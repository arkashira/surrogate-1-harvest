### 1. HF CDN-bypass loader (no auth, project at parse)
`surrogate/ui/src/lib/hf-cdn-loader.js`
```js
/**
 * HF CDN-bypass dataset loader
 * - Uses public CDN URLs (no Authorization header) to bypass /api/ rate limits
 * - Expects fileList to be pre-listed and embedded (Mac orchestration)
 * - Projects each file to {prompt, response} only at parse time
 */
export async function loadDatasetCdn({ repo, fileList, projectFn = defaultProject }) {
  const base = `https://huggingface.co/datasets/${repo}/resolve/main`;
  const records = [];

  for (const file of fileList) {
    const url = `${base}/${file.path}`;
    try {
      const res = await fetch(url, { cache: 'no-cache' });
      if (!res.ok) throw new Error(`CDN fetch failed: ${res.status} ${file.path}`);

      // Supports .jsonl, .parquet via streaming projection
      if (file.path.endsWith('.jsonl')) {
        const text = await res.text();
        for (const line of text.split('\n').filter(Boolean)) {
          const parsed = JSON.parse(line);
          const projected = projectFn(parsed);
          if (projected) records.push(projected);
        }
      } else if (file.path.endsWith('.parquet')) {
        // Lightweight browser parquet reader (e.g., apache-arrow) can be used here.
        // For now, skip parquet in UI; rely on server-side parquet ingestion.
        console.warn('Parquet skipped in browser loader; use server ingestion for', file.path);
      } else {
        console.warn('Unsupported file type:', file.path);
      }
    } catch (err) {
      console.error('Failed to load', file.path, err);
      // Continue with remaining files (best-effust)
    }
  }

  return records;
}

function defaultProject(obj) {
  // Project to {prompt, response} only; ignore mixed schema fields
  const prompt = obj.prompt ?? obj.input ?? obj.question ?? null;
  const response = obj.response ?? obj.output ?? obj.answer ?? null;
  if (prompt == null || response == null) return null;
  return { prompt, response };
}
```

### 2. Lightning Studio reuse + idle restart
`surrogate/ui/src/lib/lightning-studio.js`
```js
import { Lightning } from '@lightningai/lightning-sdk'; // adjust import to actual SDK

const STUDIO_NAME = 'surrogate-training-studio';
const MACHINE = 'lightning-lambda-prod/L40S'; // prefer H200-capable account

export async function getOrCreateStudio({ apiKey, project }) {
  const teamspace = Lightning.teamspace({ apiKey });

  // Reuse running studio
  const studios = await teamspace.studios();
  const running = studios.find(s => s.name === STUDIO_NAME && s.status === 'Running');
  if (running) {
    console.log('Reusing running studio:', running.id);
    return running;
  }

  // Restart if stopped
  const stopped = studios.find(s => s.name === STUDIO_NAME && s.status === 'Stopped');
  if (stopped) {
    console.log('Restarting stopped studio:', stopped.id);
    await stopped.start({ machine: MACHINE });
    return stopped;
  }

  // Create new
  console.log('Creating new studio:', STUDIO_NAME);
  return teamspace.studio(STUDIO_NAME, { create_ok: true }).start({ machine: MACHINE });
}

export async function runInStudio(studio, scriptPath, args = []) {
  // Guard against idle timeout death
  const status = await studio.status();
  if (status !== 'Running') {
    console.warn('Studio not running, restarting...');
    await studio.start({ machine: MACHINE });
  }

  return studio.run(scriptPath, args);
}
```

### 3. Wire into TrainingPanel (example Vue)
`surrogate/ui/src/components/TrainingPanel.vue` (excerpt)
```vue
<script setup>
import { loadDatasetCdn } from '../lib/hf-cdn-loader';
import { getOrCreateStudio, runInStudio } from '../lib/lightning-studio';
import { ref } from 'vue';

const props = defineProps({
  repo: { type: String, required: true },
  fileList: { type: Array, required: true }, // pre-listed from Mac orchestration
  apiKey: { type: String, required: true }
});

const loading = ref(false);
const logs = ref([]);

async function startTraining() {
  loading.value = true;
  logs.value.push('Loading dataset via CDN bypass...');

  try {
    const records = await loadDatasetCdn({
      repo: props.repo,
      fileList: props.fileList
    });
    logs.value.push(`Loaded ${records.length} records`);

    logs.value.push('Acquiring Lightning Studio...');
    const studio = await getOrCreateStudio({ apiKey: props.apiKey, project: 'surrogate' });

    logs.value.push('Submitting training run...');
    const run = await runInStudio(studio, 'train.py', [
      '--data', JSON.stringify(records),
      '--epochs', '3'
    ]);

    logs.value.push(`Run submitted: ${run.id}`);
  } catch (err) {
    logs.value.push(`Error: ${err.message}`);
    console.error(err);
  } finally {
    loading.value = false;
  }
}
</script>

<template>
  <div class="training-panel">
    <button @click="startTraining" :disabled="loading">
      {{ loading ? 'Starting...' : 'Start Training (CDN bypass)' }}
    </button>
    <pre class="logs">{{ logs.join('\n') }}</pre>
  </div>
</template>
```

## Notes & Fallbacks
- **File list embedding**: Mac orchestration should run `list_repo_tree` once per date folder and save to `public/file-list-{date}.json`. The UI fetches that JSON before training (zero HF API calls during load).  
- **Parquet in browser**: For large parquet files, keep ingestion server-side (HF Space) and use CDN URLs for direct download; browser loader focuses on JSONL.  
- **Lightning quota**: Reuse prevents 80hr/mo waste from repeated studio creation.  
- **Idle timeout**: Studio auto-stops kill training; the restart guard mitigates this.
