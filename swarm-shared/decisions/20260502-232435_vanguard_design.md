# vanguard / design

## Final Synthesis (Corrected + Actionable)

**Core problem:**  
- No persistent, CDN-only file index for dataset ingestion → repeated HF API calls (429s) and quota burn.  
- Training assumes homogeneous schema and uses `load_dataset` on mixed repos → runtime `pyarrow.CastError`.  
- Lightning Studio lifecycle is recreated each run instead of reused → quota waste and idle-stop deaths.  
- Data loading still hits `/api/` endpoints instead of public CDN URLs.

**Single goal:** Shift all dataset I/O to CDN-only, cache the file list once, enforce schema at read time, and reuse Lightning Studio across runs.

## 1) make_index.py — generate CDN-only index (run on Mac)

```python
#!/usr/bin/env python3
"""
Generate a CDN-ready file index for one date folder.
Usage:
    python make_index.py --repo datasets/surrogate-1 --date 2026-04-29 --out index_2026-04-29.json
"""
import argparse
import json
from pathlib import Path

try:
    from huggingface_hub import HfApi
except ImportError:
    raise SystemExit("pip install huggingface_hub")

HF_API = HfApi()

def build_index(repo: str, date: str):
    prefix = f"{date}/"
    # Single-level listing to minimize quota use
    items = HF_API.list_repo_tree(repo, path=prefix, recursive=False)
    files = []
    for item in items:
        if isinstance(item, dict) and item.get("type") == "file":
            name = item["path"]
            files.append({
                "repo": repo,
                "path": name,
                "cdn_url": f"https://huggingface.co/datasets/{repo}/resolve/main/{name}"
            })
    return sorted(files, key=lambda x: x["path"])

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True)
    parser.add_argument("--date", required=True)
    parser.add_argument("--out", required=True)
    args = parser.parse_args()

    index = build_index(args.repo, args.date)
    out_path = Path(args.out)
    out_path.write_text(json.dumps(index, indent=2))
    print(f"Wrote {len(index)} files -> {out_path}")

if __name__ == "__main__":
    main()
```

"""
CDN-only training entrypoint.
Usage:
    python train.py --index index_2026-04-29.json --epochs 1
"""
import argparse
import json
import torch
from torch.utils.data import IterableDataset, DataLoader
import requests
import pyarrow.parquet as pq
from io import BytesIO

class CDNParquetDataset(IterableDataset):
    def __init__(self, index_path, columns=("prompt", "response")):
        self.index_path = index_path
        self.columns = columns
        with open(index_path) as f:
            self.files = json.load(f)

    def __iter__(self):
        for meta in self.files:
            url = meta["cdn_url"]
            resp = requests.get(url, timeout=60)
            resp.raise_for_status()
            table = pq.read_table(BytesIO(resp.content), columns=self.columns)
            for row in table.to_pylist():
                # Enforce schema and skip malformed rows
                prompt = row.get("prompt")
                response = row.get("response")
                if prompt is None or response is None:

## 3) studio_launcher.py — reuse studios and restart on idle-stop

```python
#!/usr/bin/env python3
"""
Lightning Studio lifecycle helper.
Reuse running studios; restart if stopped (idle-timeout safe).
"""
from lightning import LightningWork, LightningFlow, LightningApp, Machine

class SurrogateTrainer(LightningWork):
    def __init__(self, index_path: str, **kwargs):
        super().__init__(**kwargs)
        self.index_path = index_path
        self.machine = Machine.L40S  # adjust as needed
        self._launched = False

    def run(self, index_path: str):
        # Import here to avoid requiring lightning in local dev
        from lightning.pytorch import Trainer
        from lightning.pytorch.demos.boring_classes import BoringModel
        # Replace with real model + CDN dataset
        model = BoringModel()
        trainer = Trainer(max_epochs=1, limit_train_batches=2)
        trainer.fit(model)

    def configure_app(self) -> None:
        pass

class VanguardFlow(LightningFlow):
    def __init__(self):
        super().__init__()
        self.trainer = SurrogateTrainer(
            index_path="index_2026-04-29.json",
            cloud_compute=Machine.L40S
        )

    def run(self):
        if not self.trainer.has_started:
            self.trainer.start()
            return

        # If studio stopped (idle timeout), restart safely
        if self.trainer.has_stopped:
            print("Studio stopped (likely idle). Restarting...")
            self.trainer.stop()
            self.trainer.start()
            return

        # Drive training once
        if not self.trainer._launched:
            self.trainer.run(self.trainer.index_path)
            self.trainer._launched = True

app = LightningApp(VanguardFlow())
```
