### 1. Pre-list HF file paths (run once from Mac)
`/opt/axentx/airship/scripts/list_hf_files.py`

```python
#!/usr/bin/env python3
"""
Run after rate-limit windows clear.
Lists files for ONE date folder (non-recursive per folder)
and saves a JSON manifest for embedding in training.
"""
import json
import os
from datetime import datetime
from huggingface_hub import HfApi

HF_REPO = "axentx/surrogate-datasets"
DATE_FOLDER = datetime.now().strftime("%Y-%m-%d")
SCRIPT_DIR = os.path.dirname(os.path.abspath(__file__))
DATA_DIR = os.path.join(SCRIPT_DIR, "..", "data")
OUTPUT_JSON = os.path.join(DATA_DIR, f"filelist_{DATE_FOLDER}.json")

def list_files() -> list[str]:
    api = HfApi()
    # Top-level items (non-recursive)
    tree = api.list_repo_tree(repo_id=HF_REPO, path=DATE_FOLDER, recursive=False)

    all_files: list[str] = []
    for item in tree:
        if item.type == "file":
            all_files.append(item.path)
        elif item.type == "directory":
            # One call per subfolder to stay within quota
            sub_tree = api.list_repo_tree(
                repo_id=HF_REPO, path=item.path, recursive=False
            )
            all_files.extend(f.path for f in sub_tree if f.type == "file")

    os.makedirs(DATA_DIR, exist_ok=True)
    with open(OUTPUT_JSON, "w", encoding="utf-8") as f:
        json.dump({"date": DATE_FOLDER, "files": sorted(all_files)}, f, indent=2)

    print(f"Saved {len(all_files)} files to {OUTPUT_JSON}")
    return all_files

if __name__ == "__main__":
    list_files()
```

### 2. CDN-only Data Loader (embedded manifest)
`/opt/axentx/airship/surrogate/data/cdn_dataset.py`

```python
import json
import os
from pathlib import Path
from typing import Tuple

import requests
import pyarrow.parquet as pq
from torch.utils.data import Dataset, DataLoader
from io import BytesIO

HF_DATASET_REPO = "axentx/surrogate-datasets"
CDN_BASE = f"https://huggingface.co/datasets/{HF_DATASET_REPO}/resolve/main"

class CDNParquetDataset(Dataset):
    """
    Bypasses HF API entirely.
    Uses CDN URLs (no Authorization header) to avoid 429s.
    """

    def __init__(self, manifest_path: str | None = None):
        if manifest_path is None:
            data_dir = Path(__file__).parent.parent / "data"
            candidates = sorted(
                data_dir.glob("filelist_*.json"),
                key=lambda p: p.stat().st_mtime,
                reverse=True,
            )
            manifest_path = str(candidates[0]) if candidates else None

        if manifest_path and os.path.exists(manifest_path):
            with open(manifest_path, encoding="utf-8") as f:
                manifest = json.load(f)
            self.file_paths: list[str] = manifest["files"]
        else:
            # Minimal fallback for dev
            self.file_paths = ["2026-05-03/batch-001.parquet"]

    def __len__(self) -> int:
        return len(self.file_paths)

    def __getitem__(self, idx: int) -> Tuple[str, str]:
        cdn_url = f"{CDN_BASE}/{self.file_paths[idx]}"
        resp = requests.get(cdn_url, timeout=30)
        resp.raise_for_status()

        table = pq.read_table(BytesIO(resp.content))
        # Project only required columns at parse time
        df = table.select(["prompt", "response"]).to_pandas()
        return df.iloc[0]["prompt"], df.iloc[0]["response"]


def get_dataloader(batch_size: int = 8, num_workers: int = 0) -> DataLoader:
    dataset = CDNParquetDataset()
    return DataLoader(
        dataset,
        batch_size=batch_size,
        num_workers=num_workers,
        shuffle=True,
        drop_last=True,
    )
```

### 3. Lightning Studio Manager (reuse + idle-resilient)
`/opt/axentx/airship/surrogate/lightning_manager.py`

```python
#!/usr/bin/env python3
"""
Lightning Studio reuse + idle-resilient launcher.
Reuses running studios and restarts stopped/idle ones.
"""
import os
import time
from pathlib import Path

from lightning_sdk import Studio, Teamspace
from lightning_sdk.workspace import StudioStatus

TEAMSPACE = "axentx"
STUDIO_NAME = "surrogate-trainer"
MACHINE = "L40S"  # Falls back to public tier if unavailable


def get_or_create_studio() -> Studio:
    teamspace = Teamspace(name=TEAMSPACE)

    for studio in teamspace.studios:
        if studio.name == STUDIO_NAME:
            if studio.status == StudioStatus.RUNNING:
                print(f"Reusing running studio: {STUDIO_NAME}")
                return studio
            if studio.status == StudioStatus.STOPPED:
                print(f"Restarting stopped studio: {STUDIO_NAME}")
                studio.start(machine=MACHINE)
                return studio

    print(f"Creating new studio: {STUDIO_NAME}")
    return Studio.create(
        name=STUDIO_NAME,
        teamspace=TEAMSPACE,
        machine=MACHINE,
        create_ok=True,
    )


def run_training_with_retry(script_path: str | Path, max_retries: int = 3):
    studio = get_or_create_studio()

    for attempt in range(1, max_retries + 1):
        try:
            if studio.status != StudioStatus.RUNNING:
                print(f"Studio not running (status={studio.status}), restarting...")
                studio.start(machine=MACHINE)
                time.sleep(60)

            result = studio.run(
                str(script_path),
                environment="surrogate-training",
                wait=True,
                stream_logs=True,
            )

            if result.exit_code == 0:
                print("Training completed successfully")
                return result

            print(f"Training failed (exit {result.exit_code}), retry {attempt}/{max_retries}")

        except Exception as exc:  # noqa: BLE001
            print(f"Attempt {attempt} failed: {exc}")

        if attempt < max_retries:
            time.sleep(300)
            studio = get_or_create_studio()

    raise RuntimeError("Training failed after all retries")


if __name__ == "__main__":
    train_script = Path(__file__).parent / "train.py"
    run_training_with_retry(train_script)
```
