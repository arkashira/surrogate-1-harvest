- Uses a **single `list_repo_tree` snapshot** (JSON manifest) generated once per date on the Mac orchestrator and committed to the repo (or passed via `file-list.json`).

#### 1) Manifest generator (run on Mac) — `bin/file-list-snapshot.py`

```python
#!/usr/bin/env python3
"""
Generate a file-list snapshot for a date folder in axentx/surrogate-1-training-pairs.
Usage:
  python bin/file-list-snapshot.py --date 20260503 --out file-list-20260503.json
"""
import argparse
import json
import os
from huggingface_hub import HfApi

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--date", required=True, help="Date folder, e.g. 20260503")
    parser.add_argument("--out", required=True, help="Output JSON path")
    parser.add_argument("--repo", default="axentx/surrogate-1-training-pairs")
    args = parser.parse_args()

    api = HfApi()
    # Single non-recursive call per date folder; rely on CDN for file downloads
    tree = api.list_repo_tree(repo_id=args.repo, path=args.date, recursive=False)

    entries = []
    for item in tree:
        if item.type == "file":
            entries.append({
                "path": f"{args.date}/{item.path}",
                "size": getattr(item, "size", None)
            })

    snapshot = {
        "date": args.date,
        "repo": args.repo,
        "generated_by": "file-list-snapshot.py",
        "files": entries
    }

    with open(args.out, "w", encoding="utf-8") as f:
        json.dump(snapshot, f, indent=2)
    print(f"Wrote {len(entries)} files to {args.out}")

if __name__ == "__main__":
    main()
```

def shard_filter(path: str, shard_id: int, total_shards: int = 16) -> bool:
    """Deterministic shard assignment by path hash."""
    h = int(hashlib.md5(path.encode("utf-8")).hexdigest(), 16)
    return (h % total_shards) == shard_id

def parse_file_to_pair(cdn_url: str, path: str) -> dict | None:
    """
    Download via CDN and project heterogeneous files to {prompt, response}.
    Supports common formats: jsonl, parquet, txt (heuristic).
    """
    try:
        resp = requests.get(cdn_url, timeout=30)
        resp.raise_for_status()
    except Exception as exc:
        print(f"[WARN] Failed to fetch {cdn_url}: {exc}", file=sys.stderr)
        return None

    suffix = Path(path).suffix.lower()
    content = resp.content

    try:
        if suffix == ".parquet":
            tbl = pq.read_table(pa.BufferReader(content))
            # Project to prompt/response heuristically
            cols = set(tbl.column_names)
            prompt_col = next((c for c in ("prompt", "instruction", "input") if c in cols), None)
            response_col = next((c for c in ("response", "output", "completion") if c in cols), None)
            if prompt_col and response_col and tbl.num_rows > 0:
                return {
                    "prompt": str(tbl[prompt_col][0].as_py()),
                    "response": str(tbl[response_col][0].as_py()),
                    "source_file": path
                }

        elif suffix == ".jsonl":
            lines = content.decode("utf-8", errors="replace").strip().splitlines()
            if not lines:
                return None
            import json as jsonlib
            obj = jsonlib.loads(lines[0])
            prompt = obj.get("prompt") or obj.get("instruction") or obj.get("input")
            response = obj.get("response") or obj.get("output") or obj.get("completion")
            if prompt and response:
                return {"prompt": str(prompt), "response": str(response), "source_file": path}

        else:
            # Fallback: treat as text; heuristic split by double newline or Q/A markers
            text = content.decode("utf-8", errors="replace").strip()
            if "\n\n" in text:
                parts = text.split("\n\n", 1)
                return {"prompt": parts[0].strip(), "response": parts[1].strip(), "source_file": path}
    except Exception as exc:
        print(f"[WARN] Parse error for {path}: {exc}", file=sys.stderr)

    return None

def upload_shard(output_dir: Path, shard_id: int, pairs: list[dict]):
