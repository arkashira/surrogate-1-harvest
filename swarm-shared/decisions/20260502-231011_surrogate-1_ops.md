### scripts/orchestrate_snapshot.sh
```bash
#!/usr/bin/env bash
set -euo pipefail

DATASET_REPO="${1:-axentx/surrogate-1-training-pairs}"
DATE_FOLDER="${2:-public-merged/$(date +%Y-%m-%d)}"
OUT_JSON="${3:-snapshots/snapshot-$(date +%Y%m%d).json}"

mkdir -p "$(dirname "$OUT_JSON")"

python3 - "$DATASET_REPO" "$DATE_FOLDER" "$OUT_JSON" <<'PY'
import json, os, sys, time
from huggingface_hub import HfApi

repo, date_folder = sys.argv[1], sys.argv[2]
out_path = sys.argv[3]

api = HfApi()
# One snapshot call; handle 429 with simple retry
for attempt in range(3):
    try:
        tree = api.list_repo_tree(repo_id=repo, path=date_folder, recursive=False)
        break
    except Exception as e:
        if attempt == 2:
            raise
        time.sleep(60 * (attempt + 1))

files = [
    {
        "path": f.path,
        "size": f.size or 0,
        "sha": (f.lfs or {}).get("oid", "") if f.lfs else ""
    }
    for f in tree
]

snapshot = {"date": date_folder, "files": files, "ts": time.time()}
with open(out_path, "w") as f:
    json.dump(snapshot, f)
print(f"Snapshot saved: {out_path} ({len(files)} files)")
PY
```

### bin/dataset-enrich.sh (CDN-first + deterministic shard)
```bash
#!/usr/bin/env bash
set -euo pipefail

DATASET_REPO="${DATASET_REPO:-axentx/surrogate-1-training-pairs}"
SNAPSHOT="${SNAPSHOT:-snapshots/snapshot-$(date +%Y%m%d).json}"
SHARD_ID="${SHARD_ID:-0}"
N_SHARDS="${N_SHARDS:-16}"

download_cdn() {
  local rel_path="$1" out="$2"
  # Retry/backoff for CDN robustness
  for i in 1 2 3; do
    if curl -L -f -s -S --retry 2 --retry-delay 1 \
         "https://huggingface.co/datasets/${DATASET_REPO}/resolve/main/${rel_path}" \
         -o "${out}"; then
      return 0
    fi
    sleep $((i * 2))
  done
  return 1
}

hash_to_shard() {
  local slug="$1" n="${2:-16}"
  # deterministic, language-agnostic modulo
  python3 -c "import mmh3, sys; print(mmh3.hash(sys.argv[1]) % int(sys.argv[2]))" "$slug" "$n"
}

process_file() {
  local f="$1" slug="$2"
  local my_shard
  my_shard=$(hash_to_shard "$slug" "$N_SHARDS")
  if [ "$my_shard" -ne "$SHARD_ID" ]; then
    return 0  # skip; not this worker's shard
  fi

  local tmp
  tmp=$(mktemp)
  if download_cdn "$f" "$tmp"; then
    python3 parse_and_project.py "$tmp"
  else
    # fallback to hf_hub_download only on CDN failure
    python3 -c "
from huggingface_hub import hf_hub_download
hf_hub_download(repo_id='$DATASET_REPO', filename='$f', local_dir='.cache')
"
    python3 parse_and_project.py ".cache/$f"
  fi
  rm -f "$tmp"
}

# export SHARD_ID=0 N_SHARDS=16
# jq -r '.files[].path' "$SNAPSHOT" | while read -r f; do process_file "$f" "some-slug"; done

### train.py (CDN-only IterableDataset + projection)
```python
import json, requests, pyarrow as pa, pyarrow.parquet as pq, io, os
from torch.utils.data import IterableDataset
from typing import Iterator

class CDNIterableDataset(IterableDataset):
    def __init__(self, snapshot_path: str, base_url: str, max_concurrent: int = 16):
        with open(snapshot_path) as f:
            self.files = json.load(f)["files"]
        self.base_url = base_url.rstrip("/")
        self.max_concurrent = max_concurrent

    def _stream_parquet_to_pairs(self, url: str) -> Iterator[dict]:
        # Stream parquet and project to {prompt,response}; adjust column names as needed
        resp = requests.get(url, stream=True, timeout=30)
        resp.raise_for_status()
        buf = io.BytesIO(resp.content)
        table = pq.read_table(buf, columns=["prompt", "response"])
        for row in table.to_pylist():
            yield {"prompt": row.get("prompt"), "response": row.get("response")}

    def __iter__(self):
        for item in self.files:
            url = f"{self.base_url}/{item['path']}"
            try:
                yield from self._stream_parquet_to_pairs(url)
            except Exception as e:
                # Log and skip; optionally fallback to hf_hub_download here
                print(f"Failed to process {url}: {e}")
                continue

### Lightning Studio reuse + deterministic repo selection
```python
from lightning import Teamspace, Studio, Machine
import mmh3

def get_or_create_studio(name: str = "surrogate-1-train"):
    for s in Teamspace.studios():
        if s.name == name and s.status == "running":
            return s
    return Studio.create(name=name, machine=Machine.L40S, script="train.py")

def pick_repo(slug: str, n_siblings: int = 5) -> str:
    idx = mmh3.hash(slug) % n_siblings
    return f"axentx/surrogate-1-training-pairs-shard{idx}"
