# vanguard / ops

## 1. Diagnosis
- Missing HF CDN-bypass: no `file-list.json` for datasets → training scripts fall back to `load_dataset`/HF API calls and hit 429s on heterogeneous repos.
- Non-idempotent Lightning Studio lifecycle: orchestration scripts create new studios instead of reusing running ones, wasting quota and risking idle-stop training death.
- No guard against Lightning idle-stop: training dies if studio stops; no pre-run status check or restart path.
- Mac=CLI rule violation risk: orchestration may accidentally trigger local model loading instead of delegating to Lightning/Kaggle/Cerebras.
- Missing deterministic HF sibling-repo sharding: ingestion writes to single repo and will hit 128 commits/hr cap.

## 2. Proposed change
File scope: `/opt/axentx/vanguard/ops/train.py` (create if absent) and `/opt/axentx/vanguard/ops/file_list.json` (generated artifact).  
Add:
- Idempotent Lightning Studio reuse + idle-stop guard.
- HF CDN-bypass via pre-listed `file_list.json` and a DataLoader that fetches only via CDN URLs.
- Deterministic HF sibling repo selector (hash-slug → repo index).
- Mac-safe guard that refuses `model.from_pretrained()` locally.

# ---- HF sibling repo sharding ----
def pick_sibling_repo(slug: str) -> str:
    """Deterministic repo selection from HF_SIBLINGS."""
    idx = int(hashlib.sha256(slug.encode()).hexdigest(), 16) % len(HF_SIBLINGS)
    return HF_SIBLINGS[idx]

# ---- HF CDN-bypass utilities ----
def build_cdn_urls(file_list: List[Dict[str, Any]]) -> List[str]:
    """Convert repo-relative paths into public CDN URLs (no auth)."""
    base = f"https://huggingface.co/datasets/{HF_DATASET_REPO}/resolve/main"
    urls = []
    for entry in file_list:
        path = entry.get("path") or entry.get("file") or entry.get("name")
        if not path:
            continue
        urls.append(f"{base}/{path.lstrip('/')}")
    return urls

def load_file_list() -> List[Dict[str, Any]]:
    """Load pre-listed file list generated on Mac after rate-limit window."""
    if not FILE_LIST_PATH.exists():
        raise FileNotFoundError(
            f"{FILE_LIST_PATH} missing. Run list_hf_files.py on Mac first "
            f"to generate file_list.json (single API call per date folder)."
        )
    with open(FILE_LIST_PATH, "r", encoding="utf-8") as f:
        data = json.load(f)
    if isinstance(data, dict) and "files" in data:
        return data["files"]
    if isinstance(data, list):
        return data
    raise ValueError("file_list.json must be a list or dict with 'files' key")

def get_or_create_studio(
    name: str = "vanguard-surrogate1",
    machine: str = "L40S",
    cloud: str = "lightning-public-prod",
) -> Any:
    """
    Reuse a running studio if present; otherwise create one.
    Returns a Studio-like object with .run() capability.
    """
    if L is None:
        raise RuntimeError("lightning not available; cannot manage Studio")

    teamspace = L.Teamspace()
    for studio in teamspace.studios:
        if studio.name == name and studio.status == "running":
            print(f"Reusing running studio: {name}")
            return studio

    print(f"No running studio '{name}' found. Creating one...")
    # Avoid recreating if one exists but is stopped: restart instead.
    for studio in teamspace.studios:
        if studio.name == name and studio.status == "stopped":
            print(f"Restarting stopped studio: {name}")
            # Lightning may require explicit machine selection on restart
            studio.start(machine=machine, cloud=cloud)
            return studio

    # Create new studio
    studio = L.Studio(
        name=name,
        cloud=cloud,
        machine=machine,
        create_ok=True,
    )
    return studio

def ensure_studio_alive(studio: Any) -> Any:
    """Guard against Lightning idle-stop killing training."""
    if getattr(studio, "status", None) == "stopped":
        print("Studio stopped (likely idle timeout). Restarting...")
        studio.start(machine=getattr(studio, "machine", "L40S"))
    return studio

# ---- Minimal CDN-only dataset iterable ----
class CDNParquetIterable:
    """
    Lightweight iterable over HF CDN parquet files.
    Uses zero HF API calls during training.
    Projects only {prompt, response} at parse time.
    """
    def __init__(self, urls: List[str], columns: List[str] = ("prompt", "response")):
        self.urls = urls
        self.columns = columns

    def __iter__(self):
        import pyarrow.parquet as pq
        import requests
        import io

        for url in self.urls:
            resp = requests.get(url, timeout=60)
            resp.raise_for_status()
            table = pq.read_table(io.BytesIO(resp.content))
            # Keep only required columns; ignore heterogeneous extras
            available = [c for c in self.columns if c in table.column_names]
            if not available:
                continue
            batch = table.select(available).to_pydict()
            # Yield per-row dicts
            n = len(next(iter(batch.values()))) if batch else 0
            for i in range(n):
                yield {k: batch[k][i] for k in available}
