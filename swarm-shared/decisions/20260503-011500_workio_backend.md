# workio / backend

## Final Synthesized Solution

### 1. Diagnosis (merged + prioritized)
- **Race on clock-in + missing DB constraint** — concurrent/retried LINE webhooks can insert duplicate open punches; app-layer checks are not safe under concurrency.
- **No idempotency for webhook deliveries** — LINE retries on timeouts create duplicate events; without idempotency, duplicates hit business logic and cause “already clocked in” errors.
- **Missing signature verification** — `/webhook/line` accepts unverified requests; spoofed events can corrupt attendance.
- **Notification failures roll back punches** — sending LINE replies inside the same transaction causes punch loss when notify fails (e.g., user blocked bot).
- **Hardcoded/single-tenant webhook path** — no tenant resolution from headers or path; multi-tenant deployments break.
- **No safe retry/backoff for outbound LINE** — transient failures are not retried, and permanent failures block punches.

### 2. DB Changes (schema + constraints)

```sql
-- 1) Idempotency table (deduplicate by LINE event + tenant)
CREATE TABLE IF NOT EXISTS line_webhook_idempotency (
  tenant_id      UUID NOT NULL REFERENCES tenants(id) ON DELETE CASCADE,
  line_event_id  TEXT NOT NULL,
  processed_at   TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  PRIMARY KEY (tenant_id, line_event_id)
);

-- 2) Partial unique index: at most one open punch per user per calendar day (Asia/Bangkok)
-- Uses date in tenant TZ to avoid boundary races vs UTC.
CREATE UNIQUE INDEX IF NOT EXISTS punches_one_open_per_day
ON punches (tenant_id, user_id, DATE(clock_in_at AT TIME ZONE 'Asia/Bangkok'))
WHERE clock_out_at IS NULL;

-- 3) Ensure clock_out_after_clock_in
ALTER TABLE punches
  ADD CONSTRAINT chk_clock_out_after_clock_in
  CHECK (clock_out_at IS NULL OR clock_out_at >= clock_in_at);
```

### 3. Signature Verification Middleware

```ts
// server/src/middleware/line-signature.ts
import crypto from 'crypto';
import { Request, Response, NextFunction } from 'express';

export function lineSignature(channelSecret: string, toleranceMs = 300_000) {
  return (req: Request, _res: Response, next: NextFunction) => {
    const signature = req.headers['x-line-signature'] as string | undefined;
    if (!signature) return next(new Error('Missing X-Line-Signature'));

    const body = JSON.stringify(req.body);
    const expected = crypto
      .createHmac('SHA256', channelSecret)
      .update(body, 'utf8')
      .digest('base64');

    if (!crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected))) {
      return next(new Error('Invalid signature'));
    }

    // Optional replay/stale protection
    const timestamp = req.headers['x-line-request-time'];
    if (timestamp && typeof timestamp === 'string') {
      const ts = parseInt(timestamp, 10);
      if (!Number.isNaN(ts) && Math.abs(Date.now() - ts) > toleranceMs) {
        return next(new Error('Stale request'));
      }
    }

    next();
  };
}
```

### 4. Punch Service (idempotent + concurrency-safe)

```ts
// server/src/services/punch.service.ts
import { PoolClient, QueryResult } from 'pg';
import { pool } from '../db';

export interface PunchInput {
  tenantId: string;
  userId: string;
  lineEventId: string;
  type: 'CLOCK_IN' | 'CLOCK_OUT';
  latitude?: number;
  longitude?: number;
  imageUrl?: string;
}

export async function upsertPunch(input: PunchInput, client?: PoolClient) {
  const db = client ?? pool;

  // Idempotency: deduplicate by LINE event + tenant
  const idem = await db.query(
    `INSERT INTO line_webhook_idempotency (tenant_id, line_event_id)
     VALUES ($1, $2)
     ON CONFLICT (tenant_id, line_event_id) DO NOTHING
     RETURNING line_event_id`,
    [input.tenantId, input.lineEventId]
  );

  if (idem.rowCount === 0) {
    // Already processed — return current open punch (best-effort)
    const existing = await db.query(
      `SELECT id, type, clock_in_at, clock_out_at
       FROM punches
       WHERE tenant_id = $1 AND user_id = $2
         AND clock_in_at >= (DATE_TRUNC('day', NOW() AT TIME ZONE 'Asia/Bangkok'))
         AND clock_in_at <  (DATE_TRUNC('day', NOW() AT TIME ZONE 'Asia/Bangkok') + INTERVAL '1 day')
       ORDER BY clock_in_at DESC LIMIT 1`,
      [input.tenantId, input.userId]
    );
    return existing.rows[0] ?? null;
  }

  // Try to close an open punch for CLOCK_OUT
  if (input.type === 'CLOCK_OUT') {
    const close = await db.query(
      `UPDATE punches
       SET clock_out_at = NOW(), updated_at = NOW()
       WHERE tenant_id = $1 AND user_id = $2
         AND clock_out_at IS NULL
         AND clock_in_at >= (DATE_TRUNC('day', NOW() AT TIME ZONE 'Asia/Bangkok'))
         AND clock_in_at <  (DATE_TRUNC('day', NOW() AT TIME ZONE 'Asia/Bangkok') + INTERVAL '1 day')
       RETURNING *`,
      [input.tenantId, input.userId]
    );
    if (close.rowCount && close.rowCount > 0) return close.rows[0];
    return null; // nothing to close
  }

  // CLOCK_IN: attempt insert; partial unique index prevents duplicates
  const insert = await db.query(
    `INSERT INTO punches (tenant_id, user_id, clock_in_at, latitude, longitude, image_url)
     VALUES ($1, $2, NOW(), $3, $4, $5)
     RETURNING *`,
    [input.tenantId, input.userId, input.latitude, input.longitude, input.imageUrl]
  ).catch((err) => {
    // Transform unique violation into graceful null (already clocked in)
    if (err.code === '23505') {
      // Return current open punch
      return db.query(
        `SELECT id, type, clock_in_at, clock_out_at
         FROM punches
         WHERE tenant_id = $1 AND user_id = $2
           AND clock_out_at IS NULL
           AND clock_in_at >= (DATE_TRUNC('day', NOW() AT TIME ZONE 'Asia/Bangkok'))
           AND clock_in_at <  (DATE_TRUNC('day', NOW() AT TIME ZONE 'Asia/Bangkok') + INTERVAL '1 day')
         ORDER BY clock_in_at DESC LIMIT 1`,
        [input.tenantId, input.userId]
      ) as Promise<QueryResult<any>>;
    }
    throw err;
  });

  return insert.rows[0] ?? null;
}
```
