## Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16` (matrix) and optional `DATE_FOLDER` (defaults to today `YYYY-MM-DD`).
- Uses a **single pre-listed manifest** (`manifest/<DATE_FOLDER>.json`) produced by the Mac orchestrator (or generated on first run) containing all file paths under that date folder. This avoids recursive `list_repo_files` and HF API rate limits.
- Each shard deterministically hashes `slug` → `bucket = hash(slug) % SHARD_TOTAL` and processes only its slice.
- Downloads files **via HF CDN** (`https://huggingface.co/datasets/axentx/surrogate-1-training-pairs/resolve/main/<path>`) with no Authorization header — bypasses `/api/` rate limits entirely.
- Projects each file to `{prompt, response}` at parse time (no schema assumptions), computes `md5` for dedup, and streams output to `batches/public-merged/<DATE_FOLDER>/shard<SHARD_ID>-<HHMMSS>.jsonl`.
- Uses the existing `lib/dedup.py` central SQLite store for cross-source dedup (or falls back to local dedup if unavailable).
- Runs in GitHub Actions matrix (16 shards) with isolated 7 GB RAM per runner — no OOM.

def get_manifest() -> List[str]:
    """Return list of file paths for DATE_FOLDER."""
    manifest_path = os.getenv("MANIFEST_PATH")
    if manifest_path and Path(manifest_path).exists():
        with open(manifest_path) as f:
            data = json.load(f)
            if isinstance(data, dict) and "files" in data:
                return [p for p in data["files"] if p.startswith(DATE_FOLDER)]
            if isinstance(data, list):
                return [p for p in data if p.startswith(DATE_FOLDER)]

    # Fallback: single API call (non-recursive) for the date folder
    if not HF_TOKEN:
        raise RuntimeError("HF_TOKEN required to list repo tree when no manifest provided")
    headers = {"Authorization": f"Bearer {HF_TOKEN}"}
    url = f"https://huggingface.co/api/datasets/{HF_DATASET_REPO}/tree"
    params = {"path": DATE_FOLDER, "recursive": False}
    r = requests.get(url, headers=headers, params=params, timeout=30)
    if r.status_code == 429:
        print("Rate limited (429). Wait 360s before retry.", file=sys.stderr)
        sys.exit(1)
    r.raise_for_status()
    items = r.json()
    files = []
    for item in items:
        if item.get("type") == "file":
            files.append(f"{DATE_FOLDER}/{item['path']}")
        elif item.get("type") == "directory":
            # one-level deeper non-recursive fetch (keeps API calls low)
            sub = requests.get(
                url, headers=headers, params={"path": f"{DATE_FOLDER}/{item['path']}", "recursive": False}, timeout=30
            )
            if sub.status_code == 200:
                for si in sub.json():
                    if si.get("type") == "file":
                        files.append(f"{DATE_FOLDER}/{item['path']}/{si['path']}")
    return files

def extract_pairs(content: bytes, filename: str) -> List[Dict[str, str]]:
    """Project arbitrary HF parquet/jsonl to {prompt,response}."""
    pairs = []
    try:
        # Try parquet first
        tbl = pq.read_table(pa.BufferReader(content))
        df = tbl.to_pandas()
    except Exception:
        # Try JSON lines
        try:
            text = content.decode("utf-8")
            lines = [ln.strip() for ln in text.splitlines() if ln.strip()]
            import json as jsonlib
            rows = [jsonlib.loads(ln) for ln in lines]
            df = pa.Table.from_pylist(rows).to_pandas()
        except Exception:
            print(f"Could not decode {filename} as parquet/jsonl", file=sys.stderr)
            return pairs

    # Normalize column names
    col_map = {}
    for c in df.columns:
        cl = c.strip().lower()
        if "prompt" in cl or "question" in cl or "instruction" in cl:
            col_map[c] = "prompt"
        elif "response" in cl or "answer" in cl or "completion" in cl:
            col_map[c] = "response"

    prompt_col = None
    response_col = None
    for c, mapped in col_map.items():
        if mapped == "prompt" and prompt_col is None:
            prompt_col = c
        if mapped == "response" and response_col is None:
            response_col = c

    if prompt_col is None or response_col is None:
        # fallback: first text-like column as prompt, second as response
        text_cols = [c for c in df.columns if df[c].dtype == "object"]
        if len(text_cols) >= 2:
            prompt_col, response_col = text_cols[0], text_cols[1]
        else:
            print(f"Not enough text columns in {filename}", file=sys.stderr)
            return pairs

    for _, row in df.iterrows():
        p = str(row.get(prompt_col, "")).strip()
        r = str(row.get(response_col, "")).strip()
        if p and r:
            pairs.append({"prompt": p, "response": r})
    return pairs
