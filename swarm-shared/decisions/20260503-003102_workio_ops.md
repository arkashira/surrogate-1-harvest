# workio / ops

## 1) Diagnosis

- No idempotency key on `/punches` write path → repeated LINE webhook deliveries or frontend retries create duplicate punch records.
- Missing DB uniqueness constraint for “one open punch per user” → allows multiple concurrent clock-ins or overlapping open punches.
- Non-transactional punch creation → partial writes (punch created but audit/notification state not committed) under concurrency.
- No retry/backoff on LINE API calls from webhook handler → transient network failures create phantom “missed” punches.
- Missing operational observability (request-id, timing, outcome) on webhook endpoints → hard to debug duplicates or failures in production.

## 2) Proposed change

File scope:  
- `workio/server/src/routes/webhook/line.ts` (or equivalent) — add idempotency + transaction + retry.  
- `workio/server/src/db/schema.sql` — add unique constraint for open punches.  
- `workio/server/src/db/migrations/` (if migrations exist) — add migration for constraint and idempotency table.

## 3) Implementation

### 3.1 DB: enforce one open punch per user + idempotency table

```sql
-- workio/server/src/db/schema.sql  (or new migration file)

-- Idempotency table for LINE webhook events
CREATE TABLE IF NOT EXISTS webhook_idempotency (
  idempotency_key VARCHAR(255) PRIMARY KEY,
  event_type      VARCHAR(100) NOT NULL,
  payload_hash    VARCHAR(64)  NOT NULL,
  created_at      TIMESTAMPTZ  NOT NULL DEFAULT NOW(),
  processed_at    TIMESTAMPTZ,
  result_status   SMALLINT,
  result_body     JSONB
);

-- Ensure one open punch per user (null clock_out_at means open)
-- If your table is named punches, add:
CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_one_open_per_user
ON punches (user_id)
WHERE clock_out_at IS NULL;
```

// workio/server/src/routes/webhook/line.ts
import { Router, Request, Response } from 'express';
import { Pool } from 'pg';
import crypto from 'crypto';
import { lineClient } from '../../lib/line/client';

const router = Router();
const pool = new Pool(); // ensure same pool as app

function sha256(body: string): string {
  return crypto.createHash('sha256').update(body, 'utf8').digest('hex');
}

async function withIdempotency<T>(
  key: string,
  payloadHash: string,
  fn: () => Promise<T>
): Promise<{ already: boolean; result?: T }> {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');
    // lock row for update to prevent concurrent processing of same key
    const exists = await client.query(
      `SELECT 1 FROM webhook_idempotency WHERE idempotency_key = $1 FOR UPDATE`,
      [key]
    );
    if (exists.rows.length > 0) {
      await client.query('ROLLBACK');
      return { already: true };
    }

    await client.query(
      `INSERT INTO webhook_idempotency(idempotency_key, event_type, payload_hash, created_at)
       VALUES ($1, $2, $3, NOW())`,
      [key, 'line.message', payloadHash]
    );
    const result = await fn();
    await client.query(
      `UPDATE webhook_idempotency
       SET processed_at = NOW(), result_status = 200, result_body = $1
       WHERE idempotency_key = $2`,
      [JSON.stringify({ ok: true }), key]
    );
    await client.query('COMMIT');
    return { already: false, result };
  } catch (err) {
    await client.query('ROLLBACK');
    // mark failure so repeated deliveries don't retry endlessly
    try {
      await pool.query(
        `UPDATE webhook_idempotency
         SET processed_at = NOW(), result_status = 500, result_body = $1
         WHERE idempotency_key = $2`,
        [JSON.stringify({ error: String(err) }), key]
      );
    } catch (_) {
      // best-effort
    }
    throw err;
  } finally {
    client.release();
  }
}

async function upsertPunch(userId: number, clockInAt: Date, location?: { lat: number; lng: number }) {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');
    // Try to close any dangling open punch (defensive)
    await client.query(
      `UPDATE punches SET clock_out_at = $1 WHERE user_id = $2 AND clock_out_at IS NULL`,
      [new Date(), userId]
    );
    // Insert new punch
    await client.query(
      `INSERT INTO punches(user_id, clock_in_at, clock_out_at, latitude, longitude)
       VALUES ($1, $2, $3, $4, $5)`,
      [userId, clockInAt, null, location?.lat ?? null, location?.lng ?? null]
    );
    await client.query('COMMIT');
  } catch (err: any) {
    await client.query('ROLLBACK');
    // If unique violation on one-open-punch index, close and retry once
    if (err.code === '23505') {
      // close open and insert again
      await client.query('BEGIN');
      await client.query(
        `UPDATE punches SET clock_out_at = $1 WHERE user_id = $2 AND clock_out_at IS NULL`,
        [new Date(), userId]
      );
      await client.query(
        `INSERT INTO punches(user_id, clock_in_at, clock_out_at, latitude, longitude)
         VALUES ($1, $2, $3, $4, $5)`,
        [userId, clockInAt, null, location?.lat ?? null, location?.lng ?? null]
      );
      await client.query('COMMIT');
    } else {
      throw err;
    }
  } finally {
    client.release();
  }
}

async function sendLineReplyWithRetry(replyToken: string, message: any, retries = 2) {
  for (let i = 0; i <= retries; i++) {
    try {
      await lineClient.replyMessage(replyToken, message);
      return;
    } catch (err: any) {
      if (i === retries) throw err;
      const delay = 500 * Math.pow(2, i);
      await new Promise((r) => setTimeout(r, delay));
    }
  }
}
