**Scope & Constraints**  
- Read-only (Sense + Signal). No mutations, no side effects.  
- Optional `?for_date=YYYY-MM-DD` (default today UTC).  
- Reuse existing `KnowledgeGraph` assets; avoid recomputing centrality per request.  
- Ship in ≤2h with minimal new code; follow repo patterns (FastAPI + services + DAL).  
- Favor correctness + concrete actionability in all outputs.

#### A) Model — `app/models/signal.py`

```python
from datetime import date, datetime
from pydantic import BaseModel, Field
from typing import Optional, List, Dict, Any


class TopHubSignal(BaseModel):
    signal_type: str = Field(
        default="cost_anomaly_top_hub",
        description="Type of signal",
    )
    hub_id: str = Field(..., description="Most-connected hub node identifier (e.g., 'MOC')")
    hub_label: Optional[str] = Field(None, description="Human-readable label")
    centrality_score: float = Field(..., ge=0.0, description="Graph centrality metric value")
    for_date: date = Field(..., description="Date this signal applies to (YYYY-MM-DD)")
    context: Dict[str, Any] = Field(
        default_factory=dict,
        description="Additional contextual insights (edges, related docs, cost tags)",
    )
    related_nodes: List[Dict[str, Any]] = Field(
        default_factory=list,
        description="Top-N related nodes with weights and types",
    )
    recommendations: List[str] = Field(
        default_factory=list,
        description="Actionable signals derived from hub context",
    )
    generated_at: str = Field(..., description="ISO-8601 UTC timestamp when signal produced")

    class Config:
        json_encoders = {
            date: lambda v: v.isoformat(),
            datetime: lambda v: v.isoformat(timespec="seconds") + "Z",
        }
```

#### B) Service — `app/services/signals/top_hub.py`
```python
from datetime import datetime, timezone
from typing import Dict, Any, List, Optional
from app.models.signal import TopHubSignal
from app.db.knowledge_graph import KnowledgeGraph  # existing/reusable


class TopHubSignalService:
    def __init__(self, graph: KnowledgeGraph):
        self.graph = graph

    def get_top_hub_signal(self, for_date: Optional[str] = None) -> TopHubSignal:
        """
        Produce a read-only top-hub signal for cost anomalies.
        Reuses existing graph; does not mutate.
        """
        target_date = for_date or datetime.utcnow().strftime("%Y-%m-%d")

        # Reuse existing graph API: top_hub(for_date) -> dict | None
        hub = self.graph.top_hub(for_date=target_date)
        if not hub:
            # Graceful fallback: neutral signal
            return TopHubSignal(
                hub_id="none",
                hub_label=None,
                centrality_score=0.0,
                for_date=target_date,
                context={"note": "No hub found for date"},
                related_nodes=[],
                recommendations=[],
                generated_at=datetime.now(timezone.utc).isoformat(),
            )

        # Lightweight related nodes (top edges)
        related: List[Dict[str, Any]] = [
            {
                "node_id": e.get("target"),
                "weight": float(e.get("weight", 0.0)),
                "type": e.get("type", ""),
            }
            for e in hub.get("edges", [])[:10]
        ]

        recommendations = self._derive_recommendations(hub)

        return TopHubSignal(
            hub_id=str(hub.get("id", "")),
            hub_label=hub.get("label"),
            centrality_score=float(hub.get("score", 0.0)),
            for_date=target_date,
            context={
                "tags": hub.get("tags", []),
                "cost_categories": hub.get("cost_categories", []),
                "anomaly_indicators": hub.get("anomaly_indicators", []),
                "doc_refs": hub.get("doc_refs", []),
            },
            related_nodes=related,
            recommendations=recommendations,
            generated_at=datetime.now(timezone.utc).isoformat(),
        )

    def _derive_recommendations(self, hub: Dict[str, Any]) -> List[str]:
        label = (hub.get("label") or "").lower()
        tags = [t.lower() for t in hub.get("tags", [])]
        recs: List[str] = []

        if "moc" in label or "moc" in tags:
            recs.append("Review MOC-linked resources for idle or oversized instances.")
        if "spend" in tags or "anomaly" in tags:
            recs.append("Investigate recent cost spikes tied to this hub.")
        if "ri" in tags or "reserved" in tags:
            recs.append("Validate RI coverage and utilization for linked accounts.")

        if not recs:
            recs.append("Review hub-linked resources and recent cost trends.")

        return recs
```

#### C) Endpoint — `app/api/v1/endpoints/cost_anomaly.py`
```python
from datetime import date
from typing import Optional
from fastapi import APIRouter, Depends, Query
from app.services.signals.top_hub import TopHubSignalService
from app.db.knowledge_graph import KnowledgeGraph  # assumed singleton/reusable
from app.models.signal import TopHubSignal

router = APIRouter()


def get_graph() -> KnowledgeGraph:
    # Reuse existing graph accessor (cached/singleton if available)
    return KnowledgeGraph()


@router.get(
    "/cost-anomaly/signal/top-hub",
    response_model=TopHubSignal,
    tags=["cost-anomaly", "signals"],
    summary="Top knowledge-graph hub as cost-anomaly signal (Sense + Signal)",
)
def get_top_hub_signal(
    for_date: Optional[date] = Query(
        None,
        description="Date to evaluate (YYYY-MM-DD). Defaults to today UTC.",
        example="2026-04-27",
    ),
    graph: KnowledgeGraph = Depends(get_graph),
    service: TopHubSignalService = Depends(lambda g=graph: TopHubSignalService(g)),
) -> TopHubSignal:
    """
    Returns the strongest knowledge-graph hub as a cost-anomaly signal.
    Read-only. No execution.
    """
    date_str = for_date.isoformat() if for_date else None
    return service.get_top_hub_signal(for_date=date_str)
```
