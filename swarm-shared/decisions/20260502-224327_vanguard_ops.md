### `/opt/axentx/vanguard/ops/prepare_training_run.sh`
```bash
#!/usr/bin/env bash
# prepare_training_run.sh
# Idempotent run preparation: CDN file list, config, studio launcher, schema guard.
set -euo pipefail

# ---- Config (override via env) ----
HF_REPO="${HF_REPO:-datasets/axentx/surrogate-1}"
DATE_FOLDER="${1:-$(date +%Y-%m-%d)}"
OUT_DIR="${2:-/opt/axentx/vanguard/config}"
HF_TOKEN="${HF_TOKEN:-}"
STUDIO_NAME="${STUDIO_NAME:-vanguard-l40s}"
MACHINE="${MACHINE:-L40S}"
SIBLING_COUNT="${SIBLING_COUNT:-5}"
SCHEMA_PROJECT_COLS="${SCHEMA_PROJECT_COLS:-prompt,response}"

mkdir -p "${OUT_DIR}"

# ---- 1) List single folder (non-recursive) once ----
echo "[+] Listing ${HF_REPO}/${DATE_FOLDER} (non-recursive)"
AUTH_HEADER=()
if [ -n "${HF_TOKEN}" ]; then
  AUTH_HEADER=(-H "Authorization: Bearer ${HF_TOKEN}")
fi

# Use tree endpoint (non-recursive) to avoid pagination
TREE_JSON=$(curl -sS "${AUTH_HEADER[@]}" \
  "https://huggingface.co/api/datasets/${HF_REPO}/tree?path=${DATE_FOLDER}&recursive=false")

echo "${TREE_JSON}" | jq -r '.[].path' > "${OUT_DIR}/file-list.json"
FILE_COUNT=$(wc -l < "${OUT_DIR}/file-list.json")
echo "[+] Found ${FILE_COUNT} files"

# ---- 2) Build CDN-only URLs (no auth, no API during training) ----
jq -r --arg base "https://huggingface.co/datasets/${HF_REPO}/resolve/main/${DATE_FOLDER}" \
  '$base + "/" + .' "${OUT_DIR}/file-list.json" > "${OUT_DIR}/cdn-urls.json"

# ---- 3) Deterministic sibling repo selector (spread HF commit writes) ----
select_sibling_repo() {
  local slug="$1"
  # deterministic 0..SIBLING_COUNT-1
  local idx_hex=$(echo -n "${slug}" | md5sum | tr -d ' -' | head -c 8)
  local idx=$(( 0x${idx_hex} % SIBLING_COUNT ))
  echo "${HF_REPO}-s${idx}"
}
SELECTED_REPO=$(select_sibling_repo "${DATE_FOLDER}")
echo "[+] Selected sibling repo for writes: ${SELECTED_REPO}"

# ---- 4) Generate train_config.py (consumed by Lightning training) ----
cat > "${OUT_DIR}/train_config.py" <<EOF
# Auto-generated by prepare_training_run.sh
HF_REPO = "${HF_REPO}"
DATE_FOLDER = "${DATE_FOLDER}"
SELECTED_REPO = "${SELECTED_REPO}"
CDN_URLS = $(jq -c '.' "${OUT_DIR}/cdn-urls.json")
FILE_PATHS = $(jq -c '.' "${OUT_DIR}/file-list.json")
MACHINE = "${MACHINE}"
EOF

# ---- 5) Schema guard: projection helper for ingestion ----
cat > "${OUT_DIR}/schema_guard.py" <<'PY'
import pyarrow as pa
import pyarrow.parquet as pq
import os

def project_to_prompt_response(input_path, output_path, cols=("prompt", "response")):
    """Project parquet to only prompt,response (drop source/ts etc)."""
    tbl = pq.read_table(input_path, columns=cols) if os.path.exists(input_path) else None
    if tbl is None:
        raise FileNotFoundError(input_path)
    # Ensure expected columns exist
    missing = set(cols) - set(tbl.column_names)
    if missing:
        raise ValueError(f"Missing columns {missing} in {input_path}")
    pq.write_table(tbl, output_path)

if __name__ == "__main__":
    import sys
    if len(sys.argv) != 3:
        print("Usage: schema_guard.py <input.parquet> <output.parquet>")
        sys.exit(1)
    project_to_prompt_response(sys.argv[1], sys.argv[2])
PY

# ---- 6) Idempotent Lightning Studio launcher ----
cat > "${OUT_DIR}/launch_studio.py" <<'PY'
import os
import sys
import time

try:
    from lightning_sdk import Teamspace, Studio, Machine
except ImportError as e:
    print("[!] lightning-sdk not available:", e)
    sys.exit(1)

STUDIO_NAME = os.getenv("STUDIO_NAME", "vanguard-l40s")
MACHINE = os.getenv("MACHINE", "L40S")
MAX_RETRIES = int(os.getenv("MAX_RETRIES", "3"))
RETRY_DELAY = int(os.getenv("RETRY_DELAY", "15"))

def get_running_studio(name):
    team = Teamspace()
    for s in team.studios:
        if s.name == name and s.status == "running":
            return s
    return None

def main():
    running = get_running_studio(STUDIO_NAME)
    if running:
        print(f"[+] Reusing running studio: {running.name}")
        print(f"[+] Studio URL: {running.url}")
        return 0

    # No running studio -> create/start idempotently
    print(f"[+] Starting studio: {STUDIO_NAME} on {MACHINE}")
    for attempt in range(1, MAX_RETRIES + 1):
        try:
            studio = Studio.create(name=STUDIO_NAME, machine=Machine(MACHINE), create_ok=True)
            break
        except Exception as e:
            print(f"[!] Attempt {attempt}/{MAX_RETRIES} failed: {e}")
            if attempt == MAX_RETRIES:
                raise
            time.sleep(RETRY_DELAY)

    # Ensure running
    if studio.status != "running":
        print("[+] Studio not running; starting...")
        studio.start(machine=Machine(MACHINE))

    print(f"[+] Studio URL: {studio.url}")
    return 0

if __name__ == "__main__":
    sys.exit(main())
PY

echo "[+] Generated configs in ${OUT_DIR}"
echo "[+] Next steps:"
echo "    bash ${OUT_DIR}/prepare_training_run.sh 2026-04-29 ${OUT_DIR}"
echo "    STUDIO_NAME=vanguard-l40s MACHINE=L40S python ${OUT_DIR}/launch_studio.py"
```
