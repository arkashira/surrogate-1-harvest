# vanguard / design

# Final Design: Manifest-First, CDN-Only, Deterministic Training Pipeline

## 1. Diagnosis (Consolidated)
- **No persisted manifest**: every training run triggers authenticated `list_repo_tree`, burning HF API quota and risking 429s.
- **Schema heterogeneity**: training/data loader uses `load_dataset` or per-file streaming across repos, causing `pyarrow.CastError` on mixed schemas.
- **Unnecessary auth during training**: authenticated API calls are fragile; public CDN URLs bypass auth and rate limits.
- **HF commit concentration**: writes focused on one repo risk the 128 commits/hour limit.
- **Lightning Studio inefficiency**: repeated studio creation wastes quota; idle-stop kills long-running training.

## 2. High-Level Solution
Add a small, self-contained module under `/opt/axentx/vanguard/` that enables:
1. **Manifest-first workflow**: single authenticated `list_repo_tree`, then CDN-only training.
2. **Schema-safe streaming**: per-file projection to canonical `{prompt, response}` and optional schema enforcement.
3. **Deterministic repo routing**: hash-based sibling selection to spread HF commit load.
4. **Lightning Studio reuse**: orchestrator that reuses running studios and handles idle-stop gracefully.

## 3. Directory Scaffold
```bash
cd /opt/axentx/vanguard
mkdir -p vanguard/data vanguard/training

### 4.1 `vanguard/data/manifest.py`
```python
from __future__ import annotations
import json
from datetime import datetime
from pathlib import Path
from typing import Dict, List

from huggingface_hub import HfApi

HF_API = HfApi()

def list_date_files(repo_id: str, date_folder: str, token: str | None = None) -> List[str]:
    """
    Single authenticated call to list files for one date folder (non-recursive).
    Returns paths like 'en/2024-01-15/file1.jsonl'.
    """
    tree = HF_API.list_repo_tree(
        repo_id=repo_id,
        path=date_folder,
        repo_type="dataset",
        token=token,
        recursive=False,
    )
    return sorted(f["path"] for f in tree if f.get("path"))

def build_manifest(
    repo_id: str,
    date_folder: str,
    out_path: Path | str,
    token: str | None = None,
) -> Dict:
    manifest = {
        "repo_id": repo_id,
        "date_folder": date_folder,
        "generated_at": datetime.utcnow().isoformat() + "Z",
        "files": list_date_files(repo_id, date_folder, token=token),
    }
    out_path = Path(out_path)
    out_path.parent.mkdir(parents=True, exist_ok=True)
    out_path.write_text(json.dumps(manifest, indent=2))
    return manifest

def load_manifest(manifest_path: Path | str) -> Dict:
    return json.loads(Path(manifest_path).read_text())

def cdn_urls(manifest: Dict) -> List[str]:
    """Convert manifest into public CDN URLs (no auth)."""
    repo = manifest["repo_id"]
    base = f"https://huggingface.co/datasets/{repo}/resolve/main"
    return [f"{base}/{f}" for f in manifest["files"]]
```

### 4.2 `vanguard/data/router.py`
```python
import hashlib
from typing import List

def pick_sibling_repo(slug: str, siblings: List[str], salt: str = "vanguard") -> str:
    """
    Deterministic repo selection to spread HF commit load.
    siblings = ["repo-a", "repo-b", "repo-c", "repo-d", "repo-e"]
    """
    if not siblings:
        raise ValueError("siblings must not be empty")
    key = f"{salt}:{slug}"
    idx = int(hashlib.sha256(key.encode()).hexdigest(), 16) % len(siblings)
    return siblings[idx]
```

### 4.3 `vanguard/training/lightning_launcher.py`
```python
from __future__ import annotations
from pathlib import Path

from lightning import LightningWork, LightningFlow, LightningApp, Machine

class SurrogateTrainer(LightningWork):
    def __init__(
        self,
        manifest_path: Path | str,
        script_path: Path | str,
        machine: Machine = Machine.L40S,
        cloud: str = "lightning-public-prod",
    ):
        super().__init__(cloud=cloud)
        self.manifest_path = Path(manifest_path)
        self.script_path = Path(script_path)
        self.machine = machine

    def run(self):
        if not self.manifest_path.exists():
            raise FileNotFoundError(f"Manifest missing: {self.manifest_path}")

        import subprocess
        cmd = ["python", str(self.script_path), "--manifest", str(self.manifest_path)]
        subprocess.run(cmd, check=True)

class SurrogateFlow(LightningFlow):
    def __init__(
        self,
        manifest_path: Path | str,
        script_path: Path | str,
    ):
        super().__init__()
        self.manifest_path = Path(manifest_path)
        self.script_path = Path(script_path)
        self.trainer = SurrogateTrainer(
            manifest_path=self.manifest_path,
            script_path=self.script_path,
            machine=Machine.L40S,
        )

    def configure_layout(self):
        return [{"name": "train", "content": self.trainer}]

    def run(self):
        # Reuse running studio; restart only if stopped after having run.
        if self.trainer.has_stopped and self.trainer.has_run:
            self.trainer.start(machine=self.trainer.machine)
        if not self.trainer.is_running:
            self.trainer.start(machine=self.trainer.machine)
        if not getattr(self.trainer, "_started", False):
            self.trainer.run()
            self.trainer._started = True

def launch_local(manifest_path: Path | str, script_path: Path | str):
    app = LightningApp(
        SurrogateFlow(
            manifest_path=manifest_path,
            script_path=script_path,
        )
    )
    return app
```

### 4.4 `vanguard/training/train.py` (CDN-only, schema-safe)
```python
from __future__ import annotations
import argparse
import json
from pathlib import Path
from typing import Dict, Iterable

import requests

def load_records_from_cdn(url: str) -> Iterable[Dict[str, str]]:
    """
    Load and project one file to canonical {prompt, response}.
    Handles JSONL; skips empty lines and malformed records.
    """
    resp = requests.get(url, timeout=30)
    resp.raise_for_status()
    for line in resp.text.strip().splitlines():
        line = line.strip()
        if not line:
            continue
        try:
            obj = json.loads(line)
        except json.JSONDecodeError:
            continue

        prompt = obj.get("prompt") or obj.get("input") or ""
        response = obj.get("response") or obj.get("output") or ""
        # Coerce to string to avoid downstream type errors
        yield {"prompt": str(prompt), "response": str(response)}

def build(manifest_path: Path) -> Iterable[Dict[str, str]]:
    manifest = json.loads(manifest_path.read_text())
    base = f"https://huggingface.co/datasets/{manifest['repo_id']}/resolve/main"
    for f in manifest["files"]:
        url = f"{base}/{f}"
        try:
            yield from load_records_from_cdn(url)
        except Exception as exc:
            # Log and skip bad files to keep training robust
            print(f"Skipping {url}: {exc}")
            continue

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--manifest", type=Path, required=True)
    parser.add_argument("--output", type=Path, default=Path("training_data.jsonl"))
    args = parser.parse_args()

    args.output.parent.mkdir(parents=True, exist_ok=True)
    with args.output.open("w") as fout:
        for record in build(args.manifest):
            fout.write(json.dumps(record) + "\n")
    print(f"Wrote {args.output}")

if __name__ == "__main__":
    main()
```
