### 2. CDN-only dataset loader
Replace `load_dataset` with direct CDN downloads.

```python
# /opt/axentx/airship/surrogate/data/cdn_dataset.py
import json
import os
import pyarrow.parquet as pq
import requests
from io import BytesIO
from typing import List, Dict, Iterator
import torch
from torch.utils.data import IterableDataset

HF_DATASET_REPO = "datasets/your-org/surrogate-mirror"
CDN_BASE = f"https://huggingface.co/{HF_DATASET_REPO}/resolve/main"

class CDNParquetDataset(IterableDataset):
    def __init__(self, file_list_path: str, start_idx: int = 0, end_idx: int = None):
        with open(file_list_path) as f:
            manifest = json.load(f)
        self.files = manifest["files"][start_idx:end_idx]
        self.date = manifest["date"]
        
    def _stream_parquet(self, file_path: str) -> List[Dict]:
        """Download single parquet via CDN (no auth, no API)"""
        url = f"{CDN_BASE}/{self.date}/{file_path}"
        resp = requests.get(url, timeout=30)
        resp.raise_for_status()
        table = pq.read_table(BytesIO(resp.content))
        # Project to {prompt, response} only
        df = table.select(["prompt", "response"]).to_pandas()
        return df.to_dict("records")
    
    def __iter__(self) -> Iterator[Dict]:
        for file_path in self.files:
            try:
                records = self._stream_parquet(file_path)
                for rec in records:
                    yield rec
            except Exception as e:
                print(f"Skipping {file_path}: {e}")
                continue

def create_dataloader(file_list_path: str, batch_size: int = 8, num_workers: int = 0):
    dataset = CDNParquetDataset(file_list_path)
    return torch.utils.data.DataLoader(
        dataset,
        batch_size=batch_size,
        num_workers=num_workers,
        collate_fn=lambda x: x  # handle variable-length dicts
    )
```

### 3. Lightning Studio lifecycle wrapper
Handle idle-stop and quota reuse.

```python
# /opt/axentx/airship/surrogate/training/studio_manager.py
import time
from lightning import LightningWork, LightningFlow, LightningApp, Machine
from lightning.fabric.utilities.cloud_io import get_filesystem
from lightning.pytorch import Trainer
from lightning.pytorch.strategies import DDPStrategy

class ResilientStudio(LightningWork):
    def __init__(self, studio_name: str = "surrogate-training", machine_type: str = "L40S"):
        super().__init__()
        self.studio_name = studio_name
        self.machine_type = machine_type
        self._studio = None
        
    def run(self, train_script: str, file_list_date: str):
        from lightning import Teamspace
        
        # Reuse running studio (saves 80hr/mo quota)
        for s in Teamspace.studios:
            if s.name == self.studio_name and s.status == "running":
                self._studio = s
                print(f"Reusing running studio: {self.studio_name}")
                break
        
        if not self._studio:
            # Check free tier clouds in priority order
            clouds = ["lightning-public-prod", "lightning-lambda-prod"]
            for cloud in clouds:
                try:
                    self._studio = Teamspace.studios.create(
                        name=self.studio_name,
                        machine=Machine(self.machine_type, cloud=cloud),
                        create_ok=True
                    )
                    print(f"Created studio on {cloud}")
                    break
                except Exception as e:
                    print(f"Cloud {cloud} failed: {e}")
                    continue
        
        # Ensure studio is running before training
        max_retries = 3
        for attempt in range(max_retries):
            if self._studio.status != "running":
                print(f"Studio not running (attempt {attempt+1}), restarting...")
                self._studio.start(machine=Machine(self.machine_type))
                time.sleep(60)  # wait for startup
            
            # Run training with CDN file list
            result = self._studio.run(
                train_script,
                arguments={"file_list_date": file_list_date},
                enable_pip_pinning=False
            )
            
            if result.exit_code == 0:
                print("Training completed successfully")
                return result
            elif "idle" in str(result).lower():
                print("Idle timeout detected, restarting studio...")
                self._studio.stop()
                time.sleep(30)
            else:
                print(f"Training failed: {result}")
                return result
        
        raise RuntimeError("Max retries exceeded for studio training")
```
