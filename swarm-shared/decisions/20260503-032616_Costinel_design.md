# Costinel / design

## Final Synthesis — Top-Hub Signal Panel (CDN-first, non-blocking, <2h)

**Chosen approach**:  
Deploy a **read-only, CDN-first Top-Hub Signal Panel** that surfaces cost/governance signals for the most-connected hub without blocking UI, without calling HF API during render, and with graceful fallbacks + lightweight telemetry.

---

### Core design decisions (resolve contradictions)

| Contradiction | Resolution |
|--------------|------------|
| Where to resolve “top hub” (build-time vs runtime) | **Hybrid**: precompute top hub at build/deploy time into `public/signals/top-hub.json`. If a lightweight runtime index (`hub-index.json`) is available and cheap, optionally enhance at runtime; otherwise fall back to static file. Avoid runtime HF API calls entirely. |
| Telemetry vs privacy/overhead | Include **lightweight, opt-in telemetry** (`signal_impression`, `signal_action`) emitted via `navigator.sendBeacon` to avoid blocking; respect Do Not Track and privacy preferences; no PII. |
| Panel placement (sidebar vs top ribbon) | **Sidebar card** (low visual weight, avoids header clutter). Make placement configurable via feature flag. |
| SSR vs client-only fetch | **Client-only fetch** to avoid blocking hydration. SSR a skeleton for perceived performance. |

### CDN-first fetcher + cache

`src/lib/signalFetcher.ts`

```ts
const CDN_SIGNAL_PATH = '/signals/top-hub.json';
const HUB_INDEX_PATH = '/signals/hub-index.json';
const CACHE_TTL_MS = 60_000;

let memoryCache: { data: any | null; ts: number } | null = null;

export async function fetchTopHubSignal(abortSignal?: AbortSignal) {
  // memory cache for session
  if (memoryCache && Date.now() - memoryCache.ts < CACHE_TTL_MS) {
    return memoryCache.data;
  }

  try {
    const controller = new AbortController();
    const timeout = setTimeout(() => controller.abort(), 5000);
    const combinedSignal = abortSignal
      ? AbortSignal.any([abortSignal, controller.signal])
      : controller.signal;

    const res = await fetch(CDN_SIGNAL_PATH, {
      method: 'GET',
      credentials: 'same-origin',
      signal: combinedSignal,
    });
    clearTimeout(timeout);

    if (!res.ok) throw new Error(`CDN fetch failed: ${res.status}`);
    const json = await res.json();
    memoryCache = { data: json, ts: Date.now() };
    return json;
  } catch (err) {
    console.warn('Top-hub signal CDN fetch failed, using fallback', err);
    memoryCache = { data: null, ts: Date.now() };
    return null;
  }
}

// Optional: lightweight top-hub resolver from index
export async function resolveTopHubFromIndex(abortSignal?: AbortSignal) {
  try {
    const controller = new AbortController();
    const timeout = setTimeout(() => controller.abort(), 3000);
    const combinedSignal = abortSignal
      ? AbortSignal.any([abortSignal, controller.signal])
      : controller.signal;

    const res = await fetch(HUB_INDEX_PATH, { signal: combinedSignal });
    clearTimeout(timeout);
    if (!res.ok) return null;
    const list = await res.json();
    if (!Array.isArray(list) || list.length === 0) return null;
    // highest degree
    const top = list.reduce((a, b) => (b.connections > a.connections ? b : a));
    return top.hubId;
  } catch {
    return null;
  }
}
```

### Telemetry (lightweight, non-blocking)

`src/lib/telemetry.ts`

```ts
export function sendSignalEvent(event: 'signal_impression' | 'signal_action', payload: Record<string, string | number | boolean>) {
  if (typeof navigator === 'undefined' || !navigator.sendBeacon) return;
  // respect Do Not Track
  if (navigator.doNotTrack === '1' || window.doNotTrack === '1') return;

  const body = JSON.stringify({
    event,
    ts: Date.now(),
    ...payload,
  });

  const blob = new Blob([body], { type: 'application/json' });
  navigator.sendBeacon('/api/telemetry/signal', blob);
}
```

'use client';

import { useEffect, useState } from 'react';
import { fetchTopHubSignal } from '@/lib/signalFetcher';
import { sendSignalEvent } from '@/lib/telemetry';

type Signal = {
  id: string;
  title: string;
  severity: 'critical' | 'high' | 'medium' | 'low';
  description: string;
  recommendation: string;
  costImpact: string;
  tags: string[];
};

type TopHub = {
  hubId: string;
  label: string;
  connections: number;
  lastUpdated: string;
  signals: Signal[];
};

const severityColor = (s: Signal['severity']) => {
  switch (s.severity) {
    case 'critical': return 'bg-red-600 text-white';
    case 'high': return 'bg-orange-600 text-white';
    case 'medium': return 'bg-yellow-500 text-black';
    case 'low': return 'bg-gray-500 text-white';
    default: return 'bg-gray-400 text-white';
  }
};

export default function TopHubSignalPanel() {
  const [hub, setHub] = useState<TopHub | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(false);

  useEffect(() => {
    let cancelled = false;
    const controller = new AbortController();

    fetchTopHubSignal(controller.signal)
      .then((data) => {
        if (cancelled) return;
        if (data) {
          setHub(data);
          setError(false);
          // impression telemetry
          sendSignalEvent('signal_impression', {
            hubId: data.hubId,
            signalCount: data.signals?.length || 0,
          });
        } else {
          setHub(null);
          setError(true);
        }
      })
      .catch(() => {
        if (cancelled) return;
        setError(true);
        setHub(null);
