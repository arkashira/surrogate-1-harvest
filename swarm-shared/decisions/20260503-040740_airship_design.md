# Final Design — CDN-only Parquet Loader + Idle-Resilient Lightning Runner

## Core Insight (highest value, <2h)
Eliminate HuggingFace API rate limits during training **and** prevent Lightning idle-timeouts from killing 24/7 autonomous runs by:

1. **Listing parquet files once on the Mac orchestrator** (single HF API call).
2. **Training exclusively via CDN URLs** (no `load_dataset`, no auth/429s during training).
3. **Running Lightning Studio with reuse + auto-restart** so idle timeouts don’t stop training.

This keeps the Mac as orchestrator only; all heavy compute stays on Lightning.

## 1) Mac orchestration: one-time snapshot (HF API, single call)

`scripts/list_parquet_snapshot.py`

```python
#!/usr/bin/env python3
"""
Snapshot parquet files for a date folder.
Run from Mac when HF API window is clear.
Output: JSON consumed by Lightning training.
"""
import json
import os
import sys
from datetime import datetime, timezone

from huggingface_hub import HfApi

REPO = os.getenv("REPO", "axentx/surrogate-datasets")
DATE_FOLDER = os.getenv("DATE_FOLDER", datetime.now(timezone.utc).strftime("%Y-%m-%d"))
OUTPUT = os.getenv("OUTPUT", f"filelist_{DATE_FOLDER}.json")

def main() -> None:
    api = HfApi()
    # Non-recursive to avoid pagination explosion
    entries = api.list_repo_tree(repo_id=REPO, path=DATE_FOLDER, recursive=False)
    files = [
        f"{DATE_FOLDER}/{e.path.split('/')[-1]}"
        for e in entries
        if e.path.lower().endswith(".parquet")
    ]
    payload = {
        "repo": REPO,
        "date_folder": DATE_FOLDER,
        "files": sorted(files),
        "snapshot_ts": datetime.now(timezone.utc).isoformat(),
    }
    with open(OUTPUT, "w", encoding="utf-8") as f:
        json.dump(payload, f, indent=2)
    print(f"Wrote {len(files)} files to {OUTPUT}")

if __name__ == "__main__":
    main()
```

Make executable:

```bash
chmod +x scripts/list_parquet_snapshot.py
```

## 2) CDN-only parquet loader (Lightning training script)

`surrogate/train_cdn.py`

```python
#!/usr/bin/env python3
"""
CDN-only data loader for Surrogate training.
No HuggingFace API calls during training — avoids 429s.
Expects filelist JSON produced by list_parquet_snapshot.py.
"""
import argparse
import json
import os
import pyarrow.parquet as pq
import requests
import torch
from torch.utils.data import IterableDataset, DataLoader
from typing import Iterator

HF_CDN = "https://huggingface.co/datasets"

class CDNParquetIterable(IterableDataset):
    def __init__(self, filelist_path: str, columns=("prompt", "response")):
        super().__init__()
        with open(filelist_path, encoding="utf-8") as f:
            manifest = json.load(f)
        self.repo = manifest["repo"]
        self.files = manifest["files"]
        self.columns = columns

    def _stream_file(self, path: str) -> Iterator[dict]:
        url = f"{HF_CDN}/{self.repo}/resolve/main/{path}"
        # CDN download; no Authorization header required
        with requests.get(url, timeout=60, stream=True) as r:
            r.raise_for_status()
            tmp_path = "/tmp/current.parquet"
            with open(tmp_path, "wb") as tmp:
                for chunk in r.iter_content(chunk_size=8192):
                    tmp.write(chunk)
            table = pq.read_table(tmp_path, columns=self.columns)
            for batch in table.to_batches(max_chunksize=1024):
                df = batch.to_pandas()
                for _, row in df.iterrows():
                    yield {k: row[k] for k in self.columns if k in row}

    def __iter__(self) -> Iterator[dict]:
        for f in self.files:
            yield from self._stream_file(f)

def make_dataloader(filelist_path: str, batch_size: int = 8, num_workers: int = 2) -> DataLoader:
    dataset = CDNParquetIterable(filelist_path=filelist_path)
    return DataLoader(
        dataset,
        batch_size=batch_size,
        num_workers=num_workers,
        pin_memory=True,
    )

if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--filelist", required=True)
    parser.add_argument("--batch-size", type=int, default=8)
    args = parser.parse_args()

    loader = make_dataloader(args.filelist, batch_size=args.batch_size)
    for i, batch in enumerate(loader):
        print(f"Batch {i}: {list(batch.keys())}")
        if i >= 2:  # smoke test
            break
```

Notes:
- No `load_dataset` → no HF API auth/rate-limit during training.
- CDN downloads are not counted against API rate limits.
- Project to `{prompt, response}` only; attribution via filename pattern.

## 3) Lightning idle-resilient runner (Mac orchestrator)

`scripts/run_lightning_studio.py`

```bash
#!/usr/bin/env bash
#
# Run Lightning Studio resiliently:
# - Reuse running studio if found
# - Restart if stopped (idle-timeout killed training)
# - Execute training with CDN-only loader
#
set -euo pipefail
SHELL=/bin/bash

REPO_ROOT="/opt/axentx/airship"
STUDIO_NAME="surrogate-l40s-training"
MACHINE="lightning-A100"   # fallback; prefer L40S if available
DATE_TAG=$(date -u +%Y-%m-%d)
FILELIST="${REPO_ROOT}/filelist_${DATE_TAG}.json"

cd "${REPO_ROOT}"

# Ensure filelist exists (run snapshot earlier from cron)
if [[ ! -f "${FILELIST}" ]]; then
  echo "Filelist missing: ${FILELIST}. Generating..."
  python3 scripts/list_parquet_snapshot.py
fi

# Use Lightning Python SDK to manage studio
python3 <<PYEOF
import os
from lightning.app import LightningStudio, Machine

studio_name = "${STUDIO_NAME}"
machine = Machine.${MACHINE}

# Reuse running studio
reuse = None
for s in LightningStudio.list():
    if s.name == studio_name and s.status == "running":
        reuse = s
        print(f"Reusing running studio: {studio_name}")
        break

if reuse is None:
    print(f"Starting studio: {studio_name}")
    reuse = LightningStudio(
        name=studio_name,
        create_ok=True,
        machine=machine,
    )

# Ensure it is running before submitting work
if reuse.status != "running":
    print(f"Studio not running (status={reuse.status}). Starting...")
    reuse.start(machine=machine)

# Submit training run (non-blocking)
reuse.run(
    script="surrogate/train_cdn.py",
    arguments=[
        "--filelist", "${FILELIST}",
        "--batch-size", "8",
    ],
    wait=False,
)
print("Training submitted.")
PYEOF
```

Make executable:

```bash
chmod +x scripts/run_lightning_studio.py
```
