# vanguard / ops

# 1. Diagnosis
- No persisted `(repo, dateFolder)` manifest exists → every training run re-enumerates via authenticated HF API → quota burn + 429 risk.
- Data loading likely uses authenticated `list_repo_tree`/`load_dataset` during training → burns API quota on CDN-capable files that could be fetched anonymously.
- No local file-list cache for Lightning training runs → repeated enumeration across retries/restarts wastes quota and increases flakiness.
- Missing ops guardrails: no pre-flight check for existing running Studio and no automatic resume on idle-stop → wastes Lightning quota and kills long jobs.
- No CDN-bypass strategy encoded in training pipeline → misses the key HF CDN bypass (public `resolve/main/` URLs require no auth and avoid rate limits).

# 2. Proposed change
Create `/opt/axentx/vanguard/ops/prepare_training_manifest.py` and update training launcher to:
- Accept `repo` and `dateFolder` args.
- Call `list_repo_tree(path=dateFolder, recursive=False)` once (authenticated, from Mac).
- Persist `{repo}_{dateFolder}_manifest.json` containing `{ "repo": "...", "dateFolder": "...", "files": [...], "cdn_prefix": "https://huggingface.co/datasets/.../resolve/main/..." }`.
- Embed this manifest path in Lightning training script so data loader uses CDN-only fetches with zero API calls.
- Add Studio reuse + idle-stop guard in launcher (`list Teamspace.studios`, reuse if running, restart if stopped).

```bash
# /opt/axentx/vanguard/ops/prepare_training_manifest.py
#!/usr/bin/env bash
# Usage: ./prepare_training_manifest.py <repo> <dateFolder> [out_dir]
# Example: ./prepare_training_manifest.py axentx/surrogate-1 2026-04-29 ./manifests

set -euo pipefail

REPO="${1:-axentx/surrogate-1}"
DATEFOLDER="${2:-$(date +%Y-%m-%d)}"
OUTDIR="${3:-./manifests}"
OUTFILE="${OUTDIR}/${REPO//\//_}_${DATEFOLDER}_manifest.json"

mkdir -p "$OUTDIR"

python3 - "$REPO" "$DATEFOLDER" "$OUTFILE" <<'PY'
import json, os, sys
from huggingface_hub import HfApi

repo = sys.argv[1]
date_folder = sys.argv[2].strip("/")
outfile = sys.argv[3]

api = HfApi()
# Non-recursive listing for the date folder
tree = api.list_repo_tree(repo=repo, path=date_folder, recursive=False)
files = [item.rfilename for item in tree if item.type == "file"]

manifest = {
    "repo": repo,
    "dateFolder": date_folder,
    "files": sorted(files),
    "cdn_prefix": f"https://huggingface.co/datasets/{repo}/resolve/main/{date_folder}",
    "generated_by": "prepare_training_manifest",
}

os.makedirs(os.path.dirname(outfile), exist_ok=True)
with open(outfile, "w") as f:
    json.dump(manifest, f, indent=2)
print(f"Manifest written to {outfile} ({len(files)} files)")
PY

echo "Done: $OUTFILE"
```

```python
#!/usr/bin/env python3
"""
Launch Lightning Studio training with manifest-based CDN fetches and Studio reuse.
Usage: python launch_lightning_training.py <manifest.json> <train_script.py> [--machine L40S]
"""
import argparse, json, subprocess, sys, time
from lightning_sdk import Client, Teamspace, Studio, Machine

def load_manifest(path: str) -> dict:
    with open(path) as f:
        return json.load(f)

def find_running_studio(name: str):
    for s in Teamspace.studios:
        if s.name == name and s.status == "Running":
            return s
    return None

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("manifest", help="Path to manifest JSON")
    parser.add_argument("train_script", help="Training script to run in Studio")
    parser.add_argument("--machine", default="L40S", help="Lightning machine type")
    parser.add_argument("--studio-name", default="vanguard-training", help="Studio name")
    args = parser.parse_args()

    manifest = load_manifest(args.manifest)
    print(f"Loaded manifest: {manifest['repo']}/{manifest['dateFolder']} ({len(manifest['files'])} files)")

    # Studio reuse
    studio = find_running_studio(args.studio_name)
    if studio:
        print(f"Reusing running Studio: {studio.name} (id={studio.id})")
    else:
        print(f"Creating Studio: {args.studio_name}")
        studio = Studio.create(
            name=args.studio_name,
            machine=getattr(Machine, args.machine, Machine.L40S),
            create_ok=True,
        )
        # Wait until running
        while studio.status != "Running":
            time.sleep(10)
            studio = Studio(id=studio.id)

    # Prepare run command that uses manifest and CDN-only fetches
    # We assume train_script accepts --manifest and uses CDN URLs from it.
    run_cmd = [
        sys.executable, args.train_script,
        "--manifest", args.manifest,
    ]

    # Execute in Studio (non-blocking run; Studio handles environment)
    # If Studio was stopped by idle timeout, restart it first.
    if studio.status != "Running":
        print("Studio not running — restarting...")
        studio.start(machine=getattr(Machine, args.machine, Machine.L40S))
        while studio.status != "Running":
            time.sleep(10)
            studio = Studio(id=studio.id)

    # Use lightning CLI or SDK to run script inside Studio.
```

```python
"""
Example train.py update: load file list from manifest and fetch via CDN (no HF API auth).
"""
import argparse, json, io, requests, pyarrow.parquet as pq
from torch.utils.data import IterableDataset

class CDNParquetDataset(IterableDataset):
    def __init__(self, manifest_path):
        with open(manifest_path) as f:
            m = json.load(f)
        self.repo = m["repo"]
        self.date = m["dateFolder"]
        self.files = m["files"]
        self.cdn_prefix = m["cdn_prefix"]

    def _stream_parquet(self, filename):
        url = f"{self.cdn_prefix}/{filename}"
        # CDN fetch — no Authorization header (bypasses API rate limits)
        resp = requests.get(url, timeout=30)
        resp.raise_for_status()
        return pq.read_table(io.BytesIO(resp.content)).to_pydict()

    def __iter__(self):
        for f in self.files:
            try:
                batch = self._stream_parquet(f)
                # Project to {prompt, response} only at parse time
                for prompt, response in zip(batch.get("prompt", []), batch.get("response", [])):
                    yield {"prompt": prompt, "response": response}
            except Exception as exc:
                # Log and continue — don't fail entire epoch on single file
                print(f"Skipping {f}: {exc}")
                continue

if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--manifest", required=True)
    args = parser.parse_args()
    dataset = CDNParquetDataset(args.manifest)
```
