# workio / frontend

### 1. Diagnosis (consolidated)
- **Stale frontend state** after LINE webhook–driven changes (clock/leave/OT) until hard refresh → perceived latency and silent failures.  
- **No real-time subscription layer**; UI relies on local React state, so multi-tab/devices diverge.  
- **Polling is inefficient/jittery** and optimistic updates are unsafe without server truth.  
- **Missing lightweight push channel** (SSE/WebSocket) to stream tenant-scoped domain events from backend to clients.  
- **No frontend event bus** to reconcile remote state changes across components and tabs.  
- **Insufficient error handling/logging** makes diagnosis and recovery difficult.

---

### 2. Proposed change (single, actionable plan)
Add a **real-time event pipeline** using **Server-Sent Events (SSE)** for unidirectional server→client push (simpler than WebSocket, HTTP-compatible, auto-reconnect) plus a **tiny frontend event bus** integrated with a shared store (Zustand or Context).  
Scope:  
- One backend route: `GET /api/events` (SSE, tenant-scoped).  
- One frontend module: `src/lib/events.ts` (SSE client + pub/sub).  
- Small integration in `App.tsx` and existing mutation handlers to emit domain events.  
- Graceful degradation: if SSE fails, fall back to safe polling or manual refresh with clear UX feedback.  
Target effort: <2 hours for minimal viable flow.

### 3. Implementation (corrected + production-ready)

#### Backend: SSE endpoint (Node/Express)
File: `/opt/axentx/workio/server/src/routes/events.ts`
```ts
import { Router } from 'express';
import { authenticate } from '../middleware/auth';

const router = Router();

// Simple in-memory pub/sub; replace with Redis/Postgres LISTEN/NOTIFY in prod
const listeners = new Set<(event: { type: string; payload: any }) => void>();
export function emitEvent(event: { type: string; payload: any }) {
  listeners.forEach((fn) => {
    try { fn(event); } catch {}
  });
}

router.get('/events', authenticate, (req, res) => {
  const tenantId = (req as any).user.tenantId;

  res.setHeader('Content-Type', 'text/event-stream');
  res.setHeader('Cache-Control', 'no-cache');
  res.setHeader('Connection', 'keep-alive');
  res.flushHeaders();

  // Immediate comment to detect connection open
  res.write(':\n\n');

  const onEvent = (event: { type: string; payload: any }) => {
    // Scope by tenant for multi-tenant safety
    if (event.payload?.tenantId !== tenantId) return;
    const data = JSON.stringify(event);
    res.write(`data: ${data}\n\n`);
  };

  listeners.add(onEvent);

  req.on('close', () => {
    listeners.delete(onEvent);
  });

  // Keepalive to avoid proxy timeouts
  const keepAlive = setInterval(() => res.write(':\n\n'), 20_000);
  req.on('close', () => clearInterval(keepAlive));
});

export default router;
```

#### Frontend: SSE client + pub/sub + store integration
File: `/opt/axentx/workio/src/lib/events.ts`
```ts
type Listener = (event: { type: string; payload: any }) => void;

class EventSourceClient {
  private es: EventSource | null = null;
  private listeners = new Set<Listener>();
  private reconnectDelay = 1000;
  private maxDelay = 30_000;
  private url: string;
  private token: string;

  constructor(url: string, token: string) {
    this.url = url;
    this.token = token;
  }

  connect() {
    if (this.es) return;

    // Use withCredentials for cookie-based auth if needed; token via search param or header polyfill
    const url = new URL(this.url, window.location.origin);
    // If your backend expects Authorization header, use a lightweight proxy or cookie auth.
    // For SSE with Bearer, many setups use a same-origin endpoint that reads cookies.
    this.es = new EventSource(url);

    this.es.onmessage = (ev) => {
      if (!ev.data || ev.data.startsWith(':')) return;
      try {
        const msg = JSON.parse(ev.data);
        this.listeners.forEach((l) => {
          try { l(msg); } catch {}
        });
      } catch {}
    };

    this.es.onerror = () => {
      this.disconnect();
      // Exponential backoff reconnect
      setTimeout(() => this.connect(), this.reconnectDelay);
      this.reconnectDelay = Math.min(this.reconnectDelay * 2, this.maxDelay);
    };
  }

  disconnect() {
    if (this.es) {
      this.es.close();
      this.es = null;
    }
  }

  subscribe(listener: Listener) {
    this.listeners.add(listener);
    return () => this.listeners.delete(listener);
  }
}

// Singleton instance; initialize in App.tsx after auth
let client: EventSourceClient | null = null;
export function initEvents(url: string, token: string) {
  if (!client) client = new EventSourceClient(url, token);
  client.connect();
  return client;
}
export function getEventClient() {
  return client;
}
```

Integrate with Zustand store (recommended) or Context  
Example store (`/opt/axentx/workio/src/stores/useWorkStore.ts`)
```ts
import { create } from 'zustand';

type ClockState = { status: string | null; ts: string | null };
type LeaveOTState = { requests: any[] };
type Store = ClockState & LeaveOTState & {
  setClockStatus: (v: ClockState) => void;
  upsertLeaveOT: (v: any) => void;
  applyRemoteEvent: (type: string, payload: any) => void;
};

export const useWorkStore = create<Store>((set) => ({
  status: null,
  ts: null,
  requests: [],
  setClockStatus: (v) => set(v),
  upsertLeaveOT: (v) => set((s) => ({ requests: [...s.requests.filter(r => r.id !== v.id), v] })),
  applyRemoteEvent: (type, payload) => {
    switch (type) {
      case 'clock:updated':
        set({ status: payload.status || payload.action, ts: payload.ts });
        break;
      case 'leave:updated':
      case 'ot:updated':
        set((s) => {
          const existing = s.requests.find((r) => r.id === payload.id);
          if (existing) {
            return { requests: s.requests.map((r) => (r.id === payload.id ? payload : r)) };
          }
          return { requests: [...s.requests, payload] };
        });
        break;
      default:
        break;
    }
  },
}));
```
