# Costinel / ops

## Implementation Plan — Top-Hub Signal Panel (CDN-first, <2h)

### Scope
Add a lightweight, non-blocking **Top-Hub Signal Panel** to the Costinel dashboard that:
- Detects the most-connected hub (default: `MOC`) from local graph metadata
- Shows 3 contextual insights (from knowledge-rag) without blocking UI
- Uses CDN-first asset loading (bypasses HF API during runtime)
- Falls back gracefully if graph/insights unavailable

### Implementation Steps (≤2h)

1. **Create graph metadata reader** (`src/lib/top-hub.ts`) — reads local JSON or computes top hub from edges.
2. **Create insights fetcher** (`src/lib/insights-cdn.ts`) — loads pre-baked insights JSON from CDN (or local fallback).
3. **Create React panel component** (`src/components/TopHubSignalPanel.tsx`) — renders hub + 3 insights, non-blocking, skeleton-first.
4. **Wire into dashboard** (`src/pages/Dashboard.tsx`) — mount panel in sidebar/top slot, lazy-load chunk.
5. **Add build/cdn asset** — include a small JSON at `/assets/top-hub-insights.json` (can be updated by ops pipeline).
6. **Ensure cron-safe script pattern** — if ops script generates insights, use Bash shebang + executable + `SHELL=/bin/bash` in crontab.

### Code Snippets

#### 1) Graph metadata reader (local, fast)
```ts
// src/lib/top-hub.ts
export interface HubNode {
  id: string;
  label: string;
  degree: number;
  tags?: string[];
}

export interface GraphMeta {
  hubs: HubNode[];
  generatedAt: string;
}

const FALLBACK_META: GraphMeta = {
  hubs: [{ id: 'MOC', label: 'MOC', degree: 1, tags: ['knowledge-rag', 'graph', 'hub'] }],
  generatedAt: new Date().toISOString(),
};

export async function getTopHub(): Promise<HubNode> {
  try {
    // Local graph metadata baked at build time or updated by ops
    const res = await fetch('/assets/graph-meta.json', { cache: 'no-store' });
    if (!res.ok) throw new Error('No graph meta');
    const meta: GraphMeta = await res.json();
    const top = meta.hubs.sort((a, b) => b.degree - a.degree)[0];
    return top || FALLBACK_META.hubs[0];
  } catch {
    return FALLBACK_META.hubs[0];
  }
}
```

#### 2) CDN-first insights loader (bypasses HF API)
```ts
// src/lib/insights-cdn.ts
export interface Insight {
  title: string;
  summary: string;
  source?: string;
  tags?: string[];
}

const FALLBACK_INSIGHTS: Insight[] = [
  {
    title: 'Review most-connected hub (MOC) before planning tasks',
    summary: 'Check the top hub to align context and reduce redundant work.',
    tags: ['knowledge-rag', 'graph', 'hub'],
  },
  {
    title: 'Use CDN URLs for public dataset files',
    summary: 'Use huggingface.co/datasets/{repo}/resolve/main/{path} to bypass API rate limits during training.',
    tags: ['huggingface', 'cdn', 'rate-limit-bypass'],
  },
  {
    title: 'Prefer Lightning Studio reuse to save quota',
    summary: 'List running studios and reuse them instead of recreating to save 80hr/mo quota.',
    tags: ['lightning-ai', 'quota'],
  },
];

export async function getTopHubInsights(): Promise<Insight[]> {
  try {
    // CDN path — can be updated by ops pipeline without touching app code
    const res = await fetch('/assets/top-hub-insights.json', { cache: 'no-store' });
    if (!res.ok) throw new Error('No CDN insights');
    const data: Insight[] = await res.json();
    return data.length > 0 ? data.slice(0, 3) : FALLBACK_INSIGHTS;
  } catch {
    return FALLBACK_INSIGHTS;
  }
}
```

#### 3) React panel (skeleton-first, non-blocking)
```tsx
// src/components/TopHubSignalPanel.tsx
import { useEffect, useState } from 'react';
import { getTopHub } from '../lib/top-hub';
import { getTopHubInsights } from '../lib/insights-cdn';

interface Insight {
  title: string;
  summary: string;
  source?: string;
  tags?: string[];
}

export default function TopHubSignalPanel() {
  const [hub, setHub] = useState<{ label: string; id: string } | null>(null);
  const [insights, setInsights] = useState<Insight[]>([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    let mounted = true;
    setLoading(true);
    Promise.all([getTopHub(), getTopHubInsights()])
      .then(([topHub, topInsights]) => {
        if (!mounted) return;
        setHub({ label: topHub.label, id: topHub.id });
        setInsights(topInsights);
      })
      .catch(() => {
        if (!mounted) return;
        setHub({ label: 'MOC', id: 'MOC' });
        setInsights([]);
      })
      .finally(() => {
        if (mounted) setLoading(false);
      });
    return () => {
      mounted = false;
    };
  }, []);

  if (loading) {
    return (
      <div className="rounded border border-gray-200 bg-gray-50 p-4">
        <div className="mb-2 h-5 w-32 animate-pulse rounded bg-gray-200" />
        <div className="space-y-2">
          <div className="h-4 w-full animate-pulse rounded bg-gray-200" />
          <div className="h-4 w-5/6 animate-pulse rounded bg-gray-200" />
        </div>
      </div>
    );
  }

  return (
    <div className="rounded border border-gray-200 bg-white p-4 shadow-sm">
      <div className="mb-3 flex items-center justify-between">
        <h3 className="text-sm font-semibold text-gray-900">Top Hub</h3>
        <span className="inline-flex items-center rounded-full bg-blue-50 px-2 py-0.5 text-xs font-medium text-blue-700">
          {hub?.id || 'MOC'}
        </span>
      </div>

      <div className="space-y-3">
        {insights.length > 0 ? (
          insights.map((insight, idx) => (
            <div key={idx} className="rounded border border-gray-100 bg-gray-50 p-3">
              <div className="text-xs font-medium text-gray-900">{insight.title}</div>
              <div className="mt-1 text-xs text-gray-600">{insight.summary}</div>
              {insight.tags && (
                <div className="mt-2 flex flex-wrap gap-1">
                  {insight.tags.map((t) => (
                    <span
                      key={t}
                      className="rounded bg-gray-200 px-1.5 py-0.5 text-[10px] font-medium text-gray-600"
                    >
                      {t}
                    </span>
                  ))}
                </div>
              )}
            </div>
          ))
        ) : (
          <div className="text-xs text-gray-500">No insights available.</div>
        )}
      </div>

      <div className="mt-3 text-[10px] text-gray-400">
        Source: knowledge-rag • Updated {new Date().toLocaleDateString()}
      </div>
    </div>
  );
}
```
