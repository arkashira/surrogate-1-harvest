# surrogate-1 / ops

## Implementation Plan (≤2h)

**Highest-value improvement**: Replace fragile shell-based ingestion with a **manifest-driven, CDN-bypass Python worker** that eliminates HF API rate limits during training data load and prevents mixed-schema `CastError`.

### What to change

- Add `bin/worker.py` — single, deterministic worker that:
  - Accepts `SHARD_ID` and `TOTAL_SHARDS` (matrix)
  - Uses one API call (after rate-limit window) to list a single date folder via `list_repo_tree(..., recursive=False)`
  - Persists file list to `manifest.json` (committed or embedded)
  - Downloads only assigned shard’s files via **CDN bypass** (`https://huggingface.co/datasets/.../resolve/main/...`) with no Authorization header
  - Projects each file to `{prompt, response}` at parse time (avoids `pyarrow` mixed-schema `CastError`)
  - Deduplicates via central md5 store (`lib/dedup.py`)
  - Writes `batches/public-merged/<date>/shard<N>-<HHMMSS>.jsonl`

- Update `.github/workflows/ingest.yml`:
  - Keep 16-shard matrix
  - Set `SHELL=/bin/bash`
  - Run via `python bin/worker.py` (not shell loops)
  - Export `SHARD_ID`, `TOTAL_SHARDS`, `DATE`, `HF_TOKEN`

- Update `requirements.txt`:
  - Add `requests` (CDN downloads), keep `datasets`, `huggingface_hub`, `pyarrow`, `numpy`

- Remove/reduce shell loops in `bin/dataset-enrich.sh` to a thin wrapper that calls `python bin/worker.py "$@"`.

### Why this is highest value

- Eliminates HF API rate limits during data loading (CDN bypass)
- Prevents `pyarrow.CastError` from heterogeneous repo files
- Cuts OOM risk by streaming per-file instead of `load_dataset(streaming=True)` across mixed schemas
- Makes retries deterministic and idempotent via manifest
- Keeps 16-shard parallelism and dedup semantics intact

#!/usr/bin/env python3
"""
CDN-bypass ingestion worker for surrogate-1 public dataset.
Usage (via GitHub Actions matrix):
  SHARD_ID=0 TOTAL_SHARDS=16 python bin/worker.py
"""
import os
import json
import hashlib
import datetime
import sys
from pathlib import Path

import requests
from huggingface_hub import HfApi, hf_hub_download

# Local dedup store (shared with HF Space)
sys.path.insert(0, str(Path(__file__).parent))
from lib.dedup import is_duplicate, mark_seen  # noqa: E402

HF_REPO = "datasets/axentx/surrogate-1-training-pairs"
API = HfApi()
HF_TOKEN = os.getenv("HF_TOKEN")
if not HF_TOKEN:
    print("HF_TOKEN required", file=sys.stderr)
    sys.exit(1)

HEADERS = {"Authorization": f"Bearer {HF_TOKEN}"}
CDN_ROOT = f"https://huggingface.co/{HF_REPO}/resolve/main"

SHARD_ID = int(os.getenv("SHARD_ID", 0))
TOTAL_SHARDS = int(os.getenv("TOTAL_SHARDS", 16))
DATE = os.getenv("DATE", datetime.datetime.utcnow().strftime("%Y-%m-%d"))
OUT_DIR = Path(f"batches/public-merged/{DATE}")
OUT_DIR.mkdir(parents=True, exist_ok=True)
TIMESTAMP = datetime.datetime.utcnow().strftime("%H%M%S")
OUT_FILE = OUT_DIR / f"shard{SHARD_ID}-{TIMESTAMP}.jsonl"

# One API call: list top-level date folder (non-recursive)
def list_date_folder(date: str):
    try:
        items = API.list_repo_tree(
            repo_id=HF_REPO,
            path=date,
            recursive=False,
            token=HF_TOKEN,
        )
        # items may be list of dicts or list of strings depending on lib version
        paths = []
        for it in items:
            p = it["path"] if isinstance(it, dict) else str(it)
            if p.endswith((".jsonl", ".parquet", ".json")):
                paths.append(p)
        return paths
    except Exception as e:
        print(f"list_repo_tree failed: {e}", file=sys.stderr)
        return []

def download_via_cdn(repo_path: str):
    url = f"{CDN_ROOT}/{repo_path}"
    # CDN does NOT require Authorization header; omit to bypass API rate limits
    resp = requests.get(url, timeout=60)
    resp.raise_for_status()
    return resp.content

def project_to_pair(content: bytes, path: str):
    """
    Project heterogeneous file to {prompt, response} only.
    Supports:
      - JSONL lines with any fields -> extract prompt/response or first text-like fields
      - Parquet -> read via pyarrow, select/coerce to string cols
    """
    if path.endswith(".parquet"):
        import pyarrow as pa
        import pyarrow.parquet as pq
        tbl = pq.read_table(pa.BufferReader(content))
        # Try common field names; fallback to first two string columns
        cols = tbl.column_names
        prompt_col = next((c for c in ("prompt", "instruction", "input", "question") if c in cols), None)
        response_col = next((c for c in ("response", "output", "answer") if c in cols), None)
        if prompt_col and response_col:
            prompts = tbl.column(prompt_col).to_pylist()
            responses = tbl.column(response_col).to_pylist()
        elif len(cols) >= 2:
            prompts = tbl.column(cols[0]).to_pylist()
            responses = tbl.column(cols[1]).to_pylist()
        else:
            return []
        out = []
        for p, r in zip(prompts, responses):
            if isinstance(p, str) and isinstance(r, str) and p.strip() and r.strip():
                out.append({"prompt": p.strip(), "response": r.strip()})
        return out

    # Assume JSONL
    lines = content.decode("utf-8", errors="replace").strip().splitlines()
    pairs = []
    for line in lines:
        line = line.strip()
        if not line:
            continue
        try:
            obj = json.loads(line)
        except Exception:
            continue
        if isinstance(obj, dict):
            prompt = obj.get("prompt") or obj.get("instruction") or obj.get("input") or obj.get("question")
            response = obj.get("response") or obj.get("output") or obj.get("answer")
            if isinstance(prompt, str) and isinstance(response, str) and prompt.strip() and response.strip():
                pairs.append({"prompt": prompt.strip(), "response": response.strip()})
    return pairs

def main():
    paths = list_date_folder(DATE)
    if not paths:
        print(f"No files found for {DATE}", file=sys.stderr)
        sys.exit(0)

    # Deterministic shard assignment by slug hash
    assigned = []
    for p in sorted(paths):
        slug = Path(p).stem
        h = int(hashlib.md5(slug.encode()).hexdigest(), 16)
        if h % TOTAL_SHARDS == SHARD_ID:
            assigned.append(p)

    print(f"Shard {SHARD_ID}/{TOTAL_SHARDS}: processing {len(assigned)} files")

    written = 0
    with OUT_FILE.open("w", encoding="utf-8") as fout:
        for repo_path in assigned:
            try:
                content = download_via_cdn(repo_path)
                pairs = project_to_pair(content, repo_path)
            except Exception as e:
                print(f"Failed {repo_path}: {e}", file=sys.stderr)
                continue

            for pair in pairs:
                text = json.dumps(pair, ensure_ascii=False)
                md5 = hashlib.md5(text.encode()).hexdigest()
                if is_duplicate(md5):
                    continue
                mark_seen(md5)
                fout.write(text + "\n")
                written += 1

    print(f"Shard {SHARD_ID} wrote {written} pairs to {OUT_FILE}")

    # Optional: commit via huggingface_hub if running
