# workio / quality

## 1. Diagnosis

- No idempotency layer for `/clock-in` and `/clock-out`; concurrent LINE webhook deliveries or retries can create duplicate punches for the same user on the same day.
- Missing partial unique index to enforce “one open punch per user” at the database level; race conditions can leave multiple `status='in'` rows.
- Punch mutations lack request-level idempotency key handling; retries from LINE or network blips cause double entries instead of safe replays.
- No lightweight audit trail for punch state transitions; debugging duplicate/ghost punches requires forensic log scraping.
- Observability gap: no structured metrics or alerts for duplicate-punch attempts or constraint violations.

## 2. Proposed change

- **Files**:  
  - `server/src/db/schema.sql` — add `punches.idempotency_key` (unique per user+day), partial unique index for open punches, and `punch_audit` table.  
  - `server/src/routes/punch.ts` — enforce idempotency key on write, upsert on conflict, and emit audit row.  
  - `server/src/middleware/lineWebhookIdempotency.ts` — extract idempotency key from LINE webhook (use `deliveryId` or hash of body) and attach to request.

## 3. Implementation

### 3.1 DB schema (`server/src/db/schema.sql`)

```sql
-- Add idempotency key to punches (unique per user per calendar day)
ALTER TABLE punches
  ADD COLUMN idempotency_key TEXT;

CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_user_day_idempotency
  ON punches (user_id, DATE(created_at), idempotency_key)
  WHERE idempotency_key IS NOT NULL;

-- Partial unique index: at most one open punch per user
CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_one_open_per_user
  ON punches (user_id)
  WHERE status = 'in';

-- Lightweight audit trail for punch state transitions
CREATE TABLE IF NOT EXISTS punch_audit (
  id            SERIAL PRIMARY KEY,
  punch_id      INTEGER REFERENCES punches(id) ON DELETE CASCADE,
  user_id       INTEGER NOT NULL,
  action        TEXT    NOT NULL, -- 'clock_in', 'clock_out'
  prev_status   TEXT,
  new_status    TEXT    NOT NULL,
  location      JSONB,
  idempotency_key TEXT,
  created_at    TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX IF NOT EXISTS idx_punch_audit_user_created
  ON punch_audit (user_id, created_at);
```

### 3.2 Idempotency middleware (`server/src/middleware/lineWebhookIdempotency.ts`)

```ts
import { Request, Response, NextFunction } from 'express';
import crypto from 'crypto';

export function lineWebhookIdempotency(
  req: Request,
  _res: Response,
  next: NextFunction
) {
  // Prefer LINE's deliveryId if present in headers; otherwise hash body
  const deliveryId = (req as any).headers['x-line-delivery-id'] as string | undefined;
  if (deliveryId) {
    (req as any).idempotencyKey = `line-delivery:${deliveryId}`;
  } else {
    const body = JSON.stringify(req.body || {});
    (req as any).idempotencyKey = `line-body:${crypto.createHash('sha256').update(body).digest('hex')}`;
  }
  next();
}
```

### 3.3 Punch route with idempotent upsert (`server/src/routes/punch.ts`)

```ts
import { Router, Request, Response } from 'express';
import { pool } from '../db';

const router = Router();

router.post(
  '/clock-in',
  async (req: Request & { idempotencyKey?: string }, res: Response) => {
    const { userId, latitude, longitude } = req.body;
    const idempotencyKey = req.idempotencyKey || req.body.idempotencyKey;

    if (!userId || !idempotencyKey) {
      return res.status(400).json({ error: 'userId and idempotencyKey required' });
    }

    const client = await pool.connect();
    try {
      await client.query('BEGIN');

      // Try insert; on idempotency conflict, return existing punch
      const insertQuery = `
        INSERT INTO punches (user_id, status, latitude, longitude, idempotency_key)
        VALUES ($1, 'in', $2, $3, $4)
        ON CONFLICT (user_id, DATE(created_at), idempotency_key)
        DO UPDATE SET user_id = EXCLUDED.user_id
        RETURNING id, status, created_at, latitude, longitude
      `;
      const insertRes = await client.query(insertQuery, [
        userId,
        latitude || null,
        longitude || null,
        idempotencyKey,
      ]);
      const punch = insertRes.rows[0];

      // Audit
      await client.query(
        `INSERT INTO punch_audit (punch_id, user_id, action, prev_status, new_status, location, idempotency_key)
         VALUES ($1, $2, 'clock_in', NULL, $3, $4, $5)`,
        [punch.id, userId, punch.status, JSON.stringify({ latitude, longitude }), idempotencyKey]
      );

      await client.query('COMMIT');
      return res.json({ punch, conflict: false });
    } catch (err: any) {
      await client.query('ROLLBACK');

      // If partial unique index violation (already clocked in), return existing open punch
      if (err.code === '23505' && err.constraint === 'idx_punches_one_open_per_user') {
        const existing = await pool.query(
          'SELECT id, status, created_at, latitude, longitude FROM punches WHERE user_id = $1 AND status = $2 ORDER BY created_at DESC LIMIT 1',
          [userId, 'in']
        );
        return res.json({ punch: existing.rows[0] || null, conflict: true, reason: 'already_clocked_in' });
      }

      console.error('Punch insert failed', err);
      return res.status(500).json({ error: 'Internal server error' });
    } finally {
      client.release();
    }
  }
);
