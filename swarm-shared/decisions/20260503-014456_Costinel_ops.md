### 2) Lightweight KG client (merged strategy)
**Strategy (resolved):**
- Try **CDN sample first** (bypasses auth/rate limits, fastest resilient path).
- If CDN fails, try **live API** with timeout/abort.
- If both fail, use **graceful fallback** proposals.
- Always attempt **localStorage cache** with TTL for offline/rapid repeat loads.
- Never block UI; surface errors non-intrusively.

```ts
// src/lib/knowledgeGraph.ts
import { CONFIG } from './config';

export interface Proposal {
  id: string;
  title: string;
  summary: string;
  impact: 'high' | 'medium' | 'low';
  action?: string;
  tags: string[];
}

export interface HubInsight {
  hub: string;
  description: string;
  proposals: Proposal[];
  updatedAt: string;
}

const SAMPLE_URL = '/mock/top-hub-sample.json';

async function fetchCDNSample(): Promise<HubInsight> {
  const res = await fetch(SAMPLE_URL, { cache: 'no-store' });
  if (!res.ok) throw new Error('CDN sample unavailable');
  return res.json();
}

async function fetchAPI(): Promise<HubInsight> {
  const controller = new AbortController();
  const timeout = setTimeout(() => controller.abort(), CONFIG.knowledgeGraph.timeoutMs);

  try {
    const res = await fetch(`${CONFIG.knowledgeGraph.apiBase}/top-hub`, {
      signal: controller.signal,
      headers: { Accept: 'application/json' },
    });
    clearTimeout(timeout);
    if (!res.ok) throw new Error(`API ${res.status}`);
    return res.json();
  } finally {
    clearTimeout(timeout);
  }
}

function isFresh(cachedAt: number) {
  return Date.now() - cachedAt < CONFIG.knowledgeGraph.cacheTTL;
}

export async function getTopHubInsight(): Promise<HubInsight | null> {
  const cacheKey = 'costinel:top-hub:insight';

  // 1) Valid cache
  try {
    const raw = localStorage.getItem(cacheKey);
    if (raw) {
      const parsed = JSON.parse(raw) as HubInsight & { _cachedAt: number };
      if (parsed && parsed.hub && Array.isArray(parsed.proposals) && isFresh(parsed._cachedAt)) {
        return parsed;
      }
    }
  } catch {
    // ignore malformed cache
  }

  // 2) Try CDN -> API -> fallback
  let result: HubInsight | null = null;
  try {
    result = await fetchCDNSample();
  } catch {
    try {
      result = await fetchAPI();
    } catch {
      // 3) Graceful fallback
      result = {
        hub: CONFIG.topHub.fallbackHub,
        description: 'Knowledge graph unavailable — showing default hub.',
        proposals: [
          {
            id: 'fallback-1',
            title: 'Review Reserved Instance coverage',
            summary: 'Check RI/SP coverage for top cost services to reduce on-demand spend.',
            impact: 'high',
            tags: ['cost-optimization', 'ri'],
          },
          {
            id: 'fallback-2',
            title: 'Enable idle resource detection',
            summary: 'Schedule weekly idle resource scans and tag owners for cleanup.',
            impact: 'medium',
            tags: ['governance', 'idle'],
          },
        ],
        updatedAt: new Date().toISOString(),
      };
    }
  }

  // 4) Update cache if we got real or fallback data
  if (result) {
    try {
      localStorage.setItem(cacheKey, JSON.stringify({ ...result, _cachedAt: Date.now() }));
    } catch {
      // ignore storage quota/errors
    }
  }

  return result;
}
```

import { useEffect, useState } from 'react';
import { getTopHubInsight, type HubInsight, type Proposal } from '@/lib/knowledgeGraph';
import { CONFIG } from '@/lib/config';

function ImpactBadge({ impact }: { impact: Proposal['impact'] }) {
  const colors = {
    high: 'bg-red-100 text-red-800',
    medium: 'bg-amber-100 text-amber-800',
    low: 'bg-green-100 text-green-800',
  };
  return <span className={`inline-flex items-center px-2 py-0.5 rounded text-xs font-medium ${colors[impact]}`}>{impact}</span>;
}

export default function TopHubSignalPanel() {
  const [insight, setInsight] = useState<HubInsight | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    if (!CONFIG.knowledgeGraph.enabled) {
      setLoading(false);
      return;
    }

    let mounted = true;
    setLoading(true);
    getTopHubInsight()
      .then((data) => {
        if (mounted) setInsight(data);
      })
      .catch((err) => {
        if (mounted) setError(err instanceof Error ? err.message : 'Unknown error');
      })
      .finally(() => {
        if (mounted) setLoading(false);
      });

    return () => {
      mounted = false;
    };
  }, []);

  if (!CONFIG.knowledgeGraph.enabled) {
    return null;
  }

  return (
    <section aria-label="Top hub signal" className="rounded-lg border bg-white p-4 shadow-sm">
      <div className="flex items-start justify-between mb-3">
        <div>
          <h2 className="text-base font-semibold text-gray-900">Top Hub Signal</h2>
          <p className="text-sm text-gray-500">Most-connected hub and actionable proposals</p>
        </div>
        {loading && <span className="text-xs text-gray-400">Loading…</span>}
      </div>

        <div className="mb-3 rounded-md bg-yellow-50 px-3 py-2 text-xs text-yellow-800" role="status">
          {error}
        </div>
