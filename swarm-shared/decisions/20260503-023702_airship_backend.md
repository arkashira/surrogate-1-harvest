# airship / backend

## Final Implementation Plan — CDN-first Surrogate Training (airship/backend)

**Scope**: `/opt/axentx/airship/surrogate`  
**Goal**: Eliminate HF 429 + Lightning quota waste; enable reproducible, zero-API training runs; ship an incremental, production-ready slice in <2h.

---

### 1) Core Design Decisions (resolved)

- **File listing**: single `list_repo_tree` call per date folder (orchestration only), saved to `file_list.json`.  
- **Training data fetch**: Lightning training uses **CDN-only** (`https://huggingface.co/datasets/.../resolve/main/...`) — no API calls, no auth, bypasses 429.  
- **Schema**: project to `{prompt, response}` at parse time; move attribution to filename (`batches/mirror-merged/{date}/{slug}.parquet`). No extra columns.  
- **Lightning reuse**: list running studios; reuse if name/status match. Auto-restart if idle-killed.  
- **HF writes**: sibling repos (5) + deterministic hash-slug routing to spread 128/hr caps.  
- **Kaggle push**: raw HTTP Bearer + `slug`/`newTitle`/`text` + `isPrivate: true`.  
- **Deterministic routing**: use content hash (or date+slug) to pick HF sibling repo and Kaggle title to avoid collisions and rate-limit bursts.  
- **Idempotency & resumability**: track processed files in a local state file; skip already-processed entries on restart.  
- **Observability**: emit structured logs (JSON) and minimal metrics (files processed, rows yielded, errors) to stdout for CI/monitoring.

#### A) Add CDN-first dataset loader with resumability (`surrogate/data/cdn_loader.py`)

```python
# surrogate/data/cdn_loader.py
import json
import pyarrow.parquet as pq
import pyarrow as pa
import requests
from io import BytesIO
from typing import List, Dict, Iterator, Optional, Set
import hashlib
import os
import logging

logging.basicConfig(level=logging.INFO, format="%(asctime)s %(levelname)s %(message)s")
logger = logging.getLogger(__name__)

HF_DATASET_REPO = os.getenv("HF_DATASET_REPO", "axentx/surrogate-mirror")
CDN_ROOT = f"https://huggingface.co/datasets/{HF_DATASET_REPO}/resolve/main"

def load_file_list(file_list_path: str) -> List[str]:
    with open(file_list_path) as f:
        data = json.load(f)
    # expected: list of relative paths under the date folder
    if isinstance(data, dict) and "files" in data:
        return data["files"]
    return data

def load_state(state_path: str) -> Set[str]:
    if os.path.exists(state_path):
        with open(state_path) as f:
            return set(json.load(f))
    return set()

def save_state(state_path: str, processed: Set[str]) -> None:
    with open(state_path, "w") as f:
        json.dump(sorted(processed), f)

def cdn_fetch(path: str, timeout: int = 30) -> bytes:
    url = f"{CDN_ROOT}/{path}"
    r = requests.get(url, timeout=timeout)
    r.raise_for_status()
    return r.content

def project_to_prompt_response(batch_bytes: bytes) -> Iterator[Dict[str, str]]:
    table = pq.read_table(BytesIO(batch_bytes))
    # strict projection; ignore extra cols
    if "prompt" not in table.column_names or "response" not in table.column_names:
        raise ValueError("Missing prompt/response in parquet")
    prompts = table["prompt"].to_pylist()
    responses = table["response"].to_pylist()
    for p, r in zip(prompts, responses):
        if isinstance(p, str) and isinstance(r, str) and p.strip() and r.strip():
            yield {"prompt": p.strip(), "response": r.strip()}

def stream_cdn_dataset(
    file_list_path: str,
    state_path: Optional[str] = None,
    flush_every: int = 100
) -> Iterator[Dict[str, str]]:
    processed = load_state(state_path) if state_path else set()
    to_process = load_file_list(file_list_path)
    total = len(to_process)
    yielded = 0
    errors = 0

    for idx, rel_path in enumerate(to_process, start=1):
        if rel_path in processed:
            continue
        try:
            blob = cdn_fetch(rel_path)
            count_before = yielded
            for item in project_to_prompt_response(blob):
                yielded += 1
                yield item
            processed.add(rel_path)
            if state_path and (yielded - count_before) > 0:
                # lightweight: save state after each successful file
                save_state(state_path, processed)
            if idx % flush_every == 0 or idx == total:
                logger.info(f"progress: {idx}/{total} files, {yielded} rows, {errors} errors")
        except Exception as exc:
            errors += 1
            logger.warning(f"SKIP {rel_path}: {exc}")
            continue

    logger.info(f"finished: {yielded} rows from {len(processed)} files, {errors} errors")
```

#### B) Lightning training launcher with robust studio reuse (`surrogate/train/launch.py`)

```python
# surrogate/train/launch.py
import os
import sys
import time
import json
from pathlib import Path
from lightning_sdk import Studio, Teamspace, Machine

def find_or_create_studio(name: str, machine: Machine = Machine.L40S) -> Studio:
    teamspace = Teamspace()
    for s in teamspace.studios:
        if s.name == name:
            if s.status == "Running":
                print(f"Reusing running studio: {name}")
                return s
            else:
                print(f"Restarting stopped studio: {name}")
                try:
                    s.delete()
                except Exception:
                    pass
                break
    print(f"Creating studio: {name}")
    return Studio(
        name=name,
        machine=machine,
        create_ok=True,
    )

def run_training(file_list_path: str, state_path: str):
    studio = find_or_create_studio("surrogate-train-l40s", Machine.L40S)
    script_path = Path(__file__).parent / "train.py"
    # pass file list and state so train.py does CDN-only fetches and is resumable
    run = studio.run(
        command=[
            "python", str(script_path),
            "--file-list", file_list_path,
            "--state", state_path,
            "--epochs", "1",
        ],
        requirements=[
            "torch",
            "pyarrow",
            "requests",
            "transformers",
            "accelerate",
        ],
    )
    print(f"Run submitted: {run.id}")
    # poll status; if studio stops, restart and resubmit
    while True:
        run.refresh()
        if run.status in ("completed", "failed", "crashed"):
            print(f"Run finished: {run.status}")
            break
        studio.refresh()
        if studio.status != "Running":
            print("Studio stopped; restarting and resubmitting...")
            studio = find_or_create_studio("surrogate-train-l40s", Machine.L40S)
            run = studio.run(
                command=[
                    "python", str(script_path),
                    "--file-list", file_list_path,
                    "--state", state_path,
                    "--epochs", "1",
                ],
                requirements=[
                    "torch",
                    "pyarrow",
                    "requests",
                    "transformers",
                    "accelerate",
                ],
            )
            print(f"Resubmitted: {run.id}")
        time.sleep(30)

if __name__ == "__main__":
    if len(sys.argv) < 2:
        print("Usage: python launch.py <file_list.json> [state.json]")
        sys.exit(1)
    file_list = sys.argv[1]
    state = sys.argv[2] if len(sys.argv) > 2 else "state_processed.json"
    run_training(file_list, state)
```
