- Frontend/training repeatedly call authenticated `list_repo_tree` on every selection/launch → burns HF API quota (1000/5min) → 429s.
- No persisted `(repo, dateFolder) → file-list` manifest → forces re-enumeration and blocks CDN-only training.

- No reuse of running Lightning Studio (wastes quota) and no idle-stop resilience (training dies if studio stops).
- Mac/local orchestration is fine; training must run in Lightning Studio; ingestion should use HF Space or CDN downloads.

### 2) `/opt/axentx/vanguard/ops/preflight.py`
```python
#!/usr/bin/env python3
"""
preflight.py
Generate and cache repo+dateFolder file manifest.

Usage:
  HF_TOKEN=<token> python3 preflight.py <repo> <date_folder> [--ttl 86400] [--out ./manifests/...]

Produces JSON:
{
  "repo": "...",
  "dateFolder": "...",
  "generated": 1712345678,
  "ttl": 86400,
  "files": ["path1", "path2", ...]
}
"""
import argparse
import json
import os
import sys
import time
import fcntl
from pathlib import Path

try:
    from huggingface_hub import HfApi
except ImportError:
    print("ERROR: huggingface_hub not installed. Install with: pip install huggingface_hub")
    sys.exit(1)

def parse_args():
    parser = argparse.ArgumentParser(description="Generate cached HF file manifest.")
    parser.add_argument("repo", help="HF repo (e.g. bigcode/the-stack)")
    parser.add_argument("date_folder", help="Folder path under repo (e.g. python/2024-05-01)")
    parser.add_argument("--ttl", type=int, default=86400, help="Cache TTL in seconds (default 86400)")
    parser.add_argument("--out", default=None, help="Output JSON path (default: manifests/{repo}__{date_folder}.json)")
    parser.add_argument("--hf-token", default=os.getenv("HF_TOKEN", None), help="HF token (or use HF_TOKEN env)")
    return parser.parse_args()

def safe_hf_api(token):
    return HfApi(token=token)

def list_files(api, repo, folder):
    # Non-recursive to minimize pages; we only need top-level enumeration for this dateFolder.
    tree = api.list_repo_tree(repo=repo, path=folder, recursive=False)
    files = sorted([item.rfilename for item in tree if item.type == "file"])
    return files

def main():
    args = parse_args()
    out_path = Path(args.out) if args.out else Path("manifests") / f"{args.repo.replace('/', '__')}__{args.date_folder.replace('/', '_')}.json"
    out_path.parent.mkdir(parents=True, exist_ok=True)

    # Lock to avoid concurrent runs corrupting manifest
    lock_path = out_path.with_suffix(".lock")
    with open(lock_path, "w") as lock_f:
        try:
            fcntl.flock(lock_f, fcntl.LOCK_EX | fcntl.LOCK_NB)
        except BlockingIOError:
            print(f"Another preflight is running (lock {lock_path}). Exiting.")
            sys.exit(0)

        try:
            # Check cache
            use_cache = False
            if out_path.exists():
                try:
                    with open(out_path) as f:
                        cached = json.load(f)
                    generated = cached.get("generated", 0)
                    ttl = cached.get("ttl", args.ttl)
                    if time.time() - generated < ttl:
                        print(f"Using cached manifest: {out_path}")
                        use_cache = True
                except Exception:
                    pass

            if use_cache:
                # Validate minimal structure
                if not isinstance(cached.get("files"), list):
                    use_cache = False

            if not use_cache:
                print(f"Generating manifest for {args.repo}/{args.date_folder} ...")
                api = safe_hf_api(args.hf_token)
                files = list_files(api, args.repo, args.date_folder)
                manifest = {
                    "repo": args.repo,
                    "dateFolder": args.date_folder,
                    "generated": int(time.time()),
                    "ttl": args.ttl,
                    "files": files
                }
                with open(out_path, "w") as f:
                    json.dump(manifest, f, indent=2)
                print(f"Wrote {len(files)} files to {out_path}")
            else:
                manifest = cached

            # Print summary for CI/automation
            print(json.dumps({
                "manifest": str(out_path),
                "repo": manifest["repo"],
                "dateFolder": manifest["dateFolder"],
                "fileCount": len(manifest["files"]),
                "generated": manifest["generated"],
                "ttl": manifest["ttl"]
            }))
        finally:
            fcntl.flock(lock_f, fcntl.LOCK_UN)
            try:
                lock_path.unlink(missing_ok=True)
            except Exception:
                pass

if __name__ == "__main__":
    main()
```

Make executable:
```bash
chmod +x /opt/axentx/vanguard/ops/preflight.py
```
