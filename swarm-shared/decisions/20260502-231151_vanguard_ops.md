# vanguard / ops

## 1. Diagnosis
- No durable ingestion manifest: every training run re-lists HF repos and re-downloads, causing 429s and wasted bandwidth.
- Training loads via HF API (`load_dataset`/`list_repo_files`) instead of CDN bypass → guaranteed rate limits during data loading.
- No Lightning Studio reuse: scripts create new studios each run, burning 80+ hrs/mo quota.
- No idle-stop guard: Lightning idle timeout kills training; no auto-restart on stopped studios.
- No deterministic routing for HF commit cap: all writes target one repo, risking 128/hr cap blocks.

## 2. Proposed change
Create `/opt/axentx/vanguard/ops/training_orchestrator.py` (single orchestrator) that:
- Persists a dated manifest of file paths to `batches/mirror-merged/{date}/files.json` after a single HF API list call.
- Embeds that manifest in training so Lightning workers fetch exclusively via CDN (zero API calls during training).
- Reuses a running Lightning Studio by name, or starts one if absent.
- Guards idle-stop by checking status before `.run()` and auto-restarts the studio if stopped.
- Routes HF writes across 5 sibling repos by deterministic hash-slug.

## 3. Implementation

```bash
# /opt/axentx/vanguard/ops/training_orchestrator.py
#!/usr/bin/env bash
set -euo pipefail
SHELL=/bin/bash

# Config
HF_REPO_BASE="datasets/axentx/mirror-merged"
DATE=$(date +%Y-%m-%d)
MANIFEST_DIR="batches/mirror-merged/${DATE}"
MANIFEST_FILE="${MANIFEST_DIR}/files.json"
STUDIO_NAME="vanguard-surrogate-train"
MACHINE="lightning-lambda-prod/L40S"  # or fallback to public tier
SIBLING_COUNT=5

# Helpers
log() { echo "[$(date -Iseconds)] $*"; }

# 1) Generate or reuse manifest (run once per date from Mac)
generate_manifest() {
  log "Checking for manifest: ${MANIFEST_FILE}"
  if python3 -c "import json,sys; json.load(open('${MANIFEST_FILE}'))" 2>/dev/null; then
    log "Manifest exists and valid. Skipping HF list_repo_tree."
    return 0
  fi

  log "Generating manifest via HF API (single call)."
  python3 <<PY
import json, os, sys
from huggingface_hub import HfApi
api = HfApi()
# Non-recursive per-date folder to avoid 100x pagination
files = api.list_repo_tree(repo_id="${HF_REPO_BASE}", path="${MANIFEST_DIR}", repo_type="dataset", recursive=False)
paths = [f.rfilename for f in files if f.rfilename.endswith(".parquet")]
os.makedirs("${MANIFEST_DIR}", exist_ok=True)
with open("${MANIFEST_FILE}", "w") as f:
    json.dump(paths, f, indent=2)
log(f"Wrote {len(paths)} files to ${MANIFEST_FILE}")
PY
}

# 2) Lightning Studio reuse + idle guard
run_training() {
  log "Checking Lightning Studio: ${STUDIO_NAME}"
  python3 <<PY
import os, time, sys
from lightning_sdk import Studio, Machine
from pathlib import Path

MANIFEST_FILE = "${MANIFEST_FILE}"
STUDIO_NAME = "${STUDIO_NAME}"
MACHINE = "${MACHINE}"

def find_running():
    for s in Studio.list():
        if s.name == STUDIO_NAME and s.status == "running":
            return s
    return None

studio = find_running()
if studio is None:
    log("No running studio found. Starting...")
    studio = Studio.create(
        name=STUDIO_NAME,
        machine=Machine(MACHINE),
        create_ok=True,
    )
    # wait until running
    while studio.status != "running":
        time.sleep(10)
        studio = Studio.get(studio.id)
else:
    log(f"Reusing running studio {studio.id}")

# Guard: if stopped, restart
if studio.status != "running":
    log("Studio not running. Restarting...")
    studio.start(machine=Machine(MACHINE))
    while studio.status != "running":
        time.sleep(10)
        studio = Studio.get(studio.id)

# Run training script with embedded manifest (CDN-only)
run = studio.run(
    command=[
        "python3", "train.py",
        "--manifest", MANIFEST_FILE,
        "--epochs", "1",
        "--batch-size", "8"
    ],
    wait=False,
)
log(f"Started run {run.id}")
PY
}

# 3) Deterministic HF sibling routing
pick_sibling_repo() {
  local slug="$1"
  local idx=$(echo -n "$slug" | sha256sum | tr -d -c 0-9 | head -c 8)
  idx=$(( 10#$idx % SIBLING_COUNT ))
  echo "${HF_REPO_BASE}-sib${idx}"
}

# Main
main() {
  generate_manifest
  run_training
}
main "$@"
```

```python
# /opt/axentx/vanguard/train.py  (minimal CDN-only loader snippet)
import json, os, requests, pyarrow.parquet as pq
from torch.utils.data import IterableDataset

class CDNParquetDataset(IterableDataset):
    def __init__(self, manifest_path):
        with open(manifest_path) as f:
            self.files = json.load(f)
        self.base = "https://huggingface.co/datasets/axentx/mirror-merged/resolve/main"

    def __iter__(self):
        for fn in self.files:
            url = f"{self.base}/{fn}"
            # stream CDN file (no HF API/auth/rate-limit)
            with requests.get(url, stream=True, timeout=30) as r:
                r.raise_for_status()
                with open("/tmp/current.parquet", "wb") as tmp:
                    for chunk in r.iter_content(chunk_size=8192):
                        tmp.write(chunk)
            table = pq.read_table("/tmp/current.parquet")
            # project to {prompt,response} only
            for batch in table.to_batches(columns=["prompt", "response"]):
                for row in zip(batch[0].to_pylist(), batch[1].to_pylist()):
                    yield {"prompt": row[0], "response": row[1]}
```
