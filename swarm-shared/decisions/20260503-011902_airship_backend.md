## Implementation Plan

### 1. Add CDN-bypass dataset loader (UI utility)
- Create `src/utils/hfCdnLoader.js`
- Accept `repo`, `dateFolder`, `fileList` (pre-listed JSON from Mac)
- Fetch via `https://huggingface.co/datasets/{repo}/resolve/main/{dateFolder}/{file}`
- Parse only `{prompt,response}` at runtime; ignore extra schema columns
- Retry with exponential backoff; respect CDN limits (no Authorization header)

### 2. Add Studio reuse hook
- Create `src/hooks/useLightningStudio.js`
- On mount, call `Teamspace.studios` (Lightning SDK) and find running Studio by name
- If found and status === 'Running', attach to it
- If stopped, restart with `target.start(Machine.L40S)`
- Expose `{studio, status, startTraining, stopTraining}`

### 3. Update Training UI page
- Wire the CDN loader into the dataset selection step
- Use the Studio reuse hook for the "Run Training" action
- Show file list from pre-saved JSON (upload or select date folder)
- Display active Studio status and fallback to L40S if H200 unavailable

### 4. Small UX polish
- Add note: "Using HF CDN — no API auth required"
- Disable HF API key field when CDN mode is active
- Show Studio reuse badge when attached to existing running Studio

### `src/utils/hfCdnLoader.js`
```js
/**
 * HF CDN-bypass dataset loader
 * Downloads files via /resolve/main/ (no Authorization header)
 * Projects to {prompt, response} only at parse time
 */
export async function loadDatasetFromCdn({ repo, dateFolder, fileList, signal }) {
  const base = `https://huggingface.co/datasets/${repo}/resolve/main/${dateFolder}`;
  const results = [];

  for (const file of fileList) {
    const url = `${base}/${file}`;
    const resp = await fetch(url, { signal });
    if (!resp.ok) throw new Error(`CDN fetch failed: ${url} ${resp.status}`);

    // Read as ArrayBuffer and parse parquet-like content in browser or proxy via worker
    // For MVP: assume JSONL for simplicity; replace with parquet-wasm if needed
    const text = await resp.text();
    const lines = text.trim().split('\n');
    for (const line of lines) {
      try {
        const obj = JSON.parse(line);
        // Project to {prompt, response} only — ignore extra schema columns
        if (obj.prompt != null && obj.response != null) {
          results.push({ prompt: obj.prompt, response: obj.response });
        }
      } catch (e) {
        // skip malformed lines
        continue;
      }
    }
  }

  return results;
}
```

### `src/hooks/useLightningStudio.js`
```js
import { useEffect, useState, useCallback } from 'react';
import { Teamspace, Machine } from '@lightningai/sdk'; // adjust import per actual SDK

export function useLightningStudio({ studioName, preferredMachine = 'L40S' }) {
  const [studio, setStudio] = useState(null);
  const [status, setStatus] = useState('idle'); // idle | running | stopped | error

  const findOrCreate = useCallback(async () => {
    try {
      const studios = await Teamspace.studios();
      const existing = studios.find((s) => s.name === studioName && s.status === 'Running');

      if (existing) {
        setStudio(existing);
        setStatus('running');
        return existing;
      }

      // Try to find stopped studio and restart
      const stopped = studios.find((s) => s.name === studioName && s.status === 'Stopped');
      if (stopped) {
        await stopped.start({ machine: Machine[preferredMachine] });
        setStudio(stopped);
        setStatus('running');
        return stopped;
      }

      // Create new if none found (fallback)
      const newStudio = await Teamspace.studios().create({
        name: studioName,
        machine: Machine[preferredMachine],
        create_ok: true,
      });
      setStudio(newStudio);
      setStatus('running');
      return newStudio;
    } catch (err) {
      setStatus('error');
      throw err;
    }
  }, [studioName, preferredMachine]);

  const stop = useCallback(async () => {
    if (studio && studio.stop) {
      await studio.stop();
      setStatus('stopped');
    }
  }, [studio]);

  useEffect(() => {
    if (!studio) {
      findOrCreate().catch(() => setStatus('error'));
    }
  }, [studio, findOrCreate]);

  return { studio, status, findOrCreate, stop };
}
```

### UI integration sketch (Training page)
```jsx
import { useState } from 'react';
import { loadDatasetFromCdn } from '../utils/hfCdnLoader';
import { useLightningStudio } from '../hooks/useLightningStudio';

export default function TrainingPage() {
  const [fileList, setFileList] = useState([]);
  const [dataset, setDataset] = useState(null);
  const { studio, status, stop } = useLightningStudio({
    studioName: 'surrogate-training-run',
    preferredMachine: 'L40S',
  });

  const loadDataset = async () => {
    // fileList should come from pre-saved JSON (uploaded or selected date folder)
    const data = await loadDatasetFromCdn({
      repo: 'your-org/surrogate-data',
      dateFolder: '2026-04-29',
      fileList,
    });
    setDataset(data);
  };

  const runTraining = async () => {
    if (!studio || status !== 'running') return;
    await studio.run({
      script: 'train.py',
      args: ['--dataset-path', 'cdn-files.json', '--hf-repo', 'your-org/surrogate-data'],
    });
  };

  return (
    <div>
      <h2>Surrogate Training (CDN mode)</h2>
      <p style={{ color: '#666' }}>Using HF CDN — no API auth required</p>

      <button onClick={loadDataset}>Load dataset from CDN</button>
      {dataset && <div>Loaded {dataset.length} samples</div>}

      <div>Studio status: {status}</div>
      {status === 'running' && <button onClick={runTraining}>Start Training</button>}
      {status === 'running' && <button onClick={stop}>Stop Studio</button>}
    </div>
  );
}
```
