# workio / backend

## 1. Diagnosis

- Missing tenant cache on webhook path: every LINE webhook hits `channelId → tenant` lookup (DB + network) on bursts; no LRU in-memory cache.
- No request-scoped correlation ID: logs across webhook → service → DB lack traceability for incident/debug.
- LINE signature verification duplicated and inconsistent across routes; no centralized defense-in-depth middleware.
- Health/readiness endpoints absent (k8s/PM2 unfriendly) — no `/healthz` or `/readyz`.
- Structured logging missing severity/context (no `level`, `traceId`, `spanId`); hard to pipe to ELK/Datadog.

## 2. Proposed change

File scope:
- `workio/server/src/middleware/`
  - `correlation.ts` (new)
  - `line-signature.ts` (new)
  - `tenant-cache.ts` (new)
- `workio/server/src/utils/logger.ts` (modify)
- `workio/server/src/index.ts` (modify) — add middleware and health routes
- `workio/server/src/services/tenantService.ts` (modify) — add cached lookup

### 3.3 Correlation middleware (`src/middleware/correlation.ts`)
```ts
import { Request, Response, NextFunction } from 'express';
import logger from '../utils/logger';

export function correlationMiddleware(req: Request, res: Response, next: NextFunction) {
  const traceId = req.headers['x-trace-id'] || req.headers['x-request-id'] || crypto.randomUUID();
  const spanId = crypto.randomUUID();

  (req as any).traceId = traceId;
  (req as any).spanId = spanId;

  const childLogger = logger.child({ traceId, spanId, method: req.method, path: req.path });
  (req as any).log = childLogger;

  res.setHeader('x-trace-id', traceId as string);
  next();
}
```

### 3.4 Tenant cache (`src/middleware/tenant-cache.ts`)
```ts
import LRU from 'lru-cache';
import { findTenantByChannelId } from '../services/tenantService';
import logger from '../utils/logger';

const cache = new LRU<string, { tenantId: string; updatedAt: number }>({
  max: 500,
  ttl: 1000 * 60 * 10, // 10m
});

export async function getTenantIdByChannelId(channelId: string): Promise<string | null> {
  const cached = cache.get(channelId);
  if (cached) return cached.tenantId;

  try {
    const tenant = await findTenantByChannelId(channelId);
    if (!tenant) return null;
    cache.set(channelId, { tenantId: tenant.id, updatedAt: Date.now() });
    return tenant.id;
  } catch (err) {
    logger.error({ err, channelId }, 'Failed to resolve tenant');
    return null;
  }
}

export function invalidateTenantCache(channelId: string) {
  cache.delete(channelId);
}
```

### 3.5 LINE signature verification (`src/middleware/line-signature.ts`)
```ts
import { Request, Response, NextFunction } from 'express';
import crypto from 'crypto';
import logger from '../utils/logger';

const CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';

export function lineSignatureMiddleware(req: Request, res: Response, next: NextFunction) {
  if (!CHANNEL_SECRET) {
    (req as any).log?.warn('LINE_CHANNEL_SECRET missing, skipping signature verification');
    return next();
  }

  const signature = req.headers['x-line-signature'] as string | undefined;
  if (!signature) {
    (req as any).log?.warn({ headers: req.headers }, 'Missing x-line-signature');
    return res.status(401).json({ error: 'Missing signature' });
  }

  const body = JSON.stringify(req.body);
  const expected = crypto
    .createHmac('sha256', CHANNEL_SECRET)
    .update(body, 'utf8')
    .digest('base64');

  if (!crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected))) {
    (req as any).log?.warn({ signature, expected: expected.slice(0, 8) + '...' }, 'Invalid LINE signature');
    return res.status(401).json({ error: 'Invalid signature' });
  }

  next();
}
```

### 3.6 Update tenant service (`src/services/tenantService.ts`)
```ts
import pool from '../db';
import logger from '../utils/logger';

export async function findTenantByChannelId(channelId: string) {
  try {
    const result = await pool.query(
      'SELECT id, name FROM tenants WHERE line_channel_id = $1 LIMIT 1',
      [channelId]
    );
    return result.rows[0] || null;
  } catch (err) {
    logger.error({ err, channelId }, 'DB error in findTenantByChannelId');
    throw err;
  }
}
```

app.get('/healthz', (req, res) => res.json({ status: 'ok', uptime: process.uptime() }));

    // lightweight DB ping
    await pool.query('SELECT 1');
    res.json({ status: 'ready' });

    logger.error({ err }, 'Readiness check failed');
    res.status(503).json({ status: 'unavailable' });
  }
});

// Apply LINE signature only to webhook route(s)
