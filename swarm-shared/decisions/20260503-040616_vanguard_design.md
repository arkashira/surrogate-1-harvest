# vanguard / design

Below is the **single, consolidated final answer**.  
It merges the strongest, non-contradictory parts of both proposals and resolves remaining differences in favor of **correctness, CDN-only data loading, quota-safe Lightning Studio reuse, and concrete actionability**.

---

## 1. Diagnosis (merged)
- No content-addressed manifest → ingestion/training re-list HF repos at runtime → 429 rate-limits and non-reproducible runs.
- Mixed-schema files from `dataset-mirror` land in `enriched/` without projection to `{prompt,response}` → breaks surrogate-1 training and causes pyarrow `CastError` on schema drift.
- Lightning Studio is recreated on every run → burns ~80 hrs/mo quota and risks idle-stop loss.
- Training uses HF API (`list/stream/load_dataset`) during data loading instead of CDN-only fetches → triggers auth-tier rate limits.
- No deterministic shard-to-repo mapping for commit-cap avoidance when writing enriched parquet shards.

---

## 2. Proposed change (merged + resolved)
Create **two small, focused files** under `/opt/axentx/vanguard/design/`:

1. `/opt/axentx/vanguard/design/manifest.py`  
   - One-time manifest builder (run from Mac/orchestration node).  
   - Single `list_repo_tree` API call per date folder.  
   - Saves `manifest-{date}.json` containing only CDN URLs (no auth required at training time).

2. `/opt/axentx/vanguard/design/train_with_manifest.py`  
   - Training entrypoint for Lightning Studio.  
   - Loads data via CDN URLs only (no HF API/auth during training).  
   - Projects mixed schemas to `{prompt,response}`.  
   - Accepts `--manifest` and optional `--limit`.

Plus a **minimal patch** to `/opt/axentx/vanguard/surrogate-1/train.py` (≤20 lines) to accept `--manifest` and use the CDN loader.

Plus **Lightning Studio reuse helper** (orchestration side) to find or restart a studio by name.

Plus **deterministic sibling repo selection** for write-side commit-cap avoidance.

### 3.2 `/opt/axentx/vanguard/design/manifest.py`
```python
# /opt/axentx/vanguard/design/manifest.py
import json
import hashlib
import os
import argparse
from pathlib import Path
from typing import List, Dict

HF_DATASET_REPO = os.getenv("HF_DATASET_REPO", "datasets/example/mirror-merged")
SIBLING_REPOS = [
    f"{HF_DATASET_REPO}-sib{i}" for i in range(5)
]  # 5 siblings => ~640 commits/hr aggregate cap

try:
    from huggingface_hub import list_repo_tree
except ImportError as e:
    raise RuntimeError("huggingface_hub required to generate manifests") from e

MANIFESTS_DIR = Path(__file__).parent.parent / "manifests"
MANIFESTS_DIR.mkdir(exist_ok=True, parents=True)

def list_date_folder(date_folder: str, recursive: bool = False) -> List[str]:
    """
    One-time API call (run after rate-limit window clears).
    Returns filenames like '2026-04-29/file-0001.parquet'.
    """
    tree = list_repo_tree(HF_DATASET_REPO, path=date_folder, recursive=recursive)
    return sorted([item.rfilename for item in tree if not item.rfilename.endswith("/")])

def build_manifest(date_folder: str, out_path: Path) -> Dict:
    paths = list_date_folder(date_folder, recursive=False)
    manifest = {
        "repo": HF_DATASET_REPO,
        "date_folder": date_folder,
        "files": paths,
        "cdn_prefix": f"https://huggingface.co/datasets/{HF_DATASET_REPO}/resolve/main",
    }
    out_path.write_text(json.dumps(manifest, indent=2))
    return manifest

def pick_sibling_repo(slug: str) -> str:
    """Deterministic shard -> sibling repo to avoid HF commit cap."""
    digest = int(hashlib.sha256(slug.encode()).hexdigest(), 16)
    return SIBLING_REPOS[digest % len(SIBLING_REPOS)]

if __name__ == "__main__":
    ap = argparse.ArgumentParser()
    ap.add_argument("--date", required=True, help="YYYY-MM-DD folder in dataset")
    ap.add_argument("--out", default=None, help="Output filename (default: manifests/manifest-{date}.json)")
    args = ap.parse_args()

    out_name = args.out or f"manifest-{args.date}.json"
    out_path = MANIFESTS_DIR / out_name
    build_manifest(args.date, out_path)
    print(f"Wrote {out_path}")
```

### 3.3 `/opt/axentx/vanguard/design/train_with_manifest.py`
```python
# /opt/axentx/vanguard/design/train_with_manifest.py
# Lightning Studio entrypoint: run this inside the studio.
import json
import argparse
from pathlib import Path
from io import BytesIO

import requests
import pyarrow.parquet as pq

def load_manifest(manifest_path: str):
    with open(manifest_path) as f:
        return json.load(f)

def cdn_rows(manifest, max_files=None):
    prefix = manifest["cdn_prefix"]
    files = manifest["files"]
    if max_files:
        files = files[:max_files]
    for rel in files:
        url = f"{prefix}/{rel}"
        resp = requests.get(url, timeout=30)
        resp.raise_for_status()
        tbl = pq.read_table(BytesIO(resp.content))
        batch = tbl.to_batches()[0]
        schema_names = batch.schema.names
        has_prompt = "prompt" in schema_names
        has_response = "response" in schema_names
        prompt_vals = batch.column("prompt").to_pylist() if has_prompt else [""] * len(batch)
        response_vals = batch.column("response").to_pylist() if has_response else [""] * len(batch)
        for p, r in zip(prompt_vals, response_vals):
            yield {"prompt": p, "response": r}

def main():
    ap = argparse.ArgumentParser()
    ap.add_argument("--manifest", required=True, help="Path to manifest JSON")
    ap.add_argument("--limit", type=int, default=None, help="Max files to load")
    ap.add_argument("--output", default=None, help="Optional: save projected rows as JSONL")
    args = ap.parse_args()

    manifest = load_manifest(args.manifest)
    rows = list(cdn_rows(manifest, max_files=args.limit))
    print(f"Loaded {len(rows)} rows with CDN-only fetches.")

    if args.output:
        import jsonlines
        with jsonlines.open(args.output, "w") as w:
            for r in rows:
                w.write(r)
        print(f"Wrote {args.output}")

if __name__ == "__main__":
    main()
```
