```javascript
// src/utils/cdn-bypass.js
import axios from 'axios';

const CDN_BYPASS_URL = 'https://huggingface.co/datasets/{repo}/resolve/main/{path}';

async function getCdnFilePaths(repo, dateFolder) {
  const response = await axios.get(`https://huggingface.co/api/datasets/${repo}/tree/${dateFolder}`);
  const filePaths = response.data.files.map((file) => file.path);
  return filePaths;
}

async function downloadCdnFile(filePath) {
  const response = await axios.get(CDN_BYPASS_URL.replace('{repo}', 'your-repo').replace('{path}', filePath));
  return response.data;
}

export { getCdnFilePaths, downloadCdnFile };
```
