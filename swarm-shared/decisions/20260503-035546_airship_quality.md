### 1) scripts/list_hf_files.py

```python
#!/usr/bin/env python3
"""
List files in a HuggingFace dataset folder (non-recursive) and emit CDN paths.
Run from Mac after rate-limit window clears. Embed output JSON in train.py.

Usage:
  python scripts/list_hf_files.py \
    --repo datasets/axentx/surrogate-mirror \
    --folder batches/mirror-merged/2026-05-03 \
    --out surrogate/file_list.json
"""

import argparse
import json
import os
import sys
from typing import List, Dict

try:
    from huggingface_hub import HfApi
except ImportError:
    print("pip install huggingface_hub")
    sys.exit(1)

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def list_cdn_files(repo: str, folder: str) -> List[Dict[str, str]]:
    api = HfApi()
    # recursive=False avoids paginating 100x; we only want one date folder
    entries = api.list_repo_tree(repo=repo, path=folder, recursive=False)
    files = []
    for e in entries:
        if e.type != "file":
            continue
        path = f"{folder}/{e.path.split('/')[-1]}"
        files.append(
            {
                "repo": repo,
                "path": path,
                "cdn_url": CDN_TEMPLATE.format(repo=repo, path=path),
                "size": getattr(e, "size", None),
            }
        )
    return files

def main() -> None:
    parser = argparse.ArgumentParser(description="List HF dataset files for CDN loader")
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g. datasets/axentx/surrogate-mirror)")
    parser.add_argument("--folder", required=True, help="Folder to list (non-recursive)")
    parser.add_argument("--out", required=True, help="Output JSON path")
    args = parser.parse_args()

    files = list_cdn_files(args.repo, args.folder)
    os.makedirs(os.path.dirname(os.path.abspath(args.out)), exist_ok=True)
    with open(args.out, "w") as f:
        json.dump(files, f, indent=2)
    print(f"Wrote {len(files)} files to {args.out}")

if __name__ == "__main__":
    main()
```

Make executable:

```bash
chmod +x scripts/list_hf_files.py
```

### 2) surrogate/train.py

```python
#!/usr/bin/env python3
"""
CDN-only dataset loader + Lightning idle-resilient runner.

- Data loading uses only CDN URLs (no HF API during training).
- Before each studio run, checks status and restarts if stopped (idle-timeout safe).
- Projects heterogeneous files to {prompt, response} at parse time.
"""

import json
import logging
import os
import tempfile
from pathlib import Path
from typing import Dict, List, Optional

import pyarrow as pa
import pyarrow.parquet as pq
import requests
import torch
from torch.utils.data import Dataset, DataLoader

# Lightning optional: only imported when USE_LIGHTNING=1
USE_LIGHTNING = os.getenv("USE_LIGHTNING", "1") == "1"
if USE_LIGHTNING:
    try:
        from lightning import LightningWork, LightningApp, Machine
        from lightning.app import LightningFlow
    except ImportError:
        USE_LIGHTNING = False
        logging.warning("lightning not installed; running in local mode")

logging.basicConfig(level=logging.INFO)
log = logging.getLogger("surrogate.train")

# -------------------------
# CDN Dataset
# -------------------------
class CDNParquetDataset(Dataset):
    """
    Loads parquet files via CDN URLs. Projects columns to {prompt, response}.
    Falls back to local cache to avoid repeated CDN fetches within epoch.
    """

    def __init__(self, file_list_path: str, cache_dir: Optional[str] = None):
        super().__init__()
        with open(file_list_path) as f:
            self.files: List[Dict] = json.load(f)

        if not self.files:
            raise ValueError("file_list is empty")

        self.cache_dir = Path(cache_dir or tempfile.mkdtemp(prefix="cdn_cache_"))
        self.cache_dir.mkdir(parents=True, exist_ok=True)

        self._cached_paths: List[Path] = []
        self._length = 0
        self._index_map: List[tuple[int, int]] = []  # (file_idx, row_idx)

        self._build_index()

    def _cdn_fetch(self, cdn_url: str, dest: Path) -> Path:
        if dest.exists():
            return dest
        # CDN downloads: no Authorization header required, bypasses API rate limits
        resp = requests.get(cdn_url, timeout=60)
        resp.raise_for_status()
        dest.write_bytes(resp.content)
        return dest

    def _project_to_prompt_response(self, table: pa.Table) -> pa.Table:
        """
        Project heterogeneous schemas to {prompt, response}.
        Expected conventions:
          - prompt column: prompt, input, question, instruction
          - response column: response, output, answer, completion
        If missing, synthesize from available text columns.
        """
        cols = table.column_names
        prompt_col = next((c for c in cols if c in {"prompt", "input", "question", "instruction"}), None)
        response_col = next((c for c in cols if c in {"response", "output", "answer", "completion"}), None)

        if prompt_col is None or response_col is None:
            # Fallback: use first two text-like columns
            text_cols = [c for c in cols if pa.types.is_string(table.schema.field(c).type)]
            if len(text_cols) >= 2:
                prompt_col, response_col = text_cols[0], text_cols[1]
            else:
                raise ValueError(f"Cannot find prompt/response in columns: {cols}")

        return table.select([prompt_col, response_col]).rename_columns(["prompt", "response"])

    def _build_index(self) -> None:
        for file_idx, meta in enumerate(self.files):
            cdn_url = meta["cdn_url"]
            slug = Path(meta["path"]).stem
            local_path = self.cache_dir / f"{slug}.parquet"

            local_path = self._cdn_fetch(cdn_url, local_path)
            table = pq.read_table(local_path)
            table = self._project_to_prompt_response(table)

            row_count = table.num_rows
            for row_idx in range(row_count):
                self._index_map.append((file_idx, row_idx))
            self._length += row_count

            # Keep one projected table in memory per file for speed
            # (trade memory for fewer disk reads). For large folders,
            # consider mmap or streaming.
            setattr(self, f"_table_{file_idx}", table)

        log.info(f"Indexed {self._length} rows across {len(self.files)} files via CDN")

    def __len__(self) -> int:
        return self._length

    def __getitem__(self, idx: int) -> Dict[str, str]:
        file_idx, row_idx = self._index
