## Implementation

### 1) Create orchestrator script

```bash
#!/usr/bin/env bash
# /opt/axentx/airship/bin/airship-discover
# Zero-runtime, CDN-safe discovery orchestrator for airship.
# Emits: status.json, manifest.json
# Usage:
#   ./airship-discover [--out-dir ./out] [--source-root .] [--verify] [--upload-cdn <base-cdn-url>]

set -euo pipefail
IFS=$'\n\t'

readonly SCRIPT_NAME="$(basename "$0")"
readonly DEFAULT_OUT_DIR="./out"
readonly DEFAULT_SOURCE_ROOT="."

log()  { echo "[$SCRIPT_NAME] $*" >&2; }
die()  { echo "[$SCRIPT_NAME] ERROR: $*" >&2; exit 1; }

check_tool() {
  command -v "$1" >/dev/null 2>&1 || die "Required tool not found: $1"
}

# ---- validation ----
check_tool jq
check_tool find
check_tool sort
check_tool sha256sum || check_tool shasum || die "No sha256 tool (sha256sum/shasum) available"

OUT_DIR="$DEFAULT_OUT_DIR"
SOURCE_ROOT="$DEFAULT_SOURCE_ROOT"
MODE="run"
UPLOAD_CDN_URL=""

while [[ $# -gt 0 ]]; do
  case "$1" in
    --out-dir)
      OUT_DIR="${2:-}"
      shift 2
      ;;
    --source-root)
      SOURCE_ROOT="${2:-}"
      shift 2
      ;;
    --verify)
      MODE="verify"
      shift
      ;;
    --upload-cdn)
      UPLOAD_CDN_URL="${2:-}"
      shift 2
      MODE="upload-cdn"
      ;;
    --help|-h)
      echo "Usage: $SCRIPT_NAME [--out-dir DIR] [--source-root DIR] [--verify] [--upload-cdn URL]"
      exit 0
      ;;
    *)
      die "Unknown option: $1"
      ;;
  esac
done

mkdir -p "$OUT_DIR"
OUT_DIR="$(cd "$OUT_DIR" && pwd)"
SOURCE_ROOT="$(cd "$SOURCE_ROOT" && pwd)"

# ---- deterministic file enumeration ----
# Exclude common non-source and CI artifacts to keep manifest stable and small.
# Uses null-delimited find + sort for determinism across environments.
enumerate_files() {
  local root="$1"
  (
    cd "$root"
    find . -type f \
      -not -path './.git/*' \
      -not -path './node_modules/*' \
      -not -path './.tox/*' \
      -not -path './venv/*' \
      -not -path './out/*' \
      -not -path './dist/*' \
      -not -path './build/*' \
      -not -path './__pycache__/*' \
      -not -name '*.pyc' \
      -not -name '*~' \
      -not -name '.*.swp' \
      -print0 | sort -z | tr '\0' '\n'
  )
}

compute_file_hash() {
  local file="$1"
  if command -v sha256sum >/dev/null 2>&1; then
    sha256sum "$file" | awk '{print $1}'
  else
    shasum -a 256 "$file" | awk '{print $1}'
  fi
}

# ---- run mode ----
run_discovery() {
  log "Discovering files under: $SOURCE_ROOT"
  local file_list_file="$OUT_DIR/file-list.txt"
  local status_file="$OUT_DIR/status.json"
  local manifest_file="$OUT_DIR/manifest.json"

  # deterministic file list
  enumerate_files "$SOURCE_ROOT" > "$file_list_file"
  local total_files
  total_files=$(wc -l < "$file_list_file" | tr -d ' ')

  log "Found $total_files files"

  # build manifest entries deterministically
  local entries="[]"
  while IFS= read -r relpath; do
    [[ -z "$relpath" ]] && continue
    local abspath="$SOURCE_ROOT/$relpath"
    local fsize
    fsize=$(stat -c%s "$abspath" 2>/dev/null || stat -f%z "$abspath" 2>/dev/null || echo 0)
    local fhash
    fhash=$(compute_file_hash "$abspath")
    local mtime
    mtime=$(stat -c%Y "$abspath" 2>/dev/null || stat -f%m "$abspath" 2>/dev/null || echo 0)

    entries=$(echo "$entries" | jq --arg p "$relpath" --arg sz "$fsize" --arg h "$fhash" --arg mt "$mtime" \
      '. + [{path: $p, size: ($sz | tonumber), sha256: $h, mtime: ($mt | tonumber)}]')
  done < "$file_list_file"

  # ensure stable ordering by path
  entries=$(echo "$entries" | jq 'sort_by(.path)')

  # compute manifest-level hash
  local manifest_hash
  manifest_hash=$(echo "$entries" | jq -c '.' | sha256sum | awk '{print $1}')

  # status.json
  jq -n \
    --arg ts "$(date -u +"%Y-%m-%dT%H:%M:%SZ")" \
    --arg root "$SOURCE_ROOT" \
    --argjson count "$total_files" \
    --arg manifest_hash "$manifest_hash" \
    '{
      timestamp: $ts,
      source_root: $root,
      discovered_files: $count,
      manifest_sha256: $manifest_hash,
      status: "ok"
    }' > "$status_file"

  # manifest.json
  jq -n \
    --arg ts "$(date -u +"%Y-%m-%dT%H:%M:%SZ")" \
    --arg root "$SOURCE_ROOT" \
    --arg manifest_hash "$manifest_hash" \
    --argjson files "$entries" \
    '{
      generated_at: $ts,
      source_root: $root,
      manifest_sha256: $manifest_hash,
      files: $files
    }' > "$manifest_file"

  log "Wrote status.json -> $status_file"
  log "Wrote manifest.json -> $manifest_file"
  log "Wrote file-list -> $file_list_file"

  # machine-readable summary to stdout (CI-friendly)
  jq -n \
    --arg status_file "$status_file" \
    --arg manifest_file "$manifest_file" \
    --arg file
