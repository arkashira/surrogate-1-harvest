# Costinel / ops

## Implementation Plan — Top-Hub Signal Panel (CDN-first, <2h)

### What we ship
- A **non-blocking Top-Hub Signal Panel** mounted on the Costinel dashboard (sidebar or top banner area).
- Defaults to hub **MOC** (configurable via `VITE_HUB_NAME`).
- Shows: hub title, short description, top 5 connected docs (title + snippet), last-updated timestamp.
- **CDN-first data strategy**: pre-list file paths once (Mac orchestration), embed JSON in build, runtime fetches via CDN (`/resolve/main/...`) — zero API calls during dashboard use, bypasses HF rate limits.
- **Non-blocking UX**: lazy-loaded, skeleton fallback, no render-blocking.

### Files to modify/create
1. `src/components/TopHubSignalPanel.vue` (new)
2. `src/data/hubFileList.json` (generated by orchestration script)
3. `scripts/generate-hub-file-list.js` (Mac orchestration script)
4. `src/main.js` or dashboard layout: mount panel in sidebar/top banner
5. `vite.config.js` (optional): expose `VITE_HUB_NAME`

#### 2) Orchestration script (run on Mac) — generates CDN file list
```js
// scripts/generate-hub-file-list.js
#!/usr/bin/env bash
# Save as scripts/generate-hub-file-list.sh and chmod +x
# Usage: bash scripts/generate-hub-file-list.sh <owner>/<repo> <path>

set -euo pipefail

REPO="${1:-axentx/knowledge-rag}"
PATH_PREFIX="${2:-hubs}"
OUTFILE="${3:-src/data/hubFileList.json}"
HUB_NAME="${VITE_HUB_NAME:-MOC}"

echo "Listing HF repo tree (non-recursive) for $REPO/$PATH_PREFIX/$HUB_NAME ..."

# Requires HF_TOKEN in env for list_repo_tree (API call).
# We run once per build; runtime uses CDN only.
node - <<NODE
const { HfApi } = require("@huggingface/hub");
const fs = require("fs");
const path = require("path");

async function main() {
  const api = new HfApi({ token: process.env.HF_TOKEN || undefined });
  const repo = "$REPO";
  const prefix = "$PATH_PREFIX/$HUB_NAME";
  try {
    // list_repo_tree non-recursive to avoid pagination/rate-limits
    const tree = await api.listRepoTree({
      repo,
      path: prefix,
      recursive: false,
    });
    const files = (tree.files || [])
      .filter(f => f.type === "file" && f.path.endsWith(".md"))
      .map(f => f.path)
      .sort();

    // Build CDN URLs (resolve/main) — no auth, bypasses API rate limits
    const items = files.map(p => ({
      path: p,
      cdn: \`https://huggingface.co/datasets/\${repo}/resolve/main/\${p}\`,
    }));

    fs.mkdirSync(path.dirname("$OUTFILE"), { recursive: true });
    fs.writeFileSync("$OUTFILE", JSON.stringify({ hub: "$HUB_NAME", generatedAt: new Date().toISOString(), items }, null, 2));
    console.log("Wrote", items.length, "files to $OUTFILE");
  } catch (err) {
    console.error("HF list error:", err.message);
    // Fallback: minimal placeholder so build doesn't break
    fs.mkdirSync(path.dirname("$OUTFILE"), { recursive: true });
    fs.writeFileSync("$OUTFILE", JSON.stringify({ hub: "$HUB_NAME", generatedAt: new Date().toISOString(), items: [] }, null, 2));
    process.exit(0);
  }
}
main();
NODE
```

Make executable:
```bash
chmod +x scripts/generate-hub-file-list.sh
```

Run once (or in CI):
```bash
bash scripts/generate-hub-file-list.sh axentx/knowledge-rag hubs
```

#### 3) TopHubSignalPanel component (Vue 3)
```vue
<!-- src/components/TopHubSignalPanel.vue -->
<template>
  <div class="top-hub-panel">
    <div class="panel-header" @click="toggle">
      <div class="hub-title">
        <span class="hub-badge">Hub</span>
        {{ hubName }}
      </div>
      <div class="panel-meta">
        <span class="doc-count">{{ visibleItems.length }} docs</span>
        <span class="updated">Updated {{ lastUpdated }}</span>
      </div>
      <svg class="chevron" :class="{ open }" viewBox="0 0 24 24"><path d="M7 10l5 5 5-5z"/></svg>
    </div>

    <transition name="slide">
      <div v-if="open" class="panel-body">
        <div v-if="loading" class="skeleton-list">
          <div v-for="i in 5" :key="i" class="skeleton-row"></div>
        </div>

        <div v-else-if="error" class="panel-error">
          Unable to load hub signals.
        </div>

        <ul v-else class="doc-list">
          <li v-for="doc in visibleItems" :key="doc.path" class="doc-item">
            <a :href="doc.cdn" target="_blank" rel="noopener" class="doc-link">
              <div class="doc-title">{{ doc.title }}</div>
              <div v-if="doc.snippet" class="doc-snippet">{{ doc.snippet }}</div>
            </a>
          </li>
        </ul>

        <div v-if="!loading && !error && items.length === 0" class="empty">
          No documents found.
        </div>
      </div>
    </transition>
  </div>
</template>

<script setup>
import { ref, onMounted, computed } from "vue";

const props = defineProps<{
  hubName?: string;
  maxDocs?: number;
}>();

const hubName = ref(props.hubName || import.meta.env.VITE_HUB_NAME || "MOC");
const maxDocs = ref(props.maxDocs || 5);

const items = ref([]);
const loading = ref(true);
const error = ref(false);
const open = ref(false);

const visibleItems = computed(() => items.value.slice(0, maxDocs.value));
const lastUpdated = computed(() => {
  if (!items.value.length) return "";
  // fallback to generated timestamp if available
  return new Date(items.value[0].generatedAt || Date.now()).toLocaleDateString();
});

async function fetchList() {
  loading.value = true;
  error.value = false;
  try {
    // CDN fetch — no Authorization header required, bypasses API rate limits
    const res = await fetch(`/data/hubFileList.json?cb=${Date.now()}`, { cache: "no-store" });
    if (!res.ok) throw new Error("Network response not ok");
    const json = await res.json();
    const docs = (json.items || []).map((it) => ({
      ...it,
      title: it.path.split("/").pop().replace(/\.md$/, "").replace(/-/g, " "),
      snippet: "", // lightweight; could fetch excerpt on demand
      generatedAt: json.generatedAt,
    }));
    items.value = docs;
  } catch (err) {
    console.warn("Hub panel fetch failed:", err);
    error.value = true;
  } finally {
    loading.value = false;
  }
}

function toggle() {
  open.value = !open.value;
  if (open.value && !items.value.length && !loading.value) {
    fetchList();
  }
}

onMounted(() => {
  // Preload minimally (non-blocking) or wait for user interaction
  // We'll lazy-load on first open to avoid initial render cost.
});
</script>
