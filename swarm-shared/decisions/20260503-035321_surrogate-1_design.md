# surrogate-1 / design

## Final Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16` (matrix) and optional `DATE_FOLDER` (defaults to today `YYYY-MM-DD`).
- Uses a **single pre-listed manifest** (`manifests/{DATE_FOLDER}.json`) produced by the Mac orchestrator (or a one-off runner) that contains only file paths for that date folder. This avoids recursive `list_repo_tree` calls from workers and eliminates per-worker API rate-limit risk.
- Performs **CDN-only fetches** (`https://huggingface.co/datasets/.../resolve/main/...`) with no Authorization header to bypass `/api/` rate limits.
- Projects each file to `{prompt, response}` only at parse time (avoids pyarrow `CastError` on mixed schemas).
- Deduplicates via the existing central md5 store (`lib/dedup.py`) and writes deterministic shard outputs:
  ```
  batches/public-merged/<date>/shard<N>-<HHMMSS>.jsonl
  ```
- Adds retry/backoff for 429 (wait 360s) and respects HF commit cap by deterministic shard → repo routing if needed.
- Keeps GitHub Actions matrix at 16 shards (isolated 7 GB each) and preserves `workflow_dispatch`.

def shard_assignment(files: List[str]) -> List[str]:
    """Deterministic shard assignment by slug hash."""
    assigned = []
    for f in files:
        # Use filename (slug) for shard bucket
        slug = Path(f).stem
        h = int(hashlib.sha256(slug.encode()).hexdigest(), 16)
        if h % SHARD_TOTAL == SHARD_ID:
            assigned.append(f)
    return assigned

def cdn_download(url: str, timeout: int = 30) -> bytes:
    for attempt in range(1, MAX_RETRIES + 1):
        try:
            resp = requests.get(url, timeout=timeout)
            if resp.status_code == 429:
                wait = RETRY_WAIT * attempt
                print(f"Rate-limited (429). Waiting {wait}s...", file=sys.stderr)
                time.sleep(wait)
                continue
            resp.raise_for_status()
            return resp.content
        except requests.RequestException as e:
            if attempt == MAX_RETRIES:
                raise
            print(f"Download failed ({e}), retry {attempt}/{MAX_RETRIES}", file=sys.stderr)
            time.sleep(5 * attempt)
    raise RuntimeError("Max retries exceeded")

def project_to_pair(content: bytes, filename: str) -> Optional[List[Dict[str, str]]]:
    """Project file content to {prompt, response} only; ignore mixed schema errors."""
    suffix = Path(filename).suffix.lower()
    try:
        if suffix == ".parquet":
            table = pq.read_table(pa.BufferReader(content))
            # Keep only prompt/response if present; ignore extra columns
            cols = set(table.column_names)
            prompt_col = next((c for c in ("prompt", "Prompt", "input", "question") if c in cols), None)
            response_col = next((c for c in ("response", "Response", "output", "answer") if c in cols), None)
            if prompt_col is None or response_col is None:
                print(f"Missing prompt/response in {filename}", file=sys.stderr)
                return None
            # Convert to dicts
            prompts = table.column(prompt_col).to_pylist()
            responses = table.column(response_col).to_pylist()
            pairs = []
            for p, r in zip(prompts, responses):
                if isinstance(p, str) and isinstance(r, str) and p.strip() and r.strip():
                    pairs.append({"prompt": p.strip(), "response": r.strip()})
            return pairs
        elif suffix in (".jsonl", ".json"):
            # Simple line-by-line JSONL handling
            lines = content.decode("utf-8").strip().splitlines()
            pairs = []
            for line in lines:
                if not line.strip():
                    continue
                obj = json.loads(line)
                prompt = obj.get("prompt") or obj.get("Prompt") or obj.get("input") or obj.get("question")
                response = obj.get("response") or obj.get("Response") or obj.get("output") or obj.get("answer")
                if isinstance(prompt, str) and isinstance(response, str) and prompt.strip() and response.strip():
                    pairs.append({"prompt": prompt.strip(), "response": response.strip()})
            return pairs
        else:
            print(f"Unsupported file type {suffix} for {filename}", file=sys.stderr)
            return None
    except Exception as e:
        print(f"Projection error in {filename}: {e}", file=sys.stderr)
        return None
