## 1. Diagnosis (consolidated)
- Missing `X-Line-Signature` HMAC-SHA256 verification on `/webhook/line` allows spoofed/replayed events.
- No idempotency guard enables duplicate clock-in/out, leave, or OT records when LINE retries or users double-tap.
- Body-parser/`express.json()` mutates the stream before HMAC can be checked; raw body must be preserved **only for this route**.
- Tenant context inferred from payload without cryptographic binding; cross-tenant spoofing possible.
- No replay-window check; stale events can be accepted.

### B. Middleware: raw-body capture + signature + replay + idempotency metadata
File: `/opt/axentx/workio/server/src/middleware/lineSignature.ts`

```ts
import crypto from 'crypto';
import { Request, Response, NextFunction } from 'express';

const LINE_CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET;
if (!LINE_CHANNEL_SECRET) {
  throw new Error('LINE_CHANNEL_SECRET is required');
}

const REPLAY_WINDOW_SECONDS = 300; // ±5 minutes

/**
 * Verify X-Line-Signature, timestamp freshness, and expose idempotency metadata.
 * Must run AFTER raw-body capture on /webhook/line and BEFORE JSON parsers for this route.
 */
export function lineSignature(req: Request, res: Response, next: NextFunction) {
  const signature = req.get('X-Line-Signature');
  const retryId = req.get('X-Line-Retry-Id') || '';
  const rawBody = (req as any).rawBody as Buffer | undefined;
  const nowSec = Math.floor(Date.now() / 1000);

  if (!rawBody || !Buffer.isBuffer(rawBody)) {
    return res.status(400).json({ error: 'Missing raw body' });
  }
  if (!signature) {
    return res.status(400).json({ error: 'Missing X-Line-Signature' });
  }

  // 1) HMAC-SHA256 verification (constant-time)
  const expected = crypto
    .createHmac('sha256', LINE_CHANNEL_SECRET)
    .update(rawBody, 'utf8')
    .digest('base64');

  // Use timingSafeEqual with fixed-length buffers to avoid timing attacks
  const sigBuf = Buffer.from(signature);
  const expBuf = Buffer.from(expected);
  const len = Math.max(sigBuf.length, expBuf.length);
  const a = Buffer.alloc(len);
  const b = Buffer.alloc(len);
  sigBuf.copy(a);
  expBuf.copy(b);
  if (!crypto.timingSafeEqual(a, b)) {
    return res.status(401).json({ error: 'Invalid signature' });
  }

  // 2) Parse payload after verification
  let payload;
  try {
    payload = JSON.parse(rawBody.toString('utf8'));
  } catch {
    return res.status(400).json({ error: 'Invalid JSON' });
  }

  const events = payload.events;
  if (!Array.isArray(events) || events.length === 0) {
    return res.status(400).json({ error: 'No events' });
  }

  // 3) Replay-window check per event
  for (const ev of events) {
    const ts = ev.timestamp;
    if (typeof ts !== 'number' || Math.abs(nowSec - Math.floor(ts / 1000)) > REPLAY_WINDOW_SECONDS) {
      return res.status(400).json({ error: 'Timestamp out of window', event: ev });
    }
  }

  // 4) Attach verified metadata for downstream idempotency + tenant binding
  (req as any)._lineVerified = {
    events,
    retryId,
    receivedAt: nowSec,
    // Include channel-secret-derived tenant binding (prevents cross-tenant replays)
    channelSecretHash: crypto
      .createHash('sha256')
      .update(LINE_CHANNEL_SECRET)
      .digest('hex')
      .slice(0, 16),
  };

  next();
}
```

### C. Wire up raw-body capture and middleware
File: `/opt/axentx/workio/server/src/app.ts` (or main server file)

```ts
import express from 'express';
import { lineSignature } from './middleware/lineSignature';
import lineWebhookRouter from './routes/webhook/line';

const app = express();

// Capture raw body ONLY for LINE webhook before any JSON parser touches it
app.use('/webhook/line', (req, res, next) => {
  const chunks: Buffer[] = [];
  req.on('data', (chunk) => chunks.push(chunk));
  req.on('end', () => {
    (req as any).rawBody = Buffer.concat(chunks);
    next();
  });
});

// General JSON parser for other routes
app.use(express.json());

// Apply signature verification to LINE webhook
app.use('/webhook/line', lineSignature);
app.use('/webhook/line', lineWebhookRouter);

export default app;
```

const router = Router();

  const verified = (req as any)._lineVerified;
  if (!verified) {
    return res.status(400).json({ error: 'Unverified request' });
  }

  const { events, retryId, channelSecretHash } = verified;

  for (const ev of events) {
    // Prefer LINE-provided retryId; fallback to composite key

    // Check for existing processed event (idempotency)

      .from('line_webhook_events')
      .select('id')
