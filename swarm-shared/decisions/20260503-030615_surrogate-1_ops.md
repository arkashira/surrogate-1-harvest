### Key Decisions (Correctness + Actionability)
1. **Use `datasets` streaming + `pyarrow` schema enforcement** instead of raw `hf_hub_download` + manual parsing. This resolves the CastError reliably while keeping CDN benefits.
2. **Deterministic sharding via `hash(slug) % 16`** — consistent across runners, no coordination needed.
3. **Manifest is authoritative** — single `list_repo_tree` call per DATE, cached to disk; training uses CDN-only paths from manifest.
4. **Strict `{prompt, response}` projection at parse time** — prevents mixed-schema errors downstream.
5. **Mac=CLI rule enforced** — orchestration only; no `model.from_pretrained()` on Mac runners.

#!/usr/bin/env python3
"""
Manifest-driven, CDN-bypass ingestion worker.

Usage:
  SHARD_ID=0 SHARD_TOTAL=16 DATE=2026-05-03 HF_TOKEN=hf_xxx \
    python bin/dataset-enrich.py

Behavior:
- Single list_repo_tree for DATE -> manifest-<DATE>.json
- Streams only assigned shard via datasets (CDN, no API during iter)
- Projects to {prompt, response} at parse time (fixes CastError)
- Deterministic sharding via hash(slug) % SHARD_TOTAL
- Deduplicates via lib.dedup (if available)
- Writes batches/public-merged/<DATE>/shard<N>-<HHMMSS>.jsonl
"""

import json
import os
import sys
import time
import hashlib
import logging
from datetime import datetime, timezone
from pathlib import Path
from typing import Iterator, Dict, Any

from huggingface_hub import HfApi, HfFolder
import pyarrow as pa
from datasets import load_dataset, Features, Value

# ----------------------------
# Config / env
# ----------------------------
REPO_DATASET = os.getenv("REPO_ID", "axentx/surrogate-1-training-pairs")
HF_TOKEN = os.getenv("HF_TOKEN")
SHARD_ID = int(os.getenv("SHARD_ID", "0"))
SHARD_TOTAL = int(os.getenv("SHARD_TOTAL", "16"))
DATE = os.getenv("DATE", datetime.now(timezone.utc).strftime("%Y-%m-%d"))

if not HF_TOKEN:
    print("ERROR: HF_TOKEN required", file=sys.stderr)
    sys.exit(1)

# Paths
BASE_DIR = Path(__file__).parent.parent
MANIFEST_DIR = BASE_DIR / "manifests"
BATCH_DIR = BASE_DIR / "batches" / "public-merged" / DATE
MANIFEST_DIR.mkdir(parents=True, exist_ok=True)
BATCH_DIR.mkdir(parents=True, exist_ok=True)

MANIFEST_PATH = MANIFEST_DIR / f"manifest-{DATE}.json"
OUTPUT_PATH = BATCH_DIR / f"shard{SHARD_ID}-{datetime.utcnow().strftime('%H%M%S')}.jsonl"

# Dedup store (optional)
sys.path.insert(0, str(BASE_DIR / "lib"))
try:
    from dedup import DedupStore
    dedup = DedupStore()
except Exception:
    dedup = None

# Logging
logging.basicConfig(
    level=logging.INFO,
    format="%(asctime)s [shard%(shard)s] %(levelname)s %(message)s",
    datefmt="%Y-%m-%dT%H:%M:%S%z",
)
log = logging.LoggerAdapter(logging.getLogger(), {"shard": SHARD_ID})

# HF API
api = HfApi(token=HF_TOKEN)

# Retry policy
MAX_RETRIES = 5
RATE_LIMIT_WAIT = 360

# ----------------------------
# Helpers
# ----------------------------
def deterministic_shard(slug: str) -> bool:
    h = int(hashlib.sha256(slug.encode()).hexdigest(), 16)
    return h % SHARD_TOTAL == SHARD_ID

def api_call_with_retry(fn, *args, **kwargs):
    for attempt in range(1, MAX_RETRIES + 1):
        try:
            return fn(*args, **kwargs)
        except Exception as e:
            msg = str(e).lower()
            if "429" in msg or "rate limit" in msg:
                if attempt < MAX_RETRIES:
                    log.info(f"Rate limited. Waiting {RATE_LIMIT_WAIT}s")
                    time.sleep(RATE_LIMIT_WAIT)
                    continue
            if attempt == MAX_RETRIES:
                raise
            sleep = 5 * (2 ** (attempt - 1))
            log.warning(f"Attempt {attempt}/{MAX_RETRIES} failed: {e}. Retrying in {sleep}s")
            time.sleep(sleep)

# ----------------------------
# Manifest: list once, CDN-only thereafter
# ----------------------------
def build_or_load_manifest() -> list:
    if MANIFEST_PATH.exists():
        log.info(f"Loading existing manifest: {MANIFEST_PATH}")
        return json.loads(MANIFEST_PATH.read_text())

    log.info(f"Building manifest for {DATE} in {REPO_DATASET}")
    try:
        tree = api_call_with_retry(
            api.list_repo_tree,
            repo_id=REPO_DATASET,
            path=DATE,
            repo_type="dataset",
            recursive=False,
        )
        items = [f.rfilename for f in tree if isinstance(f, dict) or hasattr(f, "rfilename")]
    except Exception:
        # Fallback: list root and filter by DATE prefix
        tree = api_call_with_retry(
            api.list_repo_tree,
            repo_id=REPO_DATASET,
            path="",
            repo_type="dataset",
            recursive=False,
        )
        items = [f.rfilename for f in tree if f.rfilename.startswith(f"{DATE}/")]

    # Keep only data files
    files = [f for f in items if f.endswith((".parquet", ".jsonl", ".json"))]
    MANIFEST_PATH.write_text(json.dumps(files, indent=2))
    log.info(f"Manifest saved with {len(files)} files: {MANIFEST_PATH}")
    return files

# ----------------------------
# Parse helpers (mixed-schema safe)
# ----------------------------
def extract_prompt_response(obj: Dict[str, Any]) -> Dict[str, str]:
    """Project to {prompt, response} regardless of schema."""
    prompt = obj.get("prompt") or obj.get("input") or obj.get("question") or ""
    response = obj.get("response") or obj.get("output") or obj.get("answer") or ""
    return {"prompt": str(prompt), "response": str(response)}

def stream_shard(file: str) -> Iterator[Dict[str, str]]:
    """Stream a single file and yield {prompt, response} for shard items."""
    try:
        # Use streaming + schema enforcement to avoid CastError
        ds = load_dataset(
            "parquet",
            name="stream",
            data_files={"train": file},
            split="train",
            streaming=True,
            features=Features({
                "prompt": Value("string"),
                "response": Value("string"),
            }),
        )
        for row in ds:
            # If schema already matches, use directly; else project
            d = dict(row)
            if "prompt" in d and "response" in d:
                yield {"prompt": str(d["prompt"]), "response": str(d["response"])}
            else:
                yield extract_prompt_response(d)
    except Exception as e:
        log.warning(f"Failed to stream {file}: {e}. Falling back to manual projection.")
        # Fallback: load without strict schema and project
        ds = load_dataset(
            "parquet",
            data_files={"train": file},
            split="train",
            streaming=True,
        )
        for row in ds:
            yield extract_prompt_response(dict(row))

# ----------------------------
# Main
# ----------------------------
def main():
    files = build_or_load_manifest()
    if not files:
        log.error("No files found in manifest. Exiting.")
        sys.exit(1)

    # Deterministic shard assignment by file slug
    my_files = [f for f in files if deterministic_shard(f)]
