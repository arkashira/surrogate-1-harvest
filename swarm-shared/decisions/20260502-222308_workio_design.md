### 2. middleware/tracing.ts
```ts
import { randomUUID } from 'uuid';

export interface RequestLocals {
  requestId: string;
  startTime: number;
  tenantId?: string;
}

export function attachTracing(req: any, res: any, next: () => void) {
  req.requestId = randomUUID();
  req.startTime = Date.now();
  res.setHeader('X-Request-Id', req.requestId);
  next();
}

export function logWebhookEvent(
  requestId: string,
  channelId: string | undefined,
  event: string,
  details?: Record<string, unknown>
) {
  const entry = {
    type: 'webhook',
    requestId,
    channelId,
    event,
    timestamp: Date.now(),
    ...details,
  };
  // Use structured logging (JSON) for ops filtering
  console.log(JSON.stringify(entry));
}
```

### 3. middleware/lineSignature.ts
```ts
import crypto from 'crypto';

export function verifyLineSignature(
  channelSecret: string,
  body: string,
  signature: string | undefined
): boolean {
  if (!signature) return false;
  try {
    const expected = crypto
      .createHmac('sha256', channelSecret)
      .update(body, 'utf8')
      .digest('base64');
    return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
  } catch {
    return false;
  }
}
```

### 4. middleware/tenant.ts
```ts
import LRU from 'lru-cache';
import { getTenantByChannelId } from '../services/tenantService';

interface Tenant {
  id: string;
  name: string;
  channelId: string;
}

const cache = new LRU<string, Tenant>({
  max: 500,
  ttl: 1000 * 60 * 10, // 10m
  allowStale: true,
  updateAgeOnGet: true,
});

export async function resolveTenant(
  channelId: string
): Promise<Tenant | null> {
  const cached = cache.get(channelId);
  if (cached) return cached;

  const tenant = await getTenantByChannelId(channelId);
  if (tenant) cache.set(channelId, tenant);
  return tenant || null;
}

export function attachTenant(req: any, res: any, next: () => void) {
  // channelId expected from header or earlier middleware
  const channelId = req.headers['x-line-channel-id'] as string | undefined;
  if (!channelId) {
    return next();
  }
  resolveTenant(channelId)
    .then((tenant) => {
      if (tenant) {
        req.tenant = tenant;
        (req as any).tenantId = tenant.id;
      }
      next();
    })
    .catch(() => next());
}

export function warmTenantCache(channelId: string, tenant: Tenant | null) {
  if (tenant) cache.set(channelId, tenant);
  else cache.del(channelId);
}
```

### 5. routes/health.ts
```ts
import express from 'express';
import { getDb } from '../db';

const router = express.Router();

router.get('/health', (_req, res) => {
  res.json({ status: 'ok', uptime: process.uptime() });
});

router.get('/ready', async (_req, res) => {
  try {
    const db = getDb();
    await db.query('SELECT 1');
    res.json({ status: 'ready', db: 'up' });
  } catch (err) {
    res.status(503).json({ status: 'not_ready', db: 'down', error: String(err) });
  }
});

export default router;
```

### 6. webhook/line.ts (recommended location)
```ts
import express, { Request, Response } from 'express';
import { verifyLineSignature } from '../middleware/lineSignature';
import { resolveTenant } from '../middleware/tenant';
import { logWebhookEvent } from '../middleware/tracing';

const router = express.Router();

router.post(
  '/line',
  express.raw({ type: 'application/json' }),
  async (req: Request & { requestId?: string; startTime?: number }, res: Response) => {
    const requestId = req.requestId || 'unknown';
    const channelId = req.headers['x-line-channel-id'] as string | undefined;
    const signature = req.headers['x-line-signature'] as string | undefined;
    const bodyRaw = req.body.toString('utf8');
    const startTime = req.startTime || Date.now();

    logWebhookEvent(requestId, channelId || 'unknown', 'received', { startTime });

    if (!channelId) {
      logWebhookEvent(requestId, 'unknown', 'missing_channel');
      return res.status(400).send('Missing channel');
    }

    const secret = process.env.LINE_CHANNEL_SECRET;
    if (!secret || !verifyLineSignature(secret, bodyRaw, signature)) {
      logWebhookEvent(requestId, channelId, 'invalid_signature');
      return res.status(400).send('Invalid signature');
    }

    const tenant = await resolveTenant(channelId);
    if (!tenant) {
      logWebhookEvent(requestId, channelId, 'tenant_not_found');
      return res.status(404).send('Tenant not found');
    }

    try {
      const payload = JSON.parse(bodyRaw);
      // TODO: call existing LINE webhook handler with tenant + payload
      // Example: await handleLineWebhook(tenant, payload);
      logWebhookEvent(requestId, channelId, 'processed', {
        eventsCount: payload.events?.length ?? 0,
        tenantId: tenant.id,
      });
      res.status(200).send('OK');
    } catch (err) {
      logWebhookEvent(requestId, channelId, 'processing_failed', {
        error: String(err),
        tenantId: tenant.id,
      });
      res.status(500).send('Processing failed');
    }
  }
);

export default router;
```

### 7. server/src/index.ts (apply middleware and routes)
```ts
import express from 'express';
import { attachTracing } from './middleware/tracing';
import { attachTenant } from './middleware/tenant';
import healthRoutes from './routes/health';
import lineWebhook from '../webhook/line';

const app = express();

// Global tracing and tenant resolution for all routes
app.use(attachTracing);
app.use(attachTenant);

// Health endpoints
app.use('/health', healthRoutes);

// LINE webhook (raw body handled in route)
app.use('/webhook', lineWebhook);

// Other API routes can follow below
// app.use('/api', apiRoutes);

export default app;
```

### 8. Verification
- Start server: `npm run dev` in `/opt/axentx/workio/server`.
- Health: `curl http://localhost:3000/health` → `{"status":"ok","uptime":...}`
- Readiness: `curl http://localhost:3000/ready` → `{"status":"ready","db":"up"}`
- Webhook:
  - Valid request with correct `X-Line-Signature` and `X-Line-Channel-Id` → 200 and log entry with `event:processed`.
  - Invalid signature → 400 and log entry with `event:invalid_signature`.
  - Unknown channel → 404 and log entry with `event:tenant_not_found`.
- Cache: repeated requests for same `channelId` within 10m should show cached tenant usage (no DB query in tenantService logs).
