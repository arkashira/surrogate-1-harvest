- Fits <2h: one small Python script + shell tweak + docs.
- Preserves existing architecture; no schema or dedup changes.
- Makes shard workers and local training reproducible and quota-safe.

### 1) Add `bin/list-files.py`

```python
#!/usr/bin/env python3
"""
Generate deterministic file list for a date folder in a HF dataset repo.
Usage:
  HF_TOKEN=... python bin/list-files.py \
    --repo axentx/surrogate-1-training-pairs \
    --date 2026-05-02 \
    --out file-list.json

Uses list_repo_tree(path, recursive=False) per subfolder to avoid recursive
pagination and heavy API usage. Outputs list of paths relative to repo root.
"""

import argparse
import json
import os
import sys
import time
from pathlib import Path

from huggingface_hub import HfApi, RepositoryNotFoundError

API_RATE_LIMIT_WAIT = 360  # seconds (per pattern)

def robust_list_tree(api, repo_id, path, max_retries=3):
    for attempt in range(1, max_retries + 1):
        try:
            # recursive=False avoids paginating 100x on big repos
            tree = api.list_repo_tree(repo_id=repo_id, path=path, recursive=False)
            return [item.path for item in tree]
        except Exception as exc:
            # crude 429 detection; huggingface_hub may raise HTTPError
            is_rate_limit = hasattr(exc, "response") and getattr(exc.response, "status_code", None) == 429
            if is_rate_limit and attempt < max_retries:
                wait = API_RATE_LIMIT_WAIT * attempt
                print(f"[rate-limited] waiting {wait}s (attempt {attempt}/{max_retries})", file=sys.stderr)
                time.sleep(wait)
                continue
            raise

def main() -> None:
    parser = argparse.ArgumentParser(description="List HF dataset files for a date folder.")
    parser.add_argument("--repo", required=True, help="HF dataset repo id (e.g. axentx/surrogate-1-training-pairs)")
    parser.add_argument("--date", required=True, help="Date folder name (e.g. 2026-05-02)")
    parser.add_argument("--out", required=True, help="Output JSON path")
    parser.add_argument("--token", default=os.getenv("HF_TOKEN"), help="HF token (env HF_TOKEN by default)")
    args = parser.parse_args()

    api = HfApi(token=args.token)

    # We expect layout like: /<date>/<subfolder>/... or /<date>/*.parquet
    # List top-level date folder non-recursively, then list each subfolder non-recursively.
    base_path = args.date.strip("/")
    collected = []

    try:
        top_items = robust_list_tree(api, args.repo, base_path)
    except RepositoryNotFoundError:
        print(f"Repository not found: {args.repo}", file=sys.stderr)
        sys.exit(1)
    except Exception as exc:
        print(f"Failed to list {base_path}: {exc}", file=sys.stderr)
        sys.exit(1)

    for item in top_items:
        # If item is a folder, list it non-recursively
        # list_repo_tree doesn't tell type directly; try listing it as path.
        # If empty or same as parent, skip duplicates conservatively.
        sub_items = robust_list_tree(api, args.repo, item)
        # Keep only items under the date subtree
        sub_items = [p for p in sub_items if p.startswith(base_path)]
        collected.extend(sub_items)

    # Also include any files directly under date folder from top_items
    # (some repos may place files directly in date folder)
    collected.extend([p for p in top_items if p.startswith(base_path)])

    # Deduplicate and sort for determinism
    collected = sorted(set(collected))

    out_path = Path(args.out)
    out_path.write_text(json.dumps({"date": args.date, "repo": args.repo, "files": collected}, indent=2))
    print(f"Wrote {len(collected)} file paths to {out_path}")

if __name__ == "__main__":
    main()
```
