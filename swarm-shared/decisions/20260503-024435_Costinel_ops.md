# Costinel / ops

## Final Implementation Plan — Top-Hub Signal Panel (CDN-first, <2h, telemetry-aware)

### What we ship
- A **non-blocking Top-Hub Signal Panel** on the Costinel dashboard.
- Defaults to hub **MOC** (configurable via `HUB_NAME`).
- Shows **3 actionable, cost-impact proposals** (title, impact, rationale, next-step).
- Uses **CDN-first static JSON** (`/data/top-hub-signals.json`) with graceful fallback content.
- No HF API calls at runtime; avoids rate limits and keeps ops simple.
- Optional **non-blocking telemetry opt-in** (sendBeacon) that never blocks UI or core workflows.
- Can ship end-to-end in <2h.

---

### Chosen approach (why)
- **CDN-first static JSON** is required to avoid HF API rate limits and to keep the frontend fast and reliable.
- **Graceful fallback** is required for correctness: if CDN fails, the panel must still show useful guidance.
- **Telemetry must be opt-in and non-blocking** (sendBeacon) so it never harms core cost workflows.
- **Actionability is prioritized**: each proposal must include a concrete next-step.

2) **React component**  
   Path: `src/components/TopHubSignalPanel.tsx`

```tsx
import { useEffect, useState } from "react";
import "./TopHubSignalPanel.css";

const CDN_URL = (window as any).CDN_BASE_URL
  ? `${(window as any).CDN_BASE_URL}/data/top-hub-signals.json`
  : "/data/top-hub-signals.json";

const FALLBACK: any = {
  hub: "MOC",
  proposals: [
    {
      id: "fallback-ri",
      title: "Review RI coverage for top on-demand spenders",
      impact: "HIGH",
      rationale: "Reduces cost with committed use discounts where utilization is consistently high.",
      nextStep: "Identify top 5 on-demand instance families and model RI/Savings Plan options."
    },
    {
      id: "fallback-storage",
      title: "Audit unattached storage and idle resources",
      impact: "MEDIUM",
      rationale: "Orphaned volumes and underused instances often accumulate unnoticed cost.",
      nextStep: "Run weekly unattached-resource report and tag owners for review."
    }
  ]
};

type Proposal = {
  id: string;
  title: string;
  impact: "HIGH" | "MEDIUM" | "LOW";
  rationale: string;
  nextStep: string;
};

type SignalData = {
  hub: string;
  generatedAt: string;
  ttlSeconds: number;
  proposals: Proposal[];
};

export default function TopHubSignalPanel() {
  const [data, setData] = useState<SignalData | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    let mounted = true;
    const fetchData = async () => {
      try {
        const res = await fetch(CDN_URL, { cache: "no-store" });
        if (!res.ok) throw new Error(`CDN fetch failed: ${res.status}`);
        const json = await res.json();
        if (mounted) {
          setData(json);
          // Non-blocking, opt-in telemetry (must be enabled globally)
          if ((window as any).ENABLE_SIGNAL_PING === true) {
            try {
              navigator.sendBeacon?.(
                "/api/telemetry/signal-view",
                JSON.stringify({ hub: json.hub, count: json.proposals.length, ts: Date.now() })
              );
            } catch {
              // ignore telemetry failures
            }
          }
        }
      } catch (err: any) {
        if (mounted) {
          setError(err.message);
          // Use fallback content so panel remains useful
          setData(FALLBACK);
        }
      } finally {
        if (mounted) setLoading(false);
      }
    };

    fetchData();
    return () => {
      mounted = false;
    };
  }, []);

  if (loading) {
    return (
      <div className="top-hub-panel loading">
        <div className="panel-header">
          <span className="hub-badge">MOC</span>
          <span className="panel-title">Top-Hub Signals</span>
        </div>
        <div className="panel-body skeleton">Loading signals…</div>
      </div>
    );
  }

  // data is always set (either CDN or fallback)
  const displayData = data || FALLBACK;
  const impactClass = (imp: string) => `impact-${imp.toLowerCase()}`;

  return (
    <div className="top-hub-panel">
      <div className="panel-header">
        <span className="hub-badge">{displayData.hub}</span>
        <span className="panel-title">Top-Hub Signals</span>
        {data?.generatedAt && (
          <span className="panel-meta">Updated {new Date(data.generatedAt).toLocaleDateString()}</span>
        )}
        {error && <span className="panel-meta error-note">Using fallback guidance</span>}
      </div>
      <div className="panel-body">
        <ol className="proposal-list">
          {displayData.proposals.map((p: Proposal) => (
            <li key={p.id} className="proposal-item">
              <div className="proposal-title">{p.title}</div>
              <div className="proposal-meta">
                <span className={`impact-badge ${impactClass(p.impact)}`}>{p.impact}</span>
              </div>
              <p className="proposal-rationale">{p.rationale}</p>
              <p className="proposal-next"><strong>Next:</strong> {p.nextStep}</p>
            </li>
          ))}
        </ol>
      </div>
    </div>
  );
}
```
