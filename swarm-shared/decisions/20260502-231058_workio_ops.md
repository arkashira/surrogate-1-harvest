# workio / ops

## 1. Diagnosis

- No storage-level uniqueness constraint on `punches(employee_id, punch_type, date)` → retries/races create duplicate punches.
- No idempotency key on punch writes → LINE webhook retries and client retries are treated as new punches.
- Punch write path uses `INSERT` without `ON CONFLICT` handling → duplicates fail loudly or silently create extra rows.
- No transactional upsert for clock-in/out transitions (e.g., prevent double clock-in without intervening clock-out).
- Missing index for fast conflict checks on `(employee_id, date, punch_type)` → race window larger under load.

## 2. Proposed change

File: `/opt/axentx/workio/server/src/db/schema.sql`  
Scope:  
- Add unique constraint on `(employee_id, punch_type, date)` for punch integrity.  
- Add idempotency table `punch_idempotency` with key + unique constraint to deduplicate retries at API boundary.  
- Add index for fast conflict checks.  
- Update punch write path in `/opt/axentx/workio/server/src/routes/punch.ts` (or equivalent) to use idempotency key and transactional upsert.

## 3. Implementation

### 3.1 Schema changes (`server/src/db/schema.sql`)

```sql
-- 1) Ensure punches uniqueness at storage level
ALTER TABLE punches
  DROP CONSTRAINT IF EXISTS punches_employee_type_date_unique;
ALTER TABLE punches
  ADD CONSTRAINT punches_employee_type_date_unique
  UNIQUE (employee_id, punch_type, date);

-- 2) Idempotency table for punch writes (safe retry)
CREATE TABLE IF NOT EXISTS punch_idempotency (
  idempotency_key TEXT PRIMARY KEY,
  employee_id     INTEGER NOT NULL,
  punch_type      TEXT    NOT NULL,
  date            DATE    NOT NULL,
  punch_id        INTEGER NOT NULL REFERENCES punches(id) ON DELETE CASCADE,
  created_at      TIMESTAMPTZ DEFAULT NOW()
);

-- Optional: cleanup old keys (keep 24-48h)
CREATE INDEX IF NOT EXISTS idx_punch_idempotency_created_at
  ON punch_idempotency (created_at);
```

### 3.2 Punch route upsert with idempotency (`server/src/routes/punch.ts`)

```ts
import { Request, Response } from 'express';
import { db } from '../db';
import { v4 as uuidv4 } from 'uuid';

// POST /punch
// Body: { employee_id, punch_type, date, idempotency_key? }
// If idempotency_key provided, reuse existing punch for that key.
export async function createPunch(req: Request, res: Response) {
  const { employee_id, punch_type, date } = req.body;
  const idempotencyKey = req.body.idempotency_key || uuidv4();
  const client = await db.pool.connect();

  try {
    await client.query('BEGIN');

    // 1) Check idempotency first
    const idem = await client.query(
      `SELECT punch_id FROM punch_idempotency WHERE idempotency_key = $1`,
      [idempotencyKey]
    );

    if (idem.rows.length > 0) {
      const punch = await client.query(
        `SELECT id, employee_id, punch_type, date FROM punches WHERE id = $1`,
        [idem.rows[0].punch_id]
      );
      await client.query('COMMIT');
      return res.json({ punch: punch.rows[0], idempotent: true });
    }

    // 2) Try insert; if unique violation, return existing punch
    let punchId: number;
    try {
      const ins = await client.query(
        `INSERT INTO punches (employee_id, punch_type, date, created_at)
         VALUES ($1, $2, $3, NOW())
         RETURNING id`,
        [employee_id, punch_type, date]
      );
      punchId = ins.rows[0].id;
    } catch (err: any) {
      if (err.code === '23505') { // unique violation
        const exist = await client.query(
          `SELECT id FROM punches
           WHERE employee_id = $1 AND punch_type = $2 AND date = $3
           LIMIT 1`,
          [employee_id, punch_type, date]
        );
        if (exist.rows.length === 0) throw err;
        punchId = exist.rows[0].id;
      } else {
        throw err;
      }
    }

    // 3) Record idempotency (ignore conflict if race)
    await client.query(
      `INSERT INTO punch_idempotency (idempotency_key, employee_id, punch_type, date, punch_id)
       VALUES ($1, $2, $3, $4, $5)
       ON CONFLICT (idempotency_key) DO UPDATE
       SET punch_id = EXCLUDED.punch_id`,
      [idempotencyKey, employee_id, punch_type, date, punchId]
    );

    await client.query('COMMIT');

    const punch = await client.query(
      `SELECT id, employee_id, punch_type, date FROM punches WHERE id = $1`,
      [punchId]
    );

    return res.json({ punch: punch.rows[0], idempotent: false });
  } catch (err) {
    await client.query('ROLLBACK');
    console.error('Punch creation failed', err);
    return res.status(500).json({ error: 'Punch creation failed' });
  } finally {
    client.release();
  }
}
```

### 3.3 Client/web integration (quick guidance)

- Generate and persist an `idempotency_key` per user action (e.g., per clock-in intent) and reuse it for retries.
- For LINE webhook handler, extract a deterministic idempotency key from `webhookEvent.id` (or compose from `userId + timestamp + type`) and pass it to the punch route.
