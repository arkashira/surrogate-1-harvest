def fetch_files(file_list):

        url = f"https://huggingface.co/datasets/{repo}/resolve/main/{file}"
        response = requests.get(url)
        files.append(response.content)
    return files

def project_to_prompt_response(parquet_file):

    prompt = table["prompt"]
    response = table["response"]
    return prompt, response

    with open(file_path, "r") as f:
        data = f.read()
    return {"prompt": data[:128], "response": data[128:]}

def get_file_manifest(repo, path):
    file_data = get_file_data(repo, path)
