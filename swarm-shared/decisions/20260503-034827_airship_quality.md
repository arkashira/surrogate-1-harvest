```python
#!/usr/bin/env python3
"""
Generate file_index.json for CDN-only dataset loading.
Usage:
  python list_hf_files.py <repo> <date_folder> [--out surrogate/file_index.json]
Example:
  python list_hf_files.py axentx/surrogate-dataset 2026-04-29
"""
import argparse
import json
import os
import sys
from pathlib import Path

try:
    from huggingface_hub import HfApi
except ImportError:
    print("Install: pip install huggingface_hub", file=sys.stderr)
    sys.exit(1)

HF_TOKEN = os.getenv("HF_TOKEN", "")  # optional; public repos don't require it

def main() -> None:
    parser = argparse.ArgumentParser(description="List HF dataset files for CDN loader.")
    parser.add_argument("repo", help="Hugging Face dataset repo (e.g. axentx/surrogate-dataset)")
    parser.add_argument("date_folder", help="Date folder to list (e.g. 2026-04-29)")
    parser.add_argument("--out", default="surrogate/file_index.json", help="Output JSON path")
    args = parser.parse_args()

    api = HfApi(token=HF_TOKEN if HF_TOKEN else None)
    # Non-recursive per folder to avoid 100x pagination and rate limits
    tree = api.list_repo_tree(repo=args.repo, path=args.date_folder, recursive=False)

    entries = []
    for item in tree:
        if not item.size or item.size == 0:
            continue
        entries.append(
            {
                "repo": args.repo,
                "path": f"{args.date_folder}/{item.path}",
                "size": item.size,
            }
        )

    out_path = Path(args.out)
    out_path.parent.mkdir(parents=True, exist_ok=True)
    with out_path.open("w") as f:
        json.dump({"repo": args.repo, "date_folder": args.date_folder, "files": entries}, f, indent=2)

    print(f"Wrote {len(entries)} files to {out_path}")

if __name__ == "__main__":
    main()
```

- Uses HuggingFace CDN URLs (no auth, bypasses API rate limits).

import json
import os
import sys

import requests

CDN_BASE = "https://huggingface.co/datasets"
INDEX_PATH = Path(__file__).parent / "file_index.json"

class CDNParquetIterable(IterableDataset):
    """Stream Parquet files via CDN with zero HF API calls during training."""

    def __init__(self, file_index_path: Path = INDEX_PATH, columns=("prompt", "response")):
        super().__init__()
        self.file_index_path = file_index_path
        self.columns = columns
        self._files: list[Dict] = []

    def _load_index(self) -> list[Dict]:
        if not self._files:
            with self.file_index_path.open() as f:
                idx = json.load(f)
            self._files = idx.get("files", [])
        return self._files

    def _stream_rows(self, repo: str, cdn_path: str) -> Iterator[Tuple[str, str]]:
        url = f"{CDN_BASE}/{repo}/resolve/main/{cdn_path}"
        # CDN downloads are not counted against HF API rate limits
        resp = requests.get(url, timeout=60, stream=True)
        resp.raise_for_status()
        with open("tmp.parquet", "wb") as tmp:
            for chunk in resp.iter_content(chunk_size=8192):
                tmp.write(chunk)
        table = pq.read_table("tmp.parquet", columns=self.columns)
        os.remove("tmp.parquet")
        for i in range(table.num_rows):
            row = table.slice(i, 1).to_pydict()
            prompt = row.get("prompt", [""])[0]
            response = row.get("response", [""])[0]
            if prompt and response:
                yield prompt, response

    def __iter__(self):
        files = self._load_index()
        for entry in files:
            repo = entry["repo"]
            cdn_path = entry["path"]
            try:
                for prompt, response in self._stream_rows(repo, cdn_path):
                    yield {"input_ids": prompt, "labels": response}
            except Exception as exc:
                print(f"Skipping {cdn_path}: {exc}", file=sys.stderr)
                continue
