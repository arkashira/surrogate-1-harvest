# vanguard / ops

## 1. Diagnosis
- No persisted manifest per `(repo, dateFolder)` → every training launch re-enumerates via authenticated HF API, burning quota and risking 429.
- Training script likely uses `load_dataset(streaming=True)` on heterogeneous repos → triggers PyArrow `CastError` on mixed schemas.
- Lightning Studio is probably recreated on each run instead of reused → wastes 80+ hrs/mo quota.
- No CDN-only data path → training performs authenticated fetches during data loading instead of using public CDN URLs.
- No idle-stop resilience → Lightning idle timeout kills long-running training jobs without restart logic.

## 2. Proposed change
Create `/opt/axentx/vanguard/ops/training_manifest.py` (single source of truth) and update the training launcher to:
- Generate and persist a `manifest_{repo}_{date}.json` listing only file paths for one date folder via a single `list_repo_tree` call.
- Embed that manifest in the training script so Lightning workers fetch via CDN URLs only (zero API calls during training).
- Reuse a running Lightning Studio or restart gracefully if stopped.
- Project to `{prompt, response}` at parse time (never rely on `load_dataset(streaming=True)` for heterogeneous repos).

## 3. Implementation

# /opt/axentx/vanguard/ops/training_manifest.py
#!/usr/bin/env bash
# Generate a persisted manifest for (repo, dateFolder) to enable CDN-only training.
# Usage: ./training_manifest.py <repo> <dateFolder> [outDir]
# Example: ./training_manifest.py datasets/mycorp/surrogate-1 2026-04-29 ./manifests

set -euo pipefail
SHELL=/bin/bash

REPO="${1:-datasets/mycorp/surrogate-1}"
DATEFOLDER="${2:-$(date +%Y-%m-%d)}"
OUTDIR="${3:-./manifests}"
MANIFEST="${OUTDIR}/manifest_${REPO//\//_}_${DATEFOLDER}.json"

mkdir -p "$OUTDIR"

# Mac-only orchestration: list once, embed in training script.
# Uses HF API (rate-limited) a single time per (repo,dateFolder).
python3 - "$REPO" "$DATEFOLDER" "$MANIFEST" <<'PY'
import json, os, sys
from huggingface_hub import HfApi

def main():
    repo, date_folder, out_path = sys.argv[1], sys.argv[2], sys.argv[3]
    api = HfApi()

    # List non-recursive for the date folder to avoid 100x pagination.
    tree = api.list_repo_tree(repo=repo, path=date_folder, recursive=False)
    files = [
        item.rfilename
        for item in tree
        if item.rfilename.lower().endswith((".parquet", ".jsonl", ".json"))
    ]

    # CDN-only URLs (no auth). These bypass HF API rate limits during training.
    cdn_urls = [
        f"https://huggingface.co/datasets/{repo}/resolve/main/{f}"
        for f in files
    ]

    manifest = {
        "repo": repo,
        "date_folder": date_folder,
        "files": files,
        "cdn_urls": cdn_urls,
        "generated_by": "vanguard/ops/training_manifest.py",
    }

    os.makedirs(os.path.dirname(out_path), exist_ok=True)
    with open(out_path, "w", encoding="utf-8") as fh:
        json.dump(manifest, fh, indent=2)
    print(f"Manifest written to {out_path} ({len(files)} files)")

if __name__ == "__main__":
    main()
PY

echo "✅ Manifest generated: $MANIFEST"

# /opt/axentx/vanguard/ops/launch_surrogate_training.py
#!/usr/bin/env python3
"""
Lightning-based launcher for Surrogate-1 training.
- Reuses running Studio or restarts gracefully.
- Uses CDN-only manifest (zero HF API calls during training).
- Projects to {prompt,response} at parse time to avoid mixed-schema errors.
"""

import json
import os
import sys
from pathlib import Path

from lightning import LightningWork, LightningApp, CloudCompute, Machine
from lightning.app.storage import Drive

#
# Worker that loads from CDN manifest and trains (minimal example).
#
class SurrogateTrainer(LightningWork):
    def __init__(
        self,
        manifest_path: str,
        cloud_compute: CloudCompute = CloudCompute("lightning-lambda-prod"),  # prefer H200-capable account
        machine: Machine = "lightning-lambda-prod",
    ):
        super().__init__(cloud_compute=cloud_compute, parallel=True)
        self.manifest_path = manifest_path
        self.machine = machine
        self.drive = Drive(f"lit://vanguard-training-data")

    def run(self, manifest_path: str):
        # Re-check status; if studio stopped, training died. Restart if needed.
        if not self.cloud_compute or not self.is_running:
            # In practice, orchestrator should restart with target machine.
            print("Studio not running or stopped; restart via launcher.")
            return

        # Load CDN-only manifest (generated by ops/training_manifest.py).
        with open(manifest_path, "r", encoding="utf-8") as f:
            manifest = json.load(f)

        cdn_urls = manifest["cdn_urls"]
        if not cdn_urls:
            print("No files in manifest; skipping.")
            return

        # Lightweight parquet loader that projects to {prompt,response} only.
        # Avoids load_dataset(streaming=True) on heterogeneous repos.
        import pyarrow.parquet as pq

        records = []
        for url in cdn_urls:
            # CDN fetch (no auth). If files are large, stream in chunks.
            local_path, _ = self._download_cdn_file(url)
            try:
                table = pq.read_table(local_path, columns=["prompt", "response"])
                records.extend(table.to_pylist())
            except Exception as exc:
                print(f"Skipping {url}: {exc}")
                continue

        print(f"Loaded {len(records)} prompt/response pairs from CDN.")
        # TODO: plug into your actual training loop / HF Trainer.
        # Example: train_on_records(records)

    def _download_cdn_file(self, url: str):
        # Simple CDN download to local workspace (Lightning provides ephemeral disk).
        import requests, tempfile, shutil
        resp = requests.get(url, stream=True, timeout=60)
        resp.raise_for_status()
        fd, path = tempfile.mkstemp(suffix=".parquet")
        os.close(fd)
        with open(path, "wb") as f:
            shutil.copyfileobj(resp.raw, f)
        return path, resp.headers.get("content-length")

#
# App orchestration: reuse running studio or start one.
#
def find_running_studio(name: str):
    from lightning.app import Teamspace
    for s in Teamspace.studios:
        if s.name == name and s.status == "running":
            return s
    return None

def main():
    if len(sys.argv) < 2:
        print("Usage: launch_surrogate_training.py <manifest.json>")
        sys.exit(1)

    manifest_path = Path(sys.argv[1]).expanduser().absolute()
    if not manifest_path.is_file():
        print(f"Manifest not found: {manifest_path}")
        sys.exit(1)

    studio_name = "vanguard-surrogate-training"
    studio = find_running_studio(studio_name)

    # Prefer Lightning-lambda-prod for H200; fallback handled by Lightning public tier.
    machine = Machine.L40S  # safe default; Lightning will map to available cloud account
    cloud_compute = CloudCompute("lightning-lambda-prod")

    trainer = SurrogateTrainer(
        manifest_path=str(manifest_path),
        cloud_compute=cloud_compute,
        machine=machine,
    )

    app = LightningApp(trainer)
    print(f"Started training app using manifest: {manifest_path}")
    # LightningApp.run() is handled by `lightning run app` in production.
    # This script is intended to be invoked via `python launch_surrogate_training.py manifest.json`
    # or via `lightning run app` with proper entrypoint.

if __name__ == "__main__":
    main()
