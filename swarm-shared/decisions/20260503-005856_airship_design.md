### 1. Orchestration: `scripts/build_manifest.py` (Mac/Linux)

Generates a CDN manifest for a single date folder. Run once per date when HF API limits allow.

```python
#!/usr/bin/env python3
"""
Generate CDN manifest for a Surrogate-1 date folder.
Usage:
  python scripts/build_manifest.py --repo surrogate-1-mirror --date 2026-04-29 --out surrogate/file_list.json
"""

import argparse
import json
import os
from pathlib import Path
from huggingface_hub import HfApi

HF_API = HfApi()

def build_manifest(repo_id: str, date_folder: str, out_path: str) -> None:
    entries = HF_API.list_repo_tree(repo_id=repo_id, path=date_folder, recursive=False)
    files = sorted(e.rfilename for e in entries if e.type == "file")

    base_cdn = f"https://huggingface.co/datasets/{repo_id}/resolve/main"
    manifest = {
        "repo_id": repo_id,
        "date_folder": date_folder,
        "files": files,
        "cdn_base": base_cdn,
        "urls": [f"{base_cdn}/{f}" for f in files],
    }

    out_path = Path(out_path)
    out_path.parent.mkdir(parents=True, exist_ok=True)
    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"Manifest written to {out_path} ({len(files)} files)")

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Build CDN manifest for Surrogate-1 date folder")
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g. surrogate-1-mirror)")
    parser.add_argument("--date", required=True, help="Date folder (e.g. 2026-04-29)")
    parser.add_argument("--out", default="surrogate/file_list.json", help="Output JSON path")
    args = parser.parse_args()
    build_manifest(args.repo, args.date, args.out)
```

Make executable:

```bash
chmod +x scripts/build_manifest.py
```

### 2. Training Entrypoint: `surrogate/train.py`

CDN-only loader with robust schema projection and Lightning integration.

```python
#!/usr/bin/env python3
"""
Surrogate-1 training entrypoint (Lightning Studio).
- Uses manifest with CDN URLs only (zero HF API calls during training).
- Projects heterogeneous files to {prompt, response} at parse time.
"""

import json
import os
from pathlib import Path
from typing import Dict, Any, List

import torch
from torch.utils.data import Dataset, DataLoader, IterableDataset
import requests
from lightning import LightningModule, Trainer
from lightning.pytorch.strategies import DeepSpeedStrategy, FSDPStrategy

MANIFEST_PATH = Path(__file__).parent / "file_list.json"

# ---- Dataset ----

PROMPT_KEYS = {"prompt", "instruction", "question", "input"}
RESPONSE_KEYS = {"response", "output", "answer", "completion"}

def _normalize_record(obj: Dict[str, Any]) -> Dict[str, str]:
    if not isinstance(obj, dict):
        return {"prompt": "", "response": ""}
    prompt = next((str(obj[k]) for k in PROMPT_KEYS if k in obj and obj[k] is not None), "")
    response = next((str(obj[k]) for k in RESPONSE_KEYS if k in obj and obj[k] is not None), "")
    return {"prompt": prompt, "response": response}

class CDNTextIterableDataset(IterableDataset):
    """
    Streams samples from CDN files. Each worker gets a shard of URLs.
    Yields one sample per record (flattened).
    """

    def __init__(self, urls: List[str], max_samples_per_file: int = -1):
        self.urls = urls
        self.max_samples_per_file = max_samples_per_file

    def _fetch_and_yield(self, url: str):
        try:
            resp = requests.get(url, timeout=30)
            resp.raise_for_status()
        except Exception as exc:
            # Log and skip bad files; don't crash training
            print(f"Failed to fetch {url}: {exc}")
            return

        count = 0
        for line in resp.text.strip().splitlines():
            line = line.strip()
            if not line:
                continue
            try:
                obj = json.loads(line)
                rec = _normalize_record(obj)
                if rec["prompt"] or rec["response"]:
                    yield rec
                    count += 1
                    if 0 < self.max_samples_per_file <= count:
                        break
            except json.JSONDecodeError:
                continue

    def __iter__(self):
        worker_info = torch.utils.data.get_worker_info()
        if worker_info is None:
            urls = self.urls
        else:
            per_worker = len(self.urls) // worker_info.num_workers
            urls = self.urls[worker_info.id * per_worker : (worker_info.id + 1) * per_worker]

        for url in urls:
            yield from self._fetch_and_yield(url)

# ---- DataModule ----

class SurrogateDataModule:
    def __init__(
        self,
        manifest_path: Path,
        batch_size: int = 4,
        max_samples_per_file: int = -1,
        num_workers: int = 2,
    ):
        with open(manifest_path) as f:
            manifest = json.load(f)
        self.urls = manifest.get("urls", [])
        self.batch_size = batch_size
        self.max_samples_per_file = max_samples_per_file
        self.num_workers = num_workers

    def train_dataloader(self) -> DataLoader:
        dataset = CDNTextIterableDataset(self.urls, max_samples_per_file=self.max_samples_per_file)
        return DataLoader(
            dataset,
            batch_size=self.batch_size,
            num_workers=self.num_workers,
            pin_memory=True,
            # IterableDataset doesn't support shuffle=True; shuffle via URL order or sampler if needed
        )

# ---- Model ----

class SurrogateModel(LightningModule):
    def __init__(self, lr: float = 1e-4):
        super().__init__()
        self.save_hyperparameters()
        # Replace with Surrogate-1 architecture
        self.model = torch.nn.Linear(10, 10)  # placeholder

    def training_step(self, batch, batch_idx):
        # Replace with real forward + loss
        loss = torch.tensor(0.0, requires_grad=True)
        self.log("train_loss", loss)
        return loss

    def configure_optimizers(self):
        return torch.optim.AdamW(self.parameters(), lr=self.hyperparameters.lr)

# ---- Main ----

def main():
    if not MANIFEST_PATH.exists():
        raise FileNotFoundError(f"Manifest not found: {MANIFEST_PATH}. Run scripts/build_manifest.py first.")

    data = SurrogateDataModule(
        manifest_path=MANIFEST_PATH,
        batch_size=4,
        max_samples_per_file=32,  # small for quick iteration
        num_workers=2,
    )
    model = SurrogateModel(lr=1e-4)

    # Prefer DeepSpeed or FSDP on multi-GPU; fallback to auto
    strategy = DeepSpeedStrategy() if torch.cuda.device_count() > 1 else "auto"

    trainer = Trainer(
        max_epochs=1,
        accelerator="gpu",
        devices="auto",
        strategy=strategy,
        precision="bf16-mixed",
        limit_train_batches=10,  # quick smoke test
    )
    trainer.fit(model, datamodule=data)

if __name__ == "__main__":
    main()
```
