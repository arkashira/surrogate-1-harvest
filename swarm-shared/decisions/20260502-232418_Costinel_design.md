## Implementation Plan — Costinel Top-Hub Signal Endpoint

**Highest-value incremental improvement** (ship <2h):  
Expose a **read-only, side-effect-free** `GET /api/v1/cost-anomaly/signal/top-hub` that surfaces the most-connected hub (e.g., "MOC") with contextual insights for governance decisions. No mutations, no external writes, no background jobs.

---

### 1) Architecture & Flow
- **Endpoint**: `GET /api/v1/cost-anomaly/signal/top-hub`
- **Auth**: Existing app auth (read-only scope)
- **Data source**: In-memory graph built from latest daily cost export (parquet) + existing anomaly/relationship metadata
- **Response**: JSON with hub node, top edges, and signal summary
- **Constraints**: No `POST/PUT/DELETE`; no external mutations; cacheable (30–60s)

#### B) Service: compute top-hub signal
`/opt/axentx/Costinel/src/services/top_hub.py`

```python
# src/services/top_hub.py
from pathlib import Path
import json
from datetime import datetime, timezone
import networkx as nx
import pandas as pd
from typing import Any, Dict, List

# Path conventions (adjust if project layout differs)
DATA_DIR = Path(__file__).parents[2] / "data"
ENRICHED_DIR = DATA_DIR / "enriched"

def _latest_daily_parquet() -> Path:
    # Expect pattern: batches/mirror-merged/{date}/{slug}.parquet
    candidates = sorted(ENRICHED_DIR.rglob("*.parquet"), key=lambda p: p.stat().st_mtime, reverse=True)
    if not candidates:
        raise FileNotFoundError("No parquet files found in enriched/")
    return candidates[0]

def _build_graph_from_parquet(path: Path) -> nx.Graph:
    # Project only required fields; avoid mixed-schema pitfalls
    df = pd.read_parquet(path, columns=["account_id", "service", "region", "anomaly_id", "related_ids"])
    g = nx.Graph()

    # Nodes: use typed ids to avoid collisions
    for _, row in df.iterrows():
        g.add_node(f"account:{row['account_id']}", type="account")
        g.add_node(f"service:{row['service']}", type="service")
        g.add_node(f"region:{row['region']}", type="region")
        if pd.notna(row["anomaly_id"]):
            g.add_node(f"anomaly:{row['anomaly_id']}", type="anomaly")
            g.add_edge(f"account:{row['account_id']}", f"anomaly:{row['anomaly_id']}")
            g.add_edge(f"service:{row['service']}", f"anomaly:{row['anomaly_id']}")
            g.add_edge(f"region:{row['region']}", f"anomaly:{row['anomaly_id']}")

        # related_ids expected as JSON list string or already list; best-effort
        if pd.notna(row["related_ids"]):
            try:
                related = json.loads(row["related_ids"]) if isinstance(row["related_ids"], str) else row["related_ids"]
            except Exception:
                related = []
            for rel in related:
                g.add_node(f"anomaly:{rel}", type="anomaly")
                g.add_edge(f"anomaly:{row['anomaly_id']}", f"anomaly:{rel}")

    return g

def _top_hub_by_pagerank(g: nx.Graph, top_k: int = 5) -> List[Dict[str, Any]]:
    pr = nx.pagerank(g)
    # Filter to anomaly/service nodes for governance relevance
    filtered = {n: s for n, s in pr.items() if g.nodes[n]["type"] in ("anomaly", "service")}
    sorted_nodes = sorted(filtered.items(), key=lambda kv: kv[1], reverse=True)
    return [{"node": n, "score": round(s, 6)} for n, s in sorted_nodes[:top_k]]

def get_top_hub_signal() -> Dict[str, Any]:
    parquet_path = _latest_daily_parquet()
    g = _build_graph_from_parquet(parquet_path)
    top_nodes = _top_hub_by_pagerank(g)

    if not top_nodes:
        raise ValueError("No top hubs found in graph")

    primary = top_nodes[0]
    insights = [
        f"Top hub: {primary['node']} (score {primary['score']})",
        "Connected to multiple anomalies and cost centers; prioritize governance review.",
        "Recommended action: inspect linked anomalies and RI/SP coverage for this hub."
    ]

    return {
        "hub": primary["node"],
        "score": primary["score"],
        "top_nodes": top_nodes,
        "insights": insights,
        "generatedAt": datetime.now(timezone.utc).isoformat()
    }
```

_TTL_SECONDS = 45
