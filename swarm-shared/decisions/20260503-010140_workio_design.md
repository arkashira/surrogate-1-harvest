# workio / design

## Final synthesized answer

**Core problem**: multiple open punches per user per day can occur under LINE webhook retries, concurrent taps, or race conditions in `find-then-insert` logic. The system lacks an atomic, DB-enforced uniqueness guarantee and an idempotent write path.

**Chosen approach**: combine the strongest parts of both proposals:

- Use a **partial unique index** (Candidate 2) to enforce *“at most one open punch per user per tenant per calendar day”* at the database level. This is precise, low-overhead, and preserves ability to have multiple closed punches per day (segments/history) while preventing duplicate open punches.
- Add a **deterministic `daily_punch_id` column** (Candidate 1) to make retries idempotent and to provide a stable, conflict-safe key for upserts and deduplication.
- Perform clock-in via **atomic `INSERT … ON CONFLICT` inside an explicit transaction** (Candidate 1), returning the existing row on conflict so callers can safely retry.
- Keep schema changes migration-friendly and include tenant_id in constraints (Candidate 2) for multi-tenant correctness.

### 1. Schema changes (`workio/server/src/db/schema.sql` or migration)

```sql
-- Add deterministic daily punch id for idempotency and stable conflict target
ALTER TABLE punches
  ADD COLUMN IF NOT EXISTS daily_punch_id TEXT;

-- Backfill existing rows
UPDATE punches
  SET daily_punch_id = user_id || ':' || tenant_id || ':' || DATE(COALESCE(clock_in_at, created_at))
  WHERE daily_punch_id IS NULL;

ALTER TABLE punches
  ALTER COLUMN daily_punch_id SET NOT NULL;

-- Unique constraint on daily_punch_id to prevent duplicate day entries at all
-- (This defends against retries that might otherwise try to create a second row.)
CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_daily_punch_id
  ON punches (daily_punch_id);

-- Partial unique index: at most one open punch per user per tenant per calendar day
-- This is the primary guardrail against duplicate open punches.
CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_one_open_per_user_per_day
  ON punches (user_id, tenant_id, DATE(clock_in_at))
  WHERE clock_out_at IS NULL;

-- Optional indexes for performance
CREATE INDEX IF NOT EXISTS idx_punches_user_date
  ON punches (user_id, DATE(clock_in_at));
CREATE INDEX IF NOT EXISTS idx_punches_user_open
  ON punches (user_id, tenant_id)
  WHERE clock_out_at IS NULL;
```

### 2. Service implementation (`punch.service.ts`)

```ts
// workio/server/src/services/punch.service.ts
import { Pool } from 'pg';

const pool = new Pool({ connectionString: process.env.DATABASE_URL });

export async function clockIn(userId: string, tenantId: string): Promise<{
  id: string;
  daily_punch_id: string;
  clock_out_at: Date | null;
}> {
  const client = await pool.connect();
  const today = new Date().toISOString().split('T')[0]; // YYYY-MM-DD
  const dailyPunchId = `${userId}:${tenantId}:${today}`;

  try {
    await client.query('BEGIN');

    // Try to insert; on conflict (daily_punch_id or partial unique) do nothing and return existing.
    const insertRes = await client.query(
      `INSERT INTO punches (id, user_id, tenant_id, clock_in_at, daily_punch_id)
       VALUES (gen_random_uuid(), $1, $2, NOW(), $3)
       ON CONFLICT (daily_punch_id) DO NOTHING
       RETURNING id, daily_punch_id, clock_out_at`,
      [userId, tenantId, dailyPunchId]
    );

    let punch;
    if (insertRes.rowCount === 0) {
      // Row already exists — return current state so caller can decide next action.
      const selectRes = await client.query(
        `SELECT id, daily_punch_id, clock_out_at FROM punches WHERE daily_punch_id = $1`,
        [dailyPunchId]
      );
      punch = selectRes.rows[0];
    } else {
      punch = insertRes.rows[0];
    }

    await client.query('COMMIT');
    return { id: punch.id, daily_punch_id: punch.daily_punch_id, clock_out_at: punch.clock_out_at };
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  } finally {
    client.release();
  }
}

// Optional: atomic clock-out helper
export async function clockOut(userId: string, tenantId: string): Promise<{ id: string; clock_out_at: Date }> {
  const client = await pool.connect();
  const today = new Date().toISOString().split('T')[0];
  const dailyPunchId = `${userId}:${tenantId}:${today}`;

  try {
    await client.query('BEGIN');

    // Atomically close the open punch for today, if one exists.
    const res = await client.query(
      `UPDATE punches
       SET clock_out_at = NOW()
       WHERE daily_punch_id = $1 AND clock_out_at IS NULL
       RETURNING id, clock_out_at`,
      [dailyPunchId]
    );

    if (res.rowCount === 0) {
      await client.query('ROLLBACK');
      throw new Error('No open punch found for today');
    }

    await client.query('COMMIT');
    return { id: res.rows[0].id, clock_out_at: res.rows[0].clock_out_at };
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  } finally {
    client.release();
  }
}
```

### 3. Webhook handler (`line/webhook.ts`)

```ts
// workio/server/src/routes/line/webhook.ts
import express from 'express';
import { clockIn } from '../../services/punch.service';

const router = express.Router();

router.post('/webhook/line', async (req, res) => {
  try {
    const { events } = req.body;
    for (const ev of events) {
      if (ev.type === 'message' && ev.message.type === 'text') {
        const text = ev.message.text.trim().toLowerCase();
        if (text === 'clock in' || text === 'เข้างาน') {
          const userId = ev.source.userId; // LINE user ID
          const tenantId = ev.source.groupId || ev.source.roomId || 'default'; // adapt as needed
          await clockIn(userId, tenantId);
        }
      }
    }
    res.status(200).send('OK');
  } catch (err) {
    console.error(err);
    res.status(500).send('ERR');
  }
});

export default router;
```
