**Problem**: Surrogate-1 training is blocked by HF API rate limits (429) during dataset ingestion.  
**Goal**: ≤2h fix that eliminates HF API calls during training by generating a **CDN-only file manifest** and adapting the training loader to use CDN URLs.

## Chosen Scope & Rationale

- **CLI**: `airship discover` (Python, not bash) — robust argument handling, reuseable logic.  
- **API**: FastAPI endpoint `/api/discover` (not `/api/dataset/manifest`) — consistent naming with CLI, cacheable.  
- **Training loader**: manifest-aware iterator that streams from CDN with retries and schema projection to `{prompt, response}`.  
- **CDN fetcher utility**: session with retries for resilience.

**Why Python CLI over bash**: enables reuse of `build_manifest` logic with the API, proper error handling, and easier future extension.  
**Why `/api/discover`**: matches CLI naming and avoids proliferating endpoint names.  
**Why manifest-first**: one HF API call up-front, then zero during training — directly solves the rate-limit block.

### 1. CLI: `airship/cli/discover.py`

```python
#!/usr/bin/env python3
"""
airship discover
Generate CDN-only file manifest for a HF dataset repo/date folder.

Usage:
  airship discover --repo datasets/my-org/surrogate-mirror --date 2026-04-29 --out ./manifests/mirror-2026-04-29.json
"""
import argparse
import json
import os
import sys
from datetime import datetime, timezone

from huggingface_hub import list_repo_tree

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"


def normalize_repo_id(repo_id: str) -> str:
    """Accept 'org/repo' or 'datasets/org/repo' and return 'datasets/org/repo'."""
    repo_id = repo_id.strip()
    if not repo_id.startswith("datasets/"):
        repo_id = f"datasets/{repo_id}"
    return repo_id


def build_manifest(repo_id: str, date_folder: str):
    """
    Single API call: list one folder (non-recursive).
    Returns manifest dict.
    """
    repo_id = normalize_repo_id(repo_id)
    items = list_repo_tree(repo_id=repo_id, path=date_folder, recursive=False)

    files = []
    for item in items:
        if item.type != "file":
            continue
        # item.path is relative to repo root; ensure it includes date_folder when needed
        path = item.path if item.path.startswith(date_folder) else f"{date_folder}/{item.path}"
        files.append({
            "path": path,
            "cdn_url": CDN_TEMPLATE.format(repo=repo_id, path=path),
            "size": item.size,
            "sha": getattr(item, "sha", None),
        })

    return {
        "repo_id": repo_id,
        "date_folder": date_folder,
        "generated_at": datetime.now(timezone.utc).isoformat(),
        "count": len(files),
        "files": files,
    }


def main():
    parser = argparse.ArgumentParser(description="Generate CDN manifest for HF dataset folder")
    parser.add_argument("--repo", required=True, help="HF dataset repo id (e.g. org/repo or datasets/org/repo)")
    parser.add_argument("--date", required=True, help="Date folder (e.g. 2026-04-29)")
    parser.add_argument("--out", help="Output JSON path (default: ./manifests/{repo_slug}-{date}.json)")
    args = parser.parse_args()

    if args.out is None:
        slug = normalize_repo_id(args.repo).replace("/", "-").replace("datasets-", "")
        os.makedirs("./manifests", exist_ok=True)
        args.out = f"./manifests/{slug}-{args.date}.json"

    try:
        manifest = build_manifest(args.repo, args.date)
    except Exception as exc:
        print(f"ERROR listing repo tree: {exc}", file=sys.stderr)
        sys.exit(1)

    with open(args.out, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2)
    print(f"Manifest written: {args.out} ({manifest['count']} files)")


if __name__ == "__main__":
    main()
```

### 2. FastAPI endpoint: `airship/api/discover.py`

```python
from fastapi import APIRouter, HTTPException
from starlette.responses import JSONResponse

from airship.cli.discover import build_manifest

router = APIRouter()


@router.get("/api/discover")
def discover(repo: str, date: str):
    """
    Generate CDN manifest for repo/date folder.
    Example: GET /api/discover?repo=org/repo&date=2026-04-29
    """
    try:
        manifest = build_manifest(repo, date)
    except Exception as exc:
        raise HTTPException(status_code=502, detail=f"HF API error: {exc}")

    return JSONResponse(
        content=manifest,
        headers={"Cache-Control": "public, max-age=3600"},
    )
```

### 3. CDN fetcher utility: `airship/cdn.py`

```python
import time
import requests
from requests.adapters import HTTPAdapter, Retry

_session = None


def cdn_session() -> requests.Session:
    global _session
    if _session is None:
        _session = requests.Session()
        retries = Retry(
            total=3,
            backoff_factor=1,
            status_forcelist=[429, 500, 502, 503, 504],
            allowed_methods=["GET", "HEAD"],
        )
        _session.mount("https://", HTTPAdapter(max_retries=retries))
    return _session


def fetch_cdn_file(cdn_url: str, timeout: int = 30) -> bytes:
    resp = cdn_session().get(cdn_url, timeout=timeout, stream=True)
    resp.raise_for_status()
    return resp.content
```
