## 1. Unified Diagnosis (highest-confidence items)
- **No persisted manifest** → every training run re-enumerates via authenticated HF API → quota burn + 429 risk.  
- **Training uses recursive/list or `load_dataset` during data loading** → repeated API calls, mixed-schema pyarrow errors, and CDN bypass.  
- **No CDN-only fetch path** in training scripts → authenticated API calls remain the bottleneck.  
- **No Lightning Studio reuse guard** → idle stop/start cycles waste 80+ hrs/mo quota and lose state.  
- **No deterministic sibling-repo spread** for HF commit cap (128/hr/repo) → ingestion bursts can stall.  

## 2. Single Proposed Change
Create **one orchestration script** that generates a persisted manifest, and patch the training launcher to use CDN-only fetches and reuse studios.

### File: `/opt/axentx/vanguard/ops/persist_manifest.py`
```python
#!/usr/bin/env python3
"""
Generate and cache HF dataset manifests to avoid repeated API enumeration.
Run once per (repo, dateFolder) from Mac/cron after rate-limit window clears.

Usage:
    python persist_manifest.py \
        --repo datasets/company/surrogate-1 \
        --date 2026-04-29 \
        --out ops/manifests/surrogate-1_2026-04-29.json
"""
import argparse
import hashlib
import json
import time
from pathlib import Path
from typing import Dict, List

from huggingface_hub import list_repo_tree

CDN_BASE = "https://huggingface.co/datasets"

def deterministic_repo_shard(slug: str, n_shards: int = 5) -> int:
    """Deterministically map repo slug -> shard index [0, n_shards)."""
    digest = hashlib.sha256(slug.encode()).digest()
    return int.from_bytes(digest, "big") % n_shards

def build_manifest(repo: str, date_folder: str, out_path: Path) -> Dict:
    """
    Build manifest for repo/date_folder (non-recursive).
    Writes JSON to out_path and returns manifest dict.
    """
    prefix = f"{date_folder}/"
    # Single non-recursive call to minimize API usage
    tree = list_repo_tree(repo=repo, path=prefix, recursive=False)
    files = [
        f.rfilename for f in tree
        if f.rfilename.endswith((".parquet", ".jsonl", ".json"))
    ]

    entries: List[Dict] = []
    for f in files:
        entries.append({
            "repo": repo,
            "path": f,
            "cdn_url": f"{CDN_BASE}/{repo}/resolve/main/{f}",
            "shard": deterministic_repo_shard(repo),
        })

    manifest = {
        "repo": repo,
        "date_folder": date_folder,
        "created_ts": int(time.time()),
        "files": entries,
        "note": "Use CDN URLs for training; zero authenticated API calls during data loading."
    }

    out_path.parent.mkdir(parents=True, exist_ok=True)
    with open(out_path, "w", encoding="utf-8") as fp:
        json.dump(manifest, fp, indent=2, ensure_ascii=False)

    return manifest

def load_manifest(manifest_path: Path) -> Dict:
    with open(manifest_path, encoding="utf-8") as fp:
        return json.load(fp)

def main() -> None:
    parser = argparse.ArgumentParser(
        description="Build HF dataset manifest for CDN-only training."
    )
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g., datasets/company/surrogate-1)")
    parser.add_argument("--date", required=True, help="Date folder (e.g., 2026-04-29)")
    parser.add_argument("--out", default="manifest.json", help="Output JSON path")
    args = parser.parse_args()

    out_path = Path(args.out).resolve()
    manifest = build_manifest(args.repo, args.date, out_path)
    print(f"Manifest written to {out_path} with {len(manifest['files'])} files.")

if __name__ == "__main__":
    main()
```

### Lightning Studio reuse + idle guard (in launcher)
```python
from lightning import Teamspace, Studio, Machine

def get_or_create_studio(name: str, machine: Machine = Machine.L40S):
    for s in Teamspace.studios:
        if s.name == name and s.status == "Running":
            print(f"Reusing running studio: {name}")
            return s
    print(f"Creating studio: {name}")
    return Studio(name=name, create_ok=True, machine=machine)

def run_training_on_studio(studio, script_path, args):
    if studio.status != "Running":
        studio.start(machine=studio.machine or Machine.L40S)
    run = studio.run(str(script_path), args=args)
    return run
```
