# Costinel / discovery

## Final Implementation Plan — Top-Hub Signal Panel (Costinel Dashboard)

**Scope & Value**  
Frontend-only, read-only React panel that surfaces the highest-signal/most-connected hub (default “MOC”) and its top 3 actionable proposals from the knowledge graph. CDN-first data strategy: static `top-hub.json` committed/embedded at build (or generated by CI) so the dashboard loads with **zero runtime API calls, auth, or rate-limits**. Ships in ~90–120 minutes.

---

### Acceptance Criteria
- [x] Panel appears on dashboard showing hub name, short description, and top 3 proposals.  
- [x] Each proposal has title, rationale/impact (“signal”), and an actionable label.  
- [x] Data is loaded from `/data/top-hub.json` (CDN-friendly).  
- [x] Graceful fallback UI if file missing or malformed.  
- [x] Responsive and accessible (ARIA labels, keyboard nav).

### 2) Types — `src/types/knowledge-graph.ts`

```ts
export interface KnowledgeHub {
  id: string;
  label: string;
  name?: string;
  description: string;
  signalScore?: number;
  connections?: number;
  updatedAt?: string; // ISO
}

export interface KnowledgeProposal {
  id: string;
  title: string;
  signal: string;   // one-line actionable insight / impact
  action: string;   // e.g. "Review RI coverage"
  hubId: string;
  priority: number; // lower = higher priority
}

export interface TopHubPayload {
  hub: KnowledgeHub;
  proposals: KnowledgeProposal[];
}
```

### 3) Component — `src/components/TopHubSignalPanel/TopHubSignalPanel.tsx`

```tsx
import React, { useEffect, useState } from 'react';
import { TopHubPayload, KnowledgeHub, KnowledgeProposal } from '../../types/knowledge-graph';
import styles from './TopHubSignalPanel.module.css';

const DATA_URL = '/data/top-hub.json';

export const TopHubSignalPanel: React.FC = () => {
  const [payload, setPayload] = useState<TopHubPayload | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    let mounted = true;
    setLoading(true);
    fetch(DATA_URL, { cache: 'no-store' })
      .then((res) => {
        if (!res.ok) throw new Error(`Failed to load top-hub data (${res.status})`);
        return res.json() as Promise<TopHubPayload>;
      })
      .then((json) => {
        if (!mounted) return;
        const sorted = (json.proposals || []).sort((a, b) => a.priority - b.priority);
        setPayload({ hub: json.hub, proposals: sorted.slice(0, 3) });
        setError(null);
      })
      .catch((err) => {
        if (!mounted) return;
        setError(err?.message || 'Unknown error');
        setPayload(null);
      })
      .finally(() => mounted && setLoading(false));

    return () => {
      mounted = false;
    };
  }, []);

  const hub = payload?.hub;
  const proposals = payload?.proposals || [];

  if (loading) {
    return (
      <section className={styles.panel} aria-busy="true" aria-label="Loading top hub signals">
        <div className={styles.headerSkeleton} />
        <div className={styles.descSkeleton} />
        <div className={styles.listSkeleton}>
          {Array.from({ length: 3 }).map((_, i) => (
            <div key={i} className={styles.proposalSkeleton} />
          ))}
        </div>
      </section>
    );
  }

  if (error || !hub) {
    return (
      <section className={styles.panel} role="status" aria-label="Top hub signals unavailable">
        <p className={styles.muted}>Unable to load top-hub signals.</p>
        <small>{error}</small>
      </section>
    );
  }

  return (
    <section className={styles.panel} aria-label={`Top hub: ${hub.label}`}>
      <div className={styles.header}>
        <div>
          <span className={styles.badge}>{hub.label}</span>
          <span className={styles.name}>{hub.name}</span>
        </div>
        {typeof hub.signalScore === 'number' && (
          <span className={styles.score}>{hub.signalScore.toFixed(1)} pts</span>
        )}
      </div>

      <p className={styles.description}>{hub.description}</p>

      <div className={styles.meta} aria-hidden="false">
        <small>
          {hub.connections != null ? `${hub.connections} connections` : ''}
          {hub.updatedAt ? ` • Updated ${new Date(hub.updatedAt).toLocaleDateString()}` : ''}
        </small>
      </div>

      <div className={styles.proposals} role="list" aria-label="Top proposals">
        {proposals.map((p) => (
          <a
            key={p.id}
            href={`/proposal/${p.id}`}
            className={styles.proposalCard}
            role="listitem"
            tabIndex={0}
            aria-label={`${p.title}. ${p.signal}. Action: ${p.action}`}
          >
            <div className={styles.proposalTitle}>{p.title}</div>
            <div className={styles.proposalSignal}>{p.signal}</div>
            <div className={styles.proposalAction}>{p.action}</div>
          </a>
        ))}
      </div>
    </section>
  );
};
```
