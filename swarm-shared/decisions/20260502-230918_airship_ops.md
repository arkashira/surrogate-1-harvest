# airship / ops

## Final Implementation Plan — <2h Quality Increment

**Highest-value improvement**: Harden `airship discover` into a **deterministic, CDN-safe, zero-runtime orchestrator** that eliminates HF API rate limits and PyArrow schema errors while keeping Mac=CLI.

**Core strategy** (resolved from swarm decisions):
- Single API call (after rate-limit window) → `list_repo_tree(path, recursive=False)` for one date folder.
- Save file list to `manifests/{date}/files.json`.
- Embed that list in training scripts so Lightning training does **CDN-only fetches** (`https://huggingface.co/datasets/{repo}/resolve/main/{path}`) with zero API calls during data load.
- Project each file to `{prompt, response}` only at parse time; never rely on `load_dataset(streaming=True)` for heterogeneous repos.
- Deterministic repo selection for writes: hash slug → pick sibling repo (0–5) to stay under 128 commits/hr/repo.
- Mac runs only orchestration; training → Lightning Studio; ingestion → HF Space; LLM burst → Cerebras/Groq.

def sibling_repo_for(slug: str) -> str:
    """Deterministic repo selection to spread HF commit cap."""
    digest = hashlib.sha256(slug.encode()).hexdigest()
    idx = int(digest, 16) % len(SIBLING_REPOS)
    return SIBLING_REPOS[idx]

def list_date_folder(date_str: str, folder: str = "") -> list[dict]:
    """
    Single API call: non-recursive folder listing.
    Returns list of dicts with 'path' and optional 'size'.
    """
    try:
        tree = api.list_repo_tree(
            repo_id=HF_REPO,
            path=folder.strip("/"),
            repo_type="dataset",
            recursive=False,
        )
        items = [{"path": t.path, "type": getattr(t, "type", "file")} for t in tree]
        return [i for i in items if i["type"] == "file"]
    except Exception as exc:
        print(f"[discover] HF API error: {exc}", file=sys.stderr)
        raise

def cdn_url_for(repo: str, path: str) -> str:
    """CDN URL that bypasses HF API auth/rate limits."""
    return f"https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def safe_hf_download(path: str, out_dir: Path) -> Path:
    """Download one file via hf_hub_download (uses CDN under the hood)."""
    out_dir.mkdir(parents=True, exist_ok=True)
    return hf_hub_download(
        repo_id=HF_REPO,
        repo_type="dataset",
        filename=path,
        local_dir=out_dir,
        token=HF_TOKEN if HF_TOKEN else None,
    )

def project_to_prompt_response(file_path: Path) -> list[dict]:
    """
    Parse one file and project to {prompt, response}.
    Implement per-format adapters here (jsonl, parquet, csv, etc).
    """
    suffix = file_path.suffix.lower()
    records = []

    if suffix == ".jsonl":
        import json as jsonlib
        with file_path.open() as f:
            for line in f:
                line = line.strip()
                if not line:
                    continue
                obj = jsonlib.loads(line)
                prompt = obj.get("prompt") or obj.get("input") or ""
                response = obj.get("response") or obj.get("output") or obj.get("completion") or ""
                if prompt and response:
                    records.append({"prompt": prompt, "response": response})

    elif suffix == ".parquet":
        import pyarrow.parquet as pq
        tbl = pq.read_table(file_path)
        df = tbl.to_pandas()
        # Best-effort projection; do not trust schema
        prompt_col = next((c for c in df.columns if "prompt" in c.lower()), None)
        response_col = next((c for c in df.columns if "response" in c.lower() or "completion" in c.lower()), None)
        if prompt_col and response_col:
            for _, row in df.iterrows():
                p = str(row[prompt_col]).strip()
                r = str(row[response_col]).strip()
                if p and r:
                    records.append({"prompt": p, "response": r})

    else:
        # fallback: skip
        print(f"[discover] Skipping unsupported format: {file_path}", file=sys.stderr)

    return records
