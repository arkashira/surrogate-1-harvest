# airship / design

Below is the **single, merged implementation** that keeps the fastest, safest, and most actionable parts from both proposals and resolves conflicts in favor of **correctness + concrete actionability**.

Key decisions:
- Use **CDN-only orchestration** (eliminates HF API rate limits, no model/infra changes).
- Keep discovery **deterministic and reproducible** (sorted file list, single non-recursive tree call).
- Use **httpx + tenacity** for robust CDN downloads (retry/backoff, `Retry-After`, timeouts).
- Apply **strict schema projection** (`prompt`, `response` only) at parse time to prevent `pyarrow` schema errors.
- Default to `--cdn-only` in production; keep interface minimal.

## File-by-File Changes

### 1) `airship/scripts/discover_cdn.py`
Deterministic, CDN-only discovery. One non-recursive `list_repo_tree` call. Produces reproducible `file_list.json`.

```python
#!/usr/bin/env python3
"""
CDN-only discovery for surrogate ingestion.
Produces file_list.json with CDN URLs to bypass HF API rate limits.
"""
import json
import os
import sys
from pathlib import Path
from typing import List, Dict

from huggingface_hub import HfApi

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def discover_cdn(repo: str, date_folder: str, output_path: Path) -> List[Dict[str, str]]:
    api = HfApi()
    try:
        # Single non-recursive call: 1 API request per date folder
        items = api.list_repo_tree(
            repo_id=repo,
            path=date_folder,
            repo_type="dataset",
            recursive=False,
        )
    except Exception as e:
        # Idempotent: missing folder -> empty list
        print(f"Warning: could not list {repo}/{date_folder}: {e}", file=sys.stderr)
        items = []

    files: List[Dict[str, str]] = []
    for item in items:
        if not item.path.endswith(".parquet"):
            continue
        files.append({
            "cdn_url": CDN_TEMPLATE.format(repo=repo, path=item.path),
            "hf_path": item.path,
            "basename": os.path.basename(item.path),
        })

    # Deterministic/reproducible ordering
    files.sort(key=lambda f: f["hf_path"])

    output_path.write_text(json.dumps(files, indent=2))
    print(f"Wrote {len(files)} files to {output_path}")
    return files

if __name__ == "__main__":
    if len(sys.argv) != 4:
        print("Usage: discover_cdn.py <repo> <date_folder> <output.json>")
        sys.exit(1)
    repo, date_folder, output = sys.argv[1], sys.argv[2], Path(sys.argv[3])
    discover_cdn(repo, date_folder, output)
```

### 2) `airship/cli/discover.py` (patch)
Add `--cdn-only` flag (default `True`). Generate `file_list.json` and pass path to downstream ingestion via env.

```python
import os
import subprocess
import sys
from pathlib import Path

import click

@click.command()
@click.option("--repo", default="my-org/surrogate-data")
@click.option("--date", default="2026-04-29")
@click.option("--cdn-only", is_flag=True, default=True, help="Use CDN-only discovery to bypass HF API limits.")
def discover(repo: str, date: str, cdn_only: bool) -> None:
    if cdn_only:
        output = Path("file_list.json").absolute()
        script = Path(__file__).parent.parent / "scripts" / "discover_cdn.py"
        result = subprocess.run(
            [sys.executable, str(script), repo, date, str(output)],
            check=False,
        )
        if result.returncode != 0:
            sys.exit(result.returncode)

        # Pass to surrogate ingestion
        os.environ["SURROGATE_FILE_LIST"] = str(output)
        print(f"Set SURROGATE_FILE_LIST={output}")

### 3) `surrogate/lib/cdn_download.py`
Robust CDN downloads with retries, backoff, `Retry-After`, and timeouts.

```python
from pathlib import Path
import httpx
from tenacity import (
    retry,
    stop_after_attempt,
    wait_exponential,
    retry_if_exception_type,
)

@retry(
    stop=stop_after_attempt(5),
    wait=wait_exponential(multiplier=1.5, min=2, max=60),
    retry=retry_if_exception_type((httpx.TimeoutException, httpx.HTTPStatusError)),
)
def download_cdn_file(cdn_url: str, dest: Path, timeout: float = 30.0) -> Path:
    """
    Download a single file from CDN with robust retry/backoff.
    Raises on non-retryable HTTP errors.
    """
    dest.parent.mkdir(parents=True, exist_ok=True)

    with httpx.stream(
        "GET",
        cdn_url,
        timeout=timeout,
        follow_redirects=True,
    ) as response:
        # Let tenacity retry on 5xx and 429 (HTTPStatusError)
        response.raise_for_status()

        with dest.open("wb") as f:
            for chunk in response.iter_bytes(chunk_size=8192):
                f.write(chunk)

    return dest
```

### 4) `surrogate/ingest/project_schema.py`
Strict schema projection to avoid mixed-schema / cast errors. Keeps only `prompt` and `response`.

```python
from pathlib import Path
import pyarrow.parquet as pq
import pyarrow as pa
import pandas as pd

REQUIRED_FIELDS = {"prompt", "response"}

def project_file(in_path: Path, out_path: Path) -> Path:
    """
    Read parquet, keep only {prompt,response}, write back.
    Raises if required fields are missing.
    """
    table = pq.read_table(in_path)
    names = set(table.schema.names)

    missing = REQUIRED_FIELDS - names
    if missing:
        raise ValueError(f"Missing required fields {missing} in {in_path}")

    # Keep only required fields, in fixed order
    table = table.select(sorted(REQUIRED_FIELDS))

    # Ensure consistent types (coerce to utf8 if needed)
    schema = pa.schema([
        pa.field("prompt", pa.string()),
        pa.field("response", pa.string()),
    ])
    table = table.cast(schema)

    pq.write_table(table, out_path)
    return out_path

def project_df(df: pd.DataFrame) -> pd.DataFrame:
    """
    Project pandas DataFrame to required fields with safe casting.
    """
    missing = REQUIRED_FIELDS - set(df.columns)
    if missing:
        raise ValueError(f"Missing required fields {missing}")

    df
