# airship / design — highest-value 2h increment

## Chosen improvement
Replace HF API–driven dataset ingestion in the surrogate training pipeline with a **manifest-driven, CDN-only iterable loader** (generated once on the Mac orchestrator) to eliminate 429 rate limits and mixed-schema hazards.

- **Why this is highest value**:  
  - Eliminates HF API 429 during training (the most common failure).  
  - Removes recursive `list_repo_tree` calls and per-file API overhead.  
  - Enables reliable multi-hour Lightning Studio runs with zero API overhead.  
  - Handles mixed schemas safely (project only at parse time).  
  - Fits within ~2h (small, focused change + tests).

### 1) Manifest generator (run on Mac orchestrator)

```bash
#!/usr/bin/env bash
# generate_manifest.sh
# Usage: ./generate_manifest.sh <repo> <date_folder> <out_json>
# Example: ./generate_manifest.sh axentx/datasets 2026-05-03 file_list.json

set -euo pipefail
REPO="${1:?repo required}"
DATE_FOLDER="${2:?date_folder required}"
OUT="${3:?out_json required}"

python3 -c "
import json, os, sys
from huggingface_hub import HfApi
api = HfApi()
# non-recursive per folder to avoid pagination + rate limits
tree = api.list_repo_tree(repo_id='$REPO', path='$DATE_FOLDER', recursive=False)
files = [t.rfilename for t in tree if not t.rfilename.endswith('/')]
manifest = {
    'repo': '$REPO',
    'date_folder': '$DATE_FOLDER',
    'files': files,
    'generated_by': 'generate_manifest.sh'
}
with open('$OUT', 'w') as f:
    json.dump(manifest, f, indent=2)
print(f'Wrote {len(files)} files to $OUT')
"
```

Make executable:

```bash
chmod +x generate_manifest.sh
```

### 2) CDN-only iterable dataset wrapper (`surrogate/data/hf_cdn_dataset.py`)

```python
# surrogate/data/hf_cdn_dataset.py
from __future__ import annotations

import json
import logging
from pathlib import Path
from typing import Dict, Iterator, Optional

import pyarrow as pa
import pyarrow.compute as pc
import requests
from tqdm import tqdm

logger = logging.getLogger(__name__)

class HFCDNDataset:
    """
    CDN-only dataset loader that bypasses HF API auth/rate limits.
    Manifest format:
    {
      "repo": "axentx/datasets",
      "date_folder": "2026-05-03",
      "files": ["batches/mirror-merged/2026-05-03/slug1.parquet", ...]
    }
    """

    CDN_BASE = "https://huggingface.co/datasets"

    def __init__(
        self,
        manifest_path: str | Path,
        prompt_field: str = "prompt",
        response_field: str = "response",
        cache_dir: Optional[str | Path] = None,
    ) -> None:
        manifest_path = Path(manifest_path)
        with manifest_path.open() as f:
            self.manifest = json.load(f)

        self.repo = self.manifest["repo"]
        self.date_folder = self.manifest["date_folder"]
        self.files = self.manifest["files"]
        self.prompt_field = prompt_field
        self.response_field = response_field
        self.cache_dir = Path(cache_dir) if cache_dir else Path(".hf_cdn_cache")
        self.cache_dir.mkdir(parents=True, exist_ok=True)

    def _cdn_url(self, file_path: str) -> str:
        # file_path is relative to repo root
        return f"{self.CDN_BASE}/{self.repo}/resolve/main/{file_path}"

    def _cached_path(self, file_path: str) -> Path:
        safe = file_path.replace("/", "_")
        return self.cache_dir / safe

    def _download_file(self, file_path: str) -> Path:
        out = self._cached_path(file_path)
        if out.is_file():
            return out

        url = self._cdn_url(file_path)
        logger.info("Downloading (CDN): %s", url)
        resp = requests.get(url, timeout=60)
        resp.raise_for_status()
        out.write_bytes(resp.content)
        return out

    def _project_to_pair(self, table: pa.Table) -> pa.Table:
        # Keep only prompt/response; cast safely
        available = set(table.column_names)
        if self.prompt_field not in available or self.response_field not in available:
            raise ValueError(
                f"Required fields missing. Available: {available}, "
                f"need: {self.prompt_field}, {self.response_field}"
            )

        # Select and ensure string
        selected = table.select([self.prompt_field, self.response_field])
        # Cast to string if needed (handles mixed schemas)
        for col in selected.column_names:
            col_type = selected.schema.field(col).type
            if not pa.types.is_string(col_type):
                selected = selected.set_column(
                    selected.column_names.index(col),
                    col,
                    pc.cast(selected[col], pa.string()),
                )
        return selected

    def __iter__(self) -> Iterator[Dict[str, str]]:
        for file_path in tqdm(self.files, desc="Loading CDN files"):
            local_path = self._download_file(file_path)
            try:
                table = pq.read_table(local_path)
                pair_table = self._project_to_pair(table)
                for i in range(pair_table.num_rows):
                    row = pair_table.slice(i, 1).to_pydict()
                    yield {
                        "prompt": row[self.prompt_field][0],
                        "response": row[self.response_field][0],
                    }
            except Exception as exc:
                logger.warning("Failed to process %s: %s", file_path, exc)
                continue

    def __len__(self) -> int:
        # Approximate; could be expensive to compute exactly
        return len(self.files)
```

(Requires `pyarrow`, `requests`, `tqdm`; add `import pyarrow.parquet as pq` at top.)
