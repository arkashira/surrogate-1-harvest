# vanguard / design

## 1. Diagnosis
- No persisted `(repo, dateFolder)` file manifest — every training run re-enumerates via authenticated HF API, burning quota and risking 429.
- Likely recursive `list_repo_files` or `load_dataset(streaming=True)` on heterogeneous repo schema triggers `pyarrow.CastError` and amplifies rate-limit.
- Training script couples data enumeration with training loop; prevents CDN-only fetches and prevents Lightning Studio reuse.
- No idempotent Lightning Studio lifecycle (creates new studio per run; wastes 80hr/mo quota; idle-stop kills training).
- Missing deterministic file-list artifact that can be committed/versioned and embedded so Lightning training uses CDN-only downloads (zero API calls).

## 2. Proposed change
Create `/opt/axentx/vanguard/training/manifest.py` + `/opt/axentx/vanguard/training/train.py` (or patch existing) so:
- `manifest.py` provides `build_manifest(repo, date_folder, out_json)` that uses a single `list_repo_tree(recursive=False)` per date folder and saves `{"repo": repo, "date": date_folder, "files": [...]}`.
- `train.py` accepts `--manifest` and uses only CDN URLs (`https://huggingface.co/datasets/{repo}/resolve/main/{f}`) via `hf_hub_download` or direct `requests`/`datasets` with `data_files` pointing to local manifest.
- Add idempotent Lightning Studio reuse: `get_or_create_studio(name, machine=L40S)` that lists existing running studios and reuses; falls back to create.

```bash
# /opt/axentx/vanguard/training/manifest.py
#!/usr/bin/env python3
"""
Build a deterministic file manifest for (repo, date_folder) to avoid
authenticated HF API calls during training.
"""
import argparse
import json
import os
import sys
from pathlib import Path
from typing import List, Dict, Any

try:
    from huggingface_hub import list_repo_tree, HfApi
except ImportError:
    print("Install: pip install huggingface_hub")
    sys.exit(1)

HF_API = HfApi()

def build_manifest(repo: str, date_folder: str, out_path: str) -> Dict[str, Any]:
    """
    Build manifest for repo/date_folder.
    Uses list_repo_tree per folder (non-recursive) to avoid recursive pagination.
    """
    # list top-level of date_folder (non-recursive)
    tree = list_repo_tree(repo=repo, path=date_folder, recursive=False)
    files = []
    for entry in tree:
        if entry.type == "file":
            # store repo-relative path
            files.append(entry.path)

    manifest = {
        "repo": repo,
        "date": date_folder,
        "files": sorted(files),
        "count": len(files),
        "cdn_prefix": f"https://huggingface.co/datasets/{repo}/resolve/main"
    }

    out_path = Path(out_path)
    out_path.parent.mkdir(parents=True, exist_ok=True)
    with open(out_path, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2)

    print(f"Manifest written to {out_path} ({len(files)} files)")
    return manifest

def main() -> None:
    parser = argparse.ArgumentParser(description="Build HF dataset manifest")
    parser.add_argument("--repo", required=True, help="HF dataset repo (user/ds)")
    parser.add_argument("--date", required=True, help="Date folder (e.g. 2026-04-29)")
    parser.add_argument("--out", default="manifest.json", help="Output JSON path")
    args = parser.parse_args()

    build_manifest(args.repo, args.date, args.out)

if __name__ == "__main__":
    main()
```

```python
#!/usr/bin/env python3
"""
Idempotent training entrypoint that uses a prebuilt manifest and
reuses existing Lightning Studio when possible.
"""
import argparse
import json
import os
import sys
from pathlib import Path
from typing import List

try:
    import lightning as L
    from lightning.fabric.plugins import LightningCLI
    from huggingface_hub import hf_hub_download
except ImportError:
    print("Install: pip install lightning huggingface_hub")
    sys.exit(1)

def load_manifest(manifest_path: str) -> dict:
    with open(manifest_path, "r", encoding="utf-8") as f:
        return json.load(f)

def cdn_local_paths(manifest: dict, cache_dir: str = ".cache") -> List[str]:
    """
    Download files via CDN (no auth API calls) and return local paths.
    Uses hf_hub_download which resolves to CDN; no streaming dataset.
    """
    os.makedirs(cache_dir, exist_ok=True)
    paths = []
    for rel in manifest["files"]:
        local = hf_hub_download(
            repo_id=manifest["repo"],
            filename=rel,
            cache_dir=cache_dir,
            force_download=False,
            resume_download=True,
        )
        paths.append(local)
    return paths

def get_or_create_studio(name: str, machine: str = "L40S"):
    """
    Reuse existing running studio if present; otherwise create.
    """
    teamspace = L.Teamspace()
    for s in teamspace.studios:
        if s.name == name and s.status == "Running":
            print(f"Reusing running studio: {name}")
            return s
    print(f"Creating studio: {name} on {machine}")
    # Note: adjust create params as needed for your org
    return teamspace.create_studio(
        name=name,
        machine=machine,
        create_ok=True,
    )

    return

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--manifest", required=True, help="Path to manifest.json")
    parser.add_argument("--studio", default="vanguard-train", help="Lightning Studio name")
    parser.add_argument("--machine", default="L40S", help="Machine type for studio")
    parser.add_argument("--cache", default=".cache", help="HF cache dir")
    args = parser.parse_args()

    manifest = load_manifest(args.manifest)
    print(f"Loaded manifest: {manifest['repo']} {manifest['date']} ({manifest['count']} files)")

    # Reuse studio (idempotent)
    studio = get_or_create_studio(name=args.studio, machine=args.machine)

    # Download via CDN (zero authenticated API calls during data load)

    # If studio stopped, restart before run

        print("Studio not running; restarting...")

    # Execute training on studio (or locally if not using remote)
    # For remote execution via Lightning:
    #   trainer = L.Trainer(...)
    #   trainer.run(...)
    # Here we run training step locally for simplicity.
    train_step(paths)

if __name__ == "__main__":
    main()
```
