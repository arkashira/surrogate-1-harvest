#### 1. HF CDN bypass dataset loader  
`/opt/axentx/airship/surrogate/data/hf_cdn_loader.py`
```python
import json
import os
from pathlib import Path
from typing import Dict, List, Iterator
import pyarrow as pa
import pyarrow.parquet as pq
import pyarrow.json as pj
import requests
from tqdm import tqdm

HF_CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def load_hf_manifest(manifest_path: str) -> List[Dict]:
    with open(manifest_path) as f:
        return json.load(f)

def cdn_fetch(repo: str, path: str, out_path: Path) -> Path:
    url = HF_CDN_TEMPLATE.format(repo=repo, path=path)
    out_path.parent.mkdir(parents=True, exist_ok=True)
    if out_path.exists():
        return out_path
    resp = requests.get(url, timeout=60, stream=True)
    resp.raise_for_status()
    with open(out_path, "wb") as f:
        for chunk in resp.iter_content(chunk_size=8192):
            f.write(chunk)
    return out_path

def project_to_prompt_response(file_path: Path) -> Iterator[Dict[str, str]]:
    """Project heterogeneous parquet/json to {prompt,response} only."""
    try:
        table = pq.read_table(file_path)
    except pa.ArrowInvalid:
        try:
            table = pj.read_json(file_path)
        except Exception:
            return

    cols = {c.lower(): c for c in table.column_names}
    prompt_col = cols.get("prompt") or cols.get("instruction") or cols.get("input")
    response_col = cols.get("response") or cols.get("output") or cols.get("completion")

    if not prompt_col or not response_col:
        return

    df = table.select([prompt_col, response_col]).to_pandas()
    df.columns = ["prompt", "response"]
    for _, row in df.iterrows():
        if row["prompt"] and row["response"]:
            yield {"prompt": str(row["prompt"]), "response": str(row["response"])}

def build_cdn_dataset(manifest_path: str, cache_dir: str = ".hf_cache") -> List[Dict]:
    manifest = load_hf_manifest(manifest_path)
    repo = manifest[0]["repo"] if manifest and "repo" in manifest[0] else None
    if not repo:
        raise ValueError("Manifest must include 'repo' field in entries or top-level.")

    cache_root = Path(cache_dir)
    samples = []
    for entry in tqdm(manifest, desc="Fetching CDN files"):
        path = entry["path"]
        cached = cdn_fetch(repo, path, cache_root / path.replace("/", "_"))
        for item in project_to_prompt_response(cached):
            samples.append(item)
    return samples
```

#### 2. Manifest builder  
`/opt/axentx/airship/surrogate/scripts/build_hf_manifest.py`
```python
#!/usr/bin/env python3
"""
Build HF manifest for a date folder (avoids recursive list_repo_files 429).
Run from Mac after rate-limit window clears.
"""
import argparse
import json
import os
from huggingface_hub import HfApi

def list_folder_tree(repo_id: str, folder: str) -> list:
    api = HfApi()
    items = []
    # non-recursive per folder to avoid pagination explosion
    for item in api.list_repo_tree(repo_id=repo_id, path=folder, recursive=False):
        if item.type == "file":
            items.append({"path": item.path, "repo": repo_id})
        elif item.type == "folder":
            subpath = os.path.join(folder, item.path)
            items.extend(list_folder_tree(repo_id, subpath))
    return items

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True, help="e.g., axentx/surrogate-dataset")
    parser.add_argument("--date", required=True, help="date folder, e.g. 2026-04-29")
    parser.add_argument("--out", default="hf_manifest.json")
    args = parser.parse_args()

    folder = args.date
    items = list_folder_tree(args.repo, folder)
    with open(args.out, "w") as f:
        json.dump(items, f, indent=2)
    print(f"Wrote {len(items)} files to {args.out}")

if __name__ == "__main__":
    main()
```
