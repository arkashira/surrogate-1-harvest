# vanguard / design

## Final Synthesized Design  
*(Best parts merged; contradictions resolved in favor of correctness + concrete actionability)*

### 1) Diagnosis (merged)
- **No persisted manifest**: every run performs `list_repo_tree`, burning HF API quota and risking 429s.
- **Mixed-schema loads**: training uses `load_dataset(streaming=True)` or per-file loads across heterogeneous repos → `pyarrow.CastError`.
- **Authenticated API during training**: bypasses CDN-only URLs; should use `https://huggingface.co/datasets/{repo}/resolve/main/...` to avoid auth/rate limits.
- **Single-repo ingestion**: hits HF’s 128 commits/hour cap; needs deterministic repo-sharding across sibling repos.
- **Lightning Studio reuse + idle-stop resilience**: recreation wastes quota; idle-stop kills long jobs.

### 2) Proposed change (merged + corrected)
Create two files under `/opt/axentx/vanguard/design/` (liftable to `src/` later):

1. `/opt/axentx/vanguard/design/data_manifest.py`  
   - Single `list_repo_tree` per `(repo, dateFolder)` from orchestrator.  
   - Persist `manifest_{repo}_{date}.json` with file list and CDN prefix.  
   - Deterministic shard selection: `repo_index = hash(slug) % N` across sibling repos.  
   - Optional dry-run mode when `huggingface_hub` unavailable.

2. `/opt/axentx/vanguard/design/train_cdn.py`  
   - Lightning-compatible training entry.  
   - Data loader uses CDN-only URLs (no auth) and projects `{prompt, response}` at parse time to avoid mixed-schema errors.  
   - Resilient shard skipping, worker-aware sharding, and configurable max files for fast iteration.  
   - Launcher wrapper (outside these files) handles Studio reuse + idle-stop check before `.run()`.

No changes to existing files.

### `/opt/axentx/vanguard/design/data_manifest.py`
```python
#!/usr/bin/env python3
"""
Generate and persist (repo, dateFolder) -> file-list manifest for HF datasets.
Run once per dateFolder from orchestrator (Mac) after rate-limit window clears.
"""
import json
import hashlib
import os
import sys
from pathlib import Path
from typing import List, Dict, Any, Optional

try:
    from huggingface_hub import list_repo_tree, HfApi
    HF_AVAILABLE = True
except Exception:  # noqa: BLE001
    HF_AVAILABLE = False
    list_repo_tree = None  # type: ignore[assignment]
    HfApi = None  # type: ignore[assignment]


def list_date_files(
    repo: str,
    date_folder: str,
    recursive: bool = False,
    token: Optional[str] = None,
) -> List[str]:
    """
    Single call to list files in date_folder.
    Returns paths like 'batches/mirror-merged/2026-04-28/slug.parquet'.
    """
    if not HF_AVAILABLE or list_repo_tree is None:
        raise RuntimeError(
            "huggingface_hub not available. Install with: pip install huggingface_hub"
        )

    try:
        tree = list_repo_tree(repo_id=repo, path=date_folder, recursive=recursive, token=token)
    except Exception as exc:
        raise RuntimeError(f"Failed to list {repo}/{date_folder}: {exc}") from exc

    files = [item.path for item in tree if getattr(item, "type", None) == "file"]
    return sorted(files)


def build_manifest(
    repo: str,
    date_folder: str,
    out_dir: str = ".",
    recursive: bool = False,
    token: Optional[str] = None,
) -> Dict[str, Any]:
    """Build manifest and write to disk."""
    files = list_date_files(repo, date_folder, recursive=recursive, token=token)

    manifest = {
        "repo": repo,
        "date_folder": date_folder,
        "files": files,
        "n_files": len(files),
        "cdn_prefix": f"https://huggingface.co/datasets/{repo}/resolve/main",
    }

    os.makedirs(out_dir, exist_ok=True)
    safe_name = repo.replace("/", "_")
    out_path = Path(out_dir) / f"manifest_{safe_name}_{date_folder}.json"
    out_path.write_text(json.dumps(manifest, indent=2))
    return manifest


def pick_shard_repo(slug: str, sibling_repos: List[str]) -> str:
    """
    Deterministic shard selection to avoid HF commit cap.
    sibling_repos should be full repo ids, e.g. ['org/vanguard-en-01', ..., 'org/vanguard-en-05']
    """
    if not sibling_repos:
        raise ValueError("sibling_repos required")
    h = hashlib.sha256(slug.encode()).hexdigest()
    idx = int(h, 16) % len(sibling_repos)
    return sibling_repos[idx]


if __name__ == "__main__":
    import argparse

    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True, help="HF dataset repo id")
    parser.add_argument("--date", required=True, help="dateFolder path inside repo")
    parser.add_argument("--out", default=".", help="output directory")
    parser.add_argument("--recursive", action="store_true", help="list recursively")
    parser.add_argument("--token", default=None, help="HF token (optional)")
    args = parser.parse_args()

    try:
        manifest = build_manifest(
            repo=args.repo,
            date_folder=args.date,
            out_dir=args.out,
            recursive=args.recursive,
            token=args.token,
        )
        print(f"Wrote manifest with {manifest['n_files']} files.")
    except Exception as exc:
        print(f"Error: {exc}", file=sys.stderr)
        sys.exit(1)
```

#!/usr/bin/env python3
"""
Lightning-compatible training entry that uses CDN-only fetches and avoids
load_dataset(streaming=True) on mixed-schema repos.

Usage:
- Embed manifest JSON produced by data_manifest.py.
- Lightning Studio reuse + idle-stop resilience handled in launcher wrapper.
"""
import json
import os
import random
from pathlib import Path
from typing import Dict, List, Iterator, Tuple, Optional

import torch
from torch.utils.data import IterableDataset, DataLoader
import requests


class CDNParquetDataset(IterableDataset):
    """
    Stream parquet shards via CDN URLs and yield {prompt, response}.
    Projects only required columns at parse time to avoid mixed-schema errors.
    """

    def __init__(
        self,
        manifest_path: str,
        cdn_prefix: Optional[str] = None,
        max_files: Optional[int] = None,
        seed: int = 42,
    ):
        super().__init__()
        manifest = json.loads(Path(manifest_path).read_text())
        self.files: List[str] = manifest["files"]
        if max_files is not None:
            self.files = self.files[:max_files]
        self.cdn_prefix = cdn_prefix or manifest.get("cdn_prefix")
        if not self.cdn_prefix:
            raise ValueError("cdn_prefix required in manifest or arg")
        self.seed = seed

    def _fetch_shard(self, file_path: str) -> List[Tuple[str, str]]:
        """Download one parquet via CDN and extract (prompt, response)."""
        url = f"{self.cdn_prefix}/{file_path}"
        resp = requests.get(url, stream=True, timeout=60)
        resp.raise_for_status()

        import pyarrow.parquet as pq
        import io

        buf = io.BytesIO(resp.content)

        table = pq.read_table(buf, columns=["prompt", "response"])
        prompts = table["prompt"].to_pylist()
        responses = table["response"].to_pylist()
        return list(zip(prompts, responses))
