# workio / ops

## Final synthesized solution (correct + actionable)

**Core diagnosis (merged, de-duplicated)**
- LINE webhook at-least-once delivery + no idempotency key causes duplicate punches/leave/OT rows on retries/5xx/timeouts.
- Race condition in punch state transitions (`find-then-insert/update`) allows multiple open punches per user/tenant.
- No transactional boundary around punch state changes; partial failures leave inconsistent state.
- No DB-level guard for a single open punch and no fast deduplication for LINE event IDs.

**Chosen approach (DB-first, minimal infra)**
- Use PostgreSQL partial unique index to enforce **one open punch per (user, tenant)**.
- Use a **punch_idempotency** table keyed by LINE `deliveryId` (or client-provided idempotency key) to make event handling idempotent and avoid relying on unique violation side effects for duplicates.
- Wrap each event in a transaction; return existing punch for duplicates; fail fast on invalid transitions.
- Keep it simple: no Redis, no extra queue required for correctness (queue only if you need heavy async work).

### 2) Service implementation

```typescript
// workio/server/src/services/punchService.ts
import { pool } from '../db';
import { Row } from '../types';

export type PunchEvent = {
  lineDeliveryId: string;          // from LINE (or client-generated idempotency key)
  type: 'clock_in' | 'clock_out' | 'leave_request' | 'ot_request';
  userId: number;
  tenantId: number;
  timestamp: Date;
  location?: { lat: number; lng: number };
  metadata?: Record<string, unknown>;
};

export async function handlePunchEvent(event: PunchEvent) {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // Idempotency: return existing punch if already processed
    const idem = await client.query<Row>(
      `SELECT punch_id FROM punch_idempotency WHERE idempotency_key = $1`,
      [event.lineDeliveryId]
    );

    if (idem.rows.length > 0) {
      const punch = await client.query<Row>(
        `SELECT id, user_id, tenant_id, status, clock_in_at, clock_out_at
         FROM punches WHERE id = $1`,
        [idem.rows[0].punch_id]
      );
      await client.query('ROLLBACK');
      return { ok: true, duplicate: true, punch: punch.rows[0] };
    }

    if (event.type === 'clock_in') {
      const result = await client.query<Row>(
        `INSERT INTO punches (user_id, tenant_id, status, clock_in_at, location, line_delivery_id, metadata)
         VALUES ($1, $2, 'clocked_in', $3, $4, $5, $6)
         RETURNING id, user_id, tenant_id, status, clock_in_at`,
        [
          event.userId,
          event.tenantId,
          event.timestamp,
          event.location ? JSON.stringify(event.location) : null,
          event.lineDeliveryId,
          event.metadata ? JSON.stringify(event.metadata) : null,
        ]
      );

      await client.query<Row>(
        `INSERT INTO punch_idempotency (idempotency_key, punch_id) VALUES ($1, $2)`,
        [event.lineDeliveryId, result.rows[0].id]
      );

      await client.query('COMMIT');
      return { ok: true, punch: result.rows[0] };
    }

    if (event.type === 'clock_out') {
      // Close the open punch for this user/tenant
      const result = await client.query<Row>(
        `UPDATE punches
         SET status = 'clocked_out', clock_out_at = $1, location_out = $2, updated_at = NOW()
         WHERE user_id = $3 AND tenant_id = $4 AND status = 'clocked_in'
         RETURNING id, user_id, tenant_id, status, clock_in_at, clock_out_at`,
        [
          event.timestamp,
          event.location ? JSON.stringify(event.location) : null,
          event.userId,
          event.tenantId,
        ]
      );

      if (result.rows.length === 0) {
        await client.query('ROLLBACK');
        return { ok: false, error: 'no_open_punch' };
      }

      await client.query<Row>(
        `INSERT INTO punch_idempotency (idempotency_key, punch_id) VALUES ($1, $2)`,
        [event.lineDeliveryId, result.rows[0].id]
      );

      await client.query('COMMIT');
      return { ok: true, punch: result.rows[0] };
    }

    // leave/ot_request: similar pattern — idempotent insert/update with punch_idempotency
    await client.query('ROLLBACK');
    return { ok: false, error: 'unhandled_type' };
  } catch (err: any) {
    await client.query('ROLLBACK');

    // Unique violation on open punch or line_delivery_id => duplicate or conflict
    if (err.code === '23505') {
      // Try to return the conflicting punch for consistency
      const conflict = await client.query<Row>(
        `SELECT p.id, p.user_id, p.tenant_id, p.status, p.clock_in_at
         FROM punches p
         WHERE p.line_delivery_id = $1
         UNION ALL
         SELECT p.id, p.user_id, p.tenant_id, p.status, p.clock_in_at
         FROM punch_idempotency pi
         JOIN punches p ON p.id = pi.punch_id
         WHERE pi.idempotency_key = $1
         LIMIT 1`,
        [event.lineDeliveryId]
      );
      if (conflict.rows.length > 0) {
        return { ok: true, duplicate: true, punch: conflict.rows[0] };
      }
      return { ok: false, error: 'conflict', detail: err.message };
    }

    throw err;
  } finally {
    client.release();
  }
}
```
