# vanguard / design

## Final Synthesis (single, correct, actionable)

**Core diagnosis (merged, de-duplicated):**  
- Training and ingestion re-enumerate repos at runtime via `list_repo_files`/`load_dataset`, causing HF API 429s and non-reproducible epochs.  
- No content-addressed manifest (file list + SHA256) per date folder → CDN-only fetches impossible and resume/restart unsafe.  
- Missing deterministic repo-to-sibling mapping for commit-cap scaling (128/hr/repo) → ingestion stalls on big pushes.  
- Scripts assume HF API availability during training; no local deterministic snapshot to enable CDN-only, zero-API training inside Lightning.  
- Mac orchestration layer lacks a small, versioned snapshot it can embed in `train.py`.

**Single contract (ingestion ⇄ training):**  
Ingestion (HF Space) produces one content-addressed manifest per date folder. Training (Lightning) consumes only that manifest and fetches files via CDN; it never calls `list_repo_files` or `load_dataset`.

## Implementation (merged best parts, fixed contradictions, fully actionable)

### 1) Manifest generator (`/opt/axentx/vanguard/manifest.py`)
- Uses HF API **once** (from Mac/CI) to list the date folder non-recursively.  
- Downloads each file to a local cache, records SHA256 and size.  
- Writes deterministic JSON; training uses this file as the single source of truth.  
- Supports `--max-workers` for faster ingestion and explicit repo mapping for commit-cap scaling.

```python
#!/usr/bin/env python3
"""
Generate a content-addressed manifest for a HF dataset repo/date folder.

Usage:
  python -m vanguard.manifest \
    --repo datasets/my-data \
    --date 2026-04-29 \
    --out manifest-2026-04-29.json \
    --max-workers 8

Produces:
  {
    "repo": "datasets/my-data",
    "date": "2026-04-29",
    "generated_at": "<ISO-8601-UTC>",
    "files": [
      {"path": "2026-04-29/file.parquet", "sha256": "...", "size": 12345}
    ],
    "cdn_base": "https://huggingface.co/datasets"
  }
"""

from __future__ import annotations

import argparse
import concurrent.futures
import hashlib
import json
import sys
from datetime import datetime, timezone
from pathlib import Path
from typing import Any, Dict, List

from huggingface_hub import HfApi, hf_hub_download

CDN_BASE = "https://huggingface.co/datasets"

def sha256_file(path: Path) -> str:
    h = hashlib.sha256()
    with path.open("rb") as f:
        for chunk in iter(lambda: f.read(128 * 1024), b""):
            h.update(chunk)
    return h.hexdigest()

def _download_one(
    repo: str,
    rfilename: str,
    cache_dir: Path,
) -> Dict[str, Any] | None:
    try:
        local = hf_hub_download(
            repo_id=repo,
            filename=rfilename,
            repo_type="dataset",
            cache_dir=str(cache_dir),
            force_download=False,
        )
        p = Path(local)
        if not p.is_file():
            return None
        return {
            "path": rfilename,
            "sha256": sha256_file(p),
            "size": p.stat().st_size,
        }
    except Exception as exc:
        print(f"WARNING: failed to download {repo}/{rfilename}: {exc}", file=sys.stderr)
        return None

def build_manifest(
    repo: str,
    date: str,
    out_path: Path,
    cache_dir: Path | None = None,
    max_workers: int = 8,
) -> Dict[str, Any]:
    api = HfApi()
    folder = date  # e.g. "2026-04-29"

    print(f"Listing repo={repo} folder={folder} (non-recursive)...")
    try:
        tree = api.list_repo_tree(repo=repo, path=folder, recursive=False)
    except Exception as exc:
        raise RuntimeError(
            f"Failed to list {repo}/{folder}. "
            "If rate-limited, retry after window or use a HF token with higher limits."
        ) from exc

    items = [t for t in tree if t.rfilename and not t.rfilename.endswith("/")]
    print(f"Found {len(items)} files in {repo}/{folder}")

    cache_dir = cache_dir or (Path.home() / ".cache" / "vanguard" / "manifest")
    cache_dir.mkdir(parents=True, exist_ok=True)

    files: List[Dict[str, Any]] = []
    with concurrent.futures.ThreadPoolExecutor(max_workers=max_workers) as ex:
        futures = {
            ex.submit(_download_one, repo, item.rfilename, cache_dir): item.rfilename
            for item in items
        }
        for fut in concurrent.futures.as_completed(futures):
            rfn = futures[fut]
            try:
                entry = fut.result()
                if entry:
                    files.append(entry)
                    print(f"  ✓ {rfn}")
                else:
                    print(f"  ✗ {rfn} (not a file)")
            except Exception as exc:
                print(f"  ✗ {rfn}: {exc}")

    # Deterministic ordering for reproducibility
    files.sort(key=lambda x: x["path"])

    manifest = {
        "repo": repo,
        "date": date,
        "generated_at": datetime.now(timezone.utc).isoformat(),
        "files": files,
        "cdn_base": CDN_BASE,
    }

    out_path.parent.mkdir(parents=True, exist_ok=True)
    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"Manifest written to {out_path} ({len(files)} files)")
    return manifest

def main() -> None:
    parser = argparse.ArgumentParser(description="Generate content-addressed manifest for HF dataset folder.")
    parser.add_argument("--repo", required=True, help="HF dataset repo, e.g. datasets/my-data")
    parser.add_argument("--date", required=True, help="Date folder, e.g. 2026-04-29")
    parser.add_argument("--out", required=True, help="Output manifest JSON path")
    parser.add_argument("--cache-dir", help="Local cache directory (default: ~/.cache/vanguard/manifest)")
    parser.add_argument("--max-workers", type=int, default=8, help="Parallel downloads (default: 8)")
    args = parser.parse_args()

    try:
        build_manifest(
            repo=args.repo,
            date=args.date,
            out_path=Path(args.out),
            cache_dir=Path(args.cache_dir) if args.cache_dir else None,
            max_workers=args.max_workers,
        )
    except Exception as e:
        print(f"Error: {e}", file=sys.stderr)
        sys.exit(1)

if __name__ == "__main__":
    main()
```

### 2) Training entrypoint (`/opt/axentx/vanguard/train.py`)
- Accepts `--manifest` and uses **CDN-only** fetches via `hf_hub_download`.  
- Never calls `list_repo_files` or `load_dataset` during training.  
- Supports deterministic resume by verifying SHA256 before use.  
- Lightweight, Lightning-friendly, and embeddable.
