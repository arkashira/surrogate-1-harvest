# vanguard / ops

## 1. Diagnosis

- Hash-router navigations do not restore scroll position on back/forward → disorienting jumps to top instead of restoring prior viewport.
- Dataset list state (page, pageSize, sort, dir, filters, selection) is not encoded in the URL → reloads and shared links lose context.
- No loading/error UI during async dataset fetch/navigation → users perceive stalls or broken state.
- Dataset detail view does not reflect URL params on mount (e.g., tab, section) → deep links behave inconsistently.
- Missing scroll-to-anchor handling for hash links (e.g., `#row-123`) → anchor navigations do not position target element.

## 2. Proposed change

File: `/opt/axentx/vanguard/src/lib/router.js` (or equivalent main router module)  
Scope: Add hash-scroll manager + URL-state sync for dataset list (page, pageSize, sort, dir, filters) and detail tabs. Add per-route loading boundary and scroll restoration hooks.

// /opt/axentx/vanguard/src/lib/router.js
// Minimal, framework-agnostic hash router with scroll + URL-state management.

(function () {
  const DEFAULTS = {
    page: 1,
    pageSize: 20,
    sort: 'updated',
    dir: 'desc',
    filters: '',
    selection: '',
    detailTab: 'overview'
  };

  // Utilities
  function parseHash() {
    const hash = location.hash.replace(/^#/, '');
    if (!hash) return { route: 'list', params: { ...DEFAULTS } };

    const [route, ...parts] = hash.split('?');
    const query = parts.join('?');
    const params = new URLSearchParams(query);

    const result = { route: route || 'list', params: { ...DEFAULTS } };

    // Coerce known params
    const num = (k, def) => { const v = parseInt(params.get(k), 10); return Number.isFinite(v) && v > 0 ? v : def; };
    const str = (k, def) => params.get(k) || def;

    if (result.route === 'list') {
      result.params.page = num('page', DEFAULTS.page);
      result.params.pageSize = num('pageSize', DEFAULTS.pageSize);
      result.params.sort = str('sort', DEFAULTS.sort);
      result.params.dir = str('dir', DEFAULTS.dir);
      result.params.filters = str('filters', DEFAULTS.filters);
      result.params.selection = str('selection', DEFAULTS.selection);
    } else if (result.route === 'detail') {
      // detail/<slug>?tab=overview&...
      const segs = route.split('/').filter(Boolean);
      result.slug = segs[1] || '';
      result.params.detailTab = str('tab', DEFAULTS.detailTab);
    }

    return result;
  }

  function buildHash(route, params) {
    const query = new URLSearchParams();
    Object.keys(params).forEach((k) => {
      const v = params[k];
      if (v == null || v === '' || (typeof v === 'number' && isNaN(v))) return;
      if (v === DEFAULTS[k]) return; // keep URLs clean
      query.set(k, String(v));
    });
    const qs = query.toString();
    return `#${route}${qs ? '?' + qs : ''}`;
  }

  // Scroll restoration
  let scrollPositions = new Map(); // routeKey -> y
  let isPopState = false;

  function saveScrollPosition(key) {
    scrollPositions.set(key, window.scrollY);
  }

  function restoreScrollPosition(key, immediate) {
    const y = scrollPositions.get(key) || 0;
    if (immediate) window.scrollTo(0, y);
    else window.scrollTo({ top: y, behavior: 'auto' });
  }

  function scrollToTop(immediate) {
    if (immediate) window.scrollTo(0, 0);
    else window.scrollTo({ top: 0, behavior: 'smooth' });
  }

  // Anchor handling (e.g., #row-123 within current page)
  function handleAnchors() {
    const hash = location.hash;
    if (!hash || hash.length <= 1) return false;
    try {
      const el = document.querySelector(hash);
      if (el) {
        el.scrollIntoView({ behavior: 'smooth', block: 'start' });
        return true;
      }
    } catch (e) {
      // invalid selector
    }
    return false;
  }

  // Route handlers (to be wired by app)
  let onList = null;
  let onDetail = null;
  let onLoading = null;
  let onError = null;

  // Core router
  async function route(hash, { replace = false, skipScroll = false } = {}) {
    if (hash && hash.charAt(0) !== '#') hash = '#' + hash;
    const prev = parseHash();
    const prevKey = `${prev.route}:${JSON.stringify(prev.params)}`;

    if (hash) {
      if (replace) history.replaceState(null, '', hash);
      else history.pushState(null, '', hash);
    }

    const state = parseHash();
    const key = `${state.route}:${JSON.stringify(state.params)}`;

    // Save previous scroll before leaving
    if (!isPopState) saveScrollPosition(prevKey);

    // Loading UI
    if (onLoading) onLoading(true);

    try {
      if (state.route === 'list') {
        if (onList) await onList(state.params);
      } else if (state.route === 'detail') {
        if (onDetail) await onDetail(state.slug, state.params);
      } else {
        // unknown route -> list
        return navigate(buildHash('list', DEFAULTS), { replace: true });
      }

      // Scroll handling
      if (!skipScroll) {
        if (isPopState) {
          // back/forward: restore saved position (or top)
          restoreScrollPosition(key, false);
        } else {
          // new navigation: top, unless same-route param-only change (keep position)
          if (prev.route === state.route && prev.slug === state.slug) {
            // same view, param change — don't forcibly scroll to top (preserve UX)
            // but ensure anchors are handled
            if (!handleAnchors()) {
              // small param change: do not scroll
            }
          } else {
            scrollToTop(false);
          }
        }
      }

      // If there's an explicit hash anchor after route, try to scroll to it
      if (!skipScroll && location.hash && location.hash.length > 1) {
        // If the hash looks like an element id (not route), try anchor
        const maybeId = location.hash.substring(1);
        if (!state.route.startsWith(maybeId) && !maybeId.includes('?')) {
          handleAnchors();
        }
      }
    } catch (err) {
      if (onError) onError(err);
      console.error('[router]', err);
    } finally {
      if (onLoading) onLoading(false);
    }
  }

  function navigate(hash, opts) {
    return route(hash, { replace: false, ...opts });
  }

  function replace(hash) {
    return route(hash, { replace: true });
  }

  function updateListParams(updates, { replace = false } = {}) {
    const current = parseHash();
    const params = { ...current.params, ...updates };
    const hash = buildHash('list', params);
    return route(hash, { replace });
  }

  function updateDetailParams(updates, { replace = false } = {}) {
    const current = parseHash();
    const params = { ...current.params, ...updates };
    const hash = buildHash(`detail/${current.slug || ''}`, params);
    return route(hash, { replace });
  }

  // Popstate (back/forward)
  window.addEventListener('popstate', () => {
    isPopState = true;
    route(location.hash, { skipScroll: false });
    isPopState = false;
  });

  // Public API
  window.Router = {
    init({ onList: ol, onDetail: od, onLoading: olng, onError: oe } = {}) {
      onList = ol;
      onDetail = od;
      onLoading = olng;
