Add a **CDN-only dataset loader** plus **Lightning idle-resilient runner** so training:
- Never calls the HF Hub API during training (no 429s).
- Uses a precomputed `filelist.json` (built once on your Mac) embedded in the repo.
- Automatically reuses a running Studio or restarts on idle-stop, preventing quota loss.

### `scripts/build_filelist.py`
```python
#!/usr/bin/env python3
"""
Run once on Mac (after HF API window clears) to list one date folder.
Output: surrogate/training/filelist.json
"""
import json
import os
from pathlib import Path

from huggingface_hub import HfApi

REPO = os.getenv("HF_DATASET_REPO", "myorg/surrogate-data")
DATE_FOLDER = os.getenv("DATE_FOLDER", "batches/mirror-merged/2026-05-03")
OUT_PATH = Path(__file__).parent.parent / "training" / "filelist.json"

def main() -> None:
    api = HfApi()
    entries = api.list_repo_tree(repo_id=REPO, path=DATE_FOLDER, recursive=False)
    files = sorted(e.rfilename for e in entries if e.rfilename.endswith(".parquet"))

    payload = {
        "repo": REPO,
        "date_folder": DATE_FOLDER,
        "files": files,
    }
    OUT_PATH.parent.mkdir(parents=True, exist_ok=True)
    OUT_PATH.write_text(json.dumps(payload, indent=2))
    print(f"Wrote {len(files)} files to {OUT_PATH}")

if __name__ == "__main__":
    main()
```

### `surrogate/training/cdn_dataset.py`
```python
from __future__ import annotations

import io
import json
from typing import Dict, Iterator

import pyarrow.parquet as pq
import requests
from torch.utils.data import IterableDataset

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

class CdnParquetDataset(IterableDataset):
    """
    CDN-only parquet loader.
    filelist_json format:
      {"repo": "...", "date_folder": "...", "files": ["f1.parquet", ...]}
    """

    def __init__(self, filelist_json: str, split: str | None = None) -> None:
        with open(filelist_json) as f:
            meta = json.load(f)
        self.repo = meta["repo"]
        self.base = meta["date_folder"]
        self.files = meta["files"]
        if split:
            # deterministic split by filename hash
            self.files = [f for f in self.files if hash(f) % 10 < (8 if split == "train" else 2)]

    def _stream_urls(self) -> Iterator[str]:
        for fname in self.files:
            path = f"{self.base}/{fname}"
            yield CDN_TEMPLATE.format(repo=self.repo, path=path)

    def __iter__(self) -> Iterator[Dict[str, object]]:
        for url in self._stream_urls():
            try:
                resp = requests.get(url, timeout=30)
                resp.raise_for_status()
            except Exception as exc:
                print(f"Failed to fetch {url}: {exc}")
                continue

            table = pq.read_table(pq.ParquetFile(pq.ParquetReader(pq.BufferReader(resp.content))))
            cols = set(table.column_names)
            prompt_col = next((c for c in ("prompt", "instruction", "input") if c in cols), None)
            response_col = next((c for c in ("response", "output", "completion") if c in cols), None)

            if prompt_col is None or response_col is None:
                continue

            prompts = table.column(prompt_col).to_pylist()
            responses = table.column(response_col).to_pylist()
            for p, r in zip(prompts, responses):
                if isinstance(p, str) and isinstance(r, str) and p.strip() and r.strip():
                    yield {"prompt": p.strip(), "response": r.strip()}
```

### `surrogate/training/run_lightning.py`
```python
#!/usr/bin/env python3
"""
Lightning Studio idle-resilient runner.
- Reuses running Studio if present.
- Restarts on idle-stop and re-runs training.
"""
import os
import time
from pathlib import Path

import lightning as L

REPO_ROOT = Path(__file__).parent.parent.parent
TRAIN_SCRIPT = str(REPO_ROOT / "surrogate" / "training" / "train.py")

def ensure_studio_running() -> L.Studio:
    # Try to find a running studio in the teamspace
    team = L.Team()
    studios = team.studios()
    running = [s for s in studios if s.status == "Running"]
    if running:
        print(f"Reusing running studio: {running[0].name}")
        return running[0]

    print("No running studio found. Starting new L40S Studio...")
    studio = L.Studio()
    studio.start(machine=L.Machine.L40S)
    return studio

def main() -> None:
    max_retries = 3
    for attempt in range(1, max_retries + 1):
        try:
            studio = ensure_studio_running()
            # Run training; Lightning will handle idle-stop internally
            studio.run(
                script=TRAIN_SCRIPT,
                env={
                    "HF_DATASET_REPO": os.getenv("HF_DATASET_REPO", "myorg/surrogate-data"),
                    "DATE_FOLDER": os.getenv("DATE_FOLDER", "batches/mirror-merged/2026-05-03"),
                },
            )
            print("Training completed successfully.")
            break
        except Exception as exc:
            print(f"Attempt {attempt} failed: {exc}")
            if attempt == max_retries:
                raise
            print("Restarting studio and retrying...")
            time.sleep(10)

if __name__ == "__main__":
    main()
```
