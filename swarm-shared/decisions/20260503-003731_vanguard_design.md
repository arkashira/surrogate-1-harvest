# Final Synthesized Solution

## Diagnosis (Consolidated)
- **No persisted manifest**: every training run re-enumerates via authenticated HF API → quota burn and 429 risk.
- **Data loader issues**: recursive `list_repo_files` or `load_dataset(streaming=True)` on heterogeneous repos → pyarrow schema errors and amplified rate-limit exposure.
- **Studio churn**: training script recreates Lightning Studio on each run → wastes quota on spin-up instead of reuse.
- **Mac misuse**: local `model.from_pretrained()` or heavy compute on Mac → violates Mac=CLI rule and burns local resources.
- **No CDN bypass**: all data fetches go through authenticated `/api/` endpoints instead of public CDN URLs.

## Proposed Change
Create `/opt/axentx/vanguard/manifest.py` and update training launcher to:
- **Single authenticated enumeration** per `(repo, dateFolder)` → save `manifest/{repo}/{date}.json` listing file paths only.
- **CDN-only data loading**: embed manifest path in Lightning training script; data loader uses `https://huggingface.co/datasets/.../resolve/main/...` with zero API/auth calls during training.
- **Studio reuse**: reuse a running Studio; start once, restart only if stopped.
- **Mac as orchestrator only**: no model loading or heavy compute on Mac.

### File: `/opt/axentx/vanguard/manifest.py`
```python
#!/usr/bin/env python3
"""
Generate and consume CDN-only file manifests for HF datasets.

Usage:
  python manifest.py build --repo <repo> --date <YYYY-MM-DD> --out-dir manifest/
  python manifest.py path  --repo <repo> --date <YYYY-MM-DD> --manifest-dir manifest/
"""
import argparse
import json
import sys
from pathlib import Path
from typing import List

from huggingface_hub import HfApi

CDN_BASE = "https://huggingface.co/datasets"


def build_manifest(repo: str, date_folder: str, out_dir: Path) -> Path:
    """
    Single authenticated API call: list top-level folder for date.
    Returns manifest path.
    """
    api = HfApi()
    out_dir = Path(out_dir)
    out_dir.mkdir(parents=True, exist_ok=True)

    try:
        items = api.list_repo_tree(
            repo=repo,
            path=date_folder,
            recursive=False,
            repo_type="dataset",
        )
    except Exception as e:
        print(f"HF list_repo_tree failed: {e}", file=sys.stderr)
        raise

    file_paths = [
        f"{date_folder}/{obj.path.split('/')[-1]}"
        for obj in items
        if obj.type == "file"
    ]
    file_paths.sort()

    safe_repo = repo.replace("/", "_")
    manifest_path = out_dir / f"{safe_repo}_{date_folder}.json"
    manifest_path.write_text(json.dumps(file_paths, indent=2))
    print(f"Manifest written: {manifest_path} ({len(file_paths)} files)")
    return manifest_path


def load_manifest(repo: str, date_folder: str, manifest_dir: Path) -> List[str]:
    safe_repo = repo.replace("/", "_")
    manifest_path = Path(manifest_dir) / f"{safe_repo}_{date_folder}.json"
    if not manifest_path.exists():
        raise FileNotFoundError(f"Manifest missing: {manifest_path}")
    return json.loads(manifest_path.read_text())


def cdn_urls(repo: str, file_paths: List[str]) -> List[str]:
    return [f"{CDN_BASE}/{repo}/resolve/main/{p}" for p in file_paths]


def main() -> None:
    parser = argparse.ArgumentParser(description="HF CDN manifest tool")
    sub = parser.add_subparsers(dest="cmd", required=True)

    build = sub.add_parser("build", help="Build manifest")
    build.add_argument("--repo", required=True, help="HF dataset repo (datasets/owner/name)")
    build.add_argument("--date", required=True, help="Date folder (YYYY-MM-DD)")
    build.add_argument("--out-dir", default="manifest", help="Output directory")

    path_cmd = sub.add_parser("path", help="Print manifest")
    path_cmd.add_argument("--repo", required=True)
    path_cmd.add_argument("--date", required=True)
    path_cmd.add_argument("--manifest-dir", default="manifest")

    args = parser.parse_args()

    if args.cmd == "build":
        build_manifest(args.repo, args.date, Path(args.out_dir))
    elif args.cmd == "path":
        try:
            files = load_manifest(args.repo, args.date, Path(args.manifest_dir))
            print(json.dumps(files, indent=2))
        except Exception as e:
            print(f"Error: {e}", file=sys.stderr)
            sys.exit(1)


if __name__ == "__main__":
    main()
```

### File: `/opt/axentx/vanguard/run_lightning.py` (orchestrator)
```python
#!/usr/bin/env python3
"""
Mac-side orchestrator only — no model.from_pretrained() locally.
"""
import os
import sys
from lightning import Lightning, Teamspace, Studio, Machine

LIGHTNING_ACCOUNT = os.getenv("LIGHTNING_ACCOUNT", "lightning-public-prod")


def get_or_create_studio(name: str):
    ts = Teamspace()
    for s in ts.studios:
        if s.name == name and s.status == "Running":
            print(f"Reusing running studio: {s.name}")
            return s
    print(f"Creating studio: {name}")
    return Studio(
        name=name,
        machine=Machine.L40S,
        cloud=LIGHTNING_ACCOUNT,
        create_ok=True,
    )


def run_training(repo: str, date_folder: str):
    # Build manifest once on Mac (after rate-limit window)
    os.system(
        f"python manifest.py build --repo {repo} --date {date_folder} --out-dir manifest"
    )

    studio = get_or_create_studio("vanguard-train")
    if studio.status != "Running":
        studio.start(machine=Machine.L40S)

    # Lightning training does CDN-only fetches; zero HF API calls during data load
    studio.run(
        repo="https://github.com/axentx/vanguard",
        command=[
            "python", "train.py",
            "--repo", repo,
            "--date", date_folder,
            "--manifest-dir", "manifest",
        ],
        framework="docker",
    )


if __name__ == "__main__":
    if len(sys.argv) != 3:
        print("Usage: run_lightning.py <repo> <date_folder>")
        sys.exit(1)
    run_training(sys.argv[1], sys.argv[2])
```
