# surrogate-1 / ops

## Implementation Plan (≤2h)

**Goal**: Replace fragile shell-based ingestion with a **manifest-driven, CDN-bypass Python worker** that eliminates HF API rate limits during training data loads and prevents mixed-schema CastErrors.

### Changes

1. **Add `bin/worker.py`** — manifest-driven worker that:
   - Accepts `SHARD_ID` and `TOTAL_SHARDS` (matrix strategy)
   - Single API call to `list_repo_tree` for the target date folder
   - Persists file manifest to `manifest-shard-{SHARD_ID}.json`
   - Downloads only assigned files via **CDN bypass** (`resolve/main/...`, no auth)
   - Projects to `{prompt, response}` at parse time (avoids mixed-schema CastError)
   - Streams output to `batches/public-merged/{date}/shard{N}-{HHMMSS}.jsonl`
   - Uses central dedup via `lib/dedup.py`

2. **Update `bin/dataset-enrich.sh`** — thin wrapper that:
   - Sets `SHELL=/bin/bash`
   - Invokes `python3 bin/worker.py` with matrix env vars
   - Falls back to existing behavior if Python unavailable

3. **Update `.github/workflows/ingest.yml`** — ensure matrix passes `SHARD_ID`/`TOTAL_SHARDS` and uses `python3` in job container.

### 1) `bin/worker.py` (new)

#!/usr/bin/env python3
"""
Manifest-driven, CDN-bypass ingestion worker for surrogate-1.

Usage:
  SHARD_ID=0 TOTAL_SHARDS=16 python3 bin/worker.py

Environment:
  HF_TOKEN         - write token for axentx/surrogate-1-training-pairs
  HF_REPO          - dataset repo (default: axentx/surrogate-1-training-pairs)
  TARGET_DATE      - YYYY-MM-DD folder under datasets/ (default: today)
  SHARD_ID         - matrix shard index (0..TOTAL_SHARDS-1)
  TOTAL_SHARDS     - total parallel shards (default: 16)
"""

import os
import sys
import json
import hashlib
import datetime
import itertools
from pathlib import Path

import requests
from huggingface_hub import HfApi, hf_hub_download, list_repo_tree
from datasets import Dataset, Features, Value

HF_TOKEN = os.getenv("HF_TOKEN")
HF_REPO = os.getenv("HF_REPO", "axentx/surrogate-1-training-pairs")
TARGET_DATE = os.getenv("TARGET_DATE", datetime.date.today().isoformat())
SHARD_ID = int(os.getenv("SHARD_ID", "0"))
TOTAL_SHARDS = int(os.getenv("TOTAL_SHARDS", "16"))

API = HfApi(token=HF_TOKEN)
BASE_CDN = f"https://huggingface.co/datasets/{HF_REPO}/resolve/main"
DATE_PREFIX = f"datasets/{TARGET_DATE}"

OUT_DIR = Path("batches/public-merged") / TARGET_DATE
OUT_DIR.mkdir(parents=True, exist_ok=True)
TIMESTAMP = datetime.datetime.utcnow().strftime("%H%M%S")
OUT_FILE = OUT_DIR / f"shard{SHARD_ID}-{TIMESTAMP}.jsonl"

# Central dedup store (shared across workers via mounted volume or HF Space SQLite)
sys.path.insert(0, str(Path(__file__).parent))
from lib.dedup import DedupStore  # noqa: E402

dedup = DedupStore()

def list_date_files() -> list[str]:
    """Single API call: list files in date folder (non-recursive)."""
    try:
        tree = list_repo_tree(repo_id=HF_REPO, path=DATE_PREFIX, repo_type="dataset")
    except Exception as exc:
        # Fallback: if folder doesn't exist yet, return empty
        print(f"[WARN] list_repo_tree failed: {exc}", file=sys.stderr)
        return []
    # tree items have .path
    return sorted(
        item.path for item in tree if item.type == "file" and item.path.endswith((".jsonl", ".parquet", ".json"))
    )

def assign_shard(paths: list[str], shard_id: int, total: int) -> list[str]:
    """Deterministic shard assignment by slug hash."""
    assigned = []
    for p in paths:
        # stable hash across runs
        h = int(hashlib.md5(p.encode()).hexdigest(), 16)
        if h % total == shard_id:
            assigned.append(p)
    return assigned

def download_via_cdn(repo_path: str) -> bytes:
    """CDN bypass: no Authorization header, avoids /api/ rate limits."""
    url = f"{BASE_CDN}/{repo_path}"
    resp = requests.get(url, timeout=60)
    resp.raise_for_status()
    return resp.content

def parse_to_pair(content: bytes, path: str) -> list[dict]:
    """
    Parse heterogeneous files and project to {prompt,response}.
    Avoid load_dataset(streaming=True) on mixed schemas.
    """
    suffix = Path(path).suffix.lower()
    pairs = []

    if suffix == ".jsonl":
        for line in content.decode("utf-8").splitlines():
            line = line.strip()
            if not line:
                continue
            try:
                obj = json.loads(line)
            except json.JSONDecodeError:
                continue
            prompt = obj.get("prompt") or obj.get("input") or obj.get("question")
            response = obj.get("response") or obj.get("output") or obj.get("answer")
            if prompt is not None and response is not None:
                pairs.append({"prompt": str(prompt), "response": str(response)})

    elif suffix == ".json":
        try:
            data = json.loads(content.decode("utf-8"))
        except json.JSONDecodeError:
            return []
        if isinstance(data, list):
            items = data
        else:
            items = [data]
        for obj in items:
            prompt = obj.get("prompt") or obj.get("input") or obj.get("question")
            response = obj.get("response") or obj.get("output") or obj.get("answer")
            if prompt is not None and response is not None:
                pairs.append({"prompt": str(prompt), "response": str(response)})

    elif suffix == ".parquet":
        # Avoid pyarrow CastError by reading only needed columns via pyarrow
        try:
            import pyarrow.parquet as pq
            import pyarrow as pa
            table = pq.read_table(pa.BufferReader(content), columns=["prompt", "response"])
            df = table.to_pandas()
        except Exception:
            # Fallback: try common aliases
            try:
                table = pq.read_table(pa.BufferReader(content))
                df = table.to_pandas()
            except Exception:
                return []
        prompt_col = next((c for c in df.columns if c in ("prompt", "input", "question")), None)
        response_col = next((c for c in df.columns if c in ("response", "output", "answer")), None)
        if prompt_col and response_col:
            for _, row in df.iterrows():
                p = row[prompt_col]
                r = row[response_col]
                if pd_notna(p) and pd_notna(r):
                    pairs.append({"prompt": str(p), "response": str(r)})
    return pairs

def pd_notna(v):
    try:
        import pandas as pd
        return pd.notna(v)
    except Exception:
        return v is not None and (not (isinstance(v, float) and v != v))

def main() -> None:
    print(f"[INFO] Worker shard={SHARD_ID}/{TOTAL_SHARDS} date={TARGET_DATE}")

    # 1) List files once (single API call)
    all_files = list_date_files()
    print(f"[INFO] Found {len(all_files)} files in {DATE_PREFIX}")

    # Persist manifest for reproducibility / training script embedding
    manifest_path = Path(f"manifest-shard{SHARD_ID}.json")
    manifest_path.write_text(json.dumps(all_files, indent=2))
    print(f"[INFO] Manifest saved to {manifest_path}")

    # 2) Assign shard
    my_files = assign_shard(all_files, SHARD_ID, TOTAL_SHARDS)
    print(f"[INFO] Shard {SHARD_ID} assigned {len(my_files)} files")

    # 3) Stream-process and upload
    written = 0
    skipped_dup = 0
