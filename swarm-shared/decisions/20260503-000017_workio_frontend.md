# workio / frontend

## 1. Diagnosis
- No client-side idempotency for punch submissions → retries (offline/network flaps) create duplicate punches.
- No optimistic UI or local pending queue → spinner ambiguity and zero feedback when API/webhook fails.
- Clock-in/out buttons remain interactive during in-flight requests → double-taps race.
- No persisted draft state across reloads → user loses in-progress punch if page refreshes.
- Missing visual distinction between pending/committed states → users resubmit.

## 2. Proposed change
File: `workio/src/features/punch/usePunchQueue.ts` (new) + `workio/src/features/punch/PunchButton.tsx` (modify).  
Scope: add an in-memory+localStorage queue with idempotency keys, optimistic state, and one-shot submit worker; wire PunchButton to enqueue and reflect status.

## 3. Implementation

### `workio/src/features/punch/usePunchQueue.ts`
```tsx
import { useEffect, useCallback, useState } from 'react';
import { v4 as uuidv4 } from 'uuid';

type PunchType = 'IN' | 'OUT';
type PunchStatus = 'pending' | 'committing' | 'committed' | 'failed';

interface PunchRequest {
  idempotencyKey: string;
  tenantId: string;
  userId: string;
  type: PunchType;
  latitude?: number;
  longitude?: number;
  timestamp: string; // ISO
}

interface QueueItem extends PunchRequest {
  status: PunchStatus;
  attempts: number;
  error?: string;
}

const STORAGE_KEY = 'workio:punch-queue';
const MAX_ATTEMPTS = 3;

function loadQueue(): QueueItem[] {
  try {
    const raw = localStorage.getItem(STORAGE_KEY);
    return raw ? JSON.parse(raw) : [];
  } catch {
    return [];
  }
}

function saveQueue(queue: QueueItem[]) {
  try {
    localStorage.setItem(STORAGE_KEY, JSON.stringify(queue));
  } catch {
    // ignore storage failures
  }
}

export function usePunchQueue({ tenantId, userId }: { tenantId: string; userId: string }) {
  const [queue, setQueue] = useState<QueueItem[]>(loadQueue);

  useEffect(() => {
    saveQueue(queue);
  }, [queue]);

  // Retry worker
  useEffect(() => {
    const timer = setInterval(() => {
      setQueue((prev) => {
        const next = [...prev];
        let changed = false;
        for (const item of next) {
          if (item.status === 'failed' && item.attempts < MAX_ATTEMPTS) {
            item.status = 'committing';
            changed = true;
            attemptSubmit(item);
          }
        }
        return changed ? [...next] : prev;
      });
    }, 10000); // retry every 10s
    return () => clearInterval(timer);
  }, []);

  const attemptSubmit = useCallback(async (item: QueueItem) => {
    try {
      const res = await fetch('/api/punch', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          idempotencyKey: item.idempotencyKey,
          tenantId: item.tenantId,
          userId: item.userId,
          type: item.type,
          latitude: item.latitude,
          longitude: item.longitude,
          timestamp: item.timestamp,
        }),
      });

      if (!res.ok) throw new Error(`HTTP ${res.status}`);

      setQueue((prev) =>
        prev.map((p) =>
          p.idempotencyKey === item.idempotencyKey ? { ...p, status: 'committed' } : p
        )
      );
    } catch (err: any) {
      setQueue((prev) =>
        prev.map((p) =>
          p.idempotencyKey === item.idempotencyKey
            ? {
                ...p,
                status: 'failed',
                attempts: p.attempts + 1,
                error: err.message,
              }
            : p
        )
      );
    }
  }, []);

  const enqueue = useCallback(
    (type: PunchType, coords?: { latitude: number; longitude: number }) => {
      const item: QueueItem = {
        idempotencyKey: uuidv4(),
        tenantId,
        userId,
        type,
        latitude: coords?.latitude,
        longitude: coords?.longitude,
        timestamp: new Date().toISOString(),
        status: 'committing',
        attempts: 0,
      };
      setQueue((prev) => {
        // dedupe in-flight same-type for same day (simple client-side guard)
        const today = new Date().toDateString();
        const hasPendingToday = prev.some(
          (p) =>
            p.userId === userId &&
            p.type === type &&
            new Date(p.timestamp).toDateString() === today &&
            (p.status === 'pending' || p.status === 'committing')
        );
        if (hasPendingToday) return prev;
        return [...prev, item];
      });
      attemptSubmit(item);
      return item.idempotencyKey;
    },
    [tenantId, userId, attemptSubmit]
  );

  const latestByType = useCallback(
    (type: PunchType) => {
      return queue
        .filter((p) => p.type === type && p.status === 'committed')
        .sort((a, b) => new Date(b.timestamp).valueOf() - new Date(a.timestamp).valueOf())[0];
    },
    [queue]
  );

  const statusByDay = useCallback((date: Date) => {
    const d = date.toDateString();
    return queue.filter(
      (p) => new Date(p.timestamp).toDateString() === d && p.status === 'committed'
    );
  }, [queue]);

  return { enqueue, queue, latestByType, statusByDay };
}
```

  userId: string;

  const [locError, setLocError] = useState<string | null>(null);

  const handleClick = useCallback(() => {

      enqueue(type);

          latitude: pos.coords.latitude,
