# surrogate-1 / design

## Final Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID` and `SHARD_TOTAL` (16) from GitHub Actions matrix.
- Uses a pre-generated `file-list.json` (Mac-side, single `list_repo_tree` call) to deterministically assign 1/16 of files to this shard (`hash(slug) % 16`).
- Downloads assigned files via **HF CDN bypass** (`https://huggingface.co/datasets/.../resolve/main/...`) with no Authorization header → zero API calls during ingest.
- Projects each file to `{prompt, response}` only at parse time (avoids pyarrow CastError from mixed schemas).
- Deduplicates via central md5 store (`lib/dedup.py`) and writes normalized pairs to `batches/public-merged/<date>/shard<N>-<HHMMSS>.jsonl`.
- Keeps the HF Space as the single source of truth for cross-run dedup; workers remain stateless per run.

---

### Steps (est. 90–110 min)

1. Create `bin/dataset-enrich.py` (new) — manifest loader, CDN downloader, schema projector, shard assignment, dedup writer.
2. Add small helper `bin/get_file_list.py` (Mac orchestration) — one-time `list_repo_tree` → `file-list.json` (committed or passed to workflow).
3. Update `.github/workflows/ingest.yml` — pass `SHARD_ID`/`SHARD_TOTAL`, install deps, run `python bin/dataset-enrich.py`.
4. Remove/disable old `bin/dataset-enrich.sh` (keep as backup for one release).
5. Smoke test locally with a small file-list subset.

### `bin/get_file_list.py` (run on Mac; produces file-list.json)

```python
#!/usr/bin/env python3
"""
Generate file-list.json for a date folder in axentx/surrogate-1-training-pairs.
Run on Mac (or any machine with HF token) to avoid list_repo_files recursion.

Usage:
  HF_TOKEN=hf_xxx python bin/get_file_list.py --date 2026-05-03 --out file-list.json
"""

import argparse
import json
import os
from huggingface_hub import HfApi

API = HfApi()
REPO = "datasets/axentx/surrogate-1-training-pairs"

def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("--date", required=True, help="Date folder, e.g. 2026-05-03")
    parser.add_argument("--out", default="file-list.json")
    args = parser.parse_args()

    # Non-recursive per folder to avoid pagination explosion
    folder = f"batches/public-raw/{args.date}"
    items = API.list_repo_tree(repo_id=REPO, path=folder, recursive=False)

    # Flatten only files (ignore subfolders)
    files = [it.rfilename for it in items if not it.rfilename.endswith("/")]

    # Save manifest for workers
    payload = {
        "date": args.date,
        "folder": folder,
        "files": sorted(files),
    }
    with open(args.out, "w") as f:
        json.dump(payload, f, indent=2)

    print(f"Wrote {len(files)} files to {args.out}")

if __name__ == "__main__":
    main()
```

### `bin/dataset-enrich.py` (worker; CDN-bypass, schema projection, dedup)

```python
#!/usr/bin/env python3
"""
CDN-bypass ingestion worker for surrogate-1.
Deterministic shard assignment + zero API calls during data load.

Required env:
  SHARD_ID      (0..15)
  SHARD_TOTAL   (16)
  HF_TOKEN      (write permission to axentx/surrogate-1-training-pairs)
  FILE_LIST     (path to file-list.json; default file-list.json)
"""

import hashlib
import json
import os
import sys
import time
from datetime import datetime, timezone
from pathlib import Path

import pyarrow as pa
import pyarrow.parquet as pq
import requests
from huggingface_hub import HfApi

sys.path.insert(0, str(Path(__file__).parent))
from lib.dedup import DedupStore  # noqa: E402

API = HfApi()
REPO = "datasets/axentx/surrogate-1-training-pairs"
CDN_ROOT = "https://huggingface.co/datasets"

# ---- helpers ----
def hash_slug(s: str) -> int:
    return int(hashlib.sha256(s.encode()).hexdigest(), 16)

def assign_shard(slug: str, total: int) -> int:
    return hash_slug(slug) % total

def cdn_url(repo: str, path: str) -> str:
    return f"{CDN_ROOT}/{repo}/resolve/main/{path}"

def project_to_pair(raw_bytes: bytes, filename: str) -> dict | None:
    """
    Best-effort projection to {prompt, response}.
    Avoids pyarrow mixed-schema errors by not using load_dataset.
    """
    try:
        # Try parquet first
        table = pq.read_table(pa.BufferReader(raw_bytes))
        cols = set(table.column_names)

        # Common patterns observed in surrogate-1 datasets
        prompt_col = next((c for c in ("prompt", "instruction", "input", "question") if c in cols), None)
        response_col = next((c for c in ("response", "output", "answer", "completion") if c in cols), None)

        if prompt_col is None or response_col is None:
            # Fallback: pick first two text-like columns
            text_cols = [c for c in cols if table.schema.field(c).type in (pa.string(), pa.large_string())]
            if len(text_cols) >= 2:
                prompt_col, response_col = text_cols[0], text_cols[1]
            else:
                return None

        # Take first valid row (workers can be extended to emit multiple rows)
        for i in range(table.num_rows):
            prompt = table[prompt_col][i].as_py()
            response = table[response_col][i].as_py()
            if isinstance(prompt, str) and isinstance(response, str):
                if prompt.strip() and response.strip():
                    return {"prompt": prompt.strip(), "response": response.strip()}
        return None
    except Exception:
        return None

# ---- main ----
def main() -> None:
    shard_id = int(os.environ.get("SHARD_ID", "0"))
    shard_total = int(os.environ.get("SHARD_TOTAL", "16"))
    file_list_path = os.environ.get("FILE_LIST", "file-list.json")
    hf_token = os.environ.get("HF_TOKEN")
    if not hf_token:
        print("HF_TOKEN required", file=sys.stderr)
        sys.exit(1)

    with open(file_list_path) as f:
        manifest = json.load(f)

    date = manifest["date"]
    files = manifest["files"]

    # Deterministic shard assignment
    my_files = [fn for fn in files if assign_shard(fn, shard_total) == shard_id]
    print(f"Shard {shard_id}/{shard_total} assigned {len(my_files)} files")

    dedup = DedupStore()
    api = HfApi(token=hf_token)

    out_dir = Path(f"batches/public-merged/{date}")
    out_dir.mkdir(parents=True, exist_ok=True)
    ts = datetime.now(timezone.utc).strftime("%H%M%S")
    out_file = out_dir / f"shard{shard_id}-{ts}.jsonl"

    written = 0
    skipped_dup = 0
    failed = 0

    with out_file.open("w") as out:
        for rfilename in my_files:
            try:
                # CDN bypass: no Authorization header -> no API rate limit
                url = cdn_url(REPO, rfilename)
                resp = requests.get(url, timeout=30)
                resp.raise_for_status()
                raw = resp.content

                pair = project_to_pair(raw, rfilename)
                if not pair:
                    failed += 1
                    continue

                # Dedup by content
