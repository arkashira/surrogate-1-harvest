# vanguard / discovery

## Final Consolidated Implementation

**Diagnosis (resolved)**  
- Missing durable manifest forces repeated HF API calls → 429s and quota burn.  
- Training uses `load_dataset`/`list_repo_files` instead of CDN → guaranteed rate limits.  
- No reuse guard for Lightning Studio → repeated creation burns quota.  
- No deterministic shard mapping for large-scale ingestion (commit cap 128/hr/repo).  

**Solution (single coherent plan)**  
1. One durable `file-list.json` per date folder generated by a single Mac-side API call.  
2. Training uses only CDN URLs (`https://huggingface.co/datasets/{repo}/resolve/main/{path}`) — zero HF API calls during training.  
3. Lightning launcher reuses a running Studio; never recreates if already running.  
4. Optional shard mapping for large repos to respect HF commit caps.

```python
# /opt/axentx/vanguard/scripts/discovery/manifest.py
#!/usr/bin/env python3
"""
Generate a durable CDN-only file-list manifest for a HuggingFace dataset repo.
Run once per date folder after HF API rate-limit window is clear.

Usage:
  HF_TOKEN=hf_xxx python manifest.py \
    --repo datasets/opus-mt-ko-en \
    --date 2026-05-02 \
    --out /opt/axentx/vanguard/data/file-list.json
"""
import argparse
import json
import os
import sys
from datetime import datetime, timezone
from pathlib import Path

try:
    from huggingface_hub import HfApi
except ImportError:
    print("Missing huggingface_hub. Install with: pip install huggingface_hub")
    sys.exit(1)

def build_manifest(repo_id: str, date_folder: str, out_path: Path) -> dict:
    api = HfApi()
    # Single cheap call: non-recursive listing of one date folder
    entries = api.list_repo_tree(repo_id=repo_id, path=date_folder, recursive=False)

    files = []
    for e in entries:
        if e.type == "file":
            files.append({
                "repo": repo_id,
                "path": e.path,
                "cdn_url": f"https://huggingface.co/datasets/{repo_id}/resolve/main/{e.path}",
                "size": getattr(e, "size", None)
            })

    manifest = {
        "repo": repo_id,
        "date_folder": date_folder,
        "generated_at": datetime.now(timezone.utc).isoformat(),
        "count": len(files),
        "files": files
    }

    out_path.parent.mkdir(parents=True, exist_ok=True)
    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"Wrote {len(files)} files to {out_path}")
    return manifest

def main() -> None:
    parser = argparse.ArgumentParser(description="Generate CDN file-list manifest.")
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g. datasets/opus-mt-ko-en)")
    parser.add_argument("--date", required=True, help="Date folder in repo (e.g. 2026-05-02)")
    parser.add_argument("--out", required=True, help="Output JSON path")
    args = parser.parse_args()

    token = os.getenv("HF_TOKEN")
    api = HfApi()
    if token:
        api._token = token

    build_manifest(args.repo, args.date, Path(args.out))

if __name__ == "__main__":
    main()
```

### 3. Optional: shard-to-repo mapping for large repos (respect HF commit cap)

If you must split across repos to avoid >128 commits/hr/repo:

```python
# /opt/axentx/vanguard/scripts/discovery/shard_map.py
import json
from pathlib import Path

def build_shard_map(manifest_path: Path, shards: int, out_dir: Path):
    manifest = json.loads(manifest_path.read_text())
    files = manifest["files"]
    out_dir.mkdir(parents=True, exist_ok=True)

    per_shard = len(files) // shards + (1 if len(files) % shards else 0)
    for i in range(shards):
        chunk = files[i * per_shard:(i + 1) * per_shard]
        repo_name = f"{manifest['repo'].replace('/', '_')}_shard{i}"
        shard_manifest = {
            "repo": repo_name,
            "date_folder": manifest["date_folder"],
            "generated_at": manifest["generated_at"],
            "count": len(chunk),
            "files": chunk
        }
        (out_dir / f"{repo_name}.json").write_text(json.dumps(shard_manifest, indent=2))
    print(f"Wrote {shards} shard manifests to {out_dir}")

if __name__ == "__main__":
    import argparse
    parser = argparse.ArgumentParser()
    parser.add_argument("--manifest", required=True)
    parser.add_argument("--shards", type=int, default=4)
    parser.add_argument("--out-dir", default="data/shards")
    args = parser.parse_args()
    build_shard_map(Path(args.manifest), args.shards, Path(args.out_dir))
```

Use shard manifests to distribute ingestion across repos and stay within HF commit limits.
