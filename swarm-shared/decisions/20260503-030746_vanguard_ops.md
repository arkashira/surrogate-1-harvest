## Final Consolidated Implementation (Best of Both Candidates)

### 1. Diagnosis (Resolved)
- **Deterministic CDN-first manifest missing** → Use HF API exactly once per folder to generate manifest; training uses only CDN URLs.
- **No integrity verification** → Include SHA-256 in manifest (computed once at generation) and verify on every training fetch.
- **No idempotent ops automation** → Single shebanged Python + cron-ready Bash script with lockfile to prevent concurrent runs.
- **Lightning idle-stop wasting quota** → Add checkpointing and fast resume in loader; keep fetches short and verifiable.
- **Missing single source of truth** → Manifest is canonical; training scripts consume only `manifest.json`.

# /opt/axentx/vanguard/ops/make_manifest.py
#!/usr/bin/env python3
"""
Generate a deterministic CDN-first manifest for a HuggingFace dataset folder.
Usage:
  python3 make_manifest.py \
    --repo datasets/axentx/surrogate-1 \
    --folder 2026-05-03 \
    --out manifests/2026-05-03.json
"""
import argparse
import hashlib
import json
import sys
from concurrent.futures import ThreadPoolExecutor, as_completed
from pathlib import Path

try:
    from huggingface_hub import list_repo_tree
    from huggingface_hub.utils import HFError
except ImportError:
    print("Missing huggingface_hub. Install with: pip install huggingface_hub requests")
    sys.exit(1)

CDN_BASE = "https://huggingface.co/datasets"


def _sha256_bytes(data: bytes) -> str:
    return hashlib.sha256(data).hexdigest()


def _fetch_and_hash(url: str, timeout: int = 60) -> str:
    import requests
    resp = requests.get(url, timeout=timeout)
    resp.raise_for_status()
    return _sha256_bytes(resp.content)


def main() -> None:
    parser = argparse.ArgumentParser(description="Generate CDN manifest for HF dataset folder.")
    parser.add_argument("--repo", required=True, help="e.g. datasets/axentx/surrogate-1")
    parser.add_argument("--folder", required=True, help="Folder path inside repo (e.g. 2026-05-03)")
    parser.add_argument("--out", required=True, help="Output JSON path")
    parser.add_argument("--workers", type=int, default=4, help="Parallel hash workers")
    args = parser.parse_args()

    # list once, non-recursive (rate-limit friendly)
    try:
        items = list_repo_tree(repo_id=args.repo, path=args.folder, recursive=False)
    except HFError as e:
        print(f"HF API error: {e}", file=sys.stderr)
        sys.exit(1)

    entries = []
    files = [item for item in items if item.type == "file"]
    if not files:
        print(f"No files found in {args.repo}/{args.folder}", file=sys.stderr)
        sys.exit(0)

    # Build URLs and compute SHA-256 in parallel (one-time cost)
    to_hash = []
    for item in files:
        path = item.path
        url = f"{CDN_BASE}/{args.repo}/resolve/main/{path}"
        to_hash.append((path, url))

    results = {}
    with ThreadPoolExecutor(max_workers=args.workers) as ex:
        future_to_path = {
            ex.submit(_fetch_and_hash, url): (path, url)
            for path, url in to_hash
        }
        for fut in as_completed(future_to_path):
            path, url = future_to_path[fut]
            try:
                h = fut.result()
                results[path] = (url, h)
            except Exception as e:
                print(f"WARN: failed to hash {path}: {e}", file=sys.stderr)
                # still include entry; training loader will fail fast on mismatch/empty
                results[path] = (url, None)

    for item in files:
        path = item.path
        url, h = results.get(path, (f"{CDN_BASE}/{args.repo}/resolve/main/{path}", None))
        entries.append(
            {
                "path": path,
                "cdn_url": url,
                "sha256": h,
            }
        )

    manifest = {
        "repo": args.repo,
        "folder": args.folder,
        "generated_by": "make_manifest.py",
        "entries": entries,
    }

    out_path = Path(args.out)
    out_path.parent.mkdir(parents=True, exist_ok=True)
    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"Wrote {len(entries)} entries to {out_path}")


if __name__ == "__main__":
    main()

# Idempotent manifest generator with lockfile (cron-safe).
set -euo pipefail

cd /opt/axentx/vanguard

REPO="${HF_REPO:-datasets/axentx/surrogate-1}"
FOLDER="${HF_FOLDER:-$(date +%Y-%m-%d)}"
OUT="ops/manifests/${FOLDER}.json"
LOCK="ops/locks/ingest_${FOLDER}.lock"

mkdir -p ops/locks ops/manifests

exec 200>"${LOCK}"
flock -n 200 || { echo "Another ingest is running for ${FOLDER}. Exiting."; exit 0; }

echo "[$(date -u +%Y-%m-%dT%H:%M:%SZ)] Generating manifest: repo=${REPO} folder=${FOLDER}"
python3 ops/make_manifest.py --repo "${REPO}" --folder "${FOLDER}" --out "${OUT}" --workers 4
echo "[$(date -u +%Y-%m-%dT%H:%M:%SZ)] Done."

        self.manifest = json.loads(Path(manifest_path).read_text())
        self.cache_root = Path("data/cdn_cache")

        h = hashlib.sha256()

                h.update(chunk)
        return h.hexdigest()
