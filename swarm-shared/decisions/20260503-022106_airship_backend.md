### 1) `scripts/build_file_list.py` (run on Mac)

```python
#!/usr/bin/env python3
"""
One-shot: list a date folder in HF dataset repo and save file list for CDN-only training.
Usage:
    python scripts/build_file_list.py \
        --repo datasets/eliben/code-translated-examples \
        --date 2026-04-29 \
        --out file_list.json
"""
import argparse
import json
import os
import time
from pathlib import Path

from huggingface_hub import HfApi

HF_TOKEN = os.getenv("HF_TOKEN")

def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g. datasets/eliben/code-translated-examples)")
    parser.add_argument("--date", required=True, help="Date folder (e.g. 2026-04-29)")
    parser.add_argument("--out", required=True, help="Output JSON path")
    parser.add_argument("--max-retries", type=int, default=3)
    args = parser.parse_args()

    api = HfApi(token=HF_TOKEN)
    # Avoid recursive on big repos; list one folder only.
    path = f"{args.date}"
    for attempt in range(1, args.max_retries + 1):
        try:
            items = api.list_repo_tree(repo_id=args.repo, path=path, recursive=False)
            break
        except Exception as exc:
            if attempt == args.max_retries:
                raise
            wait = 30 * attempt
            print(f"Attempt {attempt} failed: {exc}. Retry in {wait}s")
            time.sleep(wait)

    files = []
    for item in items:
        # item.path is relative to repo root; we want path inside date folder
        if item.type == "file":
            files.append(item.path)

    out_path = Path(args.out)
    out_path.parent.mkdir(parents=True, exist_ok=True)
    payload = {
        "repo": args.repo,
        "date": args.date,
        "files": sorted(files),
    }
    out_path.write_text(json.dumps(payload, indent=2))
    print(f"Wrote {len(files)} files to {args.out}")

if __name__ == "__main__":
    main()
```

Make executable:

```bash
chmod +x scripts/build_file_list.py
```

### 2) Surrogate dataset loader (CDN-only)

```python
# surrogate/data/cdn_dataset.py
import json
import os
from pathlib import Path
from typing import Dict, List

import pyarrow as pa
import pyarrow.parquet as pq
import requests
from tqdm import tqdm

HF_DATASETS_CDN = "https://huggingface.co/datasets"

class CDNParquetDataset:
    """
    Loads Parquet shards via CDN (no HF API auth/rate-limit during training).
    Provide file_list.json produced by scripts/build_file_list.py.
    """

    def __init__(self, file_list_path: str, repo: str, columns: List[str] = None):
        with open(file_list_path) as f:
            manifest = json.load(f)
        self.repo = repo or manifest["repo"]
        self.files = manifest["files"]
        self.columns = columns

    def _cdn_url(self, file_path: str) -> str:
        # file_path is repo-relative (e.g. 2026-04-29/shard-0001.parquet)
        return f"{HF_DATASETS_CDN}/{self.repo}/resolve/main/{file_path}"

    def __iter__(self):
        for file_path in tqdm(self.files, desc="CDN shards"):
            url = self._cdn_url(file_path)
            for attempt in range(3):
                try:
                    resp = requests.get(url, timeout=30)
                    resp.raise_for_status()
                    break
                except Exception as exc:
                    if attempt == 2:
                        raise
                    continue

            table = pq.read_table(pa.BufferReader(resp.content))
            if self.columns:
                table = table.select(self.columns)
            # Yield rows as dicts (or convert to your training format)
            for batch in table.to_batches(max_chunksize=1024):
                for row in zip(*[batch.column(col) for col in batch.schema.names]):
                    yield dict(zip(batch.schema.names, row))
```

### 3) Lightning Studio reuse + idle-stop guard

```python
# surrogate/lightning_launcher.py
import time
from lightning_sdk import Studio, Machine, Teamspace
from typing import Optional

def get_or_create_studio(
    name: str,
    machine: Machine = Machine.L40S,
    teamspace: Optional[str] = None,
) -> Studio:
    """
    Reuse a running studio if exists; otherwise create.
    If studio exists but is stopped, restart it.
    """
    ts = Teamspace(name=teamspace) if teamspace else Teamspace.current()
    running = None
    for s in ts.studios:
        if s.name == name:
            running = s
            break

    if running is not None:
        if running.status == "running":
            print(f"Reusing running studio: {name}")
            return running
        else:
            print(f"Studio {name} exists but status={running.status}. Starting...")
            running.start(machine=machine)
            _wait_running(running)
            return running

    print(f"Creating studio: {name}")
    studio = Studio(
        name=name,
        machine=machine,
        create_ok=True,
        teamspace=teamspace,
    )
    _wait_running(studio)
    return studio

def _wait_running(studio: Studio, timeout: int = 300, interval: int = 10):
    elapsed = 0
    while elapsed < timeout:
        studio.refresh()
        if studio.status == "running":
            return
        time.sleep(interval)
