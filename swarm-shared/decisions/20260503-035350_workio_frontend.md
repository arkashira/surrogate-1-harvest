# workio / frontend

## 1. Diagnosis
- No real-time subscription layer: frontend relies on local React state and polling; LINE webhook–driven changes (clock/leave/OT) are invisible until hard refresh → perceived latency and silent failures.
- Multi-tab/device divergence: clock/leave/OT mutations performed on one device/tab are not reflected in others until reload, causing stale UI and double-submissions.
- Polling is coarse (if present) or absent; when present it burns battery/requests and still misses short windows between polls.
- No optimistic UI for clock in/out and leave/OT requests → perceived sluggishness and accidental double-taps.
- No centralized channel for backend events (websocket/SSE) to push tenant-scoped updates; frontend cannot subscribe to “my tenant’s” stream of time/leave/OT events.

## 2. Proposed change
Add a lightweight, tenant-scoped EventSource (SSE) subscription in the frontend that connects to a new backend endpoint `/api/events/stream` (tenant inferred from auth/session), and wire it into a central `useEventStream` hook that dispatches normalized actions into Zustand (or existing store). Scope:
- New backend endpoint: `server/src/routes/events.ts` (or similar) — returns `text/event-stream`, streams tenant-scoped entity changes (clock, leave, ot).
- Frontend hook: `src/hooks/useEventStream.ts` — manages reconnect, dedupe, and dispatches to store.
- Update clock/leave/OT list components to derive from store (no breaking rewrite) and add optimistic updates for local mutations.

## 3. Implementation

### Backend: SSE stream endpoint (Node/Express)
File: `server/src/routes/events.ts`
```ts
import express from 'express';
import { db } from '../db';
import type { Request } from '../types';

const router = express.Router();

function tenantId(req: Request): string | null {
  // adapt to your auth: req.user.tenantId or req.session.tenantId
  return (req as any).user?.tenantId || (req as any).session?.tenantId || null;
}

router.get('/stream', (req, res) => {
  const tid = tenantId(req as Request);
  if (!tid) {
    res.status(401).end();
    return;
  }

  res.setHeader('Content-Type', 'text/event-stream');
  res.setHeader('Cache-Control', 'no-cache');
  res.setHeader('Connection', 'keep-alive');
  res.flushHeaders();

  // Send initial heartbeat/retry
  res.write('retry: 3000\n\n');

  // Helper to send event
  const send = (type: string, payload: any) => {
    res.write(`event: ${type}\n`);
    res.write(`data: ${JSON.stringify(payload)}\n\n`);
  };

  // On-demand broadcast helper (call from controllers when entities change)
  const broadcastChange = (entity: 'clock' | 'leave' | 'ot', action: 'upsert' | 'delete', data: any) => {
    send('entity-change', { entity, action, data, ts: Date.now() });
  };

  // Attach broadcaster to request/locals so controllers can call it
  (req as any).locals = (req as any).locals || {};
  (req as any).locals.broadcastChange = broadcastChange;

  // Keep connection alive
  const keepAlive = setInterval(() => res.write(':\n\n'), 20000);

  req.on('close', () => {
    clearInterval(keepAlive);
    res.end();
  });
});

export default router;
```

Wire into app (e.g., `server/src/app.ts`):
```ts
import eventsRouter from './routes/events';
app.use('/api/events', eventsRouter);
```

Broadcast from controllers (example in clock controller):
```ts
// after successful clock/leave/OT mutation
const broadcaster = (req as any).locals?.broadcastChange;
if (broadcaster) {
  broadcaster('clock', 'upsert', sanitizedClockRecord);
}
```

### Frontend: Hook to consume stream
File: `src/hooks/useEventStream.ts`
```ts
import { useEffect } from 'react';
import { useStore } from '../store'; // adapt to your store (Zustand/Redux)

export function useEventStream() {
  const { tenantId } = useStore((s) => s.auth); // adapt
  const upsertClock = useStore((s) => s.upsertClock);
  const upsertLeave = useStore((s) => s.upsertLeave);
  const upsertOt = useStore((s) => s.upsertOt);

  useEffect(() => {
    if (!tenantId) return;
    const source = new EventSource(`${import.meta.env.VITE_API_BASE}/api/events/stream`, {
      withCredentials: true,
    });

    source.addEventListener('entity-change', (e: any) => {
      try {
        const { entity, action, data } = JSON.parse(e.data);
        if (action === 'upsert') {
          if (entity === 'clock') upsertClock(data);
          if (entity === 'leave') upsertLeave(data);
          if (entity === 'ot') upsertOt(data);
        }
        // handle deletes if needed
      } catch {
        // ignore malformed
      }
    });

    source.onerror = () => {
      // EventSource auto-reconnects per spec; close after a few failures if desired
      setTimeout(() => source.close(), 30000);
    };

    return () => {
      source.close();
    };
  }, [tenantId, upsertClock, upsertLeave, upsertOt]);
}
```

### Frontend: Optimistic UI for clock in/out (example)
File: `src/components/ClockButton.tsx`
```tsx
import { useState } from 'react';
import { useStore } from '../store';
import { api } from '../lib/api';

export function ClockButton() {
  const [pending, setPending] = useState(false);
  const lastClock = useStore((s) => s.clocks[0]);
  const upsertClock = useStore((s) => s.upsertClock);

  const handleClock = async () => {
    const optimistic = {
      id: `temp-${Date.now()}`,
      type: lastClock?.type === 'in' ? 'out' : 'in',
      ts: new Date().toISOString(),
      _optimistic: true,
    };
    upsertClock(optimistic);
    setPending(true);

    try {
      const result = await api.post('/clock', { type: optimistic.type });
      // replace optimistic with real record
      upsertClock(result.data);
    } catch {
      // rollback: remove optimistic or mark error
      upsertClock({ ...optimistic, _error: true });
    } finally {
      setPending(false);
    }
  };

  return (
    <button onClick={handleClock} disabled={pending}>
      {lastClock?.type === 'in' ? 'Clock Out' : 'Clock In'}
    </button>
  );
}
```

### Store minimal patch (Zustand example)
File: `src/store/index.ts` (add)
```ts
export const useStore = create<Store>((set) => ({
  // existing state...
  clocks: [],
  leaves: [],
  ots: [],
  upsertClock: (c) =>
    set((s) => ({
      clocks: [c, ...s.clocks.filter((x) => x.id !== c.id && !x._optimistic)],
    })),
  upsertLeave: (l) =>
    set((s) => ({
      leaves: [l, ...s.leaves.filter((x) => x.id !== l.id && !x._optimistic)],
    })),
  upsertOt: (o) =>
    set((s) => ({
      ots: [o, ...s.ots.filter((x) => x.id !== o.id && !x._optimistic)],
    })),
}));
```
