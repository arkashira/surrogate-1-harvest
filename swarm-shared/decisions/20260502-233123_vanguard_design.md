# vanguard / design

## Final Synthesis (adopted)

**Diagnosis (merged, corrected)**
- No persistent file-list cache for HF repos → repeated `list_repo_tree`/`load_dataset` calls hit 429 rate limits.
- Lightning Studio lifecycle not reused → quota burned on redundant create/idle/stop cycles; idle-stop kills training silently.
- Training scripts use `load_dataset(streaming=True)` which still performs per-file HTTP range requests through HF’s API/CDN mix and can trigger rate limits/metadata calls; plus schema drift in `enriched/` pollutes training.
- No CDN bypass strategy during training → many data load paths go through authenticated `/api/` endpoints instead of public CDN `resolve/main/` URLs.
- Missing deterministic repo routing for HF commits → ingestion limited to 128 writes/hr/repo instead of spreading across sibling repos.

**Proposed change (merged, concrete)**
Create **one** canonical module `/opt/axentx/vanguard/design/hf_training_manifest.py` (<150 LOC) that:
- Exposes `build_manifest(repo, date_folder, tree_items)` → saves `manifests/{repo}/{date_folder}.json` with CDN paths and metadata.
- Exposes `stream_from_cdn(file_entry)` → yields file bytes via `requests.get(cdn_url, timeout=30)` without auth.
- Exposes `pick_sibling_repo(slug, n=5)` → deterministic hash routing for commit spreading.
- Adds lightweight `LightningStudioPool` context manager to reuse running studios and auto-restart on idle-stop.
- Adds `cdn_parquet_iterator(entries)` that projects to `{prompt,response}` only and ignores schema drift.

Also add **one** canonical example script `/opt/axentx/vanguard/design/train_cdn_only.py` that loads the manifest and streams parquet rows via CDN (zero HF API calls during training).

# /opt/axentx/vanguard/design/hf_training_manifest.py
import json, hashlib, os
from pathlib import Path
from typing import List, Dict, Any

import requests
from lightning_sdk import Teamspace, Studio, Machine

MANIFEST_ROOT = Path(__file__).parent.parent / "manifests"
MANIFEST_ROOT.mkdir(exist_ok=True)

def build_manifest(repo: str, date_folder: str, tree_items: List[Dict[str, Any]]) -> Path:
    """
    Build CDN-only manifest from list_repo_tree(path, recursive=False) items.
    Expects items with 'path' and optionally 'size'.
    Stores: {cdn_url, path, size, repo, date_folder}
    """
    entries = []
    for item in tree_items:
        rel = item["path"]
        cdn_url = f"https://huggingface.co/datasets/{repo}/resolve/main/{date_folder}/{rel}"
        entries.append({
            "cdn_url": cdn_url,
            "path": rel,
            "size": item.get("size"),
            "repo": repo,
            "date_folder": date_folder,
        })
    out = MANIFEST_ROOT / repo / f"{date_folder}.json"
    out.parent.mkdir(parents=True, exist_ok=True)
    out.write_text(json.dumps(entries, indent=2))
    return out

def load_manifest(repo: str, date_folder: str) -> List[Dict[str, Any]]:
    p = MANIFEST_ROOT / repo / f"{date_folder}.json"
    if not p.exists():
        raise FileNotFoundError(f"Manifest missing: {p}")
    return json.loads(p.read_text())

def stream_from_cdn(entry: Dict[str, Any], chunk_size=8192):
    """Stream file bytes from CDN (no auth)."""
    resp = requests.get(entry["cdn_url"], stream=True, timeout=30)
    resp.raise_for_status()
    for chunk in resp.iter_content(chunk_size=chunk_size):
        if chunk:
            yield chunk

def pick_sibling_repo(slug: str, n: int = 5, base: str = "vanguard-mirror") -> str:
    """Deterministic sibling routing: base -> base-0..base-(n-1)"""
    digest = int(hashlib.sha256(slug.encode()).hexdigest(), 16)
    idx = digest % n
    return f"{base}-{idx}" if idx > 0 else base

class LightningStudioPool:
    """
    Reuse a running Lightning Studio or restart if stopped.
    Avoids quota burn from repeated create/delete and idle-stop deaths.
    """
    def __init__(self, name: str, machine: Machine = Machine.L40S, teamspace: str = "default"):
        self.name = name
        self.machine = machine
        self.teamspace = Teamspace(teamspace)
        self._studio = None

    def __enter__(self) -> Studio:
        for s in self.teamspace.studios:
            if s.name == self.name:
                self._studio = s
                break
        if self._studio is None:
            self._studio = self.teamspace.studio_create(
                name=self.name,
                machine=self.machine,
                create_ok=True,
            )
        if self._studio.status != "running":
            self._studio.start(machine=self.machine)
        return self._studio

    def __exit__(self, exc_type, exc, tb):
        # Do NOT stop — leave running for reuse.
        pass

    def ensure_running(self) -> bool:
        if self._studio is None:
            return False
        if self._studio.status != "running":
            self._studio.start(machine=self.machine)
            return True
        return True

def cdn_parquet_iterator(entries):
    """Yield record batches from parquet files streamed via CDN (zero HF API calls)."""
    import pyarrow.parquet as pq
    import io

    for entry in entries:
        buf = io.BytesIO()
        for chunk in stream_from_cdn(entry):
            buf.write(chunk)
        buf.seek(0)
        try:
            table = pq.read_table(buf)
            # Project to {prompt, response} only — ignore schema drift
            cols = [c for c in table.column_names if c in ("prompt", "response")]
            if len(cols) < 2:
                continue
            projected = table.select(cols)
            yield projected
        except Exception as exc:
            print(f"Skip {entry['path']}: {exc}")
            continue
