# airship / backend

## Implementation Plan: `airship discover` — Zero-config discovery CLI

**Highest-value incremental improvement (<2h):**  
Ship a single Bash CLI (`airship discover`) that operationalizes past patterns (business research + knowledge-rag + top-hub review) into one deterministic, cron-safe command. It produces a dated insight artifact and exits 0/1 cleanly so cron/logging pipelines work without surprises.

### Why this now
- Reuses existing `granite-business-research.sh` and `knowledge-rag` tooling.
- Applies #top-hub doc insight (MOC/graph) automatically.
- Fixes #script-error/#wrapper pitfalls (shebang, executable, SHELL, Bash invocation).
- Outputs machine-readable + human-readable artifacts under `/opt/axentx/airship/out/` for downstream automation.

### Code

#### `/opt/axentx/airship/lib/airship-lib.sh`

```bash
#!/usr/bin/env bash
# airship-lib.sh — shared helpers for airship CLI
set -euo pipefail

readonly AXENTX_ROOT="/opt/axentx/airship"
readonly OUT_DIR="${AXENTX_ROOT}/out/discover"
readonly TMP_DIR="${AXENTX_ROOT}/tmp"
readonly LOG_FILE="${OUT_DIR}/discover.log"

airship::log() {
  local level="$1"
  shift
  local message="$*"
  local ts
  ts="$(date -u +"%Y-%m-%dT%H:%M:%SZ")"
  echo "${ts} [${level}] ${message}" | tee -a "${LOG_FILE}"
}

airship::ensure_dirs() {
  mkdir -p "${OUT_DIR}" "${TMP_DIR}"
}

airship::lock() {
  local name="$1"
  local lockfile="${TMP_DIR}/${name}.lock"
  if [[ -f "${lockfile}" ]]; then
    local pid
    pid="$(cat "${lockfile}" 2>/dev/null || true)"
    if [[ -n "${pid}" ]] && kill -0 "${pid}" 2>/dev/null; then
      airship::log WARN "Another instance of '${name}' is running (PID=${pid}). Exiting."
      exit 0
    fi
  fi
  echo $$ > "${lockfile}"
  trap 'rm -f "${lockfile}"' EXIT
}

airship::run_script() {
  local script_path="$1"
  shift
  if [[ ! -x "${script_path}" ]]; then
    airship::log ERROR "Script not executable or missing: ${script_path}"
    return 1
  fi
  SHELL=/bin/bash bash "${script_path}" "$@"
}

# Top-hub extraction from knowledge-rag outputs (best-effort)
airship::top_hub() {
  local base_dir="$1"
  local hub_file
  hub_file="$(find "${base_dir}" -type f -name "*.md" -o -name "*.json" | head -n 20 | xargs grep -l -i "hub\|MOC\|centrality" 2>/dev/null | head -n 1 || true)"
  if [[ -n "${hub_file}" ]]; then
    airship::log INFO "Top hub candidate file: ${hub_file}"
    # Extract short snippet around MOC/hub mentions
    grep -i -C 3 -E "hub|MOC|centrality|most.connected" "${hub_file}" 2>/dev/null | head -n 30 || true
  else
    airship::log INFO "No obvious top-hub file found; listing latest docs instead."
    find "${base_dir}" -type f -name "*.md" -printf "%T@ %p\n" | sort -rn | head -n 5 | cut -d' ' -f2- || true
  fi
}
```

#### `/opt/axentx/airship/bin/airship`

```bash
#!/usr/bin/env bash
# airship — DevOps/AI platform CLI
# Usage: airship <command> [args...]
set -euo pipefail

readonly AXENTX_ROOT="/opt/axentx/airship"
readonly LIB="${AXENTX_ROOT}/lib/airship-lib.sh"

if [[ -f "${LIB}" ]]; then
  # shellcheck source=/dev/null
  source "${LIB}"
else
  echo "ERROR: missing library ${LIB}" >&2
  exit 1
fi

readonly CMD="${1:-}"
shift || true

airship::discover() {
  local run_id
  run_id="$(date -u +"%Y%m%dT%H%M%SZ")"
  local run_dir="${OUT_DIR}/${run_id}"
  local report_file="${run_dir}/insights.md"

  airship::ensure_dirs
  airship::lock "airship-discover"
  mkdir -p "${run_dir}"

  airship::log INFO "Starting discovery run ${run_id}"

  # 1) Business research
  airship::log INFO "Running business research..."
  local research_out="${run_dir}/business-research"
  mkdir -p "${research_out}"
  if airship::run_script "${AXENTX_ROOT}/scripts/granite-business-research.sh" "${research_out}"; then
    airship::log INFO "Business research completed."
  else
    airship::log WARN "Business research script returned non-zero; continuing."
  fi

  # 2) Knowledge RAG indexing/query
  airship::log INFO "Running knowledge-rag..."
  local rag_out="${run_dir}/knowledge-rag"
  mkdir -p "${rag_out}"
  # If knowledge-rag is a CLI/script, invoke it here.
  # Example placeholder:
  if command -v knowledge-rag >/dev/null 2>&1; then
    if knowledge-rag --input "${research_out}" --output "${rag_out}" --top-k 10; then
      airship::log INFO "knowledge-rag completed."
    else
      airship::log WARN "knowledge-rag returned non-zero; continuing."
    fi
  else
    airship::log WARN "knowledge-rag command not found; skipping RAG step."
    # Create minimal placeholder so downstream steps don't fail
    echo '{"hubs":[],"docs":[]}' > "${rag_out}/result.json"
  fi

  # 3) Top-hub review
  airship::log INFO "Extracting top-hub insights..."
  {
    echo "# Discovery Run: ${run_id}"
    echo ""
    echo "## Business Research"
    find "${research_out}" -type f -name "*.md" -exec echo "### {}" \; -exec cat {} \; 2>/dev/null || echo "_No markdown outputs_"
    echo ""
    echo "## Knowledge RAG — Top Hub & Docs"
    if [[ -f "${rag_out}/result.json" ]]; then
      cat "${rag_out}/result.json"
    fi
    echo ""
    echo "## Top Hub Insight"
    airship::top_hub "${rag_out}"
    echo ""
    echo "## Raw Outputs"
    echo "- Business research: ${research_out}"
    echo "- Knowledge RAG: ${rag_out}"
  } > "${report_file}"

  # 4) Produce stable latest symlink
  local latest_link="${OUT_DIR}/latest"
  ln -sfn "${run_dir}" "${latest_link}"

  airship::log INFO "Discovery complete. Report: ${report_file}"
  airship::log INFO "Latest symlink: ${latest_link}"
  echo "${report_file}"
}

case "${CMD}" in
  discover)
    airship::discover
    ;;
  *)
    echo "Usage: airship <discover>"
    exit 1
    ;;
esac
```
