# workio / quality

## 1) Diagnosis

- **No idempotency enforcement** on clock-in/out: duplicate LINE webhook deliveries and client retries can create multiple punches for the same `(employee_id, punch_type, date)` → payroll drift.
- **Non-transactional punch writes**: concurrent requests can insert two punches before either sees the other (read-then-insert race).
- **No unique constraint** at DB level to prevent duplicates; relies on app logic only.
- **Missing deduplication key**: no `idempotency_key` or `request_id` stored with punches to detect retries.
- **Clock-out logic tolerant of duplicates**: multiple “out” punches for same day corrupt total hours and OT calculations.

## 2) Proposed change

- **File**: `workio/server/src/db/schema.sql` — add unique constraint on punches.
- **File**: `workio/server/src/db/queries/punch.js` (or equivalent) — add idempotency check and transactional upsert.
- **File**: `workio/server/src/routes/punch.js` (or webhook handler) — generate idempotency key from LINE `webhookEvent.id` or client-supplied `idempotencyKey` header and enforce single write.

## 3) Implementation

### 3.1 DB schema — add unique constraint + idempotency index

```sql
-- workio/server/src/db/schema.sql
-- Add unique constraint to prevent duplicate punches per employee/day/type
ALTER TABLE punches
  ADD CONSTRAINT uq_punch_employee_date_type
  UNIQUE (employee_id, punch_date, punch_type);

-- Optional: store idempotency keys for external retries (webhook + client)
CREATE TABLE IF NOT EXISTS punch_idempotency (
  idempotency_key TEXT PRIMARY KEY,
  punch_id        INTEGER NOT NULL REFERENCES punches(id) ON DELETE CASCADE,
  created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
```

### 3.2 Query layer — transactional upsert with idempotency

```js
// workio/server/src/db/queries/punch.js
const { pool } = require('./index');

async function createPunchWithIdempotency({
  employeeId,
  punchType,           // 'in' | 'out'
  punchDate,           // YYYY-MM-DD
  timestamp,           // full timestamp for time value
  location,            // { lat, lng }
  idempotencyKey,
  notes
}) {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // If idempotency key provided, return existing punch
    if (idempotencyKey) {
      const existing = await client.query(
        `SELECT p.* FROM punch_idempotency pi
         JOIN punches p ON p.id = pi.punch_id
         WHERE pi.idempotency_key = $1`,
        [idempotencyKey]
      );
      if (existing.rows.length > 0) {
        await client.query('ROLLBACK');
        return existing.rows[0];
      }
    }

    // Try insert; if unique violation, return existing row
    let result;
    try {
      result = await client.query(
        `INSERT INTO punches (employee_id, punch_type, punch_date, timestamp, location, notes)
         VALUES ($1, $2, $3, $4, $5, $6)
         RETURNING *`,
        [employeeId, punchType, punchDate, timestamp, JSON.stringify(location), notes]
      );
    } catch (err) {
      if (err.code === '23505') { // unique_violation
        result = await client.query(
          `SELECT * FROM punches
           WHERE employee_id = $1 AND punch_date = $2 AND punch_type = $3`,
          [employeeId, punchDate, punchType]
        );
      } else {
        throw err;
      }
    }

    const punch = result.rows[0];

    // Store idempotency key if provided
    if (idempotencyKey) {
      await client.query(
        `INSERT INTO punch_idempotency (idempotency_key, punch_id)
         VALUES ($1, $2)`,
        [idempotencyKey, punch.id]
      );
    }

    await client.query('COMMIT');
    return punch;
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  } finally {
    client.release();
  }
}

module.exports = { createPunchWithIdempotency };
```

### 3.3 Route/webhook handler — generate idempotency key

```js
// workio/server/src/routes/punch.js
const express = require('express');
const router = express.Router();
const { createPunchWithIdempotency } = require('../db/queries/punch');

// For LINE webhook: use event.id as idempotency key
router.post('/line', async (req, res) => {
  try {
    const { events } = req.body;
    if (!events || !Array.isArray(events)) return res.sendStatus(400);

    for (const ev of events) {
      if (ev.type === 'message' && ev.message.type === 'text') {
        // Example: parse "in" or "out" from text or use payload from template
        const text = ev.message.text.toLowerCase();
        const punchType = text.includes('in') ? 'in' : text.includes('out') ? 'out' : null;
        if (!punchType) continue;

        const now = new Date();
        const punchDate = now.toISOString().split('T')[0];

        await createPunchWithIdempotency({
          employeeId: ev.source.userId, // map via LINE userId -> employee_id in your tenant
          punchType,
          punchDate,
          timestamp: now.toISOString(),
          location: null, // parse location if available
          idempotencyKey: `line:${ev.id}`,
          notes: `LINE webhook ${ev.id}`
        });
      }
    }

    res.sendStatus(200);
  } catch (err) {
    console.error(err);
    res.sendStatus(500);
  }
});

// For client API: accept Idempotency-Key header
router.post('/clock', async (req, res) => {
  try {
    const { employeeId, punchType, location } = req.body;
    const idempotencyKey = req.get('Idempotency-Key') || req.get('idempotency-key');
    const now = new Date();
    const punchDate = now.toISOString().split('T')[0];

    const punch = await createPunchWithIdempotency({
      employeeId,
      punchType,
      punchDate,
      timestamp: now.toISOString(),
      location,
      idempotencyKey,
      notes: idempotencyKey ? `client idempotent ${idempotencyKey}` : 'client'
    });

    res.json(punch);
  } catch (err) {
    console.error(err);
    res.status(500).json({ error: err.message });
  }
});

module.exports = router;
```
