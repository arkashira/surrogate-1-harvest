# surrogate-1 / ops

## Implementation Plan — CDN-first snapshot + zero-HF-API ingestion (surrogate-1)

**Highest-value incremental improvement (<2h)**  
Add a Mac-side snapshot script that:
1. Lists one date folder in `axentx/surrogate-1-training-pairs` (single API call)  
2. Produces `manifests/{date}/file-manifest.json` containing `{ "path": str, "size": int, "etag": str, "url": str }`  
3. Embeds that manifest into the Lightning training script so data loading uses **only** CDN URLs (`https://huggingface.co/datasets/.../resolve/main/...`) with **zero** HF API calls during training.

This directly applies the CDN-bypass pattern and avoids `load_dataset(streaming=True)` on heterogeneous schemas.

### 1) Snapshot script (run on Mac)

`scripts/snapshot_hf_manifest.py`
```python
#!/usr/bin/env python3
"""
Create a deterministic file manifest for one date folder in
axentx/surrogate-1-training-pairs.

Usage:
    HF_TOKEN=<token> python snapshot_hf_manifest.py 2026-05-03

Produces:
    manifests/2026-05-03/file-manifest.json
"""
import os
import json
import sys
from datetime import datetime
from pathlib import Path

from huggingface_hub import HfApi, hf_hub_url

REPO_ID = "datasets/axentx/surrogate-1-training-pairs"
OUT_DIR = Path(__file__).parent.parent / "manifests"

def main(date_folder: str) -> None:
    api = HfApi(token=os.environ.get("HF_TOKEN"))
    folder = f"batches/public-merged/{date_folder}"

    # Single API call: non-recursive tree listing
    entries = api.list_repo_tree(
        repo_id=REPO_ID,
        path=folder,
        repo_type="dataset",
        recursive=False,
    )

    files = [e for e in entries if e.type == "file"]
    if not files:
        print(f"No files found under {folder}")
        sys.exit(1)

    manifest = []
    for f in sorted(files, key=lambda x: x.path):
        url = hf_hub_url(
            repo_id=REPO_ID,
            filename=f.path,
            repo_type="dataset",
        )
        manifest.append({
            "path": f.path,
            "size": f.size,
            "etag": f.etag,
            "url": str(url),
        })

    out_dir = OUT_DIR / date_folder
    out_dir.mkdir(parents=True, exist_ok=True)
    out_path = out_dir / "file-manifest.json"
    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"Wrote {len(manifest)} files -> {out_path}")

if __name__ == "__main__":
    if len(sys.argv) != 2:
        print("Usage: snapshot_hf_manifest.py <YYYY-MM-DD>")
        sys.exit(1)
    main(sys.argv[1])
```

Make executable:
```bash
chmod +x scripts/snapshot_hf_manifest.py
```

### 2) Lightning training loader (CDN-only)

`training/train.py` (excerpt)
```python
import json
from pathlib import Path
from typing import List, Dict

import pyarrow as pa
import pyarrow.parquet as pq
import requests
from torch.utils.data import IterableDataset

class CDNParquetDataset(IterableDataset):
    """
    Load parquet shards via CDN URLs only (no HF API during training).
    Manifest is produced once by snapshot_hf_manifest.py on the Mac.
    """
    def __init__(self, manifest_path: str | Path, max_shards: int = None):
        manifest_path = Path(manifest_path)
        with manifest_path.open() as f:
            self.files: List[Dict] = json.load(f)

        if max_shards:
            self.files = self.files[:max_shards]

    def __iter__(self):
        for item in self.files:
            url = item["url"]
            # Stream parquet from CDN
            with requests.get(url, stream=True) as r:
                r.raise_for_status()
                # Write to temp buffer (or memory-map via pyarrow)
                table = pq.read_table(pa.BufferReader(r.content))
                # Project to {prompt, response} only (schema heterogeneity handled here)
                batch = table.select(["prompt", "response"])
                for row in batch.to_pylist():
                    yield row
```

### 4) Orchestration (Mac)

`scripts/run_cycle.sh`
```bash
#!/usr/bin/env bash
set -euo pipefail
export SHELL=/bin/bash

DATE=$(date +%Y-%m-%d)
MANIFEST="manifests/${DATE}/file-manifest.json"

# 1) Create snapshot (single API call)
HF_TOKEN=${HF_TOKEN:?required} python scripts/snapshot_hf_manifest.py "$DATE"

# 2) Launch Lightning training (CDN-only)
cd training
HF_TOKEN=${HF_TOKEN} python launch_studio.py --manifest "../${MANIFEST}"
```
