# vanguard / ops

## 1. Diagnosis
- No durable ingestion manifest → every training run re-lists HF repos and re-downloads, causing 429s and wasted time/bandwidth.
- Training loads via HF API (`load_dataset`/`list_repo_files`) instead of CDN bypass → guaranteed rate limits during data loading.
- No Lightning Studio reuse logic → each run recreates/kills Studio, burning 80+ hrs/mo quota and risking idle-timeout deaths.
- Missing pre-flight file-list snapshot → training cannot run CDN-only; Mac must embed a stable file manifest for Lightning.
- No retry/backoff for HF 429 or commit-cap 128/hr → ingestion fails unpredictably on big repos.

## 2. Proposed change
Create `/opt/axentx/vanguard/ops/ingest_state.py` and update `/opt/axentx/vanguard/ops/train_launcher.py` (or create if absent) to:
- Snapshot repo file tree once per date-folder to `state/YYYYMMDD_filelist.json`.
- Embed that file list in training so Lightning does CDN-only fetches (zero API calls during training).
- Reuse a running Studio by name before creating a new one.
- Add exponential backoff + 360s sleep on HF 429.

```python
#!/usr/bin/env python3
"""
Durable ingestion state: snapshot HF repo file tree and produce CDN file list.
Usage:
    python ingest_state.py --repo datasets/mycorp/surrogate-1 --date 2026-05-02 --out-dir state
"""
import argparse
import json
import os
import sys
import time
from datetime import datetime
from pathlib import Path
from typing import List, Dict, Any

import requests

HF_API_BASE = "https://huggingface.co/api"
CDN_BASE = "https://huggingface.co/datasets"
RETRY_BACKOFF = [1, 2, 4, 8, 16, 32, 64, 128, 256, 360]  # seconds; final wait for 429

def hf_get(path: str, token: str = None, max_retries: int = 10) -> Dict[str, Any]:
    headers = {}
    if token:
        headers["Authorization"] = f"Bearer {token}"
    url = f"{HF_API_BASE}{path}"
    for attempt in range(max_retries):
        resp = requests.get(url, headers=headers, timeout=30)
        if resp.status_code == 429:
            wait = RETRY_BACKOFF[min(attempt, len(RETRY_BACKOFF) - 1)]
            print(f"[ingest_state] HF 429, waiting {wait}s (attempt {attempt+1})", file=sys.stderr)
            time.sleep(wait)
            continue
        resp.raise_for_status()
        return resp.json()
    raise RuntimeError(f"HF API failed after {max_retries} retries: {resp.text}")

def snapshot_repo_files(repo: str, date_folder: str, token: str = None) -> List[str]:
    """
    List files under repo/{date_folder}/ (non-recursive by folder) and return CDN URLs.
    Uses HF API only once per snapshot.
    """
    # List top-level tree for the date folder (non-recursive)
    tree = hf_get(f"/datasets/{repo}/tree/{date_folder}", token=token)
    files = []
    for entry in tree:
        if entry.get("type") == "file":
            cdn_url = f"{CDN_BASE}/{repo}/resolve/main/{date_folder}/{entry['name']}"
            files.append(cdn_url)
    return files

def save_manifest(repo: str, date_folder: str, file_urls: List[str], out_dir: Path):
    out_dir.mkdir(parents=True, exist_ok=True)
    manifest = {
        "repo": repo,
        "date_folder": date_folder,
        "snapshot_ts": datetime.utcnow().isoformat() + "Z",
        "files": file_urls,
        "note": "CDN-only file list for Lightning training (zero HF API calls during data load)"
    }
    out_path = out_dir / f"{date_folder.replace('/', '_')}_filelist.json"
    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"[ingest_state] Saved manifest to {out_path}")
    return out_path

def main():
    parser = argparse.ArgumentParser(description="Snapshot HF repo file list for CDN-only training.")
    parser.add_argument("--repo", required=True, help="HF dataset repo, e.g. datasets/mycorp/surrogate-1")
    parser.add_argument("--date", required=True, help="Date folder, e.g. 2026-05-02 or batches/mirror-merged/2026-05-02")
    parser.add_argument("--out-dir", default="state", help="Output directory for manifests")
    parser.add_argument("--hf-token", default=os.getenv("HF_TOKEN"), help="HF token (optional for public repos)")
    args = parser.parse_args()

    repo = args.repo.lstrip("datasets/").rstrip("/")
    date_folder = args.date.strip("/")
    file_urls = snapshot_repo_files(repo, date_folder, token=args.hf_token)
    if not file_urls:
        print("[ingest_state] WARNING: no files found; verify date_folder path.", file=sys.stderr)
    save_manifest(repo, date_folder, file_urls, Path(args.out_dir))

if __name__ == "__main__":
    main()
```
