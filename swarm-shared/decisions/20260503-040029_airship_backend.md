## `scripts/list_hf_files.py`

```python
#!/usr/bin/env python3
"""
List parquet files in a HuggingFace dataset folder (non-recursive)
and save paths to JSON for CDN-only training.

Usage:
    python scripts/list_hf_files.py \
        --repo "org/surrogate-dataset" \
        --folder "batches/mirror-merged/2026-05-03" \
        --out "scripts/filelist.json"
"""
import argparse
import json
import os
import sys
from typing import List, Dict

try:
    from huggingface_hub import HfApi
except ImportError:
    print("Install: pip install huggingface_hub")
    sys.exit(1)

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def build_filelist(repo: str, folder: str) -> List[Dict[str, str]]:
    api = HfApi()
    entries = api.list_repo_tree(repo_id=repo, path=folder, recursive=False)
    files = []
    for entry in entries:
        if not entry.path.endswith(".parquet"):
            continue
        files.append({
            "repo": repo,
            "path": entry.path,
            "cdn_url": CDN_TEMPLATE.format(repo=repo, path=entry.path),
        })
    return files

def main() -> None:
    parser = argparse.ArgumentParser(description="List HF folder for CDN training")
    parser.add_argument("--repo", required=True, help="HF dataset repo (org/name)")
    parser    .add_argument("--folder", required=True, help="Folder path in repo")
    parser.add_argument("--out", default="scripts/filelist.json", help="Output JSON path")
    args = parser.parse_args()

    files = build_filelist(repo=args.repo, folder=args.folder)
    if not files:
        print("No parquet files found.")
        sys.exit(1)

    out_path = os.path.abspath(args.out)
    os.makedirs(os.path.dirname(out_path) or ".", exist_ok=True)
    with open(out_path, "w") as f:
        json.dump(files, f, indent=2)

    print(f"Wrote {len(files)} parquet files to {out_path}")

if __name__ == "__main__":
    main()
```

Make executable:
```bash
chmod +x scripts/list_hf_files.py
```

def make_session() -> requests.Session:
    session = requests.Session()
    retries = Retry(
        total=CDN_RETRY,
        backoff_factor=1,
        status_forcelist=[429, 500, 502, 503, 504],
        allowed_methods={"GET"},
    )
    session.mount("https://", HTTPAdapter(max_retries=retries))
    return session

# ------------------------------
# CDN Iterable Dataset
# ------------------------------
class CDNParquetDataset(IterableDataset):
    """
    Stream parquet files via CDN URLs.
    Each row must contain {prompt, response} (projected at parse time).
    """
    def __init__(self, filelist_path: str, session: requests.Session):
        super().__init__()
        with open(filelist_path) as f:
            self.files = json.load(f)
        if not self.files:
            raise ValueError("No parquet files found in filelist")
        self.session = session
        self.required_columns = {"prompt", "response"}

    def _stream_parquet(self, url: str) -> bytes:
        with self.session.get(url, stream=True, timeout=CDN_TIMEOUT) as r:
            r.raise_for_status()
            buf = bytearray()
            for chunk in r.iter_content(chunk_size=16384):
                buf.extend(chunk)
            return bytes(buf)

    def __iter__(self) -> Iterator[Dict[str, Any]]:
        import pyarrow.parquet as pq
        import io

        for item in self.files:
            url = item["cdn_url"]

                raw = self._stream_parquet(url)
