# Costinel / quality

## Implementation Plan — Top Hub Signal Panel (CDN-first, frontend-only)

**Scope**: Add a lightweight, resilient Top Hub signal card to the Costinel dashboard that surfaces the most-connected hub (e.g., "MOC") with contextual insights.  
**Effort**: ~60–90 minutes (frontend only).  
**Mechanism**:  
- Single orchestration call (Mac/CI) to `list_repo_tree` once per day → save `top-hub.json` to repo or CDN.  
- Frontend fetches `top-hub.json` via CDN (`/resolve/main/...`) — zero API calls during runtime, bypasses 429 limits.  
- Graceful fallback UI when data is unavailable.  
- Tags: #knowledge-rag #graph #hub #cdn #rate-limit-bypass

### 3) Types (`src/types/index.ts`)

export interface TopHubInsight {
  id: string;
  text: string;
  severity: 'low' | 'medium' | 'high';
  tags: string[];
}

export interface TopHub {
  id: string;
  label: string;
  description: string;
  connections: number;
  score: number;
  updatedAt: string;
}

export interface TopHubPayload {

### 4) Component (`src/components/TopHubSignalCard.tsx`)

```tsx
import React, { useEffect, useState } from 'react';
import { TopHubPayload } from '../types';

const CDN_TOP_HUB_URL = '/data/top-hub.json';

const severityColors = {
  low: 'bg-blue-100 text-blue-800',
  medium: 'bg-yellow-100 text-yellow-800',
  high: 'bg-red-100 text-red-800',
} as const;

export const TopHubSignalCard: React.FC = () => {
  const [payload, setPayload] = useState<TopHubPayload | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    let mounted = true;
    setLoading(true);

    fetch(CDN_TOP_HUB_URL, { cache: 'no-store' })
      .then((res) => {
        if (!res.ok) throw new Error(`HTTP ${res.status}`);
        return res.json();
      })
      .then((data: TopHubPayload) => {
        if (mounted) {
          setPayload(data);
          setError(null);
        }
      })
      .catch((err) => {
        if (mounted) {
          setError(err.message || 'Failed to load top hub');
          setPayload(null);
        }
      })
      .finally(() => {
        if (mounted) setLoading(false);
      });

    return () => {
      mounted = false;
    };
  }, []);

  if (loading) {
    return (
      <div className="rounded-lg border bg-white p-4 shadow-sm">
        <div className="h-6 w-32 animate-pulse rounded bg-gray-200" />
        <div className="mt-3 h-4 w-24 animate-pulse rounded bg-gray-100" />
      </div>
    );
  }

  if (error || !payload) {
    return (
      <div className="rounded-lg border bg-white p-4 shadow-sm">
        <p className="text-sm text-gray-500">Top hub signal unavailable.</p>
      </div>
    );
  }

  const { hub, insights } = payload;

  return (
    <div className="rounded-lg border bg-white p-4 shadow-sm">
      <div className="flex items-start justify-between">
        <div>
          <h3 className="text-base font-semibold text-gray-900">{hub.label}</h3>
          <p className="text-xs text-gray-500">{hub.description}</p>
        </div>
        <span className="inline-flex items-center rounded-full bg-emerald-50 px-2 py-1 text-xs font-medium text-emerald-700">
          {hub.connections} links
        </span>
      </div>

      <div className="mt-3 flex items-center gap-2">
        <span className="text-xs text-gray-400">Score</span>
        <div className="h-2 w-20 rounded-full bg-gray-100">
          <div
            className="h-full rounded-full bg-emerald-500 transition-all"
            style={{ width: `${Math.min(100, Math.max(0, hub.score * 100))}%` }}
          />
        </div>
        <span className="text-xs font-medium text-gray-700">{hub.score.toFixed(2)}</span>
      </div>

      {insights.length > 0 && (
        <div className="mt-4 space-y-2">
          {insights.map((insight) => (
            <div key={insight.id} className="rounded border p-2">
              <div className="flex items-start justify-between gap-2">
                <p className="text-sm text-gray-800">{insight.text}</p>
                <span className={`flex-shrink-0 rounded px-1.5 py-0.5 text-xs font-medium ${severityColors[insight.severity]}`}>
                  {insight.severity}
                </span>
              </div>
              <div className="mt-2 flex flex-wrap gap-1">
                {insight.tags.map((tag) => (
                  <span
                    key={tag}
                    className="rounded bg-gray-50 px-1.5 py-0.5 text-xs text-gray-600"
                  >
                    {tag}
                  </span>
                ))}
              </div>
            </div>
          ))}
        </div>
      )}

      <p className="mt-3 text-xs text-gray-400">
        Updated {new Date(hub.updatedAt).toLocaleString()}
      </p>
    </div>
  );
};
```
