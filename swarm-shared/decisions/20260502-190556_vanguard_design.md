### 1) Diagnosis (resolved)
- **Missing canonical discovery entrypoint** → must surface top-hub (MOC) insights before planning (`#knowledge-rag #graph #hub`).
- **HF ingestion will hit API rate limits** → need CDN-bypass file-list using `resolve/main/` paths and tree pagination (not recursive per-file calls).
- **Lightning Studio reuse missing** → must detect running studio and reuse; never recreate blindly and burn quota.
- **Schema pollution risk** → local `data/raw/` must be projected to strict `{prompt, response}` before any downstream use.
- **No single orchestration** → one executable workflow that composes discovery, CDN file-list, studio reuse, and projection.

2. `list_hf_files.py` — CDN-bypass file list (tree pagination, 429 handling, deterministic).
3. `reuse_studio.py` — detect or create studio, emit status (safe defaults).
4. `project_local.py` — project `data/raw/` to `{prompt, response}` parquet.

All outputs go to `discovery/`:
- `top_hub_insights.md`
- `hf_file_list.json`

#!/usr/bin/env bash

cd /opt/axentx/vanguard

OUTDIR="discovery"
mkdir -p "$OUTDIR"

# Top-hub insights (MOC)
- Review most-connected hub (MOC) before planning tasks.
- Use knowledge-rag to query related docs for contextual insights.
- Tags: #knowledge-rag #graph #hub
EOF
fi

echo "2) HF CDN-bypass file list..."
python3 "$OUTDIR/list_hf_files.py" \
  --repo "datasets/your-org/your-dataset" \
  --date-folder "batches/mirror-merged/$(date +%Y-%m-%d)" \
  --out "$OUTDIR/hf_file_list.json"

echo "3) Lightning Studio reuse..."
python3 "$OUTDIR/reuse_studio.py" \
  --name "vanguard-train" \
  --machine "L40S" \
  --out "$OUTDIR/studio_status.json"

python3 "$OUTDIR/project_local.py" \
  --in-dir "data/raw" \
  --out "$OUTDIR/projected.parquet"

echo "== discovery complete =="

echo "HF file list (CDN): $OUTDIR/hf_file_list.json"
echo "Studio status: $OUTDIR/studio_status.json"
echo "Projected data: $OUTDIR/projected.parquet"

# /opt/axentx/vanguard/discovery/list_hf_files.py
#!/usr/bin/env python3
"""
Generate CDN-bypass file list for a Hugging Face dataset folder.
Uses tree pagination (non-recursive per folder) to avoid rate limits.
Emits CDN URLs via resolve/main/ for training ingestion.
"""
import argparse
import json
import os
import sys
import time
from typing import List, Dict, Any

try:
    from huggingface_hub import HfApi, login
except ImportError as e:
    print("Error: huggingface_hub required. Install via: pip install huggingface_hub", file=sys.stderr)
    sys.exit(1)

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def list_hf_files(repo: str, date_folder: str, retries: int = 3, sleep_base: int = 5) -> List[Dict[str, str]]:
    api = HfApi()
    token = os.getenv("HF_TOKEN")
    if token:
        login(token=token, add_to_git_credential=False)

    prefix = date_folder.rstrip("/")
    all_files: List[Dict[str, str]] = []
    cursor = None

    while True:
        attempt = 0
        tree = None
        while attempt < retries:
            try:
                tree = api.list_repo_tree(
                    repo=repo,
                    path=prefix,
                    repo_type="dataset",
                    recursive=True,
                    cursor=cursor,
                )
                break
            except Exception as e:
                attempt += 1
                if attempt >= retries:
                    raise
                # Respect 429 backoff
                if getattr(e, "response", None) and e.response.status_code == 429:
                    time.sleep(360)
                else:
                    time.sleep(sleep_base * attempt)

        if tree is None or not list(tree):
            break

        for item in tree:
            if item.type == "file" and item.path.startswith(prefix + "/") or item.path == prefix:
                all_files.append({
                    "repo": repo,
                    "path": item.path,
                    "cdn_url": CDN_TEMPLATE.format(repo=repo, path=item.path),
                })

        cursor = getattr(tree, "cursor", None)
        if not cursor:
            break

    all_files.sort(key=lambda x: x["path"])
    return all_files

def main() -> None:
    parser = argparse.ArgumentParser(description="Generate HF CDN-bypass file list")
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g., datasets/your-org/your-dataset)")
    parser.add_argument("--date-folder", required=True, help="Folder path in dataset (e.g., batches/mirror-merged/2026-05-02)")
    parser.add_argument("--out", required=True, help="Output JSON path")
    args = parser.parse_args()

    entries = list_hf_files(args.repo, args.date_folder)
    os.makedirs(os.path.dirname(os.path.abspath(args.out)), exist_ok=True)
    with open(args.out, "w", encoding="utf-8") as f:
        json.dump(entries, f, indent=2)
    print(f"Wrote {len(entries)} files to {args.out}")

if __name__ == "__main__":
    main()

"""
Reuse a running Lightning Studio or report status.
Avoids recreating studios and burning quota.
"""

import sys
from typing import Optional, Any, Dict

LIGHTNING_AVAILABLE = False
try:
    from lightning import Studio, Teamspace, Machine  # type: ignore
    LIGHTNING_AVAILABLE = True
except Exception:
    # Fallback for environments without lightning SDK
    Studio = Teamspace = Machine = None  # type: ignore

def find_running_studio(name: str) -> Optional[Any]:
    if Teamspace is None:
        return None
    try:
        for s in Teamspace.studios():
            if getattr(s, "name", None) == name:
                return s
    except Exception:
        return None
    return None
