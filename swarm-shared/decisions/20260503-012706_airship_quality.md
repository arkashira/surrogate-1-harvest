### 1) CDN-bypass dataset loader  
`surrogate/data/hf_cdn_dataset.py`
```python
# surrogate/data/hf_cdn_dataset.py
import json
from pathlib import Path
from typing import Dict, Iterator, Optional

import httpx
import pyarrow as pa
import pyarrow.parquet as pq
from torch.utils.data import IterableDataset

HF_CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

class HFCDNDataset(IterableDataset):
    """
    HF CDN-bypass dataset loader.
    Precondition: filelist_json contains list of dicts:
      [{"repo": "...", "path": "batches/mirror-merged/2026-05-01/slug.parquet"}, ...]
    Only 'prompt' and 'response' columns are projected at parse time.
    """

    def __init__(
        self,
        filelist_path: str,
        repo_column: str = "repo",
        path_column: str = "path",
        prompt_column: str = "prompt",
        response_column: str = "response",
        client: Optional[httpx.Client] = None,
        timeout: float = 30.0,
    ):
        self.filelist_path = Path(filelist_path)
        self.repo_column = repo_column
        self.path_column = path_column
        self.prompt_column = prompt_column
        self.response_column = response_column
        self.timeout = timeout
        self._client = client or httpx.Client(timeout=timeout)

        with open(self.filelist_path, "r") as f:
            self.files = json.load(f)
        if not isinstance(self.files, list) or not self.files:
            raise ValueError("filelist must be a non-empty list")

    def _download_parquet(self, repo: str, path: str) -> bytes:
        url = HF_CDN_TEMPLATE.format(repo=repo.strip("/"), path=path.lstrip("/"))
        resp = self._client.get(url)
        resp.raise_for_status()
        return resp.content

    def _parse_parquet(self, data: bytes) -> Iterator[Dict[str, str]]:
        table = pq.read_table(pa.BufferReader(data))
        cols = set(table.column_names)
        prompt_col = self.prompt_column if self.prompt_column in cols else None
        response_col = self.response_column if self.response_column in cols else None

        if not prompt_col or not response_col:
            # skip files missing required fields instead of crashing
            return

        prompts = table[prompt_col].to_pylist()
        responses = table[response_col].to_pylist()
        for p, r in zip(prompts, responses):
            if p is None or r is None:
                continue
            yield {self.prompt_column: str(p), self.response_column: str(r)}

    def __iter__(self) -> Iterator[Dict[str, str]]:
        for item in self.files:
            repo = item[self.repo_column]
            path = item[self.path_column]
            try:
                data = self._download_parquet(repo, path)
                yield from self._parse_parquet(data)
            except Exception as exc:
                # Log and skip bad files to keep pipeline robust
                print(f"Skipping {repo}/{path}: {exc}")
                continue

    def __del__(self):
        if hasattr(self, "_client"):
            self._client.close()
```

### 2) Lightning Studio reuse helper  
`surrogate/train/lightning_utils.py`
```python
# surrogate/train/lightning_utils.py
from typing import Optional

from lightning import LightningWork, Machine

def get_or_create_studio(
    name: str,
    machine: str = "L40S",
    cloud: str = "lightning-public-prod",
    shutdown_after_done: bool = False,
) -> LightningWork:
    """
    Reuse a running Lightning Studio if found; otherwise create one.
    Avoids recreating studios to save quota.
    """
    # List running studios in current teamspace
    for studio in LightningWork._running_studios or []:
        if getattr(studio, "name", None) == name and getattr(studio, "status", None) == "Running":
            print(f"Reusing running studio: {name}")
            return studio

    print(f"No running studio '{name}' found. Creating on {cloud}/{machine}.")

    class TrainingStudio(LightningWork):
        def run(self, *args, **kwargs):
            import subprocess
            cmd = ["python", "train.py"] + list(args)
            subprocess.run(cmd, check=True)

    return TrainingStudio(
        name=name,
        machine=Machine(machine, cloud=cloud),
        shutdown_after_done=shutdown_after_done,
    )
```
