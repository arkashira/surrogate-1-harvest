# vanguard / design

## Final Synthesized Design (Best of Both Candidates)

**Core Principle**: Eliminate all Hugging Face API calls at training time. Replace them with a deterministic, content-addressed CDN manifest generated once per date snapshot. Enforce remote-only compute and studio reuse to prevent quota exhaustion and local hardware misuse.

---

### 1. Diagnosis (Consolidated)
- **No CDN-first manifest**: Training scripts still call `list_repo_tree`/`load_dataset` at runtime → 429s and non-reproducible runs.
- **No content-addressed snapshots**: Training jobs cannot pin exact file sets → drift between runs.
- **No guardrails**: Mac-local `from_pretrained` or heavy compute can occur; orchestration boundaries are implicit.
- **No Lightning Studio reuse**: Each run likely recreates a studio, burning quota.
- **No idle-stop fallback**: Training dies if studio stops before `.run()`.

---

### 2. Proposed Change
Create `/opt/axentx/vanguard/design/` with:
- `cdn_manifest.py` — one-shot script to list a date folder and emit `manifest.json` (repo, date, file paths, sha256, size, ETag).
- `train_launcher.py` — Lightning launcher that reuses a running studio, validates idle state, and passes **only CDN URLs** to training.
- `guardrails.sh` — Mac-side wrapper that blocks local model downloads and enforces remote-only compute.

---

#### cdn_manifest.py
```python
#!/usr/bin/env python3
"""
Generate a deterministic CDN-first manifest for a HuggingFace dataset repo folder.
Run on Mac (or any low-rate-limit machine) once per date folder.

Usage:
  python cdn_manifest.py --repo axentx/surrogate-1 --date 2026-04-29 --out manifest.json
"""
import argparse
import hashlib
import json
import os
import sys
from datetime import datetime, timezone

try:
    from huggingface_hub import HfApi, hf_hub_download
except ImportError:
    print("Install: pip install huggingface_hub")
    sys.exit(1)

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def sha256_file(path: str) -> str:
    h = hashlib.sha256()
    with open(path, "rb") as f:
        for chunk in iter(lambda: f.read(8192), b""):
            h.update(chunk)
    return h.hexdigest()

def build_manifest(repo: str, date: str, out_path: str, local_cache_dir: str = "/tmp/hf_cache"):
    os.makedirs(local_cache_dir, exist_ok=True)
    api = HfApi()
    folder = f"{date}"
    print(f"Listing repo tree: {repo}/{folder}")

    try:
        items = api.list_repo_tree(repo=repo, path=folder, recursive=False, repo_type="dataset")
    except Exception as e:
        print(f"Failed to list repo tree: {e}")
        sys.exit(1)

    files = []
    for item in items:
        if item.type != "file":
            continue
        path = item.path
        cdn_url = CDN_TEMPLATE.format(repo=repo, path=path)

        # Try to get ETag/size via repo file metadata; fall back to local download if needed
        try:
            metadata = api.get_paths_info(repo=repo, paths=[path], repo_type="dataset")
            meta = metadata[0] if metadata else None
            size = getattr(meta, "size", None)
            etag = getattr(meta, "etag", None)
            lfs_sha256 = getattr(meta, "lfs", {}).get("sha256") if hasattr(meta, "lfs") else None
        except Exception:
            size = getattr(item, "size", None)
            etag = None
            lfs_sha256 = getattr(item, "lfs", {}).get("sha256") if hasattr(item, "lfs") else None

        # If LFS SHA256 not available, optionally download to compute it (skip for large files)
        computed_sha256 = None
        if not lfs_sha256 and size and size < 500_000_000:  # 500 MB threshold
            try:
                local_path = hf_hub_download(repo_id=repo, filename=path, repo_type="dataset", cache_dir=local_cache_dir)
                computed_sha256 = sha256_file(local_path)
            except Exception:
                computed_sha256 = None

        entry = {
            "repo": repo,
            "path": path,
            "cdn_url": cdn_url,
            "size": size,
            "etag": etag,
            "lfs_sha256": lfs_sha256 or computed_sha256,
            "date": date,
        }
        files.append(entry)

    manifest = {
        "generated_at_utc": datetime.now(timezone.utc).isoformat(),
        "repo": repo,
        "date": date,
        "count": len(files),
        "files": files,
    }

    os.makedirs(os.path.dirname(os.path.abspath(out_path)), exist_ok=True)
    with open(out_path, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Manifest written: {out_path} ({len(files)} files)")

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Create CDN-first manifest")
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g., axentx/surrogate-1)")
    parser.add_argument("--date", required=True, help="Date folder (e.g., 2026-04-29)")
    parser.add_argument("--out", default="manifest.json", help="Output JSON path")
    parser.add_argument("--cache-dir", default="/tmp/hf_cache", help="Local HF cache for optional SHA256 checks")
    args = parser.parse_args()
    build_manifest(args.repo, args.date, args.out, local_cache_dir=args.cache_dir)
```

#!/usr/bin/env python3
"""
Lightning Studio launcher for Surrogate-1 training.
- Reuses running studio when available.
- Restarts if idle-stopped.
- Passes only CDN URLs to training (no HF API calls during data load).
"""
import json
import os
import sys
import subprocess
import time

try:
    from lightning import LightningWork, LightningFlow, LightningApp
    from lightning.app.utilities.network import LightningClient
    from lightning.app.utilities.cloud import _get_project
except ImportError:
    print("Install: pip install lightning")
    sys.exit(1)

MANIFEST_PATH = os.getenv("MANIFEST_PATH", "manifest.json")
TRAIN_SCRIPT = os.getenv("TRAIN_SCRIPT", "train.py")
STUDIO_NAME = "vanguard-surrogate-trainer"

class SurrogateTrainer(LightningWork):
    def __init__(self, machine: str = "lightning-lambda-prod/L40S", **kwargs):
        super().__init__(machine=machine, **kwargs)
        self.machine = machine
        self._studio_reused = False

    def run(self, manifest_path: str):
        if not os.path.exists(manifest_path):
            raise FileNotFoundError(f"Manifest not found: {manifest_path}")

        with open(manifest_path) as f:
            manifest = json.load(f)

        # Build CDN-only file list for training script
        file_urls = [f["cdn_url"] for f in manifest["files"]]
        print(f"Prepared {len(file_urls)} CDN URLs for training")

        # Pass CDN list via env to train.py (or CLI)
        os.environ["CDN_FILE_URLS"] = ",".join(file_urls)
        os.environ["MANIFEST_REPO"] = manifest["repo"]
        os.environ["MANIFEST_DATE"] = manifest["date"]

        # Execute training script (must use CDN URLs only)
        cmd = [sys.executable, TRAIN_SCRIPT]
        print(f"Running: {' '.join(cmd)}")
        result = subprocess.run(cmd, check=False)
        if result.returncode != 0:
            raise RuntimeError(f"Training script failed with code {result.returncode}")
