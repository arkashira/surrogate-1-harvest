# airship / ops

Candidate 3:
## Highest-Value Incremental Improvement (<2h)

**Goal**: Harden `airship discover` into a **deterministic, CDN-safe, zero-runtime orchestrator** that:
- Eliminates HF API 429/128-hr commit cap during ingestion
- Avoids PyArrow `CastError` from mixed-schema repos
- Uses CDN-only fetches (no Authorization header) for training data
- Produces clean `{prompt,response}` parquet with attribution in filename

**Why this wins**: One change removes two failure modes (rate-limit + schema), enables reliable surrogate-1 training, and costs 0 extra infra.

---

## Implementation Plan

1. **Locate entrypoint**  
   Find `airship/discover` CLI or create `scripts/discover_hf_mirror.sh`.

2. **Add CDN-safe file listing**  
   - Single `list_repo_tree` call per date folder → save `file_list.json`.  
   - Embed list in training script; Lightning workers fetch via CDN only.

3. **Make `discover` deterministic & schema-safe**  
   - Download each file via `hf_hub_download` (or raw CDN) → parse only `{prompt,response}` fields.  
   - Drop extra columns (`source`, `ts`, etc.) before parquet write.

4. **Harden wrapper scripts**  
   - Ensure Bash shebang, `chmod +x`, and `SHELL=/bin/bash` in any crontab entries.  
   - Invoke via `bash <script> "$@"` for portability.

5. **Lightning quota guard**  
   - Reuse running Studios by name; restart only if stopped (idle timeout kills training).

6. **Output layout**  
   - `batches/mirror-merged/{date}/{slug}.parquet` (attribution in filename, no extra cols).

### 2. `scripts/parse_and_project.sh`
```bash
#!/usr/bin/env bash
set -euo pipefail

FILE_LIST="${FILE_LIST}"
OUT_DIR="${OUT_DIR}"
REPO="${REPO}"

mkdir -p "${OUT_DIR}/tmp"

python3 - <<PY
import os, json, sys, uuid
from huggingface_hub import hf_hub_download
import pyarrow.parquet as pq
import pyarrow as pa
import pandas as pd

repo = os.environ["REPO"]
file_list = json.load(open(os.environ["FILE_LIST"]))
out_dir = os.environ["OUT_DIR"]
tmp_dir = os.path.join(out_dir, "tmp")
os.makedirs(tmp_dir, exist_ok=True)

records = []
for i, rfile in enumerate(file_list):
    try:
        local_path = hf_hub_download(repo_id=repo, filename=rfile, repo_type="dataset")
        # Try parquet first
        try:
            df = pd.read_parquet(local_path)
        except Exception:
            # Fallback to line-based JSON/JSONL
            if rfile.endswith(".parquet"):
                raise
            df = pd.read_json(local_path, lines=rfile.endswith(".jsonl"), orient="records")
        # Project only what we need; tolerate missing cols
        prompt_col = next((c for c in df.columns if "prompt" in c.lower()), None)
        response_col = next((c for c in df.columns if "response" in c.lower() or "completion" in c.lower() or "answer" in c.lower()), None)
        if prompt_col is None or response_col is None:
            print(f"Skipping {rfile}: missing prompt/response cols")
            continue
        for _, row in df.iterrows():
            records.append({
                "prompt": str(row[prompt_col]),
                "response": str(row[response_col]),
                # slug used for attribution, not as column
            })
    except Exception as e:
        print(f"Error processing {rfile}: {e}")
        continue

# Write merged parquet with no extra metadata columns
out_path = os.path.join(out_dir, f"{uuid.uuid4().hex[:12]}.parquet")
tbl = pa.Table.from_pylist(records, schema=pa.schema([
    pa.field("prompt", pa.string()),
    pa.field("response", pa.string()),
]))
pq.write_table(tbl, out_path)
print(f"Wrote {len(records)} rows -> {out_path}")
PY
```

### 3. `train.py` (Lightning Studio entrypoint) — CDN-only data loader
```python
import json
import os
import pyarrow.parquet as pq
import requests
from torch.utils.data import IterableDataset

class CDNParquetDataset(IterableDataset):
    def __init__(self, repo: str, file_list_path: str):
        self.repo = repo
        with open(file_list_path) as f:
            self.files = json.load(f)

    def _stream_file(self, rfile: str):
        # CDN URL — no Authorization header, bypasses HF API rate limits
        url = f"https://huggingface.co/datasets/{self.repo}/resolve/main/{rfile}"
        resp = requests.get(url, stream=True, timeout=60)
        resp.raise_for_status()
        # Stream into pyarrow (or save temp) — simplified: download chunk then read
        # For large files, consider streaming to temp then pq.read_table
        import tempfile
        with tempfile.NamedTemporaryFile(suffix=".parquet", delete=False) as tmp:
            for chunk in resp.iter_content(chunk_size=8192*1024):
                tmp.write(chunk)
            tmp_path = tmp.name
        tbl = pq.read_table(tmp_path, columns=["prompt", "response"])
        os.unlink(tmp_path)
        return tbl.to_pylist()

    def __iter__(self):
        for rfile in self.files:
            try:
                rows = self._stream_file(rfile)
                for row in rows:
                    yield row["prompt"], row["response"]
            except Exception as e:
                print(f"Failed to stream {rfile}: {e}")
                continue
```

### 4. Lightning Studio reuse guard (launcher snippet)
```python
from lightning import Studio, Machine, Teamspace

def get_or_create_studio(name: str, machine: Machine = Machine.L40S):
    teamspace = Teamspace()
    for s in teamspace.studios:
        if s.name == name and s.status == "Running":
            print(f"Reusing running studio: {name}")
            return s
    print(f"Creating studio: {name}")
    return Studio(
        name=name,
        machine=machine,
        create_ok=True,
    )

# Usage
studio = get_or_create_studio("surrogate-train-l40s")
# Before each run, check status; restart if stopped (idle timeout kills training)
if studio.status != "Running":
    studio.start(machine=Machine.L40S)
```
