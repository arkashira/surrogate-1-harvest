# workio / ops

## 1. Diagnosis
- **No idempotency guard on LINE webhook ingestion** — LINE retries on 5xx/timeouts; duplicate clock-in/out or leave/OT records on replay.
- **Missing `X-Line-Signature` verification** — webhook accepts spoofed events; enables attendance fraud and phantom punches.
- **No replay-safe request-idempotency key** — concurrent clock-in/out from multiple devices can create double entries.
- **Webhook handler does not validate event schema/expiry** — stale events (e.g., old timestamps) accepted and persisted.
- **No audit trail for webhook ingestion** — hard to debug duplicates or spoofed events in production.

## 2. Proposed change
File: `/opt/axentx/workio/server/src/routes/webhook/line.ts` (or create if absent)  
Scope: add `X-Line-Signature` HMAC verification, idempotency via `X-Line-Request-Id` (or event-level dedupe), schema validation, and audit log.

// /opt/axentx/workio/server/src/routes/webhook/line.ts
import { Router, Request, Response, NextFunction } from 'express';
import crypto from 'crypto';
import { body, validationResult } from 'express-validator';
import { AppDataSource } from '../../db/data-source';
import { WebhookEvent } from '../../entities/WebhookEvent'; // audit entity
import { ClockRecord } from '../../entities/ClockRecord';
import { LeaveRequest } from '../../entities/LeaveRequest';
import { OtRequest } from '../../entities/OtRequest';

const router = Router();
const CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';

function verifyLineSignature(rawBody: Buffer, signature: string): boolean {
  if (!CHANNEL_SECRET) return false;
  const expected = crypto
    .createHmac('sha256', CHANNEL_SECRET)
    .update(rawBody)
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

function isExpiredEvent(event: any): boolean {
  const ts = event.timestamp ? Number(event.timestamp) : Date.now();
  const now = Date.now();
  // reject events older than 5 minutes (clock skew tolerance)
  return Math.abs(now - ts) > 5 * 60 * 1000;
}

async function dedupeEvent(eventId: string, tenantId: string): Promise<boolean> {
  const repo = AppDataSource.getRepository(WebhookEvent);
  const existing = await repo.findOneBy({ eventId, tenantId });
  return !!existing;
}

async function recordEvent(eventId: string, tenantId: string, payload: any) {
  const repo = AppDataSource.getRepository(WebhookEvent);
  await repo.insert({ eventId, tenantId, payload, receivedAt: new Date() });
}

router.post(
  '/line',
  // preserve raw body for signature verification
  (req: Request, res: Response, next: NextFunction) => {
    if (Buffer.isBuffer(req.body) || typeof req.body === 'string') return next();
    let data = '';
    req.setEncoding('utf8');
    req.on('data', (chunk) => (data += chunk));
    req.on('end', () => {
      (req as any).rawBody = data;
      try {
        req.body = JSON.parse(data);
      } catch {
        req.body = {};
      }
      next();
    });
  },
  [
    body().custom((_, { req }) => {
      const sig = (req as any).headers['x-line-signature'];
      const raw = (req as any).rawBody;
      if (!raw || !sig || !verifyLineSignature(Buffer.from(raw), sig)) {
        throw new Error('Invalid LINE signature');
      }
      return true;
    }),
  ],
  async (req: Request, res: Response) => {
    const errors = validationResult(req);
    if (!errors.isEmpty()) {
      return res.status(401).json({ error: 'Unauthorized' });
    }

    const events = req.body.events || [];
    if (!Array.isArray(events) || events.length === 0) {
      return res.status(200).json({ ok: true });
    }

    // Basic tenant resolution: from source.userId or webhook config mapping
    // For multi-tenant, map source.userId → tenantId via your user service.
    // Here we use a placeholder: resolveTenantId(source.userId)
    const resolveTenantId = (userId: string): string => {
      // TODO: integrate your tenant mapping
      return 'default-tenant';
    };

    for (const ev of events) {
      try {
        const eventId = ev.source?.userId + ':' + (ev.timestamp || Date.now()) + ':' + (ev.type || 'unknown');
        const tenantId = resolveTenantId(ev.source?.userId);

        // idempotency + replay protection
        if (await dedupeEvent(eventId, tenantId)) continue;
        if (isExpiredEvent(ev)) continue;

        // Process event types
        if (ev.type === 'message' && ev.message?.text) {
          const text = ev.message.text.trim().toLowerCase();
          const userId = ev.source.userId;

          if (text === 'clock in' || text === 'clock out') {
            const isClockIn = text === 'clock in';
            // Prevent double punch within short window (e.g., 5min)
            const recent = await AppDataSource.getRepository(ClockRecord).findOne({
              where: { userId, tenantId, clockIn: isClockIn ? null : { $ne: null } },
              order: { createdAt: 'DESC' },
            });
            if (recent && Math.abs(Date.now() - new Date(recent.createdAt).getTime()) < 5 * 60 * 1000) {
              continue;
            }

            const record = new ClockRecord();
            record.userId = userId;
            record.tenantId = tenantId;
            record.clockIn = isClockIn ? new Date() : null;
            record.clockOut = isClockIn ? null : new Date();
            record.location = ev.source?.area || null;
            await AppDataSource.getRepository(ClockRecord).save(record);
          }
        }

        // Leave / OT handling stubs — integrate your business logic
        if (ev.type === 'postback' && ev.postback?.data) {
          const data = ev.postback.data;
          // parse and create LeaveRequest/OtRequest as needed
        }

        // Audit
        await recordEvent(eventId, tenantId, ev);
      } catch (err) {
        // Log and continue processing other events
        console.error('Webhook event processing error:', err);
      }
    }

    // LINE expects 200 OK to stop retries
    return res.status(200).json({ ok: true });
  }
);

export default router;
