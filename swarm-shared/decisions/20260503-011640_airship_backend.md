# airship / backend

## Final Synthesis — Highest-Value, Actionable Improvement

**Goal:** Eliminate HF API 429s and `pyarrow.CastError`s during dataset loading, reduce Lightning quota burn, and keep all changes frontend-only (<2h ship).

**Chosen approach:**  
- Use **HF CDN (`/resolve/main/`)** for dataset file fetches (no auth, no rate limits).  
- **List files once** via HF tree API, then fetch parquet files directly from CDN.  
- **Project only `{prompt, response}`** at parse time to avoid schema mismatches.  
- **Reuse running Lightning Studio** instead of starting new ones each run.  
- Keep backend unchanged; expose minimal proxy endpoints only if CORS or auth requires them.

### 1) CDN-bypass dataset loader utility (15 min)
**File:** `frontend/src/utils/hf-cdn-loader.ts`

```typescript
// frontend/src/utils/hf-cdn-loader.ts
interface HFRepoFile {
  path: string;
  type: 'file' | 'dir';
  size: number;
}

export interface ParsedSample {
  prompt: string;
  response: string;
}

export class HFCDNDatasetLoader {
  private repo: string;
  private folderPath: string;

  constructor(repo: string, folderPath: string = '') {
    this.repo = repo;
    this.folderPath = folderPath;
  }

  // One authenticated/tree API call (do this once per session)
  async listParquetFiles(): Promise<string[]> {
    const url = `https://huggingface.co/api/datasets/${this.repo}/tree/${this.folderPath}?recursive=false`;
    const res = await fetch(url);
    if (!res.ok) throw new Error(`HF tree API failed: ${res.status}`);
    const files: HFRepoFile[] = await res.json();
    return files
      .filter((f) => f.type === 'file' && f.path.endsWith('.parquet'))
      .map((f) => f.path);
  }

  // CDN-only fetch — no auth, no rate limit
  async loadFile(filePath: string): Promise<ParsedSample[]> {
    const cdnUrl = `https://huggingface.co/datasets/${this.repo}/resolve/main/${filePath}`;
    const res = await fetch(cdnUrl);
    if (!res.ok) throw new Error(`CDN fetch failed: ${res.status} for ${filePath}`);

    const buffer = await res.arrayBuffer();
    return this.parseParquetProjected(buffer);
  }

  // Load all files in folder via CDN
  async loadAllFromFolder(): Promise<ParsedSample[]> {
    const paths = await this.listParquetFiles();
    const results: ParsedSample[] = [];

    for (const p of paths) {
      try {
        const samples = await this.loadFile(p);
        results.push(...samples);
      } catch (err) {
        console.warn(`Skipping ${p}:`, err);
        // Continue with remaining files
      }
    }

    return results;
  }

  private async parseParquetProjected(buffer: ArrayBuffer): Promise<ParsedSample[]> {
    // Use parquet-wasm (or arrow-js) to project only required columns
    // This avoids pyarrow.CastError from schema heterogeneity
    // @ts-ignore - ensure parquet-wasm is installed
    const parquet = await import('parquet-wasm');
    const table = parquet.readTable(new Uint8Array(buffer));

    const prompts = table.getChild('prompt')?.toArray() || [];
    const responses = table.getChild('response')?.toArray() || [];

    const len = Math.min(prompts.length, responses.length);
    const out: ParsedSample[] = [];
    for (let i = 0; i < len; i++) {
      const prompt = String(prompts[i] ?? '');
      const response = String(responses[i] ?? '');
      if (prompt || response) out.push({ prompt, response });
    }
    return out;
  }
}
```

**Notes:**  
- If `parquet-wasm` is unavailable, fall back to `apache-arrow` with column projection.  
- This projection step is what prevents `pyarrow.CastError` from unexpected columns/types.

### 2) Lightning Studio reuse hook (15 min)
**File:** `frontend/src/hooks/useLightningStudio.ts`

```typescript
// frontend/src/hooks/useLightningStudio.ts
import { useState, useEffect, useCallback } from 'react';

export interface LightningStudio {
  id: string;
  name: string;
  status: 'running' | 'stopped' | 'starting';
  machine: string;
}

export function useLightningStudio(studioName: string) {
  const [studio, setStudio] = useState<LightningStudio | null>(null);
  const [loading, setLoading] = useState(false);

  const fetchStudios = useCallback(async (): Promise<LightningStudio[]> => {
    // If backend proxy is required for auth/CORS, call /api/lightning/studios
    const res = await fetch('/api/lightning/studios');
    if (!res.ok) throw new Error('Failed to fetch studios');
    return res.json();
  }, []);

  const findRunningStudio = useCallback(async (): Promise<LightningStudio | null> => {
    const studios = await fetchStudios();
    return studios.find((s) => s.name === studioName && s.status === 'running') || null;
  }, [fetchStudios, studioName]);

  const startOrReuseStudio = useCallback(async (): Promise<LightningStudio> => {
    setLoading(true);
    try {
      const existing = await findRunningStudio();
      if (existing) {
        setStudio(existing);
        return existing;
      }

      // Start new studio (fallback machine if preferred unavailable)
      const res = await fetch('/api/lightning/studios', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          name: studioName,
          machine: 'L40S', // fallback if H200 unavailable
          cloud: 'lightning-public-prod',
        }),
      });

      if (!res.ok) throw new Error('Failed to start studio');
      const newStudio: LightningStudio = await res.json();
      setStudio(newStudio);
      return newStudio;
    } finally {
      setLoading(false);
    }
  }, [studioName, findRunningStudio]);

  const checkStudioStatus = useCallback(async (): Promise<void> => {
    if (!studio) return;
    const res = await fetch(`/api/lightning/studios/${studio.id}`);
    if (!res.ok) return;
    const updated: LightningStudio = await res.json();

    if (updated.status === 'stopped') {
      // Studio stopped unexpectedly — restart
      await startOrReuseStudio();
    } else {
      setStudio(updated);
    }
  }, [studio, startOrReuseStudio]);

  useEffect(() => {
    if (studio) {
      const interval = setInterval(checkStudioStatus, 30000);
      return () => clearInterval(interval);
    }
  }, [studio, checkStudioStatus]);

  return { studio, loading, startOrReuseStudio, checkStudioStatus };
}
```

**Backend expectations (minimal):**  
- `GET /api/lightning/studios` → list studios (proxy to Lightning if auth required).  
- `POST /api/lightning/studios` → start a studio.  
- `GET /api/lightning/studios/:id` → get status.  
If no backend exists yet, stub these endpoints or use direct SDK calls from the frontend where permitted.
