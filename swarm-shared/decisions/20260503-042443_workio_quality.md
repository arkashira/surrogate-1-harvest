# workio / quality

## 1. Diagnosis
- **Missing `X-Line-Signature` verification** — webhook accepts spoofed events; enables attendance fraud and phantom clock-ins/outs.
- **No idempotency guard on LINE webhook ingestion** — LINE retries on 5xx/timeouts; duplicate clock-in/out or leave/OT records are created on replay.
- **No transactional state transition checks** — allows invalid sequences (e.g., double clock-in without clock-out, approve already-approved leave).
- **Missing monotonic replay protection per event** — same `webhookEventId` can be processed multiple times across retries.
- **No immutable audit log for attendance/leave/OT changes** — cannot reconstruct who changed what and when for compliance.

## 2. Proposed change
Create `workio/server/src/routes/webhook/line.ts` (new) and wire it into the Express app. Scope:
- Add `X-Line-Signature` HMAC-SHA256 verification.
- Add idempotency table (`webhook_events`) keyed by `lineEventId` to deduplicate.
- Wrap attendance/leave/OT mutations in transactions with state checks.
- Emit immutable audit rows to `attendance_audit` / `leave_request_audit`.
- Return 200 only after commit; 5xx on failure so LINE retries safely.

CREATE TABLE IF NOT EXISTS webhook_events (
  id              BIGSERIAL PRIMARY KEY,
  line_event_id   TEXT NOT NULL UNIQUE,
  event_type      TEXT NOT NULL,
  payload_json    JSONB NOT NULL,
  processed_at    TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
CREATE INDEX IF NOT EXISTS idx_webhook_events_line_event_id ON webhook_events(line_event_id);

-- Immutable attendance audit
CREATE TABLE IF NOT EXISTS attendance_audit (
  id              BIGSERIAL PRIMARY KEY,
  attendance_id   BIGINT NOT NULL,
  tenant_id       BIGINT NOT NULL,
  user_id         BIGINT NOT NULL,
  action          TEXT NOT NULL, -- 'clock_in', 'clock_out', 'auto_corrected'
  device_info     JSONB,
  location        GEOGRAPHY(POINT,4326),
  metadata        JSONB,
  performed_by    TEXT, -- 'system' or userId
  created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
CREATE INDEX IF NOT EXISTS idx_attendance_audit_tenant_user ON attendance_audit(tenant_id, user_id);

-- Immutable leave request audit
CREATE TABLE IF NOT EXISTS leave_request_audit (
  id              BIGSERIAL PRIMARY KEY,
  leave_id        BIGINT NOT NULL,
  tenant_id       BIGINT NOT NULL,
  user_id         BIGINT NOT NULL,
  action          TEXT NOT NULL, -- 'created','updated','approved','rejected'
  old_status      TEXT,
  new_status      TEXT,
  reason          TEXT,
  metadata        JSONB,
  performed_by    BIGINT, -- userId who performed action (nullable for system)
  created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

import express, { Request, Response, NextFunction } from 'express';
import crypto from 'crypto';
import { Pool } from 'pg';
import { v4 as uuidv4 } from 'uuid';

const router = express.Router();
const pool = new Pool({ connectionString: process.env.DATABASE_URL });

const LINE_CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';
if (!LINE_CHANNEL_SECRET) {
  console.warn('LINE_CHANNEL_SECRET is missing; webhook verification disabled (not recommended in prod)');
}

function verifySignature(body: string, signature: string): boolean {
  if (!LINE_CHANNEL_SECRET) return false;
  const expected = crypto
    .createHmac('sha256', LINE_CHANNEL_SECRET)
    .update(body, 'utf8')
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

// Idempotency check
async function isAlreadyProcessed(lineEventId: string): Promise<boolean> {
  const res = await pool.query('SELECT 1 FROM webhook_events WHERE line_event_id = $1', [lineEventId]);
  return res.rowCount > 0;
}

// Record processed event
async function recordEvent(lineEventId: string, eventType: string, payload: any) {
  await pool.query(
    'INSERT INTO webhook_events(line_event_id, event_type, payload_json) VALUES($1,$2,$3)',
    [lineEventId, eventType, payload]
  );
}

    [attendance.id, attendance.tenant_id, attendance.user_id, 'clock_out', location, { source: 'line-webhook' }, 'system']

// Main handler
async function handleLineWebhook(req: Request, res: Response) {
  const signature = req.headers['x-line-signature'] as string || '';
  const rawBody = JSON.stringify(req.body);

  // 1) Verify signature
  if (!verifySignature(rawBody, signature)) {
    console.warn('Invalid LINE signature', { signaturePresent: !!signature });
    return res.status(401).json({ error: 'Invalid signature' });
  }

  const events = req.body.events;
  if (!Array.isArray(events) || events.length === 0) {
    return res.status(200).json({ ok: true });
  }

  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    for (const ev of events) {
      const lineEventId = ev.source?.userId ? `${ev.source.userId}:${ev.timestamp}:${ev.type}` : ev['webhookEventId'] || uuidv4();
      // Use a stable idempotency key: prefer LINE's delivery id if available; fallback to composite
      const idempotencyKey = ev['webhookEventId'] || lineEventId;

      if (await isAlreadyProcessed(idempotencyKey)) {
        // Already processed — safe to ACK so LINE stops retrying
        continue;
      }

      // Record first to guarantee id
