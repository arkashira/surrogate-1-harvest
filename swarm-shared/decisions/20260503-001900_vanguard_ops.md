# vanguard / ops

## Final Consolidated Solution

**Core diagnosis (accepted from both candidates):**  
- No persisted `(repo, dateFolder)` manifest → repeated authenticated HF API enumeration → quota burn + 429 risk.  
- Training uses authenticated `list_repo_tree`/`load_dataset` during data loading instead of CDN-only fetches with a precomputed file list.  
- No retry/backoff for HF 429 in ops code.  
- Lightning Studio reuse missing; idle-stop kills jobs with no resume.  
- Local/mac runs must never perform model/heavy compute (architecture rule).

**Chosen approach:**  
- Single manifest generator with exponential backoff (cap 360s) and one `list_repo_tree` call per folder.  
- Training launcher uses manifest + CDN-only URLs, injects manifest path, and never performs authenticated HF calls during training.  
- Launcher reuses running studios, guards idle-stop with restart on L40S, and enforces local/mac to delegate (no local compute).  
- Lightweight ops CLI; deterministic caching.

### 1. Manifest generator

`/opt/axentx/vanguard/ops/manifest.py`

```python
#!/usr/bin/env python3
"""
Generate and cache per-(repo, dateFolder) file manifests to avoid repeated
HF API enumeration. Uses a single list_repo_tree call per folder and produces
CDN URLs for training (zero auth during data load).
"""
import json
import os
import sys
import time
from pathlib import Path
from typing import Dict, List

try:
    from huggingface_hub import list_repo_tree
except ImportError:
    print("huggingface_hub required: pip install huggingface_hub", file=sys.stderr)
    sys.exit(1)

MANIFEST_ROOT = Path(os.getenv("VANGUARD_MANIFEST_ROOT", "/opt/axentx/vanguard/manifests"))
HF_API_RETRY = int(os.getenv("HF_API_RETRY", "5"))
HF_RETRY_BACKOFF = int(os.getenv("HF_RETRY_BACKOFF", "10"))


def hf_api_with_backoff(fn, *args, **kwargs):
    for attempt in range(HF_API_RETRY):
        try:
            return fn(*args, **kwargs)
        except Exception as exc:
            msg = str(exc).lower()
            if "429" in msg or "rate limit" in msg or "too many requests" in msg:
                wait = HF_RETRY_BACKOFF * (2 ** attempt)
                if wait >= 360 or attempt == HF_API_RETRY - 1:
                    wait = 360
                print(
                    f"HF rate limit (attempt {attempt + 1}/{HF_API_RETRY}), sleeping {wait}s",
                    file=sys.stderr,
                )
                time.sleep(wait)
                continue
            raise
    raise RuntimeError(f"HF API failed after {HF_API_RETRY} attempts")


def build_manifest(repo_id: str, date_folder: str, revision: str = "main") -> Dict:
    """
    repo_id: e.g. 'datasets/myorg/mirror-merged'
    date_folder: e.g. '2026-04-29'
    Returns manifest with CDN URLs and metadata.
    """
    safe_repo = repo_id.replace("/", "--")
    root = MANIFEST_ROOT / safe_repo
    root.mkdir(parents=True, exist_ok=True)
    manifest_path = root / f"{date_folder}.json"

    if manifest_path.exists():
        with open(manifest_path) as f:
            return json.load(f)

    items = hf_api_with_backoff(
        list_repo_tree,
        repo_id=repo_id,
        path=date_folder,
        revision=revision,
        recursive=False,
    )

    files = []
    for item in items:
        if item.get("type") != "file":
            continue
        path = item["path"]
        cdn_url = f"https://huggingface.co/datasets/{repo_id}/resolve/main/{path}"
        files.append(
            {
                "path": path,
                "cdn_url": cdn_url,
                "size": item.get("size"),
                "lfs": item.get("lfs", {}),
            }
        )

    manifest = {
        "repo_id": repo_id,
        "date_folder": date_folder,
        "revision": revision,
        "generated_at": time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime()),
        "files": files,
    }

    with open(manifest_path, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Manifest written to {manifest_path}")
    return manifest


if __name__ == "__main__":
    if len(sys.argv) < 3:
        print("Usage: manifest.py <repo_id> <date_folder> [revision]")
        sys.exit(1)
    repo_id = sys.argv[1]
    date_folder = sys.argv[2]
    revision = sys.argv[3] if len(sys.argv) > 3 else "main"
    manifest = build_manifest(repo_id, date_folder, revision)
    print(json.dumps(manifest, indent=2))
```

MANIFEST_ROOT = Path(os.getenv("VANGUARD_MANIFEST_ROOT", "/opt/axentx/vanguard/manifests"))


def find_running_studio(name: str) -> Optional[Studio]:
    for s in Teamspace.studios:
        if getattr(s, "name", None) == name and getattr(s, "status", None) == "Running":
            return s
    return None


def ensure_studio_running(studio_name: str, machine: str = "L40S") -> Studio:
    studio = find_running_studio(studio_name)
    if studio:
        print(f"Reusing running studio: {studio_name}")
        return studio

    print(f"Creating studio: {studio_name} on {machine}")
    studio = Studio(
        name=studio_name,
        cloud="lightning-public-prod",
        machine=Machine(machine),
        create_ok=True,
    )
    return studio


def wait_for_running(studio: Studio, timeout_s: int = 600, poll_s: int = 10) -> None:
    for _ in range(timeout_s // poll_s):
        time.sleep(poll_s)
        studio.refresh()
        if getattr(studio, "status", None) == "Running":
            return
    raise RuntimeError("Studio failed to start within timeout")
