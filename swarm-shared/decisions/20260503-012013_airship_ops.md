import os, json, pyarrow.parquet as pq, requests, io
from pathlib import Path
from lightning.pytorch import Trainer
from lightning.pytorch.loggers import CSVLogger
from torch.utils.data import Dataset, DataLoader

REPO = os.getenv("HF_REPO", "myorg/surrogate-data")
DATE = os.getenv("HF_DATE", "2026-05-03")
FILE_LIST = os.getenv("HF_FILE_LIST", f"file_list_{DATE}.json")
MACHINE = os.getenv("LIGHTNING_MACHINE", "L40S")

class CDNParquetDataset(Dataset):
    def __init__(self, file_urls):
        self.file_urls = file_urls
        self.data = []
        for url in self.file_urls:
            r = requests.get(url, timeout=30)
            r.raise_for_status()
            table = pq.read_table(io.BytesIO(r.content))
            # Project only {prompt, response} — ignore heterogeneous schema
            df = table.select(["prompt", "response"]).to_pandas()
            self.data.extend(df.to_dict(orient="records"))

    def __len__(self):
        return len(self.data)

    def __getitem__(self, idx):
        item = self.data[idx]
        return str(item["prompt"]), str(item["response"])

def main():
    with open(FILE_LIST) as f:
        paths = json.load(f)
    base = f"https://huggingface.co/datasets/{REPO}/resolve/main/{DATE}"
    urls = [f"{base}/{p}" for p in paths]

    ds = CDNParquetDataset(urls)
    loader = DataLoader(ds, batch_size=8, shuffle=True, num_workers=4)

    # Minimal model placeholder — replace with your Surrogate ensemble
    import torch.nn as nn
    model = nn.Linear(1, 1)

    logger = CSVLogger("logs", name="cdn_train")
    trainer = Trainer(
        max_epochs=1,
        logger=logger,
        accelerator="gpu",
        devices=1,
        enable_checkpointing=False,
    )
    trainer.fit(model, loader)

if __name__ == "__main__":
    main()

import os
from lightning.pytorch.cli import LightningCLI
from lightning.fabric.utilities.cloud_io import _load as _load_fabric
from lightning.pytorch.studio import Studio, Teamspace, Machine

REPO = os.getenv("HF_REPO", "myorg/surrogate-data")
DATE = os.getenv("HF_DATE", "2026-05-03")
STUDIO_NAME = f"surrogate-cdn-{DATE}"

teamspace = Teamspace()
studio = None
for s in teamspace.studios:
    if s.name == STUDIO_NAME and s.status == "Running":
        studio = s
        print(f"Reusing running studio: {STUDIO_NAME}")
        break

if studio is None:
    for s in teamspace.studios:
        if s.name == STUDIO_NAME and s.status == "Stopped":
            studio = s
            studio.target.start(machine=Machine.L40S)
            print(f"Restarted stopped studio: {STUDIO_NAME}")
            break

if studio is None:
    studio = Studio(
        name=STUDIO_NAME,
        script="train_cdn_lightning.py",
        machine=Machine.L40S,
        create_ok=True,
    )
    print(f"Created new studio: {STUDIO_NAME}")

# Ensure studio is running before run()
if studio.status != "Running":
    studio.target.start(machine=Machine.L40S)

studio.run()
print(f"Studio URL: {studio.url}")

### UI integration snippet (`surrogate/ui/train_form.html`)
```html
<form id="trainForm">
  <label>
    <input type="checkbox" name="use_cdn" checked>
    Use CDN-bypass + Studio reuse (recommended)
  </label>
  <button type="submit">Launch Training</button>
</form>

<script>
document.getElementById("trainForm").addEventListener("submit", async (e) => {
  e.preventDefault();
  const useCDN = e.target.use_cdn.checked;
  if (useCDN) {
    // Calls the CDN launcher (bash → Lightning SDK)
    await fetch("/api/train/cdn", { method: "POST", body: JSON.stringify({ repo: "myorg/surrogate-data" }) });
  } else {
    // Legacy path
    await fetch("/api/train", { method: "POST", body: JSON.stringify({ repo: "myorg/surrogate-data" }) });
  }
});
</script>
```
