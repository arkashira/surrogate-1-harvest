#### `bin/manifest.py`
```python
#!/usr/bin/env python3
"""
Generate manifest for surrogate-1-training-pairs.
Run from Mac when HF API rate-limit window is clear.
Writes manifest.json with CDN-ready paths.
"""
import json
import os
import sys
from datetime import datetime, timezone

from huggingface_hub import HfApi

REPO_ID = "axentx/surrogate-1-training-pairs"
OUTFILE = "manifest.json"

def main() -> None:
    api = HfApi()
    # Single non-recursive call per top-level folder (avoids 100x pagination)
    folders = ["batches/public-raw", "batches/mirror-merged"]
    entries = []
    for folder in folders:
        try:
            items = api.list_repo_tree(
                repo_id=REPO_ID,
                path=folder,
                recursive=False,
                repo_type="dataset",
            )
        except Exception as exc:
            print(f"Warning: failed to list {folder}: {exc}", file=sys.stderr)
            continue
        for item in items:
            if getattr(item, "type", None) != "file":
                continue
            path = getattr(item, "path", None)
            if not path:
                continue
            entries.append(
                {
                    "path": path,
                    "cdn_url": f"https://huggingface.co/datasets/{REPO_ID}/resolve/main/{path}",
                    "size": getattr(item, "size", None),
                }
            )

    manifest = {
        "repo_id": REPO_ID,
        "generated_at_utc": datetime.now(timezone.utc).isoformat(),
        "entries": entries,
        "count": len(entries),
    }
    with open(OUTFILE, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2)
    print(f"Wrote {OUTFILE} with {len(entries)} entries")

if __name__ == "__main__":
    main()
```

#### `bin/dataset_enrich.py`
```python
#!/usr/bin/env python3
"""
CDN-bypass worker for surrogate-1 ingestion.
Each shard processes deterministic slice of manifest entries.
Zero HF API calls during data load (uses CDN URLs).
"""
import argparse
import hashlib
import json
import sys
from datetime import datetime, timezone
from pathlib import Path

import numpy as np
import pyarrow as pa
import pyarrow.parquet as pq
import requests
from tqdm import tqdm

# Local dedup store
sys.path.insert(0, str(Path(__file__).parent))
from lib.dedup import DedupStore  # noqa: E402

CHUNK_SIZE = 8192
TIMEOUT = 30
RETRIES = 3
BACKOFF = 5

def shard_filter(path: str, shard_id: int, total_shards: int) -> bool:
    """Deterministic 1/N shard assignment by path hash."""
    digest = hashlib.md5(path.encode()).digest()
    bucket = int.from_bytes(digest[:2], "little") % total_shards
    return bucket == shard_id

def safe_get(url: str) -> bytes:
    for attempt in range(RETRIES):
        try:
            resp = requests.get(url, timeout=TIMEOUT, stream=True)
            resp.raise_for_status()
            chunks = []
            for chunk in resp.iter_content(chunk_size=CHUNK_SIZE):
                chunks.append(chunk)
            return b"".join(chunks)
        except Exception as exc:
            if attempt == RETRIES - 1:
                raise
            import time
            time.sleep(BACKOFF * (attempt + 1))
    raise RuntimeError("unreachable")

def parse_parquet_to_pairs(content: bytes):
    """Project heterogeneous parquet to {prompt,response} only."""
    table = pq.read_table(pa.BufferReader(content))
    rows = []
    # Best-effort column mapping
    prompt_col = None
    response_col = None
    for col in ["prompt", "instruction", "input", "question"]:
        if col in table.column_names:
            prompt_col = col
            break
    for col in ["response", "output", "answer", "completion"]:
        if col in table.column_names:
            response_col = col
            break
    if prompt_col is None or response_col is None:
        # fallback: use first two string/text columns
        text_cols = [n for n in table.column_names if pa.types.is_string(table.schema.field(n).type)]
        if len(text_cols) >= 2:
            prompt_col, response_col = text_cols[0], text_cols[1]
        else:
            return rows  # skip
    for i in range(table.num_rows):
        prompt = str(table[prompt_col][i].as_py())
        response = str(table[response_col][i].as_py())
        if prompt and response:
            rows.append({"prompt": prompt.strip(), "response": response.strip()})
    return rows

def parse_jsonl_to_pairs(content: bytes):
    rows = []
    for line in content.splitlines():
        line = line.strip()
        if not line:
            continue
        try:
            obj = json.loads(line)
        except Exception:
            continue
        prompt = obj.get("prompt") or obj.get("instruction") or obj.get("input")
        response = obj.get("response") or obj.get("output") or obj.get("answer")
        if prompt and response:
            rows.append({"prompt": str(prompt).strip(), "response": str(response).strip()})
    return rows

def worker(manifest_path: str, shard_id: int, total_shards: int, out_dir: Path):
    manifest_path = Path(manifest_path)
    if not manifest_path.exists():
        print(f"Manifest not found: {manifest_path}", file=sys.stderr)
        sys.exit(1)

    with open(manifest_path, encoding="utf-8") as f:
        manifest = json.load(f)

    entries = manifest.get("entries", [])
    my_entries = [e for e in entries if shard_filter(e["path"], shard_id, total_shards)]
    print(f"Shard {shard_id}/{total_shards}: processing {len(my_entries)} files")

    out_dir.mkdir(parents=True, exist_ok=True)
    dedup = DedupStore()
    accepted = 0
    skipped_dup = 0
    failed = 0

    ts = datetime.now(timezone.utc).strftime("%Y%m%d-%H%M%S")
    out_file = out_dir / f"shard{shard_id}-{ts}.jsonl"

    with open(out_file, "w", encoding="utf-8") as out_f:
        for entry in tqdm(my_entries, desc=f"Shard {shard_id}"):
            path = entry["path"]
            url = entry["cdn_url"]
            try:
                content = safe_get(url)
            except Exception as exc:
                print(f"Failed to fetch {path}: {exc}", file=sys.stderr)
                failed += 1
                continue

            pairs = []
            if path.endswith(".parquet"):
                pairs = parse_parquet_to_pairs(content)
            elif path.endswith(".jsonl"):
                pairs = parse_jsonl_to_pairs(content)
            else:
                continue

            for pair in pairs:
                combined = (pair["prompt"] + "\n" + pair["response"]).encode()
                md5 = hashlib.md5(combined).hexdigest()
                if dedup.exists(md5):
                    skipped_dup += 1
                    continue
                dedup.add(md5)
                out_f.write(json.dumps(pair, ensure_ascii=False) + "\n")
                accepted += 1
            out_f.flush()

    print(f"Shard {shard_id} done: accepted={accepted}, dup={skipped_dup}, failed={failed}")
