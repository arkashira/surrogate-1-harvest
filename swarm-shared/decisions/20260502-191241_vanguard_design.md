# vanguard / design

## 1. Diagnosis
- No canonical discovery entrypoint exists to surface top-hub insights (e.g., MOC) before planning — violates `#knowledge-rag #graph #hub` pattern and forces ad-hoc exploration.
- Missing CDN-bypass file-list strategy for HF datasets; any future surrogate-1 training will hit API rate limits instead of using `resolve/main/` CDN fetches.
- No reusable Lightning Studio orchestration guard to prevent quota waste via recreation and to survive idle-stop restarts.
- No wrapper-script hardening (shebang + executable + Bash invocation) for cron-run scripts (active-learning / opus-pr-reviewer patterns) — likely to fail in production.
- No clear separation of orchestration (Mac) vs compute (Lightning/Kaggle/Cerebras) boundaries — risk of accidental local training or credential misuse.

## 2. Proposed change
Add a lightweight, reusable design module and orchestration helpers under `/opt/axentx/vanguard/design/`:
- `vanguard/discovery.py` — top-hub insight fetcher (knowledge-rag pattern).
- `vanguard/hf_cdn.py` — pre-list file paths once and emit JSON for CDN-only training.
- `vanguard/lightning_orchestrator.py` — studio reuse + idle-stop guard.
- `vanguard/run_wrapper.sh` — hardened wrapper for cron scripts (shebang, exec via bash).
- Update `README.md` with usage and architecture notes.

        return [
            {"title": f"{hub} Overview", "uri": f"kb://hub/{hub.lower()}/overview", "snippet": "..."},
        ][:limit]

    hubs = client.top_hubs(limit=1)
    if not hubs:
        return {"error": "no hubs found"}

    top = hubs[0]
    docs = client.related_docs(top["hub"])
    result = {"top_hub": top, "related_docs": docs}

    if cache_path:
        cache_path.parent.mkdir(parents=True, exist_ok=True)
        cache_path.write_text(json.dumps(result, indent=2))
    return result

if __name__ == "__main__":
    import sys

    print(json.dumps(out, indent=2))

### vanguard/hf_cdn.py
```python
"""
HF CDN-bypass file-list strategy.
Pattern: pre-list once, embed list in training script; Lightning uses CDN-only fetches.
"""
import json
import logging
from pathlib import Path
from typing import List, Optional

log = logging.getLogger(__name__)

try:
    from huggingface_hub import list_repo_tree
except ImportError:
    list_repo_tree = None

def list_dataset_files(repo_id: str, folder: str = "", recursive: bool = False) -> List[str]:
    """
    List files via HF API (non-recursive preferred) once and return paths.
    Caller should cache this list and embed in training script.
    """
    if list_repo_tree is None:
        raise RuntimeError("huggingface_hub not installed")

    # Avoid recursive on big repos; paginate per folder if needed
    tree = list_repo_tree(repo_id=repo_id, path=folder, recursive=recursive)
    files = [f.rfilename for f in tree if f.type == "file"]
    return files

def build_cdn_file_list(repo_id: str, folder: str, out_json: Path) -> None:
    """
    Save file list to JSON. Training script embeds this list and uses:
      https://huggingface.co/datasets/{repo_id}/resolve/main/{path}
    No Authorization header required for public datasets.
    """
    files = list_dataset_files(repo_id=repo_id, folder=folder, recursive=False)
    payload = {
        "repo_id": repo_id,
        "folder": folder,
        "files": files,
        "note": "Embed this list in training script; use CDN resolve URLs to bypass API rate limits."
    }
    out_json.parent.mkdir(parents=True, exist_ok=True)
    out_json.write_text(json.dumps(payload, indent=2))
    logging.info("Wrote CDN file list to %s (n=%d)", out_json, len(files))

if __name__ == "__main__":
    import sys
    repo = sys.argv[1] if len(sys.argv) > 1 else "datasets/example"
    folder = sys.argv[2] if len(sys.argv) > 2 else ""
    out = Path(sys.argv[3]) if len(sys.argv) > 3 else Path("hf_file_list.json")
    build_cdn_file_list(repo, folder, out)
```

### vanguard/lightning_orchestrator.py
```python
"""
Lightning Studio orchestration helpers: reuse running studios and survive idle-stop.
Patterns: #lightning-ai #quota #idle-timeout
"""
import logging
import time
from typing import Optional

log = logging.getLogger(__name__)

try:
    from lightning import LightningWork, LightningFlow, LightningApp, Machine
    from lightning.app import Teamspace
    RUNNING_AVAILABLE = True
except Exception:
    RUNNING_AVAILABLE = False
    log.warning("lightning not available; stubbing classes for design")

if RUNNING_AVAILABLE:
    class StudioOrchestrator:
        def __init__(self, studio_name: str, machine: str = "MACHINE_L40S"):
            self.studio_name = studio_name
            self.machine = getattr(Machine, machine, Machine.L40S)

        def get_running_studio(self):
            for s in Teamspace.studios:
                if s.name == self.studio_name and s.status == "Running":
                    return s
            return None

        def ensure_running(self, target):
            """
            Reuse running studio or start target on preferred machine.
            Call before each .run() to survive idle-stop kills.
            """
            existing = self.get_running_studio()
            if existing:
                log.info("Reusing running studio: %s", self.studio_name)
                return existing

            log.info("Starting studio: %s on %s", self.studio_name, self.machine)
            target.start(machine=self.machine, create_ok=True)
            return target

        def run_with_retry(self, target, max_retries: int = 3, wait_sec: int = 10):
            for attempt in range(1, max_retries + 1):
                try:
                    studio = self.ensure_running(target)
                    # If studio was stopped by idle timeout, restart
                    if getattr(studio, "status", None) != "Running":
                        target.start(machine=self.machine)
                    return target.run()
                except Exception as exc:
                    log.warning("Run attempt %s failed: %s", attempt, exc)
                    if attempt == max_retries:
                        raise
                    time.sleep(wait_sec * attempt)
else:
    class StudioOrchestrator:
        def __init__(self, *a, **kw): pass
        def ensure_running(self, target): return target
        def run_with_retry(self, target, max_retries=3, wait_sec=10): return target.run() if hasattr(target, "run") else None
```
