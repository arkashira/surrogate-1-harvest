# Costinel / ops

## Implementation Plan — Costinel “Top-Hub Signal” Card (read-only)

**Scope**: Frontend-only card that identifies the most-connected hub from the knowledge-rag graph and surfaces the top hub with 3 contextual signals.  
**Timebox**: <2h (frontend-only, no backend changes).

### What this ships
- A new card in the dashboard that:
  - Shows the most-connected hub (e.g., “MOC”) from the knowledge-rag graph.
  - Displays 3 contextual signals (title + short summary + source link).
  - Auto-refreshes on mount (read-only; no mutations).
  - Graceful fallbacks when graph or signals are unavailable.
- Reuses existing patterns:
  - CDN-bypass file list (if graph/signal files are in HF datasets).
  - Lightweight client fetch (no API keys required for public CDN assets).
  - Uses existing UI tokens/colors for consistency.

### File changes
- `src/components/CostinelTopHubSignalCard.tsx` — new component.
- `src/pages/Dashboard.tsx` — mount the new card in the top-row or signals section.
- `src/lib/hooks/useTopHubSignals.ts` — hook to fetch/parse graph + signals.
- `src/types/knowledge-rag.ts` — minimal types.

### 2) Hook (`src/lib/hooks/useTopHubSignals.ts`)

```ts
import { useEffect, useState, useCallback } from "react";
import { TopHubPayload, KnowledgeHub, Signal } from "@/types/knowledge-rag";

const CDN_BASE = "https://huggingface.co/datasets";
const REPO = "axentx/costinel-knowledge-rag";
const GRAPH_PATH = "graph/top-hub.json";
const SIGNALS_PATH = "signals/top-context.json";

async function fetchCdnJson<T>(filePath: string): Promise<T | null> {
  try {
    const url = `${CDN_BASE}/${REPO}/resolve/main/${filePath}`;
    const res = await fetch(url, { cache: "no-store" });
    if (!res.ok) return null;
    return (await res.json()) as T;
  } catch {
    return null;
  }
}

export function useTopHubSignals(enabled = true) {
  const [data, setData] = useState<TopHubPayload>({ hub: null, signals: [] });
  const [loading, setLoading] = useState(true);

  const load = useCallback(async () => {
    if (!enabled) {
      setLoading(false);
      return;
    }

    setLoading(true);
    try {
      const [graph, signals] = await Promise.all([
        fetchCdnJson<{ hub: KnowledgeHub }>(GRAPH_PATH),
        fetchCdnJson<{ signals: Signal[] }>(SIGNALS_PATH),
      ]);

      setData({
        hub: graph?.hub ?? null,
        signals: signals?.signals?.slice(0, 3) ?? [],
        generatedAt: new Date().toISOString(),
      });
    } finally {
      setLoading(false);
    }
  }, [enabled]);

  useEffect(() => {
    load();
  }, [load]);

  return { data, loading, refetch: load };
}
```

### 3) Card component (`src/components/CostinelTopHubSignalCard.tsx`)

```tsx
import { useTopHubSignals } from "@/lib/hooks/useTopHubSignals";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Skeleton } from "@/components/ui/skeleton";
import { ExternalLink } from "lucide-react";
import { Badge } from "@/components/ui/badge";

export function CostinelTopHubSignalCard() {
  const { data, loading } = useTopHubSignals(true);

  return (
    <Card className="col-span-1">
      <CardHeader className="flex flex-row items-center justify-between pb-3">
        <CardTitle className="text-base font-semibold">
          Top-Hub Signal
        </CardTitle>
        {data.hub && (
          <Badge variant="secondary">{data.hub.label || data.hub.id}</Badge>
        )}
      </CardHeader>
      <CardContent>
        {loading ? (
          <div className="space-y-3">
            <Skeleton className="h-5 w-3/4" />
            <Skeleton className="h-4 w-full" />
            <Skeleton className="h-4 w-5/6" />
          </div>
        ) : data.hub ? (
          <div className="space-y-3">
            <p className="text-sm text-muted-foreground">
              Most-connected hub in the knowledge graph:
              <span className="ml-1 font-medium text-foreground">
                {data.hub.label || data.hub.id}
              </span>
              {data.hub.description && (
                <span className="ml-1 text-xs text-muted-foreground">
                  — {data.hub.description}
                </span>
              )}
            </p>

            <div className="border-t pt-3">
              <p className="mb-2 text-xs font-medium uppercase tracking-wide text-muted-foreground">
                Contextual signals
              </p>
              <ul className="space-y-2">
                {data.signals.length > 0 ? (
                  data.signals.map((s) => (
                    <li key={s.id} className="text-sm">
                      <div className="flex items-start justify-between gap-2">
                        <div className="pr-2">
                          <p className="font-medium leading-tight text-foreground">
                            {s.title}
                          </p>
                          <p className="mt-0.5 text-xs text-muted-foreground line-clamp-2">
                            {s.summary}
                          </p>
                        </div>
                        {s.sourceUrl && (
                          <a
                            href={s.sourceUrl}
                            target="_blank"
                            rel="noopener noreferrer"
                            className="shrink-0 text-muted-foreground hover:text-foreground"
                            title="Open source"
                          >
                            <ExternalLink className="h-3.5 w-3.5" />
                          </a>
                        )}
                      </div>
                    </li>
                  ))
                ) : (
                  <li className="text-xs text-muted-foreground">
                    No contextual signals available.
                  </li>
                )}
              </ul>
            </div>
          </div>
        ) : (
          <p className="text-sm text-muted-foreground">
            Top hub not available.
          </p>
        )}
      </CardContent>
    </Card>
  );
}
```
