## Final Implementation Plan (≤2 h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** `bin/dataset-enrich.py` that is deterministic, shard-safe, rate-limit–aware, and immediately actionable.

### Core behavior (non-negotiable)
- Inputs via env: `SHARD_ID`, `SHARD_TOTAL=16`, `DATE`, `HF_TOKEN`, optional `REPO_ID`, `BATCH_SIZE=5000`.
- One `list_repo_tree(path=DATE, recursive=False)` → deterministic file list for the date.
- Deterministic shard assignment: `hash(slug) % SHARD_TOTAL == SHARD_ID`.
- **CDN-only downloads** (no Authorization header) to bypass API rate limits:
  - `https://huggingface.co/datasets/{REPO_ID}/resolve/main/{DATE}/{file}`
- Per-file projection to `{prompt, response}` only at parse time (avoids pyarrow CastError on mixed schemas).
- Central dedup via md5 of `(prompt, response)` using a local SQLite-backed `DedupStore` (robust fallback if repo-based store unavailable).
- Output: `batches/public-merged/{DATE}/shard{N}-{YYYYmmddHHMMSS}.jsonl`, flushed every `BATCH_SIZE` rows.
- No workflow changes; reuse existing GitHub Actions matrix.

# ── dedup --
DEDUP_DB_PATH = Path("dedup_cache.sqlite")

class DedupStore:
    """Local SQLite dedup store (robust, no repo import required)."""
    def __init__(self, path: str):
        self.path = path
        self.conn = sqlite3.connect(path, timeout=30)
        self.conn.execute("CREATE TABLE IF NOT EXISTS seen (md5 TEXT PRIMARY KEY)")
        self.conn.commit()

    def exists(self, md5: str) -> bool:
        cur = self.conn.execute("SELECT 1 FROM seen WHERE md5=?", (md5,))
        return cur.fetchone() is not None

    def add(self, md5: str) -> None:
        try:
            self.conn.execute("INSERT INTO seen(md5) VALUES (?)", (md5,))
            self.conn.commit()
        except Exception:
            pass  # ignore races/dup

    def close(self) -> None:
        self.conn.close()

dedup = DedupStore(str(DEDUP_DB_PATH))

def deterministic_shard(slug: str) -> int:
    return int(hashlib.md5(slug.encode()).hexdigest(), 16) % SHARD_TOTAL

def row_md5(prompt: str, response: str) -> str:
    return hashlib.md5((prompt + "\0" + response).encode()).hexdigest()

def list_date_files(date_folder: str) -> List[str]:
    """Single API call: list top-level files in date folder."""
    items = API.list_repo_tree(
        repo_id=REPO_ID,
        path=date_folder,
        repo_type="dataset",
    )

    for it in items:
        name = it if isinstance(it, str) else it.get("path", "")

    return filenames

def cdn_url(date_folder: str, filename: str) -> str:

def parse_parquet_to_rows(url: str) -> Iterator[Dict[str, Any]]:
    """
    Download parquet via CDN and project to {prompt,response} lazily.
    Avoids load_dataset(streaming=True) on mixed-schema repos.
    """
    try:
        import pyarrow.parquet as pq
    except Exception as e:
        sys.exit(f"pyarrow is required to read parquet: {e}")

    resp = requests.get(url, timeout=120)
    resp.raise_for_status()
    buf = io.BytesIO(resp.content)

    try:
        table = pq.read_table(buf, columns=["prompt", "response"], use_threads=False)
    except Exception:
        # fallback: try common alternate names
        try:
            table = pq.read_table(buf, use_threads=False)
        except Exception as e:
            # skip unreadable
            return

    # normalize columns
    for col, alts in [("prompt", ["instruction", "input"]), ("response", ["output", "completion"])]:
        if col not in table.column_names:
            for a in alts:
                if a in table.column_names:
                    table = table.rename_columns(
                        [a if c == col else c for c in table.column_names]
                    )
                    break

    if "prompt" not in table.column_names or "response" not in table.column_names:
        return

    prompts = table["prompt"].to_pylist()
    responses = table["response"].to_pylist()
    for p, r in zip(prompts, responses):
        if p is None or r is None:
            continue
        yield {"prompt": str(p), "response": str(r)}

def parse_jsonl_lines(chunks: Iterator[bytes]) -> Iterator[Dict[str, Any]]:
    decoder = json.JSONDecoder()
    buffer = b""
    for chunk in chunks:
        buffer += chunk
        while True:
            buffer = buffer.lstrip()
            if not buffer:
                break
            try:
                obj, idx = decoder.raw_decode(buffer.decode("utf-8"))
                buffer = buffer[idx:]
                if isinstance(obj, dict):
                    yield obj
            except (json.JSONDecodeError, UnicodeDecodeError):
                # incomplete line
                break

def stream_file(url: str) -> Iterator[bytes]:
    with requests.get(url, stream=True, timeout=60) as r:
        r.raise_for_status()
        for chunk in r.iter_content(chunk_size=16384):
            if chunk:
                yield chunk

def normalize_row(raw: Dict[str, Any]) -> Dict[str, str] | None:
    prompt = raw.get("prompt") or raw.get("instruction") or raw.get("input")
    response = raw.get("response") or raw.get("output") or raw.get("completion")
    if prompt is None or response is None:
        return None
    return {"prompt": str(prompt), "response": str(response)}
