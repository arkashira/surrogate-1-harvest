# workio / ops

## 1. Diagnosis

- **No server-side idempotency key enforcement**: clock-in/out can be duplicated by retries or LINE webhook replays, creating multiple active punches and corrupting time records.
- **Active punch uniqueness relies on app logic, not DB constraint**: race conditions between concurrent requests can create >1 active punch per employee.
- **Missing idempotency table and index**: no durable record of processed webhook IDs to guarantee safe retries or replays.
- **Clock-in/out endpoints accept unbounded retries**: no `Idempotency-Key` header check or 409 conflict response for duplicates.
- **No atomic upsert for punch state**: current flow does select → decide → insert/update, which is unsafe under load.

## 2. Proposed change

- Add `punch_idempotency` table with unique constraint on `idempotency_key` and `tenant_id`.
- Add partial unique index on `punches(tenant_id, employee_id, type)` where `type='IN'` and `clock_out_at IS NULL` to enforce single active punch at DB level.
- Modify `server/src/routes/punch.ts` to require `Idempotency-Key` header and perform atomic upsert via `punch_idempotency` before creating/updating punches.
- Return `409 Conflict` for duplicate idempotency keys and `422` for constraint violations.

## 3. Implementation

### 3.1 DB schema (server/src/db/schema.sql)

```sql
-- Idempotency table for LINE webhook / punch endpoint replays
CREATE TABLE punch_idempotency (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id       UUID NOT NULL REFERENCES tenants(id) ON DELETE CASCADE,
  idempotency_key TEXT NOT NULL,
  punch_id        UUID NOT NULL REFERENCES punches(id) ON DELETE CASCADE,
  created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  CONSTRAINT uq_tenant_idempotency UNIQUE (tenant_id, idempotency_key)
);

-- Ensure only one active IN punch per employee per tenant
CREATE UNIQUE INDEX uq_active_in_punch
ON punches (tenant_id, employee_id)
WHERE type = 'IN' AND clock_out_at IS NULL;
```

### 3.3 Route handler (server/src/routes/punch.ts)

```ts
import express from 'express';
import { pool } from '../db';
import { v4 as uuidv4 } from 'uuid';

const router = express.Router();

async function upsertPunchWithIdempotency(
  tenantId: string,
  employeeId: string,
  type: 'IN' | 'OUT',
  idempotencyKey: string,
  location?: { latitude: number; longitude: number }
) {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // 1) Check idempotency key
    const idempotencyRes = await client.query(
      `SELECT punch_id FROM punch_idempotency
       WHERE tenant_id = $1 AND idempotency_key = $2`,
      [tenantId, idempotencyKey]
    );

    if (idempotencyRes.rows.length > 0) {
      // Return existing punch (safe replay)
      const punchId = idempotencyRes.rows[0].punch_id;
      const punchRes = await client.query(
        `SELECT id, tenant_id, employee_id, type, clock_in_at, clock_out_at, latitude, longitude, created_at
         FROM punches WHERE id = $1`,
        [punchId]
      );
      await client.query('COMMIT');
      return { existing: true, punch: punchRes.rows[0] };
    }

    // 2) Create or update punch atomically
    let punchId: string;
    if (type === 'IN') {
      // Create new IN punch (DB unique index prevents double active IN)
      const newId = uuidv4();
      await client.query(
        `INSERT INTO punches (id, tenant_id, employee_id, type, clock_in_at, latitude, longitude)
         VALUES ($1, $2, $3, $4, NOW(), $5, $6)`,
        [newId, tenantId, employeeId, type, location?.latitude ?? null, location?.longitude ?? null]
      );
      punchId = newId;
    } else {
      // OUT: close latest active IN
      const outRes = await client.query(
        `UPDATE punches
         SET clock_out_at = NOW(),
             latitude = COALESCE($4, latitude),
             longitude = COALESCE($5, longitude)
         WHERE id = (
           SELECT id FROM punches
           WHERE tenant_id = $1 AND employee_id = $2 AND type = 'IN' AND clock_out_at IS NULL
           ORDER BY clock_in_at DESC LIMIT 1
           FOR UPDATE SKIP LOCKED
         )
         RETURNING id`,
        [tenantId, employeeId, location?.latitude ?? null, location?.longitude ?? null]
      );

      if (outRes.rowCount === 0) {
        await client.query('ROLLBACK');
        throw new Error('No active IN punch to clock out');
      }
      punchId = outRes.rows[0].id;
    }

    // 3) Record idempotency
    await client.query(
      `INSERT INTO punch_idempotency (tenant_id, idempotency_key, punch_id)
       VALUES ($1, $2, $3)`,
      [tenantId, idempotencyKey, punchId]
    );

    await client.query('COMMIT');

    // Fetch final punch
    const finalRes = await client.query(
      `SELECT id, tenant_id, employee_id, type, clock_in_at, clock_out_at, latitude, longitude, created_at
       FROM punches WHERE id = $1`,
      [punchId]
    );

    return { existing: false, punch: finalRes.rows[0] };
  } catch (err: any) {
    await client.query('ROLLBACK');
    // Unique violation on active IN punch
    if (err.code === '23505' && err.constraint === 'uq_active_in_punch') {
      const conflictErr: any = new Error('Active IN punch already exists');
      conflictErr.status = 409;
      throw conflictErr;
    }
    throw err;
  } finally {
    client.release();
  }
}
