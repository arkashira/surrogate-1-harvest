### 1) Build CDN manifest (non-recursive, one folder)
```python
# scripts/build_hf_cdn_manifest.py
#!/usr/bin/env python3
import argparse
import json
import os
import time
from huggingface_hub import HfApi

HF_API_RATE_LIMIT_WAIT = 360

def build_manifest(repo: str, folder: str, out_path: str):
    api = HfApi()
    items = api.list_repo_tree(repo=repo, path=folder, recursive=False)
    files = [it.rfilename for it in items if it.rfilename.endswith(".parquet")]

    manifest = {
        "repo": repo,
        "folder": folder,
        "generated_at": time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime()),
        "shards": [
            {
                "filename": f,
                "cdn_url": f"https://huggingface.co/datasets/{repo}/resolve/main/{folder}/{f}"
            }
            for f in sorted(files)
        ]
    }

    os.makedirs(os.path.dirname(out_path) or ".", exist_ok=True)
    with open(out_path, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Wrote {len(files)} shards to {out_path}")

if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True)
    parser.add_argument("--folder", required=True)
    parser.add_argument("--out", default="manifest.json")
    args = parser.parse_args()

    try:
        build_manifest(args.repo, args.folder, args.out)
    except Exception as e:
        print(f"Error: {e}")
        print(f"Waiting {HF_API_RATE_LIMIT_WAIT}s and retrying once...")
        time.sleep(HF_API_RATE_LIMIT_WAIT)
        build_manifest(args.repo, args.folder, args.out)
```

### 2) CDN-only IterableDataset (streaming, minimal memory)
```python
# surrogate/data.py
import json
import io
import pyarrow.parquet as pq
import requests
from torch.utils.data import IterableDataset

class CDNParquetIterable(IterableDataset):
    def __init__(self, manifest_path, repeat=1):
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.shards = self.manifest["shards"]
        self.repeat = repeat

    def __iter__(self):
        for _ in range(self.repeat):
            for shard in self.shards:
                resp = requests.get(shard["cdn_url"], stream=True, timeout=60)
                resp.raise_for_status()
                buf = io.BytesIO(resp.content)
                table = pq.read_table(buf)
                df = table.select(["prompt", "response"]).to_pandas()
                for _, row in df.iterrows():
                    yield {"prompt": row["prompt"], "response": row["response"]}
```
