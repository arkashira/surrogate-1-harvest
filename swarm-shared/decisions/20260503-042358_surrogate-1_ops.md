### Changes

1. **`bin/dataset-enrich.sh`** â†’ **`bin/dataset-enrich.py`**  
   - Single API call (post-rate-limit window) to `list_repo_tree(path, recursive=False)` for one date folder â†’ save `manifest.json`.  
   - Worker reads manifest, downloads each file via **CDN bypass** (`https://huggingface.co/datasets/{repo}/resolve/main/{path}`) with no Authorization header.  
   - Stream-parse each file, project to `{prompt, response}` only, compute md5, dedup via central SQLite, emit `shard-N/date/slug.parquet`.  
   - No `load_dataset(streaming=True)` on heterogeneous repos â†’ avoids pyarrow CastError.

2. **`lib/dedup.py`** (unchanged API, optional perf)  
   - Keep SQLite-backed `seen(md5)` check; add batch `executemany` for speed.

3. **`.github/workflows/ingest.yml`**  
   - Matrix `shard_id: [0..15]`.  
   - Each job runs `python bin/dataset-enrich.py --shard $SHARD_ID --date $YYYYMMDD`.  
   - No recursive `list_repo_files`; use folder-level tree calls only.  
   - On 429: wait 360s then retry (single retry policy).  
   - Output to `batches/public-merged/{date}/shard{N}-{HHMMSS}.parquet`.

4. **`requirements.txt`**  
   - Add `requests`, keep `datasets`, `huggingface_hub`, `pyarrow`, `numpy`.

CDN-bypass ingestion worker for surrogate-1.

HF_REPO = "datasets/axentx/surrogate-1-training-pairs"
BASE_CDN = f"https://huggingface.co/{HF_REPO}/resolve/main"
API = HfApi(token=os.getenv("HF_TOKEN"))

DATE_FMT = "%Y%m%d"

def list_date_folder(date_str: str):
    """Single API call: folder-level tree (non-recursive)."""
    try:
        tree = API.list_repo_tree(
            repo_id=HF_REPO,
            path=date_str,
            repo_type="dataset",
        )
        return [t.path for t in tree if t.type == "file"]
    except Exception as e:
        print(f"list_repo_tree failed: {e}", file=sys.stderr)
        return []

def save_manifest(date_str: str, files):
    out = Path("manifest") / f"{date_str}.json"
    out.parent.mkdir(exist_ok=True)
    out.write_text(json.dumps(files, indent=2))
    return out

def download_cdn(path: str, dest: Path):
    url = f"{BASE_CDN}/{path}"
    r = requests.get(url, timeout=60)
    r.raise_for_status()
    dest.write_bytes(r.content)
    return dest

def parse_projection(raw_bytes, ext):
    """
    Minimal projection to {prompt,response}.
    Extend with format-specific parsers as needed.
    """
    if ext == ".jsonl":
        import json as _json
        pairs = []
        for line in raw_bytes.decode().strip().splitlines():
            obj = _json.loads(line)
            pairs.append({
                "prompt": obj.get("prompt") or obj.get("input") or "",
                "response": obj.get("response") or obj.get("output") or "",
            })
        return pairs
    elif ext == ".parquet":
        tbl = pq.read_table(pa.BufferReader(raw_bytes))
        df = tbl.to_pandas()
        prompt_col = next((c for c in df.columns if "prompt" in c.lower()), df.columns[0])
        resp_col = next((c for c in df.columns if "response" in c.lower()), df.columns[-1])
        return df[[prompt_col, resp_col]].rename(columns={prompt_col: "prompt", resp_col: "response"}).to_dict("records")
    else:
        raise ValueError(f"Unsupported ext: {ext}")

def md5_bytes(b):
    return hashlib.md5(b).hexdigest()

class Dedup:
    def __init__(self, db_path="dedup.db"):
        self.db_path = db_path
        self.conn = sqlite3.connect(db_path)
        self.conn.execute("CREATE TABLE IF NOT EXISTS seen (md5 TEXT PRIMARY KEY)")
        self.conn.commit()

    def exists_batch(self, md5s):
        placeholders = ",".join("?" * len(md5s))
        rows = self.conn.execute(
            f"SELECT md5 FROM seen WHERE md5 IN ({placeholders})", md5s
        ).fetchall()
        return {r[0] for r in rows}

    def add_batch(self, md5s):
        self.conn.executemany("INSERT OR IGNORE INTO seen (md5) VALUES (?)", [(m,) for m in md5s])
        self.conn.commit()

def _flush(batch_raw, batch_md5s, dedup, batch_rows, schema, out_file, idx):
    existing = dedup.exists_batch(batch_md5s)
    new_rows = []
    new_md5s = []
    for (prompt, response, m) in batch_raw:
        if m not in existing:
            new_rows.append((prompt, response))
            new_md5s.append(m)
    if new_rows:
        dedup.add_batch(new_md5s)
        batch_rows.extend(new_rows)
    # Clear buffers
    batch_raw.clear()
    batch_md5s.clear()
    # Write parquet incrementally
    if batch_rows:
        tbl = pa.Table.from_pylist(
            [{"prompt": p, "response": r} for p, r in batch_rows],
            schema=schema,
        )
        pq.write_table(tbl, out_file)
        batch_rows.clear()
    print(f"  flushed batch at file {idx}")
