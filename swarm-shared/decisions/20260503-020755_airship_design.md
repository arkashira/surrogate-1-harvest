### 1. CDN-first file-list generator (run from Mac)

```python
# scripts/generate_filelist.py
import json
import os
from datetime import datetime, timezone
from huggingface_hub import HfApi

API = HfApi()
REPO = "axentx/surrogate-datasets"
DATE_FOLDER = datetime.now(timezone.utc).strftime("%Y-%m-%d")  # e.g. 2026-05-03
OUT_PATH = "filelist.json"

def main():
    # Single API call (non-recursive) to avoid pagination/429
    entries = API.list_repo_tree(REPO, path=DATE_FOLDER, recursive=False)
    files = [e.rfilename for e in entries if e.type == "file"]
    payload = {
        "repo": REPO,
        "date_folder": DATE_FOLDER,
        "files": files,
        "generated_at": datetime.now(timezone.utc).isoformat(),
        "strategy": "cdn-only",
        "note": "Embed this in train.py; Lightning workers must use CDN URLs only."
    }
    os.makedirs(os.path.dirname(OUT_PATH), exist_ok=True)
    with open(OUT_PATH, "w") as f:
        json.dump(payload, f, indent=2)
    print(f"Wrote {len(files)} files to {OUT_PATH}")

if __name__ == "__main__":
    main()
```

Run once (after rate-limit window clears):

```bash
python scripts/generate_filelist.py
```

### 2. Patch Surrogate `train.py` (CDN-only dataset loader)

```python
# surrogate/train.py  (excerpt)
import json
import os
import requests
from torch.utils.data import IterableDataset

FILELIST_PATH = os.getenv("FILELIST_PATH", "filelist.json")
CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

class CDNParquetIterable(IterableDataset):
    def __init__(self, filelist_path=FILELIST_PATH, max_files=None):
        with open(filelist_path) as f:
            manifest = json.load(f)
        self.repo = manifest["repo"]
        files = manifest["files"]
        if max_files:
            files = files[:max_files]
        self.urls = [
            CDN_TEMPLATE.format(repo=self.repo, path=f)
            for f in files
            if f.endswith(".parquet")
        ]

    def __iter__(self):
        for url in self.urls:
            # Stream download without HF API auth/rate-limit
            with requests.get(url, stream=True, timeout=60) as r:
                r.raise_for_status()
                # Write to temp or memory-map via pyarrow
                import tempfile, pyarrow.parquet as pq
                with tempfile.NamedTemporaryFile(suffix=".parquet", delete=False) as tmp:
                    for chunk in r.iter_content(chunk_size=8192):
                        tmp.write(chunk)
                    tmp.flush()
                    table = pq.read_table(tmp.name, columns=["prompt", "response"])
                    os.unlink(tmp.name)
                # Yield rows
                for i in range(table.num_rows):
                    row = table.slice(i, 1).to_pydict()
                    yield {"prompt": row["prompt"][0], "response": row["response"][0]}
```

Notes:
- No `load_dataset(streaming=True)` → avoids mixed-schema CastError.
- No HF API calls during training → bypasses 429.
- Project only `prompt`/`response` at parse time.

### 3. Lightning Studio reuse + idle guard

```python
# scripts/launch_lightning.py
import time
from lightning_sdk import Teamspace, Studio, Machine

TEAMSPACE = "axentx"
STUDIO_NAME = "surrogate-train-l40s"
MACHINE = Machine.L40S
MAX_RETRIES = 3
RETRY_WAIT = 60

def get_or_create_studio():
    ts = Teamspace(TEAMSPACE)
    running = [s for s in ts.studios if s.name == STUDIO_NAME and s.status == "Running"]
    if running:
        print(f"Reusing running studio: {STUDIO_NAME}")
        return running[0]

    print(f"Creating studio: {STUDIO_NAME}")
    return Studio.create(
        name=STUDIO_NAME,
        teamspace=TEAMSPACE,
        machine=MACHINE,
        create_ok=True,
    )

def wait_for_running(studio, timeout=600):
    elapsed = 0
    while studio.status != "Running" and elapsed < timeout:
        time.sleep(10)
        elapsed += 10
        studio.refresh()
    if studio.status != "Running":
        raise RuntimeError(f"Studio failed to start: {studio.status}")
    return studio

def run_training(script_path="train.py", args=None):
    args = args or []
    for attempt in range(1, MAX_RETRIES + 1):
        studio = get_or_create_studio()
        studio = wait_for_running(studio)

        # Guard: if stopped, restart
        if studio.status != "Running":
            print(f"Studio not running ({studio.status}). Restarting...")
            studio.start(machine=MACHINE)
            studio = wait_for_running(studio)

        print(f"Running training (attempt {attempt})")
        run = studio.run(
            command=["python", script_path] + args,
            environment="python:3.10",
        )
        # Monitor run (simplified)
        while not run.is_finished:
            time.sleep(30)
            run.refresh()
            studio.refresh()
            # If studio stopped mid-run, break to retry
            if studio.status != "Running":
                print("Studio stopped during training. Will retry.")
                break

        if run.is_finished and run.status == "finished":
            print("Training completed successfully.")
            return
        else:
            print(f"Run failed (status={run.status}). Retrying in {RETRY_WAIT}s...")
            time.sleep(RETRY_WAIT)

    raise RuntimeError("Training failed after retries.")

if __name__ == "__main__":
    run_training()
```

Usage:

```bash
python scripts/launch_lightning.py
```
