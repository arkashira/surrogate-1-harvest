# Implementation Plan — CDN-first snapshot + zero-HF-API ingestion

**Estimated effort**: ~90 min (Mac orchestrator + Lightning training changes)  
**Scope**: produce deterministic file manifest once (Mac), embed in training, and fetch exclusively via CDN during Lightning training to eliminate HF API rate-limit risk.

## 1. High-level flow

1. **Mac orchestrator** (run after rate-limit window clears):
   - Single `list_repo_tree(..., recursive=False)` per date folder in `axentx/surrogate-1-training-pairs`.
   - Build deterministic manifest: `batches/public-merged/<date>/shard-*.jsonl` files only.
   - Save as `snapshot/<date>/file-manifest.json` (sorted, sha256 of entries).
   - Commit snapshot to surrogate-1 repo (or embed directly in training container).

2. **Lightning training script**:
   - Load `file-manifest.json` (local or via raw CDN URL).
   - During `DataLoader`/dataset init, fetch each file via:
     `https://huggingface.co/datasets/axentx/surrogate-1-training-pairs/resolve/main/<path>`
   - Zero `huggingface_hub` API calls during training.
   - Project to `{prompt, response}` at parse time; ignore extra columns.

3. **Studio reuse & idle handling**:
   - Reuse running Studio when present.
   - If stopped, restart with `target.start(machine=Machine.L40S)` before `.run()`.

## 2. Mac orchestrator script (single-shot manifest)

`scripts/build_snapshot.py`

```python
#!/usr/bin/env python3
"""
Build a CDN-only file manifest for axentx/surrogate-1-training-pairs.
Run from Mac orchestrator after HF API rate-limit window clears.

Usage:
    HF_TOKEN=<token> python scripts/build_snapshot.py --date 2026-05-03 --out snapshot/2026-05-03/file-manifest.json
"""

import argparse
import hashlib
import json
import os
import sys
from datetime import datetime

from huggingface_hub import HfApi, list_repo_tree

REPO_ID = "axentx/surrogate-1-training-pairs"

def build_manifest(date: str, out_path: str) -> None:
    api = HfApi(token=os.environ.get("HF_TOKEN"))
    prefix = f"batches/public-merged/{date}/"

    # Single non-recursive list per folder to avoid pagination explosion
    entries = list_repo_tree(
        repo_id=REPO_ID,
        path=prefix,
        recursive=False,
        repo_type="dataset",
    )

    files = sorted(
        e.path for e in entries if e.type == "file" and e.path.endswith(".jsonl")
    )

    if not files:
        print(f"No JSONL files found under {prefix}", file=sys.stderr)
        sys.exit(1)

    manifest = {
        "repo_id": REPO_ID,
        "date": date,
        "generated_at_utc": datetime.utcnow().isoformat() + "Z",
        "files": files,
        "cdn_prefix": "https://huggingface.co/datasets",
    }

    # Deterministic sha256 over canonical JSON
    payload = json.dumps(manifest, sort_keys=True, separators=(",", ":")).encode()
    manifest["sha256"] = hashlib.sha256(payload).hexdigest()

    os.makedirs(os.path.dirname(out_path), exist_ok=True)
    with open(out_path, "w") as f:
        json.dump(manifest, f, indent=2, sort_keys=True)

    print(f"Manifest written to {out_path} ({len(files)} files)")

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Build CDN manifest")
    parser.add_argument("--date", required=True, help="YYYY-MM-DD folder")
    parser.add_argument("--out", required=True, help="Output JSON path")
    args = parser.parse_args()
    build_manifest(date=args.date, out_path=args.out)
```

- **Why this is safe**:
  - One API call per date folder (non-recursive), avoids `list_repo_files` recursive pagination.
  - CDN downloads (`resolve/main/...`) are not counted against API rate limits.
  - Manifest is deterministic; training can pin by `sha256`.

## 3. Lightning training dataset (CDN-only)

`training/dataset_cdn.py`

```python
import json
import logging
from pathlib import Path
from typing import Dict, List

import numpy as np
import pyarrow as pa
import pyarrow.parquet as pq
import requests
from torch.utils.data import Dataset

logger = logging.getLogger(__name__)

class CDNJsonlDataset(Dataset):
    """
    Dataset that loads JSONL files from HF CDN only.
    No huggingface_hub API calls during training.
    """

    def __init__(self, manifest_path: str, max_files: int = -1):
        with open(manifest_path) as f:
            self.manifest = json.load(f)

        self.files: List[str] = self.manifest["files"]
        if max_files > 0:
            self.files = self.files[:max_files]

        self.cdn_prefix = self.manifest.get(
            "cdn_prefix", "https://huggingface.co/datasets"
        )
        self.repo_id = self.manifest["repo_id"]

    def _cdn_url(self, path: str) -> str:
        return f"{self.cdn_prefix}/{self.repo_id}/resolve/main/{path}"

    def _load_jsonl_lines(self, path: str) -> List[Dict]:
        url = self._cdn_url(path)
        resp = requests.get(url, timeout=30)
        resp.raise_for_status()

        lines = []
        for line in resp.text.splitlines():
            line = line.strip()
            if not line:
                continue
            try:
                obj = json.loads(line)
            except json.JSONDecodeError:
                continue
            # Project to {prompt, response} only at parse time
            prompt = obj.get("prompt") or obj.get("input") or obj.get("text")
            response = obj.get("response") or obj.get("output") or obj.get("completion")
            if prompt is None or response is None:
                continue
            lines.append({"prompt": str(prompt), "response": str(response)})
        return lines

    def __len__(self) -> int:
        return len(self.files)

    def __getitem__(self, idx: int) -> Dict:
        # Return file-level batch to be collated downstream
        path = self.files[idx]
        try:
            lines = self._load_jsonl_lines(path)
            if not lines:
                # fallback empty batch
                return {"prompt": [], "response": []}
            # For simplicity, concatenate prompts/responses in this file
            # Downstream can shuffle/pack as needed
            return {
                "prompt": [ln["prompt"] for ln in lines],
                "response": [ln["response"] for ln in lines],
            }
        except Exception as exc:
            logger.warning("Failed to load %s: %s", path, exc)
            return {"prompt": [], "response": []}
```

- **Key points**:
  - Zero `datasets.load_dataset` and zero `hf_hub_download` during training.
  - CDN-only fetches bypass HF API auth/rate limits.
  - Projects to `{prompt, response}` only at parse time (handles mixed schemas).
