# workio / quality

## 1. Diagnosis

- **Missing `X-Line-Signature` verification** — webhook accepts spoofed events; enables attendance fraud and phantom clock-ins/outs.
- **No idempotency guard on LINE webhook ingestion** — LINE retries on 5xx/timeouts; duplicate clock-in/out or leave/OT records are created on replay.
- **No transactional state transitions** — clock-in/out and leave/OT mutations can leave inconsistent state on partial failure (e.g., DB write succeeds, LINE notify fails).
- **Missing monotonic replay protection** — no dedupe window or event-id tracking; replay storms create garbage data.
- **Tenant enforcement absent on webhook path** — events can be applied to wrong tenant if channel secret is reused or routing is ambiguous.

## 2. Proposed change

Create `workio/server/src/routes/webhook/line.ts` (new) and wire it into Express. Scope:
- Verify `X-Line-Signature` with HMAC-SHA256 using channel secret.
- Idempotency via `X-Line-Retry-Id` (or generated from `deliveryId`+`timestamp`+`event.type`) stored in Redis/DB with 24h TTL.
- Wrap DB mutations + LINE notify in transaction; rollback on any failure.
- Enforce tenant by channel ID (`source.userId` context resolved via channel secret → tenant mapping).
- Immutable audit log table `webhook_events` for traceability.

```ts
import { Router, Request, Response, NextFunction } from 'express';
import crypto from 'crypto';
import { Pool } from 'pg';

const router = Router();
const pool = new Pool({ connectionString: process.env.DATABASE_URL });

const CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET;
if (!CHANNEL_SECRET) {
  throw new Error('LINE_CHANNEL_SECRET is required');
}

// Helpers
function verifySignature(body: string, signature: string): boolean {
  const expected = crypto
    .createHmac('sha256', CHANNEL_SECRET)
    .update(body, 'utf8')
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

function buildDedupeKey(destination: string, event: any): string {
  // Prefer LINE-provided retry id when available; fallback to stable hash
  const deliveryId = event.deliveryId || `${event.timestamp}-${event.type}-${event.source.userId || event.source.groupId || event.source.roomId || 'unknown'}`;
  return `line:webhook:dedupe:${destination}:${deliveryId}`;
}

async function isDuplicate(key: string, client: any): Promise<boolean> {
  const res = await client.query(
    `SELECT 1 FROM webhook_dedupe WHERE dedupe_key = $1`,
    [key]
  );
  return res.rowCount > 0;
}

async function markProcessed(key: string, client: any): Promise<void> {
  await client.query(
    `INSERT INTO webhook_dedupe (dedupe_key, processed_at) VALUES ($1, NOW()) ON CONFLICT (dedupe_key) DO NOTHING`,
    [key]
  );
}

// Core handlers
async function handleClockInOut(event: any, tenantId: string, client: any): Promise<void> {
  const userId = event.source.userId;
  const type = event.type; // 'message' with text 'clock_in' / 'clock_out' or via LIFF payload
  // Simplified: expect event.message.text or event.postback.data
  const action = event.message?.text || event.postback?.data;
  if (!action || !['clock_in', 'clock_out'].includes(action)) return;

  // Transactional state transition
  const now = new Date();
  await client.query('BEGIN');
  try {
    if (action === 'clock_in') {
      await client.query(
        `INSERT INTO attendances (tenant_id, user_id, clock_in_at, created_at)
         VALUES ($1, $2, $3, NOW())`,
        [tenantId, userId, now]
      );
    } else {
      await client.query(
        `UPDATE attendances
         SET clock_out_at = $3, updated_at = NOW()
         WHERE tenant_id = $1 AND user_id = $2 AND clock_out_at IS NULL
         ORDER BY clock_in_at DESC LIMIT 1`,
        [tenantId, userId, now]
      );
    }

    // Immutable audit
    await client.query(
      `INSERT INTO webhook_events (tenant_id, channel_id, event, processed)
       VALUES ($1, $2, $3, TRUE)`,
      [tenantId, event.destination, event]
    );

    await client.query('COMMIT');
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  }
}

// Main route
router.post('/line', async (req: Request, res: Response, next: NextFunction) => {
  const signature = req.headers['x-line-signature'] as string;
  const rawBody = JSON.stringify(req.body);

  // 1) Verify signature
  if (!signature || !verifySignature(rawBody, signature)) {
    return res.status(401).json({ error: 'Invalid signature' });
  }

  const events = req.body.events;
  if (!Array.isArray(events) || events.length === 0) {
    return res.status(200).json({ ok: true });
  }

  const client = await pool.connect();
  try {
    for (const event of events) {
      const destination = event.destination;
      const tenantId = await resolveTenantIdByChannel(destination);
      if (!tenantId) {
        // Unknown tenant — log and skip to avoid cross-tenant contamination
        await client.query(
          `INSERT INTO webhook_events (tenant_id, channel_id, event, processed)
           VALUES ($1, $2, $3, FALSE)`,
          [null, destination, event]
        );
        continue;
      }

      const dedupeKey = buildDedupeKey(destination, event);
      if (await isDuplicate(dedupeKey, client)) {
        // Idempotent success response
        continue;
      }

      // Route by event content
      if (event.type === 'message' || event.type === 'postback') {
        await handleClockInOut(event, tenantId, client);
        // Extend with leave/OT handling as needed
        await handleLeaveOrOT(event, tenantId, client);
      }

      await markProcessed(dedupeKey, client);
    }
    res.status(200).json({ ok: true });
  } catch (err) {
    next(err);
  } finally {
    client.release();
  }
});

export default router;
```
