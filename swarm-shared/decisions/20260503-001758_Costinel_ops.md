# Costinel / ops

## Final Implementation Plan — Costinel “Top-Hub Signal” Card (Read-Only)

**Scope**: Read-only frontend card (≤2 h)  
**Principle**: “Sense + Signal — ไม่ Execute” (strictly no writes, no runtime mutations, no self-execution)  
**Goal**: Surface the most-connected hub (e.g., “MOC”) with contextual insights from knowledge-rag as a static, CDN-friendly signal on the dashboard.

---

### 1) Design (5 min)
- Placement: right rail or dashboard header (existing grid).
- Visual: icon + “Top hub” label + hub name pill + 1–5 insight bullets + timestamp + tag pills.
- States: loading → data → empty/error (graceful null fallback).
- Accessibility: `role="region"` + `aria-label="Top hub signal"` + `aria-live="polite"` on loading; no interactive mutations.

---

### 2) Data contract (read-only, CDN-friendly)
- Source: static JSON produced by ops/knowledge-rag pipeline (committed or uploaded to CDN).
- Path: `/data/top-hub/latest.json` (override via `REACT_APP_TOP_HUB_URL`).
- Schema:
  ```json
  {
    "hub": "MOC",
    "score": 0.94,
    "insight": "Most-connected hub; central to cost anomaly patterns and reserved-instance coverage signals.",
    "insights": [
      { "text": "Highest betweenness centrality in cost graph.", "tags": ["graph", "centrality"] },
      { "text": "Strong correlation with reserved-instance coverage gaps.", "tags": ["ri", "coverage"] }
    ],
    "updatedAt": "2026-05-03T00:11:19Z",
    "tags": ["knowledge-rag", "graph", "hub"]
  }
  ```
- Notes: `insight` kept for backward compatibility; `insights` preferred for richer bullets.

#### B) Create read-only hook: `useTopHubSignal` (15 min)
`src/hooks/useTopHubSignal.ts`
```ts
import { useEffect, useState, useCallback } from 'react';

const DEFAULT_URL = '/data/top-hub/latest.json';

export interface Insight {
  text: string;
  tags?: string[];
}

export interface TopHubSignal {
  hub: string;
  score: number;
  insight: string;
  insights: Insight[];
  updatedAt: string;
  tags: string[];
}

export function useTopHubSignal(url?: string) {
  const [data, setData] = useState<TopHubSignal | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<Error | null>(null);

  const fetchData = useCallback(async () => {
    const target = url || process.env.REACT_APP_TOP_HUB_URL || DEFAULT_URL;
    setLoading(true);
    setError(null);
    try {
      const res = await fetch(target, { cache: 'no-store' });
      if (!res.ok) throw new Error(`Failed to fetch top-hub: ${res.status}`);
      const json = await res.json();
      // Basic shape guard (read-only; fail soft)
      if (!json?.hub || typeof json.insight !== 'string') {
        throw new Error('Invalid top-hub payload');
      }
      setData({
        hub: json.hub,
        score: json.score ?? 0,
        insight: json.insight,
        insights: Array.isArray(json.insights) ? json.insights : [],
        updatedAt: json.updatedAt,
        tags: Array.isArray(json.tags) ? json.tags : [],
      });
    } catch (err: any) {
      setError(err);
      setData(null);
    } finally {
      setLoading(false);
    }
  }, [url]);

  useEffect(() => {
    fetchData();
    // Optional polling (disabled by default):
    // const id = setInterval(fetchData, 5 * 60 * 1000);
    // return () => clearInterval(id);
  }, [fetchData]);

  return { data, loading, error, refetch: fetchData };
}
```

#### C) Add card component: `TopHubSignalCard` (25 min)
`src/components/TopHubSignalCard.tsx`
```tsx
import React from 'react';
import { useTopHubSignal } from '../hooks/useTopHubSignal';

export function TopHubSignalCard({ url }: { url?: string }) {
  const { data, loading, error } = useTopHubSignal(url);

  if (loading) {
    return (
      <div
        role="status"
        aria-live="polite"
        className="p-4 rounded border border-gray-200 bg-gray-50"
      >
        <div className="flex items-center gap-2 text-sm text-gray-600">
          <span className="w-2 h-2 rounded-full bg-blue-400 animate-pulse" />
          Reading top-hub signal…
        </div>
      </div>
    );
  }

  if (error || !data) {
    // Read-only: graceful fallback, no retry UI, no console spam
    return null;
  }

  const insights = data.insights.length > 0 ? data.insights : [{ text: data.insight }];

  return (
    <article
      role="region"
      aria-label="Top hub signal"
      className="p-4 rounded-lg border border-blue-100 bg-blue-50 shadow-sm"
    >
      <div className="flex items-start justify-between gap-3">
        <div className="flex-1 min-w-0">
          <div className="flex items-center gap-2 mb-1">
            <span className="text-xs font-medium uppercase tracking-wider text-blue-600">
              Top hub
            </span>
            <span className="px-2 py-0.5 rounded bg-blue-100 text-blue-800 text-sm font-semibold">
              {data.hub}
            </span>
          </div>

          <ul className="mt-2 space-y-1" aria-label="Insights">
            {insights.map((item, idx) => (
              <li key={idx} className="text-sm text-gray-700 leading-relaxed">
                • {item.text}
              </li>
            ))}
          </ul>

          <p className="mt-2 text-xs text-gray-500">
            Updated {new Date(data.updatedAt).toLocaleString(undefined, {
              dateStyle: 'short',
              timeStyle: 'short',
            })}
          </p>

          {data.tags.length > 0 && (
            <div className="flex flex-wrap gap-1 mt-2" aria-label="Tags">
              {data.tags.map((tag) => (
                <span
                  key={tag}
                  className="px-1.5 py-0.5 rounded text-xs bg-gray-100 text-gray-600"
                >
                  #{tag}
                </span>
              ))}
            </div>
          )}
        </div>
      </div>
    </article>
  );
}
```
