# vanguard / quality

## 1. Diagnosis
- No deterministic frontend build pipeline → no HMR, no asset hashing, fragile runtime resolution.
- Missing mount point and entrypoint → cannot render or iterate locally; no clear boundary between orchestration host and browser runtime.
- No CDN-first data strategy wired into frontend → training/inference assets will hit HF API rate limits instead of using `resolve/main/` CDN bypass.
- No static manifest generation on orchestration host → frontend cannot know available files without runtime API calls.
- No reproducible dev workflow (no `package.json`, no Vite config, no TypeScript) → quality gates (lint/typecheck/tests) missing.

## 2. Proposed change
Create a minimal, production-grade Vite frontend scaffold with:
- `index.html` mount point
- `src/main.tsx` + `src/App.tsx` (React + TypeScript)
- `vite.config.ts` with strict build options and asset hashing
- `package.json` with dev/build/preview/quality scripts (dev, build, preview, typecheck, lint)
- `src/lib/manifest.ts` utility to load a static file manifest (generated by orchestration) and resolve HF CDN URLs (`https://huggingface.co/datasets/{repo}/resolve/main/{path}`)
- `.gitignore`, `tsconfig.json`, `tsconfig.node.json`

Scope: add files under `/opt/axentx/vanguard/` (root) and `src/`; no changes to existing backend yet.

import React, { useEffect, useState } from 'react';
import { loadManifest, resolveCdnUrl } from './lib/manifest';
import './App.css';

export default function App() {
  const [files, setFiles] = useState<string[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    loadManifest()
      .then((list) => {
        setFiles(list);
        setLoading(false);
      })
      .catch((err) => {
        setError(err instanceof Error ? err.message : String(err));
        setLoading(false);
      });
  }, []);

  if (loading) return <div className="status">Loading manifest...</div>;
  if (error) return <div className="status error">Error: {error}</div>;

  return (
    <main className="container">
      <h1>Vanguard — CDN-first dataset viewer</h1>
      <p className="hint">
        Manifest loaded: {files.length} file(s). All URLs use HF CDN (resolve/main) — no API auth required.
      </p>
      <ul className="file-list">
        {files.map((f) => (
          <li key={f} className="file-item">
            <a href={resolveCdnUrl(f)} target="_blank" rel="noopener noreferrer">
              {f}
            </a>
          </li>
        ))}
      </ul>
    </main>
  );
}

const DEFAULT_MANIFEST_PATH = '/manifest/files.json';

export function resolveCdnUrl(filePath: string, options?: { repo?: string }): string {
  const repo = options?.repo || 'datasets/your-org/your-repo';
  // CDN bypass: https://huggingface.co/datasets/{repo}/resolve/main/{path}
  return `https://huggingface.co/datasets/${repo}/resolve/main/${filePath}`;
}
