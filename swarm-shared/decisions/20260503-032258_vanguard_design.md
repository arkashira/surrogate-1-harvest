# vanguard / design

## Final synthesized answer (strongest parts, contradictions resolved)

### 1. Diagnosis (merged, tightened)
- **No CDN-first manifest**: ingestion/training scripts still call `list_repo_tree`/`load_dataset` at runtime → 429s and non-reproducible runs.
- **No content-addressed, date/slug-pinned file list**: training jobs cannot pin an exact data slice; runs are not reproducible.
- **No deterministic fallback when HF API rate-limits**: training fails instead of falling back to CDN-only paths.
- **Studio reuse/quota discipline not enforced**: idle-stop kills training and wastes quota.
- **Schema/attribution inconsistency**: extra columns (`source`, `ts`) pollute surrogate-1 schema; attribution should live in filename/path, not schema.

### 2. Proposed change (merged)
Create **one small, focused module** and **one example script**:
- `/opt/axentx/vanguard/design/cdn_manifest.py`
  - One-time manifest builder (Mac/CI) using `list_repo_tree`.
  - CDN-only loader with **remote parquet streaming** (zero HF API calls during training).
  - Deterministic fallback: if HF API fails, use last-known-good manifest; never fail open to runtime API calls.
  - Schema projection to `{prompt, response}`; attribution via `slug` and filename pattern `batches/mirror-merged/{date}/{slug}.parquet`.
- `/opt/axentx/vanguard/examples/train_cdn_first.py`
  - Orchestrates manifest build (once) and launches training via Lightning Studio with **reuse + idle-stop guard**.
  - Uses pinned manifest path to guarantee reproducibility.

### 3. Implementation (merged + hardened)

# /opt/axentx/vanguard/design/cdn_manifest.py
import json
import os
import time
from datetime import datetime, timezone
from pathlib import Path
from typing import Dict, List, Optional

import requests

try:
    from huggingface_hub import HfApi, list_repo_tree, hf_hub_download
except ImportError:
    HfApi = None  # optional for manifest build; CDN loader does not require it

CDN_BASE = "https://huggingface.co/datasets"

def build_cdn_manifest(
    repo: str,
    date_folder: str,
    out_path: Optional[str] = None,
    api_token: Optional[str] = None,
) -> Dict:
    """
    One-time manifest build (run on Mac/CI after rate-limit window clears).
    Produces content-addressed, date/slug-pinned manifest.
    """
    if HfApi is None:
        raise RuntimeError("huggingface_hub required to list repo tree (one-time).")

    api = HfApi(token=api_token)
    tree = list_repo_tree(
        repo=repo,
        path=date_folder,
        recursive=False,
        repo_type="dataset",
    )

    files = []
    for item in tree:
        if item.type != "file":
            continue
        slug = Path(item.path).stem
        cdn_url = f"{CDN_BASE}/{repo}/resolve/main/{item.path}"
        files.append(
            {
                "slug": slug,
                "path": item.path,
                "cdn_url": cdn_url,
                "size": getattr(item, "size", None),
                "lfs": bool(getattr(item, "lfs", False)),
            }
        )

    manifest = {
        "repo": repo,
        "date_folder": date_folder,
        "generated_utc": datetime.now(timezone.utc).isoformat(),
        "generated_by": "cdn_manifest.build_cdn_manifest",
        "files": files,
        "total_files": len(files),
    }

    if out_path:
        out_path = Path(out_path)
        out_path.parent.mkdir(parents=True, exist_ok=True)
        out_path.write_text(json.dumps(manifest, indent=2))
    return manifest


def load_parquet_via_cdn(manifest_path: str):
    """
    CDN-only loader for training.
    Zero HF API calls; streams remote parquet via HTTP.
    Projects to {prompt, response}; attribution via slug.
    """
    import pyarrow.parquet as pq
    import pandas as pd

    manifest = json.loads(Path(manifest_path).read_text())
    if not manifest.get("files"):
        raise ValueError(f"No files in manifest: {manifest_path}")

    rows = []
    for f in manifest["files"]:
        url = f["cdn_url"]
        # Stream remote parquet; request only needed columns
        tbl = pq.read_table(url, columns=["prompt", "response"])
        df = tbl.to_pandas()
        # Enforce schema: only prompt/response; attribution via slug
        df = df[["prompt", "response"]].copy()
        df["slug"] = f["slug"]
        rows.append(df)

    out = pd.concat(rows, ignore_index=True)
    return out


def reuse_or_create_studio(
    name: str,

    machine: str = "L40S",
    target_script: str = "examples/train_cdn_first.py",
    manifest_path: str = "design/latest_manifest.json",
):
    """
    Reuse running studio if available; guard against idle-stop.
    """
    try:
        from lightning import Fabric, Studio, Teamspace
    except ImportError:
        print("lightning not installed; skipping studio reuse.")
        return

    ts = Teamspace(teamspace)
    running = None
    for s in ts.studios():
        if s.name == name and s.status == "Running":
            running = s
            break

    if running:
        print(f"Reusing running studio: {name}")
        studio = running
    else:
        print(f"Creating studio: {name}")
        studio = Studio(
            name=name,
            teamspace=teamspace,
            machine=machine,
            create_ok=True,
        )

    # Guard: if stopped, restart
    if studio.status != "Running":
        print(f"Studio stopped ({studio.status}); restarting...")
        studio.start(machine=machine)

    # Run training via Fabric inside studio
    fabric = Fabric(devices=1, accelerator="cuda", precision="bf16-mixed")
    fabric.launch(
