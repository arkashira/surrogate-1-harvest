# airship / design

## Final Synthesized Implementation (≤2h)

**Chosen approach**: CDN-first deterministic ingestion + Lightning Studio reuse/guard + Arkship UI idle-stop resilience.  
**Why**: This combination removes the HF API 429 bottleneck, recovers the most Lightning quota fastest, and prevents silent training death from idle-stop—while keeping Mac as orchestrator only.

---

## 1. CDN-first deterministic filelist (single API call)

```python
# scripts/generate_filelist.py
import json
import os
from datetime import datetime
from huggingface_hub import list_repo_tree

REPO = "axentx/surrogate-dataset"
DATE_FOLDER = os.getenv("SURROGATE_DATE", datetime.utcnow().strftime("%Y-%m-%d"))
OUT_PATH = "filelist.json"

def main():
    # One rate-limited call; everything else is CDN.
    tree = list_repo_tree(repo_id=REPO, path=DATE_FOLDER, recursive=False)
    files = sorted(f.rfilename for f in tree if f.rfilename.endswith(".parquet"))

    payload = {
        "repo": REPO,
        "date": DATE_FOLDER,
        "files": files,
        "generated_at": datetime.utcnow().isoformat() + "Z",
        "note": "CDN-only; use resolve/main/ URLs to bypass HF API during training"
    }

    with open(OUT_PATH, "w") as f:
        json.dump(payload, f, indent=2)

    print(f"Wrote {len(files)} files to {OUT_PATH}")

if __name__ == "__main__":
    main()
```

Usage (Mac orchestrator):
```bash
python scripts/generate_filelist.py
```

## 2. Surrogate training: CDN-only dataloader (zero HF API during training)

```python
# surrogate/train.py
import json
import pyarrow.parquet as pq
import requests
import torch
from torch.utils.data import IterableDataset, DataLoader
from typing import Iterator, Dict, Any

class CDNParquetDataset(IterableDataset):
    def __init__(self, filelist_path: str = "filelist.json"):
        with open(filelist_path) as f:
            meta = json.load(f)
        self.repo = meta["repo"]
        self.files = meta["files"]

    def _stream_file(self, path: str):
        url = f"https://huggingface.co/datasets/{self.repo}/resolve/main/{path}"
        local_path = "/tmp/current.parquet"
        with requests.get(url, stream=True, timeout=120) as r:
            r.raise_for_status()
            with open(local_path, "wb") as f:
                for chunk in r.iter_content(chunk_size=16384):
                    f.write(chunk)
        table = pq.read_table(local_path, columns=["prompt", "response"])
        return table.to_pylist()

    def __iter__(self) -> Iterator[Dict[str, Any]]:
        for f in self.files:
            for row in self._stream_file(f):
                yield row

def build_dataloader(filelist_path: str = "filelist.json", batch_size: int = 8) -> DataLoader:
    dataset = CDNParquetDataset(filelist_path)
    return DataLoader(dataset, batch_size=batch_size, num_workers=0)
```

Notes:
- No `load_dataset(streaming=True)` on heterogeneous repos.
- No HF API calls in training loop → immune to 429.
- Only `prompt`/`response` projected; attribution via filename pattern.

## 3. Lightning Studio guard + reuse (quota saver)

```python
# surrogate/lightning_utils.py
from lightning import Studio, Teamspace, Machine
from typing import Optional

MACHINE = Machine.L40S  # fallback; H200 requires lightning-lambda-prod
STUDIO_NAME = "surrogate-trainer"

def get_or_create_studio() -> Studio:
    # Reuse running studio to save quota
    for s in Teamspace.studios:
        if s.name == STUDIO_NAME and s.status == "Running":
            print(f"Reusing running studio: {s.name}")
            return s

    print(f"Creating studio: {STUDIO_NAME}")
    return Studio(
        name=STUDIO_NAME,
        machine=MACHINE,
        create_ok=True,
        exist_ok=True,
    )

def ensure_running(studio: Studio) -> Studio:
    # Guard against idle-stop death
    if studio.status != "Running":
        print(f"Studio {studio.name} is {studio.status}; restarting...")
        studio.start(machine=MACHINE)
    return studio
```

## 4. Arkship UI: training-status endpoint + idle-stop banner

```python
# arkship/api/training_status.py
from fastapi import APIRouter
from lightning_utils import get_or_create_studio, ensure_running

router = APIRouter()

@router.get("/training/status")
def training_status():
    studio = get_or_create_studio()
    studio = ensure_running(studio)
    return {
        "studio": studio.name,
        "status": studio.status,
        "machine": str(studio.machine),
        "idle_stop_warning": studio.status != "Running",
        "message": "Training active" if studio.status == "Running" else "Studio stopped by idle; restart recommended"
    }
```

Minimal frontend banner (Arkship UI):

```tsx
// arkship/frontend/components/TrainingStatusBanner.tsx
import useSWR from "swr";

export default function TrainingStatusBanner() {
  const { data } = useSWR("/api/training/status", (url) =>
    fetch(url).then((r) => r.json())
  );

  if (!data || data.status === "Running") return null;

  return (
    <div className="fixed top-4 right-4 bg-amber-600 text-white px-4 py-2 rounded shadow-lg z-50">
      <strong>Surrogate training paused</strong> — Studio idle-stopped.
      <button
        className="ml-3 underline"
        onClick={() => fetch("/api/training/status", { method: "GET" })}
      >
        Refresh
      </button>
    </div>
  );
}
```

Register route in Arkship main:

```python
# arkship/main.py
from fastapi import FastAPI
from arkship.api.training_status import router as training_router

app = FastAPI()
app.include_router(training_router, prefix="/api")
```
