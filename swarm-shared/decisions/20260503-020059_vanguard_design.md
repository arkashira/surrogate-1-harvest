# vanguard / design

# Design improvement: CDN-bypass manifest + schema-safe ingestion

## 1. Diagnosis
- No persisted `(repo, dateFolder) → file-list` manifest; every training/data-selection run triggers authenticated `list_repo_tree` against HF API, burning quota and risking 429s.
- Training/data loader likely uses `load_dataset(streaming=True)` or repeated per-file API calls on heterogeneous repos, causing `pyarrow.CastError` from mixed schemas.
- Attribution/metadata columns (`source`, `ts`) and mixed schemas written into `enriched/` break downstream surrogate-1 training expectations.
- No deterministic fallback when Lightning Studio is stopped/idle, causing lost training hours and quota waste.
- Mac/local runs may attempt `model.from_pretrained()` or heavy compute instead of delegating to Lightning/Kaggle/Cerebras.

## 2. Proposed change
Create `/opt/axentx/vanguard/design/` with:
- `file_manifest.py` — one-time Mac script to list a date folder and emit `file_manifest.json` (non-recursive) for CDN-only training.
- `project_to_pair.py` — ingestion helper that downloads via CDN URLs, projects to `{prompt, response}` only, and writes `batches/mirror-merged/{date}/{slug}.parquet` with no extra columns.
- `studio_launcher.py` — Lightning SDK wrapper that reuses running studios, checks status/idle-stop, and restarts on L40S (or Lambda H200 if account available).
- `README.md` — short design/runbook for the pattern.

Scope: new files only; no changes to existing code yet.

### file_manifest.py
```python
#!/usr/bin/env python3
"""
Generate a file manifest for a single date folder in a HF dataset repo.
Run from Mac (or any API-capable host) once per date folder.
Usage:
  HF_TOKEN=hf_xxx python file_manifest.py \
    --repo bigcode/the-stack \
    --date-folder 2024-01-15 \
    --out file_manifest.json
"""
import argparse
import json
import os
from huggingface_hub import HfApi

def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True)
    parser.add_argument("--date-folder", required=True)
    parser.add_argument("--out", default="file_manifest.json")
    args = parser.parse_args()

    api = HfApi(token=os.getenv("HF_TOKEN"))
    # Non-recursive to avoid 100x pagination and heavy rate-limit burn.
    tree = api.list_repo_tree(repo_id=args.repo, path=args.date_folder, recursive=False)
    files = [item.rfilename for item in tree if item.type == "file"]

    manifest = {
        "repo": args.repo,
        "date_folder": args.date_folder,
        "files": sorted(files),
        # CDN-only URLs (no Authorization header required)
        "cdn_base": f"https://huggingface.co/datasets/{args.repo}/resolve/main",
    }
    with open(args.out, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Wrote {len(files)} files to {args.out}")

if __name__ == "__main__":
    main()
```

### project_to_pair.py
```python
#!/usr/bin/env python3
"""
Download files via CDN, project to {prompt, response}, write parquet.
Keeps schema minimal for surrogate-1 training.
"""
import json
import hashlib
import pyarrow as pa
import pyarrow.parquet as pq
import requests
from pathlib import Path
from typing import Iterator, Tuple

CDN_TIMEOUT = 30

def cdn_url(repo: str, filepath: str) -> str:
    return f"https://huggingface.co/datasets/{repo}/resolve/main/{filepath}"

def parse_file(content: bytes, filepath: str) -> Iterator[Tuple[str, str]]:
    """
    Placeholder parser: adapt per repo/schema.
    Must return (prompt, response) pairs.
    """
    # Example: newline-delimited JSON with fields "instruction","output"
    # Replace with repo-specific logic.
    try:
        text = content.decode("utf-8")
        for line in text.splitlines():
            line = line.strip()
            if not line:
                continue
            # naive fallback: treat whole file as prompt, empty response
            yield line, ""
    except Exception:
        # If decoding/projection fails, skip file.
        return

def hash_slug(filepath: str) -> str:
    return hashlib.sha256(filepath.encode()).hexdigest()[:16]

def build_parquet(manifest_path: str, out_dir: Path) -> None:
    with open(manifest_path) as f:
        m = json.load(f)

    repo = m["repo"]
    date_folder = m["date_folder"]
    out_dir = out_dir / date_folder
    out_dir.mkdir(parents=True, exist_ok=True)

    rows = []
    for fn in m["files"]:
        url = cdn_url(repo, f"{date_folder}/{fn}")
        try:
            r = requests.get(url, timeout=CDN_TIMEOUT)
            r.raise_for_status()
        except Exception as e:
            print(f"Skip {fn}: {e}")
            continue

        for prompt, response in parse_file(r.content, fn):
            rows.append({"prompt": prompt, "response": response})

    if not rows:
        print("No rows extracted.")
        return

    table = pa.Table.from_pylist(rows, schema=pa.schema([
        pa.field("prompt", pa.string()),
        pa.field("response", pa.string()),
    ]))

    slug = hash_slug(date_folder)
    out_path = out_dir / f"{slug}.parquet"
    pq.write_table(table, out_path)
    print(f"Wrote {len(rows)} rows to {out_path}")

if __name__ == "__main__":
    import sys
    if len(sys.argv) != 3:
        print("Usage: python project_to_pair.py manifest.json out_dir")
        sys.exit(1)
    build_parquet(sys.argv[1], Path(sys.argv[2]))
```

    name: str,
    machine: L.Machine = L.Machine.L40S,
    cloud: str = "lightning-public-prod",
) -> L.Studio:
    teamspace = L.Teamspace()
    for s in teamspace.studios:
        if s.name == name and s.status == "running":
            print(f"Reusing running studio: {name}")
            return s

    # If exists but stopped (idle timeout killed training), restart.
    for s in teamspace.studios:
        if s.name == name:
            print(f"Found stopped studio {name}; restarting on {machine} in {cloud}")
            s.start(machine=machine, cloud=cloud)
            return s

    print(f"Creating studio {name} on {machine} in {cloud}")
    return L.Studio(
        name=name,
        machine=machine,
        cloud=cloud,
