#!/usr/bin/env python3
"""
CDN-bypass, manifest-driven ingestion worker for surrogate-1.

Env:
  SHARD_ID        int (0..SHARD_TOTAL-1)
  SHARD_TOTAL     int (default 16)
  DATE            YYYY-MM-DD (e.g. 2026-05-03)
  HF_TOKEN        write token for axentx/surrogate-1-training-pairs
  HF_REPO         default: datasets/axentx/surrogate-1-training-pairs
  API_BASE        default: https://huggingface.co
"""

import os
import sys
import json
import hashlib
import time
from datetime import datetime, timezone
from pathlib import Path
from typing import Any, Dict, List, Optional

import requests
from huggingface_hub import HfApi

# ---------- config ----------
HF_REPO = os.getenv("HF_REPO", "datasets/axentx/surrogate-1-training-pairs")
API_BASE = os.getenv("API_BASE", "https://huggingface.co").rstrip("/")
DATE = os.getenv("DATE")
SHARD_ID = int(os.getenv("SHARD_ID", "-1"))
SHARD_TOTAL = int(os.getenv("SHARD_TOTAL", "16"))
HF_TOKEN = os.getenv("HF_TOKEN", "")

if not DATE:
    print("ERROR: DATE env required (YYYY-MM-DD)", file=sys.stderr)
    sys.exit(1)
if SHARD_ID < 0 or SHARD_ID >= SHARD_TOTAL:
    print(f"ERROR: SHARD_ID must be in [0, {SHARD_TOTAL - 1}]", file=sys.stderr)
    sys.exit(1)

WORKDIR = Path(__file__).parent.parent
MANIFEST_PATH = WORKDIR / f"manifest-{DATE}.json"
OUT_DIR = WORKDIR / "batches" / "public-merged" / DATE
OUT_DIR.mkdir(parents=True, exist_ok=True)
TIMESTAMP = datetime.now(timezone.utc).strftime("%H%M%S")
OUT_FILE = OUT_DIR / f"shard{SHARD_ID}-{TIMESTAMP}.jsonl"

# ---------- dedup ----------
sys.path.insert(0, str(WORKDIR / "lib"))
try:
    from dedup import DedupStore
except Exception as e:
    print(f"ERROR: cannot import dedup: {e}", file=sys.stderr)
    sys.exit(1)

dedup = DedupStore()

# ---------- hf api ----------
api = HfApi(token=HF_TOKEN) if HF_TOKEN else None

# ---------- helpers ----------
def slug_for_path(path: str) -> str:
    """
    Convert repo file path to stable slug.
    Examples:
      en/2026-05-03/abc123.jsonl -> en/2026-05-03/abc123
      public/2026-05-03/abc123.parquet -> public/2026-05-03/abc123
    """
    p = Path(path)
    return str(p.with_suffix("")).replace("\\", "/")

def bucket_for_slug(slug: str) -> int:
    return int(hashlib.sha256(slug.encode()).hexdigest(), 16) % SHARD_TOTAL

def list_date_files(date_folder: str) -> List[str]:
    """
    Single API call: list top-level files in date folder (non-recursive).
    Returns relative paths.
    """
    try:
        tree = api.list_repo_tree(
            repo_id=HF_REPO,
            path=date_folder,
            recursive=False,
            repo_type="dataset",
        )
    except Exception as e:
        print(f"ERROR: list_repo_tree failed for {date_folder}: {e}", file=sys.stderr)
        # If token missing/429, fallback: try to read existing manifest
        if MANIFEST_PATH.exists():
            print(f"WARNING: using cached manifest {MANIFEST_PATH}", file=sys.stderr)
            with open(MANIFEST_PATH) as f:
                return json.load(f)
        raise

    files = [item.rfilename for item in tree if item.rfilename]
    # save manifest for reproducibility / other shards
    with open(MANIFEST_PATH, "w") as f:
        json.dump(files, f)
    return files

def cdn_download_url(repo: str, path: str) -> str:
    # repo format: datasets/owner/name
    parts = repo.split("/")
    if len(parts) == 3 and parts[0] == "datasets":
        _, owner, name = parts
    else:
        owner, name = "axentx", "surrogate-1-training-pairs"
    return f"https://huggingface.co/datasets/{owner}/{name}/resolve/main/{path}"

def parse_file_to_pairs(local_path: Path) -> List[Dict[str, str]]:
    """
    Project heterogeneous file schemas to {prompt, response}.
    Supports:
      - jsonl with 'prompt'/'response' or 'instruction'/'output' or 'input'/'target'
      - parquet via pyarrow (project only required cols)
    """
    import pyarrow.parquet as pq

    pairs = []
    suffix = local_path.suffix.lower()

    try:
        if suffix == ".jsonl":
            with open(local_path, "r", encoding="utf-8") as f:
                for line in f:
                    line = line.strip()
                    if not line:
                        continue
                    obj = json.loads(line)
                    prompt, response = None, None
                    if "prompt" in obj and "response" in obj:
                        prompt, response = obj["prompt"], obj["response"]
                    elif "instruction" in obj and "output" in obj:
                        prompt, response = obj["instruction"], obj["output"]
                    elif "input" in obj and "target" in obj:
                        prompt, response = obj["input"], obj["target"]
                    elif "text" in obj:
                        # some HF datasets provide single text col
                        prompt, response = "", obj["text"]
                    else:
                        # best-effort: use first two string cols
                        ks = [k for k in obj if isinstance(obj[k], str)]
                        if len(ks) >= 2:
                            prompt, response = obj[ks[0]], obj[ks[1]]
                        elif len(ks) == 1:
                            prompt, response = "", obj[ks[0]]
                        else:
                            continue
                    if prompt is not None and response is not None:
                        pairs.append({"prompt": str(prompt), "response": str(response)})

        elif suffix == ".parquet":

            try:
                tbl = pq.read_table(local_path, columns=["prompt", "response"])
                if tbl.num_rows > 0 and set(tbl.schema.names) == {"prompt", "response"}:
                    df = tbl.to_pandas()
                    for _, row in df.iterrows():
