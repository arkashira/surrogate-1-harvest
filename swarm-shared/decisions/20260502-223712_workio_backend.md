---

## 1. Diagnosis (merged)
- Repeated `channelId → tenantId` DB lookups on `/webhook/line` cause DB thrash and increase 429/timeout risk under traffic spikes.
- No request-scoped or shared cache for tenant metadata (timezone, feature flags) needed during clock-in/out validation and leave/OT rule checks.
- Webhook handler performs full tenant row fetches (including unused columns) instead of minimal projection.
- Missing fast-path validation for unrecognized `source.channelId` before heavier parsing/validation, wasting CPU/DB on malformed/bot traffic.
- No cache stampede protection when many events for the same tenant arrive while cache is cold.
- No cache invalidation strategy when tenant settings or LINE channel config changes.

---

## 2. Scope of change
- **New file**: `server/src/middleware/lineTenant.ts`
- **Update**: `server/src/routes/webhook/line.ts`
- **Install**: `lru-cache`
- **Scope**:
  - Add a shared LRU cache (`channelId → { tenantId, timezone, features }`) with 5-minute TTL.
  - Add per-key `Promise` memoization to prevent cache stampedes/thundering herds.
  - Use minimal DB projection for tenant+channel resolution.
  - Add early-exit for missing/invalid `channelId` before heavier work.
  - Provide explicit invalidation helpers to be called from tenant/channel update endpoints.

### Create `server/src/middleware/lineTenant.ts`
```ts
// server/src/middleware/lineTenant.ts
import { LRUCache } from 'lru-cache';
import { db } from '../db/index.js';
import { tenants, lineChannels } from '../db/schema.js';
import { and, eq } from 'drizzle-orm';

export type TenantMeta = {
  tenantId: string;
  timezone: string;
  features: Record<string, boolean>;
};

const tenantMetaCache = new LRUCache<string, TenantMeta>({
  max: 500,
  ttl: 1000 * 60 * 5, // 5 minutes (strong default; can be tuned)
});

// In-flight request memoization to prevent stampedes
const inFlight = new Map<string, Promise<TenantMeta | null>>();

export async function resolveTenantByChannelId(
  channelId: string
): Promise<TenantMeta | null> {
  if (!channelId) return null;

  const cached = tenantMetaCache.get(channelId);
  if (cached) return cached;

  // Prevent stampede: reuse in-flight promise for same channelId
  const pending = inFlight.get(channelId);
  if (pending) return pending;

  const promise = (async () => {
    try {
      const rows = await db
        .select({
          tenantId: tenants.id,
          timezone: tenants.timezone,
          features: tenants.features,
        })
        .from(lineChannels)
        .innerJoin(tenants, eq(lineChannels.tenantId, tenants.id))
        .where(
          and(
            eq(lineChannels.channelId, channelId),
            eq(lineChannels.active, true),
            eq(tenants.active, true)
          )
        )
        .limit(1);

      if (rows.length === 0) return null;
      const meta = rows[0] as TenantMeta;
      tenantMetaCache.set(channelId, meta);
      return meta;
    } finally {
      inFlight.delete(channelId);
    }
  })();

  inFlight.set(channelId, promise);
  return promise;
}

export function invalidateTenantMetaForChannel(channelId: string) {
  tenantMetaCache.delete(channelId);
  // Note: in-flight promises are allowed to finish; they will not re-cache stale data.
}

export function invalidateTenantMetaForTenant(tenantId: string) {
  // Best-effort targeted invalidation for common scale.
  // For stronger guarantees at larger scale, maintain reverse index (tenantId -> Set<channelId>).
  // Here we rely on TTL + explicit channel invalidation from admin flows.
  // Optional: iterate keys (small LRU) and remove entries where meta.tenantId === tenantId
  for (const [key, value] of tenantMetaCache.entries()) {
    if (value.tenantId === tenantId) {
      tenantMetaCache.delete(key);
    }
  }
}
```

### Update `server/src/routes/webhook/line.ts`
```ts
// server/src/routes/webhook/line.ts
import { Router } from 'express';
import crypto from 'crypto';
import { resolveTenantByChannelId } from '../../middleware/lineTenant.js';
import { db } from '../../db/index.js';
import { clockEvents } from '../../db/schema.js';

const router = Router();

function verifySignature(rawBody: Buffer, signature: string, secret: string): boolean {
  if (!secret) return false;
  const expected = crypto.createHmac('sha256', secret).update(rawBody).digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

// Raw-body preservation middleware for signature verification
function preserveRawBody(req: any, res: any, next: any) {
  if (Buffer.isBuffer(req.body)) return next();
  // If body was already parsed as JSON, we cannot verify signature reliably.
  // Reject to avoid security bypass.
  res.status(400).json({ error: 'invalid_body' });
}

router.post(
  '/line',
  preserveRawBody,
  (req, res, next) => {
    const sig = req.headers['x-line-signature'] as string;
    const secret = process.env.LINE_CHANNEL_SECRET || '';
    if (!verifySignature(req.body, sig, secret)) {
      return res.status(401).json({ error: 'invalid_signature' });
    }
    next();
  },
  (req, res) => {
    const events = req.body?.events;
    if (!Array.isArray(events) || events.length === 0) {
      return res.status(400).json({ error: 'no_events' });
    }

    // Early validation: ensure all events have required source fields before async processing
    for (const ev of events) {
      if (!ev?.source?.channelId || !ev?.source?.userId) {
        // Log malformed event but continue processing others
        console.warn('Malformed LINE event skipped', { event: ev?.type });
        continue;
      }
    }

    // Acknowledge quickly to avoid LINE webhook timeouts
    res.status(200).json({ message: 'accepted' });

    // Process async to avoid webhook timeouts
    processLineEvents(events).catch((err) => {
      console.error('LINE webhook processing error:', err);
    });
  }
);

async function processLineEvents(events: any[]) {
  for (const ev of events) {
    const channelId = ev.source?.channelId;
    const userId = ev.source?.userId;
    if (!channelId || !userId) continue;

    const meta = await resolveTenantByChannelId(channelId);
    if (!meta) {
      console.warn(`No tenant for channel ${channelId}`);
      continue;
    }

    // Minimal handling for clock in/out events (expand as needed)
    if (ev.type === 'message' && ev.message?.type === 'text') {
      const text = ev.message.text.trim().toLowerCase();
      if (text === 'clock in' || text === 'clock out') {
        await db.insert(clockEvents).values({
          tenantId: meta.tenantId,
          userId,
          eventType: text === 'clock in' ? 'in' : 'out',
          timestamp: new Date(),
          latitude: ev.message.location?.latitude ?? null,
          longitude: ev.message.location?.longitude ?? null,
        });
      }
    }
  }
}

export default router;
```
