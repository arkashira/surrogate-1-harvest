### 1. Backend: list HF folder (recursive, correct)

```python
# surrogate/api/training.py
import os
import json
from pathlib import Path
from fastapi import APIRouter, HTTPException
from huggingface_hub import list_repo_tree

router = APIRouter()

@router.post("/list-hf-folder")
async def list_hf_folder(repo: str, path: str, token: str = None):
    try:
        items = list_repo_tree(
            repo_id=repo,
            path=path,
            recursive=True,
            token=token or os.getenv("HF_TOKEN")
        )
        # items may include subdirs; keep only files
        file_paths = sorted([it.rfilename for it in items if it.type == "file"])
        out_dir = Path("training")
        out_dir.mkdir(exist_ok=True, parents=True)
        out_file = out_dir / "file-list.json"
        out_file.write_text(json.dumps(file_paths, indent=2))
        return {"files": file_paths, "saved_to": str(out_file)}
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))
```

### 2. Lightning Studio reuse + safe restart

```python
# training/launch_studio.py
import time
from pathlib import Path
from lightning_sdk import Studio, Machine, Teamspace

TEAMSPACE = "your-teamspace"
STUDIO_NAME = "surrogate-training"

def get_or_create_studio(machine: Machine = Machine.L40S):
    teamspace = Teamspace(name=TEAMSPACE)
    for s in teamspace.studios:
        if s.name == STUDIO_NAME:
            if s.status == "running":
                return s
            if s.status == "stopped":
                s.start(machine=machine)
                _wait_running(s)
                return s
            # if starting or other transient, wait briefly
            _wait_running(s)
            return s
    studio = Studio.create(
        name=STUDIO_NAME,
        teamspace=TEAMSPACE,
        machine=machine,
        create_ok=True,
    )
    _wait_running(studio)
    return studio

def _wait_running(studio, timeout=300, interval=10):
    elapsed = 0
    while elapsed < timeout:
        studio.refresh()
        if studio.status == "running":
            return
        if studio.status == "error":
            raise RuntimeError(f"Studio {studio.name} entered error state")
        time.sleep(interval)
        elapsed += interval
    raise TimeoutError("Studio failed to reach running state in time")

def run_training(script_path: str, args=None):
    studio = get_or_create_studio()
    args = args or []
    if not Path("training/file-list.json").exists():
        raise FileNotFoundError("training/file-list.json not found — run list-hf-folder first")
    run = studio.run(
        command=f"python {script_path} {' '.join(args)}",
        environment="lightning",
    )
    return run
```

### 3. CDN-only data loader (robust)

```python
# training/train.py
import json
import time
import requests
from pathlib import Path
from datasets import Dataset

HF_REPO = "your/dataset-repo"
FILE_LIST = Path("training/file-list.json")

def load_cdn_files(file_list, repo=HF_REPO, max_retries=3, timeout=30):
    rows = []
    for rel_path in file_list:
        url = f"https://huggingface.co/datasets/{repo}/resolve/main/{rel_path}"
        for attempt in range(1, max_retries + 1):
            try:
                r = requests.get(url, timeout=timeout)
                r.raise_for_status()
                items = parse_file_content(r.content, rel_path)
                rows.extend(items)
                break
            except Exception as exc:
                if attempt == max_retries:
                    raise RuntimeError(f"Failed to fetch {url}: {exc}") from exc
                time.sleep(2 ** attempt)
    return rows

def parse_file_content(content, rel_path):
    # Adapt to your actual file format(s). Example for JSONL:
    lines = content.decode().strip().splitlines()
    pairs = []
    for line in lines:
        line = line.strip()
        if not line:
            continue
        obj = json.loads(line)
        pairs.append({"prompt": obj["prompt"], "response": obj["response"]})
    return pairs

def main(date="2026-04-29", slug="batch"):
    if not FILE_LIST.exists():
        raise FileNotFoundError(f"{FILE_LIST} not found — run list-hf-folder first")
    file_list = json.loads(FILE_LIST.read_text())
    rows = load_cdn_files(file_list, HF_REPO)
    ds = Dataset.from_pylist(rows)
    # strict projection
    cols = [c for c in ds.column_names if c in {"prompt", "response"}]
    ds = ds.select_columns(cols)
    out_dir = Path(f"enriched/batches/mirror-merged/{date}")
    out_dir.mkdir(parents=True, exist_ok=True)
    ds.to_parquet(str(out_dir / f"{slug}.parquet"))

if __name__ == "__main__":
    main()
```
