# workio / ops

## 1. Diagnosis

- **No idempotency key on LINE webhook payloads** → retries/redeliveries create duplicate punch records for the same `(employee_id, punch_type, date)`.
- **Non-atomic read-then-insert** in punch creation allows race conditions between concurrent or redelivered webhooks.
- **Missing unique constraint at storage layer** for active punches; DB allows duplicates instead of rejecting them.
- **No deduplication window or traceability** (request ID / LINE delivery ID) to correlate retries to original punch.
- **No defensive handling for out-of-order or late retries** (e.g., clock-in after clock-out retry) — can corrupt state.

## 2. Proposed change

File: `/opt/axentx/workio/server/src/services/punchService.ts` (or equivalent service file that handles `createPunch`).  
Scope: add idempotency using LINE `deliveryId` (or generated idempotency key), make punch creation atomic via `INSERT ... ON CONFLICT DO NOTHING`, and add a unique constraint/index on `(employee_id, punch_type, date)` for active punches.

## 3. Implementation

```sql
-- server/src/db/schema.sql
-- Add unique constraint for active punches to prevent duplicates at DB level
-- Assumes table name: punches and has columns: employee_id, punch_type, date, created_at, delivery_id
CREATE UNIQUE INDEX IF NOT EXISTS idx_unique_active_punch
ON punches (employee_id, punch_type, date)
WHERE punch_type IN ('clock_in', 'clock_out');

-- Optional: store delivery_id for idempotency traceability
ALTER TABLE punches
ADD COLUMN IF NOT EXISTS delivery_id TEXT;
CREATE INDEX IF NOT EXISTS idx_punches_delivery_id ON punches (delivery_id) WHERE delivery_id IS NOT NULL;
```

```typescript
// server/src/services/punchService.ts
import { Pool } from 'pg';

const pool = new Pool({ connectionString: process.env.DATABASE_URL });

export interface CreatePunchInput {
  employeeId: string;
  punchType: 'clock_in' | 'clock_out';
  date: string; // YYYY-MM-DD
  timestamp: Date;
  deliveryId?: string; // LINE webhook deliveryId for idempotency
}

/**
 * Create punch idempotently.
 * Uses deliveryId when present to deduplicate; falls back to unique constraint.
 */
export async function createPunch(input: CreatePunchInput) {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // If deliveryId provided, attempt insert with it; ON CONFLICT on delivery_id or unique punch do nothing
    if (input.deliveryId) {
      const insertWithDelivery = `
        INSERT INTO punches (employee_id, punch_type, date, timestamp, delivery_id)
        VALUES ($1, $2, $3, $4, $5)
        ON CONFLICT (delivery_id) DO NOTHING
        RETURNING *;
      `;
      const res = await client.query(insertWithDelivery, [
        input.employeeId,
        input.punchType,
        input.date,
        input.timestamp,
        input.deliveryId,
      ]);

      if (res.rowCount && res.rowCount > 0) {
        await client.query('COMMIT');
        return res.rows[0];
      }

      // If delivery_id duplicate skipped, check whether a punch for same (employee, type, date) already exists
      const existing = await client.query(
        `SELECT * FROM punches WHERE employee_id = $1 AND punch_type = $2 AND date = $3`,
        [input.employeeId, input.punchType, input.date]
      );
      if (existing.rows.length > 0) {
        await client.query('COMMIT');
        return existing.rows[0]; // idempotent return
      }

      // Otherwise attempt to insert punch (unique constraint will protect)
      const insertPunch = `
        INSERT INTO punches (employee_id, punch_type, date, timestamp, delivery_id)
        VALUES ($1, $2, $3, $4, $5)
        ON CONFLICT (employee_id, punch_type, date) DO NOTHING
        RETURNING *;
      `;
      const res2 = await client.query(insertPunch, [
        input.employeeId,
        input.punchType,
        input.date,
        input.timestamp,
        input.deliveryId,
      ]);
      await client.query('COMMIT');
      return res2.rows[0] || null;
    }

    // No deliveryId: rely on unique constraint
    const insert = `
      INSERT INTO punches (employee_id, punch_type, date, timestamp)
      VALUES ($1, $2, $3, $4)
      ON CONFLICT (employee_id, punch_type, date) DO NOTHING
      RETURNING *;
    `;
    const res = await client.query(insert, [
      input.employeeId,
      input.punchType,
      input.date,
      input.timestamp,
    ]);
    await client.query('COMMIT');
    return res.rows[0] || null;
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  } finally {
    client.release();
  }
}
```

```typescript
// server/src/routes/lineWebhook.ts  (or wherever webhook is handled)
import express from 'express';
import { createPunch } from '../services/punchService';
import { verifySignature } from '../utils/lineSignature';

const router = express.Router();

router.post('/webhook/line', async (req, res) => {
  const signature = req.headers['x-line-signature'] as string;
  if (!verifySignature(JSON.stringify(req.body), signature)) {
    return res.sendStatus(401);
  }

  const events = req.body.events || [];
  for (const event of events) {
    if (event.type === 'message' && event.message.type === 'text') {
      // Example: parse clock-in/out command or use payload from LINE OA action
      // For production, map user ID to employeeId and determine punchType/date from payload or DB state.
      const employeeId = event.source.userId; // map via your user table
      const punchType = 'clock_in'; // or derive from command/state
      const date = new Date().toISOString().split('T')[0];
      const timestamp = new Date();
      const deliveryId = event.deliveryId || event.message.id; // use LINE deliveryId when available

      try {
        await createPunch({ employeeId, punchType, date, timestamp, deliveryId });
      } catch (err) {
        console.error('Failed to create punch', err);
        // Do not expose internals; LINE will retry on non-2xx
      }
    }
  }

  res.sendStatus(200);
});

export default router;
```
