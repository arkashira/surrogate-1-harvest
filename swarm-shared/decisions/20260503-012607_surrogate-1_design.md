# surrogate-1 / design

## Final, consolidated implementation (single source of truth)

**Core idea (both candidates agree):**  
Replace per-file authenticated HF Hub API calls with **one non-recursive `list_repo_tree` per date folder + deterministic shard routing + CDN-only fetches**. This eliminates 429s and removes auth overhead for reads.

**Chosen details (favoring correctness + actionability):**
- Use a **single pre-step** to list the date folder and produce `filelist.json` (not per-shard) to avoid redundant calls and race conditions.
- Deterministic shard assignment by `hash(path) % TOTAL_SHARDS` (not slug) because filenames are stable and already encode the slug/date.
- Download exclusively via **CDN** (`https://huggingface.co/datasets/.../resolve/main/...`) with no Authorization header.
- Keep per-shard idempotent output: `batches/public-merged/<date>/shard<N>-<HHMMSS>.jsonl`.
- Central md5 dedup store remains source of truth; workers remain stateless.
- Retry/backoff for CDN downloads (network/HTTP 429/5xx) separate from API ratelimit logic.
- GitHub Actions: one matrix job per shard (0–15); a single “list” step runs once and shares `filelist.json` via workspace artifact (fast, reliable).

### bin/list_date_folder.py
```python
#!/usr/bin/env python3
"""
List immediate children of a date folder in axentx/surrogate-1-training-pairs
(non-recursive) and write filelist.json for workers.
"""
import json
import os
import sys
from huggingface_hub import HfApi

REPO = "axentx/surrogate-1-training-pairs"
API = HfApi(token=os.getenv("HF_TOKEN"))

def main(date_folder: str, out_path: str = "filelist.json"):
    entries = API.list_repo_tree(repo_id=REPO, path=date_folder, recursive=False)
    files = [
        {"path": e.path, "sha": getattr(e, "commit_oid", None) or getattr(e, "sha", None), "size": getattr(e, "size", None)}
        for e in entries
        if not e.path.endswith("/")  # files only
    ]
    with open(out_path, "w") as f:
        json.dump(files, f, indent=2)
    print(f"Wrote {len(files)} entries to {out_path}")

if __name__ == "__main__":
    if len(sys.argv) < 2:
        print("Usage: list_date_folder.py <date-folder> [out.json]")
        sys.exit(1)
    main(sys.argv[1], sys.argv[2] if len(sys.argv) > 2 else "filelist.json")
```

### bin/dataset-enrich.sh
```bash
#!/usr/bin/env bash
# Worker: uses pre-listed filelist.json + CDN fetches.
set -euo pipefail

REPO="axentx/surrogate-1-training-pairs"
DATE_FOLDER="${1:-$(date +%Y-%m-%d)}"
SHARD_ID="${SHARD_ID:-0}"
TOTAL_SHARDS="${TOTAL_SHARDS:-16}"
OUT_DIR="output"
mkdir -p "$OUT_DIR"

if [[ ! -f "filelist.json" ]]; then
  echo "ERROR: filelist.json not found. Run list_date_folder.py $DATE_FOLDER first." >&2
  exit 1
fi

# Deterministic shard assignment by path hash
python3 - <<PY
import json, hashlib, sys
shard = int(sys.argv[1])
total = int(sys.argv[2])
with open("filelist.json") as f:
    files = json.load(f)
assigned = []
for f in files:
    h = int(hashlib.sha256(f["path"].encode()).hexdigest(), 16)
    if h % total == shard:
        assigned.append(f["path"])
with open("assigned.json", "w") as f:
    json.dump(assigned, f)
PY "$SHARD_ID" "$TOTAL_SHARDS"

# Stream assigned files via CDN, project to {prompt,response}, upload
python3 - <<PY
import json, urllib.request, os, io, sys, time, hashlib
import pyarrow.parquet as pq
import pyarrow as pa

CDN_BASE = "https://huggingface.co/datasets/$REPO/resolve/main"
with open("assigned.json") as f:
    paths = json.load(f)

ts = time.strftime("%Y%m%d-%H%M%S", time.gmtime())
out_path = os.path.join("$OUT_DIR", f"shard$SHARD_ID-$ts.jsonl")

def retry_fetch(url, max_tries=5):
    for attempt in range(1, max_tries + 1):
        try:
            req = urllib.request.Request(url, headers={"User-Agent": "surrogate-1-runner"})
            with urllib.request.urlopen(req, timeout=30) as resp:
                return resp.read()
        except urllib.error.HTTPError as e:
            if e.code == 429 or e.code >= 500:
                wait = 30 * (2 ** (attempt - 1))
                print(f"Retryable error {e.code} for {url}, wait {wait}s (attempt {attempt})", file=sys.stderr)
                time.sleep(wait)
                continue
            raise
        except Exception:
            if attempt == max_tries:
                raise
            time.sleep(5 * attempt)
    raise RuntimeError("Unreachable")

def fetch_and_project(path):
    url = f"{CDN_BASE}/{path}"
    data = retry_fetch(url)
    rows = []
    # Try parquet
    try:
        table = pq.read_table(io.BytesIO(data))
        df = table.to_pandas()
    except Exception:
        # Try newline-delimited JSON
        try:
            df = pa.json.read_json(io.BytesIO(b"\n".join(data.splitlines()))).to_pandas()
        except Exception:
            print(f"Cannot parse {path}", file=sys.stderr)
            return rows
    for _, r in df.iterrows():
        rows.append({"prompt": str(r.get("prompt", "")), "response": str(r.get("response", ""))})
    return rows

with open(out_path, "w") as out_f:
    for p in paths:
        for row in fetch_and_project(p):
            out_f.write(json.dumps(row) + "\n")
print(f"Wrote {out_path}")
PY

# Commit and push shard output (single commit per shard per run)
OUTPUT_FILE="$OUT_DIR/shard$SHARD_ID-$(date -u +%Y%m%d-%H%M%S).jsonl"
if [[ -f "$OUTPUT_FILE" ]]; then
  git config user.name "surrogate-1-runner"
  git config user.email "runner@axentx.dev"
  git add "$OUTPUT_FILE"
  git commit -m "shard $SHARD_ID $DATE_FOLDER $(date -u +%Y-%m-%dT%H:%M:%SZ)" || true
  git push origin main
fi
```
