```python
# costinel/services/cost_anomaly_service.py
from datetime import datetime, timezone
from typing import Any, Dict, Optional
from pydantic import BaseModel

from costinel.libs.knowledge_rag import get_top_hub_for_date  # existing helper


class TopHubSignal(BaseModel):
    hub: str
    score: float
    insight: str
    context: Dict[str, Any]
    for_date: str  # ISO date
    signal_type: str = "cost_anomaly_top_hub"
    sense_only: bool = True


def get_top_hub_signal(for_date: Optional[str] = None) -> TopHubSignal:
    """
    Sense + Signal: return top hub insight as cost-anomaly signal.
    No Execute.
    """
    if for_date is None:
        for_date = datetime.now(timezone.utc).strftime("%Y-%m-%d")

    # existing knowledge-rag call (returns hub, score, insight, related_docs)
    result = get_top_hub_for_date(for_date=for_date)

    return TopHubSignal(
        hub=result.get("hub", "unknown"),
        score=float(result.get("score", 0.0)),
        insight=result.get("insight", ""),
        context={
            "related_docs": result.get("related_docs", []),
            "graph_degree": result.get("degree", 0),
            "tags": result.get("tags", []),
        },
        for_date=for_date,
    )
```

### 2) API route: `costinel/api/v1/endpoints/cost_anomaly.py`
```python
# costinel/api/v1/endpoints/cost_anomaly.py
from fastapi import APIRouter, Query, HTTPException
from typing import Optional

from costinel.services.cost_anomaly_service import get_top_hub_signal

router = APIRouter(prefix="/cost-anomaly", tags=["cost-anomaly"])


@router.get("/signal/top-hub", response_model=dict)
def top_hub_signal(
    for_date: Optional[str] = Query(
        None,
        regex=r"^\d{4}-\d{2}-\d{2}$",
        description="Date (YYYY-MM-DD). Defaults to today UTC.",
    )
):
    """
    Sense + Signal: return top hub insight as cost-anomaly signal.
    No Execute.
    """
    try:
        signal = get_top_hub_signal(for_date=for_date)
        return signal.dict()
    except Exception as exc:
        # Fail gracefully — this is a signal endpoint, not critical path
        raise HTTPException(status_code=500, detail=f"Unable to produce signal: {exc}") from exc
```
