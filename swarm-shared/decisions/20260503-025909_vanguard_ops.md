# vanguard / ops

## Final synthesized solution

**Core principle:** Build once, verify integrity, inject into training, and reuse compute.  
All runtime HF API list calls are eliminated; training uses only CDN fetches with content-hashed integrity checks.

---

### 1. Deterministic CDN-first manifest (single source of truth)

Create `/opt/axentx/vanguard/ops/cdn_asset_manifest.py`:

```python
#!/usr/bin/env python3
"""
Generate and validate a CDN-first asset manifest for a HF dataset repo folder.

Usage:
  python cdn_asset_manifest.py \
    --repo datasets/axentx/surrogate-1 \
    --folder 2026-05-03 \
    --out manifest.json \
    [--validate] [--validate-root /data/axentx/cache]

Output manifest entry:
{
  "repo_id": "...",
  "date_folder": "...",
  "cdn_prefix": "...",
  "files": [
    {"slug": "a/1.npy", "cdn_url": "...", "sha256": "...", "size": 12345}
  ]
}
"""

from __future__ import annotations

import argparse
import hashlib
import json
import os
import sys
from dataclasses import asdict
from pathlib import Path
from typing import List

import requests
from hugginggingface import HfApi  # type: ignore

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def list_folder_files(repo: str, folder: str) -> List[str]:
    """Single API call to list files in one folder (non-recursive)."""
    api = HfApi()
    nodes = api.list_repo_tree(repo=repo, path=folder, recursive=False)
    paths: List[str] = []
    for n in nodes:
        # support both object and dict returns
        if hasattr(n, "type"):
            if n.type == "file" and hasattr(n, "rfilename"):
                paths.append(n.rfilename)
        elif isinstance(n, dict):
            if n.get("type") == "file" and "path" in n:
                paths.append(n["path"])
    return sorted(set(paths))

def sha256_file(path: Path, block: int = 65536) -> str:
    h = hashlib.sha256()
    with path.open("rb") as f:
        for chunk in iter(lambda: f.read(block), b""):
            h.update(chunk)
    return h.hexdigest()

def build_manifest(repo: str, folder: str) -> dict:
    files = list_folder_files(repo, folder)
    items = []
    for p in files:
        url = CDN_TEMPLATE.format(repo=repo, path=p)
        items.append(
            {
                "slug": p,
                "cdn_url": url,
                "sha256": None,
                "size": None,
            }
        )
    return {
        "repo_id": repo,
        "date_folder": folder,
        "cdn_prefix": CDN_TEMPLATE.format(repo=repo, path=folder),
        "files": items,
    }

def validate_local(manifest_path: Path, cache_root: Path) -> bool:
    with manifest_path.open() as f:
        manifest = json.load(f)

    ok = True
    cache_root = cache_root.expanduser().resolve()
    for fentry in manifest["files"]:
        slug = fentry["slug"]
        local_path = (cache_root / slug).resolve()
        if not local_path.is_file():
            print(f"Missing: {local_path}", file=sys.stderr)
            ok = False
            continue
        expected = fentry.get("sha256")
        if expected:
            actual = sha256_file(local_path)
            if actual != expected:
                print(f"Corrupt: {local_path} (sha256 mismatch)", file=sys.stderr)
                ok = False
    return ok

def compute_hashes(manifest: dict, cache_root: Path) -> dict:
    cache_root = cache_root.expanduser().resolve()
    for fentry in manifest["files"]:
        slug = fentry["slug"]
        local_path = (cache_root / slug).resolve()
        if local_path.is_file():
            fentry["sha256"] = sha256_file(local_path)
            fentry["size"] = local_path.stat().st_size
    return manifest

def main() -> None:
    parser = argparse.ArgumentParser(description="CDN-first asset manifest tool")
    parser.add_argument("--repo", required=True, help="HF repo id (e.g. datasets/axentx/surrogate-1)")
    parser.add_argument("--folder", required=True, help="Date folder inside repo")
    parser.add_argument("--out", required=True, help="Output manifest JSON path")
    parser.add_argument("--validate", action="store_true", help="Validate existing files")
    parser.add_argument("--validate-root", default="/data/axentx/cache", help="Local cache root for validation")
    parser.add_argument("--compute-hashes", action="store_true", help="Compute sha256/size for local files")
    args = parser.parse_args()

    out_path = Path(args.out)
    out_path.parent.mkdir(parents=True, exist_ok=True)

    if args.validate:
        if not out_path.is_file():
            print(f"Manifest not found: {out_path}", file=sys.stderr)
            sys.exit(1)
        ok = validate_local(out_path, Path(args.validate_root))
        sys.exit(0 if ok else 1)

    manifest = build_manifest(args.repo, args.folder)

    if args.compute_hashes:
        manifest = compute_hashes(manifest, Path(args.validate_root))

    with out_path.open("w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Wrote {len(manifest['files'])} files to {out_path}")

if __name__ == "__main__":
    main()
```

Make executable:
```bash
chmod +x /opt/axentx/vanguard/ops/cdn_asset_manifest.py
```

### 2. Idempotent launcher with studio reuse + idle-stop guard

Create `/opt/axentx/vanguard/ops/launch_studio.sh`:

```bash
#!/usr/bin/env bash
# vanguard ops launcher — Mac orchestration host only
# Usage: SHELL=/bin/bash bash /opt/axentx/vanguard/ops/launch_studio.sh <date_folder>
# Crontab: SHELL=/bin/bash
#   0 2 * * * bash /opt/axentx/vanguard/ops/launch_studio.sh 2026-05-03
set -euo pipefail

HF_REPO="${HF_REPO:-datasets/axentx/surrogate-1}"
DATE_FOLDER="${1:-$(date +%Y-%m-%d)}"
MANIFEST_PATH="/opt/axentx/vanguard/ops/file_manifest.json"
STUDIO_NAME="vanguard-l40s-${DATE_FOLDER}"
MACHINE="L40S"

log() { echo "[$(date -u +%Y-%m-%dT%H:%M:%SZ)] $*"; }

# 1) Generate file manifest (single API call) — CDN-only after this
generate_manifest() {
  log "Generating manifest for ${HF_REPO}/${DATE_FOLDER}"
  python3 /opt/axentx/vanguard/ops/cdn_asset_manifest.py \
    --repo "$HF_REPO" \
    --folder "$DATE_FOLDER" \
    --out "$MANIFEST_PATH"
  log "Manifest written to ${MANIFEST_PATH}"
}

# 2) Reuse running studio or start new one (Lightning SDK)
launch_training() {
  log "Checking Lightning studios..."
  python3 - "$STUDIO_NAME" "$MACHINE" "$MANIFEST_PATH" <<'PY'
import os
import sys
import time
import json
from pathlib import Path

try:
    from lightning_sdk import Studio, Machine
except ImportError as e:
    print("ERROR: lightning-sdk not available", file=sys.stderr)
    sys.exit(1)

STUDIO_NAME = sys.argv[1]
MACHINE = sys.argv[2]
MANIFEST_PATH = sys.argv[3]

if not Path(MANIFEST_PATH).is_file():
    print(f"ERROR: manifest not found: {MANIFEST_PATH}", file=sys.stderr)
    sys.exit(1)
