# workio / design

## Final Synthesized Solution

### 1. Diagnosis (Consolidated)
- **No tenant cache** — every LINE webhook performs a PostgreSQL `channelId → tenant` lookup, amplifying DB load and latency under burst traffic.
- **Missing health/readiness probes** — no `/health` (liveness) or `/ready` (readiness); operators cannot detect degraded vs dead states or drain safely.
- **No graceful shutdown/startup sequencing** — server accepts requests before DB is ready and can kill in-flight webhooks during deploy, increasing 5xx and partial writes.
- **LINE signature verification order is suboptimal** — expensive crypto runs before fast-fail checks; should reject invalid requests before cache/DB work.
- **Missing observability** — no request-id/trace correlation and minimal structured logging; debugging bursts and replaying failures is slow.
- **No retry/backoff on outbound LINE API calls** — transient network/429 errors bubble up as 5xx to LINE, risking delivery guarantees.

### 2. Proposed Change (Single Scope)
**File**: `workio/server/src/app.ts` (or main server entry)  
- Add in-memory LRU cache for `channelId → tenant` (5-minute TTL, 500-entry cap).  
- Add `/health` and `/ready` endpoints.  
- Add graceful shutdown (drain + DB close) and startup readiness guard.  
- Move LINE signature verification earlier; fast-fail before cache/DB.  
- Add structured logging with request-id and minimal tracing for webhook path.  
- Add retry with exponential backoff and jitter for outbound LINE API calls.

// workio/server/src/app.ts
import express, { Request, Response, NextFunction } from 'express';
import { LRUCache } from 'lru-cache';
import { v4 as uuidv4 } from 'uuid';
import { retry } from 'exponential-backoff';
import line from './line'; // existing LINE client/module
import db from './db';     // existing db client

const app = express();
app.use(express.json());

// Structured logger
function log(level: string, msg: string, ctx: Record<string, any> = {}) {
  console.log(JSON.stringify({ ts: new Date().toISOString(), level, msg, ...ctx }));
}

// Request ID middleware
app.use((req, res, next) => {
  const requestId = (req.headers['x-request-id'] as string) || uuidv4();
  (req as any).requestId = requestId;
  res.setHeader('X-Request-Id', requestId);
  next();
});

// Capture raw body for LINE signature verification
app.use((req, res, next) => {
  const chunks: Buffer[] = [];
  req.on('data', (chunk) => chunks.push(chunk));
  req.on('end', () => {
    (req as any).rawBody = Buffer.concat(chunks).toString('utf8');
    next();
  });
});

// ----- Tenant cache (in-memory LRU) -----
const tenantCache = new LRUCache<string, { tenantId: string; name: string; updatedAt: number }>({
  max: 500,
  ttl: 1000 * 60 * 5, // 5 minutes
});

async function resolveTenantByChannelId(channelId: string) {
  const cached = tenantCache.get(channelId);
  if (cached) return cached;

  const row = await db.oneOrNone(
    'SELECT id AS "tenantId", name FROM tenants WHERE line_channel_id = $1 LIMIT 1',
    [channelId]
  );
  if (!row) return null;

  const item = { tenantId: row.tenantId, name: row.name, updatedAt: Date.now() };
  tenantCache.set(channelId, item);
  return item;
}

// ----- LINE signature verification (early fast-fail) -----
function verifyLineSignature(req: Request, res: Response, next: NextFunction) {
  const signature = req.headers['x-line-signature'] as string;
  const rawBody = (req as any).rawBody;
  if (!rawBody || typeof rawBody !== 'string') {
    return res.status(400).json({ error: 'Invalid request body' });
  }

  const ok = line.verifySignature(rawBody, signature);
  if (!ok) {
    log('warn', 'Invalid LINE signature', { requestId: (req as any).requestId });
    return res.status(401).json({ error: 'Invalid signature' });
  }
  next();
}

// ----- LINE webhook route -----
app.post('/webhook/line', verifyLineSignature, async (req, res) => {
  const requestId = (req as any).requestId;
  try {
    const channelId = req.body?.destination; // LINE bot ID
    if (!channelId) {
      log('warn', 'Missing channelId in LINE webhook', { requestId, body: req.body });
      return res.status(400).json({ error: 'Missing channelId' });
    }

    const tenant = await resolveTenantByChannelId(channelId);
    if (!tenant) {
      log('warn', 'Unknown LINE channel', { requestId, channelId });
      return res.status(404).json({ error: 'Tenant not found' });
    }

    (req as any).tenant = tenant;

    // Delegate to existing LINE handler with retry for outbound LINE API calls
    // Example adapter:
    // await retry(() => line.handleWebhook(req, res), { delay: 200, maxDelay: 2000, numOfAttempts: 3 });
    // For now, respond OK:
    res.json({ ok: true, tenantId: tenant.tenantId });
    log('info', 'LINE webhook processed', { requestId, channelId, tenantId: tenant.tenantId });
  } catch (err) {
    log('error', 'LINE webhook failed', { requestId, error: String(err) });
    res.status(500).json({ error: 'Internal server error' });
  }
});

// ----- Health / Readiness -----
app.get('/health', (req, res) => {
  // Liveness: process is alive
  res.json({ status: 'ok', uptime: process.uptime() });
});

let isReady = false;
app.get('/ready', async (req, res) => {
  // Readiness: db reachable
  if (!isReady) {
    try {
      await db.query('SELECT 1');
      isReady = true;
    } catch (err) {
      log('error', 'Readiness check failed', { error: String(err) });
      return res.status(503).json({ status: 'not ready', error: String(err) });
    }
  }
  res.json({ status: 'ready' });
});

// ----- Graceful shutdown + startup guard -----
function startServer(port: number) {
  const server = app.listen(port, () => {
    log('info', 'Server started', { port });
  });

  async function shutdown(signal: string) {
    log('info', 'Shutting down', { signal });
    isReady = false;

    // Stop accepting new connections
    server.close(async (err) => {
      if (err) {
        log('error', 'Error during server close', { error: String(err) });
        process.exit(1);
      }
      try {
        await db.end?.(); // or db.destroy() depending on client
        log('info', 'DB connections closed');
        process.exit(0);
      } catch (e) {
        log('error', 'Error closing DB', { error: String(e) });
        process.exit(1);
      }
    });

    // Force exit after timeout
    setTimeout(() => {
      log('warn', 'Forced shutdown after timeout');
      process.exit(1);
    }, 10000).unref();
  }

  process.on('SIGTERM', () => shutdown('SIGTERM'));
  process.on('SIGINT', () => shutdown('SIGINT'));

  return server;
}
