# Costinel / design

## Costinel — Design Increment: Discovery Module (≤2h)

**Goal**  
Ship a deterministic, audit-ready discovery CLI (`discovery run`) that produces machine-readable manifests and a top-hub insight snapshot — enabling **visibility-first onboarding (Sense + Signal — ไม่ Execute)** with zero execution risk.

**Exit criteria**  
- `make discovery-run` produces `manifests/sources.json`, `manifests/discovery.json`, `audit/discovery.log`, `insights/top-hub.json`
- All outputs are deterministic (same inputs → same digests)
- Zero cloud mutations (read-only)

### 2.1 `discovery/__main__.py` (CLI entrypoint)

```python
#!/usr/bin/env python3
"""
Costinel Discovery Module
Sense + Signal — ไม่ Execute
"""
import argparse
import json
import hashlib
import datetime as dt
import logging
from pathlib import Path
from typing import Dict, Any

from discovery.sources import build_sources_manifest
from discovery.insights import top_hub_snapshot

LOG_DIR = Path("audit")
MANIFEST_DIR = Path("manifests")
INSIGHTS_DIR = Path("insights")

def _checksum(data: bytes) -> str:
    return hashlib.sha256(data).hexdigest()

def _write_atomic(path: Path, content: bytes) -> None:
    path.parent.mkdir(parents=True, exist_ok=True)
    tmp = path.with_suffix(".tmp")
    tmp.write_bytes(content)
    tmp.replace(path)

def _audit(event: str, payload: Dict[str, Any], audit_log: Path) -> None:
    ts = dt.datetime.utcnow().isoformat() + "Z"
    record = {
        "ts": ts,
        "event": event,
        "payload": payload,
    }
    line = json.dumps(record, separators=(",", ":"), sort_keys=True)
    audit_log.parent.mkdir(parents=True, exist_ok=True)
    with audit_log.open("a") as f:
        f.write(line + "\n")

def run_discovery(tag: str, dry_run: bool = True) -> int:
    LOG_DIR.mkdir(parents=True, exist_ok=True)
    audit_log = LOG_DIR / "discovery.log"

    # 1) Sources manifest
    sources = build_sources_manifest()
    sources_bytes = json.dumps(sources, separators=(",", ":"), sort_keys=True).encode()
    sources_path = MANIFEST_DIR / "sources.json"
    _write_atomic(sources_path, sources_bytes)
    _audit("sources_manifest", {"tag": tag, "sha256": _checksum(sources_bytes)}, audit_log)

    # 2) Discovery manifest
    discovery = {
        "tag": tag,
        "generated_at_utc": dt.datetime.utcnow().isoformat() + "Z",
        "sources_sha256": _checksum(sources_bytes),
        "sources_count": len(sources.get("sources", [])),
        "mode": "dry_run" if dry_run else "live",
        "policy": "sense_and_signal",
        "execute": False,
    }
    disc_bytes = json.dumps(discovery, separators=(",", ":"), sort_keys=True).encode()
    disc_path = MANIFEST_DIR / "discovery.json"
    _write_atomic(disc_path, disc_bytes)
    _audit("discovery_manifest", {"tag": tag, "sha256": _checksum(disc_bytes)}, audit_log)

    # 3) Top-hub insight snapshot (knowledge-rag pattern)
    try:
        hub = top_hub_snapshot()
        hub_bytes = json.dumps(hub, separators=(",", ":"), sort_keys=True).encode()
        hub_path = INSIGHTS_DIR / "top-hub.json"
        _write_atomic(hub_path, hub_bytes)
        _audit("top_hub_snapshot", {"tag": tag, "hub": hub.get("hub")}, audit_log)
    except Exception as exc:
        logging.warning("Top-hub snapshot skipped: %s", exc)

    print("OK")
    print(f"  manifests/sources.json  -> {sources_path}")
    print(f"  manifests/discovery.json -> {disc_path}")
    print(f"  audit/discovery.log      -> {audit_log}")
    return 0

def main() -> None:
    parser = argparse.ArgumentParser(description="Costinel Discovery (Sense + Signal)")
    parser.add_argument("--tag", default="default", help="Discovery tag")
    parser.add_argument("--execute", action="store_true", help="Allow execution (disabled by default)")
    args = parser.parse_args()

    if args.execute:
        print("ERROR: Execution is disabled by design (Sense + Signal — ไม่ Execute)")
        raise SystemExit(1)

    raise SystemExit(run_discovery(tag=args.tag, dry_run=True))

if __name__ == "__main__":
    main()
```
