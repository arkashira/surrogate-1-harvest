## Highest-value incremental improvement (≤2h)

**Problem**: Surrogate training jobs on Lightning are likely to:
1. Hit HF API 429 during dataset enumeration (big repos, recursive listing)
2. Die when Lightning Studio stops (idle timeout) and lose progress
3. Re-create studios instead of reusing running ones (wastes 80h/mo quota)

**Solution**: Add a **robust Lightning training launcher** that:
- Pre-lists files once (non-recursive, per folder) → JSON cached locally
- Uses HF CDN direct downloads (no auth, no API rate limit) during training
- Reuses a running Studio if present; otherwise creates one
- Checks studio status before each run and restarts automatically if stopped
- Emits training artifacts to deterministic `batches/mirror-merged/{date}/{slug}.parquet` (no extra metadata columns)

**Why this now**: It directly unblocks fast, repeatable training iterations (<2 min) and prevents quota/time waste. It applies the exact patterns from the knowledge base (HF CDN bypass, studio reuse, idle-stop resilience, schema projection).

# scripts/build_file_index.py
import os
import json
import argparse
from huggingface_hub import HfApi

def build_index(repo_id: str, folders: list[str], out_path: str):
    api = HfApi()
    index = []

    for folder in folders:
        tree = api.list_repo_tree(repo_id=repo_id, path=folder, recursive=False)
        for item in tree:
            if item.type == "file":
                index.append({
                    "repo_id": repo_id,
                    "path": item.path,
                    "url": f"https://huggingface.co/datasets/{repo_id}/resolve/main/{item.path}"
                })

    os.makedirs(os.path.dirname(os.path.abspath(out_path)), exist_ok=True)
    with open(out_path, "w") as f:
        json.dump(index, f, indent=2)
    print(f"Indexed {len(index)} files -> {out_path}")

if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", default="axentx/surrogate-data")
    parser.add_argument("--folders", nargs="+", default=["batches/mirror-merged/2026-05-03"])
    parser.add_argument("--out", default="data/file_index.json")
    args = parser.parse_args()
    build_index(args.repo, args.folders, args.out)

### 2) CDN dataset reader (no HF API during training)

```python
# surrogate/data/hf_cdn_dataset.py
import json
import pyarrow as pa
import pyarrow.parquet as pq
import requests
from torch.utils.data import Dataset
from typing import List, Dict

class HFCDNDataset(Dataset):
    def __init__(self, file_index_path: str):
        with open(file_index_path) as f:
            self.items = json.load(f)

    def __len__(self):
        return len(self.items)

    def __getitem__(self, idx: int) -> Dict[str, str]:
        item = self.items[idx]
        url = item["url"]
        # Stream parquet from CDN (no auth, no API)
        resp = requests.get(url, timeout=30)
        resp.raise_for_status()
        table = pq.read_table(pa.BufferReader(resp.content))
        # Project to {prompt, response} only
        row = table.slice(0, 1).to_pydict()
        return {
            "prompt": row.get("prompt", [""])[0],
            "response": row.get("response", [""])[0]
        }
```

### 3) Lightning launcher with reuse + idle-stop resilience

```python
# scripts/launch_surrogate_training.py
#!/usr/bin/env python3
import os
import time
from pathlib import Path
from lightning import LightningWork, LightningFlow, LightningApp, Machine
from lightning.app.storage import Drive
from surrogate.train.train import train_fn  # your existing train loop

class SurrogateTrainer(LightningWork):
    def __init__(self, file_index_path: str, **kwargs):
        super().__init__(**kwargs)
        self.file_index_path = file_index_path
        self.machine = Machine.L40S  # or fallback handled by sweep logic
        self._studio = None

    def run(self):
        # Ensure index is available inside work
        local_index = Path(self.file_index_path)
        if not local_index.exists():
            raise FileNotFoundError(f"File index missing: {local_index}")

        # Train with CDN-only dataset
        train_fn(file_index_path=str(local_index))

class SurrogateFlow(LightningFlow):
    def __init__(self, file_index_path: str):
        super().__init__()
        self.file_index_path = file_index_path
        self.trainer = SurrogateTrainer(
            file_index_path=file_index_path,
            cloud_compute=Machine.L40S,
            cloud_build_commands=[
                "pip install -r requirements.txt",
            ],
        )

    def configure_layout(self):
        return []

    def run_once(self):
        # Reuse if already running
        if self.trainer.status == "running":
            print("Studio running, skipping recreate.")
            return

        # If stopped/idle, restart
        if self.trainer.status in ["stopped", "failed"]:
            print(f"Studio stopped ({self.trainer.status}), restarting...")
            self.trainer.start(machine=Machine.L40S)
            return

        # Not started yet
        if self.trainer.status == "not_started":
            self.trainer.start(machine=Machine.L40S)

if __name__ == "__main__":
    import argparse
    parser = argparse.ArgumentParser()
    parser.add_argument("--file-index", default="data/file_index.json")
    args = parser.parse_args()

    # Ensure index exists before launch
    if not Path(args.file_index).exists():
        raise SystemExit(f"Build file index first: {args.file_index}")

    app = LightningApp(SurrogateFlow(file_index_path=args.file_index))
```

### 4) Minimal train entrypoint (uses CDN dataset)

```python
# surrogate/train/train.py
import os
import time
from pathlib import Path
from torch.utils.data import DataLoader
from surrogate.data.hf_cdn_dataset import HFCDNDataset

def train_fn(file_index_path: str, max_steps: int = 10):
    dataset = HFCDNDataset(file_index_path)
    loader = DataLoader(dataset, batch_size=1, shuffle=False)

    out_dir = Path("batches/mirror-merged") / time.strftime("%Y-%m-%d")
    out_dir.mkdir(parents=True, exist_ok=True)

    rows = []
    for step, batch in enumerate(loader):
        if step >= max_steps:
            break
        rows.append({
            "prompt": batch["prompt"][0],
            "response": batch["response"][0],
        })

    if rows:
        import pandas as pd
        slug = f"train-{int(time.time())}"
        df = pd.DataFrame(rows)
        out_path = out_dir / f"{slug}.parquet"
        df.to_parquet(out_path, index=False
