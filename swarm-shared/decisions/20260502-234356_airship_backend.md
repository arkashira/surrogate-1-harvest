# airship / backend

## Implementation Plan — `airship discover` (CDN-only manifest)

**Scope (≤2h)**  
Add a deterministic CLI command `airship discover` that produces a CDN-only file manifest for a given `repo_id` and `date_folder`.  
- Uses **one** `list_repo_tree(path, recursive=False)` call from Mac (or any orchestrator) after rate-limit window clears.  
- Saves list to JSON.  
- Embeds file list in training script so Lightning Studio does **zero HF API calls** during data load (CDN-only fetches via `https://huggingface.co/datasets/{repo}/resolve/main/{path}`).  
- Avoids `load_dataset(streaming=True)` on heterogeneous repos; projects to `{prompt,response}` at parse time.  
- Stores enriched parquet into `batches/mirror-merged/{date}/{slug}.parquet` (no extra metadata columns).  
- Reuses running Lightning Studio when possible; restarts if idle-stopped.

### Step 1 — Add `surrogate/discover.py` (40–50m)

- CLI: `airship discover --repo <repo_id> --date <YYYY-MM-DD> --out <path>`
- Single `list_repo_tree(path=f"{date_folder}", recursive=False)` call.
- Flatten to relative paths; filter by extension (`.jsonl`, `.json`, `.parquet`, `.csv`).
- Save manifest JSON: `{repo_id, date_folder, generated_at, files: [{rel_path, size, etag?}]}`.
- Deterministic sort for reproducibility.

```python
# surrogate/discover.py
from __future__ import annotations
import argparse
import json
import os
import sys
from datetime import datetime
from pathlib import Path
from typing import Any, Dict, List

try:
    from huggingface_hub import HfApi
except ImportError:
    print("Install huggingface_hub to use discover", file=sys.stderr)
    sys.exit(1)

HF_API = HfApi()

def list_date_folder(repo_id: str, date_folder: str) -> List[Dict[str, Any]]:
    """
    Single list_repo_tree call (non-recursive) for CDN manifest.
    Returns sorted list of file dicts with rel_path and size.
    """
    entries = HF_API.list_repo_tree(repo_id=repo_id, path=date_folder, recursive=False)
    files = []
    for e in entries:
        if e.type == "file":
            # path is like "2026-04-29/file1.jsonl"
            files.append({"rel_path": e.path, "size": e.size or 0})
    files.sort(key=lambda x: x["rel_path"])
    return files

def build_manifest(repo_id: str, date_folder: str) -> Dict[str, Any]:
    manifest = {
        "repo_id": repo_id,
        "date_folder": date_folder.rstrip("/"),
        "generated_at": datetime.utcnow().isoformat() + "Z",
        "files": list_date_folder(repo_id, date_folder),
    }
    return manifest

def main() -> None:
    parser = argparse.ArgumentParser(description="Generate CDN-only file manifest for Surrogate-1 training.")
    parser.add_argument("--repo", required=True, help="HF dataset repo id (e.g., org/dataset)")
    parser.add_argument("--date", required=True, help="Date folder (e.g., 2026-04-29)")
    parser.add_argument("--out", default="manifest.json", help="Output JSON path")
    args = parser.parse_args()

    try:
        manifest = build_manifest(args.repo, args.date)
    except Exception as exc:
        print(f"Failed to list repo tree: {exc}", file=sys.stderr)
        sys.exit(1)

    out_path = Path(args.out)
    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"Manifest written to {out_path} ({len(manifest['files'])} files)")

if __name__ == "__main__":
    main()
```

### Step 2 — Add `surrogate/cdn_loader.py` (30–40m)

- CDN-only fetches via `https://huggingface.co/datasets/{repo_id}/resolve/main/{rel_path}` (no auth).
- Retry/backoff on 429 or transient errors.
- Stream to temp file; parse only required `{prompt,response}` fields (projection).
- Support common formats: `.jsonl`, `.json`, `.parquet`, `.csv`.

```python
# surrogate/cdn_loader.py
from __future__ import annotations
import io
import json
import time
from pathlib import Path
from typing import Any, Dict, Iterable, List
import requests
import pyarrow.parquet as pq
import pandas as pd

CDN_ROOT = "https://huggingface.co/datasets"

def cdn_url(repo_id: str, rel_path: str) -> str:
    return f"{CDN_ROOT}/{repo_id}/resolve/main/{rel_path}"

def robust_get(url: str, max_retries: int = 5, backoff: float = 1.0) -> bytes:
    for attempt in range(1, max_retries + 1):
        try:
            resp = requests.get(url, timeout=30)
            if resp.status_code == 429:
                wait = backoff * (2 ** (attempt - 1))
                time.sleep(wait)
                continue
            resp.raise_for_status()
            return resp.content
        except (requests.RequestException, requests.Timeout) as exc:
            if attempt == max_retries:
                raise
            time.sleep(backoff * (2 ** (attempt - 1)))
    raise RuntimeError("Unreachable")

def project_record(record: Dict[str, Any]) -> Dict[str, str]:
    """Keep only prompt/response keys; tolerate schema variants."""
    prompt = record.get("prompt") or record.get("input") or record.get("question") or ""
    response = record.get("response") or record.get("output") or record.get("answer") or ""
    return {"prompt": str(prompt), "response": str(response)}

def load_file(repo_id: str, rel_path: str) -> List[Dict[str, str]]:
    url = cdn_url(repo_id, rel_path)
    data = robust_get(url)
    suffix = Path(rel_path).suffix.lower()

    if suffix == ".parquet":
        table = pq.read_table(io.BytesIO(data))
        df = table.to_pandas()
        records = df.to_dict(orient="records")
    elif suffix == ".jsonl":
        records = [json.loads(ln) for ln in io.BytesIO(data).read().decode("utf-8").splitlines() if ln.strip()]
    elif suffix == ".json":
        raw = json.loads(io.BytesIO(data).read().decode("utf-8"))
        records = raw if isinstance(raw, list) else [raw]
    elif suffix == ".csv":
        df = pd.read_csv(io.BytesIO(data))
        records = df.to_dict(orient="records")
    else:
        raise ValueError(f"Unsupported file type: {suffix}")

    return [project_record(r) for r in records]

def load_manifest(manifest_path: str) -> Dict[str, Any]:
    return json.loads(Path(manifest_path).read_text())

def stream_cdn_dataset(repo_id: str, files: List[Dict[str, Any]]) -> Iterable[Dict[str, str]]:
    for f in files:
        try:
            for rec in load_file(repo_id, f["rel_path"]):
                yield rec
        except Exception as exc:
            # Log and skip bad files to avoid breaking training
            print(f"Skipping {f['rel_path']}: {exc}", file=sys.stderr)
            continue
```
