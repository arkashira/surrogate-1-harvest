Deterministic pre-flight snapshot + CDN-only ingestion + Lightning Studio reuse.

### 1. `bin/list-snapshot.sh`
```bash
#!/usr/bin/env bash
set -euo pipefail

REPO="${HF_REPO:-axentx/surrogate-1-training-pairs}"
DATE="${1:-$(date +%Y-%m-%d)}"
TS="$(date +%Y%m%d-%H%M%S)"
OUT="${2:-snapshot-${DATE}-${TS}.json}"

echo "Listing ${REPO} for ${DATE}..."
python3 - "$REPO" "$DATE" "$OUT" <<'PY'
import json, os, sys
from huggingface_hub import HfApi

repo, date, out = sys.argv[1], sys.argv[2], sys.argv[3]
api = HfApi()
# Single non-recursive call per date folder
tree = api.list_repo_tree(repo=repo, path=date, recursive=False)
entries = []
for item in tree:
    if item.type == "file":
        entries.append({
            "repo": repo,
            "path": f"{date}/{item.path}",
            "sha": item.commit_id or "",
            "size": getattr(item, "size", 0)
        })
with open(out, "w") as f:
    json.dump(entries, f, indent=2)
print(f"Wrote {len(entries)} entries to {out}")
PY

echo "Snapshot created: ${OUT}"
```

### 2. Shard worker snippet (`bin/dataset-enrich.sh`)
```bash
#!/usr/bin/env bash
set -euo pipefail

SHARD_ID="${SHARD_ID:-0}"
TOTAL_SHARDS="${TOTAL_SHARDS:-16}"
SNAPSHOT="${SNAPSHOT_PATH:-snapshot.json}"

python3 - "$SHARD_ID" "$TOTAL_SHARDS" "$SNAPSHOT" <<'PY'
import json, hashlib, subprocess, sys, os, time

shard_id = int(sys.argv[1])
total_shards = int(sys.argv[2])
snapshot_path = sys.argv[3]

with open(snapshot_path) as f:
    files = json.load(f)

def shard_for(path):
    slug = os.path.basename(path)
    h = int(hashlib.sha256(slug.encode()).hexdigest(), 16)
    return h % total_shards

my_files = [f for f in files if shard_for(f["path"]) == shard_id]

def download_with_backoff(url, max_retries=8):
    for attempt in range(1, max_retries + 1):
        try:
            # stream download -> parse -> project to {prompt,response} -> write shard output
            # Example: curl -L "$url" | parse_and_project.py >> "shard-${shard_id}.jsonl"
            subprocess.run(["curl", "-L", "--fail", url], check=True)
            return
        except subprocess.CalledProcessError as e:
            # crude 429 detection by exit code / stderr; backoff + long sleep
            if attempt == max_retries:
                raise
            sleep_sec = 30 * (2 ** (attempt - 1))
            if sleep_sec > 360:
                sleep_sec = 360
            print(f"[shard {shard_id}] attempt {attempt} failed, sleeping {sleep_sec}s: {e}")
            time.sleep(sleep_sec)

for f in my_files:
    url = f"https://huggingface.co/datasets/{f['repo']}/resolve/main/{f['path']}"
    print(f"[shard {shard_id}] CDN fetch: {url}")
    download_with_backoff(url)
PY
```

### 4. Lightning Studio reuse (`launch_studio.py`)
```python
from lightning import Studio, Machine, Teamspace

def get_or_create_studio(name="surrogate-1-train", machine=Machine.L40S):
    for s in Teamspace.studios:
        if s.name == name and s.status == "Running":
            print(f"Reusing running studio: {s.id}")
            return s
    print(f"Creating studio {name} on {machine}")
    return Studio.create(name=name, machine=machine, create_ok=True)

# If stopped, restart explicitly
