# workio / design

## Final consolidated solution (correct + actionable)

### 1. Diagnosis (merged)
- LINE retries and user double-taps produce duplicate webhook deliveries.
- No DB uniqueness on `webhook_event_id` (or equivalent) → duplicate clock rows and corrupted daily totals.
- Clock-in/out state checks are not atomic with insert → concurrent events can both pass “last state” check and create invalid transitions.
- Non-DB side-effects (LINE replies, notifications) lack idempotency → retries can send duplicate user messages.
- Missing replay-window/TTL on idempotency keys → unbounded growth and false positives after restores/migrations.

### 2. DB schema (single source of truth)

```sql
-- Idempotency keys for webhook and other non-db side-effects
CREATE TABLE IF NOT EXISTS idempotency_keys (
  key          TEXT PRIMARY KEY,
  payload_hash TEXT NOT NULL,
  response_body TEXT,
  status_code  INTEGER,
  created_at   TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  expires_at   TIMESTAMPTZ NOT NULL DEFAULT (NOW() + INTERVAL '24 hours')
);
CREATE INDEX IF NOT EXISTS idx_idempotency_expires ON idempotency_keys (expires_at);

-- Raw LINE webhook events for audit and idempotency
CREATE TABLE IF NOT EXISTS line_webhook_events (
  id               BIGSERIAL PRIMARY KEY,
  webhook_event_id TEXT NOT NULL,
  user_id          TEXT NOT NULL,
  event_type       TEXT NOT NULL,
  payload          JSONB NOT NULL,
  processed_at     TIMESTAMPTZ,
  created_at       TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  CONSTRAINT uq_line_webhook_event UNIQUE (webhook_event_id)
);
CREATE INDEX IF NOT EXISTS idx_line_webhook_user_type ON line_webhook_events (user_id, event_type);
CREATE INDEX IF NOT EXISTS idx_line_webhook_pending ON line_webhook_events (processed_at) WHERE processed_at IS NULL;

-- Clock events table (adjust name if your schema differs)
-- Ensure uniqueness to prevent duplicates at DB level
ALTER TABLE IF EXISTS clock_events
  ADD CONSTRAINT uq_user_clock_type_time UNIQUE (user_id, event_type, created_at);

-- Optional: add foreign key to webhook event for traceability
ALTER TABLE clock_events
  ADD COLUMN IF NOT EXISTS source_webhook_event_id TEXT;
CREATE INDEX IF NOT EXISTS idx_clock_webhook_event ON clock_events (source_webhook_event_id);
```

### 3. Service: atomic clock record (clockService.ts)

```ts
// workio/server/src/services/clockService.ts
import { Pool } from 'pg';

const pool = new Pool({ connectionString: process.env.DATABASE_URL });

export type ClockType = 'clock_in' | 'clock_out';

export async function recordClock(
  userId: string,
  type: ClockType,
  timestamp: Date,
  location: { lat: number; lng: number } | null,
  webhookEventId: string
) {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // 1) Record webhook event once (idempotent)
    const insertEvent = `
      INSERT INTO line_webhook_events (webhook_event_id, user_id, event_type, payload, processed_at)
      VALUES ($1, $2, $3, $4, NOW())
      ON CONFLICT (webhook_event_id) DO NOTHING
      RETURNING id;
    `;
    const eventRes = await client.query(insertEvent, [
      webhookEventId,
      userId,
      type,
      JSON.stringify({ timestamp, location }),
    ]);

    const isDuplicateEvent = eventRes.rowCount === 0;

    // If duplicate event, try to return existing clock row for consistency
    if (isDuplicateEvent) {
      const find = `
        SELECT id, user_id, event_type, created_at
        FROM clock_events
        WHERE source_webhook_event_id = $1
        LIMIT 1;
      `;
      const found = await client.query(find, [webhookEventId]);
      await client.query('ROLLBACK');
      if (found.rows[0]) return { created: false, clock: found.rows[0], duplicate: true };
      // If missing (edge), continue to allow insert by constraints below
    }

    // 2) Atomic transition check + insert
    const insertClock = `
      WITH last_event AS (
        SELECT event_type
        FROM clock_events
        WHERE user_id = $1
          AND created_at < $3
        ORDER BY created_at DESC
        LIMIT 1
      ),
      allowed AS (
        SELECT
          CASE
            WHEN NOT EXISTS (SELECT 1 FROM last_event) AND $2 = 'clock_in' THEN true
            WHEN (SELECT event_type FROM last_event) = 'clock_in' AND $2 = 'clock_out' THEN true
            WHEN (SELECT event_type FROM last_event) = 'clock_out' AND $2 = 'clock_in' THEN true
            ELSE false
          END AS ok
      )
      INSERT INTO clock_events (user_id, event_type, created_at, lat, lng, source_webhook_event_id)
      SELECT $1, $2, $3, $4, $5, $6
      FROM allowed
      WHERE allowed.ok = true
      ON CONFLICT (user_id, event_type, created_at) DO NOTHING
      RETURNING id, user_id, event_type, created_at;
    `;
    const clockRes = await client.query(insertClock, [
      userId,
      type,
      timestamp,
      location?.lat ?? null,
      location?.lng ?? null,
      webhookEventId,
    ]);

    if (clockRes.rowCount === 0) {
      await client.query('ROLLBACK');
      const last = `
        SELECT id, user_id, event_type, created_at
        FROM clock_events
        WHERE user_id = $1
        ORDER BY created_at DESC
        LIMIT 1;
      `;
      const lastRes = await client.query(last, [userId]);
      return {
        created: false,
        clock: lastRes.rows[0] || null,
        reason: 'invalid_transition_or_duplicate',
      };
    }

    await client.query('COMMIT');
    return { created: true, clock: clockRes.rows[0] };
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  } finally {
    client.release();
  }
}
```
