**Why**: Eliminates HF API rate limits during training and prevents Lightning idle-stop from killing long runs.

**Files to modify/create**:
- `/opt/axentx/airship/surrogate/train.py` (main training script)

### 2) CDN-only dataset loader (`surrogate/train.py`)

```python
# surrogate/train.py
import json
import pyarrow.parquet as pq
import requests
from torch.utils.data import IterableDataset, DataLoader
from typing import Iterator

HF_CDN = "https://huggingface.co/datasets"

class CDNParquetIterable(IterableDataset):
    """
    Loads Parquet files directly from HF CDN (no API/auth/rate-limit).
    Expects file_list.json generated by the Mac helper above.
    Projects only {prompt, response} at parse time.
    """
    def __init__(self, file_list_path: str, split: str = "train"):
        super().__init__()
        with open(file_list_path) as f:
            meta = json.load(f)
        self.repo = meta["repo"]
        self.folder = meta["folder"]
        self.files = meta["files"]
        if not self.files:
            raise ValueError("No parquet files found in file_list.json")

    def _stream_urls(self) -> Iterator[str]:
        for fn in self.files:
            # CDN URL bypasses API auth and rate limits
            yield f"{HF_CDN}/{self.repo}/resolve/main/{self.folder}/{fn}"

    def __iter__(self) -> Iterator[dict]:
        worker_info = torch.utils.data.get_worker_info()
        files = list(self._stream_urls())

        # Simple sharding by worker
        if worker_info is not None:
            files = files[worker_info.id::worker_info.num_workers]

        for url in files:
            try:
                # Streaming HTTP -> Arrow -> project {prompt,response}
                resp = requests.get(url, timeout=30, stream=True)
                resp.raise_for_status()
                with pq.ParquetFile(pq.ParquetReader(resp.raw)) as pf:
                    # Read in small row-groups to limit memory
                    for batch in pf.iter_batches(batch_size=512, columns=["prompt", "response"]):
                        df = batch.to_pydict()
                        for prompt, response in zip(df["prompt"], df["response"]):
                            yield {"input_ids": prompt, "labels": response}
            except Exception as exc:
                print(f"[CDN loader] skip {url}: {exc}")
                continue

# Example usage in training loop
def make_dataloader(file_list: str, batch_size: int = 8, num_workers: int = 4):
    ds = CDNParquetIterable(file_list)
    return DataLoader(ds, batch_size=batch_size, num_workers=num_workers)
```

# surrogate/lightning_runner.py
import time
import os
from lightning import LightningApp, LightningFlow, LightningWork, Machine
from lightning.app.components import LightningTrainer
from lightning.app.storage import Drive

# Optional: reuse running studio to save quota
from lightning.pytorch import Trainer
from lightning.pytorch.callbacks import ModelCheckpoint, LearningRateMonitor
from lightning.pytorch.loggers import CSVLogger

from surrogate.train import make_dataloader

SURROGATE_STUDIO_NAME = "surrogate-training"
MACHINE = Machine.L40S  # free-tier fallback: L40S; H200 requires lightning-lambda-prod

class SurrogateTrainer(LightningWork):
    def __init__(self):
        super().__init__()
        self.trainer = None

    def run(self, file_list_path: str = "file_list.json"):
        # Re-check studio status before each run (idle-stop resilience)
        from lightning.pytorch import Trainer
        from lightning.app.components import _is_running_in_cloud

        if _is_running_in_cloud():
            # In Lightning Cloud, ensure we restart machine if stopped
            from lightning import Teamspace
            studios = Teamspace.studios
            running = [s for s in studios if s.name == SURROGATE_STUDIO_NAME and s.status == "running"]
            if not running:
                print(f"[Runner] Studio {SURROGATE_STUDIO_NAME} not running. Starting...")
                from lightning.app.components import LightningTrainer
                # Re-launch with same machine
                self.trainer = Trainer(
                    accelerator="gpu",
                    devices=1,
                    max_epochs=1,
                    logger=CSVLogger("logs"),
                    callbacks=[
                        ModelCheckpoint(monitor="loss"),
                        LearningRateMonitor(),
                    ],
                )
            else:
                self.trainer = Trainer(
                    accelerator="gpu",
                    devices=1,
                    max_epochs=1,
                    logger=CSVLogger("logs"),
                    callbacks=[
                        ModelCheckpoint(monitor="loss"),
                        LearningRateMonitor(),
                    ],
                )
        else:
            # Local / dev mode
            self.trainer = Trainer(
                accelerator="auto",
                devices="auto",
                max_epochs=1,
                logger=CSVLogger("logs"),
                callbacks=[
                    ModelCheckpoint(monitor="loss"),
                    LearningRateMonitor(),
                ],
            )

        # Build dataloader via CDN (zero HF API calls during training)
        train_loader = make_dataloader(file_list_path, batch_size=8, num_workers=4)

        # Minimal model stub — replace with your Surrogate model
        import torch.nn as nn
        import torch.optim as optim

        class TinySurrogate(nn.Module):
            def __init__(self):
                super().__init__()
                self.net = nn.Transformer(d_model=512, nhead=8, num_encoder_layers=6)

            def forward(self, x):
                return self.net(x)

        model = TinySurrogate()

        # Simple training step
        class SurrogateModule(LightningFlow):
            def __init__(self, model, train_loader):
                super().__init__()
                self.model = model
                self.train_loader = train_loader
                self.opt = optim.AdamW(model.parameters(), lr=1e-4)

            def training_step(self, batch):
                # Replace with real loss/forward
                x = batch["input_ids"]
                y = batch["labels"]
                loss = torch.nn.functional.cross_entropy(x.float(), y.long())
                return loss

            def configure_optimizers(self):
                return self.opt

        # Run training (idempotent; safe to call multiple times)
        module = SurrogateModule(model, train_loader)
        self.trainer.fit(module, train_loader)

class SurrogateApp(LightningApp):
    def __init__(self):
        super().__init__()
        self.trainer = SurrogateTrainer()

    def run(self):
        # Always use file_list.json generated on Mac
        self.trainer.run(file_list_path="file_list.json")

if __name__ == "__main__":
    # Local dev entrypoint
    app = SurrogateApp()
