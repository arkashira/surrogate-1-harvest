# surrogate-1 / design

## Final Implementation Plan (≤2 h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID` and `SHARD_TOTAL` (16) from the GitHub Actions matrix.
- Uses a precomputed `manifest.json` (generated once per date folder on a Mac orchestrator) listing **only immediate file paths** for that folder.  
  - This eliminates recursive `list_repo_files` and avoids HF API 429s during ingestion.
- Downloads each assigned file via **HF CDN** (`https://huggingface.co/datasets/{repo}/resolve/main/{path}`) with no Authorization header (bypasses `/api/` rate limits).
- Streams each file with `datasets` but **never uses `streaming=True` on heterogeneous repos**; instead downloads raw bytes and projects to `{prompt, response}` at parse time to avoid `pyarrow.CastError`.
- Deduplicates via the existing `lib/dedup.py` central md5 store.
- Writes output to `batches/public-merged/{date}/shard{N}-{HHMMSS}.jsonl` containing **only** `{prompt, response}` (no extra metadata columns) with deterministic filenames to prevent cross-shard collisions.
- Exits with standard success/failure codes so GitHub Actions can retry cleanly.

Workflow changes:
- Add a one-time “prepare manifest” job (or run locally on Mac) before the 16-shard matrix to produce `manifest.json` for the target date folder.
- Keep the 16-shard matrix but pass `MANIFEST_URL` (CDN) and `DATE` to each runner.
- Ensure `SHELL=/bin/bash` in any cron or wrapper usage (per past lessons).

#!/usr/bin/env python3
"""

def download_manifest(url: str) -> list[str]:
    resp = requests.get(url, timeout=30)
    resp.raise_for_status()
    data = resp.json()
    if isinstance(data, list):
        return data
    raise ValueError("Manifest must be a JSON list of file paths")

def assign_shard(path: str, total: int, shard_id: int) -> bool:
    """Deterministic shard assignment by path hash."""
    h = int(hashlib.sha256(path.encode()).hexdigest(), 16)
    return (h % total) == shard_id

def safe_project_to_pair(raw_obj: Dict[str, Any]) -> Optional[Dict[str, str]]:
    """
    Project heterogeneous file content to {prompt, response}.
    Returns None if no valid pair can be extracted.
    """
    if not isinstance(raw_obj, dict):
        return None

    # Common patterns (adjust to your actual schemas)
    if "prompt" in raw_obj and "response" in raw_obj:
        p, r = str(raw_obj["prompt"]).strip(), str(raw_obj["response"]).strip()
        if p and r:
            return {"prompt": p, "response": r}
        return None

    # Alternate field names
    prompt_keys = {"prompt", "input", "question", "instruction"}
    response_keys = {"response", "output", "answer", "completion"}

    p = None
    r = None
    for k, v in raw_obj.items():
        if k in prompt_keys and p is None and v is not None:
            p = str(v).strip()
        if k in response_keys and r is None and v is not None:
            r = str(v).strip()

    if p and r:
        return {"prompt": p, "response": r}
    return None

def read_jsonl_lines(path: Path) -> Iterable[Dict[str, Any]]:
    open_fn = gzip.open if path.suffix == ".jsonl.gz" else open
    mode = "rt" if path.suffix == ".jsonl.gz" else "r"
    with open_fn(path, mode, encoding="utf-8") as f:  # type: ignore
        for line in f:
            line = line.strip()
            if not line:
                continue
            try:
                yield json.loads(line)
            except Exception:
                continue

def stream_and_project(path: str) -> Iterable[Dict[str, str]]:
    """
    Download file via hf_hub_download (CDN-backed) and project to pairs.
    Avoid load_dataset(streaming=True) on heterogeneous repos.
    """
    local_path = hf_hub_download(
        repo_id=REPO,
        filename=path,
        repo_type="dataset",
        token=HF_TOKEN or None,
        force_download=False,
        cache_dir=None,
    )
    p = Path(local_path)

    # JSONL (or .jsonl.gz)
    if p.suffix in (".jsonl", ".gz") and p.suffixes != [".jsonl", ".gz"]:
        # Normalize: treat .jsonl.gz as .jsonl.gz
        pass

    if p.suffix == ".jsonl" or (len(p.suffixes) == 2 and p.suffixes[1] == ".gz" and p.suffixes[0] == ".jsonl"):
        for obj in read_jsonl_lines(p):
            pair = safe_project_to_pair(obj)
            if pair:
                yield pair
        return

    # Single JSON object/array
    if p.suffix == ".json":
        try:
            with open(p, "r", encoding="utf-8") as f:
                data = json.load(f)
        except Exception:
            return
        items = data if isinstance(data, list) else [data]
        for obj in items:
            pair = safe_project_to_pair(obj)
            if pair:
                yield pair
        return

    # Parquet
    if p.suffix == ".parquet":
        try:
            ds = Dataset.from_parquet(str(p))
            for row in ds:
                pair = safe_project_to_pair(row)
                if pair:
                    yield pair
        except Exception:
            return

    # Unknown format — skip
    return

def compute_content_hash(content: str) -> str:
    return hashlib.md5(content.encode("utf-8")).hexdigest()

def main() -> int:
    try:
        manifest = download_manifest(MANIFEST_URL)
    except Exception as e:
        print(f"Failed to download manifest: {e}", file=sys.stderr)
        return 1
