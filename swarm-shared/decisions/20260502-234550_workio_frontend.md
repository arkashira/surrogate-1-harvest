# workio / frontend

## Final Synthesized Plan (adopted)

**Core principle**: Keep existing URLs unchanged; add a single source-of-truth route catalog with compile-time types, runtime validation (Zod), typed navigation/link helpers, and route-level guards + loading/error boundaries. This merges the strongest, non-overlapping parts of both proposals and resolves contradictions in favor of correctness and concrete actionability.

### 2. Route catalog (`src/routes.ts`)

- Single canonical catalog with path, param schema, search schema.
- Compile-time-safe path builder and type helpers.
- Reuse validation schemas across frontend forms and route validation.

// Types
export type RouteKey = keyof typeof routes;
export type RoutePath = (typeof routes)[RouteKey]['path'];

// Path builder (compile-time safe)
export function buildPath<
  K extends RouteKey,
  P = (typeof routes)[K] extends { paramSchema: infer Z } ? Z : undefined,
  S = (typeof routes)[K] extends { searchSchema: infer Z } ? Z : undefined
>(
  key: K,
  params?: P extends undefined ? undefined : z.infer<Exclude<P, undefined>>,
  search?: S extends undefined ? undefined : z.infer<Exclude<S, undefined>>
): string {
  const route = routes[key];
  let path = route.path;

  if (params && 'paramSchema' in route && route.paramSchema) {
    const parsed = (route.paramSchema as z.ZodSchema).parse(params);
    Object.entries(parsed).forEach(([k, v]) => {
      path = path.replace(`:${k}`, encodeURIComponent(String(v)));
    });
  }

  if (search && 'searchSchema' in route && route.searchSchema) {
    const parsed = (route.searchSchema as z.ZodSchema).parse(search);
    const sp = new URLSearchParams();
    Object.entries(parsed).forEach(([k, v]) => {
      if (v != null) sp.set(k, String(v));
    });
    const qs = sp.toString();
    if (qs) path = `${path}?${qs}`;
  }

  return path;
}

// Utilities for extracting expected shapes (useful for forms)
export function getParamSchema<K extends RouteKey>(key: K) {
  return (routes[key] as any).paramSchema as z.ZodSchema | undefined;
}

export function getSearchSchema<K extends RouteKey>(key: K) {
  return (routes[key] as any).searchSchema as z.ZodSchema | undefined;
}
