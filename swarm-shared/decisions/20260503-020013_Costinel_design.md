# Costinel / design

## Final Implementation — Top-Hub Signal Panel (Costinel Dashboard)

**Scope & value**  
- Frontend-only, read-only React panel that surfaces the highest-scoring hub (default “MOC”) and its actionable proposals from the knowledge graph.  
- CDN-first JSON to eliminate HF API rate limits and keep the dashboard fast.  
- Zero backend changes; deployable in <2h.

### 3) Reusable CDN fetch hook

`src/hooks/useCdnJson.ts`

```ts
import { useEffect, useState, useCallback } from 'react';

export function useCdnJson<T>(url: string, options: { refreshInterval?: number } = {}) {
  const [data, setData] = useState<T | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  const fetchData = useCallback(() => {
    setLoading(true);
    fetch(url, { cache: 'no-store' })
      .then((res) => {
        if (!res.ok) throw new Error(`Failed to load: ${res.status}`);
        return res.json() as Promise<T>;
      })
      .then((json) => {
        setData(json);
        setError(null);
      })
      .catch((err) => {
        setError(err.message);
        setData(null);
      })
      .finally(() => setLoading(false));
  }, [url]);

  useEffect(() => {
    fetchData();
    if (options.refreshInterval) {
      const id = setInterval(fetchData, options.refreshInterval);
      return () => clearInterval(id);
    }
  }, [fetchData, options.refreshInterval]);

  return { data, loading, error, refetch: fetchData };
}
```
