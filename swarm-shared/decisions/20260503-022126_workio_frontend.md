## Final Synthesis: Action Plan

### 1. Core Problems to Fix (merged diagnosis)
- **Missing/frontend-only RBAC**: No systematic role/permission checks in UI.
- **Race conditions and duplicate punches**: No idempotency or inflight deduplication for clock-in/out; optimistic UI is unsafe.
- **Poor UX feedback**: No consistent loading/disabled states; users can retry and race.
- **Error handling gaps**: Unclear how API and LINE errors are handled.
- **Responsive/UI quality**: Layouts may not be optimized across devices.
- **Code quality**: Inconsistent organization, naming, and comments.

### 2. Single Proposed Change (scope)
Implement two tightly integrated deliverables:
1. **Frontend RBAC system** (centralized roles/permissions + guarded routes/components).
2. **Idempotent punch mutation layer** with safe optimistic updates and UI guards (button disabled during inflight, dedupe by employee/type/date).

Also add lightweight cross-cutting improvements:
- Consistent loading/disabled states and error boundaries/toasts.
- One shared `utils/errors.js` for API + LINE error handling.
- Basic responsive layout fixes where most impactful.

export const ROLES = {
  SUPER_ADMIN: 'super_admin',
  ADMIN: 'admin',
  MANAGER: 'manager',
  EMPLOYEE: 'employee',
};

// Permissions map: which roles can perform which actions
export const PERMISSIONS = {
  CLOCK_IN_OUT: [ROLES.EMPLOYEE, ROLES.MANAGER, ROLES.ADMIN, ROLES.SUPER_ADMIN],
  VIEW_REPORTS: [ROLES.MANAGER, ROLES.ADMIN, ROLES.SUPER_ADMIN],
  MANAGE_USERS: [ROLES.ADMIN, ROLES.SUPER_ADMIN],
  // extend as needed
};

export function hasPermission(userRole, permission) {
  return Array.isArray(PERMISSIONS[permission]) && PERMISSIONS[permission].includes(userRole);
}

// Optional: route-level guard helper
export function canAccess(userRole, requiredPermission) {
  return hasPermission(userRole, requiredPermission);
}

#### B. Idempotent punch mutation + safe optimistic UI
File: `src/features/punch/hooks/usePunchMutation.ts`
```typescript
// src/features/punch/hooks/usePunchMutation.ts
import { useCallback, useRef } from 'react';
import { useQueryClient } from '@tanstack/react-query';
import { v4 as uuidv4 } from 'uuid';
import { punchIn, punchOut } from '../api/punchApi';
import type { PunchResponse, PunchType } from '../types';
import { handleApiError } from '../../utils/errors';

type InflightKey = `${string}-${PunchType}-${string}`; // employeeId-type-YYYY-MM-DD

export function usePunchMutation() {
  const queryClient = useQueryClient();
  const inflight = useRef<Set<InflightKey>>(new Set());

  const execute = useCallback(
    async ({
      employeeId,
      type,
      location,
      imageBlob,
    }: {
      employeeId: string;
      type: PunchType;
      location: GeolocationPosition['coords'];
      imageBlob?: Blob;
    }): Promise<PunchResponse | null> => {
      const now = new Date();
      const dateFragment = now.toISOString().slice(0, 10); // YYYY-MM-DD
      const idempotencyKey = uuidv4();
      const inflightKey = `${employeeId}-${type}-${dateFragment}` as InflightKey;

      // Deduplicate in-flight requests for same employee/type/date
      if (inflight.current.has(inflightKey)) {
        return null;
      }
      inflight.current.add(inflightKey);

      const cacheKey = ['punches', employeeId, dateFragment] as const;
      const previous = queryClient.getQueryData<PunchResponse[]>(cacheKey);

      // Optimistic punch (safe: remove any pending same-type for this date)
      const optimisticPunch: PunchResponse = {
        id: `optimistic-${idempotencyKey}`,
        employeeId,
        type,
        timestamp: now.toISOString(),
        location: {
          latitude: location.latitude,
          longitude: location.longitude,
          accuracy: location.accuracy,
        },
        imageUrl: imageBlob ? URL.createObjectURL(imageBlob) : undefined,
        synced: false,
      };

      queryClient.setQueryData<PunchResponse[]>(cacheKey, (old) => {
        const syncedOnly = (old ?? []).filter((p) => p.synced !== false);
        const withoutPendingSameType = syncedOnly.filter((p) => !(p.synced === false && p.type === type));
        return [...withoutPendingSameType, optimisticPunch];
      });

      try {
        const result =
          type === 'IN'
            ? await punchIn({ employeeId, location, imageBlob, idempotencyKey })
            : await punchOut({ employeeId, location, imageBlob, idempotencyKey });

        // Replace optimistic with server truth
        queryClient.setQueryData<PunchResponse[]>(cacheKey, (old) => {
          const withoutOptimistic = (old ?? []).filter((p) => p.id !== optimisticPunch.id);
          const withoutPrevServerSameType = withoutOptimistic.filter((p) => !(p.type === type && p.synced !== false));
          return [...withoutPrevServerSameType, result];
        });

        return result;
      } catch (err) {
        // Rollback optimistic on failure
        queryClient.setQueryData<PunchResponse[]>(cacheKey, (old) =>
          (old ?? []).filter((p) => p.id !== optimisticPunch.id)
        );
        handleApiError(err);
        return null;
      } finally {
        inflight.current.delete(inflightKey);
      }
    },
    [queryClient]
  );

  return { execute };
}
```
