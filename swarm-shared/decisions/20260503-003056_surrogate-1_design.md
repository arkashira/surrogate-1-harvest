Eliminate Hugging Face API calls during training to avoid 429s and commit limits, and enable fully reproducible, CDN-based data loading.

2. A lightweight CDN-backed `IterableDataset` that streams files via `requests` (no HF API).
3. A small training entrypoint patch to accept `--manifest` and use the CDN dataset.
4. Optional CI step to generate and commit the manifest (or upload as artifact).

## Concrete steps

### 1) Create Mac/Linux orchestrator snapshot script  
Path: `scripts/snapshot_hf_dataset.py`  
- Uses `huggingface_hub.HfApi` to list repo tree once (`recursive=False`).  
- Respects 429 with exponential backoff and a 360s ceiling.  
- Emits deterministic `file-manifest.json` (sorted paths).  
- Uses public CDN URLs (`/resolve/main/...`) — no auth required for public datasets.

```python
#!/usr/bin/env python3
"""
Create a deterministic CDN file manifest for axentx/surrogate-1-training-pairs.
Usage:
    HF_TOKEN=... python scripts/snapshot_hf_dataset.py \
        --repo axentx/surrogate-1-training-pairs \
        --folder batches/public-merged/2026-05-03 \
        --out file-manifest.json
"""
import argparse
import json
import os
import sys
import time
from datetime import datetime, timezone
from pathlib import Path

from huggingface_hub import HfApi, hf_hub_url

HF_API_RATE_LIMIT_RETRY = 360  # seconds

def exponential_backoff(attempt: int) -> int:
    return min(2 ** attempt, HF_API_RATE_LIMIT_RETRY)

def list_folder_safe(api, repo_id: str, folder: str, max_retries: int = 5):
    for attempt in range(max_retries):
        try:
            tree = api.list_repo_tree(repo_id=repo_id, path=folder, recursive=False)
            return sorted(t.path for t in tree if t.type == "file")
        except Exception as exc:
            status = getattr(exc, "status", None)
            if status == 429:
                wait = exponential_backoff(attempt)
                print(f"[snapshot] HF 429 hit, waiting {wait}s", file=sys.stderr)
                time.sleep(wait)
                continue
            raise
    raise RuntimeError(f"Failed to list {repo_id}/{folder} after {max_retries} retries")

def build_manifest(repo_id: str, folder: str, output_path: str):
    api = HfApi(token=os.getenv("HF_TOKEN"))
    print(f"[snapshot] Listing {repo_id}/{folder} ...")
    files = list_folder_safe(api, repo_id, folder)

    manifest = {
        "repo_id": repo_id,
        "folder": folder,
        "generated_at": datetime.now(timezone.utc).isoformat(),
        "files": [],
    }

    for f in files:
        cdn_url = hf_hub_url(repo_id=repo_id, filename=f, repo_type="dataset")
        cdn_url = cdn_url.replace("/api/", "/resolve/")
        manifest["files"].append(
            {
                "path": f,
                "cdn_url": cdn_url,
                "local_key": Path(f).name,
            }
        )

    out = Path(output_path)
    out.parent.mkdir(parents=True, exist_ok=True)
    out.write_text(json.dumps(manifest, indent=2))
    print(f"[snapshot] Wrote {len(files)} files -> {out}")

def main():
    parser = argparse.ArgumentParser(description="Create CDN file manifest.")
    parser.add_argument("--repo", default="axentx/surrogate-1-training-pairs")
    parser.add_argument("--folder", required=True, help="Folder to snapshot, e.g. batches/public-merged/2026-05-03")
    parser.add_argument("--out", default="file-manifest.json")
    args = parser.parse_args()

    build_manifest(args.repo, args.folder, args.out)

if __name__ == "__main__":
    main()
```

### 2) Add CDN-backed streaming dataset  
Path: `surrogate_1/data/cdn_dataset.py`  
- Accepts `file-manifest.json`.  
- Streams each file via CDN URL with `requests.get(..., stream=True)`.  
- Parses JSONL line-by-line and yields `{prompt, response}`.  
- No `load_dataset` on heterogeneous repo (avoids pyarrow cast errors).  
- Optional integrity checks (size or sha/md5 from manifest if provided).

```python
import json
import logging
from typing import Dict, Iterator, Optional

import torch
from torch.utils.data import IterableDataset
import requests

logger = logging.getLogger(__name__)

class CdnDataset(IterableDataset):
    """
    Stream dataset files via CDN URLs from a manifest produced by snapshot_hf_dataset.py.

    Manifest format:
    {
      "repo_id": "...",
      "folder": "...",
      "generated_at": "...",
      "files": [
        {"path": "...", "cdn_url": "...", "local_key": "..."},
        ...
      ]
    }

    Each file is expected to be JSONL with at least {prompt, response} fields.
    """

    def __init__(
        self,
        manifest_path: str,
        start_index: int = 0,
        end_index: Optional[int] = None,
        max_retries: int = 3,
        timeout: int = 30,
    ):
        super().__init__()
        self.manifest_path = manifest_path
        self.start_index = start_index
        self.end_index = end_index
        self.max_retries = max_retries
        self.timeout = timeout
        self._manifest = None

    def _load_manifest(self):
        if self._manifest is None:
            with open(self.manifest_path, "r") as f:
                self._manifest = json.load(f)
        return self._manifest

    def _stream_file(self, cdn_url: str) -> Iterator[Dict]:
        for attempt in range(self.max_retries):
            try:
                with requests.get(cdn_url, stream=True, timeout=self.timeout) as r:
                    r.raise_for_status()
                    for line in r.iter_lines(decode_unicode=True):
                        if not line:
                            continue
                        line = line.strip()
                        if not line:
                            continue
                        obj = json.loads(line)
                        yield {"prompt": obj["prompt"], "response": obj["response"]}
                return
            except Exception as exc:
                logger.warning(f"CDN stream error (attempt {attempt+1}/{self.max_retries}): {exc}")
                if attempt + 1 == self.max_retries:
                    raise
                time.sleep(2 ** attempt)

    def _iter_manifest(self):
        manifest = self._load_manifest()
        files = manifest["files"]
        end = len(files) if self.end_index is None else self.end_index
        for idx in range(self.start_index, end):
            entry = files[idx]
            yield entry["cdn_url"]

    def __iter__(self) -> Iterator[Dict]:
        worker_info = torch.utils.data.get_worker_info()
        if worker_info is None:
            # single-process
            urls = list(self._iter_manifest())
        else:
            # shard across workers
            urls = list(self._iter_manifest())
            per_worker = len(urls) // worker_info.num_workers
            worker_id = worker_info.id
            start = worker_id * per_worker
            end = start + per_worker if worker_id < worker_info.num_workers - 1 else len(urls)
            urls = urls[start:end]

        for cdn_url in urls:
            try:
                yield from self._stream_file(cdn_url)
            except Exception as exc:
                logger.error(f"Failed to stream {cdn_url}: {exc}")
                raise
```
