# vanguard / ops

# 1. Diagnosis
- No persisted `(repo, dateFolder) → file-list` manifest; every training/data-selection run triggers authenticated `list_repo_tree` against HF API, burning quota and risking 429s.
- Training/data loader uses `load_dataset(streaming=True)` or repeated per-file API calls instead of CDN-only fetches, causing auth overhead and rate-limit exposure.
- No reuse guard for Lightning Studio; scripts create new studios instead of reusing running ones, wasting ~80hr/mo quota.
- No idle-stop resilience: Lightning Studio stops after idle and kills training; no auto-restart on `Machine.L40S` before `.run()`.
- Mac/local runs may attempt `model.from_pretrained()` or heavy compute instead of delegating to Lightning/Kaggle/Cerebras.

# 2. Proposed change
Create `/opt/axentx/vanguard/ops/file_list_cache.py` (single source of truth for repo+date → file-list JSON) and patch the training launcher to:
- Generate/refresh cache on Mac (post-rate-limit window) via one `list_repo_tree` call per dateFolder.
- Embed cache path in Lightning training script so data loader uses CDN-only URLs (`https://huggingface.co/datasets/{repo}/resolve/main/{path}`) with zero API calls.
- Reuse running LightStudios by name; restart L40S if stopped before `.run()`.
- Guard against local heavy compute with an `assert not is_mac_or("cuda")` for `.from_pretrained()`.

# 3. Implementation

```bash
# /opt/axentx/vanguard/ops/file_list_cache.py
#!/usr/bin/env python3
"""
Generate and consume (repo, dateFolder) -> file-list cache to avoid HF API
rate limits and enable CDN-only training fetches.
"""
import json
import os
from datetime import datetime
from pathlib import Path
from typing import Dict, List, Optional

try:
    from huggingface_hub import list_repo_tree
except ImportError:
    list_repo_tree = None

CACHE_DIR = Path(__file__).parent.parent / ".cache" / "file_lists"
CACHE_DIR.mkdir(parents=True, exist_ok=True)

def _cache_path(repo: str, date_folder: str) -> Path:
    safe_repo = repo.replace("/", "_")
    return CACHE_DIR / f"{safe_repo}__{date_folder}.json"

def build_file_list_cache(repo: str, date_folder: str, overwrite: bool = False) -> List[str]:
    """
    Single API call: list_repo_tree(repo, path=date_folder, recursive=False).
    Returns list of file paths (relative to repo root) under date_folder.
    """
    cache_file = _cache_path(repo, date_folder)
    if cache_file.exists() and not overwrite:
        return json.loads(cache_file.read_text())

    if list_repo_tree is None:
        raise RuntimeError("huggingface_hub not installed; cannot build file list.")

    # One paginated call for the folder (non-recursive). CDN downloads are free of API limits.
    tree = list_repo_tree(repo=repo, path=date_folder, recursive=False)
    files = [item.rfilename for item in tree if item.type == "file"]

    cache_file.write_text(json.dumps(files, indent=2))
    return files

def load_file_list_cache(repo: str, date_folder: str) -> Optional[List[str]]:
    cache_file = _cache_path(repo, date_folder)
    if not cache_file.exists():
        return None
    return json.loads(cache_file.read_text())

def cdn_url(repo: str, file_path: str) -> str:
    """CDN URL that bypasses HF API auth/rate limits."""
    return f"https://huggingface.co/datasets/{repo}/resolve/main/{file_path}"

def generate_manifest(repo: str, date_folder: str, overwrite: bool = False) -> Path:
    """
    Create manifest JSON with repo/dateFolder and CDN urls for training script embedding.
    """
    files = build_file_list_cache(repo, date_folder, overwrite=overwrite)
    manifest = {
        "repo": repo,
        "date_folder": date_folder,
        "generated_at_utc": datetime.utcnow().isoformat() + "Z",
        "files": [{"path": f, "cdn_url": cdn_url(repo, f)} for f in files],
    }
    manifest_path = CACHE_DIR / f"{repo.replace('/', '_')}__{date_folder}.manifest.json"
    manifest_path.write_text(json.dumps(manifest, indent=2))
    return manifest_path

if __name__ == "__main__":
    # Example CLI usage (run from Mac after rate-limit window clears):
    # python -m ops.file_list_cache <repo> <dateFolder> [--overwrite]
    import sys
    repo_arg = sys.argv[1] if len(sys.argv) > 1 else "datasets/mycorp/surrogate-1"
    date_arg = sys.argv[2] if len(sys.argv) > 2 else "2026-04-29"
    overwrite_flag = "--overwrite" in sys.argv
    manifest = generate_manifest(repo_arg, date_arg, overwrite=overwrite_flag)
    print(f"Manifest written: {manifest}")
```
