# Costinel / design

Below is the single, consolidated design that keeps the strongest, most actionable parts of both proposals and resolves contradictions in favor of correctness and immediate usability.

---

## Costinel Discovery Module — Final Design

**Goal**  
Ship a lightweight, deterministic discovery module that produces machine-readable manifests to enable repeatable onboarding, training, and governance without repeated API calls or rate-limit risk.

**Key outputs (stable contract)**  
- `manifests/sources.json` — canonical source catalog (cloud accounts + dataset file lists + top-hub context).  
- `manifests/discovery.json` — readiness checks + CDN-bypass file list for training.  
- Single CLI entrypoint `discover.py` usable on Mac or CI and importable by other scripts.

**Why this wins**  
- Removes repeated discovery in training/governance (deterministic artifacts).  
- Embeds top-hub context immediately for governance enrichment.  
- Fits within a short implementation window (single file + small tests).  
- Caching and CDN-bypass reduce HF rate-limit and latency risk.

## CLI contract

```bash
python discover.py --out-dir manifests --ttl-minutes 60 --repo org/dataset
```

- `--out-dir`: where `sources.json` and `discovery.json` are written.  
- `--ttl-minutes`: cache TTL for external calls (default 60).  
- `--repo`: HF dataset repo to list files for (repeatable).  
- Exit codes: 0 = success, 1 = fatal, 2 = partial (some clouds failed).

def _cache_path(out_dir: Path, name: str) -> Path:
    return out_dir / f".cache_{name}.json"


def _read_cache(out_dir: Path, name: str, ttl: int) -> Optional[Any]:
    p = _cache_path(out_dir, name)
    if not p.exists():
        return None
    try:
        mtime = p.stat().st_mtime
        if time.time() - mtime > ttl:
            return None
        with p.open() as f:
            return json.load(f)
    except Exception:
        return None


def _write_cache(out_dir: Path, name: str, payload: Any) -> None:
    p = _cache_path(out_dir, name)
    out_dir.mkdir(parents=True, exist_ok=True)
    with p.open("w") as f:
        json.dump(payload, f)


def _now_iso() -> str:

def run_knowledge_rag() -> Dict[str, Any]:
    """
    Execute knowledge-rag pipeline and return top-hub context.
    Expected script: scripts/knowledge-rag.sh -> prints JSON to stdout.
    """
    script = Path(__file__).parent.parent / "scripts" / "knowledge-rag.sh"
    if not script.exists():
        return {"top_hub": None, "note": "knowledge-rag.sh not found"}

    try:
        result = subprocess.run(
            ["/bin/bash", str(script)],
            capture_output=True,
            text=True,
            timeout=120,
            env={**os.environ, "SHELL": "/bin/bash"},
        )
        if result.returncode != 0:
            return {"top_hub": None, "error": result.stderr.strip() or "non-zero exit"}
        payload = json.loads(result.stdout.strip())
        return payload
    except Exception as exc:
        return {"top_hub": None, "error": str(exc)}


def discover_aws() -> List[Dict[str, Any]]:
    """
    Return accessible AWS account(s). Deterministic and minimal.
    """
    if boto3 is None:
        return [{"provider": "aws", "error": "boto3 not available"}]

    try:
        sts = boto3.client("sts")
        caller = sts.get_caller_identity()
        iam = boto3.client("iam")
        aliases_resp = iam.list_account_aliases()

        return [
            {
                "provider": "aws",
                "account_id": caller["Account"],
                "arn": caller["Arn"],
                "aliases": aliases_resp.get("AccountAliases", []),
                "region": boto3.Session().region_name or "us-east-1",
            }
        ]
    except (ClientError, NoCredentialsError) as exc:
        return [{"provider": "aws", "error": str(exc)}]
    except Exception as exc:
        return [{"provider": "aws", "error": str(exc)}]


def discover_azure() -> List[Dict[str, Any]]:
    """
    Return available Azure subscriptions.
    """
    if SubscriptionClient is None or DefaultAzureCredential is None:
        return [{"provider": "azure", "error": "azure SDK not available"}]

    try:
        credential = DefaultAzureCredential()
        client = SubscriptionClient(credential)
        subscriptions = list(client.subscriptions.list())
        out = []
        for sub in subscriptions:
            out.append(
                {
                    "provider": "azure",
                    "subscription_id": sub.subscription_id,
                    "display_name": sub.display_name,
                    "state": sub.state,
                }
            )
        return out
    except Exception as exc:
        return [{"provider": "azure", "error": str(exc)}]


def discover_gcp() -> List[Dict[str, Any]]:
    """
    Return GCP billing accounts accessible.
    """
    if billing_v1 is None:
        return [{"provider": "gcp", "error": "google-cloud-billing not available"}]

    try:
        client = billing_v1.CloudBillingClient()
        # List billing accounts accessible to the caller
        request = billing_v1.ListBillingAccountsRequest()
        response = client.list_billing_accounts(request=request)
        accounts = []
        for ba in response:
            accounts.append(
                {
                    "provider": "gcp",
                    "billing_account_id": ba.name,
                    "display_name": ba.display_name,
                    "open": ba.open,
                }
            )
        return accounts
    except Exception as exc:
        return [{"provider": "gcp", "error": str(exc)}]
