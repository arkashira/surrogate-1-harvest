# Costinel / ops

## Final Implementation Plan — Top-Hub Signal Panel (Costinel Dashboard)

**Scope & Value**  
- Frontend-only, read-only panel that surfaces the most-connected hub (default **MOC**) and its actionable proposals from the knowledge graph.  
- Ships in **<2h** as a resilient, self-contained widget that can be dropped into the existing dashboard layout.  
- Uses **CDN-bypass pattern** for graph data to avoid Hugging Face API rate limits during dev/preview and runtime.  
- Includes **offline-first fallback** and **graceful degradation** when CDN or index is unavailable.

---

- `src/components/dashboard/TopHubSignalPanel.vue` (new) — reusable panel component.  
- `src/views/DashboardView.vue` — mount panel into existing dashboard grid/layout.

### 2) Data Strategy (CDN-bypass + pre-list + offline fallback)

- **During development / rate-limit windows**: use local sample in `public/data/knowledge-graph/moc-hub.json`.  
- **In production / preview**: use CDN URLs without Authorization header:
  ```
  https://huggingface.co/datasets/axentx/costinel-knowledge-graph/resolve/main/moc/{file}
  ```
- **Pre-list graph folder once** on orchestrator (Mac) after rate-limit window clears and commit or host an `index.json`:
  ```bash
  python -c "
  from huggingface_hub import list_repo_tree
  import json, os
  repo = 'axentx/costinel-knowledge-graph'
  folders = list_repo_tree(repo, path='moc', recursive=False)
  files = [f.rfilename for f in folders if f.rfilename.endswith('.json')]
  os.makedirs('public/data/knowledge-graph', exist_ok=True)
  # optional: write local index for CDN-bypass during dev
  with open('public/data/knowledge-graph/moc-index.json', 'w') as idx:
      json.dump(files, idx)
  # optional: fetch latest into public for offline dev
  import requests
  if files:
      latest = files[0]
      url = f'https://huggingface.co/datasets/{repo}/resolve/main/moc/{latest}'
      r = requests.get(url, timeout=30)
      with open(f'public/data/knowledge-graph/moc-hub.json', 'wb') as f:
          f.write(r.content)
  print('Index + sample saved')
  "
  ```
- Runtime behavior:
  1. Try CDN `index.json` (or `moc-index.json`) → fetch latest file.
  2. If CDN fails, fall back to `public/data/knowledge-graph/moc-hub.json`.
  3. If both fail, show graceful error with retry button.

### 3) Composable — `useHubSignals.ts`

```ts
// src/composables/useHubSignals.ts
import { ref } from 'vue';

export interface Signal {
  id: string;
  title: string;
  description: string;
  category: string;
  severity: 'critical' | 'high' | 'medium' | 'low';
  link?: string;
}

export function useHubSignals(repo: string, hubPath: string, hubId: string) {
  const signals = ref<Signal[]>([]);
  const loading = ref(false);
  const error = ref<string | null>(null);
  const lastUpdated = ref<string>('-');

  const severityOrder: Record<string, number> = { critical: 0, high: 1, medium: 2, low: 3 };

  async function fetchFromCDN(file: string) {
    const res = await fetch(
      `https://huggingface.co/datasets/${repo}/resolve/main/${hubPath}/${file}`
    );
    if (!res.ok) throw new Error('CDN unavailable');
    return res.json();
  }

  async function fetchLocalIndex() {
    const res = await fetch(`/data/knowledge-graph/${hubPath}-index.json`);
    if (!res.ok) throw new Error('Local index unavailable');
    return res.json() as string[];
  }

  async function fetchLocalSample() {
    const res = await fetch(`/data/knowledge-graph/${hubPath}-hub.json`);
    if (!res.ok) throw new Error('Local sample unavailable');
    return res.json();
  }

  async function loadSignals() {
    loading.value = true;
    error.value = null;
    try {
      // Try CDN index first
      let files: string[] = [];
      try {
        files = await fetchLocalIndex();
      } catch {
        // fallback: try CDN root index.json
        try {
          const res = await fetch(
            `https://huggingface.co/datasets/${repo}/resolve/main/${hubPath}/index.json`
          );
          if (res.ok) files = await res.json();
        } catch {
          // continue to local sample
        }
      }

      let graph: any = null;

      if (files && files.length > 0) {
        // fetch latest from CDN
        try {
          graph = await fetchFromCDN(files[0]);
        } catch {
          // ignore, try local sample
        }
      }

      if (!graph) {
        // fallback to committed local sample
        graph = await fetchLocalSample();
      }

      const proposals = (graph?.proposals || graph?.signals || []) as any[];
      signals.value = proposals
        .filter((p: any) => (p.hub || p.hubId || hubId) === hubId && p.active !== false)
        .map((p: any) => ({
          id: p.id || p.slug || crypto.randomUUID(),
          title: p.title || p.slug || 'Untitled',
          description: p.context || p.summary || p.description || '',
          category: p.category || 'General',
          severity: (p.severity || 'medium') as Signal['severity'],
          link: p.sourceUrl || p.contextUrl || p.link || null,
        }))
        .sort((a, b) => (severityOrder[a.severity] ?? 3) - (severityOrder[b.severity] ?? 3));

      lastUpdated.value = new Date().toLocaleString();
    } catch (err: any) {
      error.value = err?.message || 'Unable to load hub signals.';
      console.error(err);
    } finally {
      loading.value = false;
    }
  }

  return {
    signals,
    loading,
    error,
    lastUpdated,
    loadSignals,
  };
}
```
