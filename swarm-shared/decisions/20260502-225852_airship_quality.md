## Code Snippets

### 1. `airship/discover.sh` (Bash-first, zero-runtime-dependency)

```bash
#!/usr/bin/env bash
# airship/discover.sh
# Discover dataset files from HF repo and emit CDN-safe manifest + status.
# Usage:
#   HF_REPO="datasets/myorg/myrepo" ./discover.sh [--date 2026-04-29] [--out ./dist]
#
# Requirements:
#   - curl, jq
#   - HF_TOKEN optional (for private repos; public repos do not require it for CDN)
#
# Notes:
#   - Uses HF API only once (list_repo_tree) to avoid rate limits.
#   - Emits CDN URLs (resolve/main/) that can be fetched without Authorization.
#   - Deterministic sort ensures reproducible manifests across runs.
#   - Idempotent: lockfile prevents concurrent runs.

set -euo pipefail

REPO="${HF_REPO:-}"
DATE="${DATE:-}"
OUT="${OUT:-./dist}"
HF_API="${HF_API:-https://huggingface.co/api}"

usage() {
  cat <<EOF
Usage: $(basename "$0") [OPTIONS]

Options:
  --repo REPO        HF dataset repo (e.g., datasets/myorg/myrepo) [env: HF_REPO]
  --date DATE        Date folder to list (YYYY-MM-DD). If omitted, lists repo root.
  --out DIR          Output directory (default: ./dist)
  --help             Show this help

Environment:
  HF_TOKEN           Optional token for private repos (not used for CDN fetches).

Outputs:
  \${OUT}/manifest.json  Deterministic list of files with CDN URLs.
  \${OUT}/status.json    Summary metadata for CI/training scripts.

Examples:
  HF_REPO="datasets/axentx/airship-mirror" DATE="2026-04-29" ./discover.sh --out ./dist
EOF
  exit 1
}

# Parse simple flags
while [[ $# -gt 0 ]]; do
  case $1 in
    --repo) REPO="$2"; shift 2 ;;
    --date) DATE="$2"; shift 2 ;;
    --out)  OUT="$2"; shift 2 ;;
    --help) usage ;;
    *) echo "Unknown option: $1"; usage ;;
  esac
done

if [[ -z "$REPO" ]]; then
  echo "ERROR: --repo or HF_REPO is required." >&2
  usage
fi

mkdir -p "$OUT"

# Idempotent lockfile
LOCKFILE="${OUT}/.discover.lock"
if [[ -f "$LOCKFILE" ]]; then
  echo "INFO: Lockfile exists. Another instance may be running. Exiting." >&2
  exit 1
fi
touch "$LOCKFILE"
trap 'rm -f "$LOCKFILE"' EXIT

# Normalize repo type prefix
if [[ "$REPO" =~ ^datasets/ ]]; then
  API_PATH="datasets/${REPO#datasets/}"
elif [[ "$REPO" =~ ^models/ ]]; then
  API_PATH="models/${REPO#models/}"
else
  API_PATH="datasets/$REPO"
fi

# Build tree API URL (non-recursive to avoid pagination explosion)
TREE_URL="${HF_API}/${API_PATH}/tree?recursive=false"
if [[ -n "$DATE" ]]; then
  TREE_URL="${TREE_URL}&path=${DATE}"
fi

echo "INFO: Fetching tree from ${TREE_URL}"

# Optional auth header for private repos
AUTH_HEADER=()
if [[ -n "${HF_TOKEN:-}" ]]; then
  AUTH_HEADER=(-H "Authorization: Bearer ${HF_TOKEN}")
fi

# Single API call (cached by CI). jq ensures deterministic ordering.
TREE_JSON=$(curl -fsSL "${AUTH_HEADER[@]:+${AUTH_HEADER[@]}}" "$TREE_URL" | jq -c '.')
if [[ -z "$TREE_JSON" || "$TREE_JSON" == "null" ]]; then
  echo "ERROR: Empty response from HF API. Check repo/path and token." >&2
  exit 1
fi

# Deterministic manifest: sort by path, include CDN URL (resolve/main/)
MANIFEST=$(echo "$TREE_JSON" | jq -c '
  (if type == "array" then . else [] end)
  | map(select(.type == "file"))
  | sort_by(.path)
  | map({
      path: .path,
      size: (.size // 0),
      lfs: (.lfs // false),
      oid: (.oid // null),
      url: ("https://huggingface.co/" + $REPO + "/resolve/main/" + .path)
    })')

# Count and totals
COUNT=$(echo "$MANIFEST" | jq 'length')
TOTAL_BYTES=$(echo "$MANIFEST" | jq '[.[] | .size] | add // 0')
GENERATED_AT=$(date -u +"%Y-%m-%dT%H:%M:%SZ")

# Status payload
STATUS=$(jq -n \
  --arg repo "$REPO" \
  --arg date "$DATE" \
  --argjson count "$COUNT" \
  --argjson total_bytes "$TOTAL_BYTES" \
  --arg generated_at "$GENERATED_AT" \
  --arg cdn_base "https://huggingface.co/${REPO}/resolve/main/" \
  '{
    repo: $repo,
    date: ($date // null),
    count: $count,
    total_bytes: $total_bytes,
    generated_at: $generated_at,
    cdn_base: $cdn_base,
    note: "CDN-safe manifest; use URLs for zero-auth fetches during training"
  }')

# Write outputs
echo "$MANIFEST" | jq '.' > "${OUT}/manifest.json"
echo "$STATUS" | jq '.' > "${OUT}/status.json"

echo "INFO: Wrote manifest (${COUNT} files, ${TOTAL_BYTES} bytes) to ${OUT}/manifest.json"
echo "INFO: Wrote status to ${OUT}/status.json"

# Smoke test: fetch one sample via CDN
SAMPLE_URL=$(echo "$MANIFEST" | jq -r '.[0].url')
if [[ -n "$SAMPLE_URL" && "$SAMPLE_URL" != "null" ]]; then
  echo "INFO: Smoke test: fetching ${SAMPLE_URL} (HEAD request)..."
  HTTP_STATUS=$(curl -s -o /dev/null -w "%{http_code}" -I "$SAMPLE_URL")
  if [[ "$HTTP_STATUS" -eq 200 ]]; then
    echo "INFO: Smoke test passed (HTTP 200)."
  else
    echo "WARN: Smoke test returned HTTP ${HTTP_STATUS}." >&2
  fi
fi
```

Make executable:

```bash
chmod +x airship/discover.sh
```
