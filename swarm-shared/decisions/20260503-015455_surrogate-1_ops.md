# surrogate-1 / ops

## Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Uses a **single `list_repo_tree` snapshot** (JSON manifest) generated once per date on the Mac orchestrator and committed to the repo (or passed via `file-list.json`).  
- Each GitHub Actions shard (`SHARD_ID=0..15`) loads the manifest, keeps only its deterministic slice (`hash(slug) % 16 == SHARD_ID`), then downloads those files **directly via HF CDN** (`https://huggingface.co/datasets/.../resolve/main/...`) with **zero HF API calls during training** — bypassing 429 rate limits entirely.  
- Projects each file to `{prompt, response}` at parse time (no schema pollution), produces `{shard}-{ts}.jsonl`, and exits.  
- Keeps the existing `lib/dedup.py` contract (central md5 store) but removes recursive `list_repo_files` and streaming `load_dataset` calls that trigger `pyarrow` schema errors and rate limits.

### Steps (1h 30m)

1. Create `bin/dataset-enrich.py` (shebang, executable).  
2. Add `bin/gen-manifest.py` (optional helper) — Mac-side: `list_repo_tree` once per date folder → `file-list-{date}.json`.  
3. Update `.github/workflows/ingest.yml` to pass `MANIFEST_PATH` and ensure `SHELL=/bin/bash`.  
4. Remove `bin/dataset-enrich.sh` (or keep as thin wrapper that calls the Python script).  
5. Smoke-test locally with a small manifest slice.

### `bin/dataset-enrich.py`

```python
#!/usr/bin/env python3
"""
CDN-bypass ingestion worker for surrogate-1.
Usage (GH Actions):
  SHARD_ID=3 MANIFEST_PATH=file-list-2026-05-03.json python bin/dataset-enrich.py
"""

import json
import os
import sys
import hashlib
import time
from pathlib import Path
from typing import List, Dict, Any

import requests
import pyarrow as pa
import pyarrow.parquet as pq
from tqdm import tqdm

# -- config --
HF_DATASET = os.getenv("HF_DATASET", "axentx/surrogate-1-training-pairs")
HF_TOKEN = os.getenv("HF_TOKEN", "")
CDN_BASE = f"https://huggingface.co/datasets/{HF_DATASET}/resolve/main"

SHARD_ID = int(os.getenv("SHARD_ID", "0"))
TOTAL_SHARDS = int(os.getenv("TOTAL_SHARDS", "16"))
MANIFEST_PATH = os.getenv("MANIFEST_PATH", "file-list.json")

OUT_DIR = Path(os.getenv("OUT_DIR", "batches/public-merged"))
RUN_TS = time.strftime("%Y%m%d-%H%M%S")
OUT_FILE = OUT_DIR / f"shard{SHARD_ID}-{RUN_TS}.jsonl"
# --

def hf_hash(slug: str) -> int:
    """Deterministic 128-bit hash -> shard bucket."""
    return int(hashlib.md5(slug.encode()).hexdigest(), 16)

def load_manifest(path: str) -> List[str]:
    with open(path) as f:
        data = json.load(f)
    # Expected: list of file paths relative to dataset root
    if isinstance(data, dict) and "tree" in data:
        # handle list_repo_tree shape
        files = [item["path"] for item in data["tree"] if item.get("type") == "file"]
    else:
        files = [item for item in data if isinstance(item, str)]
    return sorted(set(files))

def cdn_download(url: str, timeout: int = 30) -> bytes:
    # CDN does NOT require Authorization for public datasets
    resp = requests.get(url, timeout=timeout)
    resp.raise_for_status()
    return resp.content

def project_to_pair(raw_bytes: bytes, file_path: str) -> Dict[str, str]:
    """
    Lightweight projection to {prompt,response}.
    Supports: parquet, jsonl, json with minimal schema assumptions.
    """
    suffix = Path(file_path).suffix.lower()
    try:
        if suffix == ".parquet":
            tbl = pq.read_table(pa.BufferReader(raw_bytes))
            # pick first string-like columns that look like prompt/response
            cols = tbl.column_names
            prompt_col = next((c for c in cols if "prompt" in c.lower()), cols[0])
            response_col = next((c for c in cols if "response" in c.lower() or "completion" in c.lower()), cols[-1])
            # take first row only (per-file normalization handled upstream)
            return {
                "prompt": str(tbl[prompt_col][0].as_py()),
                "response": str(tbl[response_col][0].as_py()),
            }
        else:
            # assume jsonl or json
            text = raw_bytes.decode("utf-8")
            if text.strip().startswith("{"):
                # single json object
                obj = json.loads(text)
            else:
                # jsonl: take first line
                first = text.strip().splitlines()[0]
                obj = json.loads(first)
            # flexible key matching
            prompt_keys = [k for k in obj if "prompt" in k.lower()]
            response_keys = [k for k in obj if "response" in k.lower() or "completion" in k.lower()]
            prompt = obj[prompt_keys[0]] if prompt_keys else obj[list(obj.keys())[0]]
            response = obj[response_keys[0]] if response_keys else obj[list(obj.keys())[-1]]
            return {"prompt": str(prompt), "response": str(response)}
    except Exception as exc:
        # fallback: minimal safe record
        return {"prompt": file_path, "response": f"decode_error: {exc}"}

def main() -> None:
    OUT_DIR.mkdir(parents=True, exist_ok=True)

    print(f"Shard {SHARD_ID}/{TOTAL_SHARDS} | manifest={MANIFEST_PATH}")
    files = load_manifest(MANIFEST_PATH)
    my_files = [f for f in files if hf_hash(f) % TOTAL_SHARDS == SHARD_ID]
    print(f"Total files: {len(files)} | Assigned: {len(my_files)}")

    written = 0
    with OUT_FILE.open("w", encoding="utf-8") as out_f:
        for rel_path in tqdm(my_files, desc="Ingesting"):
            url = f"{CDN_BASE}/{rel_path}"
            try:
                raw = cdn_download(url)
                pair = project_to_pair(raw, rel_path)
                out_f.write(json.dumps(pair, ensure_ascii=False) + "\n")
                written += 1
            except Exception as exc:
                # log and continue to avoid shard failure
                tqdm.write(f"SKIP {rel_path}: {exc}")
                continue

    print(f"Wrote {written} records to {OUT_FILE}")
    # Optional: upload via huggingface_hub if HF_TOKEN provided
    # (kept minimal here; rely on existing workflow upload step)

if __name__ == "__main__":
    main()
```

### `bin/gen-manifest.py` (Mac orchestrator helper)

```python
#!/usr/bin/env python3
"""
Generate file-list snapshot for a date folder.
Run on Mac (or any machine with HF token + API access) to avoid 429 during training.
"""
import json
import os
from datetime import datetime
from huggingface_hub import HfApi

HF_DATASET = os.getenv("HF_DATASET", "axentx/surrogate-1-training-pairs")
DATE = datetime.utcnow().strftime("%Y-%m-%d")
OUT = f"file-list-{DATE}.json"

api = HfApi()
# non-recursive per folder to avoid pagination explosion; adjust path if needed
tree = api.list_repo_tree(repo_id=HF_DATASET, path="", repo_type="dataset", recursive=False)
# flatten: if you need recursive, do it carefully with per-folder calls
files = [item.path for item in tree if item.type == "file"]

with open(OUT, "w") as f:
    json.dump(files, f, indent=2)

print(f"Saved {len(files)} files to {OUT}")
```
