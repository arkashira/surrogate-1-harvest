# workio / quality

## Final synthesized implementation

**Chosen approach**: merge the strongest, most concrete pieces from both candidates and resolve contradictions in favor of correctness and actionability.

- **Idempotency key**: use per-minute deterministic key **with userId** (`yyyyMMddHHmm-<type>-<userId>`) to protect against retries and LINE webhook replays across clients/devices.  
- **In-flight guard**: keyed by `date+type` for the current minute to prevent concurrent races and double-taps.  
- **Local debounce**: 60s cooldown per action (per type) to satisfy UX expectations and prevent rapid retries within the same minute.  
- **Optimistic UI + clear error/retry**: local pending state, abort-on-race, visible error banner, manual retry, and optional auto-retry for transient failures.  
- **Backend contract**: `POST /api/clock` with `{ type: "in" | "out", idempotencyKey: string }`.

### workio/src/lib/useClockOptimistic.ts
```tsx
import { useState, useCallback, useRef, useEffect } from "react";

export type ClockType = "in" | "out";

interface ClockState {
  lastType: ClockType | null;
  lastIdempotencyKey: string | null;
  pending: boolean;
  error: string | null;
  lastSuccessAt: number | null; // epoch ms of last success (for debounce)
}

const DEBOUNCE_MS = 60_000; // 60s local debounce per action

export function useClockOptimistic(userId: string) {
  const [state, setState] = useState<ClockState>({
    lastType: null,
    lastIdempotencyKey: null,
    pending: false,
    error: null,
    lastSuccessAt: null,
  });

  const abortRef = useRef<AbortController | null>(null);

  // per-minute idempotency key with userId
  const makeIdempotencyKey = useCallback((type: ClockType) => {
    const now = new Date();
    const y = now.getFullYear();
    const m = String(now.getMonth() + 1).padStart(2, "0");
    const d = String(now.getDate()).padStart(2, "0");
    const h = String(now.getHours()).padStart(2, "0");
    const min = String(now.getMinutes()).padStart(2, "0");
    return `${y}${m}${d}${h}${min}-${type}-${userId}`;
  }, [userId]);

  // in-flight guard + debounce guard
  const canSend = useCallback(
    (type: ClockType) => {
      if (state.pending) return false;
      if (state.lastSuccessAt && Date.now() - state.lastSuccessAt < DEBOUNCE_MS) {
        return false;
      }
      return true;
    },
    [state.pending, state.lastSuccessAt]
  );

  const clock = useCallback(
    async (type: ClockType, options?: { force?: boolean }) => {
      if (!options?.force && !canSend(type)) {
        return { ok: false, reason: "suppressed" } as const;
      }

      const idempotencyKey = makeIdempotencyKey(type);
      setState((s) => ({ ...s, pending: true, error: null }));

      // cancel previous in-flight
      if (abortRef.current) abortRef.current.abort();
      abortRef.current = new AbortController();

      try {
        const res = await fetch("/api/clock", {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ type, idempotencyKey }),
          signal: abortRef.current.signal,
        });

        if (!res.ok) {
          const text = await res.text().catch(() => "");
          throw new Error(text || `Server error ${res.status}`);
        }

        setState({
          lastType: type,
          lastIdempotencyKey: idempotencyKey,
          pending: false,
          error: null,
          lastSuccessAt: Date.now(),
        });
        return { ok: true, reason: "success" } as const;
      } catch (err: any) {
        if (err.name === "AbortError") {
          setState((s) => ({ ...s, pending: false }));
          return { ok: false, reason: "aborted" } as const;
        }
        setState((s) => ({ ...s, pending: false, error: err.message || "Unknown error" }));
        return { ok: false, reason: "network" } as const;
      }
    },
    [canSend, makeIdempotencyKey]
  );

  const resetError = useCallback(() => setState((s) => ({ ...s, error: null })), []);

  // optional auto-retry for transient network errors (3s)
  useEffect(() => {
    if (state.error && state.error !== "aborted" && state.lastType) {
      const t = setTimeout(() => {
        clock(state.lastType!, { force: true });
      }, 3000);
      return () => clearTimeout(t);
    }
  }, [state.error, state.lastType, clock]);

  return {
    state,
    clock,
    resetError,
  };
}
```
