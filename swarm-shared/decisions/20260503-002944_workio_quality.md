# workio / quality

## Final Synthesized Solution

### Diagnosis (merged + resolved)
- **Duplicate punches**: caused by missing idempotency on `/punch` and no DB enforcement of “one open punch per user.”
- **Race conditions**: non-transactional writes allow partial or overlapping state changes under retries.
- **Security gap**: punch close must verify ownership (user_id) to prevent cross-user/cross-tenant manipulation.
- **Operational risk**: LINE webhook retries need idempotency + audit trail; transient failures must not create duplicates.

### Proposed Changes (merged + resolved)
- **DB schema**:
  - Add partial unique index: at most one open punch (`clock_out IS NULL`) per user.
  - Add idempotency table with TTL and optional request-hash for dedupe of identical payloads.
- **API contract**:
  - Require `Idempotency-Key` header (min 8 chars) for mutating endpoints.
  - Require authenticated user ID from session/auth (never trust body for ownership).
- **Route logic**:
  - Transactional upsert with explicit last-state check and ownership verification.
  - Reject stale/out-of-order events by checking last punch state.
  - Return deterministic responses for idempotent replays.
- **Middleware**:
  - Idempotency middleware that checks key, serves cached responses, and records new responses transactionally.
  - Ownership middleware that sets `req.userId` from auth (not body).
- **Observability**:
  - Log key, request hash, and punch IDs for reconciliation.
  - Metrics for idempotency hits and constraint violations.

### Implementation

#### 1) DB schema
```sql
-- workio/server/src/db/schema.sql

-- Idempotency keys with TTL (e.g., 24-72h) and optional request hash for identical-payload dedupe
CREATE TABLE IF NOT EXISTS idempotency_keys (
  key TEXT PRIMARY KEY,
  request_hash TEXT NOT NULL,          -- e.g., SHA256(method + path + body)
  user_id INTEGER NOT NULL,            -- who performed the action (for audit)
  response_status INTEGER NOT NULL,
  response_body JSONB NOT NULL,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  expires_at TIMESTAMPTZ NOT NULL      -- set by app (e.g., NOW() + interval '72 hours')
);

-- Index for cleanup and request-hash dedupe (optional)
CREATE INDEX IF NOT EXISTS idx_idempotency_expires ON idempotency_keys (expires_at);
CREATE INDEX IF NOT EXISTS idx_idempotency_user_hash ON idempotency_keys (user_id, request_hash);

-- Punches table assumed existing; add partial unique index to enforce one open punch per user
CREATE UNIQUE INDEX IF NOT EXISTS idx_one_open_punch_per_user
ON punches (user_id)
WHERE clock_out IS NULL;
```

#### 2) Idempotency middleware (lightweight, transactional recording)
```ts
// workio/server/src/middleware/idempotency.ts
import { Request, Response, NextFunction } from 'express';
import { Pool } from 'pg';
import crypto from 'crypto';

const pool = new Pool({ connectionString: process.env.DATABASE_URL });

export interface RequestWithIdempotency extends Request {
  idempotencyKey?: string;
  requestHash?: string;
  userId?: number; // should be set by auth middleware before idempotency runs
}

function hashRequest(req: RequestWithIdempotency, body: any): string {
  const str = `${req.method}:${req.originalUrl}:${JSON.stringify(body || {})}`;
  return crypto.createHash('sha256').update(str).digest('hex');
}

export function requireIdempotencyKey(req: RequestWithIdempotency, res: Response, next: NextFunction) {
  if (req.method === 'GET' || req.method === 'HEAD' || req.method === 'OPTIONS') return next();

  const key = req.header('Idempotency-Key');
  if (!key || typeof key !== 'string' || key.trim().length < 8) {
    return res.status(400).json({ error: 'Idempotency-Key header required (min 8 chars)' });
  }
  req.idempotencyKey = key.trim();
  next();
}

export async function serveIdempotentResponse(req: RequestWithIdempotency, res: Response, next: NextFunction) {
  if (!req.idempotencyKey) return next();

  const { rows } = await pool.query(
    `SELECT response_status, response_body FROM idempotency_keys WHERE key = $1`,
    [req.idempotencyKey]
  );

  if (rows.length > 0) {
    const { response_status, response_body } = rows[0];
    return res.status(response_status).json(response_body);
  }
  next();
}

export function recordIdempotentResponse() {
  return async function (req: RequestWithIdempotency, res: Response, next: NextFunction) {
    if (!req.idempotencyKey) return next();

    const originalJson = res.json.bind(res);
    let capturedBody: any = null;
    res.json = (body) => {
      capturedBody = body;
      return originalJson(body);
    };

    res.on('finish', async () => {
      try {
        if (!req.userId) return; // safety: require userId from auth
        const requestHash = hashRequest(req, req.body);
        const expiresAt = new Date(Date.now() + 72 * 60 * 60 * 1000).toISOString(); // 72h TTL

        // Record only if not already present (race-safe via PK)
        await pool.query(
          `INSERT INTO idempotency_keys (key, request_hash, user_id, response_status, response_body, expires_at)
           VALUES ($1, $2, $3, $4, $5, $6)
           ON CONFLICT (key) DO NOTHING`,
          [req.idempotencyKey, requestHash, req.userId, res.statusCode, JSON.stringify(capturedBody || {}), expiresAt]
        );
      } catch {
        // best-effort logging; don't break response
      }
    });
    next();
  };
}
```
