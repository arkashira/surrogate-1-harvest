## Resolved Contradictions (correctness first)

| Contradiction | Resolution |
|--------------|------------|
| **Manifest generation: recursive vs non-recursive** | Use **non-recursive folder listing + client-side recursion** for large folders. Avoids huge single API response; still one logical call per folder tree. |
| **Parquet parsing: pyarrow vs pandas vs fastparquet** | Use **pyarrow** (standard, robust). Tolerate missing columns; coerce types safely; never fail training on schema mismatch. |
| **HTTP: streaming vs full download vs range requests** | **Stream with retries and backoff**; do not require range requests (CDN supports full streaming). Keep memory bounded via chunked download to temp file/BytesIO. |
| **Lightning Studio: reuse vs restart vs create** | **Reuse if running**; **restart if stopped**; **create if absent**. Always wait for `running` before launching script. |
| **CLI args vs config file** | Support **both**: CLI arg `--manifest` overrides config. Keeps cron/bash usage simple and IDE/debugging flexible. |
| **Error handling: fail-fast vs best-effort** | **Fail-fast on setup** (missing manifest, studio can’t start). **Best-effort per file** during training (log + skip bad parquet). |

### 1) Manifest Generator  
`s/training/gen_manifest.py`
```python
#!/usr/bin/env python3
"""
Generate CDN-only manifest for a HuggingFace dataset folder.
Usage: python gen_manifest.py --repo <repo> --path <folder> --out <manifest.json>
"""
import argparse
import json
import os
from datetime import datetime
from huggingface_hub import HfApi

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def list_folder_recursive(api, repo_id, root):
    """Non-recursive API calls but collect all files under root."""
    to_walk = [root]
    files = []
    while to_walk:
        cur = to_walk.pop()
        items = api.list_repo_tree(repo_id=repo_id, path=cur, recursive=False)
        for item in items:
            if item.type == "file":
                files.append(item)
            else:
                to_walk.append(item.path)
    return files

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True, help="HF dataset repo (user/ds)")
    parser.add_argument("--path", required=True, help="Folder path in repo")
    parser.add_argument("--out", required=True, help="Output manifest JSON path")
    parser.add_argument("--recursive", action="store_true", help="Walk subfolders")
    args = parser.parse_args()

    api = HfApi()
    if args.recursive:
        files = list_folder_recursive(api, args.repo, args.path)
    else:
        files = api.list_repo_tree(repo_id=args.repo, path=args.path, recursive=False)
        files = [f for f in files if f.type == "file"]

    manifest = {
        "repo": args.repo,
        "folder": args.path,
        "generated_at": datetime.utcnow().isoformat() + "Z",
        "files": [],
    }

    for f in files:
        rel_path = os.path.join(args.path, f.path) if args.path else f.path
        manifest["files"].append({
            "path": rel_path,
            "cdn_url": CDN_TEMPLATE.format(repo=args.repo, path=rel_path),
            "size": f.size,
        })

    os.makedirs(os.path.dirname(os.path.abspath(args.out)), exist_ok=True)
    with open(args.out, "w") as fp:
        json.dump(manifest, fp, indent=2)
    print(f"Manifest written to {args.out} ({len(manifest['files'])} files)")

if __name__ == "__main__":
    main()
```

```python
import io
import json
import logging
import time
from typing import Dict, Iterator, List, Optional

import requests

logger = logging.getLogger(__name__)

class CDNParquetDataset:
    """
    CDN-only dataset loader for HF datasets.
    Reads manifest JSON and streams parquet files via CDN URLs.
    Projects to {prompt, response} at parse time.
    """
    def __init__(
        self,
        manifest_path: str,
        columns=("prompt", "response"),
        max_retries: int = 3,
        timeout: int = 120,
    ):
        self.manifest_path = manifest_path
        self.columns = columns
        self.max_retries = max_retries
        self.timeout = timeout
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.files = self.manifest["files"]
        self._session = requests.Session()

    def _stream_parquet(self, url: str) -> bytes:
        """Stream parquet file via CDN with retry/backoff."""
        last_exc = None
        for attempt in range(1, self.max_retries + 1):
            try:
                resp = self._session.get(url, stream=True, timeout=self.timeout)
                resp.raise_for_status()
                buf = io.BytesIO()
                for chunk in resp.iter_content(chunk_size=8192):
                    buf.write(chunk)
                return buf.getvalue()
            except Exception as e:
                last_exc = e
                wait = 2 ** attempt
                logger.warning(f"Attempt {attempt}/{self.max_retries} failed for {url}: {e}; retry in {wait}s")
                time.sleep(wait)
        raise RuntimeError(f"Failed to load {url} after {self.max_retries} attempts") from last_exc

    def _parse_projection(self, data: bytes) -> List[Dict]:
        """Parse parquet bytes and project to {prompt, response}."""
        try:
            import pyarrow.parquet as pq
            table = pq.read_table(io.BytesIO(data))
            available = [c for c in self.columns if c in table.column_names]
            if not available:
                # try common aliases
                aliases = {
                    "prompt": ["input", "question", "text"],
                    "response": ["output", "answer", "completion"],
                }
                for target, alts in aliases.items():
                    if target not in available:
                        for alt in alts:
                            if alt in table.column_names:
                                available.append(alt)
                                break
                available = list(dict.fromkeys(available))  # preserve order

            if not available:
                logger.warning("No target columns found; returning empty")
                return []

            proj = table.select(available).to_pylist()
            out = []
            for row in proj:
                prompt = row.get("prompt") or row.get("input") or row.get("question") or row.get("text") or ""
                response = row.get("response") or row.get("output") or row.get("answer") or row.get("completion") or ""
                out.append({"prompt": str(prompt), "response": str(response)})
            return out
        except Exception as e:
            logger.error(f"Failed to parse parquet: {e}")
            return []
