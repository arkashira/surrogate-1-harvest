#### `scripts/list_hf_files.py`

```python
#!/usr/bin/env python3
"""
List parquet files from a HuggingFace dataset repo (non-recursive per folder)
and emit surrogate/file_list.json for CDN-only training.

Usage:
    python scripts/list_hf_files.py --repo datasets/my-parquet --date 2026-05-03 \
        --out surrogate/file_list.json
"""
import argparse
import json
import sys
import time
from pathlib import Path

import requests

HF_API_BASE = "https://huggingface.co/api"


def list_repo_tree(repo: str, path: str = "", retries: int = 5):
    url = f"{HF_API_BASE}/datasets/{repo}/tree"
    params = {"path": path, "recursive": "false"} if path else {"recursive": "false"}
    for attempt in range(retries):
        try:
            resp = requests.get(url, params=params, timeout=30)
            if resp.status_code == 429:
                wait = 360
                print(f"[rate-limit] 429, sleeping {wait}s", file=sys.stderr)
                time.sleep(wait)
                continue
            resp.raise_for_status()
            return resp.json()
        except Exception as exc:
            wait = 2 ** attempt
            print(f"[error] {exc}, retry in {wait}s", file=sys.stderr)
            time.sleep(wait)
    raise RuntimeError(f"Failed to list {repo}/{path}")


def build_file_list(repo: str, date_folder: str):
    entries = list_repo_tree(repo, path=date_folder)
    out = []
    for e in entries:
        if e.get("type") != "file":
            continue
        name = e["path"]
        if not name.endswith(".parquet"):
            continue
        cdn_url = (
            f"https://huggingface.co/datasets/{repo}/resolve/main/"
            f"{date_folder}/{name}"
        )
        out.append(
            {
                "repo": repo,
                "path": f"{date_folder}/{name}",
                "cdn_url": cdn_url,
                "size": e.get("size", 0),
            }
        )
    return out


def main():
    parser = argparse.ArgumentParser(description="List HF parquet files for CDN training")
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g. datasets/my-data)")
    parser.add_argument("--date", required=True, help="Date folder (e.g. 2026-05-03)")
    parser.add_argument("--out", default="surrogate/file_list.json", help="Output JSON path")
    args = parser.parse_args()

    out_path = Path(args.out)
    out_path.parent.mkdir(parents=True, exist_ok=True)

    file_list = build_file_list(args.repo, args.date)
    out_path.write_text(json.dumps(file_list, indent=2))
    print(f"Wrote {len(file_list)} files to {out_path}")


if __name__ == "__main__":
    main()
```

#### `surrogate/utils/data.py`
```python
import io
import time
from typing import Dict, Iterator, List, Any

import pyarrow.parquet as pq
import requests


def _stream_parquet_rows(cdn_url: str, columns: tuple = ("prompt", "response")) -> Iterator[Dict[str, Any]]:
    backoff = 1
    max_backoff = 60
    for attempt in range(8):
        try:
            with requests.get(cdn_url, stream=True, timeout=60) as r:
                r.raise_for_status()
                # Stream into memory buffer; for very large files consider temp file.
                buf = io.BytesIO(r.content)
                table = pq.read_table(buf, columns=list(columns) if columns else None)
                for batch in table.to_batches():
                    df = batch.to_pandas()
                    for _, row in df.iterrows():
                        yield {k: row[k] for k in (columns or df.columns)}
                return
        except Exception as exc:
            wait = min(backoff, max_backoff)
            print(f"[cdn] {cdn_url} failed ({exc}), retry in {wait}s")
            time.sleep(wait)
            backoff *= 2
    raise RuntimeError(f"Failed to stream {cdn_url}")


def stream_parquet_from_cdn(file_list: List[Dict[str, Any]], columns=("prompt", "response")) -> Iterator[Dict[str, Any]]:
    for item in file_list:
        cdn_url = item["cdn_url"]
        yield from _stream_parquet_rows(cdn_url, columns=columns)
```
