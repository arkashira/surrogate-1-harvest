# vanguard / ops

## Final synthesized implementation

**Core principles adopted**
- CDN-first, zero HF-API-at-runtime: training must never call `list_repo_tree` or `load_dataset` after manifest generation.
- Deterministic, content-addressed, reproducible: pinned file list, fixed ordering, SHA-256 verification, and optional commit-ID pinning.
- Resilient: retries with backoff, range/If-None-Match where possible, atomic writes, and corruption detection with auto-recovery.
- Lightweight ops workflow: generate/validate manifests on the Mac orchestration host; training consumes only the manifest and CDN.
- Minimal, safe changes: one new ops file, one small requirements file, and a single training-launcher update.

#!/usr/bin/env python3
"""
Generate a CDN-first, deterministic manifest for a HuggingFace dataset repo.

Usage:
  python make_manifest.py \
    --repo datasets/your-org/your-repo \
    --folder batches/mirror-merged/2026-05-03 \
    --out manifest.json \
    [--commit <sha>] \
    [--out-dir data] \
    [--verify] \
    [--token $HF_TOKEN]

Produces manifest.json with:
  - repo, folder, commit (optional)
  - generated_at_utc
  - entries: cdn_url, size, sha256, local_path (relative to --out-dir)
"""

import argparse
import hashlib
import json
import os
import sys
import time
from pathlib import Path
from typing import Any, Dict, List, Optional

import requests
from tqdm import tqdm

HF_API_TREE = "https://huggingface.co/api/datasets/{repo}/tree"
HF_CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/{commit}/{path}"

DEFAULT_RETRY = 5
BACKOFF_FACTOR = 1.5

def _request_with_retry(
    method: str,
    url: str,
    *,
    headers: Optional[Dict[str, str]] = None,
    params: Optional[Dict[str, str]] = None,
    timeout: int = 30,
) -> requests.Response:
    retry = 0
    wait = 5
    while True:
        try:
            resp = requests.request(
                method, url, headers=headers, params=params, timeout=timeout
            )
            if resp.status_code == 429:
                retry_after = int(resp.headers.get("Retry-After", wait))
                print(f"Rate limited. Sleeping {retry_after}s", file=sys.stderr)
                time.sleep(retry_after)
                continue
            resp.raise_for_status()
            return resp
        except (requests.RequestException, requests.HTTPError) as exc:
            retry += 1
            if retry > DEFAULT_RETRY:
                raise RuntimeError(f"Request failed after {DEFAULT_RETRY} retries: {exc}") from exc
            print(f"Retry {retry}/{DEFAULT_RETRY} in {wait}s: {exc}", file=sys.stderr)
            time.sleep(wait)
            wait = min(wait * BACKOFF_FACTOR, 300)

def get_tree(repo: str, folder: str, token: Optional[str], commit: Optional[str]) -> List[Dict[str, Any]]:
    url = HF_API_TREE.format(repo=repo)
    params: Dict[str, str] = {"path": folder, "recursive": "false", "type": "file"}
    if commit:
        params["revision"] = commit
    headers = {"Authorization": f"Bearer {token}"} if token else {}
    resp = _request_with_retry("GET", url, headers=headers, params=params)
    return resp.json()

def get_commit_from_tree(repo: str, folder: str, token: Optional[str]) -> Optional[str]:
    """
    Best-effort commit resolution using tree endpoint. If unavailable, returns None.
    Training launcher should prefer explicit --commit for reproducibility.
    """
    url = HF_API_TREE.format(repo=repo)
    params = {"path": folder, "recursive": "false", "type": "file", "limit": "1"}
    headers = {"Authorization": f"Bearer {token}"} if token else {}
    try:
        resp = _request_with_retry("GET", url, headers=headers, params=params)
        data = resp.json()
        # Some tree responses include commit metadata; if not, fallback to None.
        if isinstance(data, dict) and "commit" in data:
            return str(data["commit"])
        if isinstance(data, list) and len(data) > 0 and isinstance(data[0], dict) and "commit" in data[0]:
            return str(data[0]["commit"])
    except Exception:
        pass
    return None

def sha256_file(path: Path, chunk_kb: int = 1024) -> str:
    h = hashlib.sha256()
    with open(path, "rb") as f:
        for chunk in iter(lambda: f.read(chunk_kb * 1024), b""):
            h.update(chunk)
    return h.hexdigest()

def build_manifest(
    repo: str,
    folder: str,
    out_manifest: Path,
    out_dir: Path,
    token: Optional[str] = None,
    commit: Optional[str] = None,
    verify: bool = False,
) -> None:
    out_dir.mkdir(parents=True, exist_ok=True)

    # Resolve commit once for reproducibility when not provided
    if not commit:
        commit = get_commit_from_tree(repo, folder, token)

    items = get_tree(repo, folder, token, commit)
    items = sorted(items, key=lambda x: x["path"])

    entries = []
    for item in tqdm(items, desc="Building manifest"):
        path = item["path"]
        size = item["size"]
        cdn_url = HF_CDN_TEMPLATE.format(repo=repo, commit=commit or "main", path=path)
        local_path = out_dir / path
        local_path.parent.mkdir(parents=True, exist_ok=True)

        sha = None
        if verify and local_path.exists():
            sha = sha256_file(local_path)

        entries.append(
            {
                "cdn_url": cdn_url,
                "size": size,
                "sha256": sha,
                "local_path": str(local_path.relative_to(out_manifest.parent)),
            }
        )

    manifest = {
        "repo": repo,
        "folder": folder,
        "commit": commit,
        "generated_at_utc": time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime()),
        "entries": entries,
    }

    out_manifest.write_text(json.dumps(manifest, indent=2))
    print(f"Manifest written to {out_manifest}")

def main() -> None:
    parser = argparse.ArgumentParser(description="Create CDN-first manifest for HF dataset repo.")
    parser.add_argument("--repo", required=True, help="e.g. datasets/your-org/your-repo")
    parser.add_argument("--folder", required=True, help="Folder path in repo (non-recursive)")
    parser.add_argument("--out", default="manifest.json", help="Output manifest path")
    parser.add_argument("--out-dir", default="data", help="Local download root (relative to manifest parent)")
    parser.add_argument("--token", default=os.getenv("HF_TOKEN"), help="HF token (optional)")
    parser.add_argument("--commit", default=None, help="Commit/revision to pin (recommended)")
    parser.add_argument("--verify", action="store_true", help="Compute sha256 for existing files")
    args = parser.parse_args()

    repo = args.repo.removeprefix("datasets/").removeprefix("https://huggingface.co/datasets/")
    repo = f"datasets/{repo}" if not repo.startswith("datasets/") else repo

    build_manifest(
        repo=repo,
        folder=args.folder,
