# Costinel / design

## Highest-Value Incremental Improvement (<2h)

**Add a lightweight discovery & readiness surface** to Costinel:
- `/ready` and `/health` probes (design-time validation)
- `/discover` endpoint that emits a machine-readable manifest of ingestible sources (cloud accounts, datasets, schemas) and top-hub RAG insights
- CLI helper (`costinel discover`) that pre-lists HF dataset folders to a JSON file for CDN-only training (bypasses HF API rate limits)

This directly unblocks onboarding, RAG contextual bootstrapping, and the surrogate-1 training pipeline without touching execution logic.

---

## Implementation Plan (≤2h)

1. **Add FastAPI probes** (`/ready`, `/health`) with design-time schema validation checks (15 min)
2. **Add `/discover` endpoint** returning:
   - cloud accounts configured
   - available datasets and inferred schemas
   - top-hub RAG doc (e.g., "MOC") summary
   - HF file manifest (date-folders) for CDN training (30 min)
3. **Add CLI helper** `costinel discover` that:
   - runs once on Mac (or CI) after rate-limit window
   - calls `list_repo_tree(path, recursive=False)` for a date folder
   - writes `hf_file_manifest.json` for embedding in Lightning training (15 min)
4. **Add RAG bootstrap** that queries top hub on startup and caches insight (15 min)
5. **Smoke test** locally with `uvicorn` and verify JSON shape (15 min)

```python
# costinel/api/endpoints/discovery.py
from fastapi import APIRouter, HTTPException
from typing import List, Dict, Any
import json
from pathlib import Path
from huggingface_hub import list_repo_tree, HfApi
from datetime import datetime, timedelta

router = APIRouter()

def _infer_schema_from_parquet(path: Path) -> Dict[str, str]:
    # lightweight schema inference for manifest
    try:
        import pyarrow.parquet as pq
        pf = pq.read_schema(path)
        return {name: str(pf.field(name).type) for name in pf.names}
    except Exception:
        return {}

def _top_hub_insight() -> Dict[str, Any]:
    # RAG top-hub lookup (stub; integrate with your graph store)
    # Pattern: top-hub doc insight (2026-04-27) — review most-connected hub (e.g., "MOC")
    return {
        "hub": "MOC",
        "summary": "Most-connected operational hub; primary cost governance patterns and dataset sources live here.",
        "tags": ["#knowledge-rag", "#graph", "#hub"]
    }

def _hf_date_folders(repo: str, days: int = 7) -> List[str]:
    """List top-level date folders (YYYY-MM-DD) in repo for CDN manifest."""
    try:
        tree = list_repo_tree(repo, recursive=False)
        # Expect paths like "batches/mirror-merged/2026-05-01/file.parquet"
        folders = {p.rstrip("/").split("/")[-2] for p in tree if p.count("/") >= 2}
        recent = [f for f in folders if _is_date(f) and _within_days(f, days)]
        return sorted(recent)
    except Exception:
        return []

def _is_date(s: str) -> bool:
    try:
        datetime.strptime(s, "%Y-%m-%d")
        return True
    except ValueError:
        return False

def _within_days(date_str: str, days: int) -> bool:
    try:
        d = datetime.strptime(date_str, "%Y-%m-%d")
        return (datetime.utcnow() - d).days <= days
    except Exception:
        return False

@router.get("/ready")
async def ready() -> Dict[str, str]:
    # Design-time validation: check required config/schema presence
    required_files = ["config/cloud_accounts.yaml", "config/schemas.yaml"]
    missing = [f for f in required_files if not Path(f).exists()]
    if missing:
        raise HTTPException(status_code=503, detail=f"Missing config: {missing}")
    return {"status": "ready"}

@router.get("/health")
async def health() -> Dict[str, str]:
    return {"status": "healthy", "ts": datetime.utcnow().isoformat()}

@router.get("/discover")
async def discover(repo: str = "datasets/costinel/mirror-merged") -> Dict[str, Any]:
    """
    Machine-readable manifest for onboarding and CDN-based training.
    """
    # Cloud accounts (stub — integrate your config loader)
    accounts = []
    ca_path = Path("config/cloud_accounts.yaml")
    if ca_path.exists():
        import yaml
        accounts = yaml.safe_load(ca_path) or []

    # Available datasets and inferred schemas
    datasets = []
    data_root = Path("data/enriched")
    if data_root.exists():
        for parquet in list(data_root.rglob("*.parquet"))[:20]:  # sample
            datasets.append({
                "path": str(parquet),
                "schema": _infer_schema_from_parquet(parquet)
            })

    # HF CDN manifest (date folders) for surrogate-1 training
    hf_folders = _hf_date_folders(repo, days=14)
    hf_manifest = {
        "repo": repo,
        "date_folders": hf_folders,
        "cdn_base": f"https://huggingface.co/datasets/{repo}/resolve/main",
        "note": "Use CDN URLs to bypass HF API rate limits during training."
    }

    return {
        "service": "Costinel",
        "version": "4.2.0",
        "cloud_accounts": accounts,
        "datasets": datasets,
        "top_hub": _top_hub_insight(),
        "hf_cdn_manifest": hf_manifest,
        "generated_at": datetime.utcnow().isoformat()
    }
```

### 2. CLI Helper: Pre-list HF Folder for CDN Training

```bash
#!/usr/bin/env bash
# costinel/cli/discover_hf.sh
# Pattern: HF CDN Bypass — pre-list file paths once, embed in training script

set -euo pipefail
REPO="${1:-datasets/costinel/mirror-merged}"
OUT="${2:-hf_file_manifest.json}"
DAYS="${3:-14}"

echo "Listing HF date folders for repo=$REPO (last $DAYS days)..."

python3 - <<PY
import json
from huggingface_hub import list_repo_tree
from datetime import datetime, timedelta

def is_date(s):
    try:
        datetime.strptime(s, "%Y-%m-%d")
        return True
    except ValueError:
        return False

def within_days(date_str, days):
    try:
        d = datetime.strptime(date_str, "%Y-%m-%d")
        return (datetime.utcnow() - d).days <= days
    except Exception:
        return False

repo = "$REPO"
tree = list_repo_tree(repo, recursive=False)
# Expect structure: batches/mirror-merged/YYYY-MM-DD/...
date_folders = sorted({p.rstrip("/").split("/")[-2] for p in tree if p.count("/") >= 2 and is_date(p.split("/")[-2]) and within_days(p.split("/")[-2], int("$DAYS"))})

manifest = {
    "repo": repo,
    "date_folders": date_folders,
    "cdn_base": f"https://huggingface.co/datasets/{repo}/resolve/main",
    "file_urls": [f"{manifest['cdn_base']}/batches/mirror-merged/{d}/" for d in date_folders],
    "generated_at": datetime.utcnow().isoformat(),
    "note": "Embed this in train.py; Lightning training should use CDN-only fetches (zero API calls)."
}

with open("$OUT", "w") as f:
    json.dump(manifest, f, indent=2)

print(f"Wrote {len(date_folders)} date folders to $OUT")
PY
```
