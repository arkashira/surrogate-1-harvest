#!/usr/bin/env python3
"""
Generate deterministic file list for a date folder to enable CDN-only training.
Run from Mac after HF API window clears; embed output in train.py.
"""
import json
import os
from huggingface_hub import HfApi

HF_REPO = os.getenv("HF_DATASET_REPO", "axentx/surrogate-dataset")
DATE_FOLDER = os.getenv("DATE_FOLDER", "2026-05-03")
OUT_PATH = os.getenv("OUT_PATH", "file_list.json")

def list_date_files(repo: str, folder: str) -> list[str]:
    api = HfApi()
    # non-recursive per folder to avoid pagination explosion
    tree = api.list_repo_tree(repo=repo, path=folder, recursive=False)
    files = []
    for item in tree:
        if item.type == "file":
            files.append(f"{folder}/{item.path}")
        elif item.type == "dir":
            sub = api.list_repo_tree(repo=repo, path=item.path, recursive=False)
            for s in sub:
                if s.type == "file":
                    files.append(f"{item.path}/{s.path}")
    return sorted(set(files))

def cdn_url(repo: str, file_path: str) -> str:
    # CDN bypass: no Authorization header required
    return f"https://huggingface.co/datasets/{repo}/resolve/main/{file_path}"

if __name__ == "__main__":
    file_paths = list_date_files(HF_REPO, DATE_FOLDER)
    payload = {
        "repo": HF_REPO,
        "date_folder": DATE_FOLDER,
        "files": file_paths,
        "cdn_urls": [cdn_url(HF_REPO, f) for f in file_paths],
    }
    with open(OUT_PATH, "w") as f:
        json.dump(payload, f, indent=2)
    print(f"Wrote {len(file_paths)} files to {OUT_PATH}")
```

### `tools/sibling_shard.py`

#!/usr/bin/env python3
"""
Deterministic HF sibling sharding to bypass 128/hr/repo commit cap.
"""
import hashlib

SIBLINGS = [
    "axentx/surrogate-dataset",
    "axentx/surrogate-dataset-s1",
    "axentx/surrogate-dataset-s2",
    "axentx/surrogate-dataset-s3",
    "axentx/surrogate-dataset-s4",
    "axentx/surrogate-dataset-s5",
]

def pick_repo(slug: str) -> str:
    """Hash slug -> pick sibling repo deterministically."""
    digest = hashlib.sha256(slug.encode()).hexdigest()
    idx = int(digest, 16) % len(SIBLINGS)
    return SIBLINGS[idx]

if __name__ == "__main__":
    # quick demo
    for s in ["batch-2026-05-03-a", "batch-2026-05-03-b", "batch-2026-05-03-c"]:
        print(s, "->", pick_repo(s))
```

### `surrogate/data/projection.py`
```python
#!/usr/bin/env python3
"""
Project raw enriched files to {prompt, response} pairs before upload.
Attribution moved to filename pattern:
  batches/mirror-merged/{date}/{slug}.parquet
No 'source' or 'ts' columns in output.
"""
import pandas as pd
from pathlib import Path

def project_to_pair(df: pd.DataFrame) -> pd.DataFrame:
    # Keep only required fields; drop metadata cols
    keep = {"prompt", "response"}
    missing = keep - set(df.columns)
    if missing:
        raise ValueError(f"Missing columns: {missing}")
    return df[list(keep)].dropna(subset=["prompt", "response"]).reset_index(drop=True)

def process_and_save(input_path: Path, output_dir: Path, date: str, slug: str) -> Path:
    df = pd.read_parquet(input_path)
    paired = project_to_pair(df)
    out_dir = output_dir / f"batches/mirror-merged/{date}"
    out_dir.mkdir(parents=True, exist_ok=True)
    out_path = out_dir / f"{slug}.parquet"
    paired.to_parquet(out_path, index=False)
    return out_path
```

### `surrogate/lightning_guard.py`
```python
#!/usr/bin/env python3
"""
Lightning Studio lifecycle guard: reuse running studios, restart on idle-stop.
"""
from lightning import LightningWork, LightningFlow, LightningApp, Machine
from lightning.app import Studio
from typing import Optional

class GuardedStudio(LightningWork):
    def __init__(self, studio_name: str, script_path: str, machine: str = "L40S"):
        super().__init__()
        self.studio_name = studio_name
        self.script_path = script_path
        self.machine = Machine(machine)
        self._studio: Optional[Studio] = None

    def run(self):
        # Reuse running studio if available
        if self._studio is None or self._studio.status != "running":
            from lightning import Teamspace
            for s in Teamspace.studios:
                if s.name == self.studio_name and s.status == "running":
                    self._studio = s
                    print(f"Reusing running studio: {self.studio_name}")
                    break
            else:
                self._studio = Studio(
                    name=self.studio_name,
                    script=self.script_path,
                    create_ok=True,
                    machine=self.machine,
                )
                print(f"Created studio: {self.studio_name}")

        # Guard against idle-stop death
        if self._studio.status == "stopped":
            print(f"Studio stopped; restarting on {self.machine}")
            self._studio.start(machine=self.machine)

        # Run (or re-run) training script
        self._studio.run()

class TrainingFlow(LightningFlow):
    def __init__(self):
        super().__init__()
        self.studio = GuardedStudio(
            studio_name="surrogate-train",
            script_path="train.py",
            machine="L40S",
        )

    def run(self):
        self.studio.run()

if __name__ == "__main__":
    # Mac orchestration only; training happens on Lightning
    app = LightningApp(TrainingFlow())
```
