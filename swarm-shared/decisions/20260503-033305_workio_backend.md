# workio / backend

### 1. Diagnosis (merged + prioritized)
- **Signature verification missing/broken**: No HMAC-SHA256 check of `X-Line-Signature`; body-parser mutates raw body so HMAC always fails.  
- **No idempotency**: LINE retries create duplicate clock-in/out, leave, or OT records.  
- **Tenant context unbound**: tenant inferred only from payload fields; allows cross-tenant event injection.  
- **Replay risk**: no timestamp tolerance or replay-window check.  
- **Poor observability + error handling**: webhook requests/errors are not logged; generic 500s returned.  

### 2. Single, concrete change to make
Add a **raw-body-preserving, signature-verifying, idempotent, tenant-binding middleware** and a small handler for `/webhook/line`.  
File locations (use these exact paths):
- Middleware: `server/src/middleware/lineSignature.ts`  
- Route: `server/src/routes/webhook/line.ts`  
- Service: `server/src/services/lineService.ts`  
- Logger: reuse or create `server/src/utils/logger.ts`  
- Wire-up: `server/src/index.ts` (or `app.ts`)

#### Middleware: `server/src/middleware/lineSignature.ts`
```ts
import crypto from 'crypto';
import { Request, Response, NextFunction } from 'express';
import LRU from 'lru-cache';
import logger from '../utils/logger';

const CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';
const TOLERANCE_MS = 5 * 60 * 1000; // 5 minutes

// In-memory dedupe cache (swap to Redis in prod)
const idempotencyCache = new LRU<string, boolean>({ max: 50000, ttl: 1000 * 60 * 60 });

export function preserveRawBody(req: Request, _res: Response, buf: Buffer) {
  (req as any).rawBody = buf;
}

export function verifyLineSignature(req: Request, res: Response, next: NextFunction) {
  const signature = req.get('X-Line-Signature');
  const deliveryId = req.get('X-Line-Delivery-Id') || '';
  const rawBody = (req as any).rawBody as Buffer;

  if (!CHANNEL_SECRET) {
    logger.error('LINE_CHANNEL_SECRET is not configured');
    return res.status(500).json({ error: 'Server configuration error' });
  }

  if (!signature || !rawBody) {
    return res.status(400).json({ error: 'Missing signature or raw body' });
  }

  // HMAC-SHA256 verification
  const hmac = crypto.createHmac('sha256', CHANNEL_SECRET);
  hmac.update(rawBody);
  const expected = hmac.digest('base64');

  if (signature !== expected) {
    logger.warn('Invalid LINE signature', { signature, expected });
    return res.status(401).json({ error: 'Invalid signature' });
  }

  // Idempotency guard
  const idemKey = deliveryId || crypto.createHash('sha256').update(rawBody).digest('hex');
  if (idempotencyCache.has(idemKey)) {
    logger.debug('Duplicate LINE webhook skipped', { idemKey });
    return res.status(200).json({ ok: true });
  }
  idempotencyCache.set(idemKey, true);

  // Timestamp tolerance (replay protection)
  const body = req.body;
  if (body?.events?.length) {
    for (const ev of body.events) {
      const ts = ev.timestamp;
      if (ts && Math.abs(Date.now() - ts) > TOLERANCE_MS) {
        logger.warn('LINE event timestamp out of tolerance', { ts, now: Date.now() });
        return res.status(400).json({ error: 'Event timestamp out of tolerance' });
      }
    }
  }

  // Attach verified context for downstream handlers
  (req as any).lineDeliveryId = idemKey;
  (req as any).lineVerified = true;
  next();
}
```

#### Route: `server/src/routes/webhook/line.ts`
```ts
import { Router, Request, Response } from 'express';
import { verifyLineSignature, preserveRawBody } from '../middleware/lineSignature';
import { handleLineEvent } from '../services/lineService';
import logger from '../utils/logger';

const router = Router();

router.post(
  '/',
  verifyLineSignature,
  async (req: Request, res: Response) => {
    try {
      const body = req.body;
      if (!body?.events || !Array.isArray(body.events)) {
        return res.status(400).json({ error: 'No events' });
      }

      // Process sequentially to preserve order and simplify idempotency
      for (const event of body.events) {
        await handleLineEvent(event, req);
      }

      return res.status(200).json({ ok: true });
    } catch (err) {
      logger.error('LINE webhook processing failed', err);
      return res.status(500).json({ error: 'Processing failed' });
    }
  }
);

export default router;
```
