# vanguard / ops

## 1. Diagnosis
- Training pipeline still calls `list_repo_tree`/`load_dataset` at runtime → exposes surrogate-1 training to HF API 429s and non-reproducible shard order.
- No content-addressed manifest per date folder → epochs drift across runs and resumable training is unreliable.
- Lightning Studio reuse is not implemented → quota waste (up to 80hr/mo) from repeated studio creation.
- Idle timeout kills training when Lightning Studio stops → training process dies without restart logic.
- Mac orchestration may still attempt local model loading instead of delegating to Lightning/CDN-only fetches.

## 2. Proposed change
Create `/opt/axentx/vanguard/ops/train_launcher.py` (single orchestrator) and modify `/opt/axentx/vanguard/train.py` to accept a pre-generated file manifest. Scope:
- New: `ops/train_launcher.py` — Mac-side orchestrator: list once, write manifest, reuse studio, restart if stopped, launch training with CDN-only mode.
- Modify: `train.py` — accept `--manifest` (JSON) and use `hf_hub_download` per file (CDN bypass) instead of `load_dataset`/`list_repo_tree`.

## 3. Implementation

### File: ops/train_launcher.py
```python
#!/usr/bin/env python3
"""
Vanguard surrogate-1 training launcher (ops).
- Runs on Mac (or any orchestrator)
- Lists HF repo tree ONCE per date folder, writes content-addressed manifest
- Reuses running Lightning Studio or starts one (L40S priority)
- Restarts studio if stopped before each run
- Launches training with CDN-only fetches (zero HF API calls during train)
"""
import json, hashlib, os, sys, time
from pathlib import Path

try:
    from lightning_sdk import Lightning, Teamspace, Machine
except ImportError:
    print("Install lightning-sdk: pip install lightning")
    sys.exit(1)

HF_REPO = os.getenv("HF_REPO", "datasets/your-org/surrogate-1")
DATE_FOLDER = os.getenv("DATE_FOLDER", "2026-04-29")  # e.g. batches/mirror-merged/2026-04-29
MANIFEST_DIR = Path(__file__).parent.parent / "manifests"
MANIFEST_DIR.mkdir(exist_ok=True)
MANIFEST_PATH = MANIFEST_DIR / f"manifest_{DATE_FOLDER}.json"

# ---- HF helpers (Mac side, minimal API calls) ----
def list_date_files(repo: str, folder: str) -> list[str]:
    """Single API call to list files in folder (non-recursive)."""
    from huggingface_hub import HfApi
    api = HfApi()
    items = api.list_repo_tree(repo=repo, path=folder, recursive=False)
    # Expecting parquet files directly in folder
    files = [it.rfilename for it in items if it.rfilename.endswith(".parquet")]
    return sorted(files)

def build_manifest(repo: str, folder: str) -> dict:
    files = list_date_files(repo, folder)
    manifest = {
        "repo": repo,
        "folder": folder,
        "generated_at": time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime()),
        "files": files,
        "sha256": hashlib.sha256(json.dumps(files, sort_keys=True).encode()).hexdigest()[:16]
    }
    return manifest

# ---- Lightning Studio helpers ----
def get_or_start_studio(name: str = "vanguard-surrogate1", machine: str = "L40S") -> Lightning:
    teamspace = Teamspace.current()
    running = [s for s in teamspace.studios if s.name == name and s.status == "running"]
    if running:
        print(f"Reusing running studio: {name}")
        studio = running[0]
    else:
        # Try to find stopped studio and restart; else create
        stopped = [s for s in teamspace.studios if s.name == name and s.status == "stopped"]
        if stopped:
            studio = stopped[0]
            print(f"Starting stopped studio: {name}")
            studio.start(machine=Machine.L40S)  # fallback mapping handled by SDK
        else:
            print(f"Creating studio: {name}")
            studio = teamspace.studios.create(
                name=name,
                machine=Machine.L40S,
                create_ok=True
            )
    return studio

def ensure_studio_running(studio: Lightning, timeout: int = 300) -> bool:
    for _ in range(timeout // 10):
        studio.refresh()
        if studio.status == "running":
            return True
        if studio.status == "stopped":
            print("Studio stopped; restarting...")
            studio.start(machine=Machine.L40S)
        time.sleep(10)
    return False

# ---- Main ----
def main():
    # 1) Build manifest once (Mac side)
    if not MANIFEST_PATH.exists():
        print(f"Building manifest for {DATE_FOLDER}...")
        manifest = build_manifest(HF_REPO, DATE_FOLDER)
        MANIFEST_PATH.write_text(json.dumps(manifest, indent=2))
        print(f"Manifest written: {MANIFEST_PATH}")
    else:
        manifest = json.loads(MANIFEST_PATH.read_text())
        print(f"Using existing manifest: {MANIFEST_PATH}")

    # 2) Reuse/start Lightning Studio
    studio = get_or_start_studio()
    if not ensure_studio_running(studio):
        print("ERROR: Studio not running after timeout")
        sys.exit(1)

    # 3) Launch training inside studio (Lightning SDK run)
    # Pass manifest path and CDN-only flag to train.py
    manifest_rel = os.path.relpath(MANIFEST_PATH, Path.cwd())
    job = studio.jobs.run(
        name="surrogate1-train",
        entry_point=[
            "python", "train.py",
            "--manifest", manifest_rel,
            "--cdn-only"
        ],
        requirements=["torch", "datasets", "huggingface_hub", "pyarrow"],
        # Optional: specify dependencies file if present
    )
    print(f"Launched job: {job.name}")
    # Optionally tail logs or exit (job runs asynchronously)
    # For simplicity we exit; use `job.stream_logs()` to follow.

if __name__ == "__main__":
    main()
```
