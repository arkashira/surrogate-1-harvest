## Final Synthesized Implementation (Best of Both Candidates)

**Goal**: Ship a zero-runtime-dependency, CDN-safe `airship discover` orchestrator that deterministically emits `status.json` + `manifest.json`. Eliminates HF API calls during training/inference by pre-listing once, embedding file lists for CDN-only fetches, and enabling Lightning Studio reuse.

**Resolution of Conttradictions**:
- **CLI location**: Use `scripts/airship-discover.sh` (Candidate 1) for cron-safety, but add a `bin/airship-discover` symlink/wrapper (Candidate 2) for PATH usability.
- **Date vs Ref**: Accept **both** — `REF` (branch/tag) and `DATE` (subfolder filter). DATE is optional; if provided, filters results.
- **API Strategy**: Use HF API **once** with token (Candidate 1), fallback to CDN HTML scrape (no 429), but add **sha256** to manifest (Candidate 2) for integrity.
- **Surrogate vs Studio**: Keep surrogate hook (Candidate 1) **and** add Studio detection/restart (Candidate 2).

## Final Code

### 1. `scripts/airship-discover.sh` (Core Orchestrator)

```bash
#!/usr/bin/env bash
# airship-discover.sh — CDN-safe deterministic discovery
# Emits: status.json + manifest.json (with sha256)
# Usage: ./airship-discover.sh <owner>/<repo> <ref> [date] [outdir]
#   date: optional YYYY-MM-DD subfolder filter (e.g., 2026-05-01)

set -euo pipefail

REPO="${1:-}"
REF="${2:-main}"
DATE="${3:-}"          # Optional date filter
OUTDIR="${4:-./discovery}"
HF_TOKEN="${HF_TOKEN:-}"

if [[ -z "$REPO" ]]; then
  echo "Usage: $0 <owner>/<repo> [ref] [date] [outdir]" >&2
  exit 1
fi

mkdir -p "$OUTDIR"
TS="$(date -u +"%Y-%m-%dT%H:%M:%SZ")"

# --- Emitters ---
emit_status() {
  local state="$1" detail="${2:-}"
  cat > "$OUTDIR/status.json" <<EOF
{
  "repo": "$REPO",
  "ref": "$REF",
  "date_filter": "${DATE:-null}",
  "state": "$state",
  "detail": "$detail",
  "ts": "$TS",
  "orchestrator": "airship-discover"
}
EOF
}

emit_manifest() {
  local files_json="$1"
  cat > "$OUTDIR/manifest.json" <<EOF
{
  "repo": "$REPO",
  "ref": "$REF",
  "date_filter": "${DATE:-null}",
  "ts": "$TS",
  "strategy": "cdn-first",
  "cdn_template": "https://huggingface.co/datasets/${REPO}/resolve/main/{date}/{file}",
  "files": $files_json
}
EOF
}

# --- File Listing ---
list_files() {
  local repo="$1" ref="$2" date_filter="${3:-}"
  local tmp_list="$OUTDIR/.raw_paths"
  > "$tmp_list"

  # 1) Try HF API (single non-recursive call per folder)
  if [[ -n "$HF_TOKEN" ]]; then
    if out=$(curl -sSf -H "Authorization: Bearer $HF_TOKEN" \
      "https://huggingface.co/api/datasets/${repo}/tree?recursive=false&ref=$ref" 2>/dev/null); then
      # Filter by date if provided, extract paths, sort deterministically
      if [[ -n "$date_filter" ]]; then
        echo "$out" | jq -r --arg d "$date_filter" '.[] | select(.path | startswith($d + "/")) | .path' | sort > "$tmp_list"
      else
        echo "$out" | jq -r '.[].path' | sort > "$tmp_list"
      fi
      [[ -s "$tmp_list" ]] && return 0
    fi
  fi

  # 2) Fallback: CDN HTML scrape (no auth, no 429)
  local url="https://huggingface.co/datasets/${repo}/tree/${ref}"
  [[ -n "$date_filter" ]] && url="${url}/${date_filter}"
  if out=$(curl -sSf "$url" 2>/dev/null); then
    # Extract hrefs that look like file paths
    echo "$out" | grep -oP 'href="[^"]+\.[^"]+"' | sed 's|href="||;s|"||' | sort > "$tmp_list" || true
  fi

  # 3) If date_filter provided and no results, try without it (shallow fallback)
  if [[ -n "$date_filter" && ! -s "$tmp_list" ]]; then
    url="https://huggingface.co/datasets/${repo}/tree/${ref}"
    if out=$(curl -sSf "$url" 2>/dev/null); then
      echo "$out" | grep -oP 'href="[^"]+\.[^"]+"' | sed 's|href="||;s|"||' | sort > "$tmp_list" || true
    fi
  fi

  [[ -s "$tmp_list" ]]
}

# --- SHA256 Calculator (best-effort, CDN-safe) ---
calculate_sha256() {
  local file_path="$1" repo="$2" ref="$3"
  # Try to fetch only headers to check existence, then compute if small
  local url="https://huggingface.co/datasets/${repo}/resolve/main/${ref}/${file_path}"
  if curl -sSI "$url" | grep -q "200 OK"; then
    # Only compute for small files (<1MB) to avoid long runtime
    local size=$(curl -sI "$url" | grep -i content-length | awk '{print $2}' | tr -d '\r')
    if [[ -n "$size" && "$size" -lt 1000000 ]]; then
      curl -s "$url" | sha256sum | awk '{print $1}'
    else
      echo "null"
    fi
  else
    echo "null"
  fi
}

# --- Main ---
emit_status "running" "listing files"
if ! list_files "$REPO" "$REF" "$DATE"; then
  emit_status "failed" "no files listed"
  echo "WARN: no files listed for $REPO@$REF (date: ${DATE:-none})" >&2
  exit 1
fi

# Build JSON array with sha256 (best-effort)
FILES_JSON="["
first=true
while IFS= read -r f; do
  [[ -z "$f" ]] && continue
  $first && first=false || FILES_JSON+=","
  sha=$(calculate_sha256 "$f" "$REPO" "$REF")
  if [[ "$sha" == "null" ]]; then
    FILES_JSON+="{\"path\":\"$f\"}"
  else
    FILES_JSON+="{\"path\":\"$f\",\"sha256\":\"$sha\"}"
  fi
done < "$OUTDIR/.raw_paths"
FILES_JSON+="]"

emit_status "completed" "ok"
emit_manifest "$FILES_JSON"

cp "$OUTDIR/manifest.json" "$SURROGATE_DISCOVERY_DIR/${REPO//\//_}_${REF}.json"
