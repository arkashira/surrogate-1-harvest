# airship / backend

**Final Implementation Plan — Deterministic CDN-Only Manifest for `airship discover`**

**Goal (≤2h):**  
Eliminate HF API rate-limit risk during training by making `airship discover` produce a **deterministic, CDN-only file manifest** that training scripts can consume with **zero API/auth calls**.

---

### Scope & Constraints
- **Single API call**: `list_repo_tree(repo_id, path=date_folder, recursive=False)` only.
- **No recursive listing** and no per-file API calls.
- **CDN-only downloads** during training (`resolve/main/...` URLs).
- **CLI unchanged** except adding `--out` and `--cache-manifest`.
- **Training script** consumes manifest and downloads via `requests` (no HF client/auth).
- **Deterministic output**: sorted paths, reproducible builds.
- **Supported file types**: filter to `.parquet`, `.jsonl` (others ignored unless explicitly needed).

---

### Concrete Steps (≤2h)

1. **Locate entrypoint**  
   Edit `airship/cli/discover.py` (create if needed). This will replace/augment any existing recursive HF-API-based listing.

2. **Implement non-recursive folder listing**  
   Use exactly one call:  
   ```python
   tree = list_repo_tree(repo_id=repo_id, path=date_folder, recursive=False)
   ```
   Filter entries to files with extensions `.parquet`, `.jsonl`.

3. **Build deterministic CDN URLs**  
   For each file:  
   ```python
   cdn_url = f"https://huggingface.co/datasets/{repo_id}/resolve/main/{entry.path}"
   ```
   Use `entry.path` directly (already repo-root relative).

4. **Include lightweight metadata**  
   - `size` from tree response (fast, available).  
   - Skip `sha256` unless required (expensive; can be added later).

5. **Write deterministic manifest**  
   Output sorted-by-path JSON:
   ```json
   {
     "repo_id": "...",
     "date_folder": "...",
     "generated_by": "airship-discover",
     "files": [
       {"path": "...", "cdn_url": "...", "size": 12345}
     ]
   }
   ```

6. **Add caching flag**  
   `--cache-manifest <file>`: if present and valid (repo/date match), skip API entirely.

7. **Update training script stub**  
   Add small loader in `train.py` that:
   - Reads manifest.
   - Downloads via `requests.get(cdn_url, stream=True)` with retries + backoff.
   - Validates `size` when provided.
   - Skips already-downloaded files.

8. **Validate**  
   Run on a small public dataset folder and confirm:
   - Manifest is produced.
   - CDN download works.
   - No HF API calls occur during download.

### Final Code (Merged + Corrected)

#### `airship/cli/discover.py`

```python
#!/usr/bin/env python3
import argparse
import json
import os
import sys
from pathlib import Path

from huggingface_hub import list_repo_tree

SUPPORTED_EXTS = {".parquet", ".jsonl"}


def build_cdn_manifest(repo_id: str, date_folder: str, out_path: Path):
    """
    Non-recursive folder listing -> deterministic CDN-only manifest.
    Uses ONE API call.
    """
    date_folder = date_folder.rstrip("/")
    tree = list_repo_tree(repo_id=repo_id, path=date_folder, recursive=False)

    files = []
    for entry in tree:
        if entry.type != "file":
            continue
        ext = Path(entry.path).suffix.lower()
        if SUPPORTED_EXTS and ext not in SUPPORTED_EXTS:
            continue

        # entry.path is repo-root-relative
        if not entry.path.startswith(date_folder + "/"):
            continue

        rel_path = os.path.relpath(entry.path, date_folder).replace("\\", "/")
        cdn_url = (
            f"https://huggingface.co/datasets/{repo_id}/resolve/main/{entry.path}"
        )
        files.append(
            {
                "path": rel_path,
                "cdn_url": cdn_url,
                "size": getattr(entry, "size", None),
            }
        )

    # Deterministic ordering
    files.sort(key=lambda f: f["path"])

    manifest = {
        "repo_id": repo_id,
        "date_folder": date_folder,
        "generated_by": "airship-discover",
        "files": files,
    }

    out_path.parent.mkdir(parents=True, exist_ok=True)
    with open(out_path, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2, ensure_ascii=False)

    print(f"Manifest written to {out_path} ({len(files)} files)")
    return manifest


def main():
    parser = argparse.ArgumentParser(
        description="Generate CDN-only manifest for dataset date folder."
    )
    parser.add_argument("--repo", required=True, help="HF dataset repo id (e.g. 'org/repo')")
    parser.add_argument(
        "--date", required=True, help="Date folder path inside repo (e.g. 'batches/mirror-merged/2026-04-29')"
    )
    parser.add_argument("--out", default="manifest.json", help="Output manifest path")
    parser.add_argument(
        "--cache-manifest", help="Skip API and use cached manifest if present and valid"
    )
    args = parser.parse_args()

    out_path = Path(args.out)
    if args.cache_manifest:
        cache_path = Path(args.cache_manifest)
        if cache_path.exists():
            with open(cache_path, encoding="utf-8") as f:
                cached = json.load(f)
            if cached.get("repo_id") == args.repo and cached.get("date_folder") == args.date.rstrip("/"):
                print(f"Using cached manifest: {cache_path}")
                # Copy to out_path if different
                if cache_path.resolve() != out_path.resolve():
                    out_path.parent.mkdir(parents=True, exist_ok=True)
                    out_path.write_text(cache_path.read_text(encoding="utf-8"), encoding="utf-8")
                sys.exit(0)
            print("Cache mismatch; regenerating.")

    try:
        build_cdn_manifest(args.repo, args.date.rstrip("/"), out_path)
    except Exception as e:
        print(f"Error: {e}", file=sys.stderr)
        sys.exit(1)


if __name__ == "__main__":
    main()
```

#### `train.py` — CDN-only loader stub

```python
import json
import time
from pathlib import Path

import requests
from tqdm import tqdm


def stream_cdn_file(cdn_url, dest_path, chunk_size=8192, max_retries=3):
    for attempt in range(1, max_retries + 1):
        try:
            with requests.get(cdn_url, stream=True, timeout=30) as r:
                r.raise_for_status()
                dest_path.parent.mkdir(parents=True, exist_ok=True)
                with open(dest_path, "wb") as f:
                    for chunk in r.iter_content(chunk_size=chunk_size):
                        f.write(chunk)
                return dest_path
        except Exception as err:
            if attempt == max_retries:
                raise
            time.sleep(2 ** attempt)


def load_from_manifest(manifest_path, cache_dir="data_cache"):
    cache_dir = Path(cache_dir)
    with open(manifest_path, encoding="utf-8") as f:
        manifest = json.load(f)

    for file_info in tqdm(manifest["files"], desc="Downloading (CDN)"):
        rel = file_info["path"]
        dest = cache_dir / manifest["repo_id"] / manifest["date_folder"] / rel
        if dest.exists():
            # Optional: validate size if provided
            if file_info.get("size") and dest.stat().st_size == file_info["size"]:
                continue
            dest.unlink()

        stream_cdn_file(file_info["cdn_url"], dest)

        if file_info.get("size") and dest.stat().st_size != file_info["size"]:
            dest.unlink()
            raise RuntimeError(f"Size mismatch: {rel}")

    return cache_dir
```
