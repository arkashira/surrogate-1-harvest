# workio / ops

## Final consolidated solution

### 1. Diagnosis (merged, de-duplicated)

- **No idempotency guard**: duplicate `X-Line-Signature` or repeated LINE event IDs cause double-apply of clock/leave/OT transitions and corrupt daily totals and approval state.
- **No strict signature time window / replay protection**: accepting signatures beyond a short window enables replay attacks; no monotonic nonce or timestamp enforcement.
- **No deduplication store**: repeated event IDs or signatures can be processed multiple times on retries.
- **Non-atomic state transitions**: partial application on retry leaves DB inconsistent (clock-in without clock-out, leave without approval state).
- **No out-of-order/late-event handling**: late events can overwrite newer state and break summaries.
- **No operational observability**: silent failures on signature mismatches or DB errors blind operations.

### 3. Implementation (single coherent patch)

#### Middleware: signature verification + replay window + idempotency

`server/src/middleware/lineWebhookGuard.ts`

```ts
import { Request, Response, NextFunction } from 'express';
import crypto from 'crypto';
import { getClient } from '../db';

const SIGNATURE_TTL_MS = 5 * 60 * 1000;     // 5 minutes
const MAX_EVENT_DRIFT_MS = 30 * 1000;       // 30 seconds
const IDEMPOTENCY_WINDOW_MS = 10 * 60 * 1000; // 10 minutes

export async function lineWebhookGuard(
  req: Request,
  _res: Response,
  next: NextFunction
) {
  const sig = req.get('X-Line-Signature');
  const rawBody = (req as any).rawBody;
  if (!sig || !rawBody) {
    return next(new Error('missing-signature-or-body'));
  }

  // 1) Verify signature immediately
  if (!verifyLineSignature(rawBody, sig)) {
    return next(new Error('invalid-signature'));
  }

  const body = req.body;
  const events = body.events;
  if (!Array.isArray(events) || events.length === 0) return next();

  // 2) Basic event freshness check (use LINE timestamp or createdTime)
  const now = Date.now();
  for (const ev of events) {
    const ts = Number(ev.timestamp || ev.createdTime);
    if (ts && Math.abs(now - ts) > MAX_EVENT_DRIFT_MS) {
      return next(new Error('stale-event'));
    }
  }

  // 3) Acquire DB client and start tx for idempotency + atomic apply
  const client = await getClient();
  try {
    await client.query('BEGIN');

    for (const ev of events) {
      // Idempotency key: prefer LINE event ID; fallback to signature+hash
      const lineEventId = ev.id || '';
      const dedupKey = lineEventId
        ? `line:${lineEventId}`
        : crypto.createHash('sha256').update(sig + ':' + JSON.stringify(ev)).digest('hex');

      const dup = await client.query(
        `SELECT 1 FROM line_webhook_events
         WHERE line_event_id = $1 OR (signature = $2 AND received_at > NOW() - ($3::text || ' ms')::interval)`,
        [lineEventId, sig, IDEMPOTENCY_WINDOW_MS]
      );

      if (dup.rows.length > 0) {
        await client.query('ROLLBACK');
        client.release();
        // Idempotent acknowledgement
        return;
      }

      // Store event record (processed_at set later)
      await client.query(
        `INSERT INTO line_webhook_events (line_event_id, signature, user_id, event_type, action, effective_at, payload)
         VALUES ($1, $2, $3, $4, $5, $6, $7)`,
        [
          lineEventId || crypto.randomUUID(),
          sig,
          ev.source?.userId || '',
          ev.type || '',
          classifyActionFromEvent(ev),
          new Date(Number(ev.timestamp || ev.createdTime || now)),
          ev
        ]
      );
    }

    (req as any).pgClient = client;
    next();
  } catch (err) {
    await client.query('ROLLBACK').catch(() => {});
    client.release();
    next(err);
  }
}

function classifyActionFromEvent(ev: any): string {
  if (ev.type === 'message' && ev.message?.type === 'text') {
    const t = ev.message.text.trim().toLowerCase();
    if (t.includes('in') || t.includes('clock')) return 'clock-in';
    if (t.includes('out') || t.includes('clock')) return 'clock-out';
    if (t.includes('leave')) return 'leave';
    if (t.includes('ot') || t.includes('overtime')) return 'ot';
  }
  return 'unknown';
}

function verifyLineSignature(rawBody: Buffer | string, signature: string): boolean {
  const expected = 'v1=' + crypto
    .createHmac('sha256', process.env.LINE_CHANNEL_SECRET || '')
    .update(rawBody)
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}
```
