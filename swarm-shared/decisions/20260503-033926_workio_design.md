# workio / design

## Final Synthesis — Corrected, Prioritized, Actionable

**Core problem**: LINE webhooks lack idempotency, signature verification, tenant isolation, and transactional safety, causing duplicates, spoofing, partial writes, and cross-tenant bleed.

**Resolution priorities**:
1. Prevent duplicates and replay (idempotency + signature).
2. Enforce tenant isolation and auditability.
3. Make processing safe (transactions) and fast (non-blocking 200 OK).
4. Keep changes minimal, reversible, and observable.

### 1. Database — append to `server/src/db/schema.sql`

```sql
-- Idempotency + audit for LINE webhooks
CREATE TABLE IF NOT EXISTS line_webhook_events (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id       UUID NOT NULL REFERENCES tenants(id) ON DELETE CASCADE,
  line_delivery_id TEXT NOT NULL,            -- X-Line-Delivery-Id or event.id
  event_type      TEXT NOT NULL,             -- CLOCK_IN, LEAVE_REQUEST, OT_REQUEST, etc.
  payload_hash    BYTEA NOT NULL,            -- SHA256 of normalized body for replay detection
  processed_at    TIMESTAMPTZ,
  created_at      TIMESTAMPTZ NOT NULL DEFAULT now(),
  UNIQUE (tenant_id, line_delivery_id)
);

-- Fast idempotency lookup
CREATE INDEX IF NOT EXISTS idx_line_webhook_events_lookup
  ON line_webhook_events (tenant_id, line_delivery_id)
  WHERE processed_at IS NOT NULL;

-- Optional: index for replay detection by hash (if you want cross-tenant duplicate detection)
CREATE INDEX IF NOT EXISTS idx_line_webhook_events_hash
  ON line_webhook_events (payload_hash)
  WHERE processed_at IS NOT NULL;
```

### 2. Middleware — `middleware/lineAuth.ts`

```ts
import crypto from 'crypto';

const LINE_CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';

export function verifyLineSignature(rawBody: string, signature: string): boolean {
  if (!LINE_CHANNEL_SECRET) return false;
  const expected = crypto
    .createHmac('sha256', LINE_CHANNEL_SECRET)
    .update(rawBody, 'utf8')
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

export function lineAuth(req, res, next) {
  const sig = req.headers['x-line-signature'] as string;
  if (!sig) return res.status(401).json({ error: 'Missing X-Line-Signature' });

  const raw = (req as any).rawBody;
  if (!raw) return res.status(400).json({ error: 'Raw body missing' });

  if (!verifyLineSignature(raw, sig)) {
    return res.status(401).json({ error: 'Invalid signature' });
  }
  next();
}
```

### 3. Service — `services/lineEventService.ts`

```ts
import { Pool } from 'pg';
import { v4 as uuidv4 } from 'uuid';

const pool = new Pool({ connectionString: process.env.DATABASE_URL });

export async function processLineEvent(
  tenantId: string,
  deliveryId: string,
  eventType: string,
  payload: any,
  rawBody: string
) {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // Idempotency check
    const check = await client.query(
      `SELECT 1 FROM line_webhook_events
       WHERE tenant_id = $1 AND line_delivery_id = $2 AND processed_at IS NOT NULL`,
      [tenantId, deliveryId]
    );
    if (check.rowCount && check.rowCount > 0) {
      await client.query('ROLLBACK');
      return { alreadyProcessed: true };
    }

    const payloadHash = crypto.createHash('sha256').update(rawBody).digest();

    // Record event (mark processed_at immediately to prevent concurrent replays)
    await client.query(
      `INSERT INTO line_webhook_events
        (tenant_id, line_delivery_id, event_type, payload_hash, processed_at)
       VALUES ($1, $2, $3, $4, now())`,
      [tenantId, deliveryId, eventType, payloadHash]
    );

    // Domain processing (expand as needed)
    if (eventType === 'CLOCK_IN' || eventType === 'CLOCK_OUT') {
      await client.query(
        `INSERT INTO attendance_records (id, tenant_id, user_id, event_type, recorded_at, line_delivery_id)
         VALUES ($1, $2, $3, $4, $5, $6)
         ON CONFLICT (tenant_id, user_id, recorded_at::date, event_type)
         DO UPDATE SET line_delivery_id = EXCLUDED.line_delivery_id`,
        [uuidv4(), tenantId, payload.userId, eventType, payload.recordedAt, deliveryId]
      );
    } else if (eventType === 'LEAVE_REQUEST') {
      await client.query(
        `INSERT INTO leave_requests (id, tenant_id, user_id, leave_type, start_date, end_date, status, line_delivery_id)
         VALUES ($1, $2, $3, $4, $5, $6, 'PENDING', $7)`,
        [uuidv4(), tenantId, payload.userId, payload.leaveType, payload.startDate, payload.endDate, deliveryId]
      );
    } else if (eventType === 'OT_REQUEST') {
      await client.query(
        `INSERT INTO ot_requests (id, tenant_id, user_id, hours, work_date, reason, status, line_delivery_id)
         VALUES ($1, $2, $3, $4, $5, $6, 'PENDING', $7)`,
        [uuidv4(), tenantId, payload.userId, payload.hours, payload.workDate, payload.reason, deliveryId]
      );
    }

    await client.query('COMMIT');
    return { processed: true };
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  } finally {
    client.release();
  }
}

// Fire-and-forget wrapper to avoid blocking LINE's retry loop
export async function enqueueLineEvent(
  tenantId: string,
  deliveryId: string,
  eventType: string,
  payload: any,
  rawBody: string
) {
  // Fast path: record idempotency entry synchronously and defer domain work
  // For simplicity, we keep synchronous transactional processing but return quickly.
  // If you need true async, push to a queue (Bull/BullMQ/SQS) here.
  return processLineEvent(tenantId, deliveryId, eventType, payload, rawBody);
}
```
