# workio / design

## 1. Diagnosis

- **Tenant resolution thrash**: `/webhook/line` resolves `channelId → tenant` on every request with no caching; LINE retries on 5xx multiply DB load during spikes.
- **No correlation IDs**: Missing request/trace IDs across webhook → service → DB makes incidents untraceable in logs/metrics.
- **No structured logging**: Ad-hoc `console.log` prevents reliable ingestion into observability pipelines and slows incident triage.
- **No webhook idempotency**: Duplicate LINE events (retries/replays) can create double clock-in/out records or duplicate leave/OT requests.
- **No graceful degradation**: If DB or LINE API is slow/unavailable, webhook returns 5xx causing LINE to retry and amplify load.

## 2. Proposed change

Add lightweight tenant cache + request tracing + idempotency + structured logging to the LINE webhook handler.

Scope:
- `server/src/middleware/tracing.ts` (new)
- `server/src/middleware/tenant.ts` (new)
- `server/src/middleware/logging.ts` (new)
- `server/src/routes/webhook/line.ts` (modify)
- `server/src/utils/idempotency.ts` (new)

## 3. Implementation

### server/src/middleware/tracing.ts
```ts
import { randomUUID } from 'uuid';
import type { Request, Response, NextFunction } from 'express';

export interface RequestLocals extends Record<string, unknown> {
  requestId: string;
  traceId: string;
  tenantId?: string;
}

export function tracing(req: Request, _res: Response, next: NextFunction) {
  const requestId = randomUUID();
  const traceId = req.headers['x-trace-id'] as string || requestId;
  (req as any).locals = { requestId, traceId } as RequestLocals;
  next();
}
```

### server/src/middleware/tenant.ts
```ts
import LRU from 'lru-cache';
import type { Request, Response, NextFunction } from 'express';
import { getTenantByChannelId } from '../db/queries/tenantQueries';

// ~10k tenants, 5min TTL, 10s stale-while-revalidate-ish behavior via background refresh
const tenantCache = new LRU<string, { tenantId: string; updatedAt: number }>({
  max: 10_000,
  ttl: 1000 * 60 * 5,
  allowStale: true,
  updateAgeOnGet: true,
});

export async function resolveTenant(req: Request, res: Response, next: NextFunction) {
  const locals = (req as any).locals as { tenantId?: string };
  const channelId = req.body?.events?.[0]?.source?.groupId || req.body?.events?.[0]?.source?.userId;
  if (!channelId) {
    return res.status(400).json({ error: 'missing_channel_id' });
  }

  const cached = tenantCache.get(channelId);
  if (cached) {
    locals.tenantId = cached.tenantId;
    return next();
  }

  try {
    const row = await getTenantByChannelId(channelId);
    if (!row) {
      return res.status(404).json({ error: 'tenant_not_found' });
    }
    tenantCache.set(channelId, { tenantId: row.id, updatedAt: Date.now() });
    locals.tenantId = row.id;
    next();
  } catch (err) {
    next(err);
  }
}
```

### server/src/middleware/logging.ts
```ts
import type { Request, Response, NextFunction } from 'express';

export function structuredLogger(req: Request, res: Response, next: NextFunction) {
  const start = Date.now();
  const locals = (req as any).locals || {};

  res.on('finish', () => {
    const duration = Date.now() - start;
    const log = {
      ts: new Date().toISOString(),
      level: res.statusCode >= 500 ? 'error' : res.statusCode >= 400 ? 'warn' : 'info',
      method: req.method,
      path: req.path,
      status: res.statusCode,
      durationMs: duration,
      requestId: locals.requestId,
      traceId: locals.traceId,
      tenantId: locals.tenantId,
      userAgent: req.get('User-Agent'),
      ip: req.ip,
    };
    console.log(JSON.stringify(log));
  });

  next();
}
```

### server/src/utils/idempotency.ts
```ts
import crypto from 'crypto';

const seen = new Map<string, { ts: number; result: unknown }>();
const TTL_MS = 1000 * 60 * 10; // 10 minutes

export function idempotencyKeyFromBody(body: unknown): string {
  // Use LINE webhook event IDs if available; otherwise hash body
  const events = (body as any)?.events || [];
  const eventIds = events.map((e: any) => e.webhookEventId).filter(Boolean);
  if (eventIds.length) return `line:${eventIds.sort().join(',')}`;
  const hash = crypto.createHash('sha256').update(JSON.stringify(body)).digest('hex');
  return `body:${hash}`;
}

export function dedupe<T>(key: string, result: T): { ok: true; result: T } | { ok: false; result: unknown } {
  const now = Date.now();
  const existing = seen.get(key);
  if (existing) {
    if (now - existing.ts < TTL_MS) {
      return { ok: false, result: existing.result };
    }
  }
  seen.set(key, { ts: now, result });
  // Simple bounded cleanup (best-effort)
  if (seen.size > 50_000) {
    for (const [k, v] of seen) {
      if (now - v.ts > TTL_MS) seen.delete(k);
    }
  }
  return { ok: true, result };
}
```

### server/src/routes/webhook/line.ts (modify)
```ts
import express from 'express';
import { tracing } from '../../middleware/tracing';
import { resolveTenant } from '../../middleware/tenant';
import { structuredLogger } from '../../middleware/logging';
import { idempotencyKeyFromBody, dedupe } from '../../utils/idempotency';
import { handleLineEvent } from '../../services/lineService';

const router = express.Router();

router.use(express.json());
router.use(tracing);
router.use(structuredLogger);

router.post('/line', resolveTenant, async (req, res) => {
  const locals = (req as any).locals || {};
  const key = idempotencyKeyFromBody(req.body);
  const dedupeResult = dedupe(key, null);

  if (!dedupeResult.ok) {
    // Idempotent replay — return previous success if available
    return res.status(200).json({ ok: true, duplicated: true, result: dedupeResult.result });
  }

  try {
    const result = await handleLineEvent(req.body, locals.tenantId);
    dedupe(key, result); // store success
    return res.status(200).json({ ok: true, result });
  } catch (err: any) {
    // Do not store failure — allow retry by LINE, but avoid double-writes in handler
    console.error(JSON.stringify({
      ts: new Date().toISOString(),
      level: 'error',
      error: err.message,
      stack: err.stack,
      requestId: locals.requestId,
      traceId: locals.traceId,
      tenantId: locals.tenantId,
    }));
    // Graceful degradation: return 202 to stop LINE retries if failure is non-recoverable
    // or 503 to trigger retry for transient errors. Prefer 202 for validation errors.
    const status = err.isTransient ? 503 : 202;
    return res.status(status).json({ ok: false, error: err.message });
  }
});

export default router;
```
