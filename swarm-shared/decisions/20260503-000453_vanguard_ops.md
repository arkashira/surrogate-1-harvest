# vanguard / ops

## Final Consolidated Solution

**Core problem**: training wastes HF API quota, fails on heterogeneous schemas, and ignores Lightning resource reuse and resilience.

**Single source of truth**: a persisted, CDN-only manifest per `(repo, dateFolder)` plus a training path that never calls the HF API during training and reuses studios.

### 1. Ops layer: manifest generator

File: `/opt/axentx/vanguard/ops/generate_manifest.py`  
Folder: `/opt/axentx/vanguard/manifests/`

```python
#!/usr/bin/env python3
"""
Generate a CDN-only file manifest for a given HF dataset repo and date folder.
Usage:
    HF_TOKEN=hf_xxx python generate_manifest.py \
        --repo my-org/surrogate-1 \
        --date-folder 2026-05-03 \
        --out manifests/manifest-2026-05-03.json
"""
import argparse
import json
import os
import sys
from datetime import datetime, timezone
from pathlib import Path

from huggingface_hub import HfApi, login

API = HfApi()

def build_manifest(repo_id: str, date_folder: str, out_path: Path):
    # Single non-recursive folder list to minimize API calls
    entries = API.list_repo_tree(repo_id, path=date_folder, recursive=False)
    files = [e for e in entries if e.type == "file" and e.path.endswith(".parquet")]

    if not files:
        print(f"No parquet files found in {repo_id}/{date_folder}")
        sys.exit(1)

    manifest = {
        "repo_id": repo_id,
        "date_folder": date_folder,
        "generated_at": datetime.now(timezone.utc).isoformat(),
        "files": [
            {
                "path": f.path,
                "cdn_url": f"https://huggingface.co/datasets/{repo_id}/resolve/main/{f.path}"
            }
            for f in files
        ]
    }

    out_path.parent.mkdir(parents=True, exist_ok=True)
    with open(out_path, "w") as fp:
        json.dump(manifest, fp, indent=2)

    print(f"Manifest written to {out_path} ({len(files)} files)")
    return manifest

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Generate HF CDN manifest for date folder.")
    parser.add_argument("--repo", required=True, help="HF dataset repo id (org/repo)")
    parser.add_argument("--date-folder", required=True, help="Date folder inside dataset (e.g. 2026-05-03)")
    parser.add_argument("--out", default="manifests/manifest-latest.json", help="Output JSON path")
    args = parser.parse_args()

    token = os.getenv("HF_TOKEN")
    if token:
        login(token)

    build_manifest(args.repo, args.date_folder, Path(args.out))
```

### 2. Training launcher: CDN-only, studio reuse, idle resilience

File: `/opt/axentx/vanguard/ops/train_cdn_only.py`

```python
#!/usr/bin/env python3
"""
Lightning training job that:
- Reuses a running Studio if available
- Loads data via CDN URLs only (zero authenticated HF API calls during training)
- Projects heterogeneous parquet files to {prompt, response} at parse time
- Auto-restarts if studio was stopped (idle-timeout resilience)
"""
import json
import os
import sys
from pathlib import Path

import lightning as L
import pyarrow.parquet as pq
import torch
from torch.utils.data import Dataset, DataLoader

MANIFEST_PATH = Path(os.getenv("MANIFEST_PATH", "manifests/manifest-latest.json"))
BATCH_SIZE = int(os.getenv("BATCH_SIZE", "4"))
MAX_STEPS = int(os.getenv("MAX_STEPS", "100"))

class CDNParquetDataset(Dataset):
    """Read parquet files via CDN URLs; project to {prompt, response} on the fly."""
    def __init__(self, manifest_path: Path):
        with open(manifest_path) as f:
            manifest = json.load(f)
        self.urls = [item["cdn_url"] for item in manifest["files"]]

    def __len__(self):
        return len(self.urls)

    def __getitem__(self, idx):
        url = self.urls[idx]
        tbl = pq.read_table(
            url,
            columns=["prompt", "response"],
            use_threads=False,
            # Avoid schema mismatches by reading only requested columns when possible
        )
        # Safe projection for heterogeneous schemas
        prompt_col, response_col = None, None
        for c in ("prompt", "response"):
            if c in tbl.column_names:
                if c == "prompt":
                    prompt_col = c
                else:
                    response_col = c

        if prompt_col is None or response_col is None:
            # Fallback: pick first two text-ish columns
            text_cols = [
                c for c in tbl.column_names
                if str(tbl.schema.field(c).type).lower() in ("string", "large_string", "utf8")
            ]
            if len(text_cols) < 2:
                raise ValueError(f"Cannot project prompt/response from {tbl.column_names}")
            prompt_col, response_col = text_cols[0], text_cols[1]

        # Single-row example; scale via sharding/workers in production
        prompt = tbl[prompt_col][0].as_py()
        response = tbl[response_col][0].as_py()
        return {"prompt": prompt, "response": response}

class SimpleSurrogate(L.LightningModule):
    def __init__(self, lr=1e-4):
        super().__init__()
        self.lr = lr
        self.net = torch.nn.Linear(1024, 1024)  # placeholder

    def training_step(self, batch, batch_idx):
        # Replace with real model forward
        loss = torch.tensor(0.0, requires_grad=True)
        self.log("train_loss", loss)
        return loss

    def configure_optimizers(self):
        return torch.optim.Adam(self.parameters(), lr=self.lr)

def reuse_or_create_studio(job_name: str):
    teamspace = L.Teamspace()
    for s in teamspace.studios():
        if s.name == job_name and s.status == "running":
            print(f"Reusing running studio: {s.name}")
            return s
    print(f"Creating new studio: {job_name}")
    return L.Studio(
        name=job_name,
        machine=L.Machine.L40S,
        create_ok=True,
    )

def main():
    if not MANIFEST_PATH.exists():
        print(f"Manifest not found: {MANIFEST_PATH}")
        sys.exit(1)

    job_name = "vanguard-surrogate-train"
    studio = reuse_or_create_studio(job_name)

    # Idle-stop resilience: restart if stopped
    if studio.status != "running":
        print(f"Studio stopped ({studio.status}), restarting...")
        studio.start(machine=L.Machine.L40S)

    dataset = CDNParquetDataset(MANIFEST_PATH)
    loader = DataLoader(dataset, batch_size=BATCH_SIZE, shuffle=True)

    model = SimpleSurrogate()

    trainer = Trainer(
        max_steps=MAX_STEPS,
        logger=False,
        enable_checkpointing=False,
    )
    trainer.fit(model, loader)

if __name__ == "__main__":
    main()
```
