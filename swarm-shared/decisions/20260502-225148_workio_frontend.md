# workio / frontend

## 1. Diagnosis

- Clock-in/out events are buffered in `localStorage` (ephemeral, quota-limited, not cross-tab safe) and submitted optimistically; page reload or storage pressure loses events.
- No idempotency key on attendance mutations ŌåÆ retries (network blips, background-sync replay) can create duplicate attendance records on the backend.
- No durable offline queue: if the request fails while offline, the event is dropped and the user is not reliably prompted to retry.
- No transactional safety between local state and server: UI can show ŌĆ£clocked inŌĆØ while the server never recorded it, causing payroll/report mismatches.
- Missing retry/backoff and deduplication layer on the frontend; current flow likely fires once and forgets.

## 2. Proposed change

- **Scope**: frontend only ŌĆö `src/lib/attendance/` (new) + `src/hooks/useAttendanceQueue.ts` + minimal changes to clock-in/out UI components and API client.
- **Goal**: Replace fragile `localStorage` buffering with an IndexedDB-backed durable queue that attaches an idempotency key to every attendance mutation, retries with exponential backoff, and de-duplicates by idempotency key before sending.
- **Files**:
  - `src/lib/attendance/queue.ts` ŌĆö queue implementation (IndexedDB via `idb` or small wrapper).
  - `src/lib/attendance/types.ts` ŌĆö types + idempotency key generator.
  - `src/hooks/useAttendanceQueue.ts` ŌĆö React hook to enqueue and process.
  - `src/lib/api/attendance.ts` ŌĆö add `idempotencyKey` to request body and optional `X-Idempotency-Key` header.
  - Update clock-in/out UI to enqueue instead of direct optimistic POST.

### `src/lib/attendance/queue.ts`

```ts
import { openDB, DBSchema, IDBPDatabase } from 'idb';
import type { QueuedAttendance } from './types';

interface AttendanceDB extends DBSchema {
  queue: {
    key: number;
    value: QueuedAttendance;
    indexes: { 'by-idempotency': string; 'by-created': number };
  };
  processed: {
    key: string; // idempotencyKey
    value: { idempotencyKey: string; ts: number };
  };
}

const DB_NAME = 'workio-attendance';
const STORE_QUEUE = 'queue';
const STORE_PROCESSED = 'processed';

async function getDb(): Promise<IDBPDatabase<AttendanceDB>> {
  return openDB<AttendanceDB>(DB_NAME, 1, {
    upgrade(db) {
      const q = db.createObjectStore(STORE_QUEUE, { keyPath: 'id', autoIncrement: true });
      q.createIndex('by-idempotency', 'idempotencyKey', { unique: true });
      q.createIndex('by-created', 'createdAt');

      db.createObjectStore(STORE_PROCESSED, { keyPath: 'idempotencyKey' });
    },
  });
}

export async function enqueueAttendance(event: QueuedAttendance): Promise<void> {
  const db = await getDb();
  try {
    await db.add(STORE_QUEUE, event);
  } catch (err: any) {
    // Duplicate idempotencyKey ŌĆö ignore (already queued)
    if (err?.name === 'ConstraintError') return;
    throw err;
  }
}

export async function* peekQueue(batchSize = 10): AsyncGenerator<QueuedAttendance[]> {
  const db = await getDb();
  const tx = db.transaction(STORE_QUEUE, 'readonly');
  const idx = tx.store.index('by-created');
  const all = await idx.getAll();
  for (let i = 0; i < all.length; i += batchSize) {
    yield all.slice(i, i + batchSize);
  }
}

export async function removeFromQueue(id: number): Promise<void> {
  const db = await getDb();
  await db.delete(STORE_QUEUE, id);
}

export async function markProcessed(idempotencyKey: string): Promise<void> {
  const db = await getDb();
  await db.put(STORE_PROCESSED, { idempotencyKey, ts: Date.now() });
}

export async function isAlreadyProcessed(idempotencyKey: string): Promise<boolean> {
  const db = await getDb();
  const rec = await db.get(STORE_PROCESSED, idempotencyKey);
  return !!rec;
}
```

### `src/lib/attendance/idempotency.ts`

```ts
export function generateIdempotencyKey(action: string, userId: string, ts: number): string {
  // Deterministic but unique per action+user+timestamp(ms). Collisions astronomically unlikely.
  // Use a short hash-like representation (base64url of 16 bytes random + ts).
  const buf = new Uint8Array(16);
  crypto.getRandomValues(buf);
  const rand = btoa(String.fromCharCode(...buf)).replace(/\+/g, '-').replace(/\//g, '_').replace(/=+$/, '');
  return `${userId}-${action}-${ts}-${rand}`;
}
```

### `src/lib/api/attendance.ts`

```ts
import type { AttendanceEvent } from '../attendance/types';

export async function submitAttendance(event: AttendanceEvent & { userId: string }) {
  // Body carries idempotencyKey for backend dedupe; header optional for proxy/gateway dedupe.
  const res = await fetch('/api/attendance', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'X-Idempotency-Key': event.idempotencyKey,
    },
    body: JSON.stringify({
      action: event.action,
      timestamp: event.timestamp,
      location: event.location,
      idempotencyKey: event.idempotencyKey,
      userId: event.userId,
    }),
  });

  if (!res.ok) {
    const err = new Error('Attendance submission failed');
    (err as any).status = res.status;
    throw err;
  }

  return res.json();
}
```
