# workio / quality

## Final consolidated implementation (best of both proposals)

**Core problem**: LINE retries + concurrent deliveries create duplicate punches and inconsistent state.  
**Solution**: strong idempotency + serialized punch transitions + transactional integrity.

---

### 1. DB schema (run once)

```sql
-- Idempotency for LINE webhook events
CREATE TABLE IF NOT EXISTS line_webhook_events (
  event_id        TEXT PRIMARY KEY,
  tenant_id       TEXT NOT NULL,
  processed_at    TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  payload_hash    TEXT NOT NULL
);

CREATE INDEX IF NOT EXISTS idx_line_webhook_events_tenant
  ON line_webhook_events(tenant_id, processed_at);

-- Generic webhook delivery idempotency (recommended for other providers)
CREATE TABLE IF NOT EXISTS webhook_deliveries (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id       TEXT NOT NULL,
  msg_id          TEXT NOT NULL,
  event_type      TEXT NOT NULL,
  processed_at    TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  CONSTRAINT uq_webhook_delivery UNIQUE (tenant_id, msg_id, event_type)
);

-- Prevent multiple open punches per user/tenant
CREATE UNIQUE INDEX IF NOT EXISTS idx_open_punch
  ON punches (user_id, tenant_id, DATE(clock_in_at))
  WHERE clock_out_at IS NULL;
```

### 2. Idempotent, transactional handler

```ts
// server/src/routes/webhook/line.ts
import { Pool } from 'pg';
import crypto from 'crypto';

const pool = new Pool({ connectionString: process.env.DATABASE_URL });

type LineWebhookBody = {
  destination: string;
  events: Array<{
    type: string;
    timestamp: number;
    source: { userId: string };
    webhookEventId: string;
    things?: { result?: string };
  }>;
};

function hashPayload(obj: unknown): string {
  return crypto.createHash('sha256').update(JSON.stringify(obj)).digest('hex');
}

export async function lineWebhookHandler(req: any, res: any) {
  const body: LineWebhookBody = req.body;
  if (!body?.events?.length) return res.status(400).send('No events');

  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    for (const ev of body.events) {
      // Idempotency: skip already-processed LINE events
      const exists = await client.query(
        `SELECT 1 FROM line_webhook_events WHERE event_id = $1`,
        [ev.webhookEventId]
      );
      if (exists.rows.length > 0) continue;

      // Record event first (makes retries safe)
      await client.query(
        `INSERT INTO line_webhook_events(event_id, tenant_id, payload_hash)
         VALUES ($1, $2, $3)`,
        [ev.webhookEventId, body.destination, hashPayload(ev)]
      );

      // Also record in generic delivery table for cross-provider safety
      await client.query(
        `INSERT INTO webhook_deliveries(tenant_id, msg_id, event_type)
         VALUES ($1, $2, $3)
         ON CONFLICT (tenant_id, msg_id, event_type) DO NOTHING`,
        [body.destination, ev.webhookEventId, `line.${ev.type}`]
      );

      // Process clock events only
      if (ev.type === 'things' && ev.things?.result === 'clock') {
        const userId = ev.source.userId;
        const tenantId = body.destination;
        const ts = new Date(ev.timestamp);

        // Advisory lock to serialize concurrent deliveries for this user/tenant
        const lockId = parseInt(
          crypto.createHash('md5').update(`${tenantId}:${userId}`).digest('hex').slice(0, 8),
          16
        );
        await client.query('SELECT pg_advisory_xact_lock($1)', [lockId]);

        // Lock latest punch row to prevent race
        const latest = await client.query(
          `SELECT id, clock_out_at FROM punches
           WHERE tenant_id = $1 AND user_id = $2
           ORDER BY clock_in_at DESC LIMIT 1
           FOR UPDATE`,
          [tenantId, userId]
        );

        const last = latest.rows[0];
        if (last && !last.clock_out_at) {
          // Close open punch
          await client.query(
            `UPDATE punches
             SET clock_out_at = $1, updated_at = NOW()
             WHERE id = $2`,
            [ts, last.id]
          );
        } else {
          // Create new punch (partial unique index prevents double-open)
          await client.query(
            `INSERT INTO punches(tenant_id, user_id, clock_in_at, created_at, updated_at)
             VALUES ($1, $2, $3, NOW(), NOW())`,
            [tenantId, userId, ts]
          );
        }
      }
    }

    await client.query('COMMIT');
    res.status(200).send('OK');
  } catch (err: any) {
    await client.query('ROLLBACK');
    console.error('LINE webhook processing failed', err);
    // Return 5xx so LINE retries (idempotency prevents duplicates)
    res.status(500).send('Processing error');
  } finally {
    client.release();
  }
}
```

### 3. Route wiring

// server/src/routes/webhook/index.ts
import express from 'express';
import { lineWebhookHandler } from './line';

const router = express.Router();
router.post('/line', express.raw({ type: 'application/json' }), (req, res, next) => {
  // Optional: verify LINE signature here
  lineWebhookHandler(req, res).catch(next);
});

### 4. Operational safeguards

- **Retention**: prune `line_webhook_events` older than 30 days.  
  ```sql
  DELETE FROM line_webhook_events WHERE processed_at < NOW() - INTERVAL '30 days';
  ```
- **Side effects**: keep notifications/shift validation **outside** this transaction (enqueue after commit) to avoid long-held locks and transaction bloat.
- **Signature verification**: add `X-Line-Signature` check before idempotency if not already present.

### 5. Verification checklist

1. **Schema applied**  
   ```bash
   psql "$DATABASE_URL" -c "\d line_webhook_events"
   psql "$DATABASE_URL" -c "\d webhook_deliveries"
   ```

2. **Idempotency**  
   Send same LINE event 3× → exactly one punch row created/updated and one `line_webhook_events` row.

3. **Concurrency**  
   Fire parallel requests for same user/tenant → no duplicate open punches:
   ```sql
   SELECT user_id, count(*) FROM punches
   WHERE clock_out_at IS NULL GROUP BY user_id HAVING count(*) > 1;
   ```
   Must return zero rows.

4. **Retry safety**  
   Return 500 once, then 200 → exactly one state transition and one `line_webhook_events` record.
