# workio / quality

## Final consolidated implementation (best of both proposals)

### 1. Diagnosis (merged)
- Missing idempotency key on `/webhook/line` allows LINE retries (network blips, 5xx, slow client) to create duplicate punches.
- No partial unique index enforcing “one open punch per user” (`clock_out_at IS NULL`) permits corrupt state on races/retries.
- Punch write path is non-transactional and non-atomic (read-then-insert) → race between duplicate deliveries can insert multiple open rows.
- No idempotency table to deduplicate by LINE `deliveryId` (preferred) or composite `mode+timestamp+userId` within a safe TTL.
- No DB-level constraint or safe upsert path to collapse duplicates into a single open punch.

### 2. Scope
- `workio/server/src/db/schema.sql` (schema + constraints)
- `workio/server/src/db/queries/punches.ts` (idempotent helper)
- `workio/server/src/controllers/lineWebhook.ts` (transactional handler)
- Optional: scheduled cleanup for idempotency TTL

### 3. Implementation

#### 3.1 DB schema — constraints and idempotency table
```sql
-- workio/server/src/db/schema.sql

-- Idempotency table for LINE webhook events
CREATE TABLE IF NOT EXISTS webhook_idempotency (
  id          SERIAL PRIMARY KEY,
  key         TEXT NOT NULL,               -- e.g. line:delivery:<deliveryId>
  handler     TEXT NOT NULL,               -- e.g. 'line:punch'
  created_at  TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  UNIQUE (key, handler)
);

CREATE INDEX IF NOT EXISTS idx_webhook_idempotency_key_handler ON webhook_idempotency (key, handler);
CREATE INDEX IF NOT EXISTS idx_webhook_idempotency_created_at ON webhook_idempotency (created_at);

-- Ensure only one open punch per user (clock_out_at IS NULL)
-- Drop duplicates first if any exist (keeps earliest by id)
-- Run this once before creating the index:
--   DELETE FROM punches a
--   USING punches b
--   WHERE a.id > b.id
--     AND a.user_id = b.user_id
--     AND a.clock_out_at IS NULL
--     AND b.clock_out_at IS NULL;

CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_one_open_per_user
  ON punches (user_id)
  WHERE clock_out_at IS NULL;
```

#### 3.2 Idempotent punch creation helper (transaction-safe)
```ts
// workio/server/src/db/queries/punches.ts
import { PoolClient, QueryResult } from 'pg';

export interface PunchRow {
  id: number;
  user_id: number;
  clock_in_at: Date;
  clock_out_at: Date | null;
  latitude: number | null;
  longitude: number | null;
  created_at: Date;
}

export async function createPunchWithIdempotency(
  client: PoolClient,
  {
    userId,
    clockInAt,
    latitude,
    longitude,
    lineDeliveryId,
    lineMode,
    lineTimestamp,
  }: {
    userId: number;
    clockInAt: Date;
    latitude?: number | null;
    longitude?: number | null;
    lineDeliveryId?: string | null;
    lineMode?: string | null;
    lineTimestamp?: string | null;
  }
): Promise<PunchRow | null> {
  // Prefer deliveryId for idempotency (stable per event). Fallback to composite key.
  const idempotencyKey = lineDeliveryId
    ? `line:delivery:${lineDeliveryId}`
    : lineMode && lineTimestamp
    ? `line:event:${lineMode}:${lineTimestamp}:${userId}`
    : null;

  // If idempotency key present, check and skip duplicate within same transaction
  if (idempotencyKey) {
    const check = await client.query(
      `SELECT 1 FROM webhook_idempotency WHERE key = $1 AND handler = 'line:punch'`,
      [idempotencyKey]
    );
    if (check.rows.length > 0) {
      // Duplicate event: return existing open punch if present, else latest punch
      const existing = await client.query<PunchRow>(
        `SELECT * FROM punches WHERE user_id = $1 ORDER BY id DESC LIMIT 1`,
        [userId]
      );
      return existing.rows[0] || null;
    }
  }

  // Insert idempotency record first (if key present) so concurrent duplicate attempts
  // will be caught by unique constraint when they try to insert the same key.
  if (idempotencyKey) {
    await client.query(
      `INSERT INTO webhook_idempotency (key, handler) VALUES ($1, 'line:punch')`,
      [idempotencyKey]
    );
  }

  // Insert punch. If another open punch exists, unique partial index will raise 23505.
  const result = await client.query<PunchRow>(
    `INSERT INTO punches (user_id, clock_in_at, latitude, longitude)
     VALUES ($1, $2, $3, $4)
     RETURNING *`,
    [userId, clockInAt, latitude || null, longitude || null]
  );

  return result.rows[0];
}
```

#### 3.3 Transactional LINE webhook handler
```ts
// workio/server/src/controllers/lineWebhook.ts
import { pool } from '../db/index.js';
import { createPunchWithIdempotency } from '../db/queries/punches.js';

export async function handleLineWebhook(req, res) {
  const events = req.body.events || [];
  const client = await pool.connect();

  try {
    await client.query('BEGIN');

    for (const ev of events) {
      if (ev.type === 'message' && ev.message?.type === 'text') {
        // Resolve user by LINE user ID
        const user = await client.query(
          `SELECT id FROM users WHERE line_user_id = $1`,
          [ev.source.userId]
        );

        if (user.rows.length === 0) {
          // Optionally auto-provision or reject. Here we skip unknown users.
          continue;
        }

        const userId = user.rows[0].id;
        const now = new Date();

        await createPunchWithIdempotency(client, {
          userId,
          clockInAt: now,
          lineDeliveryId: ev.deliveryId,
          lineMode: ev.mode,
          lineTimestamp: ev.timestamp,
        });
      }
    }

    await client.query('COMMIT');
    res.status(200).json({ ok: true });
  } catch (err: any) {
    await client.query('ROLLBACK').catch(() => {});
    console.error('LINE webhook tx failed', err);

    // Unique violation from idx_punches_one_open_per_user -> conflict
    if (err.code === '23505') {
      return res.status(409).json({ error: 'already_clocked_in' });
    }

    // Idempotency unique violation should be handled above, but guard anyway
    res.status(500).json({ error: 'internal' });
  } finally {
    client.release();
  }
}
```
