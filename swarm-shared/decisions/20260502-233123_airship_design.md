## Final Synthesized Implementation  
*(Best parts of both proposals; contradictions resolved in favor of correctness + concrete actionability)*

---

### 1) Highest-Value Improvement (≤2h)
**Goal**: Make `airship discover` a **deterministic, CDN-only orchestrator** that  
- eliminates HF API rate limits  
- eliminates PyArrow schema errors  
- produces reproducible, versioned file lists  
- enables immediate surrogate training without quota penalties  

**Why this ships fastest**  
- Pure orchestration (no model/training changes)  
- Uses existing patterns (HF CDN bypass, pre-generated manifests, schema projection)  
- Single script + one small Docker/compose tweak  
- Immediate payoff for downstream surrogate runs and Lightning Studio quota savings

#### `airship/discover.py`
```python
#!/usr/bin/env python3
"""
CDN-only discovery for HF datasets.
Pre-generates file manifest to avoid HF API rate limits during training.
"""
import hashlib
import json
import os
import sys
from datetime import datetime, timezone
from pathlib import Path
from typing import Dict, List

from huggingface_hub import HfApi

# --
# Env-driven config (CI/local parity)
# --
HF_REPO = os.getenv("HF_DATASET_REPO", "elinas/airship-dataset")
DATE_FOLDER = os.getenv("HF_DATE_FOLDER", datetime.now(timezone.utc).strftime("%Y-%m-%d"))
OUTPUT_DIR = Path(os.getenv("MANIFEST_DIR", "manifests"))
OUTPUT_DIR.mkdir(parents=True, exist_ok=True)

# --
# CDN manifest (single API call per folder, non-recursive)
# --
def build_cdn_manifest(repo: str, date_folder: str) -> Dict:
    api = HfApi()
    base_path = date_folder.rstrip("/")

    # Non-recursive folder listing
    tree = api.list_repo_tree(repo=repo, path=base_path, recursive=False)

    files: List[Dict] = []
    for entry in tree:
        if entry.type != "file":
            continue

        cdn_url = (
            f"https://huggingface.co/datasets/{repo}/resolve/main/"
            f"{base_path}/{entry.path}"
        )

        files.append(
            {
                "repo": repo,
                "path": f"{base_path}/{entry.path}",
                "cdn_url": cdn_url,
                "size": getattr(entry, "size", None),
                "lfs": getattr(entry, "lfs", None),
            }
        )

    manifest = {
        "repo": repo,
        "date_folder": date_folder,
        "generated_at_utc": datetime.now(timezone.utc).isoformat(),
        "strategy": "cdn-only",
        "file_count": len(files),
        "files": files,
    }
    return manifest

# --
# Schema-safe projection helper
# --
def project_prompt_response_local(file_path: Path) -> List[Dict]:
    """
    Lightweight projection to {prompt, response} only.
    Avoids load_dataset() on heterogeneous repos and PyArrow CastError.
    Supports .jsonl (primary), .json, .parquet (via pyarrow if available).
    """
    suffix = file_path.suffix.lower()
    records = []

    # JSONL (fast, line-by-line)
    if suffix == ".jsonl":
        with file_path.open("r", encoding="utf-8") as f:
            for line in f:
                line = line.strip()
                if not line:
                    continue
                try:
                    obj = json.loads(line)
                    records.append(
                        {
                            "prompt": obj.get("prompt") or obj.get("input") or "",
                            "response": obj.get("response") or obj.get("output") or "",
                        }
                    )
                except Exception:
                    continue
        return records

    # JSON (single object or list)
    if suffix == ".json":
        try:
            with file_path.open("r", encoding="utf-8") as f:
                data = json.load(f)
            if isinstance(data, list):
                items = data
            else:
                items = [data]
            for obj in items:
                records.append(
                    {
                        "prompt": obj.get("prompt") or obj.get("input") or "",
                        "response": obj.get("response") or obj.get("output") or "",
                    }
                )
        except Exception:
            pass
        return records

    # Parquet (optional, lightweight projection)
    if suffix == ".parquet":
        try:
            import pyarrow.parquet as pq
            table = pq.read_table(file_path, columns=["prompt", "response", "input", "output"])
            df = table.to_pandas()
            for _, row in df.iterrows():
                records.append(
                    {
                        "prompt": row.get("prompt") or row.get("input") or "",
                        "response": row.get("response") or row.get("output") or "",
                    }
                )
        except Exception:
            pass
        return records

    # Fallback: unsupported format
    return []

# --
# CLI
# --
def main() -> None:
    manifest = build_cdn_manifest(HF_REPO, DATE_FOLDER)

    manifest_path = OUTPUT_DIR / f"manifest-{DATE_FOLDER}.json"
    manifest_path.write_text(json.dumps(manifest, indent=2), encoding="utf-8")

    # SHA-256 for integrity checks
    sha = hashlib.sha256(manifest_path.read_bytes()).hexdigest()
    (OUTPUT_DIR / f"manifest-{DATE_FOLDER}.sha256").write_text(
        f"{sha}  {manifest_path.name}\n", encoding="utf-8"
    )

    print(f"Manifest: {manifest_path}")
    print(f"SHA256: {sha}")
    print(f"Files: {manifest['file_count']}")
    print("Strategy: cdn-only (zero HF API calls during training)")

if __name__ == "__main__":
    # Fail fast on import/validation errors in cron
    try:
        main()
    except Exception as exc:
        print(f"airship discover failed: {exc}", file=sys.stderr)
        sys.exit(1)
```
