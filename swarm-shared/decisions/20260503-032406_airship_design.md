```python
# surrogate/training/train.py
import json
import os
import sys
import signal
from pathlib import Path
from typing import List, Dict, Optional

import torch
from torch.utils.data import IterableDataset, DataLoader
import requests
import pyarrow.parquet as pq
import tempfile

# --
# CDN-only dataset (zero HF API calls during training)
# --
class CDNParquetDataset(IterableDataset):
    """
    Manifest produced on Mac before training:
    {
      "repo": "datasets/username/repo",
      "date_folder": "2026-04-29",
      "files": [
        {"path": "batches/mirror-merged/2026-04-29/slug1.parquet", "size": 12345},
        ...
      ]
    }
    """
    def __init__(
        self,
        manifest_path: str,
        columns: List[str] = ("prompt", "response"),
        shard_rank: int = 0,
        shard_world_size: int = 1,
        shuffle: bool = False,
        seed: int = 42,
    ):
        manifest = json.loads(Path(manifest_path).read_text())
        files = [f["path"] for f in manifest.get("files", []) if f.get("path")]
        if not files:
            raise ValueError("No files found in manifest")
        # deterministic sharding by file index
        files = sorted(files)
        files = [f for i, f in enumerate(files) if i % shard_world_size == shard_rank]
        if shuffle:
            import random
            random.Random(seed).shuffle(files)
        self.files = files
        self.repo = manifest["repo"]
        self.columns = columns
        self.shuffle_files = shuffle

    def _stream_file(self, file_path: str):
        url = f"https://huggingface.co/datasets/{self.repo}/resolve/main/{file_path}"
        with tempfile.NamedTemporaryFile(suffix=".parquet", delete=False) as tmp:
            tmp_path = tmp.name
            try:
                with requests.get(url, stream=True, timeout=(10, 120)) as r:
                    r.raise_for_status()
                    for chunk in r.iter_content(chunk_size=16384):
                        tmp.write(chunk)
                table = pq.read_table(tmp_path, columns=self.columns)
                for batch in table.to_batches(max_chunksize=1024):
                    df = batch.to_pydict()
                    keys = [c for c in self.columns if c in df]
                    n = len(df[keys[0]]) if keys else 0
                    for i in range(n):
                        row = {c: df[c][i] for c in keys}
                        if all(row.get(c) is not None for c in self.columns):
                            yield row
            except Exception as exc:
                print(f"[WARN] Failed to stream {file_path}: {exc}", file=sys.stderr)
            finally:
                try:
                    os.unlink(tmp_path)
                except Exception:
                    pass

    def __iter__(self):
        for file_path in self.files:
            yield from self._stream_file(file_path)

# --
# Checkpointing + graceful stop handling
# --
class CheckpointManager:
    def __init__(self, ckpt_dir: str = "checkpoint", save_every: int = 500):
        self.ckpt_dir = Path(ckpt_dir)
        self.ckpt_dir.mkdir(exist_ok=True, parents=True)
        self.save_every = save_every
        self.step = self._load_step()

    def _load_step(self) -> int:
        marker = self.ckpt_dir / "step.txt"
        if marker.exists():
            try:
                return int(marker.read_text().strip())
            except Exception:
                return 0
        return 0

    def save(self, model, tokenizer, step: int):
        self.ckpt_dir.joinpath("step.txt").write_text(str(step))
        model.save_pretrained(self.ckpt_dir)
        tokenizer.save_pretrained(self.ckpt_dir)
        print(f"[INFO] Checkpoint saved at step={step}")

# --
# Training loop (idle-resilient, checkpoint-aware)
# --
def run_training_step(
    model,
    dataloader,
    device,
    max_steps: int = 1000,
    ckpt_manager: Optional[CheckpointManager] = None,
):
    model.train()
    optimizer = torch.optim.AdamW(model.parameters(), lr=5e-5)
    step = ckpt_manager.step if ckpt_manager else 0
    stop_requested = False

    def _handle_stop(signum, frame):
        nonlocal stop_requested
        stop_requested = True
        print("[INFO] Stop signal received; will checkpoint and exit.")

    signal.signal(signal.SIGTERM, _handle_stop)
    signal.signal(signal.SIGINT, _handle_stop)

    for batch in dataloader:
        if stop_requested or step >= max_steps:
            break
        try:
            input_ids = batch["input_ids"].to(device)
            labels = batch["labels"].to(device)
            outputs = model(input_ids=input_ids, labels=labels)
            loss = outputs.loss
            loss.backward()
            optimizer.step()
            optimizer.zero_grad()
            step += 1
            if step % 10 == 0:
                print(f"step={step} loss={loss.item():.4f}")
            if ckpt_manager and step % ckpt_manager.save_every == 0:
                ckpt_manager.save(model, tokenizer, step)
        except Exception as exc:
            print(f"[WARN] Training step failed (will retry): {exc}", file=sys.stderr)
            continue

    if ckpt_manager:
        ckpt_manager.save(model, tokenizer, step)
    return step

# --
# Entrypoint used by Lightning launcher
# --
def main():
    manifest = os.getenv("CDN_MANIFEST_PATH", "manifest.json")
    if not Path(manifest).exists():
        raise FileNotFoundError(
            f"Manifest not found: {manifest}. Run on Mac before training: "
            "python scripts/build_cdn_manifest.py --date 2026-04-29 --output manifest.json"
        )

    rank = int(os.getenv("LOCAL_RANK", 0))
    world_size = int(os.getenv("WORLD_SIZE", 1))
    dataset = CDNParquetDataset(
        manifest,
        columns=["prompt", "response"],
        shard_rank=rank,
        shard_world_size=world_size,
        shuffle=True,
    )
    loader = DataLoader(dataset, batch_size=8, num_workers=0)

    from transformers import AutoTokenizer, AutoModelForCausalLM
    model_name = os.getenv("MODEL_NAME", "Qwen/Qwen2.5-0.5B-Instruct")
    tokenizer = AutoTokenizer.from_pretrained(model_name)
    model = AutoModelForCausalLM.from_pretrained(model_name, torch_dtype=torch.bfloat16)
    device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
    model.to(device)

    ckpt_dir = os.getenv("CHECKPOINT_DIR", "checkpoint")
    ckpt_manager = CheckpointManager(ckpt_dir=ckpt_dir, save_every=500)
    max_steps = int(os.getenv("MAX_STEPS_PER_RUN", 500))
    final_step = run_training_step(model, loader, device, max_steps=max_steps, ckpt_manager=ckpt_manager)
    print(f"Training finished at step={final_step}")

if __name__ == "__main__":
    main()
```
