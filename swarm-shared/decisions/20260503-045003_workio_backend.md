# workio / backend

## 1. Diagnosis
- LINE webhook ingestion has **no replay/idempotency guard** — duplicate `X-Line-Signature` events (network retries, client double-taps) can double-apply clock-in/out and leave/OT transitions, corrupting daily totals and approval state.
- Missing **strict signature verification** (timing-safe compare) and short replay window increases risk of forged or replayed webhooks.
- Clock-in/out and leave/OT handlers lack **transactional atomicity** for state transitions (race conditions on concurrent events for same user/day).
- No **idempotency key storage** (event ID + user + date) to deduplicate within a safe window (e.g., 10–15 minutes).
- Logging/metrics around webhook processing are minimal, making detection of duplicates or anomalies hard in production.

## 2. Proposed change
- **Files**:  
  - `server/src/routes/lineWebhook.ts` (or equivalent) — add middleware for signature verification + idempotency.  
  - `server/src/services/clockService.ts` (or similar) — wrap clock-in/out in transaction with idempotency check.  
  - `server/src/services/leaveOtService.ts` — same for leave/OT transitions.  
  - `server/src/db/schema.sql` — add `webhook_events` table (idempotency store) if not present.  
- **Scope**: implement strict `X-Line-Signature` verification, short replay window (10–15m), and idempotency by `(line_event_id, user_id, event_date)` before applying state changes.

## 3. Implementation

### 3.1 DB: idempotency table (add to schema.sql)
```sql
-- Idempotency store for LINE webhook events
CREATE TABLE IF NOT EXISTS webhook_events (
  id              BIGSERIAL PRIMARY KEY,
  line_event_id   TEXT NOT NULL,
  user_id         TEXT NOT NULL,
  event_type      TEXT NOT NULL,          -- 'clock', 'leave', 'ot'
  event_date      DATE NOT NULL,          -- logical date of action (UTC)
  payload_hash    TEXT NOT NULL,          -- sha256 of normalized payload for change detection
  created_at      TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(line_event_id, user_id, event_type, event_date)
);

-- Index for fast expiry / cleanup
CREATE INDEX IF NOT EXISTS idx_webhook_events_created ON webhook_events(created_at);
```

### 3.2 Middleware: signature verification + idempotency guard
`server/src/middleware/lineWebhookGuard.ts`
```ts
import crypto from 'crypto';
import { Request, Response, NextFunction } from 'express';
import { pool } from '../db';

const LINE_CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';
const REPLAY_WINDOW_MS = 15 * 60 * 1000; // 15 minutes

function verifySignature(rawBody: Buffer, signature: string): boolean {
  const expected = crypto
    .createHmac('sha256', LINE_CHANNEL_SECRET)
    .update(rawBody)
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

export async function lineWebhookGuard(
  req: Request,
  res: Response,
  next: NextFunction
) {
  const signature = req.get('X-Line-Signature');
  if (!signature) {
    return res.status(400).json({ error: 'Missing X-Line-Signature' });
  }

  const rawBody = req.body as Buffer;
  if (!verifySignature(rawBody, signature)) {
    return res.status(401).json({ error: 'Invalid signature' });
  }

  // Attach helpers to request for downstream handlers
  req.webhook = {
    verifySignature,
    isReplay: async (lineEventId: string, userId: string, eventType: string, eventDate: Date): Promise<boolean> => {
      const client = await pool.connect();
      try {
        const now = new Date();
        // Clean old entries outside replay window (best-effort)
        await client.query(
          `DELETE FROM webhook_events WHERE created_at < $1`,
          [new Date(now.getTime() - REPLAY_WINDOW_MS)]
        );

        const { rows } = await client.query(
          `SELECT 1 FROM webhook_events
           WHERE line_event_id = $1 AND user_id = $2 AND event_type = $3 AND event_date = $4
           LIMIT 1`,
          [lineEventId, userId, eventType, eventDate.toISOString().split('T')[0]]
        );
        return rows.length > 0;
      } finally {
        client.release();
      }
    },
    recordEvent: async (lineEventId: string, userId: string, eventType: string, eventDate: Date, payload: any): Promise<boolean> => {
      const client = await pool.connect();
      try {
        const payloadHash = crypto
          .createHash('sha256')
          .update(JSON.stringify(payload))
          .digest('hex');

        await client.query(
          `INSERT INTO webhook_events(line_event_id, user_id, event_type, event_date, payload_hash)
           VALUES($1,$2,$3,$4,$5)`,
          [lineEventId, userId, eventType, eventDate.toISOString().split('T')[0], payloadHash]
        );
        return true;
      } catch (err: any) {
        // Unique violation -> duplicate
        if (err.code === '23505') return false;
        throw err;
      } finally {
        client.release();
      }
    }
  };

  next();
}
```

### 3.3 Route: apply guard and idempotency before processing
`server/src/routes/lineWebhook.ts`
```ts
import { Router } from 'express';
import { lineWebhookGuard } from '../middleware/lineWebhookGuard';
import { handleClockEvent } from '../services/clockService';
import { handleLeaveOtEvent } from '../services/leaveOtService';

const router = Router();

// Ensure raw body is preserved for signature verification
router.post('/line', lineWebhookGuard, async (req, res) => {
  const events = req.body.events || [];
  const results = [];

  for (const ev of events) {
    try {
      const { type, source, message, ...rest } = ev;
      const userId = source?.userId;
      const lineEventId = ev?.eventId || ev?.webhookEventId || ev?.id;
      if (!userId || !lineEventId) {
        results.push({ eventId: lineEventId, status: 'skipped', reason: 'missing_user_or_event_id' });
        continue;
      }

      // Determine logical event date (use server date or message timestamp if provided)
      const eventDate = new Date(); // prefer ev.timestamp if available

      // Idempotency check
      const isReplay = await req.webhook.isReplay(lineEventId, userId, type, eventDate);
      if (isReplay) {
        results.push({ eventId: lineEventId, status: 'duplicate', reason: 'replay_detected' });
        continue;
      }

      // Route by type
      let ok = false;
      if (type === 'message' && message?.type === 'text') {
        // Example: text commands like "clock in", "leave", "ot"
        ok = await handleClockEvent(userId, message.text, eventDate, ev);
      } else if (type === 'postback') {
        // Could be leave/OT actions
        ok = await handleLeaveOtEvent(userId, ev.postback?.data, eventDate, ev);
      } else {
        results.push({ eventId: lineEventId, status: 'ignored', type });
        continue;
      }

      if (ok) {
        // Record only if processing succeeded
        await req.webhook.recordEvent(lineEventId, userId, type, eventDate, ev);
        results.push({ eventId: lineEventId, status: 'processed' });
      } else {
        results.push({ eventId: lineEventId, status: 'failed', reason: 'handler_error' });
      }
    } catch (err) {
      results.push({ eventId: ev?.eventId, status: 'error', error: String(err) });
    }
  }

  return res.json({ ok: true, results });
});

export default router;
```
