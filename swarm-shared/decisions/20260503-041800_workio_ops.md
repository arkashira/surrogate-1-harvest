# workio / ops

## 1. Diagnosis
- No idempotency guard on LINE webhook ingestion: retries, double-taps, or network replays create duplicate clock-in/out or leave/OT records.
- Missing request signature verification: LINE webhook events accepted without validating `X-Line-Signature`, enabling spoofed events.
- No per-tenant/per-day/per-employee intent guard: concurrent or replayed events can produce multiple attendance records for the same period.
- No graceful shutdown handling in the server: SIGTERM/SIGINT do not close DB/HTTP resources cleanly, risking connection leaks and partial writes.
- Ops observability gap: no structured request/response logging or health endpoint for deployment/runtime checks.

## 2. Proposed change
File scope: `/opt/axentx/workio/server/src/app.ts` (or `server.ts` if that’s the entry) and `/opt/axentx/workio/server/src/controllers/lineWebhook.ts` (create if absent). Add:
- Signature verification middleware for LINE.
- Idempotency guard (unique constraint + upsert) on attendance records keyed by `tenantId, employeeId, date, type` (clock-in/out) and similar for leave/OT requests.
- Graceful shutdown in app bootstrap.
- Lightweight health/metrics endpoint for ops.

### server/src/middleware/lineSignature.ts
```ts
import crypto from 'crypto';
import { Request, Response, NextFunction } from 'express';

const LINE_CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';

export function verifyLineSignature(req: Request, res: Response, next: NextFunction) {
  const signature = req.get('X-Line-Signature') || '';
  const body = JSON.stringify(req.body);
  const expected = crypto
    .createHmac('sha256', LINE_CHANNEL_SECRET)
    .update(body, 'utf8')
    .digest('base64');

  if (!crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected))) {
    return res.status(401).json({ error: 'Invalid signature' });
  }
  next();
}
```

### server/src/controllers/lineWebhook.ts
```ts
import { Request, Response } from 'express';
import { pool } from '../db';
import { verifyLineSignature } from '../middleware/lineSignature';

// Expose middleware for route binding
export const lineSignatureMiddleware = verifyLineSignature;

export async function handleLineWebhook(req: Request, res: Response) {
  const events = req.body.events || [];

  for (const ev of events) {
    if (ev.type === 'message' && ev.message.type === 'text') {
      await handleTextMessage(ev);
    }
    // extend for other event types as needed
  }

  res.json({ ok: true });
}

async function handleTextMessage(event: any) {
  const userId = event.source.userId;
  const text = event.message.text.trim().toLowerCase();
  const tenantId = process.env.DEFAULT_TENANT_ID; // resolve via mapping in prod

  // Resolve employee by LINE userId
  const emp = await pool.query(
    'SELECT id FROM employees WHERE tenant_id = $1 AND line_user_id = $2',
    [tenantId, userId]
  );

  if (!emp.rows.length) return;
  const employeeId = emp.rows[0].id;
  const today = new Date().toISOString().split('T')[0];

  if (text === 'clock in') {
    await upsertAttendance(tenantId, employeeId, today, 'clock_in');
  } else if (text === 'clock out') {
    await upsertAttendance(tenantId, employeeId, today, 'clock_out');
  }
  // extend for leave/OT commands
}

async function upsertAttendance(tenantId: string, employeeId: number, date: string, type: 'clock_in' | 'clock_out') {
  const now = new Date();
  await pool.query(
    `INSERT INTO attendance (tenant_id, employee_id, date, type, timestamp, created_at)
     VALUES ($1, $2, $3, $4, $5, $6)
     ON CONFLICT (tenant_id, employee_id, date, type)
     DO UPDATE SET timestamp = EXCLUDED.timestamp, updated_at = NOW()`,
    [tenantId, employeeId, date, type, now, now]
  );
}
```

### server/src/app.ts (or server.ts) — graceful shutdown + health
```ts
import express from 'express';
import { lineSignatureMiddleware, handleLineWebhook } from './controllers/lineWebhook';
import { pool } from './db';

const app = express();
app.use(express.json());

// Health endpoint for ops
app.get('/health', (req, res) => {
  res.json({ status: 'ok', uptime: process.uptime() });
});

// LINE webhook route with signature verification
app.post('/webhook/line', lineSignatureMiddleware, handleLineWebhook);

const server = app.listen(process.env.PORT || 3000, () => {
  console.log('Workio server running');
});

function shutdown(signal: string) {
  return async () => {
    console.log(`${signal} received: closing server`);
    server.close(async (err) => {
      if (err) {
        console.error('Error closing server', err);
        process.exit(1);
      }
      try {
        await pool.end();
        console.log('DB connections closed');
        process.exit(0);
      } catch (e) {
        console.error('Error closing DB pool', e);
        process.exit(1);
      }
    });
  };
}

process.on('SIGTERM', shutdown('SIGTERM'));
process.on('SIGINT', shutdown('SIGINT'));
```
