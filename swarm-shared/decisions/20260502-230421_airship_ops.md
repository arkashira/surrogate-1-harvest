### Core Principles (non-negotiable)
1. **Zero HF API calls during training** — use a pre-computed CDN-only manifest.
2. **Deterministic, cron-safe orchestration** — strict Bash, absolute paths, `SHELL=/bin/bash` in crontab.
3. **CDN-first, retry/backoff** — resilient downloads; no runtime HF auth.
4. **Schema enforced at parse time** — project to `{prompt,response}` only when reading Parquet; **do not** write `source`/`ts` into enriched files; use filename pattern `batches/mirror-merged/{date}/{slug}.parquet` for attribution.
5. **Lightning studio reuse + idle-stop handling** — avoid duplicate studios; respect running instances.

### 1) Hardened orchestrator (`scripts/airship-discover.sh`)

```bash
#!/usr/bin/env bash
# airship-discover.sh
# Orchestrates discovery + training prep. Deterministic, cron-safe, CDN-first.
# Usage: bash scripts/airship-discover.sh [--rebuild-file-list]

set -euo pipefail
IFS=$'\n\t'

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
REPO_ROOT="$(cd "${SCRIPT_DIR}/.." && pwd)"
CONFIG_DIR="${REPO_ROOT}/configs"
HF_REPO="${HF_REPO:-datasets/your-org/surrogate-mirror}"   # override via env
DATE_FOLDER="${DATE_FOLDER:-$(date +%Y-%m-%d)}"
FILE_LIST="${CONFIG_DIR}/hf-file-list.json"

log() { echo "[$(date -Iseconds)] $*"; }

ensure_dir() { mkdir -p "$1"; }

build_file_list() {
  log "Building HF file list (single API call) for ${HF_REPO}/${DATE_FOLDER}"
  ensure_dir "${CONFIG_DIR}"
  # Requires HF_TOKEN in env. Uses huggingface_hub for reliable tree listing.
  HF_TOKEN="${HF_TOKEN:?HF_TOKEN required}" python3 "${SCRIPT_DIR}/build_file_list.py" \
    --repo "${HF_REPO}" \
    --path "${DATE_FOLDER}" \
    --output "${FILE_LIST}"
  log "File list written to ${FILE_LIST}"
}

run_training() {
  if [[ ! -f "${FILE_LIST}" ]]; then
    log "ERROR: File list not found at ${FILE_LIST}. Run with --rebuild-file-list first."
    exit 1
  fi

  log "Starting surrogate training (CDN-only) with file list: ${FILE_LIST}"
  # Launch Lightning training (reuses running studio)
  cd "${REPO_ROOT}"
  HF_REPO="${HF_REPO}" python3 surrogate/train.py \
    --file-list "${FILE_LIST}" \
    --hf-repo "${HF_REPO}" \
    --machine "${MACHINE:-L40S}"
}

main() {
  ensure_dir "${CONFIG_DIR}"

  if [[ "${1:-}" == "--rebuild-file-list" ]]; then
    build_file_list
  fi

  run_training
}

main "$@"
```

Make executable and cron-safe:


### 2) File-list builder (`scripts/build_file_list.py`)

```python
#!/usr/bin/env python3
"""
build_file_list.py
List one date folder in a dataset repo and emit JSON for CDN-only training.
Uses per-folder tree calls (rate-limit friendly).
"""
import argparse
import json
import os
import sys
from datetime import datetime, timezone
from huggingface_hub import HfApi

def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True, help="e.g. datasets/your-org/surrogate-mirror")
    parser.add_argument("--path", required=True, help="Folder path in repo (e.g. 2026-05-02)")
    parser.add_argument("--output", required=True, help="Output JSON path")
    args = parser.parse_args()

    token = os.getenv("HF_TOKEN")
    if not token:
        print("ERROR: HF_TOKEN environment variable is required", file=sys.stderr)
        sys.exit(1)

    api = HfApi(token=token)

    # Non-recursive tree for this folder only (rate-limit friendly)
    tree = api.list_repo_tree(repo_id=args.repo, path=args.path, recursive=False)

    files = []
    for item in tree:
        if item.type != "file":
            continue
        # CDN URL (no auth, no API rate limit)
        cdn_url = f"https://huggingface.co/datasets/{args.repo}/resolve/main/{item.path}"
        files.append({
            "path": item.path,
            "cdn_url": cdn_url,
            "size": getattr(item, "size", None)
        })

    out = {
        "repo": args.repo,
        "folder": args.path,
        "generated_at_utc": datetime.now(timezone.utc).isoformat(),
        "files": files
    }

    os.makedirs(os.path.dirname(os.path.abspath(args.output)), exist_ok=True)
    with open(args.output, "w", encoding="utf-8") as f:
        json.dump(out, f, indent=2)

    print(f"Wrote {len(files)} files to {args.output}")

if __name__ == "__main__":
    main()
```

```bash

```
