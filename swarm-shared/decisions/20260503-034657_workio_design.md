# workio / design

## 1. Diagnosis
- Webhook state changes (clock-in/out, leave/OT) are not broadcast to connected clients; UI requires manual refresh → perceived latency and multi-tab divergence.
- No real-time channel between backend events and frontend; polling is wasteful and slow for time-sensitive actions.
- Supabase Realtime (already available via `serviceRole` key) is unused; can provide row-level events on `attendance`, `leave_requests`, `overtime_requests` for instant sync.
- No presence/connection status for LINE-linked sessions; managers can’t see who’s online or recently active.
- Missing optimistic UI + rollback strategy for transient failures during clock-in/out via LINE.

## 2. Proposed change
- Add a lightweight real-time layer using Supabase Realtime + React Query subscriptions.
- Scope:
  - Backend: enable realtime on `attendance`, `leave_requests`, `overtime_requests` (PostgreSQL publications via Supabase).
  - Frontend: create `useRealtimeAttendance(tenantId, userId)` and `useRealtimeRequests(tenantId, userId)` hooks that subscribe to relevant row events and update React Query cache.
  - Add `usePresenceChannel(tenantId)` to track active LINE-connected users via Supabase Presence (ephemeral, heartbeat-based).
- Files:
  - `workio/server/src/realtime/enable-realtime.sql` (migration)
  - `workio/src/hooks/useRealtimeAttendance.ts`
  - `workio/src/hooks/useRealtimeRequests.ts`
  - `workio/src/hooks/usePresenceChannel.ts`
  - `workio/src/lib/supabase.ts` (add presence/realtime helpers)

### 3.2 Supabase client with realtime helpers
```ts
// workio/src/lib/supabase.ts
import { createClient } from '@supabase/supabase-js';

export const supabase = createClient(
  import.meta.env.VITE_SUPABASE_URL,
  import.meta.env.VITE_SUPABASE_ANON_KEY
);

// Presence channel for tenant
export const tenantPresenceChannel = (tenantId: string) =>
  supabase.channel(`tenant:${tenantId}:presence`, {
    config: { presence: { key: tenantId } },
  });

// Track a user as online (call on mount)
export const trackPresence = async (tenantId: string, userId: string, meta?: Record<string, any>) => {
  const channel = tenantPresenceChannel(tenantId);
  await channel.subscribe();
  await channel.track({ user_id: userId, online_at: new Date().toISOString(), ...meta });
  return channel;
};

// Stop tracking
export const untrackPresence = async (channel: any) => {
  await channel.untrack();
  await channel.unsubscribe();
};
```

### 3.3 Realtime attendance hook
```ts
// workio/src/hooks/useRealtimeAttendance.ts
import { useEffect } from 'react';
import { useQueryClient } from '@tanstack/react-query';
import { supabase } from '../lib/supabase';

export function useRealtimeAttendance(tenantId: string, userId?: string) {
  const queryClient = useQueryClient();

  useEffect(() => {
    if (!tenantId) return;

    const channel = supabase
      .channel(`attendance:${tenantId}`)
      .on(
        'postgres_changes',
        {
          event: '*',
          schema: 'public',
          table: 'attendance',
          filter: userId ? `user_id=eq.${userId}` : undefined,
        },
        (payload) => {
          // Invalidate attendance queries for this tenant/user
          queryClient.invalidateQueries({ queryKey: ['attendance', tenantId, userId] });
          queryClient.invalidateQueries({ queryKey: ['today-attendance', tenantId, userId] });
        }
      )
      .subscribe();

    return () => {
      supabase.removeChannel(channel);
    };
  }, [tenantId, userId, queryClient]);
}
```

### 3.4 Realtime requests hook
```ts
// workio/src/hooks/useRealtimeRequests.ts
import { useEffect } from 'react';
import { useQueryClient } from '@tanstack/react-query';
import { supabase } from '../lib/supabase';

export function useRealtimeRequests(tenantId: string, userId?: string) {
  const queryClient = useQueryClient();

  useEffect(() => {
    if (!tenantId) return;

    const leaveChannel = supabase
      .channel(`leave:${tenantId}`)
      .on(
        'postgres_changes',
        {
          event: '*',
          schema: 'public',
          table: 'leave_requests',
          filter: userId ? `user_id=eq.${userId}` : undefined,
        },
        () => {
          queryClient.invalidateQueries({ queryKey: ['leave-requests', tenantId, userId] });
        }
      )
      .subscribe();

    const otChannel = supabase
      .channel(`overtime:${tenantId}`)
      .on(
        'postgres_changes',
        {
          event: '*',
          schema: 'public',
          table: 'overtime_requests',
          filter: userId ? `user_id=eq.${userId}` : undefined,
        },
        () => {
          queryClient.invalidateQueries({ queryKey: ['overtime-requests', tenantId, userId] });
        }
      )
      .subscribe();

    return () => {
      supabase.removeChannel(leaveChannel);
      supabase.removeChannel(otChannel);
    };
  }, [tenantId, userId, queryClient]);
}
```

### 3.5 Presence hook
```ts
// workio/src/hooks/usePresenceChannel.ts
import { useEffect, useState } from 'react';
import { supabase, tenantPresenceChannel, trackPresence, untrackPresence } from '../lib/supabase';

export function usePresenceChannel(tenantId: string, userId?: string) {
  const [presence, setPresence] = useState<Record<string, any>[]>([]);

  useEffect(() => {
    if (!tenantId || !userId) return;

    let channel = tenantPresenceChannel(tenantId);
    channel
      .on('presence', { event: 'sync' }, () => {
        const state = channel.presenceState();
        const flat = Object.values(state).flat() as Record<string, any>[];
        setPresence(flat);
      })
      .subscribe(async (status) => {
        if (status === 'SUBSCRIBED') {
          await trackPresence(tenantId, userId, { line_user_id: userId });
        }
      });

    return () => {
      untrackPresence(channel).catch(() => {});
    };
  }, [tenantId, userId]);

  return presence;
}
```

### 3.6 Integrate into main layout (example)
```tsx
// workio/src/routes/AppLayout.tsx (or similar)
import { useRealtimeAttendance } from '../hooks/useRealtimeAttendance';
import { useRealtimeRequests } from '../hooks/useRealtimeRequests';
import { usePresenceChannel } from '../hooks/usePresenceChannel';

export function AppLayout({ tenantId, userId }: { tenantId: string; userId: string }) {
  // Activate realtime subscriptions
  useRealtimeAttendance(tenantId, userId);
  useRealtimeRequests(tenantId, userId);
  const presence = usePresenceChannel(tenantId, userId);

  // presence can be used to show online indicators in UI
  return (
    <div>
      {/* your layout */}
    </div>
  );
}
```
