# airship / backend

## Final Implementation Plan — `airship discover` (CDN-only manifest)

**Scope (≤2h)**  
Add a deterministic CLI command `airship discover` that produces a CDN-only file manifest for a given `repo_id` and `date_folder`.  
- Uses **one** `list_repo_tree` API call (non-recursive) to avoid 429.  
- Outputs `manifest-{repo_slug}-{date}.json` containing CDN-resolvable paths and metadata.  
- Embeds the file list so Lightning training can do **zero API calls** during data load (CDN-only).  
- Supports sibling repo hashing for HF commit-cap scaling.

### 2) Core logic — `cli/utils/hf_cdn.py`

```python
# cli/utils/hf_cdn.py
from __future__ import annotations

import json
import hashlib
from datetime import datetime, timezone
from pathlib import Path
from typing import Dict, List, Any

from huggingface_hub import HfApi

CDN_BASE = "https://huggingface.co/datasets"


def sibling_repo_for_slug(slug: str, n_siblings: int = 5) -> int:
    """Deterministic sibling index 0..n_siblings-1."""
    digest = hashlib.sha256(slug.encode()).digest()
    return int.from_bytes(digest, "little") % n_siblings


def list_date_folder(
    repo_id: str,
    date_folder: str,
    token: str | None = None,
) -> List[Dict[str, Any]]:
    """
    Single non-recursive tree call for one date folder.
    Returns items with CDN paths and minimal metadata.
    """
    api = HfApi(token=token)
    tree = api.list_repo_tree(
        repo_id=repo_id,
        path=date_folder,
        repo_type="dataset",
        recursive=False,
    )

    items = []
    for entry in tree:
        if entry.rfilename.endswith((".parquet", ".json", ".jsonl")):
            cdn_url = f"{CDN_BASE}/{repo_id}/resolve/main/{entry.rfilename}"
            items.append(
                {
                    "path": entry.rfilename,
                    "cdn_url": cdn_url,
                    "size": getattr(entry, "size", None),
                    "lfs": getattr(entry, "lfs", None),
                }
            )
    return items


def build_manifest(
    repo_id: str,
    date_folder: str,
    output_dir: Path,
    token: str | None = None,
) -> Path:
    """
    Build manifest JSON for CDN-only training.
    """
    items = list_date_folder(repo_id, date_folder, token=token)

    manifest = {
        "repo_id": repo_id,
        "date_folder": date_folder,
        "generated_at_utc": datetime.now(timezone.utc).isoformat(),
        "cdn_strategy": "cdn-only",
        "sibling_index": sibling_repo_for_slug(f"{repo_id}/{date_folder}"),
        "files": items,
        "count": len(items),
    }

    output_dir.mkdir(parents=True, exist_ok=True)
    safe_slug = repo_id.replace("/", "-")
    out_path = output_dir / f"manifest-{safe_slug}-{date_folder}.json"
    out_path.write_text(json.dumps(manifest, indent=2))
    return out_path
```

### 3) CLI command — `cli/discover.py`

```python
# cli/discover.py
from __future__ import annotations

import argparse
import os
import sys
from pathlib import Path

from airship.cli.utils.hf_cdn import build_manifest


def main() -> None:
    parser = argparse.ArgumentParser(
        description="Generate CDN-only file manifest for HF dataset folder."
    )
    parser.add_argument(
        "repo_and_date",
        nargs="?",
        help="HF dataset repo and date, e.g. org/repo/2026-05-02",
    )
    parser.add_argument(
        "--repo-id",
        help="HF dataset repo, e.g. org/repo (alternative to positional)",
    )
    parser.add_argument(
        "--date-folder",
        help="Folder path in dataset, e.g. batches/mirror-merged/2026-05-02",
    )
    parser.add_argument(
        "--output-dir",
        default="airship/discover_manifests",
        help="Output directory for manifest JSON",
    )
    parser.add_argument(
        "--hf-token",
        default=os.getenv("HF_TOKEN"),
        help="HF token (optional; public CDN URLs do not require auth)",
    )
    args = parser.parse_args()

    # Parse combined positional arg
    repo_id = args.repo_id
    date_folder = args.date_folder

    if args.repo_and_date:
        parts = args.repo_and_date.strip().split("/")
        if len(parts) < 3:
            print(
                "ERROR: positional arg must be repo/date or org/repo/date",
                file=sys.stderr,
            )
            sys.exit(1)
        repo_id = "/".join(parts[:-1])
        date_folder = parts[-1]

    if not repo_id or not date_folder:
        print("ERROR: --repo-id and --date-folder are required", file=sys.stderr)
        parser.print_help()
        sys.exit(1)

    try:
        out = build_manifest(
            repo_id=repo_id,
            date_folder=date_folder,
            output_dir=Path(args.output_dir),
            token=args.hf_token,
        )
        print(f"Manifest written: {out}")
        sys.exit(0)
    except Exception as exc:
        print(f"Error: {exc}", file=sys.stderr)
        sys.exit(1)


if __name__ == "__main__":
    main()
```
