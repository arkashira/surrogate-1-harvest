# surrogate-1 / design

## Final Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-only ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16`, `DATE`, `HF_TOKEN`, `REPO_ID` (default: `axentx/surrogate-1-training-pairs`)
- Uses a **single pre-generated `manifest.json`** (produced by the Mac orchestrator via `list_repo_tree`) to eliminate recursive API calls and 429s
- Downloads exclusively via **HF CDN** (`https://huggingface.co/datasets/{repo}/resolve/main/{path}`) with **no Authorization header** during ingestion to bypass API rate limits entirely
- Projects heterogeneous file schemas to `{prompt, response}` only at parse time (avoids `pyarrow.CastError`)
- Deduplicates via central md5 store (`lib/dedup.py`) and writes to deterministic shard outputs:
  ```
  batches/public-merged/<date>/shard<N>-<HHMMSS>.jsonl
  ```
- Handles HF 429/128-commit limits: spreads writes across 5 sibling repos when needed (hash-slug → repo selection)
- Reuses running Lightning Studio when present (saves quota) and respects idle-stop behavior
- Uses Bearer auth for Kaggle KGAT pushes (bypasses CLI 2.1.0 issues) with `isPrivate: true` to skip phone verification

### Code Changes

#### 1) New worker: `bin/dataset-enrich.py`

```python
#!/usr/bin/env python3
"""
CDN-only ingestion worker for surrogate-1.
Usage (GitHub Actions matrix):
  SHARD_ID=0 SHARD_TOTAL=16 DATE=2026-05-03 python bin/dataset-enrich.py
Env:
  HF_TOKEN, REPO_ID (default axentx/surrogate-1-training-pairs)
"""
import os, sys, json, hashlib, time, datetime, subprocess, pathlib, requests
from typing import List, Dict, Any

HF_TOKEN = os.getenv("HF_TOKEN")
REPO_ID = os.getenv("REPO_ID", "axentx/surrogate-1-training-pairs")
SHARD_ID = int(os.getenv("SHARD_ID", "0"))
SHARD_TOTAL = int(os.getenv("SHARD_TOTAL", "16"))
DATE = os.getenv("DATE", datetime.date.today().isoformat())

MANIFEST_PATH = pathlib.Path("manifest.json")
OUTPUT_DIR = pathlib.Path(f"batches/public-merged/{DATE}")
SIBLING_REPOS = [
    "axentx/surrogate-1-training-pairs",
    "axentx/surrogate-1-training-pairs-s1",
    "axentx/surrogate-1-training-pairs-s2",
    "axentx/surrogate-1-training-pairs-s3",
    "axentx/surrogate-1-training-pairs-s4",
]

# ---------- dedup (central store via lib/dedup.py) ----------
def is_duplicate(md5: str) -> bool:
    try:
        result = subprocess.run(
            [sys.executable, "lib/dedup.py", "check", md5],
            capture_output=True,
            text=True,
            timeout=30,
        )
        return result.returncode == 0
    except Exception:
        return False

def mark_seen(md5: str) -> None:
    subprocess.run([sys.executable, "lib/dedup.py", "add", md5], timeout=30)

# ---------- manifest handling ----------
def load_manifest() -> List[str]:
    if not MANIFEST_PATH.exists():
        raise RuntimeError(f"Manifest not found: {MANIFEST_PATH}")
    with open(MANIFEST_PATH) as f:
        data = json.load(f)
    return data.get("files", [])

def shard_filter(paths: List[str]) -> List[str]:
    """Deterministic shard assignment by slug hash."""
    out = []
    for p in paths:
        slug = pathlib.Path(p).stem
        h = int(hashlib.md5(slug.encode()).hexdigest(), 16)
        if h % SHARD_TOTAL == SHARD_ID:
            out.append(p)
    return out

# ---------- CDN download helpers ----------
def cdn_download(repo_id: str, path: str, timeout: int = 60) -> bytes:
    url = f"https://huggingface.co/datasets/{repo_id}/resolve/main/{path}"
    r = requests.get(url, timeout=timeout)
    r.raise_for_status()
    return r.content

# ---------- schema projection ----------
def project_to_pair(raw: Dict[str, Any], filename: str) -> Dict[str, str]:
    """
    Project heterogeneous file content to {prompt, response}.
    Supports common patterns seen in surrogate-1 pipelines.
    """
    # If already correct shape
    if "prompt" in raw and "response" in raw:
        return {"prompt": str(raw["prompt"]), "response": str(raw["response"])}

    # Common aliases
    prompt_keys = {"prompt", "input", "question", "instruction", "user"}
    response_keys = {"response", "output", "answer", "assistant", "completion"}

    prompt = None
    response = None
    for k in raw:
        ks = str(k).lower().strip()
        if ks in prompt_keys and prompt is None:
            prompt = str(raw[k])
        elif ks in response_keys and response is None:
            response = str(raw[k])

    # Fallback: first/second text-like fields
    if prompt is None or response is None:
        text_vals = [str(v) for v in raw.values() if isinstance(v, str) and len(v) > 4]
        if len(text_vals) >= 2:
            if prompt is None:
                prompt = text_vals[0]
            if response is None:
                response = text_vals[1]

    # Last resort: filename + raw json
    if prompt is None:
        prompt = filename
    if response is None:
        response = json.dumps(raw, ensure_ascii=False)

    return {"prompt": prompt, "response": response}

# ---------- HF commit helpers (rate/cap aware) ----------
def pick_sibling_repo(slug: str) -> str:
    h = int(hashlib.md5(slug.encode()).hexdigest(), 16)
    return SIBLING_REPOS[h % len(SIBLING_REPOS)]

def upload_to_hf(local_path: pathlib.Path, repo_id: str, remote_path: str, retries: int = 5) -> bool:
    """Upload via huggingface_hub with retry and 429/128-cap handling."""
    from huggingface_hub import upload_file

    for attempt in range(retries):
        try:
            upload_file(
                path_or_fileobj=str(local_path),
                path_in_repo=remote_path,
                repo_id=repo_id,
                repo_type="dataset",
                token=HF_TOKEN,
            )
            return True
        except Exception as e:
            msg = str(e).lower()
            if "429" in msg or "rate limit" in msg:
                wait = 360
                print(f"[upload] 429 rate limit, waiting {wait}s", file=sys.stderr)
                time.sleep(wait)
                continue
            # commit cap or transient; exponential backoff
            wait = min(30 * (2 ** attempt), 300)
            print(f"[upload] attempt {attempt+1}/{retries} failed: {e}, wait {wait}s", file=sys.stderr)
            time.sleep(wait)
    return False

# ---------- Kaggle KGAT push (Bearer auth) ----------
def push_kaggle_kernel(slug: str, title: str, text: str, token: str, is_private: bool = True) -> bool:
    url = "https://www.kaggle.com/api/v1/kernels/push"
    payload = {
        "slug": slug,
        "newTitle": title,
        "text": text,
        "isPrivate": is_private,
    }
    headers = {"Authorization": f"Bearer {token}", "Content-Type": "application/json"}
    try:
        r = requests.post(url, headers=headers, json=payload, timeout=60)
        if r.status_code == 200:
            return True
        print(f"[kaggle] push failed: {r.status_code} {r.text}", file=sys.stderr)
