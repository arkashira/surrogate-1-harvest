## Highest-Value Incremental Improvement (<2h)

**Goal**: Eliminate HF API 429s and Lightning quota waste during Surrogate training by implementing **CDN-first deterministic ingestion with embedded file lists and Lightning Studio reuse**.

**Why this ships in <2h**: Single-file change to the surrogate training pipeline + one orchestration wrapper. No infra changes, no new dependencies.

---

## Implementation Plan

### 1. Add CDN-first file listing (Mac orchestration)
- Run `list_repo_tree` once per date folder → save `file_list.json`
- Embed `file_list.json` in training container
- Training script uses CDN URLs (`https://huggingface.co/datasets/.../resolve/main/...`) with zero API calls during data load

### 2. Lightning Studio reuse guard
- Before `Studio(create_ok=True)`, list `Teamspace.studios` and reuse running ones
- Add idle-stop check + restart with `target.start(machine=Machine.L40S)` if stopped

### 3. Schema-safe ingestion
- Project to `{prompt, response}` only before upload
- Use filename pattern: `batches/mirror-merged/{date}/{slug}.parquet`
- No `source` / `ts` columns (avoid mixed schema)

### surrogate/train.py (CDN-first data loader)
```python
import json
import os
from pathlib import Path
from typing import List, Dict

import torch
from torch.utils.data import Dataset, DataLoader
import pyarrow.parquet as pq
import requests
from tqdm import tqdm

HF_DATASET = "axentx/surrogate-mirror"
CDN_ROOT = f"https://huggingface.co/datasets/{HF_DATASET}/resolve/main"

class CDNParquetDataset(Dataset):
    def __init__(self, file_list_path: str = "file_list.json"):
        """CDN-first dataset: zero HF API calls during training."""
        with open(file_list_path) as f:
            self.files = json.load(f)  # list of {"path": "...", "date": "...", "slug": "..."}
        self.cache_dir = Path("./cdn_cache")
        self.cache_dir.mkdir(exist_ok=True)

    def _download_cdn(self, rel_path: str) -> Path:
        """Download via CDN (no auth, no rate limit)."""
        url = f"{CDN_ROOT}/{rel_path}"
        out = self.cache_dir / rel_path.replace("/", "_")
        if out.exists():
            return out
        out.parent.mkdir(parents=True, exist_ok=True)
        r = requests.get(url, stream=True, timeout=30)
        r.raise_for_status()
        with open(out, "wb") as f:
            for chunk in r.iter_content(chunk_size=8192):
                f.write(chunk)
        return out

    def _project_to_pair(self, parquet_path: Path) -> List[Dict[str, str]]:
        """Project to {prompt, response} only — ignore mixed schema cols."""
        table = pq.read_table(parquet_path, columns=["prompt", "response"])
        df = table.to_pandas()
        # Drop rows with nulls in either column
        df = df.dropna(subset=["prompt", "response"])
        return df.to_dict(orient="records")

    def __len__(self) -> int:
        return len(self.files)

    def __getitem__(self, idx: int) -> Dict[str, str]:
        entry = self.files[idx]
        rel_path = f"batches/mirror-merged/{entry['date']}/{entry['slug']}.parquet"
        local_path = self._download_cdn(rel_path)
        pairs = self._project_to_pair(local_path)
        # Deterministic pick: hash-based modulo for this idx
        pair = pairs[idx % len(pairs)] if pairs else {"prompt": "", "response": ""}
        return pair

def build_dataloader(file_list_path: str = "file_list.json", batch_size: int = 8, num_workers: int = 4) -> DataLoader:
    dataset = CDNParquetDataset(file_list_path)
    return DataLoader(
        dataset,
        batch_size=batch_size,
        num_workers=num_workers,
        pin_memory=True,
        persistent_workers=True if num_workers > 0 else False,
    )
```

### surrogate/launch_studio.py (Lightning Studio reuse + idle guard)
```python
#!/usr/bin/env bash
# Wrapper ensures proper shebang and executable bit for cron
# Usage: bash launch_studio.sh

set -euo pipefail
export SHELL=/bin/bash

python3 - "$@" << 'PY'
import os
import time
from lightning_sdk import Teamspace, Studio, Machine

TEAMSPACE = os.getenv("LIGHTNING_TEAMSPACE", "default")
STUDIO_NAME = os.getenv("STUDIO_NAME", "surrogate-train")
MACHINE = Machine.L40S  # Free tier falls to L40S; H200 requires lightning-lambda-prod

def reuse_or_create_studio() -> Studio:
    team = Teamspace(name=TEAMSPACE)
    # Reuse running studio to save quota (~80hr/mo)
    for s in team.studios:
        if s.name == STUDIO_NAME and s.status == "running":
            print(f"Reusing running studio: {STUDIO_NAME}")
            return s
    # Create if not running
    print(f"Creating studio: {STUDIO_NAME}")
    return Studio(
        name=STUDIO_NAME,
        teamspace=TEAMSPACE,
        create_ok=True,
        machine=MACHINE,
    )

def ensure_running(studio: Studio) -> Studio:
    # Lightning idle stop kills training; restart if stopped
    studio.refresh()
    if studio.status != "running":
        print(f"Studio stopped ({studio.status}), restarting on {MACHINE}")
        studio.start(machine=MACHINE)
        # Wait for running
        for _ in range(30):
            studio.refresh()
            if studio.status == "running":
                break
            time.sleep(10)
        studio.refresh()
    return studio

if __name__ == "__main__":
    studio = reuse_or_create_studio()
    studio = ensure_running(studio)
    print(f"Studio ready: {studio.name} ({studio.status}) on {studio.machine}")
    # Export for downstream scripts
    print(f"STUDIO_ID={studio.id}")
PY
```

### Mac orchestration: generate file_list.json (run once per date folder)
```bash
#!/usr/bin/env bash
# scripts/generate_file_list.sh
# Run from Mac after HF API rate-limit window clears

set -euo pipefail
REPO="axentx/surrogate-mirror"
DATE_FOLDER="${1:-$(date +%Y-%m-%d)}"  # e.g. 2026-05-03
OUTPUT="surrogate/file_list.json"

python3 - "$REPO" "$DATE_FOLDER" "$OUTPUT" << 'PY'
import json
import os
import sys
from huggingface_hub import HfApi

repo, date_folder, out_path = sys.argv[1], sys.argv[2], sys.argv[3]
api = HfApi()

# Single non-recursive call per folder (avoids 100x pagination)
tree = api.list_repo_tree(repo, path=f"batches/mirror-merged/{date_folder}", recursive=False)
files = []
for item in tree:
    if item.path.endswith(".parquet"):
        # slug derived from filename
        slug = item.path.split("/")[-1].replace(".parquet", "")
        files.append({
            "path": item.path,
            "date": date_folder,
            "slug": slug,
        })

os.makedirs(os.path.dirname(out_path), exist_ok=True)
with open(out_path, "w") as f:
    json.dump(files, f, indent=2)
print(f"Wrote {len(files)} files to {out_path}")
PY
```
