# surrogate-1 / design

## Final Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16` (matrix) and optional `DATE_FOLDER` (defaults to today `YYYY-MM-DD`)
- Uses **manifest-driven, CDN-bypass ingestion**: single API call to list one date folder → save `file-list.json`; worker loads manifest, deterministically shards by `hash(slug) % SHARD_TOTAL`, then downloads assigned files via **raw CDN URLs** (no auth, bypasses `/api/` rate limits)
- Projects each file to `{prompt, response}` only at parse time (avoids pyarrow CastError on mixed schemas)
- Dedups via central `lib/dedup.py` md5 store (unchanged)
- Writes `batches/public-merged/<DATE>/shard<N>-<HHMMSS>.jsonl` with deterministic ordering and no collisions
- Keeps wrapper Bash stub for GitHub Actions matrix compatibility

def deterministic_shard(key: str, total: int) -> int:
    return int(hashlib.sha256(key.encode("utf-8")).hexdigest(), 16) % total

def list_date_folder() -> List[str]:
    """Single API call: list files in one date folder (non-recursive)."""
    try:
        tree = API.list_repo_tree(repo_id=HF_REPO, path=DATE_FOLDER, recursive=False)
        # tree items have 'path'
        paths = [item["path"] for item in tree if item.get("type") == "file"]
        return sorted(paths)
    except Exception as e:
        print(f"ERROR listing repo tree: {e}", file=sys.stderr)
        sys.exit(1)

def ensure_manifest() -> List[str]:
    if os.path.exists(MANIFEST_PATH):
        with open(MANIFEST_PATH) as f:
            return json.load(f)
    paths = list_date_folder()
    with open(MANIFEST_PATH, "w") as f:
        json.dump(paths, f)
    print(f"Manifest saved: {MANIFEST_PATH} ({len(paths)} files)")
    return paths

def cdn_download(url: str, dest: Path) -> Path:
    """Download via public CDN (no auth). Retries on failure."""
    for attempt in range(3):
        try:
            r = requests.get(url, timeout=60)
            r.raise_for_status()
            dest.write_bytes(r.content)
            return dest
        except Exception as e:
            if attempt == 2:
                raise
            print(f"Retry {attempt+1}/3 for {url}: {e}")
    raise RuntimeError("unreachable")

def parse_file_to_pair(local_path: Path) -> Dict[str, str]:
    """
    Best-effort projection to {prompt, response}.
    Supports common HF dataset file formats (jsonl, parquet via `hf_hub_download` fallback).
    """
    suffix = local_path.suffix.lower()
    # If parquet or other complex format, use hf_hub_download + streaming-safe projection.
    # For this worker we prefer raw CDN text formats (jsonl/csv) and minimal parsing.
    if suffix == ".jsonl":
        import json as json_module
        objects = []
        with local_path.open() as f:
            for line in f:
                line = line.strip()
                if not line:
                    continue
                obj = json_module.loads(line)
                objects.append(obj)
        # Heuristic: pick first row or merge? For training pairs we expect one row per file.
        # If multiple rows, we'll emit one line per row later in caller.
        pairs = []
        for obj in objects:
            prompt = obj.get("prompt") or obj.get("input") or obj.get("text") or ""
            response = obj.get("response") or obj.get("output") or obj.get("completion") or ""
            if prompt or response:
                pairs.append({"prompt": prompt, "response": response})
        if not pairs:
            # fallback: treat entire file content as prompt, empty response
            text = local_path.read_text(encoding="utf-8", errors="ignore")
            return {"prompt": text, "response": ""}
        # For simplicity, return first pair; caller can iterate if needed.
        return pairs[0]
    elif suffix == ".json":
        import json as json_module
        obj = json_module.loads(local_path.read_text(encoding="utf-8", errors="ignore"))
        if isinstance(obj, list):
            obj = obj[0] if obj else {}
        prompt = obj.get("prompt") or obj.get("input") or obj.get("text") or ""
        response = obj.get("response") or obj.get("output") or obj.get("completion") or ""
        return {"prompt": prompt, "response": response}
    else:
        # Generic fallback: treat file as raw prompt
        text = local_path.read_text(encoding="utf-8", errors="ignore")
        return {"prompt": text, "response": ""}
