# workio / design

## 1. Diagnosis
- Missing `X-Line-Signature` HMAC-SHA256 verification on `/webhook/line` allows spoofed/replayed events.
- No idempotency guard enables duplicate clock-in/out or leave/OT records when LINE retries.
- Tenant context is inferred only from payload fields; no binding to the channel/tenant at verification time.
- Raw body is not preserved for HMAC; body-parser JSON parsing mutates bytes and breaks signature checks.
- No structured schema validation on webhook events; malformed events can corrupt state.

## 2. Proposed change
File: `workio/server/src/routes/webhook.line.ts` (create if absent) or update existing route handler for `/webhook/line`.  
Scope: add middleware for raw-body capture, HMAC verification, idempotency key check, tenant binding, and JSON schema validation before processing.

## 3. Implementation
Install deps (if not present):
```bash
cd /opt/axentx/workio/server
npm install raw-body ajv ajv-formats crypto
```

Create/replace `workio/server/src/routes/webhook.line.ts`:
```ts
import express, { Request, Response, NextFunction } from 'express';
import crypto from 'crypto';
import rawBody from 'raw-body';
import Ajv from 'ajv';
import addFormats from 'ajv-formats';
import { db } from '../db';

const router = express.Router();
const ajv = new Ajv({ allErrors: true });
addFormats(ajv);

// Minimal LINE event schema (extend as needed)
const LINE_EVENT_SCHEMA = {
  type: 'object',
  required: ['destination', 'events'],
  properties: {
    destination: { type: 'string' },
    events: {
      type: 'array',
      items: {
        type: 'object',
        required: ['type', 'timestamp', 'source'],
        properties: {
          type: { type: 'string' },
          timestamp: { type: 'number' },
          source: {
            type: 'object',
            required: ['type', 'userId'],
            properties: {
              type: { type: 'string' },
              userId: { type: 'string' },
              groupId: { type: 'string' },
              roomId: { type: 'string' }
            }
          },
          message: { type: 'object' },
          postback: { type: 'object' }
        }
      }
    }
  }
};

const validateLinePayload = ajv.compile(LINE_EVENT_SCHEMA);

// Resolve tenant by LINE channel (destination) and bind to req.tenantId
async function resolveTenantByDestination(destination: string): Promise<string | null> {
  // Map LINE channel (destination) to tenant_id via channel_configs or tenants table.
  // Example: SELECT tenant_id FROM line_channel_configs WHERE channel_id = $1
  const row = await db.oneOrNone(
    'SELECT tenant_id FROM line_channel_configs WHERE channel_id = $1',
    [destination]
  );
  return row ? row.tenant_id : null;
}

// Idempotency guard using LINE's webhook event-level id (or synthetic key)
async function isDuplicateEvent(eventId: string): Promise<boolean> {
  const row = await db.oneOrNone(
    'SELECT 1 FROM line_webhook_events WHERE event_id = $1',
    [eventId]
  );
  return !!row;
}

async function markEventProcessed(eventId: string, tenantId: string, payload: any): Promise<void> {
  await db.none(
    'INSERT INTO line_webhook_events(event_id, tenant_id, payload, processed_at) VALUES($1,$2,$3,now())',
    [eventId, tenantId, payload]
  );
}

// Middleware: capture raw body for HMAC
router.use('/line', async (req: Request, res: Response, next: NextFunction) => {
  try {
    const raw = await rawBody(req);
    (req as any).rawBody = raw;
    next();
  } catch (err) {
    next(err);
  }
});

// Handler: verify signature, validate, deduplicate, bind tenant, then process
router.post('/line', async (req: Request, res: Response) => {
  const signature = req.headers['x-line-signature'] as string | undefined;
  const raw = (req as any).rawBody as Buffer;
  const secret = process.env.LINE_CHANNEL_SECRET;

  if (!signature || !secret) {
    return res.status(401).json({ error: 'Missing signature or secret' });
  }

  const expected = crypto
    .createHmac('sha256', secret)
    .update(raw)
    .digest('base64');

  if (signature !== expected) {
    return res.status(401).json({ error: 'Invalid signature' });
  }

  let body: any;
  try {
    body = JSON.parse(raw.toString('utf8'));
  } catch (err) {
    return res.status(400).json({ error: 'Invalid JSON' });
  }

  if (!validateLinePayload(body)) {
    return res.status(400).json({ error: 'Invalid payload', details: validateLinePayload.errors });
  }

  const { destination, events } = body;
  const tenantId = await resolveTenantByDestination(destination);
  if (!tenantId) {
    return res.status(404).json({ error: 'Tenant not found for channel' });
  }

  // Process events sequentially; stop on first duplicate to preserve ordering semantics.
  for (const ev of events) {
    // Use LINE event id if provided; otherwise create deterministic synthetic id.
    const eventId = ev.eventId || crypto.createHash('sha256')
      .update(`${destination}:${ev.timestamp}:${ev.source.userId}:${ev.type}`)
      .digest('hex');

    if (await isDuplicateEvent(eventId)) {
      // Idempotent success: skip duplicate but continue processing others if desired.
      continue;
    }

    // TODO: domain processing (clock-in/out, leave/OT) using tenantId and ev.
    // Example: await handleClockIn(tenantId, ev);

    await markEventProcessed(eventId, tenantId, ev);
  }

  return res.status(200).json({ ok: true });
});

export default router;
```

Wire into main server (e.g., `workio/server/src/app.ts` or `index.ts`):
```ts
import lineWebhook from './routes/webhook.line';
app.use('/webhook', lineWebhook);
```

Add table for idempotency (run once):
```sql
-- server/src/db/schema.sql addition
CREATE TABLE IF NOT EXISTS line_webhook_events (
  event_id TEXT PRIMARY KEY,
  tenant_id TEXT NOT NULL,
  payload JSONB NOT NULL,
  processed_at TIMESTAMPTZ NOT NULL DEFAULT now()
);
CREATE INDEX IF NOT EXISTS idx_line_webhook_events_tenant ON line_webhook_events(tenant_id);
```

Add channel mapping table (if not present):
```sql
CREATE TABLE IF NOT EXISTS line_channel_configs (
  channel_id TEXT PRIMARY KEY,
  tenant_id TEXT NOT NULL,
  created_at TIMESTAMPTZ NOT NULL DEFAULT now()
);
```

## 4. Verification
1. Start backend: `cd /opt/axentx/workio/server && npm run dev`
2. Use ngrok or a test endpoint to expose `/webhook/line`.
3. Compute valid signature:
   ```bash
   echo -n '{"destination":"U123","events":[]}' | openssl dgst -sha256 -hmac "$LINE_CHANNEL_SECRET" -binary | base64
   ```
4. Send POST with header `X-Line-Signature: <computed>` and JSON body. Expect 200 and record in `line_webhook_events`.
5. Replay same payload — expect 200 but no new `line_webhook_events` row (idempotent).
6. Send with invalid signature — expect 401.
7. Send malformed JSON or missing fields — expect 400 with details.
