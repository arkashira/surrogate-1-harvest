# Costinel / design

**Final consolidated implementation (≤2h)**

**What to ship**  
A single, deterministic discovery module that emits `manifests/sources.json` containing:

- Cloud accounts (AWS/GCP/Azure) with aliases and status  
- Dataset file lists as HF CDN URLs (no auth/rate-limit hits during training)  
- Top-hub context stub (ready for knowledge-rag integration)  
- Readiness flags (HF token, Lightning quota, Kaggle auth)  

This satisfies onboarding, training, and governance scripts without repeated API calls or fragile assumptions.

src/discovery/__init__.py

**src/discovery/manifest.py**
```python
#!/usr/bin/env python3
"""
Lightweight discovery module for Costinel.
Produces manifests/sources.json with:
- cloud accounts
- dataset file lists (HF CDN URLs)
- top-hub context
- readiness flags
"""
from __future__ import annotations

import json
import hashlib
import os
import sys
from datetime import datetime, timezone
from pathlib import Path
from typing import Any, Dict, List, Optional

try:
    from huggingface_hub import HfApi, list_repo_tree
except Exception:  # pragma: no cover
    HfApi = None  # type: ignore
    list_repo_tree = None  # type: ignore


MANIFEST_DIR = Path(__file__).parents[2] / "manifests"
MANIFEST_DIR.mkdir(exist_ok=True, parents=True)


def _hash_slug(slug: str, n_siblings: int = 5) -> int:
    """Deterministic repo selection for HF commit-cap spreading."""
    digest = hashlib.sha256(slug.encode()).digest()
    return int.from_bytes(digest, "big") % n_siblings


def discover_cloud_accounts() -> List[Dict[str, Any]]:
    """Detect configured cloud accounts (env-based)."""
    accounts: List[Dict[str, Any]] = []
    for prefix in ("AWS", "GCP", "AZURE"):
        if os.getenv(f"{prefix}_ACCOUNT_ID") or os.getenv(f"{prefix}_PROJECT_ID"):
            accounts.append(
                {
                    "provider": prefix.lower(),
                    "account_id": os.getenv(f"{prefix}_ACCOUNT_ID")
                    or os.getenv(f"{prefix}_PROJECT_ID")
                    or "unknown",
                    "alias": os.getenv(f"{prefix}_ALIAS", f"{prefix.lower()}-prod"),
                    "status": "active",
                    "discovered_at": datetime.now(timezone.utc).isoformat(),
                }
            )
    if not accounts:
        # Default fallback for local/dev
        accounts.append(
            {
                "provider": "aws",
                "account_id": "000000000000",
                "alias": "local-dev",
                "status": "simulated",
                "discovered_at": datetime.now(timezone.utc).isoformat(),
            }
        )
    return accounts


def discover_dataset_files(
    repo_id: str,
    folder_date: str,
    hf_token: Optional[str] = None,
) -> List[Dict[str, Any]]:
    """
    List files once via HF API, return CDN URLs to bypass auth rate limits.
    Uses list_repo_tree per folder (non-recursive) to avoid pagination blowup.
    """
    if HfApi is None or list_repo_tree is None:
        return []

    api = HfApi(token=hf_token)
    cdn_base = f"https://huggingface.co/datasets/{repo_id}/resolve/main"
    entries: List[Dict[str, Any]] = []

    try:
        tree = list_repo_tree(repo_id=repo_id, path=folder_date, recursive=False)
        for node in tree:
            if node.type == "file" and node.path.endswith((".parquet", ".jsonl")):
                cdn_url = f"{cdn_base}/{node.path}"
                entries.append(
                    {
                        "path": node.path,
                        "cdn_url": cdn_url,
                        "size": getattr(node, "size", None),
                    }
                )
    except Exception as exc:
        sys.stderr.write(f"[discovery] HF tree list failed: {exc}\n")

    return entries


def top_hub_context() -> Dict[str, Any]:
    """
    Lightweight top-hub insight (stub).
    In production, integrate with knowledge-rag graph query for MOC-style hubs.
    """
    return {
        "hub": "MOC",
        "type": "cost-governance",
        "connections": 0,
        "last_review": None,
        "note": "Review top-connected hub before planning tasks (knowledge-rag/graph).",
    }


def readiness_flags(hf_token: Optional[str] = None) -> Dict[str, Any]:
    """Check minimal readiness for training/ingestion pipelines."""
    flags = {
        "hf_auth": bool(hf_token),
        "lightning_quota_ok": False,
        "kaggle_auth": bool(os.getenv("KAGGLE_USERNAME") and os.getenv("KAGGLE_KEY")),
    }

    # Reuse running studios before create; check status before .run()
    try:
        import lightning as L

        studios = L.studio.studio.LightningStudioStatus.list()
        running = [s for s in studios if s.status in ("running", "pending")]
        flags["lightning_quota_ok"] = len(running) > 0
    except Exception:
        pass

    return flags


def produce_manifest(
    repo_id: str,
    folder_date: str,
    hf_token: Optional[str] = None,
    out_path: Optional[Path] = None,
) -> Path:
    """
    Produce manifests/sources.json.
    """
    if out_path is None:
        out_path = MANIFEST_DIR / "sources.json"

    manifest = {
        "generated_at": datetime.now(timezone.utc).isoformat(),
        "cloud_accounts": discover_cloud_accounts(),
        "dataset_files": discover_dataset_files(
            repo_id=repo_id, folder_date=folder_date, hf_token=hf_token
        ),
        "top_hub_context": top_hub_context(),
        "readiness": readiness_flags(hf_token=hf_token),
    }

    out_path.write_text(json.dumps(manifest, indent=2))
    return out_path


if __name__ == "__main__":
    # Example usage (adjust repo_id/folder_date as needed)
    hf_token = os.getenv("HF_TOKEN")
    out = produce_manifest(
        repo_id="surrogate-1",
        folder_date="2024-06-01",
        hf_token=hf_token,
    )
    print(f"Manifest written to {out}")
```

---

**Key resolved choices (correctness + actionability)**
- Use `list_repo_tree` once per folder → deterministic, avoids pagination/rate limits; training consumes CDN URLs only.  
- Readiness checks include Lightning quota guard (reuse running studios) and Kaggle auth via env vars.  
- Cloud accounts are env-driven with safe local fallback; schema is extensible.  
- Top-hub context is a minimal stub (clearly marked) so governance scripts can run immediately and be upgraded later.
