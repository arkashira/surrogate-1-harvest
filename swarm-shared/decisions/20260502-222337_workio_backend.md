# workio / backend

## 1. Diagnosis
- LINE webhook handler re-resolves `channelId → tenant` on every request with no caching → DB thrash under traffic spikes.
- Missing correlation IDs (requestId) in logs and across webhook → service → DB flow → incidents untraceable.
- No structured logging (JSON) → hard to query and alert in production.
- No graceful handling of LINE signature validation failures → noisy logs and potential retry storms.
- No rate-limiting or idempotency on clock-in/out events → duplicate entries possible on retries.

### src/middleware/tracing.ts
```ts
import { randomUUID } from 'uuid';
import type { Request, Response, NextFunction } from 'express';

export interface RequestLocals {
  requestId: string;
  tenantId?: string;
}

declare global {
  namespace Express {
    interface Request {
      locals: RequestLocals;
    }
  }
}

export function tracingMiddleware(req: Request, res: Response, next: NextFunction) {
  req.locals = { requestId: req.headers['x-request-id'] as string || randomUUID() };
  res.setHeader('X-Request-Id', req.locals.requestId);
  next();
}
```

### src/middleware/tenant.ts
```ts
import type { Request, Response, NextFunction } from 'express';
import LRU from 'lru-cache';
import { getChannelTenant } from '../services/tenantService'; // assume exists or create

const tenantCache = new LRU<string, { tenantId: string; ttl: number }>({
  max: 500,
  ttl: 1000 * 60 * 10, // 10m
});

export async function tenantMiddleware(req: Request, res: Response, next: NextFunction) {
  const channelId = req.body?.source?.userId ? undefined : (req.body?.channelId || req.query.channelId || req.headers['x-line-channel-id']);
  if (!channelId) return next();

  const cached = tenantCache.get(channelId as string);
  if (cached) {
    req.locals.tenantId = cached.tenantId;
    return next();
  }

  try {
    const tenant = await getChannelTenant(channelId as string);
    if (tenant?.id) {
      tenantCache.set(channelId as string, { tenantId: tenant.id, ttl: Date.now() + 10 * 60 * 1000 });
      req.locals.tenantId = tenant.id;
    }
  } catch (err) {
    // don't block request; downstream will handle missing tenant
  }
  next();
}
```

### src/middleware/logging.ts
```ts
import type { Request, Response, NextFunction } from 'express';

export function loggingMiddleware(req: Request, res: Response, next: NextFunction) {
  const start = Date.now();
  res.on('finish', () => {
    const duration = Date.now() - start;
    const log = {
      requestId: req.locals?.requestId || '-',
      tenantId: req.locals?.tenantId || '-',
      method: req.method,
      path: req.path,
      status: res.statusCode,
      durationMs: duration,
      ip: req.ip,
      userAgent: req.get('User-Agent'),
    };
    console.log(JSON.stringify(log));
  });
  next();
}
```

### src/routes/webhook/line.ts (modify)
```ts
import express from 'express';
import { validateSignature } from '../utils/lineSignature';
import { tracingMiddleware } from '../middleware/tracing';
import { tenantMiddleware } from '../middleware/tenant';
import { loggingMiddleware } from '../middleware/logging';
import { handleClockEvent } from '../controllers/clockController';

const router = express.Router();

router.use(express.json({ limit: '1mb' }));
router.use(tracingMiddleware);
router.use(loggingMiddleware);
router.use(tenantMiddleware);

router.post('/line', async (req, res) => {
  const sig = req.headers['x-line-signature'] as string;
  const body = JSON.stringify(req.body);
  if (!validateSignature(body, sig)) {
    console.warn(JSON.stringify({ requestId: req.locals.requestId, msg: 'invalid-line-signature' }));
    return res.status(401).json({ error: 'invalid signature' });
  }

  const events = req.body.events || [];
  const results = [];
  for (const ev of events) {
    if (ev.type === 'message' && ev.message?.type === 'text') {
      // idempotency: use event.id as dedupe key for clock events
      try {
        const handled = await handleClockEvent(ev, req.locals);
        results.push({ eventId: ev.id, ok: handled });
      } catch (err) {
        results.push({ eventId: ev.id, ok: false, error: String(err) });
      }
    }
  }
  res.json({ results });
});

export default router;
```

import { prisma } from '../db'; // assume prisma client

const processed = new Set<string>(); // in-memory for single instance; replace with Redis for multi-instance

export async function handleClockEvent(event: any, locals: any) {
  if (processed.has(event.id)) return true;
  const text = event.message.text.trim().toLowerCase();
  // parse clock in/out commands
  const employeeId = await resolveEmployeeId(event.source.userId, locals.tenantId);
  await prisma.attendance.create({
    data: {
      employeeId,
      tenantId: locals.tenantId,
      eventType: text.includes('in') ? 'CLOCK_IN' : 'CLOCK_OUT',
      location: event.message?.location || null,
      lineEventId: event.id,
      createdAt: new Date(),
    },
  });
  processed.add(event.id);
  return true;
}
