"""
Manifest-driven, CDN-bypass worker for surrogate-1 public dataset ingestion.

Usage:
  python3 bin/worker.py --shard 0 --date 2026-05-03 --out-dir ./out

Behavior:
- Expects manifest.json in cwd or generated by launcher (Mac/Studio):
    {"date": "2026-05-03", "files": ["folder/file1.json", ...]}

# Known schema projections: keep only prompt/response; drop source/ts/attribution cols

def slug_for_path(path: str) -> str:
    """Deterministic slug from repo-relative path."""
    return path.rsplit(".", 1)[0].replace("/", "_").replace(" ", "_")

def shard_for_slug(slug: str, n_shards: int = 16) -> int:
    return int(hashlib.md5(slug.encode()).hexdigest(), 16) % n_shards

def download_cdn(url: str, timeout: int = 30) -> bytes:
    resp = requests.get(url, timeout=timeout)
    resp.raise_for_status()
    return resp.content

def parse_file_to_pairs(content: bytes, path: str) -> List[Dict[str, str]]:
    """
    Parse heterogeneous file content into [{prompt, response}] pairs.
    Uses pyarrow JSON reader with column projection to avoid mixed-schema CastError.
    Falls back to line-by-line JSON if arrow fails.
    """
    pairs = []
    try:
        table = paj.read_json(
            pa.py_buffer(content),
            read_options=paj.ReadOptions(column_names=["prompt", "response"]),
            parse_options=paj.ParseOptions(explicit_schema=pa.schema([
                pa.field("prompt", pa.string()),
                pa.field("response", pa.string()),
            ])),
        )
        for row in table.to_pylist():
            if row.get("prompt") and row.get("response"):
                pairs.append({"prompt": row["prompt"], "response": row["response"]})
        if pairs:
            return pairs
    except Exception:
        pass

    # Fallback: line-by-line JSON
    for line in content.decode("utf-8", errors="ignore").splitlines():
        line = line.strip()
        if not line:
            continue
        try:
            obj = json.loads(line)
            if isinstance(obj, dict):
                prompt = obj.get("prompt") or obj.get("input") or obj.get("question")
                response = obj.get("response") or obj.get("output") or obj.get("answer")
                if prompt and response:
                    pairs.append({"prompt": str(prompt), "response": str(response)})
        except Exception:
            continue
    return pairs

def load_manifest(manifest_path: Path) -> Dict[str, Any]:
    with open(manifest_path, "r", encoding="utf-8") as f:
        return json.load(f)

def run_worker(shard_id: int, date: str, out_dir: Path, manifest_path: Optional[Path] = None) -> None:
    if manifest_path is None:
        manifest_path = Path.cwd() / "manifest.json"
    if not manifest_path.exists():
        raise FileNotFoundError(f"Manifest not found: {manifest_path}")

    manifest = load_manifest(manifest_path)
    files: List[str] = manifest.get("files", [])
    if not files:
        print("No files in manifest; nothing to do.")
        return

    out_dir = out_dir.resolve()
    batch_dir = out_dir / "batches" / "public-merged" / date
    batch_dir.mkdir(parents=True, exist_ok=True)

    timestamp = datetime.now(timezone.utc).strftime("%H%M%S")
    out_file = batch_dir / f"shard{shard_id}-{timestamp}.jsonl"

    dedup = DedupStore()
    written = 0
    skipped_dup = 0
    skipped_shard = 0

    for rel_path in tqdm(files, desc=f"Shard {shard_id}"):
        slug = slug_for_path(rel_path)
        if shard_for_slug(slug) != shard_id:
            skipped_shard += 1
            continue

        url = f"{CDN_BASE}/{rel_path}"
        try:
            content = download_cdn(url)
        except Exception as exc:
            print(f"Failed to download {rel_path}: {exc}", file=sys.stderr)
            continue

        pairs = parse_file_to_pairs(content, rel_path)
        for pair in pairs:
            # Build deterministic md5 for dedup across sources
            raw = f"{pair['prompt']}\n{pair['response']}"
            md5 = hashlib.md5(raw.encode("utf-8")).hexdigest()
            if dedup.exists(md5):
                skipped_dup += 1
                continue

            dedup.add(md5)
            out_line = json.dumps(pair, ensure_ascii=False)
            with open(out_file, "a", encoding="utf-8") as f:
                f.write(out_line + "\n")
            written += 1

    print(f"Shard {shard_id} done. Written: {written}, Duplicates skipped: {skipped_dup}, Other-shard files: {skipped_shard}")
    print(f"Output: {out_file}")
