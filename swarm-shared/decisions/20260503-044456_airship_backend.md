### 2. CDN-only data loader (Lightning training)
Embed manifest and use CDN URLs exclusively — zero API calls during training.

```python
# surrogate/data/cdn_parquet_loader.py
import json
import pyarrow.parquet as pq
import requests
from pathlib import Path
from torch.utils.data import IterableDataset
import io

class CDNParquetDataset(IterableDataset):
    def __init__(self, manifest_path: str, max_files: int = None):
        with open(manifest_path) as f:
            manifest = json.load(f)
        
        self.repo = manifest["repo"]
        self.files = manifest["files"][:max_files]
        self.base_url = f"https://huggingface.co/datasets/{self.repo}/resolve/main"
    
    def _stream_parquet(self, file_rel: str):
        url = f"{self.base_url}/{file_rel}"
        # CDN download — no Authorization header, bypasses API rate limits
        resp = requests.get(url, timeout=30)
        resp.raise_for_status()
        return pq.read_table(io.BytesIO(resp.content))
    
    def __iter__(self):
        for file_rel in self.files:
            try:
                table = self._stream_parquet(file_rel)
                # Project to {prompt, response} only at parse time
                for batch in table.to_batches(max_chunksize=1024):
                    prompts = batch.column("prompt").to_pylist()
                    responses = batch.column("response").to_pylist()
                    for p, r in zip(prompts, responses):
                        if p and r:
                            yield {"prompt": p, "response": r}
            except Exception as e:
                print(f"Skipping {file_rel}: {e}")
                continue
```

### 3. Lightning Studio lifecycle reuse
Reuse running studios and restart automatically on idle-stop.

```python
# surrogate/training/studio_launcher.py
import os
from lightning import Lightning, Teamspace, Machine

LIGHTNING_CLOUD = os.getenv("LIGHTNING_CLOUD", "lightning-public-prod")  # L40S max on free

def get_or_create_studio(name: str):
    lightning = Lightning()
    teamspace = Teamspace()
    
    # Reuse if running
    for s in teamspace.studios:
        if s.name == name and s.status == "Running":
            print(f"Reusing running studio: {name}")
            return s
    
    # Create or restart if stopped
    machine = Machine.L40S if LIGHTNING_CLOUD == "lightning-public-prod" else Machine.H200
    studio = lightning.studio(
        name=name,
        cloud=LIGHTNING_CLOUD,
        machine=machine,
        create_ok=True
    )
    return studio

def run_training_with_resilience(script_path: str, studio_name: str = "surrogate-1-train"):
    studio = get_or_create_studio(studio_name)
    
    # Check status before run — idle-stop kills training
    if studio.status != "Running":
        print(f"Studio stopped, restarting on {studio.machine}")
        studio.start(machine=studio.machine)
    
    # Run with manifest baked in
    result = studio.run(
        [
            "python", script_path,
            "--manifest", "data/file_manifest_2026-04-29.json",
            "--epochs", "3"
        ],
        cwd="/workspace",
        sync=False
    )
    return result
```

### 4. Training script entrypoint
Consume manifest and CDN loader.

```python
# surrogate/training/train.py
import argparse
from pathlib import Path
from surrogate.data.cdn_parquet_loader import CDNParquetDataset
from torch.utils.data import DataLoader

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--manifest", required=True)
    parser.add_argument("--epochs", type=int, default=3)
    parser.add_argument("--batch-size", type=int, default=8)
    args = parser.parse_args()
    
    dataset = CDNParquetDataset(args.manifest, max_files=50)
    loader = DataLoader(dataset, batch_size=args.batch_size, num_workers=0)
    
    print(f"Loaded dataset from {args.manifest}")
    for epoch in range(args.epochs):
        for i, batch in enumerate(loader):
            # Surrogate-1 training step here
            if i % 10 == 0:
                print(f"Epoch {epoch} step {i}: prompts={len(batch['prompt'])}")
    
    print("Training complete (CDN-only, no HF API calls during data load)")

if __name__ == "__main__":
    main()
```
