## Final Consolidated Implementation (Corrected + Actionable)

**Problem**: Surrogate-1 training stalls due to HF API 429s during dataset loading; GPU sits idle while waiting for API pagination.

**Solution**: Eliminate all HF API calls during training by using a **pre-computed CDN-only manifest** and a **streaming Parquet loader** that fetches directly via CDN URLs.

---

### 1) One-Time Manifest Generation (Mac / Dev Machine)

Run once per date folder. Saves a `file_manifest.json` with CDN URLs and sizes.

```python
# scripts/generate_cdn_manifest.py
#!/usr/bin/env python3
import json
import os
from huggingface_hub import HfApi

REPO = "axentx/surrogate-datasets"
DATE_FOLDER = "batches/mirror-merged/2026-05-03"  # change per run
OUT_PATH = "data/file_manifest.json"

def main() -> None:
    api = HfApi()
    items = api.list_repo_tree(repo_id=REPO, path=DATE_FOLDER, recursive=False)

    manifest = []
    for item in items:
        if not item.path.lower().endswith(".parquet"):
            continue
        cdn_url = (
            f"https://huggingface.co/datasets/{REPO}"
            f"/resolve/main/{item.path}"
        )
        manifest.append({
            "repo": REPO,
            "path": item.path,
            "cdn_url": cdn_url,
            "size": getattr(item, "size", None),
        })

    os.makedirs(os.path.dirname(OUT_PATH), exist_ok=True)
    with open(OUT_PATH, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Wrote {len(manifest)} files to {OUT_PATH}")

if __name__ == "__main__":
    main()
```

Run once:

```bash
python scripts/generate_cdn_manifest.py
git add data/file_manifest.json && git commit -m "Add CDN manifest for 2026-05-03"
```

### 2) CDN-Only Streaming Dataset (Corrected)

- **No HF API calls** during training.
- Projects heterogeneous schemas to `{prompt, response}`.
- Uses HTTP range requests via `pyarrow` for efficient streaming.

```python
# surrogate/data/cdn_parquet_dataset.py
import io
import json
import pyarrow.parquet as pq
import requests
from torch.utils.data import IterableDataset
from typing import List, Dict

class CDNParquetDataset(IterableDataset):
    """
    Stream parquet files via CDN URLs with zero HF API calls.
    Projects to {prompt, response} only.
    """

    def __init__(self, manifest_path: str, start_shard: int = 0, end_shard: int = -1):
        with open(manifest_path) as f:
            self.manifest: List[Dict] = json.load(f)
        if end_shard > 0:
            self.manifest = self.manifest[start_shard:end_shard]
        self._session = None

    @property
    def session(self):
        if self._session is None:
            self._session = requests.Session()
        return self._session

    def _stream_parquet(self, cdn_url: str):
        """Download via CDN and yield projected rows as dicts."""
        resp = self.session.get(cdn_url, timeout=60)
        resp.raise_for_status()
        table = pq.read_table(io.BytesIO(resp.content))
        df = table.to_pandas()

        for _, row in df.iterrows():
            yield {
                "prompt": str(row.get("prompt", row.get("input", ""))),
                "response": str(row.get("response", row.get("output", ""))),
            }

    def __iter__(self):
        for item in self.manifest:
            yield from self._stream_parquet(item["cdn_url"])
```

### 3) Lightning DataModule + Studio Integration

- Reuses running Studio when available.
- Uses `target.start(machine=Machine.L40S)` if stopped.
- Loads data exclusively via CDN.

```python
# train.py
import lightning as L
from lightning.pytorch.cli import LightningCLI
from surrogate.data.cdn_parquet_dataset import CDNParquetDataset
from torch.utils.data import DataLoader
from lightning import Machine

class SurrogateDataModule(L.LightningDataModule):
    def __init__(self, manifest_path: str = "data/file_manifest.json", batch_size: int = 8):
        super().__init__()
        self.manifest_path = manifest_path
        self.batch_size = batch_size

    def train_dataloader(self):
        dataset = CDNParquetDataset(self.manifest_path)
        return DataLoader(dataset, batch_size=self.batch_size, num_workers=0)

if __name__ == "__main__":
    cli = LightningCLI(
        SurrogateDataModule,
        run=False,
        save_config_callback=None,
    )

    trainer: L.Trainer = cli.trainer

    # Ensure Studio is running on L40S
    studio = L.studio.get_current_studio()
    if studio and studio.status == "STOPPED":
        studio.start(machine=Machine.L40S)

    trainer.fit(cli.model, cli.datamodule)
```
