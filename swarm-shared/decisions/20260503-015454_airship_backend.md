```python
# surrogate/tools/list_repo_files.py
import json
import os
import sys
from huggingface_hub import HfApi

def list_date_folder(repo_id: str, date_folder: str, out_path: str):
    api = HfApi()
    # non-recursive to avoid pagination explosion
    tree = api.list_repo_tree(repo_id=repo_id, path=date_folder, recursive=False)
    files = [item.rfilename for item in tree if item.type == "file"]
    with open(out_path, "w") as f:
        json.dump({"repo_id": repo_id, "date_folder": date_folder, "files": files}, f)
    print(f"Saved {len(files)} files to {out_path}")

if __name__ == "__main__":
    # usage: python list_repo_files.py <repo_id> <date_folder> <out.json>
    list_date_folder(sys.argv[1], sys.argv[2], sys.argv[3])
```

### 2. CDN-only dataset loader (zero auth, zero API during training)

```python
# surrogate/data/cdn_loader.py
import aiohttp
import asyncio
from pathlib import Path
from typing import List, Dict
import pyarrow as pa
import pyarrow.parquet as pq
import json

HF_CDN_TEMPLATE = "https://huggingface.co/datasets/{repo_id}/resolve/main/{path}"

async def fetch_file(session: aiohttp.ClientSession, repo_id: str, path: str) -> bytes:
    url = HF_CDN_TEMPLATE.format(repo_id=repo_id, path=path)
    async with session.get(url) as resp:
        resp.raise_for_status()
        return await resp.read()

async def load_files_cdn(file_list_path: str, repo_id: str, output_parquet: str):
    with open(file_list_path) as f:
        meta = json.load(f)
    files = meta["files"]

    async with aiohttp.ClientSession() as session:
        tasks = [fetch_file(session, repo_id, f) for f in files]
        blobs = await asyncio.gather(*tasks, return_exceptions=False)

    # Project to {prompt, response} only at parse time
    rows = []
    for path, blob in zip(files, blobs):
        try:
            # heuristic: assume JSON lines with prompt/response
            for line in blob.decode("utf-8").strip().splitlines():
                obj = json.loads(line)
                rows.append({
                    "prompt": obj.get("prompt") or obj.get("input") or "",
                    "response": obj.get("response") or obj.get("output") or "",
                })
        except Exception:
            continue

    table = pa.Table.from_pylist(rows, schema=pa.schema([
        pa.field("prompt", pa.string()),
        pa.field("response", pa.string()),
    ]))
    pq.write_table(table, output_parquet)
    print(f"Wrote {len(rows)} rows to {output_parquet}")
```

### 3. Deterministic sibling sharding for writes

```python
# surrogate/data/sibling_shard.py
import hashlib

def pick_sibling_repo(slug: str, siblings: list) -> str:
    """Deterministic sibling selection to spread HF commit load."""
    digest = int(hashlib.sha256(slug.encode()).hexdigest(), 16)
    idx = digest % len(siblings)
    return siblings[idx]

# usage
SIBLINGS = [
    "axentx/datasets-mirror-0",
    "axentx/datasets-mirror-1",

]
```

### 4. Studio lifecycle guard (reuse + restart)

```python
# surrogate/training/studio_guard.py
from lightning.app import Lightning, Teamspace, Studio, Machine
from lightning.app.utilities.exceptions import NotFoundException

def get_or_create_studio(name: str, machine: Machine = Machine.L40S) -> Studio:
    teamspace = Teamspace()
    for s in teamspace.studios:
        if s.name == name and s.status == "running":
            print(f"Reusing running studio: {name}")
            return s
    # create if not exists
    studio = Studio.create(name=name, machine=machine, create_ok=True)
    print(f"Created studio: {name}")
    return studio

def ensure_running(studio: Studio, machine: Machine = Machine.L40S):
    if studio.status != "running":
        print(f"Studio {studio.name} stopped; restarting on {machine}")
        studio.start(machine=machine)
```

```python
# surrogate/train.py
import os
from pathlib import Path
from surrogate.data.cdn_loader import load_files_cdn
from surrogate.training.studio_guard import get_or_create_studio, ensure_running
from surrogate.data.sibling_shard import pick_sibling_repo

def main():
    repo_id = os.getenv("HF_DATASET_REPO", "axentx/datasets-mirror")
    date_folder = os.getenv("DATE_FOLDER", "2026-05-03")
    file_list = Path("file_list.json")
    out_parquet = Path("batches/mirror-merged") / date_folder / "batch.parquet"
    out_parquet.parent.mkdir(parents=True, exist_ok=True)

    # 1) Pre-listed file paths (generated by mac-side script)
    if not file_list.exists():
        raise RuntimeError("Run list_repo_files.py first to generate file_list.json")

    # 2) CDN-only load (zero HF API calls during training)
    import asyncio
    asyncio.run(load_files_cdn(str(file_list), repo_id, str(out_parquet)))

    # 3) Deterministic sibling write to bypass commit cap
    slug = f"{date_folder}/{out_parquet.name}"
    target_repo = pick_sibling_repo(slug, [
        "axentx/datasets-mirror-0",
        "axentx/datasets-mirror-1",
        "axentx/datasets-mirror-2",
        "axentx/datasets-mirror-3",
        "axentx/datasets-mirror-4",
    ])
    print(f"Selected sibling repo: {target_repo}")

    # 4) Studio guard for training compute
    studio = get_or_create_studio("surrogate-train")
    ensure_running(studio)
    # studio.run(...)  # your training command here

if __name__ == "__main__":
    main()
```
