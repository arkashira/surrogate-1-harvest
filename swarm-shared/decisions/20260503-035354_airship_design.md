## 1) scripts/list_hf_files.py

```python
#!/usr/bin/env python3
"""
One-shot file lister for HF dataset folders.
Usage:
  python scripts/list_hf_files.py \
    --repo axentx/surrogate-data \
    --date 2026-05-03 \
    --out surrogate/data/file_list.json
Must run from Mac (or any env) when HF API rate-limit window is clear.
"""
import argparse
import json
import os
from huggingface_hub import HfApi

def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True, help="HF dataset repo")
    parser.add_argument("--date", required=True, help="Date folder (e.g. 2026-05-03)")
    parser.add_argument("--out", required=True, help="Output JSON path")
    args = parser.parse_args()

    api = HfApi()
    folder = f"batches/mirror-merged/{args.date}"
    # recursive=False avoids heavy pagination; list per folder
    items = api.list_repo_tree(repo_id=args.repo, path=folder, recursive=False)

    files = []
    for item in items:
        if item.rfilename.endswith(".parquet"):
            files.append(f"{folder}/{item.rfilename}")

    out_path = os.path.abspath(args.out)
    os.makedirs(os.path.dirname(out_path), exist_ok=True)
    with open(out_path, "w") as f:
        json.dump({"repo": args.repo, "date": args.date, "files": files}, f, indent=2)
    print(f"Wrote {len(files)} files to {out_path}")

if __name__ == "__main__":
    main()
```

## 2) surrogate/data/cdn_parquet_loader.py

```python
from __future__ import annotations

import json
import pyarrow.parquet as pq
import requests
from io import BytesIO
from typing import Iterator, Dict, Any

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

class CDNParquetLoader:
    """
    CDN-only parquet loader.
    No Authorization header -> bypasses HF API rate limits.
    Projects only {prompt, response} columns.
    """

    def __init__(self, file_list_path: str):
        with open(file_list_path) as f:
            meta = json.load(f)
        self.repo = meta["repo"]
        self.files = meta["files"]

    def _stream_parquet(self, path: str) -> pq.Table:
        url = CDN_TEMPLATE.format(repo=self.repo, path=path)
        resp = requests.get(url, timeout=30)
        resp.raise_for_status()
        return pq.read_table(BytesIO(resp.content))

    def __iter__(self) -> Iterator[Dict[str, Any]]:
        for path in self.files:
            table = self._stream_parquet(path)
            # Project only required fields; ignore mixed schemas in raw files
            if "prompt" in table.column_names and "response" in table.column_names:
                df = table.select(["prompt", "response"]).to_pylist()
                yield from df
            else:
                # fallback: try lowercase
                if "Prompt" in table.column_names and "Response" in table.column_names:
                    df = table.select(["Prompt", "Response"]).to_pylist()
                    for row in df:
                        yield {"prompt": row["Prompt"], "response": row["Response"]}
```

## 3) surrogate/train.py

```python
#!/usr/bin/env python3
"""
Lightning idle-resilient training runner.
- Reuses running studio if available.
- Restarts on idle stop.
- Uses CDN-only loader (zero HF API calls during training).
"""
import os
import sys
import time
from pathlib import Path

from lightning import LightningWork, LightningFlow, LightningApp, Machine
from lightning.app.storage import Drive

# Local imports
from surrogate.data.cdn_parquet_loader import CDNParquetLoader

# Config via env
DATASET_REPO = os.getenv("DATASET_REPO", "axentx/surrogate-data")
DATE_FOLDER = os.getenv("DATE_FOLDER", "2026-05-03")
FILE_LIST = Path(__file__).parent / "data" / "file_list.json"

class SurrogateTrainer(LightningWork):
    def __init__(self):
        super().__init__(
            machine=Machine.L40S,  # free-tier safe; change to H200 in lightning-lambda-prod if quota
            cloud_compute=Machine.L40S,
        )
        self._has_run = False

    def run(self) -> None:
        if self._has_run:
            return

        # Ensure file list exists (Mac prepares this)
        if not FILE_LIST.exists():
            raise FileNotFoundError(f"Missing {FILE_LIST}. Run scripts/list_hf_files.py first.")

        loader = CDNParquetLoader(str(FILE_LIST))

        # Minimal training loop placeholder
        total = 0
        for sample in loader:
            total += 1
            if total % 1000 == 0:
                print(f"Processed {total} samples")
            # Replace with actual surrogate training logic
            if total >= 5000:  # small demo cap
                break

        print(f"Training complete on {total} samples.")
        self._has_run = True

class MonitorFlow(LightningFlow):
    def __init__(self):
        super().__init__()
        self.trainer = SurrogateTrainer()

    def configure_layout(self):
        return []

    def run(self) -> None:
        # If trainer stopped (Lightning idle timeout), restart it
        if self.trainer.has_stopped:
            print("Trainer stopped (idle). Restarting...")
            self.trainer = SurrogateTrainer()
            self._set_work(self.trainer)

app = LightningApp(MonitorFlow())
```
