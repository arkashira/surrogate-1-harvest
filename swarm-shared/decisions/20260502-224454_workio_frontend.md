# workio / frontend

## 1. Diagnosis

- **No resilient offline-first clock-in/out**: current optimistic UI likely relies on localStorage queue (per past decisions) which is fragile, size-limited, and can lose data on storage pressure or cross-tab conflicts.
- **Missing durable sync status + retry**: UI probably shows optimistic state but lacks a visible “syncing / failed / retrying” indicator and exponential backoff for failed mutations.
- **No GPS pre-check before optimistic commit**: clock-in/out can optimistically succeed then fail on server-side GPS validation, causing confusing rollback and bad UX.
- **Race conditions on rapid taps**: double-taps on clock-in/out can create duplicate pending records or conflicting optimistic states.
- **No persisted offline queue across sessions**: if user closes tab before sync completes, pending actions are lost (localStorage-only approach is not durable enough for attendance-critical flows).

## 2. Proposed change

- **File scope**: `src/hooks/useClockInOut.ts` (create if missing) + `src/components/ClockButton.tsx` + `src/lib/idbQueue.ts` (new).
- **Change**: Replace localStorage queue with a small IndexedDB queue for clock-in/out actions, add GPS pre-check, idempotency keys, and visible sync status on the button. Keep optimistic UI but make it durable and retry-aware.

### 3.2 `src/lib/idbQueue.ts`

```ts
import { openDB, DBSchema, IDBPDatabase } from 'idb';

interface WorkioDB extends DBSchema {
  clockQueue: {
    key: string;
    value: {
      id: string;
      action: 'in' | 'out';
      timestamp: number;
      lat: number;
      lng: number;
      retryCount: number;
      createdAt: number;
    };
  };
}

const DB_NAME = 'workio-queue';
const STORE = 'clockQueue';

async function getDB(): Promise<IDBPDatabase<WorkioDB>> {
  return openDB<WorkioDB>(DB_NAME, 1, {
    upgrade(db) {
      db.createObjectStore(STORE, { keyPath: 'id' });
    },
  });
}

export async function enqueueClockAction(
  action: 'in' | 'out',
  lat: number,
  lng: number
) {
  const db = await getDB();
  const id = `${action}-${Date.now()}-${Math.random().toString(36).slice(2, 9)}`;
  await db.add(STORE, {
    id,
    action,
    timestamp: Date.now(),
    lat,
    lng,
    retryCount: 0,
    createdAt: Date.now(),
  });
  return id;
}

export async function peekQueue(): Promise<
  | {
      id: string;
      action: 'in' | 'out';
      timestamp: number;
      lat: number;
      lng: number;
      retryCount: number;
      createdAt: number;
    }
  | undefined
> {
  const db = await getDB();
  const all = await db.getAll(STORE);
  all.sort((a, b) => a.createdAt - b.createdAt);
  return all[0];
}

export async function dequeueClockAction(id: string) {
  const db = await getDB();
  await db.delete(STORE, id);
}

export async function incrementRetry(id: string) {
  const db = await getDB();
  const item = await db.get(STORE, id);
  if (item) {
    item.retryCount += 1;
    await db.put(STORE, item);
  }
}

export async function getAllQueueCount(): Promise<number> {
  const db = await getDB();
  return db.count(STORE);
}
```

### 3.3 `src/hooks/useClockInOut.ts`

```ts
import { useEffect, useState, useCallback } from 'react';
import { enqueueClockAction, peekQueue, dequeueClockAction, incrementRetry, getAllQueueCount } from '../lib/idbQueue';

const MAX_RETRIES = 5;
const BASE_DELAY = 3000;

export function useClockInOut() {
  const [syncing, setSyncing] = useState(false);
  const [queueSize, setQueueSize] = useState(0);
  const [lastError, setLastError] = useState<string | null>(null);

  const syncQueue = useCallback(async () => {
    if (syncing) return;
    const item = await peekQueue();
    if (!item) {
      setQueueSize(await getAllQueueCount());
      return;
    }

    setSyncing(true);
    setLastError(null);

    try {
      const res = await fetch('/api/clock', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          action: item.action,
          timestamp: item.timestamp,
          lat: item.lat,
          lng: item.lng,
        }),
      });

      if (!res.ok) {
        throw new Error(`Server responded ${res.status}`);
      }

      await dequeueClockAction(item.id);
    } catch (err: any) {
      if (item.retryCount >= MAX_RETRIES) {
        // give up and remove to avoid blocking queue forever
        await dequeueClockAction(item.id);
        setLastError('Sync failed after retries — action discarded.');
      } else {
        await incrementRetry(item.id);
        const delay = BASE_DELAY * Math.pow(2, item.retryCount);
        setTimeout(syncQueue, delay);
      }
    } finally {
      setSyncing(false);
      setQueueSize(await getAllQueueCount());
    }
  }, [syncing]);

  // periodic sync + immediate on visibility change
  useEffect(() => {
    syncQueue();
    const iv = setInterval(syncQueue, 15000);
    const onVis = () => {
      if (document.visibilityState === 'visible') syncQueue();
    };
    document.addEventListener('visibilitychange', onVis);
    return () => {
      clearInterval(iv);
      document.removeEventListener('visibilitychange', onVis);
    };
  }, [syncQueue]);

  const clockAction = useCallback(
    async (action: 'in' | 'out', lat: number, lng: number) => {
      // optimistic local record (UI can read queueSize + last action)
      await enqueueClockAction(action, lat, lng);
      setQueueSize(await getAllQueueCount());
      // trigger immediate sync attempt
      syncQueue();
    },
    [syncQueue]
  );

  return { clockAction, syncing, queueSize, lastError, syncQueue };
}
```
