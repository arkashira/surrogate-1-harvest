# workio / ops

## 1. Diagnosis

- **No replay protection / idempotency on LINE webhook events**: duplicate deliveries (network retries, user double-taps) can create duplicate clock-in/out or leave/OT records and corrupt daily totals.
- **Signature lifetime is unbounded**: LINE signatures are accepted without a strict time window, enabling replay attacks beyond the intended freshness window.
- **Event processing is not atomic**: attendance/leave/OT state changes can be partially applied on failure, leaving inconsistent daily summaries and totals.
- **No per-event deduplication store**: missing idempotency key tracking (event id + source) means retries are processed as new events.
- **Missing operational observability**: no structured audit trail for webhook receipt, validation, and processing outcome (success/duplicate/reject) to debug production issues.

## 2. Proposed change

File: `/opt/axentx/workio/server/src/middleware/lineWebhook.ts` (create if absent) and `/opt/axentx/workio/server/src/routes/webhook/line.ts` (or equivalent).  
Scope: add idempotency + signature time-window enforcement + atomic transaction wrapper for event handlers; expose minimal audit log.

## 3. Implementation

```ts
// server/src/middleware/lineWebhook.ts
import crypto from 'crypto';
import { Request, Response, NextFunction } from 'express';
import { Pool } from 'pg';

const pool = new Pool({ connectionString: process.env.DATABASE_URL });
const SIGNATURE_TTL_MS = 5 * 60 * 1000; // 5 minutes
const IDEMPOTENCY_TTL_MS = 24 * 60 * 60 * 1000; // 24h keep dedupe keys

export async function verifyLineSignature(req: Request, res: Response, next: NextFunction) {
  const signature = req.headers['x-line-signature'] as string | undefined;
  const body = JSON.stringify(req.body);
  const secret = process.env.LINE_CHANNEL_SECRET;

  if (!signature || !secret) {
    return res.status(401).json({ error: 'Missing signature or secret' });
  }

  const expected = crypto.createHmac('sha256', secret).update(body).digest('base64');
  if (!crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected))) {
    return res.status(401).json({ error: 'Invalid signature' });
  }

  // Optional: if LINE provides timestamp in header or body, enforce TTL.
  // Many LINE webhooks include `timestamp` in the JSON body.
  const timestamp = (req.body as any)?.timestamp;
  if (timestamp) {
    const ts = Number(timestamp);
    if (Number.isFinite(ts) && Math.abs(Date.now() - ts) > SIGNATURE_TTL_MS) {
      return res.status(400).json({ error: 'Stale event' });
    }
  }

  next();
}

export async function deduplicateLineEvent(req: Request, res: Response, next: NextFunction) {
  const deliveryId = (req.body as any)?.deliveryId || (req.body as any)?.events?.[0]?.source?.userId + (req.body as any)?.events?.[0]?.timestamp;
  if (!deliveryId) return next();

  const idempotencyKey = `line:${crypto.createHash('sha256').update(deliveryId).digest('hex')}`;

  const client = await pool.connect();
  try {
    await client.query('BEGIN');
    // Advisory lock for atomic check+insert (fast, short-lived)
    const lockId = parseInt(idempotencyKey.slice(0, 8), 16) || 1;
    await client.query('SELECT pg_advisory_xact_lock($1)', [lockId]);

    const exists = await client.query(
      `SELECT 1 FROM line_event_dedupe WHERE idempotency_key = $1 AND created_at > NOW() - INTERVAL '${IDEMPOTENCY_TTL_MS} milliseconds'`,
      [idempotencyKey]
    );

    if (exists.rows.length > 0) {
      await client.query('ROLLBACK');
      client.release();
      return res.status(200).json({ ok: true, duplicate: true });
    }

    await client.query(
      `INSERT INTO line_event_dedupe (idempotency_key) VALUES ($1)`,
      [idempotencyKey]
    );

    // Attach client + idempotencyKey to request for downstream commit/rollback
    (req as any)._pgClient = client;
    (req as any)._idempotencyKey = idempotencyKey;
    next();
  } catch (err) {
    await client.query('ROLLBACK');
    client.release();
    next(err);
  }
}

export async function commitWebhookTx(req: Request, res: Response, next: NextFunction) {
  const client = (req as any)._pgClient;
  if (!client) return next();
  try {
    await client.query('COMMIT');
    client.release();
    next();
  } catch (err) {
    await client.query('ROLLBACK');
    client.release();
    next(err);
  }
}

export async function rollbackWebhookTx(req: Request, res: Response, next: NextFunction) {
  const client = (req as any)._pgClient;
  if (client) {
    try { await client.query('ROLLBACK'); } catch {}
    client.release();
  }
  next();
}
```

```ts
// server/src/routes/webhook/line.ts
import express from 'express';
import { verifyLineSignature, deduplicateLineEvent, commitWebhookTx, rollbackWebhookTx } from '../../middleware/lineWebhook';
import { processLineEvents } from '../../services/lineEvents';

const router = express.Router();

router.post(
  '/line',
  verifyLineSignature,
  deduplicateLineEvent,
  async (req, res, next) => {
    try {
      await processLineEvents(req.body, (req as any)._pgClient);
      res.json({ ok: true });
    } catch (err) {
      next(err);
    }
  },
  commitWebhookTx,
  (err: any, req: express.Request, res: express.Response, next: express.NextFunction) => {
    // structured audit log
    console.error('LINE webhook error', {
      path: req.path,
      method: req.method,
      body: req.body,
      error: err?.message,
      stack: err?.stack,
    });
    // ensure rollback middleware runs
    rollbackWebhookTx(req, res, () => {
      res.status(500).json({ error: 'Processing failed' });
    });
  }
);

export default router;
```

1. **Schema**: run `psql workio < server/src/db/schema.sql` and confirm `line_event_dedupe` exists.
2. **Signature rejection**: send a POST with wrong/missing `X-Line-Signature` → expect 401.
3. **Replay rejection**:
   - Send valid webhook payload → expect 200 `{ok:true}`.
   - Immediately resend identical payload (same body) → expect 200 `{ok:true, duplicate:true}`
