## Code

### `/opt/axentx/airship/airship`

```bash
#!/usr/bin/env bash
# airship - Arkship/Surrogate platform CLI
# Usage: airship discover [--json] [--output-dir=/opt/axentx/airship/reports]
# Tags: #business-research #knowledge-rag #graph #hub

set -euo pipefail
export SHELL=/bin/bash

ROOT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
REPORTS_DIR="${ROOT_DIR}/reports"

mkdir -p "${REPORTS_DIR}"

log() {
  echo "[$(date -u +%Y-%m-%dT%H:%M:%SZ)] [airship] $*" >&2
}

fail() {
  log "ERROR: $*"
  exit 1
}

# ---- Stage 1: Business Research ----
stage1_business_research() {
  local out="$1"
  log "Stage 1: Running granite-business-research.sh"
  local script="${ROOT_DIR}/scripts/granite-business-research.sh"
  if [[ -x "$script" ]]; then
    bash "$script" > "$out" || fail "Business research failed"
  else
    log "WARN: granite-business-research.sh not found/executable, skipping"
    echo '{}' > "$out"
  fi
}

# ---- Stage 2: Knowledge RAG Query ----
stage2_knowledge_rag() {
  local out="$1"
  log "Stage 2: Querying knowledge-rag for top hub and related docs"
  local script="${ROOT_DIR}/scripts/knowledge-rag.sh"
  if [[ -x "$script" ]]; then
    bash "$script" --query="top hub and strategic context for Arkship Surrogate platform" \
      --output="$out" || fail "Knowledge RAG query failed"
  else
    log "WARN: knowledge-rag.sh not found/executable, skipping"
    echo '{"results":[]}' > "$out"
  fi
}

# ---- Stage 3: Top-Hub (MOC) Review ----
stage3_top_hub_review() {
  local kg_file="$1"
  local out="$2"
  log "Stage 3: Extracting top-hub (MOC) insight"
  if [[ -f "$kg_file" ]]; then
    if command -v jq >/dev/null 2>&1; then
      jq '{top_hub: .results[0].hub // "MOC", connections: .results[0].connections // 0, insight: .results[0].summary // ""}' \
        "$kg_file" > "$out" 2>/dev/null || echo '{"top_hub":"MOC","connections":0,"insight":"Manual review recommended"}' > "$out"
    else
      local hub
      hub=$(grep -o '"hub":"[^"]*"' "$kg_file" 2>/dev/null | head -1 | sed 's/"hub":"//;s/"//' || true)
      hub="${hub:-MOC}"
      echo "{\"top_hub\":\"$hub\",\"connections\":0,\"insight\":\"Manual review recommended\"}" > "$out"
    fi
  else
    echo '{"top_hub":"MOC","connections":0,"insight":"No knowledge-rag output available"}' > "$out"
  fi
  log "Top hub: $(jq -r '.top_hub' "$out" 2>/dev/null || cat "$out")"
}

# ---- HF CDN Bypass Helper (if research pulls external data) ----
hf_cdn_download() {
  local repo="$1"
  local path="$2"
  local out="$3"
  curl -fsSL "https://huggingface.co/datasets/${repo}/resolve/main/${path}" -o "$out" || {
    log "WARN: HF CDN download failed for $repo/$path"
    return 1
  }
}

# ---- Lightning Studio Reuse ----
reuse_lightning_studio() {
  local out="$1"
  log "Checking for reusable Lightning Studio (quota optimization)"
  local script="${ROOT_DIR}/scripts/lightning-studio-reuse.sh"
  if [[ -x "$script" ]]; then
    bash "$script" --check-running --output="$out" || log "WARN: Lightning studio reuse check failed"
  else
    log "INFO: No lightning-studio-reuse.sh found, skipping"
    echo 'null' > "$out"
  fi
}

# ---- Main ----
discover() {
  local timestamp
  timestamp=$(date -u +%Y%m%dT%H%M%SZ)
  local run_dir="${REPORTS_DIR}/${timestamp}"
  mkdir -p "$run_dir"

  log "Starting airship discovery (run: $timestamp)"

  stage1_business_research "$run_dir/business-research.json"
  stage2_knowledge_rag "$run_dir/knowledge-rag.json"
  stage3_top_hub_review "$run_dir/knowledge-rag.json" "$run_dir/top-hub-moc.json"
  reuse_lightning_studio "$run_dir/lightning-studio.json"

  # Generate summary JSON
  local summary_file="${run_dir}/summary.json"
  cat > "$summary_file" <<EOF
{
  "run_id": "$timestamp",
  "root_dir": "$ROOT_DIR",
  "business_research": $(cat "$run_dir/business-research.json"),
  "knowledge_rag": $(cat "$run_dir/knowledge-rag.json"),
  "top_hub": $(cat "$run_dir/top-hub-moc.json"),
  "lightning_studio": $(cat "$run_dir/lightning-studio.json"),
  "generated_at": "$(date -u +%Y-%m-%dT%H:%M:%SZ)"
}
EOF

  # Human-readable Markdown report
  local report_file="${run_dir}/report.md"
  local top_hub top_conns top_insight
  top_hub=$(jq -r '.top_hub' "$run_dir/top-hub-moc.json" 2>/dev/null || echo "MOC")
  top_conns=$(jq -r '.connections' "$run_dir/top-hub-moc.json" 2>/dev/null || echo "0")
  top_insight=$(jq -r '.insight' "$run_dir/top-hub-moc.json" 2>/dev/null || echo "Manual review recommended")
  local biz_content kg_content lightning_status
  biz_content=$(jq -r '.' "$run_dir/business-research.json" 2>/dev/null || cat "$run_dir/business-research.json")
  kg_content=$(jq -r '.' "$run_dir/knowledge-rag.json" 2>/dev/null || cat "$run_dir/knowledge-rag.json")
  lightning_status=$(jq -r '.' "$run_dir/lightning-studio.json" 2>/dev/null || cat "$run_dir/lightning-studio.json")

  cat > "$report_file" <<REPORT
# Arkship Discovery Report

## Run Info
- **Run ID**: $timestamp
- **Generated**: $(date -u +%Y-%m-%dT%H:%M:%SZ)

## Business Research
\`\`\`json
$biz_content
\`\`\`

## Knowledge Graph Insights
\`\`\`json
$kg_content
\`\`\`

## Top Hub (MOC)
- **Hub**:
