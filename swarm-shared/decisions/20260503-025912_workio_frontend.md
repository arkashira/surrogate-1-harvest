# workio / frontend

## Final synthesized solution

**Core diagnosis (merged & prioritized)**
- Race-condition duplicate punches from rapid clicks and retries (both candidates).
- No client-side idempotency plumbing → backend idempotency cannot be used effectively (both).
- No optimistic UI + rollback → users retry and compound duplicates (both).
- Stale/async state from webhooks/line pushes and missing recoverable location flows (C1 + C2).
- Tenant drift risk and missing guard for tenant context (C2).

**Chosen scope (concrete, minimal, high-leverage)**
- Create `/opt/axentx/workio/src/features/punch/usePunch.ts`
- Create `/opt/axentx/workio/src/hooks/useOnlineQueue.ts`
- Update `/opt/axentx/workio/src/api/punch.ts`
- Update `/opt/axentx/workio/src/features/punch/PunchButton.tsx`
- Add lightweight route-level tenant guard (recommended file: `/opt/axentx/workio/src/lib/tenant.ts` + usage in routes/layouts).

## 1. API layer — idempotency + typed requests

```ts
import axios from 'axios';

const api = axios.create({ baseURL: '/api' });

export type PunchType = 'CLOCK_IN' | 'CLOCK_OUT';
export type PunchRequest = {
  type: PunchType;
  latitude?: number | null;
  longitude?: number | null;
  accuracy?: number | null;
};

export async function createPunch(
  payload: PunchRequest,
  idempotencyKey: string,
  signal?: AbortSignal
) {
  const res = await api.post<{ id: string; type: PunchType; createdAt: string }>(
    '/punches',
    payload,
    {
      headers: { 'Idempotency-Key': idempotencyKey },
      signal,
    }
  );
  return res.data;
}
```

## 2. Client-side queue — idempotent, retry, persist, abort

`/opt/axentx/workio/src/hooks/useOnlineQueue.ts`
```ts
import { useEffect, useRef, useCallback } from 'react';

type QueueItem<T = unknown> = {
  id: string;
  fn: (signal?: AbortSignal) => Promise<T>;
  retries: number;
};

const MAX_RETRIES = 3;
const STORAGE_KEY = 'axentx_workio_queue_v1';

function loadQueue<T>(): QueueItem<T>[] {
  try {
    const raw = localStorage.getItem(STORAGE_KEY);
    return raw ? JSON.parse(raw) : [];
  } catch {
    return [];
  }
}

export function useOnlineQueue() {
  const queueRef = useRef<QueueItem[]>(loadQueue());
  const processingRef = useRef(false);

  const persist = useCallback(() => {
    try {
      localStorage.setItem(STORAGE_KEY, JSON.stringify(queueRef.current));
    } catch {
      // ignore storage failures
    }
  }, []);

  // hydrate once
  useEffect(() => {
    // already loaded in ref; keep in sync if other tabs modify storage
    const onStorage = () => {
      queueRef.current = loadQueue();
    };
    window.addEventListener('storage', onStorage);
    return () => window.removeEventListener('storage', onStorage);
  }, []);

  const enqueue = useCallback(<T,>(fn: (signal?: AbortSignal) => Promise<T>) => {
    const id = crypto.randomUUID();
    const item: QueueItem<T> = { id, fn: (signal) => fn(signal), retries: 0 };
    queueRef.current.push(item);
    persist();
    void process();
    return id;
  }, [persist]);

  const process = useCallback(async () => {
    if (processingRef.current || queueRef.current.length === 0) return;
    processingRef.current = true;

    while (queueRef.current.length > 0) {
      const item = queueRef.current[0];
      const controller = new AbortController();

      try {
        await item.fn(controller.signal);
        queueRef.current.shift();
        persist();
      } catch (err) {
        item.retries += 1;
        if (item.retries >= MAX_RETRIES) {
          queueRef.current.shift(); // drop after max retries
        }
        persist();
        // exponential backoff
        await new Promise((r) => setTimeout(r, 1000 * 2 ** Math.min(item.retries, 5)));
      }
    }

    processingRef.current = false;
  }, [persist]);

  // retry on online
  useEffect(() => {
    const onOnline = () => void process();
    window.addEventListener('online', onOnline);
    return () => window.removeEventListener('online', onOnline);
  }, [process]);

  return { enqueue };
}
```

## 3. Punch hook — optimistic UI, idempotency, tenant-aware

`/opt/axentx/workio/src/features/punch/usePunch.ts`
```ts
import { useState, useCallback, useEffect } from 'react';
import { createPunch, PunchRequest, PunchType } from '../../api/punch';
import { useOnlineQueue } from '../../hooks/useOnlineQueue';
import { getCurrentTenantId } from '../../lib/tenant';

export type PunchState = 'idle' | 'pending' | 'success' | 'error';

export function usePunch() {
  const [state, setState] = useState<PunchState>('idle');
  const [lastPunch, setLastPunch] = useState<null | {
    id: string;
    type: PunchType;
    createdAt: string;
  }>(null);
  const [error, setError] = useState<string | null>(null);
  const { enqueue } = useOnlineQueue();

  const punch = useCallback(
    async (type: PunchType, location?: PunchRequest['latitude'] extends number ? { latitude: number; longitude: number; accuracy?: number } : never) => {
      if (state === 'pending') return; // guard rapid clicks
      setError(null);
      setState('pending');

      const tenantId = getCurrentTenantId();
      if (!tenantId) {
        setState('error');
        setError('Tenant context missing');
        return;
      }

      const idempotencyKey = `${tenantId}-${type}-${Date.now()}-${crypto.randomUUID()}`;
      const payload: PunchRequest = { type, ...location };

      // optimistic update
      const optimistic = { id: `optimistic-${idempotencyKey}`, type, createdAt: new Date().toISOString() };
      setLastPunch(optimistic);

      const id = enqueue(async (signal?) => {
        const result = await createPunch({ ...payload, tenantId }, idempotencyKey, signal);
        setLastPunch(result);
        setState('success');
      });

      if (!id) {
        setState('error');
        setError('Failed to enqueue punch');
      }
    },
    [state, enqueue]
  );

  const reset = useCallback(() => {
    setState('idle');
    setError(null);
  }, []);

  // optional: sync from server periodically or via websocket/line push events
  useEffect(() => {
    // placeholder for subscription to punch events (e.g., SSE/websocket)
    // onPunchEvent((punch) => setLastPunch(punch));
  }, []);

  return { punch, state, lastPunch, error, reset };
}
```
