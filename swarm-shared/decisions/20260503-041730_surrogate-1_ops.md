## Final Implementation Plan (≤2h)

**Highest-value improvement**: Replace fragile shell-based ingestion with a **manifest-driven, CDN-bypass Python worker** that eliminates HF API rate limits during training data loads and prevents mixed-schema CastErrors.

### Changes

1. Add `bin/manifest.py` — Mac-side script that lists one date folder via HF API (after rate-limit window), writes `manifest.json` with CDN URLs and expected row counts.
2. Replace `bin/dataset-enrich.sh` with `bin/worker.py` — deterministic shard worker that:
   - Reads `manifest.json`
   - Filters rows by `hash(shard_key) % 16 == SHARD_ID`
   - Downloads assigned parquet/jsonl files via **raw CDN URLs** (no Authorization header → bypasses API rate limit)
   - Projects to `{prompt, response}` only at parse time (avoids pyarrow CastError on mixed schemas)
   - Dedups via `lib/dedup.py`
   - Emits `batches/public-merged/<date>/shard<N>-<HHMMSS>.jsonl`
3. Update `.github/workflows/ingest.yml` to:
   - Accept `MANIFEST_URL` or `MANIFEST_PATH` input
   - Pass `SHARD_ID` matrix index 0..15
   - Use `python bin/worker.py` instead of shell script
4. Add `requirements.txt` entries: `requests`, `pyarrow`, `pandas`, `tqdm`

#### `bin/manifest.py` (run on Mac, once per date folder)

```python
#!/usr/bin/env python3
"""
Generate manifest for one date folder.
Usage:
  python bin/manifest.py \
    --repo axentx/surrogate-1-training-pairs \
    --folder batches/public-raw/2026-05-03 \
    --out manifest.json
"""
import argparse
import json
import os
import sys
import time
from typing import List, Dict

from huggingface_hub import HfApi

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def list_folder_files(repo: str, folder: str) -> List[Dict]:
    api = HfApi()
    items = api.list_repo_tree(repo=repo, path=folder, recursive=False)
    files = [i for i in items if i.type == "file"]
    out = []
    for f in files:
        out.append({
            "path": f.path,
            "cdn_url": CDN_TEMPLATE.format(repo=repo, path=f.path),
            "size": getattr(f, "size", None),
        })
    return out

def build_manifest(repo: str, folder: str, out_path: str) -> None:
    files = list_folder_files(repo, folder)
    if not files:
        print(f"No files in {repo}/{folder}", file=sys.stderr)
        sys.exit(1)

    manifest = {
        "repo": repo,
        "folder": folder,
        "generated_at": time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime()),
        "files": files,
    }
    os.makedirs(os.path.dirname(os.path.abspath(out_path)), exist_ok=True)
    with open(out_path, "w") as fh:
        json.dump(manifest, fh, indent=2)
    print(f"Manifest written to {out_path} ({len(files)} files)")

if __name__ == "__main__":
    p = argparse.ArgumentParser()
    p.add_argument("--repo", default="axentx/surrogate-1-training-pairs")
    p.add_argument("--folder", required=True)
    p.add_argument("--out", default="manifest.json")
    args = p.parse_args()
    build_manifest(args.repo, args.folder, args.out)
```

#### `bin/worker.py` (replaces `dataset-enrich.sh`)

```python
#!/usr/bin/env python3
"""
Deterministic shard worker.
Environment:
  SHARD_ID=0..15
  MANIFEST_PATH=manifest.json
  HF_TOKEN=... (for dedup store push only; CDN downloads are anonymous)
  DATE=2026-05-03
"""
import json
import os
import sys
import hashlib
import tempfile
from pathlib import Path
from typing import Any, Dict, List

import pyarrow as pa
import pyarrow.parquet as pq
import pandas as pd
import requests
from tqdm import tqdm

# local
sys.path.insert(0, str(Path(__file__).parent))
from lib.dedup import DedupStore  # noqa

CDN_TIMEOUT = 30
CHUNK_SIZE = 8 * 1024 * 1024

def shard_key(slug: str) -> int:
    return int(hashlib.md5(slug.encode()).hexdigest(), 16)

def select_shard_rows(df: pd.DataFrame, shard_id: int) -> pd.DataFrame:
    if df.empty:
        return df
    keys = df["slug"].astype(str)
    mask = keys.apply(lambda s: shard_key(s) % 16 == shard_id)
    return df[mask].copy()

def project_to_pair(row: Dict[str, Any]) -> Dict[str, Any]:
    """Return {prompt, response} only; tolerate mixed schemas."""
    prompt = row.get("prompt") or row.get("input") or row.get("question") or ""
    response = row.get("response") or row.get("output") or row.get("answer") or ""
    return {"prompt": str(prompt), "response": str(response)}

def download_cdn_parquet(url: str) -> pd.DataFrame:
    r = requests.get(url, timeout=CDN_TIMEOUT, stream=True)
    r.raise_for_status()
    with tempfile.NamedTemporaryFile(suffix=".parquet", delete=False) as tmp:
        for chunk in r.iter_content(chunk_size=CHUNK_SIZE):
            tmp.write(chunk)
        tmp_path = tmp.name
    try:
        table = pq.read_table(tmp_path)
        return table.to_pandas()
    finally:
        os.unlink(tmp_path)

def download_cdn_jsonl(url: str) -> pd.DataFrame:
    rows = []
    r = requests.get(url, timeout=CDN_TIMEOUT, stream=True)
    r.raise_for_status()
    for line in r.iter_lines(decode_unicode=True):
        line = line.strip()
        if line:
            rows.append(json.loads(line))
    return pd.DataFrame(rows)

def load_file(entry: Dict[str, str]) -> pd.DataFrame:
    url = entry["cdn_url"]
    if url.endswith(".parquet"):
        return download_cdn_parquet(url)
    elif url.endswith(".jsonl"):
        return download_cdn_jsonl(url)
    else:
        raise ValueError(f"Unsupported file type: {url}")

def main() -> None:
    shard_id = int(os.environ.get("SHARD_ID", 0))
    manifest_path = os.environ.get("MANIFEST_PATH", "manifest.json")
    date_str = os.environ.get("DATE", "2026-05-03")
    hf_token = os.environ.get("HF_TOKEN")

    with open(manifest_path) as fh:
        manifest = json.load(fh)

    dedup = DedupStore(hf_token=hf_token) if hf_token else None

    all_pairs: List[Dict[str, str]] = []
    for entry in tqdm(manifest["files"], desc="Downloading"):
        try:
            df = load_file(entry)
        except Exception as exc:
            print(f"Skipping {entry['path']}: {exc}", file=sys.stderr)
            continue

        if df.empty:
            continue

        # ensure slug exists for sharding
        if "slug" not in df.columns:
            df["slug"] = entry["path"]

        shard_df = select_shard_rows(df, shard_id)
        if shard_df.empty:
            continue

        # project to pair and dedup
        for _, row in shard_df.iterrows():
            pair = project_to_pair(row.to_dict())
            if not pair["prompt"] or not pair["response"]:
                continue
            text = f"{pair['prompt']}\n{pair['response']}"
            md5 = hashlib.md5(text.encode()).hexdigest()
            if dedup is None or dedup.add(md5):
                pair["slug"] = row["slug"]
                all_pairs.append(pair)

    if not all_pairs:
        print("No pairs for shard", shard_id, file=sys.stderr)
