## Implementation Steps (Concrete & Actionable)

### 1) Orchestration launcher (`scripts/launch_training.py`)
- Uses HF API **once** (after rate-limit window) to `list_repo_tree(path, recursive=False)` for a single date folder → saves `file_list.json`.  
- Embeds `file_list.json` into the Lightning job.  
- Reuses a running Studio if present; else creates one with L40S (falls back to free-tier clouds in priority order).  
- Sets `SHELL=/bin/bash` in environment and ensures scripts are executable.

```python
#!/usr/bin/env python3
"""
Launch Surrogate training with HF CDN-bypass and Lightning Studio reuse.
Mac/CLI only — heavy compute runs on Lightning.
"""
import os
import json
import subprocess
import sys
from pathlib import Path

# Optional: HF API only once to list a single date folder
try:
    from huggingface_hub import HfApi, list_repo_tree
except ImportError:
    subprocess.check_call([sys.executable, "-m", "pip", "install", "huggingface_hub"])
    from huggingface_hub import HfApi, list_repo_tree

from lightning import Lightning, Teamspace, Machine, Studio  # pseudo-import; adjust to actual SDK

HF_REPO = os.getenv("HF_DATASET_REPO", "axentx/surrogate-training")
DATE_FOLDER = os.getenv("TRAIN_DATE_FOLDER", "2026-05-03")  # e.g. 2026-05-03
OUTPUT_FILE_LIST = "file_list.json"
STUDIO_NAME = os.getenv("STUDIO_NAME", "surrogate-training-studio")

def list_files_once() -> list[str]:
    """Single API call to list files in DATE_FOLDER (non-recursive)."""
    api = HfApi()
    # list_repo_tree(path, recursive=False) avoids paginated file listing on big repos
    tree = list_repo_tree(repo_id=HF_REPO, path=DATE_FOLDER, recursive=False)
    files = [item.rfilename for item in tree if not item.rfilename.endswith("/")]
    with open(OUTPUT_FILE_LIST, "w") as f:
        json.dump(files, f)
    print(f"Listed {len(files)} files -> {OUTPUT_FILE_LIST}")
    return files

def ensure_bash_wrapper(script_path: Path):
    """Ensure wrapper has shebang and is executable."""
    if not script_path.exists():
        return
    content = script_path.read_text()
    if not content.startswith("#!"):
        content = "#!/usr/bin/env bash\n" + content
        script_path.write_text(content)
    script_path.chmod(0o755)

def reuse_or_create_studio():
    """Reuse running studio if available; else create with prioritized cloud/size."""
    teamspace = Teamspace()
    for s in teamspace.studios:
        if s.name == STUDIO_NAME and s.status == "Running":
            print(f"Reusing running studio: {STUDIO_NAME}")
            return s

    # Priority order: Lightning-Lambda-Prod (H200) -> public-prod (L40S) -> others
    clouds = ["lightning-lambda-prod", "lightning-public-prod"]
    machine = None
    for cloud in clouds:
        try:
            machine = Machine(cloud=cloud, name="L40S")
            print(f"Selected machine: {cloud}/L40S")
            break
        except Exception:
            continue
    if machine is None:
        machine = Machine(cloud="lightning-public-prod", name="L40S")

    studio = Studio.create(
        name=STUDIO_NAME,
        machine=machine,
        create_ok=True,
    )
    print(f"Created studio: {STUDIO_NAME} on {machine.cloud}/{machine.name}")
    return studio

def main():
    os.environ["SHELL"] = "/bin/bash"
    files = list_files_once()
    if not files:
        print("No files found; aborting.")
        sys.exit(1)

    studio = reuse_or_create_studio()

    # Ensure training wrapper is executable and has shebang
    train_wrapper = Path("training/run_training.sh")
    ensure_bash_wrapper(train_wrapper)

    # Run training in studio (adjust command to your actual entrypoint)
    cmd = [
        "bash", str(train_wrapper),
        "--file-list", OUTPUT_FILE_LIST,
        "--hf-repo", HF_REPO,
        "--date-folder", DATE_FOLDER,
    ]
    print("Running in studio:", " ".join(cmd))
    # Studio.run executes inside the Lightning environment
    job = studio.run(
        cmd,
        cwd="/workspace",
        env={
            "HF_DATASET_REPO": HF_REPO,
            "TRAIN_DATE_FOLDER": DATE_FOLDER,
            "SHELL": "/bin/bash",
        },
    )
    print(f"Started job: {job.id}")
    return job

if __name__ == "__main__":
    main()
```

### 2) Training script patch (`training/train.py` — CDN-only loader)
- Accepts `file_list_json` path or inline list.  
- During `DataLoader` init, downloads via **CDN URLs only** (`https://huggingface.co/datasets/{repo}/resolve/main/{path}`) with `requests`/`urllib` (no auth, no API calls).  
- Projects to `{prompt, response}` at parse time; ignores heterogeneous schemas.  
- Writes outputs to `batches/mirror-merged/{date}/{slug}.parquet` (no extra metadata columns).

```python
import json
import os
from pathlib import Path
from typing import List, Dict

import pyarrow as pa
import pyarrow.parquet as pq
import requests
from torch.utils.data import Dataset, DataLoader

HF_REPO = os.getenv("HF_DATASET_REPO", "axentx/surrogate-training")
CDN_BASE = f"https://huggingface.co/datasets/{HF_REPO}/resolve/main"

class CDNBackedDataset(Dataset):
    """Load files via CDN only (no HF API/auth). Project to {prompt, response}."""
    def __init__(self, file_list: List[str], date_folder: str):
        self.file_list = file_list
        self.date_folder = date_folder
        self.items = self._load_all()

    def _download_one(self, rel_path: str) -> Dict:
        url = f"{CDN_BASE}/{rel_path}"
        resp = requests.get(url, timeout=30)
        resp.raise_for_status()
        # Assume JSONL or JSON; adapt if needed
        try:
            data = resp.json()
        except Exception:
            # fallback: try JSONL lines
            lines = [ln.strip() for ln in resp.text.strip().splitlines() if ln.strip()]
            data = [json.loads(ln) for ln in lines]
        return data

    def _project(self, raw) -> List[Dict]:
        """Project heterogeneous raw records to {prompt, response} only."""
        out = []
        if isinstance(raw, dict):
            raw = [raw]
        for rec in raw:
            prompt = rec.get("prompt") or rec.get("input") or rec.get("question") or ""
            response = rec.get("response") or rec.get("output") or rec.get("answer") or ""
            if prompt and response:
                out.append({"prompt": str(prompt), "response": str(response)})
        return out

    def _load_all(self) -> List[Dict]:
        rows = []
        for rel_path in self.file_list:
            raw = self._download_one(rel_path)
            rows.extend(self._project(raw))
        return rows

    def __len__(self) -> int:
        return len(self.items)

    def __getitem__(self, idx: int) -> Dict:
        return self.items[idx]
