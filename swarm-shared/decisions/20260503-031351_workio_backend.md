# workio / backend

## 1. Diagnosis
- Missing `X-Line-Signature` verification on `/webhook/line` allows spoofed/replayed clock/leave/OT events.
- No idempotency/replay protection; LINE retries on 5xx/timeouts create duplicate punches and leave/OT records.
- No request deduplication key in DB layer → race conditions on double-tap or retry storms.
- Webhook handler mixes validation, business logic, and persistence → hard to test and extend.
- No structured logging/tracing for webhook delivery outcomes (success/retry/fraud).

## 2. Proposed change
File: `workio/server/src/routes/lineWebhook.ts` (create)  
Scope: add middleware-style handler that (a) verifies `X-Line-Signature`, (b) enforces idempotency via `X-Line-Request-Id` (or hash of body+timestamp), (c) persists dedupe key, then (d) delegates to domain services. Expose as `/webhook/line` POST.

### `workio/server/src/middleware/lineSignature.ts`
```ts
import crypto from 'crypto';

const LINE_CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';

export function verifyLineSignature(rawBody: string, signature: string): boolean {
  if (!LINE_CHANNEL_SECRET) {
    // In dev without secret, allow but log
    console.warn('LINE_CHANNEL_SECRET not set — skipping signature verification');
    return true;
  }
  const expected = crypto
    .createHmac('sha256', LINE_CHANNEL_SECRET)
    .update(rawBody, 'utf8')
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}
```

### `workio/server/src/services/lineWebhookService.ts`
```ts
import { Pool } from 'pg';
import { verifyLineSignature } from '../middleware/lineSignature.js';

const pool = new Pool({ connectionString: process.env.DATABASE_URL });

export async function handleLineWebhook(rawBody: string, signature: string, requestId?: string) {
  // 1) Verify signature
  if (!verifyLineSignature(rawBody, signature)) {
    const err = new Error('Invalid LINE signature') as any;
    err.status = 401;
    throw err;
  }

  // 2) Parse events
  let events;
  try {
    events = JSON.parse(rawBody).events;
    if (!Array.isArray(events)) throw new Error('Invalid events payload');
  } catch (err: any) {
    err.status = 400;
    throw err;
  }

  // 3) Idempotency: use X-Line-Request-Id or hash(body)
  const dedupeKey = requestId || crypto.createHash('sha256').update(rawBody).digest('hex');
  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // Check dedupe table (create if missing)
    await client.query(`
      CREATE TABLE IF NOT EXISTS line_webhook_dedupe (
        dedupe_key TEXT PRIMARY KEY,
        created_at TIMESTAMPTZ DEFAULT NOW()
      );
    `);

    const exists = await client.query(
      'SELECT 1 FROM line_webhook_dedupe WHERE dedupe_key = $1',
      [dedupeKey]
    );
    if (exists.rows.length > 0) {
      await client.query('ROLLBACK');
      return { processed: false, reason: 'duplicate' };
    }

    // Insert dedupe record early (prevents concurrent retries)
    await client.query('INSERT INTO line_webhook_dedupe (dedupe_key) VALUES ($1)', [dedupeKey]);

    // 4) Process each event (minimal mapping to domain actions)
    for (const ev of events) {
      await processLineEvent(client, ev);
    }

    await client.query('COMMIT');
    return { processed: true, events: events.length };
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  } finally {
    client.release();
  }
}

async function processLineEvent(client: any, ev: any) {
  // Map LINE event types to domain actions
  const type = ev.type;
  const userId = ev.source?.userId;
  const ts = ev.timestamp ? new Date(ev.timestamp) : new Date();

  if (type === 'message' && ev.message?.type === 'text') {
    const text = ev.message.text?.trim().toLowerCase();
    // Example: simple keyword handling; replace with real domain service calls
    if (text === 'clock in' || text === 'clock out') {
      await client.query(
        `INSERT INTO time_logs (tenant_id, user_id, event_type, created_at, source, raw_event)
         VALUES ($1, $2, $3, $4, $5, $6)`,
        ['default-tenant', userId, text === 'clock in' ? 'clock_in' : 'clock_out', ts, 'line', JSON.stringify(ev)]
      );
    }
  }

  // Add leave/OT command parsing here or delegate to domain services
}
```

### `workio/server/src/routes/lineWebhook.ts`
```ts
import express, { Request, Response, NextFunction } from 'express';
import { handleLineWebhook } from '../services/lineWebhookService.js';

const router = express.Router();

router.post('/webhook/line', express.raw({ type: 'application/json' }), async (req: Request, res: Response, next: NextFunction) => {
  try {
    const rawBody = req.body.toString('utf8');
    const signature = req.headers['x-line-signature'] as string || '';
    const requestId = req.headers['x-line-request-id'] as string || '';

    const result = await handleLineWebhook(rawBody, signature, requestId);
    if (result.processed === false && result.reason === 'duplicate') {
      return res.status(200).json({ ok: true, duplicate: true });
    }
    return res.status(200).json({ ok: true, ...result });
  } catch (err: any) {
    console.error('LINE webhook error:', err);
    const status = err.status || 500;
    return res.status(status).json({ ok: false, error: err.message });
  }
});

export default router;
```

## 4. Verification
1. Start server: `cd workio/server && npm run dev`
2. Send valid test webhook (use ngrok or local tunnel):
   ```bash
   BODY='{"events":[{"type":"message","source":{"userId":"U123"},"message":{"type":"text","text":"clock in"},"timestamp":1712345678901}]}'
   SIG=$(echo -n "$BODY" | openssl dgst -sha256 -hmac "$LINE_CHANNEL_SECRET" -binary | base64)
   curl -X POST http://localhost:3000/webhook/line \
     -H "Content-Type: application/json" \
     -H "X-Line-Signature: $SIG" \
     -d "$BODY"
   ```
   Expect `200 OK` and a row in `time_logs`.

3. Replay same request (same body + signature) → expect `200 OK` with `duplicate: true` and no new `time_logs` row.

4. Tamper signature or body → expect `401`.

5. Check logs for errors and confirm `line_webhook_dedupe` contains dedupe keys.
