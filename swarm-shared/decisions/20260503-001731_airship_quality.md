Below is the **single, canonical implementation** that merges the strongest, non-contradictory parts of both proposals.  
It prioritizes **correctness** (retry/backoff, schema validation, minimal HF API surface) and **concrete actionability** (exact file layout, CLI usage, Lightning-ready).

## Summary (why this wins)
- **Eliminates HF API 429s** during training: zero `load_dataset` calls in the training loop.
- **<2h change**: one new dataset file, one small script, one-line swap in training code.
- **Robust**: CDN retries, schema checks, skips corrupt files instead of crashing.
- **Portable**: works in Lightning Studio and local runs with identical behavior.

## 1) New dataset class (CDN-only, Iterable)

File: `surrogate/training/hf_cdn_dataset.py`

```python
# surrogate/training/hf_cdn_dataset.py
import io
import json
import time
from pathlib import Path
from typing import Dict, List

import pyarrow.parquet as pq
import requests
import torch
from torch.utils.data import IterableDataset

HF_CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

class HFCDNDataset(IterableDataset):
    """
    CDN-only IterableDataset for Surrogate-1.
    Consumes file_list.json generated by scripts/generate_file_list.py.

    Each item:
        {"prompt": str, "response": str}
    """

    def __init__(
        self,
        file_list_path: str,
        repo_column: str = "repo",
        path_column: str = "path",
        max_retries: int = 5,
        backoff_factor: float = 2.0,
        timeout: int = 30,
    ):
        super().__init__()
        self.file_list_path = Path(file_list_path)
        self.repo_column = repo_column
        self.path_column = path_column
        self.max_retries = max_retries
        self.backoff_factor = backoff_factor
        self.timeout = timeout

        with open(self.file_list_path, "r") as f:
            self.files: List[Dict] = json.load(f)

        if not self.files:
            raise ValueError(f"No files found in {self.file_list_path}")

    def _download_parquet(self, repo: str, path: str) -> bytes:
        url = HF_CDN_TEMPLATE.format(repo=repo, path=path)
        for attempt in range(self.max_retries):
            try:
                resp = requests.get(url, stream=True, timeout=self.timeout)
                if resp.status_code == 200:
                    return resp.content
                # CDN-side rate limit or transient error
                if resp.status_code in (429, 503):
                    time.sleep(self.backoff_factor ** attempt)
                    continue
                resp.raise_for_status()
            except (requests.RequestException, IOError) as exc:
                if attempt == self.max_retries - 1:
                    raise RuntimeError(f"Failed to fetch {url}: {exc}") from exc
                time.sleep(self.backoff_factor ** attempt)
        raise RuntimeError(f"Exhausted retries for {url}")

    @staticmethod
    def _parquet_to_samples(data: bytes):
        table = pq.read_table(io.BytesIO(data))
        cols = set(table.column_names)
        if "prompt" not in cols or "response" not in cols:
            raise ValueError(f"Parquet missing prompt/response columns: {cols}")
        df = table.select(["prompt", "response"]).to_pandas()
        for _, row in df.iterrows():
            yield {"prompt": str(row["prompt"]), "response": str(row["response"])}

    def __iter__(self):
        for item in self.files:
            repo = item[self.repo_column]
            path = item[self.path_column]
            try:
                data = self._download_parquet(repo, path)
                yield from self._parquet_to_samples(data)
            except Exception as exc:
                # Log and skip bad files to avoid killing epoch
                print(f"Skipping {repo}/{path}: {exc}")
                continue
```

## 2) One-shot file-list generator (run during off-peak)

File: `surrogate/scripts/generate_file_list.py`

```python
#!/usr/bin/env python3
# surrogate/scripts/generate_file_list.py
import argparse
import json
from pathlib import Path

from huggingface_hub import HfApi

def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g. org/surrogate-data)")
    parser.add_argument("--folder", required=True, help="Folder to list (e.g. batches/mirror-merged/2026-04-29)")
    parser.add_argument("--output", required=True, help="Path to save file_list.json")
    args = parser.parse_args()

    api = HfApi()
    entries = api.list_repo_tree(repo_id=args.repo, path=args.folder, recursive=False)

    files = []
    for entry in entries:
        if entry.type != "file" or not entry.path.endswith(".parquet"):
            continue
        files.append(
            {
                "repo": args.repo,
                "path": entry.path,
                # sha256 omitted here; CDN integrity is handled by HTTPS + retries
            }
        )

    out_path = Path(args.output)
    out_path.parent.mkdir(parents=True, exist_ok=True)
    with open(out_path, "w") as f:
        json.dump(files, f, indent=2)

    print(f"Wrote {len(files)} parquet files to {out_path}")

if __name__ == "__main__":
    main()
```

Usage (run once per daily folder, e.g. on a Mac during off-peak):

```bash
python surrogate/scripts/generate_file_list.py \
  --repo org/surrogate-data \
  --folder batches/mirror-merged/2026-04-29 \
  --output surrogate/training/file_lists/2026-04-29.json
```

## 3) One-line swap in training code

In your training entrypoint (`train.py` or `run_surrogate_train.py`):

**Before:**
```python
from datasets import load_dataset
dataset = load_dataset("org/surrogate-data", split="train", streaming=True)
```

**After:**
```python
from surrogate.training.hf_cdn_dataset import HFCDNDataset
dataset = HFCDNDataset("surrogate/training/file_lists/2026-04-29.json")
```

Everything else (collate, batch size, Lightning DataLoader) remains unchanged.
