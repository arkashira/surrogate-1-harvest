# workio / quality

## Final synthesized implementation (correct + actionable)

Core principles chosen:
- Verify HMAC **before** parsing JSON; buffer raw body per request.
- Enforce idempotency with a short replay window (5 min) and a persistent dedupe key.
- Bind tenant cryptographically (signature covers raw body) and explicitly via event payload; reject cross-tenant or missing tenant.
- Never let verification or dedupe failures cascade into retries that cause duplicates; return success to LINE for already-processed events, reject only auth/format errors.
- Keep domain logic separate and safe (transactions + upserts).

### 3) Signature + raw-body middleware

File: `server/src/middleware/lineSignature.ts`

```ts
import crypto from 'crypto';
import { Request, Response, NextFunction } from 'express';

const LINE_CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';

export function verifyLineSignature(rawBody: Buffer, signature: string): boolean {
  if (!LINE_CHANNEL_SECRET) return false;
  const expected = crypto
    .createHmac('sha256', LINE_CHANNEL_SECRET)
    .update(rawBody)
    .digest('base64');
  // Use timingSafeEqual to avoid timing attacks; lengths must match
  const sigBuf = Buffer.from(signature);
  const expBuf = Buffer.from(expected);
  if (sigBuf.length !== expBuf.length) return false;
  return crypto.timingSafeEqual(sigBuf, expBuf);
}

export function lineRawBody(
  req: Request,
  _res: Response,
  next: NextFunction
) {
  const chunks: Buffer[] = [];
  req.on('data', (chunk) => chunks.push(chunk));
  req.on('end', () => {
    (req as any).rawBody = Buffer.concat(chunks);
    next();
  });
  req.on('error', next);
}
```

### 4) Webhook route (single source of truth)

File: `server/src/routes/webhook/line.ts`

```ts
import express, { Request, Response } from 'express';
import { verifyLineSignature, lineRawBody } from '../../middleware/lineSignature';
import { pool } from '../../db';

const router = express.Router();

// Use raw-body only for this route; mount before global express.json()
router.use(lineRawBody);

const REPLAY_WINDOW_MS = 5 * 60 * 1000; // 5 minutes

router.post('/line', async (req: Request, res: Response) => {
  const signature = req.headers['x-line-signature'] as string | undefined;
  const rawBody = (req as any).rawBody as Buffer | undefined;

  // 1) Basic checks
  if (!signature || !rawBody) {
    return res.status(400).json({ error: 'Missing signature or body' });
  }

  // 2) Verify HMAC before parsing
  if (!verifyLineSignature(rawBody, signature)) {
    return res.status(401).json({ error: 'Invalid signature' });
  }

  // 3) Parse JSON after verification
  let body;
  try {
    body = JSON.parse(rawBody.toString('utf8'));
  } catch {
    return res.status(400).json({ error: 'Invalid JSON' });
  }

  const events = body.events;
  if (!Array.isArray(events) || events.length === 0) {
    return res.status(200).json({ ok: true });
  }

  const now = Date.now();

  for (const ev of events) {
    const webhookEventId = ev.webhookEventId;
    const timestamp = Number(ev.timestamp) || 0;
    const source = ev.source || {};
    const userId = source.userId || null;
    const tenantId = ev.tenantId || null;

    // Mandatory fields
    if (!webhookEventId || !userId || !tenantId) {
      // Skip malformed events but don't cause LINE retries for auth/format issues
      continue;
    }

    // Replay-window check
    if (Math.abs(now - timestamp) > REPLAY_WINDOW_MS) {
      continue; // stale; skip but return 200 to stop retries
    }

    // Idempotency guard (atomic upsert pattern)
    try {
      const exists = await pool.query(
        `SELECT 1 FROM webhook_events WHERE event_id = $1 AND tenant_id = $2`,
        [webhookEventId, tenantId]
      );

      if (exists.rows.length > 0) {
        continue; // already processed
      }
    } catch {
      // If check fails, continue to attempt insert with unique constraint fallback
    }

    // Record event and perform domain action in one tx to avoid duplicates
    try {
      await pool.query('BEGIN');

      // Insert event record; unique constraint prevents duplicates under race
      await pool.query(
        `INSERT INTO webhook_events (event_id, tenant_id, user_id, payload, processed_at)
         VALUES ($1, $2, $3, $4, NOW())`,
        [webhookEventId, tenantId, userId, JSON.stringify(ev)]
      );

      // Domain handler (safe to extend)
      await handleLineEvent(ev, tenantId, userId);

      await pool.query('COMMIT');
    } catch (err: any) {
      await pool.query('ROLLBACK');

      // If duplicate key, treat as already processed and continue
      if (err?.code === '23505') {
        continue;
      }

      // For other errors, log and continue processing remaining events
      // (do not throw to avoid blocking LINE retries for unrelated failures)
      console.error('Webhook processing error', {
        eventId: webhookEventId,
        tenantId,
        userId,
        error: err?.message || err,
      });
    }
  }

  return res.status(200).json({ ok: true });
});

  const postback = event.postback || {};

  // Minimal safe dispatch; expand with domain services
```
