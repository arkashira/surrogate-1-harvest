# vanguard / ops

## Final Consolidated Solution

### Diagnosis (merged)
- Frontend and training workflows trigger authenticated `list_repo_tree` on every data-source selection or training launch, exhausting the HF API quota (1000/5min) and causing 429 errors.
- No persisted `(repo, dateFolder) → file-list` artifact; each run re-enumerates via API.
- Training uses `load_dataset(streaming=True)` on repos with heterogeneous schemas, causing runtime `pyarrow.CastError`.
- Lightning Studio is recreated on every run instead of reused, wasting quota and risking idle-stop training loss.
- No CDN-only data path; training depends on authenticated API during data loading, amplifying quota pressure.

### Proposed Change
Add an ops-side **file-list cache + CDN training loader** for Surrogate-1:
- New files:
  - `/opt/axentx/vanguard/ops/generate_file_list.py`
  - `/opt/axentx/vanguard/ops/train_cdn.py`
  - `/opt/axentx/vanguard/ops/reuse_studio.py` (helper)
- Update orchestration scripts to:
  1. Call `generate_file_list.py` once per `(repo, dateFolder)`.
  2. Invoke `train_cdn.py` with the cached JSON.
- Scope: single dateFolder snapshot (e.g., `2026-04-29`) to prove the pattern; generalize later.

#### ops/generate_file_list.py
```python
#!/usr/bin/env python3
"""
Generate and cache a repo+dateFolder file list for CDN-only training.
Usage:
  python generate_file_list.py \
    --repo datasets/your-org/surrogate-1 \
    --date 2026-04-29 \
    --out data/file_lists/surrogate-1_2026-04-29.json
"""
import argparse
import json
import os
import sys
from datetime import datetime

import huggingface_hub

CACHE_DIR = os.path.join(os.path.dirname(__file__), "..", "data", "file_lists")
os.makedirs(CACHE_DIR, exist_ok=True)

def build_file_list(repo_id: str, date_folder: str, out_path: str):
    """
    List ONLY one date folder (non-recursive) to minimize API calls.
    We rely on CDN URLs for actual downloads:
      https://huggingface.co/datasets/{repo}/resolve/main/{path}
    """
    client = huggingface_hub.HfApi()
    try:
        tree = client.list_repo_tree(
            repo_id=repo_id,
            path=date_folder,
            repo_type="dataset",
            recursive=False,
        )
    except Exception as e:
        print(f"HF API error listing {repo_id}/{date_folder}: {e}", file=sys.stderr)
        sys.exit(1)

    files = [
        item.rfilename
        for item in tree
        if item.rfilename.endswith((".parquet", ".jsonl", ".json"))
    ]

    payload = {
        "repo_id": repo_id,
        "date_folder": date_folder,
        "generated_at": datetime.utcnow().isoformat() + "Z",
        "files": sorted(files),
        "note": "CDN-only; use resolve/main/ URLs to bypass HF API auth during training",
    }

    os.makedirs(os.path.dirname(out_path), exist_ok=True)
    with open(out_path, "w") as f:
        json.dump(payload, f, indent=2)

    print(f"Wrote {len(files)} files to {out_path}")

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Generate CDN file-list cache")
    parser.add_argument("--repo", required=True, help="HF dataset repo id")
    parser.add_argument("--date", required=True, help="Date folder (e.g. 2026-04-29)")
    parser.add_argument(
        "--out",
        default=None,
        help="Output JSON path (default: CACHE_DIR/{repo_slug}_{date}.json)",
    )
    args = parser.parse_args()

    if args.out is None:
        slug = args.repo.replace("/", "_")
        args.out = os.path.join(CACHE_DIR, f"{slug}_{args.date}.json")

    build_file_list(args.repo, args.date, args.out)
```

#### ops/train_cdn.py

#!/usr/bin/env python3
"""
Lightning training script that loads ONLY via CDN (no HF API auth during data load).
Expects a file-list JSON from generate_file_list.py.
Projects heterogeneous files to {prompt, response} at parse time.
"""
import argparse
import json
import os
import sys
from typing import Dict, Iterator, List

import lightning as L
import pyarrow as pa
import pyarrow.parquet as pq
import requests
import torch
from torch.utils.data import IterableDataset

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

class CDNParquetIterable(IterableDataset):
    def __init__(self, file_urls: List[str], max_files: int = None):
        self.file_urls = file_urls[:max_files] if max_files else file_urls

    def __iter__(self) -> Iterator[Dict[str, str]]:
        for url in self.file_urls:
            try:
                resp = requests.get(url, timeout=30)
                resp.raise_for_status()
            except Exception as e:
                print(f"Failed to fetch {url}: {e}", file=sys.stderr)
                continue

            table = pq.read_table(pa.BufferReader(resp.content))
            # Project to {prompt, response} regardless of source schema
            for batch in table.to_batches(max_chunksize=1024):
                cols = {n: batch.column(n) for n in batch.schema.names}
                # Flexible field mapping
                prompt_col = next((cols[k] for k in ("prompt", "input", "question") if k in cols), None)
                response_col = next((cols[k] for k in ("response", "output", "answer") if k in cols), None)

                if prompt_col is None or response_col is None:
                    continue

                for i in range(len(batch)):
                    prompt = str(prompt_col[i].as_py())
                    response = str(response_col[i].as_py())
                    if prompt and response:
                        yield {"prompt": prompt, "response": response}

class SurrogateDataModule(L.LightningDataModule):
    def __init__(self, file_list_path: str, batch_size: int = 8, max_files: int = None):
        super().__init__()
        self.file_list_path = file_list_path
        self.batch_size = batch_size
        self.max_files = max_files

    def setup(self, stage=None):
        with open(self.file_list_path) as f:
            meta = json.load(f)
        repo = meta["repo_id"]
        urls = [CDN_TEMPLATE.format(repo=repo, path=p) for p in meta["files"]]
        self.dataset = CDNParquetIterable(urls, max_files=self.max_files)

    def train_dataloader(self):
        return torch.utils.data.DataLoader(
            self.dataset,
            batch_size=self.batch_size,
            num_workers=0,  # keep simple for CDN streaming
        )

class SurrogateModel(L.LightningModule):
    def __init__(self, lr: float = 1e-4):
        super().__init__()
        self.lr = lr
        # Minimal stub; replace with real model
        self.net = torch.nn.Linear(1024, 1024)

    def training_step(self, batch, batch_idx):
        # Placeholder: real training would tokenize prompt/response
        loss = torch.tensor(0.0, requires_grad=True)
        self.log("train_loss", loss)
        return loss

    def configure_optimizers(self):
        return torch.optim.AdamW(self.parameters(), lr=self.lr)
