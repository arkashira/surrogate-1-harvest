# vanguard / ops

## Final Synthesized Implementation

### 1. Diagnosis (Consensus)
- **No durable ingestion manifest**: every training run re-lists HF repos and re-downloads, causing 429s and wasted bandwidth.
- **Training loads via HF API** (`load_dataset`/`list_repo_files`) instead of CDN bypass → guaranteed rate limits during data loading.
- **No single source of truth** for which files belong to a training slice (date/slug) → non-deterministic runs and hard-to-resume.
- **Missing retry/backoff and CDN-only fetch strategy** in data loader → fragile at scale.
- **No guardrails for Lightning Studio reuse/idle-stop** → quota burned on recreation and jobs killed by idle timeout.

### 2. Proposed Change (Unified)
Create a single Python-first manifest generator and a Lightning-compatible CDN-only dataloader with built-in resilience and Studio lifecycle hooks.

**Files to create/modify:**
- `/opt/axentx/vanguard/ops/gen_manifest.py` (manifest generator)
- `/opt/axentx/vanguard/ops/run_manifest.sh` (Bash wrapper)
- `/opt/axentx/vanguard/train/train.py` (CDN-only dataloader + Lightning hooks)

### 3. Implementation

#### 3.1 Manifest Generator (`/opt/axentx/vanguard/ops/gen_manifest.py`)
```python
#!/usr/bin/env python3
"""
Generate a CDN-only manifest for a HF dataset folder.
Avoids recursive listing and HF API rate limits by:
- list_repo_tree(..., recursive=False) per folder
- embedding CDN base for direct file fetches
"""
import argparse
import json
import os
import sys
import time
from typing import List, Dict, Any

try:
    from huggingface_hub import HfApi, login
except ImportError:
    print("ERROR: huggingface_hub not installed. Run: pip install huggingface_hub", file=sys.stderr)
    sys.exit(1)

HF_API = HfApi()

def list_folder_files(repo: str, folder: str, retries: int = 5, backoff: int = 60) -> List[str]:
    """
    Non-recursive listing of a single folder.
    On 429, wait backoff and retry (HF API: 1000 req/5m).
    """
    for attempt in range(1, retries + 1):
        try:
            tree = HF_API.list_repo_tree(
                repo_id=repo,
                path=folder,
                recursive=False
            )
            files = []
            for item in tree:
                if isinstance(item, dict):
                    if item.get("type") == "file" or "path" in item:
                        files.append(item["path"])
                elif isinstance(item, str):
                    files.append(item)
            return sorted(set(files))
        except Exception as exc:
            msg = str(exc).lower()
            if "429" in msg or "rate limit" in msg:
                if attempt < retries:
                    wait = backoff * (2 ** (attempt - 1))
                    print(f"Rate limited (attempt {attempt}/{retries}). Waiting {wait}s...", file=sys.stderr)
                    time.sleep(wait)
                    continue
            print(f"Failed to list {repo}/{folder}: {exc}", file=sys.stderr)
            raise
    raise RuntimeError(f"Exhausted retries listing {repo}/{folder}")

def build_manifest(repo: str, date_folder: str) -> Dict[str, Any]:
    """
    Build manifest for repo/date_folder.
    Example: repo='datasets/example', date_folder='2026-05-02'
    """
    folder_path = f"{date_folder}"
    files = list_folder_files(repo, folder_path)

    cdn_base = f"https://huggingface.co/datasets/{repo}/resolve/main"
    file_urls = [f"{cdn_base}/{f}" for f in files]

    manifest: Dict[str, Any] = {
        "repo": repo,
        "date": date_folder,
        "folder": folder_path,
        "files": files,
        "file_urls": file_urls,
        "cdn_base": cdn_base,
        "generated_at_utc": time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime()),
        "note": "Use file_urls for CDN-only fetches. Zero HF API calls during training data loading."
    }
    return manifest

def main() -> None:
    parser = argparse.ArgumentParser(description="Generate CDN-only manifest for HF dataset folder.")
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g. datasets/example)")
    parser.add_argument("--date-folder", required=True, help="Date folder (e.g. 2026-05-02)")
    parser.add_argument("--out", required=True, help="Output JSON path")
    parser.add_argument("--hf-token", default=os.getenv("HF_TOKEN"), help="HF token (optional; avoids login if public)")
    args = parser.parse_args()

    if args.hf_token:
        try:
            login(token=args.hf_token, add_to_git_credential=False)
        except Exception:
            print("WARNING: HF login failed; continuing (public repo may still work).", file=sys.stderr)

    manifest = build_manifest(args.repo, args.date_folder)

    os.makedirs(os.path.dirname(os.path.abspath(args.out)) or ".", exist_ok=True)
    with open(args.out, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2, ensure_ascii=False)
    print(f"Manifest written to {args.out}")
    print(f"Files: {len(manifest['files'])}")

if __name__ == "__main__":
    main()
```

#### 3.2 Bash Wrapper (`/opt/axentx/vanguard/ops/run_manifest.sh`)
```bash
#!/usr/bin/env bash
set -euo pipefail
# Usage: bash run_manifest.sh <repo> <date_folder> <out_json>
REPO="${1:-datasets/example}"
DATE_FOLDER="${2:-2026-05-02}"
OUT_JSON="${3:-manifest.json}"

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
python3 "${SCRIPT_DIR}/gen_manifest.py" \
  --repo "${REPO}" \
  --date-folder "${DATE_FOLDER}" \
  --out "${OUT_JSON}"
```

#### 3.3 Lightning Training Script (`/opt/axentx/vanguard/train/train.py`)
```python
import json
import os
import time
import requests
from pathlib import Path
from typing import Iterator, Any, Optional

import lightning as L
import torch
from torch.utils.data import IterableDataset, DataLoader

class CDNIterableDataset(IterableDataset):
    """
    CDN-only dataset that yields file contents.
    Uses manifest JSON with file_urls.
    """
    def __init__(self, manifest_path: str, max_retries: int = 5, timeout: int = 30):
        super().__init__()
        self.manifest_path = manifest_path
        self.max_retries = max_retries
        self.timeout = timeout
        self.manifest = self._load_manifest()

    def _load_manifest(self) -> dict:
        with open(self.manifest_path) as f:
            return json.load(f)

    def _stream_cdn(self) -> Iterator[bytes]:
        for url in self.manifest["file_urls"]:
            for attempt in range(1, self.max_retries + 1):
                try:
                    resp = requests.get(url, timeout=self.timeout)
                    resp.raise_for_status()
                    yield resp.content
                    break
                except Exception as exc:
                    if attempt < self.max_retries:
                        wait = 2 ** attempt
                        print(f"Retry {attempt}/{self.max_retries} for {url}: {exc}. Waiting {wait}s")
                        time.sleep(wait)
                    else:
                        raise

    def __iter__(self) -> Iterator[dict]:
        worker_info = torch.utils.data.get_worker_info()
        urls = self.manifest["file_urls"]

        if worker_info is None:
            # single-process
            selected_urls = urls
        else:
            # multi-worker: shard by worker id
            per_worker = len(urls) // worker_info.num_workers
            start = worker_info.id * per_worker
            end = start + per_worker if worker_info.id < worker_info.num_workers - 1 else len(urls)
            selected_urls = urls[start:end
