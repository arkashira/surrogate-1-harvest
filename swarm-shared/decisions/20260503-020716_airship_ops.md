### 3) Lightning Studio guard (20 min)

  - List Teamspace studios; reuse a running studio by name.
  - If stopped, start with `Machine.L40S`; if missing, create.

  - Add timeouted `wait_running` and refresh checks.

### 4) Idle-stop resilience (20 min)

  - Check studio status → restart if idle-stopped → run script → capture logs.

### 5) Smoke test (10 min)

### scripts/list_hf_snapshot.py
```python
#!/usr/bin/env python3
import json
import os
from huggingface_hub import HfApi

REPO_ID = os.getenv("HF_DATASET_REPO", "org/surrogate-dataset")
DATE_FOLDER = os.getenv("HF_DATE_FOLDER", "2026-05-03")
OUTPUT = os.getenv("OUTPUT", "file_list.json")

def main() -> None:
    api = HfApi()
    files = api.list_repo_tree(repo_id=REPO_ID, path=DATE_FOLDER, recursive=False)
    paths = [f.rfilename for f in files if f.rfilename]
    out_path = os.path.join(os.path.dirname(__file__), "..", OUTPUT)
    with open(out_path, "w") as f:
        json.dump({"repo_id": REPO_ID, "date_folder": DATE_FOLDER, "files": paths}, f, indent=2)
    print(f"Wrote {len(paths)} files to {out_path}")

if __name__ == "__main__":
    main()
```

### surrogate/data/cdn_loader.py
```python
import json
import pyarrow.parquet as pq
from huggingface_hub import hf_hub_download
from typing import List, Dict, Any

def cdn_download(repo_id: str, filename: str, cache_dir: str = ".cache/hf") -> str:
    return hf_hub_download(
        repo_id=repo_id,
        filename=filename,
        cache_dir=cache_dir,
        local_files_only=False,
    )

def project_to_pair(local_path: str) -> List[Dict[str, str]]:
    table = pq.read_table(local_path, columns=["prompt", "response"])
    df = table.to_pandas()
    return [r for r in df[["prompt", "response"]].to_dict(orient="records")
            if r.get("prompt") and r.get("response")]

def load_file_list(file_list_path: str, cache_dir: str = ".cache/hf") -> List[Dict[str, Any]]:
    with open(file_list_path) as f:
        meta = json.load(f)
    repo_id = meta["repo_id"]
    examples = []
    for fn in meta["files"]:
        local = cdn_download(repo_id, fn, cache_dir=cache_dir)
        for ex in project_to_pair(local):
            ex["_source_file"] = fn
            examples.append(ex)
    return examples
```

### scripts/lightning_studio_guard.py
```python
#!/usr/bin/env python3
import os
import time
from lightning_sdk import Teamspace, Studio, Machine

TEAMSPACE = os.getenv("LIGHTNING_TEAMSPACE", "default")
STUDIO_NAME = os.getenv("LIGHTNING_STUDIO_NAME", "surrogate-train")
MACHINE = Machine.L40S

def get_or_create_studio() -> Studio:
    teamspace = Teamspace(name=TEAMSPACE)
    running = [s for s in teamspace.studios if s.name == STUDIO_NAME and s.status == "running"]
    if running:
        print(f"Reusing running studio: {STUDIO_NAME}")
        return running[0]

    stopped = [s for s in teamspace.studios if s.name == STUDIO_NAME]
    if stopped:
        studio = stopped[0]
        print(f"Starting stopped studio: {STUDIO_NAME}")
        studio.start(machine=MACHINE)
        return studio

    print(f"Creating new studio: {STUDIO_NAME}")
    return Studio.create(
        name=STUDIO_NAME,
        teamspace=TEAMSPACE,
        machine=MACHINE,
        create_ok=True,
    )

def wait_running(studio: Studio, timeout: int = 300) -> bool:
    elapsed = 0
    while elapsed < timeout:
        studio.refresh()
        if studio.status == "running":
            return True
        time.sleep(10)
        elapsed += 10
    return False

def run_in_studio(script_path: str, args: list[str]) -> None:
    studio = get_or_create_studio()
    if not wait_running(studio):
        raise RuntimeError("Studio failed to become running")

    studio.refresh()
    if studio.status != "running":
        print("Studio stopped; restarting...")
        studio.start(machine=MACHINE)
        if not wait_running(studio):
            raise RuntimeError("Studio restart failed")

    print(f"Running {script_path} in studio {STUDIO_NAME}")
    run = studio.run(str(script_path), args)
    run.stream_logs()
```
