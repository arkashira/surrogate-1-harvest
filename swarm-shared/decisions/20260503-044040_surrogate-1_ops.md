# surrogate-1 / ops

## Final Implementation Plan (≤2h)

**Highest-value improvement**: Replace fragile shell-based ingestion with a **manifest-driven, CDN-bypass Python worker** that eliminates HF API rate limits during training data loads and prevents mixed-schema CastErrors.

### Changes

1. **Add `train_manifest.json` generation** (single API call from Mac orchestrator)  
   - `bin/build_manifest.py` — lists one date folder via `list_repo_tree(..., recursive=False)` and writes `{"date": "2026-04-29", "files": [...]}`.  
   - Embed this manifest in the repo so Lightning training does **zero HF API calls** during data loading.

2. **Replace `dataset-enrich.sh` heavy streaming with `worker/cdn_ingest.py`**  
   - Deterministic shard assignment via `hash(slug) % 16`.  
   - Each worker downloads only its shard’s files via raw CDN URLs (`https://huggingface.co/datasets/.../resolve/main/...`).  
   - Project to `{prompt, response}` at parse time; skip heterogeneous columns.  
   - Output `batches/public-merged/{date}/shard{N}-{HHMMSS}.jsonl`.

3. **Update GitHub Actions matrix** to run the Python worker instead of the shell script.

4. **Keep `lib/dedup.py`** unchanged (central md5 store) — workers call it via import.

## Code Snippets

### 1. `bin/build_manifest.py` (run from Mac orchestrator)

```python
#!/usr/bin/env python3
"""
Generate train_manifest.json for a given date folder.
Run from Mac (or any machine) after HF API rate-limit window clears.
"""
import json, os, sys
from datetime import datetime, timezone
from huggingface_hub import HfApi

REPO = "datasets/axentx/surrogate-1-training-pairs"
DATE = sys.argv[1] if len(sys.argv) > 1 else datetime.now(timezone.utc).strftime("%Y-%m-%d")
OUT = "train_manifest.json"

api = HfApi()
# Single API call: non-recursive listing for one date folder
tree = api.list_repo_tree(repo_id=REPO, path=DATE, recursive=False)
files = [f.rfilename for f in tree if f.rfilename.endswith((".parquet", ".jsonl", ".json"))]

manifest = {
    "date": DATE,
    "generated_at": datetime.now(timezone.utc).isoformat(),
    "files": sorted(files),
}

with open(OUT, "w") as f:
    json.dump(manifest, f, indent=2)

print(f"Wrote {OUT} with {len(files)} files for {DATE}")
```

### 2. `worker/cdn_ingest.py` (replaces `dataset-enrich.sh`)

```python
#!/usr/bin/env python3
"""
CDN-bypass ingest worker.
Usage (in GitHub Actions):
  python worker/cdn_ingest.py --shard 0 --shards 16 --date 2026-04-29
"""
import argparse, hashlib, json, os, sys, time
from pathlib import Path
from typing import Any, Dict

import numpy as np
import pyarrow.parquet as pq
import requests
from tqdm import tqdm

REPO = "datasets/axentx/surrogate-1-training-pairs"
BASE_CDN = f"https://huggingface.co/{REPO}/resolve/main"

# Local imports
sys.path.insert(0, str(Path(__file__).parent.parent))
from lib.dedup import DedupStore  # noqa

def hash_slug(slug: str) -> int:
    return int(hashlib.md5(slug.encode()).hexdigest(), 16)

def parse_file(path: Path, date: str, rel_path: str) -> list[Dict[str, Any]]:
    """Project heterogeneous files to {prompt,response} only."""
    suffix = path.suffix.lower()
    records = []

    try:
        if suffix == ".parquet":
            table = pq.read_table(path, columns=["prompt", "response"], use_threads=False)
            df = table.to_pandas()
            for _, row in df.iterrows():
                prompt = str(row.get("prompt") or "")
                response = str(row.get("response") or "")
                if prompt and response:
                    records.append({"prompt": prompt, "response": response})

        elif suffix in (".jsonl", ".json"):
            with open(path, "r", encoding="utf-8") as f:
                if suffix == ".jsonl":
                    lines = f
                else:
                    lines = [json.dumps(json.load(f))] if isinstance(json.load(f), list) else []
                for line in lines:
                    obj = json.loads(line)
                    prompt = str(obj.get("prompt") or obj.get("input") or "")
                    response = str(obj.get("response") or obj.get("output") or "")
                    if prompt and response:
                        records.append({"prompt": prompt, "response": response})
    except Exception as exc:
        print(f"Parse error {rel_path}: {exc}", file=sys.stderr)

    return records

def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("--shard", type=int, required=True)
    parser.add_argument("--shards", type=int, default=16)
    parser.add_argument("--date", required=True)
    parser.add_argument("--manifest", default="train_manifest.json")
    parser.add_argument("--out-dir", default="batches/public-merged")
    args = parser.parse_args()

    # Load manifest (committed by build_manifest.py)
    with open(args.manifest) as f:
        manifest = json.load(f)

    files = [p for p in manifest["files"] if p.startswith(args.date)]
    my_files = [p for p in files if hash_slug(p) % args.shards == args.shard]

    print(f"Shard {args.shard}/{args.shards}: processing {len(my_files)} files")

    dedup = DedupStore()
    ts = time.strftime("%H%M%S")
    out_path = Path(args.out_dir) / args.date / f"shard{args.shard}-{ts}.jsonl"
    out_path.parent.mkdir(parents=True, exist_ok=True)

    written = 0
    with open(out_path, "w", encoding="utf-8") as out_f:
        for rel in tqdm(my_files, desc="CDN download"):
            url = f"{BASE_CDN}/{rel}"
            tmp = Path("tmp") / rel.replace("/", "_")
            tmp.parent.mkdir(parents=True, exist_ok=True)

            resp = requests.get(url, timeout=60)
            resp.raise_for_status()
            tmp.write_bytes(resp.content)

            records = parse_file(tmp, args.date, rel)
            for rec in records:
                # Deterministic content hash for dedup
                payload = f"{rec['prompt']}\n{rec['response']}".encode()
                md5 = hashlib.md5(payload).hexdigest()
                if not dedup.seen(md5):
                    out_f.write(json.dumps(rec, ensure_ascii=False) + "\n")
                    written += 1
            tmp.unlink(missing_ok=True)

    print(f"Shard {args.shard} wrote {written} pairs to {out_path}")

if __name__ == "__main__":
    main()
```
