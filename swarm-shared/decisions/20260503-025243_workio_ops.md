## Final synthesized implementation (strongest parts, resolved contradictions)

**Chosen approach**
- Use **LINE’s `webhookEventId`** (when present) as the primary idempotency key; fall back to a deterministic composite key only when necessary.  
- Enforce idempotency **server-side in a single DB transaction** with a unique constraint; never rely on frontend-only guards.  
- Add **signature verification** and **request-id propagation** for security and traceability.  
- Keep punch logic **simple and transactional**: one webhook event → at most one punch row; use upsert (or insert-only with idempotency table) to prevent duplicates.  
- Avoid optimistic UI as a correctness mechanism; keep it for UX only.

import { Request, Response, NextFunction } from 'express';
import crypto from 'crypto';

export function requestIdMiddleware(req: Request, _res: Response, next: NextFunction) {
  const id = (req.headers['x-request-id'] as string) || crypto.randomUUID();
  (req as any).id = id;
  next();
}

// server/src/utils/lineSignature.ts
import crypto from 'crypto';

export function verifyLineSignature(rawBody: string, signature: string, channelSecret: string): boolean {
  const expected = crypto
    .createHmac('sha256', channelSecret)
    .update(rawBody, 'utf8')
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

### 3. Idempotent punch handler

```ts
// server/src/routes/line/webhook.ts
import { Pool } from 'pg';
import { verifyLineSignature } from '../../utils/lineSignature';
import crypto from 'uuid';

const CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';

export async function handleLineWebhook(req: any, res: any, pool: Pool) {
  const requestId = req.id || crypto.v4();
  const rawBody = req.rawBody || JSON.stringify(req.body);
  const signature = req.headers['x-line-signature'] as string;

  // 1) Security: verify signature
  if (!verifyLineSignature(rawBody, signature, CHANNEL_SECRET)) {
    console.warn(`[${requestId}] Invalid LINE signature`);
    return res.status(401).send('Invalid signature');
  }

  const { events } = req.body;
  if (!Array.isArray(events) || events.length === 0) {
    return res.status(400).send('No events');
  }

  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    for (const event of events) {
      // 2) Idempotency key: prefer LINE's stable ID
      const lineEventId = event.webhookEventId || [
        event.source?.userId,
        event.timestamp,
        event.type,
        event.mode
      ].filter(Boolean).join(':');

      if (!lineEventId) {
        console.warn(`[${requestId}] Missing usable event ID`, event);
        continue;
      }

      // 3) Check idempotency table
      const dup = await client.query(
        `SELECT punch_id FROM line_event_idempotency WHERE line_event_id = $1`,
        [lineEventId]
      );
      if (dup.rows.length > 0) {
        // Already processed — skip mutation
        continue;
      }

      // 4) Resolve user/tenant
      const userRes = await client.query(
        `SELECT id, tenant_id FROM users WHERE line_user_id = $1`,
        [event.source?.userId]
      );
      if (!userRes.rows.length) {
        console.warn(`[${requestId}] Unknown LINE user`, event.source?.userId);
        continue;
      }
      const { id: userId, tenant_id: tenantId } = userRes.rows[0];

      // 5) Create punch (clock-in/out inferred from event/text)
      const punchId = crypto.v4();
      const isClockIn = event.message?.text === 'clock_in' || event.type === 'clock_in';
      await client.query(
        `INSERT INTO punches (id, tenant_id, user_id, type, clock_in, clock_out, location, request_id, created_at)
         VALUES ($1, $2, $3, $4, $5, $6, $7, $8, NOW())`,
        [
          punchId,
          tenantId,
          userId,
          'clock',
          isClockIn ? 'NOW()' : null,
          isClockIn ? null : 'NOW()',
          event.message?.text || null,
          requestId
        ]
      );

      // 6) Record idempotency
      await client.query(
        `INSERT INTO line_event_idempotency (line_event_id, tenant_id, user_id, punch_id, request_id)
         VALUES ($1, $2, $3, $4, $5)`,
        [lineEventId, tenantId, userId, punchId, requestId]
      );
    }

    await client.query('COMMIT');
    return res.status(200).send('OK');
  } catch (err) {
    await client.query('ROLLBACK');
    console.error(`[${requestId}] Webhook processing failed`, err);
    return res.status(500).send('Processing failed');
  } finally {
    client.release();
  }
}
```
