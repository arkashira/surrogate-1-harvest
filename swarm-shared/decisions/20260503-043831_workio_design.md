# workio / design

## Final Synthesis (single, actionable answer)

**Core problem**: LINE retries on 5xx/timeouts and lacks client-side idempotency; missing signature verification allows spoofing; missing dedupe allows duplicate punches; missing state checks allow invalid clock transitions.

**Goal**: Accept LINE webhooks safely, prevent duplicates/fraud, enforce valid attendance state, and avoid retry storms.

### 2. Implementation (merged best parts)

**Signature verification** (`middleware/lineSignature.ts`)
```ts
import crypto from 'crypto';

const LINE_CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';

export function verifyLineSignature(rawBody: string, signature: string): boolean {
  if (!LINE_CHANNEL_SECRET) return true; // allow if not configured (dev)
  const expected = crypto
    .createHmac('sha256', LINE_CHANNEL_SECRET)
    .update(rawBody, 'utf8')
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}
```

**Webhook route** (`routes/webhook/line.ts`)
```ts
import { Router, Request, Response } from 'express';
import crypto from 'crypto';
import { Pool } from 'pg';
import { verifyLineSignature } from '../../middleware/lineSignature';
import { markProcessed } from '../../middleware/dedupe';

const router = Router();
const pool = new Pool({ connectionString: process.env.DATABASE_URL });

// Helper: current state for a user (clocked in/out)
async function currentState(client: any, tenantId: string, userId: string) {
  const r = await client.query(
    `SELECT event_type FROM attendance
     WHERE tenant_id = $1 AND user_id = $2
     ORDER BY event_time DESC LIMIT 1`,
    [tenantId, userId]
  );
  return r.rows[0]?.event_type || null;
}

// State transition rules
function isValidTransition(prev: string | null, next: string): boolean {
  if (next === 'clock_in') return prev !== 'clock_in';
  if (next === 'clock_out') return prev === 'clock_in';
  // leave/ot: allow only if currently clocked in (adjust per policy)
  if (next === 'leave' || next === 'ot') return prev === 'clock_in';
  return false;
}

router.post('/webhook/line', async (req: Request, res: Response) => {
  const signature = req.headers['x-line-signature'] as string;
  const delivery = req.headers['x-line-delivery'] as string;
  const rawBody = JSON.stringify(req.body);

  // 1) Verify signature
  if (!verifyLineSignature(rawBody, signature)) {
    console.warn('Invalid LINE signature', { delivery });
    return res.status(401).send('Invalid signature');
  }

  // 2) Build idempotency key (prefer delivery; fallback to event hash)
  const idempotencyKey = delivery || `event-${crypto.createHash('sha256').update(rawBody).digest('hex')}`;
  const isNew = await markProcessed(idempotencyKey);
  if (!isNew) {
    console.info('Duplicate webhook ignored', { delivery: idempotencyKey });
    return res.status(200).json({}); // stop LINE retries
  }

  const events = req.body.events || [];
  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    for (const ev of events) {
      // Only process message events for this simplified example
      if (ev.type === 'message' && ev.message.type === 'text') {
        const userId = ev.source.userId;
        const tenantId = 'default'; // derive from DB/config as needed
        const action = ev.message.text.trim().toLowerCase();

        // Map text to event_type (customize as needed)
        const eventTypeMap: Record<string, string> = {
          'in': 'clock_in',
          'out': 'clock_out',
          'leave': 'leave',
          'ot': 'ot'
        };
        const eventType = eventTypeMap[action];
        if (!eventType) continue; // ignore unknown commands

        // Enforce state machine
        const prev = await currentState(client, tenantId, userId);
        if (!isValidTransition(prev, eventType)) {
          console.warn('Invalid transition', { userId, tenantId, prev, eventType, delivery: idempotencyKey });
          // Handled client error: return 200 to stop LINE retries, but record nothing
          continue;
        }

        await client.query(
          `INSERT INTO attendance (tenant_id, user_id, event_type, line_delivery, event_time, created_at)
           VALUES ($1, $2, $3, $4, NOW(), NOW())`,
          [tenantId, userId, eventType, idempotencyKey]
        );
      }

      // Add other event types (postback, etc.) as needed
    }

    await client.query('COMMIT');
    return res.status(200).json({});
  } catch (err) {
    await client.query('ROLLBACK');
    console.error('Webhook processing failed', err);
    // Return 500 only for unexpected server errors so LINE may retry (idempotency prevents duplicates)
    return res.status(500).send('Processing error');
  } finally {
    client.release();
  }
});

export default router;
```
