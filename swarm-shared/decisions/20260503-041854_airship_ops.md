## Code Snippets

### 1) Manifest Generator (run from Mac — one API call)
`tools/generate_cdn_manifest.py`
```python
#!/usr/bin/env python3
"""
Generate a deterministic CDN manifest for Surrogate training.
Run once per date folder after HF rate-limit window clears.
"""
import json
import os
import sys
from datetime import datetime, timezone

from huggingface_hub import HfApi

REPO_ID = os.getenv("HF_DATASET_REPO", "axentx/surrogate-dataset")
DATE_FOLDER = os.getenv("DATE_FOLDER", datetime.now(timezone.utc).strftime("%Y-%m-%d"))
OUTPUT = os.getenv("MANIFEST_OUT", "manifest.json")

def main() -> None:
    api = HfApi()
    # Non-recursive per folder to avoid 100x pagination and rate limits
    entries = api.list_repo_tree(
        repo_id=REPO_ID,
        path=DATE_FOLDER,
        repo_type="dataset",
        recursive=False,
    )
    files = [e.rfilename for e in entries if e.type == "file"]
    manifest = {
        "repo_id": REPO_ID,
        "date_folder": DATE_FOLDER,
        "generated_at_utc": datetime.now(timezone.utc).isoformat(),
        "files": sorted(files),
        "cdn_base": f"https://huggingface.co/datasets/{REPO_ID}/resolve/main",
    }
    os.makedirs(os.path.dirname(OUTPUT) if os.path.dirname(OUTPUT) else ".", exist_ok=True)
    with open(OUTPUT, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2)
    print(f"Manifest written to {OUTPUT} with {len(files)} files")

if __name__ == "__main__":
    main()
```

### 2) Training Data Loader (CDN-only; zero API calls during train)
`surrogate/train/data.py`
```python
import json
import os
from pathlib import Path
from typing import List, Dict

import requests
from tqdm import tqdm


def load_manifest(manifest_path: str) -> Dict:
    with open(manifest_path, "r", encoding="utf-8") as f:
        return json.load(f)


def stream_parquet_from_cdn(cdn_url: str, chunk_size: int = 8192):
    """Stream parquet bytes from CDN (no auth, no HF API)."""
    resp = requests.get(cdn_url, stream=True, timeout=30)
    resp.raise_for_status()
    # If you need bytes for pyarrow:
    # content = b"".join(chunk for chunk in resp.iter_content(chunk_size=chunk_size))
    # For direct file download:
    return resp.content


def build_cdn_urls(manifest: Dict) -> List[str]:
    base = manifest["cdn_base"].rstrip("/")
    return [f"{base}/{f}" for f in manifest["files"]]


def download_dataset_to_local(manifest_path: str, out_dir: str = "data/raw") -> List[str]:
    manifest = load_manifest(manifest_path)
    urls = build_cdn_urls(manifest)
    os.makedirs(out_dir, exist_ok=True)
    local_paths = []
    for url in tqdm(urls, desc="Downloading via CDN"):
        fname = url.split("/")[-1]
        local_path = os.path.join(out_dir, fname)
        if os.path.exists(local_path):
            local_paths.append(local_path)
            continue
        data = stream_parquet_from_cdn(url)
        with open(local_path, "wb") as f:
            f.write(data)
        local_paths.append(local_path)
    return local_paths
```

Notes:
- No `load_dataset(streaming=True)` on heterogeneous repos.
- Project to `{prompt, response}` only after local parquet read.

### 3) Lightning Studio Launcher with Auto-Restart
`tools/launch_studio_training.py`
```python
#!/usr/bin/env python3
"""
Launch (or reuse) a Lightning Studio and run training.
Auto-restarts studio if stopped (idle-timeout resilience).
"""
import os
import sys
import time

from lightning_sdk import Studio, Machine, Teamspace

STUDIO_NAME = os.getenv("STUDIO_NAME", "surrogate-train-l40s")
MACHINE = Machine.L40S
MAX_RETRIES = 3
RETRY_WAIT = 60

def get_running_studio() -> Studio:
    for s in Teamspace.studios:
        if s.name == STUDIO_NAME and s.status == "Running":
            print(f"Reusing running studio: {s.name}")
            return s
    return None

def ensure_studio_running() -> Studio:
    studio = get_running_studio()
    if studio:
        return studio
    print(f"Starting new studio: {STUDIO_NAME}")
    studio = Studio.create(
        name=STUDIO_NAME,
        machine=MACHINE,
        create_ok=True,
    )
    # Wait until running
    for _ in range(30):
        studio.refresh()
        if studio.status == "Running":
            return studio
        time.sleep(10)
    raise RuntimeError("Studio failed to start")

def run_training_command(studio: Studio, cmd: str) -> None:
    for attempt in range(1, MAX_RETRIES + 1):
        try:
            print(f"Run attempt {attempt}: {cmd}")
            run = studio.run(cmd, wait=True)
            print("Training completed")
            return
        except Exception as e:
            print(f"Run failed: {e}")
            studio.refresh()
            if studio.status != "Running":
                print("Studio not running — restarting...")
                studio.start(machine=MACHINE)
                time.sleep(30)
            if attempt == MAX_RETRIES:
                raise
            time.sleep(RETRY_WAIT)

if __name__ == "__main__":
    training_cmd = os.getenv(
        "TRAIN_CMD",
        "python surrogate/train/train.py --manifest manifest.json --epochs 3",
    )
    studio = ensure_studio_running()
    run_training_command(studio, training_cmd)
```
