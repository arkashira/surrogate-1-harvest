# surrogate-1 / design

## Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16` (matrix) and optional `DATE_FOLDER` (defaults to today `YYYY-MM-DD`).
- On the orchestrator (Mac/CI): **single API call** to `list_repo_tree` for `DATE_FOLDER` (non-recursive), save `file-list.json`.
- Worker uses **only CDN URLs** (`https://huggingface.co/datasets/.../resolve/main/...`) — zero API calls during training/ingest, bypassing HF auth rate limits.
- Deterministic shard assignment: `hash(slug) % SHARD_TOTAL == SHARD_ID`.
- Per-file streaming decode with **schema projection** → `{prompt, response}` only; attribution via filename pattern (`batches/public-merged/{date}/shard{N}-{HHMMSS}.jsonl`). No extra columns.
- Central dedup via existing `lib/dedup.py` (md5 store) to prevent duplicates within the run.
- Upload produced shard file to `axentx/surrogate-1-training-pairs` via HF API (single commit per shard).
- Designed for GitHub Actions matrix (16 isolated runners) and reuse in HF Space (same script, different env).

HF_API_BASE = "https://huggingface.co"
CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def today_folder() -> str:
    return datetime.datetime.utcnow().strftime("%Y-%m-%d")

def deterministic_shard(slug: str, total: int) -> int:
    return int(hashlib.sha256(slug.encode("utf-8")).hexdigest(), 16) % total

def list_date_files(repo: str, date_folder: str) -> List[str]:
    """
    Single API call: list top-level files under date_folder (non-recursive).
    Returns repo-relative paths.
    """
    api = HfApi()
    # list_repo_tree non-recursive; may return trees and blobs
    items = api.list_repo_tree(repo=repo, path=date_folder, recursive=False)
    files = []
    for it in items:
        if it.type == "blob":
            files.append(f"{date_folder}/{it.path.split('/')[-1]}")
    return files

def build_manifest(repo: str, date_folder: str, out_path: Path) -> List[str]:
    files = list_date_files(repo, date_folder)
    out_path.write_text(json.dumps(files, indent=2))
    return files

def cdn_download(path: str, repo: str, timeout: int = 30) -> bytes:
    url = CDN_TEMPLATE.format(repo=repo, path=path)
    resp = requests.get(url, timeout=timeout, stream=True)
    resp.raise_for_status()
    chunks = []
    for chunk in resp.iter_content(chunk_size=8192):
        chunks.append(chunk)
    return b"".join(chunks)

def parse_file_to_pairs(raw: bytes, filename: str) -> List[Dict[str, str]]:
    """
    Best-effort projection to {prompt,response}. Supports:
    - JSONL / JSON arrays of objects
    - Parquet via pyarrow (streaming not possible from bytes, but files are small per shard)
    - CSV with prompt/response columns
    Falls back to empty list on parse failure (log and skip).
    """
    name = filename.lower()
    pairs = []

    try:
        if name.endswith(".parquet"):
            table = pq.read_table(pa.BufferReader(raw))
            df = table.to_pandas()
            for _, row in df.iterrows():
                prompt = row.get("prompt") or row.get("input") or row.get("text") or ""
                response = row.get("response") or row.get("output") or row.get("completion") or ""
                if prompt and response:
                    pairs.append({"prompt": str(prompt), "response": str(response)})

        elif name.endswith(".jsonl"):
            for line in raw.decode("utf-8", errors="ignore").splitlines():
                line = line.strip()
                if not line:
                    continue
                obj = json.loads(line)
                prompt = obj.get("prompt") or obj.get("input") or obj.get("text") or ""
                response = obj.get("response") or obj.get("output") or obj.get("completion") or ""
                if prompt and response:
                    pairs.append({"prompt": str(prompt), "response": str(response)})

        elif name.endswith(".json"):
            obj = json.loads(raw.decode("utf-8", errors="ignore"))
            if isinstance(obj, list):
                items = obj
            else:
                items = [obj]
            for item in items:
                prompt = item.get("prompt") or item.get("input") or item.get("text") or ""
                response = item.get("response") or item.get("output") or item.get("completion") or ""
                if prompt and response:
                    pairs.append({"prompt": str(prompt), "response": str(response)})

        elif name.endswith(".csv"):
            import csv
            from io import StringIO
            f = StringIO(raw.decode("utf-8", errors="ignore"))
            reader = csv.DictReader(f)
            for row in reader:
                prompt = row.get("prompt") or row.get("input") or row.get("text") or ""
                response = row.get("response") or row.get("output") or row.get("completion") or ""
                if prompt and response:
                    pairs.append({"prompt": str(prompt), "response": str(response)})

        else:
            # Unknown format — skip
            pass

    except Exception as exc:
        print(f"[WARN] parse failed for {filename}: {exc}", file=sys.stderr)

    return pairs
