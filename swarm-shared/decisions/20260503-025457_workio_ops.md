# workio / ops

## 1. Diagnosis
- LINE Messaging API retries on 5xx/timeouts; webhook handler lacks idempotency key enforcement → duplicate clock-in/out events and race conditions on `punches` table.
- No server-side uniqueness guard for open punches (per tenant+employee) → multiple concurrent “clock-in” rows possible, corrupting attendance reports.
- Webhook handler commits before LINE signature verification and tenant resolution, exposing endpoint to spoofing and cross-tenant data bleed.
- No idempotency table for cross-entity operations (leave/OT requests via API) — retries from frontend or LINE create duplicate leave records.
- Missing operational alerting/metrics on webhook 4xx/5xx and duplicate event rate; ops cannot detect upstream LINE issues or tenant abuse quickly.

## 2. Proposed change
File: `/opt/axentx/workio/server/src/routes/webhook/line.ts` (or equivalent)  
Scope:
- Add idempotency table (`line_webhook_deliveries`) keyed by `X-Line-Signature`+`timestamp` hash or `delivery_id` from LINE event envelope.
- Add unique constraint on `punches(tenant_id, employee_id, clock_out_at)` where `clock_out_at IS NULL` to enforce single open punch.
- Wrap handler in: verify signature → resolve tenant → idempotency check → upsert punch with `ON CONFLICT DO NOTHING` → 200 fast.
- Add lightweight operational counters (duplicates, verify-fail, tenant-miss) exported via `/ops/metrics` endpoint for ops monitoring.

CREATE INDEX IF NOT EXISTS idx_line_webhook_deliveries_tenant ON line_webhook_deliveries(tenant_id);

-- Ensure single open punch per tenant+employee
CREATE UNIQUE INDEX IF NOT EXISTS uq_open_punch

// 2) /opt/axentx/workio/server/src/routes/webhook/line.ts
import crypto from 'crypto';
import { Router, Request, Response } from 'express';
import { pool } from '../../db';
import { verifyLineSignature } from '../../utils/lineCrypto';

const router = Router();

function deliveryKey(body: string, timestamp: string): string {
  return crypto.createHash('sha256').update(`${body}:${timestamp}`).digest('hex');
}

router.post('/line', async (req: Request, res: Response) => {
  const bodyRaw = JSON.stringify(req.body);
  const timestamp = req.headers['x-line-request-timestamp'] as string;
  const signature = req.headers['x-line-signature'] as string;

  // Fast fail on missing headers
  if (!timestamp || !signature) {
    return res.status(400).json({ error: 'missing_line_headers' });
  }

  // Verify signature first
  if (!verifyLineSignature(bodyRaw, signature)) {
    // ops metric: line_verify_fail
    return res.status(401).json({ error: 'invalid_signature' });
  }

  const events = req.body?.events;
  if (!Array.isArray(events) || events.length === 0) {
    return res.status(200).json({ ok: true });
  }

  // Use first userId to resolve tenant (assumes userId -> employee -> tenant)
  const firstEvent = events[0];
  const userId = firstEvent?.source?.userId;
  if (!userId) {
    return res.status(400).json({ error: 'missing_user_id' });
  }

  const key = deliveryKey(bodyRaw, timestamp);

  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // Idempotency check
    const idempotencyCheck = await client.query(
      `SELECT 1 FROM line_webhook_deliveries WHERE delivery_key = $1 FOR UPDATE`,
      [key]
    );
    if (idempotencyCheck.rowCount && idempotencyCheck.rowCount > 0) {
      await client.query('COMMIT');
      // ops metric: line_duplicate
      return res.status(200).json({ ok: true });
    }

    // Resolve tenant via userId
    const tenantRes = await client.query(
      `SELECT e.tenant_id FROM employees e WHERE e.line_user_id = $1 LIMIT 1`,
      [userId]
    );
    if (!tenantRes.rowCount) {
      await client.query('ROLLBACK');
      // ops metric: tenant_miss
      return res.status(404).json({ error: 'tenant_not_found' });
    }
    const tenantId = tenantRes.rows[0].tenant_id;

    // Record delivery (idempotency)
    await client.query(
      `INSERT INTO line_webhook_deliveries (tenant_id, delivery_key, event_type)
       VALUES ($1, $2, $3)`,
      [tenantId, key, 'punch']
    );

    // Process punch events idempotently
    for (const ev of events) {
      if (ev.type !== 'message' || ev.message?.type !== 'text') continue;
      const text = (ev.message.text || '').trim().toLowerCase();
      const sourceUserId = ev.source?.userId;
      if (!sourceUserId) continue;

      // Resolve employee
      const empRes = await client.query(
        `SELECT id FROM employees WHERE line_user_id = $1 AND tenant_id = $2 LIMIT 1`,
        [sourceUserId, tenantId]
      );
      if (!empRes.rowCount) continue;
      const employeeId = empRes.rows[0].id;

      if (text === 'clock in' || text === 'clock-in' || text === 'เข้างาน') {
        // Try insert; if duplicate open punch, ignore safely
        await client.query(
          `INSERT INTO punches (tenant_id, employee_id, clock_in_at)
           VALUES ($1, $2, NOW())
           ON CONFLICT DO NOTHING`,
          [tenantId, employeeId]
        );
      } else if (text === 'clock out' || text === 'clock-out' || text === 'เลิกงาน') {
        // Close open punch
        await client.query(
          `UPDATE punches
           SET clock_out_at = NOW()
           WHERE tenant_id = $1 AND employee_id = $2 AND clock_out_at IS NULL
           AND id = (
             SELECT id FROM punches
             WHERE tenant_id = $1 AND employee_id = $2 AND clock_out_at IS NULL
             ORDER BY clock_in_at DESC LIMIT 1
           )`,
          [tenantId, employeeId]
        );
      }
    }

    await client.query('COMMIT');
    // ops metric: line_processed
    res.status(200).json({ ok: true });
  } catch (err) {
    await client.query('ROLLBACK');
    // ops metric: line_error
    console.error('LINE webhook error', err);
    res.status(500).json({ error: 'internal_error' });
  } finally {
    client.release();
  }
});

export default router;
