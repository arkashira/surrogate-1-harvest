# Final Synthesis (Best Parts + Corrected + Actionable)

## 1. Diagnosis (merged, corrected)

- **No persisted manifest**: every training run performs authenticated `list_repo_tree` or `load_dataset` discovery, burning HF API quota and risking 429s.
- **Heterogeneous schemas**: training uses per-file streaming/`load_dataset` across mixed repos → `pyarrow.CastError` at runtime.
- **No CDN-only data path**: authenticated API calls continue during training; public CDN bypass is unused.
- **No commit-cap mitigation**: concentrated writes on one repo risk HF’s 128-writes/hr cap blocking ingestion.
- **No studio reuse guard**: training jobs may recreate Lightning studios and waste quota.

## 2. Single Repo / Date-Folder Scope (chosen)

Target one repo + one date folder (e.g. `datasets/your-org/surrogate1-mirror/batches/mirror-merged/2026-04-29`) to:
- produce `file-list.json` once,
- train using CDN URLs only (zero HF API calls during data load),
- avoid schema mixing by enforcing `{prompt,response}` projection.

# /opt/axentx/vanguard/surrogate1/make_manifest.py
import json
import sys
from pathlib import Path

try:
    from huggingface_hub import list_repo_tree
except ImportError:
    sys.exit("huggingface_hub not installed; run: pip install huggingface_hub")

from config import HFConfig

def build_manifest(repo: str, folder: str, out_path: str) -> None:
    """
    Single authenticated list_repo_tree (non-recursive) for one folder.
    Persists relative file paths and CDN base.
    """
    print(f"Listing {repo}/{folder} (non-recursive)...")
    items = list_repo_tree(repo=repo, path=folder, recursive=False)
    files = sorted(item.rfilename for item in items if item.type == "file")

    manifest = {
        "repo": repo,
        "folder": folder,
        "files": files,
        "cdn_base": HFConfig.cdn_base,
    }

    Path(out_path).parent.mkdir(parents=True, exist_ok=True)
    with open(out_path, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Manifest written to {out_path} ({len(files)} files)")

if __name__ == "__main__":
    cfg = HFConfig()
    build_manifest(cfg.repo, cfg.date_folder, cfg.manifest_path)
```

# /opt/axentx/vanguard/surrogate1/train.py
import json
from pathlib import Path
from typing import Dict, List

import pyarrow.parquet as pq
import torch
from torch.utils.data import Dataset, DataLoader
from transformers import AutoTokenizer, AutoModelForCausalLM, TrainingArguments, Trainer

try:
    from huggingface_hub import hf_hub_download
except ImportError:
    raise RuntimeError("huggingface_hub not installed; run: pip install huggingface_hub")

from config import HFConfig, LightningConfig

# ---- CDN-only, schema-safe dataset ----
class CDNParquetDataset(Dataset):
    """
    Loads parquet shards via local cache (hf_hub_download) to avoid auth during training.
    Projects to {prompt,response} to prevent mixed-schema CastErrors.
    """
    def __init__(self, manifest_path: str, tokenizer_name: str, max_length: int = 1024):
        with open(manifest_path) as f:
            manifest = json.load(f)

        self.repo = manifest["repo"]
        self.folder = manifest["folder"]
        self.files = [f for f in manifest["files"] if f.endswith(".parquet")]
        if not self.files:
            raise ValueError("No parquet files found in manifest")

        self.tokenizer = AutoTokenizer.from_pretrained(tokenizer_name)
        if self.tokenizer.pad_token is None:
            self.tokenizer.pad_token = self.tokenizer.eos_token
        self.max_length = max_length

        self._rows: List[Dict] = []
        for file in self.files:
            local_path = hf_hub_download(
                repo_id=self.repo,
                filename=f"{self.folder}/{file}",
                repo_type="dataset",
            )
            table = pq.read_table(local_path, columns=["prompt", "response"])
            df = table.to_pandas()
            for _, row in df.iterrows():
                self._rows.append({"prompt": row["prompt"], "response": row["response"]})

    def __len__(self) -> int:
        return len(self._rows)

    def __getitem__(self, idx: int):
        item = self._rows[idx]
        text = f"Prompt: {item['prompt']}\nResponse: {item['response']}"
        enc = self.tokenizer(
            text,
            truncation=True,
            max_length=self.max_length,
            padding="max_length",
            return_tensors="pt",
        )
        return {
            "input_ids": enc["input_ids"].squeeze(0),
            "attention_mask": enc["attention_mask"].squeeze(0),
            "labels": enc["input_ids"].squeeze(0).clone(),
        }

# ---- Optional: reuse running studio to save quota ----
def find_running_studio(teamspace: str, studio_name: str):
    """
    Reuse running studio (Lightning best practice) to avoid recreation quota.
    """
    try:
        from lightning import Teamspace
        for s in Teamspace(teamspace).studios:
            if s.name == studio_name and s.status == "Running":
                return s
    except Exception:
        pass
    return None

def main():
    cfg = HFConfig()
    lit_cfg = LightningConfig()

    # Optional studio reuse
    if lit_cfg.reuse_running:
        studio = find_running_studio(lit_cfg.teamspace, lit_cfg.studio_name)
        if studio:
            print(f"Reusing running studio: {studio.name}")

    # Require pre-built manifest (run make_manifest.py once)
    manifest_path = cfg.manifest_path
    if not Path(manifest_path).exists():
        print("Manifest missing; run make_manifest.py first")
        return

    # Dataset and loader
    tokenizer_name = "meta-llama/Llama-3.2-1B-Instruct"  # change as needed
    dataset = CDNParquetDataset(manifest_path, tokenizer_name=tokenizer_name, max_length=1024)
    loader = DataLoader(dataset, batch_size=4, shuffle=True)

    # Minimal model/trainer (replace with your preferred setup)
    model = AutoModelForCausalLM.from_pretrained(tokenizer_name)
    training_args = TrainingArguments(
        output_dir="./out",
        per_device_train_batch_size=4,
        num_train_epochs=1,
        logging_steps=10,
        save_strategy="no",
        report_to="none",
    )

    trainer = Trainer(
        model=model,
        args=training_args,
        train_dataset=dataset,
    )
    trainer.train()

if __name__ == "__main__":
    main()
```
