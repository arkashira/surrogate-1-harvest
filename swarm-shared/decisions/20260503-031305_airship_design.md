## 2) Mac orchestrator (single API call, deterministic output)

`scripts/mac_orchestrator.py`

```python
#!/usr/bin/env python3
"""
Run once per date folder.
- Uses HF API only here (list_repo_tree) to build file_list.json.
- Emits CDN URLs that require no Authorization during training.
"""
import json
import os
import sys
from pathlib import Path

from huggingface_hub import HfApi

REPO_ID = os.getenv("HF_DATASET_REPO", "your-org/surrogate-dataset")
DATE_FOLDER = os.getenv("DATE_FOLDER", "2026-05-03")
OUTPUT_PATH = Path(__file__).parents[2] / "surrogate" / "train" / "file_list.json"

def main() -> None:
    api = HfApi()
    entries = api.list_repo_tree(
        repo_id=REPO_ID,
        path=DATE_FOLDER,
        repo_type="dataset",
        recursive=False,
    )

    files = []
    for entry in entries:
        if entry.type != "file":
            continue
        cdn_url = (
            f"https://huggingface.co/datasets/{REPO_ID}"
            f"/resolve/main/{DATE_FOLDER}/{entry.path}"
        )
        files.append({
            "repo": REPO_ID,
            "path": f"{DATE_FOLDER}/{entry.path}",
            "cdn_url": cdn_url,
        })

    OUTPUT_PATH.parent.mkdir(parents=True, exist_ok=True)
    with open(OUTPUT_PATH, "w") as f:
        json.dump(files, f, indent=2)

    print(f"Wrote {len(files)} files to {OUTPUT_PATH}")

if __name__ == "__main__":
    main()
```

Usage:

```bash
HF_DATASET_REPO=your-org/surrogate-dataset DATE_FOLDER=2026-05-03 python scripts/mac_orchestrator.py
```

## 3) CDN-only dataset loader (zero HF API during training)

`surrogate/train/dataset.py`

```python
from __future__ import annotations

import json
import logging
from pathlib import Path
from typing import Dict, Iterator

import pyarrow as pa
import pyarrow.parquet as pq
import requests
from torch.utils.data import IterableDataset

logger = logging.getLogger(__name__)


class CDNIterableDataset(IterableDataset):
    """
    Loads Parquet shards exclusively via HF CDN (no Authorization header).
    Expects file_list.json produced by mac_orchestrator.py.
    Projects columns to {prompt, response} at parse time.
    """

    def __init__(
        self,
        file_list_path: Path,
        max_retries: int = 3,
        timeout: float = 30.0,
    ) -> None:
        with open(file_list_path) as f:
            self.files = json.load(f)
        if not self.files:
            raise ValueError("file_list.json is empty")
        self.max_retries = max_retries
        self.timeout = timeout

    def _stream_shard(self, cdn_url: str) -> bytes:
        for attempt in range(1, self.max_retries + 1):
            try:
                resp = requests.get(cdn_url, timeout=self.timeout)
                resp.raise_for_status()
                return resp.content
            except Exception as exc:
                wait = 2 ** attempt
                logger.warning(
                    "CDN fetch failed (attempt %s/%s): %s — waiting %ss",
                    attempt,
                    self.max_retries,
                    exc,
                    wait,
                )
                if attempt == self.max_retries:
                    raise
                import time
                time.sleep(wait)
        raise RuntimeError("unreachable")

    def _project_to_pair(self, table: pa.Table) -> Iterator[Dict[str, str]]:
        prompt_col = next((c for c in table.column_names if c in ("prompt", "instruction", "input")), None)
        response_col = next((c for c in table.column_names if c in ("response", "output", "completion")), None)

        if prompt_col is None or response_col is None:
            logger.warning("No prompt/response columns in shard; skipping")
            return

        prompts = table[prompt_col].to_pylist()
        responses = table[response_col].to_pylist()
        for p, r in zip(prompts, responses):
            if p is None or r is None:
                continue
            yield {"prompt": str(p), "response": str(r)}

    def __iter__(self) -> Iterator[Dict[str, str]]:
        for item in self.files:
            cdn_url = item["cdn_url"]
            try:
                raw = self._stream_shard(cdn_url)
                table = pq.read_table(pa.BufferReader(raw))
                yield from self._project_to_pair(table)
            except Exception as exc:
                logger.error("Failed to process shard %s: %s", cdn_url, exc)
                continue
```

## 4) Lightning training script (minimal, correct)

`surrogate/train/train.py`

```python
#!/usr/bin/env python3
"""
Lightning training entrypoint.
- Uses CDNIterableDataset (zero HF API during data loading).
- Designed to be run inside Lightning Studio.
- If studio is stopped, run_lightning.py will restart and rerun this script.
"""
import logging
import os
from pathlib import Path

import lightning as L
import torch
from torch.utils.data import DataLoader

from surrogate.train.dataset import CDNIterableDataset

logger = logging.getLogger(__name__)
logging.basicConfig(level=logging.INFO)


class SurrogateModel(L.LightningModule):
    def __init__(self, lr: float = 1e-4):
        super().__init__()
        self.save_hyperparameters()
        # Replace with your actual Surrogate model architecture.
        self.net = torch.nn.Linear(1024, 1024)
        self.loss_fn = torch.nn.CrossEntropyLoss()

    def training_step(self, batch, batch_idx):
        # Minimal dummy forward; adapt to your tokenizer/model.
        x = torch.randn(batch["prompt"].shape[0], 1024, device=self.device)
        y = self.net(x)
        loss = self.loss_fn(y, torch.zeros_like(y[:, 0]).long())
        self.log("train_loss", loss)
        return loss

    def configure_optimizers(self):
        return torch.optim.AdamW(self.parameters(), lr=self.hparams.lr)


def train(
    file_list_path: Path,
    max_steps: int = 1000,
    batch_size: int = 8,
    num_workers: int = 4,
    resume_from_checkpoint: str | None = None,
) -> None:
    dataset = CDNIterableDataset(file_list_path=file_list_path)
    dataloader = DataLoader(
        dataset,
        batch_size=batch_size,
        num_workers=num_workers,
        persistent_workers=(num_workers > 0),
    )

    model = SurrogateModel()
    trainer = L.Trainer(
        max_steps=max_steps,
        accelerator="gpu" if torch.cuda.is_available() else "cpu",
        devices=1,
        default_root_dir=os.getenv("LIGHTNING_LOGS_DIR", "./lightning_logs"),
        resume_from_checkpoint=resume_from_checkpoint,
    )
    trainer.fit(model, dataloader)
