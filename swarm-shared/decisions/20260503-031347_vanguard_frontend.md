# vanguard / frontend

## 1. Diagnosis

- No CDN-first manifest exists in the frontend; data fetches likely still hit HF API (`list_repo_tree`, `load_dataset`) at runtime, risking 429s and non-reproducible runs.
- Missing deterministic file list keyed by date/slug; frontend cannot pin exact training snapshot without runtime API calls.
- No guardrails to enforce remote-only compute (frontend may trigger local model loads via `from_pretrained`-style flows or heavy inference in browser).
- No studio-reuse or idle-stop handling surfaced in UI; users may unknowingly restart running studios and burn quota.
- No content-addressed asset references (hashes) for datasets/models; cacheability and reproducibility are weak.

## 2. Proposed change

Add a CDN-first manifest loader and snapshot picker to the frontend that:
- Reads a pre-generated `manifest-{date}.json` (produced by ops) listing `{slug, cdn_url, sha256}` for each parquet file.
- Uses only CDN URLs (`https://huggingface.co/datasets/.../resolve/main/...`) for all data fetches — zero HF API calls at runtime.
- Exposes a minimal UI to select date snapshot and pin files by hash.
- Adds studio status guardrail: warns or disables training launch if a running studio exists (reuse path) and shows idle-stop guidance.

Scope:
- Add `/opt/axentx/vanguard/src/lib/manifest.ts` — loader + validation.
- Add `/opt/axentx/vanguard/src/routes/training/+page.svelte` (or equivalent) — snapshot picker + manifest-driven file list.
- Add `/opt/axentx/vanguard/src/lib/studio-guard.ts` — reuse/idle-stop helper.
- Update build/imports to include these modules.

export interface ManifestFile {
  slug: string;          // e.g. "2026-04-29/mirror-merged/file-abc123.parquet"
  cdn_url: string;       // https://huggingface.co/datasets/owner/repo/resolve/main/...
  sha256: string;        // content hash for integrity
}

export interface Manifest {
  date: string;          // snapshot date
  repo: string;          // owner/repo
  files: ManifestFile[];
  generated_at: string;  // ISO timestamp
}

export async function loadManifest(date: string): Promise<Manifest> {
  // Expect manifest placed in /static/manifests/ by ops pipeline
  const res = await fetch(`/manifests/manifest-${date}.json`, { cache: 'no-store' });
  if (!res.ok) throw new Error(`Manifest not found: ${date}`);
  const json = await res.json();
  // Minimal validation
  if (!json.date || !Array.isArray(json.files)) {
    throw new Error('Invalid manifest format');
  }
  return json;
}

export function pickFilesBySlug(manifest: Manifest, patterns: string[]): ManifestFile[] {
  if (!patterns.length) return manifest.files;
  const out: ManifestFile[] = [];
  for (const f of manifest.files) {
    if (patterns.some((p) => f.slug.includes(p))) out.push(f);
  }
  return out;
}

import { Lightning } from '@lightningai/sdk'; // adjust import per project setup

export interface StudioInfo {
  name: string;
  status: 'Running' | 'Stopped' | 'Starting' | 'Unknown';
  machine?: string;
}

export async function listRunningStudios(teamspace?: string): Promise<StudioInfo[]> {
  try {
    // Lightweight check: if SDK unavailable, return empty to avoid runtime crash
    if (typeof Lightning === 'undefined') return [];
    const studios = await Lightning.Teamspace?.studios?.();
    if (!Array.isArray(studios)) return [];
    return studios
      .map((s) => ({
        name: s.name || 'unnamed',
        status: s.status === 'Running' ? 'Running' : (s.status || 'Unknown') as any,
        machine: s.machine,
      }))
      .filter((s) => s.status === 'Running');
  } catch {
    return [];
  }
}

export async function requireStudioReuseOrWarn(
  targetName: string,
  onRunning: (studio: StudioInfo) => void
): Promise<boolean> {
  const running = await listRunningStudios();
  const existing = running.find((s) => s.name === targetName);
  if (existing) {
    onRunning(existing);
    return true;
  }
  return false;
}

<script lang="ts">
  import { onMount } from 'svelte';
  import { loadManifest, type Manifest } from '$lib/manifest';
  import { requireStudioReuseOrWarn } from '$lib/studio-guard';

  let dates: string[] = [];
  let selectedDate: string = '';
  let manifest: Manifest | null = null;
  let loading = false;
  let runningStudio: any = null;
  let message = '';

  // In a real deployment, this list can be generated by ops into /static/dates.json
  async function listAvailableDates() {
    try {
      const res = await fetch('/manifests/dates.json');
      if (res.ok) dates = await res.json();
    } catch {
      // fallback: try common recent dates
      const today = new Date();
      const recent = Array.from({ length: 7 }, (_, i) => {
        const d = new Date(today);
        d.setDate(d.getDate() - i);
        return d.toISOString().slice(0, 10);
      });
      dates = recent;
    }
  }

  async function loadSelected() {
    if (!selectedDate) return;
    loading = true;
    message = '';
    try {
      manifest = await loadManifest(selectedDate);
    } catch (e: any) {
      message = e.message || 'Failed to load manifest';
      manifest = null;
    } finally {
      loading = false;
    }
  }

  async function launchTraining() {
    const studioName = `vanguard-training-${selectedDate}`;
    const reused = await requireStudioReuseOrWarn(studioName, (s) => {
      runningStudio = s;
      message = `Reusing running studio "${s.name}" (machine: ${s.machine || 'default'}) — training will attach to it.`;
    });

    if (reused) return;

    // Proceed to launch (example: redirect to Lightning launcher or call backend)
    message = `No running studio found. Starting new training for ${selectedDate}...`;
    // Placeholder: trigger backend job or Lightning SDK call here
    console.log('Launch training for', selectedDate, manifest);
  }

  onMount(() => {
    listAvailableDates();
  });
</script>
