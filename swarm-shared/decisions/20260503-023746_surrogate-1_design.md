# surrogate-1 / design

## Final Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16`, `DATE`, `HF_TOKEN`
- Uses **one** `list_repo_tree` call (per date folder) to enumerate parquet files, saves manifest JSON, then performs **CDN-only** downloads (no `/api/` auth) to bypass HF rate limits
- Projects each file to `{prompt, response}` at parse time (avoids pyarrow `CastError` on mixed schemas)
- Deduplicates via central `lib/dedup.py` md5 store
- Writes shard output to `batches/public-merged/{DATE}/shard{N}-{HHMMSS}.jsonl`
- Adds retry/backoff for 429 (wait 360s) and 5xx
- Keeps the GitHub Actions matrix (16 shards) unchanged; only the worker script is replaced

def backoff_sleep(attempt: int) -> None:
    # exponential with jitter, min 1s, max 360s
    t = min(360, (2 ** attempt) + (hash(os.getpid() + attempt) % 1000) / 1000.0)
    log.warning("Retry attempt %s, sleeping %.2fs", attempt, t)
    time.sleep(t)

def list_date_files(date_folder: str) -> List[str]:
    """
    Single API call to list parquet files in date folder (non-recursive).
    Returns relative paths.
    """
    for attempt in range(MAX_RETRIES + 1):
        try:
            items = API.list_repo_tree(
                repo_id=DATASET_REPO,
                path=date_folder,
                recursive=False,
                repo_type="dataset",
            )
            files = [it.rfilename for it in items if it.rfilename.lower().endswith(".parquet")]
            log.info("Listed %s parquet files in %s", len(files), date_folder)
            return sorted(files)
        except Exception as exc:
            if attempt == MAX_RETRIES:
                raise
            log.warning("list_repo_tree failed: %s", exc, exc_info=True)
            backoff_sleep(attempt)
    return []

def download_cdn_file(rel_path: str, dest: Path) -> bool:
    """
    Download via CDN URL (no Authorization header).
    """
    url = f"{CDN_BASE}/{rel_path}"
    for attempt in range(MAX_RETRIES + 1):
        try:
            resp = requests.get(
                url,
                timeout=REQUEST_TIMEOUT,
                allow_redirects=True,
                # No Authorization header -> bypasses /api/ rate limits
            )
            if resp.status_code == 429:
                # CDN 429: wait longer
                log.warning("CDN 429 for %s, waiting 360s", rel_path)
                time.sleep(360)
                continue
            resp.raise_for_status()
            dest.write_bytes(resp.content)
            return True
        except Exception as exc:
            if attempt == MAX_RETRIES:
                log.error("Failed to download %s: %s", rel_path, exc)
                return False
            backoff_sleep(attempt)
    return False

def project_to_pair(file_path: Path) -> List[Dict[str, str]]:
    """
    Load parquet and project to {prompt, response} only.
    Tolerates mixed schemas by selecting available text columns.
    """
    try:
        import pyarrow.parquet as pq
    except ImportError:
        log.error("pyarrow is required")
        return []

    try:
        table = pq.read_table(file_path)
    except Exception as exc:
        log.warning("Cannot read parquet %s: %s", file_path, exc)
        return []

    cols = {c.lower(): c for c in table.column_names}
    prompt_col = None
    response_col = None

    for p in ("prompt", "instruction", "input", "question", "text"):
        if p in cols:
            prompt_col = cols[p]
            break
    for r in ("response", "output", "answer", "completion"):
        if r in cols:
            response_col = cols[r]
            break

    if prompt_col is None or response_col is None:
        log.warning("No prompt/response columns in %s (cols=%s)", file_path, list(cols))
        return []

    pairs = []
    for i in range(table.num_rows):
        row = table.slice(i, 1).to_pydict()
        prompt = str(row[prompt_col][0]) if row[prompt_col][0] is not None else ""
        response = str(row[response_col][0]) if row[response_col][0] is not None else ""
        if prompt.strip() and response.strip():
            pairs.append({"prompt": prompt.strip(), "response": response.strip()})
    return pairs

def should_process(rel_path: str, shard_id: int, shard_total: int) -> bool:
    """Deterministic shard assignment by path hash."""
    h = int(hashlib.md5(rel_path.encode()).hexdigest(), 16)
    return (h % shard_total) == shard_id
