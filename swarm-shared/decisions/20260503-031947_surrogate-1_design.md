# surrogate-1 / design

## Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16`, `DATE`, `HF_TOKEN` via env
- Single `list_repo_tree(path, recursive=False)` per date folder → saves JSON manifest
- Deterministic shard assignment: `hash(slug) % SHARD_TOTAL == SHARD_ID`
- Downloads only assigned files via **HF CDN bypass** (`resolve/main/...` — no auth, no API rate limit)
- Projects to `{prompt, response}` only at parse time (avoids pyarrow CastError on mixed schemas)
- Dedups via central md5 store (`lib/dedup.py`)
- Writes `batches/public-merged/{DATE}/shard{N}-{HHMMSS}.jsonl`
- Returns exit code 0 on success, non-zero on fatal failure (so GH Actions matrix fails fast)

### Why this is the highest-value incremental improvement
- Directly applies **HF CDN bypass** (the key 2026-04-29 insight) to eliminate API rate limits during data load.
- Eliminates `load_dataset(streaming=True)` mixed-schema hazards by per-file download + projection.
- Keeps the 16-shard parallel architecture intact (fits existing GH Actions matrix).
- Small, focused change (<2h) with immediate reliability + throughput gains.

def hf_api_get(path: str, params: dict = None) -> Any:
    url = f"https://huggingface.co/api/{path.lstrip('/')}"
    headers = {"Authorization": f"Bearer {HF_TOKEN}"}
    r = requests.get(url, headers=headers, params=params, timeout=30)
    if r.status_code == 429:
        # Respect rate limit: wait 360s as per pattern
        print("WARN: HF API 429 — waiting 360s", file=sys.stderr)
        import time
        time.sleep(360)
        r = requests.get(url, headers=headers, params=params, timeout=30)
    r.raise_for_status()
    return r.json()

def download_via_cdn(repo_path: str, dest: Path) -> None:
    """
    CDN bypass: no Authorization header.
    URL: https://huggingface.co/datasets/{owner}/{repo}/resolve/main/{repo_path}
    """
    url = f"https://huggingface.co/datasets/{REPO_OWNER}/{REPO_NAME}/resolve/main/{repo_path}"
    with requests.get(url, timeout=60, stream=True) as r:
        r.raise_for_status()
        with open(dest, "wb") as f:
            for chunk in r.iter_content(chunk_size=8192):
                f.write(chunk)

def project_to_pair(local_path: Path) -> List[Dict[str, str]]:
    """
    Load file and project to {prompt, response}.
    Supports:
      - JSONL with prompt/response fields
      - Parquet with prompt/response columns
    If schema differs, attempt best-effort extraction.
    """
    suffix = local_path.suffix.lower()
    pairs = []

    try:
        if suffix == ".parquet":
            table = pq.read_table(local_path, columns=["prompt", "response"])
            prompts = table.column("prompt").to_pylist()
            responses = table.column("response").to_pylist()
            for p, r in zip(prompts, responses):
                if p is not None and r is not None:
                    pairs.append({"prompt": str(p), "response": str(r)})
            return pairs

        if suffix == ".jsonl":
            with open(local_path, "r", encoding="utf-8") as f:
                for line in f:
                    line = line.strip()
                    if not line:
                        continue
                    obj = json.loads(line)
                    p = obj.get("prompt")
                    r = obj.get("response")
                    if p is not None and r is not None:
                        pairs.append({"prompt": str(p), "response": str(r)})
            return pairs

    except Exception as e:
        # Fallback: try generic parquet read without column projection
        try:
            table = pq.read_table(local_path)
            # Find columns that look like prompt/response
            cols = table.column_names
            prompt_col = next((c for c in cols if "prompt" in c.lower()), None)
            response_col = next((c for c in cols if "response" in c.lower()), None)
            if prompt_col and response_col:
                prompts = table.column(prompt_col).to_pylist()
                responses = table.column(response_col).to_pylist()
                for p, r in zip(prompts, responses):
                    if p is not None and r is not None:
                        pairs.append({"prompt": str(p), "response": str(r)})
                return pairs
        except Exception:
            pass

    # Last resort: skip file
    print(f"WARN: could not project {local_path} to prompt/response", file=sys.stderr)
    return pairs
