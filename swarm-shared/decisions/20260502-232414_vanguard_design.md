# vanguard / design

## Final Consolidated Implementation

### 1) Diagnosis (merged, de-duplicated)
- **No persistent file-list cache** → repeated `list_repo_tree`/`load_dataset` triggers 429 rate limits.
- **Lightning Studio lifecycle not reused** → quota burned on recreation; idle-stop deaths silently kill jobs.
- **No CDN-only data path** → training performs authenticated API calls instead of public CDN fetches.
- **Missing idle-timeout guard** → studios killed by Lightning idle-stop cause silent failures.
- **No deterministic sibling-repo sharding** → HF commit cap (128/hr/repo) throttles high-volume ingest.

### 2) Single Proposed Change
Add a lightweight ingestion/training orchestration module that:
- Pre-lists HF dataset files once, caches to JSON, and embeds the list in training scripts.
- Reuses running Lightning Studio instances (or restarts stopped ones) instead of recreating; adds idle-timeout guard and health checks.
- Uses CDN-only URLs (`resolve/main/...`) for data loading with zero API calls during training.
- Shards writes across 5 sibling repos by deterministic hash when uploading enriched parquet files.

# /opt/axentx/vanguard/vanguard/lightning_utils.py
import os
import time
from typing import Optional

try:
    from lightning_sdk import Teamspace, Studio, Machine
    _HAS_LIGHTNING = True
except Exception:
    _HAS_LIGHTNING = False
    Teamspace = Studio = Machine = None

LIGHTNING_CLOUD = os.getenv("LIGHTNING_CLOUD", "lightning-public-prod")
MACHINE_TYPE = os.getenv("LIGHTNING_MACHINE", "L40S")
STUDIO_NAME = os.getenv("LIGHTNING_STUDIO", "vanguard-train")
IDLE_TIMEOUT_MIN = int(os.getenv("IDLE_TIMEOUT_MIN", "30"))

def _now() -> float:
    return time.time()

def get_or_create_studio(
    teamspace: str,
    studio_name: str = STUDIO_NAME,
    machine: str = MACHINE_TYPE,
    cloud: str = LIGHTNING_CLOUD,
    max_retries: int = 3,
) -> Optional[Studio]:
    """
    Reuse a running studio or restart/create one.
    Returns Studio instance or None if unavailable.
    """
    if not _HAS_LIGHTNING:
        print("Lightning SDK not available; skipping studio management.")
        return None

    ts = Teamspace(teamspace)

    # reuse running
    for s in ts.studios:
        if s.name == studio_name:
            if s.status == "running":
                print(f"Reusing running studio: {studio_name}")
                return s
            elif s.status == "stopped":
                print(f"Restarting stopped studio: {studio_name}")
                try:
                    s.start(machine=Machine(machine, cloud=cloud))
                    return s
                except Exception as exc:
                    print(f"Failed to restart studio: {exc}")
                    return None

    # create if not exists
    for attempt in range(max_retries):
        try:
            print(f"Creating studio: {studio_name} on {cloud}/{machine}")
            s = Studio.create(
                teamspace=teamspace,
                name=studio_name,
                machine=Machine(machine, cloud=cloud),
                create_ok=True,
            )
            return s
        except Exception as exc:
            wait = 30 * (attempt + 1)
            print(f"Studio creation attempt {attempt+1} failed: {exc}; retry in {wait}s")
            time.sleep(wait)
    return None


def wait_for_studio_running(studio: Studio, timeout: int = 600, interval: int = 15) -> bool:
    """Wait until studio is running; return False on timeout."""
    if studio is None:
        return False
    elapsed = 0
    while elapsed < timeout:
        studio.refresh()
        if studio.status == "running":
            return True
        time.sleep(interval)
        elapsed += interval
    return False


def assert_studio_alive_with_guard(studio: Studio, last_active_ts: float) -> bool:
    """
    Guard against idle-timeout deaths.
    Returns True if studio is running and not past idle timeout.
    """
    if studio is None:
        return False
    studio.refresh()
    if studio.status != "running":
        print(f"Studio {studio.name} is not running (status={studio.status}).")
        return False
    idle_min = (time.time() - last_active_ts) / 60.0
    if idle_min > IDLE_TIMEOUT_MIN:
        print(f"Studio idle timeout exceeded ({idle_min:.1f}m > {IDLE_TIMEOUT_MIN}m).")
        return False
    return True

def list_date_files(date_folder: str, out_json: str) -> List[str]:
    """
    Single API call to list files in date_folder (non-recursive).
    Save list to out_json for embedding in training scripts.
    """
    try:
        from huggingface_hub import list_repo_tree
    except ImportError:
        raise RuntimeError("huggingface_hub required for listing")

    items = list_repo_tree(
        repo_id=HF_REPO,
        path=date_folder,
        repo_type="dataset",
        revision=HF_BRANCH,
        recursive=False,
    )
    files = [it.rfilename for it in items if it.type == "file"]
    Path(out_json).write_text(json.dumps(files, indent=2))
    print(f"Saved {len(files)} files to {out_json}")
    return files


def cdn_url(repo: str, file_path: str, branch: str = HF_BRANCH) -> str:
    """Public CDN URL — no Authorization header required."""
    return f"https://huggingface.co/datasets/{repo}/resolve/{branch}/{file_path}"


def download_via_cdn(file_path: str, local_path: str, repo: str = HF_REPO, branch: str = HF_BRANCH) -> None:
    """Download using CDN (bypasses API rate limits)."""
    url = cdn_url(repo, file_path, branch)
    resp = requests.get(url, timeout=60)
    resp.raise_for_status()
    Path(local_path).parent.mkdir(parents=True, exist_ok=True)
    Path(local_path).write_bytes(resp.content)
