### 3) CDN fetch utility (`src/lib/cdn.ts`)
```ts
// CDN-first: public/ is served at runtime with zero Authorization and no HF API calls.
const CDN_BASE = '/knowledge';

export interface RelatedDoc {
  slug: string;
  title: string;
  url: string;
  relevanceScore: number;
}

export interface TopHubPayload {
  hub: {
    slug: string;
    title: string;
    description: string;
    url: string;
  };
  related: RelatedDoc[];
  generatedAt: string;
  source: string;
}

export async function fetchTopHub(): Promise<TopHubPayload | null> {
  try {
    const res = await fetch(`${CDN_BASE}/top-hub.json`, { cache: 'no-store' });
    if (!res.ok) return null;
    return (await res.json()) as TopHubPayload;
  } catch {
    return null;
  }
}
```

### 4) Hook for robust CDN fetch (`src/hooks/useCDNTopHub.ts`)
```ts
import { useEffect, useState, useCallback } from 'react';
import { fetchTopHub, TopHubPayload } from '../lib/cdn';

export function useCDNTopHub(pollIntervalMs = 0) {
  const [payload, setPayload] = useState<TopHubPayload | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<Error | null>(null);

  const load = useCallback(async () => {
    setLoading(true);
    try {
      const data = await fetchTopHub();
      setPayload(data);
      setError(null);
    } catch (err) {
      setError(err instanceof Error ? err : new Error('Failed to fetch top-hub'));
      setPayload(null);
    } finally {
      setLoading(false);
    }
  }, []);

  useEffect(() => {
    load();
    if (pollIntervalMs > 0) {
      const id = setInterval(load, pollIntervalMs);
      return () => clearInterval(id);
    }
  }, [load, pollIntervalMs]);

  return { payload, loading, error, refetch: load };
}
```

### 5) TopHubSignalPanel component (`src/components/TopHubSignalPanel.tsx`)
```tsx
import React from 'react';
import { useCDNTopHub } from '../hooks/useCDNTopHub';
import './TopHubSignalPanel.css';

export const TopHubSignalPanel: React.FC = () => {
  const { payload, loading, error } = useCDNTopHub();

  if (loading) {
    return (
      <div className="top-hub-panel loading" aria-busy="true">
        Loading signals…
      </div>
    );
  }

  if (error || !payload) {
    return (
      <div className="top-hub-panel empty">
        No hub signals available.
      </div>
    );
  }

  return (
    <aside className="top-hub-panel" aria-label="Top hub signal">
      <header className="top-hub-header">
        <h3 className="top-hub-title">Top Hub</h3>
        <span className="top-hub-badge">{payload.hub.slug}</span>
      </header>

      <section className="top-hub-card">
        <h4 className="hub-title">
          <a href={payload.hub.url} target="_blank" rel="noopener noreferrer">
            {payload.hub.title}
          </a>
        </h4>
        <p className="hub-desc">{payload.hub.description}</p>
      </section>

      <section className="related-docs" aria-label="Related docs">
        <h4 className="related-title">Related Signals</h4>
        <ul className="related-list">
          {payload.related.map((doc) => (
            <li key={doc.slug} className="related-item">
              <a
                href={doc.url}
                target="_blank"
                rel="noopener noreferrer"
                className="related-link"
              >
                <span className="doc-title">{doc.title}</span>
                <span className="doc-score">{Math.round(doc.relevanceScore * 100)}%</span>
              </a>
            </li>
          ))}
        </ul>
      </section>

      <footer className="panel-footer">
        <small>
          Source: {payload.source} • Updated {new Date(payload.generatedAt).toLocaleDateString()}
        </small>
      </footer>
    </aside>
  );
};
```
