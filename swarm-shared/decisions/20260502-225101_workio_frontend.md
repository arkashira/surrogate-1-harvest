# workio / frontend

## 1. Diagnosis

- Attendance events are submitted directly from UI with no durable queue; offline or flaky network causes lost clock-ins/outs.
- No idempotency key on events → retries can create duplicate attendance records on the backend.
- localStorage is used for transient state; it is not transactional, can be cleared by browser/storage pressure, and is not cross-tab safe.
- No exponential backoff + jitter on retry; thundering-herd and battery drain on reconnect.
- No background flush when app regains connectivity; user must manually retry or wait for next interaction.

## 2. Proposed change

Add a lightweight IndexedDB-backed queue (`src/lib/queue/attendanceQueue.ts`) and integrate it into the existing clock-in/out flow (`src/features/attendance/hooks/useAttendance.ts`). Scope:
- Create queue module (adds ~120 lines).
- Modify `useAttendance` to enqueue instead of direct POST, and start auto-flush on mount/online.
- Keep API layer unchanged (single `submitAttendance` call) — queue will call it.

### 3.2 Create queue module

`src/lib/queue/attendanceQueue.ts`

```ts
import { openDB, DBSchema, IDBPDatabase } from 'idb';

interface AttendanceQueueDB extends DBSchema {
  events: {
    key: string; // idempotencyKey
    value: {
      idempotencyKey: string;
      type: 'CLOCK_IN' | 'CLOCK_OUT';
      payload: {
        userId: string;
        timestamp: string; // ISO
        latitude?: number;
        longitude?: number;
        imageDataUrl?: string;
      };
      attempts: number;
      createdAt: number;
      nextRetryAt: number;
      // denormalized for quick peek
      status: 'pending' | 'sending' | 'failed' | 'done';
    };
    indexes: { status: string; nextRetryAt: number };
  };
}

const DB_NAME = 'workio-attendance-queue';
const STORE = 'events';

async function getDB(): Promise<IDBPDatabase<AttendanceQueueDB>> {
  return openDB<AttendanceQueueDB>(DB_NAME, 1, {
    upgrade(db) {
      const store = db.createObjectStore(STORE, { keyPath: 'idempotencyKey' });
      store.createIndex('status', 'status', { unique: false });
      store.createIndex('nextRetryAt', 'nextRetryAt', { unique: false });
    },
  });
}

function randomJitter(baseMs: number) {
  return baseMs * (0.8 + Math.random() * 0.4); // ±20%
}

export async function enqueueAttendance(params: {
  idempotencyKey: string;
  type: 'CLOCK_IN' | 'CLOCK_OUT';
  payload: {
    userId: string;
    timestamp: string;
    latitude?: number;
    longitude?: number;
    imageDataUrl?: string;
  };
}) {
  const db = await getDB();
  await db.add(STORE, {
    idempotencyKey: params.idempotencyKey,
    type: params.type,
    payload: params.payload,
    attempts: 0,
    createdAt: Date.now(),
    nextRetryAt: Date.now(),
    status: 'pending',
  });
}

export async function peekNextBatch(limit = 10): Promise<AttendanceQueueDB['events']['value'][]> {
  const db = await getDB();
  const tx = db.transaction(STORE, 'readonly');
  const idx = tx.store.index('nextRetryAt');
  const pending = await idx.getAll(IDBKeyRange.upperBound(Date.now()), limit);
  // also include sending (in case of crash) so we can retry
  const sending = await idx.getAll(IDBKeyRange.lowerBound(0), limit);
  tx.done;
  const all = [...pending, ...sending].filter((i) => i.status !== 'done');
  // dedupe by key
  const byKey = new Map<string, AttendanceQueueDB['events']['value']>();
  for (const item of all) {
    if (!byKey.has(item.idempotencyKey)) byKey.set(item.idempotencyKey, item);
  }
  return Array.from(byKey.values()).sort((a, b) => a.nextRetryAt - b.nextRetryAt).slice(0, limit);
}

export async function markSending(idempotencyKey: string) {
  const db = await getDB();
  await db.put(STORE, { ...(await db.get(STORE, idempotencyKey)), status: 'sending' }, idempotencyKey);
}

export async function markDone(idempotencyKey: string) {
  const db = await getDB();
  const tx = db.transaction(STORE, 'readwrite');
  await tx.store.delete(idempotencyKey);
  await tx.done;
}

export async function markFailed(idempotencyKey: string, attempts: number) {
  const db = await getDB();
  const backoff = Math.min(300_000, 1_000 * Math.pow(2, attempts)); // cap 5m
  const nextRetryAt = Date.now() + randomJitter(backoff);
  await db.put(
    STORE,
    {
      ...(await db.get(STORE, idempotencyKey)),
      attempts,
      status: 'failed',
      nextRetryAt,
    },
    idempotencyKey
  );
}
```
