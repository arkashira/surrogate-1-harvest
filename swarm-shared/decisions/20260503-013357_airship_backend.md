### 1. Locate backend training entrypoint  
Check `/opt/axentx/airship/surrogate/` or `/opt/axentx/airship/arkship/` for `train.py`, `training/`, or `launcher.py`.

### 2. Create CDN-bypass dataset loader (`backend/training/cdn_loader.py`)  
- `list_date_folder(repo, date_folder, cache_dir)` → uses `list_repo_tree(recursive=False)` once, caches to JSON.  
- `stream_parquet_from_cdn(repo, file_path, columns=("prompt", "response"))` → downloads via `requests.get(CDN_URL, timeout=60)` and yields rows via `pyarrow.parquet`.  
- No `load_dataset(streaming=True)` on heterogeneous repos.  
- CDN URL template: `https://huggingface.co/datasets/{repo}/resolve/main/{file_path}`

### 3. Lightning Studio reuse wrapper (`backend/training/lightning_studio.py`)  
- `get_or_create_studio(name, machine=Machine.L40S)` → lists `Teamspace.studios`, reuses if `status == 'running'`, else creates.  
- `ensure_running(studio, machine)` → if stopped, restarts with `studio.start(machine=machine)`.  
- `.run()` only called after confirming running.

### 4. Training launcher endpoint/service (`backend/training/launcher.py`)  
- Expose a backend endpoint (FastAPI or internal CLI) that:  
  - Accepts `{repo, date_folder, epochs, ...}`.  
  - Calls `list_date_folder` once, writes `filelist.json` into the training package.  
  - Builds `train.py` arguments embedding `filelist.json` path.  
  - Uses `lightning_studio` to get running studio and launches training with `studio.run(...)`.

### 5. Embed filelist in training script (`backend/training/train.py`)  
- Reads `filelist.json`, iterates files, uses `stream_parquet_from_cdn` to build dataset iterable (zero HF API calls during training).

### 6. Error handling & retries  
- CDN download retries with backoff.  
- On HF API 429 during listing: wait 360s and retry once (only affects listing, not training).  
- Validate studio status before each `.run()`.

### 1) HF CDN-bypass loader (`backend/training/cdn_loader.py`)
```python
import json
import time
from pathlib import Path
from typing import Dict, List, Iterator

import pyarrow.parquet as pq
import requests
from huggingface_hub import HfApi

HF_API = HfApi()
CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{file_path}"

def list_date_folder(repo: str, date_folder: str, cache_dir: Path) -> List[str]:
    cache_dir.mkdir(parents=True, exist_ok=True)
    cache_path = cache_dir / f"{repo.replace('/', '_')}_{date_folder.replace('/', '_')}.json"

    if cache_path.exists():
        return json.loads(cache_path.read_text())

    items = HF_API.list_repo_tree(repo=repo, path=date_folder, recursive=False)
    files = [it.rfilename for it in items if it.type == "file"]

    cache_path.write_text(json.dumps(files, indent=2))
    return files

def stream_parquet_from_cdn(repo: str, file_path: str, columns=("prompt", "response")) -> Iterator[Dict]:
    url = CDN_TEMPLATE.format(repo=repo, file_path=file_path)
    resp = requests.get(url, timeout=60)
    resp.raise_for_status()

    table = pq.read_table(pq.ParquetFile(pq.BufferReader(resp.content)))
    if columns:
        table = table.select(columns)

    for batch in table.to_batches():
        cols = {c: batch.column(c).to_pylist() for c in batch.column_names}
        for i in range(batch.num_rows):
            yield {k: v[i] for k, v in cols.items()}
```

### 2) Lightning Studio reuse (`backend/training/lightning_studio.py`)
```python
import time
from typing import Optional

from lightning import LightningWork, Machine, Teamspace

def get_or_create_studio(name: str, machine: Machine = Machine.L40S, timeout: int = 300) -> LightningWork:
    for s in Teamspace.studios:
        if s.name == name and s.status == "running":
            return s

    studio = LightningWork(cloud_compute=machine, name=name, create_ok=True)
    studio.start(machine=machine)

    start = time.time()
    while time.time() - start < timeout:
        if studio.status == "running":
            return studio
        time.sleep(5)

    raise RuntimeError(f"Studio {name} failed to start within {timeout}s")

def ensure_running(studio: LightningWork, machine: Machine) -> None:
    if studio.status != "running":
        studio.start(machine=machine)
```

### 3) Training launcher (`backend/training/launcher.py`)
```python
import json
from pathlib import Path

from .cdn_loader import list_date_folder
from .lightning_studio import ensure_running, get_or_create_studio
from lightning import Machine

def launch_training(
    repo: str,
    date_folder: str,
    studio_name: str = "surrogate-train",
    epochs: int = 1,
) -> None:
    cache_dir = Path("/tmp/hf_filelists")
    files = list_date_folder(repo, date_folder, cache_dir)

    filelist_path = cache_dir / f"filelist_{repo.replace('/', '_')}_{date_folder.replace('/', '_')}.json"
    filelist_path.write_text(json.dumps(files, indent=2))

    train_script = Path(__file__).parent / "train.py"
    cmd = [
        "python", str(train_script),
        "--repo", repo,
        "--filelist", str(filelist_path),
        "--epochs", str(epochs),
    ]

    studio = get_or_create_studio(studio_name, machine=Machine.L40S)
    ensure_running(studio, machine=Machine.L40S)

    studio.run(command=" ".join(cmd), watch=False)
```
