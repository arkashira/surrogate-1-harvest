### 1) scripts/list_hf_files.py

#!/usr/bin/env python3
"""
One-time Mac-side script (run after rate-limit window clears).
Lists a single date folder in a HF dataset repo and saves file list to JSON.
Usage:
  python scripts/list_hf_files.py \
    --repo "org/surrogate-data" \
    --folder "batches/mirror-merged/2026-05-03" \
    --out file_list.json
"""
import argparse
import json
import os
import sys
from huggingface_hub import HfApi

def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True, help="HF dataset repo (user/repo)")
    parser.add_argument("--folder", required=True, help="Folder path inside repo")
    parser.add_argument("--out", default="file_list.json", help="Output JSON path")
    args = parser.parse_args()

    api = HfApi()
    # recursive=False to avoid 100x pagination; we only want one date folder
    tree = api.list_repo_tree(repo_id=args.repo, path=args.folder, recursive=False)
    files = [
        f.rfilename
        for f in (tree if isinstance(tree, list) else tree.to_tree())
        if not f.rfilename.endswith("/")
    ]

    # Save as simple list; training script will embed this.
    out_path = os.path.abspath(args.out)
    with open(out_path, "w", encoding="utf-8") as f:
        json.dump({"repo": args.repo, "folder": args.folder, "files": files}, f, indent=2)

    print(f"Wrote {len(files)} files to {out_path}")

if __name__ == "__main__":
    main()
```

Make executable:

### 2) surrogate/data/cdn_dataset.py

# surrogate/data/cdn_dataset.py
import json
import os
from typing import Dict, List, Optional

import pyarrow as pa
import pyarrow.parquet as pq
import torch
from datasets import Features, Sequence, Value
from huggingface_hub import hf_hub_download
from torch.utils.data import Dataset


class CDNDataset(Dataset):
    """
    CDN-only dataset loader.
    - No load_dataset(streaming=True) on heterogeneous repos.
    - Downloads individual parquet files via hf_hub_download / CDN.
    - Projects to {prompt, response} at parse time.
    - Attribution stored in filename pattern, not extra columns.
    """

    def __init__(
        self,
        file_list_path: str,
        split: str = "train",
        cache_dir: Optional[str] = None,
    ) -> None:
        super().__init__()
        with open(file_list_path, "r", encoding="utf-8") as f:
            meta = json.load(f)
        self.repo = meta["repo"]
        self.folder = meta["folder"]
        self.files: List[str] = meta["files"]
        self.cache_dir = cache_dir or os.getenv("HF_HOME", os.path.expanduser("~/.cache/huggingface"))

        # Keep only parquet files
        self.parquet_files = [f for f in self.files if f.endswith(".parquet")]
        if not self.parquet_files:
            raise ValueError(f"No parquet files found in {self.folder}")

        # Build index
        self._examples: List[Dict[str, str]] = []
        self._build_index()

    def _download_and_read(self, rfilename: str) -> List[Dict[str, str]]:
        # CDN bypass: use hf_hub_download (uses CDN, no Authorization header rate limit)
        local_path = hf_hub_download(
            repo_id=self.repo,
            filename=rfilename,
            cache_dir=self.cache_dir,
            force_download=False,
            resume_download=True,
        )
        try:
            table = pq.read_table(local_path, columns=["prompt", "response"])
        except (pa.ArrowInvalid, KeyError):
            # Fallback: try any columns and rename
            table = pq.read_table(local_path)
            cols = table.column_names
            prompt_col = next((c for c in ["prompt", "input", "question"] if c in cols), cols[0])
            response_col = next((c for c in ["response", "output", "answer"] if c in cols), cols[-1])
            table = table.select([prompt_col, response_col]).rename_columns(["prompt", "response"])

        df = table.to_pandas()
        records = df.to_dict(orient="records")
        for r in records:
            if not isinstance(r.get("prompt"), str) or not isinstance(r.get("response"), str):
                continue
            r["prompt"] = r["prompt"].strip()
            r["response"] = r["response"].strip()
        return records

    def _build_index(self) -> None:
        for rf in self.parquet_files:
            try:
                recs = self._download_and_read(rf)
                self._examples.extend(recs)
            except Exception as exc:
                # Log and skip bad files to avoid crashing training
                print(f"Skipping {rf}: {exc}")

    def __len__(self) -> int:
        return len(self._examples)

    def __getitem__(self, idx: int) -> Dict[str, str]:
        return self._examples[idx]
```
