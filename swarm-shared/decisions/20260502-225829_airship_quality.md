### 2) Implementation (single script, ~60–90 lines)

`scripts/airship-discover.sh`
```bash
#!/usr/bin/env bash
# airship-discover.sh
# Deterministic discovery + manifest generation for HF CDN-safe ingestion.
# Usage:
#   bash scripts/airship-discover.sh \
#     --repo <hf-repo> \
#     --date-folder <YYYY-MM-DD> \
#     --out-dir <path> \
#     --slug <dataset-slug>
#
# Produces:
#   <out-dir>/status.json
#   <out-dir>/manifest.json
#   <out-dir>/batches/mirror-merged/<date>/<slug>.parquet  (if --create-parquet)

set -euo pipefail
IFS=$'\n\t'

# ---- defaults ----
HF_REPO=""
DATE_FOLDER=""
OUT_DIR="./discover-output"
SLUG=""
CREATE_PARQUET=0
HF_TOKEN="${HF_TOKEN:-}"

# ---- helpers ----
log()  { echo "[$(date -u +"%Y-%m-%dT%H:%M:%SZ")] $*" >&2; }
die()  { log "ERROR: $*"; exit 1; }

usage() {
  cat <<EOF
Usage: $0 --repo <hf-repo> --date-folder <YYYY-MM-DD> --slug <slug> [options]

Options:
  --repo          HuggingFace dataset repo (e.g., axentx/airship-mirror)
  --date-folder   Date folder to list (YYYY-MM-DD)
  --slug          Dataset slug used for filename (sanitized)
  --out-dir       Output directory (default: ./discover-output)
  --create-parquet  Produce a minimal parquet placeholder (requires python/pandas/pyarrow)
  -h, --help      Show this help
EOF
  exit 1
}

sanitize_slug() {
  # keep alphanumeric, dash, underscore; collapse others to single dash
  echo "$1" | sed -E 's/[^a-zA-Z0-9_-]+/-/g' | sed -E 's/^-+|-+$//g'
}

# ---- parse args ----
while [[ $# -gt 0 ]]; do
  case $1 in
    --repo) HF_REPO="$2"; shift 2 ;;
    --date-folder) DATE_FOLDER="$2"; shift 2 ;;
    --slug) SLUG="$2"; shift 2 ;;
    --out-dir) OUT_DIR="$2"; shift 2 ;;
    --create-parquet) CREATE_PARQUET=1; shift ;;
    -h|--help) usage ;;
    *) die "Unknown option: $1" ;;
  esac
done

[[ -z "$HF_REPO" ]] && die "--repo is required"
[[ -z "$DATE_FOLDER" ]] && die "--date-folder is required"
[[ -z "$SLUG" ]] && die "--slug is required"

SLUG=$(sanitize_slug "$SLUG")
[[ -z "$SLUG" ]] && die "slug is empty after sanitization"

mkdir -p "$OUT_DIR"
mkdir -p "$OUT_DIR/batches/mirror-merged/$DATE_FOLDER"

# ---- list files (single non-recursive API call) ----
# Use HF API to list one folder; save to file-list.json for embedding in training.
# If rate-limited, fallback to CDN-only mode (no API) by using a pre-saved list.
FILE_LIST="$OUT_DIR/file-list.json"
log "Listing $HF_REPO/tree/$DATE_FOLDER (non-recursive)"

if curl -fsSL \
  -H "Authorization: Bearer ${HF_TOKEN}" \
  -H "Content-Type: application/json" \
  "https://huggingface.co/api/datasets/${HF_REPO}/tree/${DATE_FOLDER}?recursive=false" \
  -o "$FILE_LIST" 2>"$FILE_LIST.err"; then
  log "File list saved to $FILE_LIST"
else
  CODE=$?
  log "WARNING: HF API list failed (exit $CODE). Will operate in CDN-only mode."
  # produce minimal deterministic manifest without file list
  echo '[]' > "$FILE_LIST"
fi

# ---- build manifest.json (deterministic keys, sorted) ----
MANIFEST="$OUT_DIR/manifest.json"
DATE_NOW=$(date -u +"%Y-%m-%dT%H:%M:%SZ")

# extract filenames if available; otherwise empty array
FILES_JSON=$(python3 -c "
import json, sys
with open('$FILE_LIST') as f:
    data = json.load(f)
if isinstance(data, dict) and 'tree' in data:
    items = data['tree']
else:
    items = data if isinstance(data, list) else []
names = [x.get('path', '') for x in items if isinstance(x, dict) and x.get('type') == 'file']
names.sort()
print(json.dumps(names, sort_keys=True))
" 2>/dev/null || echo '[]')

# build CDN URLs for each file (bypasses API auth during training)
CDN_URLS_JSON=$(python3 -c "
import json, sys
files = json.loads('''$FILES_JSON''')
base = 'https://huggingface.co/datasets/${HF_REPO}/resolve/main/${DATE_FOLDER}'
urls = [f'{base}/{f}' for f in files]
print(json.dumps(urls, sort_keys=True))
")

cat > "$MANIFEST" <<MANIFEST_EOF
{
  "repo": "${HF_REPO}",
  "date_folder": "${DATE_FOLDER}",
  "slug": "${SLUG}",
  "generated_at_utc": "${DATE_NOW}",
  "generator": "airship-discover",
  "file_count": $(echo "$FILES_JSON" | python3 -c "import json,sys;print(len(json.load(sys.stdin)))"),
  "files": $FILES_JSON,
  "cdn_urls": $CDN_URLS_JSON,
  "parquet_path": "batches/mirror-merged/${DATE_FOLDER}/${SLUG}.parquet",
  "cdn_parquet_url": "https://huggingface.co/datasets/${HF_REPO}/resolve/main/batches/mirror-merged/${DATE_FOLDER}/${SLUG}.parquet",
  "schema_projection": ["prompt", "response"],
  "attribution": "filename-only",
  "notes": "CDN-only ingestion recommended. Do not add source/ts columns."
}
MANIFEST_EOF

# sort keys deterministically (python ensures stable ordering)
python3 -c "
import json, sys
with open('$MANIFEST') as f:
    obj = json.load(f)
with open('$MANIFEST', 'w') as f:
    json.dump(obj, f, sort_keys=True, indent=2, ensure_ascii=False)
" 2>/dev/null || true

# ---- produce minimal parquet placeholder (optional) ----
PARQUET_PATH="$OUT_DIR/batches/mirror-merged/$DATE_FOLDER/${SLUG}.parquet"
if [[ $CREATE_PARQUET -eq 1 ]]; then
  log "Creating placeholder parquet at $PARQUET_PATH"
  python3 -c "
import pandas as pd, pyarrow as pa, pyarrow.parquet as pq, sys
df = pd.DataFrame([], columns=['prompt', 'response'])
table = pa.Table.from_pandas(df)
pq.write_table(table, '$PARQUET_PATH')
" || log "WARNING: parquet creation failed (missing pandas/pyarrow?). Continuing."
fi

# ---- status.json ----
STATUS="$OUT_DIR/status.json"
cat > "$STATUS
