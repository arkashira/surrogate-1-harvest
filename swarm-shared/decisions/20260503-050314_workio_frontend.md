# workio / frontend

## Final consolidated solution

**Scope**: frontend-only changes that prevent duplicate clock events and make failures/retries safe and observable.

### 1. Core principles (resolve contradictions)
- **Correctness first**: client-side idempotency complements (never replaces) server-side enforcement. The server must remain the source of truth and must accept/require an `Idempotency-Key` header and enforce uniqueness per user/tenant/day.
- **Actionability**: provide deterministic key generation, optimistic UI, fast client dedupe, and clear disabled/loading/error states. GPS is best-effort; do not block the clock action if GPS is unavailable, but capture and send it when present.
- **Single source of truth**: one stable hook (`useClockAction`) owns in-flight state, abort/retry, and cache; one reusable button (`ClockButton`) owns UX states (acquiring/submitting/success/error) and disables while pending.

---

### 2. Files to add/modify
- `workio/src/lib/idempotency.ts` — deterministic key generator + short-lived client dedupe cache.
- `workio/src/hooks/useClockAction.ts` — mutation + idempotency + optimistic/rollback + abort.
- `workio/src/components/ClockButton.tsx` — UI with GPS orchestration and clear state feedback.
- Update `workio/src/pages/Dashboard.tsx` (or wherever clock UI lives) to use `ClockButton`.

```ts
// Lightweight, deterministic client-side idempotency helper.
// Complements server-side enforcement; not a replacement.

function cyrb53(str: string, seed = 0): string {
  let h1 = 0xdeadbeef ^ seed,
    h2 = 0x41c6ce57 ^ seed;
  for (let i = 0, ch; i < str.length; i++) {
    ch = str.charCodeAt(i);
    h1 = Math.imul(h1 ^ ch, 2654435761);
    h2 = Math.imul(h2 ^ ch, 1597334677);
  }
  h1 = Math.imul(h1 ^ (h1 >>> 16), 2246822507) ^ Math.imul(h2 ^ (h2 >>> 13), 3266489909);
  h2 = Math.imul(h2 ^ (h2 >>> 16), 2246822507) ^ Math.imul(h1 ^ (h1 >>> 13), 3266489909);
  return (h2 >>> 0).toString(16).padStart(8, '0') + (h1 >>> 0).toString(16).padStart(8, '0');
}

const CACHE_TTL = 60 * 60 * 1000; // 1h

type PendingEntry = {
  timestamp: number;
  resolve?: (value: unknown) => void;
  reject?: (reason?: any) => void;
};

class IdempotencyCache {
  private store = new Map<string, PendingEntry>();

  genKey(userId: string, tenantId: string, type: 'in' | 'out', date: string): string {
    // Deterministic per logical clock event for the day.
    return `clk_${cyrb53(`${userId}:${tenantId}:${type}:${date}`)}`;
  }

  private _expire(): void {
    const now = Date.now();
    for (const [k, v] of this.store.entries()) {
      if (now - v.timestamp > CACHE_TTL) this.store.delete(k);
    }
  }

  isPending(key: string): boolean {
    this._expire();
    return this.store.has(key);
  }

  setPending(key: string): void {
    this._expire();
    this.store.set(key, { timestamp: Date.now() });
  }

  setResult(key: string, result: unknown): void {
    const entry = this.store.get(key);
    if (entry) {
      if (entry.resolve) entry.resolve(result);
      this.store.delete(key);
    }
  }

  setError(key: string, err: any): void {
    const entry = this.store.get(key);
    if (entry) {
      if (entry.reject) entry.reject(err);
      this.store.delete(key);
    }
  }

  clear(): void {
    this.store.clear();
  }
}

export const idempotencyCache = new IdempotencyCache();

export function makeClockIdempotencyKey(
  userId: string,
  tenantId: string,
  type: 'in' | 'out',
  date: string
): string {
  return idempotencyCache.genKey(userId, tenantId, type, date);
}
```

#### workio/src/hooks/useClockAction.ts
```ts
import { useCallback, useRef } from 'react';
import { makeClockIdempotencyKey, idempotencyCache } from '../lib/idempotency';

type ClockActionOpts = {
  userId: string;
  tenantId: string;
  onSuccess?: (result: any) => void;
  onError?: (err: any) => void;
};

export function useClockAction({ userId, tenantId, onSuccess, onError }: ClockActionOpts) {
  const abortRef = useRef<AbortController | null>(null);

  const performClock = useCallback(
    async (type: 'in' | 'out', gps?: { latitude: number; longitude: number }) => {
      const today = new Date().toISOString().split('T')[0];
      const key = makeClockIdempotencyKey(userId, tenantId, type, today);

      // Fast client dedupe: do not start a second identical request while one is pending.
      if (idempotencyCache.isPending(key)) {
        return;
      }

      idempotencyCache.setPending(key);

      // Cancel previous in-flight request to reduce races for same user/tenant.
      if (abortRef.current) abortRef.current.abort();
      abortRef.current = new AbortController();

      try {
        const res = await fetch('/api/clock', {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json',
            // Server should enforce uniqueness by this key.
            'Idempotency-Key': key,
          },
          body: JSON.stringify({
            type,
            userId,
            tenantId,
            gps,
            timestamp: new Date().toISOString(),
          }),
          signal: abortRef.current.signal,
        });

        if (!res.ok) throw new Error(`Clock failed: ${res.status}`);
        const data = await res.json();
        idempotencyCache.setResult(key, data);
        onSuccess?.(data);
        return data;
      } catch (err: any) {
        if (err.name !== 'AbortError') {
          idempotencyCache.setError(key, err);
          onError?.(err);
        }
        throw err;
      }
    },
    [userId, tenantId, onSuccess, onError]
  );

  return { performClock };
}
```
