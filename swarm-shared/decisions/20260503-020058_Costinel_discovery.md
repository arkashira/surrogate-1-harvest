# Costinel / discovery

## Final Implementation Plan — Top-Hub Signal Panel (Costinel Dashboard)

**Scope & Value**  
- Frontend-only, read-only React panel that surfaces the highest-signal hub (default “MOC”) and its actionable cost-saving proposals from the knowledge graph.  
- CDN-first data delivery to eliminate HF API rate limits during dashboard loads.  
- Zero backend changes; deployable in <2h.

### 3) Utility: formatting helpers

`src/utils/format.ts`

```ts
export function formatUSD(value: number): string {
  return `$${value.toLocaleString('en-US', {
    minimumFractionDigits: 0,
    maximumFractionDigits: 0,
  })}`;
}

export function formatDate(iso: string): string {
  return new Date(iso).toLocaleString(undefined, {
    dateStyle: 'medium',
    timeStyle: 'short',
  });
}
```

### 4) Hook: CDN-first fetch with graceful fallback

`src/hooks/useTopHubSignals.ts`

```ts
import { useEffect, useState } from 'react';
import { TopHubSignal } from '../types/knowledgeGraph';

const CDN_URL = '/data/kg/top-hub-moc.json';
const FALLBACK_URL = 'https://huggingface.co/datasets/axentx/costinel-knowledge/resolve/main/top-hub-moc.json';

const FALLBACK: TopHubSignal = {
  hub: {
    id: 'MOC',
    label: 'MOC',
    description: 'Mission Operations Center — highest-signal hub by connection count and proposal density.',
    connectionCount: 0,
    lastUpdated: new Date().toISOString(),
  },
  proposals: [],
  generatedAt: new Date().toISOString(),
};

export function useTopHubSignals(enabled = true) {
  const [data, setData] = useState<TopHubSignal | null>(null);
  const [loading, setLoading] = useState(enabled);
  const [error, setError] = useState<Error | null>(null);

  useEffect(() => {
    if (!enabled) {
      setLoading(false);
      return;
    }

    let cancelled = false;

    async function loadFrom(url: string): Promise<TopHubSignal> {
      const res = await fetch(url, { cache: 'no-store' });
      if (!res.ok) throw new Error(`Fetch failed: ${res.status}`);
      return (await res.json()) as TopHubSignal;
    }

    async function load() {
      try {
        // Primary: local CDN (bypasses HF API limits)
        const primary = await loadFrom(CDN_URL);
        if (!cancelled) {
          setData(primary);
          setError(null);
        }
      } catch (err) {
        try {
          // Fallback: remote HF CDN
          const secondary = await loadFrom(FALLBACK_URL);
          if (!cancelled) {
            setData(secondary);
            setError(null);
          }
        } catch (err2) {
          if (!cancelled) {
            // Last resort: static fallback (dev-friendly)
            setData(FALLBACK);
            setError(err as Error);
          }
        }
      } finally {
        if (!cancelled) setLoading(false);
      }
    }

    load();

    return () => {
      cancelled = true;
    };
  }, [enabled]);

  return { data, loading, error };
}
```

### 5) Component: TopHubSignalPanel

`src/components/TopHubSignalPanel/TopHubSignalPanel.tsx`

```tsx
import React from 'react';
import { useTopHubSignals } from '../../hooks/useTopHubSignals';
import { formatUSD, formatDate } from '../../utils/format';
import styles from './TopHubSignalPanel.module.css';

export const TopHubSignalPanel: React.FC = () => {
  const { data, loading, error } = useTopHubSignals(true);

  if (loading) {
    return (
      <div className={styles.panel}>
        <div className={styles.loading}>Loading top-hub signals…</div>
      </div>
    );
  }

  if (!data) {
    return (
      <div className={styles.panel}>
        <div className={styles.error}>No signal data available.</div>
      </div>
    );
  }

  const { hub, proposals } = data;

  return (
    <div className={styles.panel}>
      <header className={styles.header}>
        <div className={styles.titleRow}>
          <h2 className={styles.title}>Top Hub: {hub.label}</h2>
          <span className={styles.badgeConnections}>{hub.connectionCount} connections</span>
        </div>
        <p className={styles.meta}>
          Updated {formatDate(hub.lastUpdated)}
        </p>
        <p className={styles.desc}>{hub.description}</p>
        {error && (
          <p className={styles.warning}>Using fallback data — some figures may be outdated.</p>
        )}
      </header>

      <section className={styles.proposals}>
        {proposals.length === 0 ? (
          <div className={styles.empty}>No proposals available for this hub.</div>
        ) : (
          proposals.map((p) => (
            <article key={p.id} className={styles.card}>
              <div className={styles.cardHeader}>
                <h3 className={styles.cardTitle}>{p.title}</h3>
                {p.href && (
                  <a
                    className={styles.link}
                    href={p.href}
                    target="_blank"
                    rel="noopener noreferrer"
                  >
                    Details →
                  </a>
                )}
              </div>
              <p className={styles.summary}>{p.summary}</p>

              {p.actions.length > 0 && (
                <ul className={styles.actions}>
                  {p.actions.map((a, i) => (
                    <li key={i} className={styles.actionItem}>
                      {a}
                    </li>
                  ))}
                </ul>
              )}

              <div className={styles.footer}>
                <span className={styles.badgePrimary}>
                  Save {formatUSD(p.impact.monthlySavingsUSD)}/mo
                </span>
                <span className={styles.badge}>{p.impact.effort} effort</span>
                <span className={styles.badge}>{p.impact.risk} risk</span>
                {p.tags.map((t) => (
                  <span key={t} className={styles.tag}>
                    {t}
                  </span>
                ))
