### 1. Snapshot script (Mac) — single API call

`scripts/make_snapshot.py`
```python
#!/usr/bin/env python3
"""
Create a deterministic file manifest for one date-partition.
Usage:
  HF_TOKEN=hf_xxx python scripts/make_snapshot.py \
    --repo axentx/surrogate-1-training-pairs \
    --partition 2026-05-03 \
    --out file_manifest.json
"""
import argparse
import json
import os
import sys
from datetime import datetime, timezone

from huggingface_hub import HfApi, login

def build_snapshot(repo_id: str, partition: str, out_path: str) -> None:
    api = HfApi()
    # One API call: list top-level objects in the partition folder (non-recursive)
    entries = api.list_repo_tree(
        repo_id=repo_id,
        path=partition,
        recursive=False,
        repo_type="dataset",
    )
    files = [e.rfilename for e in entries if e.type == "file"]
    files.sort()

    manifest = {
        "repo_id": repo_id,
        "partition": partition,
        "generated_at_utc": datetime.now(timezone.utc).isoformat(),
        "files": files,
        "count": len(files),
        "strategy": "cdn-only",
        "note": "Use only CDN URLs during training to avoid HF API rate limits",
    }

    os.makedirs(os.path.dirname(os.path.abspath(out_path)), exist_ok=True)
    with open(out_path, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2)
    print(f"Snapshot written to {out_path} ({len(files)} files)")

def main() -> None:
    parser = argparse.ArgumentParser(description="Create CDN-first file manifest")
    parser.add_argument("--repo", default="axentx/surrogate-1-training-pairs")
    parser.add_argument("--partition", required=True, help="Date folder, e.g. 2026-05-03")
    parser.add_argument("--out", default="file_manifest.json")
    args = parser.parse_args()

    token = os.environ.get("HF_TOKEN")
    if not token:
        print("ERROR: HF_TOKEN environment variable required", file=sys.stderr)
        sys.exit(1)
    login(token=token)
    build_snapshot(args.repo, args.partition, args.out)

if __name__ == "__main__":
    main()
```

Make executable:
```bash
chmod +x scripts/make_snapshot.py
```

Usage (Mac):
```bash
HF_TOKEN=hf_xxx python scripts/make_snapshot.py --partition 2026-05-03 --out file_manifest.json
```

### 2. Lightning training — CDN-only loader

`training/train_cdn.py`
```python
#!/usr/bin/env python3
"""
Lightning training that uses CDN URLs only (zero HF API calls during data load).
Expects file_manifest.json produced by scripts/make_snapshot.py.
"""
import json
import os
from pathlib import Path
from typing import Dict, List

import pyarrow as pa
import pyarrow.parquet as pq
import requests
import torch
from torch.utils.data import Dataset, DataLoader
import lightning as L

HF_CDN = "https://huggingface.co/datasets"

class CDNParquetDataset(Dataset):
    def __init__(self, manifest_path: str, repo_id: str, columns=("prompt", "response")):
        with open(manifest_path, "r", encoding="utf-8") as f:
            manifest = json.load(f)
        self.repo_id = repo_id or manifest["repo_id"]
        self.files: List[str] = manifest["files"]
        self.columns = columns
        # Validate expected columns exist by peeking first file
        self._peek_schema()

    def _peek_schema(self) -> None:
        if not self.files:
            return
        first = self.files[0]
        url = f"{HF_CDN}/{self.repo_id}/resolve/main/{first}"
        with requests.get(url, timeout=30) as r:
            r.raise_for_status()
            table = pq.read_table(pa.BufferReader(r.content))
        available = set(table.column_names)
        missing = [c for c in self.columns if c not in available]
        if missing:
            raise ValueError(f"Missing columns {missing} in {first}. Available: {available}")

    def __len__(self) -> int:
        return len(self.files)

    def __getitem__(self, idx: int) -> Dict[str, str]:
        relpath = self.files[idx]
        url = f"{HF_CDN}/{self.repo_id}/resolve/main/{relpath}"
        with requests.get(url, timeout=30) as r:
            r.raise_for_status()
            table = pq.read_table(pa.BufferReader(r.content))
        # Project to selected columns only
        prompt = table.column("prompt").to_pylist()
        response = table.column("response").to_pylist()
        # Return first row (each file = one sample) — adapt if multiple rows per file
        return {"prompt": prompt[0], "response": response[0], "_source_file": relpath}

class SurrogateDataModule(L.LightningDataModule):
    def __init__(self, manifest_path: str, repo_id: str, batch_size: int = 8, num_workers: int = 4):
        super().__init__()
        self.manifest_path = manifest_path
        self.repo_id = repo_id
        self.batch_size = batch_size
        self.num_workers = num_workers

    def setup(self, stage: str = "") -> None:
        self.dataset = CDNParquetDataset(self.manifest_path, self.repo_id)

    def train_dataloader(self) -> DataLoader:
        return DataLoader(
            self.dataset,
            batch_size=self.batch_size,
            num_workers=self.num_workers,
            shuffle=True,
            pin_memory=True,
        )

class SurrogateModel(L.LightningModule):
    def __init__(self, lr: float = 1e-4):
        super().__init__()
        self.lr = lr
        # Minimal stub — replace with your actual model
        self.tokenizer = None
        self.model = torch.nn.Linear(10, 10)  # placeholder

    def training_step(self, batch, batch_idx):
        # Replace with real forward + loss
        loss = torch.tensor(0.0, requires_grad=True)
        self.log("train_loss", loss)
        return loss

    def configure_optimizers(self):
        return torch.optim.AdamW(self.parameters(), lr=self.lr)

def main():
    manifest = os.environ.get("MANIFEST_PATH", "file_manifest.json")
    repo_id = os.environ.get("REPO_ID", "axentx/surrogate-1-training-pairs")
    if not Path(manifest).exists():
        print(f"Manifest not found: {manifest}", file=sys.stderr)
        print("Run scripts/make_snapshot.py first.", file=sys.stderr)
        return

    dm = SurrogateDataModule(manifest_path=manifest, repo_id=repo_id, batch_size=8, num_workers=4)
    model = SurrogateModel(lr=1e-4)

    trainer = L.Trainer(
        max_epochs=1,
        accelerator="gpu",
        devices=1,
        precision="bf16-mixed",
        log_every_n_steps=10,
    )
    trainer.fit(model, dm)

if __name__ == "__main__":
    main()
```
