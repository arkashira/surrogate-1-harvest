# workio / design

## Final synthesized answer (correct + actionable)

### 1. Diagnosis (merged, de-duplicated)

- **No idempotency enforcement on LINE webhook ingestion** — LINE retries on network/timeout; duplicate clock-in/out, leave/OT records are created.
- **Race-prone non-atomic attendance upsert** — find-then-write allows concurrent requests to both miss and insert duplicate punches for same tenant+employee.
- **Missing tenant-scoped uniqueness for open punches** — DB allows multiple active punches per employee per tenant.
- **No transactional write for punch + audit/event log** — partial failures leave attendance state inconsistent with event log.
- **Webhook handler does not validate signature end-to-end** — increases surface for replay/forged events.
- **Missing tenant isolation in webhook handler** — tenant must be resolved per event and enforced in all queries.

### 2. Proposed change (single scope)

File: `workio/server/src/controllers/line/webhook.ts` (and supporting DB/repository changes)

Scope:
- Add idempotency store keyed by `line_event_id` + `tenant_id`.
- Replace attendance upsert with atomic `INSERT ... ON CONFLICT` guarded by a partial unique index enforcing at most one open punch per `(tenant_id, employee_id)`.
- Wrap punch write + event log insert in a single DB transaction.
- Early signature verification and idempotency check before business logic.
- Resolve and enforce tenant_id for every event (isolation).

### 3. Implementation

#### 3.1 DB schema additions (`server/src/db/schema.sql`)

```sql
-- Idempotency store for LINE webhook events
CREATE TABLE IF NOT EXISTS line_webhook_events (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id       UUID NOT NULL REFERENCES tenants(id) ON DELETE CASCADE,
  line_event_id   TEXT NOT NULL,
  event_type      TEXT NOT NULL,
  payload         JSONB NOT NULL,
  processed_at    TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE UNIQUE INDEX idx_line_webhook_event_unique
  ON line_webhook_events (tenant_id, line_event_id);

-- Ensure at most one open punch per employee per tenant
-- (clock_out_at IS NULL means "currently clocked in")
CREATE UNIQUE INDEX idx_employee_open_punch
  ON attendance (tenant_id, employee_id)
  WHERE clock_out_at IS NULL;
```

#### 3.2 Idempotency + transaction helper (`server/src/db/repositories/lineEventRepository.ts`)

```ts
import { PoolClient } from 'pg';
import { pool } from '../db';

export interface LineWebhookEvent {
  tenantId: string;
  lineEventId: string;
  eventType: string;
  payload: Record<string, any>;
}

export async function createLineWebhookEvent(
  event: LineWebhookEvent,
  client?: PoolClient
): Promise<boolean> {
  const sql = `
    INSERT INTO line_webhook_events (tenant_id, line_event_id, event_type, payload)
    VALUES ($1, $2, $3, $4)
    ON CONFLICT (tenant_id, line_event_id) DO NOTHING
  `;
  const res = await (client || pool).query(sql, [
    event.tenantId,
    event.lineEventId,
    event.eventType,
    JSON.stringify(event.payload),
  ]);
  return res.rowCount === 1;
}
```

#### 3.3 Attendance repository with atomic punch (`server/src/db/repositories/attendanceRepository.ts`)

```ts
import { PoolClient } from 'pg';
import { pool } from '../db';

export interface PunchInput {
  tenantId: string;
  employeeId: string;
  clockInAt: Date;
  clockOutAt: Date | null;
  latitude?: number;
  longitude?: number;
  imageUrl?: string;
}

export async function punchIn(
  input: PunchInput,
  client?: PoolClient
): Promise<{ id: string }> {
  const sql = `
    INSERT INTO attendance (tenant_id, employee_id, clock_in_at, clock_out_at, latitude, longitude, image_url)
    VALUES ($1, $2, $3, $4, $5, $6, $7)
    RETURNING id
  `;
  const res = await (client || pool).query(sql, [
    input.tenantId,
    input.employeeId,
    input.clockInAt,
    input.clockOutAt,
    input.latitude ?? null,
    input.longitude ?? null,
    input.imageUrl ?? null,
  ]);
  return { id: res.rows[0].id };
}

export async function punchOut(
  input: {
    tenantId: string;
    employeeId: string;
    clockOutAt: Date;
    latitude?: number;
    longitude?: number;
    imageUrl?: string;
  },
  client?: PoolClient
): Promise<{ id: string } | null> {
  // Atomic update of the open punch
  const sql = `
    UPDATE attendance
    SET clock_out_at = $1, latitude = COALESCE($2, latitude), longitude = COALESCE($3, longitude), image_url = COALESCE($4, image_url)
    WHERE tenant_id = $5 AND employee_id = $6 AND clock_out_at IS NULL
    RETURNING id
  `;
  const res = await (client || pool).query(sql, [
    input.clockOutAt,
    input.latitude ?? null,
    input.longitude ?? null,
    input.imageUrl ?? null,
    input.tenantId,
    input.employeeId,
  ]);
  return res.rows[0] ? { id: res.rows[0].id } : null;
}
```

#### 3.4 Webhook controller (`server/src/controllers/line/webhook.ts`)

import { Router } from 'express';
import crypto from 'crypto';
import { pool } from '../../db/db';
import { createLineWebhookEvent } from '../../db/repositories/lineEventRepository';
import { punchIn, punchOut } from '../../db/repositories/attendanceRepository';

const router = Router();

function verifySignature(body: Buffer, signature: string, channelSecret: string): boolean {
  const expected = crypto
    .createHmac('sha256', channelSecret)
    .update(body)
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

// These should be implemented to map LINE user -> tenant/employee in your system
async function resolveTenantIdFromLineUserId(lineUserId: string, client: any): Promise<string | null> {
  const res = await client.query(
    'SELECT tenant_id FROM line_user_mappings WHERE line_user_id = $1 LIMIT 1',
    [lineUserId]
  );
  return res.rows[0]?.tenant_id || null;
}

async function resolveEmployeeIdFromLineUserId(
  lineUserId: string,
  tenantId: string,
  client: any
): Promise<string | null> {
  const res = await client.query(
    'SELECT id FROM employees WHERE tenant_id = $1 AND line_user_id = $2 LIMIT 1',
    [tenantId, lineUserId]
  );
  return res.rows[0]?.id || null;
}

  const signature = req.headers['x-line-signature'] as string;

  if (!verifySignature(rawBody, signature, channelSecret)) {
    return res.status(401).json({ error: 'Invalid signature' });
