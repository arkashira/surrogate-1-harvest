## 1. Problem & Solution (One-Sentence Each)

- **Problem:** Surrogate-1 training fails because `load_dataset(streaming=True)` hits HF API rate limits (429) on every epoch/worker.
- **Solution:** Generate a **CDN-only file manifest once**, then train with a custom `CdnParquetDataset` that fetches parquet files directly via public CDN URLs (`https://huggingface.co/datasets/{repo}/resolve/main/...`) — zero HF API calls during training.

## 3. Key Design Decisions (Correctness First)

1. **Manifest is CDN-only and pre-computed**  
   - Avoids HF API during training entirely.  
   - Uses `list_repo_tree(..., recursive=False)` once per date folder (reduces pagination/rate risk).

2. **No `load_dataset` or HF dataset wrapper during training**  
   - Prevents CastError on mixed schemas and 429s from workers.  
   - Uses `pyarrow.parquet` directly; projects to `{prompt, response}` at parse time.

3. **Lightweight caching in dataset loader**  
   - Cache one file at a time (`self._cache` keyed by CDN URL) to avoid repeated downloads within an epoch when multiple workers access same file (adjust per your DataLoader config).  
   - Optional: pin to single worker or implement per-rank caching if using multi-worker DataLoader.

4. **Studio reuse + free-tier GPU**  
   - Reuse running studio to avoid cold-start cost.  
   - Prefer `Machine.L40S` (free tier); fall back to `H200` only if `lightning-lambda-prod` account is active.

### `scripts/generate_cdn_manifest.py`
```python
#!/usr/bin/env python3
"""
Generate CDN-only file manifest for surrogate-1 training.
Run from Mac orchestrator (or any dev machine) after HF rate-limit window clears.
"""
import json
from datetime import datetime
from pathlib import Path
from huggingface_hub import HfApi

REPO = "axentx/surrogate-1-dataset"
DATE_FOLDER = datetime.utcnow().strftime("%Y-%m-%d")  # override via CLI if needed
OUTPUT_DIR = Path(__file__).parent.parent / "manifests" / "surrogate-1" / DATE_FOLDER

def main(date_folder: str | None = None):
    target = date_folder or DATE_FOLDER
    api = HfApi()

    # Single API call: list folder (non-recursive)
    files = api.list_repo_tree(REPO, path=target, recursive=False)

    manifest = []
    for f in files:
        if not f.path.endswith(".parquet"):
            continue
        cdn_url = f"https://huggingface.co/datasets/{REPO}/resolve/main/{f.path}"
        manifest.append({
            "path": f.path,
            "cdn_url": cdn_url,
            "size": f.size or 0,
            "sha256": getattr(f, "sha256", None)
        })

    OUTPUT_DIR.mkdir(parents=True, exist_ok=True)
    out_path = OUTPUT_DIR / "file_manifest.json"
    with open(out_path, "w") as fp:
        json.dump(manifest, fp, indent=2)

    # Also emit simple url list for bash workflows
    with open(OUTPUT_DIR / "urls.txt", "w") as fp:
        fp.write("\n".join(m["cdn_url"] for m in manifest))

    print(f"✅ Manifest written: {out_path} ({len(manifest)} files)")

if __name__ == "__main__":
    import sys
    main(sys.argv[1] if len(sys.argv) > 1 else None)
```

### `surrogate/data/cdn_dataset.py`
```python
import io
import json
import threading
import requests
import pyarrow.parquet as pq
from torch.utils.data import Dataset
from typing import Dict, List

class CdnParquetDataset(Dataset):
    """
    CDN-only dataset loader — zero HF API calls during training.
    Each item is one row from a parquet file with {prompt, response}.
    """
    def __init__(self, manifest_path: str, transform=None):
        with open(manifest_path) as f:
            self.files: List[Dict] = json.load(f)

        # Build row index: (file_idx, row_in_file)
        self._rows: List[tuple[int, int]] = []
        self._file_tables: Dict[int, Dict] = {}  # file_idx -> list of rows
        self._lock = threading.Lock()
        self.transform = transform

        for file_idx, entry in enumerate(self.files):
            try:
                table = self._load_table(entry["cdn_url"])
                df = table.select(["prompt", "response"]).to_pandas()
                records = df.to_dict("records")
            except Exception:
                records = []

            with self._lock:
                self._file_tables[file_idx] = records
                for row_idx in range(len(records)):
                    self._rows.append((file_idx, row_idx))

    def _load_table(self, cdn_url: str):
        resp = requests.get(cdn_url, timeout=30)
        resp.raise_for_status()
        return pq.read_table(io.BytesIO(resp.content))

    def __len__(self):
        return len(self._rows)

    def __getitem__(self, idx: int) -> Dict[str, str]:
        file_idx, row_idx = self._rows[idx]
        records = self._file_tables[file_idx]
        item = records[row_idx] if (records and 0 <= row_idx < len(records)) else {"prompt": "", "response": ""}

        if self.transform:
            item = self.transform(item)
        return item
```

### `surrogate/train.py` (excerpt)
```python
from lightning.pytorch import Trainer
from surrogate.data.cdn_dataset import CdnParquetDataset
from surrogate.models.surrogate_module import SurrogateModule

def main():
    manifest = "manifests/surrogate-1/2026-05-02/file_manifest.json"
    dataset = CdnParquetDataset(manifest)

    model = SurrogateModule()
    trainer = Trainer(
        max_epochs=1,
        accelerator="gpu",
        devices=1,
        strategy="auto"
    )
    trainer.fit(model, train_dataloaders=dataset)

if __name__ == "__main__":
    main()
```
