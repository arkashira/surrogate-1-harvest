# vanguard / design

## 1. Diagnosis

- Runtime `load_dataset`/recursive file enumeration triggers HF API 429s and non-reproducible epochs (no deterministic file list or content-addressable manifest).
- No CDN-only fetch path: training still hits `/api/` endpoints during data loading instead of using public CDN URLs (`resolve/main/...`).
- Schema heterogeneity in HF repos risks `pyarrow.CastError`; surrogate-1 expects strict `{prompt, response}` projection but ingestion may write mixed columns (`source`, `ts`, etc.).
- No reuse guard for Lightning Studio: scripts likely create new studios instead of reusing running ones, burning quota.
- Missing deterministic repo mapping for HF commit-cap mitigation: writes likely target a single repo instead of sharding across sibling repos by hash slug.

## 2. Proposed change

Add a small, self-contained **manifest + CDN loader module** and update the training launcher to use it:

- File: `/opt/axentx/vanguard/vanguard/data/manifest.py` (new)
- File: `/opt/axentx/vanguard/vanguard/train.py` (modify data-loading section)
- Optional: `/opt/axentx/vanguard/scripts/list_repo_for_date.sh` (helper for Mac orchestration)

Scope: ~120 lines total; focused on enabling reproducible, CDN-only, schema-safe data loading and deterministic repo selection.

## 3. Implementation

### 3.1 Create `vanguard/data/manifest.py`

```python
# vanguard/data/manifest.py
from __future__ import annotations

import json
import hashlib
from pathlib import Path
from typing import Dict, List, Optional, Tuple
import requests
from datasets import load_dataset

HF_DATASETS_CDN = "https://huggingface.co/datasets"

def deterministic_repo(slug: str, siblings: List[str]) -> str:
    """
    Pick one sibling repo deterministically from a list.
    siblings example: ["vanguard-mirror-00", ..., "vanguard-mirror-04"]
    """
    h = hashlib.sha256(slug.encode()).hexdigest()
    idx = int(h, 16) % len(siblings)
    return siblings[idx]

def build_manifest(
    repo: str,
    revision: str,
    folder: str,
    out_path: Path,
    *,
    recursive: bool = False,
    hf_token: Optional[str] = None,
) -> List[Dict]:
    """
    Single API call to list files in folder (non-recursive preferred).
    Produces manifest:
      [
        {"path": "batches/mirror-merged/2026-05-03/abc.parquet", "sha256": "...", "cdn_url": "..."},
        ...
      ]
    """
    from huggingface_hub import HfApi

    api = HfApi(token=hf_token)
    tree = api.list_repo_tree(repo=repo, revision=revision, path=folder, recursive=recursive)
    entries = []

    for item in tree:
        if item.rfilename.endswith(".parquet"):
            p = item.rfilename
            # Note: item.commit_hash may not be available; use path-based content hash placeholder
            # We'll fetch sha256 via CDN HEAD when needed or store ETag.
            cdn = f"{HF_DATASETS_CDN}/{repo}/resolve/main/{p}"
            entries.append({
                "path": p,
                "sha256": None,  # fill lazily or via HEAD
                "size": getattr(item, "size", None),
                "cdn_url": cdn,
                "revision": revision,
            })

    out_path.parent.mkdir(parents=True, exist_ok=True)
    with open(out_path, "w") as f:
        json.dump(entries, f, indent=2)
    return entries

def load_parquet_via_cdn(manifest: List[Dict], subset: Optional[List[str]] = None):
    """
    Load parquet files via CDN URLs (no HF API/auth during training).
    Projects to {prompt, response} and enforces schema.
    """
    import pandas as pd
    import pyarrow as pa
    import pyarrow.parquet as pq

    urls = [m["cdn_url"] for m in (manifest if subset is None else subset)]
    if not urls:
        raise ValueError("No files to load")

    # Stream via fsspec HTTPFileSystem (CDN) — no auth header by default
    fs = pa.fs.FileSystem.from_uri(urls[0])[0]
    # Use dataset to project schema early
    try:
        ds = load_dataset(
            "parquet",
            data_files={"train": urls},
            split="train",
            streaming=False,
        )
    except (ValueError, TypeError) as exc:
        # Fallback: manual projection per file to avoid mixed-schema errors
        rows = []
        for url in urls:
            pf = pq.read_table(url, filesystem=fs)
            cols = pf.column_names
            # Keep only prompt/response; tolerate missing one by filling null
            if "prompt" not in cols or "response" not in cols:
                # Best-effort: rename common aliases
                rename = {}
                for c in cols:
                    low = c.lower()
                    if "prompt" in low and "prompt" not in rename:
                        rename[c] = "prompt"
                    if "response" in low and "response" not in rename:
                        rename[c] = "response"
                pf = pf.rename_columns(rename)
            # Ensure columns exist
            if "prompt" not in pf.column_names:
                pf = pf.append_column("prompt", pa.array([None] * len(pf)))
            if "response" not in pf.column_names:
                pf = pf.append_column("response", pa.array([None] * len(pf)))
            # Select only these two
            pf = pf.select(["prompt", "response"])
            rows.append(pf.to_pandas())
        df = pd.concat(rows, ignore_index=True)
        return df

    # Normal path: HF datasets loader succeeded
    df = ds.to_pandas()
    # Enforce projection
    for col in list(df.columns):
        if col not in {"prompt", "response"}:
            df.drop(columns=[col], inplace=True)
    for col in ("prompt", "response"):
        if col not in df.columns:
            df[col] = None
    return df
```

```python
# vanguard/train.py  (excerpt — replace data-loading section)

from pathlib import Path
from vanguard.data.manifest import build_manifest, load_parquet_via_cdn

# --
# Config (tune per run)
HF_REPO = "org/vanguard-mirror-merged"     # or use sibling list + slug
REVISION = "main"
FOLDER   = "batches/mirror-merged/2026-05-03"
MANIFEST_PATH = Path("cache/manifest_2026-05-03.json")
HF_TOKEN = None  # or os.getenv("HF_TOKEN") for private repos
SIBLINGS = [f"vanguard-mirror-{i:02d}" for i in range(5)]  # if sharding

# --
# Step 1: Build manifest once (Mac orchestration) or reuse cached
if not MANIFEST_PATH.exists():
    # If sharding across repos, pick deterministic repo per slug (example):
    # repo = deterministic_repo(slug="2026-05-03", siblings=SIBLINGS)
    repo = HF_REPO
    entries = build_manifest(
        repo=repo,
        revision=REVISION,
        folder=FOLDER,
        out_path=MANIFEST_PATH,
        recursive=False,
        hf_token=HF_TOKEN,
    )
else:
    with open(MANIFEST_PATH) as f:
        entries = json.load(f)

# --
# Step 2: Load via CDN (zero API calls during training)
train_df = load_parquet_via_cdn(entries)

# --
# Continue training using train_df
# (convert to datasets.Dataset if needed)
from datasets import Dataset
train_ds = Dataset.from_pandas(train_df)
```
