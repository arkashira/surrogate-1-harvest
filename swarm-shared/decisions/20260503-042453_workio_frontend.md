# workio / frontend

## 1. Diagnosis
- Clock-in/out and leave/OT forms lack optimistic UI + rollback on failure → slow perceived performance and jarring full-page reflows on error.
- No local-first draft persistence for partially-completed requests (leave/OT/clock notes) → data loss on navigation or accidental close.
- Repeated tenant/role checks and feature-flag queries run inline in components → duplicated fetches and inconsistent permission UX.
- Missing skeleton states and progressive loading on dashboard/reports → layout jumps and poor perceived performance on slow networks.
- No centralized error boundary + toast layer for API failures → uncaught promise rejections and silent failures in forms.

## 2. Proposed change
Add a frontend **optimistic UI + rollback layer** and **local-first draft store** for clock/leave/OT actions, scoped to:
- `workio/src/features/attendance/` (clock-in/out)
- `workio/src/features/requests/` (leave/OT)
- `workio/src/lib/store/drafts.ts` (local-first draft persistence)
- `workio/src/lib/hooks/useOptimisticAction.ts` (reusable optimistic hook)

Single highest-value file to create/modify first: `workio/src/lib/hooks/useOptimisticAction.ts` + integrate into one critical flow (clock-in/out) to prove the pattern.

## 3. Implementation

### Create reusable optimistic hook
`workio/src/lib/hooks/useOptimisticAction.ts`
```ts
import { useCallback, useRef } from 'react';
import { toast } from 'react-hot-toast';

type ActionFn<TArgs, TResult> = (args: TArgs) => Promise<TResult>;
type OptimisticUpdate<TArgs, TState> = (args: TArgs, current: TState) => TState;
type Rollback<TArgs, TState> = (args: TArgs, current: TState) => TState;

export function useOptimisticAction<TArgs, TResult, TState>({
  action,
  optimisticUpdate,
  rollback,
  onSuccess,
  onError,
}: {
  action: ActionFn<TArgs, TResult>;
  optimisticUpdate: OptimisticUpdate<TArgs, TState>;
  rollback?: Rollback<TArgs, TState>;
  onSuccess?: (result: TResult, args: TArgs) => void;
  onError?: (error: unknown, args: TArgs) => void;
}) {
  // in-flight promise ref to avoid double-submit
  const inflightRef = useRef<Promise<TResult> | null>(null);

  const execute = useCallback(
    async (args: TArgs, currentState: TState, setState: (s: TState) => void) => {
      if (inflightRef.current) {
        toast.error('Action already in progress');
        return;
      }

      // optimistic update
      const nextState = optimisticUpdate(args, currentState);
      setState(nextState);

      const task = action(args);
      inflightRef.current = task;

      try {
        const result = await task;
        onSuccess?.(result, args);
        toast.success('Saved');
        return result;
      } catch (error) {
        // rollback
        const reverted = rollback ? rollback(args, nextState) : currentState;
        setState(reverted);
        onError?.(error, args);
        toast.error('Action failed — reverted');
        throw error;
      } finally {
        if (inflightRef.current === task) inflightRef.current = null;
      }
    },
    [action, optimisticUpdate, rollback, onSuccess, onError]
  );

  return { execute, pending: !!inflightRef.current };
}
```

### Local-first draft store (localStorage-backed)
`workio/src/lib/store/drafts.ts`
```ts
const PREFIX = 'workio:draft:';

export function saveDraft<T>(key: string, value: T) {
  try {
    localStorage.setItem(PREFIX + key, JSON.stringify(value));
  } catch {
    // ignore quota errors
  }
}

export function loadDraft<T>(key: string): T | null {
  try {
    const raw = localStorage.getItem(PREFIX + key);
    return raw ? JSON.parse(raw) : null;
  } catch {
    return null;
  }
}

export function clearDraft(key: string) {
  localStorage.removeItem(PREFIX + key);
}
```

### Integrate into ClockInOut component (example)
`workio/src/features/attendance/ClockInOutButton.tsx`
```tsx
import { useState } from 'react';
import { useOptimisticAction } from '@/lib/hooks/useOptimisticAction';
import { clockInOut } from '@/lib/api/attendance';

type AttendanceState = {
  lastEvent: 'IN' | 'OUT' | null;
  lastAt: string | null;
  todayHours: number;
};

export function ClockInOutButton({
  tenantId,
  current,
  onChange,
}: {
  tenantId: string;
  current: AttendanceState;
  onChange: (s: AttendanceState) => void;
}) {
  const { execute, pending } = useOptimisticAction({
    action: (payload: { type: 'IN' | 'OUT'; location?: GeolocationPosition }) =>
      clockInOut({ tenantId, ...payload }),
    optimisticUpdate: (payload, state) => ({
      ...state,
      lastEvent: payload.type,
      lastAt: new Date().toISOString(),
      todayHours: state.lastEvent === payload.type ? state.todayHours : state.todayHours + 0.5,
    }),
    rollback: (_, prev) => prev,
    onSuccess: (res) => {
      // optionally sync with server truth
      onChange({ ...current, ...res });
    },
  });

  const handle = async (type: 'IN' | 'OUT') => {
    // quick geolocation check (optional)
    let location: GeolocationPosition | undefined;
    try {
      location = await new Promise((res, rej) =>
        navigator.geolocation.getCurrentPosition(res, rej, { timeout: 5000 })
      );
    } catch {
      // proceed without location if not required
    }
    execute({ type, location }, current, onChange);
  };

  return (
    <div className="flex gap-3">
      <button
        onClick={() => handle('IN')}
        disabled={pending}
        className="btn btn-primary"
      >
        {pending ? 'Saving…' : 'Clock In'}
      </button>
      <button
        onClick={() => handle('OUT')}
        disabled={pending}
        className="btn btn-secondary"
      >
        {pending ? 'Saving…' : 'Clock Out'}
      </button>
    </div>
  );
}
```

### Add lightweight error boundary + toast root
Wrap app in `workio/src/main.tsx`:
```tsx
import { Toaster } from 'react-hot-toast';
import ErrorBoundary from '@/components/ErrorBoundary';

// inside root render:
<ErrorBoundary fallback={<div className="p-4 text-red-600">Something went wrong</div>}>
  <App />
  <Toaster position="top-right" />
</ErrorBoundary>
```

`workio/src/components/ErrorBoundary.tsx` (minimal):
```tsx
import { Component, ReactNode } from 'react';

type Props = { fallback: ReactNode; children: ReactNode };
type State = { hasError: boolean };

export default class ErrorBoundary extends Component<Props, State> {
  state = { hasError: false };
  static getDerivedStateFromError() {
    return { hasError: true };
  }
  componentDidCatch(error: Error) {
    // log to your error service
    console.error(error);
  }
  render() {
    return this.state.hasError ? this.props.fallback : this.props.children;
  }
}
```
