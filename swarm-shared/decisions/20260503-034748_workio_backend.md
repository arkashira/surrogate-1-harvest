# workio / backend

## 1. Diagnosis

- **No idempotency on LINE webhook handler**: retries on 5xx/timeouts replay clock-in/out or leave/OT events and corrupt state (duplicate punches, double approvals).
- **Missing `X-Line-Signature` verification**: webhook requests can be spoofed and processed without cryptographic proof, enabling tampering.
- **No real-time broadcast to clients**: backend state changes (clock/leave/OT) are invisible to other tabs/devices until hard refresh → multi-tab divergence and perceived latency.
- **No deduplication key in punch/leave/OT writes**: concurrent or retried webhooks create duplicate rows instead of being treated as same event.
- **No atomic upsert for time events**: clock-in/out can race and produce two active punches for same user/tenant.

## 2. Proposed change

**Scope**: `workio/server/src/routes/webhook/line.ts` (create if absent) and `workio/server/src/services/punchService.ts` + `workio/server/src/services/leaveService.ts`.

- Add `X-Line-Signature` verification middleware.
- Add idempotency via `X-Line-Webhook-Id` (or hash of `timestamp+event.type+userId`) and idempotency table/key.
- Add atomic upsert for punches (one active punch per user/tenant).
- Emit WebSocket/SSE event after state change so frontend can update in real time.

## 3. Implementation

### 3.2 Idempotency + signature verification middleware

```ts
// server/src/middleware/lineWebhook.ts
import crypto from 'crypto';
import { Request, Response, NextFunction } from 'express';
import { pool } from '../db';

const LINE_CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';

export function verifyLineSignature(req: Request, res: Response, next: NextFunction) {
  const signature = req.headers['x-line-signature'] as string;
  if (!signature) return res.status(401).json({ error: 'Missing X-Line-Signature' });

  const body = JSON.stringify(req.body);
  const expected = crypto
    .createHmac('sha256', LINE_CHANNEL_SECRET)
    .update(body)
    .digest('base64');

  if (!crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected))) {
    return res.status(401).json({ error: 'Invalid signature' });
  }
  next();
}

export async function deduplicateWebhook(req: Request, res: Response, next: NextFunction) {
  // Use delivery id if provided, else deterministic hash
  const deliveryId = (req.headers['x-line-webhook-id'] ||
    req.headers['x-line-delivery-id']) as string;
  const events = req.body.events || [];
  if (!events.length) return next();

  // Build a single idempotency key for the batch (or per-event if you prefer)
  const key = deliveryId || crypto.createHash('sha256')
    .update(JSON.stringify(events.map((e: any) => ({
      type: e.type,
      source: e.source?.userId,
      timestamp: e.timestamp,
      mode: e.mode
    }))))
    .digest('hex');

  const client = await pool.connect();
  try {
    await client.query('BEGIN');
    const { rows } = await client.query(
      'SELECT 1 FROM webhook_idempotency WHERE idempotency_key = $1',
      [key]
    );
    if (rows.length) {
      await client.query('ROLLBACK');
      return res.status(200).json({ ok: true, reason: 'duplicate' });
    }

    await client.query(
      'INSERT INTO webhook_idempotency(idempotency_key, event_type) VALUES ($1, $2)',
      [key, events.map((e: any) => e.type).join(',')]
    );
    await client.query('COMMIT');
    (req as any)._idempotencyKey = key;
    next();
  } catch (err) {
    await client.query('ROLLBACK');
    next(err);
  } finally {
    client.release();
  }
}
```

### 3.3 Punch service with atomic upsert

```ts
// server/src/services/punchService.ts
import { pool } from '../db';

export interface PunchInput {
  tenantId: string;
  userId: string;
  type: 'IN' | 'OUT';
  timestamp: Date;
  latitude?: number;
  longitude?: number;
  imageUrl?: string;
}

export async function recordPunch(input: PunchInput) {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');
    // Close any open punch for same user/tenant
    await client.query(
      `UPDATE punches
       SET clock_out = NOW(), updated_at = NOW()
       WHERE tenant_id = $1 AND user_id = $2 AND clock_out IS NULL`,
      [input.tenantId, input.userId]
    );

    if (input.type === 'IN') {
      await client.query(
        `INSERT INTO punches(tenant_id, user_id, clock_in, latitude, longitude, image_url)
         VALUES ($1, $2, $3, $4, $5, $6)`,
        [input.tenantId, input.userId, input.timestamp, input.latitude, input.longitude, input.imageUrl]
      );
    } else {
      // clock-out the latest open punch
      const { rows } = await client.query(
        `UPDATE punches
         SET clock_out = $3, updated_at = NOW()
         WHERE tenant_id = $1 AND user_id = $2 AND clock_out IS NULL
         RETURNING *`,
        [input.tenantId, input.userId, input.timestamp]
      );
      if (!rows.length) {
        // no open punch; create a pair with clock_in = clock_out (graceful handling)
        await client.query(
          `INSERT INTO punches(tenant_id, user_id, clock_in, clock_out, latitude, longitude, image_url)
           VALUES ($1, $2, $3, $3, $4, $5, $6)`,
          [input.tenantId, input.userId, input.timestamp, input.latitude, input.longitude, input.imageUrl]
        );
      }
    }
    await client.query('COMMIT');
    return { ok: true };
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  } finally {
    client.release();
  }
}
```

### 3.4 WebSocket/SSE broadcast helper (lightweight)

```ts
// server/src/lib/broadcast.ts
import type { Server } from 'socket.io';

let io: Server | null = null;

export function setIO(server: Server) {
  io = server;
}

export function broadcastTenantEvent(tenantId: string, event: string, payload: any) {
  if (!io) return;
  io.to(`tenant:${tenantId}`).emit(event, payload);
}
```

Wire into your main server file where Socket.IO is created:

```ts
// server/src/index.ts (excerpt)
import { createServer } from 'http';
import { Server } from 'socket.io';
import { setIO, broadcastTenantEvent } from './lib/broadcast';

const httpServer = createServer(app);
const io = new Server(httpServer, {
  cors: { origin: process.env.FRONTEND_URL || '*' }
});
setIO(io);

io.on('connection', (socket) => {
  const tenantId = socket.handshake.auth.tenantId as string;
  if (tenantId) socket.join(`tenant:${tenantId}`);
});
```

Emit after punch/leave changes:

```ts
// inside punchService.recordPunch after commit

```
