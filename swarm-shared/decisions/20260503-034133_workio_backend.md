# workio / backend

## Final consolidated implementation

**Scope**: `workio/server/src/routes/webhook/line.ts` (create or replace) + minimal realtime + concurrency guard.  
**Goal**: Correctness first, then concrete actionability; ship in <2h with safe fallbacks.

---

### 1. Diagnosis (merged + resolved)
- **Idempotency missing** → LINE retries on 5xx/timeouts replay webhooks and corrupt attendance state (duplicate punches, double approvals).  
  **Fix**: Per-event deduplication keyed by stable event id (or deterministic hash) with DB unique constraint as source-of-truth; short TTL in-memory/Redis for fast rejection.
- **Signature verification absent** → Spoofed payloads can mutate state.  
  **Fix**: Verify `X-Line-Signature` with HMAC-SHA256 using `LINE_CHANNEL_SECRET`. Require raw body (use `express.raw()` or equivalent middleware for this route).
- **No observable telemetry** → Silent failures, retries, and latency are invisible.  
  **Fix**: Structured JSON logs per request (requestId, eventType, status, latency, signature result) and error tracking.
- **Frontend state staleness** → DB commits from webhooks are not pushed to clients; multi-tab/devices diverge.  
  **Fix**: After successful DB commit, emit realtime event via WebSocket or SSE to tenant/employee room.
- **No concurrency guard** → Clock-in/out races across retries/tabs produce inconsistent daily punch state.  
  **Fix**: DB row-level lock (`SELECT ... FOR UPDATE`) or atomic upsert for daily punch record; enforce business rules in transaction.

import { Router, Request, Response, NextFunction } from 'express';
import crypto from 'crypto';
import { db } from '../db';
import { broadcastPunchUpdate } from '../ws';
import { v4 as uuidv4 } from 'uuid';

const router = Router();
const IDEMPOTENCY_TTL_MS = 10 * 60 * 1000; // 10m
const LINE_CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';

// In-memory fast dedupe (use Redis in multi-instance)
const seen = new Map<string, number>();
function isDuplicate(key: string): boolean {
  const now = Date.now();
  if (seen.has(key)) return true;
  seen.set(key, now);
  setTimeout(() => seen.delete(key), IDEMPOTENCY_TTL_MS);
  return false;
}

function verifySignature(rawBody: Buffer, signature: string): boolean {
  if (!LINE_CHANNEL_SECRET) return false;
  const expected = crypto
    .createHmac('sha256', LINE_CHANNEL_SECRET)
    .update(rawBody)
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

function logger(req: Request, res: Response, next: NextFunction) {
  const requestId = uuidv4();
  const start = Date.now();
  res.on('finish', () => {
    console.log(
      JSON.stringify({
        requestId,
        method: req.method,
        url: req.originalUrl,
        status: res.statusCode,
        latencyMs: Date.now() - start,
        ip: req.ip,
        lineSignature: req.get('X-Line-Signature'),
      })
    );
  });
  next();
}

router.use(logger);

router.post('/webhook/line', async (req: Request, res: Response) => {
  const signature = req.get('X-Line-Signature') || '';
  // req.body is Buffer when using express.raw(); fallback for dev convenience:
  const rawBody = Buffer.isBuffer(req.body) ? req.body : Buffer.from(JSON.stringify(req.body || {}));

  if (!verifySignature(rawBody, signature)) {
    console.warn('Invalid LINE signature');
    return res.status(401).send('Invalid signature');
  }

  // Parse JSON from raw body
  let payload;
  try {
    payload = JSON.parse(rawBody.toString('utf8'));
  } catch (err) {
    console.error('Invalid JSON in LINE webhook', err);
    return res.status(400).send('Invalid JSON');
  }

  const events = payload?.events || [];
  if (!Array.isArray(events) || events.length === 0) {
    return res.status(200).send('OK');
  }

  try {
    for (const ev of events) {
      // Deterministic idempotency key: prefer event.id; fallback to composite
      const eventId = ev?.source?.userId && ev?.timestamp
        ? `${ev.source.userId}:${ev.timestamp}:${ev.type}:${ev.message?.text || ''}`
        : null;

      if (eventId && isDuplicate(eventId)) {
        console.log('Duplicate event skipped', { eventId });
        continue;
      }

      await processLineEvent(ev);
    }
    res.status(200).send('OK');
  } catch (err) {
    console.error('Webhook processing failed', err);
    // Return 500 so LINE retries on transient failures
    res.status(500).send('Processing error');
  }
});

async function processLineEvent(event: any) {
  const type = event.type;
  const userId = event.source?.userId;
  const ts = event.timestamp;

  if (!userId) return;

  if (type === 'message' && event.message?.type === 'text') {
    const text = event.message.text.trim().toLowerCase();
    if (text === 'clock in' || text === 'clock out') {
      await atomicClockToggle(userId, text === 'in' ? 'in' : 'out', ts, event);
    }
  }

}

async function atomicClockToggle(userId: string, direction: 'in' | 'out', ts: number, event: any) {
  const today = new Date().toISOString().split('T')[0];
  const client = await db.connect();
  try {
    await client.query('BEGIN');
    // Lock today's row for this user
    const lockRes = await client.query(
      `SELECT id, clock_in, clock_out FROM punches WHERE user_id = $1 AND punch_date = $2 FOR UPDATE`,
      [userId, today]
    );

    if (direction === 'in') {
      if (lockRes.rows.length === 0) {
        // No row -> insert
        await client.query(
          `INSERT INTO punches (user_id, punch_date, clock_in, clock_in_ts, line_event_id)
           VALUES ($1, $2, NOW(), $3, $4)`,
          [userId, today, new Date(ts), event?.eventId || 'unknown']
        );
      } else {
        const row = lockRes.rows[0];
        if (row.clock_out) {
          // Already clocked out — allow new in for next session (business rule)
          await client.query(
            `INSERT INTO punches (user_id, punch_date, clock_in, clock_in_ts, line_event_id)
             VALUES ($1, $2, NOW(), $3, $4)`,
            [userId, today, new Date(ts), event?.eventId || 'unknown']
          );
        }
        // else: already clocked in and not out -> ignore duplicate in
      }
    } else {
      // clock out
      if (lockRes.rows.length === 0) {
