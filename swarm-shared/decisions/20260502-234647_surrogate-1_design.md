# surrogate-1 / design

## Implementation Plan (≤2h)

**Highest-value improvement**: Add a pre-flight snapshot generator (`bin/snapshot.sh`) that lists all dataset files once, embeds the file manifest into training, and enables CDN-only downloads during parallel shard processing. This eliminates HF API rate-limit pressure during ingestion and guarantees deterministic file lists across all 16 shards.

### Why this now
- Removes the 429/1000 req/5min risk when 16 runners concurrently list files
- Enables CDN bypass (`resolve/main/`) for actual file downloads (uncapped tier)
- Makes shard slicing deterministic (hash(slug) % 16) instead of relying on runtime `list_repo_files` pagination order
- Fits in <2h: single script + small changes to `dataset-enrich.sh` + workflow wiring

### Concrete changes

#### 1) `bin/snapshot.sh` — generate file manifest
```bash
#!/usr/bin/env bash
# bin/snapshot.sh
# Usage: HF_TOKEN=... ./bin/snapshot.sh axentx surrogate-1-training-pairs main
# Produces: snapshots/YYYY-MM-DD_HHMMSS_files.json

set -euo pipefail

REPO_OWNER="${1:-axentx}"
REPO_NAME="${2:-surrogate-1-training-pairs}"
BRANCH="${3:-main}"
OUTDIR="snapshots"
TIMESTAMP=$(date -u +"%Y-%m-%d_%H%M%S")
OUTFILE="${OUTDIR}/${TIMESTAMP}_files.json"

mkdir -p "${OUTDIR}"

echo "[$(date -u)] Listing ${REPO_OWNER}/${REPO_NAME}@${BRANCH} ..."

# Use huggingface_hub from Python to list top-level tree (non-recursive) once.
# We only need the top-level date folders (e.g. 2026-04-27/) to build the manifest.
python3 - <<PY | tee "${OUTFILE}"
import os, json, sys
from huggingface_hub import HfApi

api = HfApi(token=os.environ.get("HF_TOKEN"))
owner = "${REPO_OWNER}"
repo = "${REPO_NAME}"
branch = "${BRANCH}"

# List top-level folders only (fast, minimal pagination)
tree = api.list_repo_tree(
    repo_id=f"{owner}/{repo}",
    path="",
    recursive=False,
    revision=branch,
)

# Expect folders like 2026-04-27/; include only dirs
folders = [item.rstrip("/") for item in tree if item.endswith("/")]
folders.sort()

# Build manifest: for each folder, list files under it (one extra hop, still deterministic)
manifest = {}
for folder in folders:
    sub = api.list_repo_tree(
        repo_id=f"{owner}/{repo}",
        path=folder,
        recursive=False,
        revision=branch,
    )
    files = [f for f in sub if not f.endswith("/")]
    if files:
        manifest[folder] = sorted(files)

result = {
    "repo": f"{owner}/{repo}",
    "branch": branch,
    "snapshot_ts": "${TIMESTAMP}",
    "folders": folders,
    "manifest": manifest,
}

json.dump(result, sys.stdout, indent=2)
PY

echo "[$(date -u)] Snapshot written to ${OUTFILE}"
```

Make executable:
```bash
chmod +x bin/snapshot.sh
```

#### 2) Update `bin/dataset-enrich.sh` to consume snapshot and use CDN

Key changes:
- Accept snapshot file as optional arg (fallback to inline listing for backward compat)
- Compute deterministic shard assignment from manifest
- Download via CDN (`resolve/main/...`) instead of HF dataset streaming
- Keep existing schema normalization and dedup logic unchanged

```bash
#!/usr/bin/env bash
# bin/dataset-enrich.sh
# Usage: SHARD_ID=0 SHARD_TOTAL=16 ./bin/dataset-enrich.sh [snapshot.json]

set -euo pipefail

SHARD_ID="${SHARD_ID:-0}"
SHARD_TOTAL="${SHARD_TOTAL:-16}"
SNAPSHOT="${1:-}"

REPO_OWNER="axentx"
REPO_NAME="surrogate-1-training-pairs"
BRANCH="main"
BASE_CDN="https://huggingface.co/datasets/${REPO_OWNER}/${REPO_NAME}/resolve/main"

# If snapshot provided, use it; otherwise fall back to live listing (legacy)
if [[ -n "${SNAPSHOT}" && -f "${SNAPSHOT}" ]]; then
    echo "[$(date -u)] Using snapshot ${SNAPSHOT}"
    # Extract folder→file map for this shard
    mapfile -t ALL_FILES < <(
        python3 -c "
import json, sys, itertools
with open('${SNAPSHOT}') as f:
    data = json.load(f)
files = []
for folder, names in data['manifest'].items():
    for n in names:
        files.append(f'{folder}/{n}')
files.sort()
for f in files:
    print(f)
"
    )
else
    echo "[$(date -u)] No snapshot provided; listing live (slower, rate-limited)"
    mapfile -t ALL_FILES < <(
        python3 -c "
from huggingface_hub import HfApi
import os
api = HfApi(token=os.environ.get('HF_TOKEN'))
tree = api.list_repo_tree('${REPO_OWNER}/${REPO_NAME}', path='', recursive=True, revision='${BRANCH}')
for p in sorted(tree):
    if not p.endswith('/'):
        print(p)
"
    )
fi

TOTAL_FILES="${#ALL_FILES[@]}"
echo "[$(date -u)] Total files: ${TOTAL_FILES}; Shard ${SHARD_ID}/${SHARD_TOTAL}"

# Deterministic shard assignment by filename hash
process_file() {
    local relpath="$1"
    # Stable hash across runs
    local hash
    hash=$(echo -n "${relpath}" | sha256sum | cut -c1-8)
    local bucket=$(( 0x${hash} % SHARD_TOTAL ))
    if [[ "${bucket}" -eq "${SHARD_ID}" ]]; then
        echo "${relpath}"
    fi
}

mapfile -t MY_FILES < <(
    for f in "${ALL_FILES[@]}"; do
        process_file "${f}"
    done
)

echo "[$(date -u)] Shard ${SHARD_ID} processing ${#MY_FILES[@]} files"

# Process each assigned file via CDN download
for relpath in "${MY_FILES[@]}"; do
    url="${BASE_CDN}/${relpath}"
    outfile=$(mktemp)
    echo "[$(date -u)] Downloading ${relpath} ..."
    curl -fsSL --retry 3 --retry-delay 2 -o "${outfile}" "${url}"

    # Project to {prompt,response} here (existing schema logic)
    # Example placeholder: call python normalizer
    python3 - <<PY
import json, pyarrow as pa, pyarrow.parquet as pq, sys, os, tempfile
path = "${outfile}"
relpath = "${relpath}"
# Placeholder: existing normalize logic goes here.
# For now, emit minimal valid JSONL lines if parquet/json.
if path.endswith('.parquet'):
    try:
        tbl = pq.read_table(path, columns=['prompt','response'])
        for i in range(tbl.num_rows):
            row = tbl.slice(i,1).to_pydict()
            print(json.dumps({'prompt': row['prompt'][0], 'response': row['response'][0]}))
    except Exception:
        # fallback: skip malformed
        pass
else:
    # try json/jsonl
    try:
        with open(path) as f:
            for line in f:
                line = line.strip()
                if not line: continue
                obj = json.loads(line)
                if 'prompt' in obj and 'response' in obj:
                    print(json.dumps({'prompt': obj['prompt'], 'response': obj['response']}))
    except Exception:
        pass
PY

    rm -f "${outfile}"
done

echo "[$(date -u)] Shard ${SHARD_ID} done"
```

Make executable:
```bash
chmod +x bin/dataset-enrich.sh
```
