## 2. CDN Manifest Generator (Mac/Linux orchestration)

**File**: `surrogate/scripts/generate_cdn_manifest.py`

```python
#!/usr/bin/env python3
"""
Generate a deterministic CDN manifest for a date folder in a HF dataset repo.

Usage:
    HF_TOKEN=hf_xxx python generate_cdn_manifest.py \
        --repo "org/surrogate-data" \
        --date "2026-04-29" \
        --out "manifests/manifest_2026-04-29.json"
"""
import argparse
import json
import os
import sys
import time
from pathlib import Path
from typing import List, Dict

from huggingface_hub import HfApi, HFValidationError

HF_TOKEN = os.getenv("HF_TOKEN")
if not HF_TOKEN:
    print("ERROR: HF_TOKEN environment variable is required.", file=sys.stderr)
    sys.exit(1)

API = HfApi(token=HF_TOKEN)

def list_date_folder(repo: str, date: str) -> List[str]:
    """List immediate files under repo/date/ (non-recursive)."""
    prefix = f"{date}/"
    try:
        entries = API.list_repo_tree(repo_id=repo, path=prefix, recursive=False)
    except HFValidationError as e:
        raise ValueError(f"Invalid repo or path for {repo}/{prefix}: {e}") from e

    files = sorted(e.rfilename for e in entries if e.type == "file")
    return files

def build_manifest(repo: str, date: str, files: List[str]) -> Dict:
    base = f"https://huggingface.co/datasets/{repo}/resolve/main"
    items = [
        {
            "repo": repo,
            "path": f,
            "cdn_url": f"{base}/{f}",
            "date": date,
        }
        for f in files
    ]
    return {
        "generated_at_utc": time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime()),
        "repo": repo,
        "date": date,
        "count": len(items),
        "items": items,
    }

def main() -> None:
    parser = argparse.ArgumentParser(description="Generate CDN manifest for HF dataset date folder.")
    parser.add_argument("--repo", required=True, help="HF dataset repo (org/repo)")
    parser.add_argument("--date", required=True, help="Date folder (YYYY-MM-DD)")
    parser.add_argument("--out", required=True, help="Output JSON path")
    args = parser.parse_args()

    Path(args.out).parent.mkdir(parents=True, exist_ok=True)

    # Retry strategy: 1 retry after backoff on rate-limit (429) or transient errors.
    for attempt in range(2):
        try:
            files = list_date_folder(args.repo, args.date)
            break
        except Exception as exc:
            if attempt == 0 and ("429" in str(exc) or "rate" in str(exc).lower()):
                wait = 360
                print(f"Rate-limited (429) — waiting {wait}s before retry", file=sys.stderr)
                time.sleep(wait)
                continue
            print(f"Failed to list repo tree: {exc}", file=sys.stderr)
            sys.exit(1)

    manifest = build_manifest(args.repo, args.date, files)
    Path(args.out).write_text(json.dumps(manifest, indent=2))
    print(f"Wrote {len(files)} entries to {args.out}")

if __name__ == "__main__":
    main()
```

Make executable:
```bash
chmod +x surrogate/scripts/generate_cdn_manifest.py
```

**Why this is best**:
- Uses `HfApi` (stable) and validates inputs.
- Fails early if `HF_TOKEN` missing.
- Creates output directory if missing.
- Clear retry policy (single retry after 360s) aligned with HF guidance.
- Deterministic sorted file list.

def load_cdn_manifest(path: str) -> List[Dict]:
    if not Path(path).exists():
        raise FileNotFoundError(f"CDN manifest not found: {path}")
    with open(path) as f:
        manifest = json.load(f)
    items = manifest.get("items", [])
    if not items:
        raise ValueError("CDN manifest contains no items.")
    return items

def cdn_only_collate(batch):
    """Expects list of dicts with at least 'prompt' and 'response'."""
    # Keep minimal projection; do not add extra transforms here.
    return batch

class CDNDataModule(LightningDataModule):
    """
    Loads parquet/jsonl via CDN URLs listed in manifest.
    Streams samples into memory (lightweight) — suitable for moderate-size shards.
    For huge datasets, replace setup() with a streaming IterableDataset.
    """

    def __init__(self, manifest_items: List[Dict], batch_size: int = 8, num_workers: int = 4):
        super().__init__()
        self.items = manifest_items
        self.batch_size = batch_size
        self.num_workers = num_workers
        self.samples: List[Dict] = []

    def prepare_data(self):
        # No downloads — CDN URLs used directly.
        pass

    def setup(self, stage=None):
        samples = []
        for item in self.items:
            url = item["cdn_url"]
            try:
                if url.endswith(".parquet"):
                    import pyarrow.parquet as pq
                    table = pq.read_table(url, columns=["prompt", "response"])
                    df = table.to_pandas()
                elif url.endswith(".jsonl"):
                    import pandas as pd
                    df = pd.read_json(url, lines=True, usecols=["prompt", "response"])
                else:
                    print(f"Skipping unsupported file: {url}")
                    continue

                for _, row in df.iterrows():
                    samples.append({"prompt": str(row["prompt"]), "response": str(row["response"])})
            except Exception as exc:
                print(f"Skipping {url}: {exc}")
                continue
        self.samples = samples
        print(f"Loaded {len(self.samples)} samples from CDN manifest.")

    def train_dataloader(self):
        return tud.DataLoader(
            self.samples,
            batch_size=self.batch_size,
            num_workers=self.num_workers,
            collate_fn=cdn_only_collate,
            shuffle=True,
        )

def ensure_running_studio(name: str, machine: str = "L40S"):
    """
    Reuse or start a Lightning Studio. If stopped, restart it.
    Avoids recreating studios and prevents idle-timeout training loss.
    """
    from lightning import Teamspace, Studio, Machine

    teamspace = Teamspace()
    running = None
    for s in teamspace.studios:
        if s.name == name:
            running = s
            break

    if running is not None:
        if running.status == "Running":
            print(f"Reusing running studio: {name}")
            return running
        else:
            print(f"Studio {name} is {running.status}. Restarting...")
            running.start(machine=Machine(machine))
            for _ in range(15):
                running.refresh()
                if running.status == "Running":
                    print(f"Studio {name} is running.")
                    return running
                time.sleep(10)
            raise RuntimeError(f"Failed to restart studio {name}")
    else:
        print(f"Creating studio {name} on {machine}")
