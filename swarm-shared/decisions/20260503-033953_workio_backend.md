# workio / backend

## 1. Diagnosis

- **No idempotency guard on LINE webhook**: LINE retries on timeouts/network errors → duplicate clock-in/out records and double leave/OT approvals.
- **No signature verification** (`X-Line-Signature`): webhook body can be spoofed to fake punches or approvals.
- **No transactional boundary** for clock/leave/OT mutations: partial writes on error corrupt state (e.g., punch recorded but notification failed).
- **No tenant isolation in webhook path**: shared handler risks cross-tenant event injection when multiple tenants share a bot or during routing errors.
- **No replay protection / time-window enforcement**: stale or replayed webhook payloads accepted beyond valid clock skew.

## 2. Proposed change

File: `/opt/axentx/workio/server/src/routes/webhook/line.ts` (or create if missing)  
Scope: add middleware + handler wrapper that enforces signature verification, idempotency (per-event dedupe), tenant resolution, and transactional write for clock/leave/OT events.

## 3. Implementation

```ts
// server/src/routes/webhook/line.ts
import { Router, Request, Response, NextFunction } from 'express';
import crypto from 'crypto';
import { Pool } from 'pg';
import { v4 as uuidv4 } from 'uuid';

const router = Router();
const pool = new Pool({ connectionString: process.env.DATABASE_URL });

const LINE_CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';
const IDEMPOTENCY_TTL_SECONDS = 300; // 5m
const CLOCK_SKEW_SECONDS = 300; // 5m

// -- Helpers --

function verifySignature(rawBody: Buffer, signature: string): boolean {
  if (!LINE_CHANNEL_SECRET) return false;
  const expected = crypto
    .createHmac('sha256', LINE_CHANNEL_SECRET)
    .update(rawBody)
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

async function isDuplicateEvent(eventId: string): Promise<boolean> {
  const res = await pool.query(
    `SELECT 1 FROM webhook_events WHERE event_id = $1 AND created_at > NOW() - ($2 || ' seconds')::interval`,
    [eventId, IDEMPOTENCY_TTL_SECONDS]
  );
  return res.rowCount > 0;
}

async function recordEvent(eventId: string, tenantId: string, payload: unknown) {
  await pool.query(
    `INSERT INTO webhook_events (event_id, tenant_id, payload, created_at)
     VALUES ($1, $2, $3, NOW())`,
    [eventId, tenantId, payload]
  );
}

function resolveTenantId(req: Request): string | null {
  // Prefer explicit header; fallback to channel-based mapping if needed.
  const tenantId = req.headers['x-workio-tenant-id'] as string;
  if (tenantId) return tenantId;
  // TODO: map channelId -> tenantId via DB if multi-bot
  return null;
}

// -- Middleware --

function lineSignatureMiddleware(
  req: Request,
  res: Response,
  next: NextFunction
) {
  const signature = req.headers['x-line-signature'] as string;
  if (!signature) return res.status(401).json({ error: 'Missing signature' });

  const rawBody = (req as any).rawBody as Buffer;
  if (!rawBody || !verifySignature(rawBody, signature)) {
    return res.status(401).json({ error: 'Invalid signature' });
  }
  next();
}

// -- Transactional worker --

async function handleClockEvent(
  client: any,
  tenantId: string,
  userId: string,
  timestamp: Date,
  metadata: any
) {
  // Idempotent upsert: one punch per user per minute window (business rule)
  const windowStart = new Date(
    Math.floor(timestamp.getTime() / 60000) * 60000
  ).toISOString();

  await client.query(
    `INSERT INTO punches (tenant_id, user_id, punched_at, window_start, metadata)
     VALUES ($1, $2, $3, $4, $5)
     ON CONFLICT (tenant_id, user_id, window_start) DO UPDATE
       SET punched_at = EXCLUDED.punched_at, metadata = EXCLUDED.metadata`,
    [tenantId, userId, timestamp, windowStart, metadata]
  );
}

async function handleLeaveEvent(
  client: any,
  tenantId: string,
  userId: string,
  leaveId: string,
  action: 'approve' | 'reject',
  approverId: string
) {
  await client.query(
    `UPDATE leave_requests
     SET status = $1, approved_by = $2, updated_at = NOW()
     WHERE tenant_id = $3 AND id = $4 AND user_id = $5`,
    [action, approverId, tenantId, leaveId, userId]
  );
}

// -- Routes --

router.post(
  '/line',
  // Capture raw body for signature verification
  (req, res, next) => {
    const chunks: Buffer[] = [];
    req.on('data', (chunk) => chunks.push(chunk));
    req.on('end', () => {
      (req as any).rawBody = Buffer.concat(chunks);
      next();
    });
  },
  lineSignatureMiddleware,
  async (req: Request, res: Response) => {
    const client = await pool.connect();
    try {
      const body = req.body;
      const events = body.events || [];
      const tenantId = resolveTenantId(req);
      if (!tenantId) return res.status(400).json({ error: 'Tenant not resolved' });

      await client.query('BEGIN');

      for (const ev of events) {
        // Basic replay/time-window protection
        const ts = ev.timestamp ? new Date(ev.timestamp) : new Date();
        const now = new Date();
        const skew = Math.abs(now.getTime() - ts.getTime()) / 1000;
        if (skew > CLOCK_SKEW_SECONDS) continue;

        // Idempotency guard
        if (await isDuplicateEvent(ev.source?.userId + ':' + ev.timestamp + ':' + ev.type)) {
          continue;
        }

        // Minimal event projection: only handle known types
        if (ev.type === 'message' && ev.message?.type === 'text') {
          const text = ev.message.text.toLowerCase();
          const userId = ev.source.userId;

          // Example quick commands (real impl should use richer parser)
          if (text === 'clock in' || text === 'clock out') {
            await handleClockEvent(client, tenantId, userId, ts, {
              rawText: text,
              lineEventId: ev.source.eventId,
            });
          }
        }

        if (ev.type === 'postback') {
          const data = ev.postback?.data || '';
          // Example: approve-leave:{leaveId}:{userId}
          if (data.startsWith('approve-leave:')) {
            const [, leaveId, userId] = data.split(':');
            await handleLeaveEvent(client, tenantId, userId, leaveId, 'approve', ev.source.userId);
          }
        }

        // Record processed event
        await recordEvent(
          ev.source?.userId + ':' + ev.timestamp + ':' + ev.type,
          tenantId,
          ev
        );
      }

      await client.query('COMMIT');
      res.status(200).json({ ok: true });
    } catch (err) {
      await client.query('ROLLBACK');
      console.error('LINE webhook error:', err);
      res.status(500).json({ error: 'Processing failed' });
    } finally {
      client.release();
    }
  }
);

export default router;
```
