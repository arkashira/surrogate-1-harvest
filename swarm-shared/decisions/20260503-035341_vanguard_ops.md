# vanguard / ops

**Final, consolidated plan (correctness + concrete actionability)**

**1. Core diagnosis (resolved)**
- No content-addressed manifest → non-reproducible ingestion and 429s from repeated HF API calls.
- Mixed schemas written to `enriched/` without projection → `pyarrow.CastError` during surrogate-1 training.
- Training uses `load_dataset`/`list_repo_files` at runtime → burns HF API quota instead of using CDN-only fetches.
- Lightning Studio is recreated each run → wastes ~80 hr/mo quota.
- No idle-stop guard → long jobs killed by Studio shutdown.

**2. Single source of truth: manifest-first ingestion**
- One script produces a deterministic, content-addressed manifest per date folder.
- Manifest contains `{path, sha256, size, cdn_url}` and a local projected parquet filename.
- Projection keeps only `{prompt, response}`; all other columns dropped at parse time.
- Manifest enables zero HF API calls during training: training loads via CDN URLs or local cache.

**3. Implementation (merged and hardened)**

#!/usr/bin/env python3
import argparse
import json
import hashlib
import os
import pyarrow as pa
import pyarrow.parquet as pq
import requests
from pathlib import Path
from typing import List, Dict, Any, Optional

try:
    from huggingface_hub import list_repo_tree, hf_hub_download
except ImportError:
    raise SystemExit("pip install huggingface_hub pyarrow requests")

HF_REPO = "dataset-mirror"
CDN_ROOT = f"https://huggingface.co/datasets/{HF_REPO}/resolve/main"

def list_date_folder(date: str) -> List[Dict[str, Any]]:
    """Single non-recursive API call per date folder."""
    items = list_repo_tree(repo_id=HF_REPO, path=date, recursive=False)
    return [i for i in items if i.type == "file"]

def cdn_url(path: str) -> str:
    return f"{CDN_ROOT}/{path}"

def safe_project_to_prompt_response(file_path: str, tmp_dir: Path) -> Optional[Path]:
    """Download and project to {prompt, response} only."""
    try:
        local_path = hf_hub_download(repo_id=HF_REPO, filename=file_path, cache_dir=str(tmp_dir))
    except Exception:
        return None

    try:
        tbl = pq.read_table(local_path, columns=["prompt", "response"])
    except (pa.ArrowInvalid, KeyError, OSError, ValueError):
        return None

    slug = hashlib.sha256(file_path.encode()).hexdigest()[:16]
    out_path = tmp_dir / f"{slug}.parquet"
    pq.write_table(tbl, out_path)
    return out_path

def build_manifest(date: str, manifest_dir: str, batch_dir: str) -> None:
    manifest_path = Path(manifest_dir) / date / f"{HF_REPO}_files.json"
    batch_date_dir = Path(batch_dir) / date
    batch_date_dir.mkdir(parents=True, exist_ok=True)

    entries: List[Dict[str, Any]] = []
    files = list_date_folder(date)

    tmp_dir = Path("/tmp/vanguard_ingest") / date
    tmp_dir.mkdir(parents=True, exist_ok=True)

    for f in files:
        path = f.rfilename
        url = cdn_url(path)
        entry: Dict[str, Any] = {
            "path": path,
            "size": f.size,
            "sha256": getattr(f, "sha256", None),
            "cdn_url": url,
        }

        projected = safe_project_to_prompt_response(path, tmp_dir)
        if projected is not None:
            entry["batch_file"] = str(projected.name)
            dest = batch_date_dir / projected.name
            if not dest.exists():
                dest.write_bytes(projected.read_bytes())

        entries.append(entry)

    manifest_path.parent.mkdir(parents=True, exist_ok=True)
    manifest_path.write_text(json.dumps(entries, indent=2))
    print(f"Manifest written: {manifest_path}")

    # Emit train snippet (CDN/local cache first)
    snippet = f"""import json, os, io, pyarrow.parquet as pq, requests
MANIFEST = r"{manifest_path}"
BATCH_DIR = r"{batch_date_dir}"
with open(MANIFEST) as f:
    files = json.load(f)

def load_projected_batches():
    for e in files:
        if "batch_file" not in e:
            continue
        local_pq = os.path.join(BATCH_DIR, os.path.basename(e["batch_file"]))
        if os.path.exists(local_pq):
            yield pq.read_table(local_pq)
        else:
            resp = requests.get(e["cdn_url"], timeout=30)
            resp.raise_for_status()
            yield pq.read_table(io.BytesIO(resp.content))
"""
    snippet_path = manifest_path.parent / "train_snippet.py"
    snippet_path.write_text(snippet)
    print(f"Train snippet: {snippet_path}")

if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--date", required=True)
    parser.add_argument("--manifest-dir", required=True)
    parser.add_argument("--batch-dir", required=True)
    args = parser.parse_args()
    build_manifest(args.date, args.manifest_dir, args.batch_dir)
