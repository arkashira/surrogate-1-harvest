### 2. Idempotency core module

```ts
// workio/server/src/core/idempotency.ts
import { PoolClient } from 'pg';
import crypto from 'crypto';

export interface IdempotencyOpts {
  tenantId: number;
  key: string;
  payload: unknown;
}

/**
 * Enforce idempotency within a transaction.
 * Returns whether this is a duplicate delivery (same tenant+key and same payload hash).
 * Callers should skip business logic when isDuplicate === true.
 */
export async function enforceIdempotency(
  client: PoolClient,
  { tenantId, key, payload }: IdempotencyOpts
): Promise<{ isDuplicate: boolean }> {
  const payloadHash = crypto.createHash('sha256').update(JSON.stringify(payload)).digest('hex');
  const q = `
    INSERT INTO idempotency_keys (tenant_id, key, payload_hash)
    VALUES ($1, $2, $3)
    ON CONFLICT (tenant_id, key) DO UPDATE
    SET payload_hash = EXCLUDED.payload_hash
    WHERE idempotency_keys.payload_hash IS DISTINCT FROM EXCLUDED.payload_hash
    RETURNING (xmax = 0) AS inserted;
  `;
  const res = await client.query(q, [tenantId, key, payloadHash]);
  const row = res.rows[0];
  // inserted=true => new row (not duplicate). inserted=false => existing row unchanged (duplicate).
  return { isDuplicate: row ? !row.inserted : false };
}
```

> **Why this implementation**:  
> - Uses `xmax` trick to distinguish insert vs. no-op update reliably in Postgres.  
> - Stores hash so accidental payload changes for same key surface as non-duplicates (safe).  
> - Runs inside the same transaction as business writes → atomic.

### 3. LINE webhook handler (idempotent, tenant-scoped)

```ts
// workio/server/src/routes/webhook/line.ts
import { Router } from 'express';
import { Pool } from 'pg';
import { enforceIdempotency } from '../core/idempotency';

const router = Router();
const pool = new Pool({ connectionString: process.env.DATABASE_URL });

router.post('/line', async (req, res) => {
  const events = req.body?.events;
  if (!events || !Array.isArray(events)) return res.status(400).send('Bad Request');

  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    for (const ev of events) {
      // Stable idempotency key from LINE delivery context
      const idempotencyKey = `line:${ev.deliveryId || ev.timestamp}:${ev.type}:${ev.source?.userId || 'unknown'}`;

      // Resolve tenant from channel header or stored mapping
      const tenantId = await resolveTenantIdByChannel(client, req.get('X-Line-Channel-Id') || '');
      if (!tenantId) {
        await client.query('ROLLBACK');
        return res.status(404).send('Tenant not found');
      }

      const { isDuplicate } = await enforceIdempotency(client, {
        tenantId,
        key: idempotencyKey,
        payload: ev,
      });
      if (isDuplicate) continue;

      // Process clock-in/out text commands
      if (ev.type === 'message' && ev.message?.type === 'text') {
        const text = ev.message.text.trim().toLowerCase();
        const lineUid = ev.source?.userId;
        if (!lineUid) continue;

        const employee = await resolveEmployeeByLineUid(client, tenantId, lineUid);
        if (!employee) continue;

        if (text === 'clock in' || text === 'clock out') {
          await handleClock(client, tenantId, employee.id, text === 'clock in');
        }
      }
    }

    await client.query('COMMIT');
    res.status(200).json({ ok: true });
  } catch (err) {
    await client.query('ROLLBACK').catch(() => {});
    console.error('LINE webhook error:', err);
    // 500 -> LINE will retry; idempotency prevents duplicates
    res.status(500).send('Server Error');
  } finally {
    client.release();
  }
});

async function resolveTenantIdByChannel(client: any, channelId: string): Promise<number | null> {
  const r = await client.query('SELECT id FROM tenants WHERE line_channel_id = $1 LIMIT 1', [channelId]);
  return r.rows[0]?.id || null;
}

async function resolveEmployeeByLineUid(client: any, tenantId: number, lineUid: string) {
  const r = await client.query(
    'SELECT id FROM employees WHERE tenant_id = $1 AND line_user_id = $2 LIMIT 1',
    [tenantId, lineUid]
  );
  return r.rows[0];
}

async function handleClock(client: any, tenantId: number, employeeId: number, isClockIn: boolean) {
  if (isClockIn) {
    // DB partial unique index prevents concurrent open punches
    await client.query(
      `INSERT INTO punches (tenant_id, employee_id, clock_in_at, clock_in_source)
       VALUES ($1, $2, NOW(), 'line')`,
      [tenantId, employeeId]
    );
  } else {
    // Close latest open punch for this tenant+employee
    await client.query(
      `UPDATE punches
       SET clock_out_at = NOW(), clock_out_source = 'line'
       WHERE tenant_id = $1 AND employee_id = $2 AND clock_out_at IS NULL
       ORDER BY clock_in_at DESC LIMIT 1`,
      [tenantId, employeeId]
    );
  }
}

export default router;
```
