# workio / backend

## Final Synthesis — Correct, Actionable Implementation

**Core problems (agreed by both candidates):**
1. LINE webhook retries cause duplicate punches (no idempotency at message-id level).
2. Punch upsert is non-atomic and race-prone.
3. No database-level uniqueness to enforce one active punch per tenant+employee.
4. No idempotency key support for client/API retries (punch/leave/OT).
5. Tenant isolation relies on app checks rather than atomic tenant-scoped queries.

**Chosen approach:**  
- Add lightweight idempotency table + middleware (keyed by tenant+entity).  
- Convert punch upsert to atomic `INSERT ... ON CONFLICT` with a unique constraint that enforces one open punch per tenant+employee.  
- Add `line_message_id` uniqueness to deduplicate LINE webhook deliveries.  
- Perform punch operations in a transaction with tenant-scoped queries and row-level checks.  
- Keep middleware minimal and safe: reserve idempotency key early, attach entity_id after commit, and handle races via unique constraint + retry lookup.

### 1. Database schema (single source of truth)

```sql
-- server/src/db/schema.sql

-- Idempotency keys (lightweight)
CREATE TABLE IF NOT EXISTS idempotency_keys (
  id          SERIAL PRIMARY KEY,
  key         TEXT NOT NULL,
  tenant_id   INTEGER NOT NULL,
  entity      TEXT NOT NULL,           -- 'punch', 'leave', 'ot'
  entity_id   INTEGER,                 -- populated after operation
  created_at  TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(key, tenant_id, entity)
);

-- Ensure one active (open) punch per tenant+employee
-- Assumes punches has: id, tenant_id, employee_id, clock_in, clock_out, created_at, line_message_id
CREATE UNIQUE INDEX IF NOT EXISTS idx_one_active_punch
ON punches (tenant_id, employee_id)
WHERE clock_out IS NULL;

-- Prevent duplicate LINE deliveries for the same tenant+message
CREATE UNIQUE INDEX IF NOT EXISTS idx_unique_line_message
ON punches (tenant_id, line_message_id)
WHERE line_message_id IS NOT NULL;

-- Ensure created_at exists
ALTER TABLE punches ADD COLUMN IF NOT EXISTS created_at TIMESTAMPTZ DEFAULT NOW();
```

### 2. Idempotency middleware (safe, minimal)

```ts
// server/src/middleware/idempotency.ts
import { Request, Response, NextFunction } from 'express';
import { PoolClient } from 'pg';
import { getPool } from '../db';

export async function idempotency(
  req: Request & { tenantId?: number },
  res: Response,
  next: NextFunction
) {
  const key = req.header('Idempotency-Key') || req.body.idempotencyKey;
  const tenantId = req.tenantId;
  const path = req.path || '';
  const entity = path.includes('punch') ? 'punch'
                : path.includes('leave') ? 'leave'
                : path.includes('ot') ? 'ot'
                : 'unknown';

  if (!key || !tenantId) return next();

  const pool = getPool();
  let client: PoolClient | undefined;
  try {
    client = await pool.connect();

    // Serialize concurrent attempts for same key
    const sel = await client.query(
      `SELECT entity_id FROM idempotency_keys WHERE key = $1 AND tenant_id = $2 AND entity = $3`,
      [key, tenantId, entity]
    );

    if (sel.rows.length > 0) {
      (req as any).idempotentResult = { entityId: sel.rows[0].entity_id };
      client.release();
      return next();
    }

    // Reserve key (entity_id filled later)
    await client.query(
      `INSERT INTO idempotency_keys (key, tenant_id, entity) VALUES ($1, $2, $3)`,
      [key, tenantId, entity]
    );

    (req as any)._idempotency = { key, tenantId, entity, client };
    next();
  } catch (err: any) {
    if (client) client.release();

    if (err.code === '23505') {
      // Race: another request inserted same key
      return pool.query(
        `SELECT entity_id FROM idempotency_keys WHERE key = $1 AND tenant_id = $2 AND entity = $3`,
        [key, tenantId, entity]
      ).then((retry) => {
        if (retry.rows.length > 0) {
          (req as any).idempotentResult = { entityId: retry.rows[0].entity_id };
          return next();
        }
        return res.status(409).json({ error: 'Idempotency conflict' });
      }).catch(() => res.status(409).json({ error: 'Idempotency conflict' }));
    }
    next(err);
  }
}

// Call after successful mutation to attach entity_id
export async function attachIdempotencyResult(entityId: number, req: Request) {
  const state: { key?: string; tenantId?: number; entity?: string; client?: PoolClient } = (req as any)._idempotency || {};
  const { key, tenantId, entity, client } = state;
  if (!key || !tenantId || !entity) return;

  try {
    if (client) {
      await client.query(
        `UPDATE idempotency_keys SET entity_id = $1 WHERE key = $2 AND tenant_id = $3 AND entity = $4`,
        [entityId, key, tenantId, entity]
      );
    } else {
      const pool = getPool();
      await pool.query(
        `UPDATE idempotency_keys SET entity_id = $1 WHERE key = $2 AND tenant_id = $3 AND entity = $4`,
        [entityId, key, tenantId, entity]
      );
    }
  } catch {
    // best-effort update; uniqueness already enforced
  } finally {
    if (client) client.release();
  }
}
```

### 3. Atomic punch upsert (controller)

```ts
// server/src/controllers/punchController.ts
import { Request, Response } from 'express';
import { getPool } from '../db';
import { attachIdempotencyResult } from '../middleware/idempotency';

export async function clockInOut(req: Request & { tenantId?: number; userId?: number }, res: Response) {
  const tenantId = req.tenantId;
  const employeeId = req.userId;
  const now = new Date();
  const location = req.body.location;
  const lineMessageId = req.body.lineMessageId; // optional, for LINE webhook dedup

  if (!tenantId || !employeeId) return res.status(400).json({ error: 'Missing tenant or user' });

  // Idempotent replay: return existing punch if key already processed
  if ((req as any).idempotentResult) {
    const existing = await getPool().query(
      `SELECT * FROM punches WHERE id = $1 AND tenant_id = $2`,
      [(req as any).idempotentResult.entityId, tenantId]
    );
    if (existing.rows.length === 0) return res.status(404).json({ error: 'Punch not found' });
    return res.json(existing.rows[0]);
  }

  const client = await getPool().connect();
  try {
    await client.query('BEGIN');

    // Try to close an open punch for this tenant+employee
    const closeRes = await client.query(
      `UPDATE punches
       SET clock_out = $1, updated_at = NOW()
       WHERE tenant_id = $2 AND employee_id = $3 AND clock_out IS NULL
       RETURNING *`,
      [now, tenantId, employeeId]
    );

    if (closeRes.rows.length > 0) {
      await client.query('COMMIT');
      client.release();
      return res.json(closeRes.rows[0]);
    }

    // No open punch -> insert new clock-in
    // Use ON CONFLICT to handle races and line_message_id uniqueness
    const insertRes = await client.query(
      `INSERT INTO punches (tenant_id, employee_id, clock_in, location, line_message_id, created_at)
       VALUES ($1, $2, $3, $4, $5, $6)
       ON CONFLICT (tenant_id, employee_id)
