# airship / backend

## Final Implementation Plan — Manifest-Driven CDN-Only Dataset Loader

**Scope**: Replace `load_dataset`/`list_repo_files` in surrogate training with a manifest-driven, CDN-only iterable loader to eliminate HF API rate limits and mixed-schema ingestion failures.

---

### 1. Strategy (≤2h)
- **Manifest generation** (Mac orchestration): one-time `list_repo_tree` per date folder → `file_manifest.json` saved to repo root.
- **Training script**: reads manifest, streams files via CDN URLs (`resolve/main/...`) with **zero HF API calls** during training.
- **Schema projection**: parse each file → extract `{prompt, response}` only; drop extra columns; write to `batches/mirror-merged/{date}/{slug}.parquet`.
- **Lightning Studio reuse**: check running studios, restart only if stopped.

---

#### 3.1 Manifest Generator (run once on Mac)
```python
# surrogate/generate_manifest.py
import json
import os
import sys
from datetime import datetime
from huggingface_hub import HfApi

HF_REPO = os.getenv("HF_REPO", "datasets/your-org/your-mirror-repo")
DATE_FOLDER = os.getenv("DATE_FOLDER", "2026-04-29")
OUTPUT = "file_manifest.json"

def build_manifest():
    api = HfApi()
    entries = api.list_repo_tree(
        repo_id=HF_REPO,
        path=DATE_FOLDER,
        repo_type="dataset",
        recursive=False
    )

    files = []
    for e in entries:
        if e.type != "file":
            continue
        cdn_url = f"https://huggingface.co/datasets/{HF_REPO}/resolve/main/{DATE_FOLDER}/{e.path}"
        files.append({
            "path": e.path,
            "size": e.size,
            "cdn_url": cdn_url,
            "date_folder": DATE_FOLDER
        })

    manifest = {
        "generated_at": datetime.utcnow().isoformat() + "Z",
        "repo": HF_REPO,
        "date_folder": DATE_FOLDER,
        "count": len(files),
        "files": files
    }

    with open(OUTPUT, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Wrote {len(files)} files to {OUTPUT}")

if __name__ == "__main__":
    build_manifest()
```

Make executable:

#### 3.2 CDN-Only Iterable Dataset Loader
```python
# surrogate/train.py
import json
import os
import pyarrow as pa
import pyarrow.parquet as pq
import requests
from io import BytesIO
from typing import Iterator, Dict, Any

MANIFEST = "file_manifest.json"
BATCH_SIZE = 1024
OUTPUT_DIR = "batches/mirror-merged"

def cdn_record_stream(manifest_path: str) -> Iterator[Dict[str, Any]]:
    with open(manifest_path) as f:
        manifest = json.load(f)

    for item in manifest["files"]:
        url = item["cdn_url"]
        resp = requests.get(url, timeout=30)
        resp.raise_for_status()

        # Try parquet first, then JSON lines
        table = None
        try:
            table = pq.read_table(BytesIO(resp.content))
        except Exception:
            try:
                import pandas as pd
                df = pd.read_json(BytesIO(resp.content), lines=True)
                table = pa.Table.from_pandas(df)
            except Exception:
                continue  # skip malformed

        df = table.to_pandas()

        # Project to {prompt, response}
        if "prompt" not in df.columns or "response" not in df.columns:
            col_map = {}
            for c in df.columns:
                low = c.lower()
                if "prompt" in low or "instruction" in low or "input" in low:
                    col_map[c] = "prompt"
                if "response" in low or "output" in low or "completion" in low:
                    col_map[c] = "response"
            if "prompt" not in col_map or "response" not in col_map:
                continue
            df = df.rename(columns=col_map)

        df = df[["prompt", "response"]].dropna()
        for _, row in df.iterrows():
            yield {"prompt": str(row["prompt"]), "response": str(row["response"])}

def batched_parquet_writer(records: Iterator[Dict[str, Any]], output_dir: str):
    os.makedirs(output_dir, exist_ok=True)
    batch = []
    batch_idx = 0

    schema = pa.schema([
        pa.field("prompt", pa.string()),
        pa.field("response", pa.string())
    ])

    for rec in records:
        batch.append(rec)
        if len(batch) >= BATCH_SIZE:
            table = pa.Table.from_pylist(batch, schema=schema)
            out_path = os.path.join(output_dir, f"batch_{batch_idx:06d}.parquet")
            pq.write_table(table, out_path)
            print(f"Wrote {len(batch)} records to {out_path}")
            batch = []
            batch_idx += 1

    if batch:
        table = pa.Table.from_pylist(batch, schema=schema)
        out_path = os.path.join(output_dir, f"batch_{batch_idx:06d}.parquet")
        pq.write_table(table, out_path)
        print(f"Wrote {len(batch)} records to {out_path}")

if __name__ == "__main__":
    records = cdn_record_stream(MANIFEST)
    batched_parquet_writer(records, OUTPUT_DIR)
```

#### 3.3 Lightning Studio Launcher (Reuse-Aware)
```python
# surrogate/launch_studio.py
import os
from lightning_sdk import Studio, Teamspace

def get_or_create_studio(name="surrogate-train", machine=None):
    teamspace = Teamspace()
    running = [s for s in teamspace.studios if s.name == name and s.status == "Running"]
    if running:
        print(f"Reusing running studio: {running[0].id}")
        return running[0]

    stopped = [s for s in teamspace.studios if s.name == name and s.status == "Stopped"]
    if stopped:
        s = stopped[0]
        s.start(machine=machine)
        print(f"Restarted studio {s.id}")
        return s

    s = Studio(name=name, machine=machine, create_ok=True)
    print(f"Created studio {s.id}")
    return s

if __name__ == "__main__":
    studio = get_or_create_studio()
    studio.upload("train.py")
    studio.upload("file_manifest.json")
    studio.run("python train.py")
```
