### 1) `scripts/list_hf_files.py`
```python
#!/usr/bin/env python3
"""
Single API call to list files in a date folder.
Save to file_list.json for CDN-only ingestion.
Usage:
  python list_hf_files.py --repo <dataset_repo> --date 2026-05-01 --out file_list.json
"""
import argparse
import json
import os
import time
from huggingface_hub import HfApi

HF_TOKEN = os.getenv("HF_TOKEN", "")
api = HfApi(token=HF_TOKEN)

def list_date_folder(repo: str, date: str, out_path: str):
    folder_path = f"batches/mirror-merged/{date}"
    print(f"Listing {repo}/{folder_path} ...")
    try:
        tree = api.list_repo_tree(repo=repo, path=folder_path, recursive=False)
    except Exception as e:
        print(f"HF API error: {e}")
        time.sleep(360)
        tree = api.list_repo_tree(repo=repo, path=folder_path, recursive=False)

    files = [item.rfilename for item in tree if not item.rfilename.endswith("/")]
    payload = {
        "repo": repo,
        "date": date,
        "folder": folder_path,
        "files": files,
        "listed_at": time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime())
    }
    with open(out_path, "w") as f:
        json.dump(payload, f, indent=2)
    print(f"Wrote {len(files)} files to {out_path}")

if __name__ == "__main__":
    p = argparse.ArgumentParser()
    p.add_argument("--repo", required=True, help="HF dataset repo, e.g. org/surrogate-data")
    p.add_argument("--date", required=True, help="Date folder, e.g. 2026-05-01")
    p.add_argument("--out", default="file_list.json")
    args = p.parse_args()
    list_date_folder(args.repo, args.date, args.out)
```

### 2) `scripts/cdn_ingest.py`
```python
#!/usr/bin/env python3
"""
CDN-only ingestion: download via public resolve URLs (no auth),
project to {prompt,response}, write parquet shards.
Usage:
  python cdn_ingest.py --file-list file_list.json --out-dir ./data/shards
"""
import argparse
import json
import os
import time
import requests
import pyarrow as pa
import pyarrow.parquet as pq
from tqdm import tqdm
from typing import Dict

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def download_cdn(url: str, retries: int = 3, backoff: float = 2.0) -> bytes:
    for attempt in range(retries):
        try:
            resp = requests.get(url, timeout=30)
            if resp.status_code == 429:
                wait = backoff * (2 ** attempt)
                print(f"CDN 429, waiting {wait}s")
                time.sleep(wait)
                continue
            resp.raise_for_status()
            return resp.content
        except Exception as e:
            if attempt == retries - 1:
                raise
            time.sleep(backoff * (2 ** attempt))
    raise RuntimeError("Max retries exceeded")

def project_to_pair(raw_bytes: bytes, filename: str) -> Dict[str, str]:
    """
    Heuristic projection to {prompt,response}.
    Replace with repo-specific parser as needed.
    """
    import json as jsonlib
    try:
        data = jsonlib.loads(raw_bytes.decode("utf-8"))
        if isinstance(data, list):
            data = data[-1] if data else {}
        return {
            "prompt": str(data.get("prompt", "")),
            "response": str(data.get("response", ""))
        }
    except Exception:
        return {
            "prompt": f"raw_content_from_{filename}",
            "response": raw_bytes.decode("utf-8", errors="replace")[:2000]
        }

def run(file_list_path: str, out_dir: str, repo: str):
    os.makedirs(out_dir, exist_ok=True)
    with open(file_list_path) as f:
        meta = json.load(f)
    files = meta["files"]
    date = meta["date"]
    repo_name = repo or meta["repo"]

    rows = []
    skipped = []
    for fname in tqdm(files, desc="CDN ingest"):
        cdn_url = CDN_TEMPLATE.format(repo=repo_name, path=fname)
        try:
            raw = download_cdn(cdn_url)
            pair = project_to_pair(raw, fname)
            rows.append({
                "slug": os.path.splitext(os.path.basename(fname))[0],
                "prompt": pair["prompt"],
                "response": pair["response"],
                "source_file": fname,
                "date": date
            })
        except Exception as e:
            skipped.append({"file": fname, "error": str(e)})
            continue

    if rows:
        table = pa.Table.from_pylist(rows)
        out_path = os.path.join(out_dir, f"shard_{date}.parquet")
        pq.write_table(table, out_path)
        print(f"Wrote {len(rows)} rows to {out_path}")
    if skipped:
        with open(os.path.join(out_dir, f"skipped_{date}.json"), "w") as f:
            json.dump(skipped, f, indent=2)
        print(f"Skipped {len(skipped)} files")

if __name__ == "__main__":
    p = argparse.ArgumentParser()
    p.add_argument("--file-list", required=True)
    p.add_argument("--out-dir", default="./data/shards")
    p.add_argument("--repo", help="Override repo (otherwise from file_list)")
    args = p.parse_args()
    run(args.file_list, args.out_dir, args.repo)
```
