# airship / quality

## Final Synthesized Implementation (Best of Both Candidates)

**Value**: Eliminates HF API 429s during training, prevents Lightning idle-stop waste, reduces Mac→Lightning iteration from ~15min to <2min per cycle.  
**Effort**: <2h, zero breaking changes.

---

## Architecture Decisions (Resolved Contradictions)

1. **Manifest location**: Place `cdn_manifest.json` in `surrogate/train/` (not root) so Lightning workers find it via relative path without repo checkout churn.
2. **Fetch strategy**: Use direct CDN `requests` (not `hf_hub_download`) during training to guarantee zero API calls. Keep `huggingface-hub` only for manifest generation on Mac.
3. **Studio lifecycle**: Reuse running studios by name; restart if stopped (never recreate) to prevent 80hr/mo quota waste.
4. **Schema safety**: Project to `{prompt, response}` at load time with explicit column selection; drop nulls and cast to string to avoid Arrow schema/CastError issues.

## Code

### `surrogate/scripts/build_cdn_manifest.py` (run on Mac)

```python
#!/usr/bin/env python3
"""
Generate CDN-only manifest for one date folder.
Run on Mac after rate-limit window clears.
"""
import json
import os
import sys
from pathlib import Path
from huggingface_hub import HfApi

REPO = "axentx/surrogate-datasets"
DATE_FOLDER = os.getenv("SURROGATE_DATE_FOLDER", "batches/mirror-merged/2026-05-03")
OUT_PATH = Path(__file__).parent.parent / "train" / "cdn_manifest.json"

def main() -> None:
    api = HfApi()
    # Single non-recursive call to avoid pagination/429
    tree = api.list_repo_tree(REPO, path=DATE_FOLDER, recursive=False)
    files = sorted(
        f.rfilename for f in tree
        if f.rfilename.endswith(".parquet")
    )

    if not files:
        print("No parquet files found.", file=sys.stderr)
        sys.exit(1)

    manifest = {
        "repo": REPO,
        "date_folder": DATE_FOLDER,
        "files": files,
        "cdn_base": f"https://huggingface.co/datasets/{REPO}/resolve/main",
    }

    OUT_PATH.parent.mkdir(parents=True, exist_ok=True)
    OUT_PATH.write_text(json.dumps(manifest, indent=2))
    print(f"Manifest written: {OUT_PATH}")
    print(f"Files: {len(files)}")

if __name__ == "__main__":
    main()
```

### `surrogate/train/data/project.py` (CDN fetch + projection)

```python
import os
import json
from pathlib import Path
from typing import List, Dict

import pyarrow.parquet as pq
import requests

_MANIFEST_PATH = Path(__file__).parent.parent / "cdn_manifest.json"
_CACHE_DIR = Path(__file__).parent.parent / ".cache"

def _load_manifest() -> dict:
    if not _MANIFEST_PATH.exists():
        raise FileNotFoundError(
            f"Manifest not found: {_MANIFEST_PATH}. "
            "Run scripts/build_cdn_manifest.py on Mac first."
        )
    return json.loads(_MANIFEST_PATH.read_text())

def cdn_fetch(filepath: str, cdn_base: str) -> Path:
    """
    Download single file via CDN (no Authorization header).
    Returns local path.
    """
    _CACHE_DIR.mkdir(parents=True, exist_ok=True)
    local_path = _CACHE_DIR / os.path.basename(filepath)
    if local_path.exists():
        return local_path

    url = f"{cdn_base}/{filepath}"
    resp = requests.get(url, timeout=60)
    resp.raise_for_status()
    local_path.write_bytes(resp.content)
    return local_path

def project_to_chat_pairs() -> List[Dict[str, str]]:
    """
    Load all parquet files from manifest and project to [{prompt, response}].
    Avoids mixed-schema CastError from load_dataset(streaming=True).
    """
    manifest = _load_manifest()
    all_pairs: List[Dict[str, str]] = []

    for fname in manifest["files"]:
        local_path = cdn_fetch(
            filepath=f"{manifest['date_folder']}/{fname}",
            cdn_base=manifest["cdn_base"],
        )
        # Explicit columns only; drop nulls; ensure strings
        table = pq.read_table(
            str(local_path),
            columns=["prompt", "response"],
        )
        df = table.to_pandas()
        df = df.dropna(subset=["prompt", "response"])
        df["prompt"] = df["prompt"].astype(str)
        df["response"] = df["response"].astype(str)
        all_pairs.extend(df[["prompt", "response"]].to_dict(orient="records"))

    return all_pairs
```

### `surrogate/train/train_cdn.py` (Lightning entrypoint)

```python
#!/usr/bin/env python3
import lightning as L
from lightning.fabric.plugins import TorchTensorBoardLogger
from torch.utils.data import DataLoader, Dataset

from data.project import project_to_chat_pairs

class ChatDataset(Dataset):
    def __init__(self, pairs):
        self.pairs = pairs

    def __len__(self):
        return len(self.pairs)

    def __getitem__(self, idx):
        item = self.pairs[idx]
        # TODO: tokenize here; return input_ids, attention_mask, labels
        return item["prompt"], item["response"]

class SurrogateDataModule(L.LightningDataModule):
    def __init__(self):
        super().__init__()
        self.data = None

    def prepare_data(self):
        # Runs once per node; no HF API calls (CDN-only)
        self.data = project_to_chat_pairs()

    def train_dataloader(self):
        return DataLoader(ChatDataset(self.data), batch_size=8, shuffle=True)

def main():
    # Reuse running Studio; never recreate to avoid quota waste
    teamspace = L.Teamspace()
    studio_name = "surrogate-train-l40s"
    target = None

    for s in teamspace.studios:
        if s.name == studio_name:
            target = s
            break

    if target is None:
        print(f"Creating studio: {studio_name}")
        target = L.Studio(
            name=studio_name,
            machine=L.Machine.L40S,
            create_ok=True,
        )

    # Guard against idle-stop death
    if target.status != "running":
        print("Studio stopped — restarting...")
        target.start(machine=L.Machine.L40S)

    # Minimal run for fast iteration
    fabric = L.Fabric(
        devices=1,
        accelerator="cuda",
        logger=TorchTensorBoardLogger("logs/"),
    )
    fabric.launch()

    dm = SurrogateDataModule()
    dm.prepare_data()

    # Replace with real model init; do not run model code from Mac
    model = None

    trainer = L.Trainer(
        max_epochs=1,
        limit_train_batches=10,
        logger=fabric.logger,
    )
    trainer.fit(model, dm)

if __name__ == "__main__":
    main()
```
