## Final Synthesis — airship / ops (≤2h)

**Highest-value improvement (merged):**  
Embed a deterministic CDN file manifest into the Surrogate training launcher and add Lightning Studio lifecycle guards so training jobs bypass HF API rate limits and survive idle timeouts.

---

### Concrete, Correct, Actionable Plan

1. **Create CDN manifest generator**  
   - Runs on Mac/CI after rate-limit window clears.  
   - Uses **one** `list_repo_tree(..., recursive=False)` per date folder to avoid pagination explosion.  
   - Outputs `manifests/cdn-manifest-{date}.json` with:  
     `repo`, `date`, `generated_at`, `shards` (sorted, relative paths).  
   - Embed this file into the training container at `/app/manifest.json`.

2. **Add CDN-only DataLoader**  
   - Reads `/app/manifest.json`.  
   - Downloads shards via `https://huggingface.co/datasets/{repo}/resolve/main/{path}` (no auth).  
   - Projects to `{prompt, response}` only; drops extra columns.  
   - Streams via PyTorch `IterableDataset` with retry/backoff and **360s sleep on 429** (CDN tier rarely 429, but safe).

3. **Lightning resilient launcher**  
   - Before `.run()`, list `Teamspace.studios` and reuse a running studio with matching name.  
   - If studio exists but stopped, start with `machine=L40S` (fallback L40S → A100 priority order).  
   - Wrap training in a loop with **max 3 restarts**; on idle-stop detection (status != 'Running'), restart studio and resubmit from last checkpoint.  
   - Poll status every 60s; if stopped, restart + rerun.

4. **Integrate into training entrypoint**  
   - Accept `--manifest /app/manifest.json`.  
   - Use `cdn_loader.CDNIterableDataset` instead of `load_dataset`.  
   - Save checkpoints to `batches/mirror-merged/{date}/{slug}.parquet`-style outputs (no `source`/`ts` cols).  
   - On startup, validate manifest exists; fail fast with clear message if missing.

5. **Ops/cron hygiene**  
   - All wrapper scripts: `#!/usr/bin/env bash`, `chmod +x`, invoked via `bash script.sh "$@"`.  
   - Set `SHELL=/bin/bash` in crontab for scheduled manifest builds or training jobs.

#### `scripts/build_cdn_manifest.py`
```python
#!/usr/bin/env python3
"""
Build a CDN-only manifest for a HuggingFace dataset repo.
Run on Mac/CI (rate-limit window cleared). Non-recursive per folder.
"""
import json
import os
import sys
from datetime import datetime
from pathlib import Path

from huggingface_hub import HfApi

REPO = os.getenv("HF_DATASET_REPO", "org/surrogate-dataset")
DATE_FOLDER = os.getenv("DATE_FOLDER", datetime.utcnow().strftime("%Y-%m-%d"))
OUT_DIR = Path(os.getenv("OUT_DIR", "manifests"))
OUT_DIR.mkdir(parents=True, exist_ok=True)

def main() -> None:
    api = HfApi()
    try:
        tree = api.list_repo_tree(repo_id=REPO, path=DATE_FOLDER, recursive=False)
    except Exception as e:
        print(f"Failed to list repo tree: {e}", file=sys.stderr)
        sys.exit(1)

    shards = [
        f"{DATE_FOLDER}/{f.rfilename}"
        for f in tree
        if f.rfilename.endswith(".parquet")
    ]

    manifest = {
        "repo": REPO,
        "date": DATE_FOLDER,
        "generated_at": datetime.utcnow().isoformat() + "Z",
        "shards": sorted(shards),
    }

    out_path = OUT_DIR / f"cdn-manifest-{DATE_FOLDER}.json"
    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"Manifest written to {out_path} ({len(shards)} shards)")

if __name__ == "__main__":
    main()
```

#### `surrogate/training/data/cdn_loader.py`
```python
import json
import time
from pathlib import Path
from typing import Dict, Iterator

import requests
import pyarrow.parquet as pq
from torch.utils.data import IterableDataset

CDN_BASE = "https://huggingface.co/datasets"

class CDNIterableDataset(IterableDataset):
    def __init__(self, manifest_path: str, columns=("prompt", "response"), max_retries=5):
        self.manifest_path = Path(manifest_path)
        self.columns = columns
        self.max_retries = max_retries
        with open(self.manifest_path) as f:
            self.manifest = json.load(f)
        self.repo = self.manifest["repo"]
        self.shards = self.manifest["shards"]

    def _download_shard(self, shard_path: str) -> bytes:
        url = f"{CDN_BASE}/{self.repo}/resolve/main/{shard_path}"
        for attempt in range(1, self.max_retries + 1):
            try:
                resp = requests.get(url, timeout=30)
                if resp.status_code == 429:
                    wait = 360
                    print(f"CDN 429, sleeping {wait}s (attempt {attempt})")
                    time.sleep(wait)
                    continue
                resp.raise_for_status()
                return resp.content
            except Exception as e:
                if attempt == self.max_retries:
                    raise
                print(f"Download failed {shard_path}: {e}, retry {attempt}")
                time.sleep(2**attempt)
        raise RuntimeError(f"Failed to download {shard_path}")

    def _iter_shard(self, shard_path: str) -> Iterator[Dict[str, str]]:
        data = self._download_shard(shard_path)
        table = pq.read_table(pq.ParquetFile(pq.ParquetReader(pq.BufferReader(data))))
        # Ensure required columns exist
        for col in self.columns:
            if col not in table.column_names:
                table = table.append_column(col, [None] * len(table))
        for i in range(len(table)):
            row = {col: str(table[col][i]) for col in self.columns}
            yield row

    def __iter__(self) -> Iterator[Dict[str, str]]:
        for shard in self.shards:
            yield from self._iter_shard(shard)
```
