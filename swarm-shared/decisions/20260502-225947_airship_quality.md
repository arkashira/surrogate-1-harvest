### 1) `scripts/airship-discover`

```bash
#!/usr/bin/env bash
# airship-discover
# Orchestrator: emits status.json + manifest.json for local/CI.
# Zero runtime deps beyond bash + curl + jq. CDN-safe (no HF API during training).
#
# Usage:
#   bash scripts/airship-discover <dataset_repo> <date_folder> [out_dir]
#
# Example:
#   bash scripts/airship-discover axentx/surrogate-data 2026-05-02 ./out
#
# Notes:
# - Uses HF CDN (resolve/main) to bypass API rate limits during training.
# - Pre-lists files once (single API call) and embeds list into manifest.json.
# - Deterministic ordering and canonical timestamps for reproducible builds.

set -euo pipefail

# ---- config ----
DATASET_REPO="${1:-}"
DATE_FOLDER="${2:-}"
OUT_DIR="${3:-./out}"
MANIFEST_NAME="manifest.json"
STATUS_NAME="status.json"
TIMESTAMP="$(date -u +"%Y-%m-%dT%H:%M:%SZ")"
CDN_BASE="https://huggingface.co/datasets/${DATASET_REPO}/resolve/main"
HF_API_BASE="https://huggingface.co/api/datasets/${DATASET_REPO}/tree"

# ---- validation ----
if [ -z "${DATASET_REPO}" ] || [ -z "${DATE_FOLDER}" ]; then
  echo "ERROR: Usage: $0 <dataset_repo> <date_folder> [out_dir]" >&2
  exit 1
fi

mkdir -p "${OUT_DIR}"

# ---- helpers ----
retry_cdn() {
  local url="$1"
  local out="$2"
  local max_attempts=5
  local attempt=1
  local wait_sec=2

  while [ $attempt -le $max_attempts ]; do
    if curl -fsSL --retry 3 --retry-delay 1 -o "${out}" "${url}"; then
      return 0
    fi
    echo "WARN: CDN fetch failed (attempt ${attempt}/${max_attempts}): ${url}" >&2
    sleep $wait_sec
    wait_sec=$((wait_sec * 2))
    attempt=$((attempt + 1))
  done
  echo "ERROR: CDN fetch exhausted: ${url}" >&2
  return 1
}

# ---- step 1: list folder (single API call) ----
# Use non-recursive tree call to avoid pagination explosion. Caller should
# scope to a single date folder to keep list small and deterministic.
echo "INFO: Listing ${DATASET_REPO}/${DATE_FOLDER} (non-recursive)..."
TREE_JSON="$(mktemp)"
if ! curl -fsSL "${HF_API_BASE}/${DATE_FOLDER}?recursive=false" -o "${TREE_JSON}"; then
  echo "ERROR: Failed to list repo tree (API). Check repo/folder and HF API limits." >&2
  rm -f "${TREE_JSON}"
  exit 2
fi

# Extract paths and sort for deterministic manifest
mapfile -t FILES < <(jq -r '.[] | select(.type=="file") | .path' "${TREE_JSON}" | sort)
rm -f "${TREE_JSON}"

if [ "${#FILES[@]}" -eq 0 ]; then
  echo "ERROR: No files found in ${DATASET_REPO}/${DATE_FOLDER}" >&2
  exit 3
fi

# ---- step 2: build manifest (CDN paths only) ----
MANIFEST_PATH="${OUT_DIR}/${MANIFEST_NAME}"
echo "{" > "${MANIFEST_PATH}"
echo "  \"dataset_repo\": \"${DATASET_REPO}\"," >> "${MANIFEST_PATH}"
echo "  \"date_folder\": \"${DATE_FOLDER}\"," >> "${MANIFEST_PATH}"
echo "  \"generated_at\": \"${TIMESTAMP}\"," >> "${MANIFEST_PATH}"
echo "  \"cdn_base\": \"${CDN_BASE}\"," >> "${MANIFEST_PATH}"
echo "  \"files\": [" >> "${MANIFEST_PATH}"

for idx in "${!FILES[@]}"; do
  f="${FILES[$idx]}"
  comma=""
  if [ $((idx + 1)) -lt "${#FILES[@]}" ]; then
    comma=","
  fi
  # CDN path for this file (no Authorization header required)
  echo "    {\"path\": \"${f}\", \"cdn_url\": \"${CDN_BASE}/${f}\"}${comma}" >> "${MANIFEST_PATH}"
done

echo "  ]" >> "${MANIFEST_PATH}"
echo "}" >> "${MANIFEST_PATH}"

# ---- step 3: build status.json ----
STATUS_PATH="${OUT_DIR}/${STATUS_NAME}"
cat > "${STATUS_PATH}" <<EOF
{
  "status": "ok",
  "timestamp": "${TIMESTAMP}",
  "dataset_repo": "${DATASET_REPO}",
  "date_folder": "${DATE_FOLDER}",
  "file_count": ${#FILES[@]},
  "outputs": {
    "manifest": "${MANIFEST_NAME}",
    "status": "${STATUS_NAME}"
  },
  "notes": "CDN-safe manifest generated. Training should use embedded file list and CDN URLs only (zero HF API calls during data load)."
}
EOF

# ---- step 4: validate outputs ----
if ! jq -e . "${MANIFEST_PATH}" >/dev/null 2>&1; then
  echo "ERROR: Generated invalid JSON: ${MANIFEST_PATH}" >&2
  exit 4
fi

if ! jq -e . "${STATUS_PATH}" >/dev/null 2>&1; then
  echo "ERROR: Generated invalid JSON: ${STATUS_PATH}" >&2
  exit 4
fi

echo "INFO: Done. Wrote ${MANIFEST_PATH} and ${STATUS_PATH}"
echo "INFO: File count: ${#FILES[@]}"
exit 0
```

Make it executable:

```bash
chmod +x /opt/axentx/airship/scripts/airship-discover
```
