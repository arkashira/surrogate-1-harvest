## `/opt/axentx/airship/bin/airship`
```bash
#!/usr/bin/env bash
# airship — Arkship + Surrogate AI Platform CLI
# Usage: airship <command> [args...]
# Commands: discover

set -euo pipefail

# Cron-safe environment (Candidate 2)
export SHELL=/bin/bash
export PATH="/usr/local/sbin:/usr/local/bin:/usr/sbin:/usr/bin:/sbin:/bin"

AXENTX_ROOT="$(cd "$(dirname "${BASH_SOURCE[0]}")/.." && pwd)"
export AXENTX_ROOT

log() {
  printf '[%s] [airship] %s\n' "$(date -u +"%Y-%m-%dT%H:%M:%SZ")" "$*" >&2
}

fail() {
  log "ERROR: $*"
  exit 1
}

command_discover() {
  # shellcheck source=../lib/airship/discover.sh
  . "${AXENTX_ROOT}/lib/airship/discover.sh"
  airship::discover::run "$@"
}

main() {
  local cmd="${1:-}"
  case "${cmd}" in
    discover) shift; command_discover "$@" ;;
    *) fail "unknown command: ${cmd}. Usage: airship <discover>" ;;
  esac
}

main "$@"
```

## `/opt/axentx/airship/lib/airship/discover.sh`
```bash
#!/usr/bin/env bash
# airship discover — business research + knowledge-rag + top-hub insight
# Exports: airship::discover::run
# Dependencies: jq, curl, (optional) knowledge-rag CLI, (optional) Neo4j

set -euo pipefail

AIRSHIP_LIB_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
AXENTX_ROOT="$(cd "${AIRSHIP_LIB_DIR}/../.." && pwd)"

# Logging: system log (production) + project log (dev)
AIRSHIP_LOG_DIR="${AIRSHIP_LOG_DIR:-/var/log/airship}"
AIRSHIP_LOG="${AIRSHIP_LOG_DIR}/discover.log"
PROJECT_LOG="${AXENTX_ROOT}/logs/discover.log"

mkdir -p "${AIRSHIP_LOG_DIR}" "$(dirname "${PROJECT_LOG}")" "${AXENTX_ROOT}/state"

airship::discover::log() {
  local msg
  msg="$(printf '[%s] [discover] %s' "$(date -u +"%Y-%m-%dT%H:%M:%SZ")" "$*")"
  echo "$msg" >> "${AIRSHIP_LOG}"
  echo "$msg" >> "${PROJECT_LOG}"
}

airship::discover::run_granite() {
  local script="${AXENTX_ROOT}/scripts/granite-business-research.sh"
  if [[ -x "${script}" ]]; then
    airship::discover::log "Running granite-business-research.sh"
    if ! bash "${script}" >> "${AIRSHIP_LOG}" 2>&1; then
      airship::discover::log "granite-business-research.sh exited non-zero (logged)"
    fi
  else
    airship::discover::log "granite-business-research.sh not found or not executable — skipping"
  fi
}

airship::discover::run_knowledge_rag() {
  if command -v knowledge-rag >/dev/null 2>&1; then
    airship::discover::log "Querying top hub via knowledge-rag CLI"
    knowledge-rag --query "top hub and related docs" --limit 5 --format json >> "${AIRSHIP_LOG}" 2>&1 || \
      airship::discover::log "knowledge-rag CLI returned non-zero (logged)"
    return
  fi

  airship::discover::log "knowledge-rag CLI not found — attempting local service query"
  local surrogate_ai_url="http://localhost:8001"
  if curl -fsSL --max-time 10 "${surrogate_ai_url}/health" >/dev/null 2>&1; then
    airship::discover::log "Surrogate AI reachable; querying knowledge endpoint"
    curl -fsSL --max-time 30 -X POST "${surrogate_ai_url}/knowledge/query" \
      -H "Content-Type: application/json" \
      -d '{"query":"top hub and related docs","limit":5}' >> "${AIRSHIP_LOG}" 2>&1 || true
  else
    airship::discover::log "Surrogate AI not reachable — skipping knowledge-rag query"
  fi
}

airship::discover::extract_top_hub() {
  # Try log heuristics first
  local hub
  hub="$(grep -i -m1 -o -E '"hub":"[^"]+"|"name":"[^"]*MOC[^"]*"|most-connected[[:space:]]*[:=][[:space:]]*"[^"]+"' "${AIRSHIP_LOG}" 2>/dev/null | head -1 || true)"
  if [[ -n "${hub}" ]]; then
    airship::discover::log "Top hub (extracted from log): ${hub}"
    printf '%s\n' "${hub}"
    return 0
  fi

  # Fallback: Neo4j
  if curl -fsSL --max-time 10 http://localhost:7474/db/data/ >/dev/null 2>&1; then
    airship::discover::log "Neo4j reachable — attempting top-hub query"
    local payload='{"statements":[{"statement":"MATCH (h:Hub) RETURN h.name AS name, size((h)--()) AS connections ORDER BY connections DESC LIMIT 1"}]}'
    local result
    result="$(curl -fsSL --max-time 15 -X POST http://localhost:7474/db/data/transaction/commit \
      -H "Content-Type: application/json" \
      -d "${payload}" 2>/dev/null || true)"
    if echo "${result}" | grep -q '"name"'; then
      hub="$(echo "${result}" | jq -r '.results[0].data[0].row[0]' 2>/dev/null || true)"
      if [[ -n "${hub}" && "${hub}" != "null" ]]; then
        airship::discover::log "Top hub (Neo4j): ${hub}"
        printf '%s\n' "${hub}"
        return 0
      fi
    fi
  fi

  airship::discover::log "Top hub not found — defaulting to MOC"
  printf 'MOC\n'
}
