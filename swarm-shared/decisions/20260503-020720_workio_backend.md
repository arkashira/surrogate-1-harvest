## Final Synthesized Solution (Correct + Actionable)

### 1) Core Diagnosis (merged, de-duplicated)
- **LINE retries + client retries** create duplicate punches, leave, and OT rows.
- **Read-then-write upserts** are not atomic; concurrent webhooks can interleave and create double entries or invalid state transitions.
- **Missing request-level idempotency**: no stored `deliveryId`/`eventId` to reject replays; business-level uniqueness (`user+date+type`) is insufficient under retries.
- **No transactional boundary** for side effects (notifications/updates) → partial application and inconsistent state.
- **Clock-in/out state derived in app logic** → concurrent webhooks can miscompute current status and allow invalid transitions (e.g., double clock-in).
- **Approval flows are race-prone** (Candidate 2 fragment) — concurrent approval attempts can double-process or skip state checks.

### 2) Single Source of Truth & Constraints
- **Idempotency table** with unique `line_delivery_id` (and optional `event_id`) to deduplicate at request level.
- **Unique business constraint** on `(line_user_id, tenant_id, event_type, event_date, clock_session_id)` for punches and `(line_user_id, tenant_id, request_type, request_date)` for leave/OT.
- **All writes occur inside a single DB transaction**; side effects (notifications) are enqueued only after commit.
- **Deterministic clock_session_id** (e.g., `YYYY-MM-DD-userId`) to group punches per shift and enforce one valid transition per session.

```ts
// server/src/routes/line/webhook.ts
import { pool } from '../../db';
import { v4 as uuidv4 } from 'uuid';

export async function handleLineWebhook(req: any, res: any) {
  const { events } = req.body;
  const client = await pool.connect();

  try {
    await client.query('BEGIN');

    for (const ev of events) {
      const { type, source, message, webhookEventId, deliveryId } = ev;
      const lineUserId = source.userId;
      const tenantId = await resolveTenantId(lineUserId); // implement per your auth/context

      const dedupKey = deliveryId || webhookEventId || uuidv4();
      const eventDate = new Date().toISOString().split('T')[0];
      const eventType = mapLineEventType(type, message);
      if (eventType === 'unknown') {
        // skip unknown but continue processing other events
        continue;
      }

      // 1) Idempotency check (fast, unique index)
      const idem = await client.query(
        `SELECT 1 FROM line_event_idempotency WHERE line_delivery_id = $1`,
        [dedupKey]
      );
      if (idem.rowCount > 0) {
        continue; // already processed
      }

      if (eventType === 'clock_in' || eventType === 'clock_out') {
        const clockSessionId = `${eventDate}-${lineUserId}`;

        // Atomic punch insert (no duplicate session punch)
        await client.query(
          `INSERT INTO punches (tenant_id, line_user_id, event_type, event_date, clock_session_id, created_at)
           VALUES ($1, $2, $3, $4, $5, NOW())
           ON CONFLICT (line_user_id, tenant_id, event_type, event_date, clock_session_id) DO NOTHING`,
          [tenantId, lineUserId, eventType, eventDate, clockSessionId]
        );

        // Record idempotency
        await client.query(
          `INSERT INTO line_event_idempotency (tenant_id, line_user_id, event_type, event_date, clock_session_id, line_delivery_id)
           VALUES ($1, $2, $3, $4, $5, $6)`,
          [tenantId, lineUserId, eventType, eventDate, clockSessionId, dedupKey]
        );
      } else if (eventType === 'leave_request' || eventType === 'ot_request') {
        // Atomic request insert
        await client.query(
          `INSERT INTO requests (tenant_id, line_user_id, request_type, request_date, payload, created_at)
           VALUES ($1, $2, $3, $4, $5, NOW())
           ON CONFLICT (line_user_id, tenant_id, request_type, request_date) DO NOTHING`,
          [tenantId, lineUserId, eventType, eventDate, JSON.stringify(message)]
        );

        await client.query(
          `INSERT INTO line_event_idempotency (tenant_id, line_user_id, event_type, event_date, line_delivery_id)
           VALUES ($1, $2, $3, $4, $5)`,
          [tenantId, lineUserId, eventType, eventDate, dedupKey]
        );
      }

      // Note: approval state changes should use atomic DB updates (e.g.,
      // UPDATE ... WHERE current_status = 'pending' AND id = $1) and be
      // performed inside the same transaction if triggered by this webhook.
    }

    await client.query('COMMIT');

    // Enqueue side effects (notifications) after commit to avoid partial state.
    // Example: await queue.add('notify-punch', {...});
    res.status(200).json({ ok: true });
  } catch (err) {
    await client.query('ROLLBACK');
    console.error('Webhook transaction failed', err);
    // Return 500 so LINE may retry (idempotency prevents duplicates)
    res.status(500).json({ error: 'processing_failed' });
  } finally {
    client.release();
  }
}

function mapLineEventType(type: string, message: any): string {
  if (type === 'message' && message.type === 'text') {
    const text = (message.text || '').toLowerCase().trim();
    if (text.includes('in') || text.includes('clock in')) return 'clock_in';
    if (text.includes('out') || text.includes('clock out')) return 'clock_out';
    if (text.includes('leave')) return 'leave_request';
    if (text.includes('ot') || text.includes('overtime')) return 'ot_request';
  }
  return 'unknown';
}
```
