# surrogate-1 / design

## Final Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16` (matrix) and optional `DATE_FOLDER` (defaults to today `YYYY-MM-DD`)
- Uses **manifest-first strategy**: single API call to `list_repo_tree` for the date folder → saves `manifest.json` → Lightning training uses CDN-only fetches with zero API calls during data load
- Downloads files via **HF CDN** (`https://huggingface.co/datasets/.../resolve/main/...`) with no Authorization header (bypasses API 429)
- Projects heterogeneous schemas to `{prompt, response}` only at parse time (avoids PyArrow CastError)
- Deduplicates via central md5 store (`lib/dedup.py`) and writes to deterministic shard output:
  ```
  batches/public-merged/<date>/shard<N>-<HHMMSS>.jsonl
  ```
- Adds retry/backoff for 429 (wait 360s) and resiliency for transient CDN failures
- Keeps wrapper Bash-compatible: executable, `#!/usr/bin/env bash` shebang preserved for cron usage

```bash
# Wrapper for dataset-enrich.py — keeps cron/ci compatibility.
# Usage: ./dataset-enrich.sh [SHARD_ID] [SHARD_TOTAL] [DATE_FOLDER]

set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
PYTHON_SCRIPT="${SCRIPT_DIR}/dataset-enrich.py"

SHARD_ID="${1:-${SHARD_ID:-0}}"
SHARD_TOTAL="${2:-${SHARD_TOTAL:-16}}"
DATE_FOLDER="${3:-${DATE_FOLDER:-$(date +%Y-%m-%d)}}"

if [[ ! -f "${PYTHON_SCRIPT}" ]]; then
  echo "ERROR: Python worker not found at ${PYTHON_SCRIPT}" >&2
  exit 1
fi

exec python3 "${PYTHON_SCRIPT}" \
  --shard-id "${SHARD_ID}" \
  --shard-total "${SHARD_TOTAL}" \
  --date-folder "${DATE_FOLDER}"
```

#!/usr/bin/env python3
"""
Manifest-driven, CDN-bypass ingestion worker for surrogate-1 public dataset.

Behavior:
- List date folder once via HF API (list_repo_tree) -> manifest.json
- Deterministic shard assignment by slug-hash % SHARD_TOTAL
- Download assigned files via HF CDN (no auth header) to bypass 429
- Project heterogeneous schemas to {prompt, response}
- Deduplicate via central md5 store (lib/dedup.py)
- Upload shard output: batches/public-merged/<date>/shard<N>-<HHMMSS>.jsonl
"""

import argparse
import hashlib
import json

import sys
import time
from datetime import datetime, timezone

import numpy as np
import pyarrow as pa

import requests

from tqdm import tqdm

# Local dedup store
DEDUP_PY = Path(__file__).parent / "lib" / "dedup.py"
if DEDUP_PY.exists():
    # Import dedup helper if available
    import importlib.util

    spec = importlib.util.spec_from_file_location("dedup", str(DEDUP_PY))
    dedup_module = importlib.util.module_from_spec(spec)
    spec.loader.exec_module(dedup_module)
    is_duplicate = getattr(dedup_module, "is_duplicate", None)
    mark_seen = getattr(dedup_module, "mark_seen", None)
else:
    # Fallback in-memory dedup for worker isolation (cross-run state is best-effort)
    _seen = set()

    def is_duplicate(md5: str) -> bool:
        return md5 in _seen

        _seen.add(md5)

HF_REPO = "axentx/surrogate-1-training-pairs"
API_RATE_LIMIT_RETRY = 360  # seconds
CDN_BASE = f"https://huggingface.co/datasets/{HF_REPO}/resolve/main"


def slug_hash(slug: str) -> int:
    return int(hashlib.sha256(slug.encode("utf-8")).hexdigest(), 16)


def list_date_folder(api: HfApi, date_folder: str) -> List[str]:
    """List files in date folder (non-recursive)."""
    for attempt in range(8):
        try:
            tree = api.list_repo_tree(repo_id=HF_REPO, path=date_folder, recursive=False)
            return [item.rfilename for item in tree if item.rfilename.endswith((".parquet", ".jsonl"))]
        except Exception as exc:
            wait = min(API_RATE_LIMIT_RETRY, (2**attempt) + np.random.uniform(0, 1))
            print(f"list_repo_tree failed (attempt {attempt+1}): {exc}. Retry in {wait:.1f}s", file=sys.stderr)
            time.sleep(wait)
    raise RuntimeError(f"Failed to list {date_folder} after retries")


def download_via_cdn(cdn_path: str, timeout: int = 30) -> bytes:
    url = f"{CDN_BASE}/{cdn_path}"
    for attempt in range(6):
        try:
            resp = requests.get(url, timeout=timeout, stream=True)
            if resp.status_code == 429:
                wait = API_RATE_LIMIT_RETRY
                print(f"CDN 429 on {url}. Waiting {wait}s", file=sys.stderr)
                time.sleep(wait)
                continue
            resp.raise_for_status()
            return resp.content
        except Exception as exc:
            wait = min(30, (2**attempt) + np.random.uniform(0, 1))
            print(f"CDN download failed {url} (attempt {attempt+1}): {exc}. Retry in {wait:.1f}s", file=sys.stderr)
            time.sleep(wait)
    raise RuntimeError(f"Failed to download {cdn_path} after retries")


def project_to_pair(raw: Dict[str, Any], filename: str) -> Optional[Dict[str, Any]]:
    """
    Project heterogeneous schema to {prompt, response}.
    Heuristic:
    - Prefer keys: prompt/response, question/answer, input/output, user/assistant
    - If single text column and metadata-like other columns, use text as prompt and metadata as response or vice versa
    - Fallback: serialize row as prompt and empty response
    """
    keys = {k.lower(): k for k in raw.keys()}

    prompt_candidates = [keys.get(k) for k in ("prompt", "question", "input", "user", "text") if keys.get(k)]
    response_candidates = [keys.get(k) for k in ("response", "answer", "output", "assistant") if keys.get(k)]

    prompt_key = next((k for k in prompt_candidates if k), None)
    response_key = next((k for k in response_candidates if k), None)

        return {"prompt": str(raw[prompt_key]), "response": str(raw[response_key])}

        return {"prompt": str(raw[prompt_key]), "response": str(other)}

        single = next(iter(raw.values()))
