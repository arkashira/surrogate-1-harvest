# airship / backend

## Decision: ship `airship discover` (backend)

**Highest-value incremental improvement** (≤2h):  
Add a zero-config, cron-safe orchestrator (`airship/discover.py`) that:

- runs `granite-business-research.sh` (if present) and tags outputs with `#business-research #knowledge-rag #graph`
- executes `knowledge-rag` to query top hub and related docs (tags: `#knowledge-rag #graph #hub`)
- produces a **CDN-only training manifest** (HF dataset file list via single `list_repo_tree` → JSON) so Lightning training can fetch via CDN without API/auth during data load
- is idempotent, logs to stdout/stderr, exits non-zero on hard failures, and is safe for cron (no interactive prompts, short-circuits when already running via lockfile)

---

## Implementation plan (≤2h)

1. **Create `airship/discover.py`** (single entrypoint)
   - Shebang `#!/usr/bin/env python3`
   - Lockfile `/tmp/airship-discover.lock` to prevent overlapping cron runs
   - Idempotent: skip steps if outputs already exist and are fresh (configurable staleness, default 24h)
   - Runs:
     - `granite-business-research.sh` (if executable) → capture stdout/stderr → tag outputs
     - `knowledge-rag` CLI (if available) → query top hub (e.g., "MOC") and related docs → emit tagged insights
     - HF manifest: single `list_repo_tree` call for a date folder (or repo root) → save `training-manifest-{date}.json` with CDN-resolvable paths
   - Emits structured JSON lines to stdout and a dated log file in `logs/`

2. **Create lightweight HF manifest builder**
   - Uses `huggingface_hub.list_repo_tree(repo_id, path, recursive=False)` (single API call)
   - Walks folders one level and collects file paths
   - Saves `{"repo_id": "...", "files": ["path1", ...], "cdn_base": "https://huggingface.co/datasets/{repo_id}/resolve/main/"}`
   - Avoids `load_dataset` and recursive listing to stay within rate limits

3. **Cron-safe invocation wrapper**
   - Ensure cron uses `SHELL=/bin/bash`
   - Invoke via `python3 /opt/axentx/airship/discover.py >> /var/log/airship/discover.log 2>&1`

4. **Add minimal unit tests** (optional but recommended)
   - Mock HF API and test manifest shape
   - Test lockfile behavior

def acquire_lock() -> Optional[int]:
    """Create and hold lockfile; return fd or None if already locked."""
    try:
        fd = os.open(LOCKFILE, os.O_CREAT | os.O_WRONLY | os.O_EXCL)
        fcntl.lockf(fd, fcntl.LOCK_EX | fcntl.LOCK_NB)
        os.write(fd, str(os.getpid()).encode())
        return fd
    except (FileExistsError, BlockingIOError, OSError):
        return None

def release_lock(fd: Optional[int]) -> None:
    if fd is not None:
        try:
            fcntl.lockf(fd, fcntl.LOCK_UN)
            os.close(fd)
        except OSError:
            pass
        try:
            os.unlink(LOCKFILE)
        except OSError:
            pass

def run_cmd(cmd: List[str], cwd: Optional[Path] = None, check: bool = False) -> subprocess.CompletedProcess:
    logging.info("Running: %s (cwd=%s)", " ".join(cmd), cwd)
    return subprocess.run(
        cmd,
        cwd=cwd or REPO_ROOT,
        capture_output=True,
        text=True,
        check=check
    )

def tag_file(path: Path, tags: List[str]) -> None:
    """Append a small tagged header to a file if not already tagged."""
    try:
        content = path.read_text()
        tag_line = f"# TAGS: {', '.join(tags)}\n"
        if tag_line not in content:
            path.write_text(tag_line + content)
            logging.info("Tagged %s with %s", path.name, tags)
    except Exception as exc:
        logging.warning("Failed to tag %s: %s", path, exc)

def run_business_research() -> None:
    script = REPO_ROOT / "granite-business-research.sh"
    if not script.is_file() or not os.access(script, os.X_OK):
        logging.info("granite-business-research.sh not found or not executable; skipping")
        return

    result = run_cmd([str(script)])
    ts = datetime.now(timezone.utc).strftime("%Y%m%dT%H%M%S")
    out_path = REPO_ROOT / "research" / f"business-research-{ts}.md"
    out_path.parent.mkdir(exist_ok=True)
    out_path.write_text(result.stdout)
    tag_file(out_path, ["#business-research", "#knowledge-rag", "#graph"])
    logging.info("Business research saved to %s", out_path)

    # Also tag stderr insights if present
    if result.stderr.strip():
        err_path = REPO_ROOT / "research" / f"business-research-{ts}.stderr.md"
        err_path.write_text(result.stderr)
        tag_file(err_path, ["#business-research", "#knowledge-rag", "#graph"])

def run_knowledge_rag() -> None:
    # Try common executable names
    for exe in ["knowledge-rag", "knowledge_rag", "./knowledge-rag"]:
        cmd = None
        p = (REPO_ROOT / exe)
        if p.is_file() and os.access(p, os.X_OK):
            cmd = [str(p)]
            break
        # also check PATH
        try:
            subprocess.run(["which", exe], capture_output=True, check=True)
            cmd = [exe]
            break
        except Exception:
            continue

    if not cmd:
        logging.info("knowledge-rag not found; skipping")
        return

    # Query top hub (e.g., MOC) and related docs
    result = run_cmd(cmd + ["query", "top-hub", "--limit", "5"])
    ts = datetime.now(timezone.utc).strftime("%Y%m%dT%H%M%S")
    out_path = REPO_ROOT / "research" / f"knowledge-rag-top-hub-{ts}.md"
    out_path.parent.mkdir(exist_ok=True)
    out_path.write_text(result.stdout)
    tag_file(out_path, ["#knowledge-rag", "#graph", "#hub"])
    logging.info("Knowledge RAG top-hub saved to %s", out_path)
