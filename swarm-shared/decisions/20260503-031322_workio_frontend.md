# workio / frontend

## Final Synthesized Implementation  
File: `workio/src/components/TimeClock.tsx` (create/update)

### 1. Diagnosis (merged)
- No optimistic UI or local pending queue → double-taps create duplicate attempts and perceived slowness.
- No offline/weak-network handling; failed LINE/webhook round-trips lose intent and force manual retry.
- No immediate local timestamp + GPS capture; relies on backend round-trip, increasing perceived latency.
- No visual distinction between pending/committing/succeeded/failed; users can’t tell if punch was accepted.
- Duplicate submissions can race before backend idempotency; frontend should enforce client-side dedupe by a stable `client_event_id`.

### 2. Core Principles
- **Correctness**: One punch per user action. Use stable, unique `client_event_id` and enforce client-side dedupe.
- **Actionability**: Immediate optimistic state, local persistence, background submit with retry, clear status UI, and graceful degradation when offline.
- **Performance**: Capture timestamp + GPS at button press (not after round-trip). Show local time immediately.
- **Resilience**: Persist pending/failed punches to `localStorage` (best-effort) and retry with exponential backoff + jitter.

import { useState, useEffect, useCallback, useRef } from 'react';
import { useLocation } from 'react-router-dom';

type PunchType = 'clock_in' | 'clock_out';
type PunchStatus = 'idle' | 'pending' | 'committing' | 'succeeded' | 'failed';

interface LocalPunch {
  client_event_id: string;
  type: PunchType;
  timestamp: number;
  latitude?: number;
  longitude?: number;
  status: PunchStatus;
  error?: string | null;
  retries?: number;
}

const STORAGE_KEY = 'workio:pending-punches';
const MAX_RETRIES = 5;
const BASE_DELAY_MS = 1500;

function usePendingPunches() {
  const [punches, setPunches] = useState<LocalPunch[]>([]);

  useEffect(() => {
    try {
      const raw = localStorage.getItem(STORAGE_KEY);
      if (raw) setPunches(JSON.parse(raw));
    } catch {
      setPunches([]);
    }
  }, []);

  const persist = useCallback((next: LocalPunch[]) => {
    try {
      localStorage.setItem(STORAGE_KEY, JSON.stringify(next));
    } catch {
      // ignore storage quota/incognito failures
    }
    setPunches(next);
  }, []);

  const add = useCallback((p: Omit<LocalPunch, 'status' | 'retries'>) => {
    const item: LocalPunch = { ...p, status: 'pending', retries: 0 };
    persist((prev) => [item, ...prev.filter((x) => x.client_event_id !== item.client_event_id)]);
    return item;
  }, [persist]);

  const update = useCallback((client_event_id: string, patch: Partial<LocalPunch>) => {
    persist((prev) =>
      prev.map((p) => (p.client_event_id === client_event_id ? { ...p, ...patch } as LocalPunch : p))
    );
  }, [persist]);

  const remove = useCallback((client_event_id: string) => {
    persist((prev) => prev.filter((p) => p.client_event_id !== client_event_id));
  }, [persist]);

  return { punches, add, update, remove };
}

function useNetworkStatus() {
  const [online, setOnline] = useState<boolean>(true);
  useEffect(() => {
    const onOnline = () => setOnline(true);
    const onOffline = () => setOnline(false);
    window.addEventListener('online', onOnline);
    window.addEventListener('offline', onOffline);
    setOnline(navigator.onLine);
    return () => {
      window.removeEventListener('online', onOnline);
      window.removeEventListener('offline', onOffline);
    };
  }, []);
  return online;
}
