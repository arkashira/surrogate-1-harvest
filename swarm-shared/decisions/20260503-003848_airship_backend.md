## Implementation Plan — Manifest-Driven CDN-Only Dataset Loader

**Scope**: Replace `load_dataset`/`list_repo_files` usage in surrogate training with a manifest-driven, CDN-only iterable loader to eliminate HF API rate limits and mixed-schema ingestion failures.

**Estimated effort**: 1.5h (implementation + tests)

---

### 1. Strategy (HF CDN Bypass + Manifest)

- **Pre-list once on Mac orchestrator**: `list_repo_tree(path, recursive=False)` for a single date folder → save `manifest.json`
- **Embed manifest in training script**: Lightning Studio fetches files via CDN URLs only (`resolve/main/...`) — zero API calls during training
- **Schema projection at parse time**: Download individual files via `hf_hub_download` (or raw CDN) and project to `{prompt, response}` only; ignore heterogeneous columns
- **No `load_dataset(streaming=True)`** on repos with mixed schemas

### 3. Code — `dataset/manifest_dataset.py`

```python
# surrogate/training/dataset/manifest_dataset.py
import json
import os
from typing import Dict, Iterator, Optional
from pathlib import Path

import pyarrow as pa
import pyarrow.parquet as pq
from huggingface_hub import hf_hub_download
import requests


class ManifestDataset:
    """
    Manifest-driven, CDN-only iterable dataset.

    Manifest format:
    {
      "repo_id": "org/surrogate-data",
      "revision": "main",
      "files": [
        {"path": "batches/mirror-merged/2026-05-01/s1.parquet", "sha256": "..."},
        ...
      ]
    }

    Behavior:
    - No load_dataset()
    - No list_repo_files() during training
    - CDN-only downloads (resolve/main/...) or hf_hub_download
    - Projects each file to {prompt, response} at parse time
    """

    def __init__(
        self,
        manifest_path: str,
        repo_id: Optional[str] = None,
        revision: str = "main",
        use_cdn: bool = True,
        cache_dir: Optional[str] = None,
    ):
        self.manifest_path = Path(manifest_path)
        self.repo_id = repo_id
        self.revision = revision
        self.use_cdn = use_cdn
        self.cache_dir = cache_dir or os.getenv("HF_HOME", Path.home() / ".cache" / "huggingface")
        self.manifest = self._load_manifest()

    def _load_manifest(self) -> dict:
        with open(self.manifest_path, "r", encoding="utf-8") as f:
            return json.load(f)

    def _cdn_url(self, file_path: str) -> str:
        repo = self.manifest.get("repo_id") or self.repo_id
        rev = self.manifest.get("revision", self.revision)
        # Public CDN URL — no auth header required
        return f"https://huggingface.co/datasets/{repo}/resolve/{rev}/{file_path}"

    def _download_file(self, file_path: str, local_path: Path) -> Path:
        if self.use_cdn:
            # CDN fetch (bypasses API rate limits)
            url = self._cdn_url(file_path)
            resp = requests.get(url, timeout=60)
            resp.raise_for_status()
            local_path.write_bytes(resp.content)
        else:
            # Fallback to hf_hub_download (uses API but reliable)
            local_path.parent.mkdir(parents=True, exist_ok=True)
            hf_hub_download(
                repo_id=self.manifest.get("repo_id") or self.repo_id,
                filename=file_path,
                revision=self.manifest.get("revision", self.revision),
                cache_dir=str(self.cache_dir),
                local_dir=local_path.parent,
                local_dir_use_symlinks=False,
            )
        return local_path

    def _project_to_pair(self, file_path: Path) -> Iterator[Dict[str, str]]:
        """
        Project parquet file to {prompt, response} only.
        Tolerates heterogeneous schemas; missing fields become empty strings.
        """
        try:
            table = pq.read_table(file_path, columns=["prompt", "response"])
        except (pa.ArrowInvalid, KeyError, FileNotFoundError):
            # Fallback: read all and select if columns exist
            table = pq.read_table(file_path)
            cols = table.column_names
            prompt_col = next((c for c in ["prompt", "Prompt", "input", "question"] if c in cols), None)
            response_col = next((c for c in ["response", "Response", "output", "answer"] if c in cols), None)
            if prompt_col is None or response_col is None:
                # Cannot project; skip file
                return
            table = table.select([prompt_col, response_col]).rename_columns(["prompt", "response"])

        df = table.to_pandas()
        for _, row in df.iterrows():
            prompt = str(row.get("prompt", "")) if pd_notna(row.get("prompt")) else ""
            response = str(row.get("response", "")) if pd_notna(row.get("response")) else ""
            if prompt.strip() or response.strip():
                yield {"prompt": prompt, "response": response}

    def __iter__(self) -> Iterator[Dict[str, str]]:
        import pandas as pd
        global pd_notna
        pd_notna = pd.notna

        files = self.manifest.get("files", [])
        if not files:
            return

        for entry in files:
            file_path = entry["path"]
            cache_subpath = Path("manifest_files") / file_path
            local_path = Path(self.cache_dir) / cache_subpath

            if not local_path.exists():
                local_path.parent.mkdir(parents=True, exist_ok=True)
                try:
                    self._download_file(file_path, local_path)
                except Exception as exc:
                    # Skip corrupted/unavailable files; log in real usage
                    continue

            try:
                yield from self._project_to_pair(local_path)
            except Exception:
                # Skip unreadable files
                continue
```

### 4. Code — `train.py` (minimal integration)

```python
# surrogate/training/train.py
import os
from pathlib import Path

from torch.utils.data import IterableDataset

from dataset.manifest_dataset import ManifestDataset


MANIFEST_PATH = os.getenv(
    "SURROGATE_MANIFEST_PATH",
    Path(__file__).parent / "manifest.json"
)


def build_dataset() -> IterableDataset:
    return ManifestDataset(
        manifest_path=MANIFEST_PATH,
        use_cdn=True,
        cache_dir=os.getenv("HF_HOME", Path.home() / ".cache" / "huggingface"),
    )


# Example usage in training loop
if __name__ == "__main__":
    dataset = build_dataset()
    for i, item in enumerate(dataset):
        if i >= 10:
            break
        print(item)
```

### 5. Orchestrator snippet (Mac) — generate manifest

```bash
#!/usr/bin/env bash
# scripts/generate_manifest.sh
set -euo pipefail

REPO_ID="org/surrogate-data"
DATE_FOLDER="2026-05-01"
OUT="surrogate/training/manifest.json"

python - <<PY
import json
from huggingface_hub import HfApi

api = HfApi()
tree = api.list_repo_tree(
    repo_id="$REPO_ID",
    path="batches/mirror-merged/$DATE_FOLDER",
    recursive=False,
)

files = [{"path": item.path} for item in tree if item.type == "file"]
manifest = {
    "repo_id": "$REPO_ID",
    "revision": "main",
    "files": files,
}

with open("$OUT", "w") as f:
    json.dump(manifest, f, indent=2)

print(f"Manifest written: $OUT ({len(files)} files)")
PY
```
