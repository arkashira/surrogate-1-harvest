## Final Integrated Implementation

**Goal:** Make Surrogate-1 training resilient to HF API 429s and Lightning Studio idle-stop by implementing deterministic CDN-only data loading + studio lifecycle resilience.

**Why:**  
- Eliminates HF API rate-limit failures during data loads  
- Prevents quota loss from idle-stop/studio recreation  
- Reduces training flakiness to near-zero with minimal code changes  

---

## Implementation Plan (≤2h)

### 1. Pre-list file paths once (or fallback gracefully)
Single API call to list one date folder → save JSON. Embed in training script for CDN-only fetches. If listing fails, fall back to remote directory listing via CDN HTML or a small local allow-list so training never blocks.

### 2. CDN-only data loader with robust retries and projection
Use `https://huggingface.co/datasets/{repo}/resolve/main/{path}` with `aiohttp` + `pyarrow` projection to `{prompt, response}`. Add:
- Exponential backoff + jitter for CDN 429/503  
- Per-file integrity checks (size/row count)  
- Optional streaming for large files  
- Clear error messages for missing columns  

### 3. Lightning Studio lifecycle resilience
- Reuse running studios  
- Auto-restart on idle-stop  
- Use L40S (free tier) with fallback priority  
- Emit run IDs and log outputs to persistent storage  

### 4. Cron-safe execution and observability
- Wrapper script that never fails the cron job (captures exit codes)  
- Structured logs + optional metrics push  
- Idempotent run: reusing CDN file list and partial checkpoints

### 1. Pre-list file paths (with CDN fallback)

```python
# scripts/list_cdn_files.py
#!/usr/bin/env python3
"""
Run once after rate-limit window clears.
Saves file list for CDN-only training. Falls back to CDN HTML listing
if HF API is unavailable.
"""
import json
import os
import sys
from datetime import datetime, timezone
from typing import List, Dict, Any

try:
    from huggingface_hub import HfApi
except ImportError:
    HfApi = None

import aiohttp
import asyncio
from bs4 import BeautifulSoup

REPO_ID = "axentx/surrogate-1-data"
DATE_FOLDER = os.getenv("DATA_DATE_FOLDER", "2026-05-03")
OUT_PATH = "cdn_file_list.json"

def _normalize_path(p: str) -> str:
    return p.lstrip("/")

async def _cdn_html_list(session: aiohttp.ClientSession, prefix: str) -> List[str]:
    """Fallback: parse CDN HTML directory listing."""
    url = f"https://huggingface.co/datasets/{REPO_ID}/tree/main/{prefix}"
    try:
        async with session.get(url, timeout=15) as resp:
            if resp.status != 200:
                return []
            html = await resp.text()
    except Exception:
        return []

    soup = BeautifulSoup(html, "html.parser")
    files = []
    for a in soup.select('a[href]'):
        href = a["href"]
        # href pattern: /datasets/.../resolve/main/... or /tree/main/...
        if "/resolve/main/" in href and href.endswith(".parquet"):
            # extract path after /resolve/main/
            parts = href.split("/resolve/main/")
            if len(parts) == 2:
                files.append(_normalize_path(parts[1]))
        elif "/tree/main/" in href and href.endswith(".parquet"):
            # tree links need conversion to resolve
            parts = href.split("/tree/main/")
            if len(parts) == 2:
                files.append(_normalize_path(parts[1]))
    return files

async def _fallback_list_files() -> List[str]:
    connector = aiohttp.TCPConnector(limit=5)
    async with aiohttp.ClientSession(connector=connector) as session:
        files = await _cdn_html_list(session, f"{DATE_FOLDER}")
        # Deduplicate and sort for determinism
        return sorted(set(files))

def list_files_hf_api() -> List[str]:
    if HfApi is None:
        raise RuntimeError("huggingface_hub not installed")
    api = HfApi()
    tree = api.list_repo_tree(
        repo_id=REPO_ID,
        path=DATE_FOLDER,
        recursive=False
    )
    files = [
        _normalize_path(f"{DATE_FOLDER}/{entry.path.split('/')[-1]}")
        for entry in tree
        if entry.path.endswith(".parquet")
    ]
    return sorted(set(files))

def build_manifest(files: List[str]) -> Dict[str, Any]:
    return {
        "repo_id": REPO_ID,
        "date_folder": DATE_FOLDER,
        "generated_at_utc": datetime.now(timezone.utc).isoformat(),
        "files": files,
        "total": len(files)
    }

def main() -> None:
    files: List[str] = []
    used_fallback = False

    # Try HF API first
    try:
        if HfApi is None:
            raise RuntimeError("huggingface_hub not installed")
        files = list_files_hf_api()
        print(f"Listed {len(files)} files via HF API")
    except Exception as e:
        print(f"HF API listing failed ({e}), falling back to CDN HTML listing...")
        try:
            files = asyncio.run(_fallback_list_files())
            used_fallback = True
            print(f"Listed {len(files)} files via CDN fallback")
        except Exception as e2:
            print(f"CDN fallback failed: {e2}")
            sys.exit(1)

    if not files:
        print("No parquet files found.")
        sys.exit(1)

    manifest = build_manifest(files)
    with open(OUT_PATH, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2)

    print(f"Saved {len(files)} files to {OUT_PATH}" + (" (fallback)" if used_fallback else ""))

if __name__ == "__main__":
    main()
```
