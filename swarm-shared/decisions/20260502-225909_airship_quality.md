## Concrete Steps (≤2h)

1. **Harden entrypoint**
   - Use strict POSIX shell: `#!/usr/bin/env bash` + `set -euo pipefail`
   - Replace non-POSIX constructs; use `mktemp`, `printf`, `date -u +%Y-%m-%dT%H:%M:%SZ`
   - Make executable: `chmod +x airship/discover`

2. **Deterministic JSON emitters**
   - `emit_status()`: writes `status.json` with stable keys `{"ts","status","step","checksum","message"}`
   - `emit_manifest()`: writes `manifest.json` with `{"generated_at","root_path","files":[{"path","sha256","size"}]}` sorted by path
   - Add `file-list.json` (array of `{"path","size"}`) for embedding into training

3. **CDN-safe path + deterministic sibling routing**
   - Compute `sha256_slug=$(printf '%s' "$payload" | sha256sum | awk '{print $1}')`
   - Target path: `batches/mirror-merged/$(date -u +%Y-%m-%d)/${sha256_slug}.parquet`
   - Sibling repo: `crc32(sha256_slug) % N` (N=5) → deterministic routing to avoid HF commit cap

4. **Pre-list & embed file list (single HF API call)**
   - One `list_repo_tree` (or equivalent) for the date folder
   - Save to `file-list.json`
   - Embed into `train.py` as constant; Lightning data loader uses CDN URLs only:


5. **Retry/backoff for CDN downloads**
   - Use `curl --retry 5 --retry-delay 2 --retry-max-time 120 -fSL "$url" -o "$out"`
   - On 429 from CDN, wait 360s before retry (per HF CDN guidance)

6. **Exit-code contract & CI/CD**
   - Exit 0 = success; non-zero = failure with error in `status.json`
   - Update GitHub Actions / cron to invoke via `bash airship/discover "$@"`
   - Ensure `SHELL=/bin/bash` in crontab

## Final Code Snippets

### `airship/discover` (hardened entrypoint)
```bash
#!/usr/bin/env bash
set -euo pipefail

# Deterministic timestamp UTC
ts() { date -u +%Y-%m-%dT%H:%M:%SZ; }

# Emit status.json with stable key order
emit_status() {
  local status="$1" step="$2" message="$3"
  local checksum="${4:-}"
  jq -n \
    --arg ts "$(ts)" \
    --arg status "$status" \
    --arg step "$step" \
    --arg message "$message" \
    --arg checksum "$checksum" \
    '{
      ts: $ts,
      status: $status,
      step: $step,
      message: $message,
      checksum: $checksum
    }' > status.json
}

# Emit manifest.json sorted by path
emit_manifest() {
  local root_path="$1"
  shift
  local files=("$@")
  local entries=()
  for f in "${files[@]}"; do
    local sha256 size
    sha256=$(sha256sum "$f" | awk '{print $1}')
    size=$(stat -c%s "$f" 2>/dev/null || stat -f%z "$f" 2>/dev/null)
    entries+=("$(jq -n --arg p "$f" --arg s "$sha256" --arg sz "$size" '{path:$p,sha256:$s,size:$sz}')")
  done

  jq -n \
    --arg generated_at "$(ts)" \
    --arg root_path "$root_path" \
    --argjson files "$(printf '%s\n' "${entries[@]}" | jq -s 'sort_by(.path)')" \
    '{
      generated_at: $generated_at,
      root_path: $root_path,
      files: $files
    }' > manifest.json
}

# Emit file-list.json for CDN-only fetches
emit_file_list() {
  local root_path="$1"
  shift
  local files=("$@")
  local entries=()
  for f in "${files[@]}"; do
    local size
    size=$(stat -c%s "$f" 2>/dev/null || stat -f%z "$f" 2>/dev/null)
    # Use relative path from root_path for CDN consistency
    local rel
    rel=$(realpath --relative-to="$root_path" "$f")
    entries+=("$(jq -n --arg p "$rel" --arg sz "$size" '{path:$p,size:$sz}')")
  done

  jq -n \
    --argjson files "$(printf '%s\n' "${entries[@]}" | jq -s 'sort_by(.path)')" \
    '$files' > file-list.json
}

# CDN-safe upload path
cdn_path_for() {
  local payload="$1"
  local sha256_slug
  sha256_slug=$(printf '%s' "$payload" | sha256sum | awk '{print $1}')
  local day
  day=$(date -u +%Y-%m-%d)
  echo "batches/mirror-merged/${day}/${sha256_slug}.parquet"
}

# Deterministic sibling repo routing (avoid HF commit cap)
pick_sibling_repo() {
  local sha256_slug="$1"
  local n_siblings="${2:-5}"
  local idx
  idx=$(printf '%s' "$sha256_slug" | cksum | awk -v n="$n_siblings" '{print $1 % n}')
  echo "sibling-repo-${idx}"
}

# Main
main() {
  local root="${1:-./discover-root}"
  emit_status "running" "discover-start" "Starting discovery" ""

  # Collect files deterministically
  mapfile -t files < <(find "$root" -type f -print0 | sort -z | xargs -0 -I{} printf '%s\n' "{}")

  emit_status "running" "discover-files" "Found ${#files[@]} files" ""
  emit_manifest "$root" "${files[@]}"
  emit_file_list "$root" "${files[@]}"

  # Example: build payload and compute CDN path
  local payload
  payload=$(jq -n --arg root "$root" --slurpfile mf manifest.json '$mf[0] | tostring')
  local cdn_dest
  cdn_dest=$(cdn_path_for "$payload")
  emit_status "running" "discover-path" "CDN path: $cdn_dest" "$(printf '%s' "$payload" | sha256sum | awk '{print $1}')"

  # Pre-list HF repo tree (single API call) — example placeholder
  # hf_repo="org/dataset"
  # curl -s -H "Authorization: Bearer $HF_TOKEN" "https://huggingface.co/api/datasets/$hf_repo/tree?recursive=false" > hf-tree.json

  emit_status "ok" "discover-complete" "Discovery complete" "$(sha256sum manifest.json | awk '{print $1}')"
}

main "$@"
```
