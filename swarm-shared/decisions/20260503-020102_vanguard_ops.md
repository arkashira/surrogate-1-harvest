# vanguard / ops

## 1. Diagnosis
- No persisted `(repo, dateFolder) → file-list` manifest: every training/data-selection run triggers authenticated `list_repo_tree` against HF API, burning quota and risking 429s.
- Data loader likely uses `load_dataset(streaming=True)` or repeated per-file API calls on heterogeneous repos, causing `pyarrow.CastError` from mixed schemas.
- Training runs on Lightning risk quota waste and idle-stop deaths because studios are recreated instead of reused and idle timeouts aren’t handled.
- No CDN bypass strategy: training still relies on HF `/api/` endpoints during data loading instead of raw CDN URLs, exposing the pipeline to rate limits.
- Missing deterministic shard-to-repo mapping to bypass HF’s 128-commit/hr repo cap when ingesting large volumes.

## 2. Proposed change
Create `/opt/axentx/vanguard/ops/persist_filelist.py` (single orchestration script) and update the training launcher stub in `/opt/axentx/vanguard/train/train.py` to consume a persisted file list and use CDN-only fetches. Scope:
- Add `ops/persist_filelist.py` (mac-safe, orchestration only) to list one date folder via `list_repo_tree`, save `manifests/{repo}/{date}.json`, and embed repo/file metadata.
- Add `train/train.py` changes to read the manifest and fetch parquet files via `hfcdn` URLs (no auth, no API) and project `{prompt, response}` only.
- Add `train/lightning_launcher.py` stub to reuse running studios and restart on idle-stop.

### ops/persist_filelist.py
```python
#!/usr/bin/env python3
"""
Orchestration-only (run on Mac).
Produce manifest: manifests/{repo_slug}/{date}.json
containing CDN paths for one date folder.
"""
import argparse
import json
import os
import time
from pathlib import Path

try:
    from huggingface_hub import HfApi, login
except ImportError:
    raise SystemExit("pip install huggingface_hub")

HF_API = HfApi()
MANIFEST_ROOT = Path(__file__).parent.parent / "manifests"

def persist_filelist(
    repo_id: str,
    date_folder: str,
    out_dir: Path = MANIFEST_ROOT,
    retries: int = 3,
    backoff: int = 360,
) -> Path:
    """
    List one date folder (non-recursive) and persist CDN paths.
    date_folder example: '2026-04-29'
    """
    out_dir.mkdir(parents=True, exist_ok=True)
    out_path = out_dir / f"{repo_id.replace('/', '_')}_{date_folder}.json"

    # Optional: login(token=os.getenv("HF_TOKEN")) if private repo
    # Public dataset files do not require auth for CDN downloads.

    for attempt in range(1, retries + 1):
        try:
            # Single API call: non-recursive tree for the date folder
            tree = HF_API.list_repo_tree(
                repo_id=repo_id,
                path=date_folder,
                recursive=False,
            )
            files = [item.rfilename for item in tree if item.rfilename.endswith(".parquet")]

            manifest = {
                "repo_id": repo_id,
                "date_folder": date_folder,
                "cdn_base": f"https://huggingface.co/datasets/{repo_id}/resolve/main",
                "files": sorted(files),
                "generated_at": time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime()),
            }

            out_path.write_text(json.dumps(manifest, indent=2))
            print(f"Manifest written: {out_path}")
            return out_path

        except Exception as exc:
            wait = backoff if attempt == retries else (2 ** attempt) * 10
            print(f"Attempt {attempt}/{retries} failed: {exc}. Retry in {wait}s")
            if attempt == retries:
                raise
            time.sleep(wait)

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Persist HF file list for CDN-only training.")
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g., 'username/dataset')")
    parser.add_argument("--date", required=True, help="Date folder (e.g., '2026-04-29')")
    args = parser.parse_args()

    persist_filelist(args.repo, args.date)
```

# Add near top
import json
import random
from pathlib import Path
from typing import List, Dict

import pyarrow.parquet as pq
import torch
from torch.utils.data import IterableDataset, DataLoader

# CDN-only dataset loader (no HF API/auth during training)
class CDNParquetDataset(IterableDataset):
    def __init__(self, manifest_path: str, columns=("prompt", "response"), shuffle=True):
        manifest = json.loads(Path(manifest_path).read_text())
        self.cdn_base = manifest["cdn_base"]
        self.files = [f"{self.cdn_base}/{f}" for f in manifest["files"]]
        self.columns = columns
        self.shuffle = shuffle

    def _stream(self):
        files = self.files.copy()
        if self.shuffle:
            random.Random(42).shuffle(files)

        for url in files:
            # CDN fetch (no auth). Consider using fsspec or direct HTTP range reads for large files.
            # For simplicity, download to temp path; replace with streaming HTTP if needed.
            import tempfile
            import urllib.request

            with tempfile.NamedTemporaryFile(suffix=".parquet", delete=False) as tmp:
                urllib.request.urlretrieve(url, tmp.name)
                tbl = pq.read_table(tmp.name, columns=self.columns)
                df = tbl.to_pandas()
                for _, row in df.iterrows():
                    yield {k: row[k] for k in self.columns}

    def __iter__(self):
        return self._stream()

# Example usage in training loop
def build_dataloader(manifest_path: str, batch_size: int = 8, num_workers: int = 0) -> DataLoader:
    dataset = CDNParquetDataset(manifest_path, columns=("prompt", "response"), shuffle=True)
    return DataLoader(dataset, batch_size=batch_size, num_workers=num_workers)
```

#!/usr/bin/env python3
"""
Lightning launcher stub: reuse running studios and restart on idle-stop.
"""
import time
from lightning import Lightning, Teamspace, Studio, Machine

def get_or_start_studio(name: str, machine: Machine = Machine.L40S) -> Studio:
    ts = Teamspace()
    for s in ts.studios:
        if s.name == name and s.status == "Running":
            print(f"Reusing running studio: {name}")
            return s

    print(f"Creating studio: {name}")
    return Studio(
        name=name,
        machine=machine,
        create_ok=True,
    )

def run_with_idle_restart(studio: Studio, target, max_restarts: int = 3):
    restarts = 0
    while restarts <= max_restarts:
        if studio.status != "Running":
            print(f"Studio stopped. Restarting ({restarts+1}/{max_restarts})...")
            studio.start(machine=studio.machine)
            restarts += 1
            time.sleep(30)
            continue
        try:
            studio.run(target)
            break
        except Exception as exc:
            print(f"Run failed: {exc}")
            restarts += 1
            time.sleep(60)
```
