## Code Snippets

### `scripts/generate_manifest.py`
```python
#!/usr/bin/env python3
"""
Generate CDN-only manifest for a date folder in a HuggingFace dataset repo.
Usage:
  HF_REPO=datasets/your/repo python generate_manifest.py --date 2026-05-01 --out manifest.json
"""
import argparse
import json
import os
import sys
from datetime import datetime

from huggingface_hub import HfApi

HF_REPO = os.getenv("HF_REPO", "datasets/your/repo")

def main() -> None:
    parser = argparse.ArgumentParser(description="Generate CDN manifest for HF dataset folder.")
    parser.add_argument("--date", required=True, help="Date folder (YYYY-MM-DD)")
    parser.add_argument("--out", default="manifest.json", help="Output JSON path")
    parser.add_argument("--repo", default=HF_REPO, help="HF repo (e.g., datasets/your/repo)")
    args = parser.parse_args()

    api = HfApi()
    folder_path = f"{args.date}"  # top-level date folder; adjust if nested
    try:
        tree = api.list_repo_tree(repo_id=args.repo, path=folder_path, recursive=False)
    except Exception as e:
        print(f"Error listing repo tree: {e}", file=sys.stderr)
        sys.exit(1)

    files = []
    for item in tree:
        if item.rfilename and item.rfilename.endswith(".parquet"):
            files.append({
                "path": item.rfilename,
                "cdn_url": f"https://huggingface.co/{args.repo}/resolve/main/{item.rfilename}"
            })

    manifest = {
        "repo": args.repo,
        "date": args.date,
        "generated_at": datetime.utcnow().isoformat() + "Z",
        "files": files,
        "total_files": len(files)
    }

    with open(args.out, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2)

    print(f"Wrote manifest with {len(files)} files to {args.out}")

if __name__ == "__main__":
    main()
```

### `surrogate/train.py` (excerpt)

import argparse
import json
import pyarrow.parquet as pq
import requests
from io import BytesIO
from typing import Iterator, Tuple

def stream_parquet_rows(cdn_url: str, max_retries: int = 5) -> Iterator[Tuple[str, str]]:
    for attempt in range(max_retries):
        try:
            resp = requests.get(cdn_url, timeout=30)
            resp.raise_for_status()
            table = pq.read_table(BytesIO(resp.content))
            # Project to {prompt, response} at parse time; ignore other columns
            prompts = table.column("prompt").to_pylist()
            responses = table.column("response").to_pylist()
            for p, r in zip(prompts, responses):
                yield str(p), str(r)
            return
        except Exception as exc:
            if attempt == max_retries - 1:
                raise
            wait = 2 ** attempt
            print(f"Retry {attempt+1}/{max_retries} after {wait}s for {cdn_url}: {exc}")
            import time; time.sleep(wait)

def load_dataset_from_manifest(manifest_path: str) -> Iterator[Tuple[str, str]]:
    with open(manifest_path, "r", encoding="utf-8") as f:
        manifest = json.load(f)
    for fobj in manifest.get("files", []):
        yield from stream_parquet_rows(fobj["cdn_url"])

### `scripts/run_surrogate_training.py`
```python
#!/usr/bin/env python3
"""
Lightning-idle-resilient runner for Surrogate training.
Reuses running Studio; restarts if stopped.
"""
import time
from lightning import Lightning, Teamspace, Machine, Studio

LIGHTNING_PROJECT = "surrogate-training"
STUDIO_NAME = "surrogate-train-studio"
MACHINE = Machine.L40S

def get_or_create_studio() -> Studio:
    ts = Teamspace()
    for s in ts.studios:
        if s.name == STUDIO_NAME:
            if s.status == "Running":
                print(f"Reusing running studio: {s.name}")
                return s
            else:
                print(f"Studio {s.name} is {s.status}. Restarting...")
                s.start(machine=MACHINE)
                return s
    print(f"Creating studio {STUDIO_NAME}...")
    return Studio.create(
        name=STUDIO_NAME,
        machine=MACHINE,
        project=LIGHTNING_PROJECT,
        create_ok=True
    )

def run_training():
    studio = get_or_create_studio()
    # Ensure studio is running before run
    if studio.status != "Running":
        studio.start(machine=MACHINE)
        # Wait for ready
        for _ in range(60):
            studio.refresh()
            if studio.status == "Running":
                break
            time.sleep(10)
        else:
            raise RuntimeError("Studio failed to start")

    # Run training command inside studio
    # Adjust command to your training entrypoint
    cmd = [
        "python", "train.py",
        "--manifest", "manifest.json",
        "--output", "train.parquet"
    ]
    run = studio.run(command=cmd, wait=False)
    print(f"Started training run {run.id}")
    return run

if __name__ == "__main__":
    run_training()
```
