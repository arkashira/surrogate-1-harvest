# workio / frontend

## 1. Diagnosis

- No optimistic UI for clock in/out — users wait on network round-trip and can double-tap causing duplicate requests/race punches.
- No local pending-state or error recovery; if LINE webhook succeeds but client times out, employee sees failure and may retry creating duplicate punches.
- Missing visual feedback for GPS/geofence validation before submitting punch (users learn only after round-trip).
- Punch timestamp shown from server only — no immediate local echo makes UI feel sluggish.
- No request deduplication key (idempotency) on client; retries can queue multiple identical mutations.

## 2. Proposed change

File: `workio/src/components/PunchButton.tsx` (or equivalent main clock-in/out component).  
Scope: add optimistic update, in-flight request dedupe, local timestamp echo, and inline validation states.

## 3. Implementation

```tsx
// workio/src/components/PunchButton.tsx
import { useState, useCallback, useRef } from 'react';
import { usePunch } from '../hooks/usePunch';
import { CheckCircle, Clock, AlertCircle } from 'lucide-react';

type PunchState = 'idle' | 'pending' | 'success' | 'error';

export function PunchButton() {
  const { punch, isLoading, error } = usePunch();
  const [state, setState] = useState<PunchState>('idle');
  const [localTime, setLocalTime] = useState<string | null>(null);
  const inFlightRef = useRef<Promise<unknown> | null>(null);

  const handlePunch = useCallback(async () => {
    // Dedupe: if there's an in-flight request, don't start another
    if (inFlightRef.current) return;

    // Optimistic: set local state immediately
    const now = new Date();
    const timeStr = now.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' });
    setLocalTime(timeStr);
    setState('pending');

    const payload = { timestamp: now.toISOString() };
    const req = punch(payload)
      .then(() => {
        setState('success');
        setTimeout(() => setState('idle'), 2500);
      })
      .catch((err) => {
        console.error(err);
        setState('error');
        setTimeout(() => setState('idle'), 3500);
      })
      .finally(() => {
        inFlightRef.current = null;
      });

    inFlightRef.current = req;
  }, [punch]);

  const isGeofenceOk = true; // replace with real geofence check if available

  return (
    <div className="flex flex-col items-center gap-3">
      <button
        onClick={handlePunch}
        disabled={isLoading || state === 'pending' || !isGeofenceOk}
        className={`
          flex items-center gap-2 px-6 py-3 rounded-xl font-semibold text-lg
          transition-all transform active:scale-95
          ${
            state === 'pending' || isLoading
              ? 'bg-amber-400 text-amber-900 cursor-wait'
              : state === 'success'
              ? 'bg-emerald-500 text-white'
              : state === 'error'
              ? 'bg-red-500 text-white'
              : 'bg-blue-600 text-white hover:bg-blue-700 shadow-lg'
          }
          ${(!isGeofenceOk || isLoading) && 'opacity-60 cursor-not-allowed'}
        `}
      >
        {state === 'pending' || isLoading ? (
          <>
            <Clock className="w-5 h-5 animate-spin" />
            กำลังบันทึก...
          </>
        ) : state === 'success' ? (
          <>
            <CheckCircle className="w-5 h-5" />
            บันทึกแล้ว {localTime}
          </>
        ) : state === 'error' ? (
          <>
            <AlertCircle className="w-5 h-5" />
            ลองใหม่
          </>
        ) : (
          <>
            <Clock className="w-5 h-5" />
            ลงเวลา
          </>
        )}
      </button>

      {localTime && state === 'idle' && (
        <span className="text-sm text-slate-500">ครั้งล่าสุด: {localTime}</span>
      )}

      {!isGeofenceOk && (
        <div className="flex items-center gap-1 text-amber-600 text-sm">
          <AlertCircle className="w-4 h-4" />
          อยู่นอกพื้นที่ที่กำหนด
        </div>
      )}

      {error && state !== 'error' && (
        <div className="text-red-600 text-sm">{String(error)}</div>
      )}
    </div>
  );
}
```

Hook adjustment (if not already supporting optimistic/mutation-level dedupe):

```ts
// workio/src/hooks/usePunch.ts
import { useMutation, useQueryClient } from '@tanstack/react-query';
import { punchAPI } from '../api/punch';

export function usePunch() {
  const queryClient = useQueryClient();

  return useMutation({
    mutationFn: punchAPI,
    // Optimistic update: immediately update local cache so UI reflects new punch
    onMutate: async (newPunch) => {
      await queryClient.cancelQueries({ queryKey: ['currentPunch'] });
      const previous = queryClient.getQueryData(['currentPunch']);
      queryClient.setQueryData(['currentPunch'], newPunch);
      return { previous };
    },
    onError: (err, _newPunch, context) => {
      if (context?.previous) {
        queryClient.setQueryData(['currentPunch'], context.previous);
      }
    },
    onSettled: () => {
      queryClient.invalidateQueries({ queryKey: ['currentPunch'] });
    },
  });
}
```
