### 1) Entrypoint: `/opt/axentx/airship/bin/airship-discover`
```bash
#!/usr/bin/env bash
# airship-discover — zero-config, cron-safe discovery + CDN training manifest
# Usage: airship-discover [YYYY-MM-DD] [HF_REPO]
#
# Cron-safe: flock, no TTY, idempotent, structured logs (syslog + file)
# Tags: #business-research #knowledge-rag #graph #training #cdn #api-strategy

set -euo pipefail
SHELL=/bin/bash
export SHELL

readonly SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
readonly PROJECT_ROOT="$(cd "${SCRIPT_DIR}/.." && pwd)"
readonly LIB_DIR="${PROJECT_ROOT}/lib"
readonly LOG_TAG="airship-discover"

# Paths
readonly VAR_DIR="${PROJECT_ROOT}/var/discover"
readonly LOCKFILE="/var/lock/airship-discover.lock"

# Logging
log_info()  { logger -t "${LOG_TAG}" -p user.info "$*"; echo "[INFO]  $(date -u +%Y-%m-%dT%H:%M:%SZ) $*"; }
log_error() { logger -t "${LOG_TAG}" -p user.err "$*";  echo "[ERROR] $(date -u +%Y-%m-%dT%H:%M:%SZ) $*" >&2; }

# Defaults
TARGET_DATE="${1:-$(date -u +%Y-%m-%d)}"
HF_REPO="${2:-huggingface/datasets}"
DATE_DIR="${VAR_DIR}/${TARGET_DATE}"
MANIFEST_FILE="${DATE_DIR}/manifest.json"
RESEARCH_FILE="${DATE_DIR}/research.md"

# Ensure directories
mkdir -p "${DATE_DIR}"

# Source core library
if [[ -f "${LIB_DIR}/airship-discover.sh" ]]; then
  source "${LIB_DIR}/airship-discover.sh"
else
  log_error "Core library not found: ${LIB_DIR}/airship-discover.sh"
  exit 1
fi

main() {
  # Cron-safe single instance + idempotency
  exec 200>"${LOCKFILE}"
  if ! flock -n 200; then
    log_info "Another instance is running (lock held). Exiting."
    exit 0
  fi

  # Idempotency: skip if manifest already exists and is non-empty
  if [[ -s "${MANIFEST_FILE}" ]]; then
    log_info "Manifest already exists for ${TARGET_DATE}. Skipping (idempotent)."
    exit 0
  fi

  log_info "Starting airship-discover date=${TARGET_DATE} repo=${HF_REPO}"

  # 1) Business research (tagged)
  run_business_research "${TARGET_DATE}" "${RESEARCH_FILE}"

  # 2) Knowledge RAG — top hub insight (MOC) (tagged)
  run_knowledge_rag "${TARGET_DATE}" "${DATE_DIR}/insight.md"

  # 3) Build CDN-only training manifest (single API call)
  build_training_manifest "${TARGET_DATE}" "${HF_REPO}" "${MANIFEST_FILE}"

  log_info "Completed airship-discover. Manifest: ${MANIFEST_FILE}"
}

main "$@"
```

#!/usr/bin/env bash
# airship-discover core library
# Tags: #business-research #knowledge-rag #graph #training #cdn #rate-limit-bypass

set -euo pipefail

# ---- Configuration ----
readonly HF_API="https://huggingface.co/api"
readonly HF_CDN="https://huggingface.co/datasets"

# ---- Helpers ----
prepend_tags() {
  local file="$1"
  local tags="$2"
  if [[ -f "${file}" ]] && ! grep -q "#${tags%% *}" "${file}" 2>/dev/null; then
    local tmpf
    tmpf="$(mktemp)"
    echo -e "# Tags: ${tags}\n" > "${tmpf}"
    cat "${file}" >> "${tmpf}"
    mv "${tmpf}" "${file}"
  fi
}

# ---- Business Research ----
run_business_research() {
  local target_date="$1"
  local research_file="$2"
  local research_script="${PROJECT_ROOT}/scripts/granite-business-research.sh"
  local out_dir="${PROJECT_ROOT}/outputs/business-research/${target_date}"

  mkdir -p "${out_dir}"

  if [[ -x "${research_script}" ]]; then
    log_info "Running granite-business-research.sh for ${target_date}"
    if "${research_script}" --date "${target_date}" --out "${out_dir}"; then
      # Combine key outputs into research_file and tag
      {
        echo "## Business Research (${target_date})"
        find "${out_dir}" -type f \( -name "*.json" -o -name "*.md" -o -name "*.txt" \) -print0 | \
          while IFS= read -r -d '' f; do
            echo ""
            echo "### $(basename "${f}")"
            cat "${f}"
          done
      } > "${research_file}.tmp"
      mv "${research_file}.tmp" "${research_file}"
      prepend_tags "${research_file}" "#business-research #knowledge-rag #graph"
      log_info "Business research completed. Output: ${research_file}"
    else
      log_error "Business research script failed (non-fatal)"
    fi
  else
    log_info "No executable granite-business-research.sh found, skipping"
  fi
}

# ---- Knowledge RAG (top hub) ----
run_knowledge_rag() {
  local target_date="$1"
  local insight_file="$2"

  # Try surrogate AI first (fast local), fallback to static MOC insight
  if curl -s -f -m 3 http://localhost:8001/health >/dev/null 2>&1; then
    log_info "Surrogate AI reachable — querying top-hub insight"
    if curl -s -m 10 -X POST http://localhost:8001/v1/rag/top-hub \
         -H "Content-Type: application/json" \
         -d "{\"date\":\"${target_date}\",\"hub_hint\":\"MOC\"}" > "${insight_file}.tmp" 2>/dev/null; then
      mv "${insight_file}.tmp" "${insight_file}"
    else
      fallback_top_hub_insight "${target_date}" > "${insight_file}"
    fi
  else
    log_info "Surrogate AI not reachable — using fallback top-hub insight"
    fallback_top_hub_insight "${target_date}" > "${insight_file}"
  fi

  prepend_tags "${insight_file}" "#knowledge-rag #graph #hub"
  log_info "Top hub insight written to ${insight_file}"
}

fallback_top_hub_insight() {
  local target_date="$1"
  cat <<EOF
## Top Hub: MOC (Mission Operations Center)

**Date**: ${target_date}  
**Tags**: #knowledge-rag #graph #hub

### Summary
The Mission Operations Center (MOC) is the most-connected hub across platform telemetry,
incident workflows, and AI-assisted decision loops.

### Recommended Pre-task Review
- Review active incidents and playbooks in MOC before planning tasks.
- Check latest knowledge-graph links for tool integrations (250+ tools).
- Validate runbooks against current environment registry.

### Actionable Notes
- Use Arkship API (port 8000) to list active runbooks and owners.
- Confirm on-call rotations and escalation paths for the date above.
- Attach relevant runbook IDs to tasks when filing handoffs.
EOF
}
