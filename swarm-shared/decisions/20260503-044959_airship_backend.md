## Implementation Plan: Resilient Surrogate-1 Training Pipeline (≤2h)

### Goal
Eliminate HF API 429s and Lightning idle-stop quota waste by:
1. Pre-listing file paths once (Mac orchestration) → JSON manifest
2. Training script uses CDN-only fetches (zero API calls during data load)
3. Lightning studio reuse + idle-aware restart
4. Deterministic write sharding across sibling repos for HF ingestion

### Scope (backend changes only)
- `surrogate/training/train.py` (or equivalent) — data loader + Lightning launcher
- `surrogate/orchestration/prepare_manifest.py` — Mac-side manifest generator
- `surrogate/ingestion/upload_shard.py` — deterministic sibling-repo uploader
- `surrogate/lightning/launcher.py` — studio reuse + idle-resilient runner

## 1. Manifest Generator (run on Mac before training window)

```python
# surrogate/orchestration/prepare_manifest.py
import os
import json
from datetime import datetime, timedelta
from huggingface_hub import HfApi

HF_REPO = "datasets/axentx/surrogate-mirror-merged"
OUTPUT_MANIFEST = "surrogate/training/file_manifest.json"

def build_manifest(target_date: str):
    """
    Single API call: list_repo_tree for one date folder (non-recursive).
    Returns CDN URLs for all parquet files.
    """
    api = HfApi()
    prefix = f"{target_date}/"
    entries = api.list_repo_tree(
        repo_id=HF_REPO,
        path=prefix,
        repo_type="dataset",
        recursive=False
    )

    files = []
    for entry in entries:
        if entry.path.endswith(".parquet"):
            cdn_url = (
                f"https://huggingface.co/datasets/{HF_REPO}/resolve/main/"
                f"{entry.path}"
            )
            files.append({
                "path": entry.path,
                "cdn_url": cdn_url,
                "size": getattr(entry, "size", None)
            })

    manifest = {
        "generated_at": datetime.utcnow().isoformat(),
        "target_date": target_date,
        "repo": HF_REPO,
        "files": files,
        "total_files": len(files)
    }

    os.makedirs(os.path.dirname(OUTPUT_MANIFEST), exist_ok=True)
    with open(OUTPUT_MANIFEST, "w") as f:
        json.dump(manifest, f, indent=2)

    print(f"Manifest written: {OUTPUT_MANIFEST} ({len(files)} files)")
    return manifest

if __name__ == "__main__":
    # yesterday by default; override via env or CLI
    target = os.getenv("TARGET_DATE", (datetime.utcnow() - timedelta(days=1)).strftime("%Y-%m-%d"))
    build_manifest(target)
```

## 2. CDN-Only Data Module (training script)

```python
# surrogate/training/data_module.py
import json
import pyarrow.parquet as pq
import torch
from torch.utils.data import Dataset, DataLoader
from pathlib import Path
import requests
from tempfile import NamedTemporaryFile

MANIFEST_PATH = Path(__file__).parent / "file_manifest.json"

class CDNParquetDataset(Dataset):
    def __init__(self, manifest_path=MANIFEST_PATH, columns=("prompt", "response")):
        with open(manifest_path) as f:
            manifest = json.load(f)
        self.files = [f["cdn_url"] for f in manifest["files"]]
        self.columns = columns

    def __len__(self):
        return len(self.files)

    def __getitem__(self, idx):
        url = self.files[idx]
        # Stream parquet via CDN (no auth, no API calls)
        resp = requests.get(url, timeout=30)
        resp.raise_for_status()
        with NamedTemporaryFile(suffix=".parquet", delete=False) as tmp:
            tmp.write(resp.content)
            tmp_path = tmp.name

        try:
            table = pq.read_table(tmp_path, columns=self.columns)
            # Convert to list of {prompt, response}
            prompts = table.column("prompt").to_pylist()
            responses = table.column("response").to_pylist()
            # Return first row for this file (or extend to yield all rows)
            return {"prompt": prompts[0], "response": responses[0]}
        finally:
            Path(tmp_path).unlink(missing_ok=True)

def build_dataloader(manifest_path=MANIFEST_PATH, batch_size=4, num_workers=2):
    dataset = CDNParquetDataset(manifest_path)
    return DataLoader(dataset, batch_size=batch_size, num_workers=num_workers, shuffle=True)
```

## 3. Lightning Studio Resilient Launcher

```python
# surrogate/lightning/launcher.py
import time
from lightning import LightningWork, LightningFlow, LightningApp, Machine
from lightning.app.storage import Drive

class TrainingWork(LightningWork):
    def __init__(self, machine: Machine = None, **kwargs):
        super().__init__(machine=machine or Machine("lightning-lambda-prod", disk_size=200, shm_size=0), **kwargs)
        self.trained = False

    def run(self, manifest_path: str):
        # Import here to avoid Mac-side heavy deps
        from surrogate.training.train import train  # expects manifest_path
        train(manifest_path=manifest_path)
        self.trained = True

class ResilientFlow(LightningFlow):
    def __init__(self):
        super().__init__()
        self.work = TrainingWork()
        self.last_check = 0

    def run(self):
        manifest = "surrogate/training/file_manifest.json"

        # Reuse running studio; restart if idle-stopped
        if not self.work.has_started:
            self.work.start()
            return

        if self.work.status == "stopped":
            print("Studio stopped (idle timeout). Restarting...")
            self.work.stop()
            self.work = TrainingWork()
            self.work.start()
            return

        if self.work.status == "running" and not self.work.trained:
            # Trigger training once
            self.work.run(manifest_path=manifest)

        # Periodic health check (every 60s)
        now = time.time()
        if now - self.last_check > 60:
            self.last_check = now
            print(f"Studio status: {self.work.status}")

app = LightningApp(ResilientFlow())
```

## 4. Deterministic HF Ingestion Sharding (for writes)

```python
# surrogate/ingestion/upload_shard.py
import hashlib
from pathlib import Path
from huggingface_hub import HfApi

HF_REPO_PREFIX = "datasets/axentx/surrogate-enriched"
SIBLINGS = 5  # 5 sibling repos: surrogate-enriched-0 .. surrogate-enriched-4

def pick_repo(slug: str) -> str:
    """Deterministic repo assignment by hash."""
    digest = int(hashlib.sha256(slug.encode()).hexdigest(), 16)
    idx = digest % SIBLINGS
    return f"{HF_REPO_PREFIX}-{idx}"

def upload_shard(parquet_path: Path, slug: str, date: str):
    repo = pick_repo(slug)
    api = HfApi()
    target_path = f"batches/mirror-merged/{date}/{slug}.parquet"
    api.upload_file(
        path_or_fileobj=str(parquet_path),
        path_in_repo=target_path,
        repo_id=repo,
        repo_type="dataset",
        commit_message=f"Add {slug} ({date})"
    )
    print(f"Uploaded to {repo}/{target_path}")
    return repo, target_path
```
