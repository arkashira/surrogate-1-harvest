## Final Unified Implementation (Best Parts + Resolved Contradictions)

**Chosen approach**: Manifest-driven, CDN-only dataset loader with strict schema projection and zero HF API calls during training.  
**Why this wins**: Fixes `pyarrow.CastError` on mixed schemas, bypasses HF 429/128-commit caps, and works within “Mac orchestration / remote compute” constraints. Contradictions resolved in favor of correctness and concrete actionability:

- **Manifest generation**: Use `list_repo_tree` (not `list_files`) to avoid pagination issues and ensure reliable file enumeration.
- **Schema projection**: Require explicit column names and validate at parse time; fail fast if columns missing.
- **CDN downloads**: Use `resolve/main/...` URLs with `requests` (no auth header) to bypass rate limits.
- **Lightning integration**: Provide a proper `IterableDataset` + `DataLoader` that works with Lightning’s training loop and avoids schema surprises.
- **Execution order**: Manifest built on Mac first; training on Studio never calls HF API.

### 1) Orchestration script (run on Mac)

`scripts/build_file_list.py`

```python
#!/usr/bin/env python3
"""
Generate file_list.json for a given date folder in a HF dataset repo.
Run on Mac (or any orchestration host) before training.

Usage:
  HF_TOKEN=<token> python scripts/build_file_list.py \
    --repo oxbot/surrogate-mirror \
    --date 2026-04-29 \
    --out surrogate/training/file_list.json
"""

import argparse
import json
import os
import sys

from huggingface_hub import HfApi

def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g. oxbot/surrogate-mirror)")
    parser.add_argument("--date", required=True, help="Date folder (e.g. 2026-04-29)")
    parser.add_argument("--out", required=True, help="Output JSON path")
    parser.add_argument("--prefix", default="batches/mirror-merged", help="Base folder inside repo")
    args = parser.parse_args()

    api = HfApi(token=os.getenv("HF_TOKEN"))
    folder = f"{args.prefix}/{args.date}"
    print(f"Listing {folder} in {args.repo} ...")

    # Use list_repo_tree (non-recursive) to avoid pagination surprises
    items = api.list_repo_tree(repo_id=args.repo, path=folder, repo_type="dataset")
    files = sorted(
        it.rfilename for it in items
        if it.type == "file" and it.rfilename.endswith(".parquet")
    )

    if not files:
        print("No parquet files found.")
        sys.exit(1)

    manifest = {
        "repo": args.repo,
        "date": args.date,
        "folder": folder,
        "files": files,
        "cdn_prefix": f"https://huggingface.co/datasets/{args.repo}/resolve/main",
    }

    os.makedirs(os.path.dirname(os.path.abspath(args.out)), exist_ok=True)
    with open(args.out, "w") as f:
        json.dump(manifest, f, indent=2)

    print(f"Wrote {len(files)} files to {args.out}")

if __name__ == "__main__":
    main()
```

### 2) CDN-only dataset loader (Lightning training)

`surrogate/training/cdn_dataset.py`

```python
"""
CDN-only dataset loader for Surrogate-1 training.
- Uses file_list.json generated by build_file_list.py
- Downloads via HF CDN (no Authorization header; bypasses API rate limits)
- Projects each parquet file to {prompt, response} at parse time
- Never uses load_dataset on heterogeneous repos (avoids schema/cast errors)
"""

import json
from pathlib import Path
from typing import Dict, Iterator, List, Tuple

import pyarrow as pa
import pyarrow.parquet as pq
import requests
from torch.utils.data import IterableDataset

class CDNParquetDataset(IterableDataset):
    """
    IterableDataset that yields {prompt, response} rows from parquet files
    listed in manifest, fetched via HF CDN.
    """

    def __init__(
        self,
        manifest_path: str | Path,
        columns: Tuple[str, str] = ("prompt", "response"),
    ):
        manifest_path = Path(manifest_path)
        if not manifest_path.is_file():
            raise FileNotFoundError(f"Manifest not found: {manifest_path}")

        with manifest_path.open() as f:
            self.manifest = json.load(f)

        self.cdn_prefix = self.manifest["cdn_prefix"].rstrip("/")
        self.files: List[str] = self.manifest["files"]
        self.prompt_col, self.response_col = columns

    def _stream_file(self, rel_path: str) -> Iterator[Dict[str, str]]:
        url = f"{self.cdn_prefix}/{rel_path}"
        # CDN downloads bypass HF API rate limits
        resp = requests.get(url, timeout=120)
        resp.raise_for_status()

        table = pq.read_table(pa.BufferReader(resp.content))
        df = table.to_pandas()

        # Validate required columns; fail fast on schema mismatch
        missing = {self.prompt_col, self.response_col} - set(df.columns)
        if missing:
            raise KeyError(
                f"Required columns missing in {rel_path}: {list(df.columns)} "
                f"(missing {sorted(missing)})"
            )

        for _, row in df.iterrows():
            yield {
                "prompt": str(row[self.prompt_col]),
                "response": str(row[self.response_col]),
            }

    def __iter__(self) -> Iterator[Dict[str, str]]:
        for rel_path in self.files:
            yield from self._stream_file(rel_path)
```
