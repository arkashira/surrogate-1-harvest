# workio / frontend

## 1. Diagnosis

- Clock-in/out UX has no optimistic UI or local pending state → users perceive latency and may double-tap, creating race conditions that manifest as duplicate events before idempotency layer settles.
- No visual feedback when LINE webhook is in-flight or failed; errors surface only via generic toast or silently, leaving employees unsure whether clock-in succeeded.
- GPS verification prompt is blocking and un-cancellable on poor signal; users cannot proceed or retry gracefully, leading to support tickets and manual overrides.
- Tailwind class drift and missing dark-mode tokens in clock-in/out card break consistency with design system (colors, spacing, radius) and hurt perceived quality.
- No client-side debounce or disabled state on primary clock action button → rapid double-taps queue multiple requests and can bypass optimistic locking.

## 2. Proposed change

File: `src/features/clock/ClockCard.tsx` (create if absent) or `src/pages/Dashboard.tsx` — whichever houses the clock-in/out UI.  
Scope: add optimistic state, pending/disabled controls, inline error recovery, and consistent Tailwind tokens.

## 3. Implementation

```tsx
// src/features/clock/ClockCard.tsx
import { useState, useCallback } from 'react';
import { useMutation, useQueryClient } from '@tanstack/react-query';
import { CheckCircle, Clock, AlertCircle, MapPin } from 'lucide-react';

const CLOCK_ENDPOINT = '/api/clock';

export function ClockCard({ tenantId, userId }: { tenantId: string; userId: string }) {
  const queryClient = useQueryClient();
  const [lastError, setLastError] = useState<string | null>(null);

  const clockMutation = useMutation({
    mutationFn: async (type: 'IN' | 'OUT') => {
      const res = await fetch(CLOCK_ENDPOINT, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ tenantId, userId, type }),
      });
      if (!res.ok) {
        const text = await res.text().catch(() => res.statusText);
        throw new Error(text || `Clock ${type} failed`);
      }
      return res.json();
    },
    onMutate: async (type) => {
      setLastError(null);
      // Optimistic update for local cache (if you keep last clock in cache)
      queryClient.setQueryData(['last-clock', userId], {
        type,
        ts: new Date().toISOString(),
        pending: true,
      });
    },
    onSuccess: (data) => {
      queryClient.setQueryData(['last-clock', userId], { ...data, pending: false });
      queryClient.invalidateQueries({ queryKey: ['today-events'] });
    },
    onError: (err: Error) => {
      setLastError(err.message || 'Unknown error');
      // Rollback optimistic state
      queryClient.setQueryData(['last-clock', userId], (old: any) => ({
        ...old,
        pending: false,
      }));
    },
  });

  const handleClock = useCallback(
    (type: 'IN' | 'OUT') => {
      if (clockMutation.isPending) return;
      clockMutation.mutate(type);
    },
    [clockMutation.isPending]
  );

  const last = queryClient.getQueryData(['last-clock', userId]) as
    | { type: 'IN' | 'OUT'; ts: string; pending?: boolean }
    | undefined;

  return (
    <div className="w-full max-w-sm rounded-2xl border border-gray-200 bg-white p-6 shadow-sm dark:border-gray-700 dark:bg-gray-800">
      <div className="mb-4 flex items-center justify-between">
        <h2 className="text-lg font-semibold text-gray-900 dark:text-gray-100">ลงเวลาทำงาน</h2>
        {last?.pending && <Clock className="h-5 w-5 animate-spin text-blue-500" />}
      </div>

      <div className="mb-4 flex gap-3">
        <button
          onClick={() => handleClock('IN')}
          disabled={clockMutation.isPending || last?.type === 'IN'}
          className={`flex-1 rounded-xl px-4 py-3 font-medium transition-all focus:outline-none focus:ring-2 focus:ring-blue-500 focus:ring-offset-2 disabled:opacity-60 disabled:cursor-not-allowed ${
            last?.type === 'IN' && !last?.pending
              ? 'bg-green-100 text-green-700 dark:bg-green-900/40 dark:text-green-300'
              : 'bg-blue-600 text-white hover:bg-blue-700 dark:bg-blue-500 dark:hover:bg-blue-600'
          }`}
        >
          <span className="flex items-center justify-center gap-2">
            {last?.type === 'IN' && !last?.pending && <CheckCircle className="h-4 w-4" />}
            ลงเวลาเข้า
          </span>
        </button>

        <button
          onClick={() => handleClock('OUT')}
          disabled={clockMutation.isPending || last?.type === 'OUT'}
          className={`flex-1 rounded-xl px-4 py-3 font-medium transition-all focus:outline-none focus:ring-2 focus:ring-red-500 focus:ring-offset-2 disabled:opacity-60 disabled:cursor-not-allowed ${
            last?.type === 'OUT' && !last?.pending
              ? 'bg-red-100 text-red-700 dark:bg-red-900/40 dark:text-red-300'
              : 'bg-gray-600 text-white hover:bg-gray-700 dark:bg-gray-500 dark:hover:bg-gray-600'
          }`}
        >
          <span className="flex items-center justify-center gap-2">
            {last?.type === 'OUT' && !last?.pending && <CheckCircle className="h-4 w-4" />}
            ลงเวลาออก
          </span>
        </button>
      </div>

      {lastError && (
        <div className="mb-3 flex items-start gap-2 rounded-lg bg-red-50 px-3 py-2 text-sm text-red-700 dark:bg-red-900/30 dark:text-red-300">
          <AlertCircle className="mt-0.5 h-4 w-4 flex-none" />
          <div className="flex-1">
            <p className="font-medium">เกิดข้อผิดพลาด</p>
            <p className="text-xs opacity-90">{lastError}</p>
            <button
              onClick={() => window.location.reload()}
              className="mt-1 text-xs underline underline-offset-2 hover:no-underline"
            >
              รีเฟรช
            </button>
          </div>
        </div>
      )}

      <div className="flex items-center gap-2 rounded-lg border border-dashed border-gray-200 px-3 py-2 text-xs text-gray-500 dark:border-gray-600 dark:text-gray-400">
        <MapPin className="h-4 w-4" />
        <span>ระบบจะตรวจสอบพิกัด GPS อัตโนมัติ</span>
      </div>

      {last && (
        <p className="mt-3 text-center text-xs text-gray-400">
          ครั้งล่าสุด: {last.type === 'IN' ? 'เข้า' : 'ออก'} •{' '}
          {last.ts ? new Date(last.ts).toLocaleTimeString() : '-'}
        </p>
      )}
    </div>
  );
}
```
