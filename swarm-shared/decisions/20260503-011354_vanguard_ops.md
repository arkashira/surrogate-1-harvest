# vanguard / ops

## 1. Diagnosis

- No persisted `(repo, dateFolder)` manifest exists → every backend request and training job re-enumerates HF API via `list_repo_tree`, burning the 1000/5 min quota and risking 429s.
- Backend uses authenticated `/api/` paths instead of public CDN → avoidable auth overhead and rate-limit exposure for file fetches that could be anonymous.
- Training scripts likely call `load_dataset(streaming=True)` on heterogeneous repos → pyarrow `CastError` on mixed-schema files (per surrogate-1 lessons).
- No reuse guard for Lightning Studio → each run may create new studio instead of reusing running ones, wasting 80hr/mo quota.
- Mac-side orchestration may invoke model training locally (`.from_pretrained`) instead of delegating to Lightning/Kaggle → violates Mac=CLI rule and risks resource contention.

## 2. Proposed change

Create `/opt/axentx/vanguard/ops/manifest.py` + update `/opt/axentx/vanguard/ops/train.py` (or create if absent) to:

- Add `build_manifest(repo, date_folder)` that calls `list_repo_tree` **once**, saves `manifest/{repo}/{date_folder}.json` with public CDN URLs.
- Add `load_manifest(repo, date_folder)` for cached reads; fallback to build if stale (>24h).
- Replace any `load_dataset` usage with per-file `hf_hub_download` + projection to `{prompt, response}` only.
- Add `get_or_reuse_studio(name, machine)` that reuses running studios before creating.
- Ensure training launcher is invoked via `bash` and sets `SHELL=/bin/bash`.

```python
#!/usr/bin/env python3
"""
Persist (repo, dateFolder) manifests to avoid repeated HF API list calls.
Uses public CDN URLs to bypass auth/rate limits during training.
"""
from __future__ import annotations

import json
import os
import time
from datetime import datetime, timedelta, timezone
from pathlib import Path
from typing import Any, Dict, List

from huggingface_hub import list_repo_tree

MANIFEST_ROOT = Path(__file__).parent.parent / "manifests"
MANIFEST_ROOT.mkdir(exist_ok=True, parents=True)

CDN_BASE = "https://huggingface.co/datasets"


def _manifest_path(repo: str, date_folder: str) -> Path:
    safe_repo = repo.replace("/", "_")
    return MANIFEST_ROOT / f"{safe_repo}_{date_folder}.json"


def _is_fresh(p: Path, max_age_seconds: int = 86400) -> bool:
    if not p.exists():
        return False
    mtime = p.stat().st_mtime
    return (time.time() - mtime) < max_age_seconds


def build_manifest(repo: str, date_folder: str) -> List[Dict[str, str]]:
    """
    Single API call to list top-level folder for date_folder.
    Returns list of CDN URLs and relative paths.
    """
    tree = list_repo_tree(repo=repo, path=date_folder, recursive=False)
    entries: List[Dict[str, str]] = []
    for item in tree:
        if item.type != "file":
            continue
        cdn_url = f"{CDN_BASE}/{repo}/resolve/main/{item.path}"
        entries.append(
            {
                "path": item.path,
                "cdn_url": cdn_url,
                "size": item.size or 0,
            }
        )

    manifest = {
        "repo": repo,
        "date_folder": date_folder,
        "generated_at": datetime.now(timezone.utc).isoformat(),
        "files": entries,
    }

    out = _manifest_path(repo, date_folder)
    out.parent.mkdir(parents=True, exist_ok=True)
    out.write_text(json.dumps(manifest, indent=2))
    return entries


def load_manifest(repo: str, date_folder: str, max_age_seconds: int = 86400) -> List[Dict[str, str]]:
    p = _manifest_path(repo, date_folder)
    if _is_fresh(p, max_age_seconds=max_age_seconds):
        data = json.loads(p.read_text())
        return data.get("files", [])
    return build_manifest(repo, date_folder)
```

```python
#!/usr/bin/env python3
"""
Lightning Studio reuse helpers to avoid quota waste.
"""
from lightning import Studio, L40S, LightningMachine
from lightning.app import Teamspace


def get_or_reuse_studio(name: str, machine: LightningMachine = L40S) -> Studio:
    for s in Teamspace.studios:
        if s.name == name and s.status == "running":
            return s
    return Studio(name=name, machine=machine, create_ok=True)
```
