### 1) Mac-side file-list generator (run once per date folder)
```bash
# scripts/list_hf_date_folder.py
#!/usr/bin/env python3
import argparse
import json
from huggingface_hub import HfApi

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True, help="HF repo id")
    parser.add_argument("--date-folder", required=True, help="Path to date folder")
    parser.add_argument("--out", required=True, help="Output JSON path")
    parser.add_argument("--recursive", action="store_true", help="Recurse subfolders")
    args = parser.parse_args()

    api = HfApi()
    tree = api.list_repo_tree(repo=args.repo, path=args.date_folder, recursive=args.recursive)
    files = [item.path for item in tree if item.path.endswith(".parquet")]

    with open(args.out, "w") as f:
        json.dump(files, f, indent=2)
    print(f"Saved {len(files)} files to {args.out}")

if __name__ == "__main__":
    main()
```

**Usage:**
```bash
python scripts/list_hf_date_folder.py \
  --repo "axentx/surrogate-1-ingest" \
  --date-folder "batches/mirror-merged/2026-05-03" \
  --out "data/file_list.json" \
  --recursive
```

### 2) CDN-only dataset loader (with retry/backoff)
```python
# surrogate/data/cdn_loader.py
import json
import time
import pyarrow.parquet as pq
from pathlib import Path
from typing import List, Dict
from huggingface_hub import hf_hub_download
import tenacity

class CDNParquetLoader:
    def __init__(self, repo: str, file_list_path: str, cache_dir: str = "./hf_cache"):
        self.repo = repo
        self.cache_dir = Path(cache_dir)
        self.cache_dir.mkdir(parents=True, exist_ok=True)
        with open(file_list_path) as f:
            self.file_paths: List[str] = json.load(f)

    @tenacity.retry(
        stop=tenacity.stop_after_attempt(5),
        wait=tenacity.wait_exponential(multiplier=1, min=2, max=30),
        retry=tenacity.retry_if_exception_type((OSError, ConnectionError)),
        reraise=True,
    )
    def _download_cdn(self, file_path: str) -> Path:
        """Download via CDN (does NOT count against HF API rate limits)."""
        return Path(hf_hub_download(
            repo_id=self.repo,
            filename=file_path,
            cache_dir=str(self.cache_dir),
            force_download=False,
            resume_download=True,
        ))

    def load_and_project(self, limit: int = None) -> List[Dict[str, str]]:
        """Load parquet files and project to {prompt, response} with slug attribution."""
        records = []
        paths = self.file_paths[:limit] if limit else self.file_paths
        
        for i, file_path in enumerate(paths):
            try:
                local_path = self._download_cdn(file_path)
                table = pq.read_table(local_path, columns=["prompt", "response"])
                batch = table.to_pylist()
                
                # Attribution via filename pattern: batches/mirror-merged/{date}/{slug}.parquet
                slug = Path(file_path).stem
                for row in batch:
                    row["_slug"] = slug
                    row["_source_file"] = file_path
                records.extend(batch)
                
                if (i + 1) % 10 == 0:
                    print(f"  Loaded {i+1}/{len(paths)} files ({len(records)} records)")
                    
            except Exception as e:
                print(f"⚠️  Skipping {file_path}: {e}")
                continue
                
        return records
```

### 3) Lightning Studio lifecycle wrapper (corrected)
```python
# surrogate/train/lightning_launcher.py
import os
from lightning import LightningWork, LightningFlow, LightningApp, Machine
from lightning.app import Teamspace

class SurrogateTrainer(LightningWork):
    def __init__(self, file_list_path: str, **kwargs):
        super().__init__(**kwargs)
        self.file_list_path = file_list_path
        self.cloud_compute = kwargs.get("cloud_compute", Machine.L40S)

    def run(self):
        from surrogate.train.trainer import train  # your training entrypoint
        train(file_list_path=self.file_list_path)

class SurrogateFlow(LightningFlow):
    def __init__(self):
        super().__init__()
        self.trainer = None

    def run(self):
        teamspace = Teamspace()
        studio_name = "surrogate-training-studio"

        # Check for existing running studio
        existing = next(
            (s for s in teamspace.studios if s.name == studio_name and s.status == "running"),
            None
        )

        if existing:
            print(f"✅ Reusing running studio: {existing.name}")
            machine = existing.cloud_compute
        else:
            print("🚀 Creating new studio (L40S preferred)")
            machine = Machine.L40S

        # Initialize or reuse trainer
        if not self.trainer:
            self.trainer = SurrogateTrainer(
                file_list_path="data/file_list.json",
                cloud_compute=machine,
            )

        # Start if not running
        if not self.trainer.is_running:
            self.trainer.start()

app = LightningApp(SurrogateFlow())
```

### 4) Commit-cap mitigation (sibling repos)
```python
# surrogate/ingest/sibling_router.py
import hashlib

SIBLINGS = [
    "axentx/surrogate-mirror-0",
    "axentx/surrogate-mirror-1",
    "axentx/surrogate-mirror-2",
    "axentx/surrogate-mirror-3",
    "axentx/surrogate-mirror-4",
]

def repo_for_slug(slug: str) -> str:
    """Deterministically route slug to sibling repo for commit-cap distribution."""
    digest = int(hashlib.sha256(slug.encode()).hexdigest(), 16)
    return SIBLINGS[digest % len(SIBLINGS)]
```
