### 1) Build-Time: Fetch Once, Store in CDN Artifact (Cross-Platform)

Use a Node.js script for cross-platform compatibility (works on Mac/Linux CI runners) that falls back to a deterministic static path if API limits are hit.

**File**: `scripts/build-top-hub.js`

```js
// scripts/build-top-hub.js
// Cross-platform build script. Run during CI or pre-build.
// Uses HF API minimally (1 list call), then CDN for file fetch.
import { HfApi } from "@huggingface/hub";
import fs from "fs/promises";
import path from "path";

const REPO = "axentx/knowledge-rag";
const OUT_DIR = "public/data";
const OUT_FILE = path.join(OUT_DIR, "top-hub.json");

async function main() {
  const api = new HfApi();
  await fs.mkdir(OUT_DIR, { recursive: true });

  // 1) List root to find latest date folder (batches/YYYY-MM-DD)
  let entries;
  try {
    entries = await api.listRepoTree(REPO, "main", { recursive: false });
  } catch (err) {
    console.warn("HF API unavailable, using static fallback:", err.message);
    await writeFallback();
    return;
  }

  const batchFolders = entries
    .filter((e) => e.type === "directory" && e.path.startsWith("batches/"))
    .map((e) => e.path)
    .sort()
    .reverse();

  const folder = batchFolders[0];
  if (!folder) {
    console.warn("No batches folder found, using fallback.");
    await writeFallback();
    return;
  }

  // 2) List files in that folder (non-recursive)
  let files;
  try {
    files = await api.listRepoTree(REPO, "main", { path: folder, recursive: false });
  } catch (err) {
    console.warn("Failed to list folder, using fallback:", err.message);
    await writeFallback();
    return;
  }

  const jsonFiles = files.filter((f) => f.type === "file" && f.path.endsWith(".json"));
  const target = jsonFiles[0];
  if (!target) {
    await writeFallback();
    return;
  }

  // 3) Fetch via CDN (no auth, no rate limit)
  const cdnUrl = `https://huggingface.co/datasets/${REPO}/resolve/main/${target.path}`;
  let hubRaw;
  try {
    const res = await fetch(cdnUrl);
    if (!res.ok) throw new Error(`CDN fetch failed: ${res.status}`);
    hubRaw = await res.json();
  } catch (err) {
    console.warn("CDN fetch failed, using fallback:", err.message);
    await writeFallback();
    return;
  }

  // 4) Normalize payload
  const slug = path.basename(target.path).replace(/^hub-/, "").replace(/\.json$/, "");
  const title = hubRaw.title || hubRaw.hub || hubRaw.name || slug;
  const summary = hubRaw.summary || hubRaw.insight || hubRaw.description || "";
  const connections = hubRaw.connections || hubRaw.degree || 0;

  const payload = {
    slug,
    title,
    summary,
    connections,
    updated: new Date().toISOString(),
    source: "knowledge-rag",
    type: "top-hub",
  };

  await fs.writeFile(OUT_FILE, JSON.stringify(payload, null, 2));
  console.log(`Top-hub baked to ${OUT_FILE}`);

  async function writeFallback() {
    const payload = {
      slug: "MOC",
      title: "Mission Operations Center",
      summary: "Primary coordination hub for mission-critical operations.",
      connections: 1,
      updated: new Date().toISOString(),
      source: "knowledge-rag",
      type: "top-hub",
      fallback: true,
    };
    await fs.writeFile(OUT_FILE, JSON.stringify(payload, null, 2));
  }
}

main().catch((err) => {
  console.error("build-top-hub failed:", err);
  process.exit(1);
});
```

**Usage** (add to package.json scripts):
```json
"scripts": {
  "build:top-hub": "node scripts/build-top-hub.js"
}
```

Run during CI or pre-build. Output: `public/data/top-hub.json`.

### 2) Frontend: Non-Blocking Top-Hub Signal Panel (React)

Fails silently and does not block rendering.

**File**: `src/components/TopHubSignalPanel.jsx`

```jsx
import React, { useEffect, useState } from "react";
import "./TopHubSignalPanel.css";

export default function TopHubSignalPanel() {
  const [hub, setHub] = useState(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    const controller = new AbortController();
    const timeout = setTimeout(() => controller.abort(), 4000);

    fetch("/data/top-hub.json", { signal: controller.signal, cache: "no-cache" })
      .then((res) => {
        if (!res.ok) throw new Error("No top-hub");
        return res.json();
      })
      .then((data) => {
        setHub(data);
        setLoading(false);
      })
      .catch(() => {
        setLoading(false);
      });

    return () => clearTimeout(timeout);
  }, []);

  if (loading) return null;
  if (!hub) return null;

  return (
    <div className="top-hub-panel" role="complementary" aria-label="Top hub insight">
      <div className="top-hub-header">
        <span className="top-hub-badge">Top Hub</span>
        <span className="top-hub-slug">{hub.slug.toUpperCase()}</span>
      </div>
      <div className="top-hub-body">
        <h4 className="top-hub-title">{hub.title}</h4>
        {hub.summary && <p className="top-hub-summary">{hub.summary}</p>}
        <div className="top-hub-meta">
          <span className="top-hub-metric">{hub.connections} connections</span>
          <span className="top-hub-updated">Updated {new Date(hub.updated).toLocaleDateString()}</span>
        </div>
      </div>
    </div>
  );
}
```
