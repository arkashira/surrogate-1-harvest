- On idle stop, restart with `target.start(machine=Machine.L40S)`.
- Avoids quota waste from repeated studio creation.

### `scripts/generate_file_list.py` (run from Mac)
```python
#!/usr/bin/env python3
"""
Generate file_list.json for a given date folder.
Run from Mac when HF API rate-limit window is clear.
"""
import argparse
import json
from pathlib import Path
from huggingface_hub import HfApi

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", default="axentx/surrogate-dataset")
    parser.add_argument("--date", required=True, help="Date folder (e.g., 2026-04-29)")
    parser.add_argument("--output", default="file_list.json")
    args = parser.parse_args()

    api = HfApi()
    tree = api.list_repo_tree(
        repo_id=args.repo,
        path=args.date,
        recursive=False  # avoid pagination explosion
    )

    files = [
        f"{args.date}/{entry.path.split('/')[-1]}"
        for entry in tree
        if entry.path.lower().endswith(".parquet")
    ]

    out = {
        "repo": args.repo,
        "date": args.date,
        "files": sorted(files)  # deterministic order
    }

    Path(args.output).write_text(json.dumps(out, indent=2))
    print(f"Wrote {len(files)} files to {args.output}")

if __name__ == "__main__":
    main()
```

### `surrogate/lightning_studio.py` (reuse + idle handling)
```python
#!/usr/bin/env python3
from lightning import Studio, Teamspace, Machine

STUDIO_NAME = "surrogate-train-l40s"

def get_or_create_studio():
    # Reuse existing running studio
    for s in Teamspace.studios:
        if s.name == STUDIO_NAME and s.status == "Running":
            print(f"Reusing running studio: {STUDIO_NAME}")
            return s

    # Create new if not running
    print(f"Creating studio: {STUDIO_NAME}")
    return Studio(
        name=STUDIO_NAME,
        machine=Machine.L40S,
        create_ok=True,
    )

def run_training():
    studio = get_or_create_studio()

    # Handle idle stop (Lightning kills training on stop)
    if studio.status != "Running":
        print("Studio stopped, restarting...")
        studio.start(machine=Machine.L40S)

    # Execute training script inside studio
    studio.run(
        command="python train.py",
        sync=False,
    )


if __name__ == "__main__":
    run_training()
```
