# workio / frontend

## Final synthesized implementation

**Diagnosis (merged, prioritized correctness + actionability)**
- No optimistic UI for clock in/out → perceived latency and double-taps on flaky networks.
- No local-first draft for leave/OT → unsaved form loss on navigation/offline.
- Clock state can drift between UI and server; no fast revalidation on focus/foreground.
- No clear loading/disabled affordance on punch button.
- No retry/backoff UX for failed punches; silent failures lead to manual admin fixes.

**Chosen approach**
- Keep optimistic UI for immediate feedback, but **do not let optimistic state replace source-of-truth**; always reconcile with server on success and on focus/visibility/online.
- Add localStorage draft for leave/OT with debounced save and clear-on-submit.
- Add explicit loading state + disabled button to prevent double-tap.
- Add retry with exponential backoff and user-facing toast/action; auto-retry on reconnect.
- Revalidate clock status on focus, visibility change, and online events to correct drift.
- Keep changes contained to attendance/leave-ot features; no schema/API changes.

### `workio/src/features/attendance/useClockActions.ts`
```ts
import { useState, useCallback, useEffect, useRef } from 'react';
import useSWR from 'swr';

const ENDPOINT = '/api/attendance/clock';

type ClockState = 'in' | 'out' | null;

export function useClockActions(employeeId: string) {
  const [optimistic, setOptimistic] = useState<ClockState>(null);
  const [retryCount, setRetryCount] = useState(0);
  const retryQueueRef = useRef<{ type: 'in' | 'out'; payload: any } | null>(null);

  const { data, error, mutate } = useSWR<{ status: ClockState; ts: string }>(
    employeeId ? `${ENDPOINT}/status?employeeId=${employeeId}` : null,
    (url) => fetch(url).then((r) => r.json()),
    { revalidateOnFocus: true, dedupingInterval: 2000, revalidateOnReconnect: true }
  );

  // Reconcile optimistic vs server: optimistic only while pending
  const current = optimistic ?? data?.status ?? null;

  const clock = useCallback(
    async (type: 'in' | 'out') => {
      const payload = { employeeId, type };
      const prevOptimistic = optimistic;
      const prevRetryCount = retryCount;

      // Optimistic UI
      setOptimistic(type);
      setRetryCount(0);
      retryQueueRef.current = { type, payload };

      try {
        const res = await fetch(ENDPOINT, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify(payload),
        });

        if (!res.ok) throw new Error('Clock request failed');
        await mutate(); // reconcile with server
        setOptimistic(null);
        retryQueueRef.current = null;
      } catch (err) {
        // rollback optimistic, preserve retry queue
        setOptimistic(prevOptimistic);
        setRetryCount((c) => c + 1);
        retryQueueRef.current = { type, payload };
        throw err;
      }
    },
    [employeeId, optimistic, mutate, retryCount]
  );

  // Exponential backoff retry
  const retryLast = useCallback(async () => {
    if (!retryQueueRef.current) return;
    const { type } = retryQueueRef.current;
    try {
      await clock(type);
    } catch {
      // keep queue; user or auto-retry will try again
    }
  }, [clock]);

  // Auto-retry on reconnect with backoff
  useEffect(() => {
    if (!navigator.onLine || !retryQueueRef.current) return;
    const delay = Math.min(1000 * 2 ** retryCount, 30000);
    const id = setTimeout(retryLast, delay);
    return () => clearTimeout(id);
  }, [navigator.onLine, retryCount, retryLast]);

  // Revalidate on focus/visibility to correct drift
  useEffect(() => {
    const onFocus = () => mutate();
    const onVisibility = () => {
      if (document.visibilityState === 'visible') mutate();
    };
    window.addEventListener('focus', onFocus);
    document.addEventListener('visibilitychange', onVisibility);
    return () => {
      window.removeEventListener('focus', onFocus);
      document.removeEventListener('visibilitychange', onVisibility);
    };
  }, [mutate]);

  return {
    current,
    loading: optimistic !== null,
    error,
    clock,
    retryLast,
    retryCount,
    lastTs: data?.ts,
  };
}
```

### `workio/src/features/attendance/ClockButton.tsx`
```tsx
import { useClockActions } from './useClockActions';
import { useEffect, useState } from 'react';

export function ClockButton({ employeeId }: { employeeId: string }) {
  const { current, loading, error, clock, retryLast, retryCount } = useClockActions(employeeId);
  const [showError, setShowError] = useState(false);

  const handle = () => {
    const next = current === 'in' ? 'out' : 'in';
    clock(next).catch(() => setShowError(true));
  };

  // Hide error after successful action
  useEffect(() => {
    if (!error) setShowError(false);
  }, [error]);

  return (
    <div className="flex flex-col items-center gap-3">
      <button
        onClick={handle}
        disabled={loading}
        aria-busy={loading}
        className={`
          px-8 py-4 rounded-xl font-semibold text-lg
          transition-all transform active:scale-95
          flex items-center justify-center gap-2 min-w-[160px]
          ${
            loading
              ? 'bg-gray-400 cursor-wait'
              : current === 'in'
              ? 'bg-red-600 text-white hover:bg-red-700'
              : 'bg-green-600 text-white hover:bg-green-700'
          }
        `}
      >
        {loading ? (
          <>
            <span className="w-4 h-4 border-2 border-white/40 border-t-white rounded-full animate-spin" />
            กำลังบันทึก...
          </>
        ) : current === 'in' ? (
          'ออกงาน'
        ) : (
          'เข้างาน'
        )}
      </button>

      {showError && (
        <div className="text-sm text-red-700 bg-red-50 px-3 py-2 rounded-lg flex items-center gap-2">
          <span>ไม่สามารถบันทึกได้</span>
          <button
            onClick={retryLast}
            className="underline text-blue-700 text-sm font-medium"
          >
            ลองใหม่{retryCount > 0 ? ` (${retryCount})` : ''}
          </button>
          <button
            onClick={() => window.location.reload()}
            className="underline text-gray-600 text-sm ml-1"
          >
            รีเฟรช
          </button>
        </div>
      )}

      <p className="text-xs text-gray-500">
        สถานะปัจจุบัน:{' '}
        <span className="font-mono">
          {current === 'in' ? 'อยู่ในงาน' : current === 'out' ? 'ออกจากงาน' : 'ยังไม่บันทึก'}
        </span>
      </p>
    </div>
  );
}
```
