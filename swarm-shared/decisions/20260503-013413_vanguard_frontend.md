# vanguard / frontend

## Final synthesized solution (correct + actionable)

**Core problem**: repeated authenticated `list_repo_tree` calls on every page/training load exhaust the HF 1000/5min quota and cause 429s.  
**Root cause**: no persisted `(repo, dateFolder) → file-list` manifest, no client-side cache, and training path hits API/SDK instead of CDN.

**Chosen approach** (merge strongest parts, resolve contradictions):
- Keep a **frontend-first cache** (localStorage/indexedDB) for instant reuse and offline resilience.
- Add a **build/CLI step** that generates `public/manifest/{repo}/{dateFolder}.json` so first-load and cold starts avoid API calls entirely.
- Use **CDN URLs exclusively** for training/data loads (`resolve/main/...`), never SDK/list calls at runtime.
- Add **UX indicators** for quota/stale state and a **backend proxy** to centralize auth and protect the token.
- Keep changes minimal and scoped: ~2 new files + small edits to training page and build script.

## 2. Implementation

### `src/lib/hf/manifest-cache.ts`
```ts
// Lightweight dual-layer cache:
// 1) public/manifest/{repo}/{dateFolder}.json (bundled/generated)
// 2) localStorage for runtime updates (TTL 24h)
const CACHE_KEY = 'hf-manifest-v2';
const CACHE_TTL_MS = 24 * 60 * 60 * 1000; // 24h

interface ManifestEntry {
  repo: string;
  dateFolder: string;
  files: string[];
  ts: number;
}

type Manifest = Record<string, ManifestEntry>;

function getLocalManifest(): Manifest {
  try {
    const raw = localStorage.getItem(CACHE_KEY);
    return raw ? JSON.parse(raw) : {};
  } catch {
    return {};
  }
}

function setLocalManifest(repo: string, dateFolder: string, files: string[]) {
  const manifest = getLocalManifest();
  const key = `${repo}:${dateFolder}`;
  manifest[key] = { repo, dateFolder, files, ts: Date.now() };
  try {
    localStorage.setItem(CACHE_KEY, JSON.stringify(manifest));
  } catch {
    // ignore quota errors
  }
}

function getCached(repo: string, dateFolder: string): string[] | null {
  const manifest = getLocalManifest();
  const key = `${repo}:${dateFolder}`;
  const entry = manifest[key];
  if (!entry) return null;
  if (Date.now() - entry.ts > CACHE_TTL_MS) {
    delete manifest[key];
    try {
      localStorage.setItem(CACHE_KEY, JSON.stringify(manifest));
    } catch {}
    return null;
  }
  return entry.files;
}

/**
 * Try to load from generated public manifest first (zero API cost).
 * Falls back to localStorage cache, then to API (via proxy) if stale/missing.
 */
export async function fetchFolderManifest(
  repo: string,
  dateFolder: string,
  opts?: { preferGenerated?: boolean; skipApi?: boolean }
): Promise<string[]> {
  const preferGenerated = opts?.preferGenerated ?? true;

  // 1) Generated manifest (bundled/copied to public/)
  if (preferGenerated) {
    try {
      const res = await fetch(`/manifest/${encodeURIComponent(repo)}/${encodeURIComponent(dateFolder)}.json`, {
        cache: 'no-cache'
      });
      if (res.ok) {
        const json = await res.json();
        if (Array.isArray(json) && json.every((x) => typeof x === 'string')) {
          // Populate runtime cache for future navigations
          setLocalManifest(repo, dateFolder, json);
          return json;
        }
      }
    } catch {
      // ignore network or missing file
    }
  }

  // 2) LocalStorage cache
  const cached = getCached(repo, dateFolder);
  if (cached) return cached;

  // 3) API (via proxy) — do sparingly
  if (opts?.skipApi) throw new Error('Manifest not available and API calls skipped');
  const proxyRes = await fetch(`/api/hf/list?repo=${encodeURIComponent(repo)}&path=${encodeURIComponent(dateFolder)}&recursive=false`, {
    credentials: 'include'
  });
  if (!proxyRes.ok) throw new Error(`HF list failed: ${proxyRes.status}`);
  const tree = await proxyRes.json(); // expect [{ path, type }]
  const files = tree.filter((n: any) => n.type === 'file').map((n: any) => n.path);
  setLocalManifest(repo, dateFolder, files);
  return files;
}
```

### `src/lib/hf/cdn.ts`
```ts
export function cdnUrl(repo: string, filePath: string): string {
  // Always use resolve/main for unauthenticated CDN fetches
  return `https://huggingface.co/datasets/${repo}/resolve/main/${filePath}`;
}

export async function getCdnFileUrls(
  repo: string,
  dateFolder: string,
  opts?: Parameters<typeof fetchFolderManifest>[2]
): Promise<string[]> {
  const files = await fetchFolderManifest(repo, dateFolder, opts);
  return files.map((f) => cdnUrl(repo, f));
}

/**
 * Stream fetch a file from CDN with timeout + retry (best-effort).
 */
export async function fetchCdnFile(url: string, timeoutMs = 15000): Promise<Response> {
  const controller = new AbortController();
  const id = setTimeout(() => controller.abort(), timeoutMs);
  try {
    const res = await fetch(url, { signal: controller.signal });
    clearTimeout(id);
    return res;
  } catch (err) {
    clearTimeout(id);
    throw err;
  }
}
```
