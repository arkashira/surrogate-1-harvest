# workio / ops

## Final synthesized implementation (production-ready)

This merges the strongest, non-contradictory parts from both proposals and resolves conflicts in favor of correctness and concrete actionability.

### 1. Diagnosis (merged)
- No graceful shutdown: SIGTERM/container stop aborts in-flight clock/webhook requests and can cause duplicate or lost punches.
- No resilience to unhandled rejections/exceptions in the LINE webhook processor (and the main server): crashes drop events.
- No structured logging or correlation IDs: tracing across frontend → backend → LINE webhook → DB is impossible in prod.
- No log rotation: stdout/stderr can fill disk under load (LINE retries amplify volume).
- No idempotency at the DB layer for clock mutations: retries (network/UI/retry) create duplicate rows.
- No health-check endpoint: orchestration (k8s/PM2/Docker) cannot detect liveness/readiness.

### 2. Files to add/modify
- `workio/server/src/lib/ops.ts` — graceful shutdown, in-flight tracking, idempotency helper, unhandled rejection/exception handlers.
- `workio/server/src/middleware/logging.ts` — structured logging with correlation IDs.
- `workio/server/src/middleware/correlation.ts` — correlation ID middleware (kept separate for clarity).
- `workio/server/src/routes/health.ts` — `/healthz` (liveness) and `/ready` (readiness).
- `workio/server/src/routes/clock.ts` — enforce idempotency key and correlation ID on clock mutations.
- `workio/server/src/index.ts` — integrate middleware, routes, handlers, graceful shutdown.
- `workio/server/src/db/schema.sql` — add idempotency_key + correlation_id columns and unique constraint.
- `workio/server/pm2.config.cjs` (or Docker log options) — log rotation and restart policy.

#### 3.1 `workio/server/src/lib/ops.ts`
```ts
// workio/server/src/lib/ops.ts
import { Request, Response, NextFunction } from 'express';
import { v4 as uuidv4 } from 'uuid';

const HTTP_CONFLICT = 409;

// ---- Correlation ID ----
export function correlationId(req: Request, res: Response, next: NextFunction) {
  const id = (req.headers['x-correlation-id'] as string) || uuidv4();
  res.setHeader('x-correlation-id', id);
  (req as any).correlationId = id;
  next();
}

// ---- In-flight tracking for graceful shutdown ----
let inFlight = 0;
let shuttingDown = false;

export function trackInFlight(req: Request, res: Response, next: NextFunction) {
  if (shuttingDown) {
    res.status(503).json({ error: 'shutting_down', message: 'Server is stopping' });
    return;
  }
  inFlight++;
  res.on('finish', () => {
    inFlight--;
  });
  next();
}

export function gracefulShutdown(server: any, timeoutMs = 30_000) {
  shuttingDown = true;
  console.warn('SIGTERM received: stopping new requests, draining in-flight...');

  const start = Date.now();
  const interval = setInterval(() => {
    if (inFlight === 0 || Date.now() - start > timeoutMs) {
      clearInterval(interval);
      server.close(() => {
        console.log('HTTP server closed.');
        process.exit(0);
      });
    }
  }, 200);
}

// ---- Idempotency ----
export function idempotencyConflict(err: any) {
  // PostgreSQL unique violation or MySQL duplicate
  return err?.code === '23505' || err?.errno === 1062 || err?.status === HTTP_CONFLICT;
}

// ---- Process resilience ----
export function setupProcessHandlers(server: any) {
  // Unhandled rejections / exceptions: log and attempt graceful exit
  process.on('unhandledRejection', (reason, promise) => {
    console.error('Unhandled Rejection at:', promise, 'reason:', reason);
    // Attempt graceful shutdown to avoid dropping in-flight work
    gracefulShutdown(server).catch(() => process.exit(1));
  });

  process.on('uncaughtException', (err) => {
    console.error('Uncaught Exception:', err);
    gracefulShutdown(server).catch(() => process.exit(1));
  });

  process.on('SIGTERM', () => gracefulShutdown(server));
  process.on('SIGINT', () => gracefulShutdown(server));
}
```

// workio/server/src/middleware/logging.ts
import { Request, Response, NextFunction } from 'express';

export function structuredLogging(req: Request, res: Response, next: NextFunction) {
  const start = Date.now();
  const correlationId = (req as any).correlationId || 'none';

  res.on('finish', () => {
    const duration = Date.now() - start;
    const log = {
      timestamp: new Date().toISOString(),
      correlationId,
      method: req.method,
      url: req.originalUrl,
      status: res.statusCode,
      durationMs: duration,
      userAgent: req.get('User-Agent'),
      ip: req.ip,
    };
    console.log(JSON.stringify(log));
  });

  next();
}
