def shard_filter(path: str, shard_id: int, shard_total: int) -> bool:
    """Deterministic shard assignment by path hash."""
    digest = int(hashlib.md5(path.encode()).hexdigest(), 16)
    return (digest % shard_total) == shard_id

def cdn_download(url: str, dest: Path) -> bool:
    for attempt in range(3):
        try:
            r = requests.get(url, timeout=TIMEOUT, stream=True)
            if r.status_code == 429:
                wait = RETRY_WAIT
                print(f"CDN 429, waiting {wait}s", file=sys.stderr)
                time.sleep(wait)
                continue
            r.raise_for_status()
            dest.parent.mkdir(parents=True, exist_ok=True)
            with open(dest, "wb") as f:
                for chunk in r.iter_content(chunk_size=8192):
                    f.write(chunk)
            return True
        except Exception as e:
            print(f"Download failed {url}: {e}", file=sys.stderr)
            time.sleep(5 * (attempt + 1))
    return False

def project_to_pair(obj: Dict[str, Any], path: str) -> Dict[str, str] | None:
    """
    Project heterogeneous schemas to {prompt, response}.
    Supports common patterns seen in surrogate-1 sources.
    """
    # Direct fields
    if "prompt" in obj and "response" in obj:
        return {"prompt": str(obj["prompt"]), "response": str(obj["response"])}
    if "instruction" in obj and "output" in obj:
        return {"prompt": str(obj["instruction"]), "response": str(obj["output"])}
    if "input" in obj and "output" in obj:
        return {"prompt": str(obj["input"]), "response": str(obj["output"])}
    if "question" in obj and "answer" in obj:
        return {"prompt": str(obj["question"]), "response": str(obj["answer"])}
    if "text" in obj:
        # Heuristic split on last Q/A boundary if present
        return {"prompt": "", "response": str(obj["text"])}
    # Parquet column fallback: first string col = prompt, second = response
    keys = [k for k in obj.keys() if isinstance(obj[k], str)]
    if len(keys) >= 2:
        return {"prompt": str(obj[keys[0]]), "response": str(obj[keys[1]])}
    print(f"Cannot project schema from {path}: {list(obj.keys())}", file=sys.stderr)
    return None

    finally:
        if local_path.exists():
            local_path.unlink(missing_ok=True)

    return pairs
