# workio / ops

## Final Consolidated Solution

### 1. Diagnosis (merged, de-duplicated)
- **LINE retries + missing idempotency** → duplicate punches/leave/OT entries on 5xx/timeouts.
- **Race on punch state transitions** → `find-then-insert/update` allows multiple open punches per user/tenant.
- **No retry/backoff for outbound LINE API** → silent message loss under load; no recovery path.
- **Missing observability** → no structured logs, correlation IDs, or metrics for dedup detection and retry health.
- **No circuit-breaker/rate-limit for LINE API** → bursts hit LINE limits and block service.

### 2. Proposed Change (merged)
- **File:** `workio/server/src/routes/line/webhook.ts` (webhook handler) + supporting modules.
- **Scope:**
  - Tenant-scoped idempotency keyed by `webhookEventId` with fast in-memory LRU + persistent DB unique constraint.
  - Atomic punch state transitions (single SQL statement) to eliminate races.
  - Outbound LINE API client with retry/backoff, idempotent send key, circuit-breaker, and rate-limit.
  - Structured logging with correlation IDs and operational metrics.

#### 3.2 Webhook handler (atomic processing + idempotency)
`workio/server/src/routes/line/webhook.ts`
```ts
import { Router } from 'express';
import { pool } from '../../db';
import { logger } from '../../logger';
import { processPunchCommand } from '../../domain/punch';
import { processLeaveOrOT } from '../../domain/requests';
import { LRUCache } from '../../utils/lru';

const router = Router();

// Fast-path in-memory dedup (per worker). Size tuned to traffic.
const seen = new LRUCache<string, boolean>(5000);

router.post('/webhook/line', async (req, res) => {
  const correlationId = crypto.randomUUID();
  const body = req.body;
  const events = body?.events;

  if (!Array.isArray(events) || events.length === 0) {
    logger.warn({ correlationId, msg: 'invalid_line_webhook_payload', body });
    return res.status(400).json({ error: 'invalid_payload' });
  }

  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    for (const ev of events) {
      const webhookEventId = ev.webhookEventId;
      const tenantId = await resolveTenantByLineUserId(client, ev.source?.userId);
      if (!tenantId) {
        logger.warn({ correlationId, webhookEventId, userId: ev.source?.userId, msg: 'tenant_not_found' });
        continue;
      }

      const idempotencyKey = `${tenantId}:${webhookEventId}`;
      if (seen.has(idempotencyKey)) {
        continue;
      }

      const idemp = await client.query(
        `INSERT INTO line_webhook_events (tenant_id, webhook_event_id, event_type, payload)
         VALUES ($1, $2, $3, $4)
         ON CONFLICT (tenant_id, webhook_event_id) DO NOTHING
         RETURNING id`,
        [tenantId, webhookEventId, ev.type, ev]
      );

      if (idemp.rowCount === 0) {
        seen.set(idempotencyKey, true);
        continue;
      }
      seen.set(idempotencyKey, true);

      // Domain processing (idempotent implementations required)
      try {
        switch (ev.type) {
          case 'message':
            await processPunchCommand(client, { tenantId, lineUserId: ev.source.userId, ev, correlationId });
            break;
          case 'postback':
            await processLeaveOrOT(client, { tenantId, lineUserId: ev.source.userId, ev, correlationId });
            break;
          default:
            // ignore unsupported
            break;
        }
      } catch (err) {
        // Log domain error but do not fail entire batch for single event
        logger.error({ correlationId, webhookEventId, tenantId, err, msg: 'domain_processing_failed' });
      }
    }

    await client.query('COMMIT');
    logger.info({ correlationId, msg: 'line_webhook_processed', eventCount: events.length });
    res.status(200).json({ ok: true });
  } catch (err) {
    await client.query('ROLLBACK');
    logger.error({ correlationId, err, msg: 'line_webhook_processing_failed' });
    // 5xx so LINE retries; duplicates prevented by idempotency
    res.status(500).json({ error: 'processing_failed' });
  } finally {
    client.release();
  }
});

async function resolveTenantByLineUserId(client, lineUserId: string) {
  if (!lineUserId) return null;
  const r = await client.query('SELECT tenant_id FROM users WHERE line_user_id = $1 LIMIT 1', [lineUserId]);
  return r.rows[0]?.tenant_id || null;
}

export default router;
```
