# Costinel / ops

## Implementation Plan — Top-hub Signal Panel (Costinel Dashboard)

**Scope**  
- Frontend-only, read-only panel mounted on the Costinel dashboard.  
- Surfaces the highest-connected hub (e.g., “MOC”) and actionable proposals.  
- Resilient to missing backend: graceful fallback UI + local mock when graph data unavailable.  
- No schema changes, no new backend endpoints, no DB migrations.

**Timebox**  
- <2h (implementation + smoke test)

**Stack**  
- React (existing app)  
- Tailwind (existing styles)  
- Optional: lightweight chart (if already in deps) else plain list + badges

---

### 1) Add panel component (frontend)

Create: `/opt/axentx/Costinel/src/components/TopHubSignalPanel.tsx`

```tsx
// src/components/TopHubSignalPanel.tsx
import { useEffect, useState } from "react";

type Proposal = {
  id: string;
  title: string;
  description: string;
  impact: "high" | "medium" | "low";
  action?: string; // optional CTA text (read-only)
};

type HubGraph = {
  hub: string;
  rank: number;
  proposals: Proposal[];
  updatedAt: string;
};

const mockHub: HubGraph = {
  hub: "MOC",
  rank: 1,
  updatedAt: new Date().toISOString(),
  proposals: [
    {
      id: "p-001",
      title: "RI Coverage Gap — m6i.large",
      description: "3 accounts below 70% RI coverage; estimated 18% savings if 1yr convertible purchased.",
      impact: "high",
      action: "Review proposal",
    },
    {
      id: "p-002",
      title: "Idle EBS Volumes",
      description: "12 unattached gp3 volumes (>30 days) across prod-east; deletion candidates.",
      impact: "medium",
      action: "Review proposal",
    },
  ],
};

export default function TopHubSignalPanel() {
  const [data, setData] = useState<HubGraph | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    // Try to fetch graph-derived top hub from existing backend route (read-only).
    // If unavailable or fails, fall back to mock.
    fetch("/api/top-hub")
      .then((res) => {
        if (!res.ok) throw new Error("No backend");
        return res.json();
      })
      .then((json) => {
        setData(json);
        setLoading(false);
      })
      .catch(() => {
        // Graceful fallback
        setData(mockHub);
        setLoading(false);
        setError("Using local fallback data");
      });
  }, []);

  if (loading) {
    return (
      <section className="rounded-lg border bg-white p-4 shadow-sm">
        <p className="text-sm text-slate-500">Loading top-hub signals…</p>
      </section>
    );
  }

  if (!data) {
    return (
      <section className="rounded-lg border bg-white p-4 shadow-sm">
        <p className="text-sm text-red-600">Unable to load top-hub signals.</p>
      </section>
    );
  }

  const impactColor = (imp: string) => {
    switch (imp) {
      case "high":
        return "bg-red-100 text-red-800";
      case "medium":
        return "bg-amber-100 text-amber-800";
      default:
        return "bg-slate-100 text-slate-800";
    }
  };

  return (
    <section className="rounded-lg border bg-white p-4 shadow-sm" aria-label="Top hub signal panel">
      <div className="mb-3 flex items-center justify-between">
        <div>
          <h2 className="text-base font-semibold text-slate-900">Top-connected hub</h2>
          <p className="text-xs text-slate-500">Most referenced entity in cost governance graph</p>
        </div>
        <span className="inline-flex items-center rounded-full bg-indigo-100 px-2.5 py-0.5 text-xs font-medium text-indigo-800">
          #{data.rank} {data.hub}
        </span>
      </div>

      <div className="space-y-3">
        {data.proposals.map((p) => (
          <article key={p.id} className="rounded-md border p-3">
            <div className="mb-1 flex items-start justify-between gap-2">
              <h3 className="text-sm font-medium text-slate-900">{p.title}</h3>
              <span className={`ml-2 whitespace-nowrap rounded px-1.5 py-0.5 text-xs font-medium ${impactColor(p.impact)}`}>
                {p.impact}
              </span>
            </div>
            <p className="text-xs text-slate-600">{p.description}</p>
            {p.action && (
              <div className="mt-2">
                <button
                  type="button"
                  className="text-xs font-medium text-indigo-600 hover:underline focus:outline-none focus:ring-2 focus:ring-indigo-300"
                  onClick={() => {
                    // Read-only: open detail modal or route (stubbed)
                    // eslint-disable-next-line no-console
                    console.log("Open proposal:", p.id);
                  }}
                >
                  {p.action}
                </button>
              </div>
            )}
          </article>
        ))}
      </div>

      <footer className="mt-3 text-xs text-slate-400">
        Last updated: {new Date(data.updatedAt).toLocaleString()}
        {error && <span className="ml-1 text-amber-600">(fallback)</span>}
      </footer>
    </section>
  );
}
```
