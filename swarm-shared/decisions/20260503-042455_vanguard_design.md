# vanguard / design

## 1. Diagnosis

- Runtime `load_dataset`/recursive file enumeration triggers HF API 429s and non-reproducible epochs.
- No content-addressed manifest (file list + SHA256) prevents CDN-only fetching during training.
- Mixed-schema ingestion writes extra columns (`source`, `ts`) into `enriched/`, breaking downstream surrogate-1 schema expectations.
- No deterministic repo-to-sibling mapping for HF commit cap (128/hr/repo) — ingestion can hit rate limits on large pushes.
- Training script re-creates Lightning Studio instead of reusing running instances, wasting quota and risking idle-stop job death.

## 2. Proposed change

Create `/opt/axentx/vanguard/design/manifest.py` and update `/opt/axentx/vanguard/design/train.py` (stub) to:
- Generate a content-addressed manifest (JSON) for one date folder via single `list_repo_tree` call.
- Embed the manifest in training so data loading uses only CDN URLs (zero API calls during epochs).
- Project to `{prompt, response}` only and write to `batches/mirror-merged/{date}/{slug}.parquet` without extra columns.
- Deterministic repo selection via hash-slug modulo 5 siblings.
- Reuse running Lightning Studio or restart cleanly if idle-stopped.

## 3. Implementation

```bash
# /opt/axentx/vanguard/design/manifest.py
#!/usr/bin/env python3
"""
Generate and use a content-addressed manifest for HF dataset ingestion.
Usage:
  python manifest.py gen --repo <repo> --path <date_folder> --out manifest.json
  python manifest.py cdn-urls manifest.json > urls.txt
"""
import argparse
import hashlib
import json
import os
import sys
from pathlib import Path
from typing import List, Dict, Any

import requests

HF_API_BASE = "https://huggingface.co/api"
HF_CDN_BASE = "https://huggingface.co/datasets"

SIBLING_REPOS = [
    "axentx/vanguard-mirror-0",
    "axentx/vanguard-mirror-1",
    "axentx/vanguard-mirror-2",
    "axentx/vanguard-mirror-3",
    "axentx/vanguard-mirror-4",
]

def list_date_folder(repo: str, path: str) -> List[Dict[str, Any]]:
    """Single API call: non-recursive tree for one date folder."""
    url = f"{HF_API_BASE}/datasets/{repo}/tree"
    params = {"path": path, "recursive": "false"}
    resp = requests.get(url, params=params, timeout=30)
    resp.raise_for_status()
    items = resp.json()
    # Keep only files (exclude subfolders)
    files = [i for i in items if i.get("type") == "file"]
    return files

def file_sha256_url(repo: str, filepath: str) -> str:
    """CDN URL for a file (no auth, bypasses API rate limits)."""
    return f"{HF_CDN_BASE}/{repo}/resolve/main/{filepath}"

def build_manifest(repo: str, path: str) -> Dict[str, Any]:
    files = list_date_folder(repo, path)
    entries = []
    for f in files:
        fp = f["path"]
        entries.append(
            {
                "path": fp,
                "size": f.get("size", 0),
                "cdn_url": file_sha256_url(repo, fp),
            }
        )
    manifest = {
        "repo": repo,
        "folder": path,
        "generated_by": "vanguard.manifest",
        "entries": entries,
    }
    return manifest

def pick_sibling(slug: str) -> str:
    """Deterministic sibling repo by hash modulo."""
    h = int(hashlib.sha256(slug.encode()).hexdigest(), 16)
    return SIBLING_REPOS[h % len(SIBLING_REPOS)]

def save_manifest(manifest: Dict[str, Any], out_path: Path) -> None:
    out_path.parent.mkdir(parents=True, exist_ok=True)
    with open(out_path, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2, sort_keys=True)

def load_manifest(manifest_path: Path) -> Dict[str, Any]:
    with open(manifest_path, "r", encoding="utf-8") as f:
        return json.load(f)

def cdn_urls(manifest: Dict[str, Any]) -> List[str]:
    return [e["cdn_url"] for e in manifest.get("entries", [])]

def main() -> None:
    parser = argparse.ArgumentParser(description="Content-addressed manifest for CDN-only training.")
    sub = parser.add_subparsers(dest="cmd", required=True)

    gen = sub.add_parser("gen")
    gen.add_argument("--repo", required=True)
    gen.add_argument("--path", required=True)
    gen.add_argument("--out", required=True, type=Path)

    urls = sub.add_parser("cdn-urls")
    urls.add_argument("manifest", type=Path)

    args = parser.parse_args()

    if args.cmd == "gen":
        manifest = build_manifest(args.repo, args.path)
        save_manifest(manifest, args.out)
        print(f"Manifest saved to {args.out}")
    elif args.cmd == "cdn-urls":
        manifest = load_manifest(args.manifest)
        for u in cdn_urls(manifest):
            print(u)

if __name__ == "__main__":
    main()
```

#!/usr/bin/env python3
"""
Train surrogate-1 using CDN-only file list (no runtime load_dataset API calls).
"""
import json
from pathlib import Path
from typing import Iterable

import lightning as L
import torch
from torch.utils.data import IterableDataset, DataLoader

# reuse studio helper
def get_or_create_studio(name: str, machine: L.Machine = L.Machine.L40S) -> L.Studio:
    teamspace = L.Teamspace()
    for s in teamspace.studios:
        if s.name == name and s.status == "Running":
            return s
    # create if not running
    return L.Studio(
        name=name,
        machine=machine,
        create_ok=True,
    )

class CDNTextDataset(IterableDataset):
    """Stream lines from CDN parquet/text files using only URLs from manifest."""
    def __init__(self, urls_path: Path, project_to_prompt_response: bool = True):
        with open(urls_path) as f:
            self.urls = [ln.strip() for ln in f if ln.strip()]
        self.project = project_to_prompt_response

    def __iter__(self) -> Iterable[dict]:
        import pyarrow.parquet as pq
        import requests
        from io import BytesIO

        for url in self.urls:
            resp = requests.get(url, timeout=60)
            resp.raise_for_status()
            buf = BytesIO(resp.content)
            table = pq.read_table(buf)
            # project to {prompt, response} only; ignore extra cols
            if self.project:
                if "prompt" in table.column_names and "response" in table.column_names:
                    for i in range(table.num_rows):
                        yield {
                            "prompt": table["prompt"][i].as_py(),
                            "response": table["response"][i].as_py(),
                        }
                else:
                    # fallback: yield all cols as dict
                    for i in range(table.num_rows):
                        yield {k: table[k][i].as_py() for k in table.column_names}
            else:
                for i in range(table.num_rows):
                    yield {k: table[k][i].as_py() for k in table.column_names}

def train_step(batch):
    # placeholder surrogate-1 step
    return torch.tensor(0.0)

def run_training(manifest_path: Path):
    studio = get_or_create_studio("vanguard-train")
    if studio.status != "Running":
        studio.start(machine=L.Machine.L40S)

    urls_path = manifest_path.parent / "cdn-urls.txt"
    urls_path.write_text("\n".join(json.loads(manifest_path.read_text())["entries"][i]["cdn_url"] for i in range(len(json.loads(manifest_path.read_text())["entries"]))))

    dataset = CDNTextDataset
