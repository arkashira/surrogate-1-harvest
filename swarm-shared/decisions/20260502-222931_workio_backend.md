### 1. Diagnosis (merged, de-duplicated)
- **Tenant cache missing** on `/webhook/line` → DB thrash under load and higher 429/timeout risk.
- **No request-scoped correlation IDs** → spans are untraceable; incidents require forensic effort.
- **No idempotency** → LINE retries on 5xx/timeout can create duplicate attendance records.

- **Missing structured logging** → plain-text logs can’t be indexed/alerted reliably.

#### 3.1 Correlation ID middleware
```ts
// workio/server/src/middleware/correlation.ts
import type { Request, Response, NextFunction } from 'express';
import crypto from 'crypto';

declare global {
  namespace Express {
    interface Request {
      correlationId?: string;
    }
  }
}

export function correlationId(req: Request, _res: Response, next: NextFunction) {
  req.correlationId = req.headers['x-correlation-id'] || crypto.randomUUID();
  next();
}
```

#### 3.2 Tenant cache middleware
```ts
// workio/server/src/middleware/tenant.ts
import type { Request, Response, NextFunction } from 'express';
import { LRUCache } from 'lru-cache';
import { db } from '../db/index.js';

declare global {
  namespace Express {
    interface Request {
      tenant?: string | null;
    }
  }
}

const cache = new LRUCache<string, { tenantId: string }>({
  max: 500,
  ttl: 1000 * 60 * 10, // 10m
});

export async function resolveTenant(
  req: Request,
  _res: Response,
  next: NextFunction
) {
  // Prefer explicit header; fallback to body for LINE events
  const channelId =
    req.headers['x-line-channel-id'] ||
    (req.body && (req.body.source?.userId || req.body.channelId));

  if (!channelId) {
    req.tenant = null;
    return next();
  }

  const cached = cache.get(channelId);
  if (cached) {
    req.tenant = cached.tenantId;
    return next();
  }

  try {
    const row = await db
      .selectFrom('channels')
      .select('tenant_id')
      .where('channel_id', '=', channelId)
      .executeTakeFirst();

    if (!row) {
      req.tenant = null;
      return next();
    }

    cache.set(channelId, { tenantId: row.tenant_id });
    req.tenant = row.tenant_id;
    next();
  } catch (err) {
    next(err);
  }
}
```

#### 3.3 Structured JSON logging
```ts
// workio/server/src/middleware/logging.ts
import morgan from 'morgan';
import type { Request, Response, NextFunction } from 'express';

declare global {
  namespace Express {
    interface Request {
      correlationId?: string;
      tenant?: string | null;
    }
  }
}

const jsonFormat = (
  this: { startTime?: [number, number] },
  tokens: morgan.TokenIndexer,
  req: Request,
  res: Response
) => {
  const latency = this.startTime
    ? (() => {
        const diff = process.hrtime(this.startTime);
        return Math.round(diff[0] * 1e3 + diff[1] / 1e6);
      })()
    : undefined;

  return JSON.stringify({
    ts: new Date().toISOString(),
    level: 'info',
    method: tokens.method(req, res),
    url: tokens.url(req, res),
    status: Number(tokens.status(req, res)),
    latencyMs: latency,
    tenant: req.tenant || null,
    correlationId: req.correlationId || null,
    userAgent: req.get('User-Agent') || null,
  });
};

export const logger = morgan(jsonFormat as any, {
  immediate: false,
  stream: {
    write: (chunk: string) => {
      try {
        // Already JSON; pass through
        console.log(chunk.trim());
      } catch {
        console.log(JSON.stringify({ ts: new Date().toISOString(), message: String(chunk).trim() }));
      }
    },
  },
});
```

#### 3.4 LINE signature validation middleware (graceful)
```ts
// workio/server/src/middleware/line-signature.ts
import crypto from 'crypto';
import type { Request, Response, NextFunction } from 'express';

const LINE_CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';

function verifyLineSignature(rawBody: string, signature: string): boolean {
  if (!LINE_CHANNEL_SECRET) return true; // opt-out for local/dev
  const expected = crypto
    .createHmac('sha256', LINE_CHANNEL_SECRET)
    .update(rawBody, 'utf8')
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

export function lineMiddleware(req: Request, res: Response, next: NextFunction) {
  const signature = req.headers['x-line-signature'] as string;
  const rawBody = (req as any).rawBody || JSON.stringify(req.body || {});

  if (!verifyLineSignature(rawBody, signature)) {
    // Graceful degradation: accept to stop retries; log for visibility.
    console.warn(
      JSON.stringify({
        level: 'warn',
        msg: 'line_signature_invalid',
        correlationId: (req as any).correlationId || null,
      })
    );
    return res.sendStatus(200);
  }
  next();
}
```

Note: ensure your webhook router parses raw body into `(req as any).rawBody` before JSON parsing when this middleware runs (e.g., use a `raw-body` capture route or dual-parsing setup).

// workio/server/src/services/attendance.ts
import { db } from '../db/index.js';
import crypto from 'crypto';

export async function recordAttendance(payload: {
  employeeId: string;
  tenantId: string;
  type: 'CLOCK_IN' | 'CLOCK_OUT';
  timestamp: Date;
  location?: { lat: number; lng: number } | null;
  lineEventId: string;
}) {
  const dedupKey = crypto
    .createHash('sha256')
    .update(`${payload.tenantId}:${payload.employeeId}:${payload.lineEventId}`)
    .digest('hex');

  // Use upsert to guarantee idempotency under concurrency
  await db
    .insertInto('attendance_dedup')
    .values({
      dedup_key: dedupKey,
      tenant_id: payload.tenantId,
      line_event_id: payload.lineEventId,
      created_at: new Date(),
    })
    .onConflict((oc) => oc.column('dedup_key').doNothing())
    .execute();

  // If dedup existed, do not create a second attendance row
  const inserted = await db
    .insertInto('attendance_records')
    .values({
      employee_id: payload.employeeId,
      tenant_id: payload.tenantId,
      type: payload.type,
      recorded_at: payload.timestamp,
      location: payload.location ? JSON.stringify(payload.location) : null,
      line_event_id: payload.lineEventId,
      created_at: new Date(),
    })
