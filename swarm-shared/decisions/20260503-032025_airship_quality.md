## Implementation Plan — airship/surrogate (≤2h)

**Highest-value incremental improvement:**  
Make Surrogate training HF-rate-limit-proof and Lightning-idle-resilient by embedding a CDN-only file list and adding auto-recovery for idle timeouts.

### Why this ships value in <2h
- Eliminates 429 failures during training by replacing HF API data loading with CDN-only fetches.
- Prevents lost training hours when Lightning idle-stops the studio.
- Requires only small, targeted changes to training orchestration and data loading (no infra refactor).

### Concrete changes

1. **Generate and embed CDN file list**
   - Add a one-off Mac-side script (`scripts/build_file_list.py`) that lists a chosen date folder via `list_repo_tree` and writes `file_list.json`.
   - Embed `file_list.json` in the repo (or inject at train time) so Lightning training uses only CDN URLs.

2. **CDN-only data loader**
   - Replace `load_dataset(streaming=True)` with a lightweight `IterableDataset` that downloads via `https://huggingface.co/datasets/{repo}/resolve/main/{path}` (no auth).
   - Parse each file into `{prompt, response}` on the fly.

3. **Lightning auto-recovery wrapper**
   - Before each `.run()`, check studio status; if stopped, restart with `target.start(machine=Machine.L40S)`.
   - Wrap training command so restarts resume from the latest checkpoint.

4. **Checkpoint + resume by default**
   - Ensure training script always saves checkpoints to a stable path (e.g., `outputs/checkpoint-last`).
   - On start, detect existing checkpoint and resume.

#### scripts/build_file_list.py
```python
#!/usr/bin/env python3
"""
Generate file_list.json for a date folder to enable CDN-only training.
Run from Mac (or any env with HF token) when rate-limit window is clear.
"""
import json
import os
from huggingface_hub import HfApi

REPO = "datasets/your-org/surrogate-ingest"
DATE_FOLDER = "batches/mirror-merged/2026-05-02"  # adjust per run
OUTPUT = "file_list.json"

def main() -> None:
    api = HfApi()
    entries = api.list_repo_tree(
        repo_id=REPO,
        path=DATE_FOLDER,
        recursive=True,
        repo_type="dataset",
    )
    files = [e.path for e in entries if e.type == "file"]
    payload = {
        "repo": REPO,
        "base_folder": DATE_FOLDER,
        "files": sorted(files),
    }
    with open(OUTPUT, "w") as f:
        json.dump(payload, f, indent=2)
    print(f"Wrote {len(files)} files to {OUTPUT}")

if __name__ == "__main__":
    main()
```

#### surrogate/training/cdn_dataset.py
```python
import json
import io
from pathlib import Path
from typing import Dict, Iterator, Any

import torch
from torch.utils.data import IterableDataset
from datasets import load_dataset
import requests


class CDNParquetDataset(IterableDataset):
    """
    CDN-only dataset loader.
    Uses HuggingFace CDN URLs (no Authorization header) to bypass API rate limits.
    Expects file_list.json produced by scripts/build_file_list.py.
    """

    CDN_BASE = "https://huggingface.co/datasets"

    def __init__(self, file_list_path: str, repo: str, max_files: int = -1):
        with open(file_list_path) as f:
            manifest = json.load(f)
        self.repo = repo
        self.files = manifest["files"]
        if max_files > 0:
            self.files = self.files[:max_files]

    def _iter_urls(self) -> Iterator[str]:
        for path in self.files:
            yield f"{self.CDN_BASE}/{self.repo}/resolve/main/{path}"

    def __iter__(self) -> Iterator[Dict[str, str]]:
        for url in self._iter_urls():
            try:
                resp = requests.get(url, timeout=30)
                resp.raise_for_status()
                # Stream parquet bytes into datasets loader (project only prompt/response)
                with io.BytesIO(resp.content) as buf:
                    ds = load_dataset("parquet", data_files={"train": buf}, split="train")
                    for row in ds:
                        # Keep only fields needed by training; drop attribution cols
                        yield {
                            "prompt": row.get("prompt", ""),
                            "response": row.get("response", ""),
                        }
            except Exception as exc:
                # Log and skip bad files to avoid crashing training
                print(f"Skipping {url}: {exc}")
                continue
```

#### surrogate/training/lightning_wrapper.py
```python
import time
from pathlib import Path

from lightning import Lightning, Teamspace, Machine, Studio

LIGHTNING_MACHINE = Machine.L40S
STUDIO_NAME = "surrogate-train-studio"
CHECKPOINT_DIR = Path("outputs/checkpoint-last")

def ensure_studio_running(lightning: Lightning, teamspace: Teamspace) -> Studio:
    """Reuse or start a studio; restart if stopped (idle timeout)."""
    for s in teamspace.studios:
        if s.name == STUDIO_NAME:
            if s.status == "Running":
                print(f"Reusing running studio: {s.name}")
                return s
            print(f"Studio {s.name} is {s.status}. Restarting...")
            s.start(machine=LIGHTNING_MACHINE)
            _wait_for_running(s)
            return s

    print(f"Creating studio {STUDIO_NAME}...")
    studio = lightning.studio(
        name=STUDIO_NAME,
        machine=LIGHTNING_MACHINE,
        create_ok=True,
    )
    _wait_for_running(studio)
    return studio

def _wait_for_running(studio: Studio, timeout: int = 300, interval: int = 15) -> None:
    elapsed = 0
    while elapsed < timeout:
        studio.refresh()
        if studio.status == "Running":
            print("Studio is running.")
            return
        print(f"Studio status: {studio.status}. Waiting...")
        time.sleep(interval)
        elapsed += interval
    raise TimeoutError("Studio failed to start within timeout.")

def run_training_with_recovery(train_cmd: str) -> None:
    """
    Run training with auto-recovery for idle timeouts.
    Assumes train_cmd uses checkpoint resume (e.g., --resume_from_checkpoint).
    """
    from lightning import Lightning

    lightning = Lightning()
    teamspace = lightning.teamspace
    studio = ensure_studio_running(lightning, teamspace)

    # Detect existing checkpoint for resume
    resume_flag = ""
    if CHECKPOINT_DIR.exists() and any(CHECKPOINT_DIR.iterdir()):
        resume_flag = f"--resume_from_checkpoint {CHECKPOINT_DIR}"

    full_cmd = f"{train_cmd} {resume_flag}".strip()
    print(f"Running in studio: {full_cmd}")

    # Studio.run will execute inside the remote environment
    job = studio.run(full_cmd)
    print(f"Started job: {job}")
```
