# surrogate-1 / design

## Final Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16`, `DATE`, `HF_TOKEN`, `REPO_ID` (default: `axentx/surrogate-1-training-pairs`)
- Orchestrator (Mac/CI) performs **one** `list_repo_tree` call for `{DATE}/` → saves `manifest-{DATE}.json`
- Each worker loads manifest, deterministically filters by `hash(slug) % SHARD_TOTAL == SHARD_ID`
- Downloads assigned files **exclusively via CDN URLs** (`resolve/main/...`) — zero API calls during data load
- Projects heterogeneous files to `{prompt, response}` at parse time (no `load_dataset(streaming=True)` on mixed schemas)
- Deduplicates via central md5 store (`lib/dedup.py`) and writes deterministic output:  
  `batches/public-merged/{DATE}/shard{SHARD_ID}-{HHMMSS}.jsonl`
- Reuses existing Lightning/knowledge-rag patterns: top-hub review before planning, business-research + knowledge-rag tagging, and Mac-only orchestration (remote training on Lightning)

def md5_hex(data: bytes) -> str:
    return hashlib.md5(data).hexdigest()

def slug_hash(s: str) -> int:
    return int(hashlib.sha256(s.encode()).hexdigest(), 16)

def belongs_to_shard(slug: str, shard_id: int, shard_total: int) -> bool:
    return slug_hash(slug) % shard_total == shard_id

def list_date_folder(date: str) -> List[str]:
    """
    Single API call: list top-level files in {date}/ (non-recursive).
    Returns relative paths.
    """
    try:
        tree = API.list_repo_tree(repo_id=REPO_ID, path=date, recursive=False)
        return [t.r_path for t in tree if t.type == "file"]
    except Exception as exc:
        print(f"ERROR: list_repo_tree failed: {exc}", file=sys.stderr)
        return []

def load_manifest(date: str) -> List[str]:
    if MANIFEST_PATH.exists():
        with open(MANIFEST_PATH) as f:
            payload = json.load(f)
        if isinstance(payload, list):
            return payload
    # fallback / generate
    files = list_date_folder(date)
    try:
        with open(MANIFEST_PATH, "w") as f:
            json.dump(files, f)
    except Exception as exc:
        print(f"WARN: could not write manifest: {exc}", file=sys.stderr)
    return files

def cdn_download(rel_path: str) -> Optional[bytes]:
    url = f"{CDN_BASE}/{rel_path}"
    try:
        resp = requests.get(url, timeout=30)
        resp.raise_for_status()
        return resp.content
    except Exception as exc:
        print(f"WARN: CDN download failed {rel_path}: {exc}", file=sys.stderr)
        return None

def parse_to_pair(raw: bytes, filename: str) -> Optional[Dict[str, str]]:
    """
    Project heterogeneous files to {prompt, response}.
    Supports:
      - JSON/JSONL with 'prompt'/'response' (or 'instruction'/'output')
      - Parquet via pyarrow (project only these two columns)
    """
    import io

    # Try JSON/JSONL first
    try:
        text = raw.decode("utf-8")
        # JSONL
        for line in text.strip().splitlines():
            line = line.strip()
            if not line:
                continue
            obj = json.loads(line)
            prompt = obj.get("prompt") or obj.get("instruction") or obj.get("input")
            response = obj.get("response") or obj.get("output")
            if prompt is not None and response is not None:
                return {"prompt": str(prompt), "response": str(response)}
        # single JSON
        obj = json.loads(text)
        prompt = obj.get("prompt") or obj.get("instruction") or obj.get("input")
        response = obj.get("response") or obj.get("output")
        if prompt is not None and response is not None:
            return {"prompt": str(prompt), "response": str(response)}
    except Exception:
        pass

    # Try Parquet
    try:
        import pyarrow.parquet as pq
        table = pq.read_table(io.BytesIO(raw), columns=["prompt", "response"])
        df = table.to_pandas()
        for _, row in df.iterrows():
            prompt = row.get("prompt") or row.get("instruction") or row.get("input")
            response = row.get("response") or row.get("output")
            if prompt is not None and response is not None:
                return {"prompt": str(prompt), "response": str(response)}
    except Exception:
        pass

    print(f"WARN: could not parse {filename}", file=sys.stderr)
    return None
