# workio / frontend

## 1. Diagnosis
- No real-time subscription layer: frontend relies on local React state and polling; LINE webhook–driven changes (clock/leave/OT) are invisible until hard refresh → perceived latency and silent failures.
- Multi-tab/device divergence: actions performed in one tab do not propagate to others, causing inconsistent UI state across devices.
- Missing optimistic UI + rollback: clock/leave/OT actions block on network with no instant feedback; failures leave UI stuck and require manual refresh.
- No idempotency key coordination between client and backend: retries can duplicate punches or approvals when combined with LINE webhook replays.
- Clock/location UX is fragile: GPS acquisition is not cancellable, and errors surface late (after user action), degrading perceived reliability.

## 2. Proposed change
Add a lightweight real-time + optimistic layer to the frontend:
- Introduce a shared Zustand store with cross-tab sync (BroadcastChannel) for tenant-scoped state (punches, leave/OT requests).
- Add optimistic mutations for clock and leave/OT actions with rollback on failure.
- Add a WebSocket/SSE client that subscribes to tenant events (fallback to short poll) and reconciles server state.
- Add an idempotency key generator (crypto.randomUUID) attached to each mutation and passed via header (`Idempotency-Key`).
- Scope: `src/stores/` (new), `src/hooks/useRealtime.ts`, `src/hooks/useOptimisticAction.ts`, update relevant components (`ClockButton`, `LeaveRequestForm`, `Dashboard`).

```ts
// src/stores/tenantStore.ts
import { create } from 'zustand';
import { persist, createJSONStorage } from 'zustand/middleware';

export interface Punch {
  id: string;
  type: 'IN' | 'OUT';
  ts: string;
  location?: { lat: number; lng: number };
  synced: boolean;
}

export interface LeaveRequest {
  id: string;
  type: 'ANNUAL' | 'SICK' | 'OTHER';
  start: string;
  end: string;
  status: 'PENDING' | 'APPROVED' | 'REJECTED';
  synced: boolean;
}

interface TenantState {
  punches: Punch[];
  leaveRequests: LeaveRequest[];
  addPunch: (p: Omit<Punch, 'id' | 'synced'>) => void;
  updatePunch: (id: string, patch: Partial<Punch>) => void;
  addLeaveRequest: (l: Omit<LeaveRequest, 'id' | 'synced'>) => void;
  updateLeaveRequest: (id: string, patch: Partial<LeaveRequest>) => void;
  reconcile: (server: { punches: Punch[]; leaveRequests: LeaveRequest[] }) => void;
}

export const useTenantStore = create<TenantState>()(
  persist(
    (set) => ({
      punches: [],
      leaveRequests: [],
      addPunch: (p) =>
        set((s) => ({
          punches: [
            ...s.punches,
            { ...p, id: crypto.randomUUID(), synced: false },
          ],
        })),
      updatePunch: (id, patch) =>
        set((s) => ({
          punches: s.punches.map((p) =>
            p.id === id ? { ...p, ...patch } : p
          ),
        })),
      addLeaveRequest: (l) =>
        set((s) => ({
          leaveRequests: [
            ...s.leaveRequests,
            { ...l, id: crypto.randomUUID(), synced: false },
          ],
        })),
      updateLeaveRequest: (id, patch) =>
        set((s) => ({
          leaveRequests: s.leaveRequests.map((l) =>
            l.id === id ? { ...l, ...patch } : l
          ),
        })),
      reconcile: (server) =>
        set((s) => {
          // merge strategy: server wins for synced=true; keep optimistic local-only entries
          const serverPunchIds = new Set(server.punches.map((p) => p.id));
          const localOnly = s.punches.filter((p) => !serverPunchIds.has(p.id));
          const mergedPunches = [...server.punches, ...localOnly];

          const serverLeaveIds = new Set(server.leaveRequests.map((l) => l.id));
          const localOnlyLeaves = s.leaveRequests.filter(
            (l) => !serverLeaveIds.has(l.id)
          );
          const mergedLeaves = [...server.leaveRequests, ...localOnlyLeaves];

          return { punches: mergedPunches, leaveRequests: mergedLeaves };
        }),
    }),
    {
      name: 'tenant-storage',
      storage: createJSONStorage(() => localStorage),
      partialize: (s) => ({ punches: s.punches, leaveRequests: s.leaveRequests }),
    }
  )
);
```

### Cross-tab sync via BroadcastChannel
```ts
// src/hooks/useCrossTabSync.ts
import { useEffect } from 'react';
import { useTenantStore } from '../stores/tenantStore';

export function useCrossTabSync() {
  useEffect(() => {
    const channel = new BroadcastChannel('workio_tenant_channel');
    channel.onmessage = (ev) => {
      if (ev.data.type === 'RECONCILE') {
        useTenantStore.getState().reconcile(ev.data.payload);
      }
    };

    // broadcast local optimistic changes to other tabs (optional)
    const unsub = useTenantStore.subscribe(
      (s) => ({ punches: s.punches, leaveRequests: s.leaveRequests }),
      (state) => {
        channel.postMessage({ type: 'RECONCILE', payload: state });
      },
      { equalityFn: (a, b) => JSON.stringify(a) === JSON.stringify(b) }
    );

    return () => {
      unsub();
      channel.close();
    };
  }, []);
}
```

### Realtime client (SSE fallback to polling)
```ts
// src/hooks/useRealtime.ts
import { useEffect } from 'react';
import { useTenantStore } from '../stores/tenantStore';

export function useRealtime(tenantId: string) {
  useEffect(() => {
    const controller = new AbortController();
    const source = new EventSource(
      `/api/tenants/${tenantId}/events`,
      { withCredentials: true }
    );

    source.onmessage = (ev) => {
      try {
        const data = JSON.parse(ev.data);
        if (data.type === 'STATE_SYNC') {
          useTenantStore.getState().reconcile(data.payload);
        }
      } catch {
        // ignore malformed
      }
    };

    source.onerror = () => {
      source.close();
      // fallback to polling
      startPolling();
    };

    function startPolling() {
      const poll = () => {
        if (controller.signal.aborted) return;
        fetch(`/api/tenants/${tenantId}/state`, {
          signal: controller.signal,
          credentials: 'include',
        })
          .then((r) => r.json())
          .then((json) =>
            useTenantStore.getState().reconcile(json)
          )
          .catch(() => {
            /* ignore */
          })
          .finally(() => setTimeout(poll, 15000));
      };
      poll();
    }

    return () => {
      controller.abort();
      source.close();
    };
  }, [tenantId]);
}
```
