## 1. Pre-list date folder → JSON manifest (Mac orchestration)
Single non-recursive `list_repo_tree` call per date folder → save `manifest-{date}.json` with CDN URLs. Embed in training script so Lightning workers perform **zero API calls** during data load.

**File:** `scripts/build_cdn_manifest.py`
```python
#!/usr/bin/env python3
"""
Usage: python build_cdn_manifest.py \
    --repo datasets/axentx/surrogate-1 \
    --date 2026-05-03 \
    --out manifest-2026-05-03.json
"""

import argparse
import json
import os
import time
from datetime import datetime, timezone
from huggingface_hub import HfApi

HF_TOKEN = os.getenv("HF_TOKEN")
CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def build_manifest(repo: str, date: str, out_path: str):
    api = HfApi(token=HF_TOKEN)
    folder = f"{date}/"
    print(f"Listing {repo}/{folder} ...")

    # single non-recursive call
    items = list(api.list_repo_tree(repo=repo, path=folder, recursive=False))
    files = [i for i in items if i.path.endswith(".parquet")]

    manifest = {
        "repo": repo,
        "date": date,
        "generated_at": datetime.now(timezone.utc).isoformat(),
        "files": [
            {
                "path": f.path,
                "cdn_url": CDN_TEMPLATE.format(repo=repo, path=f.path),
                "size": getattr(f, "size", None),
            }
            for f in files
        ],
    }

    os.makedirs(os.path.dirname(out_path) or ".", exist_ok=True)
    with open(out_path, "w") as f:
        json.dump(manifest, f, indent=2)

    print(f"Wrote {len(files)} files -> {out_path}")

if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", default="datasets/axentx/surrogate-1")
    parser.add_argument("--date", required=True, help="YYYY-MM-DD")
    parser.add_argument("--out", default="manifest.json")
    args = parser.parse_args()

    # gentle retry on 429
    for attempt in range(3):
        try:
            build_manifest(args.repo, args.date, args.out)
            break
        except Exception as e:
            if "429" in str(e) and attempt < 2:
                wait = 360
                print(f"Rate-limited, waiting {wait}s...")
                time.sleep(wait)
                continue
            raise
```

## 2. Training script: CDN-only data loader
Embed manifest; use `datasets` with `data_files` pointing to CDN URLs. No `list_repo_files` during training.

**File:** `surrogate/train.py` (excerpt)
```python
import json
from pathlib import Path
from datasets import load_dataset

def load_cdn_dataset(manifest_path: str, split="train"):
    with open(manifest_path) as f:
        manifest = json.load(f)

    cdn_urls = [f["cdn_url"] for f in manifest["files"]]
    if not cdn_urls:
        raise ValueError("No parquet files in manifest")

    # CDN-only; no HF API calls during data load
    ds = load_dataset(
        "parquet",
        name=manifest["date"],
        data_files={"train": cdn_urls},
        split=split,
    )
    # project to {prompt, response} only
    ds = ds.select_columns(["prompt", "response"])
    return ds
```

## 3. Lightning Studio lifecycle resilience
Reuse running studios; restart automatically if idle-stop killed.

**File:** `scripts/launch_surrogate_training.py`
```python
#!/usr/bin/env python3
"""
Launch Surrogate-1 training on Lightning with studio reuse + idle resilience.
"""

import lightning as L
from pathlib import Path

MANIFEST = Path("manifests/manifest-2026-05-03.json")
STUDIO_NAME = "surrogate-1-train"

def launch():
    teamspace = L.Teamspace()
    studio = None

    # reuse running studio
    for s in teamspace.studios:
        if s.name == STUDIO_NAME and s.status == "running":
            studio = s
            print(f"Reusing running studio: {s.id}")
            break

    if studio is None:
        print("Creating studio...")
        studio = teamspace.studios.create(
            name=STUDIO_NAME,
            machine=L.Machine.L40S,
            region="us-east-1",
        )

    # ensure running
    if studio.status != "running":
        print("Starting studio...")
        studio.start(machine=L.Machine.L40S)

    # run training (idempotent)
    run = studio.run(
        run_name="surrogate-1-cdn-train",
        entry_point="train.py",
        arguments=[
            "--manifest", str(MANIFEST),
            "--epochs", "1",
            "--batch-size", "8",
        ],
        requirements=["torch", "datasets", "lightning"],
    )
    print(f"Run started: {run.id}")
    return run

if __name__ == "__main__":
    launch()
```
