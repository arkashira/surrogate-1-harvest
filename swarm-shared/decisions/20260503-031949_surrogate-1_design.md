# surrogate-1 / design

## Final Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16`, `DATE`, `HF_TOKEN` via env  
- **Pre-generated manifest**: run once (locally or in CI) to produce `manifests/{DATE}.json` containing all file paths + sizes + md5. Worker loads this manifest (no `list_repo_tree` at runtime), eliminating HF API rate limits entirely.  
- Deterministic shard assignment: `hash(slug) % SHARD_TOTAL == SHARD_ID`  
- Downloads eligible files via **HF CDN bypass** (`https://huggingface.co/datasets/.../resolve/main/...`) with no Authorization header (public files).  
- Projects each file to `{prompt, response}` only at parse time (avoids pyarrow `CastError` from mixed schemas).  
- Deduplicates via central md5 store (`lib/dedup.py`).  
- Writes `batches/public-merged/{DATE}/shard{N}-{HHMMSS}.jsonl`.  
- Exits 0 on success; non-zero on fatal error (GitHub Actions will retry).

DEDUP_SCRIPT = Path(__file__).parent / "lib" / "dedup.py"


def is_duplicate(md5_hex: str) -> bool:
    """Query central dedup store via lib/dedup.py."""
    if not DEDUP_SCRIPT.exists():
        return False
    try:
        result = subprocess.run(
            [sys.executable, str(DEDUP_SCRIPT), "exists", md5_hex],
            capture_output=True,
            text=True,
            timeout=10,
        )
        return result.returncode == 0
    except Exception:
        # Fail open to avoid blocking ingestion
        return False


def register_md5(md5_hex: str) -> None:
    """Register md5 in central dedup store (best-effort)."""
    try:
        subprocess.run(
            [sys.executable, str(DEDUP_SCRIPT), "add", md5_hex],
            capture_output=True,
            timeout=5,
        )
    except Exception:
        pass

# ── manifest --
def load_manifest(path: str) -> list[Dict[str, Any]]:
    """Load manifest JSON: list of {path, size?, md5?}."""
    try:
        with open(path, "r", encoding="utf-8") as f:
            data = json.load(f)
    except Exception as exc:
        print(f"Failed to load manifest {path}: {exc}", file=sys.stderr)
        sys.exit(1)

    if not isinstance(data, list):
        print(f"Manifest must be a list, got {type(data)}", file=sys.stderr)
        sys.exit(1)

    entries = []
    for i, entry in enumerate(data):
        if isinstance(entry, dict):
            p = entry.get("path")
        elif isinstance(entry, str):
            p = entry
            entry = {"path": p}
        else:
            print(f"Invalid manifest entry at index {i}", file=sys.stderr)
            continue
        if not p or not isinstance(p, str):
            continue
        entries.append(entry)
    return entries


# ── CDN download --
def cdn_download(repo: str, path: str) -> bytes:
    """
    Download via HF CDN (no Authorization header) to bypass API rate limits.
    Public dataset files are accessible without token.
    """
    url = f"https://huggingface.co/datasets/{repo}/resolve/main/{path}"
    resp = requests.get(url, timeout=30)
    resp.raise_for_status()
    return resp.content


# ── parse helpers --
def normalize_pair(obj: Dict[str, Any]) -> Dict[str, str]:
    """
    Project arbitrary schema to {prompt, response}.
    Tolerates common field names.
    """
    prompt = obj.get("prompt") or obj.get("input") or obj.get("question") or ""
    response = obj.get("response") or obj.get("output") or obj.get("answer") or ""
    # Ensure strings
    return {"prompt": str(prompt), "response": str(response)}
