# workio / backend

## 1. Diagnosis
- **Tenant cache missing on `/webhook/line`** — every LINE event triggers a DB lookup for `channelId → tenant`; traffic spikes cause DB thrash and increase 429/timeout risk.
- **No request-scoped correlation IDs** — webhook → service → DB spans are untraceable; incidents require log grepping and slow RCA.
- **No structured logging** — plain `console.log` makes metrics/alerting and log-aggregation brittle.
- **No defense-in-depth signature verification** — relies only on Express middleware; replay/malformed payloads can reach handlers.
- **No readiness/liveness for k8s/PM2** — missing `/healthz` and `/readyz` endpoints complicates orchestration and zero-downtime deploys.

## 2. Proposed change
File: `/opt/axentx/workio/server/src/routes/webhook/line.ts` (or create if absent)  
Scope: add tenant cache (memory + optional Redis fallback), correlation ID middleware, structured logger, signature verification guard, and health endpoints in `/src/server.ts`.

### `/opt/axentx/workio/server/src/middleware/correlation.ts`
```ts
import { Request, Response, NextFunction } from 'express';
import { randomUUID } from 'crypto';
import { logger } from '../logger';

export function correlationMiddleware(req: Request, res: Response, next: NextFunction) {
  const cid = req.headers['x-correlation-id'] as string || randomUUID();
  (req as any).correlationId = cid;
  res.setHeader('X-Correlation-ID', cid);
  next();
}

export function requestLogger(req: Request, res: Response, next: NextFunction) {
  const start = Date.now();
  res.on('finish', () => {
    const duration = Date.now() - start;
    logger.info({
      cid: (req as any).correlationId,
      method: req.method,
      url: req.originalUrl,
      status: res.statusCode,
      durationMs: duration,
      ip: req.ip,
      userAgent: req.get('User-Agent')
    }, 'request');
  });
  next();
}
```

### `/opt/axentx/workio/server/src/cache/tenantCache.ts`
```ts
import { LRUCache } from 'lru-cache';
import { findTenantByChannelId } from '../db/repositories/tenantRepo';
import { logger } from '../logger';

const cache = new LRUCache<string, { tenantId: string; name: string; updatedAt: number }>({
  max: 500,
  ttl: 1000 * 60 * 10 // 10m
});

export async function resolveTenantId(channelId: string): Promise<string | null> {
  const cached = cache.get(channelId);
  if (cached) return cached.tenantId;

  try {
    const tenant = await findTenantByChannelId(channelId);
    if (!tenant) return null;
    cache.set(channelId, { tenantId: tenant.id, name: tenant.name, updatedAt: Date.now() });
    return tenant.id;
  } catch (err) {
    logger.error({ err, channelId }, 'tenantCache.resolveTenantId failed');
    return null;
  }
}

export function invalidateTenantCache(channelId: string) {
  cache.delete(channelId);
}
```

### `/opt/axentx/workio/server/src/middleware/lineSignature.ts`
```ts
import { Request, Response, NextFunction } from 'express';
import crypto from 'crypto';
import { logger } from '../logger';

export function lineSignatureGuard(channelSecret: string) {
  return (req: Request, res: Response, next: NextFunction) => {
    const sig = req.headers['x-line-signature'] as string;
    if (!sig) {
      logger.warn({ cid: (req as any).correlationId }, 'missing line signature');
      return res.status(401).json({ error: 'missing signature' });
    }

    const body = JSON.stringify(req.body);
    const expected = crypto
      .createHmac('sha256', channelSecret)
      .update(body, 'utf8')
      .digest('base64');

    if (!crypto.timingSafeEqual(Buffer.from(sig), Buffer.from(expected))) {
      logger.warn({ cid: (req as any).correlationId }, 'invalid line signature');
      return res.status(401).json({ error: 'invalid signature' });
    }
    next();
  };
}
```

### `/opt/axentx/workio/server/src/routes/webhook/line.ts`
```ts
import { Router } from 'express';
import { logger } from '../../logger';
import { correlationMiddleware, requestLogger } from '../../middleware/correlation';
import { lineSignatureGuard } from '../../middleware/lineSignature';
import { resolveTenantId } from '../../cache/tenantCache';
import { handleLineEvent } from '../../handlers/lineHandler';
import { config } from '../../config/env';

const router = Router();

router.use(correlationMiddleware);
router.use(requestLogger);
router.use(lineSignatureGuard(config.line.channelSecret));

router.post('/', async (req, res) => {
  const cid = (req as any).correlationId;
  const events = req.body.events || [];

  if (!events.length) {
    return res.status(200).json({ ok: true });
  }

  // All events in one webhook share same destination (channel)
  const channelId = req.body.destination;
  const tenantId = await resolveTenantId(channelId);

  if (!tenantId) {
    logger.warn({ cid, channelId }, 'tenant not found for channel');
    return res.status(404).json({ error: 'tenant not found' });
  }

  try {
    for (const evt of events) {
      await handleLineEvent({ ...evt, tenantId }, cid);
    }
    res.status(200).json({ ok: true });
  } catch (err) {
    logger.error({ cid, channelId, tenantId, err }, 'webhook processing failed');
    // LINE will retry on non-2xx
    res.status(500).json({ error: 'processing failed' });
  }
});

export default router;
```

### `/opt/axentx/workio/server/src/server.ts` (add health + mount)
```ts
import express from 'express';
import lineWebhook from './routes/webhook/line';
import { logger } from './logger';

const app = express();
app.use(express.json());

// Health endpoints (k8s/PM2 friendly)
app.get('/healthz', (_, res) => res.status(200).json({ status: 'ok', uptime: process.uptime() }));
app.get('/readyz', (_, res) => res.status(200).json({ status: 'ready' })); // extend with DB/redis checks later

// Webhook
app.use('/webhook/line', lineWebhook);

const PORT = process.env.PORT || 3000;
app.listen(PORT, () => {
  logger.info({ port: PORT }, 'server started');
});
```
