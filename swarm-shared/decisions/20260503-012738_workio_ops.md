# workio / ops

## 1. Diagnosis
- No automated health/readiness checks for LINE webhook or PostgreSQL → silent failures in production.
- No log rotation or structured logging → disk saturation and hard-to-debug incidents.
- No process supervision or graceful shutdown handling → Node crashes leave sockets hanging and DB connections leaking.
- No deploy-time smoke tests or rollback guardrails → bad deploys can persist for hours.
- No centralized env validation → missing/invalid `.env` surfaces only at runtime (crashes or misbehavior).

## 2. Proposed change
Add an ops-ready health/logging/supervision layer:
- `workio/server/src/ops/health.ts` — `/health` and `/ready` endpoints (LINE webhook + DB checks).
- `workio/server/src/ops/logger.ts` — structured pino logger with log rotation (100 MB, 7 files).
- `workio/server/src/ops/process.ts` — graceful shutdown (SIGTERM/SIGINT), in-flight request drain (30s), DB pool end.
- `workio/server/src/ops/validateEnv.ts` — fail-fast env validation at startup.
- Update `workio/server/src/index.ts` to wire ops modules and expose endpoints.
- Add `workio/server/ops/Procfile` and `workio/server/ops/ecosystem.config.js` for process supervision (PM2) and deploy smoke.

## 3. Implementation

### workio/server/src/ops/validateEnv.ts
```ts
import { config } from 'dotenv';
config();

const required = [
  'DATABASE_URL',
  'LINE_CHANNEL_ACCESS_TOKEN',
  'LINE_CHANNEL_SECRET',
  'PORT',
  'NODE_ENV',
];

export function validateEnv() {
  const missing = required.filter((k) => !process.env[k]);
  if (missing.length > 0) {
    console.error('Missing required env vars:', missing.join(', '));
    process.exit(1);
  }
}
```

import pino from 'pino';
import path from 'path';
import { fileURLToPath } from 'url';

const __filename = fileURLToPath(import.meta.url);
const __dirname = path.dirname(__filename);

const logDir = path.resolve(__dirname, '../../../logs');

export const logger = pino({
  level: process.env.LOG_LEVEL || 'info',
  formatters: {
    level(label) {
      return { level: label };
    },
  },
  transport: {
    target: 'pino/file',
    options: {
      destination: path.join(logDir, 'app.log'),
      mkdir: true,
    },
  },

import { Pool } from 'pg';
import { logger } from './logger.js';

export const pool = new Pool({
  connectionString: process.env.DATABASE_URL,
});

export async function checkDb(): Promise<boolean> {
  try {
    await pool.query('SELECT 1');
    return true;
  } catch (err) {
    logger.error({ err }, 'Database health check failed');
    return false;
  }

### workio/server/src/ops/health.ts
```ts
import express, { Request, Response } from 'express';
import axios from 'axios';
import { checkDb } from './db.js';
import { logger } from './logger.js';

const router = express.Router();

async function checkLine(): Promise<boolean> {
  try {
    const token = process.env.LINE_CHANNEL_ACCESS_TOKEN!;
    await axios.get('https://api.line.me/v2/bot/info', {
      headers: { Authorization: `Bearer ${token}` },
      timeout: 5000,
    });
    return true;
  } catch (err) {
    logger.warn({ err }, 'LINE health check failed');
    return false;
  }
}

router.get('/health', async (_req: Request, res: Response) => {
  // Liveness: process is alive
  res.json({ status: 'ok', uptime: process.uptime() });
});

router.get('/ready', async (_req: Request, res: Response) => {
  // Readiness: dependencies available
  const dbOk = await checkDb();
  const lineOk = await checkLine();

  if (dbOk && lineOk) {
    res.json({ status: 'ready' });
  } else {
    res.status(503).json({ status: 'not_ready', db: dbOk, line: lineOk });
  }
});

export default router;
```

### workio/server/src/ops/process.ts
```ts
import { logger } from './logger.js';
import { pool } from './db.js';

let shuttingDown = false;

export function setupGracefulShutdown(server: { close: (cb?: () => void) => void }) {
  const shutdown = async (signal: string) => {
    if (shuttingDown) return;
    shuttingDown = true;
    logger.info({ signal }, 'Shutdown signal received. Draining...');

    // Stop accepting new work
    server.close(async (err) => {
      if (err) logger.error({ err }, 'Error closing server');
      try {
        await pool.end();
        logger.info('DB pool closed');
        process.exit(err ? 1 : 0);
      } catch (e) {
        logger.error({ err: e }, 'Error during shutdown cleanup');
        process.exit(1);
      }
    });

    // Force exit after drain timeout
    setTimeout(() => {
      logger.warn('Forced shutdown after drain timeout');
      process.exit(1);
    }, 30_000).unref();
  };

  process.on('SIGTERM', () => shutdown('SIGTERM'));
  process.on('SIGINT', () => shutdown('SIGINT'));
}
```
