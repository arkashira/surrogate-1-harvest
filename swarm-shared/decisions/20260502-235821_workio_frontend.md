## 1. Core diagnosis (consensus)

- No optimistic UI or rollback → ambiguous spinners / duplicate retries when webhooks or API calls fail.  
- No offline/low-connectivity queue → punches are lost if network is poor.  
- No idempotency on client → double-taps or retries can create duplicate records.  
- No visual state distinction (pending / synced / failed) → users can’t tell if punch was recorded.  
- No persisted retry with exponential backoff → transient failures require manual retries and increase support load.

---

## 2. Architecture decisions (resolved)

- **Queue location & ownership**: single `ClockQueue` (persistent, retry, idempotent) in `src/lib/clockQueue.ts`.  
  - Uses localStorage (widely available, synchronous, good enough for queue durability across reloads).  
  - Optional Background Sync registration (best-effort) for browsers that support it.  
- **Idempotency scope**: **per punch type per day** (`tenant:user:date:type`).  
  - This prevents duplicate IN or OUT while still allowing an IN+OUT pair each day.  
  - If business rules require stricter uniqueness (e.g. one punch per N minutes), enforce that on the backend; frontend key is stable and safe.
- **Retry strategy**: exponential backoff + jitter, max 5 retries, serial processing to reduce thundering-herd and API bursts.
- **Optimistic UI**: immediate local state update; queue runs in background.  
  - Status observed by lightweight polling (or can be upgraded to event emitter later).  
  - Failed items remain in queue and surface “failed” so user can retry manually if desired.
- **Location**: accept lat/lng optionally from caller (geolocation can be added at call site); queue stores it when provided.

### 3.1 Clock queue (persistent, retry, idempotent)

```ts
// src/lib/clockQueue.ts
import { v4 as uuidv4 } from 'uuid';

const STORAGE_KEY = 'workio_clock_queue';
const MAX_RETRIES = 5;
const BASE_DELAY = 1000; // 1s

export type PunchType = 'IN' | 'OUT';
export type ClockStatus = 'pending' | 'synced' | 'failed';

export interface ClockRecord {
  id: string;
  punchIdempotencyKey: string; // tenant:user:date:type
  tenantId: string;
  userId: string;
  type: PunchType;
  timestamp: number; // local device time (ms)
  latitude?: number;
  longitude?: number;
  retries: number;
  status: ClockStatus;
  createdAt: number;
}

class ClockQueue {
  private queue: ClockRecord[] = [];
  private processing = false;

  constructor() {
    this.load();
    this.maybeRegisterBackgroundSync();
  }

  private load() {
    try {
      const raw = localStorage.getItem(STORAGE_KEY);
      this.queue = raw ? JSON.parse(raw) : [];
    } catch {
      this.queue = [];
    }
  }

  private save() {
    try {
      localStorage.setItem(STORAGE_KEY, JSON.stringify(this.queue));
    } catch {
      // ignore quota / private mode failures
    }
  }

  private maybeRegisterBackgroundSync() {
    if (!('serviceWorker' in navigator)) return;
    // Best-effort registration; not required for core functionality.
    void (async () => {
      try {
        const reg = await navigator.serviceWorker?.ready;
        if (reg && 'sync' in reg) {
          // @ts-ignore - Background Sync API availability varies
          await reg.sync.register('clock-retry').catch(() => {});
        }
      } catch {
        // ignore
      }
    })();
  }

  enqueue(record: Omit<ClockRecord, 'id' | 'retries' | 'status' | 'createdAt'>): ClockRecord {
    const item: ClockRecord = {
      ...record,
      id: uuidv4(),
      retries: 0,
      status: 'pending',
      createdAt: Date.now(),
    };
    this.queue.push(item);
    this.save();
    void this.process();
    return item;
  }

  updateStatus(id: string, status: ClockStatus) {
    const item = this.queue.find((q) => q.id === id);
    if (item) {
      item.status = status;
      if (status === 'synced') {
        this.queue = this.queue.filter((q) => q.id !== id);
      }
      this.save();
    }
  }

  getPending(): ClockRecord[] {
    return this.queue.filter((q) => q.status === 'pending');
  }

  getForDate(tenantId: string, userId: string, date: string): ClockRecord[] {
    return this.queue.filter(
      (q) =>
        q.tenantId === tenantId &&
        q.userId === userId &&
        new Date(q.timestamp).toISOString().slice(0, 10) === date
    );
  }

  getById(id: string): ClockRecord | undefined {
    return this.queue.find((q) => q.id === id);
  }

  async process() {
    if (this.processing) return;
    this.processing = true;

    try {
      const pending = this.getPending();
      for (const item of pending) {
        if (item.retries >= MAX_RETRIES) {
          this.updateStatus(item.id, 'failed');
          continue;
        }

        try {
          await this.submit(item);
          this.updateStatus(item.id, 'synced');
        } catch {
          item.retries += 1;
          this.save();
          const delay = BASE_DELAY * 2 ** item.retries + Math.random() * 500;
          // schedule next attempt and pause this loop
          await new Promise((resolve) => setTimeout(resolve, delay));
        }
      }
    } finally {
      this.processing = false;
    }
  }

  private async submit(record: ClockRecord): Promise<void> {
    const res = await fetch('/api/clock/punch', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        tenantId: record.tenantId,
        userId: record.userId,
        type: record.type,
        timestamp: record.timestamp,
        latitude: record.latitude,
        longitude: record.longitude,
        idempotencyKey: record.punchIdempotencyKey,
      }),
    });

    if (!res.ok) {
      const text = await res.text().catch(() => '');
      throw new Error(`Clock submit failed: ${res.status} ${text}`);
    }
  }
}

export const clockQueue = new ClockQueue();
```
