```tsx
// workio/src/components/ClockInOutButton.tsx
import React, { useState, useCallback, useRef } from 'react';
import { useMutation, useQueryClient } from '@tanstack/react-query';
import { clockIn, clockOut } from '../api/punches';
import type { Punch } from '../types';

type Mode = 'IN' | 'OUT';

export interface ClockInOutButtonProps {
  tenantId: string;
  employeeId: string;
  isClockedIn: boolean; // server truth
}

export default function ClockInOutButton({
  tenantId,
  employeeId,
  isClockedIn,
}: ClockInOutButtonProps) {
  const queryClient = useQueryClient();
  const [pending, setPending] = useState(false);
  const [error, setError] = useState<Error | null>(null);
  const inFlightRef = useRef<{ controller: AbortController; idempotencyKey: string } | null>(null);

  const mode: Mode = isClockedIn ? 'OUT' : 'IN';

  const punchMutation = useMutation({
    mutationFn: mode === 'IN' ? clockIn : clockOut,
    onMutate: async () => {
      setError(null);
      setPending(true);

      const idempotencyKey = `punch-${tenantId}-${employeeId}-${Date.now()}-${Math.random().toString(36).slice(2, 9)}`;
      const controller = new AbortController();

      // Deduplicate in-flight
      if (inFlightRef.current) {
        inFlightRef.current.controller.abort();
      }
      inFlightRef.current = { controller, idempotencyKey };

      const now = new Date();
      const optimisticPunch: Punch = {
        id: idempotencyKey,
        tenantId,
        employeeId,
        type: mode,
        timestamp: now.toISOString(),
        location: null,
        status: 'PENDING',
        createdAt: now.toISOString(),
        updatedAt: now.toISOString(),
      };

      await queryClient.cancelQueries({ queryKey: ['punches', tenantId, employeeId] });
      const previous = queryClient.getQueryData<Punch[]>(['punches', tenantId, employeeId]);
      queryClient.setQueryData<Punch[]>(['punches', tenantId, employeeId], (old) => [
        optimisticPunch,
        ...(old || []),
      ]);

      return { previous, idempotencyKey };
    },
    onError: (err, _variables, context) => {
      setError(err instanceof Error ? err : new Error(String(err)));
      if (context?.previous) {
        queryClient.setQueryData<Punch[]>(['punches', tenantId, employeeId], context.previous);
      }
    },
    onSettled: (_data, _error, _variables, context) => {
      setPending(false);
      if (inFlightRef.current) {
        inFlightRef.current.controller = null as any;
        inFlightRef.current = null;
      }
      // Keep optimistic item until server confirms, then invalidate to replace with canonical data
      queryClient.invalidateQueries({ queryKey: ['punches', tenantId, employeeId] });
    },
  });

  const handleClick = useCallback(() => {
    if (pending || inFlightRef.current) return;
    punchMutation.mutate();
  }, [punchMutation, pending]);

  return (
    <div className="flex flex-col items-center gap-2">
      <button
        onClick={handleClick}
        disabled={pending}
        className={`
          inline-flex items-center justify-center gap-2 px-6 py-3 rounded-lg font-semibold text-white transition-all
          ${pending ? 'opacity-80 cursor-wait' : ''}
          ${mode === 'IN'
            ? 'bg-green-600 hover:bg-green-700'
            : 'bg-red-600 hover:bg-red-700'
          }
        `}
        aria-busy={pending}
      >
        {pending ? (
          <>
            <span className="w-4 h-4 border-2 border-white border-t-transparent rounded-full animate-spin" />
            {mode === 'IN' ? 'กำลังลงเวลาเข้า...' : 'กำลังลงเวลาออก...'}
          </>
        ) : mode === 'IN' ? (
          <>
            <Clock className="w-4 h-4" />
            ลงเวลาเข้า
          </>
        ) : (
          <>
            <Clock className="w-4 h-4" />
            ลงเวลาออก
          </>
        )}
      </button>

      {error && (
        <div className="flex items-center gap-1 text-sm text-red-600">
          <AlertCircle className="w-4 h-4" />
          <span>เกิดข้อผิดพลาด กรุณาลองใหม่</span>
        </div>
      )}

      {pending && (
        <div className="flex items-center gap-1 text-sm text-gray-500">
          <Check className="w-4 h-4 opacity-50" />
          <span>กำลังบันทึก...</span>
        </div>
      )}
    </div>
  );
}
```

```ts
// workio/src/api/punches.ts
import axios from 'axios';

const api = axios.create();

export async function clockIn() {
  const { data } = await api.post('/api/punches', { type: 'IN' });
  return data;
}

export async function clockOut() {
  const { data } = await api.post('/api/punches', { type: 'OUT' });
  return data;
}
```
