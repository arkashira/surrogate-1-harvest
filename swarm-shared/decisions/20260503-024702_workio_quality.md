# workio / quality

## Final synthesized solution (best parts, resolved contradictions)

**Core principle**: enforce idempotency at the database layer (strongest guarantee) and make the client cooperate with a deterministic idempotency key and optimistic UI. Use a single, narrow idempotency table keyed by `(user_id, punch_date, punch_type, idempotency_key)` and upsert atomically in one statement. This prevents duplicates from webhook retries, client double-taps, and concurrent requests without fragile locking.

---

### 1. DB schema (migration)

```sql
-- workio/server/src/db/schema.sql
-- Punches table (add stricter uniqueness to prevent logical duplicates)
CREATE TABLE IF NOT EXISTS punches (
  id            SERIAL PRIMARY KEY,
  user_id       INTEGER NOT NULL,
  type          VARCHAR(10) NOT NULL, -- 'in' | 'out'
  created_at    TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Optional: prevent multiple punches of same type on same calendar day for same user
-- (relax or remove if business allows multiple punches per day)
CREATE UNIQUE INDEX IF NOT EXISTS uniq_user_day_type
  ON punches (user_id, DATE(created_at), type);

-- Idempotency table (narrow, fast, and safe for retries)
CREATE TABLE IF NOT EXISTS punch_idempotency (
  id            SERIAL PRIMARY KEY,
  user_id       INTEGER NOT NULL,
  punch_date    DATE NOT NULL,
  punch_type    VARCHAR(10) NOT NULL,
  idempotency_key VARCHAR(64) NOT NULL,
  punch_id      INTEGER NOT NULL,
  created_at    TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Strong uniqueness for idempotency (prevents duplicates from concurrent deliveries)
CREATE UNIQUE INDEX uniq_user_day_type_key
  ON punch_idempotency (user_id, punch_date, punch_type, idempotency_key);

-- FK to keep data consistent (optional but recommended)
ALTER TABLE punch_idempotency
  ADD CONSTRAINT fk_punch_id
  FOREIGN KEY (punch_id) REFERENCES punches(id) ON DELETE CASCADE;
```

Notes:
- Keep `punch_date` as DATE (not timestamp) so idempotency is scoped to the calendar day, matching the expected user behavior.
- Store `punch_id` in the idempotency row so we can return the exact punch on replay without extra lookups or nondeterministic `ORDER BY`.

### 2. Backend: atomic upsert route

Use a single `INSERT ... ON CONFLICT DO NOTHING` for the punch, then insert into idempotency. If idempotency already exists, return the stored punch immediately. This avoids races and extra round-trips.

```ts
// workio/server/src/routes/punch.ts
import { db } from '../db';
import { RowDataPacket } from 'mysql2/promise';

export async function punchHandler(req: any, res: any) {
  const { userId, type } = req.body; // type: 'in' | 'out'
  const idempotencyKey = (req.headers['x-idempotency-key'] as string) || '';
  if (!idempotencyKey) {
    return res.status(400).json({ error: 'Missing X-Idempotency-Key' });
  }
  if (!['in', 'out'].includes(type)) {
    return res.status(400).json({ error: 'Invalid type' });
  }

  const today = new Date().toISOString().slice(0, 10); // YYYY-MM-DD
  const conn = await db.getConnection();
  try {
    await conn.beginTransaction();

    // 1) Check idempotency first (fast path)
    const [existing] = await conn.query<RowDataPacket[]>(
      'SELECT punch_id FROM punch_idempotency WHERE user_id = ? AND punch_date = ? AND punch_type = ? AND idempotency_key = ?',
      [userId, today, type, idempotencyKey]
    );
    if (existing.length > 0 && existing[0].punch_id) {
      const [punch] = await conn.query<RowDataPacket[]>(
        'SELECT * FROM punches WHERE id = ?',
        [existing[0].punch_id]
      );
      await conn.commit();
      return res.json({ duplicate: true, punch: punch[0] });
    }

    // 2) Try to create punch (unique constraint prevents duplicates)
    const [result] = await conn.query(
      'INSERT INTO punches (user_id, type, created_at) VALUES (?, ?, NOW())',
      [userId, type]
    );
    const punchId = result.insertId;

    // 3) Record idempotency (unique index prevents races)
    await conn.query(
      'INSERT INTO punch_idempotency (user_id, punch_date, punch_type, idempotency_key, punch_id) VALUES (?, ?, ?, ?, ?)',
      [userId, today, type, idempotencyKey, punchId]
    );

    await conn.commit();
    return res.json({ duplicate: false, punchId });
  } catch (err: any) {
    await conn.rollback();

    // Duplicate punch (unique constraint) or duplicate idempotency (race)
    if (err.code === 'ER_DUP_ENTRY') {
      // Fetch the existing punch deterministically
      const [punch] = await conn.query<RowDataPacket[]>(
        'SELECT * FROM punches WHERE user_id = ? AND DATE(created_at) = ? AND type = ? ORDER BY created_at DESC LIMIT 1',
        [userId, today, type]
      );
      if (punch.length) {
        // Ensure idempotency row exists for future replays (best-effort)
        try {
          await conn.query(
            'INSERT IGNORE INTO punch_idempotency (user_id, punch_date, punch_type, idempotency_key, punch_id) VALUES (?, ?, ?, ?, ?)',
            [userId, today, type, idempotencyKey, punch[0].id]
          );
          await conn.commit();
        } catch {
          // ignore
        }
        return res.json({ duplicate: true, punch: punch[0] });
      }
    }

    res.status(500).json({ error: 'Server error' });
  } finally {
    conn.release();
  }
}
```

Why this is stronger:
- Atomic transaction prevents interleaving.
- Idempotency table stores the exact `punch_id`, removing nondeterministic lookups.
- Uses DB constraints as the source of truth; retries are safe even under concurrency.
- Gracefully handles `ER_DUP_ENTRY` by returning the existing punch and backfilling idempotency row.
