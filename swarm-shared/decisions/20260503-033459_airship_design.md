## Final Code Snippets

### 1) Generate CDN manifest (run once on Mac after rate-limit window)

`scripts/generate_manifest.py`
```python
#!/usr/bin/env python3
"""
Generate a CDN-only manifest for a single date folder.
Run on Mac (or any dev machine) when HF API limits allow.

Usage:
  HF_TOKEN=hf_xxx python scripts/generate_manifest.py \
    --repo surrogate-dataset \
    --date 2026-04-29 \
    --out manifest.json
"""

import argparse
import json
import os
import sys

from huggingface_hub import HfApi

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def main() -> None:
    parser = argparse.ArgumentParser(description="Generate CDN manifest for a date folder.")
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g. surrogate-dataset)")
    parser.add_argument("--date", required=True, help="Date folder (e.g. 2026-04-29)")
    parser.add_argument("--out", default="manifest.json", help="Output JSON path")
    parser.add_argument("--folder", default="batches/mirror-merged", help="Base folder in dataset")
    args = parser.parse_args()

    api = HfApi(token=os.getenv("HF_TOKEN"))
    base_path = f"{args.folder}/{args.date}"

    # Non-recursive list for the single date folder (avoids 100x pagination)
    try:
        entries = api.list_repo_tree(repo_id=args.repo, path=base_path, recursive=False)
    except Exception as e:
        print(f"HF API error: {e}", file=sys.stderr)
        sys.exit(1)

    files = [e.rfilename for e in entries if e.type == "file"]
    if not files:
        print(f"No files found at {base_path}", file=sys.stderr)
        sys.exit(1)

    manifest = {
        "repo": args.repo,
        "date": args.date,
        "base_path": base_path,
        "files": [
            {
                "path": f,
                "cdn_url": CDN_TEMPLATE.format(repo=args.repo, path=f),
                "size": next((e.size for e in entries if e.rfilename == f), None),
            }
            for f in sorted(files)
        ],
    }

    with open(args.out, "w") as fp:
        json.dump(manifest, fp, indent=2)

    print(f"Wrote {len(files)} files to {args.out}")

if __name__ == "__main__":
    main()
```

    def project_to_prompt_response(item):
        return {"prompt": "stub", "response": "stub"}

MANIFEST_PATH = os.getenv("MANIFEST_PATH", "manifest.json")
CACHE_DIR = Path(os.getenv("CACHE_DIR", tempfile.gettempdir())) / "surrogate_cdn"
CACHE_DIR.mkdir(parents=True, exist_ok=True)

class CDNIterableDataset(IterableDataset):
    def __init__(self, manifest_path: str):
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.files = self.manifest["files"]

    def __iter__(self):
        import requests

        for item in self.files:
            url = item["cdn_url"]
            local = CACHE_DIR / Path(item["path"]).name
            if not local.exists():
                # CDN download: no Authorization header required
                resp = requests.get(url, timeout=60)
                resp.raise_for_status()
                local.write_bytes(resp.content)

            # Project to {prompt, response} only at parse time
            # Supports mixed schemas: only read what we need
            try:
                # Minimal: assume parquet for this example
                import pandas as pd
                df = pd.read_parquet(local)
                for _, row in df.iterrows():
                    projected = project_to_prompt_response(dict(row))
                    yield projected
            except Exception as exc:
                print(f"Skipping {local}: {exc}")
                continue

class SurrogateDataModule(L.LightningDataModule):
    def __init__(self, manifest_path: str, batch_size: int = 8):
        super().__init__()
        self.manifest_path = manifest_path
        self.batch_size = batch_size

    def train_dataloader(self):
        return DataLoader(CDNIterableDataset(self.manifest_path), batch_size=self.batch_size)

class SurrogateTask(L.LightningModule):
    def __init__(self):
        super().__init__()
        self.model = SurrogateModel()
        self.loss_fn = torch.nn.CrossEntropyLoss()

    def training_step(self, batch, batch_idx):
        # Replace with real tokenization + forward
        x = torch.randn(self.batch_size, 1024)  # dummy
        y = torch.randint(0, 1024, (self.batch_size,))
        preds = self.model(x)
        loss = self.loss_fn(preds, y)
        self.log("train_loss", loss)
        return loss

    def configure_optimizers(self):
        return torch.optim.AdamW(self.parameters(), lr=1e-4)

if __name__ == "__main__":
    # Lightning Studio will run this script directly
    manifest = os.getenv("MANIFEST_PATH", "manifest.json")
    if not Path(manifest).exists():
        raise FileNotFoundError(f"Manifest not found: {manifest}")

    task = SurrogateTask()

    trainer = L.Trainer(
        max_epochs=1,
        accelerator="gpu" if torch.cuda.is_available() else "cpu",
        devices=1,
        enable_checkpointing=False,
        logger=False,
    )
    trainer.fit(task, dm)
