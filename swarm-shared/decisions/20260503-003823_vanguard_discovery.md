# vanguard / discovery

## Final Synthesized Answer

### 1. Diagnosis (Consolidated)
- **No persisted file manifest**: Every training run re-enumerates via authenticated HF API → quota burn + 429 risk.
- **Heterogeneous schema + recursive listing**: `list_repo_files`/`load_dataset(streaming=True)` on mixed repos triggers PyArrow casting errors and multiplies rate-limit exposure.
- **Training launcher wastes quota**: Likely recreates Lightning Studio instead of reusing running instances → burns 80+ hrs/mo free-tier quota.
- **No CDN-only data path**: Training still makes authenticated API calls during data loading instead of using public CDN URLs.
- **No deterministic repo routing**: Without per-repo commit cap (128/hr/repo) strategy, ingestion stalls when limit hit.

### 2. Proposed Change
Create `/opt/axentx/vanguard/manifest.py` + update launcher to:
- Persist `(repo, dateFolder)` file manifest via **one** authenticated `list_repo_tree` call.
- Embed manifest so training uses **CDN-only fetches** (zero HF API calls during data load).
- Reuse running Lightning Studio; create with `L40S` (free tier) if absent; restart on idle-stop.
- Add deterministic slug→repo routing (5 sibling repos) to spread HF commit load.

### 3. Implementation

# /opt/axentx/vanguard/manifest.py
#!/usr/bin/env python3
"""
Persist (repo, dateFolder) file manifest for CDN-only training.
Usage:
  python manifest.py build --repo datasets/xyz --date 2026-05-01 --out manifest.json
  python manifest.py train --manifest manifest.json --script train.py
"""
import argparse
import hashlib
import json
import os
import sys
from pathlib import Path

try:
    from huggingface_hub import list_repo_tree, HfApi
except ImportError:
    print("pip install huggingface_hub")
    sys.exit(1)

HF_API = HfApi()

def deterministic_repo(slug: str, n_siblings: int = 5) -> str:
    """Map slug -> sibling repo index to spread HF commit load."""
    idx = int(hashlib.sha256(slug.encode()).hexdigest(), 16) % n_siblings
    if idx == 0:
        return slug
    return f"{slug}-sibling-{idx}"

def build_manifest(repo: str, date_folder: str, out_path: Path):
    """
    Single authenticated call to list top-level date folder.
    Emits JSON list of CDN paths relative to repo root.
    """
    prefix = date_folder.rstrip("/") + "/"
    tree = list_repo_tree(repo=repo, path=prefix, recursive=False)
    files = [
        item.rfilename
        for item in tree
        if item.rfilename.endswith((".parquet", ".jsonl", ".json"))
    ]
    manifest = {
        "repo": repo,
        "date_folder": date_folder,
        "files": sorted(files),
        "cdn_base": f"https://huggingface.co/datasets/{repo}/resolve/main",
    }
    out_path.parent.mkdir(parents=True, exist_ok=True)
    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"Manifest written: {out_path} ({len(files)} files)")
    return manifest

def load_manifest(manifest_path: Path):
    return json.loads(manifest_path.read_text())

def generate_train_shim(manifest_path: Path, train_script: Path, out_script: Path):
    """
    Generate a lightweight train shim that:
    - loads manifest
    - streams files via CDN (no HF API auth during training)
    - projects to {prompt,response} at parse time
    """
    manifest = load_manifest(manifest_path)
    repo = manifest["repo"]
    cdn = manifest["cdn_base"]
    files = manifest["files"]

    shim = f"""#!/usr/bin/env python3
# Auto-generated by manifest.py — CDN-only training shim
import pyarrow.parquet as pq
import requests
import io
import os
import json
from typing import Iterator

REPO = "{repo}"
FILES = {json.dumps(files)}
CDN_BASE = "{cdn}"

def cdn_stream(file_rel: str) -> bytes:
    url = f"{{CDN_BASE}}/{{file_rel}}"
    resp = requests.get(url, timeout=30)
    resp.raise_for_status()
    return resp.content

def iter_rows() -> Iterator[dict]:
    for rel in FILES:
        raw = cdn_stream(rel)
        # project only required cols; ignore heterogeneous schema
        tbl = pq.read_table(io.BytesIO(raw), columns=["prompt", "response"], use_threads=False)
        df = tbl.to_pandas()
        for _, row in df.iterrows():
            yield {{"prompt": str(row.get("prompt", "")), "response": str(row.get("response", ""))}}

if __name__ == "__main__":
    # delegate to original training script logic
    import runpy
    os.environ["VANGUARD_MANIFEST"] = "{str(manifest_path)}"
    runpy.run_path(str({str(train_script)}), run_name="__main__")
"""
    out_script.parent.mkdir(parents=True, exist_ok=True)
    out_script.write_text(shim)
    out_script.chmod(0o755)
    print(f"Train shim written: {out_script}")

def main():
    parser = argparse.ArgumentParser(description="Vanguard manifest tool")
    sub = parser.add_subparsers(dest="cmd", required=True)

    build = sub.add_parser("build", help="Build manifest")
    build.add_argument("--repo", required=True, help="HF dataset repo (user/repo)")
    build.add_argument("--date", required=True, help="Date folder (YYYY-MM-DD)")
    build.add_argument("--out", default="manifest.json", help="Output JSON path")

    train = sub.add_parser("train", help="Generate CDN train shim")
    train.add_argument("--manifest", required=True, help="Manifest JSON path")
    train.add_argument("--script", required=True, help="Original train script path")
    train.add_argument("--out", default="train_shim.py", help="Output shim path")

    args = parser.parse_args()

    if args.cmd == "build":
        build_manifest(args.repo, args.date, Path(args.out))
    elif args.cmd == "train":
        generate_train_shim(Path(args.manifest), Path(args.script), Path(args.out))

if __name__ == "__main__":
    main()

# /opt/axentx/vanguard/lightning_launcher.py
#!/usr/bin/env python3
"""
Lightning launcher with studio reuse + idle restart.
"""
import os
import sys
import time

try:
    from lightning import Fabric, LightningWork, LightningFlow, LightningApp, Machine
    from lightning.app.teams import Teamspace
except ImportError:
    print("pip install lightning")
    sys.exit(1)

VANGUARD_TEAMSPACE = os.getenv("VANGUARD_TEAMSPACE", "default")
STUDIO_NAME = os.getenv("VANGUARD_STUDIO", "vanguard-train")
MACHINE = Machine.L40S  # free-tier fallback; H200 requires lightning-lambda-prod

def reuse_or_create_studio():
    ts = Teamspace(name=VANGUARD_TEAMSPACE)
    for s in ts.studios:
        if s.name == STUDIO_NAME:
            if s.status == "running":
                print(f"Reusing running studio: {STUDIO_NAME}")
                return s
            else:
                print(f"Restarting stopped studio: {STUDIO_NAME}")
                s.start(machine=MACHINE)
                return s
    print(f"Creating new studio: {STUDIO_NAME}")
    return ts.create_studio(name=STUDIO_NAME, machine=MACHINE, create_ok=True)

def run_training(train_shim: str):
    studio = reuse_or_create_studio()
    # Ensure studio is running before run
    if studio.status != "running":
        studio.start(machine=MACHINE)
        time.sleep(10)
    # Execute training shim inside studio context
    fabric = Fabric(devices=1, accelerator="cuda", precision="bf16-mixed")
    fabric.launch(train_shim)

if __name__ == "__main__":
    train_shim = sys.argv[1] if len(sys.argv) > 1 else "train_shim.py"
    run_training(train_shim)
