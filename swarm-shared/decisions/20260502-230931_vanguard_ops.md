# vanguard / ops

## Final Consolidated Implementation

**Core diagnosis (accepted from both):**
- No durable ingestion state → retries re-list and re-download, causing HF API 429s and wasted time.
- Training loads via HF API (no CDN bypass) → rate limits and quota exhaustion.
- No Studio reuse guard → `.run(create_ok=True)` spawns new studios, burning quota.
- No idempotent ops manifest → partial failures require manual cleanup.
- Missing pre-flight file manifest → training cannot guarantee CDN-only fetches.

**Single source of truth (chosen approach):**
- One lightweight JSON state file (`state.json`) for durable run metadata.
- One manifest generator that lists a date folder **once**, records CDN URLs, and persists `manifest-{date}.json`.
- Training launcher patched to accept a manifest and use CDN URLs only (no HF API calls during training).
- Studio reuse guard that lists running studios and reuses a running one before creating a new one.
- Idempotent ops wrapper so ingestion/training runs are tracked and resumable.

**Resolved contradictions in favor of correctness + actionability:**
- Use `huggingface_hub` for listing (simple, reliable) but **only once per folder**; persist results so training never calls HF API again.
- Studio reuse: prefer listing running studios and reusing; fall back to create only if none running (avoids quota burn).
- State is minimal JSON (not a database) to avoid extra dependencies and keep ops simple.
- Manifest contains CDN URLs and metadata; training code can stream via `fsspec`/`wget`/`requests` without HF client.
- Keep everything under `/opt/axentx/vanguard/ops/` for clear ownership and path stability.

### 2) `/opt/axentx/vanguard/ops/state.py`

```python
# /opt/axentx/vanguard/ops/state.py
import json
import hashlib
from datetime import datetime, timezone
from pathlib import Path

STATE_PATH = Path(__file__).parent / "state.json"

def _load():
    if STATE_PATH.exists():
        with open(STATE_PATH) as f:
            return json.load(f)
    return {}

def _save(state):
    with open(STATE_PATH, "w") as f:
        json.dump(state, f, indent=2)

def get(key, default=None):
    return _load().get(key, default)

def set(key, value):
    state = _load()
    state[key] = value
    _save(state)

def record_tree(repo, path, tree_json):
    blob = json.dumps(tree_json, sort_keys=True).encode()
    tree_hash = hashlib.sha256(blob).hexdigest()
    set("last_tree", {
        "repo": repo,
        "path": path,
        "tree_hash": tree_hash,
        "ts": datetime.now(timezone.utc).isoformat(),
        "tree": tree_json
    })
    return tree_hash

def last_tree_for(repo, path):
    last = get("last_tree")
    if last and last.get("repo") == repo and last.get("path") == path:
        return last
    return None

def record_run(run_id, kind, status, details=None):
    runs = get("runs", [])
    entry = {
        "run_id": run_id,
        "kind": kind,
        "status": status,
        "ts": datetime.now(timezone.utc).isoformat(),
        "details": details or {}
    }
    # replace if same run_id exists
    runs = [r for r in runs if r.get("run_id") != run_id] + [entry]
    set("runs", runs)
    return entry
```

### 3) `/opt/axentx/vanguard/ops/ingest_manifest.py`

```python
#!/usr/bin/env python3
# /opt/axentx/vanguard/ops/ingest_manifest.py
import argparse
import json
import os
import sys
from datetime import datetime, timezone
from pathlib import Path

try:
    from huggingface_hub import list_repo_tree
except ImportError:
    sys.exit("pip install huggingface_hub")

from .state import record_tree, last_tree_for, record_run

HF_REPO = os.getenv("HF_REPO", "datasets/axentx/surrogate-1")
OUT_DIR = Path(os.getenv("OUT_DIR", Path(__file__).parent / "manifests"))
OUT_DIR.mkdir(exist_ok=True, parents=True)

def build_manifest(repo: str, folder: str, out_dir: Path):
    # List non-recursive to avoid huge payloads; caller should provide specific date folder
    tree = list_repo_tree(repo, recursive=False, path=folder)
    entries = getattr(tree, "entries", []) or []

    files = []
    for entry in entries:
        if entry.type != "file":
            continue
        cdn_url = f"https://huggingface.co/datasets/{repo}/resolve/main/{folder}/{entry.path}"
        files.append({
            "path": f"{folder}/{entry.path}",
            "cdn_url": cdn_url,
            "size": getattr(entry, "size", None),
            "lfs": getattr(entry, "lfs", None)
        })

    manifest = {
        "repo": repo,
        "folder": folder,
        "generated_at": datetime.now(timezone.utc).isoformat(),
        "files": files
    }

    # Record state
    record_tree(repo, folder, [{"path": e["path"], "type": "file"} for e in files])
    record_run(
        run_id=f"ingest-{folder.rstrip('/').replace('/', '-')}",
        kind="ingest",
        status="completed",
        details={"repo": repo, "folder": folder, "file_count": len(files)}
    )

    out_path = out_dir / f"manifest-{folder.rstrip('/').replace('/', '-')}.json"
    with open(out_path, "w") as f:
        json.dump(manifest, f, indent=2)

    print(json.dumps({"manifest": str(out_path), "files": len(files)}))
    return manifest

if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", default=HF_REPO)
    parser.add_argument("--folder", default=os.getenv("DATE_FOLDER", ""))
    parser.add_argument("--out-dir", default=OUT_DIR, type=Path)
    args = parser.parse_args()

    if not args.folder:
        sys.exit("Provide DATE_FOLDER or --folder (e.g. batches/mirror-merged/2026-05-02)")

    build_manifest(args.repo, args.folder, args.out_dir)
```

Make executable:

#!/usr/bin/env python3
# /opt/axentx/vanguard/ops/run_lightning.py
import json
import os
import sys
from pathlib import Path

MANIFEST_PATH = os.getenv("MANIFEST_PATH")
if not MANIFEST_PATH or not Path(MANIFEST_PATH).exists():
    sys.exit("Set MANIFEST_PATH to a generated manifest JSON")

with open(MANIFEST_PATH) as f:
    manifest = json.load(f)

cdn_urls = [f["cdn_url"] for f in manifest["files"]]
print(f"Loaded {len(cdn_urls)} CDN URLs from manifest")

# Optional: lightweight reuse guard for Lightning Studio
def reuse_or_create_studio(name="vanguard-training", machine="lightning-public-prod"):
    try:
        import lightning as L
    except ImportError:
        sys.exit("pip install lightning")

    try:
        studios = L.Teamspace.studios()
        for s in studios:
            if s.name == name and s.status == "Running":
                print(f"Reusing running studio: {s.id}")
                return s
    except Exception as e:
        print(f"Studio listing failed (may need auth): {e}")

    print(f"Creating studio {name} (machine={machine})")
    return L.Studio(
        name=name,
        machine=machine,
        create_ok=True
    )
