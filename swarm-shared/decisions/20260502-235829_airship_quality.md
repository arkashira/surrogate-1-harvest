## Final Synthesis: CDN-Only Training Pipeline (Zero HF API During Training)

**Core Problem**: Surrogate-1 training is blocked by HF API 429 errors during dataset ingestion.  
**Goal**: ≤2h fix that eliminates API calls during training, enables continuous runs, and wastes zero quota.

**Resolution of Contradictions**:
- **Manifest generation**: Use a *single* `list_repo_tree` call per date folder (respects rate limits) rather than per-file calls.
- **CDN vs HF download**: Prefer direct CDN fetches during training (bypasses HF API entirely). Fallback to `hf_hub_download` only for cache misses or private repos.
- **Schema handling**: Project to `{prompt, response}` at parse time with flexible column mapping, not rigid schema assumptions.
- **Studio reuse**: Prevent duplicate studio spawns to save quota; do not start new runs if one is already active.

```python
# scripts/generate_cdn_manifest.py
#!/usr/bin/env python3
"""
Generate CDN-only manifest for surrogate-1 dataset.
Uses single API call per folder, then CDN URLs for training.
"""
import json
import os
from datetime import datetime
from pathlib import Path
from huggingface_hub import HfApi, list_repo_tree

API = HfApi()
REPO = "axentx/surrogate-1"
OUTPUT_DIR = Path("manifests")
OUTPUT_DIR.mkdir(exist_ok=True)

def generate_manifest(date_folder: str = None):
    """Generate manifest for a date folder (e.g., '2026-04-29')."""
    if date_folder is None:
        date_folder = datetime.utcnow().strftime("%Y-%m-%d")
    
    print(f"📋 Listing repo tree for {date_folder}...")
    try:
        tree = list_repo_tree(
            repo_id=REPO,
            path=f"batches/mirror-merged/{date_folder}",
            recursive=True,
            repo_type="dataset"
        )
    except Exception as e:
        print(f"⚠️  API call failed: {e}")
        print("💡 Tip: Wait 360s after 429, or use existing manifest")
        return None
    
    files = [f for f in tree if f.path.endswith('.parquet')]
    print(f"📄 Found {len(files)} parquet files")
    
    manifest = {
        "repo": REPO,
        "date": date_folder,
        "generated_at": datetime.utcnow().isoformat(),
        "files": [],
        "total_files": len(files)
    }
    
    for f in files:
        # CDN URL - no auth, no API rate limit
        cdn_url = f"https://huggingface.co/datasets/{REPO}/resolve/main/{f.path}"
        manifest["files"].append({
            "path": f.path,
            "size": f.size,
            "cdn_url": cdn_url,
            "slug": Path(f.path).stem
        })
    
    output_path = OUTPUT_DIR / f"surrogate-1-{date_folder}.json"
    with open(output_path, 'w') as fp:
        json.dump(manifest, fp, indent=2)
    
    print(f"✅ Manifest saved to {output_path}")
    print(f"🚀 Training can now use CDN-only (zero API calls)")
    return output_path

if __name__ == "__main__":
    import sys
    date = sys.argv[1] if len(sys.argv) > 1 else None
    generate_manifest(date)
```

### 2. CDN-Aware Dataset Loader

```python
# surrogate/data/dataset.py
import json
import pyarrow.parquet as pq
from pathlib import Path
from typing import Iterator, Dict
import requests
from huggingface_hub import hf_hub_download

class CDNParquetDataset:
    """Dataset loader using CDN URLs (zero HF API calls during training)."""
    
    def __init__(self, manifest_path: str, use_cdn: bool = True):
        self.manifest_path = Path(manifest_path)
        with open(self.manifest_path) as f:
            self.manifest = json.load(f)
        self.use_cdn = use_cdn
        self.cache_dir = Path(".cache/hf")
        self.cache_dir.mkdir(parents=True, exist_ok=True)
    
    def _download_via_cdn(self, cdn_url: str, local_path: Path) -> Path:
        """Download via CDN (bypasses HF API rate limits)."""
        if local_path.exists():
            return local_path
        
        local_path.parent.mkdir(parents=True, exist_ok=True)
        response = requests.get(cdn_url, timeout=30)
        response.raise_for_status()
        local_path.write_bytes(response.content)
        return local_path
    
    def _download_via_hf(self, repo_path: str) -> Path:
        """Fallback: download via HF hub (uses API but respects caching)."""
        return hf_hub_download(
            repo_id=self.manifest["repo"],
            filename=repo_path,
            repo_type="dataset",
            cache_dir=str(self.cache_dir)
        )
    
    def __iter__(self) -> Iterator[Dict[str, str]]:
        for file_info in self.manifest["files"]:
            try:
                if self.use_cdn:
                    # CDN bypass - no Authorization header, separate rate limits
                    local_path = self.cache_dir / file_info["path"].replace("/", "_")
                    parquet_path = self._download_via_cdn(file_info["cdn_url"], local_path)
                else:
                    parquet_path = self._download_via_hf(file_info["path"])
                
                # Load and project to {prompt, response} only
                table = pq.read_table(parquet_path)
                
                # Handle mixed schemas gracefully
                for batch in table.to_batches():
                    cols = set(batch.schema.names)
                    
                    # Extract prompt/response regardless of schema variations
                    prompt_col = next((c for c in ["prompt", "input", "question"] if c in cols), None)
                    response_col = next((c for c in ["response", "output", "answer"] if c in cols), None)
                    
                    if prompt_col and response_col:
                        for i in range(batch.num_rows):
                            yield {
                                "prompt": batch[prompt_col][i].as_py(),
                                "response": batch[response_col][i].as_py(),
                                "source_file": file_info["slug"]
                            }
                            
            except Exception as e:
                print(f"⚠️  Failed to process {file_info['path']}: {e}")
                continue
    
    def __len__(self):
        return self.manifest["total_files"]
```
