## Final Implementation (Synthesized)

**Core Strategy**: Replace all Hugging Face API calls during training with a **manifest-driven, CDN-only data pipeline**. This eliminates rate limits and bypasses mixed-schema ingestion errors by projecting to `{prompt, response}` at parse time.

---

### 1. Manifest Generator (`scripts/build_file_manifest.py`)
Run on Mac orchestrator after rate-limit window clears. Uses non-recursive tree listing to minimize API hits.

```python
#!/usr/bin/env python3
import argparse, json, os, time
from huggingface_hub import HfApi

def build_manifest(repo: str, folder: str, out_path: str) -> None:
    items = HfApi().list_repo_tree(repo=repo, path=folder, recursive=False)
    files = [{"path": i.path, "size": getattr(i, "size", None)} 
             for i in items if i.type == "file"]
    
    manifest = {
        "repo": repo, "folder": folder,
        "generated_at": time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime()),
        "files": files
    }
    os.makedirs(os.path.dirname(os.path.abspath(out_path)) or ".", exist_ok=True)
    with open(out_path, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Wrote {len(files)} files to {out_path}")

if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True)
    parser.add_argument("--folder", required=True)
    parser.add_argument("--out", default="file_manifest.json")
    build_manifest(**vars(parser.parse_args()))
```

### 2. CDN Dataset Loader (`surrogate/data/cdn_dataset.py`)
Lightning-compatible `IterableDataset` that streams from public CDN URLs. Projects schema at parse time; retries on CDN failures.

```python
import json, logging, time
from pathlib import Path
from typing import Dict, Iterable
import pandas as pd
import pyarrow as pa
import pyarrow.parquet as pq
import requests
from torch.utils.data import IterableDataset

logger = logging.getLogger(__name__)
CDN = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

class CDNIterableDataset(IterableDataset):
    def __init__(self, manifest_path: str, repo: str = None,
                 max_retries: int = 5, backoff: float = 1.5):
        manifest = json.loads(Path(manifest_path).read_text())
        self.repo = repo or manifest["repo"]
        self.files = [f["path"] for f in manifest["files"]]
        self.max_retries, self.backoff = max_retries, backoff

    def _stream_file(self, path: str) -> Iterable[Dict[str, str]]:
        url = CDN.format(repo=self.repo, path=path)
        for attempt in range(1, self.max_retries + 1):
            try:
                r = requests.get(url, timeout=30)
                r.raise_for_status()
                tbl = pq.read_table(pa.BufferReader(r.content))
                cols = [c for c in ("prompt", "response") if c in tbl.column_names]
                if not cols:
                    logger.warning("No prompt/response in %s", path)
                    return
                for _, row in tbl.select(cols).to_pandas().iterrows():
                    yield {k: str(v) if not pd.isna(v) else "" for k, v in row.items()}
                return
            except Exception as e:
                wait = self.backoff ** attempt
                logger.warning("Failed %s (attempt %s): %s — retry in %.1fs", path, attempt, e, wait)
                time.sleep(wait)
        logger.error("Exhausted retries for %s", path)

    def __iter__(self) -> Iterable[Dict[str, str]]:
        for path in self.files:
            yield from self._stream_file(path)
```

### 3. Training Entrypoint (`train.py`)
```python
from surrogate.data.cdn_dataset import CDNIterableDataset
from torch.utils.data import DataLoader

def build_dataloader(manifest_path: str, batch_size: int = 8, num_workers: int = 4):
    return DataLoader(CDNIterableDataset(manifest_path), 
                      batch_size=batch_size, num_workers=num_workers)

# CLI: python train.py --manifest file_manifest.json
```
