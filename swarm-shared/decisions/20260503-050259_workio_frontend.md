# workio / frontend

## Final Synthesized Solution

### 1. Diagnosis (merged + resolved)
- **Duplicate submissions**: rapid taps, retry on slow networks, and missing inflight locks cause duplicate events that corrupt daily totals/OT/leave balances.
- **Race conditions**: no client-side request deduplication, no inflight tracking, and no optimistic state.
- **Server-only validation is too late**: GPS freshness and clock-in/out window rules (e.g., no double clock-in within 5 min) must be checked locally before submitting.
- **Poor UX**: button does not show loading/disabled state; users perceive lag and retry; no immediate feedback or safe rollback on failure.
- **Clock skew**: device/server time differences can reorder timelines; server timestamp should be canonical after success.
- **Missing resiliency**: no idempotency, no retry with backoff, no clear handling of duplicates (409).

### 2. Architecture (single source of truth)
- **Frontend**: one `useAttendance` hook that owns inflight lock, idempotency key, optimistic state, validation, and rollback.
- **API layer**: lightweight `withIdempotency` wrapper that adds `Idempotency-Key`, retries transient errors, and treats `409` as success (duplicate accepted).
- **UI component**: `AttendanceControls` consumes hook, shows clear loading/disabled states, validation errors, success confirmations, and retry affordances.

### 3. Implementation

#### `workio/src/lib/api.ts`
```ts
import axios from 'axios';

export const api = axios.create({
  baseURL: import.meta.env.VITE_API_BASE_URL || '/api',
  timeout: 15000,
});

export function withIdempotency<T = any>(
  request: () => Promise<T>,
  idempotencyKey: string,
  retries = 1,
  backoffMs = 800
): Promise<T> {
  return request().catch(async (err) => {
    // Server accepted duplicate — treat as success
    if (err.response?.status === 409) {
      return err.response.data as T;
    }
    // Network/transient failure: retry with backoff
    if (retries > 0 && (!err.response || err.code === 'ECONNABORTED' || err.code === 'ERR_NETWORK')) {
      await new Promise((r) => setTimeout(r, backoffMs));
      return withIdempotency(request, idempotencyKey, retries - 1, backoffMs * 2);
    }
    throw err;
  });
}
```

#### `workio/src/hooks/useAttendance.ts`
```ts
import { useState, useCallback, useRef } from 'react';
import { api, withIdempotency } from '../lib/api';

type AttendanceAction = 'CLOCK_IN' | 'CLOCK_OUT';

interface AttendanceState {
  status: 'idle' | 'loading' | 'success' | 'error';
  lastAction?: AttendanceAction;
  lastTimestamp?: string;
  error?: string;
  validationError?: string;
}

const MIN_WINDOW_MINUTES = 5;

function nowISO() {
  return new Date().toISOString();
}

function minutesBetween(a: string, b: string) {
  return Math.abs(new Date(a).getTime() - new Date(b).getTime()) / 60000;
}

export function useAttendance({ lastServerEvent }: { lastServerEvent?: { action: AttendanceAction; timestamp: string } | null } = {}) {
  const [state, setState] = useState<AttendanceState>({ status: 'idle' });
  const inflightRef = useRef<AbortController | null>(null);

  const generateIdempotencyKey = useCallback((action: AttendanceAction) => {
    const day = new Date().toISOString().slice(0, 10); // YYYY-MM-DD
    const rand = Math.random().toString(36).slice(2, 9);
    return `${action}:${day}:${rand}`;
  }, []);

  const validateLocally = useCallback(
    (action: AttendanceAction): string | null => {
      if (!lastServerEvent) return null;

      const sameActionTooSoon =
        lastServerEvent.action === action &&
        minutesBetween(lastServerEvent.timestamp, nowISO()) < MIN_WINDOW_MINUTES;

      if (sameActionTooSoon) {
        return `Cannot ${action === 'CLOCK_IN' ? 'clock in' : 'clock out'} again within ${MIN_WINDOW_MINUTES} minutes.`;
      }

      return null;
    },
    [lastServerEvent]
  );

  const perform = useCallback(
    async (action: AttendanceAction) => {
      if (inflightRef.current) return;

      const validationError = validateLocally(action);
      if (validationError) {
        setState({ status: 'error', validationError, lastAction: action });
        return;
      }

      const abortController = new AbortController();
      inflightRef.current = abortController;
      const idempotencyKey = generateIdempotencyKey(action);
      const optimisticTs = nowISO();

      setState({ status: 'loading', lastAction: action, lastTimestamp: optimisticTs, validationError: undefined });

      try {
        const result = await withIdempotency(
          () =>
            api.post(
              '/attendance/record',
              { action, timestamp: optimisticTs },
              {
                headers: { 'Idempotency-Key': idempotencyKey },
                signal: abortController.signal,
              }
            ),
          idempotencyKey,
          1
        );

        // Use server timestamp when available to avoid skew
        const serverTs = result.data?.timestamp || optimisticTs;
        setState({ status: 'success', lastAction: action, lastTimestamp: serverTs });
      } catch (err: any) {
        if (err.name !== 'AbortError') {
          setState({
            status: 'error',
            lastAction: action,
            lastTimestamp: optimisticTs,
            error: err.message || 'Failed to record attendance',
          });
        }
      } finally {
        inflightRef.current = null;
      }
    },
    [generateIdempotencyKey, validateLocally]
  );

  const resetError = useCallback(() => {
    setState((s) => (s.status === 'error' ? { status: 'idle' } : s));
  }, []);

  return {
    state,
    clockIn: () => perform('CLOCK_IN'),
    clockOut: () => perform('CLOCK_OUT'),
    resetError,
    isLoading: state.status === 'loading',
  };
}
```
