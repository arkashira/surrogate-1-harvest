# workio / backend

### 1. Diagnosis (merged)
- **Signature verification missing** — `/webhook/line` accepts unverified payloads, enabling spoofing/replay.
- **Idempotency missing** — LINE retries or user double-taps create duplicate clock-in/out, leave, or OT records.
- **Raw body destroyed** — `express.json()` mutates `req.body`, removing the buffer required for HMAC verification.
- **Tenant context not enforced** — tenant inferred from payload fields rather than validated against a tenant-scoped channel secret, risking cross-tenant injection.
- **Replay window absent** — events can be replayed arbitrarily far in past/future.
- **Error handling/logging weak** — crashes or data corruption possible; critical events not logged.

### 2. Single proposed change
Add a hardened `/webhook/line` handler that:
1. Preserves raw body for HMAC-SHA256 verification.
2. Verifies `X-Line-Signature` against a tenant-scoped channel secret.
3. Enforces a 5-minute replay window using `event.timestamp`.
4. Deduplicates by `event.deliveryId` (or derived stable key) with a tenant-scoped, TTL-backed store.
5. Returns `401` (invalid signature), `400` (stale/malformed), `409` (duplicate), `200` (accepted).
6. Logs critical events and errors; fails safely without crashing.

### 3. Implementation (single file)
File: `server/src/routes/webhook/line.ts` (or `.js` equivalent).  
Dependencies: Redis (preferred) or DB for multi-instance safety; fallback to in-memory for single-instance dev.

```ts
// server/src/routes/webhook/line.ts
import { Router, Request, Response, NextFunction } from 'express';
import crypto from 'crypto';
import Redis from 'ioredis';
import { v4 as uuidv4 } from 'uuid';
import winston from 'winston';

const router = Router();

// Logger
const logger = winston.createLogger({
  level: 'info',
  format: winston.format.json(),
  transports: [new winston.transports.Console()],
});

// Redis client (configure URL via env)
const redis = new Redis(process.env.REDIS_URL || 'redis://localhost:6379');

// Preserve raw body for HMAC
export function preserveRawBody(req: Request, _res: Response, buf: Buffer) {
  (req as any).rawBody = buf;
}

// HMAC-SHA256 verification
function verifyLineSignature(rawBody: Buffer, signature: string, channelSecret: string): boolean {
  if (!signature || !channelSecret) return false;
  try {
    const expected = crypto.createHmac('sha256', channelSecret).update(rawBody).digest('base64');
    return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
  } catch {
    return false;
  }
}

// Tenant-scoped channel secret lookup (example)
async function getChannelSecret(tenantId: string): Promise<string | null> {
  // Replace with your tenant lookup (DB/env/secret manager)
  // For now, assume one secret from env; extend to tenant-scoped lookup.
  return process.env.LINE_CHANNEL_SECRET || null;
}

// Deduplication (Redis-backed)
const DEDUP_TTL_SECONDS = 60 * 10; // 10 minutes
async function isDuplicate(tenantId: string, eventId: string): Promise<boolean> {
  if (!eventId) return false;
  const key = `webhook:line:dedup:${tenantId}:${eventId}`;
  const set = await redis.set(key, '1', 'PX', DEDUP_TTL_SECONDS * 1000, 'NX');
  return set !== 'OK'; // already exists
}

// Replay window (ms)
const REPLAY_WINDOW_MS = 5 * 60 * 1000; // 5 minutes

// Core handler
router.post(
  '/line',
  preserveRawBody,
  async (req: Request, res: Response, next: NextFunction) => {
    const requestId = uuidv4();
    const signature = req.headers['x-line-signature'] as string;
    const rawBody = (req as any).rawBody as Buffer;

    // Basic body validation
    if (!rawBody || !req.body?.events || !Array.isArray(req.body.events)) {
      logger.warn({ requestId, msg: 'Invalid payload', body: req.body });
      return res.status(400).json({ error: 'Invalid payload' });
    }

    // Determine tenant (example: from header or payload)
    // Prefer header for early binding; fallback to payload with strict validation.
    const tenantId = (req.headers['x-tenant-id'] as string) || req.body.tenantId;
    if (!tenantId) {
      logger.warn({ requestId, msg: 'Missing tenant' });
      return res.status(400).json({ error: 'Missing tenant' });
    }

    const channelSecret = await getChannelSecret(tenantId);
    if (!channelSecret) {
      logger.error({ requestId, tenantId, msg: 'Channel secret missing' });
      return res.status(500).json({ error: 'Server configuration error' });
    }

    // Signature verification
    if (!verifyLineSignature(rawBody, signature, channelSecret)) {
      logger.warn({ requestId, tenantId, msg: 'Invalid signature' });
      return res.status(401).json({ error: 'Invalid signature' });
    }

    const body = req.body;
    const now = Date.now();
    const results: Array<{ status: number; error?: string; eventType?: string }> = [];

    for (const event of body.events) {
      try {
        // Event shape validation
        if (!event?.type || !event.timestamp || !event.source) {
          results.push({ status: 400, error: 'Malformed event' });
          continue;
        }

        // Replay protection
        if (Math.abs(now - event.timestamp) > REPLAY_WINDOW_MS) {
          results.push({ status: 400, error: 'Stale event' });
          continue;
        }

        // Idempotency
        const eventId = event.deliveryId || event.message?.id || `${event.source.userId}-${event.timestamp}-${event.type}`;
        if (await isDuplicate(tenantId, eventId)) {
          results.push({ status: 409, error: 'Duplicate event' });
          continue;
        }

        // Business handling (example)
        switch (event.type) {
          case 'clock_in':
          case 'clock_out':
            // await insertAttendance({ tenantId, event, requestId });
            logger.info({ requestId, tenantId, eventType: event.type, eventId, msg: 'Attendance processed' });
            results.push({ status: 200, eventType: event.type });
            break;
          case 'leave_request':
            // await insertLeaveRequest({ tenantId, event, requestId });
            logger.info({ requestId, tenantId, eventType: event.type, eventId, msg: 'Leave request processed' });
            results.push({ status: 200, eventType: event.type });
            break;
          case 'ot_request':
            // await insertOtRequest({ tenantId, event, requestId });
            logger.info({ requestId, tenantId, eventType: event.type, eventId, msg: 'OT request processed' });
            results.push({ status: 200, eventType: event.type });
            break;
          default:
            results.push({ status: 400, error: 'Unsupported event type' });
            break;
        }
      } catch (err) {
        logger.error({ requestId, tenantId, event, error: err instanceof Error ? err.message : err, msg: 'Event processing failed' });
        results.push({ status: 500, error: 'Internal Server Error' });
      }
    }

    // If any event-level failure, return multi-status summary; otherwise 200.
    const hasError = results.some((r) => r.status >= 400);
    return res.status(hasError ? 207 : 200).json({ requestId, results });
  }
);

export default router;
```
