```python
#!/usr/bin/env python3
"""
Generate CDN file list for a HuggingFace dataset repo.
Usage:
    python scripts/list_hf_files.py --repo <org/repo> --date 2026-04-29 --out filelist.json
"""
import argparse
import json
import os
from huggingface_hub import HfApi

def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True, help="HF dataset repo (org/name)")
    parser.add_argument("--date", required=True, help="Date folder (YYYY-MM-DD)")
    parser.add_argument("--out", default="filelist.json", help="Output JSON path")
    args = parser.parse_args()

    api = HfApi()
    folder = args.date
    # Non-recursive to avoid pagination explosion; CDN downloads are not rate-limited.
    entries = api.list_repo_tree(repo_id=args.repo, path=folder, recursive=False)

    files = [e.rfilename for e in entries if e.rfilename.endswith(".parquet")]
    payload = {
        "repo": args.repo,
        "date": folder,
        "files": sorted(files),
    }

    os.makedirs(os.path.dirname(os.path.abspath(args.out)), exist_ok=True)
    with open(args.out, "w") as f:
        json.dump(payload, f, indent=2)
    print(f"Wrote {len(files)} parquet files to {args.out}")

if __name__ == "__main__":
    main()
```

#!/usr/bin/env python3
"""
CDN-only surrogate-1 training with Lightning idle-resilient runner.

- Uses CDN URLs to bypass HF API rate limits.
- Projects to {prompt, response} only.
- Auto-restarts Lightning Studio if stopped by idle timeout.
"""
import json
import os
import tempfile
from pathlib import Path
from typing import Any, Dict, List, Optional

import pyarrow.parquet as pq
import requests
import torch
from lightning import Fabric, LightningModule, LightningDataModule
from lightning.fabric.plugins import TorchCheckpointIO
from lightning.pytorch.strategies import DDPStrategy
from lightning.pytorch.loggers import CSVLogger

try:
    from lightning import Studio, Machine, Teamspace
    _LIGHTNING_AVAILABLE = True
except Exception:
    _LIGHTNING_AVAILABLE = False

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

# ---- Data utilities ----
def cdn_parquet_to_table(cdn_url: str, columns: Optional[List[str]] = None):
    with tempfile.NamedTemporaryFile(suffix=".parquet") as tmp:
        resp = requests.get(cdn_url, stream=True, timeout=60)
        resp.raise_for_status()
        for chunk in resp.iter_content(chunk_size=8192):
            tmp.write(chunk)
        tmp.flush()
        return pq.read_table(tmp.name, columns=columns)

class SurrogateDataModule(LightningDataModule):
    def __init__(self, filelist_path: str, repo: str, batch_size: int = 8, num_workers: int = 4):
        super().__init__()
        self.filelist_path = filelist_path
        self.repo = repo
        self.batch_size = batch_size
        self.num_workers = num_workers
        self.columns = ["prompt", "response"]

    def prepare_data(self):
        with open(self.filelist_path) as f:
            self.meta = json.load(f)
        self.cdn_files = [
            CDN_TEMPLATE.format(repo=self.repo, path=fn)
            for fn in self.meta["files"]
        ]

    def setup(self, stage: Optional[str] = None):
        # Simple concatenation; for large data, use streaming iterators.
        tables = []
        for url in self.cdn_files:
            try:
                t = cdn_parquet_to_table(url, columns=self.columns)
                tables.append(t)
            except Exception as e:
                print(f"Skipping {url}: {e}")
        if not tables:
            raise RuntimeError("No valid parquet files found via CDN.")
        import pyarrow as pa
        full = pa.concat_tables(tables)
        # Ensure string columns
        self.dataset = full.to_batches()

    def train_dataloader(self):
        # Minimal torch Dataset wrapper
        class ParquetDataset(torch.utils.data.IterableDataset):
            def __init__(self, batches, columns):
                self.batches = batches
                self.columns = columns

            def __iter__(self):
                for batch in self.batches:
                    df = batch.to_pydict()
                    for i in range(len(df[self.columns[0]])):
                        prompt = str(df[self.columns[0]][i])
                        response = str(df[self.columns[1]][i])
                        yield {"input_ids": prompt, "labels": response}

        ds = ParquetDataset(self.dataset, self.columns)
        return torch.utils.data.DataLoader(
            ds,
            batch_size=self.batch_size,
            num_workers=self.num_workers,
            pin_memory=True,
        )

# ---- Model stub ----
class SurrogateModel(LightningModule):
    def __init__(self, lr: float = 1e-4):
        super().__init__()
        self.lr = lr

        self.log("train_loss", loss)

    def configure_optimizers(self):
        return torch.optim.AdamW(self.parameters(), lr=self.lr)
