# vanguard / ops

# 1. Diagnosis
- No persisted `(repo, dateFolder)` manifest exists → every training run re-enumerates via authenticated HF API → quota burn + 429 risk.
- Training likely uses recursive `list_repo_tree` or `load_dataset(streaming=True)` during data loading → mixed-schema CastError and rate-limit exposure.
- No CDN bypass strategy in place → all data fetches go through `/api/` endpoints subject to 1000 req/5min limit.
- No Lightning Studio reuse guard → idle stop kills training and recreation burns 80hr/mo quota.
- Missing wrapper script hygiene (shebang, executable bit, SHELL) for cron-launched ops scripts → silent exec failures.

# 2. Proposed change
Create `/opt/axentx/vanguard/ops/training_manifest.py` (single file, <150 LOC) that:
- Accepts `repo` and `dateFolder` on CLI.
- Calls `list_repo_tree(recursive=False)` **once** (Mac-side) and writes `manifests/{repo}/{dateFolder}.json` containing only file paths.
- Emits a companion `train.py` snippet that reads the manifest and downloads via CDN URLs (`resolve/main/...`) with zero authenticated calls during training.
- Adds a small Lightning launcher that reuses a running Studio or starts L40S in `lightning-public-prod`, with idle-stop guard.

```python
#!/usr/bin/env python3
"""
ops/training_manifest.py
Generate and use a CDN-only file manifest for HF dataset training.
Usage:
  python training_manifest.py build --repo <repo> --date <YYYY-MM-DD>
  python training_manifest.py train --repo <repo> --date <YYYY-MM-DD> --script train.py
"""
import argparse
import json
import os
import sys
from pathlib import Path

try:
    from huggingface_hub import list_repo_tree, hf_hub_download
except ImportError:
    print("pip install huggingface_hub")
    sys.exit(1)

BASE = Path(__file__).parent
MANIFEST_DIR = BASE / "manifests"
MANIFEST_DIR.mkdir(exist_ok=True)

def build_manifest(repo: str, date_folder: str):
    """Single authenticated call -> CDN-only manifest."""
    out_dir = MANIFEST_DIR / repo / date_folder
    out_dir.mkdir(parents=True, exist_ok=True)
    manifest_path = out_dir / "files.json"

    # Non-recursive: one API call for the date folder
    tree = list_repo_tree(repo=repo, path=date_folder, recursive=False)
    files = [item.rfilename for item in tree if item.type == "file"]

    manifest_path.write_text(json.dumps(files, indent=2))
    print(f"Built manifest: {manifest_path} ({len(files)} files)")
    return manifest_path

def cdn_download(repo: str, file_path: str, target_dir: Path):
    """Download via CDN (no auth/rate-limit on public repos)."""
    target_dir.mkdir(parents=True, exist_ok=True)
    url = f"https://huggingface.co/datasets/{repo}/resolve/main/{file_path}"
    out_file = target_dir / Path(file_path).name
    if out_file.exists():
        return out_file
    import requests
    r = requests.get(url, timeout=60)
    r.raise_for_status()
    out_file.write_bytes(r.content)
    return out_file

def generate_train_snippet(repo: str, date_folder: str, script_name: str):
    """Emit a Lightning-compatible train.py snippet using CDN-only downloads."""
    manifest_path = MANIFEST_DIR / repo / date_folder / "files.json"
    if not manifest_path.exists():
        build_manifest(repo, date_folder)

    files = json.loads(manifest_path.read_text())
    snippet = f"""# Auto-generated CDN-only training loader for {repo}/{date_folder}
from pathlib import Path
import json
import torch
from torch.utils.data import Dataset, DataLoader

MANIFEST = {json.dumps(files)}
REPO = "{repo}"
DATE_FOLDER = "{date_folder}"
CACHE_ROOT = Path("cache")

class CDNTextDataset(Dataset):
    def __init__(self):
        self.items = []
        for rel in MANIFEST:
            local = CACHE_ROOT / Path(rel).name
            if not local.exists():
                import requests
                url = f"https://huggingface.co/datasets/{{REPO}}/resolve/main/{{rel}}"
                local.parent.mkdir(parents=True, exist_ok=True)
                r = requests.get(url, timeout=60)
                r.raise_for_status()
                local.write_bytes(r.content)
            # Project to (prompt, response) here; skip mixed-schema cols
            # Example: parse JSONL line -> yield {\"prompt\": ..., \"response\": ...}
            # For demo, append path only:
            self.items.append(str(local))

    def __len__(self):
        return len(self.items)

    def __getitem__(self, i):
        return self.items[i]

if __name__ == "__main__":
    ds = CDNTextDataset()
    print(f"Loaded {{len(ds)}} files via CDN")
"""
    out_path = BASE / script_name
    out_path.write_text(snippet)
    print(f"Train snippet written: {out_path}")
    return out_path

def launch_lightning(script_path: str, repo: str, date_folder: str):
    """Reuse running Studio or start L40S in lightning-public-prod."""
    try:
        from lightning import Studio, Machine, Teamspace
    except ImportError:
        print("pip install lightning")
        return

    name = f"vanguard-{repo.replace('/', '-')}-{date_folder}"
    reused = False
    for s in Teamspace.studios:
        if s.name == name and s.status == "running":
            print(f"Reusing running studio: {s.name}")
            studio = s
            reused = True
            break

    if not reused:
        studio = Studio(
            name=name,
            machine=Machine.L40S,
            cloud="lightning-public-prod",
            script_path=str(script_path),
            create_ok=True,
        )
        print(f"Started studio: {studio.name}")

    # Guard against idle-stop death
    if studio.status != "running":
        print("Studio not running; restarting...")
        studio.start(machine=Machine.L40S)

    print(f"Studio URL: {studio.url}")
    return studio

def main():
    parser = argparse.ArgumentParser(description="Vanguard training manifest ops")
    sub = parser.add_subparsers(dest="cmd", required=True)

    build = sub.add_parser("build", help="Build manifest")
    build.add_argument("--repo", required=True)
    build.add_argument("--date", required=True, help="YYYY-MM-DD")

    train = sub.add_parser("train", help="Generate train snippet and launch")
    train.add_argument("--repo", required=True)
    train.add_argument("--date", required=True)
    train.add_argument("--script", default="train.py")

    args = parser.parse_args()

    if args.cmd == "build":
        build_manifest(args.repo, args.date)
    elif args.cmd == "train":
        sp = generate_train_snippet(args.repo, args.date, args.script)
        launch_lightning(sp, args.repo, args.date)

if __name__ == "__main__":
    main()
```
