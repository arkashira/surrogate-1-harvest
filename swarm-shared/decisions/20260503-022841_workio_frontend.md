# workio / frontend

## 1. Diagnosis

- No frontend idempotency feedback: when LINE webhook retries cause duplicate punches, the UI shows optimistic updates that make duplicates appear valid to employees.
- Clock-in/out actions lack client-side request deduplication (same user can double-click or retry while offline), increasing duplicate punch risk before the DB constraint even applies.
- No visual distinction between “pending” vs “committed” punch state; users can’t tell if a retry is safe or harmful.
- Missing optimistic-UI rollback on 409/422 (idempotency/unique violation) — errors are shown but stale optimistic state remains.
- No local pending-punch queue with stable request IDs to correlate retries and suppress duplicates across page reloads (within session).

## 2. Proposed change

**Scope**: `workio/src/features/attendance/` (new) + `workio/src/hooks/` + integrate into existing punch components.

- Add `usePendingPunch` hook: client-side idempotent queue keyed by `employeeId+date+type` with `requestId` (UUID) to deduplicate in-flight and persisted-to-sessionStorage retries.
- Add `PunchStatus` optimistic state machine: `idle → pending → committed | conflict`.
- Update punch buttons/components to use the hook and show clear pending/committed/conflict UI.
- Add automatic rollback on 409/422 and surface actionable message (“Punch already recorded — refresh to see latest”).

Files to touch/create:
- `src/features/attendance/hooks/usePendingPunch.ts`
- `src/features/attendance/components/PunchButton.tsx`
- Update existing punch form/modal to consume hook (likely `src/features/attendance/components/...` or `src/pages/AttendancePage.tsx`).

## 3. Implementation

### 3.1 `usePendingPunch.ts`

```ts
// src/features/attendance/hooks/usePendingPunch.ts
import { useEffect, useState, useCallback } from 'react';

type PunchType = 'IN' | 'OUT';
type PunchStatus = 'idle' | 'pending' | 'committed' | 'conflict';

interface PendingPunch {
  requestId: string;
  employeeId: string;
  date: string; // YYYY-MM-DD
  type: PunchType;
  timestamp: string; // ISO
  location?: { lat: number; lng: number };
}

const STORAGE_KEY = 'workio:pending-punches';

function loadPending(): Record<string, PendingPunch> {
  try {
    return JSON.parse(sessionStorage.getItem(STORAGE_KEY) || '{}');
  } catch {
    return {};
  }
}

function savePending(record: Record<string, PendingPunch>) {
  sessionStorage.setItem(STORAGE_KEY, JSON.stringify(record));
}

export function usePendingPunch(employeeId: string, date: string) {
  const [status, setStatus] = useState<PunchStatus>('idle');
  const [conflict, setConflict] = useState<boolean>(false);

  const key = `${employeeId}:${date}`;

  // reconcile on mount: if a pending exists for this key, mark pending so UI reflects it
  useEffect(() => {
    const pending = loadPending();
    if (pending[key]) {
      setStatus('pending');
    }
  }, [key]);

  const submitPunch = useCallback(
    async (type: PunchType, payload: { timestamp: string; location?: { lat: number; lng: number } }) => {
      const requestId = crypto.randomUUID();
      const pendingRecord: PendingPunch = {
        requestId,
        employeeId,
        date,
        type,
        timestamp: payload.timestamp,
        location: payload.location,
      };

      const pendingMap = loadPending();
      // deduplicate: if already pending for same employee+date+type on same date, skip new request
      const existing = Object.values(pendingMap).find(
        (p) => p.employeeId === employeeId && p.date === date && p.type === type
      );
      if (existing) {
        setStatus('pending');
        return { requestId: existing.requestId, duplicate: true };
      }

      pendingMap[key] = pendingRecord;
      savePending(pendingMap);
      setStatus('pending');
      setConflict(false);

      try {
        const res = await fetch('/api/attendance/punch', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify(pendingRecord),
        });

        if (res.ok) {
          // remove from pending
          const next = loadPending();
          delete next[key];
          savePending(next);
          setStatus('committed');
          return { requestId, ok: true };
        }

        // idempotency/unique violation
        if (res.status === 409 || res.status === 422) {
          const next = loadPending();
          delete next[key];
          savePending(next);
          setStatus('conflict');
          setConflict(true);
          return { requestId, conflict: true };
        }

        // other error: keep pending so user can retry explicitly
        setStatus('pending');
        return { requestId, ok: false, status: res.status };
      } catch (err) {
        // network error: keep pending for retry
        setStatus('pending');
        return { requestId, ok: false, error: err };
      }
    },
    [employeeId, date, key]
  );

  const clearConflict = useCallback(() => {
    setConflict(false);
    setStatus('idle');
  }, []);

  const retryPending = useCallback(async () => {
    const pendingMap = loadPending();
    const p = pendingMap[key];
    if (!p) {
      setStatus('idle');
      return;
    }
    setStatus('pending');
    setConflict(false);
    await submitPunch(p.type, { timestamp: p.timestamp, location: p.location });
  }, [key, submitPunch]);

  return {
    status,
    conflict,
    submitPunch,
    clearConflict,
    retryPending,
    isPending: status === 'pending',
    isCommitted: status === 'committed',
    isConflict: status === 'conflict',
  };
}
```
