# generate_file_index.py
# Mac-only orchestration script. Run after HF API rate-limit clears.
# Produces file_index.json for a single date folder.

set -euo pipefail

REPO="${HF_DATASET_REPO:-datasets/your-org/your-dataset}"
DATE_FOLDER="${1:-$(date +%Y-%m-%d)}"  # e.g. 2026-04-29
OUTFILE="${2:-file_index.json}"

python3 - "$REPO" "$DATE_FOLDER" "$OUTFILE" <<'PY'
import os, json, sys
from huggingface_hub import list_repo_tree

repo, date_folder, outfile = sys.argv[1], sys.argv[2], sys.argv[3]

# non-recursive listing for the date folder
tree = list_repo_tree(repo, path=date_folder, recursive=False)
files = [item.rfilename for item in tree if item.type == "file"]

index = {
    "repo": repo,
    "date_folder": date_folder,
    "files": sorted(files)
}

os.makedirs(os.path.dirname(outfile) if os.path.dirname(outfile) else ".", exist_ok=True)
with open(outfile, "w") as f:
    json.dump(index, f, indent=2)

print(f"Wrote {len(files)} files to {outfile}")
PY

#!/usr/bin/env python3
"""
CDN-only training entrypoint for Surrogate AI.
- No HF Hub API calls during training (bypasses 429).
- Projects heterogeneous files to {prompt, response} only.
- Lightning idle-stop resilient (auto-restart on stopped studio).
"""

import json
import os
import hashlib
import requests
from pathlib import Path
from typing import Iterator, Dict, Any

import torch
from torch.utils.data import IterableDataset, DataLoader
from lightning import Trainer, seed_everything
from lightning.pytorch.strategies import DDPStrategy

# Optional: surrogate model imports

HF_DATASET_REPO = os.getenv("HF_DATASET_REPO", "datasets/your-org/your-dataset")
FILE_INDEX = os.getenv("FILE_INDEX", "file_index.json")
HF_CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"
SIBLING_REPOS = [
    "datasets/your-org/your-dataset",
    "datasets/your-org/your-dataset-sib1",
    "datasets/your-org/your-dataset-sib2",
    "datasets/your-org/your-dataset-sib3",
    "datasets/your-org/your-dataset-sib4",
    "datasets/your-org/your-dataset-sib5",
]

seed_everything(42, workers=True)


def pick_sibling_repo(slug: str) -> str:
    """Deterministic sibling repo by hash-slug to bypass HF 128/hr commit cap."""
    h = hashlib.sha256(slug.encode()).hexdigest()
    idx = int(h, 16) % len(SIBLING_REPOS)
    return SIBLING_REPOS[idx]


def hfcdn_read(repo: str, path: str) -> bytes:
    """Download via CDN (no Authorization header)."""
    url = HF_CDN_TEMPLATE.format(repo=repo, path=path)
    resp = requests.get(url, timeout=60)
    resp.raise_for_status()
    return resp.content


def project_to_pair(raw: Dict[str, Any]) -> Dict[str, str]:
    """
    Project heterogeneous file content to {prompt, response}.
    Accepts JSON/JSONL/parquet read via other means; here we assume
    raw is already parsed into dict-like structure.
    """
    # Common field names seen in our mirrors
    prompt_keys = {"prompt", "instruction", "input", "question", "user"}
    response_keys = {"response", "completion", "output", "answer", "assistant"}

    prompt = None
    response = None

    for k in prompt_keys:
        if k in raw and raw[k] is not None:
            prompt = str(raw[k]).strip()
            break
    for k in response_keys:
        if k in raw and raw[k] is not None:
            response = str(raw[k]).strip()
            break

    # Fallbacks
    if prompt is None:
        prompt = json.dumps({k: v for k, v in raw.items() if k not in response_keys})
    if response is None:
        response = json.dumps({k: v for k, v in raw.items() if k in response_keys})

    return {"prompt": prompt, "response": response}


class CDNPairDataset(IterableDataset):
    def __init__(self, file_index_path: str, repo: str, max_files: int = -1):
        super().__init__()
        with open(file_index_path) as f:
            self.index = json.load(f)
        self.repo = repo or self.index.get("repo") or HF_DATASET_REPO
        files = self.index.get("files", [])
        if max_files > 0:
            files = files[:max_files]
        self.files = files

    def __iter__(self) -> Iterator[Dict[str, str]]:
        worker_info = torch.utils.data.get_worker_info()
        if worker_info is None:
            iter_files = self.files
        else:
            per_worker = len(self.files) // worker_info.num_workers
            wid = worker_info.id
            iter_files = self.files[wid * per_worker : (wid + 1) * per_worker]

        for path in iter_files:
            try:
                raw_bytes = hfcdn_read(self.repo, path)
                # Lightweight projection: try JSON lines first, else fallback
                text = raw_bytes.decode("utf-8", errors="replace").strip()
                if not text:
                    continue

                # Support both single JSON and JSONL
                lines = [ln for ln in text.splitlines() if ln.strip()]
                for ln in lines:
                    try:
                        obj = json.loads(ln)
                        pair = project_to_pair(obj)
                        if pair["prompt"] and pair["response"]:
                            yield pair
                    except json.JSONDecodeError:
                        # If not JSON, treat whole file as prompt (unlikely)
                        yield {"prompt": text, "response": ""}
            except Exception as exc:
                # Log and skip bad files to keep training robust
                print(f"Skipping {path}: {exc}")
                continue


def build_dataloader(file_index_path: str, batch_size: int = 8, max_files: int = -1) -> DataLoader:
    dataset = CDNPairDataset(file_index_path=file_index_path, repo=HF_DATASET_REPO, max_files=max_files)
    return DataLoader(dataset, batch_size=batch_size, num_workers=4)
