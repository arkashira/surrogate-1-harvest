## 1. File List Generator (Mac orchestrator)

`/opt/axentx/airship/scripts/generate_file_list.py`

```python
#!/usr/bin/env python3
"""
Generate CDN file list for Surrogate training.
Run from Mac orchestrator after HF rate-limit window clears.
Embeds file list into training to avoid API calls during Lightning runs.
"""
import json
import os
import sys
from datetime import datetime, timedelta
from pathlib import Path

try:
    from huggingface_hub import HfApi
except ImportError:
    print("Installing huggingface_hub...")
    os.system("pip install huggingface_hub -q")
    from huggingface_hub import HfApi

HF_REPO = os.getenv("HF_DATASET_REPO", "axentx/surrogate-training-data")
OUTPUT_DIR = Path(__file__).parent.parent / "config" / "file_lists"
OUTPUT_DIR.mkdir(parents=True, exist_ok=True)

def list_date_folder(date_str: str, folder_path: str = "batches/mirror-merged") -> dict:
    """
    List files in a specific date folder using single tree call.
    Returns dict with CDN URLs and local paths for Lightning training.
    """
    api = HfApi()
    full_path = f"{folder_path}/{date_str}" if date_str else folder_path

    try:
        tree = api.list_repo_tree(
            repo_id=HF_REPO,
            path=full_path,
            recursive=False,
            repo_type="dataset"
        )
    except Exception as e:
        print(f"Error listing {full_path}: {e}")
        return {"date": date_str, "files": [], "cdn_base": f"https://huggingface.co/datasets/{HF_REPO}/resolve/main"}

    files = []
    for item in tree:
        if item.rfilename.endswith(".parquet"):
            cdn_url = f"https://huggingface.co/datasets/{HF_REPO}/resolve/main/{item.rfilename}"
            files.append({
                "path": item.rfilename,
                "cdn_url": cdn_url,
                "size": getattr(item, "size", 0)
            })

    result = {
        "date": date_str,
        "generated_at": datetime.utcnow().isoformat(),
        "repo": HF_REPO,
        "cdn_base": f"https://huggingface.co/datasets/{HF_REPO}/resolve/main",
        "files": files,
        "total_files": len(files)
    }

    out_file = OUTPUT_DIR / f"file_list_{date_str or 'latest'}.json"
    out_file.write_text(json.dumps(result, indent=2))
    print(f"Generated {out_file} with {len(files)} files")
    return result

def generate_latest():
    """Generate file list for yesterday (safe window after HF commits)."""
    yesterday = (datetime.utcnow() - timedelta(days=1)).strftime("%Y-%m-%d")
    return list_date_folder(yesterday)

if __name__ == "__main__":
    date_arg = sys.argv[1] if len(sys.argv) > 1 else None
    if date_arg == "latest":
        generate_latest()
    else:
        list_date_folder(date_arg or "latest")
```

## 2. CDN-Only Training Script (Lightning)

`/opt/axentx/airship/scripts/train_cdn_only.py`

#!/usr/bin/env python3
"""
Lightning Surrogate training with CDN-only fetches.
Zero HF API calls during training — bypasses 429 rate limits.
"""
import json
import os
import sys
from pathlib import Path

import pyarrow.parquet as pq
import torch
from lightning import Fabric, LightningModule
from lightning.pytorch.utilities import rank_zero_only
from torch.utils.data import Dataset, DataLoader

# Project root
PROJECT_ROOT = Path(__file__).parent.parent
sys.path.insert(0, str(PROJECT_ROOT))

HF_REPO = os.getenv("HF_DATASET_REPO", "axentx/surrogate-training-data")
FILE_LIST = os.getenv("FILE_LIST", "config/file_lists/file_list_latest.json")

class CDNParquetDataset(Dataset):
    """Load parquet files directly from CDN URLs — no HF API."""
    def __init__(self, file_list_path: str, max_files: int = 50):
        list_path = Path(file_list_path)
        if not list_path.exists():
            raise FileNotFoundError(f"File list not found: {file_list_path}")

        with open(list_path) as f:
            manifest = json.load(f)

        self.files = [f["cdn_url"] for f in manifest["files"][:max_files]]
        print(f"CDN dataset: {len(self.files)} files from {manifest['date']}")

    def __len__(self):
        return len(self.files)

    def __getitem__(self, idx):
        url = self.files[idx]
        # Direct CDN download — no auth, no API rate limit
        table = pq.read_table(url, columns=["prompt", "response"])
        # Convert to tensors (simplified)
        text = table.to_pydict()
        # Tokenization placeholder — replace with actual tokenizer
        return {
            "prompt": text["prompt"][0] if text["prompt"] else "",
            "response": text["response"][0] if text["response"] else ""
        }

class SurrogateModel(LightningModule):
    def __init__(self, lr: float = 1e-4):
        super().__init__()
        self.lr = lr
        # Placeholder model — replace with actual surrogate architecture
        self.encoder = torch.nn.Linear(768, 512)

    def training_step(self, batch, batch_idx):
        # Simplified training step
        loss = torch.tensor(0.0, requires_grad=True)
        self.log("train_loss", loss)
        return loss

    def configure_optimizers(self):
        return torch.optim.Adam(self.parameters(), lr=self.lr)

def train():
    fabric = Fabric(devices=1, accelerator="gpu", precision="bf16-mixed")
    fabric.launch()

    dataset = CDNParquetDataset(FILE_LIST)
    loader = DataLoader(dataset, batch_size=8, num_workers=4)

    model = SurrogateModel()

    # Use Lightning Studio if available, else local
    try:
        from lightning import Studio, Machine
        studios = Studio.list()
        running = [s for s in studios if s.name == "surrogate-train" and s.status == "Running"]

        if running:
            studio = running[0]
            print(f"Reusing running studio: {studio.name}")
        else:
            studio = Studio.create(
                name="surrogate-train",
                machine=Machine.L40S,
                cloud="lightning-public-prod"
            )

        studio.run(
            train_fn=lambda: fabric.run(model, loader),
            requirements=["pyarrow", "torch", "lightning"]
        )
    except Exception as e:
        print(f"Studio unavailable, training locally: {e}")
        trainer = fabric
        trainer.fit(model, loader)

if __name__ == "__main__":
    train()
