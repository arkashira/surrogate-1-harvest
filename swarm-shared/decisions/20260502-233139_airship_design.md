# airship / design

## Highest-value incremental improvement (≤2h)

**Goal**: Harden `airship discover` into a deterministic, CDN-only ingestion workflow that eliminates HF API rate limits and produces a reproducible training manifest.

**Why this ships fastest**:
- Reuses existing patterns (CDN bypass, file-list pre-generation, Lightning Studio reuse).
- Avoids schema/casting pitfalls by projecting `{prompt,response}` at parse time.
- One small CLI + one small orchestrator change + one training-side manifest consumer.

---

## Implementation plan

1. **Add `airship discover` CLI command**
   - Single date-scoped run: `airship discover --repo <repo> --date <YYYY-MM-DD> --out ./manifests/`
   - Uses HF API **once** (after rate-limit window) to list top-level folder for that date.
   - Persists deterministic JSON manifest: `{repo, date, files: [{path, cdn_url, sha256, size}]}`.
   - Computes per-file deterministic shard assignment (hash-slug → 1 of 5 sibling repos) to respect HF commit cap.

2. **Embed manifest in training script**
   - `train.py` reads manifest JSON and downloads via CDN URLs only (`/resolve/main/...`).
   - No `load_dataset(streaming=True)` on heterogeneous repos.
   - Projects to `{prompt, response}` at parse time; attribution via filename pattern `batches/mirror-merged/{date}/{slug}.parquet`.

3. **Lightning Studio reuse guard**
   - Launcher checks running studios and reuses if name/status match; avoids quota burn.
   - On idle-stop, restarts with `target.start(machine=Machine.L40S)` before `.run()`.

4. **Mac-only orchestration contract**
   - Mac runs only this CLI + Lightning SDK launcher.
   - All heavy compute (training, ingestion) offloaded to Lightning/Kaggle/Colab.

```python
#!/usr/bin/env python3
"""
airship discover --repo <repo> --date 2026-04-29 --out ./manifests/

- Lists one date folder via HF API (non-recursive).
- Produces CDN-only manifest for deterministic training fetches.
- Assigns files to sibling repos to respect HF commit cap.
"""

import argparse
import hashlib
import json
import os
import sys
from datetime import datetime
from pathlib import Path
from typing import List, Dict

import requests
from tqdm import tqdm

HF_API_BASE = "https://huggingface.co/api"
CDN_BASE = "https://huggingface.co/datasets"

SIBLING_REPOS = [
    "axentx/surrogate-mirror-0",
    "axentx/surrogate-mirror-1",
    "axentx/surrogate-mirror-2",
    "axentx/surrogate-mirror-3",
    "axentx/surrogate-mirror-4",
]


def hf_api_get(path: str, token: str = None) -> dict:
    headers = {}
    if token:
        headers["Authorization"] = f"Bearer {token}"
    resp = requests.get(f"{HF_API_BASE}{path}", headers=headers, timeout=30)
    if resp.status_code == 429:
        print("HF API 429 — wait 360s before retry.", file=sys.stderr)
        sys.exit(1)
    resp.raise_for_status()
    return resp.json()


def list_date_folder(repo: str, date: str, token: str = None) -> List[dict]:
    """List non-recursive top-level files for a date folder."""
    path = f"/datasets/{repo}/tree/{date}"
    items = hf_api_get(path, token)
    # Expect files; ignore nested folders for safety
    return [i for i in items if i.get("type") == "file"]


def file_cdn_url(repo: str, filepath: str) -> str:
    return f"{CDN_BASE}/{repo}/resolve/main/{filepath}"


def hash_slug_to_repo(slug: str) -> str:
    """Deterministic sibling repo assignment."""
    digest = hashlib.sha256(slug.encode()).hexdigest()
    idx = int(digest[:8], 16) % len(SIBLING_REPOS)
    return SIBLING_REPOS[idx]


def build_manifest(repo: str, date: str, files: List[dict]) -> dict:
    manifest = {
        "repo": repo,
        "date": date,
        "created_at": datetime.utcnow().isoformat() + "Z",
        "files": [],
    }
    for f in files:
        path = f["path"]
        slug = f"{repo}/{path}"
        entry = {
            "path": path,
            "cdn_url": file_cdn_url(repo, path),
            "size": f.get("size", 0),
            "sha256": f.get("sha256", ""),
            "sibling_repo": hash_slug_to_repo(slug),
        }
        manifest["files"].append(entry)
    return manifest


def main() -> None:
    parser = argparse.ArgumentParser(description="CDN-only dataset manifest builder")
    parser.add_argument("--repo", required=True, help="HF dataset repo (user/repo)")
    parser.add_argument("--date", required=True, help="Date folder (YYYY-MM-DD)")
    parser.add_argument("--out", default="./manifests", help="Output directory")
    parser.add_argument("--hf-token", default=os.getenv("HF_TOKEN"), help="HF token (optional)")
    args = parser.parse_args()

    out_dir = Path(args.out)
    out_dir.mkdir(parents=True, exist_ok=True)

    print(f"Listing {args.repo}/{args.date} ...")
    files = list_date_folder(args.repo, args.date, args.hf_token)
    if not files:
        print("No files found.", file=sys.stderr)
        sys.exit(1)

    manifest = build_manifest(args.repo, args.date, files)
    out_path = out_dir / f"{args.repo.replace('/', '_')}_{args.date}.json"
    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"Manifest written: {out_path}")
    print(f"Total files: {len(files)}")
    print(f"CDN-only fetches enabled (zero API calls during training).")


if __name__ == "__main__":
    main()
```
