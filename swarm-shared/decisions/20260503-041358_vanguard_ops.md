# vanguard / ops

## 1. Diagnosis

- No content-addressed manifest exists → training and UI hit HF API at runtime (429s, non-reproducible epochs, no shareable snapshots).
- Dataset ingestion produces mixed-schema files in `enriched/` (extra `source`, `ts` cols) instead of projecting to `{prompt, response}` only, breaking downstream surrogate-1 schema expectations.
- No CDN-only data path for training → Lightning Studio jobs still call `load_dataset`/`list_repo_tree` during dataload, risking quota and rate limits.
- No deterministic file list snapshot → each run may see different file set if repo changes mid-sweep, causing silent epoch drift.
- Missing ops guardrails: no pre-flight check for running Studio reuse and no idle-stop restart logic, wasting Lightning quota when jobs resume after idle timeout.

## 2. Proposed change

Create a single canonical manifest generator and a training launcher that uses CDN-only paths with zero HF API calls during data load.

Files to add/modify:
- `/opt/axentx/vanguard/ops/make_manifest.py` — one-shot Mac orchestration script: list repo tree for a date folder, produce `manifest-{date}.json` with `{sha256, cdn_url, slug}` entries.
- `/opt/axentx/vanguard/train/train.py` — accept `--manifest` and use CDN URLs via `datasets` `IterableDataset` that fetches parquet shards directly (no `hf` auth).
- `/opt/axentx/vanguard/ops/launch_studio.py` — reuse running Studio or start L40S, with idle-stop guard and restart-on-stopped logic.

### ops/make_manifest.py
```python
#!/usr/bin/env python3
"""
Generate a content-addressed manifest for a date folder.
Run from Mac (or any orchestration host) after HF API window clears.

Usage:
  python make_manifest.py --repo <datasets/repo> --date 2026-04-29 \
    --out data/manifests/manifest-2026-04-29.json
"""

import argparse
import hashlib
import json
import os
import sys
from datetime import datetime
from pathlib import Path

# HF: use tree (non-recursive per folder) to avoid 429 on big repos
from huggingface_hub import HfApi

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def sha256_bytes(b: bytes) -> str:
    return hashlib.sha256(b).hexdigest()

def build_manifest(repo: str, date: str, out_path: Path):
    api = HfApi()
    folder = f"{date}/"
    entries = []

    # List top-level objects in date folder (avoids recursive paginate explosion)
    items = api.list_repo_tree(repo=repo, path=folder, recursive=False)
    for item in items:
        if not item.path.endswith(".parquet"):
            continue

        cdn_url = CDN_TEMPLATE.format(repo=repo, path=item.path)
        # Use path as slug; keep deterministic
        slug = item.path.replace("/", "__")

        # Note: we do NOT download here on Mac; CDN-only during training.
        # We include sha256 placeholder; populate via HEAD or first read if needed.
        entries.append({
            "slug": slug,
            "path": item.path,
            "cdn_url": cdn_url,
            "sha256": None,  # optional: fill later by training worker on first fetch
            "size": getattr(item, "size", None),
        })

    manifest = {
        "repo": repo,
        "date": date,
        "created_at": datetime.utcnow().isoformat() + "Z",
        "entries": entries,
    }

    out_path.parent.mkdir(parents=True, exist_ok=True)
    with open(out_path, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2)

    print(f"Wrote {len(entries)} entries to {out_path}")

def main():
    parser = argparse.ArgumentParser(description="Create CDN-only manifest")
    parser.add_argument("--repo", required=True, help="HF dataset repo (user/ds)")
    parser.add_argument("--date", required=True, help="Date folder (YYYY-MM-DD)")
    parser.add_argument("--out", required=True, help="Output manifest JSON path")
    args = parser.parse_args()

    try:
        build_manifest(args.repo, args.date, Path(args.out))
    except Exception as e:
        print(f"Error: {e}", file=sys.stderr)
        sys.exit(1)

if __name__ == "__main__":
    main()
```

### train/train.py (minimal CDN-only loader)
```python
#!/usr/bin/env python3
"""
Lightning training script that uses CDN-only parquet files.
No HF API calls during data loading.

Usage:
  python train.py --manifest data/manifests/manifest-2026-04-29.json
"""

import argparse
import json
import sys
from pathlib import Path

import torch
from torch.utils.data import IterableDataset, DataLoader
import pyarrow.parquet as pq
import requests

class CDNParquetIterable(IterableDataset):
    def __init__(self, entries, project_cols=("prompt", "response")):
        self.entries = entries
        self.project_cols = project_cols

    def __iter__(self):
        for ent in self.entries:
            url = ent["cdn_url"]
            # Stream parquet from CDN (no auth)
            resp = requests.get(url, timeout=60)
            resp.raise_for_status()
            table = pq.read_table(pq.ParquetFile(pq.ParquetBuffer(resp.content)))
            df = table.select(self.project_cols).to_pandas()
            for _, row in df.iterrows():
                yield {k: row[k] for k in self.project_cols}

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--manifest", required=True)
    parser.add_argument("--batch-size", type=int, default=8)
    parser.add_argument("--max-batches", type=int, default=100)
    args = parser.parse_args()

    manifest_path = Path(args.manifest)
    if not manifest_path.exists():
        print(f"Manifest not found: {manifest_path}", file=sys.stderr)
        sys.exit(1)

    with open(manifest_path, encoding="utf-8") as f:
        manifest = json.load(f)

    entries = manifest["entries"]
    if not entries:
        print("No parquet entries in manifest", file=sys.stderr)
        sys.exit(1)

    ds = CDNParquetIterable(entries)
    loader = DataLoader(ds, batch_size=args.batch_size, num_workers=0)

    # Minimal training loop placeholder
    device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
    print(f"Using device: {device}")

    for i, batch in enumerate(loader):
        if i >= args.max_batches:
            break
        # Replace with real model step
        print(f"Batch {i}: prompts={len(batch.get('prompt', []))}")

    print("Training step completed (CDN-only).")

if __name__ == "__main__":
    main()
```

    teamspace = Teamspace(name=TEAMSPACE)
