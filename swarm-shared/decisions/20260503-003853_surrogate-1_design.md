## Implementation Plan (≤2h)

**Highest-value change**: Add a Mac-side `tools/snapshot_manifest.py` that lists one date-partition via a **single** HF API call, emits `file_manifest.json` with CDN URLs + integrity metadata, and patch Lightning training to use **zero-API CDN-only** data loading. This eliminates HF API rate-limit risk during training and keeps within the 128-commit cap by avoiding repeated `list_repo_files` calls.

   - Single `list_repo_tree(path=date_partition, recursive=True)` call from Mac.
   - Filter to parquet/jsonl files.
   - Emit `file_manifest.json`:
     - `cdn_url` (`https://huggingface.co/datasets/{repo}/resolve/main/{path}`)
     - `size_bytes`
     - `etag`/`md5` if available
     - `local_path` (relative)
   - Save to `manifests/{date}/file_manifest.json`.
   - Include retry/backoff for 429 (wait 360s).

   - Validates size/etag.
   - Streams to local temp dir.

### 1. `tools/snapshot_manifest.py`

```python
#!/usr/bin/env python3
"""
snapshot_manifest.py
List one date-partition of a HuggingFace dataset via a single API call
and emit file_manifest.json with CDN URLs and integrity metadata.

Usage:
    python tools/snapshot_manifest.py \
        --repo axentx/surrogate-1-training-pairs \
        --date 2026-04-29 \
        --out manifests/2026-04-29/file_manifest.json
"""

import argparse
import json
import os
import sys
import time
from datetime import datetime
from pathlib import Path

import requests
from huggingface_hub import HfApi, hf_hub_url

API = HfApi()
CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"
RETRY_WAIT = 360  # seconds after 429

def list_partition(repo_id: str, date_partition: str):
    """Single API call to list files in one date folder (non-recursive tree + walk)."""
    # Use recursive=True to get all files under the partition in one call
    for attempt in range(3):
        try:
            tree = API.list_repo_tree(
                repo_id=repo_id,
                path=date_partition,
                recursive=True,
                repo_type="dataset",
            )
            return tree
        except requests.exceptions.HTTPError as e:
            if e.response.status_code == 429:
                print(f"[{datetime.utcnow().isoformat()}] 429 rate limit — waiting {RETRY_WAIT}s", file=sys.stderr)
                time.sleep(RETRY_WAIT)
                continue
            raise
    raise RuntimeError("Exhausted retries after 429")

def build_manifest(repo_id: str, date_partition: str, out_path: Path):
    tree = list_partition(repo_id, date_partition)
    out_path.parent.mkdir(parents=True, exist_ok=True)

    entries = []
    for item in tree:
        if item.type != "file":
            continue
        path = item.path
        if not (path.endswith(".parquet") or path.endswith(".jsonl")):
            continue

        size = getattr(item, "size", None)
        etag = getattr(item, "etag", None)

        entries.append({
            "path": path,
            "cdn_url": CDN_TEMPLATE.format(repo=repo_id, path=path),
            "size_bytes": size,
            "etag": etag,
            "last_modified": getattr(item, "last_modified", None),
        })

    manifest = {
        "repo_id": repo_id,
        "partition": date_partition,
        "generated_at": datetime.utcnow().isoformat() + "Z",
        "total_files": len(entries),
        "files": entries,
    }

    with open(out_path, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2, ensure_ascii=False)

    print(f"Wrote manifest with {len(entries)} files to {out_path}")
    return manifest

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Snapshot HF dataset partition manifest")
    parser.add_argument("--repo", default="axentx/surrogate-1-training-pairs")
    parser.add_argument("--date", required=True, help="Date partition path, e.g. 2026-04-29")
    parser.add_argument("--out", required=True, help="Output manifest JSON path")
    args = parser.parse_args()

    build_manifest(args.repo, args.date, Path(args.out))
```

### 2. `tools/download_cdn_partition.py` (optional helper)

```python
#!/usr/bin/env python3
"""
Download files listed in file_manifest.json via CDN (no auth).
"""

import json
import hashlib
import argparse
import requests
from pathlib import Path
from concurrent.futures import ThreadPoolExecutor, as_completed

def download_file(entry, dest_dir):
    url = entry["cdn_url"]
    rel = entry["path"]
    out = dest_dir / rel
    out.parent.mkdir(parents=True, exist_ok=True)

    r = requests.get(url, stream=True, timeout=60)
    r.raise_for_status()

    sha256 = hashlib.sha256()
    with open(out, "wb") as f:
        for chunk in r.iter_content(chunk_size=8192):
            f.write(chunk)
            sha256.update(chunk)

    actual_size = out.stat().st_size
    expected = entry.get("size_bytes")
    if expected is not None and actual_size != expected:
        out.unlink()
        raise ValueError(f"Size mismatch for {rel}: expected {expected}, got {actual_size}")

    return {"path": rel, "sha256": sha256.hexdigest(), "size": actual_size}

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--manifest", required=True, help="file_manifest.json")
    parser.add_argument("--dest", default="data/partition", help="Destination directory")
    parser.add_argument("--workers", type=int, default=8)
    args = parser.parse_args()

    manifest = json.loads(Path(args.manifest).read_text())
    dest = Path(args.dest)
    dest.mkdir(parents=True, exist_ok=True)

    results = []
    with ThreadPoolExecutor(max_workers=args.workers) as ex:
        futures = {ex.submit(download_file, e, dest): e for e in manifest["files"]}
        for fut in as_completed(futures):
            try:
                results.append(fut.result())
            except Exception as exc:
                print(f"Failed {futures[fut]['path']}: {exc}")
                raise

    print(f"Downloaded {len(results)} files to {dest}")

if __name__ == "__main__":
    main()
```
