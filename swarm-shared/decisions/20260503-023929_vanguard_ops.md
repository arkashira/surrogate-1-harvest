# vanguard / ops

## 1. Diagnosis
- Frontend still triggers runtime HF API calls (`list_repo_tree`, dataset endpoints) from the browser, burning quota and risking 429s on user machines.
- No static file manifest embedded in the bundle → every session re-enumerates repos instead of using a pre-computed file list.
- Missing `index.html` and client mount point means no fast feedback loop for frontend changes and no ops-deployable artifact.
- No bundler/dev tooling configured; any JS/TS changes require manual reloads and lack cache-busting or CDN-friendly output.
- No ops-level guardrails (env-based API strategy, build-time embedding) to enforce HF CDN bypass pattern for training/inference workloads.

## 2. Proposed change
Create a minimal ops-deployable frontend bundle with build-time embedding of repo file lists and runtime HF CDN-only fetches. Scope:
- Add `/opt/axentx/vanguard/public/index.html` (client mount)
- Add `/opt/axentx/vanguard/scripts/build-frontend.sh` (build + manifest embed)
- Add `/opt/axentx/vanguard/src/frontend.js` (runtime: CDN fetches only, no API calls)
- Add `/opt/axentx/vanguard/.env.ops` (ops config: `HF_REPO`, `HF_FOLDER`, `PORT`)
- Update crontab/systemd helper to run `knowledge-rag`-style pre-list once and embed into the bundle before deploy.

// Runtime behavior: CDN-only fetches. No HF API calls from frontend.
// Manifest is embedded at build time by build-frontend.sh.
const CONFIG = {
  repo: '__HF_REPO__',
  folder: '__HF_FOLDER__',
  manifestPath: '/static/manifest.json'
};

const $ = (s) => document.querySelector(s);
const statusEl = $('#status');
const filesEl = $('#files');
const previewEl = $('#preview');
const previewBody = $('#preview-body');
const refreshBtn = $('#refresh');

function setStatus(msg) { if (statusEl) statusEl.textContent = msg; }

async function loadManifest() {
  try {
    const res = await fetch(CONFIG.manifestPath + '?t=' + Date.now());
    if (!res.ok) throw new Error('Manifest fetch failed: ' + res.status);
    return await res.json();
  } catch (err) {
    console.error(err);
    setStatus('Error loading manifest — run build step on ops host.');
    return null;
  }
}

function renderFiles(files) {
  if (!filesEl) return;
  filesEl.innerHTML = '';
  if (!files || files.length === 0) {
    filesEl.innerHTML = '<p>No files found.</p>';
    return;
  }
  const ul = document.createElement('ul');
  ul.style.listStyle = 'none';
  ul.style.padding = '0';
  files.forEach((f) => {
    const li = document.createElement('li');
    li.style.padding = '0.5rem 0';
    li.style.borderBottom = '1px solid #334155';
    const btn = document.createElement('button');
    btn.textContent = f.name || f.path.split('/').pop() || f.path;
    btn.onclick = () => previewFile(f);
    li.appendChild(btn);
    ul.appendChild(li);
  });
  filesEl.appendChild(ul);
}

async function previewFile(file) {
  setStatus('Fetching ' + file.path + ' (CDN)...');
  previewEl.style.display = 'block';
  previewBody.textContent = '';
  try {
    const res = await fetch(file.cdnUrl);
    if (!res.ok) throw new Error('CDN fetch failed: ' + res.status);
    const text = await res.text();
    previewBody.textContent = text;
    setStatus('Loaded ' + file.path);
  } catch (err) {
    previewBody.textContent = 'Error: ' + err.message;
    setStatus('CDN fetch error');
  }
}

async function run() {
  setStatus('Loading manifest...');
  const manifest = await loadManifest();
  if (manifest && manifest.files) {
    renderFiles(manifest.files);
    setStatus('Ready — ' + manifest.files.length + ' files (CDN-only)');
  } else {
    setStatus('No manifest — run build step on ops host.');
  }
}

if (refreshBtn) {
  refreshBtn.onclick = async () => {
    setStatus('Refreshing manifest (this triggers build step on ops host)...');
    // In ops flow, this button should call an ops endpoint or instruct user to run build-frontend.sh.
    // For local dev, we simply reload manifest.
    await run();
  };
}

// Start
run();
JS

#!/usr/bin/env bash
set -euo pipefail
# Build frontend bundle and embed repo file manifest (CDN-only strategy).
# Usage: bash scripts/build-frontend.sh [--serve]
# Requires: curl, jq, python3 (simple http server optional)

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
PROJECT_ROOT="$(cd "$SCRIPT_DIR/.." && pwd)"
SRC="$PROJECT_ROOT/src/frontend.js"
PUBLIC="$PROJECT_ROOT/public"
STATIC="$PUBLIC/static"
OUT_JS="$STATIC/frontend.js"
MANIFEST_OUT="$STATIC/manifest.json"

# Load ops config
ENV_FILE="$PROJECT_ROOT/.env.ops"
if [[ -f "$ENV_FILE" ]]; then
  export $(grep -v '^#' "$ENV_FILE" | xargs) || true
fi

HF_REPO="${HF_REPO:-datasets/example/repo}"
HF_FOLDER="${HF_FOLDER:-enriched}"
PORT="${PORT:-3000}"

mkdir -p "$STATIC"

echo "== Building frontend (CDN-only) for $HF_REPO/$HF_FOLDER =="
