**Highest-value improvement**: Add deterministic pre-flight file listing + CDN-only ingestion to eliminate HF API 429s during training and make shard workers resilient.

### Changes (3 files, ~120 lines total)

1. **`bin/list_files.py`** — Mac-side helper: single API call to `list_repo_tree` for a date folder, emit JSON of `{path, sha256}` for CDN download. Embeddable by training scripts and shard workers.
2. **`bin/dataset-enrich.sh`** — Shard worker: accept optional file-list JSON; if provided, use CDN URLs (`/resolve/main/...`) for zero-API ingestion; fall back to `load_dataset` only if list missing.
3. **`lib/dedup.py`** — Minor: expose `bulk_exists` + `bulk_insert` so workers can batch-check duplicates before upload (reduces SQLite contention and API writes).

### 1) `bin/list_files.py` (new)

```python
#!/usr/bin/env python3
"""
Usage:
  python bin/list_files.py axentx/surrogate-1-training-pairs 2026-05-02 > file-list.json

Outputs:
  [
    {"path": "raw/2026-05-02/foo.parquet", "sha256": "..."},
    ...
  ]

Notes:
- Uses huggingface_hub list_repo_tree (non-recursive per folder) to avoid 429.
- Falls back to repo root listing if subpath missing.
- sha256 is repo file LFS pointer hash when available; otherwise None.
"""
import json
import sys
from pathlib import Path

from huggingface_hub import HfApi, list_repo_tree

API = HfApi()

def list_files(repo_id: str, subpath: str = "") -> list[dict]:
    try:
        items = list_repo_tree(repo_id=repo_id, path=subpath, recursive=False)
    except Exception:
        # If subpath invalid, list root
        items = list_repo_tree(repo_id=repo_id, path="", recursive=False)

    out = []
    for item in items:
        if item.type != "file":
            continue
        out.append({
            "path": str(Path(subpath) / item.path) if subpath else item.path,
            "sha256": getattr(item, "lfs", {}).get("sha256") if hasattr(item, "lfs") else None,
        })
    return out

def main() -> None:
    if len(sys.argv) < 2:
        print("Usage: list_files.py <repo_id> [subpath]", file=sys.stderr)
        sys.exit(1)
    repo_id = sys.argv[1]
    subpath = sys.argv[2] if len(sys.argv) > 2 else ""
    result = list_files(repo_id, subpath)
    json.dump(result, sys.stdout, indent=None, separators=(",", ":"))

if __name__ == "__main__":
    main()
```

Make executable:

```bash
chmod +x bin/list_files.py
```

### 2) `bin/dataset-enrich.sh` (updated)

```bash
#!/usr/bin/env bash
# surrogate-1 shard worker: normalize + dedup + upload
#
# Optional env:
#   FILE_LIST_JSON   path to JSON produced by list_files.py (CDN mode)
#   HF_DATASET_REPO  dataset repo (default: axentx/surrogate-1-training-pairs)
#   SHARD_ID         0..15
#   DATE_FOLDER      e.g. 2026-05-02

set -euo pipefail
SHELL=/bin/bash

REPO="${HF_DATASET_REPO:-axentx/surrogate-1-training-pairs}"
SHARD_ID="${SHARD_ID:-0}"
DATE_FOLDER="${DATE_FOLDER:-$(date +%Y-%m-%d)}"
OUT_DIR="$(mktemp -d)"
TIMESTAMP="$(date +%H%M%S)"
OUTPUT="${OUT_DIR}/shard${SHARD_ID}-${TIMESTAMP}.jsonl"

cd /opt/axentx/surrogate-1

log() { echo "[$(date -Iseconds)] $*"; }

# Dedup helper
python3 -c "
import sys
sys.path.insert(0, 'lib')
from dedup import bulk_exists, bulk_insert
import json, hashlib, os

def hash_pair(prompt, response):
    return hashlib.md5((str(prompt) + '||' + str(response)).encode()).hexdigest()

def run():
    file_list = os.getenv('FILE_LIST_JSON')
    if file_list and os.path.exists(file_list):
        # CDN mode: zero-API ingestion
        import requests
        with open(file_list) as f:
            files = json.load(f)
        # deterministic shard slice
        shard_files = [f for f in files if hash(f['path']) % 16 == int(os.getenv('SHARD_ID', 0))]
        pairs = []
        hashes = []
        for fobj in shard_files:
            url = f'https://huggingface.co/datasets/{os.getenv(\"HF_DATASET_REPO\", \"axentx/surrogate-1-training-pairs\")}/resolve/main/{fobj[\"path\"]}'
            try:
                import pyarrow.parquet as pq
                import pyarrow as pa
                tbl = pq.read_table(url, columns=['prompt', 'response'])
                for row in tbl.to_pylist():
                    h = hash_pair(row.get('prompt', ''), row.get('response', ''))
                    pairs.append(row)
                    hashes.append(h)
            except Exception as e:
                print(f'WARN: CDN read failed {url}: {e}')
                continue
        if not hashes:
            print('No rows from CDN files')
            return
        seen = bulk_exists(hashes)
        new_pairs = [p for p, h in zip(pairs, hashes) if h not in seen]
        if new_pairs:
            bulk_insert([(h, json.dumps(p, ensure_ascii=False)) for p, h in zip(new_pairs, hashes) if h not in seen])
        # Write output
        with open(os.path.join(os.getenv('OUT_DIR', '/tmp'), os.getenv('OUTPUT_FILE', 'out.jsonl')), 'w') as out:
            for p in new_pairs:
                out.write(json.dumps(p, ensure_ascii=False) + '\n')
        print(f'Uploaded {len(new_pairs)} new pairs (CDN mode)')
    else:
        # Fallback: streaming dataset (may hit 429)
        from datasets import load_dataset
        ds = load_dataset(os.getenv('HF_DATASET_REPO', 'axentx/surrogate-1-training-pairs'), streaming=True)
        split = ds.get('train') or ds.get('validation') or next(iter(ds.values()))
        shard_split = (row for i, row in enumerate(split) if i % 16 == int(os.getenv('SHARD_ID', 0)))
        pairs = []
        hashes = []
        for row in shard_split:
            h = hash_pair(row.get('prompt', ''), row.get('response', ''))
            pairs.append(row)
            hashes.append(h)
        seen = bulk_exists(hashes)
        new_pairs = [p for p, h in zip(pairs, hashes) if h not in seen]
        if new_pairs:
            bulk_insert([(h, json.dumps(p, ensure_ascii=False)) for p, h in zip(new_pairs, hashes) if h not in seen])
        with open(os.path.join(os.getenv('OUT_DIR', '/tmp'), os.getenv('OUTPUT_FILE', 'out.jsonl')), 'w') as out:
            for p in new_pairs:
                out.write(json.dumps(p, ensure_ascii=False) + '\n')
        print(f'Uploaded {len(new_pairs)} new pairs (streaming mode)')

if __name__ == '__main__':
    run()
"

# Upload output to dataset repo (unique filename prevents collisions)
if [[ -s "${OUTPUT}" ]]; then
  log "Uploading shard ${SHARD_ID} output"
  huggingface-cli upload \
    --repo-type dataset \
    "${REPO}" \
    "${OUTPUT}" \
    "batches/public-merged/${DATE_FOLDER}/shard${SHARD_ID}-${TIMESTAMP}.jsonl" \
    --token "${HF_TOKEN}"
else
  log "No new pairs for shard ${SHARD_ID}"
fi

rm -rf "${OUT
