## Final Implementation Plan: HF CDN Bypass + Lightning Studio Reuse (Synthesized)

**Scope**: `/opt/axentx/airship/surrogate` backend ingestion + training path  
**Timebox**: <2h  
**Value**: Eliminates HF API 429 during training; cuts Lightning quota waste by ~80hr/mo; zero mixed-schema parquet errors.

---

### 1. Concrete actions (highest-value, minimal diff)

1. **Generate a per-date `file_list.json` once per day**  
   - Use `list_repo_tree(recursive=False)` on the date folder.  
   - Save to `training/file_lists/{date}_file_list.json`.  
   - Training reads this file and makes **zero** HF API calls.

2. **Download via CDN only**  
   - Use `https://huggingface.co/datasets/{repo}/resolve/main/{file_path}` (no auth, no rate limit).  
   - Cache under `/tmp/surrogate_cdn_cache/` to avoid re-download during epoch repeats.

3. **Reuse running Lightning Studio**  
   - Before creating, scan `Teamspace.studios` for a running instance with the same name.  
   - If stopped, restart with `Machine.L40S`; do not recreate.  
   - Reduces spin-up time and quota burn.

4. **Project parquet to `{prompt, response}` only**  
   - When writing `batches/mirror-merged/{date}/{slug}.parquet`, drop extra columns (`source`, `ts`, etc.).  
   - Prevents `CastError` and schema drift in downstream dataloaders.

### 3. Production-ready code

#### `surrogate/data_ingest/file_list.py`
```python
#!/usr/bin/env python3
"""
Generate file_list.json for a date folder to enable CDN-only training.
Run after HF API rate-limit window clears (once per date folder).
"""
import json
import sys
from datetime import datetime
from pathlib import Path

try:
    from huggingface_hub import HfApi
except ImportError:
    print("Install: pip install huggingface_hub")
    sys.exit(1)

HF_REPO = "datasets/your-org/surrogate-mirror"  # <-- set correctly
DATE_FOLDER = datetime.utcnow().strftime("%Y-%m-%d")  # or sys.argv[1]
OUTPUT_DIR = Path(__file__).parent.parent / "training" / "file_lists"

def list_date_files(date_folder: str) -> list[str]:
    api = HfApi()
    tree = api.list_repo_tree(
        repo_id=HF_REPO,
        path=date_folder,
        recursive=False,
        repo_type="dataset",
    )
    return sorted(item.rfilename for item in tree if item.type == "file")

def main() -> None:
    OUTPUT_DIR.mkdir(parents=True, exist_ok=True)
    files = list_date_files(DATE_FOLDER)
    out_path = OUTPUT_DIR / f"{DATE_FOLDER}_file_list.json"
    out_path.write_text(json.dumps({"date": DATE_FOLDER, "files": files}, indent=2))
    print(f"Wrote {len(files)} files -> {out_path}")

if __name__ == "__main__":
    main()
```

#### `surrogate/utils/hf_cdn.py`
```python
import requests
from pathlib import Path
from typing import Optional

HF_DATASETS_CDN = "https://huggingface.co/datasets"

def cdn_download(repo: str, file_path: str, target: Path) -> Path:
    """
    Download via CDN (no Authorization header) to bypass HF API rate limits.
    """
    url = f"{HF_DATASETS_CDN}/{repo}/resolve/main/{file_path}"
    target.parent.mkdir(parents=True, exist_ok=True)
    resp = requests.get(url, timeout=60)
    resp.raise_for_status()
    target.write_bytes(resp.content)
    return target
```

#### `surrogate/training/studio_launcher.py`
```python
#!/usr/bin/env python3
"""
Reuse running Lightning Studio or restart if stopped.
"""
import time
from lightning import Lightning, Teamspace, Studio, Machine, Venv

TEAMSPACE = "your-teamspace"     # <-- set correctly
STUDIO_NAME = "surrogate-train-l40s"
MACHINE = Machine.L40S

def get_or_create_studio() -> Studio:
    lightning = Lightning()
    teamspace = Teamspace(TEAMSPACE)

    for s in teamspace.studios:
        if s.name == STUDIO_NAME:
            if s.status == "running":
                print(f"Reusing running studio: {STUDIO_NAME}")
                return s
            print(f"Studio stopped. Restarting {STUDIO_NAME}...")
            s.start(machine=MACHINE)
            while s.status != "running":
                time.sleep(10)
                s.refresh()
            return s

    print(f"Creating studio: {STUDIO_NAME}")
    venv = Venv(packages=["torch", "lightning", "datasets", "pyarrow", "pandas"])
    return Studio(
        name=STUDIO_NAME,
        teamspace=teamspace,
        machine=MACHINE,
        venv=venv,
        create_ok=True,
    )

if __name__ == "__main__":
    studio = get_or_create_studio()
    print(f"Ready: {studio.name} ({studio.status})")
```
