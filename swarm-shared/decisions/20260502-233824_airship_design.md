import os

def list_repo_files(repo, path):
    # Use CDN-only file paths
    url = f"https://huggingface.co/datasets/{repo}/resolve/main/{path}"

# airship/discover/airship_discover.py
import pyarrow.parquet as pq

def project_files(file_list):
    # Project files to {prompt, response} schema
    projected_files = []
    for file in file_list:
        # Load file using PyArrow
        table = pq.read_table(file)
        # Project table to {prompt, response} schema
        projected_table = table.to_pandas().rename(columns={"text": "prompt", "label": "response"})
        # Save projected table to new file
        projected_files.append(pq.write_table(projected_table, file))
    return projected_files
```

# airship/discover/airship_discover.py
import hashlib

def produce_manifest(projected_files):
    # Produce reproducible file manifest
    manifest = {}
    for file in projected_files:
        # Calculate file hash
        file_hash = hashlib.sha256(open(file, "rb").read()).hexdigest()
        # Add file hash to manifest
        manifest[file] = file_hash
    return manifest
```
