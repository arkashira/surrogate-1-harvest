# Costinel / design

## Final Implementation Plan — Top-Hub Signal Panel (CDN-first, <2h)

### What we ship
- A **non-blocking Top-Hub Signal Panel** mounted on the Costinel dashboard (top banner or sidebar).
- Defaults to hub **MOC** (configurable via `VITE_HUB_NAME`).
- Shows: hub title, short description, top 3 signals (anomalies/recommendations), last updated timestamp.
- **CDN-first data fetch** using public JSON:  
  `https://huggingface.co/datasets/axentx/costinel-hubs/resolve/main/{hub}/latest.json`  
  (no Authorization header → bypasses HF API rate limits).
- Graceful fallback: if CDN fails, shows cached/placeholder data and logs (never blocks render).
- Auto-refresh every **5 minutes** while page is visible; respects `prefers-reduced-motion` and `save-data`.
- Zero backend changes; pure frontend addition.

### src/composables/useHubSignals.ts
```ts
import { ref, computed } from 'vue';
import type { HubSnapshot, HubSignal } from '@/types/hub';

const HUB_NAME = import.meta.env.VITE_HUB_NAME || 'MOC';
const CDN_URL = `https://huggingface.co/datasets/axentx/costinel-hubs/resolve/main/${HUB_NAME}/latest.json`;
const LOCAL_FALLBACK_URL = `/data/hubs/${HUB_NAME}/latest.json`;

const snapshot = ref<HubSnapshot | null>(null);
const loading = ref(false);
const error = ref<string | null>(null);
const lastFetchAt = ref<number | null>(null);

const topSignals = computed<HubSignal[]>(() => snapshot.value?.signals?.slice(0, 3) || []);
const lastUpdated = computed(() => snapshot.value?.generated_at || null);

async function fetchFromCDN(url: string, timeoutMs = 3000): Promise<HubSnapshot | null> {
  const controller = new AbortController();
  const timeout = setTimeout(() => controller.abort(), timeoutMs);

  try {
    const res = await fetch(url, {
      cache: 'no-store',
      signal: controller.signal,
    });
    clearTimeout(timeout);

    if (!res.ok) throw new Error(`CDN fetch failed: ${res.status}`);
    return res.json();
  } finally {
    clearTimeout(timeout);
  }
}

async function load() {
  if (loading.value) return;
  loading.value = true;
  error.value = null;

  try {
    // Try CDN first (bypasses HF API auth/rate limits)
    snapshot.value = await fetchFromCDN(CDN_URL);
  } catch {
    try {
      // Fallback to local bundled stub
      snapshot.value = await fetchFromCDN(LOCAL_FALLBACK_URL);
    } catch {
      error.value = 'Unable to load hub signals';
      // Keep existing snapshot if available (graceful degradation)
    }
  } finally {
    loading.value = false;
    lastFetchAt.value = Date.now();
  }
}

// Auto-refresh while page visible (respects reduced-motion/save-data)
function startAutoRefresh(intervalMs = 5 * 60 * 1000) {
  if (typeof window === 'undefined') return;

  const prefersReduced = window.matchMedia('(prefers-reduced-motion: reduce)').matches;
  const saveData = (navigator as any)?.connection?.saveData || false;
  if (prefersReduced || saveData) return;

  load();
  const id = setInterval(() => {
    if (document.visibilityState === 'visible') load();
  }, intervalMs);

  // Return cleanup
  return () => clearInterval(id);
}

export function useHubSignals() {
  return {
    hub: HUB_NAME,
    snapshot,
    loading,
    error,
    topSignals,
    lastUpdated,
    lastFetchAt,
    load,
    startAutoRefresh,
  };
}
```

### src/components/TopHubSignalPanel.vue
```vue
<template>
  <section class="top-hub-panel" aria-label="Top hub signals">
    <header class="panel-header">
      <h3 class="panel-title">{{ hubName }}</h3>
      <span v-if="lastUpdated" class="panel-meta">Updated {{ formatTime(lastUpdated) }}</span>
    </header>

    <div v-if="loading && !hasData" class="panel-loading">Loading signals…</div>
    <div v-else-if="error && !hasData" class="panel-error">{{ error }}</div>

    <ul v-if="topSignals.length" class="signals-list" role="list">
      <li
        v-for="s in topSignals"
        :key="s.id"
        class="signal-item"
        :class="`severity-${s.severity}`"
      >
        <div class="signal-badge" :aria-label="`Severity: ${s.severity}`" />
        <div class="signal-body">
          <div class="signal-title">{{ s.title }}</div>
          <div class="signal-desc">{{ s.description }}</div>
          <div v-if="s.recommendation" class="signal-reco">{{ s.recommendation }}</div>
          <div v-if="s.resource" class="signal-resource">{{ s.resource }}</div>
        </div>
      </li>
    </ul>

    <div v-else-if="!loading" class="panel-empty">No signals available.</div>

    <footer class="panel-footer">
      <button
        @click="refresh"
        :disabled="loading"
        class="refresh-btn"
        aria-label="Refresh signals"
      >
        Refresh
      </button>
    </footer>
  </section>
</template>

<script setup lang="ts">
import { computed, onMounted, onUnmounted, watch } from 'vue';
import { useHubSignals } from '@/composables/useHubSignals';

const {
  hub,
  loading,
  error,
  topSignals,
  lastUpdated,
  load,
  startAutoRefresh,
} = useHubSignals();

const hubName = computed(() => hub);
const hasData = computed(() => topSignals.value.length > 0);

function formatTime(iso?: string | null) {
  if (!iso) return '';
  try {
    return new Intl.RelativeTimeFormat(undefined, { numeric: 'auto' }).format(
      Math.round((new Date(iso).getTime() - Date.now()) / 60000),
      'minute'
    );
  } catch {
    return new Date(iso).toLocaleString();
  }
}

function refresh() {
  load();
}

let stopAutoRefresh: (() => void) | undefined;
onMounted(() => {
  stopAutoRefresh = startAutoRefresh();
});
onUnmounted(() => {
  if (stopAutoRefresh) stopAutoRefresh();
});

// Re-fetch when hub changes (if made reactive later)
watch(hub, () => load(), { immediate: false });
</script>
