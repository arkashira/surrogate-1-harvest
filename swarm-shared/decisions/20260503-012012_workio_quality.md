# workio / quality

## Final consolidated implementation (best of both proposals, corrected + actionable)

### 1. Diagnosis (resolved)

- **LINE retries** → duplicate webhook deliveries must be deduplicated.
- **Race conditions** on punch create (find-then-insert) → need DB-level uniqueness + atomic upsert/locking.
- **Multiple open punches** → enforce at DB level with a partial unique index.
- **Partial writes** (punch + notification) → wrap in transaction.
- **Auditability** → record idempotency events and punch state transitions.

Chosen approach:
- Use a **webhook idempotency table** keyed by deterministic LINE event key (event-id when available; otherwise stable fingerprint of employee+event-type+timestamp).
- Add **partial unique index** to enforce single open punch per employee per tenant at the DB level.
- Use **atomic upsert** (`INSERT ... ON CONFLICT DO NOTHING`) + **row-level locking** (`SELECT ... FOR UPDATE`) where business logic requires read-before-write.
- Perform punch + notification writes in **one transaction**; record idempotency entry within the same transaction.
- Add **punch_audit** table for state transition traceability.

```sql
-- 1) Prevent multiple open punches per employee per tenant
CREATE UNIQUE INDEX IF NOT EXISTS idx_one_open_punch_per_employee
ON punches (employee_id, tenant_id)
WHERE clock_out_time IS NULL;

-- 2) Idempotency for LINE/webhook events (lightweight)
CREATE TABLE IF NOT EXISTS webhook_idempotency (
  idempotency_key TEXT PRIMARY KEY,
  event_type      TEXT NOT NULL,
  employee_id     TEXT NOT NULL,
  tenant_id       TEXT NOT NULL,
  punch_id        TEXT,               -- optional: link to affected punch
  created_at      TIMESTAMPTZ DEFAULT NOW(),
  payload_hash    TEXT
);

-- Optional cleanup: expire idempotency keys after 48h (run as periodic job)
CREATE INDEX IF NOT EXISTS idx_webhook_idempotency_expiry
ON webhook_idempotency (created_at);

-- 3) Audit trail for punch state transitions
CREATE TABLE IF NOT EXISTS punch_audit (
  id             UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  punch_id       TEXT NOT NULL,
  employee_id    TEXT NOT NULL,
  tenant_id      TEXT NOT NULL,
  transition     TEXT NOT NULL,      -- e.g. 'clock_in', 'clock_out', 'rejected_duplicate'
  previous_state JSONB,
  new_state      JSONB,
  created_at     TIMESTAMPTZ DEFAULT NOW(),
  created_by     TEXT                -- e.g. 'line_webhook'
);
```

### 3. Service implementation (TypeScript)

File: `/opt/axentx/workio/server/src/services/punchService.ts`

```ts
import { Pool, PoolClient } from 'pg';
import crypto from 'crypto';

const pool = new Pool({ connectionString: process.env.DATABASE_URL });

type PunchRow = {
  id: string;
  employee_id: string;
  tenant_id: string;
  clock_in_time: Date;
  clock_out_time: Date | null;
};

function makeIdempotencyKey(employeeId: string, eventType: string, lineTimestamp: string, lineUserId: string) {
  // Prefer stable LINE event.id if available; fallback to deterministic fingerprint
  return `line:${eventType}:${employeeId}:${lineUserId}:${lineTimestamp}`;
}

function hashPayload(payload: any) {
  return crypto.createHash('sha256').update(JSON.stringify(payload)).digest('hex');
}

async function recordIdempotency(
  trx: PoolClient,
  key: string,
  eventType: string,
  employeeId: string,
  tenantId: string,
  punchId: string | null,
  payload: any
) {
  await trx.query(
    `INSERT INTO webhook_idempotency (idempotency_key, event_type, employee_id, tenant_id, punch_id, payload_hash)
     VALUES ($1, $2, $3, $4, $5, $6)`,
    [key, eventType, employeeId, tenantId, punchId, hashPayload(payload)]
  );
}

async function recordAudit(
  trx: PoolClient,
  punchId: string,
  employeeId: string,
  tenantId: string,
  transition: string,
  previousState: any,
  newState: any,
  createdBy: string
) {
  await trx.query(
    `INSERT INTO punch_audit (punch_id, employee_id, tenant_id, transition, previous_state, new_state, created_by)
     VALUES ($1, $2, $3, $4, $5, $6, $7)`,
    [punchId, employeeId, tenantId, transition, JSON.stringify(previousState), JSON.stringify(newState), createdBy]
  );
}

export async function handleClockIn(
  employeeId: string,
  tenantId: string,
  lineEventId: string | null,
  lineTimestamp: string,
  lineUserId: string,
  rawPayload: any
) {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    const idemKey = lineEventId || makeIdempotencyKey(employeeId, 'clock_in', lineTimestamp, lineUserId);

    // Idempotency check
    const idem = await client.query(
      `SELECT punch_id FROM webhook_idempotency WHERE idempotency_key = $1`,
      [idemKey]
    );
    if (idem.rows.length > 0) {
      // Fetch existing open punch to return
      const existing = await client.query<PunchRow>(
        `SELECT * FROM punches WHERE employee_id = $1 AND tenant_id = $2 AND clock_out_time IS NULL`,
        [employeeId, tenantId]
      );
      await client.query('ROLLBACK');
      return { alreadyProcessed: true, punch: existing.rows[0] || null };
    }

    // Try atomic insert; partial unique index prevents double-open
    const now = new Date();
    const insertResult = await client.query<PunchRow>(
      `INSERT INTO punches (employee_id, tenant_id, clock_in_time, clock_out_time, created_at)
       VALUES ($1, $2, $3, NULL, $4)
       ON CONFLICT DO NOTHING
       RETURNING *`,
      [employeeId, tenantId, now, now]
    );

    let punch: PunchRow;
    if (insertResult.rows.length > 0) {
      punch = insertResult.rows[0];
    } else {
      // Conflict: fetch existing open punch under lock to return consistent state
      const existing = await client.query<PunchRow>(
        `SELECT * FROM punches WHERE employee_id = $1 AND tenant_id = $2 AND clock_out_time IS NULL FOR UPDATE`,
        [employeeId, tenantId]
      );
      if (existing.rows.length === 0) {
        // Should not happen if index is present; defensive fallback
        throw new Error('Unable to create or locate open punch after conflict');
      }
      punch = existing.rows[0];
    }

    // Record idempotency and audit within same transaction
    await recordIdempotency(client, idemKey, 'clock_in', employeeId, tenantId, punch.id, rawPayload);
    await recordAudit(client, punch.id, employeeId, tenantId, 'clock_in', null, punch, 'line_webhook');

    await client.query('COMMIT');
    return { processed: true, punch };
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  } finally {
    client.release();
  }
}
