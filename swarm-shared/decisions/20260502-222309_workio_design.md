# workio / design

## Final Synthesized Implementation (Best of Both Candidates)

### 1. Diagnosis (Consensus)
- **Critical**: No tenant cache on `/webhook/line` → DB burst risk on LINE traffic spikes.
- **Critical**: Missing correlation IDs → untraceable incidents across webhook → service → DB.
- **Critical**: No `/health` or `/ready` → unsafe for k8s/PM2 rolling deploys and load balancer probes.
- **High**: Ad-hoc logging (mixed `console.log`/`error`) → unparseable in production observability.
- **High**: LINE signature verification lacks early rejection + structured audit trail.
- **Medium**: Clock-in/out endpoint lacks idempotency → duplicate attendance on retries.

### 2. Scope (Concrete Files)
- `workio/server/src/middleware/tenant.ts`

#### `workio/server/src/middleware/tenant.ts`
```ts
import { LRUCache } from 'lru-cache';
import type { Request, Response, NextFunction } from 'express';
import { pool } from '../db';

type Tenant = {
  id: string;
  name: string;
  channelId: string;
  config: Record<string, any>;
};

const tenantCache = new LRUCache<string, Tenant>({
  max: 500,
  ttl: 1000 * 60 * 10, // 10 minutes
});

export async function resolveTenant(
  req: Request,
  res: Response,
  next: NextFunction
) {
  const channelId =
    req.headers['x-line-channel-id'] ||
    (req.body as any)?.destination ||
    req.query.channelId;

  if (!channelId) {
    return res.status(400).json({ error: 'channelId missing' });
  }

  const cached = tenantCache.get(String(channelId));
  if (cached) {
    (req as any).tenant = cached;
    (req as any).tenantSource = 'cache';
    return next();
  }

  try {
    const { rows } = await pool.query(
      `SELECT id, name, channel_id AS "channelId", config
       FROM tenants
       WHERE channel_id = $1 AND deleted_at IS NULL
       LIMIT 1`,
      [channelId]
    );

    if (!rows.length) {
      return res.status(404).json({ error: 'tenant not found' });
    }

    const tenant: Tenant = {
      id: rows[0].id,
      name: rows[0].name,
      channelId: rows[0].channelId,
      config: rows[0].config || {},
    };

    tenantCache.set(String(channelId), tenant);
    (req as any).tenant = tenant;
    (req as any).tenantSource = 'db';
    next();
  } catch (err) {
    next(err);
  }
}
```

import { v4 as uuidv4 } from 'uuid';
import type { Request, Response, NextFunction } from 'express';

export function requestLogger(req: Request, res: Response, next: NextFunction) {
  const correlationId = req.headers['x-correlation-id'] || uuidv4();
  (req as any).correlationId = correlationId;

  const start = Date.now();

  res.on('finish', () => {
    const duration = Date.now() - start;
    const log = {
      ts: new Date().toISOString(),
      correlationId,
      method: req.method,
      path: req.path,
      status: res.statusCode,
      durationMs: duration,
      tenantId: (req as any).tenant?.id || null,
      tenantSource: (req as any).tenantSource || null,
      ip: req.ip,
      userAgent: req.get('User-Agent'),
    };
    console.log(JSON.stringify(log));
  });

  next();
}

#### `workio/server/src/middleware/idempotency.ts`
```ts
import type { Request, Response, NextFunction } from 'express';
import { pool } from '../db';

const IDEMPOTENCY_TTL_MINUTES = 60;

export async function idempotencyGuard(
  req: Request,
  res: Response,
  next: NextFunction
) {
  // Apply only to state-changing endpoints that need it (e.g., attendance)
  if (req.method !== 'POST' || !req.path.includes('/attendance')) {
    return next();
  }

  const idempotencyKey = req.headers['idempotency-key'] as string;
  if (!idempotencyKey) {
    return res.status(400).json({ error: 'idempotency-key header required' });
  }

  const tenantId = (req as any).tenant?.id;
  if (!tenantId) {
    return res.status(400).json({ error: 'tenant resolution required for idempotency' });
  }

  try {
    // Upsert idempotency record; if conflict and recent, return previous response
    const upsert = await pool.query(
      `INSERT INTO idempotency_keys (key, tenant_id, response_status, response_body, expires_at)
       VALUES ($1, $2, $3, $4, NOW() + INTERVAL '${IDEMPOTENCY_TTL_MINUTES} minutes')
       ON CONFLICT (key, tenant_id) DO UPDATE
       SET last_used_at = NOW()
       WHERE idempotency_keys.expires_at > NOW()
       RETURNING response_status, response_body, (xmax = 0) AS inserted`,
      [idempotencyKey, tenantId, 0, '']
    );

    const row = upsert.rows[0];
    // If xmax > 0 (update on existing, non-expired), this is a replay
    if (row && !row.inserted && row.response_status) {
      return res
        .status(row.response_status)
        .json(JSON.parse(row.response_body || '{}'));
    }

    // Attach idempotency metadata for response capture
    (req as any).idempotencyKey = idempotencyKey;
    (req as any).idempotencyTenantId = tenantId;
    next();
  } catch (err) {
    // Fail open on idempotency subsystem errors to avoid blocking business flow
    console.warn(JSON.stringify({
      ts: new Date().toISOString(),
      correlationId: (req as any).correlationId,
      tenantId,
      event: 'idempotency_error',
      error: String(err),
    }));
    next();
  }
}

export async function recordIdempotentResponse(
  req: Request,
  res: Response,
  next: NextFunction
) {
  // Capture final response for idempotency
  const originalSend = res.json;
  // eslint-disable-next-line @typescript-eslint/no-explicit-any
  (res as any).json = function (body: any) {
    const key = (req as any).idempotencyKey;
    const tenantId = (req as any).idempotencyTenantId;
    if (key && tenantId && res.statusCode >= 200 && res.statusCode < 300) {
      pool.query(
        `UPDATE idempotency_keys
         SET response_status = $1, response_body = $2
         WHERE key = $3 AND tenant_id = $4`,
        [res.statusCode, JSON.stringify(body), key, tenantId]
      ).catch((err) => {
        console.warn(JSON.stringify({
          ts: new Date().toISOString(),
          event: 'idempotency_record_failed',
          key,
          tenantId,
          error: String(err),
        }));
      });
    }
    return originalSend.call(this, body);
  };
  next();
}
```
