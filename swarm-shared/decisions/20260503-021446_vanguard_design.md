# vanguard / design

## 1. Diagnosis

- No persisted `(repo, dateFolder) → file-list` manifest: every training run triggers authenticated `list_repo_tree` calls, burning HF API quota and risking 429s.
- Training/data loader likely uses `load_dataset(streaming=True)` or per-file authenticated fetches instead of CDN-only downloads, exposing training to auth rate limits.
- Heterogeneous repo schemas can cause `pyarrow.CastError` if raw files are loaded without projection to `{prompt, response}` at parse time.
- No reuse guard for Lightning Studio: scripts may create new studios instead of reusing running ones, wasting 80hr/mo quota.
- No idle-stop resilience: Lightning Studio idle timeout kills training; no pre-run status check or auto-restart.

## 2. Proposed change

Create `/opt/axentx/vanguard/train.py` (new) and `/opt/axentx/vanguard/manifest.json` (generated) with the following scope:
- Single Mac-side script to generate `manifest.json` for a chosen `(repo, dateFolder)` using one authenticated `list_repo_tree` call.
- Lightning training script that reads `manifest.json` and downloads only via HF CDN (`resolve/main/`) with zero authenticated calls.
- Project each file to `{prompt, response}` at parse time to avoid schema mismatches.
- Reuse running Lightning Studio and auto-restart if stopped.

## 3. Implementation

```bash
# /opt/axentx/vanguard/generate_manifest.py
#!/usr/bin/env python3
"""
Generate manifest.json for a repo+dateFolder to avoid repeated HF API calls.
Run from Mac (or any env with huggingface_hub) after rate-limit window clears.
"""
import json
import os
import sys
from pathlib import Path

from huggingface_hub import HfApi

REPO = os.getenv("HF_DATASET_REPO", "datasets/your-repo")
DATE_FOLDER = os.getenv("HF_DATE_FOLDER", "2026-04-29")
OUT_PATH = Path(__file__).parent / "manifest.json"

def main() -> None:
    api = HfApi()
    # non-recursive list for the date folder; paginated safely
    files = api.list_repo_tree(repo_id=REPO, path=DATE_FOLDER, recursive=False)
    entries = []
    for f in files:
        if f.path.endswith((".parquet", ".jsonl", ".json")):
            entries.append(
                {
                    "repo": REPO,
                    "path": f.path,
                    "cdn_url": f"https://huggingface.co/datasets/{REPO}/resolve/main/{f.path}",
                }
            )
    manifest = {"repo": REPO, "date_folder": DATE_FOLDER, "files": entries}
    OUT_PATH.write_text(json.dumps(manifest, indent=2))
    print(f"Wrote {len(entries)} entries to {OUT_PATH}")

if __name__ == "__main__":
    main()
```

```python
# /opt/axentx/vanguard/train.py
"""
Lightning training script that uses CDN-only fetches via manifest.json.
Run inside Lightning Studio (reuse existing).
"""
import json
import os
import pyarrow as pa
import pyarrow.parquet as pq
import requests
from pathlib import Path
from typing import Iterator

import lightning as L

MANIFEST_PATH = Path(__file__).parent / "manifest.json"
CACHE_DIR = Path(__file__).parent / ".cdn_cache"
CACHE_DIR.mkdir(exist_ok=True)

def load_manifest() -> dict:
    if not MANIFEST_PATH.exists():
        raise FileNotFoundError(f"Generate manifest first: {MANIFEST_PATH}")
    return json.loads(MANIFEST_PATH.read_text())

def cdn_fetch(url: str, cache_name: str) -> Path:
    out = CACHE_DIR / cache_name
    if out.exists():
        return out
    resp = requests.get(url, timeout=60)
    resp.raise_for_status()
    out.write_bytes(resp.content)
    return out

def project_to_prompt_response(file_path: Path):
    """Project heterogeneous file to {prompt, response} only."""
    try:
        table = pq.read_table(file_path, columns=["prompt", "response"])
    except (pa.ArrowInvalid, KeyError, OSError):
        # fallback: try jsonl
        try:
            with file_path.open() as f:
                rows = [json.loads(l) for l in f if l.strip()]
            pairs = [(r.get("prompt", ""), r.get("response", "")) for r in rows]
            table = pa.table({"prompt": [p for p, _ in pairs], "response": [r for _, r in pairs]})
        except Exception:
            table = pa.table({"prompt": [], "response": []})
    return table

def cdn_data_iterator(manifest: dict) -> Iterator[pa.Table]:
    for f in manifest["files"]:
        cache_name = f"{manifest['date_folder']}_{Path(f['path']).name}"
        local_path = cdn_fetch(f["cdn_url"], cache_name)
        table = project_to_prompt_response(local_path)
        if len(table) > 0:
            yield table

class SurrogateDataModule(L.LightningDataModule):
    def __init__(self, batch_size: int = 8):
        super().__init__()
        self.batch_size = batch_size

    def setup(self, stage=None):
        self.manifest = load_manifest()

    def train_dataloader(self):
        # simple concatenation; for large data, stream and tokenize on-the-fly
        tables = list(cdn_data_iterator(self.manifest))
        if not tables:
            raise RuntimeError("No data found in manifest")
        full = pa.concat_tables(tables)
        # convert to torch dataset (example)
        import torch
        from torch.utils.data import Dataset, DataLoader

        class SimpleDs(Dataset):
            def __len__(self):
                return len(full)

            def __getitem__(self, idx):
                row = full.slice(idx, 1).to_pydict()
                return {
                    "prompt": row["prompt"][0],
                    "response": row["response"][0],
                }

        return DataLoader(SimpleDs(), batch_size=self.batch_size, shuffle=True)

def main():
    # Reuse running studio; avoid quota waste
    teamspace = L.Teamspace(name=os.getenv("LIGHTNING_TEAMSPACE", "default"))
    existing = [s for s in teamspace.studios if s.name == "vanguard-train" and s.status == "running"]
    if existing:
        studio = existing[0]
        print(f"Reusing running studio: {studio.name}")
    else:
        studio = L.Studio(
            name="vanguard-train",
            script=str(Path(__file__).absolute()),
            create_ok=True,
        )

    # Guard against idle-stop death
    if studio.status != "running":
        print("Studio not running; starting L40S")
        studio.start(machine=L.Machine.L40S)

    # Run training (minimal example)
    dm = SurrogateDataModule(batch_size=8)
    dm.setup()
    loader = dm.train_dataloader()
    for batch in loader:
        print(batch)
        break
    print("CDN-only training step completed (no authenticated HF API calls).")

if __name__ == "__main__":
    main()
```
