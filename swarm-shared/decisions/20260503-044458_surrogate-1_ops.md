# surrogate-1 / ops

## Final Implementation Plan (≤2h)

**Highest-value improvement**: Replace fragile shell-based ingestion with a **manifest-driven, CDN-bypass Python worker** that eliminates HF API rate limits during training data loads and prevents mixed-schema CastErrors.

### Changes

1. **Add `bin/worker.py`** — single-file, manifest-first worker:
   - Reads `manifest.json` (date + repo + folder list) produced by a one-time Mac script.
   - Downloads only files matching its `SHARD_ID` via raw CDN URLs (`resolve/main/...`), bypassing `/api/` rate limits entirely.
   - Projects heterogeneous schemas to `{prompt, response}` at parse time (no `load_dataset(streaming=True)` on mixed schemas).
   - Emits `batches/public-merged/{date}/shard{N}-{HHMMSS}.jsonl` with deterministic slug-hash sharding.
   - Uses central `lib/dedup.py` for cross-run dedup (SQLite md5 store).

2. **Add `bin/gen-manifest.py`** — run once from Mac after rate-limit window clears:
   - Calls `list_repo_tree(path, recursive=False)` per date folder (non-recursive to avoid 100× pagination).
   - Saves `manifest.json` mapping `date -> [files]`.
   - Commits to repo so workers never call HF listing APIs during ingestion.

3. **Update `bin/dataset-enrich.sh`** → thin wrapper:
   - Sets `SHARD_ID`, `TOTAL_SHARDS=16`, `SHELL=/bin/bash`.
   - Executes `python3 bin/worker.py` (no inline complex logic).

4. **Update `.github/workflows/ingest.yml`**:
   - Matrix `shard: [0..15]`.
   - Uses `actions/checkout` to pick up `manifest.json`.
   - No HF API listing during workflow — only CDN downloads + single push per shard.

5. **Update `requirements.txt`**:
   - Add `requests`, keep `datasets`, `huggingface_hub`, `pyarrow`, `numpy`.

#### `bin/gen-manifest.py`
```python
#!/usr/bin/env python3
"""
Generate manifest.json once per date folder to avoid HF API listing during ingestion.
Run from Mac after rate-limit window clears.
"""
import json, os, sys
from datetime import datetime, timezone
from huggingface_hub import HfApi

REPO = "datasets/axentx/surrogate-1-training-pairs"
OUT = "manifest.json"

def main():
    api = HfApi()
    # Top-level tree (non-recursive) to list date folders
    root = api.list_repo_tree(REPO, path="", recursive=False)
    manifest = {"generated_at": datetime.now(timezone.utc).isoformat(), "repos": {}}

    for item in root:
        if not item.path or item.type != "directory":
            continue
        date_path = item.path
        # List files in this date folder (non-recursive)
        files = api.list_repo_tree(REPO, path=date_path, recursive=False)
        file_list = [f.path for f in files if f.type == "file"]
        if file_list:
            manifest["repos"][REPO] = manifest["repos"].get(REPO, {})
            manifest["repos"][REPO][date_path] = sorted(file_list)

    with open(OUT, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Wrote {OUT} with {sum(len(v) for r in manifest['repos'].values() for v in r.values())} files")

if __name__ == "__main__":
    main()
```

#### `bin/worker.py`
```python
#!/usr/bin/env python3
"""
Manifest-driven, CDN-bypass worker.
Usage: SHARD_ID=0 TOTAL_SHARDS=16 python3 bin/worker.py
"""
import hashlib, json, os, sys, requests, datetime, pyarrow as pa, pyarrow.parquet as pq
from pathlib import Path

sys.path.insert(0, str(Path(__file__).parent))
from lib.dedup import DedupStore

MANIFEST = Path(__file__).parent.parent / "manifest.json"
BASE_CDN = "https://huggingface.co"
REPO = "datasets/axentx/surrogate-1-training-pairs"
OUT_DIR = Path(__file__).parent.parent / "batches" / "public-merged"

SHARD_ID = int(os.getenv("SHARD_ID", "0"))
TOTAL_SHARDS = int(os.getenv("TOTAL_SHARDS", "16"))
HF_TOKEN = os.getenv("HF_TOKEN", "")

def shard_for(slug: str) -> int:
    return int(hashlib.md5(slug.encode()).hexdigest(), 16) % TOTAL_SHARDS

def project_to_pair(raw: dict) -> dict:
    """Project heterogeneous schema to {prompt, response} only."""
    prompt = raw.get("prompt") or raw.get("input") or raw.get("question") or ""
    response = raw.get("response") or raw.get("output") or raw.get("answer") or ""
    return {"prompt": str(prompt), "response": str(response)}

def download_cdn(path: str) -> bytes:
    url = f"{BASE_CDN}/{REPO}/resolve/main/{path}"
    r = requests.get(url, timeout=60)
    r.raise_for_status()
    return r.content

def parse_file(path: str) -> list[dict]:
    data = download_cdn(path)
    try:
        table = pq.read_table(pa.BufferReader(data))
    except Exception:
        # fallback: try jsonl
        lines = [ln for ln in data.decode("utf-8", errors="replace").splitlines() if ln.strip()]
        rows = [json.loads(ln) for ln in lines]
        return [project_to_pair(r) for r in rows]

    rows = table.to_pylist()
    return [project_to_pair(r) for r in rows]

def upload_artifact(date_folder: str, shard: int, pairs: list[dict]):
    if not pairs:
        return
    OUT_DIR.mkdir(parents=True, exist_ok=True)
    ts = datetime.datetime.utcnow().strftime("%H%M%S")
    out_path = OUT_DIR / date_folder / f"shard{shard}-{ts}.jsonl"
    out_path.parent.mkdir(parents=True, exist_ok=True)
    with open(out_path, "w") as f:
        for p in pairs:
            f.write(json.dumps(p, ensure_ascii=False) + "\n")

    # Push via huggingface_hub (single commit per shard)
    from huggingface_hub import upload_file
    repo_id = REPO
    repo_path = f"batches/public-merged/{date_folder}/{out_path.name}"
    upload_file(
        path_or_fileobj=str(out_path),
        path_in_repo=repo_path,
        repo_id=repo_id,
        token=HF_TOKEN,
        repo_type="dataset",
    )
    print(f"Uploaded {repo_path}")

def main():
    if not HF_TOKEN:
        print("HF_TOKEN required", file=sys.stderr)
        sys.exit(1)

    if not MANIFEST.exists():
        print(f"Manifest not found: {MANIFEST}", file=sys.stderr)
        sys.exit(1)

    with open(MANIFEST) as f:
        manifest = json.load(f)

    dedup = DedupStore()
    total_pairs = 0

    for repo_key, dates in manifest.get("repos", {}).items():
        for date_folder, files in dates.items():
            shard_pairs = []
            for file_path in files:
                slug = file_path
                if shard_for(slug) != SHARD_ID:
                    continue
                try:
                    pairs = parse_file(file_path)
                    for p in pairs:
                        # dedup by content hash
                        h = hashlib.md5(json.dumps(p, sort_keys=True).encode()).hexdigest()
                        if not dedup.seen(h):
                            shard_pairs.append(p)
                except Exception as e:
                    print(f"Skip {file_path}: {e}", file=sys.stderr)
                    continue

            total_pairs += len(shard_pairs)
            upload_artifact(date_folder, SHARD_ID, shard_pairs)

    print(f"Shard {SHARD_ID}/{TOTAL_SHARDS} completed: {total_pairs} new pairs")

if __name__ == "__main__":
    main()
```
