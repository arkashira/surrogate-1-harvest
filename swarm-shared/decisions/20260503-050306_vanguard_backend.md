# vanguard / backend

Below is the **single, consolidated implementation** that merges the strongest, non-contradictory parts of both proposals and prioritizes **correctness + concrete actionability**.

Key resolution choices:
- **Do not call HF `datasets`/`list_repo_tree` at training time** (both agree).  
  → Manifest is generated **once** (or on cadence) by an offline script; training consumes only the manifest + CDN URLs.
- **Use true content-addressing for cache-bust/reproducibility** (Candidate 1: hash file bytes; Candidate 2: use repo commit/LFS pointer).  
  → Hybrid: prefer `etag`/`commit` from HEAD request for CDN objects; fall back to deterministic path hash if unavailable. This avoids reading every parquet during manifest generation while remaining reproducible.
- **CDN-only, zero-auth training loop** (both agree).  
  → Use `requests` with streaming + `pyarrow.parquet` to avoid `CastError` via explicit schema selection.
- **Lightning Studio reuse** (both agree).  
  → Use `lightning` Teamspace lookup; if running studio exists, attach; else create.
- **File layout**:  
  - `/opt/axentx/vanguard/make_manifest.py` (new, offline, run on Mac/orchestrator)  
  - `/opt/axentx/vanguard/train.py` (modified training entrypoint)

### 1) `/opt/axentx/vanguard/make_manifest.py`
Generate a content-addressed manifest for one date folder. Run this **outside** training (e.g., on Mac after rate-limit window clears).

```python
#!/usr/bin/env python3
"""
Generate content-addressed manifest for a date folder.
Usage:
  DATE_FOLDER=2026-04-29 OUT_MANIFEST=manifest-2026-04-29.json \
  python make_manifest.py
"""
import json
import hashlib
import os
import sys
from pathlib import Path
from typing import Dict, List

import requests
from huggingface_hub import HfApi, DatasetFilter, HfFolder

REPO = os.getenv("HF_REPO", "datasets/surrogate-1")
DATE_FOLDER = os.getenv("DATE_FOLDER", "2026-04-29")
OUT_MANIFEST = Path(os.getenv("OUT_MANIFEST", f"manifest-{DATE_FOLDER}.json"))
HF_TOKEN = os.getenv("HF_TOKEN", HfFolder().get_token())

API = HfApi(token=HF_TOKEN)

HEADERS = {"Authorization": f"Bearer {HF_TOKEN}"} if HF_TOKEN else {}

def list_parquet_files(base_path: str) -> List[str]:
    """Non-recursive list of .parquet files under base_path."""
    try:
        items = API.list_repo_tree(repo_id=REPO, path=base_path, repo_type="dataset")
    except Exception as e:
        print(f"Failed to list {base_path}: {e}", file=sys.stderr)
        return []

    if not isinstance(items, list):
        return []

    # Deterministic order
    files = sorted(
        p.path for p in items
        if p.path.endswith(".parquet") and p.path.startswith(base_path)
    )
    return files

def content_addressable_tag(file_path: str) -> str:
    """
    Prefer commit/etag for content address without reading full file.
    Fallback: deterministic path hash.
    """
    cdn_url = f"https://huggingface.co/datasets/{REPO}/resolve/main/{file_path}"
    try:
        r = requests.head(cdn_url, headers=HEADERS, timeout=10)
        r.raise_for_status()
        etag = r.headers.get("etag", "")
        if etag:
            return etag.strip('"')
    except Exception:
        pass
    return hashlib.sha256(file_path.encode()).hexdigest()[:16]

def build_manifest() -> Dict:
    base = f"batches/mirror-merged/{DATE_FOLDER}"
    files = list_parquet_files(base)

    entries = []
    for p in files:
        entries.append({
            "path": p,
            "cdn_url": f"https://huggingface.co/datasets/{REPO}/resolve/main/{p}",
            "slug": Path(p).stem,
            "tag": content_addressable_tag(p),
        })

    # Reproducible manifest-level hash
    h = hashlib.sha256()
    for e in entries:
        h.update(e["path"].encode())
        h.update(e["tag"].encode())

    manifest = {
        "date_folder": DATE_FOLDER,
        "repo": REPO,
        "generated_by": "make_manifest.py",
        "entries": entries,
        "count": len(entries),
        "manifest_hash": h.hexdigest(),
    }
    return manifest

def main() -> None:
    manifest = build_manifest()
    OUT_MANIFEST.write_text(json.dumps(manifest, indent=2))
    print(f"Wrote {manifest['count']} entries to {OUT_MANIFEST}")

if __name__ == "__main__":
    main()
```

### 2) `/opt/axentx/vanguard/train.py`
Lightning training with **CDN-only fetches**, explicit schema selection (avoid `CastError`), and **studio reuse**.

```python
#!/usr/bin/env python3
"""
Lightning training with CDN-only fetches and studio reuse.
Requires manifest generated by make_manifest.py.
"""
import os
import json
import sys
import hashlib
from pathlib import Path
from typing import Dict, Any, Iterator

import torch
from torch.utils.data import IterableDataset, DataLoader
import pyarrow as pa
import pyarrow.parquet as pq
import lightning as L
from huggingface_hub import Teamspace

# ----------
# Manifest (embedded at launch)
# ----------
MANIFEST_PATH = Path(os.getenv("MANIFEST_PATH", "manifest-2026-04-29.json"))
if not MANIFEST_PATH.exists():
    print(f"Manifest not found: {MANIFEST_PATH}. Run make_manifest.py first.", file=sys.stderr)
    sys.exit(1)

with MANIFEST_PATH.open() as f:
    MANIFEST = json.load(f)

ENTRIES = MANIFEST["entries"]
if not ENTRIES:
    print("No entries in manifest.", file=sys.stderr)
    sys.exit(1)

# ----------
# CDN-only dataset (zero HF API calls during training)
# ----------
class CDNParquetDataset(IterableDataset):
    """
    Stream parquet files directly from CDN.
    - Avoids HF datasets API (no 429, no shard-order nondeterminism).
    - Explicit schema selection reduces CastError risk.
    """
    def __init__(
        self,
        entries: list[Dict[str, Any]],
        columns: list[str] | None = None,
        batch_size: int = 128,
        shuffle: bool = True,
        buffer_size: int = 4,  # number of files to prefetch
    ):
        super().__init__()
        self.entries = entries
        self.columns = columns
        self.batch_size = batch_size
        self.shuffle = shuffle
        self.buffer_size = max(1, buffer_size)

        # Deterministic epoch seed from manifest
        h = hashlib.sha256()
        h.update(json.dumps([e["path"] for e in entries], sort_keys=True).encode())
        self.epoch_seed = int(h.hexdigest()[:8], 16)

    def _stream_files(self) -> Iterator[Dict[str, Any]]:
        worker_info = torch.utils.data.get_worker_info()
        if worker_info is None:
            it = iter(self.entries)
        else:
            # Deterministic per-worker sharding
            it = (
                e for i, e in enumerate(self.entries)
                if i % worker_info.num_workers == worker_info.id
            )

        for entry in it:
            url = entry["cdn_url"]
            try:
                # Stream + selective column read
                with pa.input_stream(url) as source:
                    pf = pq.ParquetFile(source)
                    # Use only requested columns; if None, use all
                    cols = self.columns or pf.schema.names
                    # Avoid type surprises: read as pyarrow Table then convert
                    table = pf.read(columns=cols)
                    for batch in table.to_batches(max_chunksize=self.batch_size):
                        yield batch.to_pydict
