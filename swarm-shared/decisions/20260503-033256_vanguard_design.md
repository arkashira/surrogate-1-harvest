# vanguard / design

## Final Synthesized Implementation

Below is the single, canonical solution. It merges the strongest architectural choices from both candidates:

- **CDN-first, zero-API training** (both agree): manifest lists files once; training consumes only `resolve/main/...` URLs.
- **Deterministic, content-addressed pinning** (Candidate 2): manifest includes SHA-256 digests and a manifest-level hash so runs are reproducible and verifiable.
- **Studio reuse guard** (Candidate 2): lightweight helper to reuse running Lightning Studios and avoid quota waste.
- **Concrete, copy-paste-ready code** (Candidate 1): one new module plus a minimal training stub.
- **Correctness fixes**: use `huggingface_hub` for reliable tree listing and optional HEAD-based size/hash population; omit `Authorization` header for public CDN downloads; add retries and timeouts; handle 429 explicitly.

# /opt/axentx/vanguard/vanguard/manifest.py
import hashlib
import json
import os
import time
from datetime import datetime, timezone
from pathlib import Path
from typing import Dict, List, Optional

import requests

try:
    from huggingface_hub import list_repo_tree
    HF_HUB_AVAILABLE = True
except Exception:  # pragma: no cover
    HF_HUB_AVAILABLE = False

HF_DATASETS_CDN = "https://huggingface.co/datasets"


def _sha256_file(path: Path) -> str:
    h = hashlib.sha256()
    with open(path, "rb") as f:
        for chunk in iter(lambda: f.read(8192), b""):
            h.update(chunk)
    return h.hexdigest()


def list_date_folder(
    repo: str,
    date_folder: str,
    token: Optional[str] = None,
) -> List[str]:
    """
    Non-recursive file listing for one date folder.
    Returns paths relative to repo root, e.g. ['2026-04-29/file.parquet'].
    Prefer huggingface_hub; fall back to REST API if unavailable.
    """
    if HF_HUB_AVAILABLE:
        items = list_repo_tree(
            repo_id=repo,
            path=date_folder,
            repo_type="dataset",
            token=token,
        )
        paths = sorted(
            item.rfilename
            for item in items
            if item.type == "file" and item.rfilename.startswith(date_folder)
        )
        return paths

    # Fallback: direct REST API
    headers = {"Authorization": f"Bearer {token}"} if token else {}
    url = f"https://huggingface.co/api/datasets/{repo}/tree"
    params = {"path": date_folder, "recursive": "false"}
    for attempt in range(3):
        resp = requests.get(url, headers=headers, params=params, timeout=30)
        if resp.status_code == 429:
            wait = 60 * (2 ** attempt)
            time.sleep(wait)
            continue
        resp.raise_for_status()
        items = resp.json()
        paths = [item["path"] for item in items if item.get("type") == "file"]
        return sorted(paths)
    raise RuntimeError("Failed to list folder after retries")


def populate_file_info(
    repo: str,
    file_path: str,
    token: Optional[str] = None,
) -> Dict:
    """
    Return CDN URL, size, and SHA-256 for one file.
    Uses HEAD when possible; falls back to local download for hash.
    """
    cdn_url = f"{HF_DATASETS_CDN}/{repo}/resolve/main/{file_path}"
    info = {"path": file_path, "cdn_url": cdn_url, "size": None, "sha256": None}

    # Try HEAD for size (no auth needed for public CDN)
    try:
        head = requests.head(cdn_url, timeout=15, allow_redirects=True)
        if head.status_code == 200:
            info["size"] = int(head.headers.get("content-length", 0)) or None
    except Exception:
        pass

    # Compute SHA-256 by downloading if not available via headers
    if info["sha256"] is None:
        try:
            from huggingface_hub import hf_hub_download
            local = hf_hub_download(
                repo_id=repo,
                filename=file_path,
                repo_type="dataset",
                token=token,
            )
            info["sha256"] = _sha256_file(Path(local))
            info["size"] = Path(local).stat().st_size
        except Exception:
            # Best-effort: leave sha256 as None if unavailable
            pass
    return info


def build_cdn_manifest(
    repo: str,
    date_folder: str,
    out_path: Path,
    token: Optional[str] = None,
    populate_hashes: bool = True,
) -> Dict:
    """
    Build a CDN-first manifest:
    {
      "repo": "...",
      "date_folder": "2026-04-29",
      "created_utc": "...",
      "files": [
        {"path": "...", "cdn_url": "...", "size": ..., "sha256": "..."}
      ],
      "manifest_sha256": "..."
    }
    """
    paths = list_date_folder(repo, date_folder, token=token)
    files: List[Dict] = []
    for p in paths:
        if populate_hashes:
            files.append(populate_file_info(repo, p, token=token))
        else:
            files.append(
                {
                    "path": p,
                    "cdn_url": f"{HF_DATASETS_CDN}/{repo}/resolve/main/{p}",
                    "size": None,
                    "sha256": None,
                }
            )

    manifest = {
        "repo": repo,
        "date_folder": date_folder,
        "created_utc": datetime.now(timezone.utc).isoformat(),
        "files": files,
    }
    manifest_bytes = json.dumps(manifest, indent=2, sort_keys=True).encode()
    manifest["manifest_sha256"] = hashlib.sha256(manifest_bytes).hexdigest()

    out_path.parent.mkdir(parents=True, exist_ok=True)
    out_path.write_bytes(manifest_bytes)
    return manifest


def cdn_only_loader(manifest_path: Path):
    """
    Yield (rel_path, cdn_url) from manifest for CDN-only streaming.
    """
    manifest = json.loads(manifest_path.read_text())
    for f in manifest["files"]:
        yield f["path"], f["cdn_url"]


def safe_cdn_download(
    repo: str,
    file_path: str,
    out_dir: Path,
    token: Optional[str] = None,
) -> Path:
    """
    Download one file via public CDN (no Authorization header).
    """
    out_dir.mkdir(parents=True, exist_ok=True)
    out_file = out_dir / Path(file_path).name
    if out_file.exists():
        return out_file

    url = f"{HF_DATASETS_CDN}/{repo}/resolve/main/{file_path}"
    # Public CDN does not require Authorization; omit header.
    resp = requests.get(url, timeout=120, stream=True)
    resp.raise_for_status()
    with open(out_file, "wb") as f:
        for chunk in resp.iter_content(chunk_size=8192):
            f.write(chunk)
    return out_file
