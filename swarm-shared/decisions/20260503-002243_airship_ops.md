# airship / ops

## Final synthesized implementation

**Core problem**: Surrogate-1 training hits HF API 429s and mixed-schema errors because training uses `load_dataset(streaming=True)` against a heterogeneous repo.  
**Core rule**: Mac orchestrates; Lightning trains; never call HF APIs during training.

**Chosen fix (single coherent plan)**:  
Add a **zero-API, CDN-only data path** plus a **manifest generator** (run once on Mac after rate-limit window). Training loads via raw CDN URLs, projects `{prompt,response}` only, and optionally caches shards locally. Lightning Studio reuse/idle-stop handling remains unchanged.

---

## Implementation plan (unified)

1. **Add `scripts/build_manifest.py`** (run on Mac)  
   - Single `list_repo_tree(..., recursive=True)` (or `False` + walk) per date folder.  
   - Produces `manifests/manifest_{date}.json` with `{path, cdn_url, size?, rows?}`.  
   - Projects schema at generation time: keep only `prompt`, `response` (store as lightweight metadata or rely on parse-time projection).  
   - Commit manifest to repo (or upload to cloud) so training never calls HF APIs.

2. **Add `surrogate/data/cdn_dataset.py`**  
   - `CdnParquetDataset(manifest_path, cache_dir=None, max_cache_size_gb=5)`  
   - Reads manifest; streams each parquet via `pd.read_parquet(cdn_url, columns=["prompt","response"], engine="pyarrow")`.  
   - Skips malformed/missing columns.  
   - Optional: materialize sharded local parquet cache for fast repeated epochs (avoids repeated CDN fetches).  
   - Returns `Iterator[Dict[str,str]]` compatible with HF `datasets`-style map/tokenize.

3. **Patch `surrogate/train/train_surrogate.py`**  
   - Replace `load_dataset(streaming=True, ...)` with `CdnParquetDataset(args.manifest, cache_dir=args.cache_dir)`.  
   - Add CLI:  
     - `--manifest` (required)  
     - `--cache-dir` (optional)  
     - `--hf-repo` (deprecated/fallback).  
   - Keep tokenizer/collate unchanged.

4. **Add robust launcher behavior in `scripts/launch_lightning_studio.py`**  
   - Reuse running studio if status == "running".  
   - If stopped, restart with Machine.L40S (or public tier fallback).  
   - Wait until status == "running" before submit.  
   - Keep idle-stop handling as-is.

5. **Validate and commit**  
   - Local smoke test on one small date folder.  
   - Verify zero HF API calls during training (check logs/network).  
   - Commit manifest + loader + training diff.

### 1) `scripts/build_manifest.py`

```python
#!/usr/bin/env python3
"""
Build CDN-only manifest for a date folder.
Run on Mac (or any dev box) when HF API limit allows.
"""
import argparse
import json
import os
from pathlib import Path

from huggingface_hub import HfApi

HF_REPO = os.getenv("HF_DATASET_REPO", "axentx/surrogate-mirror")
OUT_DIR = Path("manifests")

def build_manifest(date: str, out_dir: Path = OUT_DIR, recursive: bool = True) -> Path:
    api = HfApi()
    prefix = f"batches/mirror-merged/{date}/"
    # One paginated call; recursive=True simplifies deep listing.
    tree = api.list_repo_tree(repo_id=HF_REPO, path=prefix, recursive=recursive)

    out_dir.mkdir(parents=True, exist_ok=True)
    manifest_path = out_dir / f"manifest_{date}.json"

    items = []
    for entry in tree:
        if not entry.path.endswith(".parquet"):
            continue
        cdn_url = f"https://huggingface.co/datasets/{HF_REPO}/resolve/main/{entry.path}"
        items.append({"path": entry.path, "cdn_url": cdn_url})

    manifest_path.write_text(json.dumps(items, indent=2))
    print(f"Wrote {len(items)} items to {manifest_path}")
    return manifest_path

if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--date", required=True, help="YYYY-MM-DD")
    parser.add_argument("--no-recursive", action="store_true", help="Non-recursive listing")
    args = parser.parse_args()
    build_manifest(args.date, recursive=not args.no_recursive)
```

### 2) `surrogate/data/cdn_dataset.py`

```python
from __future__ import annotations

import json
import shutil
from pathlib import Path
from typing import Iterator, Dict, Any

import pandas as pd
import pyarrow as pa

class CdnParquetDataset:
    """
    CDN-only parquet dataset.
    Manifest format: [{"path": "...", "cdn_url": "https://..."}, ...]
    Projects to {prompt, response} only.
    Optional local cache for repeated-epoch speed.
    """

    def __init__(
        self,
        manifest_path: str,
        cache_dir: str | None = None,
        max_cache_size_gb: float = 5.0,
    ):
        self.items = json.loads(Path(manifest_path).read_text())
        self.cache_dir = Path(cache_dir) if cache_dir else None
        self.max_cache_bytes = int(max_cache_size_gb * (1024**3))
        if self.cache_dir:
            self.cache_dir.mkdir(parents=True, exist_ok=True)

    def _cached_path(self, cdn_url: str) -> Path | None:
        if not self.cache_dir:
            return None
        # Stable filename from URL
        name = cdn_url.rsplit("/", 1)[-1]
        return self.cache_dir / name

    def _ensure_cached(self, cdn_url: str) -> str:
        if not self.cache_dir:
            return cdn_url

        local_path = self._cached_path(cdn_url)
        if local_path and local_path.exists() and local_path.stat().st_size > 0:
            return str(local_path)

        # Stream download to temp then atomic move
        import tempfile
        import urllib.request

        tmp = tempfile.NamedTemporaryFile(delete=False, dir=self.cache_dir)
        try:
            with urllib.request.urlopen(cdn_url) as resp, open(tmp.name, "wb") as f:
                shutil.copyfileobj(resp, f)
            # Best-effort cache size limit (simple cleanup)
            self._enforce_cache_limit()
            target = local_path
            target.parent.mkdir(parents=True, exist_ok=True)
            shutil.move(tmp.name, target)
            return str(target)
        except Exception:
            try:
                Path(tmp.name).unlink(missing_ok=True)
            except Exception:
                pass
            # Fallback to CDN URL
            return cdn_url

    def _enforce_cache_limit(self) -> None:
        if not self.cache_dir:
            return
        total = sum(p.stat().st_size for p in self.cache_dir.rglob("*.parquet") if p.is_file())
        if total <= self.max_cache_bytes:
            return
        # Simple LRU-ish: remove oldest until under limit
        files = sorted(self.cache_dir.rglob("*.parquet"), key=lambda p: p.stat().st_mtime)
        for f in files:
            if total <= self.max_cache_bytes:
                break
            try:
                total -= f.stat().st_size
                f.unlink()
            except Exception:
                pass

    def __iter__(self) -> Iterator[Dict[str, Any]]:
        for item in self.items:
            source = self._ensure_cached(item["cdn_url"])
            try:
                df = pd.read_parquet(
                    source,
                    columns=["prompt", "response"],
                    engine="pyarrow",
                )
            except (pa.lib.ArrowInvalid, KeyError, OSError, ValueError):
                continue

            for _, row in df.iterrows():
                prompt = str(row.get("prompt", "")).strip()
                response = str(row.get("response", "")).strip()
                if prompt and response:
                    yield {"prompt": prompt, "response": response}
```
