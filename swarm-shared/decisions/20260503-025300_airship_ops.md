```python
#!/usr/bin/env python3
"""
CDN-only dataset loader for Surrogate training.
- Pre-list files once on Mac (HF API) -> file_list.json
- Lightning training uses ONLY CDN URLs (zero API/auth calls).
- Projects heterogeneous repo files to {prompt,response} at parse time.
"""
from __future__ import annotations

import json
import os
import re
from pathlib import Path
from typing import Dict, List, Optional, Tuple

import pyarrow as pa
import pyarrow.parquet as pq
import requests
from tqdm import tqdm

HF_DATASET_REPO = os.getenv("HF_DATASET_REPO", "axentx/surrogate-mirror")
CDN_ROOT = f"https://huggingface.co/datasets/{HF_DATASET_REPO}/resolve/main"

# Patterns for attribution via filename (no source/ts cols in schema)
SLUG_RE = re.compile(r"batches/mirror-merged/[\d-]+/([a-zA-Z0-9_-]+)\.parquet")


def load_file_list(file_list_path: Path) -> List[Dict[str, str]]:
    with open(file_list_path) as f:
        data = json.load(f)
    # Expected: [{"path": "batches/mirror-merged/2026-04-29/abc.parquet", "size": 12345}]
    return data


def cdn_url(rel_path: str) -> str:
    return f"{CDN_ROOT}/{rel_path}"


def safe_parquet_read_bytes(url: str, max_retries: int = 3) -> bytes:
    for attempt in range(1, max_retries + 1):
        try:
            r = requests.get(url, timeout=30)
            r.raise_for_status()
            return r.content
        except Exception as exc:
            if attempt == max_retries:
                raise
            # simple backoff
            import time
            time.sleep(2**attempt)
    raise RuntimeError("unreachable")


def project_to_prompt_response(parquet_bytes: bytes) -> List[Dict[str, str]]:
    """
    Heterogeneous schema tolerance:
    - Try common prompt/response field names.
    - Drop all other fields.
    - Attribution moved to filename slug.
    """
    table = pq.read_table(pa.BufferReader(parquet_bytes))
    df = table.to_pandas()

    # Normalize column names
    col_map = {c: c.lower().strip() for c in df.columns}
    df = df.rename(columns=col_map)

    prompt_col = next((c for c in df.columns if "prompt" in c), None)
    response_col = next((c for c in df.columns if "response" in c or "completion" in c), None)

    if prompt_col is None or response_col is None:
        # If missing, skip file instead of crashing
        return []

    out = []
    for _, row in df.iterrows():
        prompt = str(row[prompt_col]) if not pd_isna(row[prompt_col]) else ""
        response = str(row[response_col]) if not pd_isna(row[response_col]) else ""
        if prompt and response:
            out.append({"prompt": prompt, "response": response})
    return out


def pd_isna(x) -> bool:
    try:
        import pandas as pd
        return pd.isna(x)
    except Exception:
        return x is None or (isinstance(x, float) and x != x)


class CDNDataset:
    def __init__(self, file_list_path: Path):
        self.file_list = load_file_list(file_list_path)

    def __len__(self) -> int:
        return len(self.file_list)

    def iter_rows(self):
        for item in tqdm(self.file_list, desc="CDN loading"):
            url = cdn_url(item["path"])
            try:
                blob = safe_parquet_read_bytes(url)
                rows = project_to_prompt_response(blob)
                slug_match = SLUG_RE.search(item["path"])
                slug = slug_match.group(1) if slug_match else "unknown"
                for r in rows:
                    r["slug"] = slug
                    yield r
            except Exception as exc:
                # Log and skip bad files to avoid crashing training
                print(f"WARN: skip {url} -> {exc}")
                continue
```

### 2. `surrogate/training/ops/lightning_runner.py`
```python
#!/usr/bin/env python3
"""
Lightning Studio reuse + idle-safe runner.
- Reuse running studios to save quota.
- Restart if stopped (idle-timeout safe).
"""
import os
import sys
from typing import Optional

try:
    from lightning_sdk import Client, Machine
    from lightning_sdk.workspace import Teamspace
except ImportError as e:
    print("Missing lightning-ai. Install via: pip install lightning-ai")
    sys.exit(1)


LIGHTNING_TEAMSPACE = os.getenv("LIGHTNING_TEAMSPACE", "default")
STUDIO_NAME = os.getenv("LIGHTNING_STUDIO_NAME", "surrogate-train")


def get_or_create_studio() -> Teamspace:
    client = Client()
    team = client.teamspace(LIGHTNING_TEAMSPACE)

    # Reuse running studio
    for s in team.studios:
        if s.name == STUDIO_NAME:
            if s.status == "Running":
                print(f"Reusing running studio: {STUDIO_NAME}")
                return s
            else:
                print(f"Studio {STUDIO_NAME} exists but stopped ({s.status}). Restarting...")
                s.start(machine=Machine.L40S)
                return s

    # Create if not exists
    print(f"Creating studio {STUDIO_NAME}...")
    return team.create_studio(
        name=STUDIO_NAME,
        machine=Machine.L40S,
        create_ok=True,
    )


def run_training_script(
    studio: Teamspace,
    script_path: str,
    args: Optional[list] = None,
) -> None:
    args = args or []
    # Ensure studio is running before run
    if studio.status != "Running":
        print(f"Studio not running ({studio.status}). Starting...")
        studio.start(machine=Machine.L40S)

    run = studio.run(
        script_path,
        arguments=args,
        name="surrogate-train-run",
        requirements=["pyarrow", "tqdm", "pyyaml"],
    )
    print(f"Started run: {run.id}")
    run.watch()
```
