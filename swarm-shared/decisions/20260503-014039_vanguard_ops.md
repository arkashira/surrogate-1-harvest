## Final Synthesized Answer

### Diagnosis (merged, de-duplicated)
- **Quota exhaustion & 429s**: repeated authenticated `list_repo_tree` calls on every training run exhaust the 1000/5min HF API quota and cause intermittent 429s.
- **No persisted manifest**: no `(repo, dateFolder) → file-list` snapshot exists, forcing re-enumeration each run.
- **Schema hazards**: training scripts risk `load_dataset(streaming=True)` on heterogeneous repos, which can trigger pyarrow `CastError` on mixed schemas.
- **Studio reuse not enforced**: new runs may create fresh Lightning Studios instead of reusing running ones, wasting quota and time.
- **No CDN-only fallback**: training still relies on authenticated API during data loading instead of using `resolve/main/` CDN URLs.

### Proposed Change (merged)
Add a lightweight ops manifest generator + training launcher under `/opt/axentx/vanguard/ops/` and update training entrypoints:

- `ops/constants.py` — repo/dateFolder config, cloud priority list, paths.
- `ops/manifest.py` — one-time `list_repo_tree` per `(repo, dateFolder)`, save to `manifests/{repo}_{dateFolder}.json`, embed in training scripts.
- `ops/train_launcher.py` — reuse running Lightning Studio (or start on preferred cloud/machine), inject manifest path, enforce CDN-only data loading.
- Update training entrypoint to accept `--manifest` and `--cdn-only`, and use a `CDNParquetDataset` that projects required columns to avoid schema mismatches.

Scope: create three new files (~120 lines total) and update existing training entrypoint.

# /opt/axentx/vanguard/ops/manifest.py
import json
import time
from datetime import datetime, timezone
from pathlib import Path
from typing import List

from huggingface_hub import HfApi

from .constants import HF_DATASET_REPO, MANIFESTS_DIR

api = HfApi()

def list_date_folder(repo: str, date_folder: str) -> List[str]:
    """
    Single authenticated call to list files in one date folder.
    Returns relative paths under that folder.
    """
    try:
        tree = api.list_repo_tree(
            repo_id=repo,
            path=date_folder,
            recursive=False,
        )
        files = [item.rfilename for item in tree if not item.rfilename.endswith("/")]
        return files
    except Exception as exc:
        # If 429, wait 360s as per pattern
        if getattr(exc, "status_code", None) == 429:
            time.sleep(360)
            return list_date_folder(repo, date_folder)
        raise

def build_manifest(date_folder: str, out_dir: Path = MANIFESTS_DIR) -> Path:
    """
    Persist manifest for (repo, dateFolder).
    """
    files = list_date_folder(HF_DATASET_REPO, date_folder)
    manifest = {
        "repo": HF_DATASET_REPO,
        "date_folder": date_folder,
        "created_at": datetime.now(timezone.utc).isoformat(),
        "files": files,
    }
    out_dir.mkdir(parents=True, exist_ok=True)
    safe_repo = HF_DATASET_REPO.replace("/", "_")
    out_path = out_dir / f"{safe_repo}_{date_folder}.json"
    out_path.write_text(json.dumps(manifest, indent=2))
    return out_path

if __name__ == "__main__":
    import sys
    if len(sys.argv) != 2:
        print("Usage: python manifest.py <dateFolder>")
        sys.exit(1)
    mf = build_manifest(sys.argv[1])
    print(f"Manifest written: {mf}")

import json
import time
from pathlib import Path

from lightning import LightningStudio, Teamspace

from .constants import CLOUD_PRIORITY, STUDIO_NAME, MANIFESTS_DIR

def find_running_studio(name: str) -> LightningStudio | None:
    for studio in Teamspace.studios:
        if studio.name == name and studio.status == "Running":
            return studio
    return None

def launch_or_reuse_studio(
    script_path: str,
    manifest_path: Path,
    cloud_priority=None,
) -> None:
    cloud_priority = cloud_priority or CLOUD_PRIORITY
    studio = find_running_studio(STUDIO_NAME)

    if studio is None:
        # Start with preferred cloud/size
        for cloud, machine in cloud_priority:
            try:
                studio = LightningStudio(
                    name=STUDIO_NAME,
                    script=script_path,
                    machine=machine,
                    cloud=cloud,
                    create_ok=True,
                )
                break
            except Exception:
                continue
        else:
            raise RuntimeError("Could not allocate studio on any preferred cloud/size.")
    else:
        # Reuse running studio; ensure it's alive before run
        if studio.status != "Running":
            cloud, machine = cloud_priority[0]
            studio.start(machine=machine, cloud=cloud)

    # Inject manifest path and run
    # Assumes training script accepts --manifest and uses CDN-only loading
    studio.run(
        script_args=[f"--manifest={manifest_path}", "--cdn-only"],
        wait=False,
    )
    print(f"Studio '{STUDIO_NAME}' running (reused={studio.status == 'Running'}).")

if __name__ == "__main__":
    import sys
    if len(sys.argv) != 3:
        print("Usage: python train_launcher.py <dateFolder> <train_script.py>")
        sys.exit(1)

    date_folder = sys.argv[1]
    train_script = sys.argv[2]

    # Build manifest once (Mac orchestration)
    from .manifest import build_manifest
    manifest_path = build_manifest(date_folder)

class CDNParquetDataset(IterableDataset):
    def __init__(self, manifest_path, cdn_only=True):
        with open(manifest_path) as f:
            manifest = json.load(f)
        self.repo = manifest["repo"]
        self.files = manifest["files"]
        self.cdn_only = cdn_only

    def _iter_urls(self):
        for rel in self.files:
            # CDN bypass: no Authorization header
            yield f"https://huggingface.co/datasets/{self.repo}/resolve/main/{rel}"

    def __iter__(self):
        for url in self._iter_urls():
            yield from self._stream_parquet(url)

    def _stream_parquet(self, url):
        import pyarrow.parquet as pq
        import pyarrow as pa
        r = requests.get(url, stream=True)

        # Project only needed columns to avoid schema issues
        with pa.BufferReader(r.content) as buf:
            table = pq.read_table(buf, columns=["prompt", "response"])
            for row in table.to_pylist():
                yield row
