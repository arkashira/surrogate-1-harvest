# Costinel / ops

**Final Implementation Plan — Top-Hub Signal Panel (Costinel Dashboard)**

**Core value & scope**  
- Frontend-only, read-only panel that surfaces the most-connected hub (e.g., “MOC”) and its actionable proposals so teams can triage cloud-cost governance signals without leaving the dashboard.  
- Ship in ~1–2h. No backend changes, no schema migrations.  
- Resilient by design: network-first, then localStorage cache, then local sample. Graceful degraded UI.

### 3) API helpers (network + cache)

```ts
// src/lib/api/topHubApi.ts
import type { TopHubSignal } from '../types/topHub';

const ENDPOINT = '/api/v1/top-hub-signal';
const CACHE_KEY = 'costinel:top-hub-signal';
const CACHE_TTL_MS = 5 * 60 * 1000; // 5m

export async function fetchTopHubSignal(signal?: AbortSignal): Promise<TopHubSignal> {
  const res = await fetch(ENDPOINT, { signal });
  if (!res.ok) {
    throw new Error(`Top-hub signal fetch failed: ${res.status}`);
  }
  return res.json();
}

export function getCachedTopHubSignal(): TopHubSignal | null {
  try {
    const raw = localStorage.getItem(CACHE_KEY);
    if (!raw) return null;
    const { timestamp, payload } = JSON.parse(raw);
    if (Date.now() - timestamp > CACHE_TTL_MS) return null;
    return payload as TopHubSignal;
  } catch {
    return null;
  }
}

export function setCachedTopHubSignal(payload: TopHubSignal) {
  try {
    localStorage.setItem(
      CACHE_KEY,
      JSON.stringify({ timestamp: Date.now(), payload })
    );
  } catch {
    // ignore storage quota/incognito errors
  }
}
```

### 4) Hook (network-first, cache fallback, sample fallback)

```ts
// src/lib/hooks/useTopHubSignal.ts
import { useEffect, useState, useCallback } from 'react';
import { fetchTopHubSignal, getCachedTopHubSignal, setCachedTopHubSignal } from '../api/topHubApi';
import type { TopHubSignal } from '../types/topHub';
import { sampleTopHubSignal } from '../sample/topHubSample';

export function useTopHubSignal() {
  const [data, setData] = useState<TopHubSignal | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  const load = useCallback(async (signal?: AbortSignal) => {
    setLoading(true);
    setError(null);

    try {
      const remote = await fetchTopHubSignal(signal);
      setCachedTopHubSignal(remote);
      setData(remote);
    } catch (err: any) {
      const cached = getCachedTopHubSignal();
      if (cached) {
        setData({ ...cached, fromCache: true });
      } else {
        setData({ ...sampleTopHubSignal, fromCache: true });
      }
      setError(err?.message || 'Failed to fetch top-hub signal');
    } finally {
      setLoading(false);
    }
  }, []);

  useEffect(() => {
    const controller = new AbortController();
    load(controller.signal);
    return () => controller.abort();
  }, [load]);

  const refresh = () => load();

  return { data, loading, error, refresh };
}
```

// src/components/TopHubSignalPanel.tsx
import React from 'react';
import { useTopHubSignal } from '../lib/hooks/useTopHubSignal';
import type { TopHubSignal } from '../types/topHub';
import './TopHubSignalPanel.scss';

export const TopHubSignalPanel: React.FC = () => {
  const { data, loading, error, refresh } = useTopHubSignal();

  if (loading) {
    return (
      <div className="top-hub-signal-panel card">
        <div className="panel-header">
          <h3>Top-hub signal</h3>
          <span className="loading-spinner" />
        </div>
        <div className="panel-body skeleton" />
      </div>
    );
  }

  if (error || !data) {
    // Graceful fallback with retry
    return (
      <div className="top-hub-signal-panel card">
        <div className="panel-header">
          <h3>Top-hub signal</h3>
          <button className="btn-ghost btn-sm" onClick={refresh} title="Retry" aria-label="Retry">
            ↻
          </button>
        </div>
        <div className="panel-body">
          <p className="muted">Unable to load top-hub insights. Using local sample.</p>
        </div>
      </div>
    );

    <div className="top-hub-signal-panel card">
      <div className="panel-header">
        <h3>Top-hub signal</h3>

          ↻
        </button>
        {data.fromCache && <span className="cache-badge">cached</span>}
      </div>

      <div className="panel-body">
        <HubHeader hub={data.hub} />
        <ProposalsList proposals={data.proposals} />
      </div>
    </div>
  );
};
