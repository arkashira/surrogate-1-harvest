## Implementation Plan (≤2h)

**Highest-value incremental improvement**: Ship a manifest-driven, CDN-only dataset loader for Surrogate training that eliminates HF API rate limits and mixed-schema errors.

**Scope**: Add `scripts/build_manifest.py` + modify `surrogate/data/dataset.py` to use a local `manifest.json` (date-folder → file list) and download via CDN URLs only. Lightning training will use zero HF API calls during data loading.

**Why this now**: The README confirms Surrogate AI is a standalone service (port 8001) with auto-training; the past patterns show HF API 429 and pyarrow CastError are current blockers. This fix removes both.

### 1. Add manifest builder (run once on Mac after rate-limit window clears)

`/opt/axentx/airship/scripts/build_manifest.py`

```python
#!/usr/bin/env python3
"""
Build a manifest for a HuggingFace dataset repo folder (non-recursive).
Usage:
    python scripts/build_manifest.py \
        --repo datasets/your-org/your-repo \
        --folder 2026-05-03 \
        --out surrogate/data/manifest_2026-05-03.json
"""
import argparse
import json
import os
import sys
from datetime import datetime

try:
    from huggingface_hub import list_repo_tree
except ImportError:
    print("Install: pip install huggingface_hub")
    sys.exit(1)

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def build_manifest(repo: str, folder: str, out_path: str):
    # folder should be like "2026-05-03" (non-recursive)
    print(f"Listing {repo}/{folder} (non-recursive)...")
    items = list_repo_tree(repo=repo, path=folder, recursive=False)

    files = []
    for item in items:
        if item.type != "file":
            continue
        # Only include parquet (or extend as needed)
        if not item.path.lower().endswith(".parquet"):
            continue
        files.append({
            "repo": repo,
            "path": item.path,
            "cdn_url": CDN_TEMPLATE.format(repo=repo, path=item.path),
            "size": getattr(item, "size", None),
        })

    manifest = {
        "repo": repo,
        "folder": folder,
        "generated_at": datetime.utcnow().isoformat() + "Z",
        "count": len(files),
        "files": files,
    }

    os.makedirs(os.path.dirname(os.path.abspath(out_path)), exist_ok=True)
    with open(out_path, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2)

    print(f"Wrote {len(files)} files -> {out_path}")

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Build CDN manifest for HF dataset folder.")
    parser.add_argument("--repo", required=True, help="e.g. datasets/your-org/your-repo")
    parser.add_argument("--folder", required=True, help="Folder in repo (non-recursive), e.g. 2026-05-03")
    parser.add_argument("--out", required=True, help="Output JSON path")
    args = parser.parse_args()
    build_manifest(args.repo, args.folder, args.out)
```

### 2. Add CDN-only iterable dataset loader

`/opt/axentx/airship/surrogate/data/dataset.py`

```python
"""
CDN-only dataset loader for Surrogate training.
Uses a local manifest to avoid HF API calls during training.
"""
import json
import logging
from pathlib import Path
from typing import Dict, Iterator, Optional

import pyarrow as pa
import pyarrow.parquet as pq
import requests
from tqdm import tqdm

logger = logging.getLogger(__name__)

class CDNParquetIterable:
    """
    Iterate rows from parquet files listed in a manifest using CDN URLs.
    Projects to {prompt, response} at parse time to avoid mixed-schema errors.
    """

    def __init__(self, manifest_path: str, columns=("prompt", "response")):
        self.manifest_path = Path(manifest_path)
        self.columns = columns
        self._manifest = self._load_manifest()

    def _load_manifest(self) -> Dict:
        with open(self.manifest_path, "r", encoding="utf-8") as f:
            return json.load(f)

    def _download_parquet(self, cdn_url: str, local_path: Path) -> Path:
        if local_path.exists():
            return local_path
        local_path.parent.mkdir(parents=True, exist_ok=True)
        logger.info("Downloading %s -> %s", cdn_url, local_path)
        resp = requests.get(cdn_url, timeout=60)
        resp.raise_for_status()
        local_path.write_bytes(resp.content)
        return local_path

    def _iter_file(self, file_info: Dict) -> Iterator[Dict]:
        cdn_url = file_info["cdn_url"]
        # Cache under surrogate/data/.cache/{slug}.parquet
        slug = Path(file_info["path"]).stem
        cache_dir = Path(__file__).parent / ".cache"
        local_path = cache_dir / f"{slug}.parquet"

        try:
            local_path = self._download_parquet(cdn_url, local_path)
        except Exception as exc:
            logger.warning("Failed to download %s: %s", cdn_url, exc)
            return

        try:
            table = pq.read_table(local_path, columns=self.columns)
        except (pa.ArrowInvalid, KeyError) as exc:
            logger.warning("Schema/projection error in %s: %s", local_path, exc)
            return

        df = table.to_pandas()
        for _, row in df.iterrows():
            yield {k: row.get(k) for k in self.columns}

    def __iter__(self) -> Iterator[Dict]:
        files = self._manifest.get("files", [])
        logger.info("Iterating %d files from manifest %s", len(files), self.manifest_path.name)
        for file_info in tqdm(files, desc="CDN files"):
            yield from self._iter_file(file_info)
```

### 4. Usage example for Lightning training script

`/opt/axentx/airship/surrogate/train.py` (snippet)

```python
from torch.utils.data import IterableDataset
from surrogate.data import CDNParquetIterable

class SurrogateIterableDataset(IterableDataset):
    def __init__(self, manifest_path):
        self.manifest_path = manifest_path

    def __iter__(self):
        for item in CDNParquetIterable(self.manifest_path):
            # Map to your model input format
            prompt = item.get("prompt")
            response = item.get("response")
            if prompt is None or response is None:
                continue
            yield {"input_text": prompt, "target_text": response}

# In your Lightning DataModule or Trainer:
# dataset = SurrogateIterableDataset("surrogate/data/manifest_2026-05-03.json")
```

   ```bash
   python scripts/build_manifest.py \
       --repo datasets/your-org/surrogate-mirror \
       --folder 2026-05-03 \
       --out surrogate/data/manifest_2026-05-03.json
   ```
2. Commit `manifest_2026-05-03.json` (or keep in data/ and reference by path).
3. In Lightning Studio, run training with `SurrogateIterableDataset`. No `load_dataset`, no `list_repo_files` during training → zero HF API calls, no pyarrow CastError from mixed schemas.
