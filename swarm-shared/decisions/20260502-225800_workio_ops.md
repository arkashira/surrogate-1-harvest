# workio / ops

## 1) Diagnosis

- **No server-side idempotency key enforcement**: clock-in/out endpoints accept duplicate requests for the same `(employee, punch_type, date, time_window)` → double punches and payroll drift.
- **No transactional upsert on punch writes**: concurrent retries or retries after network blips can insert two rows instead of one; no unique constraint to prevent it.
- **Missing operational observability**: no request-id propagation, no structured logs for punch events, and no metrics (latency/errors/duplicates) → hard to detect regressions or abuse.
- **Webhook replay risk from LINE**: LINE may redeliver webhook events; backend does not deduplicate by `LINE-delivery-id` or signature window → duplicate punches on retries.
- **No durable client queue for offline/weak connectivity**: frontend fire-and-forget `fetch` loses punches when offline or on flaky mobile networks.

## 2) Proposed change

Add server-side idempotency for punch writes and a lightweight operational log/metrics surface. Scope:

- `workio/server/src/routes/attendance.ts` — POST `/punch` handler
- `workio/server/src/db/schema.sql` — add unique constraint + idempotency table
- `workio/server/src/lib/logger.ts` — structured logger (if not present)
- `workio/server/src/lib/metrics.ts` — simple counters (Node.js `process`-based, exportable for Prometheus)

```sql
-- workio/server/src/db/schema.sql
-- Add idempotency table
CREATE TABLE IF NOT EXISTS punch_idempotency (
  idempotency_key VARCHAR(255) PRIMARY KEY,
  employee_id     INTEGER NOT NULL,
  punch_type      VARCHAR(20) NOT NULL,
  punch_time      TIMESTAMPTZ NOT NULL,
  request_id      VARCHAR(255),
  created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Prevent duplicate punches per employee+type+date+15min window
-- (adjust window to business rules; here 15min)
CREATE UNIQUE INDEX IF NOT EXISTS uq_employee_punch_window
ON punches (employee_id, punch_type, date_trunc('hour', punch_time), floor(EXTRACT(minute FROM punch_time) / 15));
```

```ts
// workio/server/src/lib/metrics.ts
const counters = new Map<string, number>();

export function inc(key: string, n = 1) {
  counters.set(key, (counters.get(key) || 0) + n);
}

export function snapshot() {
  return Object.fromEntries(counters.entries());
}

// Prometheus text exposition (simple)
export function prometheusText() {
  const lines: string[] = [];
  for (const [k, v] of counters.entries()) {
    lines.push(`workio_${k.replace(/[^a-z0-9_]/gi, '_')} ${v}`);
  }
  return lines.join('\n') + '\n';
}
```

### 3.3 Idempotent punch handler

```ts
// workio/server/src/routes/attendance.ts
import { Router } from 'express';
import { db } from '../db/index.js';
import logger from '../lib/logger.js';
import { inc, prometheusText } from '../lib/metrics.js';
import type { Request, Response, NextFunction } from 'express';

const router = Router();

// Helper: derive stable idempotency key from body or header
function deriveIdempotencyKey(req: Request): string {
  // Prefer client-supplied key; fallback to deterministic hash of payload
  if (req.headers['idempotency-key'] && typeof req.headers['idempotency-key'] === 'string') {
    return req.headers['idempotency-key'];
  }
  const payload = JSON.stringify(req.body);
  // lightweight deterministic hash (not crypto)
  let hash = 0;
  for (let i = 0; i < payload.length; i++) {
    const c = payload.charCodeAt(i);
    hash = ((hash << 5) - hash) + c;
    hash |= 0;
  }
  return `auto-${hash}`;
}

router.post('/punch', async (req: Request, res: Response, next: NextFunction) => {
  const { employeeId, punchType, punchTime } = req.body;
  const requestId = req.headers['x-request-id'] as string | undefined;
  const idempotencyKey = deriveIdempotencyKey(req);
  const log = logger.child({ requestId, idempotencyKey, employeeId, punchType });

  if (!employeeId || !punchType || !punchTime) {
    inc('punch_validation_errors');
    log.warn({ body: req.body }, 'missing required fields');
    return res.status(400).json({ error: 'Missing required fields' });
  }

  try {
    // Idempotency check: return previous result if key exists
    const existing = await db`
      SELECT punch_id, employee_id, punch_type, punch_time
      FROM punch_idempotency
      WHERE idempotency_key = ${idempotencyKey}
    `.then((r) => r[0]);

    if (existing) {
      inc('punch_idempotent_hits');
      log.info({ punchId: existing.punch_id }, 'idempotent duplicate, returning previous');
      return res.json({ punchId: existing.punch_id, duplicate: true });
    }

    // Transactional upsert: insert punch + idempotency record atomically
    const result = await db.begin(async (trx) => {
      // Try insert; unique constraint will block duplicates if race occurs
      const [punch] = await trx`
        INSERT INTO punches (employee_id, punch_type, punch_time)
        VALUES (${employeeId}, ${punchType}, ${punchTime})
        ON CONFLICT DO NOTHING
        RETURNING punch_id, employee_id, punch_type, punch_time
      `.then((r) => r);

      // If punch already existed (race), fetch it
      const finalPunch = punch || await trx`
        SELECT punch_id, employee_id, punch_type, punch_time
        FROM punches
        WHERE employee_id = ${employeeId}
          AND punch_type = ${punchType}
          AND punch_time = ${punchTime}
        ORDER BY punch_id DESC
        LIMIT 1
      `.then((r) => r[0]);

      // Record idempotency mapping
      await trx`
        INSERT INTO punch_idempotency (idempotency_key, employee_id, punch_type, punch_time, request_id)
        VALUES (${idempotencyKey}, ${employeeId}, ${punchType}, ${punchTime}, ${requestId})
      `.onConflict((oc) => oc.doNothing());

      return { punch: finalPunch, inserted: !!punch };
    });

    inc(result.inserted ? 'punch_inserts' : 'punch_races_resolved');
    log.info({ punchId: result.punch.punch_id }, 'punch recorded');
    return res.json({ punchId: result.punch.punch_id, duplicate: false });
  } catch (err) {
    inc('punch_errors');
    log.error({ err }, 'punch failed');
    return next(err);
  }
});

// Optional: expose metrics endpoint
router.get('/metrics', (_req, res) => {
  res.set('Content-Type', 'text/plain; version=0.0.4');
  res.send(prometheusText());
});

export default router;
```
