# workio / ops

## Final Synthesized Implementation

### 1. Diagnosis (merged)
- **Missing signature verification** (`X-Line-Signature`) — allows spoofed/replayed events.
- **No idempotency/replay protection** — LINE retries create duplicate `attendance_punches`, `leave_requests`, `ot_requests`.
- **Race conditions** — non-atomic upserts and concurrent retries corrupt daily punch state.
- **No observability** — missing structured logs, request IDs, and metrics for signature/idempotency decisions.
- **Startup validation gap** — `LINE_CHANNEL_SECRET` not validated at boot; misconfiguration surfaces late.
- **Replay window absent** — stale events can manipulate timesheets.

### 2. Proposed Change (merged)
File: `workio/server/src/routes/webhook/line.ts` (create if missing)  
Scope:
- Verify `X-Line-Signature` with HMAC-SHA256 using `LINE_CHANNEL_SECRET`.
- Enforce idempotency via unique `line_event_id` on `attendance_punches`, `leave_requests`, `ot_requests`.
- Atomic upsert for punch state to prevent race conditions.
- Replay-window check (±5 min) on event timestamp.
- Structured logging with request IDs and timing.
- Validate `LINE_CHANNEL_SECRET` at startup; fail fast if missing.
- Return 200 for duplicates/invalid signatures (fail closed, log).

```ts
// server/src/routes/webhook/line.ts
import { Router, Request, Response, NextFunction } from 'express';
import crypto from 'crypto';
import { z } from 'zod';
import { db } from '../db';
import { logger } from '../../utils/logger';
import { v4 as uuidv4 } from 'uuid';

const router = Router();
const LINE_CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET;
const REPLAY_WINDOW_MS = 5 * 60 * 1000; // 5 minutes

if (!LINE_CHANNEL_SECRET) {
  logger.error('LINE_CHANNEL_SECRET is required. Exiting.');
  process.exit(1);
}

function verifySignature(rawBody: Buffer, signature: string): boolean {
  const expected = crypto
    .createHmac('sha256', LINE_CHANNEL_SECRET)
    .update(rawBody)
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

const LineEventSchema = z.object({
  destination: z.string(),
  events: z.array(
    z.object({
      type: z.string(),
      timestamp: z.number(),
      source: z.object({
        type: z.string(),
        userId: z.string().optional(),
        groupId: z.string().optional(),
        roomId: z.string().optional(),
      }),
      message: z.object({
        type: z.string(),
        text: z.string().optional(),
        id: z.string(),
      }).optional(),
      postback: z.object({
        data: z.string().optional(),
        params: z.record(z.any()).optional(),
      }).optional(),
    })
  ),
});

type LineEvent = z.infer<typeof LineEventSchema>;

router.post(
  '/line',
  (req: Request & { rawBody?: Buffer }, res: Response, next: NextFunction) => {
    if (req.rawBody) return next();
    const chunks: Buffer[] = [];
    req.on('data', (chunk) => chunks.push(chunk));
    req.on('end', () => {
      req.rawBody = Buffer.concat(chunks);
      next();
    });
  },
  async (req: Request & { rawBody?: Buffer }, res: Response) => {
    const requestId = uuidv4();
    const start = Date.now();
    const signature = req.headers['x-line-signature'] as string;
    const raw = req.rawBody!;

    // 1) Signature verification
    if (!signature || !verifySignature(raw, signature)) {
      logger.warn({ requestId, signature }, 'Invalid LINE signature');
      return res.status(401).json({ error: 'Invalid signature' });
    }

    // 2) Parse & validate schema
    let parsed: LineEvent;
    try {
      parsed = LineEventSchema.parse(JSON.parse(raw.toString()));
    } catch (err) {
      logger.warn({ requestId, err }, 'Invalid LINE payload schema');
      return res.status(400).json({ error: 'Invalid payload' });
    }

    // 3) Process events idempotently and atomically
    (async () => {
      const client = await db.connect();
      try {
        await client.query('BEGIN');
        for (const ev of parsed.events) {
          const now = Date.now();
          if (Math.abs(now - ev.timestamp) > REPLAY_WINDOW_MS) {
            logger.warn({ requestId, event: ev, now }, 'Replayed/out-of-window LINE event');
            continue;
          }

          const lineEventId = ev.message?.id || ev.postback?.data || `${ev.type}:${ev.timestamp}`;
          const userId = ev.source.userId;
          const type = ev.type;
          const text = ev.message?.text || ev.postback?.data || '';

          if (!userId) {
            logger.debug({ requestId, ev }, 'LINE event without userId; skipping');
            continue;
          }

          // Determine target table and punch_type based on text or type
          let table: 'attendance_punches' | 'leave_requests' | 'ot_requests' = 'attendance_punches';
          let actionType = 'clock_in_out';
          if (text.toLowerCase().includes('leave')) {
            table = 'leave_requests';
            actionType = 'leave';
          } else if (text.toLowerCase().includes('ot') || text.toLowerCase().includes('overtime')) {
            table = 'ot_requests';
            actionType = 'overtime';
          }

          // Atomic upsert: if line_event_id exists, skip
          await client.query(
            `INSERT INTO ${table} (user_id, line_event_id, punch_type, raw_payload, created_at)
             VALUES ($1, $2, $3, $4, to_timestamp($5 / 1000.0))
             ON CONFLICT (line_event_id) DO NOTHING`,
            [userId, lineEventId, actionType, raw.toString(), ev.timestamp]
          );
          logger.info({ requestId, lineEventId, userId, type, table }, 'Processed LINE event (idempotent)');
        }
        await client.query('COMMIT');
      } catch (err) {
        await client.query('ROLLBACK');
        logger.error({ requestId, err }, 'Failed to persist LINE events');
      } finally {
        client.release();
      }
    })().catch((err) => logger.error({ requestId, err }, 'Unhandled LINE webhook processing error'));

    logger.info({ requestId, durationMs: Date.now() - start }, 'LINE webhook processed');
    res.status(200).json({ ok: true });
  }
);

export default router;
```
