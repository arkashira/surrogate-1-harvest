# airship / backend

## Implementation Plan — `airship discover` (backend)

**Scope**: ≤2h  
**Goal**: Deterministic CLI command `airship discover` + HTTP endpoint `/api/discover` that emits a CDN-only file manifest for a given `repo_id` and `date_folder` (supports surrogate/kaggle training pipelines and zero-API training).

### Why this is highest-value
- Unblocks surrogate-1 training by providing deterministic CDN file lists (bypasses HF API rate limits).
- Reuses existing `/api/discover` backend surface (implied by frontend decision).
- Small, focused, and shippable in <2h.

---

### Concrete steps (backend)

1. Add CLI entrypoint `airship discover` (single file, no heavy deps).
2. Implement `/api/discover` GET handler (FastAPI) returning deterministic manifest.
3. Use `list_repo_tree(path, recursive=False)` per date folder (avoid recursive on big repos).
4. Emit manifest: `{ repo_id, date_folder, generated_at, files: [{path, cdn_url, size, sha}], cdn_only: true }`.
5. Include retry/backoff for 429 (wait 360s) and deterministic pagination.
6. Add tests for manifest shape and CDN URL construction.

### Code snippets

#### 1) CLI: `airship/cli/discover.py`

```python
#!/usr/bin/env python3
"""
airship discover
Generate deterministic CDN-only file manifest for a repo + date folder.
Usage:
  airship discover --repo_id datasets/myrepo --date_folder 2026-05-01 --out manifest.json
"""

import argparse
import json
import sys
import time
from datetime import datetime, timezone
from pathlib import Path

import requests

HF_API_BASE = "https://huggingface.co/api"


def hf_get(path: str, retries: int = 3, backoff: int = 5):
    url = f"{HF_API_BASE}{path}"
    for attempt in range(1, retries + 1):
        resp = requests.get(url, timeout=30)
        if resp.status_code == 429:
            wait = 360
            print(f"[WARN] HF 429 rate limit — waiting {wait}s", file=sys.stderr)
            time.sleep(wait)
            continue
        resp.raise_for_status()
        return resp.json()
        # small backoff on transient errors
        time.sleep(backoff * attempt)
    raise RuntimeError(f"Failed to fetch {url} after {retries} retries")


def list_date_folder(repo_id: str, date_folder: str):
    # non-recursive per folder to avoid heavy pagination
    tree = hf_get(f"/datasets/{repo_id}/tree/{date_folder}?recursive=false")
    return tree or []


def build_manifest(repo_id: str, date_folder: str):
    files = list_date_folder(repo_id, date_folder)
    items = []
    for f in files:
        if f.get("type") != "file":
            continue
        path = f["path"]
        items.append(
            {
                "path": path,
                "cdn_url": f"https://huggingface.co/datasets/{repo_id}/resolve/main/{path}",
                "size": f.get("size"),
                "sha": f.get("oid"),
            }
        )

    return {
        "repo_id": repo_id,
        "date_folder": date_folder,
        "generated_at": datetime.now(timezone.utc).isoformat(),
        "files": items,
        "cdn_only": True,
        "count": len(items),
    }


def main():
    parser = argparse.ArgumentParser(description="Generate CDN-only file manifest")
    parser.add_argument("--repo_id", required=True, help="HF dataset repo id (e.g. datasets/myrepo)")
    parser.add_argument("--date_folder", required=True, help="Date folder path in repo (e.g. 2026-05-01)")
    parser.add_argument("--out", default="-", help="Output file (default stdout)")
    args = parser.parse_args()

    manifest = build_manifest(args.repo_id, args.date_folder)
    out = sys.stdout if args.out == "-" else open(args.out, "w", encoding="utf-8")
    json.dump(manifest, out, indent=2, ensure_ascii=False)
    if out is not sys.stdout:
        out.close()


if __name__ == "__main__":
    main()
```

#### 2) API: `airship/api/discover.py`

```python
from fastapi import APIRouter, HTTPException, Query
from datetime import datetime, timezone
import requests

router = APIRouter()

HF_API_BASE = "https://huggingface.co/api"

def hf_get(path: str):
    url = f"{HF_API_BASE}{path}"
    resp = requests.get(url, timeout=30)
    if resp.status_code == 429:
        raise HTTPException(status_code=429, detail="HF API rate limit (429)")
    resp.raise_for_status()
    return resp.json()

@router.get("/discover")
def discover(
    repo_id: str = Query(..., description="HF dataset repo id"),
    date_folder: str = Query(..., description="Date folder in repo")
):
    """
    Generate deterministic CDN-only file manifest for training pipelines.
    Uses non-recursive tree listing per folder to avoid heavy pagination.
    """
    try:
        tree = hf_get(f"/datasets/{repo_id}/tree/{date_folder}?recursive=false")
    except requests.HTTPError as exc:
        raise HTTPException(status_code=exc.response.status_code, detail=str(exc)) from exc

    files = []
    for f in (tree or []):
        if f.get("type") != "file":
            continue
        path = f["path"]
        files.append(
            {
                "path": path,
                "cdn_url": f"https://huggingface.co/datasets/{repo_id}/resolve/main/{path}",
                "size": f.get("size"),
                "sha": f.get("oid"),
            }
        )

    return {
        "repo_id": repo_id,
        "date_folder": date_folder,
        "generated_at": datetime.now(timezone.utc).isoformat(),
        "files": files,
        "cdn_only": True,
        "count": len(files),
    }
```
