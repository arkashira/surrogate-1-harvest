HF_CDN = "https://huggingface.co/datasets"

def shard_assign(file_path: str, shard_id: int, shard_total: int) -> bool:
    """Deterministic shard assignment by file path hash."""
    digest = int(hashlib.md5(file_path.encode()).hexdigest(), 16)
    return (digest % shard_total) == shard_id

def cdn_download_url(repo: str, file_path: str) -> str:
    """CDN URL that bypasses HF API auth/rate limits."""
    return f"{HF_CDN}/{repo}/resolve/main/{file_path}"

def project_to_pair(raw: Dict[str, Any]) -> Dict[str, Any]:
    """
    Project heterogeneous file to {prompt, response} only.
    Handles common schema variants seen in surrogate-1.
    """
    # Normalize keys
    text = raw.get("text") or raw.get("content") or raw.get("prompt") or ""
    completion = raw.get("completion") or raw.get("response") or raw.get("answer") or ""

    # If nested 'messages' or 'conversation', flatten last user/assistant
    if not text and "messages" in raw and isinstance(raw["messages"], list):
        msgs = raw["messages"]
        user_msgs = [m for m in msgs if m.get("role") == "user"]
        assistant_msgs = [m for m in msgs if m.get("role") == "assistant"]
        text = user_msgs[-1].get("content") if user_msgs else ""
        completion = assistant_msgs[-1].get("content") if assistant_msgs else ""

    # Ensure strings
    text = str(text) if text is not None else ""
    completion = str(completion) if completion is not None else ""

    return {"prompt": text.strip(), "response": completion.strip()}

def process_parquet(url: str, dedup: DedupStore) -> List[Dict[str, str]]:
    """Download parquet via CDN, project rows, dedup by md5."""
    pairs = []
    try:
        resp = requests.get(url, timeout=60)
        resp.raise_for_status()
    except Exception as exc:
        log.warning("Failed to download %s: %s", url, exc)
        return pairs

    try:
        table = pq.read_table(pa.BufferReader(resp.content))
    except Exception as exc:
        log.warning("Failed to read parquet %s: %s", url, exc)
        return pairs

    for batch in table.to_batches():
        cols = batch.column_names
        # Accept common column names; project row-by-row
        for i in range(batch.num_rows):
            raw = {}
            for c in cols:
                try:
                    raw[c] = batch.column(c)[i].as_py()
                except Exception:
                    raw[c] = None

            pair = project_to_pair(raw)
            if not pair["prompt"] and not pair["response"]:
                continue

            # Deterministic md5 for dedup
            payload = f"{pair['prompt']}\n{pair['response']}".strip()
            md5 = hashlib.md5(payload.encode()).hexdigest()
            if dedup.seen(md5):
                continue
            dedup.add(md5)
            pairs.append(pair)

    return pairs

def process_jsonlines(url: str, dedup: DedupStore) -> List[Dict[str, str]]:
    """Process line-delimited JSON via CDN."""
    pairs = []
    try:
        resp = requests.get(url, timeout=60, stream=True)
        resp.raise_for_status()
    except Exception as exc:
        log.warning("Failed to download %s: %s", url, exc)
        return pairs

    for line in resp.iter_lines(decode_unicode=True):
        if not line:
            continue
        try:
            raw = json.loads(line)
        except Exception:
            continue
        pair = project_to_pair(raw)
        if not pair["prompt"] and not pair["response"]:
            continue
        payload = f"{pair['prompt']}\n{pair['response']}".strip()
        md5 = hashlib.md5(payload.encode()).hexdigest()
        if dedup.seen(md5):
            continue
        dedup.add(md5)
        pairs.append(pair)
    return pairs
