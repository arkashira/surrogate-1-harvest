# Costinel / ops

## Final Implementation Plan — Top-hub Signal Panel (Costinel Dashboard)

**Scope**  
- Frontend-only, read-only panel mounted on the Costinel dashboard.  
- Surfaces the highest-connected hub (e.g., “MOC”) and its actionable proposals.  
- Resilient to missing backend: graceful fallback UI + local mock when graph data unavailable.  
- No schema changes, no new backend endpoints, no DB migrations.

**Timebox**  
<2h (implementation + smoke test)

**Stack**  
- React (existing app shell)  
- Tailwind (existing styles)  
- Optional: lightweight state/localStorage cache for last-known hub

---

### 1) Add panel component (frontend)

Create: `/opt/axentx/Costinel/src/components/TopHubSignalPanel.jsx`

```jsx
import { useEffect, useState } from "react";

const MOCK_HUB = {
  id: "MOC",
  name: "Mission Operations Center",
  rank: 1,
  connections: 42,
  proposals: [
    {
      id: "P-2026-05-03-01",
      title: "Shift non-prod workloads to Savings Plans",
      impact: "$12.4k/mo savings",
      risk: "low",
      action: "Review & approve",
    },
    {
      id: "P-2026-05-03-02",
      title: "Right-size over-provisioned EKS node groups",
      impact: "$8.1k/mo savings",
      risk: "medium",
      action: "Schedule review",
    },
  ],
};

async function fetchTopHub() {
  try {
    // Primary: try backend graph endpoint when available
    const res = await fetch("/api/hubs/top", { cache: "no-store" });
    if (!res.ok) throw new Error("Not found");
    return res.json();
  } catch {
    // Graceful fallback to local mock
    return null;
  }
}

export default function TopHubSignalPanel() {
  const [data, setData] = useState(null);
  const [isDemo, setIsDemo] = useState(false);

  useEffect(() => {
    fetchTopHub().then((res) => {
      if (res) {
        setData(res);
        setIsDemo(false);
      } else {
        setData(MOCK_HUB);
        setIsDemo(true);
      }
    });
  }, []);

  if (!data) {
    return (
      <div className="rounded-lg border border-gray-200 bg-white p-5 shadow-sm">
        <p className="text-sm text-gray-500">No active signals.</p>
      </div>
    );
  }

  return (
    <div className="rounded-lg border border-gray-200 bg-white p-5 shadow-sm">
      <div className="mb-4 flex items-center justify-between">
        <div>
          <h3 className="text-base font-semibold text-gray-900">Top hub</h3>
          <p className="text-2xl font-bold text-blue-600">{data.name || data.id}</p>
          <p className="text-xs text-gray-500">
            Rank #{data.rank} · {data.connections} connections
          </p>
        </div>
        {isDemo && (
          <span className="rounded bg-gray-100 px-2 py-1 text-xs font-medium text-gray-600">
            Demo data
          </span>
        )}
      </div>

      <div className="space-y-3">
        {data.proposals.map((p) => (
          <a
            key={p.id}
            href={`/proposals/${p.id}`}
            className="block rounded border border-gray-100 bg-gray-50 p-3 hover:border-gray-200 hover:bg-gray-100"
          >
            <div className="flex items-start justify-between gap-2">
              <div>
                <p className="text-sm font-medium text-gray-900">{p.title}</p>
                <p className="text-xs text-gray-600">{p.impact}</p>
              </div>
              {p.risk && (
                <span
                  className={`whitespace-nowrap rounded-full px-2 py-1 text-xs font-medium ${
                    p.risk === "low"
                      ? "bg-green-100 text-green-700"
                      : p.risk === "medium"
                      ? "bg-yellow-100 text-yellow-700"
                      : "bg-red-100 text-red-700"
                  }`}
                >
                  {p.risk}
                </span>
              )}
            </div>
          </a>
        ))}

        {data.proposals.length === 0 && (
          <p className="text-sm text-gray-500">No proposals available.</p>
        )}
      </div>
    </div>
  );
}
```
