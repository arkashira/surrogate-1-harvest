## Highest-Value Incremental Improvement (<2h)
**Eliminate HF API 429s during Surrogate training** by switching from `load_dataset(streaming=True)` to a **CDN-only ingestion pipeline** with pre-listed file manifests. This removes auth-check rate limits, avoids HF commit caps, and keeps training quota-free.

## Implementation Plan

### 1. Mac Orchestrator: Pre-list & Cache File Manifest (15 min)
Run once per date folder after rate-limit window clears; save JSON for Lightning training.

```bash
# /opt/axentx/airship/scripts/generate_manifest.py
#!/usr/bin/env python3
"""
Generate CDN-only file manifest for Surrogate training.
Usage: python generate_manifest.py <date_folder> > manifest.json
"""
import json, os, sys
from huggingface_hub import list_repo_tree

REPO = os.getenv("HF_DATASET_REPO", "axentx/surrogate-ingest")
DATE_FOLDER = sys.argv[1] if len(sys.argv) > 1 else "latest"

def main():
    entries = list_repo_tree(
        repo_id=REPO,
        path=DATE_FOLDER,
        recursive=False
    )
    files = [
        {
            "cdn_url": f"https://huggingface.co/datasets/{REPO}/resolve/main/{DATE_FOLDER}/{entry.path.split('/')[-1]}",
            "filename": entry.path.split('/')[-1],
            "size": getattr(entry, 'size', 0)
        }
        for entry in entries
        if entry.path.endswith(".parquet")
    ]
    manifest = {
        "repo": REPO,
        "date_folder": DATE_FOLDER,
        "files": files,
        "total_files": len(files)
    }
    json.dump(manifest, sys.stdout, indent=2)

if __name__ == "__main__":
    main()
```

### 2. Lightning Training Script: CDN-Only Data Loader (45 min)
Embed manifest; fetch via CDN with `datasets` + `pyarrow` projection to `{prompt, response}` only.

```python
# /opt/axentx/airship/surrogate/train.py
import json, os
from pathlib import Path
from datasets import load_dataset, DatasetDict
import lightning as L

MANIFEST_PATH = Path(os.getenv("MANIFEST_PATH", "data/manifest_2026-05-03.json"))

def load_cdn_dataset(manifest_path: Path):
    with open(manifest_path) as f:
        manifest = json.load(f)

    cdn_urls = [f["cdn_url"] for f in manifest["files"]]
    if not cdn_urls:
        raise ValueError("No parquet files found in manifest")

    # CDN-only: no HF API auth checks during data load
    ds = load_dataset(
        "parquet",
        data_files={"train": cdn_urls},
        split="train",
        streaming=False  # avoid mixed-schema issues; small enough for memory
    )

    # Project to {prompt, response} only; drop everything else
    def project(batch):
        return {
            "prompt": batch.get("prompt", batch.get("input", "")),
            "response": batch.get("response", batch.get("output", ""))
        }

    ds = ds.map(project, remove_columns=ds.column_names)
    return DatasetDict({"train": ds})

class SurrogateDataModule(L.LightningDataModule):
    def __init__(self, manifest_path: Path = MANIFEST_PATH, batch_size: int = 8):
        super().__init__()
        self.manifest_path = manifest_path
        self.batch_size = batch_size

    def prepare_data(self):
        load_cdn_dataset(self.manifest_path)  # validate early

    def setup(self, stage=None):
        ds = load_cdn_dataset(self.manifest_path)
        self.train = ds["train"].with_format("torch")

    def train_dataloader(self):
        from torch.utils.data import DataLoader
        return DataLoader(self.train, batch_size=self.batch_size, shuffle=True)

# Lightning Studio reuse guard
def get_or_create_studio(name: str = "surrogate-train"):
    from lightning.fabric.utilities.cloud_io import _load as load_teamspace
    # Lightweight check: list running studios via Teamspace (pseudo)
    # In practice: Teamspace.studios list; reuse if running
    L.LightningModule  # ensure import
    studio = L.Studio(
        name=name,
        create_ok=True,
        cloud="lightning-public-prod",
        machine="L40S"
    )
    return studio

if __name__ == "__main__":
    dm = SurrogateDataModule()
    dm.prepare_data()
    dm.setup()
    print(f"Loaded {len(dm.train)} samples from CDN")
```

### 3. Dataset-Mirror Writer: Schema-Hardened Parquet (30 min)
Ensure only `{prompt, response}` written; attribution via filename pattern.

```python
# /opt/axentx/airship/surrogate/mirror_writer.py
import pyarrow as pa
import pyarrow.parquet as pq
from pathlib import Path
import uuid
from datetime import datetime

def write_mirror_batch(prompt_response_pairs, output_root="batches/mirror-merged"):
    date_str = datetime.utcnow().strftime("%Y-%m-%d")
    out_dir = Path(output_root) / date_str
    out_dir.mkdir(parents=True, exist_ok=True)

    slug = f"{uuid.uuid4().hex[:16]}.parquet"
    path = out_dir / slug  # attribution by path, not columns

    table = pa.table({
        "prompt": [p for p, _ in prompt_response_pairs],
        "response": [r for _, r in prompt_response_pairs]
    })
    pq.write_table(table, path)
    return path
```
