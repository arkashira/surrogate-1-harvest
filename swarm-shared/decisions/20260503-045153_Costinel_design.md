# Costinel / design

## Top-Hub Signal Panel — Implementation Plan (≤2h)

### Scope
- Add a frontend-only panel to Costinel dashboard that shows the most-connected hub (e.g., “MOC”) and a short contextual insight.
- CDN-first: fetch JSON via `https://huggingface.co/datasets/{repo}/resolve/main/{path}` (no Authorization header) to bypass HF API rate limits.
- Graceful local fallback + client cache (15 min) so UI never blocks.
- Zero backend changes; deployable with existing static build pipeline.

### File changes
- `src/components/TopHubSignalPanel.tsx` (new)
- `src/lib/topHubClient.ts` (new)
- `src/pages/Dashboard.tsx` (import + mount panel)
- `public/data/top-hub-fallback.json` (fallback payload)

### 2) CDN-first client (no auth)

`src/lib/topHubClient.ts`
```ts
const HUB_DATA_URL =
  'https://huggingface.co/datasets/axentx/costinel-knowledge/resolve/main/top-hub/latest.json';
const FALLBACK_URL = '/data/top-hub-fallback.json';
const CACHE_KEY = 'costinel:top-hub:cached';
const CACHE_TTL_MS = 15 * 60 * 1000; // 15m

export interface TopHubPayload {
  hub: string;
  score: number;
  insight: string;
  updatedAt: string;
  source?: string;
}

async function fetchWithTimeout(url: string, timeoutMs = 6000): Promise<Response> {
  const controller = new AbortController();
  const id = setTimeout(() => controller.abort(), timeoutMs);
  try {
    const res = await fetch(url, { signal: controller.signal, cache: 'no-store' });
    clearTimeout(id);
    return res;
  } catch {
    clearTimeout(id);
    throw new Error('fetch_timeout');
  }
}

async function loadFromCDN(): Promise<TopHubPayload | null> {
  try {
    const res = await fetchWithTimeout(HUB_DATA_URL);
    if (!res.ok) throw new Error('cdn_non_ok');
    return (await res.json()) as TopHubPayload;
  } catch {
    return null;
  }
}

function loadFromCache(): TopHubPayload | null {
  try {
    const raw = localStorage.getItem(CACHE_KEY);
    if (!raw) return null;
    const { ts, payload } = JSON.parse(raw);
    if (Date.now() - ts > CACHE_TTL_MS) return null;
    return payload as TopHubPayload;
  } catch {
    return null;
  }
}

function saveToCache(payload: TopHubPayload) {
  try {
    localStorage.setItem(
      CACHE_KEY,
      JSON.stringify({ ts: Date.now(), payload })
    );
  } catch {
    // ignore storage errors
  }
}

async function loadFallback(): Promise<TopHubPayload> {
  const res = await fetch(FALLBACK_URL);
  if (!res.ok) {
    return {
      hub: 'N/A',
      score: 0,
      insight: 'Top hub insight unavailable.',
      updatedAt: new Date().toISOString(),
      source: 'fallback',
    };
  }
  return (await res.json()) as TopHubPayload;
}

export async function fetchTopHub(): Promise<TopHubPayload> {
  // 1) try cache first (fastest)
  const cached = loadFromCache();
  if (cached) return cached;

  // 2) try CDN (no auth, bypasses HF API rate limit)
  const cdn = await loadFromCDN();
  if (cdn) {
    saveToCache(cdn);
    return cdn;
  }

  // 3) fallback
  const fallback = await loadFallback();
  saveToCache(fallback);
  return fallback;
}
```

### 3) Panel component

`src/components/TopHubSignalPanel.tsx`
```tsx
import { useEffect, useState } from 'react';
import { fetchTopHub, type TopHubPayload } from '../lib/topHubClient';

export function TopHubSignalPanel() {
  const [data, setData] = useState<TopHubPayload | null>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    let mounted = true;
    setLoading(true);
    fetchTopHub()
      .then((v) => {
        if (mounted) setData(v);
      })
      .finally(() => {
        if (mounted) setLoading(false);
      });
    return () => {
      mounted = false;
    };
  }, []);

  if (loading || !data) {
    return (
      <div className="rounded-lg border bg-card p-4">
        <div className="h-5 w-32 animate-pulse rounded bg-muted" />
        <div className="mt-2 h-4 w-48 animate-pulse rounded bg-muted" />
      </div>
    );
  }

  return (
    <div className="rounded-lg border bg-card p-4 shadow-sm">
      <div className="flex items-start justify-between gap-4">
        <div>
          <div className="flex items-center gap-2">
            <span className="font-semibold">Top Hub</span>
            <span className="rounded bg-primary/10 px-2 py-0.5 text-xs font-medium text-primary">
              {data.hub}
            </span>
          </div>
          <p className="mt-1 text-sm text-muted-foreground">{data.insight}</p>
          <p className="mt-2 text-xs text-muted-foreground">
            Score: {(data.score * 100).toFixed(0)}% · Updated{' '}
            {new Date(data.updatedAt).toLocaleDateString()}
          </p>
        </div>
        <div className="shrink-0">
          <div className="flex size-10 items-center justify-center rounded-full bg-primary/10 text-primary">
            <svg
              className="size-5"
              fill="none"
              stroke="currentColor"
              viewBox="0 0 24 24"
            >
              <path
                strokeLinecap="round"
                strokeLinejoin="round"
                strokeWidth={2}
                d="M13 10V3L4 14h7v7l9-11h-7z"
              />
            </svg>
          </div>
        </div>
      </div>
    </div>
  );
}
```
