## Implementation Plan — CDN-First Top-Hub Signal Panel (<2h)

**Goal**: Ship a resilient “Top Hub” signal panel into Costinel that shows the most-connected hub (e.g., “MOC”) and related docs using CDN-fetched artifacts (zero runtime HF API calls).

### 2) `src/lib/cdn.ts` — CDN fetch utilities

```ts
// src/lib/cdn.ts
const CDN_ROOT = 'https://huggingface.co/datasets/axentx/knowledge-rag/resolve/main';

export interface TopHubPayload {
  hub: {
    id: string;
    label: string;
    degree: number;
    description?: string;
  };
  related: Array<{
    id: string;
    label: string;
    type: 'doc' | 'metric' | 'alert';
    snippet?: string;
    url?: string;
  }>;
  generated_at: string;
}

export async function fetchTopHubSignal(dateFolder: string): Promise<TopHubPayload | null> {
  // dateFolder example: "2026-04-27"
  const url = `${CDN_ROOT}/top-hub/${dateFolder}/top-hub.json`;
  try {
    const res = await fetch(url, { cache: 'no-store' });
    if (!res.ok) return null;
    return (await res.json()) as TopHubPayload;
  } catch {
    return null;
  }
}

export function latestAvailableFolder(): string {
  // Lightweight heuristic: use UTC YYYY-MM-DD for "today - 1 day" to allow daily batch to land.
  // Training/ingestion jobs write to dated folders; this avoids 404 during same-day runs.
  const d = new Date();
  d.setUTCDate(d.getUTCDate() - 1);
  return d.toISOString().slice(0, 10); // "YYYY-MM-DD"
}
```

### 3) `src/hooks/useTopHubSignal.ts` — polling hook (Sense + Signal)

```ts
// src/hooks/useTopHubSignal.ts
import { useEffect, useState, useRef } from 'react';
import { fetchTopHubSignal, latestAvailableFolder } from '../lib/cdn';
import { TopHubPayload } from '../lib/cdn';

const POLL_MS = 60_000; // 1m polling for fresh signals (non-mutating)

export function useTopHubSignal() {
  const [payload, setPayload] = useState<TopHubPayload | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  const poll = async (folder?: string) => {
    const f = folder ?? latestAvailableFolder();
    try {
      const data = await fetchTopHubSignal(f);
      if (data) {
        setPayload(data);
        setError(null);
      } else {
        // try previous day as fallback (robustness)
        const d = new Date();
        d.setUTCDate(d.getUTCDate() - 2);
        const prev = d.toISOString().slice(0, 10);
        const prevData = await fetchTopHubSignal(prev);
        setPayload(prevData);
        setError(prevData ? null : 'No top-hub signal available');
      }
    } catch (err) {
      setError(String(err));
      setPayload(null);
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => {
    poll();
    const id = setInterval(() => poll(), POLL_MS);
    return () => clearInterval(id);
  }, []);

  return { payload, loading, error, refetch: () => poll() };
}
```
