# airship / design

## Highest-Value Incremental Improvement (≤2h)
**Resilient Surrogate-1 Training Pipeline**  
Make training immune to HF API 429s and Lightning idle-stop quota loss by switching to deterministic CDN-only data loading + studio lifecycle resilience.

---

## Implementation Plan

### 1. Pre-list file paths once (Mac orchestration)
- Single API call to `list_repo_tree` for one date folder after rate-limit window clears.
- Save list to `file_list.json`.
- Embed in training script.

### 2. CDN-only data loader (Lightning training)
- Replace `load_dataset(streaming=True)` with direct CDN fetches via `datasets` `IterableDataset` that uses `hf_hub_download` or raw CDN URLs.
- Project to `{prompt, response}` only at parse time.
- No Authorization header → bypasses `/api/` rate limits.

### 3. Studio lifecycle resilience
- Before each `.run()`, check studio status.
- If stopped, restart with `target.start(machine=Machine.L40S)`.
- Reuse running studios to save quota.

### 4. Output schema hygiene
- Move attribution to filename pattern: `batches/mirror-merged/{date}/{slug}.parquet`.
- Do not add `source` / `ts` columns.

## Code Snippets

### `scripts/pre_list_files.py` (run on Mac)
```python
#!/usr/bin/env python3
"""
Pre-list HF repo files for a date folder and save to JSON.
Run after HF API rate-limit window clears.
"""
import json
import os
from huggingface_hub import HfApi

REPO_ID = os.getenv("HF_DATASET_REPO", "my-org/surrogate-data")
DATE_FOLDER = os.getenv("DATE_FOLDER", "2026-04-29")
OUT_PATH = os.getenv("OUT_JSON", "file_list.json")

api = HfApi()
tree = api.list_repo_tree(
    repo_id=REPO_ID,
    path=DATE_FOLDER,
    recursive=False,  # one level; adapt if nested
    repo_type="dataset",
)

files = [
    f.rfilename
    for f in tree
    if f.rfilename.endswith((".parquet", ".jsonl"))
]

with open(OUT_PATH, "w") as f:
    json.dump({"repo_id": REPO_ID, "date": DATE_FOLDER, "files": files}, f, indent=2)

print(f"Saved {len(files)} files to {OUT_PATH}")
```

### `surrogate/train.py` (CDN-only loader + studio resilience)
```python
#!/usr/bin/env python3
import json
import os
from pathlib import Path
from typing import Iterator, Dict, Any

import torch
from datasets import IterableDataset
from huggingface_hub import hf_hub_download
from lightning import Fabric, LightningModule
from lightning.pytorch.loggers import CSVLogger

# ---- CDN-only IterableDataset ----
class CDNParquetDataset(IterableDataset):
    """
    Loads parquet files via CDN (hf_hub_download) and yields {prompt, response}.
    Avoids load_dataset(streaming=True) to prevent HF API 429s.
    """
    def __init__(self, file_list_path: str, repo_id: str):
        with open(file_list_path) as f:
            meta = json.load(f)
        self.repo_id = repo_id
        self.files = meta["files"]

    def _parquet_to_examples(self, local_path: str) -> Iterator[Dict[str, Any]]:
        import pyarrow.parquet as pq
        table = pq.read_table(local_path, columns=["prompt", "response"])
        for row in table.to_pylist():
            # Project to {prompt, response} only
            yield {"prompt": row["prompt"], "response": row["response"]}

    def __iter__(self) -> Iterator[Dict[str, Any]]:
        for rfile in self.files:
            local = hf_hub_download(
                repo_id=self.repo_id,
                filename=rfile,
                repo_type="dataset",
                # CDN URL is used internally; no auth for public files
            )
            yield from self._parquet_to_examples(local)

# ---- Minimal Surrogate-1 LightningModule ----
class Surrogate1Module(LightningModule):
    def __init__(self, lr: float = 1e-4):
        super().__init__()
        self.lr = lr
        self.model = torch.nn.Linear(1024, 1024)  # placeholder

    def training_step(self, batch, batch_idx):
        # dummy step; replace with real model/tokenizer
        loss = torch.tensor(0.0, requires_grad=True)
        self.log("train_loss", loss)
        return loss

    def configure_optimizers(self):
        return torch.optim.AdamW(self.parameters(), lr=self.lr)

# ---- Studio lifecycle helpers ----
def ensure_studio_running(studio_name: str, fabric: Fabric) -> None:
    """
    Reuse or restart Lightning Studio to avoid idle-stop quota loss.
    """
    from lightning.pytorch.cloud import Teamspace

    teamspace = Teamspace()
    running = [s for s in teamspace.studios if s.name == studio_name and s.status == "Running"]
    if running:
        print(f"Reusing running studio: {studio_name}")
        return

    print(f"Starting studio: {studio_name}")
    # Start with L40S (free-tier fallback handled by Lightning priority order)
    from lightning.pytorch.cloud import Studio, Machine
    studio = Studio(
        name=studio_name,
        machine=Machine.L40S,
        create_ok=True,
    )
    studio.start()

# ---- Main ----
def main():
    fabric = Fabric(accelerator="cuda", devices=1, precision="bf16-mixed")
    ensure_studio_running("surrogate-1-train", fabric)

    file_list = "file_list.json"
    repo_id = os.getenv("HF_DATASET_REPO", "my-org/surrogate-data")
    dataset = CDNParquetDataset(file_list, repo_id)

    # Use DataLoader with small batch to avoid OOM
    loader = torch.utils.data.DataLoader(dataset, batch_size=4, num_workers=0)

    model = Surrogate1Module()
    trainer = fabric._trainer  # simplified; use full Trainer in prod
    # For full Lightning run:
    # trainer = Trainer(
    #     max_epochs=1,
    #     logger=CSVLogger("logs", name="surrogate-1"),
    #     accelerator="gpu",
    #     devices=1,
    #     strategy="auto",
    # )
    # trainer.fit(model, loader)

    # Minimal fit loop for demo
    optimizer = model.configure_optimizers()
    model.train()
    for batch in loader:
        # dummy tokenized batch
        loss = model.training_step(batch, 0)
        fabric.backward(loss)
        optimizer.step()
        optimizer.zero_grad()
        break

    print("Training step complete (CDN-only, resilient).")

if __name__ == "__main__":
    main()
```
