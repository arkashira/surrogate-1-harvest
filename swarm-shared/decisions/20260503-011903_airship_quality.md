Orchestration-only script (runs on Mac).

from lightning import Lightning

HF_FOLDER = os.getenv("HF_FOLDER", "batches/mirror-merged/2026-05-03")
STUDIO_NAME = os.getenv("STUDIO_NAME", "surrogate-l40s-train")
MACHINE = os.getenv("MACHINE", "lightning-lambda-prod/L40S")

api = HfApi()

def list_parquet_files(repo: str, folder: str):
    tree = list_repo_tree(repo=repo, path=folder, recursive=False)
    return [
        f"{folder.rstrip('/')}/{entry.path.split('/')[-1]}"
        for entry in tree
        if entry.path.lower().endswith(".parquet")
    ]

def build_cdn_urls(file_paths):
    base = f"https://huggingface.co/datasets/{HF_REPO}/resolve/main"
    return {p: f"{base}/{p}" for p in file_paths}

def reuse_or_create_studio(name: str, machine: str):
    teamspace = Lightning().teamspace
    for s in teamspace.studios:
        if s.name == name and s.status == "Running":
            print(f"Reusing running studio: {name}")
            return s
    print(f"Creating new studio: {name} on {machine}")
    return teamspace.create_studio(name=name, machine=machine)

def ensure_running(studio):
    studio.refresh()
    if studio.status != "Running":
        print(f"Studio {studio.name} stopped; restarting on {studio.machine}.")
        studio.start(machine=studio.machine)
        for _ in range(30):
            studio.refresh()
            if studio.status == "Running":
                print("Studio is running.")
                return
            time.sleep(2)
        raise RuntimeError("Studio failed to start.")

def main():
    print("Listing HF dataset folder (single API call)...")
    files = list_parquet_files(HF_REPO, HF_FOLDER)
    if not files:
        raise RuntimeError("No parquet files found.")
    mapping = build_cdn_urls(files)

    out_path = "workspace/file_list.json"
    os.makedirs(os.path.dirname(out_path), exist_ok=True)
    with open(out_path, "w") as f:
        json.dump(mapping, f, indent=2)
    print(f"Saved {len(mapping)} CDN paths to {out_path}")

    studio = reuse_or_create_studio(STUDIO_NAME, MACHINE)
    ensure_running(studio)

    # Upload workspace (or mount) and run training script
    # Example: studio.upload("workspace", "/workspace")
    studio.run(
        target="python /workspace/train.py --file-list /workspace/file_list.json",
        machine=MACHINE,
    )
    print("Training started via Studio.")

if __name__ == "__main__":
    main()

### 2) Training script — CDN-only dataset loader with schema projection and retries
`train.py`
```python
#!/usr/bin/env python3
"""
Lightning Studio training script.
- Uses CDN URLs from file_list.json (zero HF API calls during training).
- Projects only {prompt, response} to avoid pyarrow CastError on mixed schemas.
- Retries CDN fetches and gracefully skips malformed files.
"""
import json
import os
import sys
import time
import requests
import pyarrow.parquet as pq
from io import BytesIO
from torch.utils.data import IterableDataset, DataLoader

class CDNParquetDataset(IterableDataset):
    def __init__(self, file_list_path, max_retries=3, timeout=30):
        with open(file_list_path) as f:
            self.cdn_map = json.load(f)  # rel_path -> cdn_url
        self.paths = list(self.cdn_map.values())
        self.max_retries = max_retries
        self.timeout = timeout

    def fetch_parquet(self, url):
        for attempt in range(self.max_retries):
            try:
                resp = requests.get(url, timeout=self.timeout)
                resp.raise_for_status()
                return pq.read_table(BytesIO(resp.content))
            except Exception as e:
                if attempt == self.max_retries - 1:
                    raise
                time.sleep(2 ** attempt)
        raise RuntimeError(f"Failed to fetch {url}")

    def project_record(self, table):
        # Keep only prompt/response; tolerate missing/alternate column names
        cols = set(table.column_names)
        prompt_col = next((c for c in ("prompt", "input", "question") if c in cols), None)
        response_col = next((c for c in ("response", "output", "answer") if c in cols), None)

        if not prompt_col or not response_col:
            # Skip malformed files instead of crashing
            return []

        prompts = table[prompt_col].to_pylist()
        responses = table[response_col].to_pylist()
        return [
            {"prompt": p, "response": r}
            for p, r in zip(prompts, responses)
            if isinstance(p, str) and isinstance(r, str) and p.strip() and r.strip()
        ]

    def __iter__(self):
        for url in self.paths:
            try:
                table = self.fetch_parquet(url)
                for rec in self.project_record(table):
                    yield rec
            except Exception as e:
                print(f"Skipping {url}: {e}", file=sys.stderr)
                continue

def main():
    import argparse
    parser = argparse.ArgumentParser()
    parser.add_argument("--file-list", required=True, help="Path to file_list.json")
    parser.add_argument("--batch-size", type=int, default=8)
    parser.add_argument("--max-steps", type=int, default=1000)
    args = parser.parse_args()

    dataset = CDNParquetDataset(args
