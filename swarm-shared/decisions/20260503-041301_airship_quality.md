### Final Consolidated Implementation (Best of Both Candidates)

**Goal**: Eliminate HF API 429s during Surrogate training and make Lightning training resilient to idle timeouts—ship in <2h with zero schema/infra changes.

**Core Decisions (correctness + actionability)**
- Use CDN-only dataset fetches (bypass Authorization headers) to eliminate 429s during training.  
- Pre-list file paths once per date folder and embed a manifest; do not call HF API during training.  
- Keep data loading simple and robust: stream per-file via CDN, parse only `{prompt, response}`.  
- Add Lightning idle-resilience by checking studio status and auto-restarting before each run.  
- All changes are additive to existing `train.py`; no breaking changes.

### 1) Pre-list file manifest (Mac orchestration)
Single API call after rate-limit window clears → save JSON. Training uses CDN-only fetches with zero API calls during data load.

```bash
#!/usr/bin/env bash
# tools/build_file_manifest.sh
set -euo pipefail

REPO="${HF_REPO:-axentx/surrogate-data}"
DATE_DIR="${1:-$(date +%Y-%m-%d)}"
OUT_FILE="${OUT_PATH:-data/file_manifest_${DATE_DIR}.json}"

mkdir -p data

python3 - <<PY
import os, json, sys
from huggingface_hub import list_repo_tree

repo = os.environ.get("HF_REPO", "$REPO")
date_dir = os.environ.get("DATE_DIR", "$DATE_DIR")

tree = list_repo_tree(repo, path=date_dir, recursive=False)
files = [f.rfilename for f in tree if f.rfilename.endswith('.parquet')]

manifest = {
    "repo": repo,
    "date_dir": date_dir,
    "files": sorted(files),
    "cdn_base": f"https://huggingface.co/datasets/{repo}/resolve/main/{date_dir}"
}

out_path = os.environ.get("OUT_PATH", "$OUT_FILE")
with open(out_path, "w") as f:
    json.dump(manifest, f, indent=2)
print(f"Wrote {len(files)} files to {out_path}")
PY
```

### 2) CDN-only DataLoader in Lightning training script
Replace `load_dataset(streaming=True)` with direct CDN fetches. Minimal, robust, and bypasses API rate limits.

```python
# surrogate/train.py  (excerpt to integrate)
import json
import pyarrow.parquet as pq
import requests
from torch.utils.data import Dataset, DataLoader
import io
import os

class CDNParquetDataset(Dataset):
    def __init__(self, manifest_path: str, cache_limit: int = 4):
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.cdn_base = self.manifest["cdn_base"]
        self.files = self.manifest["files"]
        if not self.files:
            raise ValueError("No parquet files found in manifest")
        self._cache = {}
        self._cache_order = []
        self.cache_limit = cache_limit

    def _fetch_parquet(self, filename: str):
        if filename in self._cache:
            return self._cache[filename]

        url = f"{self.cdn_base}/{filename}"
        resp = requests.get(url, timeout=30)
        resp.raise_for_status()
        table = pq.read_table(io.BytesIO(resp.content))
        # Project only required columns at parse time
        df = table.select(["prompt", "response"]).to_pandas()

        # Simple LRU cache
        self._cache[filename] = df
        self._cache_order.append(filename)
        if len(self._cache_order) > self.cache_limit:
            evict = self._cache_order.pop(0)
            self._cache.pop(evict, None)
        return df

    def __len__(self):
        # Logical length: total rows across files (lightweight)
        total = 0
        for f in self.files:
            df = self._fetch_parquet(f)
            total += len(df)
        return total

    def __getitem__(self, idx):
        # Map global idx to file + row
        for f in self.files:
            df = self._fetch_parquet(f)
            if idx < len(df):
                row = df.iloc[idx]
                return str(row["prompt"]), str(row["response"])
            idx -= len(df)
        raise IndexError("Index out of range")

# Usage in LightningModule
def train_dataloader(self):
    manifest = os.getenv("MANIFEST_PATH", "data/file_manifest_latest.json")
    dataset = CDNParquetDataset(manifest)
    return DataLoader(dataset, batch_size=8, shuffle=True, num_workers=0)
```

### 3) Lightning idle-resilient launcher
Check studio status before each `.run()`; auto-restart if stopped. Uses Lightning SDK; no external infra changes.

```python
# tools/lightning_train_launcher.py
import time
from lightning import LightningWork, LightningFlow, LightningApp, Machine
from lightning import LightningStudio

class ResilientTrainer(LightningWork):
    def __init__(self):
        super().__init__(machine=Machine.L40S, cloud="lightning-lambda-prod")
        self.studio_name = "surrogate-train-studio"

    def run(self):
        from lightning import Teamspace
        # Reuse running studio if available
        studio = None
        for s in Teamspace.studios:
            if s.name == self.studio_name and s.status == "Running":
                print(f"Reusing running studio: {self.studio_name}")
                studio = s
                break

        if studio is None:
            studio = LightningStudio(
                name=self.studio_name,
                machine=Machine.L40S,
                cloud="lightning-lambda-prod",
                create_ok=True
            )

        # Ensure studio is running before executing
        max_retries = 3
        for attempt in range(max_retries):
            if studio.status != "Running":
                print(f"Studio not running (status={studio.status}), restarting...")
                studio.start(machine=Machine.L40S, cloud="lightning-lambda-prod")
                time.sleep(60)
            else:
                break
        else:
            raise RuntimeError("Could not start studio after retries")

        # Execute training script inside studio
        manifest = os.getenv("MANIFEST_PATH", "data/file_manifest_latest.json")
        cmd = f"python surrogate/train.py --manifest {manifest}"
        studio.run(cmd, wait=True)

class TrainFlow(LightningFlow):
    def __init__(self):
        super().__init__()
        self.trainer = ResilientTrainer()

    def run(self):
        self.trainer.run()

app = LightningApp(TrainFlow())
```
