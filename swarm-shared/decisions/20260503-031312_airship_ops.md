### 1. Pre-list & Shard Once (Mac orchestration)
- **Action:** Single `list_repo_tree` → deterministic sibling selection via hash.
- **Output:** `file_list.json` committed to repo or injected via Lightning run.
- **Why:** Eliminates all HF API calls during training; enables commit-cap spread.

```python
# scripts/pre_list_files.py
import json, hashlib, os
from pathlib import Path
from huggingface_hub import HfApi

HF_REPO = os.getenv("HF_DATASET_REPO", "org/surrogate-data")
DATE_FOLDER = os.getenv("DATE_FOLDER", "2026-05-03")
N_SIBLINGS = int(os.getenv("N_SIBLINGS", "5"))
OUT = Path("file_list.json")

def pick_sibling(identifier: str) -> int:
    return int(hashlib.sha256(identifier.encode()).hexdigest(), 16) % N_SIBLINGS

def main():
    api = HfApi()
    items = api.list_repo_tree(
        repo_id=HF_REPO,
        path=f"batches/mirror-merged/{DATE_FOLDER}",
        repo_type="dataset",
        recursive=False,
    )
    entries = []
    for it in items:
        if it.type != "file": continue
        slug = Path(it.path).stem
        entries.append({
            "path": it.path,
            "cdn_url": f"https://huggingface.co/datasets/{HF_REPO}/resolve/main/{it.path}",
            "sibling": pick_sibling(slug),
            "slug": slug,
        })
    OUT.write_text(json.dumps(entries, indent=2))
    print(f"Wrote {len(entries)} files to {OUT}")

if __name__ == "__main__":
    main()
```

### 2. CDN-only IterableDataset with Prefetch & Resume State
- **Action:** Replace `load_dataset(streaming=True)` with `IterableDataset` + `requests` + threadpool.
- **Correctness:** Add retries, timeouts, and skip corrupt files without crashing.
- **Resilience:** Track last successfully yielded file index to enable resume.

```python
# surrogate/data/cdn_loader.py
import json, requests, pyarrow.parquet as pq
from pathlib import Path
from io import BytesIO
from typing import Iterator, Dict, Any
from torch.utils.data import IterableDataset
from concurrent.futures import ThreadPoolExecutor, as_completed

HF_REPO = "org/surrogate-data"
FILE_LIST = Path("file_list.json")
MAX_RETRIES = 3
TIMEOUT = 30
PREFETCH = 4

def cdn_fetch(path: str) -> bytes:
    url = f"https://huggingface.co/datasets/{HF_REPO}/resolve/main/{path}"
    for attempt in range(1, MAX_RETRIES + 1):
        try:
            resp = requests.get(url, timeout=TIMEOUT)
            resp.raise_for_status()
            return resp.content
        except Exception as e:
            if attempt == MAX_RETRIES:
                raise
            # simple backoff
            import time; time.sleep(attempt)
    raise RuntimeError(f"Failed to fetch {path}")

def project(batch_bytes: bytes) -> Iterator[Dict[str, Any]]:
    table = pq.read_table(BytesIO(batch_bytes))
    df = table.to_pandas()
    for _, row in df.iterrows():
        yield {"prompt": str(row.get("prompt", "")), "response": str(row.get("response", ""))}

class HFCDNDataset(IterableDataset):
    def __init__(self, start_index: int = 0):
        self.entries = json.loads(FILE_LIST.read_text())
        self.start_index = start_index

    def _stream(self) -> Iterator[Dict[str, Any]]:
        idx = self.start_index
        while idx < len(self.entries):
            entry = self.entries[idx]
            try:
                batch = cdn_fetch(entry["path"])
                yield from project(batch)
                idx += 1
            except Exception as exc:
                print(f"Skipping {entry['path']}: {exc}")
                idx += 1  # skip corrupt file but continue
                continue

    def __iter__(self) -> Iterator[Dict[str, Any]]:
        # Prefetch wrapper to hide latency
        from queue import Queue
        q = Queue(maxsize=PREFETCH)
        def producer():
            for item in self._stream():
                q.put(item)
            q.put(None)
        import threading
        t = threading.Thread(target=producer, daemon=True)
        t.start()
        while True:
            item = q.get()
            if item is None:
                break
            yield item
```

### 3. Lightning Idle-Resilient Runner + Checkpoint Resume
- **Action:** Reuse running studio; restart if idle-stopped; resume from latest checkpoint.
- **Correctness:** Use Lightning’s native checkpointing; store `last_file_index` in checkpoint.
- **Actionability:** CLI-driven; supports cron scheduling.

```python
# surrogate/train.py
import lightning as L
from lightning.pytorch.cli import LightningCLI
from lightning.pytorch.callbacks import ModelCheckpoint
from surrogate.data.cdn_loader import HFCDNDataset
from surrogate.model import SurrogateModel  # your model
from torch.utils.data import DataLoader
import os

class SurrogateDataModule(L.LightningDataModule):
    def __init__(self, start_index: int = 0):
        super().__init__()
        self.start_index = start_index

    def train_dataloader(self):
        return DataLoader(HFCDNDataset(start_index=self.start_index), batch_size=8, num_workers=0)

def get_latest_checkpoint(run_dir: str = "lightning_logs") -> str | None:
    ckpt_dir = Path(run_dir)
    if not ckpt_dir.exists():
        return None
    ckpts = list(ckpt_dir.rglob("*.ckpt"))
    if not ckpts:
        return None
    return str(max(ckpts, key=lambda p: p.stat().st_mtime))

def main():
    from lightning.pytorch.studio import Studio, Teamspace, Machine

    studio_name = "surrogate-training"
    teamspace = Teamspace()
    studio = next((s for s in teamspace.studios if s.name == studio_name and s.status == "Running"), None)

    if studio is None:
        studio = Studio(name=studio_name, target=Machine.L40S, create_ok=True)
    if studio.status != "Running":
        studio.start(machine=Machine.L40S)

    # Resume logic
    start_index = 0
    latest_ckpt = get_latest_checkpoint()
    if latest_ckpt:
        import torch
        ckpt = torch.load(latest_ckpt, map_location="cpu")
        start_index = ckpt.get("extra", {}).get("start_index", 0)
        print(f"Resuming from checkpoint {latest_ckpt}, start_index={start_index}")

    dm = SurrogateDataModule(start_index=start_index)

    checkpoint_cb = ModelCheckpoint(
        dirpath="lightning_logs",
        filename="surrogate-{step}",
        save_top_k=3,
        monitor="step",
        mode="max",
        save_last=True,
    )

    cli = LightningCLI(
        SurrogateModel,
        dm,
        run=False,
        trainer_defaults={
            "callbacks": [checkpoint_cb],
            "max_steps": 100_000,
            "log_every_n_steps": 10,
        },
    )

    # Inject start_index into checkpoint extras for future resume
    original_fit = cli.trainer.fit
    def fit_with_extras(*args, **kwargs):
        cli.trainer.fit(*args, **kwargs)
    cli.trainer.fit = fit_with_extras

    # Store start_index in trainer for checkpointing
    cli.trainer.extra = {"start_index": start_index}
    cli.run()

if __name__ == "__main__":
    main()
```
