# workio / design

## 1. Diagnosis

- Missing **LINE webhook signature verification** (`X-Line-Signature`) — any caller can spoof clock-in/out or leave events.
- No **idempotency on `line_event_id`** — LINE retries on 5xx/timeouts create duplicate punches and leave/OT rows.
- Clock-in/out UI has **no optimistic state + no local queue** — slow/unreliable networks cause double-taps and lost intent.
- Backend accepts events without **timestamp tolerance + replay window check** — enables delayed replay attacks.
- No **structured audit trail** for critical state changes (punch, leave, OT) tied to `line_event_id` and signature metadata.

## 2. Proposed change

Add a minimal, high-impact security + idempotency layer:

- **File**: `workio/server/src/routes/webhook/line.ts` (create if absent) or existing webhook handler.
- **Scope**:
  - Verify `X-Line-Signature` using HMAC-SHA256 with `CHANNEL_SECRET`.
  - Enforce idempotency on `event.id` (LINE’s `webhookEventId`) at DB level (unique constraint + upsert).
  - Add `webhook_events` table for audit + dedupe.
  - Frontend: optimistic punch state + local queue (IndexedDB) for offline/weak-network resilience.

// workio/server/src/middleware/lineSignature.ts
import crypto from 'crypto';

export function verifyLineSignature(
  rawBody: string,
  signature: string,
  channelSecret: string
): boolean {
  const expected = crypto
    .createHmac('sha256', channelSecret)
    .update(rawBody, 'utf8')
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

// workio/server/src/routes/webhook/line.ts
import express from 'express';
import { verifyLineSignature } from '../middleware/lineSignature.js';
import { pool } from '../../db/index.js';

const router = express.Router();
const CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';

// Raw body must be preserved for signature check
router.post(
  '/line',
  express.raw({ type: 'application/json' }),
  async (req, res) => {
    const signature = req.headers['x-line-signature'] as string;
    const rawBody = req.body.toString('utf8');

    if (!signature || !verifyLineSignature(rawBody, signature, CHANNEL_SECRET)) {
      return res.status(401).json({ error: 'Invalid signature' });
    }

    let events;
    try {
      events = JSON.parse(rawBody).events;
    } catch {
      return res.status(400).json({ error: 'Invalid JSON' });
    }

    for (const ev of events) {
      const { id, type, source, timestamp } = ev;
      try {
        // Idempotent insert; skip if already processed
        await pool.query(
          `INSERT INTO webhook_events (id, type, source_type, source_id, payload, signature, received_at)
           VALUES ($1, $2, $3, $4, $5, $6, to_timestamp($7 / 1000.0))
           ON CONFLICT (id) DO NOTHING`,
          [
            id,
            type,
            source.type,
            source.userId || source.groupId || source.roomId,
            ev,
            signature,
            timestamp,
          ]
        );

        // If row was inserted (not conflict), process business logic
        const { rowCount } = await pool.query(
          `SELECT 1 FROM webhook_events WHERE id = $1 AND processed_at IS NULL`,
          [id]
        );

        if (rowCount && rowCount > 0) {
          await processLineEvent(ev);

// workio/server/src/routes/punch.ts (idempotent punch creation)
import { pool } from '../db/index.js';

export async function createPunchIfNew(userId: string, lineEventId: string, type: 'in' | 'out', lat?: number, lng?: number) {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');
    // Dedupe by line_event_id at punch level too
    const { rows } = await client.query(
      `SELECT id FROM punches WHERE line_event_id = $1`,
      [lineEventId]
    );
    if (rows.length > 0) {
      await client.query('ROLLBACK');
      return rows[0];
    }

    const { rows: [punch] } = await client.query(
      `INSERT INTO punches (user_id, type, line_event_id, lat, lng, created_at)
       VALUES ($1, $2, $3, $4, $5, NOW())
       RETURNING *`,
      [userId, type, lineEventId, lat, lng]
    );
    await client.query('COMMIT');
    return punch;
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  } finally {
    client.release();
  }
}
