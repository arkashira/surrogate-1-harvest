# workio / ops

## 1. Diagnosis

- **No resilient offline clock-in/out**: current optimistic UI likely relies on `localStorage` queue (per past decisions) which is fragile, size-limited, and not cross-tab safe.
- **Missing idempotency for clock actions**: duplicate submissions on retry can create double punches; no idempotency keys or deduplication layer.
- **No durable retry/backoff**: transient network failures likely lose punches; no exponential backoff + jitter or persisted retry state.
- **No background flush strategy**: relies on user-triggered actions only; no online-triggered + periodic flush to recover stranded records.
- **Ops observability gap**: no lightweight local telemetry (counts, last flush, failures) to detect queue health or trigger alerts.

## 2. Proposed change

- **File**: `workio/src/lib/queue/attendanceQueue.ts` (new)
- **Scope**: implement an IndexedDB-backed, idempotent attendance queue with durable retry state, exponential backoff + jitter, and cross-tab safe operations.
- **Integration points**:
  - Replace existing `localStorage` usage in clock-in/out flows with `AttendanceQueue.enqueue()`.
  - Add `flushLoop()` started on app init + online event + periodic timer (30s).
  - Expose minimal ops helper `getQueueHealth()` for dashboard/debug.

## 3. Implementation

```ts
// workio/src/lib/queue/attentionQueue.ts
import { openDB, DBSchema, IDBPDatabase } from 'idb';

const DB_NAME = 'workio-attendance';
const STORE_NAME = 'outbox';
const RETRY_BASE_MS = 1_000;
const RETRY_MAX_MS = 60_000;
const RETRY_JITTER = 0.3;
const FLUSH_INTERVAL_MS = 30_000;
const IDB_MAX_RETRIES = 3;

interface AttendanceRecord {
  id: string; // idempotency key (uuid)
  action: 'clock-in' | 'clock-out' | 'leave' | 'ot';
  payload: Record<string, unknown>;
  createdAt: number;
  attempts: number;
  nextRetryAt: number | null;
  lastError: string | null;
  completed: boolean;
}

interface WorkioDB extends DBSchema {
  [STORE_NAME]: {
    key: string;
    value: AttendanceRecord;
    indexes: {
      byCompleted: boolean;
      byNextRetry: number;
    };
  };
}

function jitter(base: number): number {
  const sign = Math.random() > 0.5 ? 1 : -1;
  return base + sign * base * RETRY_JITTER * Math.random();
}

async function withIdbRetry<T>(fn: (db: IDBPDatabase<WorkioDB>) => Promise<T>): Promise<T> {
  let lastErr: unknown;
  for (let i = 0; i < IDB_MAX_RETRIES; i++) {
    try {
      const db = await getDB();
      return await fn(db);
    } catch (err) {
      lastErr = err;
      await new Promise((r) => setTimeout(r, 100 * Math.pow(2, i)));
    }
  }
  throw lastErr;
}

let _dbPromise: Promise<IDBPDatabase<WorkioDB>> | null = null;

function getDB() {
  if (!_dbPromise) {
    _dbPromise = openDB<WorkioDB>(DB_NAME, 1, {
      upgrade(db) {
        const store = db.createObjectStore(STORE_NAME, { keyPath: 'id' });
        store.createIndex('byCompleted', 'completed', { unique: false });
        store.createIndex('byNextRetry', 'nextRetryAt', { unique: false });
      },
    });
  }
  return _dbPromise;
}

export async function enqueue(
  action: AttendanceRecord['action'],
  payload: Record<string, unknown>
): Promise<string> {
  const id = crypto.randomUUID();
  const record: AttendanceRecord = {
    id,
    action,
    payload,
    createdAt: Date.now(),
    attempts: 0,
    nextRetryAt: null,
    lastError: null,
    completed: false,
  };
  await withIdbRetry(async (db) => db.add(STORE_NAME, record));
  // Best-effort immediate flush attempt
  void flushOnce();
  return id;
}

export async function flushOnce(): Promise<{ processed: number; failed: number }> {
  const db = await getDB();
  const tx = db.transaction(STORE_NAME, 'readwrite');
  const store = tx.objectStore(STORE_NAME);
  const idx = store.index('byNextRetry');
  const now = Date.now();

  // Fetch incomplete records that are ready to retry (nextRetryAt <= now or null)
  let cursor = await idx.openCursor(IDBKeyRange.upperBound(now));
  let processed = 0;
  let failed = 0;

  while (cursor) {
    const record = cursor.value;
    if (record.completed) {
      cursor = await cursor.continue();
      continue;
    }

    try {
      await sendToServer(record);
      record.completed = true;
      processed++;
      await cursor.update(record);
      await cursor.delete(); // remove completed
    } catch (err: any) {
      record.attempts += 1;
      record.lastError = err?.message || String(err);
      const delay = Math.min(RETRY_BASE_MS * Math.pow(2, record.attempts - 1), RETRY_MAX_MS);
      record.nextRetryAt = Date.now() + jitter(delay);
      failed++;
      await cursor.update(record);
    }

    cursor = await cursor.continue();
  }

  await tx.done;
  return { processed, failed };
}

async function sendToServer(record: AttendanceRecord): Promise<void> {
  const res = await fetch('/api/attendance', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      action: record.action,
      ...record.payload,
      _idempotencyKey: record.id,
    }),
  });

  if (!res.ok) {
    const text = await res.text().catch(() => '');
    throw new Error(`HTTP ${res.status}: ${text}`);
  }
}

export async function getQueueHealth() {
  const db = await getDB();
  const tx = db.transaction(STORE_NAME, 'readonly');
  const store = tx.objectStore(STORE_NAME);
  const all = await store.getAll();
  const pending = all.filter((r) => !r.completed);
  const completed = all.filter((r) => r.completed);
  return {
    total: all.length,
    pending: pending.length,
    completed: completed.length,
    oldestPending: pending.reduce((a, b) => (a.createdAt < b.createdAt ? a : b), pending[0] || null)?.createdAt ?? null,
  };
}

// Background flush loop (safe to call multiple times)
let _flushInterval: ReturnType<typeof setInterval> | null = null;

export function startFlushLoop() {
  if (_flushInterval) return;
  void flushOnce();
  _flushInterval = setInterval(() => void flushOnce(), FLUSH_INTERVAL_MS);
  window.addEventListener('online', () => void flushOnce());
}

export function stopFlushLoop() {
  if (_flushInterval) {
    clearInterval(_flushInterval);
    _flushInterval = null;
  }
}
```
