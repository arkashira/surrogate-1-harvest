## 2) API client (optional, future-ready) with graceful fallback

`src/lib/knowledgeRag.ts`
```ts
// Types shared by API and static data
export interface Signal {
  id?: string;
  title: string;
  snippet?: string;
  summary?: string;
  href?: string;
  url?: string;
}

export interface Hub {
  id: string;
  name: string;
  label?: string;
  description: string;
  inDegree?: number;
  outDegree?: number;
  signals: Signal[];
}

export interface TopHubResponse {
  hub: Hub;
  signals: Signal[];
}

export interface KnowledgeGraph {
  hubs: Hub[];
  generatedAt: string;
  source: string;
}

const API_ENDPOINT = "/api/knowledge-rag/top-hub";

export async function fetchTopHub(): Promise<TopHubResponse | null> {
  try {
    const res = await fetch(API_ENDPOINT, { method: "GET", cache: "no-store" });
    if (!res.ok) {
      // treat 429/5xx as unavailable to avoid UI instability
      console.warn(`knowledge-rag endpoint returned ${res.status}`);
      return null;
    }
    const data = (await res.json()) as TopHubResponse;
    return data;
  } catch (err) {
    console.warn("knowledge-rag fetch failed", err);
    return null;
  }
}
```

function getTopHubFromStatic(graph: KnowledgeGraph): Hub | null {
  if (!graph.hubs.length) return null;
  return graph.hubs.reduce((best, cur) => {
    const bestScore = (best.inDegree ?? 0) + (best.outDegree ?? 0);
    const curScore = (cur.inDegree ?? 0) + (cur.outDegree ?? 0);
    return curScore > bestScore ? cur : best;
  });
}

}
