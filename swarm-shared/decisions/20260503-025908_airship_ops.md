### 1) Pre-list HF file paths once → embed JSON for CDN-only training  
**Why**: Avoids recursive/paginated `list_repo_files` (100× calls → 429). Uses CDN bypass (`/resolve/main/...`) for zero-auth, high-limit downloads during training.  

**Location**: `surrogate/training/prepare_file_list.py`

```python
#!/usr/bin/env python3
"""
Run from Mac (or any dev box) after rate-limit window clears.
Produces: surrogate/training/file_list_YYYYMMDD.json
"""
import json
import os
import argparse
from datetime import datetime
from huggingface_hub import HfApi

DEFAULT_REPO = "axentx/surrogate-dataset-mirror"

def build_file_list(repo_id: str, date_folder: str, out_path: str):
    api = HfApi()
    entries = api.list_repo_tree(
        repo_id=repo_id,
        path=date_folder,
        recursive=False
    )
    file_paths = [e.rfilename for e in entries if e.rfilename.endswith(".parquet")]
    payload = {
        "repo_id": repo_id,
        "date_folder": date_folder,
        "files": sorted(file_paths),
        "generated_utc": datetime.utcnow().isoformat() + "Z"
    }
    os.makedirs(os.path.dirname(out_path), exist_ok=True)
    with open(out_path, "w") as f:
        json.dump(payload, f, indent=2)
    print(f"Wrote {len(file_paths)} files -> {out_path}")

if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", default=DEFAULT_REPO)
    parser.add_argument("--date", default=datetime.utcnow().strftime("%Y-%m-%d"))
    parser.add_argument("--out", help="Output JSON path")
    args = parser.parse_args()

    if args.out is None:
        args.out = os.path.join(
            os.path.dirname(__file__),
            f"file_list_{args.date.replace('-','')}.json"
        )
    build_file_list(args.repo, args.date, args.out)
```

**Usage (Mac orchestration only)**:
```bash
cd /opt/axentx/airship/surrogate/training
python prepare_file_list.py --date 2026-05-03
```

Commit or keep `file_list_YYYYMMDD.json` locally; training script will read it and fetch via CDN URLs without any HF API calls.

### 2) Lightning Studio reuse + idle-stop guard  
**Why**: Prevents quota waste and training death when idle timeout kills the studio.

**Location**: `surrogate/training/launch_lightning.py`

```python
#!/usr/bin/env python3
"""
Launch or reuse a Lightning Studio for Surrogate training.
Prevents recreation when already running; restarts only if stopped.
"""
import os
import sys
import time
from lightning_sdk import Studio, Machine, Teamspace

STUDIO_NAME = "surrogate-train-l40s"
MACHINE = Machine.L40S
MAX_RETRIES = 3
RETRY_WAIT = 60

def get_running_studio():
    teamspace = Teamspace()
    for s in teamspace.studios:
        if s.name == STUDIO_NAME:
            return s
    return None

def ensure_studio_running():
    studio = get_running_studio()
    if studio is None:
        print(f"Creating studio {STUDIO_NAME}...")
        studio = Teamspace().create_studio(
            name=STUDIO_NAME,
            machine=MACHINE,
            create_ok=True
        )
        studio.start(machine=MACHINE)
        return studio

    if studio.status == "running":
        print(f"Reusing running studio {STUDIO_NAME}")
        return studio

    if studio.status == "stopped":
        print(f"Studio stopped — restarting {STUDIO_NAME}")
        studio.start(machine=MACHINE)
        return studio

    print(f"Studio status={studio.status} — waiting...")
    time.sleep(30)
    return ensure_studio_running()

def run_training_script(script_path: str, args: list[str]):
    studio = ensure_studio_running()
    for attempt in range(1, MAX_RETRIES + 1):
        try:
            run = studio.run(
                name="surrogate-train-run",
                command=["python", script_path] + args,
                environment={
                    "HF_DATASETS_OFFLINE": "1",
                    "LIGHTNING_IDLE_TIMEOUT_MIN": "30"
                }
            )
            print(f"Run started: {run.id}")
            run.wait_until_finished()
            if run.status == "failed":
                raise RuntimeError(f"Run failed: {run.id}")
            return run
        except Exception as exc:
            print(f"Attempt {attempt} failed: {exc}")
            if attempt == MAX_RETRIES:
                raise
            time.sleep(RETRY_WAIT)
            studio = ensure_studio_running()

if __name__ == "__main__":
    script = sys.argv[1] if len(sys.argv) > 1 else "train.py"
    extra = sys.argv[2:] if len(sys.argv) > 2 else []
    run_training_script(script, extra)
```

**Mac orchestration** (example):
```bash
cd /opt/axentx/airship/surrogate/training
python launch_lightning.py train.py --file-list file_list_20260503.json --epochs 1
```

### 3) Training script: CDN-only dataset loading (zero API calls)  
**Why**: Uses pre-listed JSON and direct CDN URLs to avoid HF API during data loading.

**Location**: `surrogate/training/train.py` (excerpt / changes)

```python
import json
import os
import torch
from torch.utils.data import Dataset, DataLoader
import pyarrow.parquet as pq
import requests
from tqdm import tqdm

class CDNParquetDataset(Dataset):
    def __init__(self, file_list_path, max_files=None):
        with open(file_list_path) as f:
            meta = json.load(f)
        self.repo_id = meta["repo_id"]
        files = meta["files"]
        if max_files:
            files = files[:max_files]
        self.urls = [
            f"https://huggingface.co/datasets/{self.repo_id}/resolve/main/{f}"
            for f in files
        ]
        self.samples = self._load_all()

    def _load_all(self):
        samples = []
        for url in tqdm(self.urls, desc="Downloading parquet via CDN"):
            resp = requests.get(url, timeout=120)
            resp.raise_for_status()
            table = pq.read_table(resp.content)
            samples.extend(table.to_pylist())
        return samples

    def __len__(self):
        return len(self.samples)

    def __getitem__(self, idx):
        return self.samples[idx]

# Example usage in training loop
if __name__ == "__main__":
    import argparse
    parser = argparse.ArgumentParser()
    parser.add_argument("--file-list", required=True)
    parser.add_argument("--epochs", type=int, default=1)
    parser.add_argument("--batch-size", type=int, default=8)
    args = parser.parse_args()

    dataset = CDNParquetDataset(args.file_list)
    loader = DataLoader(dataset, batch_size=args.batch_size, shuffle=True)

    model = torch.nn.Linear(1024, 1)  # placeholder
    opt = torch.optim.Adam(model.parameters())

    for epoch in range(args.epochs):
        for batch in tqdm(loader, desc=f"Epoch {epoch}"):
            # Replace with real feature extraction
            x = torch.randn(len(batch), 1024)
            y = model(x).sum()
            loss = y.mean()
            opt.zero_grad()
            loss.backward()
            opt.step()
```
