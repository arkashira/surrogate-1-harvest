# workio / frontend

## 1. Diagnosis

- Attendance events are currently buffered in `localStorage` (ephemeral, no durability guarantees) and submitted with naive `fetch` — duplicates and lost events occur on flaky networks or page reloads.
- No idempotency: retries resend identical payloads; server may double-insert clock-in/out records.
- No durable retry/backoff: offline or 5xx responses lose events or spam the server with unbounded retries.
- Cross-tab unsafe: multiple browser tabs can race on the same queue and submit duplicates.
- No visibility: UI doesn’t surface pending/synced/failed events, so users can’t know whether clock-in/out succeeded.

## 2. Proposed change

File scope: `workio/src/lib/queue/attendanceQueue.ts` (new) + integrate into `workio/src/lib/api/attendance.ts` and `workio/src/components/AttendanceButton.tsx`.  
Scope: add an IndexedDB-backed, idempotent queue with exponential backoff + jitter, cross-tab coordination via BroadcastChannel, and online-triggered + periodic flush; surface pending state in UI.

### 3.2 Create queue module

`src/lib/queue/attendanceQueue.ts`

```ts
import { openDB, DBSchema, IDBPDatabase } from 'idb';

interface AttendanceDB extends DBSchema {
  queue: {
    key: string; // idempotency key (uuid)
    value: {
      idempotencyKey: string;
      type: 'clock-in' | 'clock-out';
      payload: {
        userId: string;
        timestamp: string; // ISO
        latitude?: number;
        longitude?: number;
        locationName?: string;
      };
      attempts: number;
      createdAt: number;
      nextRetryAt: number | null;
      lastError?: string;
    };
  };
  meta: {
    key: string;
    value: { flushedAt: number };
  };
}

const DB_NAME = 'workio-attendance';
const QUEUE_NAME = 'attendance-queue';
const MAX_ATTEMPTS = 5;
const BASE_DELAY_MS = 2000; // 2s

function jitter(delay: number) {
  return delay * (0.5 + Math.random() * 0.5); // 0.5x - 1.0x
}

function backoff(attempts: number) {
  return jitter(Math.min(BASE_DELAY_MS * 2 ** Math.min(attempts - 1, 6), 30_000));
}

async function getDb(): Promise<IDBPDatabase<AttendanceDB>> {
  return openDB<AttendanceDB>(DB_NAME, 1, {
    upgrade(db) {
      db.createObjectStore('queue', { keyPath: 'key' });
      db.createObjectStore('meta');
    },
  });
}

// Cross-tab coordination
const channel = new BroadcastChannel('workio-attendance-queue');

export async function enqueueAttendance(
  userId: string,
  type: 'clock-in' | 'clock-out',
  payload: {
    timestamp: string;
    latitude?: number;
    longitude?: number;
    locationName?: string;
  }
) {
  const idempotencyKey = crypto.randomUUID();
  const db = await getDb();
  await db.add('queue', {
    idempotencyKey,
    type,
    payload,
    attempts: 0,
    createdAt: Date.now(),
    nextRetryAt: null,
  });
  db.close();
  channel.postMessage({ type: 'enqueue', idempotencyKey });
  // Best-effort immediate flush if online
  if (navigator.onLine) void flushQueue();
}

export async function flushQueue() {
  const db = await getDb();
  const tx = db.transaction('queue', 'readwrite');
  const store = tx.objectStore('queue');
  const now = Date.now();
  const cursor = await store.openCursor();

  async function process(cursor: IDBCursorWithValue | null): Promise<void> {
    if (!cursor) return;
    const item = cursor.value;
    if (item.nextRetryAt && item.nextRetryAt > now) {
      await cursor.continue();
      return;
    }
    if (item.attempts >= MAX_ATTEMPTS) {
      // Keep for visibility; could move to dead-letter
      await cursor.continue();
      return;
    }

    try {
      const res = await fetch('/api/attendance/events', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          ...item.payload,
          type: item.type,
          idempotencyKey: item.idempotencyKey,
        }),
      });

      if (!res.ok) {
        throw new Error(`HTTP ${res.status}`);
      }

      await cursor.delete();
      // Continue processing remaining items in same tx
      await cursor.continue();
    } catch (err: any) {
      const attempts = item.attempts + 1;
      const nextRetryAt = attempts < MAX_ATTEMPTS ? now + backoff(attempts) : null;
      await store.put({
        ...item,
        attempts,
        nextRetryAt,
        lastError: err?.message ?? 'unknown',
      });
      await cursor.continue();
    }
  }

  await process(await cursor);
  await tx.done;
  db.close();
  channel.postMessage({ type: 'flush' });
}

// Periodic flush (every 30s when online)
if (typeof window !== 'undefined') {
  setInterval(() => {
    if (navigator.onLine) void flushQueue();
  }, 30_000);
  window.addEventListener('online', () => void flushQueue());
}
```

### 3.3 Update API client to use queue

`src/lib/api/attendance.ts`

```ts
import { enqueueAttendance } from '../queue/attendanceQueue';

export async function recordAttendance(
  userId: string,
  type: 'clock-in' | 'clock-out',
  payload: {
    timestamp: string;
    latitude?: number;
    longitude?: number;
    locationName?: string;
  }
) {
  // Always enqueue for durability; flush will submit when possible.
  await enqueueAttendance(userId, type, payload);
}
```

  userId: string;
  currentStatus: 'in' | 'out';
  onSuccess?: () => void;
}

  const [pending, setPending] = useState(false);

  useEffect(() => {
    const channel = new BroadcastChannel('workio-attendance-queue');
    channel.onmessage = () => refreshPending();

    return () => channel.close();

    const db = await getDb();
    const count = await db.count('queue');
    db.close();
    setPending(count > 0);

    onSuccess?.();
