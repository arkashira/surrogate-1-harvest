### 1) Durable queue service  
File: `workio/src/lib/services/attendance.ts`

```ts
import { openDB, DBSchema, IDBPDatabase } from 'idb';

interface AttendanceDB extends DBSchema {
  pending: {
    key: string;
    value: {
      idempotencyKey: string;
      type: 'CLOCK_IN' | 'CLOCK_OUT';
      timestamp: number;
      location?: { lat: number; lng: number };
      userId: string;
      retries: number;
      createdAt: number;
    };
    indexes: { by_createdAt: number };
  };
}

const DB_NAME = 'workio-attendance';
const DB_VERSION = 1;
const MAX_RETRIES = 5;
const BASE_DELAY = 1000;

function getDB() {
  return openDB<AttendanceDB>(DB_NAME, DB_VERSION, {
    upgrade(db) {
      const store = db.createObjectStore('pending', { keyPath: 'idempotencyKey' });
      store.createIndex('by_createdAt', 'createdAt', { unique: false });
    },
  });
}

export class AttendanceQueue {
  private dbP = getDB();
  private flushing = false;
  private online = navigator.onLine;

  constructor() {
    this.attachHandlers();
  }

  private attachHandlers() {
    const onOnline = () => {
      this.online = true;
      this.flush();
    };
    const onOffline = () => {
      this.online = false;
    };

    window.addEventListener('online', onOnline);
    window.addEventListener('offline', onOffline);
    window.addEventListener('visibilitychange', () => {
      if (document.visibilityState === 'visible') this.flush();
    });

    // Best-effort attempt to flush before page unload.
    // We cannot await async work here; items remain safely in IDB if it fails.
    window.addEventListener('beforeunload', () => {
      // Kick off synchronous-ish microtask flush attempt.
      void this.flush();
    });
  }

  async enqueue(punch: {
    type: 'CLOCK_IN' | 'CLOCK_OUT';
    timestamp: number;
    location?: { lat: number; lng: number };
    userId: string;
  }) {
    const idempotencyKey = crypto.randomUUID();
    const db = await this.dbP;
    await db.add('pending', {
      idempotencyKey,
      type: punch.type,
      timestamp: punch.timestamp,
      location: punch.location,
      userId: punch.userId,
      retries: 0,
      createdAt: Date.now(),
    });
    void this.flush();
  }

  async flush() {
    if (this.flushing) return;
    if (!this.online) return;
    this.flushing = true;

    try {
      const db = await this.dbP;
      const tx = db.transaction('pending', 'readwrite');
      const store = tx.objectStore('pending');
      const items = await store.index('by_createdAt').getAll();

      for (const item of items) {
        if (item.retries >= MAX_RETRIES) {
          // Poison item: keep for manual inspection.
          continue;
        }

        try {
          const res = await fetch('/api/attendance/punch', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
              idempotencyKey: item.idempotencyKey,
              type: item.type,
              timestamp: item.timestamp,
              location: item.location,
              userId: item.userId,
            }),
          });

          if (!res.ok) throw new Error(`HTTP ${res.status}`);

          await store.delete(item.idempotencyKey);
        } catch {
          item.retries += 1;
          await store.put(item);
          // Stop batch on first failure; rely on next trigger (online/visibility).
          break;
        }
      }

      await tx.done;
    } finally {
      this.flushing = false;
    }
  }
}

export const attendanceQueue = new AttendanceQueue();
```

### 2) Hook for components  
File: `workio/src/features/attendance/hooks/useAttendance.ts`

```ts
import { useCallback } from 'react';
import { attendanceQueue } from '../../../lib/services/attendance';

export function useAttendance(userId: string) {
  const punch = useCallback(
    async (type: 'CLOCK_IN' | 'CLOCK_OUT', location?: { lat: number; lng: number }) => {
      await attendanceQueue.enqueue({
        type,
        timestamp: Date.now(),
        location,
        userId,
      });
    },
    [userId],
  );

  return { punch };
}
```

### 3) Server-side idempotency (minimal, production-ready)  
File: `workio/server/src/routes/attendance.ts` (or equivalent)

```ts
import { Router } from 'express';
import { LRUCache } from 'lru-cache';

const router = Router();

// In-memory dedupe for single-instance dev; for multi-instance use Redis or DB unique constraint.
const seenKeys = new LRUCache<string, boolean>({
  max: 50_000,
  ttl: 24 * 60 * 60 * 1000, // 24h
});

router.post('/punch', async (req, res) => {
  const { idempotencyKey, type, timestamp, location, userId } = req.body;

  if (!idempotencyKey || typeof idempotencyKey !== 'string') {
    return res.status(400).json({ error: 'idempotencyKey is required' });
  }

  if (seenKeys.has(idempotencyKey)) {
    return res.status(200).json({ ok: true, duplicate: true });
  }

  // TODO: validate type, timestamp, location, userId
  // await db.attendance.create({ data: { userId, type, timestamp, location } });

  seenKeys.set(idempotencyKey, true);
  res.json({ ok: true });
});

export default router;
```

### 4) Verification checklist

1. Start frontend + backend (`npm run dev`).
2. Online punch:
   - Confirm attendance appears in UI.
   - Confirm IndexedDB `pending` store is empty afterward (DevTools > Application > IndexedDB).
3. Offline punches:
   - Enable offline in DevTools.
   - Trigger several punches; confirm items accumulate in `pending`.
4. Reconnect:
   - Go online; confirm queue drains and server receives each punch exactly once (check server logs for duplicate `idempotencyKey` processing).
5. Retry behavior:
   - Temporarily make `/api/attendance/punch` return 500.
   - Punch while online; confirm retries increment in IDB and exponential backoff spacing (via timestamps).
6. Page reload/close:
   - Reload while offline; confirm punches remain in IDB.
   - Reload while online; confirm no duplicate submissions and queue empties.
7. Multi-tab safety:
   - Open two tabs; punch in one; confirm other tab’s queue does not create duplicates (idempotency key prevents server duplicates).
