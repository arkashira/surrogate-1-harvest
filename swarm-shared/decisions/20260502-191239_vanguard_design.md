### Key decisions
- **Keep Candidate 1’s directory layout and new-file scope** (no edits to existing files).  
- **Adopt Candidate 1’s CDN file-list + caching pattern** (single API call → JSON cache → CDN URLs) — resolves rate-limit risk.  
- **Adopt Candidate 1’s Studio reuse + idle-stop restart guard** — prevents quota waste.  
- **Fix Candidate 1’s wrapper script bug** (`${BASH[@]}` → `$(basename "$0")`/`dirname`) and make it truly cron-safe and relocatable.  
- **Enforce Mac/remote separation**: orchestrator runs locally (Mac/CI) and only submits remote jobs; remote code never imports HF API at runtime.  
- **Hardening**: strict `set -euo pipefail`, timeouts, minimal deps in hot path, clear failure messages.

### 2) `/opt/axentx/vanguard/design/entrypoints.py`
```python
"""
Canonical discovery entrypoints for vanguard.
Patterns: #knowledge-rag #graph #hub #cdn #rate-limit-bypass
"""
from __future__ import annotations

import json
import os
from pathlib import Path
from typing import Dict, List, Optional

try:
    from huggingface_hub import list_repo_tree
except ImportError:
    list_repo_tree = None  # type: ignore


def get_top_hub_insight(hub_name: str = "MOC") -> Dict:
    """
    Surface top-connected hub insight before planning.
    Replace with actual knowledge-rag graph query when available.
    """
    return {
        "hub": hub_name,
        "type": "top-connected",
        "insight": f"Review {hub_name} context and linked docs before planning tasks.",
        "tags": ["#knowledge-rag", "#graph", "#hub"],
    }


def build_cdn_file_list(
    repo_id: str,
    folder_path: str,
    cache_dir: Optional[str] = None,
) -> List[str]:
    """
    List once, cache, return CDN URLs.
    Embed this list in training scripts to avoid HF API calls during data load.
    """
    if list_repo_tree is None:
        raise RuntimeError("huggingface_hub not installed")

    cache_path = None
    if cache_dir:
        os.makedirs(cache_dir, exist_ok=True)
        safe_name = repo_id.replace("/", "_")
        safe_folder = folder_path.strip("/").replace("/", "_") or "root"
        cache_path = Path(cache_dir) / f"{safe_name}_{safe_folder}.json"
        if cache_path.exists():
            try:
                cached = json.loads(cache_path.read_text())
                if isinstance(cached, list) and cached:
                    return cached
            except Exception:
                cache_path.unlink(missing_ok=True)

    items = list_repo_tree(repo_id=repo_id, path=folder_path, recursive=False)
    paths = [it["path"] for it in items if isinstance(it, dict) and it.get("type") == "file"]
    urls = [f"https://huggingface.co/datasets/{repo_id}/resolve/main/{p}" for p in paths]

    if cache_path:
        cache_path.write_text(json.dumps(urls, indent=2))
    return urls


def cdn_download_single(repo_id: str, file_path: str, target_dir: str) -> str:
    """
    Download a single file via CDN (no Authorization header required).
    Returns local path.
    """
    os.makedirs(target_dir, exist_ok=True)
    local_path = os.path.join(target_dir, os.path.basename(file_path))
    url = f"https://huggingface.co/datasets/{repo_id}/resolve/main/{file_path}"
    import requests

    resp = requests.get(url, timeout=60)
    resp.raise_for_status()
    with open(local_path, "wb") as f:
        f.write(resp.content)
    return local_path
```

### 3) `/opt/axentx/vanguard/design/train_orchestrator.py`
```python
"""
Lightning Studio orchestration guard + CDN-only data strategy.
Patterns: #lightning-ai #quota #cdn #rate-limit-bypass

Run this on Mac/CI (orchestrator). Remote Studio runs train.py which must:
  - accept --file-list <json> produced here
  - load data via CDN URLs (no HF API calls in training loop)
"""
from __future__ import annotations

import json
import os
import sys
from pathlib import Path
from typing import Optional

try:
    from lightning import Studio, Teamspace, Machine
except ImportError:
    # Design-time import tolerance
    Studio = object  # type: ignore
    Teamspace = object  # type: ignore
    Machine = object  # type: ignore


def reuse_or_create_studio(
    name: str,
    machine: str = "L40S",
    cloud: str = "lightning-public-prod",
) -> Studio:
    """
    Reuse running studio to save quota; restart if idle-stopped.
    """
    team = Teamspace()
    running = next((s for s in team.studios if s.name == name and s.status == "running"), None)
    if running is not None:
        print(f"Reusing running studio: {name}")
        return running

    stopped = next((s for s in team.studios if s.name == name), None)
    if stopped is not None:
        print(f"Restarting stopped studio: {name}")
        stopped.start(machine=Machine(machine, cloud=cloud))
        return stopped

    print(f"Creating studio: {name}")
    return Studio(
        name=name,
        machine=Machine(machine, cloud=cloud),
        create_ok=True,
    )


def run_cdn_training(
    repo_id: str,
    date_folder: str,
    studio_name: str,
    train_script: str = "train.py",
) -> None:
    """
    Orchestrate CDN-only training:
      1) List files once and cache CDN URLs.
      2) Launch Lightning Studio with file-list argument.
    """
    from .entrypoints import build_cdn_file_list

    file_urls = build_cdn_file_list(
        repo_id=repo_id,
        folder_path=date_folder,
        cache_dir=".cache/vanguard",
    )
    print(f"Prepared {len(file_urls)} CDN file URLs")

    list_path = Path(".cache/vanguard") / f"filelist_{date_folder.strip('/').replace('/', '_') or 'root'}.json"
    list_path.parent.mkdir(parents=True, exist_ok=True)
    list_path.write_text(json.dumps(file_urls, indent=2))

    studio = reuse_or_create_studio(name=studio_name)
    if getattr(studio, "status", None) != "running":
        print("Studio not running; restarting...")
        studio = reuse_or_create_studio(name=studio_name)

    studio.run(
        [
            sys.executable,
            train_script,
            "--file-list",
            str(list_path),
            "--repo",
            repo_id,
        ],
        cwd=os.getcwd(),
    )
```

#!/usr/bin/env bash
# Hardened wrapper for surrogate-1 training.
# Patterns: #bash #script-error #wrapper #cron

set -euo pipefail

# Relocate to project root reliably
SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
PROJECT_ROOT="$(cd "${SCRIPT_DIR}/../.." && pwd)"
cd "${PROJECT_ROOT}" || { echo "Cannot cd to ${PROJECT_ROOT}"; exit 1; }

# Optional: activate environment if present
if [[ -f /opt/axentx/venv/bin/activate ]]; then
    # shellcheck source=/dev/null
    source /opt/axentx/venv/bin/activate
fi

exec python -m design.train_orchestrator "$@"
