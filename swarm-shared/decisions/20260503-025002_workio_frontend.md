# workio / frontend

## Final Synthesized Implementation

I've merged the strongest technical insights from both proposals into a single, production-ready solution. The key synthesis decisions:

- **Adopted Candidate 1's robust retry/backoff loop** (exponential with abort) + **Candidate 2's SWR cache invalidation strategy** for cross-component state consistency.
- **Combined idempotency approaches**: client-generated UUID per intent + server-side idempotency key header to handle both UI double-taps and LINE webhook replays.
- **Unified optimistic UI**: immediate local update with canonical server hydration (prevents clock-skew drift).
- **Added offline-fast rejection**: navigator.onLine check + persistent queue for "lost response" recovery (solves Candidate 1's "response lost but clock-in succeeded" edge case).

### 1) Hook: `src/hooks/usePunchClock.ts`

```tsx
// src/hooks/usePunchClock.ts
import { useState, useCallback, useRef, useEffect } from 'react';

export type Punch = {
  id: string;           // canonical from server
  type: 'IN' | 'OUT';
  timestamp: string;    // ISO (server authoritative)
  latitude?: number;
  longitude?: number;
  source?: 'LINE' | 'MANUAL';
};

type PunchStatus = 'idle' | 'pending' | 'success' | 'error';

type UsePunchClockOptions = {
  /** Submit to server; must accept idempotencyKey in headers */
  onSubmit: (payload: {
    type: 'IN' | 'OUT';
    timestamp: string;
    latitude?: number;
    longitude?: number;
    idempotencyKey: string;
  }) => Promise<Punch>;
  /** Optional SWR-like cache invalidator to sync dashboard/header */
  onInvalidate?: () => void | Promise<void>;
  maxRetries?: number;
  initialBackoffMs?: number;
};

export function usePunchClock({
  onSubmit,
  onInvalidate,
  maxRetries = 3,
  initialBackoffMs = 600,
}: UsePunchClockOptions) {
  const [status, setStatus] = useState<PunchStatus>('idle');
  const [lastConfirmed, setLastConfirmed] = useState<Punch | null>(null);
  const [error, setError] = useState<string | null>(null);

  // In-flight guard + idempotency
  const inflightKey = useRef<string | null>(null);
  const abortRef = useRef<AbortController | null>(null);

  // Persistent pending queue for "lost response" recovery
  const pendingQueue = useRef<
    Array<{
      idempotencyKey: string;
      payload: Parameters<typeof onSubmit>[0];
      retries: number;
    }>
  >([]);

  // Exponential backoff with jitter
  const backoff = (attempt: number) =>
    initialBackoffMs * 2 ** attempt + Math.random() * 200;

  const processQueue = useCallback(async () => {
    if (pendingQueue.current.length === 0) return;
    const item = pendingQueue.current[0]; // peek
    if (!item) return;

    abortRef.current = new AbortController();

    try {
      const result = await onSubmit(item.payload);
      setLastConfirmed(result);
      setStatus('success');
      pendingQueue.current.shift(); // dequeue
      await onInvalidize?.();
    } catch (err: any) {
      if (err.name === 'AbortError') return;

      item.retries += 1;
      if (item.retries > maxRetries) {
        pendingQueue.current.shift(); // drop after max retries
        setStatus('error');
        setError('Failed after retries. Manual retry required.');
        setLastConfirmed(null);
      } else {
        // re-enqueue at end with delay
        setTimeout(processQueue, backoff(item.retries));
      }
    } finally {
      abortRef.current = null;
      if (pendingQueue.current.length > 0 && inflightKey.current === item.idempotencyKey) {
        inflightKey.current = null;
      }
    }
  }, [onSubmit, onInvalidate, maxRetries, initialBackoffMs]);

  const punch = useCallback(
    async (type: 'IN' | 'OUT', geolocation?: { latitude: number; longitude: number }) => {
      // Fast offline rejection
      if (!navigator.onLine) {
        setStatus('error');
        setError('Offline. Check connection and retry.');
        return { success: false, error: 'offline', punch: null };
      }

      // Prevent double-submit from UI
      if (inflightKey.current) {
        return { success: false, error: 'already_in_flight', punch: null };
      }

      const idempotencyKey = `${type}-${Date.now()}-${crypto.randomUUID?.() || Math.random().toString(36).slice(2)}`;
      const timestamp = new Date().toISOString();

      const payload = {
        type,
        timestamp,
        latitude: geolocation?.latitude,
        longitude: geolocation?.longitude,
        idempotencyKey,
      };

      // Optimistic local state (prevents UI sluggishness)
      const optimistic: Punch = {
        id: `pending-${idempotencyKey}`,
        type,
        timestamp,
        latitude: geolocation?.latitude,
        longitude: geolocation?.longitude,
        source: 'MANUAL',
      };

      inflightKey.current = idempotencyKey;
      setStatus('pending');
      setError(null);
      setLastConfirmed(optimistic);

      // Enqueue for resilient submission
      pendingQueue.current.push({
        idempotencyKey,
        payload,
        retries: 0,
      });

      // Trigger processing (serialized)
      await processQueue();

      // Determine final outcome for caller
      const success = lastConfirmed?.id !== optimistic.id || status === 'success';
      return success
        ? { success: true, punch: lastConfirmed, error: null }
        : { success: false, punch: null, error };
    },
    [processQueue, status, error, lastConfirmed]
  );

  const reset = useCallback(() => {
    setStatus('idle');
    setError(null);
    if (abortRef.current) abortRef.current.abort();
    abortRef.current = null;
    inflightKey.current = null;
    pendingQueue.current = [];
  }, []);

  // Retry on reconnect for items stuck in queue
  useEffect(() => {
    const onOnline = () => {
      if (pendingQueue.current.length > 0 && !inflightKey.current) {
        processQueue();
      }
    };
    window.addEventListener('online', onOnline);
    return () => window.removeEventListener('online', onOnline);
  }, [processQueue]);

  return {
    status,
    lastConfirmed,
    error,
    punch,
    reset,
    isPending: status === 'pending',
    pendingCount: pendingQueue.current.length,
  };
}
```
