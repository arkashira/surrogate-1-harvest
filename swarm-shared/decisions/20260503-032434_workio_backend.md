// Signature verification
function verifySignature(body: string, signature: string): boolean {
  if (!CHANNEL_SECRET) {
    logger.warn('LINE_CHANNEL_SECRET not set; skipping signature verification');
    return true;
  }
  try {
    const expected = crypto
      .createHmac('sha256', CHANNEL_SECRET)
      .update(body, 'utf8')
      .digest('base64');
    return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
  } catch (err) {
    logger.error({ err }, 'Signature verification error');
    return false;
  }
}

const recentIds = new Map<string, number>(); // id -> expiryMs

function isDuplicate(id: string): boolean {
  const now = Date.now();
  const expiry = recentIds.get(id);
  if (expiry && now < expiry) return true;
  recentIds.set(id, now + IDEMPOTENCY_TTL_MS);
  return false;
}

function cleanupExpiredIds() {
  const now = Date.now();
  for (const [id, expiry] of recentIds.entries()) {
    if (now >= expiry) recentIds.delete(id);
  }
}

async function handleLineEvent(event: z.infer<typeof LineEventSchema>, tenantId: string) {
  const userId = event.source?.userId;
  if (!userId) {
    logger.warn({ eventType: event.type }, 'LINE event missing userId');
    return;
  }

  const user = await db('users')
    .where({ tenant_id: tenantId, line_user_id: userId })
    .first();

  if (!user) {
    logger.warn({ tenantId, lineUserId: userId }, 'User not found for tenant');
    return;
  }

  const text = (event.message?.text || event.postback?.data || '').toLowerCase().trim();
  if (!text) return;

  // Clock punch handling (example)
  if (text === 'in' || text === 'out') {
    const punchId = `line:${event.message?.id || event.timestamp || Date.now()}`;
    if (isDuplicate(punchId)) {
      logger.info({ tenantId, userId, punchId }, 'Duplicate punch ignored');
      return;
    }

    await db('attendance_punch').insert({
      tenant_id: tenantId,
      user_id: user.id,
      punch_type: text === 'in' ? 'clock_in' : 'clock_out',
      punched_at: event.timestamp ? new Date(event.timestamp) : new Date(),
      source: 'line',
      source_event_id: event.message?.id || null,
      created_at: new Date(),
    });

    logger.info({ tenantId, userId, punchType: text }, 'Punch recorded');
    return;
  }

  // TODO: extend for leave/OT requests with idempotency and validation
}
