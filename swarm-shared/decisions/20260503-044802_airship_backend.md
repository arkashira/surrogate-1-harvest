## Code Snippets

### `scripts/list_hf_files.py` (run on Mac)
```python
#!/usr/bin/env python3
"""
List HF dataset files for a date folder (non-recursive) and save to JSON.
Run once per folder after rate-limit window clears.
"""
import json
import os
import sys
from huggingface_hub import HfApi

REPO = os.getenv("HF_DATASET_REPO", "my-org/surrogate-mirror")
FOLDER = sys.argv[1] if len(sys.argv) > 1 else "batches/mirror-merged/2026-05-03"
OUT = sys.argv[2] if len(sys.argv) > 2 else "file_list.json"

api = HfApi()
# Non-recursive to avoid pagination explosion
tree = api.list_repo_tree(repo_id=REPO, path=FOLDER, recursive=False)
files = [item.rfilename for item in tree if item.type == "file"]

with open(OUT, "w") as f:
    json.dump({"repo": REPO, "folder": FOLDER, "files": files}, f, indent=2)

print(f"Saved {len(files)} files to {OUT}")
```

### `surrogate/train.py` (Lightning-compatible)
```python
import json
import os
from pathlib import Path
from typing import Dict, Any

import torch
from torch.utils.data import Dataset, DataLoader
from lightning.pytorch import LightningModule, Trainer
from lightning.pytorch.loggers import CSVLogger

HF_REPO = os.getenv("HF_DATASET_REPO", "my-org/surrogate-mirror")
FILE_LIST_PATH = os.getenv("FILE_LIST_PATH", "file_list.json")
CDN_BASE = "https://huggingface.co/datasets"

class CDNJsonlDataset(Dataset):
    """Load JSONL files from HF CDN (no API/auth)."""
    def __init__(self, file_list_path: str, max_files: int = -1):
        with open(file_list_path) as f:
            meta = json.load(f)
        self.repo = meta["repo"]
        self.files = meta["files"]
        if max_files > 0:
            self.files = self.files[:max_files]

    def __len__(self) -> int:
        return len(self.files)

    def _parse_line(self, line: str) -> Dict[str, Any]:
        # Project to {prompt, response} only; ignore extra fields
        try:
            obj = json.loads(line.strip())
            return {
                "prompt": str(obj.get("prompt", obj.get("input", ""))),
                "response": str(obj.get("response", obj.get("output", ""))),
            }
        except Exception:
            return {"prompt": "", "response": ""}

    def __getitem__(self, idx: int) -> Dict[str, Any]:
        import requests
        rel_path = self.files[idx]
        url = f"{CDN_BASE}/{self.repo}/resolve/main/{rel_path}"
        resp = requests.get(url, timeout=30)
        resp.raise_for_status()
        lines = resp.text.strip().split("\n")
        # Simple: take first valid line; extend as needed
        for line in lines:
            parsed = self._parse_line(line)
            if parsed["prompt"] and parsed["response"]:
                return parsed
        return {"prompt": "", "response": ""}

class SurrogateDataModule(LightningModule):
    def __init__(self, batch_size: int = 8, max_files: int = -1):
        super().__init__()
        self.batch_size = batch_size
        self.max_files = max_files

    def setup(self, stage=None):
        self.train_ds = CDNJsonlDataset(FILE_LIST_PATH, max_files=self.max_files)

    def train_dataloader(self):
        return DataLoader(self.train_ds, batch_size=self.batch_size, shuffle=True, num_workers=4)

class SurrogateModel(LightningModule):
    def __init__(self, lr: float = 1e-4):
        super().__init__()
        self.lr = lr
        # Minimal model for demo; replace with your surrogate architecture
        self.net = torch.nn.Linear(1024, 1024)

    def training_step(self, batch, batch_idx):
        # Dummy forward; replace with real tokenization + model forward
        loss = torch.tensor(0.0, requires_grad=True)
        self.log("train_loss", loss)
        return loss

    def configure_optimizers(self):
        return torch.optim.AdamW(self.parameters(), lr=self.lr)

def run_training():
    from lightning.pytorch import Trainer
    from lightning.pytorch.callbacks import ModelCheckpoint

    # Studio lifecycle resilience: reuse running studio, restart if idle-stopped
    try:
        from lightning.pytorch.studio import Studio, Machine, Teamspace
        studios = Teamspace.studios()
        studio = next((s for s in studios if s.name == "surrogate-train"), None)
        if studio and studio.status != "Running":
            studio.start(machine=Machine.L40S)
        elif not studio:
            studio = Studio.create(name="surrogate-train", machine=Machine.L40S, create_ok=True)
    except Exception as e:
        print(f"Studio setup skipped: {e}")
        studio = None

    dm = SurrogateDataModule(batch_size=4, max_files=100)
    model = SurrogateModel(lr=1e-4)

    trainer = Trainer(
        max_epochs=1,
        logger=CSVLogger("logs", name="surrogate"),
        callbacks=[ModelCheckpoint(monitor="train_loss")],
        accelerator="gpu" if torch.cuda.is_available() else "cpu",
        devices=1,
    )
    trainer.fit(model, dm)

    if studio:
        # Optional: keep alive or stop to manage quota
        pass

if __name__ == "__main__":
    run_training()
```
