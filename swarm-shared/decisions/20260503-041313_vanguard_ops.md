# vanguard / ops

## Final Synthesis (single, actionable plan)

**Core problem**: training and frontend hit HF API at runtime (429s, non-reproducible snapshots) because no content-addressed manifest or CDN-only data path exists, and ingestion produces mixed-schema files that break downstream expectations.

**Goal**: deterministic, rate-limit-safe, reproducible snapshots with strict `{prompt,response}` projection and CDN-only consumption.

---

## 1. Canonical manifest (content-addressed, CDN-only)

Create one generator that runs **after** mirror ingestion and produces a single JSON manifest per date.

- Location:  
  `/opt/axentx/vanguard/scripts/generate_manifest.py`  
  `/opt/axentx/vanguard/scripts/verify_manifest.py`  
  `/opt/axentx/vanguard/ops/generate_manifest.sh`

- Manifest schema (minimal, deterministic):
  ```json
  {
    "version": "1.0",
    "date": "YYYY-MM-DD",
    "dataset_repo": "datasets/axentx/vanguard-mirror",
    "commit_repo": "datasets/axentx/vanguard-mirror",
    "generated_by": "generate_manifest.py",
    "entries": [
      {
        "slug": "uuid-or-filename",
        "sha256": "...",
        "size": 12345,
        "cdn_url": "https://huggingface.co/datasets/.../resolve/main/...",
        "rel_path": "batches/mirror-merged/YYYY-MM-DD/file.parquet",
        "date": "YYYY-MM-DD"
      }
    ]
  }
  ```

- Behavior:
  - Non-recursive scan of `batches/mirror-merged/{date}/` for `.parquet` only.
  - Content-addressed by SHA256; idempotent.
  - Outputs to `manifests/{date}.json` (committed or pushed to HF sibling repo).
  - Training/frontend consume **only** `cdn_url` fields; zero HF API calls during data loading.

### `/opt/axentx/vanguard/ops/generate_manifest.sh`
```bash
#!/usr/bin/env bash
set -euo pipefail
SHELL=/bin/bash

REPO_ROOT="$(cd "$(dirname "$0")/../.." && pwd)"
SCRIPT="${REPO_ROOT}/scripts/generate_manifest.py"
MANIFEST_DIR="${REPO_ROOT}/manifests"

mkdir -p "${MANIFEST_DIR}"

DATE="${1:-$(date +%Y-%m-%d)}"
BATCH_DIR="/data/vanguard/batches/mirror-merged/${DATE}"

if [[ ! -d "${BATCH_DIR}" ]]; then
  echo "No batch dir: ${BATCH_DIR}" >&2
  exit 1
fi

python3 "${SCRIPT}" \
  --batch-dir "${BATCH_DIR}" \
  --date "${DATE}" \
  --out "${MANIFEST_DIR}/${DATE}.json"
```

### `/opt/axentx/vanguard/scripts/generate_manifest.py`
```python
#!/usr/bin/env python3
"""
Generate content-addressed manifest for a mirror-merged batch.
Produces CDN-only URLs and SHA256 hashes for deterministic, rate-limit-safe consumption.
"""
import argparse
import hashlib
import json
import os
import sys
from pathlib import Path
from typing import List, Dict

HF_DATASET_REPO = os.getenv("HF_DATASET_REPO", "datasets/axentx/vanguard-mirror")
HF_COMMIT_REPO = os.getenv("HF_COMMIT_REPO", HF_DATASET_REPO)

def sha256_file(path: Path) -> str:
    h = hashlib.sha256()
    with path.open("rb") as f:
        for chunk in iter(lambda: f.read(8192), b""):
            h.update(chunk)
    return h.hexdigest()

def cdn_url_for(repo: str, filepath: str) -> str:
    return f"https://huggingface.co/datasets/{repo}/resolve/main/{filepath}"

def collect_parquet_files(batch_dir: Path) -> List[Path]:
    files = sorted([p for p in batch_dir.iterdir() if p.is_file() and p.suffix == ".parquet"])
    if not files:
        print(f"No parquet files in {batch_dir}", file=sys.stderr)
    return files

def build_manifest(batch_dir: str, date: str, out_path: str) -> None:
    batch_path = Path(batch_dir).expanduser().resolve()
    out_path = Path(out_path).expanduser().resolve()
    out_path.parent.mkdir(parents=True, exist_ok=True)

    files = collect_parquet_files(batch_path)
    entries: List[Dict] = []

    for f in files:
        slug = f.stem
        rel_path = f"batches/mirror-merged/{date}/{f.name}"
        entries.append({
            "slug": slug,
            "sha256": sha256_file(f),
            "size": f.stat().st_size,
            "cdn_url": cdn_url_for(HF_DATASET_REPO, rel_path),
            "commit_repo": HF_COMMIT_REPO,
            "rel_path": rel_path,
            "date": date,
        })

    manifest = {
        "version": "1.0",
        "date": date,
        "dataset_repo": HF_DATASET_REPO,
        "commit_repo": HF_COMMIT_REPO,
        "generated_by": "generate_manifest.py",
        "entries": entries,
    }

    with out_path.open("w") as fp:
        json.dump(manifest, fp, indent=2, sort_keys=True)

    print(f"Manifest written: {out_path} ({len(entries)} entries)")

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Generate CDN-only content-addressed manifest.")
    parser.add_argument("--batch-dir", required=True, help="Path to mirror-merged/{date} folder")
    parser.add_argument("--date", required=True, help="Date (YYYY-MM-DD) for manifest filename")
    parser.add_argument("--out", required=True, help="Output manifest JSON path")
    args = parser.parse_args()
    build_manifest(args.batch_dir, args.date, args.out)
```

"""
Quick integrity check for a manifest: validate hashes (local) and CDN reachability (sample).
"""
import argparse
import hashlib
import json
import sys
import requests
from pathlib import Path

def verify_hashes(manifest_path: Path) -> bool:
    with manifest_path.open() as f:
        manifest = json.load(f)

    ok = True
    for e in manifest.get("entries", []):
        local_path = Path("/data/vanguard/batches/mirror-merged") / e["date"] / f"{e['slug']}.parquet"
        if local_path.is_file():
            h = hashlib.sha256()
            with local_path.open("rb") as fp:
                for chunk in iter(lambda: fp.read(8192), b""):
                    h.update(chunk)
            if h.hexdigest() != e["sha256"]:
                print(f"HASH MISMATCH: {e['slug']}", file=sys.stderr)
                ok = False
        else:
            print(f"SKIP (local absent): {e['slug']}")
    return ok

    with manifest_path.open() as f:
        manifest = json.load(f)

    entries = manifest.get("entries", [])
    if not entries:
        return True

                ok = False

            print(f"CDN error: {url} -> {ex}", file=sys.stderr)
            ok = False
    return ok
