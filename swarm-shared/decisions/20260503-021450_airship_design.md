def list_date_folder(date_folder: str, recursive: bool = False) -> List[str]:
    """
    Single API call to list one date folder (non-recursive).
    Save result to JSON for Lightning training script to embed.
    """
    items = list_repo_tree(
        repo_id=HF_REPO,
        path=date_folder,
        repo_type="dataset",
        recursive=recursive,
    )
    files = [it["path"] for it in items if it["type"] == "file"]
    out_path = Path("file_list.json")
    out_path.write_text(json.dumps({"date_folder": date_folder, "files": files}, indent=2))
    return files

def stream_cdn_file(file_path: str, chunk_size: int = 8192):
    """
    Stream file via CDN (no Authorization header) to bypass HF API rate limits.
    Yields raw bytes.

    url = f"{CDN_BASE}/{file_path}"
    with requests.get(url, stream=True, timeout=30) as r:
        r.raise_for_status()

### 2. Lightning Studio reuse + idle-stop guard (`surrogate/training/launcher.py`)

```python
# surrogate/training/launcher.py
import os
import time

from lightning import Lightning, Teamspace, Machine, Studio

LIGHTNING_ACCOUNT = os.getenv("LIGHTNING_ACCOUNT", "lightning-public-prod")
STUDIO_NAME = os.getenv("STUDIO_NAME", "surrogate-trainer")
MACHINE = Machine.L40S  # fallback; H200 only in lightning-lambda-prod

def get_or_create_studio() -> Studio:
    lightning = Lightning(account=LIGHTNING_ACCOUNT)
    teamspace = Teamspace()

    # Reuse running studio to save quota
    for s in teamspace.studios:
        if s.name == STUDIO_NAME and s.status == "Running":
            print(f"Reusing running studio: {s.name}")
            return s

    # Create if not exists
    studio = Studio(
        name=STUDIO_NAME,
        machine=MACHINE,
        create_ok=True,
    )
    return studio

def run_with_idle_guard(target_script: str, args: list[str] | None = None):
    """
    Guard against Lightning idle-stop killing training.
    If studio stopped, restart machine and re-run.
    """
    studio = get_or_create_studio()
    args = args or []

    max_retries = 3
    for attempt in range(1, max_retries + 1):
        if studio.status != "Running":
            print(f"Studio stopped (attempt {attempt}). Restarting...")
            studio.start(machine=MACHINE)
            time.sleep(30)  # allow startup

        try:
            run = studio.run(
                target=target_script,
                arguments=args,
                name=f"surrogate-train-{int(time.time())}",
            )
            print(f"Run started: {run.id}")
            return run
        except Exception as exc:
            print(f"Run failed (attempt {attempt}): {exc}")
            if attempt == max_retries:
                raise
            time.sleep(60)
            studio.start(machine=MACHINE)
```

### 3. Arkship idle-stop resilience middleware (`arkship/middleware/surrogate_client.py`)

# arkship/middleware/surrogate_client.py
import time
import logging
import requests
from typing import Optional

logger = logging.getLogger(__name__)

SURROGATE_BASE = "http://localhost:8001"
HEALTH_ENDPOINT = f"{SURROGATE_BASE}/health"
MAX_RETRIES = 3
BACKOFF = 5

def resilient_post(endpoint: str, payload: dict, timeout: int = 10) -> Optional[dict]:
    """
    POST to Surrogate with idle-stop resilience.
    If Surrogate is unreachable, retry with backoff.
    """
    url = f"{SURROGATE_BASE}{endpoint}"
    for attempt in range(1, MAX_RETRIES + 1):
        try:
            resp = requests.post(url, json=payload, timeout=timeout)
            resp.raise_for_status()
            return resp.json()
        except requests.RequestException as exc:
            logger.warning(f"Surrogate request failed (attempt {attempt}): {exc}")
            if attempt < MAX_RETRIES:
                time.sleep(BACKOFF * attempt)
                # optional: trigger studio restart via Lightning SDK here
            else:
                logger.error("Surrogate unavailable after retries")
                return None

### 4. Training script wired for CDN-only mode (`surrogate/training/train.py` snippet)

```python
# surrogate/training/train.py
import json
from pathlib import Path
from surrogate.training.ingest import stream_cdn_file, project_to_prompt_response

def load_dataset_from_cdn(file_list_path: str = "file_list.json"):
    file_list = json.loads(Path(file_list_path).read_text())
    for file_path in file_list["files"]:
        raw = b"".join(stream_cdn_file(file_path))
        try:
            example = project_to_prompt_response(raw, file_path)
            yield example
        except Exception as exc:
            print(f"Skip {file_path}: {exc}")
```

Ensure `list_date_folder()` is called once from Mac (after rate-limit window) to produce `file_list.json`. Lightning training uses only CDN URLs — zero HF API calls during dataload.
