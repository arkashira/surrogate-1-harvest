# airship / design

## Final Consolidated Implementation (≤2h)

**Goal**: Eliminate HF API 429 and Lightning quota waste during Surrogate training by implementing **CDN-first deterministic ingestion + Lightning Studio reuse/guard + Arkship UI idle-stop resilience**.

**Why this ships fastest**:
- No schema migrations or infra changes — pure orchestration + config + small code patches.
- Uses existing patterns (HF CDN bypass, Studio reuse, idle-stop guard) and applies them to the Surrogate training path.
- Delivers immediate reliability + cost savings for daily training runs.

### 1. `scripts/generate_file_list.py`
```python
#!/usr/bin/env python3
"""
Generate file_list.json for a date folder to enable CDN-only fetches.
Run from Mac (or any orchestrator) after HF API rate-limit window clears.
"""
import json
import os
import sys
from datetime import datetime

from huggingface_hub import HfApi

HF_REPO = os.getenv("HF_DATASET_REPO", "axentx/surrogate-mirror")
DATE_FOLDER = os.getenv("DATE_FOLDER", datetime.utcnow().strftime("%Y-%m-%d"))
OUTPUT = os.getenv("OUTPUT", "file_list.json")

def main() -> None:
    api = HfApi()
    # Single API call; recursive=False keeps it cheap
    tree = api.list_repo_tree(repo_id=HF_REPO, path=DATE_FOLDER, recursive=False)
    files = [
        {"path": f.path, "size": getattr(f, "size", None)}
        for f in tree
        if f.type == "file" and f.path.endswith((".jsonl", ".parquet", ".json"))
    ]
    payload = {
        "repo": HF_REPO,
        "date_folder": DATE_FOLDER,
        "generated_at": datetime.utcnow().isoformat() + "Z",
        "files": files,
    }
    with open(OUTPUT, "w", encoding="utf-8") as fp:
        json.dump(payload, fp, indent=2)
    print(f"Wrote {len(files)} files to {OUTPUT}")

if __name__ == "__main__":
    main()
```

### 2. `surrogate/training/train.py` (CDN-only dataloader excerpt)
```python
import json
import os
from pathlib import Path

import pyarrow.parquet as pq
import requests
from torch.utils.data import IterableDataset

HF_DATASET_REPO = os.getenv("HF_DATASET_REPO", "axentx/surrogate-mirror")
CDN_BASE = f"https://huggingface.co/datasets/{HF_DATASET_REPO}/resolve/main"

class CDNParquetDataset(IterableDataset):
    def __init__(self, file_list_path: str):
        with open(file_list_path) as f:
            manifest = json.load(f)
        self.files = [f["path"] for f in manifest["files"] if f["path"].endswith(".parquet")]

    def __iter__(self):
        for path in self.files:
            url = f"{CDN_BASE}/{path}"
            # CDN download: no Authorization header -> bypasses /api/ rate limits
            resp = requests.get(url, timeout=30)
            resp.raise_for_status()
            with open("/tmp/temp.parquet", "wb") as f:
                f.write(resp.content)
            table = pq.read_table("/tmp/temp.parquet", columns=["prompt", "response"])
            for row in table.to_pylist():
                # Project to {prompt, response} only at parse time
                yield row
```

### 3. `surrogate/training/lightning_utils.py`
```python
import time
from lightning import Lightning, Machine, Teamspace, Studio

LIGHTNING_ACCOUNT = os.getenv("LIGHTNING_ACCOUNT", "lightning-public-prod")

def get_or_create_studio(name: str, machine: Machine = Machine.L40S) -> Studio:
    ts = Teamspace(name=".")
    for s in ts.studios:
        if s.name == name and s.status == "Running":
            print(f"Reusing running studio: {name}")
            return s
    print(f"Creating studio: {name}")
    return Studio(
        name=name,
        machine=machine,
        cloud_account=LIGHTNING_ACCOUNT,
        create_ok=True,
    )

def ensure_running(studio: Studio, machine: Machine = Machine.L40S) -> Studio:
    if studio.status != "Running":
        print(f"Studio {studio.name} not running (status={studio.status}). Restarting...")
        studio.start(machine=machine)
        # Wait until running
        for _ in range(30):
            studio.refresh()
            if studio.status == "Running":
                break
            time.sleep(10)
        else:
            raise RuntimeError(f"Studio {studio.name} failed to start")
    return studio
```

### 4. `surrogate/training/ingest.py` (projection only)
```python
import pyarrow as pa
import pyarrow.parquet as pq
from pathlib import Path

def project_to_prompt_response(input_path: str, output_dir: str, date: str, slug: str) -> str:
    table = pq.read_table(input_path, columns=["prompt", "response"])
    out_path = Path(output_dir) / f"batches/mirror-merged/{date}/{slug}.parquet"
    out_path.parent.mkdir(parents=True, exist_ok=True)
    pq.write_table(table, out_path)
    return str(out_path)
```
