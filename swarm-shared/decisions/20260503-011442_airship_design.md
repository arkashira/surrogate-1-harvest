### 1) Training manifest generator (Node helper)
- **What:** One-time Mac-side script to list today’s parquet files and emit a deterministic manifest with CDN URLs.  
- **Why:** CDN URLs bypass `/api/` rate limits; manifest removes per-epoch HF API calls during training.  
- **Correctness fix:** Use `datasets/` repo path consistently and resolve CDN URLs with `resolve/main/` so files are fetched without auth headers.

**File:** `scripts/generate-training-manifest.js`
```js
#!/usr/bin/env node
// scripts/generate-training-manifest.js
// Run: HF_TOKEN=... node scripts/generate-training-manifest.js --repo=datasets/your/repo --date=2026-04-29 --out=manifests/2026-04-29.json

import { program } from 'commander';
import { HfApi } from '@huggingface/hub';
import fs from 'fs';
import path from 'path';

program
  .requiredOption('--repo <repo>', 'HF dataset repo (e.g., datasets/your/repo)')
  .requiredOption('--date <date>', 'Date folder (e.g., 2026-04-29)')
  .option('--out <out>', 'Output JSON path', 'manifests/latest.json')
  .parse();

const opts = program.opts();
const api = new HfApi({ token: process.env.HF_TOKEN });

async function run() {
  const base = opts.repo.replace(/^datasets\//, '');
  const folder = `batches/mirror-merged/${opts.date}/`;
  console.log(`Listing ${folder} (non-recursive)...`);
  const tree = await api.listRepoTree({
    repo: base,
    path: folder,
    recursive: false,
  });

  const files = (tree.files || [])
    .filter((f) => f.path.endsWith('.parquet'))
    .map((f) => ({
      filename: path.basename(f.path),
      cdn_url: `https://huggingface.co/datasets/${base}/resolve/main/${f.path}`,
      size: f.size,
    }));

  const manifest = {
    repo: base,
    date: opts.date,
    generated_at: new Date().toISOString(),
    files,
  };

  fs.mkdirSync(path.dirname(opts.out), { recursive: true });
  fs.writeFileSync(opts.out, JSON.stringify(manifest, null, 2));
  console.log(`Wrote ${files.length} files to ${opts.out}`);
}

run().catch((err) => {
  console.error(err);
  process.exit(1);
});
```
Make executable:
```bash
chmod +x scripts/generate-training-manifest.js
```

### 2) Training script: CDN-only + schema projection
- **What:** Replace `load_dataset(streaming=True)` with a `Dataset` that streams parquet via CDN URLs from the manifest and projects to `{prompt, response}` at parse time.  
- **Why:** Avoids mixed-schema `pyarrow.CastError` and HF API 429s during data loads.  
- **Correctness fixes:**
  - Import `pyarrow as pa` (was missing).  
  - Use robust column detection and a safe per-row decoder.  
  - Keep batchable `__getitem__` (return one row) and handle decoding errors gracefully.

**File:** `training/train.py`
```python

import json
import os
import hashlib
from pathlib import Path

import pyarrow as pa
import pyarrow.parquet as pq
import requests
import torch
from torch.utils.data import Dataset, DataLoader

HF_DATASET_REPO = os.getenv("HF_DATASET_REPO", "datasets/your/repo")
MANIFEST_PATH = os.getenv("MANIFEST_PATH", "manifests/latest.json")
SIBLING_REPOS = [
    "datasets/your/repo",
    "datasets/your/repo-sib1",
    "datasets/your/repo-sib2",
    "datasets/your/repo-sib3",
    "datasets/your/repo-sib4",
    "datasets/your/repo-sib5",
]

def pick_sibling_repo(slug: str) -> str:
    """Deterministic sibling repo selection to spread HF commit cap."""
    h = int(hashlib.sha256(slug.encode()).hexdigest(), 16)
    return SIBLING_REPOS[h % len(SIBLING_REPOS)]

class CDNParquetDataset(Dataset):
    def __init__(self, manifest_path):
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.items = [entry["cdn_url"] for entry in self.manifest["files"]]

    def __len__(self):
        return len(self.items)

    def _decode_or_fallback(self, col):
        # Prefer string; fallback to utf8 conversion or repr
        if pa.types.is_string(col.type) or pa.types.is_large_string(col.type):
            return col
        try:
            return pc.cast(col, pa.string())
        except Exception:
            # Last resort: convert to Python objects and then to strings
            return pa.array([str(v) for v in col.to_pylist()])

    def __getitem__(self, idx):
        url = self.items[idx]
        resp = requests.get(url, timeout=60)
        resp.raise_for_status()
        table = pq.read_table(pq.ParquetFile(pq.ParquetReader(pq.BufferReader(resp.content))))

        cols = set(table.column_names)
        prompt_col = next((c for c in ["prompt", "Prompt", "input"] if c in cols), None)
        response_col = next((c for c in ["response", "Response", "output"] if c in cols), None)

        if prompt_col is None or response_col is None:
            text_cols = [c for c in cols if table.schema.field(c).type in (pa.string(), pa.large_string())]
            if len(text_cols) >= 2:
                prompt_col, response_col = text_cols[0], text_cols[1]
            else:
                raise ValueError(f"Cannot find prompt/response columns in {url}")

        # Project only needed columns and decode safely
        prompts = self._decode_or_fallback(table.column(prompt_col))
        responses = self._decode_or_fallback(table.column(response_col))

        # Return one row to keep __getitem__ batchable
        return {
            "prompt": prompts[0].as_py(),
            "response": responses[0].as_py(),
        }
```
