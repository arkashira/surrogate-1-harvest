# surrogate-1 / design

## Implementation Plan (≤2h)

**Highest-value improvement**: Add deterministic pre-flight file listing + CDN-only ingestion to eliminate HF API 429s during training and make shard workers resilient.

### What we’ll change
1. Add `scripts/list_files.py` — single Mac-side call to `list_repo_tree` per date folder, outputs `file_list.json` (path, size, sha256).
2. Update `bin/dataset-enrich.sh` to accept an optional `FILE_LIST` path; if provided, workers iterate the local list and download via raw CDN URLs (no `datasets.load_dataset`).
3. Add `lib/cdn_loader.py` — streaming CDN fetch with backoff, projects to `{prompt, response}` only, produces normalized JSONL rows.
4. Add `lib/dedup.py` enhancement — keep deterministic shard assignment by `hash(slug) % 16` so retries are idempotent.
5. Update workflow to generate `file_list.json` once (in a prior step or on runner 0) and pass to all shards via artifact.

### Why this wins
- Zero HF API calls during data streaming → immune to 429 while training/ingesting.
- CDN limits are much higher; listing is one small API call.
- Deterministic shard + idempotent retry prevents duplicates across reruns.
- Fits <2h: small focused scripts + minimal workflow change.

### 1) scripts/list_files.py
```python
#!/usr/bin/env python3
"""
Generate deterministic file list for a date folder in axentx/surrogate-1-training-pairs.
Usage:
  python scripts/list_files.py --date 2026-05-02 --out file_list.json
"""
import argparse
import json
import os
import time
from huggingface_hub import HfApi

REPO = "datasets/axentx/surrogate-1-training-pairs"

def list_date_folder(date: str, token: str | None = None):
    api = HfApi(token=token or os.getenv("HF_TOKEN"))
    prefix = f"batches/public-merged/{date}/"
    # non-recursive per pattern: list_repo_tree(path, recursive=False) per folder
    items = api.list_repo_tree(repo_id=REPO, path=prefix, recursive=False, repo_type="dataset")
    files = []
    for item in items:
        if item.type != "file":
            continue
        # prefer sha256 if available
        files.append({
            "path": item.path,
            "size": getattr(item, "size", None),
            "lfs": getattr(item, "lfs", None),
        })
    # sort for determinism
    files.sort(key=lambda x: x["path"])
    return files

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--date", required=True, help="Date folder, e.g. 2026-05-02")
    parser.add_argument("--out", default="file_list.json", help="Output JSON path")
    parser.add_argument("--token", default=None, help="HF token (env HF_TOKEN preferred)")
    args = parser.parse_args()

    token = args.token or os.getenv("HF_TOKEN")
    if not token:
        print("WARNING: HF_TOKEN not set; listing may fail for private repos.")

    for attempt in range(3):
        try:
            files = list_date_folder(args.date, token=token)
            break
        except Exception as exc:
            wait = 360 if "429" in str(exc) else (2 ** attempt)
            print(f"Attempt {attempt+1} failed: {exc}. Retry in {wait}s")
            time.sleep(wait)
    else:
        raise RuntimeError("Failed to list repo after retries")

    out = {"date": args.date, "prefix": f"batches/public-merged/{args.date}/", "files": files}
    with open(args.out, "w", encoding="utf-8") as f:
        json.dump(out, f, indent=2)
    print(f"Wrote {len(files)} entries to {args.out}")

if __name__ == "__main__":
    main()
```

### 2) lib/cdn_loader.py
```python
import json
import time
import urllib.request
from typing import Iterator, Dict, Any
from hashlib import md5

CDN_TEMPLATE = "https://huggingface.co/datasets/axentx/surrogate-1-training-pairs/resolve/main/{path}"

def robust_cdn_fetch(path: str, timeout: int = 30, max_retries: int =5) -> bytes:
    url = CDN_TEMPLATE.format(path=path)
    for attempt in range(max_retries):
        try:
            req = urllib.request.Request(url, headers={"User-Agent": "axentx-surrogate-1/1.0"})
            with urllib.request.urlopen(req, timeout=timeout) as resp:
                return resp.read()
        except Exception as exc:
            wait = 2 ** attempt
            if attempt == max_retries - 1:
                raise RuntimeError(f"CDN fetch failed for {path}: {exc}") from exc
            time.sleep(wait)
    raise RuntimeError(f"CDN fetch exhausted for {path}")

def normalize_record(raw: Dict[str, Any], source_path: str) -> Dict[str, Any]:
    """
    Project heterogeneous schemas to {prompt, response, md5}.
    Keep attribution via filename pattern only (batches/.../{slug}.parquet).
    """
    # Common projections
    prompt = raw.get("prompt") or raw.get("input") or raw.get("text") or raw.get("question")
    response = raw.get("response") or raw.get("output") or raw.get("answer") or raw.get("completion")

    if prompt is None or response is None:
        # fallback: try to find any string fields
        strs = [v for v in raw.values() if isinstance(v, str) and len(v) > 10]
        if len(strs) >= 2:
            prompt, response = strs[0], strs[1]
        else:
            raise ValueError(f"Cannot project prompt/response from {source_path}: {raw}")

    text = f"{prompt}\n\n{response}"
    row = {
        "prompt": str(prompt).strip(),
        "response": str(response).strip(),
        "md5": md5(text.encode("utf-8")).hexdigest(),
    }
    return row

def stream_cdn_jsonl(file_list: list[dict], shard_id: int, total_shards: int = 16) -> Iterator[Dict[str, Any]]:
    """
    Deterministic shard assignment by path hash to ensure idempotency.
    """
    for entry in file_list:
        path = entry["path"]
        if hash(path) % total_shards != shard_id:
            continue
        try:
            payload = json.loads(robust_cdn_fetch(path))
            # If file is a JSONL list or single object
            if isinstance(payload, list):
                items = payload
            else:
                items = [payload]
            for item in items:
                yield normalize_record(item, path)
        except Exception as exc:
            # Log and skip; don't kill whole shard
            print(f"SKIP {path}: {exc}")
            continue
```
