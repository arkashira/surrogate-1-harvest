# surrogate-1 / ops

## Implementation Plan (≤2h)

**Highest-value improvement**: Replace fragile shell-based ingestion with a **manifest-driven, CDN-bypass Python worker** that eliminates HF API rate limits during training data loads and prevents mixed-schema CastErrors.

### Changes

1. **Add `bin/generate_manifest.py`**  
   - Single API call from Mac (after rate-limit window) to `list_repo_tree(path, recursive=False)` for one date folder.  
   - Save list to `manifests/<date>/files.json`.  
   - Embed this manifest in training scripts so Lightning workers do **CDN-only fetches** with zero API calls during data load.

2. **Add `bin/cdn_worker.py`**  
   - Deterministic shard assignment via `slug_hash % 16 == SHARD_ID`.  
   - Download files directly from CDN (`https://huggingface.co/datasets/.../resolve/main/...`) with no Authorization header.  
   - Project to `{prompt, response}` only at parse time; skip mixed-schema columns.  
   - Output `batches/public-merged/<date>/shard<N>-<HHMMSS>.jsonl`.

3. **Update `bin/dataset-enrich.sh`**  
   - Thin wrapper that calls `cdn_worker.py` with `SHARD_ID` and manifest path.  
   - Keep executable + Bash shebang for GitHub Actions compatibility.

4. **Update `.github/workflows/ingest.yml`**  
   - Pass `SHARD_ID` matrix variable.  
   - Ensure `SHELL=/bin/bash` in job defaults.

5. **Add `requirements-dev.txt`**  
   - `requests`, `pyarrow`, `tqdm`, `python-slugify`.

### `bin/generate_manifest.py`
```python
#!/usr/bin/env python3
"""
Generate a manifest for a given date folder in surrogate-1-training-pairs.
Run from Mac (or any env with HF token) once per date folder.
Usage:
    HF_TOKEN=hf_xxx python bin/generate_manifest.py --date 2026-05-03
"""
import argparse
import json
import os
from datetime import datetime
from pathlib import Path

from huggingface_hub import HfApi

API = HfApi()
REPO = "datasets/axentx/surrogate-1-training-pairs"

def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("--date", required=True, help="YYYY-MM-DD")
    args = parser.parse_args()

    folder = f"public-raw/{args.date}"
    out_dir = Path("manifests") / args.date
    out_dir.mkdir(parents=True, exist_ok=True)
    out_path = out_dir / "files.json"

    # Single API call: non-recursive per folder to avoid pagination/rate-limit
    items = API.list_repo_tree(repo_id=REPO, path=folder, recursive=False)
    files = [it.r_path for it in items if it.type == "file"]

    manifest = {
        "date": args.date,
        "generated_at": datetime.utcnow().isoformat() + "Z",
        "folder": folder,
        "files": sorted(files),
    }

    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"Wrote {len(files)} files -> {out_path}")

if __name__ == "__main__":
    main()
```

### `bin/cdn_worker.py`
```python
#!/usr/bin/env python3
"""
CDN-bypass worker for surrogate-1 ingestion.
- Uses manifest to avoid HF API during training data loads.
- Projects to {prompt, response} only.
- Deterministic sharding by slug hash.
Usage (in GitHub Actions):
    python bin/cdn_worker.py --manifest manifests/2026-05-03/files.json --shard 3
"""
import argparse
import hashlib
import json
import sys
from pathlib import Path
from typing import Any, Dict, List

import pyarrow as pa
import pyarrow.parquet as pq
import requests
from tqdm import tqdm

CDN_ROOT = "https://huggingface.co/datasets/axentx/surrogate-1-training-pairs/resolve/main"
OUTPUT_ROOT = Path("batches/public-merged")

def slug_hash(slug: str) -> int:
    return int(hashlib.sha256(slug.encode()).hexdigest(), 16)

def parse_file_cdn(path: str) -> List[Dict[str, str]]:
    """Download via CDN and project to {prompt, response}."""
    url = f"{CDN_ROOT}/{path}"
    resp = requests.get(url, timeout=60)
    resp.raise_for_status()

    # Try parquet first (common), fallback to jsonl
    if path.endswith(".parquet"):
        table = pq.read_table(pa.BufferReader(resp.content))
        # Project only; ignore mixed-schema extras
        rows = []
        for i in range(table.num_rows):
            row = {}
            # Best-effort column names
            for col in ("prompt", "response"):
                if col in table.column_names:
                    val = table[col][i].as_py()
                    row[col] = "" if val is None else str(val)
            if row.get("prompt") and row.get("response"):
                rows.append(row)
        return rows

    # Assume JSONL
    rows = []
    for line in resp.text.splitlines():
        line = line.strip()
        if not line:
            continue
        try:
            obj = json.loads(line)
        except json.JSONDecodeError:
            continue
        row = {k: obj.get(k, "") for k in ("prompt", "response")}
        if row.get("prompt") and row.get("response"):
            rows.append(row)
    return rows

def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("--manifest", required=True)
    parser.add_argument("--shard", type=int, required=True, choices=range(16))
    args = parser.parse_args()

    manifest = json.loads(Path(args.manifest).read_text())
    date = manifest["date"]
    files = manifest["files"]

    # Deterministic shard assignment
    my_files = [f for f in files if slug_hash(f) % 16 == args.shard]
    print(f"Shard {args.shard}: processing {len(my_files)}/{len(files)} files")

    out_dir = OUTPUT_ROOT / date
    out_dir.mkdir(parents=True, exist_ok=True)
    ts = datetime.utcnow().strftime("%H%M%S")
    out_path = out_dir / f"shard{args.shard}-{ts}.jsonl"

    with out_path.open("w") as fout:
        for path in tqdm(my_files, desc="CDN fetch"):
            try:
                rows = parse_file_cdn(path)
                for row in rows:
                    fout.write(json.dumps(row, ensure_ascii=False) + "\n")
            except Exception as exc:
                print(f"SKIP {path}: {exc}", file=sys.stderr)

    print(f"Wrote {out_path}")

if __name__ == "__main__":
    main()
```

### `bin/dataset-enrich.sh`
```bash
#!/usr/bin/env bash
# Thin wrapper for GitHub Actions.
# Usage: SHARD_ID=3 ./bin/dataset-enrich.sh manifests/2026-05-03/files.json
set -euo pipefail

MANIFEST="${1:?missing manifest path}"
SHARD_ID="${SHARD_ID:?missing SHARD_ID}"

export SHELL=/bin/bash
cd "$(dirname "$0")/.."

python bin/cdn_worker.py --manifest "$MANIFEST" --shard "$SHARD_ID"
```
