### 1) Create the manifest generator

`/opt/axentx/vanguard/ops/training_manifest.py`

```python
#!/usr/bin/env python3
"""
Generate and persist a deterministic file manifest for a given
(repo, date_folder) so training can use CDN-only fetches and avoid
authenticated HF API calls during data loading.

Usage:
    python training_manifest.py \
        --repo datasets/axentx/surrogate-mirror \
        --date 2026-04-29 \
        --out manifest-2026-04-29.json
"""
import argparse
import json
import os
import sys
import time
from pathlib import Path
from typing import List, Dict

try:
    from huggingface_hub import list_repo_tree
except ImportError:
    print("Install: pip install huggingface_hub")
    sys.exit(1)

HF_API_RETRY_WAIT = 360  # seconds after 429


def robust_list_tree(repo: str, path: str = "", max_retries: int = 3) -> List[Dict]:
    """List non-recursive tree with retry on rate-limit."""
    for attempt in range(1, max_retries + 1):
        try:
            tree = list_repo_tree(repo=repo, path=path, recursive=False)
            return [{"path": str(item.path)} for item in tree if hasattr(item, "path")]
        except Exception as exc:
            if "429" in str(exc) or "rate limit" in str(exc).lower():
                if attempt == max_retries:
                    raise
                print(f"Rate-limited (attempt {attempt}/{max_retries}). Waiting {HF_API_RETRY_WAIT}s...")
                time.sleep(HF_API_RETRY_WAIT)
                continue
            raise


def build_manifest(repo: str, date_folder: str, out_path: Path) -> Path:
    """
    Build manifest for repo/date_folder/*.parquet (or all files).
    Persists JSON list of objects:
      {"repo": "...", "file": "date/file.parquet", "cdn_url": "..."}
    """
    prefix = date_folder.rstrip("/") + "/"
    items = robust_list_tree(repo=repo, path=date_folder)

    # Keep only files (heuristically: contain '.' and not ending with '/')
    files = [i for i in items if "." in Path(i["path"]).name and not i["path"].endswith("/")]
    if not files:
        # fallback: include everything under prefix from non-recursive listing
        files = [i for i in items if i["path"].startswith(prefix)]

    manifest = []
    for item in files:
        fpath = item["path"]
        manifest.append(
            {
                "repo": repo,
                "file": fpath,
                "cdn_url": f"https://huggingface.co/datasets/{repo}/resolve/main/{fpath}",
            }
        )

    out_path.parent.mkdir(parents=True, exist_ok=True)
    with open(out_path, "w") as f:
        json.dump(manifest, f, indent=2)

    print(f"Wrote {len(manifest)} entries to {out_path}")
    return out_path


def main() -> None:
    parser = argparse.ArgumentParser(description="Generate training file manifest (CDN-only).")
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g., datasets/axentx/surrogate-mirror)")
    parser.add_argument("--date", required=True, help="Date folder (e.g., 2026-04-29)")
    parser.add_argument("--out", default=None, help="Output JSON path (default: manifest-{date}.json)")
    args = parser.parse_args()

    out = Path(args.out) if args.out else Path.cwd() / f"manifest-{args.date}.json"
    build_manifest(repo=args.repo, date_folder=args.date, out_path=out)


if __name__ == "__main__":
    main()
```

```python
#!/usr/bin/env python3
"""
Idempotent Lightning Studio launcher for Surrogate-1 training.
- Reuses running Studio when present.
- Uses manifest (CDN-only) for data loading.
- Restarts studio if stopped before run.
"""
import json
import subprocess
import sys
from pathlib import Path

try:
    from lightning import LightningWork, LightningApp, Machine
    from lightning.app import Teamspace
except ImportError:
    print("Install: pip install lightning")
    sys.exit(1)

BASE_DIR = Path(__file__).parent
MANIFEST_PATH = BASE_DIR / "manifest-latest.json"
TRAIN_SCRIPT = BASE_DIR.parent / "train.py"


def ensure_manifest_exists(repo: str, date_folder: str) -> Path:
    """Generate manifest once if missing."""
    if MANIFEST_PATH.exists():
        return MANIFEST_PATH

    manifest_script = BASE_DIR / "training_manifest.py"
    subprocess.run(
        [
            sys.executable,
            str(manifest_script),
            "--repo",
            repo,
            "--date",
            date_folder,
            "--out",
            str(MANIFEST_PATH),
        ],
        check=True,
    )
    return MANIFEST_PATH


class SurrogateTrainer(LightningWork):
    def __init__(self, repo: str, date_folder: str, machine_type: str = "L40S", **kwargs):
        super().__init__(**kwargs)
        self.repo = repo
        self.date_folder = date_folder
        self.machine_type = machine_type

    def run(self) -> None:
        # Ensure manifest exists before training (single API call)
        manifest = ensure_manifest_exists(self.repo, self.date_folder)
        with open(manifest) as f:
            file_list = json.load(f)
        print(f"Loaded {len(file_list)} files from manifest")

        # Launch training with manifest path injected (train.py should read CDN URLs only)
        cmd = [
            sys.executable,
            str(TRAIN_SCRIPT),
            "--manifest",
            str(manifest),
        ]
        subprocess.run(cmd, check=True)


def get_or_create_studio(repo: str, date_folder: str, name: str = "surrogate-trainer") -> SurrogateTrainer:
    """Idempotent studio lookup/create."""
    teamspace = Teamspace()
    for studio in teamspace.studios:
        if studio.name == name and studio.status == "running":
            print(f"Reusing running studio: {name}")
            return studio

    print(f"Creating studio: {name}")
    return SurrogateTrainer(
        repo=repo,
        date_folder=date_folder,
        name=name,
        cloud_compute=Machine.L40S,
    )


def main() -> None:
    repo = "datasets/axentx/surrogate-mirror"
    date_folder = "2026-04-29"
    studio = get_or_create_studio(repo, date_folder)
    app = LightningApp(studio)


if __name__ == "__main__":
    main()
```
