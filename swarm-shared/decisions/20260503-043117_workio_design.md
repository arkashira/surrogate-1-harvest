// Middleware: verify X-Line-Signature (timing-safe)
function verifyLineSignature(req: Request, res: Response, next: NextFunction) {
  const signature = req.get('X-Line-Signature');
  if (!signature) {
    log.warn('Missing X-Line-Signature');
    return res.status(401).json({ error: 'Missing signature' });
  }

  // Use raw body for signature verification (configure body-parser to keep raw)
  const rawBody = (req as any).rawBody || JSON.stringify(req.body || {});
  const expected = crypto
    .createHmac('sha256', CHANNEL_SECRET)
    .update(rawBody)
    .digest('base64');

  if (!crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected))) {
    log.warn('Invalid LINE signature', { received: signature });
    return res.status(401).json({ error: 'Invalid signature' });
  }
  next();
}

    event?.message?.id || event?.source?.groupId || ''

  const eventType = event?.type;

     ON CONFLICT (webhook_event_id) DO UPDATE
     SET attempts = line_webhook_idempotency.attempts + 1,
