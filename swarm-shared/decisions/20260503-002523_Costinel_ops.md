# Costinel / ops

## Final Implementation Plan — Costinel “Top-Hub Signal” Card (read-only)

**Scope**: Frontend-only card that identifies the most-connected hub from the knowledge-rag graph and surfaces the top hub with 3 contextual signals.  
**Timebox**: <2h  
**Constraints**: Zero backend changes; no mutations; uses existing data (static JSON or existing graph endpoint).  
**Risk**: None (pure read-only UI).

---

### 1) Data strategy (choose one, default to static)
- **Preferred static approach** (fastest, zero infra):  
  Use `public/data/top-hub.json` generated by ops/knowledge-rag and committed or copied during CI/CD.  
- **Alternative live approach** (if graph endpoint already exists):  
  Parent dashboard fetches `/api/knowledge-rag/graph` and passes `graphData` prop; card computes degrees client-side.  
- **Card accepts either** `hubPayload` (precomputed JSON) **or** `graphData` (raw graph) to stay flexible.

import { useEffect, useState, useMemo } from "react";
import { TrendingUp, AlertCircle, CheckCircle, Info } from "lucide-react";

const SEVERITY_ICON = {
  high: AlertCircle,
  medium: Info,
  low: CheckCircle,
} as const;

const SEVERITY_COLOR = {
  high: "text-red-500",
  medium: "text-amber-500",
  low: "text-emerald-500",
} as const;

const SEVERITY_BG = {
  high: "bg-red-50",
  medium: "bg-amber-50",
  low: "bg-emerald-50",
} as const;

type Signal = {
  id: string;
  title: string;
  severity: keyof typeof SEVERITY_ICON;
  snippet: string;
};

type HubPayload = {
  hub: string;
  label: string;
  connections: number;
  rank: number;
  signals: Signal[];
  updatedAt: string;
};

type GraphData = {
  nodes: Array<{ id: string; label?: string; type?: string }>;
  edges: Array<{ source: string; target: string }>;
};

type TopHubSignalCardProps = {
  hubPayload?: HubPayload;
  graphData?: GraphData;
  preferPayload?: boolean;
};

export default function TopHubSignalCard({
  hubPayload,
  graphData,
  preferPayload = true,
}: TopHubSignalCardProps) {
  const [payload, setPayload] = useState<HubPayload | null>(null);
  const [loading, setLoading] = useState(true);

  // Compute top hub from graph if no payload provided and graphData available
  const computedHub = useMemo(() => {
    if (!graphData?.nodes?.length || !graphData?.edges?.length) return null;

    const degree: Record<string, number> = {};
    const labels: Record<string, string> = {};
    graphData.nodes.forEach((n) => {
      degree[n.id] = 0;
      if (n.label) labels[n.id] = n.label;
    });
    graphData.edges.forEach((e) => {
      degree[e.source] = (degree[e.source] || 0) + 1;
      degree[e.target] = (degree[e.target] || 0) + 1;
    });

    const topId = Object.keys(degree).reduce((a, b) => (degree[a] > degree[b] ? a : b));
    const neighbors = graphData.edges
      .filter((e) => e.source === topId || e.target === topId)
      .map((e) => (e.source === topId ? e.target : e.source));

    const related = graphData.nodes
      .filter((n) => neighbors.includes(n.id) && n.type === "doc")
      .slice(0, 3)
      .map((n, i) => ({
        id: `doc-${i}`,
        title: n.label || "Untitled",
        severity: "low" as const,
        snippet: `Connected document referenced by ${labels[topId] || topId}.`,
      }));

    return {
      hub: topId,
      label: labels[topId] || topId,
      connections: degree[topId] || 0,
      rank: 1,
      signals: related.length
        ? related
        : [
            {
              id: "fallback",
              title: "No related docs found",
              severity: "low" as const,
              snippet: "Consider enriching graph metadata for richer insights.",
            },
          ],
      updatedAt: new Date().toISOString(),
    } satisfies HubPayload;
  }, [graphData]);

  useEffect(() => {
    setLoading(true);

    // If payload provided, use it
    if (preferPayload && hubPayload) {
      setPayload(hubPayload);
      setLoading(false);
      return;
    }

    // If graphData provided and payload not preferred, use computed
    if (!preferPayload && graphData) {
      setPayload(computedHub);
      setLoading(false);
      return;
    }

    // Default: try to fetch static payload
    fetch("/data/top-hub.json", { cache: "no-store" })
      .then((r) => {
        if (!r.ok) throw new Error("No payload");
        return r.json();
      })
      .then((data) => setPayload(data))
      .catch(() => {
        // fallback to computed if available
        if (graphData && computedHub) setPayload(computedHub);
        else setPayload(null);
      })
      .finally(() => setLoading(false));
  }, [hubPayload, graphData, preferPayload, computedHub]);

  if (loading) {
    return (
      <div className="animate-pulse rounded-xl border border-gray-200 bg-white p-6 shadow-sm">
        <div className="flex items-start justify-between mb-4">
          <div>
            <div className="h-5 w-20 bg-gray-200 rounded mb-2"></div>
            <div className="h-8 w-32 bg-gray-200 rounded mb-1"></div>
            <div className="h-4 w-24 bg-gray-200 rounded"></div>
          </div>
          <div className="h-6 w-20 bg-gray-200 rounded"></div>
        </div>
        <div className="space-y-3">
          {[...Array(3)].map((_, i) => (
            <div key={i} className="h-16 bg-gray-100 rounded"></div>
          ))
