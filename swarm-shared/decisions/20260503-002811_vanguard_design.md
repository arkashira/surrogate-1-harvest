# vanguard / design

## Final synthesized implementation (best parts, contradictions resolved)

**Core principles**
- One authenticated HF API call per `(repo, dateFolder)` to produce a persisted manifest.
- Training never calls HF API during data loading; only CDN URLs are used.
- Lightning Studio reuse is explicit, robust, and never blocks training.
- Deterministic repo selection is used to spread load and respect HF commit caps.
- Manifest is versioned and embedded-friendly; loader is schema-strict and column-projection–first.

---

### 1. Manifest module (`vanguard/design/manifest.py`)
- Uses `huggingface_hub` for listing; falls back to `requests` if unavailable.
- Validates entries and rejects non-files.
- Produces deterministic repo index for commit-cap spreading.
- CLI included; outputs stable JSON.

```python
from __future__ import annotations

import argparse
import hashlib
import json
from pathlib import Path
from typing import Any, Dict, List, Optional

try:
    from huggingface_hub import list_repo_tree
    HF_AVAILABLE = True
except Exception:
    HF_AVAILABLE = False

import requests


def deterministic_repo_index(slug: str, n: int = 5) -> int:
    """Stable mapping from slug -> [0, n) for commit-cap spreading."""
    digest = hashlib.sha256(slug.encode()).digest()
    return int.from_bytes(digest, "big") % max(1, n)


def _list_via_api(repo: str, folder_path: str) -> List[Dict[str, Any]]:
    if not HF_AVAILABLE:
        raise RuntimeError("huggingface_hub not available and no fallback implemented for repo listing")
    tree = list_repo_tree(repo, path=folder_path, recursive=False)
    files = []
    for entry in tree:
        if getattr(entry, "type", None) != "file":
            continue
        files.append(
            {
                "filename": Path(entry.path).name,
                "repo": repo,
                "path": entry.path,
                "size": getattr(entry, "size", None),
                "lfs": getattr(entry, "lfs", None),
            }
        )
    return files


def _list_via_web(repo: str, folder_path: str) -> List[Dict[str, Any]]:
    """Fallback: scrape repo tree page (best-effort). Prefer API."""
    base = f"https://huggingface.co/datasets/{repo}/tree/main"
    url = f"{base}/{folder_path}" if folder_path else base
    resp = requests.get(url, timeout=30)
    resp.raise_for_status()
    # Minimal scraping: this is a fallback only.
    # For production, require huggingface_hub.
    raise RuntimeError("Web fallback not implemented; install huggingface_hub")


def _cdn_url(repo: str, path: str) -> str:
    return f"https://huggingface.co/datasets/{repo}/resolve/main/{path}"


def build_manifest(repo: str, date_folder: str, out_path: Path, use_api: bool = True) -> Dict[str, Any]:
    folder_path = date_folder.rstrip("/")
    if use_api:
        files = _list_via_api(repo, folder_path)
    else:
        files = _list_via_web(repo, folder_path)

    if not files:
        raise ValueError(f"No files found in {repo}@{folder_path}")

    for f in files:
        f["cdn_url"] = _cdn_url(repo, f["path"])

    manifest = {
        "repo": repo,
        "date_folder": date_folder,
        "created_by": "manifest.py",
        "count": len(files),
        "files": files,
    }

    out_path.parent.mkdir(parents=True, exist_ok=True)
    out_path.write_text(json.dumps(manifest, indent=2))
    return manifest


def load_manifest(manifest_path: Path) -> Dict[str, Any]:
    return json.loads(manifest_path.read_text())


if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g., datasets/company/surrogate-1)")
    parser.add_argument("--date-folder", required=True, help="Folder in repo (e.g., 2026-04-29)")
    parser.add_argument("--out", required=True, help="Output manifest JSON path")
    parser.add_argument("--no-api", action="store_true", help="Use web fallback (not recommended)")
    args = parser.parse_args()

    build_manifest(args.repo, args.date_folder, Path(args.out), use_api=not args.no_api)
    print(f"Manifest written to {args.out}")
```

### 2. CLI helper (`vanguard/scripts/gen_manifest.py`)
- Simple, safe, and portable.

```bash
#!/usr/bin/env bash
set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
PROJECT_ROOT="$(cd "$SCRIPT_DIR/../.." && pwd)"

REPO="${1:-datasets/company/surrogate-1}"
DATE_FOLDER="${2:-2026-04-29}"
OUT="${3:-$PROJECT_ROOT/design/manifest_$(date +%Y%m%d_%H%M%S).json}"

cd "$PROJECT_ROOT"
python -m vanguard.design.manifest --repo "$REPO" --date-folder "$DATE_FOLDER" --out "$OUT"
```

### 3. Training launcher updates (`vanguard/train.py`)
- Robust Studio reuse (no crash if Studio API changes).
- Manifest loading with clear error messages.
- CDN-only dataset with column projection and schema checks.
- No HF API calls during training.

```python

# Manifest handling
from vanguard.design.manifest import load_manifest, deterministic_repo_index

MANIFEST_PATH = Path(__file__).parent / "design" / "manifest_latest.json"
if not MANIFEST_PATH.exists():
    # Allow override by env for flexibility
    alt = os.getenv("VANGUARD_MANIFEST_PATH")
    if alt:
        MANIFEST_PATH = Path(alt)

if not MANIFEST_PATH.exists():
    raise FileNotFoundError(
        f"Manifest not found: {MANIFEST_PATH}. "
        "Generate with vanguard.design.manifest and place as manifest_latest.json or set VANGUARD_MANIFEST_PATH."
    )

manifest = load_manifest(MANIFEST_PATH)
print(f"Loaded manifest for {manifest['repo']}@{manifest['date_folder']} with {manifest['count']} files")

# CDN-only dataset with schema checks and column projection
class CDNParquetIterable(IterableDataset):
    def __init__(self, manifest: Dict[str, Any], max_files: Optional[int] = None, required_columns=("prompt", "response")):
        files = manifest["files"]
        if max_files is not None:
            files = files[:max_files]
        self.urls = [f["cdn_url"] for f in files]
        self.required_columns = required_columns

    def _fetch_and_yield(self, url: str):
        resp = requests.get(url, timeout=30)
        resp.raise_for_status()
        table = pq.read_table(io.BytesIO(resp.content), columns=self.required_columns)
        for row in table.to_pylist():
            # Basic schema enforcement
            if not all(k in row for k in self.required_columns):
                continue
            yield row

    def __iter__(self):
        for url in self.urls:
            yield from self
