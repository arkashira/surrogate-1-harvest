## Final consolidated solution

**Core diagnosis (merged)**
- LINE webhook retries and concurrent requests cause duplicate punches because ingestion is not idempotent and the upsert is non-atomic.
- There is no DB-level uniqueness to prevent duplicates and no tracking of `line_message_id` (or equivalent idempotency key) across the write path.
- Tenant isolation must be strict; queries must always scope by `tenant_id` to avoid cross-tenant contamination.

**Chosen approach (correct + actionable)**
- Enforce idempotency at the database layer with a **single atomic `INSERT … ON CONFLICT`** using `line_message_id` as the idempotency key.
- Add a **unique constraint/index** to guarantee exactly-once ingestion for non-null `line_message_id`.
- Keep business-level uniqueness for one active punch per tenant+employee (open/close) via a partial unique index to prevent double-IN/double-OUT in the same session.
- Perform all writes scoped by `tenant_id` and run in a transaction to avoid races and maintain consistency.

### 1) Schema changes (run once)

```sql
-- Add idempotency column if not present
ALTER TABLE punches
  ADD COLUMN IF NOT EXISTS line_message_id TEXT NULL;

-- Idempotency guard: one row per non-null line_message_id globally
CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_unique_line_message
  ON punches (line_message_id)
  WHERE line_message_id IS NOT NULL;

-- Business rule: at most one open punch (type='IN' without matching OUT) per tenant+employee
-- Assumes application closes previous punch or uses state transitions; adjust if your model differs.
CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_one_open_per_employee
  ON punches (tenant_id, employee_id)
  WHERE type = 'IN'
    AND deleted_at IS NULL;
```

### 2) Service implementation

File: `/opt/axentx/workio/server/src/services/punchService.ts`

```typescript
import { Pool, PoolClient } from 'pg';

const pool = new Pool({ connectionString: process.env.DATABASE_URL });

export interface RecordPunchInput {
  tenantId: string;
  employeeId: string;
  type: 'IN' | 'OUT';
  location?: { lat: number; lng: number; address?: string } | null;
  lineMessageId?: string | null;
  timestamp?: Date;
}

/**
 * Idempotent, atomic punch record.
 * - Uses lineMessageId as idempotency key when provided.
 * - Enforces tenant isolation and business uniqueness via DB constraints.
 * - Safe under concurrency and LINE retry storms.
 */
export async function recordPunch(
  input: RecordPunchInput,
  client?: PoolClient
): Promise<{ id: string; created_at: Date; updated_at: Date }> {
  const {
    tenantId,
    employeeId,
    type,
    location,
    lineMessageId,
    timestamp = new Date(),
  } = input;

  if (!tenantId || !employeeId) {
    throw new Error('tenantId and employeeId are required');
  }

  const db = client || pool;
  const isExternalTx = !!client;
  const tx = isExternalTx ? client : await pool.connect();

  async function run() {
    // If idempotency key provided, attempt insert first; conflict means retry/no-op.
    // We still upsert on tenant+employee+line_message_id to allow safe re-delivery
    // with possible metadata correction (e.g., location/time), but never duplicate.
    const result = await tx.query(
      `INSERT INTO punches (
         tenant_id,
         employee_id,
         type,
         event_time,
         location,
         line_message_id,
         created_at,
         updated_at
       ) VALUES (
         $1, $2, $3, $4, $5, $6, NOW(), NOW()
       )
       ON CONFLICT (line_message_id) DO UPDATE
       SET
         tenant_id = EXCLUDED.tenant_id,
         employee_id = EXCLUDED.employee_id,
         type = EXCLUDED.type,
         event_time = EXCLUDED.event_time,
         location = EXCLUDED.location,
         updated_at = NOW()
       RETURNING id, created_at, updated_at`,
      [
        tenantId,
        employeeId,
        type,
        timestamp,
        location ? JSON.stringify(location) : null,
        lineMessageId || null,
      ]
    );

    return result.rows[0];
  }

  try {
    if (!isExternalTx) {
      await tx.query('BEGIN');
    }
    const row = await run();
    if (!isExternalTx) {
      await tx.query('COMMIT');
    }
    return row;
  } catch (err: any) {
    if (!isExternalTx) {
      await (tx as PoolClient).query('ROLLBACK');
    }
    // Unique violation from line_message_id or business rule index => treat as duplicate
    if (err.code === '23505') {
      // Fetch existing to return stable response for idempotent retries
      const existing = await tx.query(
        `SELECT id, created_at, updated_at FROM punches
         WHERE tenant_id = $1 AND employee_id = $2 AND line_message_id = $3
         LIMIT 1`,
        [tenantId, employeeId, lineMessageId || null]
      );
      if (existing.rows.length) {
        return existing.rows[0];
      }
    }
    throw err;
  } finally {
    if (!isExternalTx && tx instanceof PoolClient) {
      tx.release();
    }
  }
}
```

### 3) Webhook handler (idempotent consumption)

File: `/opt/axentx/workio/server/src/routes/lineWebhook.ts`

```typescript
import { Router } from 'express';
import { recordPunch } from '../services/punchService';
import { getLastPunch } from '../services/punchService'; // implement as needed
import { resolveTenantIdByChannelId, resolveEmployeeIdByUserId } from '../services/tenantService';

const router = Router();

/**
 * LINE webhook consumer.
 * - Acknowledges quickly (200) after processing.
 * - Uses lineMessageId for idempotency.
 * - Determines punch type by last punch state per tenant+employee.
 */
router.post('/webhook/line', async (req, res) => {
  try {
    const events = req.body.events || [];
    for (const ev of events) {
      const lineMessageId = ev.message?.id || `${ev.source?.userId}-${ev.timestamp}`;
      const tenantId = await resolveTenantIdByChannelId(ev.source?.channelId);
      const employeeId = await resolveEmployeeIdByUserId(ev.source?.userId, tenantId);

      const lastPunch = await getLastPunch(tenantId, employeeId);
      const type = !lastPunch || lastPunch.type === 'OUT' ? 'IN' : 'OUT';

      await recordPunch({
        tenantId,
        employeeId,
        type,
        location: ev.message?.location
          ? { lat: ev.message.location.latitude, lng: ev.message.location.longitude }
          : null,
        lineMessageId,
        timestamp: ev.timestamp ? new Date(ev.timestamp) : new Date(),
      });
    }
    res.status(200).json({ ok: true });
  } catch (err) {
    console.error('LINE webhook failed', err);
    // Return 200 for idempotent duplicates to prevent LINE retry storms;
    // return 5xx only for unexpected errors that require investigation.
    if (err && (err as any).code === '23505') {
      return res.status(200).json({ ok: true });
    }
    res.status(500).json({ error: 'internal' });
  }
});

export default router;
```
