# workio / design

## Final Consolidated Solution (Correct + Actionable)

### 1. Diagnosis (merged, de-duplicated)

- **No idempotency key on LINE webhooks** → at-least-once delivery + retries create duplicate punch rows.
- **Non-atomic find-then-insert/update** → concurrent webhooks can create overlapping active punches or double-close the same session.
- **Clock state derived without snapshot isolation** → UI/dashboard can show stale state and trigger conflicting actions.
- **Missing optimistic concurrency/versioning on attendance rows** → concurrent edits (leave/OT/approvals) can silently overwrite each other.
- **No transactional boundary across attendance + audit/log** → partial writes break auditability and leave inconsistent state on failure.

### 2. Scope & Priority

- **Must fix now**: clock-in/out idempotency + atomic state transition + transactional integrity for punches.
- **Should fix now**: optimistic concurrency for attendance edits (leave/OT) and unified audit log within same transaction.
- **Defer**: broader schema refactors unrelated to attendance consistency.

### 3. Schema Changes (single source of truth)

```sql
-- workio/server/src/db/schema.sql

-- Idempotency for LINE/webhook retries (global)
CREATE TABLE IF NOT EXISTS webhook_deliveries (
  idempotency_key TEXT PRIMARY KEY,
  event_type      TEXT      NOT NULL,
  user_id         INTEGER   NOT NULL,
  payload_hash    TEXT      NOT NULL,
  created_at      TIMESTAMPTZ DEFAULT NOW()
);

-- Attendance row versioning for optimistic concurrency
ALTER TABLE punches ADD COLUMN IF NOT EXISTS version INTEGER NOT NULL DEFAULT 1;
ALTER TABLE punches ADD COLUMN IF NOT EXISTS updated_at TIMESTAMPTZ DEFAULT NOW();

-- One active punch per user (no overlapping active sessions)
CREATE UNIQUE INDEX IF NOT EXISTS idx_one_active_punch_per_user
  ON punches (user_id)
  WHERE active = TRUE;

-- Unified audit log (append-only)
CREATE TABLE IF NOT EXISTS attendance_audit (
  id             BIGSERIAL PRIMARY KEY,
  attendance_id  INTEGER    NOT NULL,
  user_id        INTEGER    NOT NULL,
  action         TEXT       NOT NULL,  -- 'clock_in','clock_out','edit','approve','reject'
  actor_id       INTEGER,              -- who performed action (user/admin)
  old_values     JSONB,
  new_values     JSONB,
  idempotency_key TEXT,
  created_at     TIMESTAMPTZ DEFAULT NOW()
);

-- Indexes for cleanup/performance
CREATE INDEX IF NOT EXISTS idx_webhook_deliveries_created ON webhook_deliveries (created_at);
CREATE INDEX IF NOT EXISTS idx_attendance_audit_attendance ON attendance_audit (attendance_id);
CREATE INDEX IF NOT EXISTS idx_attendance_audit_user ON attendance_audit (user_id);
```

// workio/server/src/services/attendanceService.ts
import { pool } from '../db';
import { formatISO } from 'date-fns';

export interface ClockPayload {
  userId: number;
  lineUserId: string;
  idempotencyKey: string;
  latitude?: number;
  longitude?: number;
  timestamp?: string; // ISO
  actorId?: number;  // who triggered (user or admin)
}

export async function processClock(payload: ClockPayload) {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // 1) Idempotency guard (global)
    const idem = await client.query(
      `SELECT 1 FROM webhook_deliveries WHERE idempotency_key = $1`,
      [payload.idempotencyKey]
    );
    if (idem.rows.length > 0) {
      await client.query('ROLLBACK');
      return { status: 'duplicate', message: 'Already processed' };
    }

    // 2) Lock user to serialize concurrent transitions
    await client.query(`SELECT id FROM users WHERE id = $1 FOR UPDATE`, [payload.userId]);

    // 3) Find active punch (snapshot via transaction)
    const activeRes = await client.query(
      `SELECT id, version, clock_in, clock_out, active
       FROM punches
       WHERE user_id = $1 AND active = TRUE
       LIMIT 1`,
      [payload.userId]
    );

    const now = payload.timestamp ? new Date(payload.timestamp) : new Date();
    const isoNow = formatISO(now);

    if (activeRes.rows.length > 0) {
      // clock-out
      const punch = activeRes.rows[0];
      await client.query(
        `UPDATE punches
         SET clock_out = $1, active = FALSE, version = version + 1, updated_at = NOW()
         WHERE id = $2 AND version = $3
         RETURNING id, version`,
        [isoNow, punch.id, punch.version]
      );

      // audit
      await client.query(
        `INSERT INTO attendance_audit (attendance_id, user_id, action, actor_id, old_values, new_values, idempotency_key)
         VALUES ($1, $2, $3, $4, $5, $6, $7)`,
        [
          punch.id,
          payload.userId,
          'clock_out',
          payload.actorId ?? payload.userId,
          JSON.stringify({ clock_in: punch.clock_in, active: true }),
          JSON.stringify({ clock_out: isoNow, active: false }),
          payload.idempotencyKey
        ]
      );
    } else {
      // clock-in
      const insertRes = await client.query(
        `INSERT INTO punches (user_id, clock_in, active, latitude, longitude, version, updated_at)
         VALUES ($1, $2, TRUE, $3, $4, 1, NOW())
         RETURNING id, version`,
        [payload.userId, isoNow, payload.latitude, payload.longitude]
      );

      const punchId = insertRes.rows[0].id;

      // audit
      await client.query(
        `INSERT INTO attendance_audit (attendance_id, user_id, action, actor_id, old_values, new_values, idempotency_key)
         VALUES ($1, $2, $3, $4, $5, $6, $7)`,
        [
          punchId,
          payload.userId,
          'clock_in',
          payload.actorId ?? payload.userId,
          null,
          JSON.stringify({ clock_in: isoNow, active: true, latitude: payload.latitude, longitude: payload.longitude }),
          payload.idempotencyKey
        ]
      );
    }

    // 4) Record idempotency (after successful business writes)
    await client.query(
      `INSERT INTO webhook_deliveries (idempotency_key, event_type, user_id, payload_hash)
       VALUES ($1, $2, $3, $4)`,
      [payload.idempotencyKey, 'clock', payload.userId, 'hash-placeholder']
    );

    await client.query('COMMIT');
    return { status: 'ok', clockedIn: activeRes.rows.length === 0 };
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  } finally {
    client.release();
  }
}
