### 1) Scope (single highest-value deliverable)
Add a **non-blocking Top-Hub Signal Panel** to the Costinel dashboard that:
- Detects and surfaces the most-connected knowledge-hub (e.g., `"MOC"`) from the local RAG/knowledge graph.
- Shows **3–5 actionable cost-governance signals** with context, traceability, and direct links.
- Uses **CDN-first data fetching** (bypasses HF API rate limits) with graceful fallback to locally cached JSON.
- **Never blocks dashboard render**; renders async and degrades silently.
- Emits an audit event (`signal_impression`) for every signal shown (for compliance/telemetry).

### 4) Backend — `/api/top-hub` (CDN-first)

File: `/server/api/top-hub.ts` (Next.js API route)

```ts
// server/api/top-hub.ts
import type { NextApiRequest, NextApiResponse } from 'next';
import { audit } from '@/lib/audit';

const CDN_BASE = 'https://huggingface.co/datasets/axentx/costinel/resolve/main';
const REPO_FALLBACK_TREE = 'org/costinel-knowledge'; // optional HF repo tree fallback

export interface TopHubPayload {
  hub: string;
  score: number;
  signals: Array<{
    id: string;
    title: string;
    snippet: string;
    links: Array<{ label: string; href: string }>;
    tags: string[];
  }>;
  generatedAt: string;
  ttl: number;
}

async function fetchFromCDN(path = 'hub-graph/top-hub.json', timeout = 5000): Promise<TopHubPayload | null> {
  const url = `${CDN_BASE}/${path.replace(/^\/+/, '')}`;
  try {
    const controller = new AbortController();
    const id = setTimeout(() => controller.abort(), timeout);
    const res = await fetch(url, { method: 'GET', signal: controller.signal, cache: 'no-store' });
    clearTimeout(id);

    if (res.status === 429) {
      // Respect retry-after if present; caller may retry later
      return null;
    }
    if (!res.ok) return null;
    const json = (await res.json()) as TopHubPayload;
    if (json && typeof json.hub === 'string' && Array.isArray(json.signals)) return json;
    return null;
  } catch {
    return null;
  }
}

export default async function handler(req: NextApiRequest, res: NextApiResponse) {
  // 1) CDN-first
  const cdnData = await fetchFromCDN();
  if (cdnData) {
    // Optional: emit audit for panel load (not per signal here)
    audit.emit('top_hub_served', { source: 'cdn', hub: cdnData.hub }).catch(() => {});
    return res.status(200).json(cdnData);
  }

  // 2) Fallback: local static artifact bundled in repo (non-blocking)
  try {
    // In production, this file is built into the service image at build time
    const local = require('../../../data/hub-graph/top-hub.json') as TopHubPayload;
    audit.emit('top_hub_served', { source: 'local', hub: local.hub }).catch(() => {});
    return res.status(200).json(local);
  } catch {
    // 3) Final fallback: empty response (panel will hide)
    return res.status(200).json(null);
  }
}
```

Notes:
- `audit.emit` must be non-blocking (fire-and-forget or batched queue).
- If HF 429 occurs, CDN fetch returns `null`; endpoint falls back to local artifact (bundled at build) so panel still works offline.
