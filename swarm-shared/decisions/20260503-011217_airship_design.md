# airship / design

### Final Consolidated Plan  
**Title:** Airship end-to-end — HF CDN bypass + Training manifest + Studio reuse + Safe schema projection  
**Goal:** Eliminate HF API 429s and `pyarrow.CastError` during Surrogate-1 training, reduce Lightning Studio quota burn, and enable frontend-driven launch from Arkship UI.  
**Timebox:** ≤2 hours  

---

### 1) High-level flow (what runs where)
1. **Frontend** (Arkship UI) → simple form (repo, date folder, preset).  
2. **Backend** (Arkship, port 8000) → one HF `list_repo_tree` call, emits `manifests/{repo}-{date}.json` with CDN-only URLs.  
3. **Studio orchestration** → reuse running Studio by name; restart only if stopped (idle timeout).  
4. **Surrogate training script** → reads manifest, fetches parquet via CDN (no auth header), projects to `{prompt, response}` at parse time, writes attribution parquet.  

---

// arkship/frontend/src/pages/TrainPage.tsx
import { useState } from "react";

export default function TrainPage() {
  const [repo, setRepo] = useState("");
  const [date, setDate] = useState("");
  const [preset, setPreset] = useState("surrogate-l40s");
  const [status, setStatus] = useState("");

  const onSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    setStatus("Generating manifest...");
    const res = await fetch("/api/training/manifest", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ repo, date, preset }),
    });
    const data = await res.json();
    if (data.manifestUrl) {
      setStatus("Launching studio...");
      const runRes = await fetch("/api/training/run", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ manifestUrl: data.manifestUrl, jobId: data.jobId }),
      });
      const runData = await runRes.json();
      setStatus(runData.statusUrl ? `Running: ${runData.statusUrl}` : `Error: ${runData.error}`);
    } else {
      setStatus(`Error: ${data.error}`);
    }
  };

  return (
    <form onSubmit={onSubmit}>
      <label>
        Repo:
        <input value={repo} onChange={(e) => setRepo(e.target.value)} required />
      </label>
      <label>
        Date folder (YYYY-MM-DD):
        <input value={date} onChange={(e) => setDate(e.target.value)} required />
      </label>
      <label>
        Preset:
        <select value={preset} onChange={(e) => setPreset(e.target.value)}>
          <option value="surrogate-l40s">L40S (Lightning free-tier fallback)</option>
          <option value="surrogate-h200">H200 (lightning-lambda-prod)</option>
        </select>
      </label>
      <button type="submit">Generate Manifest & Launch</button>
      {status && <p>{status}</p>}
    </form>
  );
}

#### Backend manifest endpoint (single HF tree call, CDN-only)
```ts
// arkship/src/training/manifest.ts
import { Router } from "express";
import { listRepoTree } from "../utils/hf";
import { writeFileSync, ensureDirSync } from "fs-extra";
import { join } from "path";

const router = Router();

router.post("/manifest", async (req, res) => {
  const { repo, date } = req.body;
  const folderPath = `${date}`;

  try {
    // One API call; respect rate limits
    const tree = await listRepoTree({ repo: `datasets/${repo}`, path: folderPath, recursive: false });

    const files = (tree.files || []).map((f: any) => ({
      path: `${folderPath}/${f.path}`,
      cdn_url: `https://huggingface.co/datasets/${repo}/resolve/main/${folderPath}/${f.path}`,
    }));

    const manifest = {
      repo,
      date,
      createdAt: new Date().toISOString(),
      files,
      cdnPrefix: `https://huggingface.co/datasets/${repo}/resolve/main/`,
    };

    ensureDirSync(join(process.cwd(), "manifests"));
    const manifestPath = `manifests/${repo}-${date}.json`;
    writeFileSync(join(process.cwd(), manifestPath), JSON.stringify(manifest, null, 2));

    res.json({ manifestUrl: `/manifests/${repo}-${date}.json`, jobId: `${repo}-${date}-${Date.now()}` });
  } catch (err) {
    console.error(err);
    res.status(500).json({ error: "Failed to generate manifest" });
  }
});

export default router;
```

#### Studio reuse + guard (correctness + quota saving)
```python
# arkship/src/training/studio.py
import lightning as L
import time

def get_or_create_studio(name: str, machine: str = "L40S", cloud: str = "lightning-public-prod") -> L.Studio:
    teamspace = L.Teamspace(cloud=cloud)
    for studio in teamspace.studios:
        if studio.name == name:
            if studio.status == "Running":
                return studio
            # restart if stopped (idle timeout)
            studio.start(machine=machine)
            # wait until running
            while studio.status != "Running":
                time.sleep(5)
                studio.refresh()
            return studio
    # create new if none exists
    return teamspace.create_studio(name, machine=machine, cloud=cloud)

def run_with_studio(name: str, entrypoint: str, args: list[str], machine: str = "L40S"):
    studio = get_or_create_studio(name, machine=machine)
    # guard before run
    if studio.status != "Running":
        studio.start(machine=machine)
        while studio.status != "Running":
            time.sleep(5)
            studio.refresh()
    return studio.run(entrypoint=entrypoint, args=args)
```

#### Backend run endpoint (trigger training in Studio)
```ts
// arkship/src/training/run.ts
import { Router } from "express";
import { run_with_studio } from "./studio";

const router = Router();

router.post("/run", async (req, res) => {
  const { manifestUrl, jobId } = req.body;
  try {
    const run = await run_with_studio(
      `surrogate-${jobId}`,
      "python train.py",
      ["--manifest", manifestUrl],
      "L40S"
    );
    res.json({ statusUrl: `/training/status/${jobId}`, runId: run.name });
  } catch (err: any) {
    res.status(500).json({ error: err.message });
  }
});

export default router;
```
