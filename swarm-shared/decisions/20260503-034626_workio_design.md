# workio / design

**Final synthesized implementation (correct + actionable)**

This merges the strongest, production-ready choices from both proposals and resolves contradictions in favor of correctness and concrete actionability.

- **Idempotency**: enforce at the handler boundary with **persistent storage** (not in-memory) so multi-instance deployments remain safe.
- **Signature verification**: strict `X-Line-Signature` check using timing-safe comparison; reject early with 401.
- **Observability**: structured request/response/error logging with consistent fields; include idempotency key and verification outcome.
- **Realtime sync**: lightweight **SSE broadcast** from backend after state changes; frontend uses `EventSource` + React Query invalidation (not polling).
- **Frontend integration**: small hook to subscribe once; invalidate only affected queries to reduce thrash.
- **Resilience**: idempotency key required for client-initiated mutations; webhook processor also uses an internal idempotency key derived from LINE event ID to prevent replays across instances.

### `server/src/middleware/lineWebhook.ts` (new)
```ts
import crypto from 'crypto';
import { Request, Response, NextFunction } from 'express';
import { logger } from '../utils/logger.js';
import { IdempotencyStore } from '../services/idempotencyStore.js';

const LINE_CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';
if (!LINE_CHANNEL_SECRET) {
  throw new Error('LINE_CHANNEL_SECRET is required');
}

export function verifyLineSignature(req: Request, res: Response, next: NextFunction) {
  const sig = req.get('X-Line-Signature') || '';
  // body must be the raw buffer for correct signature; ensure raw body middleware runs before this.
  const raw = (req as any).rawBody || JSON.stringify(req.body);
  const expected = crypto
    .createHmac('sha256', LINE_CHANNEL_SECRET)
    .update(Buffer.from(raw))
    .digest('base64');

  const ok = sig && crypto.timingSafeEqual(Buffer.from(sig), Buffer.from(expected));
  if (!ok) {
    logger.warn({
      msg: 'line.webhook.invalid_signature',
      sigPresent: !!sig,
      body: req.body,
    });
    return res.status(401).json({ error: 'invalid_signature' });
  }
  next();
}

export async function lineWebhookIdempotency(req: Request, res: Response, next: NextFunction) {
  // Prefer client-provided Idempotency-Key; fallback to body.idempotencyKey for webhook replays.
  const key = req.get('Idempotency-Key') || req.body.idempotencyKey;
  if (!key) return next();

  const cached = await IdempotencyStore.get(key);
  if (cached) {
    logger.info({ msg: 'line.webhook.idempotent_replay', key });
    return res.status(cached.status).json(cached.response);
  }

  // Capture response to cache
  const originalJson = res.json.bind(res);
  res.json = (payload: any) => {
    // Best-effort cache; don't block response on cache write
    IdempotencyStore.set(key, { status: res.statusCode, response: payload }).catch((err) => {
      logger.error({ msg: 'idempotency_cache_failed', key, error: err });
    });
    return originalJson(payload);
  };
  next();
}
```

### `server/src/services/idempotencyStore.ts` (new)
```ts
// Persistent, multi-instance-safe idempotency store.
// Uses Redis if available; otherwise falls back to DB (implement DB adapter for prod).
import { redis } from './redis.js';

const TTL_SECONDS = 5 * 60; // 5 minutes

export type CachedResponse = {
  status: number;
  response: unknown;
};

export const IdempotencyStore = {
  async get(key: string): Promise<CachedResponse | null> {
    if (redis) {
      const raw = await redis.get(key);
      return raw ? JSON.parse(raw) : null;
    }
    // TODO: DB fallback implementation for multi-instance safety
    return null;
  },

  async set(key: string, value: CachedResponse): Promise<void> {
    if (redis) {
      await redis.set(key, JSON.stringify(value), 'EX', TTL_SECONDS);
      return;
    }
    // TODO: DB fallback
  },
};
```

### `server/src/routes/webhook/line.ts` (modified)
```ts
import express from 'express';
import { verifyLineSignature, lineWebhookIdempotency } from '../middleware/lineWebhook.js';
import { logger } from '../utils/logger.js';
import { processLineEvent } from '../services/lineService.js';

const router = express.Router();

router.post(
  '/',
  // Ensure rawBody middleware runs before signature verification in app setup
  verifyLineSignature,
  lineWebhookIdempotency,
  async (req, res) => {
    const start = Date.now();
    const evtId = req.body?.events?.[0]?.source?.userId || req.body?.id || 'unknown';
    logger.info({ msg: 'line.webhook.received', evtId, body: req.body });

    try {
      // Use LINE event ID as internal idempotency key to prevent cross-instance replays
      await processLineEvent(req.body, { idempotencyKey: `line:${evtId}` });
      const duration = Date.now() - start;
      logger.info({ msg: 'line.webhook.processed', evtId, duration });
      res.json({ ok: true });
    } catch (err: any) {
      logger.error({ msg: 'line.webhook.failed', evtId, error: err?.message || err, body: req.body });
      res.status(500).json({ error: 'processing_failed' });
    }
  }
);

export default router;
```
