# vanguard / ops

## 1. Diagnosis
- No deterministic CDN-first manifest exists for training data → Lightning training scripts still risk runtime HF API calls (429s) during data loading.
- Missing content-hash integrity verification for downloaded parquet files → silent corruption can poison surrogate-1 training runs.
- No studio reuse guard in training launcher → idle timeout kills runs and quota is wasted by repeated `Studio(create_ok=True)` instead of reusing running instances.
- No local file-list cache for date-partitioned mirror folders → Mac orchestration must call `list_repo_tree` repeatedly and risks rate limits.
- Missing idempotent mount-point/entrypoint binding for ops scripts → wrapper scripts fail when invoked from cron or CI due to cwd/env differences.

## 2. Proposed change
Create `/opt/axentx/vanguard/ops/cdn_manifest.py` and `/opt/axentx/vanguard/ops/launch_studio.py`:
- `cdn_manifest.py`: single Mac-side script that calls `list_repo_tree` once for a date folder, writes `manifest-{date}.json` with `{path, sha256, size, url}` entries using CDN resolve URLs.
- `launch_studio.py`: Lightning launcher that lists running studios, reuses a running one if present, verifies studio is alive before `.run()`, and falls back to L40S on `lightning-public-prod`.
- Add integrity check in training data loader (small diff to surrogate-1 loader) that verifies `sha256` against downloaded parquet before use.

## 3. Implementation

```bash
# /opt/axentx/vanguard/ops/cdn_manifest.py
#!/usr/bin/env bash
set -euo pipefail

REPO="${1:-datasets/axentx/surrogate-1-mirror}"
DATE="${2:-$(date +%Y-%m-%d)}"
OUT="${3:-manifest-${DATE}.json}"

python3 - "$REPO" "$DATE" "$OUT" <<'PY'
import json, hashlib, os, sys
from huggingface_hub import HfApi, hf_hub_download

REPO, DATE, OUT = sys.argv[1], sys.argv[2], sys.argv[3]
api = HfApi()

# single non-recursive call per date folder
items = api.list_repo_tree(repo_id=REPO, path=DATE, recursive=False)
manifest = []
for item in items:
    if not item.path.endswith(".parquet"):
        continue
    # CDN URL (no auth, bypasses /api/ rate limits)
    url = f"https://huggingface.co/datasets/{REPO}/resolve/main/{item.path}"
    # local temp download for hash (lightweight; one small file)
    local = hf_hub_download(repo_id=REPO, filename=item.path, cache_dir="/tmp/hf_cache")
    sha256 = hashlib.sha256(open(local, "rb").read()).hexdigest()
    manifest.append({
        "path": item.path,
        "sha256": sha256,
        "size": os.path.getsize(local),
        "url": url
    })

with open(OUT, "w") as f:
    json.dump(manifest, f, indent=2)
print(f"Wrote {len(manifest)} entries to {OUT}")
PY
```

```python
# /opt/axentx/vanguard/ops/launch_studio.py
#!/usr/bin/env python3
"""
Reuse running Lightning Studio or start one deterministically.
Ensures studio is alive before .run() to avoid idle-timeout deaths.
"""
import time
from lightning_sdk import Studio, Teamspace, Machine
from pathlib import Path

MANIFEST = Path(__file__).parent / "manifest-latest.json"

def reuse_or_start_studio(name="vanguard-surrogate-train", machine=Machine.L40S) -> Studio:
    teamspace = Teamspace()
    for s in teamspace.studios:
        if s.name == name and s.status == "Running":
            print(f"Reusing running studio: {s.id}")
            return s
    print(f"No running studio '{name}'; creating...")
    # free tier falls to lightning-public-prod (L40S max)
    return Studio(
        name=name,
        machine=machine,
        create_ok=True,
        teamspace=teamspace,
    )

def wait_alive(studio: Studio, timeout=120):
    for _ in range(timeout // 5):
        studio.refresh()
        if studio.status == "Running":
            return
        time.sleep(5)
    raise RuntimeError(f"Studio {studio.id} not alive after {timeout}s")

def run_training(script: str, args=None):
    studio = reuse_or_start_studio()
    wait_alive(studio)

    # verify manifest exists for CDN-only loader
    if not MANIFEST.exists():
        raise FileNotFoundError(f"Manifest missing: {MANIFEST}")

    # run training; Lightning will use CDN-only fetches if loader uses manifest URLs
    run = studio.run(
        command=f"python {script} --manifest {MANIFEST} {' '.join(args or [])}",
        environment="lightning-public-prod",
    )
    print(f"Started run {run.id}")
    return run

if __name__ == "__main__":
    import sys
    run_training(sys.argv[1] if len(sys.argv) > 1 else "train.py", sys.argv[2:])
```

```python
# /opt/axentx/vanguard/surrogate1/data.py  (small diff)
import json, hashlib, pyarrow.parquet as pq
from pathlib import Path
import requests

def load_with_integrity(manifest_path):
    manifest = json.loads(Path(manifest_path).read_text())
    rows = []
    for entry in manifest:
        url = entry["url"]
        expected = entry["sha256"]
        data = requests.get(url, timeout=30).content
        actual = hashlib.sha256(data).hexdigest()
        if actual != expected:
            raise ValueError(f"Integrity mismatch: {entry['path']}")
        table = pq.read_table(pq.ParquetFile(pq.ParquetReader(pq.BufferReader(data))))
        # project to {prompt, response} only (schema normalization)
        rows.append(table.select(["prompt", "response"]).to_pylist())
    return rows
```
