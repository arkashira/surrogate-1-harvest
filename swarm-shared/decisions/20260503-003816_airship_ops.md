# airship / ops

## Final Implementation Plan  
**Scope**: Replace `load_dataset` / `list_repo_files` in surrogate training with a **manifest-driven, CDN-only iterable loader** to eliminate HF API rate limits and mixed-schema ingestion failures.  
**Estimated effort**: ~1.5–2h

---

### 1. High-level approach (concrete, actionable)
1. **One-time manifest generation** (run on Mac or CI):  
   - Use `HfApi.list_repo_tree(path=date_folder, recursive=False)` **once per date folder** to list parquet files.  
   - Emit `manifest-{date}.json` containing `{repo, path, size?, url, num_rows?}` for every parquet file.  
   - Persist to `data/manifests/` and commit to repo so training never calls the HF API.

2. **CDN-only iterable dataset** (`surrogate/data/cdn_dataset.py`):  
   - Accepts a manifest JSON.  
   - Streams parquet files via `https://huggingface.co/datasets/{repo}/resolve/main/{path}` (no auth).  
   - Projects each row to `{prompt, response}` at parse time (ignores extra columns).  
   - Implements approximate `__len__` and supports file-level shuffling + optional row buffering.  
   - Compatible with `num_workers > 0` (workers open files lazily).

3. **Training script update** (`surrogate/train.py` or equivalent):  
   - Remove `load_dataset(streaming=True)` and HF API usage.  
   - Load manifest → pass to `CdnIterableDataset`.  
   - Use with PyTorch `DataLoader`.

4. **Ingestion/mirror writer cleanup** (if present):  
   - Ensure only `{prompt, response}` written to `batches/mirror-merged/{date}/{slug}.parquet`.  
   - No `source`, `ts`, or extra metadata columns (keeps schema homogeneous).

5. **CLI utility** (`scripts/build_manifest.py`):  
   - `python scripts/build_manifest.py --repo <repo> --date 2026-05-01 --out data/manifests/manifest-2026-05-01.json`

#### `surrogate/data/cdn_dataset.py`

```python
# surrogate/data/cdn_dataset.py
from __future__ import annotations

import json
import random
from dataclasses import dataclass
from pathlib import Path
from typing import Any, Dict, Iterator, List, Optional

import pyarrow.parquet as pq
import requests
from torch.utils.data import IterableDataset

CDN_BASE = "https://huggingface.co/datasets"

@dataclass
class ParquetShard:
    repo: str          # e.g. "myorg/surrogate-data"
    path: str          # e.g. "batches/mirror-merged/2026-05-01/part-0001.parquet"
    url: str
    num_rows: Optional[int] = None

    @classmethod
    def from_manifest_entry(cls, repo: str, entry: Dict[str, Any]) -> "ParquetShard":
        path = entry["path"].lstrip("/")
        url = f"{CDN_BASE}/{repo}/resolve/main/{path}"
        return cls(repo=repo, path=path, url=url, num_rows=entry.get("num_rows"))
class CdnIterableDataset(IterableDataset):
    """
    CDN-only iterable dataset for HuggingFace-hosted parquet shards.
    - No HF API calls during training.
    - Projects rows to {prompt, response} at parse time.
    - Supports file-level shuffling and optional row buffer shuffle.
    """

    def __init__(
        self,
        manifest_path: str | Path,
        repo: str,
        buffer_size: int = 10_000,
        shuffle_files: bool = True,
        seed: int = 42,
    ):
        self.repo = repo
        self.buffer_size = buffer_size
        self.shuffle_files = shuffle_files
        self.seed = seed

        with open(manifest_path) as f:
            entries = json.load(f)
        self.shards: List[ParquetShard] = [ParquetShard.from_manifest_entry(repo, e) for e in entries]
        self._epoch = 0

    def _shard_order(self, epoch: int) -> List[ParquetShard]:
        rng = random.Random(self.seed + epoch)
        shards = list(self.shards)
        if self.shuffle_files:
            rng.shuffle(shards)
        return shards

    def _project_row(self, row: Dict[str, Any]) -> Dict[str, str]:
        # Keep only prompt/response; tolerate different casing/keys
        prompt = (
            row.get("prompt")
            or row.get("Prompt")
            or row.get("input")
            or row.get("instruction")
            or ""
        )
        response = (
            row.get("response")
            or row.get("Response")
            or row.get("output")
            or row.get("completion")
            or ""
        )
        return {"prompt": str(prompt), "response": str(response)}

    def _iter_shard(self, shard: ParquetShard) -> Iterator[Dict[str, str]]:
        # Stream download and parse with pyarrow
        resp = requests.get(shard.url, stream=True, timeout=60)
        resp.raise_for_status()

        # Use streaming-friendly approach: write to temp file if large, else memory buffer
        # For simplicity and robustness, write to temp file to avoid large memory spikes.
        import tempfile
        with tempfile.NamedTemporaryFile(delete=False, suffix=".parquet") as tmp:
            for chunk in resp.iter_content(chunk_size=8192):
                tmp.write(chunk)
            tmp_path = tmp.name

        try:
            table = pq.read_table(tmp_path)
            df = table.to_pandas()
            for _, row in df.iterrows():
                yield self._project_row(row.to_dict())
        finally:
            Path(tmp_path).unlink(missing_ok=True)

    def __iter__(self) -> Iterator[Dict[str, str]]:
        shards = self._shard_order(self._epoch)
        for shard in shards:
            try:
                yield from self._iter_shard(shard)
            except Exception as exc:
                # Log and skip bad shard to avoid killing training
                print(f"[WARN] skipping shard {shard.path}: {exc}")
                continue

    def set_epoch(self, epoch: int):
        self._epoch = epoch

    def __len__(self) -> int:
        # Approximate length; used for epoch steps estimation
        total = sum(s.num_rows for s in self.shards if s.num_rows)
        return total if total > 0 else 100_000
```

#!/usr/bin/env python3
"""
Build a manifest for a date folder in a HuggingFace dataset repo.
Usage:
    python scripts/build_manifest.py --repo org/surrogate-data --date 2026-05-01 --out data/manifests/manifest-2026-05-01.json
"""

import argparse
import json
from pathlib import Path

from huggingface_hub import HfApi

CDN_BASE = "https://huggingface.co/datasets"

def list_parquet_files(api: HfApi, repo_id: str, date_folder: str) -> list[dict]:
    """
    List parquet files in repo:date_folder (non-recursive per immediate folder).
    """
    entries = api.list_repo_tree(repo_id=repo_id, path=date_folder, recursive=False)
    out = []
    for e in entries:
        if e.path.endswith(".parquet"):
            # Normalize path relative to repo root
            full_path = e.path if e.path.startswith(date_folder) else f"{date_folder}/{e.path}"
            out.append({
                "path": full_path.lstrip("/"),
                "size": getattr(e, "size", None),
                "num_rows": None,  # optional; could fetch metadata if needed
            })
    return out

    parser.add_argument("--date", required=True, help="Date folder, e.g. 2026-05-01")
    parser.add_argument("--out", required
