# airship / ops

Below is the **single, consolidated implementation** that absorbs the strongest, most actionable parts of the candidate proposals while eliminating contradictions and fragile choices.  

It is organized for immediate execution: one deterministic build step (run on the Mac orchestrator), one robust loader (CDN-only, no HF API during training), and one integration path into the surrogate training script.

## 1) One-shot manifest build (run once on Mac orchestrator)

File: `tools/build_manifest.py`

```python
#!/usr/bin/env python3
"""
Deterministic manifest builder for CDN-only dataset loading.

Produces manifest.json with:
  - repo
  - path
  - size
  - etag (or sha)
  - cdn_url

Usage:
  python tools/build_manifest.py \
    --repo datasets/your-repo \
    --folder batches/mirror-merged/2026-05-03 \
    --out manifest.json
"""

import argparse
import json
import os
import sys
from typing import Dict, List

try:
    from huggingface_hub import list_repo_tree
except ImportError:
    sys.exit("ERROR: huggingface_hub required. pip install huggingface_hub")

CDN_BASE = "https://huggingface.co/datasets"

def build_manifest(repo: str, folder: str, out_path: str) -> None:
    owner_repo = repo.strip("/")
    print(f"Listing {owner_repo}/{folder} (non-recursive)...")

    entries = list_repo_tree(
        repo_id=owner_repo,
        path=folder,
        recursive=False,
    )

    files = [e for e in entries if e.type == "file"]
    print(f"Found {len(files)} files.")

    manifest: List[Dict[str, object]] = []
    for f in files:
        if not f.path.lower().endswith((".parquet", ".jsonl", ".json")):
            continue

        # Prefer etag when available; fall back to lfs.oid or empty
        etag = getattr(f, "etag", None) or f.lfs.get("oid", "")

        manifest.append({
            "repo": owner_repo,
            "path": f.path,
            "size": f.size or 0,
            "etag": etag,
            "cdn_url": f"{CDN_BASE}/{owner_repo}/resolve/main/{f.path}",
        })

    os.makedirs(os.path.dirname(os.path.abspath(out_path)), exist_ok=True)
    with open(out_path, "w", encoding="utf-8") as fp:
        json.dump(manifest, fp, indent=2)

    print(f"Wrote {len(manifest)} entries to {out_path}")

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Build CDN manifest")
    parser.add_argument("--repo", required=True, help="Repo (e.g. datasets/your-repo)")
    parser.add_argument("--folder", required=True, help="Folder path inside repo")
    parser.add_argument("--out", default="manifest.json", help="Output JSON path")
    args = parser.parse_args()
    build_manifest(args.repo, args.folder, args.out)
```

**Why this wins**  
- Single API call per folder (avoids rate limits).  
- Uses `etag` when present (more stable than LFS OID).  
- Explicit file-type filter keeps manifest clean.  
- Manifest is static, versionable, and portable.

## 2) CDN-only iterable dataset loader

File: `surrogate/data/cdn_dataset.py`

```python
#!/usr/bin/env python3
"""
Manifest-driven, CDN-only iterable dataset.

Guarantees:
  - Zero HF API calls during training.
  - Bounded memory (streaming file-by-file, row-by-row).
  - Deterministic retry/backoff and per-file timeout.
  - Safe schema projection to {prompt, response}.
"""

import io
import json
import logging
import random
import time
from pathlib import Path
from typing import Any, Dict, Iterator, List, Optional

import pyarrow.parquet as pq
import requests
from requests.adapters import HTTPAdapter, Retry

logger = logging.getLogger(__name__)

DEFAULT_TIMEOUT = 30
MAX_RETRIES = 5
BACKOFF_FACTOR = 1
STATUS_FORCELIST = {429, 500, 502, 503, 504}

def _make_session() -> requests.Session:
    session = requests.Session()
    retries = Retry(
        total=MAX_RETRIES,
        backoff_factor=BACKOFF_FACTOR,
        status_forcelist=list(STATUS_FORCELIST),
        allowed_methods=["GET", "HEAD"],
        raise_on_status=False,
    )
    adapter = HTTPAdapter(max_retries=retries, pool_maxsize=16)
    session.mount("https://", adapter)
    session.mount("http://", adapter)
    return session

class CdnIterableDataset:
    """
    Manifest format (list of dicts):
    [
      {
        "repo": "datasets/your-repo",
        "path": "...",
        "cdn_url": "https://huggingface.co/datasets/.../resolve/main/...",
        "etag": "..."
      },
      ...
    ]
    """

    def __init__(
        self,
        manifest_path: str,
        max_files: Optional[int] = None,
        shuffle: bool = False,
        seed: int = 42,
        timeout: int = DEFAULT_TIMEOUT,
    ) -> None:
        manifest_path = Path(manifest_path)
        if not manifest_path.exists():
            raise FileNotFoundError(f"Manifest not found: {manifest_path}")

        with open(manifest_path, "r", encoding="utf-8") as f:
            self.entries: List[Dict[str, Any]] = json.load(f)

        if not self.entries:
            raise ValueError("Manifest is empty")

        if max_files is not None:
            self.entries = self.entries[:max_files]

        if shuffle:
            random.Random(seed).shuffle(self.entries)

        self.timeout = timeout
        self.session = _make_session()
        self.stats = {"ok": 0, "skipped": 0, "failed": 0}

    def _project_dict(self, item: Dict[str, Any]) -> Optional[Dict[str, str]]:
        prompt = None
        response = None

        for pkey in ("prompt", "instruction", "input", "question"):
            if pkey in item and item[pkey] is not None:
                prompt = str(item[pkey]).strip()
                break
        for rkey in ("response", "output", "answer", "completion"):
            if rkey in item and item[rkey] is not None:
                response = str(item[rkey]).strip()
                break

        if prompt and response:
            return {"prompt": prompt, "response": response}
        return None

    def _project_to_pairs(self, content: bytes, path: str) -> Iterator[Dict[str, str]]:

                table = pq.read_table(io.BytesIO(content))
                cols = {c.lower(): c for c in table.column_names}

                prompt_col = None
                response_col = None
                for cand in ("prompt", "instruction", "input", "question"):
                    if cand in cols:
                        prompt_col = cols[cand]
                        break
                for cand in ("response", "output", "answer", "completion"):
                    if cand in cols:
                        response_col = cols[cand]
                        break

                if prompt_col is None or response_col is None:
                    logger.warning("Missing prompt/response cols in %s (cols=%s)", path, list(cols))
                    return

                for _, row in df.iterrows():

                        yield pair
