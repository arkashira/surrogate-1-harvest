# airship / design

Below is the **single, merged implementation** that keeps the strongest parts of both proposals and resolves contradictions in favor of **correctness + concrete actionability**.

- **CLI entrypoint**: Bash wrapper (Candidate 1) for universal availability and cron safety.  
- **Implementation language**: Python (Candidate 2) for robustness, JSON parsing, and Neo4j/py2neo integration.  
- **Output**: Markdown to `stdout` by default, optional `--output <file.md>` (Candidate 1), with deterministic timestamped fallback file in `./reports/` (Candidate 2).  
- **Business research**: Try `granite-business-research.sh` first; on failure or absence, use embedded strategic summary (Candidate 2).  
- **Knowledge graph**: Query Neo4j for top hub (most-connected node) and related docs; degrade gracefully if Neo4j/py2neo unavailable (Candidate 2).  
- **Paths & permissions**: Correct shebang, `SHELL=/bin/bash`, `set -euo pipefail`, and `chmod +x` (both candidates).  
- **Cron/portability**: No interactive deps; all timeouts and fallbacks explicit.

## 2) CLI wrapper (Bash)

**File**: `/opt/axentx/airship/bin/airship`

```bash
#!/usr/bin/env bash
# Arkship CLI - zero-config discovery and operations
# Cron-safe: explicit shebang, SHELL, no interactive assumptions
# Tags: #bash #cron #cli

set -euo pipefail
export SHELL=/bin/bash

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
PROJECT_ROOT="$(cd "$SCRIPT_DIR/.." && pwd)"

export PYTHONPATH="${PROJECT_ROOT}:${PYTHONPATH}"

case "${1:-}" in
    discover)
        shift
        exec python3 "${PROJECT_ROOT}/airship/cli/discover.py" "$@"
        ;;
    *)
        echo "Usage: $0 {discover} [--output <file.md>]"
        echo ""
        echo "Commands:"
        echo "  discover              Run business research + knowledge RAG + top-hub insight"
        exit 1
        ;;
esac
```

Make executable:

```bash
chmod +x /opt/axentx/airship/bin/airship
```

#!/usr/bin/env python3
"""
Zero-config discovery CLI for Arkship/Surrogate platform.
Operationalizes:
  - #business-research
  - #knowledge-rag
  - #graph (top-hub insight)
"""

import argparse
import json
import subprocess
import sys
from datetime import datetime
from pathlib import Path
from typing import Dict, Any

try:
    from py2neo import Graph
except Exception:  # pragma: no cover
    Graph = None

REPORTS_DIR = Path(__file__).parent.parent.parent / "reports"
REPORTS_DIR.mkdir(exist_ok=True)

# Timeouts (seconds)
TIMEOUT_GRANITE = 120
TIMEOUT_NEO4J = 30

def run_business_research() -> Dict[str, Any]:
    """Run granite-business-research.sh or fallback to embedded summary."""
    print("[1/3] Running business research...", file=sys.stderr)

    script_path = (
        Path(__file__).parent.parent.parent / "scripts" / "granite-business-research.sh"
    )
    if script_path.is_file():
        try:
            result = subprocess.run(
                ["bash", str(script_path)],
                capture_output=True,
                text=True,
                timeout=TIMEOUT_GRANITE,
            )
            if result.returncode == 0 and result.stdout.strip():
                return {"source": "granite", "output": result.stdout.strip()}
            else:
                print(
                    f"  Warning: granite script returned {result.returncode}",
                    file=sys.stderr,
                )
        except Exception as ex:  # pragma: no cover
            print(f"  Warning: granite script failed: {ex}", file=sys.stderr)

    # Embedded strategic summary (always available)
    return {
        "source": "embedded",
        "output": (
            "## Arkship + Surrogate Platform\n"
            "- **Market**: DevSecOps/SRE/Platform Engineering with AI assistance\n"
            "- **Value Prop**: Zero-config infrastructure automation + multi-model AI ensemble\n"
            "- **Differentiator**: Temporal workflows + Neo4j knowledge graph + 6 AI roles\n"
            "- **Target**: Teams managing multi-cloud infrastructure with incident response needs"
        ),
    }

def query_knowledge_graph() -> Dict[str, Any]:
    """Query Neo4j for top hub (most-connected node) and related docs."""
    print("[2/3] Querying knowledge graph...", file=sys.stderr)

    if Graph is None:
        return {"status": "skipped", "reason": "py2neo not available"}

    try:
        graph = Graph("bolt://localhost:7687", auth=None, max_connection_lifetime=TIMEOUT_NEO4J)

        # Top hub: most-connected node with Hub label (or fallback)
        top_hub_query = """
        MATCH (h:Hub)
        RETURN h.name AS name, h.description AS description,
               size((h)--()) AS connections
        ORDER BY connections DESC
        LIMIT 1
        """
        top_hub_result = list(graph.run(top_hub_query))

        related_docs_query = """
        MATCH (h:Hub)-[:REFERENCES|MENTIONS]-(d:Document)
        WHERE h.name = $hub_name
        RETURN d.title AS title, d.url AS url, d.summary AS summary
        LIMIT 5
        """
        if top_hub_result:
            hub_name = top_hub_result[0]["name"]
            related_docs = list(graph.run(related_docs_query, hub_name=hub_name))
        else:
            # Fallback: any hub-like node
            fallback_query = """
            MATCH (n)
            WHERE n:Hub OR n:MostConnected OR n.name =~ '(?i).*moc.*'
            RETURN n.name AS name, labels(n) AS labels, size((n)--()) AS connections
            ORDER BY connections DESC
            LIMIT 1
            """
            fb = list(graph.run(fallback_query))
            if fb:
                hub_name = fb[0]["name"]
                related_docs = list(graph.run(related_docs_query, hub_name=hub_name))
            else:
                hub_name = "MOC"
                related_docs = []

        return {
            "status": "ok",
            "top_hub": {
                "name": hub_name,
                "details": top_hub_result[0] if top_hub_result else {"connections": 0},
            },
            "related_documents": [
                {"title": r["title"], "url": r.get("url", ""), "summary": r.get("summary", "")}
                for r in related_docs
            ],
        }
    except Exception as ex:  # pragma: no cover
        return {"status": "error", "reason": str(ex)}
