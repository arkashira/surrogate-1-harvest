## Code Snippets

### 1) CDN-first ingestion guard
```bash
#!/usr/bin/env bash
# /opt/axentx/airship/ops/cdn_ingest_guard.sh
# Usage: ./cdn_ingest_guard.sh <hf_repo> <date_folder> <out_json>
# Example: ./cdn_ingest_guard.sh datasets/axentx/surrogate-mirror 2026-05-03 file_list.json

set -euo pipefail
export HF_REPO="${1:-datasets/axentx/surrogate-mirror}"
export DATE_FOLDER="${2:-$(date +%Y-%m-%d)}"
export OUT_JSON="${3:-file_list.json}"

# Single API call (rate-limit safe) — non-recursive per folder
echo "Listing ${HF_REPO}/${DATE_FOLDER} ..."
files=$(python3 -c "
import os, json, sys
from huggingface_hub import list_repo_tree
repo = os.environ['HF_REPO'].replace('datasets/', '')
items = list_repo_tree(repo=repo, path='${DATE_FOLDER}', repo_type='dataset', recursive=False)
print(json.dumps([i.rfilename for i in items if i.rfilename.endswith('.parquet') or i.rfilename.endswith('.jsonl')]))
")

# Build CDN-only URLs (no Authorization header; bypasses /api/ rate limits)
python3 -c "
import os, json, sys
items = json.loads('''$files''')
base = 'https://huggingface.co/datasets/${HF_REPO}/resolve/main'
out = []
for f in items:
    out.append({
        'path': f,
        'cdn_url': f'{base}/{f}'
    })
with open('${OUT_JSON}', 'w') as fh:
    json.dump(out, fh, indent=2)
print(f'Wrote {len(out)} files to ${OUT_JSON}')
"
```

### 2) Deterministic HF sharding (128/hr cap)
```python
#!/usr/bin/env python3
# /opt/axentx/airship/ops/deterministic_shard.py
import hashlib

def pick_sibling_repo(slug: str, siblings: int = 5) -> str:
    """
    Deterministically pick sibling repo to spread HF commit cap.
    Returns repo name like surrogate-mirror-0 .. surrogate-mirror-4
    """
    h = int(hashlib.sha256(slug.encode()).hexdigest(), 16)
    idx = h % siblings
    return f"surrogate-mirror-{idx}"

# Example usage:
# repo = pick_sibling_repo("batches/mirror-merged/2026-05-03/slug.parquet")
# upload_to_hf(repo=repo, path_in_repo=f"batches/mirror-merged/2026-05-03/slug.parquet")
```

### 3) Training script patch (CDN-only)
```python
# In your train.py or data loader
import json
from pathlib import Path
import pyarrow.parquet as pq
import requests

def load_file_list(file_list_path: str):
    with open(file_list_path) as f:
        return json.load(f)

def cdn_parquet_reader(cdn_url: str):
    # Stream from CDN without HF API auth/rate-limit
    resp = requests.get(cdn_url, stream=True, timeout=60)
    resp.raise_for_status()
    with open("/tmp/temp.parquet", "wb") as tmp:
        for chunk in resp.iter_content(chunk_size=8192):
            tmp.write(chunk)
    return pq.read_table("/tmp/temp.parquet").to_pandas()
```

### 4) Studio guard (prevent idle-stop death)
```bash
#!/usr/bin/env bash
# /opt/axentx/airship/ops/studio_guard.sh
# Ensure Lightning Studio is running (L40S on lightning-lambda-prod) before training

set -euo pipefail
export SHELL=/bin/bash

STUDIO_NAME="${1:-surrogate-l40s-training}"
MACHINE="${2:-L40S}"
CLOUD="${3:-lightning-lambda-prod}"

python3 -c "
import os
from lightning_sdk import Studio, Machine, Teamspace

studio_name = os.environ['STUDIO_NAME']
machine = Machine.${MACHINE}
cloud = '${CLOUD}'

teamspace = Teamspace()
running = None
for s in teamspace.studios:
    if s.name == studio_name and s.status == 'Running':
        running = s
        break

if running:
    print(f'Reusing running studio: {studio_name}')
else:
    print(f'Starting studio: {studio_name} on {machine} ({cloud})')
    studio = Studio.create(name=studio_name, machine=machine, cloud=cloud, create_ok=True)
    studio.start()
"
```
