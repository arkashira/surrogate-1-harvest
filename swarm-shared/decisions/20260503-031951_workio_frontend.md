# workio / frontend

## 1. Diagnosis

- Frontend has no visual feedback when LINE webhook events are being processed, so users can’t distinguish between “pending” and “failed” states after clock-in/out or leave requests.
- No optimistic UI or local rollback on network errors; failed requests leave the UI in a stale state with no clear recovery path.
- Missing client-side idempotency key handling — duplicate clicks can fire multiple requests before the first resolves, risking double-punch or double-request races.
- No inline validation for GPS/location permission before clock-in, leading to avoidable LINE-side failures that surface only after the request.
- Toast/notification system is absent or inconsistent; errors from `/webhook/line` failures (e.g., signature rejections, retries) are not surfaced to the user.

## 2. Proposed change

File scope: `src/components/Attendance/AttendanceButton.tsx` (or equivalent) and `src/hooks/useAttendance.ts` (new).  
Goal: add optimistic UI + client idempotency + error toasts for clock-in/out and leave/OT requests.

## 3. Implementation

Create `src/hooks/useAttendance.ts`:

```tsx
// src/hooks/useAttendance.ts
import { useState, useCallback, useRef } from 'react';
import { toast } from '@/components/ui/toast';

type RequestState = 'idle' | 'pending' | 'success' | 'error';

export function useAttendance() {
  const [state, setState] = useState<RequestState>('idle');
  const inflightRef = useRef<Set<string>>(new Set());

  const requestAttendance = useCallback(async (payload: {
    type: 'clock_in' | 'clock_out' | 'leave' | 'ot';
    userId: string;
    tenantId: string;
    location?: GeolocationCoordinates | null;
    idempotencyKey: string;
  }) => {
    const { idempotencyKey, type } = payload;

    // Client-side idempotency: prevent duplicate inflight requests
    if (inflightRef.current.has(idempotencyKey)) {
      return { ok: false, reason: 'duplicate_request' };
    }

    inflightRef.current.add(idempotencyKey);
    setState('pending');

    try {
      const res = await fetch('/api/attendance', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(payload),
      });

      if (!res.ok) {
        throw new Error(`HTTP ${res.status}`);
      }

      setState('success');
      toast({
        title: 'Success',
        description: `${type === 'clock_in' ? 'Clocked in' : type === 'clock_out' ? 'Clocked out' : type} request submitted.`,
        variant: 'default',
      });
      return { ok: true };
    } catch (err) {
      setState('error');
      toast({
        title: 'Request failed',
        description: 'Unable to submit attendance. Please try again.',
        variant: 'destructive',
      });
      return { ok: false, reason: (err as Error).message };
    } finally {
      // Keep key for a short window to prevent rapid duplicates, then clear
      setTimeout(() => inflightRef.current.delete(idempotencyKey), 5000);
    }
  }, []);

  return { state, requestAttendance };
}
```

Update `src/components/Attendance/AttendanceButton.tsx`:

```tsx
// src/components/Attendance/AttendanceButton.tsx
import { useAttendance } from '@/hooks/useAttendance';
import { Button } from '@/components/ui/button';
import { useEffect, useState } from 'react';

export function AttendanceButton({
  userId,
  tenantId,
  actionType,
}: {
  userId: string;
  tenantId: string;
  actionType: 'clock_in' | 'clock_out' | 'leave' | 'ot';
}) {
  const { state, requestAttendance } = useAttendance();
  const [locationOk, setLocationOk] = useState<boolean | null>(null);

  useEffect(() => {
    if (!navigator.geolocation) {
      setLocationOk(false);
      return;
    }
    navigator.geolocation.getCurrentPosition(
      () => setLocationOk(true),
      () => setLocationOk(false),
      { enableHighAccuracy: true, timeout: 8000 }
    );
  }, []);

  const handleAction = async () => {
    if (locationOk === false && (actionType === 'clock_in' || actionType === 'clock_out')) {
      toast({
        title: 'Location required',
        description: 'Please enable location access to clock in/out.',
        variant: 'destructive',
      });
      return;
    }

    const idempotencyKey = `${userId}-${actionType}-${Date.now()}`;
    await requestAttendance({
      type: actionType,
      userId,
      tenantId,
      location: locationOk ? (navigator.geolocation ? await getCurrentCoords() : null) : null,
      idempotencyKey,
    });
  };

  const getCurrentCoords = (): Promise<GeolocationCoordinates | null> =>
    new Promise((resolve) => {
      navigator.geolocation.getCurrentPosition(
        (pos) => resolve(pos.coords),
        () => resolve(null),
        { enableHighAccuracy: true, timeout: 8000 }
      );
    });

  const label = {
    clock_in: 'Clock In',
    clock_out: 'Clock Out',
    leave: 'Request Leave',
    ot: 'Request OT',
  }[actionType];

  return (
    <Button
      onClick={handleAction}
      disabled={state === 'pending' || (locationOk === false && (actionType === 'clock_in' || actionType === 'clock_out'))}
      className="w-full"
    >
      {state === 'pending' ? 'Submitting...' : label}
    </Button>
  );
}
```
