# workio / design

## Final Synthesis (Correctness + Actionability)

**Chosen priority**: secure, idempotent, tenant-scoped webhook ingestion with strict schema validation and real-time frontend sync. Contradictions resolved in favor of defense-in-depth (signature + idempotency + tenant isolation before any domain logic) and concrete, copy-pasteable implementation.

---

## 1. Minimal, Correct Architecture

1. **Early tenant resolution** (before signature check) so tenant-scoped secrets and limits apply immediately.
2. **Strict signature verification** (`X-Line-Signature`) with raw-body capture.
3. **Schema validation** (Zod) on the outer LINE envelope and inner event shapes.
4. **Idempotency** at two levels:
   - `webhook_events(line_event_id, tenant_id)` unique index to deduplicate ingestion.
   - Domain-level idempotency key on `attendance_punches` to prevent double punches.
5. **Transactional domain handling** + **audit log** for every punch/leave/OT mutation.
6. **Frontend real-time hydration** via EventSource (SSE) + optimistic store update.

-- 1) Idempotency/ingestion log
CREATE TABLE IF NOT EXISTS webhook_events (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  line_event_id   TEXT NOT NULL,
  tenant_id       UUID NOT NULL REFERENCES tenants(id) ON DELETE CASCADE,
  event_type      TEXT NOT NULL,
  payload         JSONB NOT NULL,
  processed_at    TIMESTAMPTZ,
  created_at      TIMESTAMPTZ DEFAULT NOW(),
  CONSTRAINT uq_line_event_tenant UNIQUE (line_event_id, tenant_id)
);
CREATE INDEX IF NOT EXISTS idx_webhook_events_tenant_created ON webhook_events(tenant_id, created_at);

-- 2) Attendance punches (idempotent)
ALTER TABLE attendance_punches
  ADD COLUMN IF NOT EXISTS idempotency_key TEXT,
  ADD COLUMN IF NOT EXISTS source_event_id TEXT,
  ADD COLUMN IF NOT EXISTS tenant_id UUID REFERENCES tenants(id) ON DELETE CASCADE;
CREATE UNIQUE INDEX IF NOT EXISTS idx_attendance_punches_idempotency
  ON attendance_punches (tenant_id, idempotency_key)
  WHERE idempotency_key IS NOT NULL;

-- 3) Audit log
CREATE TABLE IF NOT EXISTS audit_logs (
  id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id     UUID NOT NULL REFERENCES tenants(id) ON DELETE CASCADE,
  actor_id      UUID REFERENCES users(id) ON DELETE SET NULL,
  action        TEXT NOT NULL,
  target_type   TEXT NOT NULL,
  target_id     TEXT NOT NULL,
  metadata      JSONB,
  ip_address    INET,
  user_agent    TEXT,
  created_at    TIMESTAMPTZ DEFAULT NOW()
);
CREATE INDEX IF NOT EXISTS idx_audit_logs_tenant_created ON audit_logs(tenant_id, created_at);

### 3.1 Tenant Resolver (early)

```ts
// server/src/middleware/tenantResolver.ts
import { Request, Response, NextFunction } from 'express';
import { pool } from '../db';

export async function tenantResolver(req: Request, res: Response, next: NextFunction) {
  // Webhook: resolve by channel secret header or first match on configured secret
  if (req.path.includes('/webhook/line')) {
    const providedSecret = req.headers['x-line-channel-secret'] as string;
    const secret = process.env.LINE_CHANNEL_SECRET || '';

    // Basic binding: require configured secret to match a tenant
    const { rows } = await pool.query(
      `SELECT id FROM tenants WHERE line_channel_secret = $1 LIMIT 1`,
      [secret]
    );
    if (rows.length === 0) {
      return res.status(400).json({ error: 'Tenant not found' });
    }
    res.locals.tenantId = rows[0].id;
    return next();
  }

  // Authenticated routes: rely on existing auth middleware to set res.locals.tenantId
  next();
}
```

### 3.2 Signature Verification (raw-body required)

```ts
// server/src/middleware/lineAuth.ts
import crypto from 'crypto';
import { Request, Response, NextFunction } from 'express';

const LINE_CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';

function verifyLineSignature(rawBody: Buffer, signature: string): boolean {
  if (!LINE_CHANNEL_SECRET || !signature) return false;
  const expected = crypto
    .createHmac('sha256', LINE_CHANNEL_SECRET)
    .update(rawBody)
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

export function lineAuth(req: Request, res: Response, next: NextFunction) {
  const signature = req.headers['x-line-signature'] as string;
  if (!signature) {
    return res.status(401).json({ error: 'Missing X-Line-Signature' });
  }

  const raw = (req as any).rawBody as Buffer;
  if (!raw || !verifyLineSignature(raw, signature)) {
    return res.status(401).json({ error: 'Invalid signature' });
  }
  next();
}
```

### 3.3 Schema Validation (Zod)

```ts
// server/src/validators/lineValidator.ts
import { z } from 'zod';

export const LineMessageSchema = z.object({
  id: z.string(),
  type: z.string(),
  message: z.object({
    type: z.string(),
    text: z.string().optional(),
    id: z.string().optional()
  }).optional(),
  source: z.object({
    type: z.string(),
    userId: z.string().optional(),
    groupId: z.string().optional(),
    roomId: z.string().optional()
  }).optional(),
  replyToken: z.string().optional(),
  timestamp: z.number().optional()
});

export const LineWebhookSchema = z.object({
  destination: z.string().optional(),
  events: z.array(LineMessageSchema)
});

export type LineWebhookPayload = z.infer<typeof LineWebhookSchema>;
export type LineEvent = LineWebhookPayload['events'][number];
```
