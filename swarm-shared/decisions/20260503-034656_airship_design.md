## 1. File list utility (run on Mac)

`scripts/build_filelist.py`
```python
#!/usr/bin/env python3
"""
Generate train_files.json for a single date folder.
Run on Mac (or any dev machine) when HF API window is clear.
"""
import json
import os
import sys
from huggingface_hub import HfApi

REPO = "datasets/axentx/surrogate-mirror"
DATE_FOLDER = sys.argv[1] if len(sys.argv) > 1 else "2026-05-03"
OUT = "train_files.json"

api = HfApi()
# One non-recursive call per folder (avoids 100x pagination)
entries = api.list_repo_tree(repo_id=REPO, path=DATE_FOLDER, recursive=False)

files = [
    f"{DATE_FOLDER}/{e.path.split('/')[-1]}"
    for e in entries
    if e.path.lower().endswith(".parquet")
]

with open(OUT, "w") as f:
    json.dump(files, f, indent=2)

print(f"Wrote {len(files)} files to {OUT}")
```

Usage:
```bash
python scripts/build_filelist.py 2026-05-03
```

## 2. CDN-only dataset loader

`surrogate/training/cdn_dataset.py`
```python
from torch.utils.data import IterableDataset
import pyarrow.parquet as pq
import requests
from io import BytesIO
import json
import os

HF_DATASET = "datasets/axentx/surrogate-mirror"
CDN_ROOT = f"https://huggingface.co/{HF_DATASET}/resolve/main"

class CDNParquetIterable(IterableDataset):
    """
    Streams parquet files directly from HF CDN (no Authorization header).
    Expects train_files.json produced by build_filelist.py.
    Projects to {prompt, response} only at parse time.
    """

    def __init__(self, filelist_path="train_files.json", start=0, end=None):
        with open(filelist_path) as f:
            self.files = json.load(f)[start:end]
        self._len = len(self.files)

    def __len__(self):
        return self._len

    def _stream_file(self, rel_path):
        url = f"{CDN_ROOT}/{rel_path}"
        # CDN downloads bypass /api/ auth and rate limits
        resp = requests.get(url, timeout=30)
        resp.raise_for_status()
        return BytesIO(resp.content)

    def __iter__(self):
        worker_info = torch.utils.data.get_worker_info()
        if worker_info is None:
            files = self.files
        else:
            per_worker = len(self.files) // worker_info.num_workers
            files = self.files[
                worker_info.id * per_worker : (worker_info.id + 1) * per_worker
            ]

        for rel_path in files:
            buf = self._stream_file(rel_path)
            table = pq.read_table(buf, columns=["prompt", "response"])
            for row in table.to_pylist():
                # Keep only prompt/response; attribution via filename pattern
                yield {
                    "prompt": row.get("prompt", ""),
                    "response": row.get("response", ""),
                }
```

## 3. Lightning idle-resilient runner

`surrogate/training/lightning_runner.py`
```python
import time
from lightning import Lightning, Teamspace, L40S, Machine

LIGHTNING_ACCOUNT = "lightning-lambda-prod"  # H200 available; falls back to public
STUDIO_NAME = "surrogate-train-studio"

def get_or_create_studio():
    ts = Teamspace()
    for s in ts.studios:
        if s.name == STUDIO_NAME:
            return s
    return Lightning.Studio(
        name=STUDIO_NAME,
        machine=Machine.L40S,
        cloud=LIGHTNING_ACCOUNT,
        create_ok=True,
    )

def run_with_idle_resilience(script: str, args: list[str], max_retries=3):
    """
    Run a training script in a Lightning Studio.
    If studio is stopped (idle timeout), restart and retry.
    """
    studio = get_or_create_studio()
    for attempt in range(1, max_retries + 1):
        try:
            if studio.status != "running":
                print(f"Studio not running (status={studio.status}). Restarting...")
                studio.start(machine=Machine.L40S, cloud=LIGHTNING_ACCOUNT)
                # Wait until running
                while studio.status != "running":
                    time.sleep(10)
                    studio.refresh()

            print(f"Running {script} (attempt {attempt})")
            job = studio.run(script, args)
            job.wait()
            if job.status == "succeeded":
                print("Training succeeded.")
                return
            else:
                print(f"Job failed: {job.status}")
                if attempt == max_retries:
                    raise RuntimeError(f"Training failed after {max_retries} attempts")
        except Exception as exc:
            print(f"Attempt {attempt} failed: {exc}")
            if attempt == max_retries:
                raise
            time.sleep(30)

if __name__ == "__main__":
    run_with_idle_resilience(
        "train.py",
        ["--data", "train_files.json", "--epochs", "1"],
    )
```

## 4. Deterministic sibling-repo writer (for future ingestion)

`surrogate/training/sibling_writer.py`
```python
import hashlib

SIBLINGS = [
    "datasets/axentx/surrogate-mirror",
    "datasets/axentx/surrogate-mirror-sib1",
    "datasets/axentx/surrogate-mirror-sib2",
    "datasets/axentx/surrogate-mirror-sib3",
    "datasets/axentx/surrogate-mirror-sib4",
    "datasets/axentx/surrogate-mirror-sib5",
]

def pick_repo(slug: str) -> str:
    """Deterministic repo selection to spread writes across siblings."""
    digest = int(hashlib.sha256(slug.encode()).hexdigest(), 16)
    return SIBLINGS[digest % len(SIBLINGS)]

```
