### surrogate/ingestion/repo_selector.py
```python
import hashlib
from typing import List

SIBLING_REPOS = [
    "surrogate-data-0",
    "surrogate-data-1",
    "surrogate-data-2",
    "surrogate-data-3",
    "surrogate-data-4",
]

def select_repo(slug: str) -> str:
    """Deterministic repo selection from slug."""
    digest = hashlib.sha256(slug.encode()).hexdigest()
    idx = int(digest, 16) % len(SIBLING_REPOS)
    return SIBLING_REPOS[idx]

def batch_repos(slugs: List[str]) -> dict:
    """Group slugs by selected repo."""
    batches = {r: [] for r in SIBLING_REPOS}
    for slug in slugs:
        repo = select_repo(slug)
        batches[repo].append(slug)
    return batches
```

### scripts/pre_list_hf_paths.py
```python
#!/usr/bin/env python3
"""
Pre-list HF dataset paths for a date folder and save to JSON.
Run from Mac after rate-limit window clears.
"""
import json
import argparse
from huggingface_hub import HfApi

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", default="axentx/surrogate-mirror")
    parser.add_argument("--date", required=True, help="YYYY-MM-DD folder")
    parser.add_argument("--out", default="file_list.json")
    args = parser.parse_args()

    api = HfApi()
    # Non-recursive per folder to avoid pagination explosion
    files = api.list_repo_tree(repo_id=args.repo, path=args.date, recursive=False)
    paths = [f.rfilename for f in files if not f.rfilename.endswith("/")]

    out_data = {
        "repo": args.repo,
        "date": args.date,
        "paths": paths,
    }
    with open(args.out, "w") as f:
        json.dump(out_data, f, indent=2)
    print(f"Saved {len(paths)} paths to {args.out}")

if __name__ == "__main__":
    main()
```

### surrogate/ingestion/cdn_dataset.py
```python
import json
import requests
import pyarrow as pa
import pyarrow.parquet as pq
from pathlib import Path
from typing import List, Dict

HF_CDN = "https://huggingface.co/datasets"

def download_via_cdn(repo: str, path: str) -> bytes:
    url = f"{HF_CDN}/{repo}/resolve/main/{path}"
    resp = requests.get(url, timeout=30)
    resp.raise_for_status()
    return resp.content

def project_prompt_response(raw: bytes, ext: str) -> Dict[str, str]:
    """Project raw file to {prompt, response}. Extend per format."""
    text = raw.decode("utf-8")
    # Minimal heuristic: split by last assistant turn boundary
    # Replace with format-specific parser as needed
    if ext == ".jsonl":
        lines = [json.loads(l) for l in text.strip().splitlines()]
        # Assume last two turns or known schema
        prompt = lines[-2]["content"] if len(lines) >= 2 else ""
        response = lines[-1]["content"] if lines else ""
    else:
        # Fallback: treat whole as response, prompt empty
        prompt, response = "", text
    return {"prompt": prompt, "response": response}

def build_cdn_dataset(file_list_path: str, out_dir: Path):
    out_dir = Path(out_dir)
    out_dir.mkdir(parents=True, exist_ok=True)

    with open(file_list_path) as f:
        meta = json.load(f)

    rows = []
    for path in meta["paths"]:
        raw = download_via_cdn(meta["repo"], path)
        ext = Path(path).suffix.lower()
        rec = project_prompt_response(raw, ext)
        rec["source_file"] = path
        rows.append(rec)

    table = pa.Table.from_pylist(rows, schema=pa.schema([
        ("prompt", pa.string()),
        ("response", pa.string()),
        ("source_file", pa.string()),
    ]))
    out_file = out_dir / f"{meta['date']}.parquet"
    pq.write_table(table, out_file)
    print(f"Wrote {len(rows)} rows to {out_file}")
    return out_file
```

### surrogate/train/train.py (excerpt)

```python
# Embed file_list.json in Lightning job assets
import json
from pathlib import Path
from surrogate.ingestion.cdn_dataset import build_cdn_dataset

def prepare_dataset():
    # file_list.json shipped with job
    list_path = Path("file_list.json")
    if not list_path.exists():
        raise FileNotFoundError("Embed file_list.json in job assets")
    out_dir = Path("data/cdn")
    parquet = build_cdn_dataset(str(list_path), out_dir)
    return parquet
```

### Studio reuse guard (surrogate/train/launcher.py)
```python
from lightning import Studio, Teamspace

def get_or_create_studio(name: str, machine):
    for s in Teamspace.studios:
        if s.name == name and s.status == "Running":
            print(f"Reusing running studio: {name}")
            return s
    print(f"Creating studio: {name}")
    return Studio(
        name=name,
        machine=machine,
        create_ok=True,
    )
```
