# vanguard / design

## 1. Diagnosis
- No content-addressed manifest per date folder → training repeatedly calls `list_repo_tree`/`load_dataset` at runtime, causing HF API 429s and non-reproducible epochs.
- Data loader uses Hugging Face `datasets` API during training (streaming/list calls) instead of CDN-only fetches, exposing training to auth rate limits.
- Missing deterministic file list embedded in training artifacts → shard order drifts across runs and resumable training is unreliable.
- Frontend triggers HF API `list_repo_tree`/`load_dataset` at runtime for dataset discovery → exposes UI to 429s and non-reproducible shard order.
- No local cache/manifest layer to decouple orchestration (Mac) from training (Lightning) and allow zero-API training runs.

## 2. Proposed change
Create a lightweight manifest generator + CDN-based data loader for Surrogate-1 training:
- Add `/opt/axentx/vanguard/scripts/generate_manifest.py` — Mac-side script that lists one date folder via HF API once, produces `manifest-{date}.json` with CDN URLs and byte ranges.
- Add `/opt/axentx/vanguard/train/data_loader.py` — Lightning-side loader that reads the manifest and fetches parquet files directly via CDN (`https://huggingface.co/datasets/{repo}/resolve/main/{path}`) with zero HF API calls.
- Update training entrypoint to accept `--manifest` and use the CDN loader.

Scope: two new files + one small edit to training launcher (~150 lines total).

## 3. Implementation

### 3.1 Manifest generator (run on Mac)
`/opt/axentx/vanguard/scripts/generate_manifest.py`
```python
#!/usr/bin/env python3
"""
Generate content-addressed manifest for a date folder.
Usage:
  HF_REPO="datasets/owner/repo" python generate_manifest.py --date 2026-04-29 --out manifest-2026-04-29.json
"""
import argparse
import json
import os
import time
from datetime import datetime
from pathlib import Path
from typing import List, Dict

try:
    from huggingface_hub import HfApi
except ImportError:
    raise SystemExit("pip install huggingface_hub")

HF_API = HfApi()
CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def list_date_folder(repo: str, date: str) -> List[Dict]:
    """
    Single API call: list one date folder non-recursively.
    Expects folder layout: {date}/file-00000-of-00001.parquet, ...
    """
    items = HF_API.list_repo_tree(repo=repo, path=date, recursive=False)
    files = []
    for item in items:
        if getattr(item, "type", None) != "file":
            continue
        path = f"{date}/{item.path.split('/')[-1]}"
        files.append({
            "path": path,
            "cdn_url": CDN_TEMPLATE.format(repo=repo, path=path),
            "size": getattr(item, "size", None),
        })
    # Deterministic order
    files.sort(key=lambda f: f["path"])
    return files

def build_manifest(repo: str, date: str) -> Dict:
    ts = datetime.utcnow().isoformat(timespec="seconds") + "Z"
    files = list_date_folder(repo, date)
    if not files:
        raise SystemExit(f"No files found for {repo}@{date}")

    manifest = {
        "repo": repo,
        "date": date,
        "generated_at": ts,
        "generated_by": "generate_manifest.py",
        "files": files,
        "total_files": len(files),
        "total_bytes": sum(f["size"] or 0 for f in files),
    }
    return manifest

def main() -> None:
    parser = argparse.ArgumentParser(description="Generate CDN manifest for HF dataset date folder")
    parser.add_argument("--repo", default=os.getenv("HF_REPO"), help="HF dataset repo (e.g. datasets/owner/repo)")
    parser.add_argument("--date", required=True, help="Date folder (YYYY-MM-DD)")
    parser.add_argument("--out", required=True, help="Output JSON path")
    args = parser.parse_args()

    if not args.repo:
        raise SystemExit("Set HF_REPO or pass --repo")

    manifest = build_manifest(args.repo, args.date)
    Path(args.out).write_text(json.dumps(manifest, indent=2))
    print(f"Wrote {args.out} ({manifest['total_files']} files, {manifest['total_bytes']} bytes)")

if __name__ == "__main__":
    main()
```

### 3.2 CDN-based data loader (Lightning side)
`/opt/axentx/vanguard/train/data_loader.py`
```python
import json
import pyarrow as pa
import pyarrow.parquet as pq
import requests
from io import BytesIO
from typing import List, Dict, Iterator, Optional
import torch
from torch.utils.data import IterableDataset

class CDNParquetIterable(IterableDataset):
    """
    Zero HF API calls during training.
    Reads parquet files directly from HF CDN using manifest.
    Yields {prompt: str, response: str} dicts.
    """
    def __init__(self, manifest_path: str, start_index: int = 0):
        self.manifest_path = manifest_path
        self.start_index = start_index
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.files: List[Dict] = self.manifest["files"]
        if not self.files:
            raise ValueError("No files in manifest")

    def _stream_file(self, file_info: Dict) -> Iterator[Dict]:
        url = file_info["cdn_url"]
        # CDN fetch; no Authorization header required for public datasets
        resp = requests.get(url, timeout=60)
        resp.raise_for_status()
        table = pq.read_table(BytesIO(resp.content))
        # Project only required columns; tolerate missing cols
        has_prompt = "prompt" in table.column_names
        has_response = "response" in table.column_names
        if not (has_prompt and has_response):
            # Best-effort fallback for mixed schemas
            for col in table.column_names:
                if "prompt" in col.lower():
                    has_prompt = True
                if "response" in col.lower():
                    has_response = True
        for batch in table.to_batches(max_chunksize=1024):
            cols = batch.to_pydict()
            for i in range(len(batch)):
                prompt = cols.get("prompt", [None] * len(batch))[i] if has_prompt else None
                response = cols.get("response", [None] * len(batch))[i] if has_response else None
                if prompt is None or response is None:
                    continue
                yield {"prompt": str(prompt), "response": str(response)}

    def __iter__(self) -> Iterator[Dict]:
        start = self.start_index % len(self.files)
        for idx in range(start, len(self.files)):
            yield from self._stream_file(self.files[idx])
        # Optionally loop infinitely for epochs; caller can reseed start_index
        # For finite epoch, stop here.

def build_dataloader(manifest_path: str, batch_size: int = 8, num_workers: int = 2) -> torch.utils.data.DataLoader:
    dataset = CDNParquetIterable(manifest_path=manifest_path)
    return torch.utils.data.DataLoader(
        dataset,
        batch_size=batch_size,
        num_workers=num_workers,
        pin_memory=True,
    )
```
