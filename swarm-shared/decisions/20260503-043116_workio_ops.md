# workio / ops

## Final Synthesis (Best Parts + Correctness + Actionability)

**Core diagnosis (merged, de-duplicated):**
- No `X-Line-Signature` verification → spoofable webhooks and attendance fraud.
- No idempotency on LINE event IDs → replay/retry creates duplicate clock/leave/OT records.
- No request-level idempotency key → concurrent or out-of-order events race and double-insert.
- No state transition validation → invalid sequences (double clock-in, clock-out without prior clock-in).
- No structured audit trail for webhook payloads and decisions → hard to debug replays/forgeries.
- No backpressure/circuit-breaker behavior → LINE retries can amplify load during downstream incidents.

**Chosen scope and file:**
- Implement in `/opt/axentx/workio/server/src/routes/line/webhook.ts`.
- Add a small middleware layer: signature verification, raw-body capture, structured logging, idempotency, and state-safe upsert with transition checks.

import { Router, Request, Response, NextFunction } from 'express';
import crypto from 'crypto';
import { Pool, PoolClient } from 'pg';
import { Logger } from '../../utils/logger';

const router = Router();
const pool = new Pool({ connectionString: process.env.DATABASE_URL });
const CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';
const LOG = Logger.child({ service: 'line-webhook' });

// ---- Signature verification ----
function verifySignature(rawBody: Buffer, signature: string): boolean {
  if (!CHANNEL_SECRET) return false;
  const expected = crypto
    .createHmac('sha256', CHANNEL_SECRET)
    .update(rawBody)
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

// ---- State-safe upsert with transition validation ----
async function upsertClockEvent(
  client: PoolClient,
  tenantId: string,
  userId: string,
  eventId: string,
  type: 'in' | 'out',
  ts: Date,
  location?: { lat: number; lon: number }
) {
  // 1) Idempotency: dedupe by LINE event id
  const dup = await client.query(
    `SELECT id, type FROM clock_events WHERE line_event_id = $1 AND tenant_id = $2 FOR UPDATE`,
    [eventId, tenantId]
  );
  if (dup.rows.length > 0) {
    LOG.info({ tenantId, userId, eventId, type, existingType: dup.rows[0].type }, 'duplicate_line_event_skipped');
    return { inserted: false, reason: 'duplicate' };
  }

  // 2) Transition validation: prevent invalid sequences
  const last = await client.query(
    `SELECT type FROM clock_events
     WHERE tenant_id = $1 AND user_id = $2
     ORDER BY ts DESC, id DESC
     LIMIT 1`,
    [tenantId, userId]
  );

  const lastType = last.rows[0]?.type;
  if (lastType === type) {
    // Same-type replay or out-of-order duplicate; accept insert but mark reconcilable
    LOG.warn({ tenantId, userId, eventId, type, lastType }, 'same_type_transition_inserted');
  }

  // 3) Insert
  await client.query(
    `INSERT INTO clock_events (tenant_id, user_id, line_event_id, type, ts, location, created_at)
     VALUES ($1, $2, $3, $4, $5, $6, NOW())`,
    [tenantId, userId, eventId, type, ts, location ? JSON.stringify(location) : null]
  );

  LOG.info({ tenantId, userId, eventId, type }, 'clock_event_inserted');
  return { inserted: true };
}

// ---- Request idempotency (client-retry safety) ----
async function ensureRequestIdempotency(
  client: PoolClient,
  requestId: string,
  tenantId: string
): Promise<boolean> {
  const exists = await client.query(
    `SELECT 1 FROM webhook_requests WHERE request_id = $1 AND tenant_id = $2`,
    [requestId, tenantId]
  );
  if (exists.rows.length > 0) return false;
  await client.query(
    `INSERT INTO webhook_requests (request_id, tenant_id, received_at) VALUES ($1, $2, NOW())`,
    [requestId, tenantId]
  );
  return true;
}

// ---- Raw-body middleware ----
export function rawBodyMiddleware(
  req: Request,
  res: Response,
  next: NextFunction
) {
  const chunks: Buffer[] = [];
  req.on('data', (chunk) => chunks.push(chunk));
  req.on('end', () => {
    (req as any).rawBody = Buffer.concat(chunks);
    next();
  });
  req.on('error', next);
}
