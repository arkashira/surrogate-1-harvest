## 1. Diagnosis (resolved)
- **No persisted `(repo, dateFolder)` manifest** → repeated authenticated enumeration → quota burn + 429 risk.  
- **Recursive/heavy enumeration** exposes mixed-schema files and wastes I/O during data discovery.  
- **`load_dataset(streaming=True)` on heterogeneous repos** causes runtime PyArrow `CastError`.  
- **Authenticated API calls during training** violate rate-limit budget; no CDN-only fetch path.  
- **Missing lightweight orchestration wrapper** to pre-list once, embed file list, and launch Lightning Studio with reuse logic (saves quota).

---

## 2. Proposed change (resolved)
Add **one** Python orchestration module that:
- Builds and caches a **non-recursive `(repo, dateFolder)` manifest** (folder-level) to JSON.  
- **Projects only `{prompt, response}` at parse time** (avoids mixed-schema issues).  
- Generates a `train.py` that uses **CDN-only URLs** (`resolve/main/...`) with **zero authenticated calls during training**.  
- Reuses a running Lightning Studio or starts one (L40S priority) and runs training with **idle-stop resilience**.

File scope:  
- `/opt/axentx/vanguard/orchestrate.py`  
- `/opt/axentx/vanguard/train.py` (template emitted by orchestrator)

HF_DATASETS_CDN = "https://huggingface.co/datasets"
HF_API_BASE = "https://huggingface.co/api"

def list_date_folder(repo: str, date_folder: str, token: str = None) -> list[str]:
    """
    Non-recursive folder listing. Returns relative file paths under date_folder.
    Uses authenticated API only once (cached). Falls back to unauth if token absent.
    """
    headers = {"Authorization": f"Bearer {token}"} if token else {}
    url = f"{HF_API_BASE}/datasets/{repo}/tree/{date_folder}"
    resp = requests.get(url, headers=headers, timeout=30)
    if resp.status_code == 429:
        print("Rate limited (429). Waiting 360s...")
        time.sleep(360)
        resp = requests.get(url, headers=headers, timeout=30)
    resp.raise_for_status()
    items = resp.json()
    # Keep only files (skip subfolders)
    files = [it["path"] for it in items if it.get("type") == "file"]
    return files

def build_manifest(repo: str, date_folder: str, token: str = None) -> dict:
    files = list_date_folder(repo, date_folder, token=token)
    # Keep likely parquet/jsonl; ignore non-data files
    data_files = [f for f in files if f.lower().endswith((".parquet", ".jsonl"))]
    manifest = {
        "repo": repo,
        "date_folder": date_folder,
        "created_at": datetime.now(timezone.utc).isoformat(),
        "files": sorted(data_files),
        "cdn_prefix": f"{HF_DATASETS_CDN}/{repo}/resolve/main/{date_folder}",
    }
    return manifest

def write_manifest(manifest: dict, out_path: Path):
    out_path.parent.mkdir(parents=True, exist_ok=True)
    with open(out_path, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2)
    print(f"Manifest written: {out_path}")

def launch_lightning_train(train_path: Path, manifest_path: Path, machine: Machine = Machine.L40S):
    """
    Reuse running studio or start one, then run training with idle-stop resilience.
    """
    teamspace = Teamspace()
    studio_name = f"vanguard-{manifest_path.parent.name}"
    running = None
    for s in teamspace.studios():
        if s.name == studio_name and s.status == "running":
            running = s
            break

    if running:
        print(f"Reusing running studio: {running.name} (id={running.id})")
        studio = running
    else:
        print(f"Creating studio: {studio_name}")
        studio = teamspace.studios().create(
            name=studio_name,
            machine=machine,
        )

    # Ensure studio is running before .run()
    if studio.status != "running":
        print(f"Studio not running (status={studio.status}). Starting...")
        studio.start(machine=machine)
        # Poll briefly
        for _ in range(10):
            time.sleep(10)
            studio.refresh()
            if studio.status == "running":
                break
        if studio.status != "running":
            raise RuntimeError("Studio failed to start")

    # Copy training script into studio (simple approach: run from local path via SDK)
    print("Running training in studio...")
    run = studio.run(str(train_path), name="surrogate-1-train")
    print(f"Run started: {run.id}")
    return run
