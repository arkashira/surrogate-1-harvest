## Final Code Snippets

### 1. `scripts/generate_cdn_manifest.py`
```python
#!/usr/bin/env python3
"""
Generate CDN-only manifest for surrogate training.
Run on Mac orchestrator (single API call per folder).
"""
import json
from pathlib import Path
from huggingface_hub import HfApi

API = HfApi()
REPO = "axentx/surrogate-data"  # adjust
OUT_DIR = Path("manifest")

def generate_manifest(date_folder: str):
    print(f"Listing {REPO}/{date_folder} (non-recursive)...")
    items = API.list_repo_tree(REPO, path=date_folder, recursive=False)

    files = []
    for item in items:
        if item.type != "file":
            continue
        cdn_url = f"https://huggingface.co/datasets/{REPO}/resolve/main/{date_folder}/{item.path}"
        slug = Path(item.path).stem
        files.append({
            "cdn_url": cdn_url,
            "slug": slug,
            "date": date_folder,
            "path": item.path,
        })

    out_dir = OUT_DIR / date_folder
    out_dir.mkdir(parents=True, exist_ok=True)
    out_file = out_dir / "filelist.json"
    out_file.write_text(json.dumps(files, indent=2))
    print(f"Wrote {len(files)} entries -> {out_file}")
    return out_file

if __name__ == "__main__":
    import sys
    date_folder = sys.argv[1] if len(sys.argv) > 1 else "2026-04-29"
    generate_manifest(date_folder)
```

### 2. `surrogate/data/cdn_dataset.py`
```python
from torch.utils.data import IterableDataset
import pyarrow.parquet as pq
import requests
import io
import json
from typing import List, Dict

class CdnIterableDataset(IterableDataset):
    """
    CDN-only iterable dataset for surrogate training.
    Projects to {prompt, response} on the fly to avoid mixed-schema errors.
    """
    def __init__(self, manifest_path: str, start: int = None, end: int = None):
        with open(manifest_path) as f:
            self.files: List[Dict] = json.load(f)

        if start is not None or end is not None:
            self.files = self.files[start:end]

    def __iter__(self):
        for entry in self.files:
            url = entry["cdn_url"]
            try:
                resp = requests.get(url, stream=True, timeout=30)
                resp.raise_for_status()
                data = io.BytesIO(resp.content)
                table = pq.read_table(data, columns=["prompt", "response"])
                for row in table.to_pylist():
                    prompt = row.get("prompt") or ""
                    response = row.get("response") or ""
                    if prompt and response:
                        yield {"prompt": prompt, "response": response}
            except Exception as exc:
                print(f"Skipping {url}: {exc}")
                continue
```
