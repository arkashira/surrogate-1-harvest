# workio / frontend

## Final consolidated implementation

### Diagnosis (merged)
- No optimistic/pending UI → double-taps and perceived latency; duplicates can slip before backend idempotency settles.
- No inline location guard or recovery when LINE/geolocation permission is missing or GPS fails; users retry blindly.
- No clear loading or skeleton state during the webhook round-trip; users can’t tell if the tap “worked”.
- Errors show only generic toasts; no inline affordance to retry or fix location.
- Missing client-side idempotency + click debounce allows double-submit within the same second.

### Chosen approach
- Keep API surface unchanged (same props/behaviors) but make UX resilient.
- Add optimistic UI, location pre-check, per-action idempotency key, debounced/locked click, inline error + recovery affordance, and toast feedback.
- Resolve contradictions in favor of correctness + concrete actionability:
  - Use per-action idempotency key (not only per-session) to absorb retries across sessions and webhook replays; backend should enforce TTL (e.g., 30s).
  - Use a click lock + status guard (not only debounce) to prevent races while allowing immediate retry after errors.
  - Prefer watchPosition with timeout/fallback to getCurrentPosition to handle drift and keep location fresh; clear watch on unmount.
  - Provide inline recovery button for location errors and clear loading/pending states; keep toasts for non-inline global feedback.

// workio/src/components/ClockInOutButton.tsx
import { useState, useEffect, useCallback, useRef } from 'react';
import { toast } from 'react-hot-toast';

type Status = 'idle' | 'locating' | 'ready' | 'pending' | 'success' | 'error';

interface Props {
  type: 'in' | 'out';
  onSubmit: (opts: { type: 'in' | 'out'; lat: number; lng: number; idempotencyKey: string }) => Promise<void>;
  disabled?: boolean;
}

const DEBOUNCE_MS = 800;
const LOCATION_TIMEOUT = 12000;
const LOCATION_MAX_AGE = 30000;
const SUCCESS_HOLD_MS = 1400;

export function ClockInOutButton({ type, onSubmit, disabled }: Props) {
  const [status, setStatus] = useState<Status>('idle');
  const [location, setLocation] = useState<{ lat: number; lng: number } | null>(null);
  const watchIdRef = useRef<number | null>(null);
  const clickLockRef = useRef<boolean>(false);
  const lastClickAtRef = useRef<number>(0);

  const generateIdempotencyKey = useCallback(() => {
    // Per-action uniqueness to absorb retries and webhook replays
    return `${type}-${Date.now()}-${crypto.randomUUID?.() ?? Math.random().toString(36).slice(2)}`;
  }, [type]);

  const setError = useCallback((message: string) => {
    setStatus('error');
    toast.error(message);
  }, []);

  // Location acquisition with watch + timeout fallback
  useEffect(() => {
    if (!navigator.geolocation) {
      setError('Geolocation not supported by this browser.');
      return;
    }

    setStatus('locating');
    let settled = false;

    const options: PositionOptions = {
      enableHighAccuracy: true,
      timeout: LOCATION_TIMEOUT,
      maximumAge: LOCATION_MAX_AGE,
    };

    const onSuccess = (pos: GeolocationPosition) => {
      if (settled) return;
      const { latitude, longitude } = pos.coords;
      setLocation({ lat: latitude, lng: longitude });
      setStatus('ready');
      settled = true;
    };

    const onError = (err: GeolocationPositionError) => {
      if (settled) return;
      settled = true;
      if (watchIdRef.current !== null) {
        navigator.geolocation.clearWatch(watchIdRef.current);
        watchIdRef.current = null;
      }
      if (err.code === err.PERMISSION_DENIED) {
        setError('Location permission required to clock in/out.');
      } else {
        setError('Unable to get location. Please try again.');
      }
    };

    // Use watch to adapt to drift; also attempt a single getCurrentPosition as fast fallback
    watchIdRef.current = navigator.geolocation.watchPosition(onSuccess, onError, options);
    navigator.geolocation.getCurrentPosition(onSuccess, () => {
      // ignore fallback failure if watch already succeeded
    }, options);

    return () => {
      settled = true;
      if (watchIdRef.current !== null) {
        navigator.geolocation.clearWatch(watchIdRef.current);
        watchIdRef.current = null;
      }
    };
  }, []);

  const handleClick = useCallback(async () => {
    if (clickLockRef.current || disabled || status !== 'ready' || !location) {
      if (status === 'locating') toast.loading('Waiting for location...');
      if (status === 'error') toast.error('Fix location to continue.');
      return;
    }

    // Debounce + lock to prevent races
    const now = Date.now();
    if (now - lastClickAtRef.current < DEBOUNCE_MS) return;
    lastClickAtRef.current = now;
    clickLockRef.current = true;

    const idempotencyKey = generateIdempotencyKey();
    setStatus('pending');

    try {
      await onSubmit({ type, ...location, idempotencyKey });
      setStatus('success');
      toast.success(`Clocked ${type === 'in' ? 'in' : 'out'} successfully.`);
    } catch (err: any) {
      setStatus('error');
      toast.error(err?.message ?? `Failed to clock ${type === 'in' ? 'in' : 'out'}.`);
    } finally {
      setTimeout(() => {
        setStatus(location ? 'ready' : 'locating');
        clickLockRef.current = false;
      }, SUCCESS_HOLD_MS);
    }
  }, [type, status, location, disabled, onSubmit, generateIdempotencyKey]);

  // Inline recovery for location errors
  const handleRetryLocation = useCallback(() => {
    if (disabled) return;
    setStatus('locating');
    setLocation(null);
    // Trigger re-run of location effect by forcing a state-driven remount-like behavior
    // Simpler: rely on effect re-running when component key changes; expose via parent if needed.
    // Here we attempt a manual location refresh:
    if (!navigator.geolocation) {
      setError('Geolocation not supported by this browser.');
      return;
    }
    const options: PositionOptions = {
      enableHighAccuracy: true,
      timeout: LOCATION_TIMEOUT,
      maximumAge: 0,
    };
    navigator.geolocation.getCurrentPosition(
      (pos) => {
        setLocation({ lat: pos.coords.latitude, lng: pos.coords.longitude });
        setStatus('ready');
      },
      (err) => {
        if (err.code === err.PERMISSION_DENIED) {
          setError('Location permission required to clock in/out.');
        } else {
          setError('Unable to get location. Please try again.');
        }
      },
      options
    );
  }, [disabled]);

  const isLoading = status === 'locating' || status === 'pending';
  const isSuccess = status === 'success';
  const isError = status === 'error';
  const isReady = status === 'ready';

  return (
    <div className="inline-flex flex-col items-start gap-2">
      <button
        onClick={handleClick}
        disabled={disabled || !isReady || isLoading}
        className={`
          inline-flex items-center justify-center rounded-lg px-6 py-3 font-medium transition-all
          focus:outline-none focus:ring-2 focus:ring-offset-2
          ${disabled || (!isReady && !isError)
            ? 'cursor-not-allowed bg-gray-200 text-gray-500'
            : isError
            ? 'bg-red-600 text-white hover:bg-red-700 focus:ring-red-500'
            : type === 'in'
            ? 'bg-green-600 text-white hover:bg-green-700 focus
