# Costinel / design

## Final Implementation Plan — Top Hub Signal Panel (CDN-first, frontend-only)

**Scope**: Add a lightweight, resilient Top Hub signal card to the Costinel dashboard.  
**Effort**: ~60–90 minutes (frontend only).  
**Mechanism**: CDN JSON fetch (no auth, no backend) with local fallback and graceful degradation.

---

### 1) Data contract (single, canonical payload)

Use a stable, minimal schema that supports display, links, and future automation.

`public/data/top-hub/latest/moc.json` (committed to repo; mirrored to CDN):

```json
{
  "hub": "MOC",
  "label": "Most-Connected Hub",
  "insight": "MOC connects 37% of cross-account cost signals this week. Primary drivers: shared EKS control-plane tagging gaps and unallocated NAT egress. Recommended action: enforce cost-center tag policy on cluster IAM roles.",
  "score": 0.87,
  "connections": 128,
  "updated": "2026-05-03T04:51:53Z",
  "sourceFile": "knowledge-rag/2026-05-03/top-hub-moc.json",
  "links": [
    { "label": "Explore graph", "href": "/knowledge-rag?hub=MOC" },
    { "label": "Run market analysis", "href": "/business-research?context=MOC" }
  ]
}
```

Notes:
- `score` is a 0–1 cohesion/centrality metric.
- `connections` is the link count (from Candidate 1) for quick context.
- `updated` is ISO-8601.
- `links` are relative app routes (safe for SPA) and can be extended.

### 3) Frontend component (React/TypeScript)

Create `src/components/TopHubSignalPanel.tsx`:

```tsx
import { useEffect, useState } from 'react';
import './TopHubSignalPanel.css';

interface HubPayload {
  hub: string;
  label: string;
  insight: string;
  score: number;
  connections: number;
  updated: string;
  sourceFile?: string;
  links?: Array<{ label: string; href: string }>;
}

const CDN_URL =
  'https://huggingface.co/datasets/AXENTX/Costinel/resolve/main/public/data/top-hub/latest/moc.json';
const LOCAL_FALLBACK = '/data/top-hub/latest/moc.json';

export default function TopHubSignalPanel() {
  const [data, setData] = useState<HubPayload | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    let cancelled = false;

    async function fetchData(url: string): Promise<HubPayload> {
      const res = await fetch(url, { cache: 'no-cache' });
      if (!res.ok) throw new Error(`HTTP ${res.status}`);
      return res.json();
    }

    async function load() {
      try {
        const payload = await fetchData(CDN_URL);
        if (!cancelled) {
          setData(payload);
          setLoading(false);
        }
      } catch {
        try {
          const payload = await fetchData(LOCAL_FALLBACK);
          if (!cancelled) {
            setData(payload);
            setLoading(false);
          }
        } catch (err) {
          if (!cancelled) {
            setError('Unable to load top-hub signal');
            setLoading(false);
          }
        }
      }
    }

    load();
    return () => {
      cancelled = true;
    };
  }, []);

  if (loading) {
    return (
      <div className="top-hub-panel loading" aria-busy="true">
        Loading signal…
      </div>
    );
  }

  if (error || !data) {
    return (
      <div className="top-hub-panel error" role="status">
        {error || 'Signal unavailable'}
      </div>
    );
  }

  return (
    <div className="top-hub-panel" role="region" aria-label="Top hub signal">
      <div className="top-hub-header">
        <span className="top-hub-badge">Top Hub</span>
        <span className="top-hub-label">{data.label}</span>
        <span className="top-hub-name">{data.hub}</span>
        <span className="top-hub-meta">
          {data.connections} links · {(data.score * 100).toFixed(0)}% cohesion
        </span>
      </div>
      <p className="top-hub-insight">{data.insight}</p>
      {data.links && data.links.length > 0 && (
        <p className="top-hub-links">
          {data.links.map((l, i) => (
            <a key={i} className="top-hub-link" href={l.href}>
              {l.label}
            </a>
          ))}
        </p>
      )}
      <small className="top-hub-updated">
        Updated {new Date(data.updated).toLocaleDateString(undefined, { dateStyle: 'medium' })}
      </small>
    </div>
  );
}
```
