## Code snippets

### `scripts/build_file_manifest.py`
```python
#!/usr/bin/env python3
"""
Generate CDN-only file manifest for a date folder in a HuggingFace dataset repo.
Usage:
    python scripts/build_file_manifest.py \
        --repo-id axentx/surrogate-data \
        --date 2026-05-03 \
        --out-dir manifests
"""
import argparse
import json
import os
from pathlib import Path

from huggingface_hub import list_repo_tree

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo_id}/resolve/main/{path}"

def build_manifest(repo_id: str, date: str, out_dir: str):
    path = f"batches/mirror-merged/{date}"
    entries = list_repo_tree(repo_id=repo_id, path=path, recursive=False)

    files = []
    for entry in entries:
        if entry.path.lower().endswith(".parquet"):
            files.append({
                "repo_path": entry.path,
                "cdn_url": CDN_TEMPLATE.format(repo_id=repo_id, path=entry.path),
                "size": getattr(entry, "size", None)
            })

    manifest = {
        "repo_id": repo_id,
        "date": date,
        "path_prefix": path,
        "files": files,
        "cdn_base": f"https://huggingface.co/datasets/{repo_id}/resolve/main"
    }

    out_path = Path(out_dir) / date / "file_manifest.json"
    out_path.parent.mkdir(parents=True, exist_ok=True)
    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"Manifest written to {out_path} ({len(files)} files)")
    return manifest

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Build CDN manifest for HF dataset date folder")
    parser.add_argument("--repo-id", default="axentx/surrogate-data")
    parser.add_argument("--date", required=True, help="Date folder, e.g. 2026-05-03")
    parser.add_argument("--out-dir", default="manifests")
    args = parser.parse_args()
    build_manifest(args.repo_id, args.date, args.out_dir)
```

### `surrogate/data/cdn_iterable_dataset.py`
```python
import pyarrow.parquet as pq
import pandas as pd
from torch.utils.data import IterableDataset
import json
from typing import Optional, List, Dict

class CDNParquetIterableDataset(IterableDataset):
    """
    IterableDataset that reads parquet files via CDN URLs (no HF API/auth).
    Projects rows to {prompt, response} only.
    """
    def __init__(self, manifest_path: str, start_index: int = 0, max_files: Optional[int] = None):
        with open(manifest_path) as f:
            manifest = json.load(f)
        self.file_urls = [f["cdn_url"] for f in manifest["files"]]
        if max_files:
            self.file_urls = self.file_urls[:max_files]
        self.start_index = start_index

    def _project_record(self, row: Dict) -> Dict:
        # Keep only prompt/response; adapt keys if needed
        return {
            "prompt": row.get("prompt") or row.get("input") or row.get("text"),
            "response": row.get("response") or row.get("output") or row.get("target"),
        }

    def _stream_file(self, url: str):
        try:
            # Stream via pandas (can use pyarrow for lower memory)
            for batch in pd.read_parquet(url, chunksize=500):
                if isinstance(batch, pd.DataFrame):
                    for _, row in batch.iterrows():
                        projected = self._project_record(row.to_dict())
                        if projected["prompt"] is not None and projected["response"] is not None:
                            yield projected
                else:
                    # single-row edge case
                    row = batch.to_dict() if hasattr(batch, "to_dict") else batch
                    projected = self._project_record(row)
                    if projected["prompt"] is not None and projected["response"] is not None:
                        yield projected
        except Exception as exc:
            # Log and skip corrupt/unavailable files
            print(f"Skipping {url} due to error: {exc}")

    def __iter__(self):
        worker_info = None
        # Optional: support multi-worker sharding via torch.utils.data.get_worker_info()
        for url in self.file_urls:
            yield from self._stream_file(url)
```

### Wire into training script (excerpt)
```python
import argparse
from pathlib import Path
from surrogate.data.cdn_iterable_dataset import CDNParquetIterableDataset

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--manifest", type=str, help="Path to file_manifest.json")
    parser.add_argument("--max-files", type=int, default=None)
    args = parser.parse_args()

    if not args.manifest:
        # default to latest manifest
        manifests = sorted(Path("manifests").glob("*/*/file_manifest.json"))
        args.manifest = str(manifests[-1]) if manifests else None

    assert args.manifest and Path(args.manifest).exists(), "Manifest not found"

    dataset = CDNParquetIterableDataset(
        manifest_path=args.manifest,
        max_files=args.max_files,
    )

    # Use with DataLoader (batch_size, num_workers=0 recommended for streaming)
    from torch.utils.data import DataLoader
    loader = DataLoader(dataset, batch_size=8, num_workers=0)
