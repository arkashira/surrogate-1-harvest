## 1. Deterministic File-List Generator (mac orchestration)

```bash
# surrogate/scripts/generate_file_list.py
#!/usr/bin/env python3
"""
Run from Mac after rate-limit window clears.
Generates deterministic file list for a date folder.
Embed this JSON in train.py for CDN-only fetches.
"""
import json
import os
import sys
from datetime import datetime, timezone

try:
    from huggingface_hub import HfApi
except ImportError:
    print("❌ huggingface_hub not installed: pip install huggingface_hub")
    sys.exit(1)

def main(date_str: str = None, repo_id: str = "axentx/surrogate-dataset-mirror"):
    date_str = date_str or datetime.now(timezone.utc).strftime("%Y-%m-%d")
    folder_path = f"batches/mirror-merged/{date_str}"

    api = HfApi()
    # Single API call, non-recursive to avoid pagination explosion
    try:
        tree = api.list_repo_tree(repo_id, path=folder_path, recursive=False)
    except Exception as e:
        print(f"❌ Failed to list repo tree: {e}")
        sys.exit(1)

    files = []
    total = 0
    for item in tree:
        if item.path.endswith(".parquet"):
            size = getattr(item, "size", None) or 0
            total += size
            # CDN URL (no auth, no API rate limit)
            cdn_url = f"https://huggingface.co/datasets/{repo_id}/resolve/main/{item.path}"
            files.append({"path": item.path, "cdn_url": cdn_url, "size": size})

    if not files:
        print(f"⚠️  No parquet files found in {folder_path}")
        sys.exit(1)

    out_dir = Path("surrogate/training")
    out_dir.mkdir(parents=True, exist_ok=True)
    out_path = out_dir / f"file_list_{date_str}.json"

    manifest = {
        "generated_at": datetime.now(timezone.utc).isoformat(),
        "repo_id": repo_id,
        "folder": folder_path,
        "file_count": len(files),
        "total_bytes": total,
        "files": files,
    }

    with open(out_path, "w") as f:
        json.dump(manifest, f, indent=2)

    print(f"✅ Generated {out_path} with {len(files)} files")
    print(f"   Total size: {total / 1024 ** 2:.1f} MB")

if __name__ == "__main__":
    import argparse
    from pathlib import Path

    parser = argparse.ArgumentParser(description="Generate CDN file list for surrogate training")
    parser.add_argument("--date", help="UTC date YYYY-MM-DD (defaults to today)")
    parser.add_argument("--repo", default="axentx/surrogate-dataset-mirror", help="HF repo id")
    args = parser.parse_args()
    main(date_str=args.date, repo_id=args.repo)
```
