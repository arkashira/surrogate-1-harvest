# airship / frontend

## Implementation Plan (≤2h)

**Goal**: Eliminate HF API calls during training and make Lightning Studio resilient to idle timeouts.

### Steps
1. Create `scripts/list_hf_files.py` — one-time Mac script to list parquet files for a date folder and emit `train_files.json`.
2. Update `/opt/axentx/airship/surrogate/train.py`:
   - Load `train_files.json` (embedded file list).
   - Use CDN-only parquet fetches (`https://huggingface.co/datasets/{repo}/resolve/main/{path}`) with `pyarrow`/`pandas` (no `datasets` streaming).
   - Project to `{prompt, response}` at parse time.
   - Wrap Lightning `Studio` usage with idle-resilient runner: check status before `.run()`, auto-restart on stopped.
3. Add small utility: deterministic repo shard selection (hash slug → sibling repo) for future writes (no-op today, scaffolded).

import json
import os
import sys
from huggingface_hub import HfApi

def main(repo: str, date_folder: str, out_json: str) -> None:

    # If deeper nesting is required, list per subfolder separately.

    files = []
    for item in items:

            files.append(f"{date_folder}/{item.path}")

    payload = {

        "date_folder": date_folder,
        "files": sorted(files),

    }
    os.makedirs(os.path.dirname(os.path.abspath(out_json)) if os.path.dirname(out_json) else ".", exist_ok=True)
    with open(out_json, "w", encoding="utf-8") as f:
        json.dump(payload, f, indent=2)
    print(f"Wrote {len(files)} files to {out_json}")

# ---------- CDN parquet loader ----------
def deterministic_shard(slug: str, n_shards: int = 5) -> int:
    """Hash slug -> [0, n_shards) for deterministic sibling repo selection."""
    digest = hashlib.sha256(slug.encode("utf-8")).hexdigest()
    return int(digest, 16) % n_shards

class CDNParquetIterable:
    """
    Lightweight iterable over parquet files via CDN URLs.
    No HF dataset/streaming usage; avoids API auth and rate limits.
    """

    def __init__(self, repo: str, file_rel_paths: List[str], columns: Iterable[str] = OUTPUT_COLS) -> None:
        self.repo = repo
        self.file_rel_paths = file_rel_paths
        self.columns = list(columns)

    def _cdn_url(self, rel_path: str) -> str:
        return f"{CDN_BASE}/{self.repo}/resolve/main/{rel_path}"

    def __iter__(self):
        for rel_path in self.file_rel_paths:
            url = self._cdn_url(rel_path)
            try:
                # Stream remote parquet directly into pyarrow without full download to disk.
                with tempfile.NamedTemporaryFile(suffix=".parquet", delete=False) as tmp:
                    tmp_path = tmp.name
                resp = requests.get(url, stream=True, timeout=60)
                resp.raise_for_status()
                with open(tmp_path, "wb") as f:
                    for chunk in resp.iter_content(chunk_size=8192):
                        f.write(chunk)

                table = pq.read_table(tmp_path, columns=self.columns if self.columns else None)
                df = table.to_pandas()
                # Project to prompt/response at parse time; drop extras.
                for _, row in df.iterrows():
                    # Normalize missing fields to empty string to avoid schema heterogeneity issues.
                    prompt = str(row.get("prompt", ""))
                    response = str(row.get("response", ""))
                    if prompt.strip() or response.strip():
                        yield {"prompt": prompt, "response": response}
            finally:
                if os.path.exists(tmp_path):
                    try:
                        os.unlink(tmp_path)
                    except OSError:
                        pass

    def to_dataloader(self, batch_size: int = BATCH_SIZE, **kwargs):
        # Minimal torch dataset wrapper
        from torch.utils.data import IterableDataset, DataLoader

        class _IterableDataset(IterableDataset):
            def __init__(self, generator):
                self.generator = generator

            def __iter__(self):
                yield from self.generator

        ds = _IterableDataset(self)
        return DataLoader(ds, batch_size=batch_size, **kwargs)
