# workio / backend

## Final Synthesized Solution

### 1. Diagnosis (merged)
- **Webhook retries cause duplicates**: LINE retries on non-2xx/timeout; endpoints are not idempotent.
- **No tenant-scoped uniqueness**: concurrent/retried deliveries can double-insert clock/leave/OT events.
- **State derivation is racy**: `ORDER BY created_at DESC LIMIT 1` can flip state incorrectly under retries.
- **Commit ordering risk**: returning 200 before DB commit risks LINE treating rollback as success.
- **Frontend/server sync gap**: webhook changes invisible until polling refresh → perceived latency/silent failures.
- **Partial failure**: side-effects (notifications) before DB commit can leave DB inconsistent.

### 2. Proposed change (merged scope)
- **Files**:
  - `/opt/axentx/workio/server/src/db/migrations/003_add_idempotency_and_clock_state.sql`
  - `/opt/axentx/workio/server/src/db/queries/clock.ts` (new)
  - `/opt/axentx/workio/server/src/routes/webhook/line.ts`
- **Core**: tenant-scoped idempotency key, transactional upsert, commit-before-200, deterministic latest-state query, and safe retry handling.

```sql
-- Idempotency keys for LINE webhooks (tenant-scoped)
CREATE TABLE IF NOT EXISTS line_webhook_idempotency (
  id              SERIAL PRIMARY KEY,
  tenant_id       INTEGER NOT NULL,
  idempotency_key VARCHAR(255) NOT NULL,
  payload_hash    VARCHAR(64)  NOT NULL,
  created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  CONSTRAINT uq_tenant_idempotency UNIQUE (tenant_id, idempotency_key)
);

-- Extend clock_events for multi-event types and idempotency link
ALTER TABLE clock_events
  ADD COLUMN IF NOT EXISTS idempotency_id INTEGER REFERENCES line_webhook_idempotency(id),
  ADD COLUMN IF NOT EXISTS event_type VARCHAR(20) NOT NULL DEFAULT 'CLOCK' CHECK (event_type IN ('CLOCK', 'LEAVE', 'OT'));

-- Prevent duplicate events for same tenant+user+idempotency
CREATE UNIQUE INDEX IF NOT EXISTS uq_event_tenant_user_idempotency
  ON clock_events (tenant_id, user_id, idempotency_id)
  WHERE idempotency_id IS NOT NULL;

-- Fast latest-state lookup per tenant+user+event_type
CREATE INDEX IF NOT EXISTS idx_event_tenant_user_type_created
  ON clock_events (tenant_id, user_id, event_type, created_at DESC);
```

```ts
import { pool } from '../db';

export async function getLatestClockState(tenantId: number, userId: number) {
  const q = `
    SELECT clock_in, created_at
    FROM clock_events
    WHERE tenant_id = $1 AND user_id = $2 AND event_type = 'CLOCK'
    ORDER BY created_at DESC
    LIMIT 1;
  `;
  const res = await pool.query(q, [tenantId, userId]);
  return res.rows[0] || null;
}

export async function recordClockEvent(
  tenantId: number,
  userId: number,
  clockIn: boolean,
  idempotencyId: number,
  client = pool
) {
  const q = `
    INSERT INTO clock_events (tenant_id, user_id, clock_in, event_type, idempotency_id, created_at)
    VALUES ($1, $2, $3, 'CLOCK', $4, NOW())
    RETURNING id;
  `;
  const res = await client.query(q, [tenantId, userId, clockIn, idempotencyId]);
  return res.rows[0];
}
```

import crypto from 'crypto';

function hashPayload(body: any): string {
  return crypto.createHash('sha256').update(JSON.stringify(body)).digest('hex');
}

async function upsertIdempotency(
  tenantId: number,
  idempotencyKey: string,
  payloadHash: string,
  client: any
) {
  const q = `
    INSERT INTO line_webhook_idempotency (tenant_id, idempotency_key, payload_hash)
    VALUES ($1, $2, $3)
    ON CONFLICT (tenant_id, idempotency_key) DO UPDATE
      SET payload_hash = EXCLUDED.payload_hash,
          created_at = line_webhook_idempotency.created_at -- keep original insert time
    RETURNING id;
  `;
  const res = await client.query(q, [tenantId, idempotencyKey, payloadHash]);
  return res.rows[0].id;
}

async function processClockIntent(
  tenantId: number,
  userId: number,
  intent: string,
  idempotencyId: number,
  client: any
) {
  const clockIn = intent === 'in';
  const latest = await client.query(
    `SELECT clock_in
     FROM clock_events
     WHERE tenant_id = $1 AND user_id = $2 AND event_type = 'CLOCK'
     ORDER BY created_at DESC
     LIMIT 1`,
    [tenantId, userId]
  );
  const latestState = latest.rows[0] || null;

  // Prevent double-in or double-out (business rule)
  if (latestState && latestState.clock_in === clockIn) {
    // Record for idempotency but do not change state
    await client.query(
      `INSERT INTO clock_events (tenant_id, user_id, clock_in, event_type, idempotency_id, created_at)
       VALUES ($1, $2, $3, 'CLOCK', $4, NOW())`,
      [tenantId, userId, clockIn, idempotencyId]
    );
    return;
  }

  await recordClockEvent(tenantId, userId, clockIn, idempotencyId, client);
}
