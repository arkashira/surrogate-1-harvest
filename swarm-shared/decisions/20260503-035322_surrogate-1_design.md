# surrogate-1 / design

## Final Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16` (matrix) and optional `DATE_FOLDER` (defaults to today `YYYY-MM-DD`).
- Uses a **single pre-listed manifest** (`manifests/{DATE_FOLDER}.json`) produced by the Mac orchestrator (or the first successful runner) containing all file paths to ingest for that date.
- Each shard deterministically hashes `path` → `shard = hash(path) % SHARD_TOTAL` and processes only its slice.
- Downloads files **directly via HF CDN** (`https://huggingface.co/datasets/axentx/surrogate-1-training-pairs/resolve/main/{path}`) with no Authorization header to bypass API rate limits.
- Projects each file to `{prompt, response}` at parse time (no mixed-schema storage), computes content hash for dedup, and streams output to:
  - Local: `batches/public-merged/{DATE_FOLDER}/shard{N}-{HHMMSS}.jsonl`
  - Remote: `axentx/surrogate-1-training-pairs` via `huggingface_hub` commit (one commit per shard/run).
- Reuses existing central dedup (`lib/dedup.py`) for cross-run duplicate detection.
- Fails gracefully on HF 429 (respects `Retry-After`) and on malformed files (skip + log).
- Adds shebang, `chmod +x`, and keeps GitHub Actions matrix unchanged.

## Code Changes

### 1) New worker: `bin/dataset-enrich.py`

```python
#!/usr/bin/env python3
"""
surrogate-1 CDN-bypass ingestion worker.

Usage (GH Actions matrix):
  SHARD_ID=0 SHARD_TOTAL=16 python bin/dataset-enrich.py

Env:
  SHARD_ID            - required, 0..SHARD_TOTAL-1
  SHARD_TOTAL         - optional, default 16
  DATE_FOLDER         - optional, default YYYY-MM-DD (today)
  HF_TOKEN            - required for upload
  MANIFEST_PATH       - optional, default manifests/{DATE_FOLDER}.json
  UPLOAD_REPO         - optional, default axentx/surrogate-1-training-pairs
"""

import os
import sys
import json
import hashlib
import logging
import datetime
import time
from pathlib import Path
from typing import Dict, Any, List, Optional

import requests
from huggingface_hub import HfApi, login

# Local dedup store
sys.path.insert(0, str(Path(__file__).parent.parent))
from lib.dedup import DedupStore  # type: ignore

logging.basicConfig(
    level=logging.INFO,
    format="%(asctime)s [%(levelname)s] %(message)s",
    datefmt="%Y-%m-%dT%H:%M:%S%z",
)
log = logging.getLogger("dataset-enrich")

# ---------- config ----------
HF_TOKEN = os.getenv("HF_TOKEN")
if not HF_TOKEN:
    log.error("HF_TOKEN is required")
    sys.exit(1)

SHARD_ID = int(os.getenv("SHARD_ID", -1))
SHARD_TOTAL = int(os.getenv("SHARD_TOTAL", 16))
if SHARD_ID < 0 or SHARD_ID >= SHARD_TOTAL:
    log.error("Invalid SHARD_ID/SHARD_TOTAL")
    sys.exit(1)

DATE_FOLDER = os.getenv("DATE_FOLDER", datetime.date.today().isoformat())
MANIFEST_PATH = os.getenv("MANIFEST_PATH", f"manifests/{DATE_FOLDER}.json")
UPLOAD_REPO = os.getenv("UPLOAD_REPO", "axentx/surrogate-1-training-pairs")
OUTPUT_DIR = Path("batches/public-merged") / DATE_FOLDER
OUTPUT_DIR.mkdir(parents=True, exist_ok=True)

TIMESTAMP = datetime.datetime.utcnow().strftime("%H%M%S")
OUTPUT_FILE = OUTPUT_DIR / f"shard{SHARD_ID}-{TIMESTAMP}.jsonl"

CDN_BASE = "https://huggingface.co/datasets"
DATASET_REPO = "axentx/surrogate-1-training-pairs"

# ---------- helpers ----------
def content_hash(obj: Dict[str, Any]) -> str:
    j = json.dumps(obj, sort_keys=True, separators=(",", ":"))
    return hashlib.md5(j.encode()).hexdigest()

def shard_for_path(path: str) -> int:
    return hash(path) % SHARD_TOTAL

def load_manifest(path: str) -> List[str]:
    try:
        with open(path) as f:
            data = json.load(f)
        if isinstance(data, dict) and "files" in data:
            return data["files"]
        if isinstance(data, list):
            return data
        log.error("Unexpected manifest format")
        sys.exit(1)
    except Exception as e:
        log.error("Failed to load manifest %s: %s", path, e)
        sys.exit(1)

def download_cdn(path: str, timeout: int = 30) -> Optional[bytes]:
    """Download via HF CDN (no auth)."""
    url = f"{CDN_BASE}/{DATASET_REPO}/resolve/main/{path}"
    try:
        resp = requests.get(
            url,
            timeout=timeout,
            headers={"User-Agent": "axentx-surrogate-ingest"},
        )
        if resp.status_code == 429:
            retry_after = int(resp.headers.get("Retry-After", 360))
            log.warning("CDN 429, sleeping %ss", retry_after)
            time.sleep(retry_after)
            return download_cdn(path, timeout)
        resp.raise_for_status()
        return resp.content
    except Exception as e:
        log.error("CDN download failed %s: %s", path, e)
        return None

def parse_to_pair(raw: bytes, filename: str) -> Optional[Dict[str, str]]:
    """
    Project arbitrary file to {prompt, response}.
    Extend per known schema as needed.
    """
    import io
    try:
        import pyarrow.parquet as pq
        table = pq.read_table(io.BytesIO(raw))
        cols = table.column_names
        # Common patterns
        if "prompt" in cols and "response" in cols:
            rec = {
                "prompt": str(table["prompt"].to_pylist()[0]),
                "response": str(table["response"].to_pylist()[0]),
            }
            return rec
        # fallback: try text/jsonl single-record
    except Exception:
        pass

    try:
        text = raw.decode("utf-8")
        # try jsonl single line
        for line in text.strip().splitlines():
            line = line.strip()
            if not line:
                continue
            obj = json.loads(line)
            if isinstance(obj, dict):
                p = obj.get("prompt") or obj.get("input") or obj.get("question")
                r = obj.get("response") or obj.get("output") or obj.get("answer")
                if p is not None and r is not None:
                    return {"prompt": str(p), "response": str(r)}
    except Exception:
        pass

    log.warning("Could not project %s to prompt/response", filename)
    return None

# ---------- main ----------
def main() -> None:
    login(token=HF_TOKEN)
    api = HfApi()
    dedup = DedupStore()

    files = load_manifest(MANIFEST_PATH)
    my_files = [f for f in files if shard_for_path(f) == SHARD_ID]
    log.info("Shard %s/%s -> %s files", SHARD_ID, SHARD_TOTAL, len(my_files))

    written = 0
    skipped_dup = 0
    failed = 0

    with OUTPUT_FILE.open("w", buffering=1) as out:
        for i, path in enumerate(my_files):
            if i > 0 and i % 50 == 0:
                log.info("Progress %s/%s", i, len(my_files))

            raw = download_cdn(path)
            if raw is None:
                failed += 1
                continue

            pair = parse_to_pair(raw, path)
            if pair is None:
                failed += 1
                continue

            h = content_hash(pair)
            if dedup.exists(h):
                skipped_dup
