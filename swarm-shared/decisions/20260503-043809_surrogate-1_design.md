# surrogate-1 / design

## Final Implementation Plan (≤2h)

**Highest-value improvement**: Replace fragile shell-based ingestion with a **manifest-driven, CDN-bypass Python worker** that eliminates HF API rate limits during training data loads and prevents mixed-schema CastErrors.

### Changes

1. **Add `manifest.json` generation** (single API call from orchestrator)  
   - `bin/build_manifest.py` — lists one date folder via `list_repo_tree(recursive=False)`, writes `{"date": "...", "files": [...]}` to `manifest.json`.  
   - Run once per cron tick (or manually) from Mac/CI; commit alongside code so Lightning training uses it without any HF API calls.

2. **Replace `dataset-enrich.sh` with `worker.py`**  
   - Deterministic shard assignment: `hash(slug) % 16 == SHARD_ID`.  
   - CDN-only downloads: `https://huggingface.co/datasets/axentx/surrogate-1-training-pairs/resolve/main/<path>`.  
   - Per-file schema projection: read only `{prompt, response}` fields; ignore extra columns to avoid `pyarrow.CastError`.  
   - Central dedup via existing `lib/dedup.py` (SQLite).  
   - Output: `batches/public-merged/{date}/shard{N}-{HHMMSS}.jsonl`.

3. **Update GitHub Actions matrix**  
   - Keep 16 shards; each runs `python worker.py --shard N --manifest manifest.json`.  
   - No HF API list calls inside workers → zero 429 risk during ingestion.

4. **Add Lightning training snippet**  
   - `train.py` loads the same `manifest.json`, uses CDN URLs with `datasets.load_dataset('json', data_files=...)` or direct `parquet` reads — zero HF API calls during epoch loops.

## Code Snippets

### 1. `bin/build_manifest.py`
```python
#!/usr/bin/env python3
"""
Generate manifest.json for a single date folder.
Run from Mac/CI (once per cron tick) to avoid HF API rate limits in workers.
"""
import json, os, sys
from datetime import datetime, timezone
from huggingface_hub import HfApi

REPO = "datasets/axentx/surrogate-1-training-pairs"
DATE = datetime.now(timezone.utc).strftime("%Y-%m-%d")
OUT = "manifest.json"

def main():
    api = HfApi()
    # Single API call: non-recursive, one folder
    entries = api.list_repo_tree(
        repo_id=REPO,
        path=f"batches/public-raw/{DATE}",
        repo_type="dataset",
        recursive=False,
    )
    files = [e.path for e in entries if e.path.endswith((".jsonl", ".parquet"))]
    manifest = {"date": DATE, "files": sorted(files)}
    with open(OUT, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Wrote {OUT} with {len(files)} files for {DATE}")

if __name__ == "__main__":
    main()
```

### 2. `bin/worker.py`
```python
#!/usr/bin/env python3
"""
CDN-bypass worker: deterministic shard, schema-projection, dedup, upload.
Usage: python worker.py --shard 3 --manifest manifest.json
"""
import argparse, hashlib, json, os, sys, time
from pathlib import Path

import numpy as np
import pyarrow as pa
import pyarrow.parquet as pq
import requests
from datasets import load_dataset

HF_REPO = "datasets/axentx/surrogate-1-training-pairs"
CDN_ROOT = f"https://huggingface.co/{HF_REPO}/resolve/main"

# Local imports
sys.path.insert(0, str(Path(__file__).parent))
from lib.dedup import DedupStore  # central sqlite store

def shard_for(slug: str, n_shards: int = 16) -> int:
    return int(hashlib.md5(slug.encode()).hexdigest(), 16) % n_shards

def cdn_url(path: str) -> str:
    return f"{CDN_ROOT}/{path}"

def project_to_pair(obj, schema_hint=None):
    """Return (prompt, response) only; ignore extra fields."""
    prompt = obj.get("prompt") or obj.get("input") or obj.get("question") or ""
    response = obj.get("response") or obj.get("output") or obj.get("answer") or ""
    return {"prompt": str(prompt), "response": str(response)}

def process_file(path: str, dedup: DedupStore):
    url = cdn_url(path)
    pairs = []
    if path.endswith(".jsonl"):
        # Stream line-by-line to keep memory low
        with requests.get(url, stream=True, timeout=60) as r:
            r.raise_for_status()
            for line in r.iter_lines(decode_unicode=True):
                if not line.strip():
                    continue
                try:
                    obj = json.loads(line)
                except json.JSONDecodeError:
                    continue
                pair = project_to_pair(obj)
                if not pair["prompt"] or not pair["response"]:
                    continue
                md5 = hashlib.md5(f"{pair['prompt']}\0{pair['response']}".encode()).hexdigest()
                if not dedup.seen(md5):
                    pairs.append(pair)
                    dedup.add(md5)
    elif path.endswith(".parquet"):
        # Use pyarrow directly; project columns to avoid CastError on mixed schemas
        try:
            pf = pq.read_table(url, columns=["prompt", "response"], use_threads=False)
        except (pa.ArrowInvalid, KeyError):
            # Fallback: read all and project
            pf = pq.read_table(url, use_threads=False)
            cols = set(pf.column_names)
            prompt_col = next((c for c in ["prompt", "input", "question"] if c in cols), None)
            response_col = next((c for c in ["response", "output", "answer"] if c in cols), None)
            if not prompt_col or not response_col:
                return []
            pf = pf.select([prompt_col, response_col]).rename_columns(["prompt", "response"])
        df = pf.to_pandas()
        for _, row in df.iterrows():
            pair = {"prompt": str(row["prompt"]), "response": str(row["response"])}
            if not pair["prompt"] or not pair["response"]:
                continue
            md5 = hashlib.md5(f"{pair['prompt']}\0{pair['response']}".encode()).hexdigest()
            if not dedup.seen(md5):
                pairs.append(pair)
                dedup.add(md5)
    return pairs

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--shard", type=int, required=True, help="Shard ID 0..15")
    parser.add_argument("--manifest", required=True, help="Path to manifest.json")
    parser.add_argument("--out-dir", default="output", help="Output directory")
    args = parser.parse_args()

    with open(args.manifest) as f:
        manifest = json.load(f)

    date = manifest["date"]
    files = manifest["files"]
    my_files = [p for p in files if shard_for(p) == args.shard]

    dedup = DedupStore()  # central sqlite store
    os.makedirs(args.out_dir, exist_ok=True)
    stamp = time.strftime("%H%M%S")
    out_path = Path(args.out_dir) / f"shard{args.shard}-{stamp}.jsonl"

    total = 0
    with out_path.open("w") as out_f:
        for path in my_files:
            try:
                pairs = process_file(path, dedup)
                for p in pairs:
                    out_f.write(json.dumps(p, ensure_ascii=False) + "\n")
                    total += 1
            except Exception as exc:
                print(f"ERROR processing {path}: {exc}", file=sys.stderr)

    print(f"Shard {args.shard}: wrote {total} pairs to {out_path}")

if __name__ == "__main__":
    main()
```
