### Key design decisions (resolve contradictions)

- **One API call only** (both candidates agree): use `list_repo_tree(..., recursive=False)` for the date folder.  
- **CDN-only during training** (both agree): embed `file_manifest.json`; never call HF APIs inside training loop.  
- **Schema**: include `cdn_url`, `path`, `size`, `repo` (Candidate 2) — strictly richer and more actionable than Candidate 1’s flat list.  
- **Streaming + retries** (Candidate 2): mandatory for robustness; Candidate 1 omitted it. Use `requests` with `urllib3` Retry and backoff.  
- **Parquet/JSONL support** (Candidate 2): Candidate 1 omitted file-format specifics; include both.  
- **CLI + tests** (Candidate 2): Candidate 1 omitted; include dry-run and validation step.  
- **No implicit embedding of manifest into `train.py`** (Candidate 1): prefer explicit `--manifest` flag (Candidate 2) for reusability.

#### tools/snapshot_manifest.py
```python
#!/usr/bin/env python3
"""
snapshot_manifest.py

List one date-partition of axentx/surrogate-1-training-pairs via a single
HF API call and emit file_manifest.json with CDN URLs and metadata.

Usage:
    python tools/snapshot_manifest.py --date 2026-05-03 --out file_manifest.json
"""

import argparse
import json
import os
import sys
from datetime import datetime, timezone
from pathlib import Path

from huggingface_hub import HfApi

REPO_ID = "datasets/axentx/surrogate-1-training-pairs"
CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def build_manifest(date: str, out_path: Path) -> None:
    api = HfApi()
    prefix = f"batches/public-merged/{date}/"

    # Single API call: non-recursive listing for the date folder
    entries = api.list_repo_tree(
        repo_id=REPO_ID,
        path=prefix,
        repo_type="dataset",
        recursive=False,
    )

    files = []
    total_size = 0
    for e in entries:
        if getattr(e, "type", None) != "file":
            continue
        path = e.path
        size = getattr(e, "size", 0)
        total_size += size
        files.append(
            {
                "repo": REPO_ID,
                "path": path,
                "filename": os.path.basename(path),
                "size": size,
                "cdn_url": CDN_TEMPLATE.format(repo=REPO_ID, path=path),
            }
        )

    if not files:
        print(f"Warning: no files found under {prefix}", file=sys.stderr)

    manifest = {
        "generated_at": datetime.now(timezone.utc).isoformat(),
        "date": date,
        "prefix": prefix,
        "n_files": len(files),
        "total_size": total_size,
        "files": files,
    }

    out_path.parent.mkdir(parents=True, exist_ok=True)
    with open(out_path, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2)

    print(f"Wrote {len(files)} files ({total_size / (1024**2):.1f} MB) -> {out_path}")

def main() -> None:
    parser = argparse.ArgumentParser(description="Create CDN manifest for a date partition")
    parser.add_argument("--date", required=True, help="Date folder, e.g. 2026-05-03")
    parser.add_argument("--out", default="file_manifest.json", help="Output JSON path")
    args = parser.parse_args()

    try:
        build_manifest(args.date, Path(args.out))
    except Exception as exc:
        print(f"Error: {exc}", file=sys.stderr)
        sys.exit(1)

if __name__ == "__main__":
    main()
```

from urllib3.util import Retry

def make_session() -> requests.Session:
    session = requests.Session()
    retries = Retry(
        total=5,
        backoff_factor=1,
        status_forcelist=[500, 502, 503, 504],
        allowed_methods=["GET", "HEAD"],
    )
    session.mount("https://", HTTPAdapter(max_retries=retries))
    session.mount("http://", HTTPAdapter(max_retries=retries))
    return session

SESSION = make_session()
