# airship / quality

## Final Synthesized Implementation

**Highest-Value Improvement**: Implement a unified `airship discover` CLI and `/discover` HTTP endpoint that generates a **deterministic, CDN-only manifest** for a HuggingFace dataset `repo_id` and `date_folder`. This enables the surrogate-1 training pipeline to pre-list files once, embed CDN paths in training scripts, and completely bypass HF API rate limits during distributed training.

---

### Unified Implementation Plan (≤2 hours)

**1. Shared Core Module** (`airship/core/hf_cdn.py`)
- Single source of truth for manifest generation.
- Uses `list_repo_tree(recursive=False)` for one API call.
- Deterministic sort by path.
- Validates inputs to prevent path traversal.
- Constructs CDN URLs in canonical format.

**2. CLI Command** (`airship/cli/discover.py`)
- `airship discover <repo_id> <date_folder> [--output manifest.json]`
- Reuses core module.
- Clear exit codes and stderr messages.
- Human-readable summary on success.

**3. FastAPI Endpoint** (`arkship/routes/discover.py`)
- `GET /discover?repo_id=...&date_folder=...&format=json|html`
- 5-minute in-memory cache keyed by `(repo_id, date_folder)`.
- Returns JSON or simple HTML table with sortable columns.
- Proper HTTP error codes (400/500).

**4. Frontend Template** (`arkship/templates/discover.html`)
- Minimal form + results table.
- CDN links open in new tab.
- No JS dependencies.

**5. Wiring & Quality**
- Register CLI in `airship/cli/__init__.py` or `__main__.py`.
- Mount router in Arkship FastAPI app.
- Input sanitization, HF 429 handling via single-call backoff.
- CDN URL format: `https://huggingface.co/datasets/{repo_id}/resolve/main/{path}`

### Final Code

#### `airship/core/hf_cdn.py`
```python
import os
import re
from typing import List, Dict, Any
from huggingface_hub import HfApi
from huggingface_hub.utils import HFError

HF_API = HfApi()
_ALLOWED_REPO_ID = re.compile(r"^(datasets/)?[a-zA-Z0-9][a-zA-Z0-9._-]+/[a-zA-Z0-9._-]+$")
_ALLOWED_PATH = re.compile(r"^[a-zA-Z0-9._/-]+$")

def _validate_repo_id(repo_id: str) -> str:
    """Normalize and validate repo_id."""
    if not repo_id:
        raise ValueError("repo_id is required")
    if not repo_id.startswith("datasets/"):
        repo_id = f"datasets/{repo_id}"
    if not _ALLOWED_REPO_ID.match(repo_id):
        raise ValueError(f"Invalid repo_id format: {repo_id}")
    return repo_id

def _validate_path(path: str) -> str:
    """Validate path segment to prevent traversal."""
    if not path or not _ALLOWED_PATH.match(path):
        raise ValueError(f"Invalid path: {path}")
    normalized = os.path.normpath(path).lstrip("/")
    if ".." in normalized or normalized.startswith("/"):
        raise ValueError(f"Invalid path: {path}")
    return normalized

def build_cdn_manifest(repo_id: str, date_folder: str) -> List[Dict[str, Any]]:
    """
    Build a deterministic CDN-only manifest for files in repo_id/date_folder.
    Single HF API call (list_repo_tree, non-recursive).
    """
    repo_id = _validate_repo_id(repo_id)
    date_folder = _validate_path(date_folder)

    try:
        tree = HF_API.list_repo_tree(
            repo_id=repo_id,
            path=date_folder,
            recursive=False
        )
    except HFError as e:
        raise RuntimeError(f"HF API error listing {repo_id}/{date_folder}: {e}") from e

    manifest = []
    for item in tree:
        if item.type != "file":
            continue
        path = item.path
        cdn_url = f"https://huggingface.co/{repo_id}/resolve/main/{path}"
        entry = {
            "path": path,
            "filename": os.path.basename(path),
            "cdn_url": cdn_url,
            "size": getattr(item, "size", None),
            "lfs": getattr(item, "lfs", None),
        }
        manifest.append(entry)

    manifest.sort(key=lambda x: x["path"])
    return manifest
```

#### `airship/cli/discover.py`
```python
import argparse
import json
import sys
from airship.core.hf_cdn import build_cdn_manifest

def main() -> None:
    parser = argparse.ArgumentParser(
        description="Build CDN-only file manifest for a HuggingFace dataset folder."
    )
    parser.add_argument("repo_id", help="HF dataset repo ID (e.g. 'myorg/mydata')")
    parser.add_argument("date_folder", help="Date folder in repo (e.g. '2026-04-29')")
    parser.add_argument("--output", default="manifest.json", help="Output JSON path")
    args = parser.parse_args()

    try:
        manifest = build_cdn_manifest(args.repo_id, args.date_folder)
        with open(args.output, "w", encoding="utf-8") as f:
            json.dump(
                {
                    "repo_id": args.repo_id,
                    "date_folder": args.date_folder,
                    "files": manifest,
                },
                f,
                indent=2,
                ensure_ascii=False,
            )
        print(f"OK: {len(manifest)} files -> {args.output}", file=sys.stderr)
        sys.exit(0)
    except Exception as e:
        print(f"ERROR: {e}", file=sys.stderr)
        sys.exit(1)

if __name__ == "__main__":
    main()
```

from fastapi import APIRouter, HTTPException, Query
from fastapi.responses import JSONResponse, HTMLResponse
from airship.core.hf_cdn import build_cdn_manifest
import time

router = APIRouter()
_CACHE: dict = {}
_CACHE_TTL = 300  # 5 minutes

@router.get("/discover")
async def discover(
    repo_id: str = Query(..., description="HF dataset repo ID"),
    date_folder: str = Query(..., description="Date folder in repo"),
    format: str = Query("json", regex="^(json|html)$"),
):
    key = (repo_id, date_folder)
    now = time.time()

    if key in _CACHE:
        ts, cached = _CACHE[key]
        if now - ts < _CACHE_TTL:
            if format == "html":
                return _render_html(repo_id, date_folder, cached)
            return JSONResponse(
                content={"repo_id": repo_id, "date_folder": date_folder, "manifest": cached}
            )

    try:
        manifest = build_cdn_manifest(repo_id, date_folder)
    except ValueError as e:
        raise HTTPException(status_code=400, detail=str(e))
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

    _CACHE[key] = (now, manifest)

    if format == "html":
        return _render_html(repo_id, date_folder, manifest)
    return JSONResponse(
        content={"repo_id": repo_id, "date_folder": date_folder, "manifest": manifest}
    )
