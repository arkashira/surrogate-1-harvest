## Final Synthesis (Best Parts + Corrected + Actionable)

**What to ship (≤2h)**  
Add a CDN-first training slice to Surrogate that:

1. Exposes `/api/training/file-list` (POST) to list one HF date-folder once and persist `training/file-lists/{dateFolder}.json`.  
2. Exposes `/api/training/start-cdn` (POST) to launch (or resume) a Lightning Studio that trains using **CDN-only URLs** (`https://huggingface.co/datasets/{repo}/resolve/main/{file}`) with zero HF API calls during data loading.  
3. Reuses an existing running studio by name; if stopped, restarts it on `Machine.L40S`.  
4. Projects every parquet file to strict `{prompt, response}` schema (drop extras) to keep ingestion clean and deterministic.

Why this wins:  
- Directly fixes HF API 429s and Lightning idle-timeout waste.  
- Minimal surface area; no auth needed for CDN reads.  
- Reuse prevents quota churn and cold-start delays.

### 1) Backend: file-list endpoint (FastAPI)

```python
# surrogate/api/training.py
import json
from pathlib import Path

from fastapi import APIRouter, HTTPException
from huggingface_hub import HfApi

router = APIRouter()
HF_API = HfApi()
LIST_DIR = Path("training/file-lists")
LIST_DIR.mkdir(parents=True, exist_ok=True)

@router.post("/file-list")
async def create_file_list(repo: str, path: str):
    """
    repo: e.g. "datasets/username/repo"
    path: e.g. "batches/mirror-merged/2026-05-03"
    """
    try:
        tree = HF_API.list_repo_tree(repo=repo, path=path, recursive=False)
        files = [f.rstrip("/") for f in tree if f and not f.endswith("/")]
        if not files:
            raise HTTPException(status_code=404, detail="No files found")

        date_folder = path.rstrip("/").split("/")[-1]
        out_path = LIST_DIR / f"{date_folder}.json"
        payload = {"repo": repo, "dateFolder": date_folder, "files": files}
        out_path.write_text(json.dumps(payload, indent=2))
        return payload
    except Exception as exc:
        raise HTTPException(status_code=500, detail=str(exc))
```

class CDNTrainingWork(LightningWork):
    def __init__(self, file_list_path: str, **kwargs):
        super().__init__(**kwargs)
        self.file_list_path = file_list_path
        self.parallel = False
        # Use L40S by default (widely available on free/paid tiers).
        # If you have paid quota and need H200, change to Machine.H200.
        self._cloud_build_config = {"machine": "L40S"}

    def run(self):
        import json
        from pathlib import Path
        import tempfile

        import pyarrow.parquet as pq
        import requests
        import torch
       from torch.utils.data import IterableDataset, DataLoader

        cfg_path = Path(self.file_list_path)
        if not cfg_path.exists():
            raise FileNotFoundError(f"file-list not found: {cfg_path}")

        with open(cfg_path) as f:
            cfg = json.load(f)
        repo = cfg["repo"]
        files = cfg["files"]
        base = "https://huggingface.co/datasets"

        class CDNParquetDataset(IterableDataset):
            def __init__(self, repo, files, base=base):
                self.repo = repo
                self.files = files
                self.base = base

            def _project_row(self, row):
                # Strict projection; drop extras.
                prompt = row.get("prompt") or row.get("input") or row.get("question") or ""
                response = row.get("response") or row.get("output") or row.get("answer") or ""
                return {"prompt": str(prompt), "response": str(response)}

            def __iter__(self):
                for fname in self.files:
                    url = f"{self.base}/{self.repo}/resolve/main/{fname}"
                    with tempfile.NamedTemporaryFile(suffix=".parquet", delete=False) as tmp:
                        with requests.get(url, stream=True, timeout=60) as r:
                            r.raise_for_status()
                            for chunk in r.iter_content(chunk_size=16384):
                                tmp.write(chunk)
                        tmp_path = tmp.name
                    try:
                        table = pq.read_table(tmp_path, columns=["prompt", "response", "input", "output", "question", "answer"])
                        df = table.to_pandas()
                        for _, row in df.iterrows():
                            yield self._project_row(row)
                    finally:
                        Path(tmp_path).unlink(missing_ok=True)

        dataset = CDNParquetDataset(repo, files)
        loader = DataLoader(dataset, batch_size=8, num_workers=0)

class CDNTrainingFlow(LightningFlow):
    def __init__(self):
        super().__init__()
        self.work = None

    def configure_layout(self):
        return [{"name": "CDN Training", "content": "Training running via CDN."}]

    def run(self, file_list_path: str):
        if self.work is None or not self.work.has_started:
            self.work = CDNTrainingWork(file_list_path, cloud_build_config={"machine": "L40S"})
