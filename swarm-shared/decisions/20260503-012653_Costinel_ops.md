# Costinel / ops

**Final merged implementation (best of both)**

- **Tech stack**: React + TypeScript (Candidate 2) — better type safety, modern UI primitives, and easier to maintain.  
- **API contract + fallback**: Keep Candidate 1’s simple, resilient contract and fallback JSON.  
- **UI/UX**: Use existing Card/Skeleton/Badge primitives (Candidate 2) with Candidate 1’s clear proposal fields (impact, effort, reason) and a concise hub header.  
- **Correctness + actionability**: Prefer concrete, immediately implementable choices; resolve contradictions in favor of working defaults (fetch in `useEffect`, scoped error/fallback, no backend changes).

## 3) Component

`src/components/dashboard/TopHubSignalPanel.tsx`

```tsx
import React, { useEffect, useState, useCallback } from 'react';
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card';
import { Badge } from '@/components/ui/badge';
import { Skeleton } from '@/components/ui/skeleton';
import { AlertCircle } from 'lucide-react';

type Proposal = {
  id: string;
  title: string;
  action?: string;
  impact: 'high' | 'medium' | 'low';
  reason: string;
  effort: 'low' | 'medium' | 'high';
};

type Hub = {
  id: string;
  label: string;
  type: string;
  score: number;
  description: string;
  proposals: Proposal[];
};

type TopHubResponse = {
  hub: Hub | null;
};

type Props = {
  /** Optional: inject data for tests/stories */
  initialData?: Hub | null;
  /** Optional: override fetcher */
  fetcher?: () => Promise<Hub | null>;
};

const impactColors: Record<string, string> = {
  high: 'bg-red-100 text-red-800',
  medium: 'bg-amber-100 text-amber-800',
  low: 'bg-emerald-100 text-emerald-800',
};

const effortColors: Record<string, string> = {
  low: 'bg-emerald-100 text-emerald-800',
  medium: 'bg-amber-100 text-amber-800',
  high: 'bg-red-100 text-red-800',
};

export function TopHubSignalPanel({ initialData = null, fetcher }: Props) {
  const [hub, setHub] = useState<Hub | null>(initialData);
  const [loading, setLoading] = useState(!initialData);
  const [error, setError] = useState(false);

  const load = useCallback(async () => {
    // If caller provided a fetcher, use it
    if (fetcher) {
      try {
        const result = await fetcher();
        setHub(result);
        setError(false);
      } catch {
        setError(true);
        setHub(null);
      } finally {
        setLoading(false);
      }
      return;
    }

    // Default loader: primary endpoint -> fallback JSON
    setLoading(true);
    setError(false);
    try {
      const res = await fetch('/api/knowledge-graph/top-hub', {
        headers: { Accept: 'application/json' },
        cache: 'no-store',
      });
      if (!res.ok) throw new Error(`HTTP ${res.status}`);
      const json: TopHubResponse = await res.json();
      if (json?.hub) {
        setHub(json.hub);
        return;
      }
      throw new Error('No hub in response');
    } catch {
      // fallback
      try {
        const res = await fetch('/data/top-hub-fallback.json', {
          headers: { Accept: 'application/json' },
          cache: 'default',
        });
        if (res.ok) {
          const json: TopHubResponse = await res.json();
          setHub(json?.hub || null);
          setError(false);
          return;
        }
      } catch {
        // ignore
      }
      setHub(null);
      setError(true);
    } finally {
      setLoading(false);
    }
  }, [fetcher]);

  useEffect(() => {
    if (!initialData) load();
  }, [load, initialData]);

  if (loading) {
    return (
      <Card>
        <CardHeader>
          <Skeleton className="h-5 w-32" />
        </CardHeader>
        <CardContent className="space-y-3">
          <Skeleton className="h-10 w-full" />
          <Skeleton className="h-10 w-full" />
        </CardContent>
      </Card>
    );
  }

  if (error && !hub) {
    return (
      <Card>
        <CardContent className="flex items-center gap-2 py-6 text-sm text-muted-foreground">
          <AlertCircle className="h-4 w-4" />
          Unable to load top-hub signals.
        </CardContent>
      </Card>
    );
  }

  if (!hub) {
    return (
      <Card>
        <CardContent className="py-6 text-sm text-muted-foreground">
          No hub data available.
        </CardContent>
      </Card>
    );
  }

  return (
    <Card>
      <CardHeader className="pb-3">
        <div className="flex items-center gap-2">
          <Badge variant="secondary">Top Hub</Badge>
          <CardTitle className="text-lg">{hub.label}</CardTitle>
        </div>
        <p className="text-sm text-muted-foreground">
          Relevance: {Math.round((hub.score || 0) * 100)}%
        </p>
        <p className="text-sm">{hub.description}</p>
      </CardHeader>

        <div className="space-y-2">
