# workio / frontend

## Final Synthesized Implementation

**Chosen approach**: Merge the strongest elements from both candidates into one production-ready solution.

**Key decisions**:
- **Optimistic UI + explicit synced/failed states** (both candidates) — gives immediate feedback and prevents double-taps.
- **Client-side idempotency key per punch** (both) — required for safe retries and backend dedupe.
- **Local persistence of last punch** (Candidate 1) — survives refresh and re-mount; Candidate 2’s `usePunch` hook is merged in to keep logic reusable.
- **Disabled button + aria-busy + clear loading label** (both) — prevents races and improves accessibility.
- **Explicit rollback on failure + retry affordance** (Candidate 2) — more actionable than Candidate 1’s silent revert; keeps failed entry locally so user can retry.
- **Storage key namespaced by tenant+user** (Candidate 1) — prevents cross-tenant contamination.
- **Backend contract: `Idempotency-Key` header** (Candidate 1) — standard, simple, and effective.

### 1. Reusable hook: `usePunch`

File: `workio/src/features/attendance/hooks/usePunch.ts`

```ts
// workio/src/features/attendance/hooks/usePunch.ts
import { useCallback, useRef } from 'react';
import { postPunch } from '@/features/attendance/lib/api';

export type PunchType = 'CLOCK_IN' | 'CLOCK_OUT';

export interface PunchRecord {
  idempotencyKey: string;
  type: PunchType;
  timestamp: string;
  synced: boolean;
}

export function getLastPunchStorageKey(tenantId: string, userId: string) {
  return `workio:last-punch:${tenantId}:${userId}`;
}

export function usePunch(tenantId: string, userId: string) {
  // in-flight guard to prevent concurrent calls from multiple sources
  const inFlightRef = useRef(false);

  const punch = useCallback(
    async (type: PunchType): Promise<PunchRecord> => {
      if (inFlightRef.current) {
        throw new Error('Punch already in progress');
      }

      const idempotencyKey = crypto.randomUUID();
      const timestamp = new Date().toISOString();
      const record: PunchRecord = {
        idempotencyKey,
        type,
        timestamp,
        synced: false,
      };

      inFlightRef.current = true;
      try {
        await postPunch({ type, idempotencyKey });
        record.synced = true;
        return record;
      } catch (err) {
        record.synced = false;
        throw err;
      } finally {
        inFlightRef.current = false;
      }
    },
    [tenantId, userId]
  );

  return { punch };
}
```

### 2. UI component: `PunchButton`

File: `workio/src/features/attendance/components/PunchButton.tsx`

```tsx
// workio/src/features/attendance/components/PunchButton.tsx
import { useEffect, useState, useCallback } from 'react';
import { useToast } from '@/hooks/use-toast';
import { usePunch, getLastPunchStorageKey, type PunchRecord } from '@/features/attendance/hooks/usePunch';

export function PunchButton({
  tenantId,
  userId,
}: {
  tenantId: string;
  userId: string;
}) {
  const { toast } = useToast();
  const { punch } = usePunch(tenantId, userId);
  const [lastPunch, setLastPunch] = useState<PunchRecord | null>(null);
  const [pending, setPending] = useState(false);

  // Hydrate last punch from localStorage
  useEffect(() => {
    try {
      const raw = localStorage.getItem(getLastPunchStorageKey(tenantId, userId));
      if (raw) setLastPunch(JSON.parse(raw));
    } catch {
      // ignore corrupted storage
    }
  }, [tenantId, userId]);

  const saveLastPunch = useCallback(
    (record: PunchRecord | null) => {
      try {
        if (record) {
          localStorage.setItem(getLastPunchStorageKey(tenantId, userId), JSON.stringify(record));
        } else {
          localStorage.removeItem(getLastPunchStorageKey(tenantId, userId));
        }
      } catch {
        // ignore storage failures
      }
    },
    [tenantId, userId]
  );

  const handlePunch = useCallback(async () => {
    if (pending) return;
    const type = lastPunch?.type === 'CLOCK_IN' ? 'CLOCK_OUT' : 'CLOCK_IN';
    setPending(true);

    // Optimistic update
    const optimistic: PunchRecord = {
      idempotencyKey: crypto.randomUUID(),
      type,
      timestamp: new Date().toISOString(),
      synced: false,
    };
    setLastPunch(optimistic);
    saveLastPunch(optimistic);

    try {
      const confirmed = await punch(type);
      // confirmed.synced === true
      setLastPunch(confirmed);
      saveLastPunch(confirmed);
      toast({
        title: type === 'CLOCK_IN' ? 'เข้างานแล้ว' : 'ออกงานแล้ว',
        description: new Date(confirmed.timestamp).toLocaleTimeString(),
      });
    } catch (err) {
      // Keep failed optimistic record locally so user can retry explicitly
      const failed: PunchRecord = { ...optimistic, synced: false };
      setLastPunch(failed);
      saveLastPunch(failed);
      toast({
        variant: 'destructive',
        title: 'เกิดข้อผิดพลาด',
        description: 'ไม่สามารถบันทึกการลงเวลาได้',
      });
    } finally {
      setPending(false);
    }
  }, [pending, lastPunch, punch, saveLastPunch, toast]);

  const isClockedIn = lastPunch?.type === 'CLOCK_IN' && lastPunch.synced;
  const showUnsynced = lastPunch && !lastPunch.synced;

  return (
    <div className="flex flex-col items-center gap-2">
      <button
        onClick={handlePunch}
        disabled={pending}
        aria-busy={pending}
        className={`
          rounded-full px-8 py-4 text-lg font-semibold text-white shadow-lg transition-all
          ${pending ? 'cursor-not-allowed opacity-70' : ''}
          ${isClockedIn ? 'bg-red-500 hover:bg-red-600' : 'bg-green-500 hover:bg-green-600'}
        `}
      >
        {pending
          ? 'กำลังบันทึก...'
          : isClockedIn
          ? 'ออกงาน'
          : 'เข้างาน'}
      </button>

      {lastPunch && (
        <div className="text-xs text-muted-foreground">
          ครั้งล่าสุด: {lastPunch.type === 'CLOCK_IN' ? 'เข้างาน' : 'ออกงาน'}{' '}
          {new Date(lastPunch.timestamp).toLocaleString()}
          {showUnsynced && <span className="ml-1 text-amber-600">(ยังไม่ซิงค์)</span>}
        </div>
      )}
    </div>
  );
}
```

### 3. API helper

File: `workio/src/features/attendance/lib/api.ts`

```ts
// workio/src/features/attendance/lib/api.ts
export async function postPunch(payload: { type: 'CLOCK_IN' | 'CLOCK_OUT'; idempotencyKey: string }) {
  const res = await fetch('/api/punches', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Idempotency-Key': payload.idempotencyKey,
    },
    body: JSON.stringify(payload),
  });

  if (!res.ok) {
    throw new Error('Punch failed');
  }

  return res.json();
}
```
