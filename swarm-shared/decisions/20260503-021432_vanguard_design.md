# vanguard / design

# Final Synthesized Solution

## 1. Diagnosis (Consolidated)
- **No manifest cache**: every training run calls `list_repo_tree`, burning HF API quota and risking 429s.
- **Heterogeneous schemas**: `load_dataset(..., streaming=True)` across mixed repos causes `pyarrow.CastError`.
- **No CDN fetch path**: training continues to use authenticated API calls instead of public CDN URLs.
- **Lightning Studio reuse missing**: scripts may spawn new studios and waste quota.
- **HF write rate risk**: ingestion can hit the 128 commit/hr cap on a single repo.

## 2. Design Decisions (Correctness + Actionability)
- **Persisted manifest**: one authenticated `list_repo_tree` per `(repo, dateFolder)`; saved as JSON.
- **CDN-only training path**: training scripts consume only public CDN URLs (`https://huggingface.co/datasets/.../resolve/main/...`).
- **Schema normalization at parse time**: project heterogeneous inputs to `{prompt, response}` before materializing batches; drop unusable rows.
- **Lightning Studio reuse**: detect running studio by name and reuse; create only if absent.
- **Deterministic sibling-repo sharding**: hash-based selection across 5 repos to stay under HF commit caps.
- **Streaming-friendly, chunked parsing**: avoid full materialization; support JSONL/JSON/Parquet with safe fallbacks.

# /opt/axentx/vanguard/src/vanguard/data_manifest.py
from __future__ import annotations

import json
import hashlib
import os
from pathlib import Path
from typing import Dict, List, Iterator, Any

from huggingface_hub import HfApi, list_repo_tree, hf_hub_download

HF_API = HfApi()
MANIFEST_DIR = Path(os.getenv("VANGUARD_MANIFEST_DIR", "/opt/axentx/vanguard/manifests"))
MANIFEST_DIR.mkdir(parents=True, exist_ok=True)

SIBLING_REPOS = [
    "axentx/vanguard-ds-0",
    "axentx/vanguard-ds-1",
    "axentx/vanguard-ds-2",
    "axentx/vanguard-ds-3",
    "axentx/vanguard-ds-4",
]


def _manifest_path(repo: str, date_folder: str) -> Path:
    safe_repo = repo.replace("/", "_")
    return MANIFEST_DIR / f"{safe_repo}_{date_folder}.json"


def persist_file_list(repo: str, date_folder: str, token: str | None = None) -> List[str]:
    """
    Single authenticated list_repo_tree for one date folder.
    Manifest: {"repo":..., "date_folder":..., "files":[...]}
    Returns list of file paths.
    """
    manifest_path = _manifest_path(repo, date_folder)
    if manifest_path.exists():
        try:
            with manifest_path.open("r") as f:
                data = json.load(f)
                if isinstance(data, dict) and "files" in data:
                    return data["files"]
        except Exception:
            pass  # regenerate on corruption

    tree = list_repo_tree(repo=repo, path=date_folder, recursive=False, token=token)
    files = [
        str(entry.path)
        for entry in tree
        if entry.type == "file" and Path(entry.path).suffix.lower() in {".jsonl", ".json", ".parquet"}
    ]

    manifest = {"repo": repo, "date_folder": date_folder, "files": files}
    with manifest_path.open("w") as f:
        json.dump(manifest, f, indent=2)
    return files


def cdn_urls(repo: str, date_folder: str, files: List[str]) -> Iterator[str]:
    """Yield public CDN URLs (no auth) for training to fetch."""
    for fpath in files:
        yield f"https://huggingface.co/datasets/{repo}/resolve/main/{fpath}"


def project_to_prompt_response(raw: Dict[str, Any]) -> Dict[str, str] | None:
    """
    Best-effort projection to {prompt, response}.
    Returns None if unusable.
    """
    if not isinstance(raw, dict):
        return None

    prompt = raw.get("prompt") or raw.get("input") or raw.get("question") or raw.get("instruction")
    response = raw.get("response") or raw.get("output") or raw.get("answer") or raw.get("completion")

    if prompt is None or response is None:
        return None
    return {"prompt": str(prompt), "response": str(response)}


def pick_sibling_repo(slug: str) -> str:
    """Deterministic sharding across sibling repos to avoid HF commit cap."""
    digest = hashlib.sha256(slug.encode()).hexdigest()
    idx = int(digest[:8], 16) % len(SIBLING_REPOS)
    return SIBLING_REPOS[idx]


def reuse_or_create_studio(name: str, teamspace: str = "default") -> Any:
    """
    Reuse a running Studio to save quota.
    Returns a Studio-like object; caller should handle Lightning import.
    """
    try:
        from lightning.pytorch.studio import Studio
        from lightning.pytorch.studio.teamspace import Teamspace
    except Exception as e:
        raise RuntimeError("Lightning SDK not available; cannot manage Studio.") from e

    ts = Teamspace(teamspace)
    for s in ts.studios:
        if s.name == name and getattr(s, "status", None) == "Running":
            return s
    return Studio(name=name, teamspace=teamspace, create_ok=True)


def download_and_project(repo: str, file_path: str, token: str | None = None) -> Iterator[Dict[str, str]]:
    """
    Download file and stream-parse to {prompt, response}.
    Supports JSONL/JSON/Parquet with safe fallbacks.
    """
    local_path = hf_hub_download(repo_id=repo, filename=file_path, token=token)
    suffix = Path(file_path).suffix.lower()

    if suffix == ".jsonl":
        with open(local_path, "r", encoding="utf-8") as f:
            for line in f:
                line = line.strip()
                if not line:
                    continue
                try:
                    obj = json.loads(line)
                    projected = project_to_prompt_response(obj)
                    if projected:
                        yield projected
                except Exception:
                    continue

    elif suffix == ".json":
        with open(local_path, "r", encoding="utf-8") as f:
            try:
                data = json.load(f)
            except Exception:
                return
            items = data if isinstance(data, list) else [data]
            for obj in items:
                projected = project_to_prompt_response(obj)
                if projected:
                    yield projected

    elif suffix == ".parquet":
        try:
            import pyarrow.parquet as pq
            table = pq.read_table(local_path)
            df = table.to_pandas()
            for _, row in df.iterrows():
                projected = project_to_prompt_response(row.to_dict())
                if projected:
                    yield projected
        except Exception:
            return

    else:
        return

# /opt/axentx/vanguard/src/vanguard/__main__.py
import sys
from vanguard.data_manifest import persist_file_list, cdn_urls

if __name__ == "__main__":
    if len(sys.argv) < 3:
        print("Usage: python -m vanguard.data_manifest <repo> <date_folder> [token]")
        sys.exit(1)
    repo, datef = sys.argv[1], sys.argv[2]
    token = sys.argv[3] if len(sys.argv) > 3 else None
    files = persist_file_list(repo, datef, token)
    for url in cdn_urls(repo, datef, files):
        print(url)
