## 1) ClockInOutButton — Optimistic + GPS + Rollback

**Key decisions (resolve contradictions):**
- Use **optimistic UI immediately on click** (not after server) to reflect intent and prevent double-taps.
- Keep **GPS acquisition as a hard requirement before mutation**; show clear acquiring/acquired/error states.
- Rollback optimistically on error and surface a compact retry affordance.
- Do **not** rely only on server state for button appearance; derive from `optimisticState` first, then server.

**Implementation (refined from Candidate 1, hardened):**

```tsx
// ClockInOutButton.tsx
import { useState, useCallback } from 'react';
import { useMutation, useQueryClient } from '@tanstack/react-query';
import { Check, Clock, MapPin, AlertCircle } from 'lucide-react';
import { cn } from '@/lib/utils';
import { useOfflineQueue } from '@/hooks/useOfflineQueue';

interface ClockInOutButtonProps {
  isClockedIn: boolean;
  lastClockTime?: string;
  onClock: (action: 'in' | 'out', location: GeolocationPosition) => Promise<void>;
}

export function ClockInOutButton({ isClockedIn, lastClockTime, onClock }: ClockInOutButtonProps) {
  const queryClient = useQueryClient();
  const queue = useOfflineQueue();

  const [optimisticState, setOptimisticState] = useState<{
    state: 'idle' | 'pending' | 'success' | 'error';
    isClockedIn?: boolean;
  }>({ state: 'idle' });

  const [gpsStatus, setGpsStatus] = useState<'idle' | 'acquiring' | 'acquired' | 'error'>('idle');

  const mutation = useMutation({
    mutationFn: (position: GeolocationPosition) => onClock(isClockedIn ? 'out' : 'in', position),
    onMutate: async () => {
      await queryClient.cancelQueries({ queryKey: ['attendance'] });
      const previous = queryClient.getQueryData(['attendance']);
      setOptimisticState({ state: 'pending', isClockedIn: !isClockedIn });
      return { previous };
    },
    onError: (err, _vars, context) => {
      setOptimisticState({ state: 'error', isClockedIn: isClockedIn });
      queryClient.setQueryData(['attendance'], context?.previous);
    },
    onSuccess: () => {
      setOptimisticState({ state: 'success', isClockedIn: !isClockedIn });
      setTimeout(() => setOptimisticState({ state: 'idle' }), 1000);
    },
    onSettled: () => {
      queryClient.invalidateQueries({ queryKey: ['attendance'] });
    },
  });

  const getLocation = useCallback(
    (): Promise<GeolocationPosition> =>
      new Promise((resolve, reject) => {
        if (!navigator.geolocation) {
          reject(new Error('Geolocation not supported'));
          return;
        }
        setGpsStatus('acquiring');
        navigator.geolocation.getCurrentPosition(
          (pos) => {
            setGpsStatus('acquired');
            resolve(pos);
          },
          (err) => {
            setGpsStatus('error');
            reject(err);
          },
          { enableHighAccuracy: true, timeout: 10000, maximumAge: 30000 }
        );
      }),
    []
  );

  const handleClick = async () => {
    if (mutation.isPending || optimisticState.state === 'pending') return;

    try {
      const position = await getLocation();
      // If offline, enqueue instead of mutating immediately
      if (!navigator.onLine) {
        queue.enqueue({
          type: 'clock',
          payload: { action: isClockedIn ? 'out' : 'in', position },
        });
        setOptimisticState({ state: 'pending', isClockedIn: !isClockedIn });
        return;
      }
      mutation.mutate(position);
    } catch {
      // GPS or other errors handled by gpsStatus and optimistic error state
    }
  };

  const effectiveIsClockedIn = optimisticState.state === 'pending'
    ? optimisticState.isClockedIn
    : isClockedIn;

  const isPending = optimisticState.state === 'pending' || mutation.isPending;

  return (
    <div className="flex flex-col items-center gap-3">
      <button
        onClick={handleClick}
        disabled={isPending || gpsStatus === 'acquiring'}
        className={cn(
          'group relative flex min-h-[48px] w-full max-w-[280px] items-center justify-center gap-2 rounded-xl px-6 py-3 font-medium transition-all',
          effectiveIsClockedIn
            ? 'bg-red-500/10 text-red-600 hover:bg-red-500/15 active:scale-95'
            : 'bg-emerald-500/10 text-emerald-600 hover:bg-emerald-500/15 active:scale-95',
          (isPending || gpsStatus === 'acquiring') && 'cursor-wait opacity-80'
        )}
      >
        {isPending && (
          <span className="absolute left-4 top-1/2 -translate-y-1/2 animate-spin rounded-full border-2 border-current/30 border-t-current" />
        )}

        {effectiveIsClockedIn ? (
          <>
            <Clock className="h-5 w-5" />
            <span className="hidden sm:inline">Clock Out</span>
            <span className="sm:hidden">Out</span>
          </>
        ) : (
          <>
            <Check className="h-5 w-5" />
            <span className="hidden sm:inline">Clock In</span>
            <span className="sm:hidden">In</span>
          </>
        )}

        {gpsStatus === 'acquired' && (
          <MapPin className="ml-1 h-4 w-4 text-emerald-500/60" />
        )}
        {gpsStatus === 'error' && (
          <AlertCircle className="ml-1 h-4 w-4 text-amber-500" title="Location unavailable" />
        )}
      </button>

      {effectiveIsClockedIn && lastClockTime && (
        <span className="text-xs text-muted-foreground">
          Last: {new Date(lastClockTime).toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' })}
        </span>
      )}

      {(optimisticState.state === 'error' || mutation.isError) && (
        <div className="flex items-center gap-2 text-xs text-destructive">
          <AlertCircle className="h-3 w-3" />
          Failed
          <button
            onClick={() => {
              setOptimisticState({ state: 'idle' });
              // allow retry by clicking main button again
            }}
            className="underline underline-offset-2"
          >
            Retry
          </button>
        </div>
      )}

      {queue.pendingCount > 0 && (
        <span className="text-xs text-muted-foreground">
          {queue.pendingCount} pending action{queue.pendingCount > 1 ? 's' : ''}
        </span>
      )}
    </div
