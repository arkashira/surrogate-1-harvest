# surrogate-1 / design

## Final Implementation Plan (≤2h)

**Highest-value improvement**: Replace fragile shell-based ingestion with a **manifest-driven, CDN-bypass Python worker** that eliminates HF API rate limits during training data loads and prevents mixed-schema CastErrors.

### Changes

1. **Add `bin/worker.py`** — deterministic shard worker that:
   - Uses a single `list_repo_tree` call (per date folder) from the runner, embeds file list in a `manifest.json`
   - Downloads only assigned shard files via HF CDN (`resolve/main/...`) with **no Authorization header** (bypasses API rate limits)
   - Projects to `{prompt, response}` at parse time; never loads full heterogeneous schemas into a single `load_dataset`
   - Writes `batches/public-merged/<date>/shard<N>-<HHMMSS>.jsonl` (existing path)
   - Uses central `lib/dedup.py` for cross-source md5 dedup

2. **Update `bin/dataset-enrich.sh`** → thin wrapper that:
   - Invokes `python3 bin/worker.py` with `SHARD_ID`, `TOTAL_SHARDS`, `DATE`, `MANIFEST`
   - Keeps executable bit; Bash shebang retained for cron/workflow compatibility

3. **Update `.github/workflows/ingest.yml`** → matrix runner:
   - Pre-step: one lightweight job (or first matrix entry) generates `manifest.json` for the target date folder using HF API (rate-limit safe: single tree call) and uploads as artifact
   - Each shard downloads the artifact and runs `worker.py`
   - Sets `SHELL=/bin/bash` in job env (per pattern)

4. **Add `requirements.txt`** extras if needed: `requests`, `pyarrow`, `tqdm`

```python
#!/usr/bin/env python3
"""
CDN-bypass shard worker for surrogate-1 public dataset ingestion.

Usage:
  SHARD_ID=0 TOTAL_SHARDS=16 DATE=2026-04-29 python bin/worker.py

Behavior:
- Reads manifest.json (list of files in the date folder)
- Takes shard slice by deterministic hash(filename) % TOTAL_SHARDS
- Downloads each file via HF CDN (no auth) and projects to {prompt, response}
- Deduplicates via lib.dedup and writes shard-N jsonl
"""
import json
import os
import sys
import hashlib
import datetime
from pathlib import Path

import requests
import pyarrow.parquet as pq
from tqdm import tqdm

HF_DATASET = "axentx/surrogate-1-training-pairs"
BASE_CDN = f"https://huggingface.co/datasets/{HF_DATASET}/resolve/main"

# Local modules
sys.path.insert(0, str(Path(__file__).parent))
from lib.dedup import DedupStore  # noqa

def deterministic_shard(filename: str, total_shards: int, shard_id: int) -> bool:
    h = int(hashlib.md5(filename.encode()).hexdigest(), 16)
    return (h % total_shards) == shard_id

def parse_file_to_pairs(local_path: Path):
    """Project heterogeneous file to {prompt, response} pairs."""
    ext = local_path.suffix.lower()
    try:
        if ext == ".parquet":
            # Avoid load_dataset(streaming=True) across mixed schemas
            tbl = pq.read_table(local_path, columns=["prompt", "response"])
            df = tbl.to_pandas()
            for _, row in df.iterrows():
                prompt = str(row.get("prompt") or "")
                response = str(row.get("response") or "")
                if prompt and response:
                    yield {"prompt": prompt, "response": response}
        elif ext == ".jsonl":
            with open(local_path, "r", encoding="utf-8") as f:
                for line in f:
                    line = line.strip()
                    if not line:
                        continue
                    obj = json.loads(line)
                    prompt = str(obj.get("prompt") or obj.get("input") or "")
                    response = str(obj.get("response") or obj.get("output") or "")
                    if prompt and response:
                        yield {"prompt": prompt, "response": response}
        else:
            # Fallback: try json
            with open(local_path, "r", encoding="utf-8") as f:
                obj = json.load(f)
                if isinstance(obj, list):
                    for item in obj:
                        prompt = str(item.get("prompt") or item.get("input") or "")
                        response = str(item.get("response") or item.get("output") or "")
                        if prompt and response:
                            yield {"prompt": prompt, "response": response}
                else:
                    prompt = str(obj.get("prompt") or obj.get("input") or "")
                    response = str(obj.get("response") or obj.get("output") or "")
                    if prompt and response:
                        yield {"prompt": prompt, "response": response}
    except Exception as exc:
        print(f"[WARN] failed to parse {local_path}: {exc}", file=sys.stderr)

def download_via_cdn(repo_path: str, dest: Path):
    url = f"{BASE_CDN}/{repo_path}"
    resp = requests.get(url, timeout=60)
    resp.raise_for_status()
    dest.write_bytes(resp.content)

def build_output_path(shard_id: int) -> Path:
    date_str = datetime.datetime.utcnow().strftime("%Y-%m-%d")
    ts = datetime.datetime.utcnow().strftime("%H%M%S")
    out_dir = Path("batches") / "public-merged" / date_str
    out_dir.mkdir(parents=True, exist_ok=True)
    return out_dir / f"shard{shard_id}-{ts}.jsonl"

def main():
    shard_id = int(os.environ.get("SHARD_ID", "0"))
    total_shards = int(os.environ.get("TOTAL_SHARDS", "16"))
    date_folder = os.environ.get("DATE", datetime.datetime.utcnow().strftime("%Y-%m-%d"))
    manifest_path = Path(os.environ.get("MANIFEST", "manifest.json"))

    if not manifest_path.exists():
        print(f"[ERROR] manifest not found: {manifest_path}", file=sys.stderr)
        sys.exit(1)

    with open(manifest_path) as f:
        manifest = json.load(f)
    files = manifest.get("files", [])
    my_files = [f for f in files if deterministic_shard(f, total_shards, shard_id)]

    print(f"[INFO] shard {shard_id}/{total_shards} -> {len(my_files)} files")

    dedup = DedupStore()
    out_path = build_output_path(shard_id)

    written = 0
    with open(out_path, "w", encoding="utf-8") as out_f:
        for repo_path in tqdm(my_files, desc=f"shard{shard_id}"):
            tmp = Path("tmp") / repo_path.replace("/", "_")
            tmp.parent.mkdir(parents=True, exist_ok=True)
            try:
                download_via_cdn(repo_path, tmp)
                for pair in parse_file_to_pairs(tmp):
                    # dedup by content hash
                    payload = json.dumps(pair, sort_keys=True, separators=(",", ":"))
                    md5 = hashlib.md5(payload.encode()).hexdigest()
                    if dedup.seen(md5):
                        continue
                    dedup.add(md5)
                    out_f.write(json.dumps(pair, ensure_ascii=False) + "\n")
                    written += 1
                tmp.unlink(missing_ok=True)
            except Exception as exc:
                print(f"[WARN] failed {repo_path}: {exc}", file=sys.stderr)
                tmp.unlink(missing_ok=True)

    print(f"[INFO] shard {shard_id} wrote {written} pairs -> {out_path}")

if __name__ == "__main__":
    main()
```
