### 1. `bin/worker.py` (core worker)

```python
#!/usr/bin/env python3
"""
Surrogate-1 CDN-bypass worker.
- Reads manifest JSON listing files to fetch (generated by gen-manifest.py)
- Downloads via raw CDN URLs (no HF API auth/rate-limit)
- Projects to {prompt,response}, dedups via central md5 store, outputs shard JSONL
"""
import json
import os
import sys
import hashlib
import sqlite3
import time
from pathlib import Path
from typing import Dict, Any

import pyarrow as pa
import pyarrow.parquet as pq
import requests
from tqdm import tqdm

HF_DATASET = os.getenv("HF_DATASET", "axentx/surrogate-1-training-pairs")
BASE_CDN = f"https://huggingface.co/datasets/{HF_DATASET}/resolve/main"
MANIFEST_PATH = os.getenv("MANIFEST_PATH", "manifest.json")
OUTPUT_DIR = Path(os.getenv("OUTPUT_DIR", "batches/public-merged"))
SHARD_ID = int(os.getenv("SHARD_ID", "0"))
TOTAL_SHARDS = int(os.getenv("TOTAL_SHARDS", "16"))
DEADLINE_SECONDS = int(os.getenv("DEADLINE_SECONDS", "1800"))  # 30m safety

# Central dedup store (shared via mounted volume or HF Space SQLite)
DEDUP_DB = Path(os.getenv("DEDUP_DB", "/tmp/dedup.db"))

def ensure_db() -> sqlite3.Connection:
    DEDUP_DB.parent.mkdir(parents=True, exist_ok=True)
    conn = sqlite3.connect(str(DEDUP_DB))
    conn.execute("CREATE TABLE IF NOT EXISTS seen_md5 (md5 TEXT PRIMARY KEY)")
    conn.commit()
    return conn

def is_duplicate(conn: sqlite3.Connection, md5: str) -> bool:
    cur = conn.execute("SELECT 1 FROM seen_md5 WHERE md5=?", (md5,))
    return cur.fetchone() is not None

def mark_seen(conn: sqlite3.Connection, md5: str) -> None:
    try:
        conn.execute("INSERT INTO seen_md5 (md5) VALUES (?)", (md5,))
    except sqlite3.IntegrityError:
        pass

def download_parquet(url: str) -> pa.Table:
    """Download parquet via CDN with retry/backoff."""
    for attempt in range(5):
        try:
            resp = requests.get(url, timeout=30)
            resp.raise_for_status()
            buf = pa.BufferReader(resp.content)
            return pq.read_table(buf)
        except Exception as e:
            wait = 2 ** attempt
            print(f"[WARN] retry {attempt} for {url}: {e}, waiting {wait}s", file=sys.stderr)
            time.sleep(wait)
    raise RuntimeError(f"Failed to download {url}")

def normalize_row(table: pa.Table, row_idx: int) -> Dict[str, str]:
    """Project heterogeneous schema to {prompt,response} only."""
    row = {k: table[k][row_idx].as_py() for k in table.schema.names}
    # Heuristic field names across datasets
    prompt_keys = {"prompt", "instruction", "input", "question", "text"}
    response_keys = {"response", "output", "answer", "completion", "target"}

    prompt = None
    response = None
    for k, v in row.items():
        nk = k.lower().strip()
        if nk in prompt_keys and v is not None:
            prompt = str(v).strip()
        if nk in response_keys and v is not None:
            response = str(v).strip()

    # Fallback: first/second non-null str col
    str_cols = [k for k in table.schema.names if pa.types.is_string(table.schema.field(k).type)]
    if prompt is None and str_cols:
        val = row.get(str_cols[0])
        if val is not None:
            prompt = str(val).strip()
    if response is None and len(str_cols) > 1:
        val = row.get(str_cols[1])
        if val is not None:
            response = str(val).strip()

    return {"prompt": prompt or "", "response": response or ""}

def hash_pair(prompt: str, response: str) -> str:
    return hashlib.md5(f"{prompt}\0{response}".encode("utf-8")).hexdigest()

def main() -> None:
    if not Path(MANIFEST_PATH).exists():
        print(f"[ERROR] Manifest not found: {MANIFEST_PATH}", file=sys.stderr)
        sys.exit(1)

    with open(MANIFEST_PATH) as f:
        manifest = json.load(f)

    files = manifest.get("files", [])
    if not files:
        print("[WARN] No files in manifest", file=sys.stderr)
        return

    # Deterministic shard assignment by slug hash
    my_files = []
    for f in files:
        slug = f.get("slug", f.get("path", ""))
        h = int(hashlib.md5(slug.encode("utf-8")).hexdigest(), 16)
        if h % TOTAL_SHARDS == SHARD_ID:
            my_files.append(f)

    print(f"[INFO] Shard {SHARD_ID}/{TOTAL_SHARDS} assigned {len(my_files)} files")

    conn = ensure_db()
    OUTPUT_DIR.mkdir(parents=True, exist_ok=True)
    ts = time.strftime("%Y%m%d-%H%M%S")
    out_path = OUTPUT_DIR / f"shard{SHARD_ID}-{ts}.jsonl"

    start = time.time()
    written = 0
    skipped_dup = 0
    failed = 0

    with out_path.open("w", encoding="utf-8") as out_f:
        for meta in tqdm(my_files, desc=f"Shard-{SHARD_ID}"):
            if time.time() - start > DEADLINE_SECONDS:
                print("[WARN] Deadline exceeded, stopping", file=sys.stderr)
                break

            path = meta.get("path") or meta.get("slug")
            if not path:
                continue

            url = f"{BASE_CDN}/{path}"
            try:
                table = download_parquet(url)
            except Exception as e:
                print(f"[ERROR] download failed {url}: {e}", file=sys.stderr)
                failed += 1
                continue

            for i in range(table.num_rows):
                try:
                    norm = normalize_row(table, i)
                except Exception:
                    continue
                prompt = norm.get("prompt", "")
                response = norm.get("response", "")
                if not prompt or not response:
                    continue

                md5 = hash_pair(prompt, response)
                if is_duplicate(conn, md5):
                    skipped_dup += 1
                    continue

                mark_seen(conn, md5)
                out_f.write(json.dumps({"prompt": prompt, "response": response}, ensure_ascii=False) + "\n")
                written += 1

            # Periodic commit
            conn.commit()

    conn.commit()
    conn.close()

    print(f"[DONE] Shard-{SHARD_ID}: written={written}, dup={skipped_dup}, failed={failed}, out={out_path}")

if __name__ == "__main__":
    main()
```
