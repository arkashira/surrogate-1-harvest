## Final Synthesized Solution

### 1. Diagnosis (merged, prioritized)
- **Race conditions**: `find-then-insert/update` for punches allows multiple open punches per employee under concurrent LINE retries or client double-taps.
- **No idempotency**: LINE webhook retries (same event payload) create duplicate punches/leave/OT entries.
- **No database enforcement**: missing unique constraint/index to enforce “at most one open punch per employee” at the DB level.
- **Weak webhook security**: insufficient payload signature validation and no replay protection.
- **Observability gaps**: missing request-id/trace-id and structured logging for webhook flows.

### 2. Core Principles for Correctness + Actionability
- **Enforce invariants at the database** (constraints + row locking) — app logic is secondary.
- **Idempotency first**: reject duplicates cheaply before business logic; use a short-lived idempotency table with unique key.
- **Atomicity**: all punch state changes and idempotency recording must occur in a single transaction with row-level locking (`SELECT ... FOR UPDATE`).
- **Security**: verify LINE signature on every webhook; reject unknown/unsigned requests early.
- **Observability**: emit request-id, event-id, and employee-id on all webhook processing logs.

CREATE TABLE IF NOT EXISTS webhook_idempotency (
  idempotency_key TEXT PRIMARY KEY,
  event_type      TEXT NOT NULL,
  employee_id     INTEGER NOT NULL,
  created_at      TIMESTAMPTZ DEFAULT NOW()
);
CREATE INDEX IF NOT EXISTS idx_webhook_idempotency_created_at
  ON webhook_idempotency(created_at);

-- At most one open punch per employee
-- DEFERRABLE allows constraint check at COMMIT for upsert safety
ALTER TABLE punches
  ADD CONSTRAINT uq_employee_open_punch
  UNIQUE (employee_id)
  DEFERRABLE INITIALLY DEFERRED
  WHERE (clock_out_at IS NULL);

### 4. Punch Service — Atomic Upsert (`server/src/services/punchService.ts`)

```ts
import { pool } from '../db';

export async function clockInEmployee(
  employeeId: number,
  location?: { lat: number; lng: number },
  client?: any
) {
  const releaseClient = !client;
  const c = client || (await pool.connect());
  try {
    if (releaseClient) await c.query('BEGIN');

    // Lock any open punch to prevent TOCTOU
    await c.query(
      `SELECT id FROM punches
       WHERE employee_id = $1 AND clock_out_at IS NULL
       FOR UPDATE`,
      [employeeId]
    );

    // Try insert; unique constraint prevents second open punch
    const insertRes = await c.query(
      `INSERT INTO punches (employee_id, clock_in_at, clock_in_location)
       VALUES ($1, NOW(), $2)
       ON CONFLICT DO NOTHING
       RETURNING id, clock_in_at`,
      [employeeId, location ? JSON.stringify(location) : null]
    );

    let punch;
    if (insertRes.rowCount === 0) {
      // Already open — return existing
      const getRes = await c.query(
        `SELECT id, clock_in_at FROM punches
         WHERE employee_id = $1 AND clock_out_at IS NULL`,
        [employeeId]
      );
      punch = getRes.rows[0];
    } else {
      punch = insertRes.rows[0];
    }

    if (releaseClient) await c.query('COMMIT');
    return punch;
  } catch (err) {
    if (releaseClient) await c.query('ROLLBACK');
    throw err;
  } finally {
    if (releaseClient) c.release();
  }
}

export async function clockOutEmployee(
  employeeId: number,
  location?: { lat: number; lng: number },
  client?: any
) {
  const releaseClient = !client;
  const c = client || (await pool.connect());
  try {
    if (releaseClient) await c.query('BEGIN');

    // Lock open punch
    const open = await c.query(
      `SELECT id FROM punches
       WHERE employee_id = $1 AND clock_out_at IS NULL
       FOR UPDATE`,
      [employeeId]
    );

    if (open.rowCount === 0) {
      await c.query(releaseClient ? 'ROLLBACK' : 'ROLLBACK');
      throw new Error('No open punch to clock out');
    }

    const punchId = open.rows[0].id;
    const res = await c.query(
      `UPDATE punches
       SET clock_out_at = NOW(),
           clock_out_location = $1
       WHERE id = $2
       RETURNING id, clock_out_at`,
      [location ? JSON.stringify(location) : null, punchId]
    );

    if (releaseClient) await c.query('COMMIT');
    return res.rows[0];
  } catch (err) {
    if (releaseClient) await c.query('ROLLBACK');
    throw err;
  } finally {
    if (releaseClient) c.release();
  }
}

### 5. Idempotent LINE Webhook Handler (`server/src/controllers/lineWebhookController.ts`)

```ts
import { Request, Response } from 'express';
import crypto from 'crypto';
import { pool } from '../db';
import { clockInEmployee, clockOutEmployee } from '../services/punchService';

const LINE_CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';

function verifyLineSignature(rawBody: string, signature: string): boolean {
  if (!LINE_CHANNEL_SECRET) return false;
  const expected = crypto
    .createHmac('sha256', LINE_CHANNEL_SECRET)
    .update(rawBody)
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

function makeIdempotencyKey(event: any): string {
  // Prefer stable, retry-invariant key from LINE
  // Use mode + sourceId + timestamp (or messageId if present)
  const sourceId = event.source?.userId || event.source?.groupId || '';
  const ts = event.timestamp || Date.now();
  const detail = event.message?.id || event.type;
  return `line:${event.type}:${sourceId}:${ts}:${detail}`;
}

export async function handleLineWebhook(req: Request, res: Response) {
  const requestId = req.headers['x-request-id'] || crypto.randomUUID();
  const rawBody = JSON.stringify(req.body);
  const signature = req.headers['x-line-signature'] as string;

  // 1) Security: verify signature early
  if (!verifyLineSignature(rawBody, signature)) {
    console.warn({ requestId, event: 'invalid_signature' });
    return res.status(401).json({ error: 'invalid_signature' });
  }

  const body = req.body;
  const events = body.events || [];
  if (!Array.isArray(events) || events.length === 0) {
    return res.status(200).json({ ok: true });
  }

  // 2) Idempotency: use first event key (or hash for batches) as dedupe key
  const idempotencyKey = events.length === 1
    ? makeIdempotencyKey(events[0])
    : `line:batch:sha256:${crypto.createHash('sha256').update(rawBody).digest('hex')}`;

  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // 3) Fast duplicate check + insert idempotency record atomically
    const dup = await client.query(
      `SELECT 1 FROM webhook_idempotency WHERE idempotency_key = $1`,
      [idempotencyKey]
    );

    if (dup.rowCount > 0) {
      await client.query('COMMIT');
      console.info({ requestId, event: 'duplicate', idempotencyKey });
      return res.status(200).json({ ok: true, duplicate: true });
    }

      `INSERT INTO webhook_idempotency(idempotency_key, event_type, employee_id)
       VALUES ($1, $2, $3)`,
      [idempotencyKey, 'line_webhook', 0]
