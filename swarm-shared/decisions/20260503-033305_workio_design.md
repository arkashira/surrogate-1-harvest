### 1. Signature verification middleware

`workio/server/src/middleware/line-signature.ts`

```ts
import { Request, Response, NextFunction } from 'express';
import crypto from 'crypto';

const LINE_CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';

export function lineSignature() {
  return (req: Request, res: Response, next: NextFunction) => {
    const signature = req.get('X-Line-Signature');
    const rawBody = (req as any).rawBody;

    if (!LINE_CHANNEL_SECRET) {
      return res.status(500).json({ error: 'Server configuration error' });
    }
    if (!signature || typeof rawBody !== 'string') {
      return res.status(401).json({ error: 'Missing signature or raw body' });
    }

    const expected = crypto
      .createHmac('sha256', LINE_CHANNEL_SECRET)
      .update(rawBody, 'utf8')
      .digest('base64');

    // Constant-time comparison to avoid timing attacks
    const match = crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
    if (!match) {
      return res.status(401).json({ error: 'Invalid signature' });
    }
    next();
  };
}
```

### 2. Idempotency middleware

`workio/server/src/middleware/idempotency.ts`

```ts
import { Request, Response, NextFunction } from 'express';
import crypto from 'crypto';
import { getPool } from '../db';

const IDEMPOTENCY_TTL_SECONDS = 300; // 5 minutes

export async function idempotencyGuard(req: Request, res: Response, next: NextFunction) {
  const body = req.body;
  const events = body?.events;
  if (!Array.isArray(events) || events.length === 0) {
    return res.status(400).json({ error: 'No events' });
  }

  const pool = getPool();
  const client = await pool.connect();
  try {
    await client.query('BEGIN');
    const processed: any[] = [];

    for (const ev of events) {
      // Stable dedup key: userId + type + timestamp + (replyToken or deliveryId)
      const dedupKey = crypto
        .createHash('sha256')
        .update(
          [
            ev.source?.userId || '',
            ev.type || '',
            String(ev.timestamp || ''),
            ev.replyToken || ev.deliveryId || '',
          ].join(':')
        )
        .digest('hex');

      const exists = await client.query(
        `SELECT 1 FROM webhook_events_dedup WHERE dedup_key = $1 AND created_at > NOW() - INTERVAL '${IDEMPOTENCY_TTL_SECONDS} seconds'`,
        [dedupKey]
      );

      if (exists.rows.length > 0) {
        continue; // skip duplicate
      }

      await client.query(
        `INSERT INTO webhook_events_dedup (dedup_key, event_payload, created_at)
         VALUES ($1, $2, NOW())`,
        [dedupKey, JSON.stringify(ev)]
      );
      processed.push(ev);
    }

    await client.query('COMMIT');
    // Attach processed events for downstream handlers
    (req as any).processedEvents = processed;
    next();
  } catch (err) {
    await client.query('ROLLBACK').catch(() => {});
    console.error('Idempotency check failed', err);
    return res.status(500).json({ error: 'Idempotency check failed' });
  } finally {
    client.release();
  }
}
```

### 3. Replay/timestamp protection middleware

`workio/server/src/middleware/timestamp-validator.ts`

```ts
import { Request, Response, NextFunction } from 'express';

const MAX_SKEW_MS = 5 * 60 * 1000; // 5 minutes

export function validateTimestamp(req: Request, res: Response, next: NextFunction) {
  const events = (req as any).processedEvents || req.body.events || [];
  const now = Date.now();

  for (const ev of events) {
    const ts = ev.timestamp;
    if (typeof ts !== 'number' || Math.abs(now - ts) > MAX_SKEW_MS) {
      return res.status(400).json({ error: 'Stale or missing timestamp' });
    }
  }
  next();
}
```

### 4. Raw-body capture and route wiring

`workio/server/src/routes/webhook/line.ts`

```ts
import express, { Request, Response } from 'express';
import { lineSignature } from '../middleware/line-signature';
import { idempotencyGuard } from '../middleware/idempotency';
import { validateTimestamp } from '../middleware/timestamp-validator';

const router = express.Router();

// Capture raw body before JSON parsing
function rawBodyCapture(
  req: Request,
  _res: Response,
  buf: Buffer,
  encoding: string | undefined
) {
  if (buf && buf.length) {
    (req as any).rawBody = buf.toString(encoding || 'utf8');
  }
}

router.post(
  '/line',
  // 1) Capture raw body
  express.raw({ type: '*/*', verify: rawBodyCapture }),
  // 2) Verify LINE signature
  lineSignature(),
  // 3) Parse JSON (safe after signature)
  express.json(),
  // 4) Idempotency + dedup
  idempotencyGuard,
  // 5) Timestamp/replay protection
  validateTimestamp,
  // 6) Handler (tenant-aware)
  async (req: Request, res: Response) => {
    try {
      const events = (req as any).processedEvents || [];
      // TODO: map source.userId -> tenant (validate against LINE channel binding)
      // await processLineEvents(events, req);
      res.status(200).json({ ok: true });
    } catch (err) {
      console.error('Webhook processing failed', err);
      res.status(500).json({ error: 'Processing failed' });
    }
  }
);

export default router;
```

### 5. Database schema (run once)

`workio/server/src/db/schema.sql`

```sql
CREATE TABLE IF NOT EXISTS webhook_events_dedup (
  id BIGSERIAL PRIMARY KEY,
  dedup_key TEXT NOT NULL UNIQUE,
  event_payload JSONB,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX IF NOT EXISTS idx_dedup_key ON webhook_events_dedup (dedup_key);
CREATE INDEX IF NOT EXISTS idx_dedup_created ON webhook_events_dedup (created_at);
```

### 6. Verification checklist

1. **Schema**  
   - Confirm `webhook_events_dedup` exists and indexes are present.

2. **Signature verification**  
   - Valid `X-Line-Signature` → 200.  
   - Invalid/missing signature → 401.

3. **Idempotency**  
   - Send identical event twice within 5 minutes → second request skips duplicate application records; dedup table contains one entry.

4. **Replay/timestamp protection**  
   - Event with timestamp older than 5 minutes → 400
