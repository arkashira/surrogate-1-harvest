# workio / backend

**Final consolidated implementation (best of both proposals, corrected + actionable)**

- **Core fixes applied**
  - HMAC-SHA256 verification on raw body (requires buffering before JSON parse).
  - Idempotency + replay-window (5 min) using a deterministic key and DB uniqueness.
  - Tenant binding via channel/source with cross-check against `channels.tenant_id`.
  - No reliance on in-memory only state; DB-backed idempotency for multi-instance safety.
  - Clear separation of concerns and safe error handling.

### 2) Middleware

`/opt/axentx/workio/server/src/middleware/lineSignature.ts`
```ts
import crypto from 'crypto';
import { Request, Response, NextFunction } from 'express';

const LINE_CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET;
if (!LINE_CHANNEL_SECRET) {
  throw new Error('LINE_CHANNEL_SECRET is required');
}

export function verifyLineSignature(rawBody: Buffer, signature?: string): boolean {
  if (!signature) return false;
  const expected = crypto
    .createHmac('sha256', LINE_CHANNEL_SECRET)
    .update(rawBody)
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

/**
 * Middleware that:
 * - expects raw body buffer on req.rawBody (set by bodyParser.json verify hook)
 * - verifies X-Line-Signature
 * - rejects stale events (>5min)
 * - attaches parsed payload + tenantId to req
 */
export function lineSignatureMiddleware(
  req: Request,
  res: Response,
  next: NextFunction
) {
  const raw = req.rawBody as Buffer | undefined;
  const sig = req.headers['x-line-signature'] as string | undefined;

  if (!raw || !sig) {
    return res.status(401).json({ error: 'Missing raw body or X-Line-Signature' });
  }

  if (!verifyLineSignature(raw, sig)) {
    return res.status(401).json({ error: 'Invalid signature' });
  }

  let payload: any;
  try {
    payload = JSON.parse(raw.toString());
  } catch {
    return res.status(400).json({ error: 'Invalid JSON' });
  }

  // Basic event shape validation
  const events = payload.events;
  if (!Array.isArray(events) || events.length === 0) {
    return res.status(400).json({ error: 'Missing events array' });
  }

  // Replay-window: reject events older than 5 minutes (LINE retries are short; this prevents stale replays)
  const now = Date.now();
  const fiveMin = 5 * 60 * 1000;
  for (const ev of events) {
    const ts = Number(ev.timestamp);
    if (!ts || Number.isNaN(ts) || Math.abs(now - ts) > fiveMin) {
      return res.status(400).json({ error: 'Stale event rejected' });
    }
  }

  // Deterministic idempotency key per event (use first event for single-event webhook pattern)
  const first = events[0];
  const eventId = first.source?.userId + first.timestamp + first.type;
  const idempotencyKey = crypto
    .createHash('sha256')
    .update(eventId)
    .digest('hex');

  // Resolve tenant from channel mapping (cryptographic binding)
  const channelId = first.source?.channelId || payload.destination;
  if (!channelId) {
    return res.status(400).json({ error: 'Missing channelId/destination' });
  }

  // Attach for downstream handlers and idempotency middleware
  (req as any).linePayload = payload;
  (req as any).lineChannelId = channelId;
  (req as any).lineIdempotencyKey = idempotencyKey;
  (req as any).lineEvents = events;
  (req as any).lineTenantId = null; // filled by tenant-binding middleware

  next();
}
```

`/opt/axentx/workio/server/src/middleware/tenantBinding.ts`
```ts
import { Request, Response, NextFunction } from 'express';
import { getClient } from '../db';

/**
 * Bind channel to tenant and enforce tenant context.
 * Must run after lineSignatureMiddleware (which sets lineChannelId).
 */
export async function tenantBindingMiddleware(
  req: Request,
  res: Response,
  next: NextFunction
) {
  const channelId = (req as any).lineChannelId as string | undefined;
  if (!channelId) {
    return res.status(400).json({ error: 'Missing channelId for tenant binding' });
  }

  const client = await getClient();
  try {
    const r = await client.query(
      'SELECT tenant_id FROM channels WHERE channel_id = $1',
      [channelId]
    );
    if (r.rows.length === 0) {
      return res.status(403).json({ error: 'Unknown channel' });
    }
    (req as any).lineTenantId = r.rows[0].tenant_id;
    next();
  } catch (err) {
    next(err);
  } finally {
    client.release();
  }
}
```

`/opt/axentx/workio/server/src/middleware/idempotency.ts`
```ts
import { Request, Response, NextFunction } from 'express';
import { getClient } from '../db';

/**
 * DB-backed idempotency guard.
 * Uses lineIdempotencyKey and lineTenantId set by prior middleware.
 */
export async function idempotencyGuard(
  req: Request,
  res: Response,
  next: NextFunction
) {
  const key = (req as any).lineIdempotencyKey as string | undefined;
  const tenantId = (req as any).lineTenantId as string | undefined;
  const channelId = (req as any).lineChannelId as string | undefined;
  const events = (req as any).lineEvents as any[];

  if (!key || !tenantId || !channelId || !events) {
    return res.status(400).json({ error: 'Missing required context for idempotency' });
  }

  const client = await getClient();
  try {
    // Try insert; duplicate key will throw and we treat as already-processed
    await client.query(
      'INSERT INTO webhook_events(event_id, tenant_id, channel_id, line_event_type, line_event_id, received_at) VALUES ($1, $2, $3, $4, $5, NOW())',
      [key, tenantId, channelId, events[0]?.type || null, events[0]?.source?.userId || null]
    );
    (req as any).idempotencyKey = key;
    next();
  } catch (err: any) {
    if (err.code === '23505') {
      // Unique violation: duplicate event
      return res.status(200).json({ ok: true, reason: 'duplicate' });
    }
    next(err);
  } finally {
    client.release();
  }
}
```
