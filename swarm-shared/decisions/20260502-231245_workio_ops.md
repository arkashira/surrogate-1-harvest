# workio / ops

## Final Synthesis — Correct, Actionable Solution

**Core diagnosis (accepted from both candidates):**  
- No idempotency key → retries create duplicate punches.  
- No storage-level uniqueness for active punches → `(employee_id, punch_type, date)` duplicates possible on retries/races.  
- Plain `INSERT` without conflict handling → non-idempotent and race-prone.  
- Missing index → slow duplicate checks at scale.  

**Key corrections / decisions (favor correctness + actionability):**  
1. **Use a deterministic `punch_id` (idempotency key) as the primary deduplication mechanism** — client-supplied when possible (e.g., LINE event hash), server-generated otherwise.  
2. **Add a partial unique index for active punches** to enforce business rule “one active punch per employee per type per date” (`clock_out_ts IS NULL`). This is a safety net, not the primary idempotency mechanism.  
3. **Use `INSERT ... ON CONFLICT (punch_id) DO NOTHING`** for idempotent inserts; fetch existing row after conflict to return stable state.  
4. **Clock-out must be idempotent** — use `punch_id` when available and only update when `clock_out_ts IS NULL`.  
5. **Schema changes must be safe for existing deployments** — backfill missing `punch_id` and make column `NOT NULL` only after backfill.  
6. **Index `(employee_id, punch_type, date)`** for fast lookups and to support the partial unique index efficiently.

### 1. Schema changes (`/opt/axentx/workio/server/src/db/schema.sql`)

```sql
-- 1) Add deterministic idempotency column (if missing)
ALTER TABLE punches ADD COLUMN IF NOT EXISTS punch_id UUID;

-- 2) Backfill missing punch_id values safely
UPDATE punches
SET punch_id = gen_random_uuid()
WHERE punch_id IS NULL;

-- 3) Enforce NOT NULL after backfill
ALTER TABLE punches ALTER COLUMN punch_id SET NOT NULL;

-- 4) Global uniqueness for idempotency key
ALTER TABLE punches ADD CONSTRAINT punches_punch_id_key UNIQUE (punch_id);

-- 5) Partial unique index: only one active punch per employee/type/date
CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_active_unique
ON punches (employee_id, punch_type, date)
WHERE clock_out_ts IS NULL;

-- 6) General lookup index
CREATE INDEX IF NOT EXISTS idx_punches_employee_type_date
ON punches (employee_id, punch_type, date);
```

### 2. Punch route (`/opt/axentx/workio/server/src/routes/punch.ts`)

```ts
import { Request, Response } from 'express';
import { pool } from '../db';
import { v4 as uuidv4 } from 'uuid';

// POST /punch
// Body: { employee_id, punch_type, date, lat?, lng?, line_user_id?, idempotency_key? }
export async function createPunch(req: Request, res: Response) {
  const { employee_id, punch_type, date, lat, lng, line_user_id } = req.body;
  const clientKey = req.body.idempotency_key;
  const punch_id = clientKey ? clientKey : uuidv4();

  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // Idempotent insert
    const insertSQL = `
      INSERT INTO punches (punch_id, employee_id, punch_type, date, clock_in_ts, lat, lng, line_user_id)
      VALUES ($1, $2, $3, $4, NOW(), $5, $6, $7)
      ON CONFLICT (punch_id) DO NOTHING
      RETURNING *;
    `;
    const result = await client.query(insertSQL, [punch_id, employee_id, punch_type, date, lat || null, lng || null, line_user_id || null]);

    let punch = result.rows[0];
    if (!punch) {
      // Fetch existing active punch for this employee/type/date
      const getSQL = `
        SELECT * FROM punches
        WHERE employee_id = $1 AND punch_type = $2 AND date = $3 AND clock_out_ts IS NULL
        ORDER BY clock_in_ts DESC LIMIT 1;
      `;
      const existing = await client.query(getSQL, [employee_id, punch_type, date]);
      punch = existing.rows[0] || null;
    }

    await client.query('COMMIT');
    return res.json({ ok: true, punch });
  } catch (err) {
    await client.query('ROLLBACK');
    console.error('createPunch error', err);
    return res.status(500).json({ ok: false, error: String(err) });
  } finally {
    client.release();
  }
}

// POST /punch/:id/clock_out
// Body: { idempotency_key? }
export async function clockOutPunch(req: Request, res: Response) {
  const { id } = req.params;
  const clientKey = req.body.idempotency_key;
  const punch_id = clientKey ? clientKey : undefined;

  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    let result;
    if (punch_id) {
      // Idempotent clock-out by punch_id
      const updateSQL = `
        UPDATE punches
        SET clock_out_ts = NOW()
        WHERE punch_id = $1 AND clock_out_ts IS NULL
        RETURNING *;
      `;
      result = await client.query(updateSQL, [punch_id]);
    } else {
      // Legacy: update by numeric id, ensure still active
      const updateSQL = `
        UPDATE punches
        SET clock_out_ts = NOW()
        WHERE id = $1 AND clock_out_ts IS NULL
        RETURNING *;
      `;
      result = await client.query(updateSQL, [id]);
    }

    if (!result.rowCount) {
      // Already clocked out or not found — return current state idempotently
      const getSQL = punch_id
        ? 'SELECT * FROM punches WHERE punch_id = $1'
        : 'SELECT * FROM punches WHERE id = $1';
      const getRes = await client.query(getSQL, punch_id ? [punch_id] : [id]);
      await client.query('COMMIT');
      return res.json({ ok: true, punch: getRes.rows[0] || null, already: true });
    }

    await client.query('COMMIT');
    return res.json({ ok: true, punch: result.rows[0] });
  } catch (err) {
    await client.query('ROLLBACK');
    console.error('clockOutPunch error', err);
    return res.status(500).json({ ok: false, error: String(err) });
  } finally {
    client.release();
  }
}
```

### 3. Operational guidance

- **LINE webhook handler**: generate a stable `idempotency_key` per event (e.g., hash of LINE event ID or use the event ID itself) and include it in requests. This ensures retries from LINE do not create duplicates.  
- **Frontend clients**: persist a client-generated UUID for a punch session and reuse it on retries.  
- **Deploy order**:  
  1. Apply schema changes first (safe, backfills existing rows).  
  2. Deploy updated route code.  
  3. Verify idempotency with repeated requests using the same `idempotency_key`.
