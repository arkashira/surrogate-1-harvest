# workio / backend

## 1. Diagnosis

- Missing **database-level guard** for open punches: no partial unique index prevents multiple `clock_out_at IS NULL` rows per `(user_id, date(clock_in_at))`, allowing duplicates under LINE webhook retries or client double-taps.
- Race-prone **find-then-insert** logic in clock-in/clock-out handlers (app-layer check + insert) is not safe under concurrency.
- No **idempotency key** or deduplication on LINE webhook events, so retries create duplicate punches.
- Clock-in validation (location/GPS) runs after the race window, so duplicates can bypass checks.
- No **atomic upsert** path for “close open punch” that is safe under concurrent webhook deliveries.

## 2. Proposed change

- **Scope**: `server/src/db/schema.sql` (add partial unique index) + `server/src/routes/punch.ts` (replace find-then-insert with atomic upsert + idempotency guard).
- **Goal**: enforce at most one open punch per user per calendar day at the DB level and make clock-in/clock-out handlers safe under concurrency and LINE retries.

## 3. Implementation

### 3.1 Add partial unique index (schema.sql)

```sql
-- server/src/db/schema.sql
-- Add after punches table definition

-- Ensure at most one open punch per user per calendar day
CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_one_open_per_user_per_day
ON punches (user_id, date(clock_in_at))
WHERE clock_out_at IS NULL;
```

### 3.2 Idempotent atomic clock-in handler (punch.ts)

```ts
// server/src/routes/punch.ts
import { Request, Response } from 'express';
import { db } from '../db';
import { sql } from 'drizzle-orm'; // or raw SQL if preferred

// Expects body: { userId, lineUserId, latitude, longitude, lineEventId }
// lineEventId is used as idempotency key (store processed IDs to reject replays)
export async function clockIn(req: Request, res: Response) {
  const { userId, lineUserId, latitude, longitude, lineEventId } = req.body;

  if (!userId || !lineEventId) {
    return res.status(400).json({ error: 'userId and lineEventId required' });
  }

  const today = new Date().toISOString().split('T')[0]; // YYYY-MM-DD

  return db.transaction(async (tx) => {
    // 1) Idempotency: reject already-processed LINE event
    const seen = await tx.execute<{ id: string }>`
      SELECT id FROM processed_line_events WHERE id = ${lineEventId}
    `;
    if (seen.rows.length > 0) {
      return res.status(200).json({ ok: true, message: 'already processed' });
    }

    // 2) Try to close any stale open punch for same user+calendar day (defensive)
    await tx.execute(sql`
      UPDATE punches
      SET clock_out_at = NOW(),
          updated_at = NOW(),
          note = COALESCE(note, '') || ' auto-closed by new clock-in'
      WHERE user_id = ${userId}
        AND date(clock_in_at) = ${today}::date
        AND clock_out_at IS NULL
    `);

    // 3) Insert new punch (will fail on partial unique index if race inserts concurrently)
    try {
      const [punch] = await tx.execute<{ id: string }>`
        INSERT INTO punches (user_id, line_user_id, clock_in_at, latitude, longitude)
        VALUES (${userId}, ${lineUserId}, NOW(), ${latitude}, ${longitude})
        RETURNING id
      `;

      // 4) Record processed event (best-effort; keep small TTL in prod)
      await tx.execute(sql`
        INSERT INTO processed_line_events (id, created_at)
        VALUES (${lineEventId}, NOW())
        ON CONFLICT (id) DO NOTHING
      `);

      return res.status(201).json({ ok: true, punchId: punch.id });
    } catch (err: any) {
      // Unique violation from partial index => duplicate open punch
      if (err.code === '23505') {
        return res.status(409).json({ error: 'open punch already exists' });
      }
      throw err;
    }
  });
}
```

### 3.3 Atomic clock-out handler (punch.ts)

```ts
export async function clockOut(req: Request, res: Response) {
  const { userId, lineEventId, latitude, longitude } = req.body;

  if (!userId || !lineEventId) {
    return res.status(400).json({ error: 'userId and lineEventId required' });
  }

  return db.transaction(async (tx) => {
    // Idempotency
    const seen = await tx.execute<{ id: string }>`
      SELECT id FROM processed_line_events WHERE id = ${lineEventId}
    `;
    if (seen.rows.length > 0) {
      return res.status(200).json({ ok: true, message: 'already processed' });
    }

    // Atomically close the open punch for this user (today)
    const [result] = await tx.execute(sql`
      UPDATE punches
      SET clock_out_at = NOW(),
          latitude_out = ${latitude},
          longitude_out = ${longitude},
          updated_at = NOW()
      WHERE user_id = ${userId}
        AND date(clock_in_at) = date(NOW())
        AND clock_out_at IS NULL
      RETURNING id
    `);

    if (!result?.id) {
      return res.status(404).json({ error: 'no open punch to close' });
    }

    await tx.execute(sql`
      INSERT INTO processed_line_events (id, created_at)
      VALUES (${lineEventId}, NOW())
      ON CONFLICT (id) DO NOTHING
    `);

    return res.status(200).json({ ok: true, punchId: result.id });
  });
}
```

### 3.4 Supporting table for idempotency (schema.sql)

```sql
-- server/src/db/schema.sql
CREATE TABLE IF NOT EXISTS processed_line_events (
  id TEXT PRIMARY KEY,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Optional: index for fast expiry if you plan to prune
CREATE INDEX IF NOT EXISTS idx_processed_line_events_created_at
ON processed_line_events (created_at);
```
