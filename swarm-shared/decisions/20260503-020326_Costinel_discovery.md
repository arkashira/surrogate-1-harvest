### 3) Mac orchestration script (run once / on schedule)

Produces `public/knowledge/hub-signal.json` with CDN URLs and file list.

```bash
#!/usr/bin/env bash
set -euo pipefail

OWNER="axentx"
REPO="costinel-knowledge"
DATE="2026-05-03"
BASE_PATH="batches/mirror-merged/${DATE}"
OUT_DIR="public/knowledge"
OUT_FILE="${OUT_DIR}/hub-signal.json"

mkdir -p "${OUT_DIR}"

# List parquet files non-recursively
FILES=$(python -c "
import json
from huggingface_hub import list_repo_tree
items = list_repo_tree(
    repo='${REPO}',
    path='${BASE_PATH}',
    repo_type='dataset',
    recursive=False
)
print(json.dumps([f.rfilename for f in items if f.rfilename.endswith('.parquet')]))
")

CDN_PREFIX="https://huggingface.co/datasets/${OWNER}/${REPO}/resolve/main/${BASE_PATH}"

# Build hub-signal.json (proposals can be enriched later by knowledge-rag)
python -c "
import json
files = ${FILES}
data = {
    'hub': 'MOC',
    'title': 'Mission Operations Center',
    'summary': 'Central coordination for cloud ops; highest graph centrality for cost signals.',
    'date': '${DATE}',
    'cdnPrefix': '${CDN_PREFIX}',
    'files': files,
    'proposals': [],
    'generatedAt': '$(date -u +%Y-%m-%dT%H:%M:%SZ)'
}
with open('${OUT_FILE}', 'w') as f:
    json.dump(data, f, indent=2)
print('Wrote ${OUT_FILE}')
"
```
