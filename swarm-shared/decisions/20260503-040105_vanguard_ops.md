# vanguard / ops

## 1. Diagnosis
- No content-addressed manifest: ingestion/training scripts re-list HF repos at runtime, causing 429 rate-limits and non-reproducible runs.
- Mixed-schema files from `dataset-mirror` land in `enriched/` without projection to `{prompt,response}`, breaking downstream surrogate-1 training.
- No CDN-bypass strategy: training jobs hit `/api/` endpoints during data loading instead of using public CDN URLs, wasting rate-limit budget.
- Lightning Studio lifecycle is not guarded: idle timeouts kill long-running training and scripts don’t check/reuse running studios, burning quota.
- No single-file, ops-first orchestration script that glues HF pre-listing → CDN fetch → Lightning launch with zero runtime API calls.

## 2. Proposed change
Create `/opt/axentx/vanguard/ops/surrogate1_train_ops.py` (single orchestration script) and add `/opt/axentx/vanguard/ops/requirements.txt` if missing.  
Scope: one new file (~180 lines), no changes to existing application code. This is the highest-value ops improvement deliverable in <2h.

# ── Helpers ────────────────────────────────────────────────────────
def hf_api_get(path: str, params: dict | None = None, retries: int = 3):
    url = f"https://hugface.co/api/{path}" if "hugface" in "https://huggingface.co" else f"https://huggingface.co/api/{path}"
    headers = {"Authorization": f"Bearer {HF_TOKEN}"} if HF_TOKEN else {}
    for attempt in range(retries):
        r = requests.get(url, headers=headers, params=params, timeout=30)
        if r.status_code == 429:
            wait = 360
            print(f"[HF API] 429 rate-limit — waiting {wait}s (attempt {attempt+1}/{retries})")
            time.sleep(wait)
            continue
        r.raise_for_status()
        return r.json()
    raise RuntimeError(f"HF API failed after {retries} retries")

def list_hf_folder(repo: str, folder: str) -> list[str]:
    """
    List files non-recursively. Returns relative paths.
    Uses HF API (once) so we can embed into manifest.
    """
    # tree endpoint may not exist for all repos; fallback to repo/tree
    try:
        tree = hf_api_get(f"datasets/{repo}/tree", params={"path": folder, "recursive": False})
    except Exception:
        # fallback: repo/files with pagination=False equivalent via recursive=False
        tree = hf_api_get(f"datasets/{repo}/tree", params={"path": folder, "recursive": False})
    files = []
    for entry in tree:
        if entry.get("type") == "file":
            files.append(f"{folder}/{entry['path']}" if not entry["path"].startswith(folder) else entry["path"])
    return sorted(files)

def build_manifest():
    """Single API call to list folder; save manifest for reproducible runs."""
    print(f"[manifest] listing {HF_REPO}/{HF_FOLDER} (non-recursive)")
    files = list_hf_folder(HF_REPO, HF_FOLDER)
    manifest = {
        "repo": HF_REPO,
        "folder": HF_FOLDER,
        "created_utc": datetime.now(timezone.utc).isoformat(),
        "files": files,
        "cdn_base": HF_CDN_BASE,
    }
    Path(MANIFEST_PATH).parent.mkdir(parents=True, exist_ok=True)
    with open(MANIFEST_PATH, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"[manifest] saved {len(files)} files -> {MANIFEST_PATH}")
    return manifest

def project_parquet_to_prompt_response(cdn_url: str, local_out: Path):
    """
    Download single parquet via CDN (no auth header required for public)
    and project to {prompt,response}. Keeps minimal memory footprint.
    """
    import pyarrow.parquet as pq
    import pandas as pd

    local_out.parent.mkdir(parents=True, exist_ok=True)
    # CDN download (no Authorization header)
    r = requests.get(cdn_url, timeout=120)
    r.raise_for_status()
    local_out.write_bytes(r.content)

    df = pq.read_table(local_out).to_pandas()
    # Heuristic projection: keep only prompt/response-like columns
    prompt_col = next((c for c in df.columns if "prompt" in c.lower()), None)
    response_col = next((c for c in df.columns if "response" in c.lower()), None)
    if prompt_col is None or response_col is None:
        # fallback: first two text/string cols
        text_cols = [c for c in df.columns if pd.api.types.is_string_dtype(df[c].dtype)]
        if len(text_cols) >= 2:
            prompt_col, response_col = text_cols[0], text_cols[1]
        else:
            raise ValueError(f"Cannot find prompt/response columns in {local_out}")
    projected = df[[prompt_col, response_col]].rename(columns={prompt_col: "prompt", response_col: "response"})
    projected = projected.dropna(subset=["prompt", "response"]).reset_index(drop=True)
    projected.to_parquet(local_out, index=False)
    print(f"[project] {local_out.name} -> {len(projected)} rows")
    return local_out
