## 1. Diagnosis (merged & corrected)
- **Non-idempotent webhooks**: LINE retries on 5xx/timeouts/network blips → duplicate clock/leave/OT events.  
- **Race conditions**: “Check-then-insert” without transactional upsert + unique constraints allows double punches under retries/concurrency.  
- **No deduplication key**: Missing server-side idempotency key on webhook payloads.  
- **Frontend staleness**: DB commits succeed, but frontend doesn’t see updates until hard refresh (no real-time push).  
- **Premature 200 risk**: Returning 200 before commit completion can cause LINE to retry on network blips; return 200 **after** commit and broadcast.

### 3.1 DB schema additions

```sql
-- Idempotency for LINE webhook deliveries
CREATE TABLE IF NOT EXISTS webhook_deliveries (
  id              SERIAL PRIMARY KEY,
  tenant_id       INTEGER NOT NULL,
  line_event_id   TEXT   NOT NULL,   -- stable id from LINE (event timestamp + type + source.userId)
  event_type      TEXT   NOT NULL,
  processed_at    TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  payload_hash    TEXT,
  UNIQUE(tenant_id, line_event_id)
);

-- Attendance uniqueness: one record per user per date per shift
CREATE UNIQUE INDEX IF NOT EXISTS uq_attendance_user_date_shift
  ON attendances(user_id, date, shift_type);

-- Leave/OT uniqueness: prevent overlapping requests for same type+start
CREATE UNIQUE INDEX IF NOT EXISTS uq_leave_user_start_type
  ON leave_requests(user_id, start_time, request_type);
```

### 3.2 Idempotent webhook handler (`server/src/routes/webhook/line.ts`)
```ts
import { db } from '../../db';
import { attendanceService } from '../../services/attendanceService';
import { broadcastToTenant } from '../../services/sseService';

export async function handleLineWebhook(req, res) {
  const { events } = req.body;
  const tenantId = req.headers['x-tenant-id'] || process.env.DEFAULT_TENANT_ID;

  if (!events?.length) return res.status(200).send('OK');

  try {
    for (const ev of events) {
      // Stable idempotency key derived from LINE payload
      const lineEventId = `${ev.source?.userId}:${ev.timestamp}:${ev.type}`;

      await db.tx(async (t) => {
        // Idempotency check inside transaction
        const exists = await t.oneOrNone(
          'SELECT 1 FROM webhook_deliveries WHERE tenant_id = $1 AND line_event_id = $2',
          [tenantId, lineEventId]
        );
        if (exists) return;

        // Record delivery first
        await t.none(
          'INSERT INTO webhook_deliveries(tenant_id, line_event_id, event_type) VALUES($1,$2,$3)',
          [tenantId, lineEventId, ev.type]
        );

        if (ev.type === 'message' && ev.message?.text) {
          const result = await attendanceService.handleTextCommand(t, {
            tenantId,
            userId: ev.source.userId,
            text: ev.message.text,
            timestamp: ev.timestamp
          });

          // Broadcast after successful commit (tx completes before leaving handler)
          broadcastToTenant(tenantId, {
            type: 'attendance_update',
            payload: result
          });
        }
        // Extend here for leave/OT commands
      });
    }

    res.status(200).send('OK');
  } catch (err) {
    // 5xx will trigger LINE retry; log for observability
    console.error('LINE webhook processing failed', err);
    res.status(500).send('Processing failed');
  }
}
```

### 3.3 Attendance service with transactional upsert (`server/src/services/attendanceService.ts`)
```ts
export const attendanceService = {
  async handleTextCommand(tx, { tenantId, userId, text, timestamp }) {
    const normalized = (text || '').trim().toLowerCase();
    const today = new Date().toISOString().slice(0, 10);

    if (normalized === 'clock in' || normalized === 'clock out') {
      const action = normalized === 'clock in' ? 'in' : 'out';
      const col = action === 'in' ? 'clock_in' : 'clock_out';

      // Try update existing today row
      const updated = await tx.oneOrNone(
        `UPDATE attendances
            SET ${col} = to_timestamp($1), updated_at = NOW()
            WHERE user_id = $2 AND date = $3
            RETURNING *`,
        [timestamp, userId, today]
      );
      if (updated) return updated;

      // Insert new row if none exists
      const inserted = await tx.one(
        `INSERT INTO attendances(tenant_id, user_id, date, shift_type, clock_in, clock_out)
         VALUES($1, $2, $3, 'regular',
           CASE WHEN $4 = 'in' THEN to_timestamp($5) ELSE NULL END,
           CASE WHEN $4 = 'out' THEN to_timestamp($5) ELSE NULL END)
         RETURNING *`,
        [tenantId, userId, today, action, timestamp]
      );
      return inserted;
    }

    // Extend for leave/OT with similar upsert + unique constraint handling
    return { ok: false, reason: 'unhandled' };
  }
};
```

### 3.4 Lightweight SSE broadcaster (`server/src/services/sseService.ts`)
```ts
const clients = new Map<string, Set<any>>(); // tenantId -> Set<res>

export function addClient(tenantId: string, res: any) {
  if (!clients.has(tenantId)) clients.set(tenantId, new Set());
  clients.get(tenantId)!.add(res);
  res.on('close', () => clients.get(tenantId)?.delete(res));
}

export function broadcastToTenant(tenantId: string, data: any) {
  const payload = `data: ${JSON.stringify(data)}\n\n`;
  for (const res of clients.get(tenantId) || []) {
    try {
      res.write(payload);
    } catch {
      // ignore broken connections
    }
  }
}
```

### 3.5 SSE route (`server/src/routes/events.ts`)
```ts
import express from 'express';
import { addClient } from '../services/sseService';

const router = express.Router();

router.get('/stream', (req, res) => {
  const tenantId = req.query.tenant_id as string;
  if (!tenantId) return res.status(400).send('tenant_id required');

  res.setHeader('Content-Type', 'text/event-stream');
  res.setHeader('Cache-Control', 'no-cache');
  res.setHeader('Connection', 'keep-alive');
  res.flushHeaders();

  addClient(tenantId, res);

  // Heartbeat to keep connection alive
  res.write('retry: 15000\n\n');
});

export default router;
```

## 4. Verification (actionable checks)
- **Idempotency**: Replay same LINE webhook payload (same derived `line_event_id`) twice → second must not create duplicate attendance row. Confirm exactly one `webhook_deliveries` entry and unchanged attendances count.  
- **Uniqueness/constraints**: Simulate concurrent clock-ins for same user/date (e.g., `ab -n 10 -c 5`). Verify no double rows; duplicates either update existing row or are rejected by unique index
