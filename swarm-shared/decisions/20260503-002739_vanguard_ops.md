# vanguard / ops

## Final Consolidated Implementation (Best Parts + Correctness + Actionability)

### 1. Diagnosis (Consolidated)
- **No persisted `(repo, dateFolder)` manifest** → every training run re-enumerates via authenticated HF API → quota burn + 429 risk.
- **Data loading uses recursive `list_repo_tree`/`load_dataset` during training** instead of CDN-only fetches → unnecessary API calls and rate-limit exposure.
- **No Lightning Studio reuse / idle-stop resilience** → repeated studio creation wastes quota; idle timeouts kill long jobs with no auto-restart.
- **Missing wrapper hygiene** (shebang, executable bit, deterministic env) → cron jobs (e.g., `opus-pr-reviewer`, active-learning) fail silently.
- **Missing deterministic HF ingestion sharding** for sibling-repo writes (risk of collisions and non-reproducible commits).

---

### 2. Single Orchestration Artifact
Create **one** canonical module:

```
/opt/axentx/vanguard/ops/training_manifest.py
```

Responsibilities:
- Generate and cache `manifest-{repo}-{dateFolder}.json` after a **single authenticated** `list_repo_tree` call.
- Provide deterministic HF ingestion sharding (sibling repo writes).
- Expose a small Python API for reuse by training wrappers and cron jobs.

---

### 3. Implementation

#### 3.1 Manifest generator (single source of truth)

```python
# /opt/axentx/vanguard/ops/training_manifest.py
#!/usr/bin/env python3
"""
Generate and cache repo+dateFolder manifest for CDN-only training.
Usage:
  python training_manifest.py <repo> <dateFolder> [--force]
  python training_manifest.py --shard <sibling_repo> <base_repo> <dateFolder> [--shards N]
"""
import json, os, sys, hashlib
from pathlib import Path
from huggingface_hub import HfApi, hf_hub_download, CommitOperationAdd

MANIFEST_DIR = Path(__file__).parent.parent / "manifests"
MANIFEST_DIR.mkdir(parents=True, exist_ok=True)

def _list_files(repo: str, date_folder: str) -> list[str]:
    api = HfApi()
    top = api.list_repo_tree(repo=repo, path=date_folder, recursive=False)
    files = []
    for e in top:
        if getattr(e, "type", None) == "file":
            files.append(e.path)
        elif getattr(e, "type", None) == "directory":
            sub = api.list_repo_tree(repo=repo, path=e.path, recursive=False)
            for s in sub:
                if getattr(s, "type", None) == "file":
                    files.append(s.path)
    return sorted(set(files))

def get_manifest_path(repo: str, date_folder: str) -> Path:
    safe = f"{repo.replace('/', '-')}-{date_folder}.json"
    return MANIFEST_DIR / safe

def build_manifest(repo: str, date_folder: str, force: bool = False) -> dict:
    manifest_path = get_manifest_path(repo, date_folder)
    if manifest_path.exists() and not force:
        return json.loads(manifest_path.read_text())

    files = _list_files(repo, date_folder)
    manifest = {
        "repo": repo,
        "dateFolder": date_folder,
        "files": files,
        "cdn_prefix": f"https://huggingface.co/datasets/{repo}/resolve/main",
    }
    manifest_path.write_text(json.dumps(manifest, indent=2))
    return manifest

def deterministic_shard(
    sibling_repo: str,
    base_repo: str,
    date_folder: str,
    shards: int = 1,
    shard_index: int | None = None,
) -> dict:
    """
    Deterministic sibling-repo write sharding for HF ingestion.
    Returns subset of files assigned to this shard.
    """
    manifest = build_manifest(base_repo, date_folder)
    files = manifest["files"]
    if shards < 1:
        raise ValueError("shards must be >= 1")

    assignments = {}
    for i, f in enumerate(files):
        assignments[f] = i % shards

    if shard_index is None:
        # infer from sibling repo name for reproducibility
        h = hashlib.sha256(sibling_repo.encode()).digest()
        shard_index = int.from_bytes(h[:2], "big") % shards

    shard_files = [f for f in files if assignments[f] == shard_index]
    out = {
        "sibling_repo": sibling_repo,
        "base_repo": base_repo,
        "dateFolder": date_folder,
        "shard_index": shard_index,
        "shards": shards,
        "files": shard_files,
        "cdn_prefix": manifest["cdn_prefix"],
    }
    # persist shard manifest for traceability
    shard_path = MANIFEST_DIR / f"{sibling_repo.replace('/', '-')}-{date_folder}-shard{shard_index}.json"
    shard_path.write_text(json.dumps(out, indent=2))
    return out

def commit_shard_to_sibling(
    sibling_repo: str,
    base_repo: str,
    date_folder: str,
    shards: int = 1,
    commit_message: str | None = None,
) -> None:
    """
    Deterministically assign and commit this shard's files to sibling_repo
    (avoids collisions; idempotent within shard).
    """
    shard_index = None  # auto-determine from sibling_repo name
    shard = deterministic_shard(sibling_repo, base_repo, date_folder, shards, shard_index)

    api = HfApi()
    operations = []
    for f in shard["files"]:
        # Use CDN URL to stream content without auth during ingestion preparation
        url = f"{shard['cdn_prefix']}/{f}"
        local_path = hf_hub_download(
            repo_id=sibling_repo,
            filename=f,
            repo_type="dataset",
            local_files_only=False,
            force_download=False,
        )
        # If file doesn't exist in sibling, create empty placeholder to stage write path
        # (real ingestion pipelines will replace with processed content)
        operations.append(
            CommitOperationAdd(
                path_in_repo=f,
                path_or_fileobj=local_path if os.path.exists(local_path) else b"",
            )
        )

    if not operations:
        print("No operations for this shard.")
        return

    api.create_commit(
        repo_id=sibling_repo,
        repo_type="dataset",
        operations=operations,
        commit_message=commit_message or f"shard {shard['shard_index']}/{shards} for {base_repo}/{date_folder}",
    )
    print(f"Committed shard {shard['shard_index']} to {sibling_repo}")

if __name__ == "__main__":
    # CLI entrypoint
    if len(sys.argv) < 3:
        print(__doc__)
        sys.exit(1)

    if sys.argv[1] == "--shard":
        if len(sys.argv) < 5:
            print("Usage: training_manifest.py --shard <sibling_repo> <base_repo> <dateFolder> [--shards N]")
            sys.exit(1)
        sibling, base, datef = sys.argv[2], sys.argv[3], sys.argv[4]
        shards = 1
        if "--shards" in sys.argv:
            idx = sys.argv.index("--shards")
            shards = int(sys.argv[idx + 1])
        result = deterministic_shard(sibling, base, datef, shards=shards)
        print(json.dumps(result, indent=2))
    else:
        repo, datef = sys.argv[1], sys.argv[2]
        force = "--force" in sys.argv
        result = build_manifest(repo, datef, force=force)
        print(json.dumps(result, indent=2))
```

Make executable:

```bash
chmod +x /opt/axentx/vanguard/ops/training_manifest.py
```
