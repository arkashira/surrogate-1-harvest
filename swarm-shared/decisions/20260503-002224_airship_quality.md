## Final synthesized implementation (best of both proposals)

**Core fix**: Eliminate HF API calls during training and mixed-schema errors by (1) converting ingestion to uniform `{prompt,response}` parquet with deterministic shard names, (2) generating a CDN file manifest once on the Mac orchestrator, and (3) training exclusively from CDN URLs. Reuse a running Lightning Studio to avoid quota waste and idle-stop churn.

## Code snippets

### 1) Ingestion writer (uniform schema + deterministic shards)

```python
# dataset_mirror/writer.py
import pyarrow as pa
import pyarrow.parquet as pq
import hashlib
from datetime import datetime, timezone
from pathlib import Path

HF_REPO = "axentx/surrogate-1-mirror"
OUT_PREFIX = "batches/mirror-merged"

def project_to_pair(raw: dict) -> dict:
    return {
        "prompt": str(raw.get("prompt", "")),
        "response": str(raw.get("response", "")),
    }

def deterministic_slug(content: dict) -> str:
    h = hashlib.sha256()
    h.update(content["prompt"].encode("utf-8"))
    h.update(b"\0")
    h.update(content["response"].encode("utf-8"))
    return h.hexdigest()[:16]

def write_pair(raw: dict, date: str):
    pair = project_to_pair(raw)
    slug = deterministic_slug(pair)
    table = pa.table(pair)
    table = table.cast(
        pa.schema([pa.field("prompt", pa.string()), pa.field("response", pa.string())])
    )
    out_dir = Path(f"{OUT_PREFIX}/{date}")
    out_dir.mkdir(parents=True, exist_ok=True)
    out_path = out_dir / f"{slug}.parquet"
    pq.write_table(table, out_path)
    return str(out_path)
```

### 2) File lister (run once on Mac; commit manifest)

```python
# tools/list_train_files.py
import json
from datetime import datetime, timedelta, timezone
from huggingface_hub import list_repo_tree

HF_REPO = "axentx/surrogate-1-mirror"
BASE = "batches/mirror-merged"

def list_date_folder(date: str):
    entries = list_repo_tree(repo_id=HF_REPO, path=f"{BASE}/{date}", recursive=False)
    return [e.rfilename for e in entries if e.rfilename.endswith(".parquet")]

def build_manifest(dates):
    manifest = []
    for d in dates:
        for f in list_date_folder(d):
            manifest.append(
                f"https://huggingface.co/datasets/{HF_REPO}/resolve/main/{f}"
            )
    return manifest

if __name__ == "__main__":
    today = datetime.now(timezone.utc).date()
    dates = [(today - timedelta(days=i)).isoformat() for i in range(7)]
    manifest = build_manifest(dates)
    with open("train_files.json", "w") as fp:
        json.dump(manifest, fp, indent=2)
    print(f"Wrote {len(manifest)} files to train_files.json")
```

Run once after rate-limit window; commit `train_files.json`.

### 3) Training loader (CDN-only, no HF API)

```python
# surrogate/train.py
import json
import pyarrow.dataset as ds
import polars as pl
from torch.utils.data import IterableDataset

class CDNParquetIterable(IterableDataset):
    def __init__(self, manifest_path="train_files.json", batch_size=1024):
        with open(manifest_path) as f:
            self.urls = json.load(f)
        self.batch_size = batch_size

    def __iter__(self):
        for url in self.urls:
            dataset = ds.dataset(url, format="parquet")
            for batch in dataset.to_batches(batch_size=self.batch_size):
                df = pl.from_arrow(batch)
                for row in df.iter_rows(named=True):
                    yield row["prompt"], row["response"]
```

Use this in your training loop; no `load_dataset`, no auth, no 429s.

### 4) Lightning Studio reuse + idle guard

```python
# surrogate/lightning_launcher.py
from lightning import Lightning, Teamspace, Machine, Studio

lightning = Lightning()
teamspace = Teamspace()

def get_or_start_studio(name="surrogate-train-l40s"):
    for s in teamspace.studios:
        if s.name == name and s.status == "Running":
            return s
    # restart if stopped
    machine = Machine.L40S  # fallback to public tier if unavailable
    return Studio(name=name, machine=machine, create_ok=True)

studio = get_or_start_studio()
if studio.status != "Running":
    studio.start(machine=Machine.L40S)

job = studio.run(
    entry_point="train.py",
    arguments=["--manifest", "train_files.json"],
    environment="surrogate-train-env",
)
```
