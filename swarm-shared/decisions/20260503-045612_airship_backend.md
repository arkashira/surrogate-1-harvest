## Final Implementation Plan: Resilient Surrogate-1 Training Pipeline (HF CDN Bypass)

**Highest-value improvement**: Eliminate HF API rate-limit failures during training by generating a file manifest once and embedding CDN-only fetches in Lightning training. This bypasses rate limits, respects commit caps, and enables fast iteration.

**Why this ships <2h**: Single new script + small train.py patch. No infra changes, no new deps.

### 1) Generate file manifest (run once from Mac after rate-limit window clears)

```bash
# /opt/axentx/airship/surrogate/scripts/generate_manifest.py
#!/usr/bin/env python3
"""
Generate training file manifest for Surrogate-1.
Run from Mac (or any dev box) after HF API window clears.
Outputs: manifests/training_files_YYYYMMDD.json
"""
import json, os, datetime
from huggingface_hub import HfApi

REPO = "axentx/surrogate-1-ingest"
DATE_FOLDER = "batches/mirror-merged/2026-05-03"  # or derive from arg
OUT_DIR = "manifests"
os.makedirs(OUT_DIR, exist_ok=True)

api = HfApi()
# Single API call: non-recursive per folder to avoid pagination explosion
folders = api.list_repo_tree(repo_id=REPO, path=DATE_FOLDER, recursive=False)

# Collect parquet files only (skip attribution/source cols; schema already projected)
files = [
    f.rfilename
    for f in folders
    if f.rfilename.endswith(".parquet")
]

manifest = {
    "repo": REPO,
    "date_folder": DATE_FOLDER,
    "generated_at": datetime.datetime.utcnow().isoformat() + "Z",
    "files": sorted(files),
    "total_files": len(files),
    # CDN-only URLs (no auth, bypass /api/ rate limits)
    "cdn_base": f"https://huggingface.co/datasets/{REPO}/resolve/main"
}

out_path = os.path.join(OUT_DIR, f"training_files_{datetime.date.today().isoformat()}.json")
with open(out_path, "w") as f:
    json.dump(manifest, f, indent=2)

print(f"Manifest written: {out_path} ({len(files)} files)")
```

### 2) Embed manifest in Lightning training (CDN-only loader)

```python
# /opt/axentx/airship/surrogate/train.py  (patch)
import json, os
from pathlib import Path
import torch
from torch.utils.data import IterableDataset
import pyarrow.parquet as pq
import requests
from io import BytesIO

MANIFEST_PATH = Path(__file__).parent / "manifests" / "training_files_2026-05-03.json"
with open(MANIFEST_PATH) as f:
    manifest = json.load(f)

CDN_BASE = manifest["cdn_base"]
FILES = manifest["files"]

class CDNParquetIterable(IterableDataset):
    """Zero HF API calls during training. CDN-only downloads."""
    def __init__(self, files, cdn_base, shuffle=True):
        self.files = files
        self.cdn_base = cdn_base
        self.shuffle = shuffle
        self.epoch = 0

    def __iter__(self):
        worker_info = torch.utils.data.get_worker_info()
        # Deterministic per-epoch shuffling without API
        epoch_seed = self.epoch + (worker_info.seed if worker_info else 0)
        rng = torch.Generator()
        rng.manual_seed(epoch_seed)
        indices = torch.randperm(len(self.files), generator=rng).tolist()

        for idx in indices:
            url = f"{self.cdn_base}/{self.files[idx]}"
            # CDN fetch (no auth header) — high CDN limits, no 429
            resp = requests.get(url, timeout=30)
            resp.raise_for_status()
            buf = BytesIO(resp.content)
            table = pq.read_table(buf, columns=["prompt", "response"])
            for row in table.to_pylist():
                # Project to {prompt, response} only (schema already clean)
                yield row["prompt"], row["response"]
        self.epoch += 1

# Usage in Lightning DataModule
train_dataset = CDNParquetIterable(FILES, CDN_BASE, shuffle=True)
```

### 3) Reuse running Lightning Studio (quota-safe)

```python
# /opt/axentx/airship/surrogate/scripts/launch_training.py
#!/usr/bin/env python3
from lightning_sdk import Teamspace, Studio, Machine
import sys

teamspace = Teamspace()
studio_name = "surrogate-1-train"
running = [s for s in teamspace.studios if s.name == studio_name and s.status == "Running"]

if running:
    studio = running[0]
    print(f"Reusing running studio: {studio.id}")
else:
    studio = teamspace.studios.create(
        name=studio_name,
        machine=Machine.L40S,  # free tier fallback; H200 requires lightning-lambda-prod
        container="pytorch/pytorch:latest",
    )
    studio.start()
    print(f"Started new studio: {studio.id}")

# Run training (idempotent)
job = studio.run(
    command="python3 /workspace/surrogate/train.py --epochs 3",
    environment={"PYTHONPATH": "/workspace"},
)
print(f"Job submitted: {job.id}")
```
