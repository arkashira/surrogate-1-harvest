# workio / design

## 1. Diagnosis

- **Missing idempotency key**: punch records rely on `(employee_id, date)` uniqueness but LINE retries can deliver the same event multiple times within seconds; no stable request-level dedupe key exists.
- **Non-atomic upsert path**: upsert uses `ON CONFLICT DO UPDATE` but application logic can race under concurrent webhook deliveries (same millisecond) and produce duplicate rows before constraint is checked.
- **No webhook replay protection**: no persisted `line_delivery_id` or signature verification window, so retries and replays are processed as new punches.
- **Tenant isolation not enforced at DB level**: uniqueness and row-level security depend on app code; a missing tenant filter in queries can leak data across tenants.
- **Missing audit trail**: punch changes (create/update) have no immutable log for compliance or debugging.

## 2. Proposed change

- **Files**:
  - `server/src/db/schema.sql` — add `line_delivery_id`, tenant-scoped uniqueness, RLS, and audit table.
  - `server/src/db/migrations/` (optional) — migration to backfill safely.
  - `server/src/services/punchService.ts` — switch to atomic upsert using `line_delivery_id` and tenant.
  - `server/src/middleware/lineWebhook.ts` — verify signature and reject replays quickly.
- **Scope**: schema + service layer (no UI changes). Target: <2h.

## 3. Implementation

### 3.1 Schema changes (`server/src/db/schema.sql`)

```sql
-- Add immutable delivery id and tenant to punches
ALTER TABLE punches
  ADD COLUMN IF NOT EXISTS line_delivery_id TEXT,
  ADD COLUMN IF NOT EXISTS tenant_id UUID NOT NULL DEFAULT '00000000-0000-0000-0000-000000000000',
  ADD COLUMN IF NOT EXISTS created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  ADD COLUMN IF NOT EXISTS updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW();

-- Enforce tenant isolation + idempotency
CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_tenant_employee_date_delivery
  ON punches (tenant_id, employee_id, punch_date, line_delivery_id)
  WHERE line_delivery_id IS NOT NULL;

-- Fallback unique constraint for legacy rows (delivery id absent)
CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_tenant_employee_date_type
  ON punches (tenant_id, employee_id, punch_date, punch_type)
  WHERE line_delivery_id IS NULL;

-- Row-level security (basic)
ALTER TABLE punches ENABLE ROW LEVEL SECURITY;

CREATE POLICY tenant_isolation_policy ON punches
  USING (tenant_id = (current_setting('app.current_tenant', TRUE))::UUID);

-- Audit table for punch changes
CREATE TABLE IF NOT EXISTS punch_audit (
  id BIGSERIAL PRIMARY KEY,
  punch_id UUID NOT NULL,
  tenant_id UUID NOT NULL,
  employee_id UUID NOT NULL,
  action TEXT NOT NULL, -- 'INSERT', 'UPDATE'
  old_data JSONB,
  new_data JSONB,
  changed_by UUID,
  changed_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX IF NOT EXISTS idx_punch_audit_punch_id ON punch_audit (punch_id);
CREATE INDEX IF NOT EXISTS idx_punch_audit_tenant_id ON punch_audit (tenant_id);
```

### 3.2 Service: atomic upsert (`server/src/services/punchService.ts`)

```ts
import { pool } from '../db';

export interface CreatePunchInput {
  tenantId: string;
  employeeId: string;
  punchDate: string; // YYYY-MM-DD
  punchType: 'clock_in' | 'clock_out' | 'break_start' | 'break_end';
  lineDeliveryId?: string;
  latitude?: number;
  longitude?: number;
}

export async function createOrUpdatePunch(input: CreatePunchInput) {
  const client = await pool.connect();
  try {
    await client.query('SET LOCAL app.current_tenant = $1', [input.tenantId]);

    // Atomic upsert keyed by line_delivery_id when present
    const upsertQuery = `
      INSERT INTO punches (
        tenant_id,
        employee_id,
        punch_date,
        punch_type,
        line_delivery_id,
        latitude,
        longitude
      )
      VALUES ($1, $2, $3, $4, $5, $6, $7)
      ON CONFLICT (tenant_id, employee_id, punch_date, line_delivery_id)
      DO UPDATE
        SET punch_type = EXCLUDED.punch_type,
            latitude = EXCLUDED.latitude,
            longitude = EXCLUDED.longitude,
            updated_at = NOW()
      WHERE punches.tenant_id = EXCLUDED.tenant_id
      RETURNING *
    `;

    const values = [
      input.tenantId,
      input.employeeId,
      input.punchDate,
      input.punchType,
      input.lineDeliveryId || null,
      input.latitude ?? null,
      input.longitude ?? null,
    ];

    const { rows } = await client.query(upsertQuery, values);
    const punch = rows[0];

    // Audit log
    await client.query(
      `INSERT INTO punch_audit (punch_id, tenant_id, employee_id, action, new_data, changed_by)
       VALUES ($1, $2, $3, 'INSERT', $4, $5)`,
      [punch.id, input.tenantId, input.employeeId, JSON.stringify(punch), input.employeeId]
    );

    return punch;
  } finally {
    client.release();
  }
}
```

### 3.3 Webhook handler guard (`server/src/middleware/lineWebhook.ts`)

```ts
import { Request, Response, NextFunction } from 'express';
import crypto from 'crypto';

const LINE_CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';

export function verifyLineSignature(req: Request, res: Response, next: NextFunction) {
  const signature = req.headers['x-line-signature'] as string;
  if (!signature) return res.status(401).json({ error: 'Missing signature' });

  const body = JSON.stringify(req.body);
  const expected = crypto
    .createHmac('sha256', LINE_CHANNEL_SECRET)
    .update(body)
    .digest('base64');

  if (signature !== expected) {
    return res.status(401).json({ error: 'Invalid signature' });
  }

  // Basic replay protection: reject events older than 5m
  const now = Date.now();
  const events = req.body.events || [];
  for (const ev of events) {
    if (ev.deliveryId && Math.abs(now - ev.timestamp) > 5 * 60 * 1000) {
      return res.status(400).json({ error: 'Possible replay' });
    }
  }

  next();
}
```

import { createOrUpdatePunch } from '../services/punchService';

export async function handleLineEvent(req, res) {
  const events = req.body.events || [];
  for (const ev of events) {
    if (ev.type !== 'message' || ev.message?.type !== 'text') continue;

    // Parse clock in/out command (example)
    const text = ev.message.text.trim();
    if (!['in', 'out'].includes(text)) continue;

    const tenantId = process.env.DEFAULT_TENANT_ID; // derive from source
    const employeeId = await resolveEmployeeByLineUserId(ev.source.userId);
    const punchDate = new Date().toISOString().slice(0, 10);
    const punchType = text === 'in' ? 'clock_in' : 'clock_out';

    await createOrUpdatePunch({
      tenantId,
      employeeId,
      punchDate,
      punchType,
      lineDeliveryId: ev.deliveryId || null,
    });
  }

  res.status(200).json({ ok: true });
}
