### 2) CDN-only DataLoader (surrogate/train.py)
```python
import json
import pyarrow.parquet as pq
import requests
from torch.utils.data import IterableDataset
from tqdm import tqdm

class CDNParquetDataset(IterableDataset):
    def __init__(self, file_list_path, max_retries=3):
        with open(file_list_path) as f:
            cfg = json.load(f)
        self.repo = cfg["repo"]
        self.base = f"https://huggingface.co/datasets/{self.repo}/resolve/main"
        self.files = [f"{cfg['date_folder']}/{fn}" for fn in cfg["files"]]
        self.max_retries = max_retries

    def _download_parquet(self, path):
        url = f"{self.base}/{path}"
        for attempt in range(self.max_retries):
            try:
                r = requests.get(url, timeout=30)
                r.raise_for_status()
                return pq.read_table(pq.ParquetFile(pq.BufferReader(r.content)))
            except Exception as e:
                if attempt == self.max_retries - 1:
                    raise
                # simple backoff
                import time; time.sleep(2 ** attempt)

    def __iter__(self):
        for fpath in self.files:
            tbl = self._download_parquet(fpath)
            # project to {prompt, response} only
            tbl = tbl.select(["prompt", "response"])
            for batch in tbl.to_batches(max_chunksize=512):
                for i in range(batch.num_rows):
                    yield {
                        "prompt": batch["prompt"][i].as_py(),
                        "response": batch["response"][i].as_py(),
                    }
```

### 3) Lightning idle-resilient runner (surrogate/lightning_runner.py)
```python
import time
from lightning import Lightning, Teamspace, Machine, Venv

def run_training_with_resilience(script_path, studio_name, max_restarts=3):
    lightning = Lightning()
    team = Teamspace()

    restarts = 0
    while restarts <= max_restarts:
        # reuse running studio if exists
        studio = None
        for s in team.studios:
            if s.name == studio_name:
                studio = s
                break

        if studio is None or studio.status != "Running":
            if studio and studio.status != "Running":
                print(f"Studio {studio_name} stopped (status={studio.status}). Restarting...")
            studio = lightning.studio(
                name=studio_name,
                create_ok=True,
                machine=Machine.L40S,
                venv=Venv(requirements=["requirements.txt"]),
            )

        try:
            run = studio.run(script_path, wait=False)
            # poll until done or stopped
            while True:
                run.refresh()
                studio.refresh()
                if studio.status != "Running":
                    print(f"Studio stopped during training (status={studio.status}). Will restart.")
                    break
                if run.status in ("finished", "failed", "crashed"):
                    print(f"Run finished with status={run.status}")
                    return run
                time.sleep(30)
        except Exception as e:
            print(f"Run error: {e}")

        restarts += 1
        if restarts <= max_restarts:
            print(f"Restart {restarts}/{max_restarts} after 60s...")
            time.sleep(60)
    raise RuntimeError("Max restarts exceeded")
```

### 4) Deterministic sibling-repo writer (surrogate/upload.py)
```python
import hashlib
from huggingface_hub import HfApi

SIBLINGS = [
    "my-org/surrogate-dataset",
    "my-org/surrogate-dataset-s1",
    "my-org/surrogate-dataset-s2",
    "my-org/surrogate-dataset-s3",
    "my-org/surrogate-dataset-s4",
]

def pick_repo(slug: str) -> str:
    h = int(hashlib.sha256(slug.encode()).hexdigest(), 16)
    return SIBLINGS[h % len(SIBLINGS)]

def upload_parquet(local_path, slug, subpath="batches/mirror-merged"):
    repo = pick_repo(slug)
    api = HfApi()
    # filename pattern: {date}/{slug}.parquet
    # (date extracted from subpath or slug as needed)
    api.upload_file(
        path_or_fileobj=local_path,
        path_in_repo=f"{subpath}/{slug}.parquet",
        repo_id=repo,
        repo_type="dataset",
    )
    return repo
```
