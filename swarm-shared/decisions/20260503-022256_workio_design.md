# workio / design

## Final Synthesized Solution (Correct + Actionable)

Below is the single, production-ready plan. It merges the strongest technical choices from both proposals, removes contradictions, and prioritizes **correctness** (atomicity, isolation, idempotency) and **concrete actionability** (exact schema, code, verification steps).

---

## 1. Diagnosis (merged, resolved)

- **Duplicate root cause**: LINE retries + non-atomic upsert + no dedup layer.
- **Race condition**: `findOne` → `insert`/`update` is not atomic; concurrent retries can double-insert.
- **Missing enforcement**: no unique constraint/index to guarantee one `(tenant, employee, date, event)` record.
- **Tenant isolation risk**: queries must always include `tenant_id` in predicates.
- **Stale client state**: endpoints must return the effective row after upsert.

## 2. Schema (single source of truth)

File: `workio/server/src/db/schema.sql`

```sql
-- Idempotency for LINE webhook deliveries (at-least-once -> exactly-once)
CREATE TABLE IF NOT EXISTS line_deliveries (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id       UUID NOT NULL REFERENCES tenants(id) ON DELETE CASCADE,
  message_id      TEXT NOT NULL,            -- X-Line-Message-Id or derived stable id
  event_type      TEXT NOT NULL,            -- clock_in, clock_out, etc.
  processed_at    TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  payload_hash    TEXT NOT NULL,            -- sha256(normalized_body)
  UNIQUE (tenant_id, message_id)
);

-- Guarantee one clock_in OR clock_out per tenant/employee/day
-- clock_in and clock_out are independent; allow one of each per day
CREATE UNIQUE INDEX IF NOT EXISTS uq_punch_tenant_employee_date_event
ON punches (tenant_id, employee_id, DATE(punched_at), event_type)
WHERE event_type IN ('clock_in', 'clock_out');
```

Notes:
- Use `ON CONFLICT` on the **index expression** `(tenant_id, employee_id, DATE(punched_at), event_type)` in the upsert.
- Keep `event_type` in the index predicate to allow one clock_in and one clock_out per day (not one total).

## 3. Service layer (single place for punch logic)

File: `workio/server/src/services/punchService.ts`

```ts
import { pool } from '../db';

export type PunchInput = {
  tenantId: string;
  employeeId: string;
  eventType: 'clock_in' | 'clock_out';
  punchedAt: string; // ISO
  latitude?: number;
  longitude?: number;
  imageUrl?: string;
};

export async function upsertPunch(input: PunchInput) {
  const { tenantId, employeeId, eventType, punchedAt, latitude, longitude, imageUrl } = input;
  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    const result = await client.query(
      `INSERT INTO punches (tenant_id, employee_id, event_type, punched_at, latitude, longitude, image_url)
       VALUES ($1, $2, $3, $4, $5, $6, $7)
       ON CONFLICT (tenant_id, employee_id, DATE(punched_at), event_type)
       DO UPDATE SET
         punched_at = EXCLUDED.punched_at,
         latitude = EXCLUDED.latitude,
         longitude = EXCLUDED.longitude,
         image_url = EXCLUDED.image_url,
         updated_at = NOW()
       RETURNING *`,
      [tenantId, employeeId, eventType, punchedAt, latitude, longitude, imageUrl]
    );

    await client.query('COMMIT');
    return result.rows[0];
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  } finally {
    client.release();
  }
}
```

Why this wins:
- Atomic upsert + `RETURNING` (no stale state).
- Explicit transaction (safe for future expansion).
- Reusable by any handler (webhook, API, CLI).

## 4. Idempotent LINE webhook middleware

File: `workio/server/src/middleware/lineIdempotency.ts`

```ts
import { Request, Response, NextFunction } from 'express';
import crypto from 'crypto';
import { pool } from '../db';

function sha256(body: string) {
  return crypto.createHash('sha256').update(body, 'utf8').digest('hex');
}

export async function ensureLineIdempotency(req: Request, res: Response, next: NextFunction) {
  const tenantId = req.headers['x-tenant-id'] as string;
  // Prefer explicit message-id; fallback to signature-derived stable id if needed
  const messageId = (req.headers['x-line-message-id'] as string) || deriveIdFromSignature(req);
  const rawBody = typeof req.body === 'string' ? req.body : JSON.stringify(req.body);
  const bodyHash = sha256(rawBody);

  if (!tenantId || !messageId) {
    // If no idempotency key, continue but prefer rejecting in prod
    return next();
  }

  const client = await pool.connect();
  try {
    const exists = await client.query(
      `SELECT 1 FROM line_deliveries WHERE tenant_id = $1 AND message_id = $2`,
      [tenantId, messageId]
    );

    if (exists.rows.length > 0) {
      // Already processed: short-circuit to stop LINE retries
      return res.status(200).json({ ok: true, reason: 'duplicate' });
    }

    // Record delivery before processing (best-effort; failure should not drop events)
    await client.query(
      `INSERT INTO line_deliveries (tenant_id, message_id, event_type, payload_hash)
       VALUES ($1, $2, $3, $4)`,
      [tenantId, messageId, req.path.includes('clock') ? 'clock' : 'webhook', bodyHash]
    );

    (req as any).lineDelivery = { tenantId, messageId };
    return next();
  } catch (err) {
    // Log but continue to avoid dropping legitimate events
    console.warn('Idempotency insert failed', err);
    return next();
  } finally {
    client.release();
  }
}

function deriveIdFromSignature(req: Request): string | null {
  const sig = req.headers['x-line-signature'] as string;
  if (!sig || !req.body) return null;
  const body = typeof req.body === 'string' ? req.body : JSON.stringify(req.body);
  // Deterministic, stable id derived from signature + body
  return crypto.createHash('sha256').update(sig + body).digest('hex');
}
```

Route wiring:

```ts
// workio/server/src/routes/lineWebhook.ts
import express from 'express';
import { ensureLineIdempotency } from '../middleware/lineIdempotency';
import { upsertPunch } from '../services/punchService';

const router = express.Router();
router.post('/webhook/line', ensureLineIdempotency, async (req, res) => {
  try {
    const punch = await upsertPunch(req.body);
    return res.json({ punch });
  } catch (err) {
    console.error('Punch upsert failed', err);
    return res.status(500).json({ error: 'Internal server error' });
  }
});
export default router;
```
