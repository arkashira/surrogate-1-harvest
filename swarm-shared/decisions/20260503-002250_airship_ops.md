## Final consolidated implementation (best of both proposals)

**Core problem**: Surrogate-1 ingestion and training still risk HF API 429s and slow reads because scripts may use `load_dataset(streaming=True)` or recursive listing.  
**Core solution**: Eliminate HF API calls during training by pre-listing one date folder and fetching files exclusively via CDN with deterministic sibling-repo upload and robust 429 handling.

## 1) CLI utility: list once, train zero-API

Add `tools/cdn_file_list.py` (run on Mac orchestrator or any dev machine).  
- Lists a single date folder non-recursively (one HF API call).  
- Writes `file-list-{date}.json` with CDN URLs.  
- Validates no recursive listing is used.  
- Supports dry-run and limit for safety.

```python
#!/usr/bin/env python3
"""
Generate a CDN-only file list for one date folder to avoid HF API calls during training.

Usage:
  python tools/cdn_file_list.py \
    --repo org/surrogate-data \
    --date-folder 2026-05-03 \
    --out file-list-2026-05-03.json \
    [--limit N] [--dry-run]
"""
import argparse
import json
import sys
from pathlib import Path
from typing import List, Dict, Any

from huggingface_hub import list_repo_tree

HF_TOKEN = Path(os.getenv("HF_TOKEN", "")).expanduser().read_text().strip() if os.getenv("HF_TOKEN") else None

def build_file_list(repo: str, date_folder: str, out_path: Path, limit: int = None, dry_run: bool = False) -> Dict[str, Any]:
    # Single non-recursive call (paginated 100x safe)
    tree = list_repo_tree(
        repo_id=repo,
        path=date_folder,
        recursive=False,
        token=HF_TOKEN,
    )

    entries: List[Dict[str, str]] = []
    for item in tree:
        if item.type != "file":
            continue
        path = item.path
        entries.append({
            "repo": repo,
            "path": path,
            "cdn_url": f"https://huggingface.co/datasets/{repo}/resolve/main/{path}",
        })
        if limit is not None and len(entries) >= limit:
            break

    manifest = {
        "repo": repo,
        "date_folder": date_folder,
        "generated_at": time.time(),
        "recursive": False,
        "files": entries,
    }

    if dry_run:
        print(json.dumps(manifest, indent=2))
        return manifest

    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"Wrote {len(entries)} files to {out_path}")
    return manifest

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Generate CDN file list for one date folder.")
    parser.add_argument("--repo", required=True, help="HF dataset repo, e.g. org/surrogate-data")
    parser.add_argument("--date-folder", required=True, help="Folder like 2026-05-03")
    parser.add_argument("--out", default="file-list.json", help="Output JSON path")
    parser.add_argument("--limit", type=int, default=None, help="Limit number of files (for testing)")
    parser.add_argument("--dry-run", action="store_true", help="Print manifest without writing")
    args = parser.parse_args()

    try:
        build_file_list(args.repo, args.date_folder, Path(args.out), limit=args.limit, dry_run=args.dry_run)
    except Exception as e:
        print(f"Error: {e}", file=sys.stderr)
        sys.exit(1)
```

## 2) Shared ingestion + projection module

Add `surrogate_data.py` (used by ingestion scripts and training).  
- Reads `file-list-{date}.json`.  
- Downloads via CDN (no HF API during training).  
- Projects heterogeneous files to `{prompt, response}`.  
- Writes sharded parquet to `batches/mirror-merged/{date}/{slug}.parquet`.  
- Deterministic sibling-repo selection for uploads.  
- Robust 429/backoff handling.

def _pick_sibling(slug: str) -> str:
    h = int(hashlib.sha256(slug.encode()).hexdigest(), 16)
    return SIBLING_REPOS[h % len(SIBLING_REPOS)]

def _wait_429():
    print("Encountered 429; waiting 360s")
    time.sleep(360)

def download_via_cdn(url: str, dest: Path, max_tries: int = 5) -> Path:
    for attempt in range(1, max_tries + 1):
        try:
            resp = requests.get(url, timeout=60)
            if resp.status_code == 429:
                _wait_429()
                continue
            resp.raise_for_status()
            dest.write_bytes(resp.content)
            return dest
        except Exception as exc:
            if attempt == max_tries:
                raise
            time.sleep(2 ** attempt)
    raise RuntimeError("Unreachable")

def project_to_pair(raw: Any) -> Dict[str, str]:
    """
    Best-effort projection for heterogeneous files.
    Returns {"prompt": ..., "response": ...}
    """
    if isinstance(raw, dict):
        prompt = raw.get("prompt") or raw.get("input") or raw.get("question") or ""
        response = raw.get("response") or raw.get("output") or raw.get("answer") or ""
        return {"prompt": str(prompt), "response": str(response)}
    if isinstance(raw, (list, tuple)) and len(raw) >= 2:
        return {"prompt": str(raw[0]), "response": str(raw[1])}
    return {"prompt": "", "response": str(raw)}

def build_parquet_from_filelist(
    file_list_path: Path,
    out_dir: Path,
    limit: int = None,
    chunk_size: int = 100_000,
) -> Path:
    manifest = json.loads(file_list_path.read_text())
    if manifest.get("recursive", False):
        raise ValueError("Manifest must be non-recursive. Regenerate with non-recursive listing.")

    files = manifest["files"]
    if limit:
        files = files[:limit]

    out_dir.mkdir(parents=True, exist_ok=True)
    rows: List[Dict[str, str]] = []
    total = 0

    for f in files:
        try:
            # Prefer direct CDN download to avoid HF API during ingestion/training
            local = download_via_cdn(f["cdn_url"], Path(f["path"]).name)
            content = local.read_text()
            for line in content.strip().splitlines():
                if not line.strip():
                    continue
                raw = json.loads(line)
                rows.append(project_to_pair(raw))

            # Periodic flush to avoid OOM
            if len(rows) >= chunk_size:
                _flush_rows(rows, out_dir, manifest["date_folder"], total)
                total += len(rows)
                rows.clear()
        except Exception as exc:
            print(f"Skipping {f['path']} due to {exc}")
            continue

    # Final flush
    if rows:
        _flush_rows(rows, out_dir, manifest["date_folder"], total)
        total += len(rows)

    slug = manifest["date_folder"]
    out_file = out_dir / f"{slug}.parquet"
    print(f"Wrote {total} rows to {out_file}")
    return out_file
