# Costinel / ops

## Final Implementation Plan — `GET /api/v1/cost-anomaly/signal/top-hub`

**Scope & constraints**
- Read-only, no side effects.
- Optional `?for_date=YYYY-MM-DD` (default today UTC) and `?top_n` (1–10, default 1).
- Reuse existing knowledge-graph assets; do not recompute heavy analytics.
- Return compact signal payload with hub context and top actionable anomalies.
- **Correctness priority**: deterministic hub selection, safe query patterns, explicit defaults, and concrete actionability.

**Estimated effort**: <2h (including tests + docs).

### 1) Add route + handler (FastAPI)

File: `app/api/v1/endpoints/cost_anomaly.py`

```python
from fastapi import APIRouter, Depends, Query
from datetime import date, datetime
from typing import List, Optional
from sqlalchemy.orm import Session

from app.schemas.cost_anomaly import HubInsight, CostAnomalySignal, TopHubSignal
from app.services.cost_anomaly.top_hub import build_top_hub_signal
from app.core.deps import get_db

router = APIRouter()

@router.get("/top-hub", response_model=TopHubSignal)
def get_cost_anomaly_top_hub(
    for_date: Optional[date] = Query(
        None,
        description="Date (YYYY-MM-DD), defaults to today UTC"
    ),
    top_n: int = Query(
        1,
        ge=1,
        le=10,
        description="Number of top anomalies to include"
    ),
    db: Session = Depends(get_db),
):
    """
    Sense + Signal: return the most-connected hub for the date and top-N actionable anomalies.
    No mutations. No execution.
    """
    target_date = for_date or datetime.utcnow().date()
    return build_top_hub_signal(db=db, for_date=target_date, top_n=top_n)
```

### 2) Service: build_top_hub_signal

File: `app/services/cost_anomaly/top_hub.py`

```python
from datetime import date, datetime
from typing import List
from sqlalchemy.orm import Session
from sqlalchemy import func

from app.models.knowledge import Hub, Edge
from app.models.cost import Anomaly
from app.schemas.cost_anomaly import HubInsight, CostAnomalySignal, TopHubSignal

def _to_hub_insight(hub: Hub) -> HubInsight:
    return HubInsight(
        id=hub.id,
        name=hub.name,
        slug=hub.slug,
        description=hub.description or "",
        tags=hub.tags or [],
        centrality=float(hub.centrality or 0.0),
        last_seen=hub.updated_at,
    )

def _to_anomaly_signal(anomaly: Anomaly) -> CostAnomalySignal:
    return CostAnomalySignal(
        id=anomaly.id,
        detected_at=anomaly.detected_at,
        for_date=anomaly.for_date,
        service=anomaly.service,
        account_id=anomaly.account_id,
        region=anomaly.region,
        metric=anomaly.metric,
        severity=anomaly.severity,
        score=float(anomaly.score),
        expected=float(anomaly.expected),
        actual=float(anomaly.actual),
        delta_pct=float(anomaly.delta_pct),
        context=anomaly.context or {},
        suggested_action=anomaly.suggested_action or "Review",
        status=anomaly.status,
    )

def build_top_hub_signal(db: Session, for_date: date, top_n: int = 1) -> TopHubSignal:
    # 1) Deterministic hub selection:
    # Prefer active hubs by centrality; fallback to edge count.
    hub = (
        db.query(Hub)
        .filter(Hub.active.is_(True))
        .order_by(Hub.centrality.desc(), Hub.updated_at.desc())
        .first()
    )

    if not hub:
        hub = (
            db.query(Hub)
            .join(Edge, (Edge.source_id == Hub.id) | (Edge.target_id == Hub.id))
            .group_by(Hub.id)
            .order_by(func.count(Edge.id).desc(), Hub.updated_at.desc())
            .first()
        )

    # Graceful fallback if no hub exists
    if not hub:
        hub_insight = HubInsight(
            id="unknown",
            name="N/A",
            slug="n/a",
            description="No hub available",
            tags=[],
            centrality=0.0,
            last_seen=None,
        )
    else:
        hub_insight = _to_hub_insight(hub)

    # 2) Top-N actionable anomalies for the date
    anomalies = (
        db.query(Anomaly)
        .filter(Anomaly.for_date == for_date, Anomaly.status.in_(("new", "investigating")))
        .order_by(Anomaly.score.desc(), Anomaly.detected_at.desc())
        .limit(top_n)
        .all()
    )

    signals = [_to_anomaly_signal(a) for a in anomalies]

    return TopHubSignal(
        request_at=datetime.utcnow(),
        for_date=for_date,
        top_hub=hub_insight,
        top_anomalies=signals,
        signal_summary=f"Top hub '{hub_insight.name}' with {len(signals)} actionable anomaly signals on {for_date}",
    )
```

### 3) Schemas

File: `app/schemas/cost_anomaly.py`

```python
from pydantic import BaseModel, Field
from datetime import datetime, date
from typing import List, Optional, Dict, Any

class HubInsight(BaseModel):
    id: str
    name: str
    slug: str
    description: str
    tags: List[str]
    centrality: float
    last_seen: Optional[datetime]

class CostAnomalySignal(BaseModel):
    id: str
    detected_at: datetime
    for_date: date
    service: str
    account_id: str
    region: str
    metric: str
    severity: str
    score: float
    expected: float
    actual: float
    delta_pct: float
    context: Dict[str, Any]
    suggested_action: str
    status: str

class TopHubSignal(BaseModel):
    request_at: datetime
    for_date: date
    top_hub: HubInsight
    top_anomalies: List[CostAnomalySignal]
    signal_summary: str
```
