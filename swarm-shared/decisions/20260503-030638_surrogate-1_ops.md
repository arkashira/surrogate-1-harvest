# surrogate-1 / ops

## Final Decision
Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16`, `DATE`, `HF_TOKEN`
- Pre-lists target folder once via HF API (single paginated call), saves manifest JSON, then performs **CDN-only fetches** during streaming to bypass `/api/` rate limits
- Projects heterogeneous repo files to `{prompt, response}` at parse time (avoids `pyarrow.CastError` from mixed schemas)
- Deduplicates via central md5 store (`lib/dedup.py`) and writes `batches/public-merged/{DATE}/shard{N}-{HHMMSS}.jsonl`
- Is idempotent, logs clearly, and exits with non-zero on fatal errors (so Actions can retry)

This directly applies lessons:
- HF API rate-limit bypass via CDN
- Avoid `load_dataset(streaming=True)` on mixed schemas
- Deterministic sharding + no cross-run state
- Mac=CLI orchestration only; heavy I/O in CI runners

   - Shebang `#!/usr/bin/env python3`
   - CLI: `--shard-id`, `--shard-total`, `--date`, `--hf-token`, `--repo`, `--out-dir`
   - Steps:
     1. `list_repo_tree(path=f"public-merged/{DATE}", recursive=False)` once; save manifest JSON locally
     2. Filter files by shard: `hash(slug) % SHARD_TOTAL == SHARD_ID`
     3. For each file:
        - Download via `requests.get(cdn_url, timeout=60, stream=True)`
        - Parse line-by-line or parquet via `pyarrow` (project to `{prompt, response}`)
        - Normalize fields, compute md5 of canonical JSON
        - Skip if md5 exists in central dedup store
        - Append accepted pairs to shard output file
     4. After processing, upload output file via `huggingface_hub` upload (single commit per shard)
     5. Print summary and exit code

   - Optional: add step to pre-compute manifest and cache as artifact (reduces per-run API calls)

def list_folder_files(api: HfApi, repo: str, folder: str) -> List[str]:
    """List files in a folder (non-recursive) with rate-limit awareness."""
    try:
        tree = api.list_repo_tree(repo=repo, path=folder, recursive=False)
        return [item.rfilename for item in tree if item.type == "file"]
    except Exception as e:
        if "429" in str(e):
            print(f"[WARN] Rate limited; waiting {API_RETRY_WAIT}s", file=sys.stderr)
            time.sleep(API_RETRY_WAIT)
            return list_folder_files(api, repo, folder)
        raise

def sha256_bytes(b: bytes) -> str:

def canonical_json_bytes(obj: Dict[str, Any]) -> bytes:
    return json.dumps(obj, sort_keys=True, separators=(",", ":")).encode("utf-8")

def project_to_pair(record: Dict[str, Any]) -> Optional[Dict[str, str]]:
    """
    Project heterogeneous schema to {prompt, response}.
    Heuristic:
      - prompt in {prompt, input, question, instruction}
      - response in {response, output, answer, completion}
    """
    prompt_keys = {"prompt", "input", "question", "instruction"}
    response_keys = {"response", "output", "answer", "completion"}

    prompt = None
    for k in prompt_keys:
        if k in record and isinstance(record[k], str) and record[k].strip():
            prompt = record[k].strip()
            break
    response = None
    for k in response_keys:
        if k in record and isinstance(record[k], str) and record[k].strip():
            response = record[k].strip()
            break

    if prompt is None or response is None:
        return None
    return {"prompt": prompt, "response": response}

def parse_parquet(path: Path) -> List[Dict[str, str]]:
    """Parse parquet and project to pairs."""
    try:
        table = pq.read_table(path, columns=None)
    except Exception as e:
        print(f"[WARN] Failed to read parquet {path}: {e}", file=sys.stderr)
        return []

    pairs = []
    for batch in table.to_batches():
        cols = {name: batch.column(name).to_pylist() for name in batch.schema.names}
        n = len(next(iter(cols.values()))) if cols else 0
        for i in range(n):
            rec = {k: cols[k][i] for k in cols}
            pair = project_to_pair(rec)
            if pair:
                pairs.append(pair)
    return pairs

def parse_jsonl(path: Path) -> List[Dict[str, str]]:
    pairs = []
    try:
        with path.open("r", encoding="utf-8") as f:
            for line in f:
                line = line.strip()
                if not line:
                    continue
                try:
                    rec = json.loads(line)
                except Exception:
                    continue
                pair = project_to_pair(rec)
                if pair:
                    pairs.append(pair)
    except Exception as e:
        print(f"[WARN] Failed to read JSONL {path}: {e}", file=sys.stderr)
    return pairs
