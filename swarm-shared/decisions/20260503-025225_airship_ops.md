## Code Snippets

### 1. CDN manifest generator (run from Mac after rate-limit window clears)

`scripts/build_cdn_manifest.py`
```python
#!/usr/bin/env python3
"""
Generate a CDN-only manifest for a date folder in a HuggingFace dataset repo.
Usage:
  HF_REPO="my-org/surrogate-data" python scripts/build_cdn_manifest.py \
    --date-folder 2026-05-03 --out manifest.json
"""
import argparse
import json
import os
import time
from pathlib import Path

from huggingface_hub import HfApi, hf_hub_download

HF_REPO = os.getenv("HF_REPO")
API = HfApi()

def list_date_files(date_folder: str):
    # Single non-recursive call per date folder (avoids 100x pagination)
    tree = API.list_repo_tree(repo_id=HF_REPO, path=date_folder, recursive=False)
    files = [f.rfilename for f in tree if f.type == "file"]
    return files

def build_manifest(date_folder: str, out_path: Path):
    files = list_date_files(date_folder)
    manifest = {
        "repo_id": HF_REPO,
        "date_folder": date_folder,
        "files": files,
        "generated_at": int(time.time()),
        "cdn_base": f"https://huggingface.co/datasets/{HF_REPO}/resolve/main",
    }
    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"Wrote {len(files)} files to {out_path}")

if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--date-folder", required=True)
    parser.add_argument("--out", default="manifest.json")
    args = parser.parse_args()

    if not HF_REPO:
        raise RuntimeError("HF_REPO env var required")

    build_manifest(args.date_folder, Path(args.out))
```

Make executable:
```bash
chmod +x scripts/build_cdn_manifest.py
```

### 2. Training loader: CDN-only, no API during data load

`surrogate/training/data.py`
```python
import json
import os
from pathlib import Path
from typing import Dict, Iterator

import pyarrow as pa
import pyarrow.parquet as pq
import requests
from huggingface_hub import hf_hub_download

MANIFEST_PATH = Path(os.getenv("CDN_MANIFEST", "manifest.json"))

def load_manifest() -> Dict:
    if not MANIFEST_PATH.exists():
        raise FileNotFoundError(f"CDN manifest missing: {MANIFEST_PATH}")
    return json.loads(MANIFEST_PATH.read_text())

def cdn_fetch(repo_id: str, filename: str, cache_dir: str = ".cache") -> Path:
    # Uses CDN URL under the hood; no Authorization header required for public files.
    return hf_hub_download(
        repo_id=repo_id,
        filename=filename,
        repo_type="dataset",
        cache_dir=cache_dir,
        local_files_only=False,  # allow download if not cached
    )

def stream_parquet_rows(manifest: Dict) -> Iterator[Dict]:
    repo_id = manifest["repo_id"]
    for fn in manifest["files"]:
        local_path = cdn_fetch(repo_id, fn)
        try:
            table = pq.read_table(local_path, columns=["prompt", "response"])
        except (pa.ArrowInvalid, KeyError):
            # Skip malformed or schema-mismatched files (surrogate-1 pattern)
            continue

        for batch in table.to_batches():
            df = batch.to_pandas()
            for _, row in df.iterrows():
                yield {"prompt": row["prompt"], "response": row["response"]}
```

### 3. Lightning Studio reuse + idle-resume guard

`surrogate/training/launch.py`
```python
import os
import time

from lightning import Lightning, Machine, Teamspace, Studio, Run

LIGHTNING_CLOUD = os.getenv("LIGHTNING_CLOUD", "lightning-public-prod")
STUDIO_NAME = os.getenv("STUDIO_NAME", "surrogate-train-l40s")
MACHINE = Machine.L40S

def get_or_create_studio() -> Studio:
    ts = Teamspace()
    running = [s for s in ts.studios if s.name == STUDIO_NAME and s.status == "Running"]
    if running:
        print(f"Reusing running studio: {running[0].id}")
        return running[0]

    existing = [s for s in ts.studios if s.name == STUDIO_NAME]
    if existing:
        studio = existing[0]
        if studio.status != "Running":
            print(f"Restarting stopped studio: {studio.id}")
            studio.start(machine=MACHINE, cloud=LIGHTNING_CLOUD)
        return studio

    print(f"Creating new studio: {STUDIO_NAME}")
    return Studio.create(
        name=STUDIO_NAME,
        machine=MACHINE,
        cloud=LIGHTNING_CLOUD,
        create_ok=True,
    )

def run_training():
    studio = get_or_create_studio()
    # Guard: studio may stop during long runs; check before each .run()
    if studio.status != "Running":
        studio.start(machine=MACHINE, cloud=LIGHTNING_CLOUD)

    run = Run(
        script="train.py",
        studio=studio,
        env={
            "CDN_MANIFEST": "manifest.json",
            "HF_REPO": os.getenv("HF_REPO", "my-org/surrogate-data"),
        },
    )
    run.execute()

if __name__ == "__main__":
    run_training()
```

### 4. Deterministic HF sibling repo selection (commit-cap spread)

`surrogate/training/upload.py`
```python
import hashlib

SIBLING_REPOS = [
    "my-org/surrogate-data",
    "my-org/surrogate-data-1",
    "my-org/surrogate-data-2",
    "my-org/surrogate-data-3",
    "my-org/surrogate-data-4",
]

def pick_repo(slug: str) -> str:
    # Deterministic, uniform spread across siblings
    digest = hashlib.sha256(slug.encode()).digest()
    idx = int.from_bytes(digest[:2], "big") % len(SIBLING_REPOS)
    return SIBLING_REPOS[idx]
```
