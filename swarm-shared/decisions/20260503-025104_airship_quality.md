## Code Snippets

### 1. CDN Manifest Generator (`scripts/gen_cdn_manifest.py`)
```python
#!/usr/bin/env python3
"""
Generate CDN manifest for Surrogate training.
Run from Mac (or any orchestration host) after HF rate-limit window clears.
"""
import json
from datetime import datetime
from pathlib import Path

from huggingface_hub import HfApi

API = HfApi()
REPO = "datasets/your-org/surrogate-mirror"
DATE = datetime.utcnow().strftime("%Y-%m-%d")  # or pass as arg

def main():
    folder = f"batches/mirror-merged/{DATE}"
    items = API.list_repo_tree(REPO, path=folder, recursive=False)
    
    manifest = []
    for item in items:
        if not item.path.endswith(".parquet"):
            continue
        slug = Path(item.path).stem  # e.g., "shard-abc123"
        cdn_url = f"https://huggingface.co/{REPO}/resolve/main/{item.path}"
        manifest.append({"slug": slug, "cdn_url": cdn_url})
    
    out_path = Path("training") / f"manifest-{DATE}.json"
    out_path.parent.mkdir(parents=True, exist_ok=True)
    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"Wrote {len(manifest)} entries to {out_path}")

if __name__ == "__main__":
    main()
```

### 2. Lightning-Resilient Launcher (`training/launch_surrogate_studio.py`)
```python
#!/usr/bin/env python3
import os
from lightning_sdk import Teamspace, Studio, Machine

TEAMSPACE = "surrogate-team"
STUDIO_NAME = "surrogate-train-l40s"
MACHINE = Machine.L40S

def get_or_create_studio():
    ts = Teamspace(name=TEAMSPACE)
    for s in ts.studios:
        if s.name == STUDIO_NAME:
            if s.status == "running":
                print(f"Reusing running studio: {STUDIO_NAME}")
                return s
            else:
                print(f"Restarting stopped studio: {STUDIO_NAME}")
                return s.start(machine=MACHINE)
    print(f"Creating new studio: {STUDIO_NAME}")
    return Studio.create(
        name=STUDIO_NAME,
        teamspace=TEAMSPACE,
        machine=MACHINE,
        create_ok=True,
    )

def run_training(script_path: str, args: list[str] | None = None):
    studio = get_or_create_studio()
    args = args or []
    # Studio.run executes inside the studio environment
    run = studio.run(
        command=f"python {script_path} {' '.join(args)}",
        environment="lightning",
    )
    print(f"Started run: {run.id}")
    return run

if __name__ == "__main__":
    import sys
    run_training("train_surrogate.py", sys.argv[1:])
```

### 3. CDN-Only Dataset Loader (`training/data.py`)
```python
import json
import pyarrow.parquet as pq
from pathlib import Path
from typing import Iterator, Dict
import requests
from io import BytesIO

def load_cdn_parquet(cdn_url: str, columns=("prompt", "response")) -> list[Dict]:
    """Download single parquet via CDN and project columns."""
    resp = requests.get(cdn_url, timeout=30)
    resp.raise_for_status()
    table = pq.read_table(BytesIO(resp.content), columns=columns)
    return table.to_pylist()

def iter_manifest_records(manifest_path: Path) -> Iterator[Dict]:
    manifest = json.loads(manifest_path.read_text())
    for entry in manifest:
        for record in load_cdn_parquet(entry["cdn_url"]):
            # Ensure schema consistency
            yield {
                "prompt": record.get("prompt", ""),
                "response": record.get("response", ""),
            }

# Example usage in train_surrogate.py
if __name__ == "__main__":
    manifest = Path("manifest-2026-05-03.json")
    for i, rec in enumerate(iter_manifest_records(manifest)):
        if i >= 10:
            break
        print(rec)
```
