# vanguard / design

## Final Synthesized Implementation

**Core Diagnosis (unified):**
- No persisted `(repo, dateFolder) → file-list` manifest: repeated authenticated `list_repo_tree` burns HF API quota and risks 429s.
- Training uses per-file/streaming loads on heterogeneous repos, causing `pyarrow.CastError` from mixed schemas.
- Authenticated API calls during training instead of CDN-only (`resolve/main/...`) URLs.
- Missing schema projection: raw files with attribution/metadata written to `enriched/` instead of projecting to `{prompt, response}` before training.
- No Lightning Studio reuse: jobs create new studios, wasting quota.

**Chosen Strategy (correct + actionable):**
- Single manifest build per `(repo, dateFolder)` with disk cache.
- CDN-only data loading via HTTP range requests + pyarrow (zero HF API calls during training).
- Strict schema projection to `{prompt, response}`; drop everything else.
- Reuse running Lightning Studio; fallback to local Trainer if unavailable.
- Batch-efficient row access (avoid per-row file reads) and robust error handling.

```python
#!/usr/bin/env python3
"""
Build and cache (repo, date_folder) -> file-list manifest.
Run once from orchestrator after rate-limit window clears.
"""
import json, os, hashlib
from pathlib import Path
from huggingface_hub import HfApi

CACHE_DIR = Path(__file__).parent.parent / ".cache" / "manifests"
CACHE_DIR.mkdir(parents=True, exist_ok=True)

def _cache_path(repo: str, date_folder: str) -> Path:
    slug = hashlib.sha256(f"{repo}::{date_folder}".encode()).hexdigest()[:12]
    return CACHE_DIR / f"{slug}.json"

def build_manifest(repo: str, date_folder: str, token: str = None) -> dict:
    """
    Returns:
      {
        "repo": "...",
        "date_folder": "...",
        "files": ["path/to/file.parquet", ...],
        "cdn_prefix": "https://huggingface.co/datasets/{repo}/resolve/main/{date_folder}/"
      }
    """
    cache = _cache_path(repo, date_folder)
    if cache.exists():
        return json.loads(cache.read_text())

    api = HfApi(token=token)
    # Single non-recursive call per date_folder (avoids heavy pagination)
    items = api.list_repo_tree(repo=repo, path=date_folder, recursive=False)
    files = sorted(
        f.rfilename for f in items if f.rfilename.endswith(".parquet")
    )

    manifest = {
        "repo": repo,
        "date_folder": date_folder,
        "files": files,
        "cdn_prefix": f"https://huggingface.co/datasets/{repo}/resolve/main/{date_folder}/"
    }
    cache.write_text(json.dumps(manifest, indent=2))
    return manifest

if __name__ == "__main__":
    import sys
    repo = sys.argv[1]
    date_folder = sys.argv[2]
    token = os.getenv("HF_TOKEN")
    m = build_manifest(repo, date_folder, token=token)
    print(json.dumps(m, indent=2))
```

```python
#!/usr/bin/env python3
"""
CDN-only training loader + Lightning Studio reuse.
Mac orchestrator: builds manifest once, then invokes this on Lightning.
"""
import json, os, sys
from pathlib import Path
from typing import Dict, List
import pyarrow.parquet as pq
import pyarrow as pa
import numpy as np
import torch
from torch.utils.data import Dataset, DataLoader, IterableDataset
import lightning as L

# -- Dataset (CDN-only, zero HF API calls during training) --
class CDNParquetIterable(IterableDataset):
    """
    Stream rows from CDN-hosted parquet files with schema projection.
    Uses HTTP range requests via pyarrow; no full downloads.
    Batches rows efficiently to avoid per-row overhead.
    """
    def __init__(self, manifest_path: str, columns=("prompt", "response"), batch_size=512):
        super().__init__()
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.prefix = self.manifest["cdn_prefix"]
        self.files = self.manifest["files"]
        self.columns = columns
        self.batch_size = batch_size

    def _stream_file(self, url: str):
        try:
            pf = pq.ParquetFile(url, memory_map=False, use_buffered_stream=True)
            for batch in pf.iter_batches(
                batch_size=self.batch_size,
                columns=self.columns,
                use_threads=False
            ):
                # Convert to dict of lists
                tbl = pa.Table.from_batches([batch])
                d = tbl.to_pydict()
                # Yield per-example dicts
                n = len(next(iter(d.values())))
                for i in range(n):
                    yield {k: d[k][i] for k in self.columns}
        except Exception as e:
            print(f"Error reading {url}: {e}", file=sys.stderr)

    def __iter__(self):
        worker_info = torch.utils.data.get_worker_info()
        if worker_info is None:
            files = self.files
        else:
            per_worker = len(self.files) // worker_info.num_workers
            files = self.files[worker_info.id * per_worker:(worker_info.id + 1) * per_worker]

        for fname in files:
            url = self.prefix + fname
            yield from self._stream_file(url)

# -- Lightweight collate --
def collate_fn(batch: List[Dict[str, str]]):
    # Simple dict-of-lists; tokenizer will handle padding later
    out = {}
    for k in batch[0].keys():
        out[k] = [b[k] for b in batch]
    return out

# -- Lightning task (stub) --
class SurrogateTrainer(L.LightningModule):
    def __init__(self):
        super().__init__()
        # placeholder model
        self.layer = torch.nn.Linear(1, 1)

    def training_step(self, batch, batch_idx):
        # Replace with real model forward/loss
        self.log("train/loss", 0.1 * batch_idx, prog_bar=True)
        return torch.tensor(0.1 * batch_idx, requires_grad=True)

    def configure_optimizers(self):
        return torch.optim.Adam(self.parameters(), lr=1e-4)

# -- Studio reuse + launch --
def launch_training(manifest_path: str, repo: str, studio_name: str = "vanguard-surrogate"):
    from lightning.pytorch import Trainer

    # Reuse running studio if available
    studio = None
    try:
        from lightning.pytorch import Studio, Teamspace
        studios = Teamspace.studios()
        running = [s for s in studios if s.name == studio_name and getattr(s, "status", None) == "Running"]
        if running:
            studio = running[0]
            print(f"Reusing running studio: {studio.name}")
        else:
            # Create with L40S (free tier fallback)
            studio = Studio.create_ok(
                name=studio_name,
                machine="L40S",
                cloud="lightning-public-prod"
            )
    except Exception as e:
        print(f"Studio reuse failed ({e}), falling back to local Trainer")

    dataset = CDNParquetIterable(manifest_path, columns=("prompt", "response"), batch_size=512)
    loader = DataLoader(
        dataset,
        batch_size=8,
        num_workers=0,
        collate_fn=collate_fn
    )

    model = SurrogateTrainer()
    trainer = Trainer(
        max_epochs=1,
        devices=1,
        accelerator="gpu" if torch.cuda.is_available() else "cpu",
        limit_train_batches=10
    )
    trainer.fit(model, loader)

if __name__ == "__main__":
    manifest = sys.argv[1] if len(sys.argv) > 1 else "manifest.json"
    launch_training(manifest, repo="datasets/your-repo")
```
