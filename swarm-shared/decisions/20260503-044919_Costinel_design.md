# Costinel / design

## Final Decision (Synthesized)

**Ship a frontend-only “Top Hub Signal Panel”** on Costinel’s dashboard within <2 hours.  
- **Primary mechanism**: CDN-first JSON (no Authorization header) to bypass Hugging Face API rate limits.  
- **Patterns applied**: 2026-04-27 “top-hub doc insight” + CDN bypass.  
- **Scope**: Zero backend changes, zero secrets, deployable via static hosting or existing Docker/nginx.  
- **UX contract**: “Sense + Signal — Execute” (visible governance signal, no automatic execution).

## Component (resolved for correctness + actionability)

File: `src/components/TopHubSignalPanel.tsx`

```tsx
'use client';

import { useEffect, useState, useCallback } from 'react';
import { ExternalLink, RefreshCw, AlertCircle } from 'lucide-react';

const CDN_URL =
  'https://huggingface.co/datasets/axentx/knowledge-rag/resolve/main/top-hub/latest.json';
const LOCAL_FALLBACK = '/static/top-hub/latest.json';
const CACHE_KEY = 'costinel:top-hub:cached';
const TIMEOUT_MS = 3000;
const MAX_RETRIES = 2;

type DocItem = {
  title: string;
  slug: string;
  snippet: string;
  url: string;
};

type TopHubPayload = {
  hub: string;
  updated_at: string;
  docs: DocItem[];
};

async function fetchWithTimeout(url: string, signal: AbortSignal, retries = MAX_RETRIES): Promise<Response> {
  let lastError: Error | undefined;
  for (let attempt = 0; attempt <= retries; attempt++) {
    if (attempt > 0) {
      // exponential backoff: 200ms, 400ms
      await new Promise((r) => setTimeout(r, 200 * Math.pow(2, attempt - 1)));
    }
    try {
      const controller = new AbortController();
      const timeout = setTimeout(() => controller.abort(), TIMEOUT_MS);
      const res = await fetch(url, {
        signal: AbortSignal.any([signal, controller.signal]),
        cache: 'no-store',
      });
      clearTimeout(timeout);
      if (res.ok) return res;
      throw new Error(`HTTP ${res.status}`);
    } catch (err: any) {
      lastError = err;
      if (err.name === 'AbortError') throw err;
    }
  }
  throw lastError || new Error('Fetch failed after retries');
}

export default function TopHubSignalPanel() {
  const [data, setData] = useState<TopHubPayload | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  const fetchData = useCallback(async (signal: AbortSignal) => {
    setLoading(true);
    setError(null);

    try {
      // 1) CDN-first (no auth)
      const res = await fetchWithTimeout(CDN_URL, signal);
      const json = (await res.json()) as TopHubPayload;
      setData(json);
      try {
        localStorage.setItem(CACHE_KEY, JSON.stringify(json));
      } catch (_) {}
      return;
    } catch (err: any) {
      if (err.name === 'AbortError') return;
      console.warn('CDN failed', err);
    }

    try {
      // 2) localStorage cache
      const cached = localStorage.getItem(CACHE_KEY);
      if (cached) {
        setData(JSON.parse(cached));
        return;
      }
    } catch (_) {}

    try {
      // 3) local static fallback
      const res = await fetchWithTimeout(LOCAL_FALLBACK, signal);
      const json = (await res.json()) as TopHubPayload;
      setData(json);
    } catch (err: any) {
      if (err.name === 'AbortError') return;
      setError('Unable to load Top Hub signal.');
    }
  }, []);

  useEffect(() => {
    const controller = new AbortController();
    fetchData(controller.signal);
    return () => controller.abort();
  }, [fetchData]);

  const handleRefresh = () => {
    const controller = new AbortController();
    fetchData(controller.signal);
  };

  // Emit event for loose coupling
  useEffect(() => {
    if (!data) return;
    const ev = new CustomEvent('signal:top-hub', { detail: data });
    window.dispatchEvent(ev);
  }, [data]);

  if (loading) {
    return (
      <div className="rounded-lg border bg-card p-4">
        <div className="flex items-center gap-2 text-sm text-muted-foreground">
          <RefreshCw className="h-3 w-3 animate-spin" />
          Loading Top Hub signal...
        </div>
      </div>
    );
  }

  return (
    <div className="rounded-lg border bg-card p-4">
      <div className="mb-3 flex items-start justify-between">
        <div className="flex items-center gap-2">
          <span className="inline-flex h-6 items-center rounded-full bg-primary/10 px-2 text-xs font-semibold text-primary">
            {data?.hub || '—'}
          </span>
          <span className="text-xs text-muted-foreground">Top Hub</span>
        </div>
        <button
          onClick={handleRefresh}
          disabled={loading}
          className="text-muted-foreground hover:text-foreground transition-colors"
          title="Refresh"
