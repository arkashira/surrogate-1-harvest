# Final synthesized plan (single, actionable)

## 1. Diagnosis (merged, prioritized)
- **No persisted manifest**: every training run performs authenticated `list_repo_tree`/`load_dataset` discovery → burns HF API quota and risks 429s.
- **Heterogeneous schema ingestion**: `load_dataset(streaming=True)` triggers per-file schema checks and `CastError` on mixed schemas.
- **Authenticated fetches during training**: amplifies rate-limit exposure; should use CDN-only URLs.
- **No studio reuse guard**: training may recreate Lightning Studio each run, wasting quota.
- **No idle-stop resilience**: Lightning idle timeout kills training; no automatic restart.

## 2. Single change: manifest + CDN-only training guard
Create `/opt/axentx/vanguard/ops/training_manifest.py` (one file, <150 LOC) that:
- `build_manifest(repo, date_folder, out_path)` → writes `{repo}-{date_folder}.json` with CDN URLs for every parquet file.
- `train_with_manifest(manifest_path, ...)` → Lightning job that uses **only CDN URLs** (zero HF API calls during training) and reuses running studios.
- `check_studio_and_start(...)` guard to reuse running studios and auto-restart on idle-stop.

## 3. Implementation (concrete, correct, minimal)

# ---- Manifest builder ----
def list_repo_tree(repo: str, path: str = "", recursive: bool = False) -> List[Dict]:
    """Unauthenticated tree call (non-recursive) to avoid pagination/429."""
    url = HF_API_TREE.format(repo=repo)
    params = {"recursive": int(recursive)}
    if path:
        params["path"] = path
    resp = requests.get(url, params=params, timeout=30)
    if resp.status_code == 429:
        retry_after = int(resp.headers.get("Retry-After", 60))
        print(f"HF 429, sleeping {retry_after}s")
        time.sleep(retry_after)
        return list_repo_tree(repo, path, recursive)
    resp.raise_for_status()
    return resp.json()

def build_manifest(repo: str, date_folder: str, out_dir: str) -> str:
    """Write manifest: {repo}-{date_folder}.json -> list of CDN URLs for parquet files."""
    tree = list_repo_tree(repo, path=date_folder, recursive=False)
    entries = []
    for node in tree:
        if node.get("type") != "file":
            continue
        name = node["path"]
        if not name.endswith(".parquet"):
            continue
        cdn_url = HF_CDN_TEMPLATE.format(repo=repo, path=f"{date_folder}/{name}")
        entries.append({"repo": repo, "date_folder": date_folder, "file": name, "cdn_url": cdn_url})

    os.makedirs(out_dir, exist_ok=True)
    safe_repo = repo.replace("/", "_")
    out_name = f"{safe_repo}_{date_folder}.json"
    out_path = os.path.join(out_dir, out_name)
    with open(out_path, "w") as f:
        json.dump(entries, f, indent=2)
    print(f"Manifest written: {out_path} ({len(entries)} files)")
    return out_path

class CDNParquetDataModule(LightningWork):
    def __init__(
        self,
        manifest_path: str,
        batch_size: int = 256,
        num_workers: int = 4,
        **kwargs,
    ):
        super().__init__(**kwargs)
        self.manifest_path = manifest_path
        self.batch_size = batch_size
        self.num_workers = num_workers
        self._dataloader = None

    def _build_dataloader(self):
        import pyarrow.parquet as pq
        import torch
        from torch.utils.data import IterableDataset, DataLoader

        with open(self.manifest_path) as f:
            files = json.load(f)

        class CDNParquetIterable(IterableDataset):
            def __init__(self, cdn_urls):
                self.cdn_urls = cdn_urls

            def __iter__(self):
                for item in self.cdn_urls:
                    url = item["cdn_url"]
                    try:
                        with pq.ParquetFile(url) as pf:
                            # Project only needed columns to avoid heterogeneous schema errors
                            cols_needed = {"prompt", "response"}
                            available = set(pf.schema.names)
                            if cols_needed.issubset(available):
                                table = pf.read(columns=["prompt", "response"])
                            else:
                                table = pf.read()
                                missing = cols_needed - available
                                if missing:
                                    continue
                                table = table.select(["prompt", "response"])
                            for batch in table.to_batches():
                                d = batch.to_pydict()
                                for prompt, response in zip(d.get("prompt", []), d.get("response", [])):
                                    yield {"prompt": prompt, "response": response}
                    except Exception as e:
                        print(f"Skipping {url}: {e}")
                        continue

        ds = CDNParquetIterable([x["cdn_url"] for x in files])
        return DataLoader(ds, batch_size=self.batch_size, num_workers=self.num_workers)

    def run(self):
        self._dataloader = self._build_dataloader()
        print(f"CDNParquetDataModule ready: dataloader built")
