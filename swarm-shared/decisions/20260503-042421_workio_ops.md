# workio / ops

## Final Synthesized Implementation

Below is the single, authoritative version that merges the strongest parts of both proposals, resolves contradictions, and prioritizes **correctness** and **concrete actionability**.

Key synthesis decisions:
- **Signature verification**: mandatory (fail closed if secret missing).
- **Idempotency + monotonic replay**: unified via `webhook_events` table with TTL and monotonic timestamp checks.
- **Transactional state transitions**: all DB changes occur inside explicit transactions with rollback on error.
- **Tenant enforcement**: required; events must map to a valid tenant via user or group.
- **Immutable audit trail**: append-only audit tables for attendances, leave requests, and OT.
- **Schema fixes**: complete missing columns and constraints; use `timestamptz`; avoid unbounded TTL scans where practical.

const IDEMPOTENCY_TTL_HOURS = 24;

function verifySignature(body: Buffer, signature: string): boolean {
  if (!CHANNEL_SECRET) return false;
  const expected = crypto
    .createHmac('sha256', CHANNEL_SECRET)
    .update(body)
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

async function isDuplicateEvent(client: PoolClient, tenantId: string, eventId: string): Promise<boolean> {
  const res = await client.query(
    `SELECT 1 FROM webhook_events
     WHERE tenant_id = $1 AND event_id = $2
       AND processed_at > NOW() - INTERVAL '${IDEMPOTENCY_TTL_HOURS} hours'`,
    [tenantId, eventId]
  );
  return res.rowCount > 0;
}

async function recordEvent(
  client: PoolClient,
  tenantId: string,
  eventId: string,
  source: string,
  payload: unknown
): Promise<void> {
  await client.query(
    `INSERT INTO webhook_events (event_id, source, tenant_id, payload, processed_at)
     VALUES ($1, $2, $3, $4, NOW())`,
    [eventId, source, tenantId, payload]
  );
}

function resolveTenantId(ev: any): string | null {
  // Prefer explicit tenantId; otherwise derive from user or group mapping
  if (ev.tenantId) return ev.tenantId;
  if (ev.source?.tenantId) return ev.source.tenantId;
  if (ev.userId) {
    // In production, map via a users table; here we require it to be provided
    return null;
  }
  return null;
}

async function handleAttendance(client: PoolClient, tenantId: string, ev: any): Promise<void> {
  const { userId, timestamp, type } = ev; // type: 'clock_in' | 'clock_out'
  if (!['clock_in', 'clock_out'].includes(type)) return;

  // Monotonic guard: prevent older replays from overriding newer state
  const latest = await client.query(
    `SELECT id, timestamp FROM attendances
     WHERE tenant_id = $1 AND user_id = $2 AND type = $3
     ORDER BY timestamp DESC LIMIT 1`,
    [tenantId, userId, type]
  );

  const ts = new Date(timestamp);
  if (latest.rowCount > 0 && new Date(latest.rows[0].timestamp) >= ts) {
    // Older or duplicate event — skip state change (idempotency record already written)
    return;
  }

  await client.query(
    `INSERT INTO attendances (id, tenant_id, user_id, type, timestamp)
     VALUES ($1, $2, $3, $4, $5)`,
    [uuidv4(), tenantId, userId, type, ts]
  );

  await client.query(
    `INSERT INTO attendance_audit (id, tenant_id, user_id, action, old_value, new_value, event_id)
     VALUES ($1, $2, $3, $4, $5, $6, $7)`,
    [uuidv4(), tenantId, userId, 'webhook_attendance', null, JSON.stringify({ type, timestamp: ts }), ev.eventId]
  );
}
