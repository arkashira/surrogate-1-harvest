# workio / quality

## Final Synthesized Implementation

### 1. Diagnosis (merged, highest severity first)
- **No signature verification** (`X-Line-Signature`): allows spoofed clock/leave/OT events.
- **No idempotency** for LINE retries: duplicates corrupt state (double punches, double approvals).
- **Raw body unavailable**: Express JSON parser mutates body before HMAC; verification always fails.
- **No structured audit trail**: missing provenance (event id, tenant, timestamp, idempotency key).
- **Race conditions on retries**: concurrent retries can interleave DB updates without atomic transitions/locking.

### 2. Proposed changes
- **New file**: `server/src/middleware/lineAuth.ts` — signature verification + raw-body capture helper.
- **Modify**: `server/src/routes/line/webhook.ts` — raw-body middleware, idempotency, atomic DB upserts, audit.
- **Modify**: `server/src/db/schema.sql` — add `line_webhook_events` table with unique idempotency key and `attendance_logs` optimistic lock (`version`).
- **Scope**: verify signatures, dedupe retries (short TTL), ensure atomic state transitions, full audit trail.

#### `server/src/middleware/lineAuth.ts`
```ts
import { Request, Response, NextFunction } from 'express';
import crypto from 'crypto';

const LINE_CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET || '';

export function lineSignature(rawBody: Buffer, signature: string): boolean {
  if (!LINE_CHANNEL_SECRET) return false;
  try {
    const expected = crypto
      .createHmac('sha256', LINE_CHANNEL_SECRET)
      .update(rawBody)
      .digest('base64');
    return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
  } catch {
    return false;
  }
}

export function lineAuth(req: Request, res: Response, next: NextFunction) {
  const signature = req.headers['x-line-signature'] as string;
  const rawBody = (req as any).rawBody as Buffer;
  if (!rawBody || !signature) {
    return res.status(401).json({ error: 'Missing signature or raw body' });
  }
  if (!lineSignature(rawBody, signature)) {
    return res.status(401).json({ error: 'Invalid signature' });
  }
  next();
}

/**
 * Raw-body capture middleware (use before JSON parser).
 * Attach rawBody to request, then parse JSON for downstream handlers.
 */
export function rawBodyCapture(req: Request, res: Response, next: NextFunction) {
  const chunks: Buffer[] = [];
  req.on('data', (chunk) => chunks.push(chunk));
  req.on('end', () => {
    const raw = Buffer.concat(chunks);
    (req as any).rawBody = raw;
    try {
      req.body = JSON.parse(raw.toString('utf8'));
    } catch {
      req.body = {};
    }
    next();
  });
  // Ensure stream errors don't hang
  req.on('error', next);
}
```

      console.error('Idempotency insert failed', { evId: ev.id, err });

    // Process event atomically
    try {
      await handleLineEvent(ev, tenantId, req);
      results.push({ eventId: ev.id, status: 'processed' });
    } catch (err) {
      console.error('LINE event handling failed', { evId: ev.id, err });
      results.push({ eventId: ev.id, status: 'error', error: String(err) });
    }
  }

  return res.json({ ok: true, results });
