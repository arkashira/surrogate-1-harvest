#### 1. Hardened orchestrator (`/opt/axentx/airship/airship/discover`)
```bash
#!/usr/bin/env bash
# airship/discover — deterministic HF dataset manifest builder + CDN-safe training launcher
# Usage: ./discover <dataset_repo> <date_folder> <output_dir>
# Example: ./discover datasets/airship-mirror 2024-01-15 ./manifests

set -euo pipefail
IFS=$'\n\t'

LOGFILE="/var/log/airship-discover.log"
exec 1> >(tee -a "$LOGFILE")
exec 2>&1

log() {
  echo "[$(date -u +"%Y-%m-%dT%H:%M:%SZ")] $*"
}

HF_REPO="${1:?Missing HF_REPO}"
DATE_FOLDER="${2:?Missing DATE_FOLDER}"
OUT_DIR="${3:-./manifests}"
MANIFEST_FILE="${OUT_DIR}/${HF_REPO//\//_}_${DATE_FOLDER}.json"

mkdir -p "$OUT_DIR"

# Retry wrapper for HF API (list tree) — avoids 429 hammering
hf_api_get() {
  local url="$1"
  local max_retries=5
  local backoff=2
  for i in $(seq 1 "$max_retries"); do
    if resp=$(curl -fsSL -H "Authorization: Bearer ${HF_TOKEN:-}" "$url" 2>/dev/null); then
      echo "$resp"
      return 0
    fi
    log "HF API request failed (attempt $i/$max_retries): $url"
    if [ "$i" -lt "$max_retries" ]; then
      sleep "$(( backoff ** i ))"
    fi
  done
  log "ERROR: HF API exhausted retries for $url"
  return 1
}

# List single-level tree (non-recursive) to avoid pagination explosion
log "Listing HF repo tree: $HF_REPO/$DATE_FOLDER (non-recursive)"
TREE_JSON=$(hf_api_get "https://huggingface.co/api/datasets/${HF_REPO}/tree/${DATE_FOLDER}?recursive=false")

# Extract file paths (relative to repo root) — only files, not dirs
FILE_PATHS=$(echo "$TREE_JSON" | python3 -c "
import sys, json
tree = json.load(sys.stdin)
for entry in tree:
    if entry.get('type') == 'file':
        print(entry['path'])
")

# Build CDN manifest
python3 - "$HF_REPO" "$DATE_FOLDER" "$FILE_PATHS" "$MANIFEST_FILE" <<'PY'
import json, os, sys

hf_repo = sys.argv[1]
date_folder = sys.argv[2]
paths = [p.strip() for p in sys.argv[3].splitlines() if p.strip()]
out = sys.argv[4]

items = []
for p in paths:
    full_path = f"{date_folder}/{p}" if not p.startswith(date_folder) else p
    cdn_url = f"https://huggingface.co/datasets/{hf_repo}/resolve/main/{full_path}"
    items.append({
        "path": full_path,
        "cdn_url": cdn_url
    })

os.makedirs(os.path.dirname(out), exist_ok=True)
with open(out, "w") as f:
    json.dump(items, f, indent=2)
print(f"Wrote {len(items)} files to {out}")
PY

log "Manifest written to $MANIFEST_FILE"

# Optional: trigger training or upload step here (CDN-only training uses manifest)
```

#### 2. CDN-only training loader with retry/backoff (`surrogate/train.py` excerpt)
```python
import json
import time
import requests
from pathlib import Path
from torch.utils.data import IterableDataset

class CDNParquetDataset(IterableDataset):
    def __init__(self, manifest_path="file-list.json", max_retries=5, timeout=30):
        manifest_path = Path(manifest_path)
        with manifest_path.open() as f:
            self.files = json.load(f)
        self.max_retries = max_retries
        self.timeout = timeout

    def _fetch_with_backoff(self, url):
        backoff = 1
        for attempt in range(1, self.max_retries + 1):
            try:
                resp = requests.get(url, timeout=self.timeout)
                resp.raise_for_status()
                return resp.content
            except requests.HTTPError as e:
                if resp.status_code in (429, 500, 502, 503, 504):
                    wait = backoff * (2 ** (attempt - 1))
                    print(f"Retry {attempt}/{self.max_retries} for {url} after {wait}s ({e})")
                    time.sleep(wait)
                    continue
                raise
            except requests.RequestException as e:
                if attempt == self.max_retries:
                    raise
                wait = backoff * (2 ** (attempt - 1))
                time.sleep(wait)
        raise RuntimeError(f"Failed to fetch {url} after {self.max_retries} attempts")

    def __iter__(self):
        for item in self.files:
```

#### 3. Deterministic HF repo sharding for writes (`scripts/shard_upload.py`)

```python
import hashlib

SIBLING_REPOS = [
    "datasets/airship-mirror",
    "datasets/airship-mirror-1",
    "datasets/airship-mirror-2",
    "datasets/airship-mirror-3",
    "datasets/airship-mirror-4",
]

def pick_repo(slug: str) -> str:
    """Deterministic repo assignment from slug."""
    digest = hashlib.sha256(slug.encode()).hexdigest()
    idx = int(digest, 16) % len(SIBLING_REPOS)
    return SIBLING_REPOS[idx]
```
