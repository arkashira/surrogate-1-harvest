### 1. Mac orchestration script (one-time, ~10 min)
Lists a single date folder via HF API (after rate-limit window) and writes `file_list.json` for Lightning training. Embeds this list so training uses CDN-only fetches (zero API calls during data load).

`scripts/list_hf_date_folder.py`
```python
#!/usr/bin/env python3
"""
Mac-only orchestration: list one date folder from HF dataset repo
and produce file_list.json for CDN-only training.

Usage:
  python scripts/list_hf_date_folder.py \
    --repo datasets/your-org/surrogate-mirror \
    --date 2026-05-03 \
    --out surrogate/file_list.json
"""

import argparse
import json
import os
import sys
import time
from typing import List

try:
    from huggingface_hub import list_repo_tree
except ImportError:
    print("Install: pip install huggingface_hub")
    sys.exit(1)

HF_TOKEN = os.getenv("HF_TOKEN", "")  # optional for public repos; required for private

def list_date_folder(repo: str, date: str) -> List[str]:
    """
    List files in repo under `batches/mirror-merged/{date}/` (non-recursive).
    Returns list of dicts: [{"path": "...", "size": ...}, ...]
    """
    folder = f"batches/mirror-merged/{date}/"
    print(f"Listing {repo}/{folder} ...")

    # Use non-recursive to avoid pagination explosion
    items = list_repo_tree(
        repo_id=repo,
        path=folder,
        recursive=False,
        token=HF_TOKEN or None,
    )

    files = [{"path": it["path"], "size": it.get("size", 0)} for it in items if it.get("type") == "file"]
    print(f"Found {len(files)} files.")
    return files

def save_file_list(files: List[dict], out_path: str) -> None:
    os.makedirs(os.path.dirname(os.path.abspath(out_path)), exist_ok=True)
    with open(out_path, "w") as f:
        json.dump(files, f, indent=2)
    print(f"Saved {len(files)} entries to {out_path}")

def main() -> None:
    parser = argparse.ArgumentParser(description="List HF date folder for CDN-only training.")
    parser.add_argument("--repo", required=True, help="HF dataset repo (e.g. datasets/your-org/surrogate-mirror)")
    parser.add_argument("--date", required=True, help="Date folder (YYYY-MM-DD)")
    parser.add_argument("--out", default="surrogate/file_list.json", help="Output JSON path")
    parser.add_argument("--retry-wait", type=int, default=360, help="Wait seconds on 429")
    args = parser.parse_args()

    for attempt in range(3):
        try:
            files = list_date_folder(args.repo, args.date)
            save_file_list(files, args.out)
            return
        except Exception as exc:
            # crude 429 detection; HF SDK may raise different exception types
            if "429" in str(exc) or "rate limit" in str(exc).lower():
                wait = args.retry_wait
                print(f"Rate limited (attempt {attempt+1}). Waiting {wait}s...")
                time.sleep(wait)
                continue
            print(f"Failed: {exc}")
            sys.exit(1)

    print("Exceeded retries.")
    sys.exit(1)

if __name__ == "__main__":
    main()
```

Make executable:
```bash

```

### 2. CDN-only parquet loader (Lightning training)
Uses the pre-computed `file_list.json` and downloads via public CDN URLs (`resolve/main/...`) with no Authorization header and no HF API calls during training.

`surrogate/data/cdn_parquet_loader.py`
```python
from __future__ import annotations

import json
import logging
from pathlib import Path
from typing import Iterator, Tuple

import pyarrow.parquet as pq
import requests
from torch.utils.data import IterableDataset

logger = logging.getLogger(__name__)

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

class CDNParquetIterableDataset(IterableDataset):
    """
    IterableDataset that yields {prompt, response} rows from parquet files
    using HF CDN URLs (no HF API/auth during training).

    file_list.json format:
    [
      {"path": "batches/mirror-merged/2026-05-03/abc.parquet", "size": 1234},
      ...
    ]
    """

    def __init__(
        self,
        repo: str,
        file_list_path: str | Path,
        columns: Tuple[str, str] = ("prompt", "response"),
        buffer_size: int = 8 * 1024 * 1024,
    ) -> None:
        self.repo = repo
        self.file_list_path = Path(file_list_path)
        self.columns = columns
        self.buffer_size = buffer_size

        if not self.file_list_path.is_file():
            raise FileNotFoundError(f"file_list.json not found: {self.file_list_path}")

        with open(self.file_list_path) as f:
            self.files = json.load(f)

        if not self.files:
            raise ValueError("file_list.json is empty")

        logger.info(f"Loaded {len(self.files)} files from {self.file_list_path}")

    def _stream_file(self, path: str) -> Iterator[dict]:
        url = CDN_TEMPLATE.format(repo=self.repo, path=path)
        logger.debug(f"Streaming {url}")

        # CDN downloads do NOT require Authorization header
        with requests.get(url, stream=True, timeout=60) as resp:
            resp.raise_for_status()
            # Stream into pyarrow without full materialization when possible
            with pq.ParquetFile(resp.raw) as pf:
                # Project only required columns to avoid mixed-schema issues
                batch_iter = pf.iter_batches(
                    batch_size=1024,
                    columns=list(self.columns),
                    use_threads=True,
                )
                for batch in batch_iter:
                    df = batch.to_pydict()
                    n = len(df[self.columns[0]])
                    for i in range(n):
                        yield {c: df[c][i] for c in self.columns}

    def __iter__(self) -> Iterator[dict]:
        worker_info = torch.utils.data.get_worker_info()
        if worker_info is None:
            files = self.files
        else:
            # shard across workers
            per_worker = len(self.files) // worker_info.num_workers
            files = self.files[worker_info.id * per_worker : (worker_info.id + 1) * per_worker]

        for entry in files:
            path = entry["path"]
            try:
                yield from self._stream_file(path)
            except Exception as exc:
                logger.warning(f"Failed to stream {path}: {exc}", exc_info=True)
                continue
```
