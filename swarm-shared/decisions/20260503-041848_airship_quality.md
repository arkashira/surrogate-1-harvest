### 1. Generate manifest (run once from Mac/Laptop)
```bash
# Install if needed
pip install -q huggingface_hub

# Generate manifest for a date folder (e.g., 2026-05-03)
python - <<'PY'
import json, os
from huggingface_hub import HfApi

REPO = "axentx/surrogate-dataset"
DATE_PATH = "batches/mirror-merged/2026-05-03"   # change as needed
OUT = "manifest.json"

api = HfApi()
files = api.list_repo_tree(repo_id=REPO, path=DATE_PATH, recursive=False)

manifest = {
    "repo": REPO,
    "date_path": DATE_PATH,
    "files": [
        {
            "path": f.rfilename,
            "cdn_url": f"https://huggingface.co/datasets/{REPO}/resolve/main/{DATE_PATH}/{f.rfilename}"
        }
        for f in files
    ]
}

with open(OUT, "w") as fp:
    json.dump(manifest, fp, indent=2)

print(f"Wrote {len(manifest['files'])} files to {OUT}")
PY
```

### 2. `train.py` — CDN-only loader + schema projection

# surrogate/train.py
import json, os, io, pyarrow.parquet as pq
import requests
import torch
from torch.utils.data import IterableDataset, DataLoader
from pathlib import Path

MANIFEST_PATH = Path(__file__).parent / "manifest.json"

class CDNParquetDataset(IterableDataset):
    def __init__(self, manifest_path=MANIFEST_PATH, columns=("prompt", "response")):
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.columns = columns

    def _stream_parquet(self, url):
        # CDN fetch: no Authorization header -> bypasses API rate limits
        resp = requests.get(url, timeout=30)
        resp.raise_for_status()
        return pq.read_table(io.BytesIO(resp.content)).to_pydict()

    def __iter__(self):
        for item in self.manifest["files"]:
            url = item["cdn_url"]
            try:
                batch = self._stream_parquet(url)
                for i in range(len(batch[self.columns[0]])):
                    yield {k: batch[k][i] for k in self.columns}
            except Exception as exc:
                # Log and skip bad files; don't kill training
                print(f"Skip {url}: {exc}")
                continue

def build_dataloader(manifest_path=MANIFEST_PATH, batch_size=8, num_workers=0):
    dataset = CDNParquetDataset(manifest_path)
    return DataLoader(dataset, batch_size=batch_size, num_workers=num_workers)

### 3. Lightning Studio resilient launcher
```python
# surrogate/launch_studio_train.py
import time
from lightning_sdk import Teamspace, Studio, Machine

TEAMSPACE = "your-teamspace"
STUDIO_NAME = "surrogate-train-l40s"
MANIFEST_PATH = "surrogate/manifest.json"

def ensure_running_studio():
    teamspace = Teamspace(TEAMSPACE)
    running = None
    for s in teamspace.studios:
        if s.name == STUDIO_NAME and s.status == "running":
            running = s
            break

    if running is None:
        print(f"Starting {STUDIO_NAME}...")
        # Free tier fallback -> L40S; paid tier can use Machine.H200 in lightning-lambda-prod
        studio = teamspace.studios.create(
            STUDIO_NAME,
            machine=Machine.L40S,
            script="surrogate/train.py",
            # optional: enable_idle_timeout=False if available
        )
        # Wait until running
        while studio.status != "running":
            time.sleep(10)
            studio = teamspace.studios.get(STUDIO_NAME)
        return studio
    return running

def run_training():
    studio = ensure_running_studio()
    # Studio.run executes script inside the environment
    run = studio.run(
        "surrogate/train.py",
        arguments=[f"--manifest={MANIFEST_PATH}"],
        wait=False,
    )
    print(f"Started run {run.id}")
    return run

if __name__ == "__main__":
    run_training()
```
