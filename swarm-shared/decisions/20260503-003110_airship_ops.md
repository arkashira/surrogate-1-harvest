### 1) `tools/build_file_manifest.py`
```python
#!/usr/bin/env python3
"""
Generate file_manifest.json for a single date folder in a HF dataset repo.
Run from Mac orchestrator after rate-limit window clears.
"""
import json, os, sys, argparse
from huggingface_hub import HfApi

HF_REPO = os.getenv("HF_DATASET_REPO", "axentx/surrogate-ingest")
DATE_FOLDER = os.getenv("INGEST_DATE_FOLDER", "2026-05-03")  # e.g. batches/mirror-merged/2026-05-03
OUT_PATH = os.getenv("MANIFEST_OUT", "surrogate/data/file_manifest.json")

def build():
    api = HfApi()
    entries = api.list_repo_tree(repo_id=HF_REPO, path=DATE_FOLDER, recursive=False)
    files = []
    for e in entries:
        if e.type != "file":
            continue
        # CDN URL (no auth)
        cdn_url = f"https://huggingface.co/datasets/{HF_REPO}/resolve/main/{e.path}"
        files.append({
            "path": e.path,
            "cdn_url": cdn_url,
            "size": e.size or 0,
        })
    manifest = {
        "repo": HF_REPO,
        "date_folder": DATE_FOLDER,
        "generated_by": "tools/build_file_manifest.py",
        "files": files
    }
    os.makedirs(os.path.dirname(os.path.abspath(OUT_PATH)), exist_ok=True)
    with open(OUT_PATH, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Wrote {len(files)} files -> {OUT_PATH}")

if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", default=HF_REPO)
    parser.add_argument("--folder", default=DATE_FOLDER)
    parser.add_argument("--out", default=OUT_PATH)
    args = parser.parse_args()
    HF_REPO = args.repo
    DATE_FOLDER = args.folder
    OUT_PATH = args.out
    build()
```

### 2) `surrogate/data/cdn_dataset.py`
```python
import json, io, gzip, time, requests
from typing import Dict, Iterator, Optional
import torch
from torch.utils.data import IterableDataset

class CdnDataset(IterableDataset):
    """
    CDN-only dataset loader.
    Reads file_manifest.json and streams each file via requests (no HF API auth).
    Projects each line to {prompt, response} at parse time.
    """
    def __init__(self, manifest_path: str, max_retries: int = 3, backoff: float = 5.0):
        super().__init__()
        self.manifest_path = manifest_path
        self.max_retries = max_retries
        self.backoff = backoff
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.files = self.manifest["files"]

    def _stream_file(self, entry: Dict) -> Iterator[str]:
        url = entry["cdn_url"]
        expected_size = entry.get("size", 0)
        for attempt in range(self.max_retries):
            try:
                with requests.get(url, stream=True, timeout=60) as r:
                    r.raise_for_status()
                    # optional size check
                    if expected_size and int(r.headers.get("content-length", 0)) != expected_size:
                        raise ValueError("Size mismatch")
                    # handle .gz transparently
                    if url.endswith(".gz"):
                        buf = io.BytesIO()
                        for chunk in r.iter_content(chunk_size=8192):
                            buf.write(chunk)
                        buf.seek(0)
                        with gzip.GzipFile(fileobj=buf, mode="rb") as gz:
                            for line in io.TextIOWrapper(gz, encoding="utf-8"):
                                yield line
                    else:
                        for line in r.iter_lines(decode_unicode=True):
                            if line:
                                yield line
                    return
            except Exception as exc:
                if attempt == self.max_retries - 1:
                    raise
                time.sleep(self.backoff * (2 ** attempt))

    def _parse_line(self, line: str) -> Optional[Dict]:
        line = line.strip()
        if not line:
            return None
        try:
            obj = json.loads(line)
        except json.JSONDecodeError:
            return None
        # project to {prompt, response} only
        prompt = obj.get("prompt") or obj.get("input") or obj.get("question")
        response = obj.get("response") or obj.get("output") or obj.get("answer")
        if prompt is None or response is None:
            return None
        return {"prompt": str(prompt), "response": str(response)}

    def __iter__(self) -> Iterator[Dict]:
        for entry in self.files:
            for line in self._stream_file(entry):
                parsed = self._parse_line(line)
                if parsed:
                    yield parsed


class CdnCollator:
    def __call__(self, batch):
        prompts = [b["prompt"] for b in batch]
        responses = [b["response"] for b in batch]
        # tokenizer will be applied in training loop
        return {"prompts": prompts, "responses": responses}
```

### 4) `scripts/generate_manifest.sh`
```bash
#!/bin/bash
# Generate file_manifest.json for a given date folder and copy into surrogate workspace.
# Usage: ./scripts/generate_manifest.sh [date_folder] [output_path]

set -euo pipefail
SHELL=/bin/bash

HF_DATASET_REPO="${HF_DATASET_REPO:-axentx/surrogate-ingest}"
INGEST_DATE_FOLDER="${1:-2026-05-03}"
OUT_PATH="${2:-surrogate/data/file_manifest.json}"

cd "$(dirname "$0")/.."

python tools/build_file_manifest.py \
  --repo "$HF_DATASET_REPO" \
  --folder "$INGEST_DATE_FOLDER" \
  --out "$OUT_PATH"

echo "Manifest generated at $OUT_PATH"
```
