## 1) One-time CDN manifest generation (Mac / CI)

- **Why**: Avoid `/api/` auth and rate limits during training; make runs reproducible.
- **How**: Use `huggingface_hub` to list **non-recursive** per date folder (avoids pagination explosion). Save only file paths + CDN URLs. Commit to repo.
- **File**: `scripts/generate_manifest.py`

```python
#!/usr/bin/env python3
"""
Generate CDN-only file manifest for a HuggingFace dataset folder.
Usage: python generate_manifest.py <repo> <folder> <out_json>
"""
import json
import sys
from pathlib import Path
from huggingface_hub import HfApi

def main():
    if len(sys.argv) != 4:
        print("Usage: generate_manifest.py <repo> <folder> <out_json>")
        sys.exit(1)

    repo, folder, out_path = sys.argv[1], sys.argv[2], Path(sys.argv[3])
    api = HfApi()

    # Non-recursive to avoid pagination explosion
    files = api.list_repo_tree(repo=repo, path=folder, recursive=False)
    entries = []
    for f in files:
        if f.get("type") == "file":
            cdn_url = f"https://huggingface.co/datasets/{repo}/resolve/main/{f['path']}"
            entries.append({"repo": repo, "path": f["path"], "cdn_url": cdn_url})

    out_path.parent.mkdir(parents=True, exist_ok=True)
    with open(out_path, "w") as fp:
        json.dump(entries, fp, indent=2)

    print(f"Wrote {len(entries)} files to {out_path}")

if __name__ == "__main__":
    main()
```

**Run once**:
```bash
python scripts/generate_manifest.py my-org/surrogate-data 2024-06-01 manifests/train_files_20240601.json
# commit manifests/
```

## 2) `train.py` — manifest-driven, CDN-only, streaming

- **Why**: Zero API calls during data load; memory-safe streaming; schema-tolerant parsing.
- **Changes**:
  - Load manifest at startup.
  - Fetch via CDN URLs (no auth header).
  - Use `IterableDataset` to stream.
  - Parse to `{prompt, response}` at parse time; skip bad files.

```python
import json
import os
from pathlib import Path
from typing import Dict, Iterator

import requests
from torch.utils.data import IterableDataset

MANIFEST_PATH = Path(__file__).parent.parent / "manifests" / "train_files_latest.json"

class CDNTextDataset(IterableDataset):
    def __init__(self, manifest_path: Path = MANIFEST_PATH, max_retries: int = 3):
        with open(manifest_path) as f:
            self.files = json.load(f)
        self.max_retries = max_retries

    def _fetch_cdn(self, url: str) -> str:
        for attempt in range(self.max_retries):
            try:
                resp = requests.get(url, timeout=30)
                resp.raise_for_status()
                return resp.text
            except Exception:
                if attempt == self.max_retries - 1:
                    raise
        raise RuntimeError("unreachable")

    def _parse_to_pair(self, text: str, path: str) -> Dict[str, str]:
        # Project to {prompt, response} only; ignore mixed schemas.
        # Customize per repo if needed. Minimal fallback:
        return {"prompt": text.strip(), "response": ""}

    def __iter__(self) -> Iterator[Dict[str, str]]:
        for item in self.files:
            try:
                text = self._fetch_cdn(item["cdn_url"])
                pair = self._parse_to_pair(text, item["path"])
                if pair["prompt"]:
                    yield pair
            except Exception as exc:
                # Log and skip bad files to avoid crashing training
                print(f"Skipping {item['path']}: {exc}")
                continue
```

## 3) Idle-resilient Lightning launcher

- **Why**: Lightning Studio kills training on idle stop. Auto-reuse or restart to keep runs alive.
- **Behavior**:
  - Before run, check for existing running studio with target name; reuse if found.
  - On idle stop, restart automatically.
  - Prefer `Machine.L40S` (free tier); fallback list for resilience.
  - Avoid H200 unless explicitly targeting `lightning-lambda-prod`.

```python
from pathlib import Path
from lightning import Lightning, Teamspace, Machine

def launch_training_with_reuse(
    entry_point: str = "train.py",
    studio_name: str = "surrogate-train-l40s",
    machine: Machine = Machine.L40S,
    fallback_machines: list[Machine] | None = None,
):
    """
    Launch or resume a Lightning Studio training run with idle-resilience.
    """
    lightning = Lightning()
    teamspace = Teamspace()

    if fallback_machines is None:
        fallback_machines = [Machine.A10G, Machine.A100]  # priority order

    # Reuse running studio if exists
    studio = None
    for s in teamspace.studios:
        if s.name == studio_name and s.status == "Running":
            studio = s
            print(f"Reusing running studio: {studio_name}")
            break

    if studio is None:
        # Start new studio
        print(f"Starting new studio: {studio_name} on {machine}")
        studio = lightning.run(
            target=Path(entry_point),
            name=studio_name,
            machine=machine,
            cloud_checks=False,
        )

    # Basic idle-resilience wrapper: if stopped, restart on next machine in priority
    if studio.status == "Stopped":
        print(f"Studio {studio_name} stopped (likely idle). Restarting...")
        # Rotate through fallback machines on restart
        next_machine = fallback_machines[0] if fallback_machines else machine
        studio = lightning.run(
            target=Path(entry_point),
            name=studio_name,
            machine=next_machine,
            cloud_checks=False,
        )

    return studio

if __name__ == "__main__":
    launch_training_with_reuse()
```
