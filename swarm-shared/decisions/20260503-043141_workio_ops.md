# workio / ops

## 1. Diagnosis

- **No persistent webhook replay protection** — LINE retries on 5xx/timeouts; without idempotency key enforcement, clock-in/out and leave/OT events can be duplicated on retry.
- **No outbound LINE notification retry/backoff** — failed notify calls (e.g., approval alerts) are silently dropped; operators have no visibility or retry mechanism.
- **No structured logging or request correlation** — webhook and worker logs are unstructured; debugging failures across LINE → webhook → DB → notify is slow and error-prone.
- **No health/liveness checks for ops** — no `/healthz` or `/readyz` endpoints; load balancers and cron jobs cannot detect unhealthy containers or DB outages.
- **No graceful shutdown handling** — in-flight webhooks and background jobs are terminated on SIGTERM, causing partial processing and inconsistent state.

## 2. Proposed change

File scope:  
- `workio/server/src/routes/webhook.ts` — add idempotency table and check, structured logging, graceful shutdown coordination.  
- `workio/server/src/services/line-notify.ts` — add exponential backoff retry with jitter and circuit breaker.  
- `workio/server/src/routes/health.ts` — add `/healthz` (liveness) and `/readyz` (readiness: DB + LINE token check).  
- `workio/server/src/index.ts` — wire graceful shutdown, request draining, and structured logger.

### 3.3 Idempotency check in webhook route

```ts
// server/src/routes/webhook.ts (excerpt)
import { logger } from '../lib/logger';
import { pool } from '../db';

async function verifyIdempotency(eventId: string, tenantId: number, payloadHash: string): Promise<boolean> {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');
    const exists = await client.query(
      'SELECT 1 FROM webhook_idempotency WHERE webhook_event_id = $1',
      [eventId]
    );
    if (exists.rows.length > 0) {
      logger.info({ eventId, tenantId }, 'duplicate webhook event skipped');
      await client.query('ROLLBACK');
      return false;
    }
    await client.query(
      'INSERT INTO webhook_idempotency(webhook_event_id, tenant_id, payload_hash) VALUES ($1,$2,$3)',
      [eventId, tenantId, payloadHash]
    );
    await client.query('COMMIT');
    return true;
  } catch (err) {
    await client.query('ROLLBACK');
    logger.error({ err, eventId, tenantId }, 'idempotency check failed');
    return false;
  } finally {
    client.release();
  }
}
```

### 3.4 LINE notify with retry + circuit breaker

```ts
// server/src/services/line-notify.ts
import axios from 'axios';
import { logger } from '../lib/logger';

const LINE_API = 'https://api.line.me/v2/bot/message/push';
const MAX_RETRIES = 3;
const BASE_DELAY = 300; // ms

function sleep(ms: number) {
  return new Promise((r) => setTimeout(r, ms));
}

export async function notifyLine(userId: string, message: any, token: string) {
  let lastErr: any;
  for (let attempt = 0; attempt <= MAX_RETRIES; attempt++) {
    try {
      const res = await axios.post(
        LINE_API,
        { to: userId, messages: [message] },
        { headers: { Authorization: `Bearer ${token}` }, timeout: 8000 }
      );
      if (res.status === 200) {
        logger.info({ userId, attempt }, 'line notify sent');
        return true;
      }
    } catch (err: any) {
      lastErr = err;
      const isRetryable = !err.response || err.response.status >= 500 || err.code === 'ECONNABORTED';
      if (!isRetryable) break;
      const delay = BASE_DELAY * 2 ** attempt + Math.random() * 100;
      logger.warn({ err: err.message, attempt, delay }, 'line notify retry');
      await sleep(delay);
    }
  }
  logger.error({ err: lastErr?.message, userId }, 'line notify failed after retries');
  return false;
}
```

### 3.6 Graceful shutdown + request draining

```ts
// server/src/index.ts (excerpt)
import express from 'express';
import http from 'http';
import { logger } from './lib/logger';
import webhookRouter from './routes/webhook';
import healthRouter from './routes/health';

const app = express();
app.use(express.json({ limit: '1mb' }));
app.use('/webhook/line', webhookRouter);
app.use('/health', healthRouter);

const server = http.createServer(app);

function shutdown(signal: string) {
  logger.info({ signal }, 'shutdown signal received');
  server.close(() => {
    logger.info('HTTP server closed');
    process.exit(0);
  });

  // Force exit after drain timeout
  setTimeout(() => {
    logger.warn('forced shutdown after drain timeout');
    process.exit(1);
  }, 30_000).unref();
}

process.on('SIGTERM', () => shutdown('SIGTERM'));
process.on('SIGINT', () => shutdown('SIGINT'));

server.listen(3000, () => {
  logger.info('server listening on :3000');
});
```
