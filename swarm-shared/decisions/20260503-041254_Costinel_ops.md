## Implementation

### 1) Build-time bake script (`scripts/bake-top-hub.js`)
- CDN-first; no runtime HF API.  
- Uses a **single precomputed tree file** from CDN when available (`knowledge-rag/hubs/tree.json`).  
- Falls back to local `knowledge-rag/hubs/*.md` if offline or CDN unavailable.  
- Outputs minified JSON to `public/data/top-hub-latest.json`.

```js
#!/usr/bin/env node
/**
 * Bake top-hub signal to public/data/top-hub-latest.json
 * CDN-first, zero-runtime-HF-API.
 */
const fs = require('fs');
const path = require('path');
const https = require('https');

const OUTPUT = path.join(__dirname, '../public/data/top-hub-latest.json');
const HUBS_DIR = path.join(__dirname, '../knowledge-rag/hubs');
const CDN_BASE = 'https://huggingface.co/datasets/axentx/costinel-knowledge/resolve/main';
const TREE_FILE = 'knowledge-rag/hubs/tree.json';

function fetchCDN(url) {
  return new Promise((resolve, reject) => {
    https.get(url, (res) => {
      if (res.statusCode === 200) {
        let data = '';
        res.on('data', (chunk) => (data += chunk));
        res.on('end', () => resolve(data));
      } else {
        reject(new Error(`CDN ${res.statusCode}`));
      }
    }).on('error', reject);
  });
}

async function tryCDNTree() {
  try {
    const raw = await fetchCDN(`${CDN_BASE}/${TREE_FILE}`);
    return JSON.parse(raw);
  } catch {
    return null;
  }
}

function readLocalHubs() {
  if (!fs.existsSync(HUBS_DIR)) return [];
  return fs.readdirSync(HUBS_DIR)
    .filter((f) => f.endsWith('.md'))
    .map((f) => {
      const content = fs.readFileSync(path.join(HUBS_DIR, f), 'utf8');
      const name = f.replace(/\.md$/, '');
      const connMatch = content.match(/connections:\s*(\d+)/i);
      const scoreMatch = content.match(/score:\s*([0-9.]+)/i);
      return {
        hub: name,
        connections: connMatch ? Number(connMatch[1]) : 0,
        score: scoreMatch ? Number(scoreMatch[1]) : 0,
        sourceFile: `knowledge-rag/hubs/${f}`,
      };
    });
}

async function bake() {
  let hubs = [];

  // 1) Try precomputed tree from CDN (fastest, no HF API at runtime)
  const tree = await tryCDNTree();
  if (tree && Array.isArray(tree)) {
    hubs = tree.map((item) => ({
      hub: item.hub || path.basename(item.path || '', '.md'),
      connections: item.connections || 0,
      score: item.score || 0,
      sourceFile: item.path || `knowledge-rag/hubs/${item.hub || 'unknown'}.md`,
    }));
  }

  // 2) Fallback to local files
  if (!hubs.length) {
    hubs = readLocalHubs();
  }

  // 3) Pick top hub: primary connections, secondary score
  const top = hubs.sort((a, b) => {
    if (b.connections !== a.connections) return b.connections - a.connections;
    return b.score - a.score;
  })[0] || { hub: 'N/A', connections: 0, score: 0, sourceFile: '' };

  const out = {
    hub: top.hub,
    score: top.score,
    connections: top.connections,
    lastUpdated: new Date().toISOString(),
    sourceFile: top.sourceFile,
    insight:
      top.hub !== 'N/A'
        ? `${top.hub} is the most-connected hub; prioritize cross-team alignment signals.`
        : 'No hub data available.',
  };

  fs.mkdirSync(path.dirname(OUTPUT), { recursive: true });
  fs.writeFileSync(OUTPUT, JSON.stringify(out), 'utf8');
  console.log('Baked top-hub to', OUTPUT, out);
}

bake().catch((err) => {
  console.error('Bake failed:', err);
  process.exit(1);
});
```

### 2) CDN utility (`src/lib/cdn.ts`)
- Exposes typed fetch with CDN-first + local fallback.  
- No HF API calls at runtime.

```ts
// src/lib/cdn.ts
const CDN_BASE = 'https://huggingface.co/datasets/axentx/costinel-knowledge/resolve/main';
const TOP_HUB_PATH = 'public/data/top-hub-latest.json';

export interface TopHubData {
  hub: string;
  score: number;
  connections: number;
  lastUpdated: string;
  sourceFile: string;
  insight: string;
}

export async function fetchTopHubCDN(signal?: AbortSignal): Promise<TopHubData | null> {
  try {
    const res = await fetch(`${CDN_BASE}/${TOP_HUB_PATH}`, { signal, cache: 'no-store' });
    if (!res.ok) throw new Error(`CDN ${res.status}`);
    return (await res.json()) as TopHubData;
  } catch {
    return null;
  }
}

export async function fetchTopHubWithFallback(signal?: AbortSignal): Promise<TopHubData | null> {
  // 1) Runtime CDN fetch (zero HF API)
  const cdn = await fetchTopHubCDN(signal);
  if (cdn) return cdn;

  // 2) Local baked file fallback (bundled or served from public/)
  try {
    const localRes = await fetch(`/data/top-hub-latest.json`, { signal });
    if (!localRes.ok) throw new Error('Local fetch failed');
    return (await localRes.json()) as TopHubData;
  } catch {
    return null;
  }
}
```
