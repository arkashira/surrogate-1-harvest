#### 2) Service — `app/services/knowledge_rag.py`
```python
from typing import Any, Dict, List, Optional
from datetime import datetime


def get_top_hub(for_date: Optional[str] = None, limit_docs: int = 5, limit_edges: int = 10) -> Dict[str, Any]:
    """
    Resolve the strongest hub by pagerank (preferred) or degree centrality.
    Optionally filter nodes by date property == for_date.
    Returns serializable dict for signal payload.
    """
    graph = _get_graph()

    # Candidate nodes
    candidates = [n for n in graph.nodes()]
    if for_date:
        candidates = [n for n in candidates if str(graph.nodes[n].get("date")) == str(for_date)]

    if not candidates:
        # fallback to all nodes when date filter yields empty
        candidates = list(graph.nodes())

    if not candidates:
        raise ValueError("No graph nodes available to compute top hub.")

    # Centrality: prefer pagerank; fallback to degree
    try:
        ranks = graph.pagerank(nodes=candidates)
    except Exception:
        ranks = {n: graph.degree(n) for n in candidates}

    top_hub = max(ranks, key=ranks.get)
    top_score = float(ranks[top_hub])

    # Neighbors sorted by edge weight
    neighbors = graph.neighbors(top_hub)
    scored: List[tuple] = []
    for nb in neighbors:
        w = float(graph.edge_weight(top_hub, nb) or 1.0)
        meta = graph.nodes[nb]
        scored.append((w, nb, meta))
    scored.sort(key=lambda x: x[0], reverse=True)

    top_edges = [
        {"target_id": str(nb), "target_label": str(meta.get("label", meta.get("title", ""))), "weight": float(w)}
        for w, nb, meta in scored[:limit_edges]
    ]

    top_docs = [
        {
            "doc_id": str(nb),
            "title": str(meta.get("title", str(nb))),
            "score": float(w),
            "snippet": str(meta.get("snippet", "")),
        }
        for w, nb, meta in scored[:limit_docs]
    ]

    hub_meta = graph.nodes[top_hub]
    context_summary = (
        f"Top hub '{top_hub}' (score {top_score:.3f}) connects to {len(neighbors)} related items. "
        f"Key docs and edges suggest primary focus areas for cost-anomaly context."
    )

    return {
        "hub_id": str(top_hub),
        "hub_label": str(hub_meta.get("label", hub_meta.get("title", ""))),
        "hub_type": str(hub_meta.get("type", "")),
        "score": top_score,
        "top_edges": top_edges,
        "top_docs": top_docs,
        "context_summary": context_summary,
    }


def _get_graph() -> Any:
    """
    Replace with your actual graph accessor.
    Examples:
      - return app.state.knowledge_graph
      - from app.graph import graph; return graph
    Must expose:
      - graph.nodes() -> Iterable[node_id]
      - graph.nodes[node_id] -> dict metadata
      - graph.neighbors(node_id) -> Iterable[node_id]
      - graph.edge_weight(u, v) -> Optional[float]
      - graph.pagerank(nodes=...) -> Dict[node_id, float]  (optional)
      - graph.degree(node_id) -> int
    """
    raise NotImplementedError(
        "Wire _get_graph() to your existing knowledge-graph accessor in knowledge_rag.py"
    )
```

#### 3) API route — `app/api/v1/endpoints/cost_anomaly.py`
```python
from fastapi import APIRouter, HTTPException, Query
from datetime import date, datetime, timezone
from typing import Optional

from app.schemas.cost_anomaly import TopHubSignal
from app.services.knowledge_rag import get_top_hub

router = APIRouter()


@router.get("/cost-anomaly/signal/top-hub", response_model=TopHubSignal)
def top_hub_signal(
    for_date: Optional[date] = Query(
        None, description="Optional date filter (YYYY-MM-DD). If omitted, uses all graph nodes."
    ),
):
    """
    Sense + Signal: return strongest knowledge-graph hub as a cost-anomaly signal.
    Read-only. No state changes.
    """
    try:
        for_date_str = for_date.isoformat() if for_date else None
        result = get_top_hub(for_date=for_date_str, limit_docs=5, limit_edges=10)

        payload = TopHubSignal(
            for_date=for_date,
            generated_at=datetime.now(timezone.utc).isoformat().replace("+00:00", "Z"),
            hub_id=result["hub_id"],
            hub_label=result["hub_label"],
            hub_type=result["hub_type"],
            score=result["score"],
            top_edges=result["top_edges"],
            top_docs=result["top_docs"],
            context_summary=result["context_summary"],
        )
        return payload
    except NotImplementedError as exc:
        raise HTTPException(status_code=501, detail=str(exc))
    except ValueError as exc:
        raise HTTPException(status_code=404, detail=str(exc))
    except Exception as exc:
        raise HTTPException(status_code=500, detail=f"Failed to produce top-hub signal: {exc}")
```
