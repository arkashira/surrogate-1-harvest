## Final Decision  
Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16`, `DATE`, `HF_TOKEN`, `REPO`
- Pre-lists the target folder once via the HF API (respecting rate limits) and saves a per-shard manifest JSON
- Assigns files to shards deterministically by hash(slug) % SHARD_TOTAL
- Downloads only assigned files via HF CDN (`resolve/main/...`) with **no Authorization header** (bypasses auth rate limits)
- Projects heterogeneous schemas to `{prompt, response}` at parse time (avoids pyarrow CastError / mixed-schema failures)
- Deduplicates via central md5 store (`lib/dedup.py`)
- Streams output to `batches/public-merged/{DATE}/shard{N}-{HHMMSS}.jsonl` (no extra columns; attribution via filename)
- Emits structured logs and proper exit codes for GitHub Actions

This directly applies lessons: HF rate-limit bypass, schema heterogeneity fix, deterministic shard assignment, and removes fragile bash parallelism/streaming bugs.

MAX_RETRIES = 5
RETRY_BACKOFF = 5  # seconds (exponential)

def deterministic_shard(slug: str, shard_total: int) -> int:
    digest = hashlib.md5(slug.encode("utf-8")).hexdigest()
    return int(digest, 16) % shard_total

def list_folder(repo: str, folder: str) -> List[Dict[str, Any]]:
    url = f"{HF_API_BASE}/datasets/{repo}/tree"
    params = {"path": folder, "recursive": "false"}
    for attempt in range(1, MAX_RETRIES + 1):
        try:
            resp = httpx.get(url, params=params, headers=HEADERS_API, timeout=30.0)
            if resp.status_code == 429:
                wait = 360
                print(f"[WARN] HF API 429 rate-limited. Waiting {wait}s.", file=sys.stderr)
                time.sleep(wait)
                continue
            resp.raise_for_status()
            return resp.json()
        except httpx.HTTPStatusError as e:
            if e.response.status_code >= 500:
                wait = RETRY_BACKOFF * (2 ** (attempt - 1))
                print(f"[WARN] Server error {e.response.status_code}, retry {attempt}/{MAX_RETRIES} in {wait}s", file=sys.stderr)
                time.sleep(wait)
                continue
            raise
        except Exception as e:
            print(f"[ERROR] Failed to list folder: {e}", file=sys.stderr)
            raise
    raise RuntimeError("Exhausted retries listing folder.")

def download_cdn(repo: str, path: str, out_path: Path) -> Path:
    url = HF_CDN_TEMPLATE.format(repo=repo, path=path)
    for attempt in range(1, MAX_RETRIES + 1):
        try:
            with httpx.stream("GET", url, headers=HEADERS_CDN, timeout=30.0, follow_redirects=True) as resp:
                resp.raise_for_status()
                out_path.parent.mkdir(parents=True, exist_ok=True)
                with open(out_path, "wb") as f:
                    for chunk in resp.iter_bytes(chunk_size=8192):
                        f.write(chunk)
                return out_path
        except httpx.HTTPStatusError as e:
            if e.response.status_code >= 500:
                wait = RETRY_BACKOFF * (2 ** (attempt - 1))
                print(f"[WARN] CDN server error {e.response.status_code}, retry {attempt}/{MAX_RETRIES} in {wait}s", file=sys.stderr)
                time.sleep(wait)
                continue
            raise
        except Exception as e:
            print(f"[ERROR] CDN download failed for {url}: {e}", file=sys.stderr)
            raise
    raise RuntimeError(f"Exhausted retries downloading {url}")
