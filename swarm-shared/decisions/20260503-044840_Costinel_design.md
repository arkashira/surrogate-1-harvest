# Costinel / design

## Final Synthesis — Production-Ready Top-Hub Signal Panel (CDN-first)

**Goal**: Ship in <2h a resilient, zero-compute “Top Hub” panel that surfaces the most-connected hub (e.g., MOC) and related docs using CDN-only fetches, graceful fallbacks, and an audit trail — aligned to *Sense + Signal (ไม่ Execute)*.

---

### 1) CDN Artifact (single source of truth)

- **Path (versioned)**  
  `https://huggingface.co/datasets/axentx/costinel-signals/resolve/main/top-hub/v1/top-hub.json`

- **Schema (minimal, <30KB)**
  ```json
  {
    "hub": {
      "title": "MOC (Method of Costing)",
      "slug": "moc",
      "connections": 128,
      "summary": "Central hub for cost allocation methods and tagging strategies across multi-cloud environments.",
      "url": "https://axentx.dev/docs/moc"
    },
    "related": [
      { "title": "Tagging Governance", "slug": "tagging", "connections": 94 },
      { "title": "RI Coverage Analysis", "slug": "ri-coverage", "connections": 72 },
      { "title": "Anomaly Detection Rules", "slug": "anomaly-rules", "connections": 61 }
    ],
    "updatedAt": "2025-01-01T12:00:00.000Z",
    "ttl": 3600
  }
  ```

- **Build/CI step**  
  Add `scripts/generate-top-hub-json.py` that outputs the above schema and uploads to the CDN path during CI.

### 2) Backend: lightweight proxy + fallback

`/opt/axentx/Costinel/src/server/routes/topHub.ts`

```ts
import express from 'express';
import fs from 'fs/promises';
import path from 'path';

const router = express.Router();
const CDN_BASE = process.env.CDN_BASE_URL || 'https://huggingface.co/datasets/axentx/costinel-signals/resolve/main/top-hub/v1';
const CDN_JSON = 'top-hub.json';
const LOCAL_FALLBACK = path.join(__dirname, '../../static/top-hub.json');

router.get('/panel', async (req, res) => {
  // 1) Try CDN (fast, zero-auth)
  try {
    const controller = new AbortController();
    const timeout = setTimeout(() => controller.abort(), 4000);
    const resp = await fetch(`${CDN_BASE}/${CDN_JSON}`, { signal: controller.signal });
    clearTimeout(timeout);

    if (resp.ok) {
      const data = await resp.json();
      return res.json({ source: 'cdn', data });
    }
  } catch {
    // noop → fallback
  }

  // 2) Fallback to local bundled copy
  try {
    const raw = await fs.readFile(LOCAL_FALLBACK, 'utf8');
    return res.json({ source: 'local', data: JSON.parse(raw) });
  } catch {
    return res.status(503).json({ error: 'Top hub unavailable' });
  }
});

export default router;
```

Wire into main server:

```ts
import topHubRouter from './routes/topHub';
app.use('/api/top-hub', topHubRouter);
```

### 3) Frontend: resilient TopHubPanel

`/opt/axentx/Costinel/src/components/TopHubPanel.tsx`

```tsx
import React, { useEffect, useState } from 'react';
import './TopHubPanel.css';

interface HubDoc {
  title: string;
  slug: string;
  connections: number;
  summary?: string;
  url?: string;
}

interface PanelData {
  hub: HubDoc;
  related: HubDoc[];
  updatedAt: string;
  ttl: number;
}

export const TopHubPanel: React.FC = () => {
  const [data, setData] = useState<PanelData | null>(null);
  const [loading, setLoading] = useState(true);
  const [source, setSource] = useState<'cdn' | 'local' | 'none'>('none');
  const [stale, setStale] = useState(false);

  useEffect(() => {
    const fetchData = () =>
      fetch('/api/top-hub/panel')
        .then((r) => r.json())
        .then((json) => {
          setData(json.data);
          setSource(json.source);
          const age = Date.now() - new Date(json.data.updatedAt).getTime();
          setStale(age > (json.data.ttl * 1000));
        })
        .catch(() => setSource('none'))
        .finally(() => setLoading(false));

    fetchData();
  }, []);

  // Audit: log impression
  useEffect(() => {
    if (data) {
      navigator.sendBeacon?.('/api/audit/signal', JSON.stringify({
        signal: 'top-hub-impression',
        hub: data.hub.slug,
        source,
        ts: new Date().toISOString()
      }));
    }
  }, [data, source]);

  const dismiss = () => {
    const until = Date.now() + 24 * 3600_000;
    localStorage.setItem('topHubDismissedUntil', String(until));
    setSource('none');
  };

  if (loading) return <div className="top-hub-panel loading">Loading insights…</div>;
  if (!data || source === 'none') return null;

  return (
    <div className="top-hub-panel">
      <div className="top-hub-header">
        <h3>Top Hub</h3>
        <span className={`source-badge ${source}`}>{source}</span>
      </div>

      <div className="hub-card primary">
        <h4>{data.hub.title}</h4>
        <p className="summary">{data.hub.summary}</p>
        <div className="meta">
          <span>{data.hub.connections} connections</span>
          {data.hub.url && (
            <a href={data.hub.url} target="_blank" rel="noopener noreferrer">Open doc</a>
          )}
        </div>
        {stale && <div className="stale-badge">Stale — refresh recommended</div>}
      </div>

      <div className="related-list">
        <h5>Related</h5>
        {data.related.map((doc) => (
          <div key={doc.slug} className="related-item">
            <strong>{doc.title}</strong>
            <span>{doc.connections} connections</span>
          </div>
        ))}
      </div>

      <div className="panel-footer">
        <span>Updated {new Date(data.updatedAt).toLocaleDateString()}</span>
        <button className="dismiss-btn" onClick={dismiss}>Dismiss 24h</button>
      </div>
    </div>
  );
};
```
