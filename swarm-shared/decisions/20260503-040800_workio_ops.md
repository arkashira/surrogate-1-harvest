### 1.1 Graceful shutdown and structured JSON logging with correlation IDs

Add to `server/src/app.ts` (or main server entry):

```ts
import express, { Request, Response, NextFunction } from 'express';
import morgan from 'morgan';
import { createLogger, format, transports } from 'winston';
import { v4 as uuidv4 } from 'uuid';

const { combine, timestamp, json, errors, printf } = format;

// Correlation ID middleware
const correlationMiddleware = (req: Request, res: Response, next: NextFunction) => {
  const cid = req.headers['x-correlation-id'] || uuidv4();
  (req as any).correlationId = cid;
  res.setHeader('X-Correlation-ID', cid);
  next();
};

// Structured logger
const logger = createLogger({
  level: process.env.LOG_LEVEL || 'info',
  format: combine(timestamp(), errors({ stack: true }), json()),
  transports: [new transports.Console()],
  exceptionHandlers: [new transports.Console({ format: combine(timestamp(), json()) })],
});

// HTTP log format with correlation ID
const httpFormat = printf(({ level, message, timestamp, ...meta }) => {
  const cid = (meta.req as any)?.correlationId || (meta as any).correlationId || '-';
  return JSON.stringify({
    timestamp,
    level,
    correlationId: cid,
    message,
    ...meta,
  });
});

const app = express();
app.use(express.json());
app.use(correlationMiddleware);

// Optional: HTTP request logging (JSON)
app.use(
  morgan('combined', {
    stream: {
      write: (msg) => logger.info(msg.trim()),
    },
    skip: (req) => req.url === '/health', // reduce noise
  })
);

// Health endpoint (lightweight)
app.get('/health', async (req, res) => {
  try {
    // Add real DB/liveness checks as needed
    // await db.query('SELECT 1');
    res.json({ status: 'ok', timestamp: new Date().toISOString() });
  } catch (err) {
    logger.error({ msg: 'health check failed', error: err });
    res.status(503).json({ status: 'error', error: String(err) });
  }
});

// Graceful shutdown handler
function gracefulShutdown(server: any, signal: string) {
  logger.info({ msg: `Received ${signal}. Starting graceful shutdown.` });
  server.close(() => {
    logger.info({ msg: 'HTTP server closed.' });
    // Close DB/pool connections here if needed
    process.exit(0);
  });

  // Force exit after timeout
  setTimeout(() => {
    logger.error({ msg: 'Forced shutdown after timeout.' });
    process.exit(1);
  }, 30_000).unref();
}

// Start server (example)
const server = app.listen(process.env.PORT || 3000, () => {
  logger.info({ msg: 'Server started', port: process.env.PORT || 3000 });
});

process.once('SIGTERM', () => gracefulShutdown(server, 'SIGTERM'));
process.once('SIGINT', () => gracefulShutdown(server, 'SIGINT'));

export { app, server, logger };
```

## 2. Lightweight ops script (health + recent LINE webhook checks)

Create `scripts/check-service.sh` (make executable: `chmod +x`):

```bash
#!/usr/bin/env bash
# Lightweight ops check: service health + recent LINE webhook deliveries
# Usage: ./scripts/check-service.sh [--json]

set -euo pipefail

SERVICE_URL=${SERVICE_URL:-http://localhost:3000}
LOG_DIR=${LOG_DIR:-./logs}
LINEWEBHOOK_LOG="${LOG_DIR}/linewebhook.log"
JSON_OUTPUT=false

if [[ "${1:-}" == "--json" ]]; then
  JSON_OUTPUT=true
fi

check_health() {
  if curl -fsSL "${SERVICE_URL}/health" > /dev/null 2>&1; then
    echo "OK"
    return 0
  else
    echo "FAIL"
    return 1
  fi
}

recent_webhooks() {
  if [[ -f "${LINEWEBHOOK_LOG}" ]]; then
    # Show last 10 entries; adapt path/parser to your logging format
    tail -n 10 "${LINEWEBHOOK_LOG}" 2>/dev/null || echo "[]"
  else
    echo "No webhook log found at ${LINEWEBHOOK_LOG}"
  fi
}

if $JSON_OUTPUT; then
  HEALTH_STATUS=$(check_health || true)
  WEBHOOKS=$(recent_webhooks || true)
  cat <<EOF
{
  "timestamp": "$(date -u +"%Y-%m-%dT%H:%M:%SZ")",
  "service_url": "${SERVICE_URL}",
  "health": "${HEALTH_STATUS}",
  "recent_line_webhooks": $(echo "${WEBHOOKS}" | jq -R -s 'split("\n") | map(select(. != ""))' 2>/dev/null || echo "[]")
}
EOF
else
  echo "=== Service Check ==="
  echo "URL: ${SERVICE_URL}"
  echo "Health: $(check_health)"
  echo ""
  echo "=== Recent LINE Webhooks (last 10) ==="
  recent_webhooks
fi
```

Notes:
- If you log LINE webhooks to a structured file (JSON lines), adjust `recent_webhooks` to use `jq` for filtering.
- For production, prefer querying your DB or log aggregation instead of flat files.
