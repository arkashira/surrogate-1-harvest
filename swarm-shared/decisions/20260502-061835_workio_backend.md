```bash
// server/src/routes/line/setup.js
const express = require('express');
const router = express.Router();
const lineService = require('../services/line');
const { validateChannelAccessToken, validateChannelSecret } = require('../utils/validation');

router.post('/setup', async (req, res) => {
  const { channelAccessToken, channelSecret } = req.body;
  try {
    validateChannelAccessToken(channelAccessToken);
    validateChannelSecret(channelSecret);
    await lineService.setupLineOA(channelAccessToken, channelSecret);
    res.json({ message: 'LINE OA setup successful' });
  } catch (error) {
    res.status(500).json({ message: 'Error setting up LINE OA' });
  }
});

module.exports = router;
```

Create a new file `line-oa-setup.js` in the `client` directory:

```javascript
import axios from 'axios';

const lineOASetup = async () => {
  const channelAccessToken = process.env.LINE_CHANNEL_ACCESS_TOKEN;
  const channelSecret = process.env.LINE_CHANNEL_SECRET;

  try {
    const response = await axios.post('https://your-domain.com/line/setup', {
      channelAccessToken,
      channelSecret,
    });

    console.log(`LINE OA setup successful: ${response.data.message}`);
  } catch (error) {
    console.error(`Error setting up LINE OA: ${error.message}`);
  }
};

export default lineOASetup;
```
