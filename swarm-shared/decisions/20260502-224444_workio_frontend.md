# workio / frontend

## 1. Diagnosis

- **No resilient offline-first clock-in/out**: current optimistic UI relies on localStorage queue (per past decisions) which is fragile, synchronous, and can lose data on storage pressure or cross-tab conflicts.
- **Missing durable operation queue**: IndexedDB was recommended in synthesis but not implemented; without it, offline/low-connectivity clock events are lost.
- **No GPS pre-check before optimistic commit**: UX allows optimistic clock-in even when GPS is invalid, causing later rollbacks and confusing state.
- **Missing local draft + sync-status indicator**: users can’t see whether a clock event is pending, synced, or failed, reducing trust.
- **No idempotency key on client-generated events**: retries on flaky networks can create duplicate clock records.

## 2. Proposed change

File scope:
- `src/hooks/useClockEvents.ts` (create or update) — implement durable queue with IndexedDB, idempotency keys, GPS pre-check, and sync status.
- `src/components/ClockButton.tsx` — consume hook, show pending/synced/failed states, disable when GPS invalid.
- `src/lib/idbQueue.ts` — small reusable IndexedDB queue (add, peek, remove, retry).

## 3. Implementation

### src/lib/idbQueue.ts
```ts
// Minimal IndexedDB queue for clock events
const DB_NAME = 'workio_queue';
const STORE_NAME = 'clock_events';
const DB_VERSION = 1;

function openDB(): Promise<IDBDatabase> {
  return new Promise((resolve, reject) => {
    const req = indexedDB.open(DB_NAME, DB_VERSION);
    req.onupgradeneeded = () => {
      const db = req.result;
      if (!db.objectStoreNames.contains(STORE_NAME)) {
        db.createObjectStore(STORE_NAME, { keyPath: 'id' });
      }
    };
    req.onsuccess = () => resolve(req.result);
    req.onerror = () => reject(req.error);
  });
}

export async function enqueueClockEvent(event: {
  id: string;
  type: 'clock_in' | 'clock_out';
  timestamp: number;
  latitude: number;
  longitude: number;
  attempt?: number;
}) {
  const db = await openDB();
  const tx = db.transaction(STORE_NAME, 'readwrite');
  tx.store.put({ ...event, attempt: event.attempt || 0, createdAt: Date.now() });
  return new Promise<void>((resolve, reject) => {
    tx.oncomplete = () => resolve();
    tx.onerror = () => reject(tx.error);
  });
}

export async function peekQueue(): Promise<any[]> {
  const db = await openDB();
  const tx = db.transaction(STORE_NAME, 'readonly');
  const req = tx.store.getAll();
  return new Promise((resolve, reject) => {
    req.onsuccess = () => resolve(req.result.sort((a, b) => a.createdAt - b.createdAt));
    req.onerror = () => reject(req.error);
  });
}

export async function removeFromQueue(id: string) {
  const db = await openDB();
  const tx = db.transaction(STORE_NAME, 'readwrite');
  tx.store.delete(id);
  return new Promise<void>((resolve, reject) => {
    tx.oncomplete = () => resolve();
    tx.onerror = () => reject(tx.error);
  });
}
```

### src/hooks/useClockEvents.ts
```ts
import { useEffect, useState, useCallback } from 'react';
import { enqueueClockEvent, peekQueue, removeFromQueue } from '../lib/idbQueue';

type ClockStatus = 'idle' | 'pending' | 'synced' | 'failed';

export function useClockEvents() {
  const [status, setStatus] = useState<ClockStatus>('idle');
  const [lastEvent, setLastEvent] = useState<{ id: string; type: 'clock_in' | 'clock_out'; timestamp: number } | null>(null);
  const [isSyncing, setIsSyncing] = useState(false);

  // Generate idempotency key
  const generateId = () => `clock_${Date.now()}_${Math.random().toString(36).slice(2, 9)}`;

  // GPS pre-check
  const getValidPosition = (): Promise<{ latitude: number; longitude: number }> => {
    return new Promise((resolve, reject) => {
      if (!navigator.geolocation) {
        reject(new Error('Geolocation not supported'));
        return;
      }
      navigator.geolocation.getCurrentPosition(
        (pos) => resolve({ latitude: pos.coords.latitude, longitude: pos.coords.longitude }),
        (err) => reject(err),
        { enableHighAccuracy: true, timeout: 10000, maximumAge: 30000 }
      );
    });
  };

  // Attempt to send to backend
  const sendToServer = async (event: any): Promise<boolean> => {
    try {
      const res = await fetch('/api/clock', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          idempotencyKey: event.id,
          type: event.type,
          timestamp: event.timestamp,
          latitude: event.latitude,
          longitude: event.longitude,
        }),
      });
      if (!res.ok) throw new Error(`HTTP ${res.status}`);
      return true;
    } catch (err) {
      return false;
    }
  };

  // Process queue
  const processQueue = useCallback(async () => {
    if (isSyncing) return;
    setIsSyncing(true);
    try {
      const items = await peekQueue();
      for (const item of items) {
        const ok = await sendToServer(item);
        if (ok) {
          await removeFromQueue(item.id);
        } else {
          // stop on first failure (preserve order)
          break;
        }
      }
    } finally {
      setIsSyncing(false);
    }
  }, [isSyncing]);

  // Clock action
  const clock = async (type: 'clock_in' | 'clock_out') => {
    setStatus('pending');
    const id = generateId();
    const timestamp = Date.now();

    try {
      const pos = await getValidPosition();
      // Optimistic local state
      const event = { id, type, timestamp, ...pos };
      setLastEvent({ id, type, timestamp });
      setStatus('pending');

      // Persist to durable queue
      await enqueueClockEvent(event);
      // Try immediate send
      const ok = await sendToServer(event);
      if (ok) {
        await removeFromQueue(id);
        setStatus('synced');
      } else {
        setStatus('failed');
        // will be retried by queue processor
      }
    } catch (err) {
      // GPS or other failure — do not commit optimistic event
      setStatus('failed');
      console.error('Clock failed', err);
    }
  };

  // Background retry
  useEffect(() => {
    processQueue();
    const interval = setInterval(processQueue, 15000);
    // Also retry on visibility change
    const onVis = () => document.visibilityState === 'visible' && processQueue();
    document.addEventListener('visibilitychange', onVis);
    return () => {
      clearInterval(interval);
      document.removeEventListener('visibilitychange', onVis);
    };
  }, [processQueue]);

  return { clock, status, lastEvent, isSyncing };
}
```
