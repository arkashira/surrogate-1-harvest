`scripts/list_parquet_files.py`
```python
#!/usr/bin/env python3
"""
Run from Mac after rate-limit window clears.
Lists parquet files in a date folder and emits file_list.json for Lightning training.
Uses single API call per folder (non-recursive) to avoid pagination/rate limits.
"""
import json
import os
from datetime import date
from huggingface_hub import HfApi

REPO_ID = "axentx/surrogate-datasets"
DATE_FOLDER = str(date.today())  # e.g. "2026-05-03"
OUTPUT = "file_list.json"

def main() -> None:
    api = HfApi()
    # Non-recursive: one call, returns up to 100 files in this folder only
    files = api.list_repo_tree(REPO_ID, path=DATE_FOLDER, recursive=False)
    parquet_files = [
        f"{DATE_FOLDER}/{f.rstrip('.parquet')}"
        for f in files
        if f.endswith(".parquet")
    ]
    payload = {
        "repo_id": REPO_ID,
        "date_folder": DATE_FOLDER,
        "files": parquet_files,
        "total": len(parquet_files),
        "note": "CDN-only; use https://huggingface.co/datasets/{repo_id}/resolve/main/{date}/{file}.parquet"
    }
    with open(OUTPUT, "w") as f:
        json.dump(payload, f, indent=2)
    print(f"Wrote {len(parquet_files)} files to {OUTPUT}")

if __name__ == "__main__":
    main()
```

#!/usr/bin/env python3
"""
Lightning Studio training script.
Uses CDN-only parquet fetches (zero HF API calls during training).
Embeds file list generated by list_parquet_files.py.
"""
import json
import pyarrow.parquet as pq
import requests
from pathlib import Path
from typing import List, Dict
import torch
from torch.utils.data import Dataset, DataLoader

# Embed the file list generated on Mac (committed or injected at runtime)
with open("file_list.json") as f:
    FILE_LIST = json.load(f)

REPO_ID = FILE_LIST["repo_id"]
DATE_FOLDER = FILE_LIST["date_folder"]
FILES = FILE_LIST["files"]

CDN_ROOT = f"https://huggingface.co/datasets/{REPO_ID}/resolve/main/{DATE_FOLDER}"

class CDNParquetDataset(Dataset):
    def __init__(self, files: List[str], cache_dir: str = ".cache"):
        self.files = files
        self.cache_dir = Path(cache_dir)
        self.cache_dir.mkdir(exist_ok=True)
        self.records: List[Dict[str, str]] = []
        self._load_all()

    def _download(self, file_slug: str) -> Path:
        # file_slug is like "2026-05-03/some-batch"
        fname = file_slug.split("/")[-1] + ".parquet"
        out = self.cache_dir / fname
        if out.exists():
            return out
        url = f"{CDN_ROOT}/{fname}"
        r = requests.get(url, timeout=60)
        r.raise_for_status()
        out.write_bytes(r.content)
        return out

    def _load_all(self) -> None:
        for slug in self.files:
            path = self._download(slug)
            table = pq.read_table(path, columns=["prompt", "response"])
            df = table.to_pandas()
            for _, row in df.iterrows():
                self.records.append({
                    "prompt": str(row["prompt"]),
                    "response": str(row["response"]),
                })

    def __len__(self) -> int:
        return len(self.records)

    def __getitem__(self, idx: int) -> Dict[str, str]:
        return self.records[idx]

# Minimal collate + model placeholder (replace with Surrogate training logic)
def train_one_epoch(loader: DataLoader) -> None:
    for batch in loader:
        # Replace with actual surrogate training step
        _ = batch["prompt"], batch["response"]

def main() -> None:
    dataset = CDNParquetDataset(FILES)
    loader = DataLoader(dataset, batch_size=8, shuffle=True, num_workers=0)
    print(f"Loaded {len(dataset)} records from {len(FILES)} parquet files via CDN")

### 3. Lightning idle-resilient runner

`surrogate/run_lightning_studio.py`
```python
#!/usr/bin/env python3
"""
Idempotent Lightning Studio launcher with idle-resilience.
Reuses running studios; restarts if stopped by idle timeout before each run.
"""
import time
from lightning_sdk import Studio, Teamspace, Machine

TEAMSPACE = "default"
STUDIO_NAME = "surrogate-trainer"
MACHINE = Machine.L40S  # Free tier falls to L40S; H200 requires lightning-lambda-prod

def get_or_create_studio() -> Studio:
    teamspace = Teamspace(TEAMSPACE)
    # Reuse if already running (saves quota)
    for s in teamspace.studios:
        if s.name == STUDIO_NAME:
            if s.status == "running":
                print(f"Reusing running studio: {STUDIO_NAME}")
                return s
            else:
                print(f"Studio {STUDIO_NAME} is {s.status}. Restarting...")
                s.delete()
                break
    print(f"Creating studio {STUDIO_NAME}...")
    return Studio(
        name=STUDIO_NAME,
        teamspace=TEAMSPACE,
        machine=MACHINE,
        create_ok=True,
    )

def ensure_running(studio: Studio) -> None:
    """Lightning idle stop kills training; restart if not running."""
    if studio.status != "running":
        print(f"Studio not running ({studio.status}). Starting...")
        studio.start(machine=MACHINE)
        # Wait until running
        while studio.status != "running":
            time.sleep(10)
            studio.refresh()
    print("Studio is running.")

def run_training() -> None:
    studio = get_or_create_studio()
    ensure_running(studio)
    # Run the CDN-based training script
    run = studio.run(
        command="python surrogate/train_cdn.py",
        cwd="/workspace/airship/surrogate",
    )
    print(f"Started run {run.id}")
    # Optionally stream logs
    for line in run.stream_logs():
        print(line)

if __name__ == "__main__":
    run_training()
```
