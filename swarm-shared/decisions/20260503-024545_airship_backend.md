# airship / backend

## Final Integrated Plan — CDN-first Surrogate training slice (≤2h)

**What we will ship**
- A backend endpoint that lists one HF date-folder once, persists `file-list.json`, and returns a short-lived token/path.
- A training launcher that reuses a running L40S Studio (or starts one) and passes the file list.
- A training script that fetches only via HuggingFace CDN (no HF API calls during data loading) and projects every file to `{prompt, response}` before batching.

**Why this works**
- Removes HF API 429s during training data loading (CDN-only fetches).
- Prevents Lightning Studio idle-timeout waste by reusing a running L40S.
- Keeps scope minimal and testable in ≤2 hours.
- Uses proven patterns: CDN bypass, strict schema projection, studio reuse, shared-volume triggers.

### 1) HF file-list endpoint (FastAPI)

```python
# surrogate/api/training.py
import json
from pathlib import Path
from fastapi import APIRouter, HTTPException
from huggingface_hub import HfApi

router = APIRouter()
HF_API = HfApi()
LIST_DIR = Path("/opt/axentx/airship/surrogate/training/file_lists")
LIST_DIR.mkdir(parents=True, exist_ok=True)

@router.post("/file-list")
async def create_file_list(repo: str, date_folder: str):
    """
    List one date folder (non-recursive) and persist file-list.json.
    Example: repo="myorg/surrogate-data", date_folder="2026-04-29"
    """
    try:
        tree = HF_API.list_repo_tree(repo=repo, path=date_folder, recursive=False)
        files = [
            f.rf_path for f in tree
            if f.rf_path and not f.rf_path.endswith("/") and not f.rf_path.startswith(".")
        ]
        payload = {"repo": repo, "date_folder": date_folder, "files": files}
        out = LIST_DIR / f"{date_folder}.json"
        out.write_text(json.dumps(payload, indent=2))
        return {"ok": True, "count": len(files), "file_list": str(out)}
    except Exception as exc:
        raise HTTPException(status_code=500, detail=str(exc))
```

```python
# surrogate/training/launch_studio.py
import json
from pathlib import Path
from lightning import LightningWork, LightningFlow

SHARED = Path("/shared/surrogate")
SHARED.mkdir(parents=True, exist_ok=True)

class CDNTrainingWork(LightningWork):
    def __init__(self, file_list_path: str, repo: str, **kwargs):
        super().__init__(**kwargs)
        self.file_list_path = file_list_path
        self.repo = repo
        # L40S is available on free-tier fallbacks; H200 requires lambda-prod quota
        self.machine = "L40S"

    def run(self):
        import subprocess
        cmd = [
            "python", "train_cdn.py",
            "--file-list", self.file_list_path,
            "--repo", self.repo,
            "--output-dir", str(SHARED / "outputs"),
            "--batch-size", "8"
        ]
        subprocess.run(cmd, check=True)

class CDNTrainingFlow(LightningFlow):
    def __init__(self):
        super().__init__()
        self.work = None

    def start_training(self, repo: str, file_list_path: str):
        from lightning import Teamspace
        studio_name = "surrogate-cdn-train"
        running = [
            s for s in Teamspace.studios
            if s.name == studio_name and s.status == "running"
        ]
        if running:
            self._trigger_in_running(running[0], repo, file_list_path)
        else:
            self.work = CDNTrainingWork(
                name=studio_name,
                file_list_path=file_list_path,
                repo=repo
            )
            self._spawn_work()

    def _spawn_work(self):
        if self.work and not self.work.has_started:
            self.work.start()

    def _trigger_in_running(self, studio, repo: str, file_list_path: str):
        # Signal running studio via shared trigger file (studio should poll or watch)
        trigger = SHARED / "trigger.json"
        trigger.write_text(json.dumps({
            "action": "start_cdn_training",
            "repo": repo,
            "file_list": file_list_path
        }, indent=2))
```

# surrogate/training/train_cdn.py
import argparse
import json
import sys
import time
from pathlib import Path
from typing import Dict, List, Iterator

import requests
import pytorch_lightning as pl
from torch.utils.data import IterableDataset, DataLoader

HF_CDN = "https://huggingface.co/datasets"

def _safe_request(url: str, retries: int = 3, timeout: int = 30) -> bytes:
    for attempt in range(1, retries + 1):
        try:
            resp = requests.get(url, timeout=timeout)
            resp.raise_for_status()
            return resp.content
        except Exception as exc:
            if attempt == retries:
                raise
            time.sleep(2 ** attempt)
    raise RuntimeError("unreachable")

def _extract_fields(obj: Dict) -> Dict[str, str]:
    if not isinstance(obj, dict):
        return {"prompt": "", "response": ""}
    prompt = obj.get("prompt") or obj.get("input") or obj.get("question") or ""
    response = obj.get("response") or obj.get("output") or obj.get("answer") or ""
    return {"prompt": str(prompt), "response": str(response)}

def _project_blob(content: bytes) -> List[Dict[str, str]]:
    text = content.decode("utf-8", errors="replace").strip()
    if not text:
        return []

    # Try single JSON object
    try:
        data = json.loads(text)
        if isinstance(data, list):
            return [_extract_fields(item) for item in data if isinstance(item, dict)]
        if isinstance(data, dict):
            return [_extract_fields(data)]
    except json.JSONDecodeError:
        pass

    # Try JSON Lines
    rows = []
    for line in text.splitlines():
        line = line.strip()
        if not line:
            continue
        try:
            obj = json.loads(line)
            if isinstance(obj, dict):
                rows.append(_extract_fields(obj))
            elif isinstance(obj, list):
                rows.extend(_extract_fields(item) for item in obj if isinstance(item, dict))
        except json.JSONDecodeError:
            continue
    return rows
