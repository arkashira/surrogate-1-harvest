# workio / ops

## 1) Diagnosis

- Missing **idempotency key** on `/webhook/line` POST allows LINE retries (network blips, 5xx, slow client) to create duplicate punch records.
- No **partial unique index** enforcing “one open punch per user” (`clock_out_at IS NULL`) permits multiple concurrent clock-ins and corrupts timesheet state.
- No **transactional upsert** for clock-in/out flows; race conditions under load can produce double punches or lost updates.
- Webhook handler lacks **short-circuit duplicate detection** (dedupe window) before DB write, increasing DB load and inconsistency risk.
- No **ops alerting/metrics** for duplicate punch events or constraint violations, delaying detection of data-quality incidents.

## 2) Proposed change

- **File scope**: `workio/server/src/routes/line/webhook.ts` (or equivalent) and `workio/server/src/db/schema.sql`.
- **Changes**:
  - Add partial unique index `idx_punches_user_open` on `punches(user_id) WHERE clock_out_at IS NULL`.
  - Add `idempotency_key` column (or use dedupe cache) to webhook handler; reject duplicates within a sliding window (e.g., 5–10 minutes).
  - Wrap clock-in/out logic in a transaction with `ON CONFLICT DO NOTHING` / application-level upsert to enforce single-open-punch invariant.
  - Emit structured log + metric (counter) on duplicate attempts for ops visibility.

## 3) Implementation

```sql
-- workio/server/src/db/schema.sql
-- Add partial unique index to enforce one open punch per user
CREATE UNIQUE INDEX IF NOT EXISTS idx_punches_user_open
ON punches(user_id)
WHERE clock_out_at IS NULL;
```

```ts
// workio/server/src/routes/line/webhook.ts
import { Router, Request, Response } from 'express';
import { pool } from '../../db';
import { v4 as uuidv4 } from 'uuid';

const router = Router();

// In-memory dedupe cache (short-term). Replace with Redis if available.
const dedupeCache = new Map<string, number>(); // key -> timestamp
const DEDUPE_WINDOW_MS = 10 * 60 * 1000; // 10 minutes

function isDuplicate(key: string): boolean {
  const now = Date.now();
  const ts = dedupeCache.get(key);
  if (ts && now - ts < DEDUPE_WINDOW_MS) return true;
  dedupeCache.set(key, now);
  // simple cleanup (best-effort)
  if (dedupeCache.size > 5000) {
    for (const [k, t] of dedupeCache.entries()) {
      if (now - t > DEDUPE_WINDOW_MS) dedupeCache.delete(k);
    }
  }
  return false;
}

router.post('/line', async (req: Request, res: Response) => {
  try {
    const events = req.body?.events;
    if (!Array.isArray(events) || events.length === 0) {
      return res.status(400).json({ error: 'invalid payload' });
    }

    const results = [];
    for (const event of events) {
      if (event?.type !== 'message' || event?.message?.type !== 'text') continue;

      const userId = event.source?.userId;
      const text = (event.message.text || '').trim().toLowerCase();
      if (!userId) continue;

      // Idempotency key: LINE message id + user + intent (clock in/out)
      const intent = text.includes('ออก') || text.includes('out') ? 'out' : 'in';
      const idemKey = `line:${userId}:${event.message.id}:${intent}`;

      if (isDuplicate(idemKey)) {
        // duplicate delivery — acknowledge but skip processing
        console.warn('Duplicate webhook skipped', { userId, idemKey });
        // emit metric: duplicate_line_webhook_total++
        results.push({ status: 'duplicate', userId });
        continue;
      }

      const client = await pool.connect();
      try {
        await client.query('BEGIN');

        // Enforce one open punch per user via partial unique index + upsert logic
        if (intent === 'in') {
          // Close any dangling open punch (defensive) then insert new clock-in
          await client.query(
            `UPDATE punches SET clock_out_at = NOW()
             WHERE user_id = $1 AND clock_out_at IS NULL`,
            [userId]
          );

          const insertRes = await client.query(
            `INSERT INTO punches (user_id, clock_in_at, clock_out_at, source, line_event_id)
             VALUES ($1, NOW(), NULL, 'line', $2)
             RETURNING id`,
            [userId, event.message.id]
          );
          results.push({ status: 'clocked_in', punchId: insertRes.rows[0].id });
        } else {
          // Clock out the latest open punch
          const updateRes = await client.query(
            `UPDATE punches
             SET clock_out_at = NOW()
             WHERE user_id = $1 AND clock_out_at IS NULL
             ORDER BY clock_in_at DESC
             LIMIT 1
             RETURNING id`,
            [userId]
          );
          if (updateRes.rowCount === 0) {
            // no open punch — create a clock-in then out (or reject). Here we create one for safety.
            await client.query(
              `INSERT INTO punches (user_id, clock_in_at, clock_out_at, source, line_event_id)
               VALUES ($1, NOW(), NOW(), 'line', $2)`,
              [userId, event.message.id]
            );
            results.push({ status: 'auto_paired', note: 'no open punch found' });
          } else {
            results.push({ status: 'clocked_out', punchId: updateRes.rows[0].id });
          }
        }

        await client.query('COMMIT');
      } catch (err) {
        await client.query('ROLLBACK');
        // emit metric: line_webhook_error_total++
        console.error('Webhook tx error', err);
        results.push({ status: 'error', error: String(err) });
      } finally {
        client.release();
      }
    }

    return res.json({ ok: true, results });
  } catch (err) {
    console.error('Webhook handler error', err);
    return res.status(500).json({ error: 'internal' });
  }
});

export default router;
```
