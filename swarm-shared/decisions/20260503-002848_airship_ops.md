## Final Decision (unified)

**Replace any HF API–driven dataset ingestion in the surrogate training pipeline with a manifest‑driven, CDN‑only iterable loader** (generated once on the Mac orchestrator) to eliminate 429 rate limits and allow Lightning Studio training to run with zero HF API calls during data loading.

```bash
#!/usr/bin/env bash
# generate_manifest.sh
# Usage: HF_REPO=datasets/your/repo HF_FOLDER=mirror-merged/2026-05-03 ./generate_manifest.sh
set -euo pipefail

HF_REPO="${HF_REPO:?required}"
HF_FOLDER="${HF_FOLDER:?required}"
OUT="${OUT:-file_manifest.json}"

python3 - "$HF_REPO" "$HF_FOLDER" "$OUT" <<'PY'
import json, os, sys
from huggingface_hub import HfApi

repo_id, folder = sys.argv[1], sys.argv[2]
out_path = sys.argv[3]

api = HfApi()
items = api.list_repo_tree(repo_id=repo_id, path=folder, recursive=False)

files = []
for item in items:
    path = getattr(item, "path", None)
    if not path:
        continue
    files.append({
        "filename": os.path.basename(path),
        "repo_id": repo_id,
        "repo_type": "dataset",
        "path": path,
        "cdn_url": f"https://huggingface.co/datasets/{repo_id}/resolve/main/{path}"
    })

with open(out_path, "w") as f:
    json.dump({"repo_id": repo_id, "folder": folder, "files": files}, f, indent=2)

print(f"Wrote {len(files)} files to {out_path}")
PY
```

### 2) CDN‑only iterable dataset (surrogate/training/data.py)

```python
# surrogate/training/data.py
import json
import io
from typing import Dict, Iterator, Any
from pathlib import Path

import torch
from torch.utils.data import IterableDataset
import pyarrow.parquet as pq
import requests

class CdnOnlyIterableDataset(IterableDataset):
    """
    Zero HF API calls during training.
    Manifest is generated once on orchestrator (Mac) and shipped with the job.
    Each sample is projected to {prompt, response} on the fly.
    """

    def __init__(self, manifest_path: str, columns: tuple = ("prompt", "response")):
        super().__init__()
        self.manifest_path = Path(manifest_path)
        self.columns = columns
        with open(self.manifest_path, "r") as f:
            self.manifest = json.load(f)
        self.files = self.manifest["files"]
        if not self.files:
            raise ValueError("No files found in manifest")

    def _stream_file(self, file_info: Dict[str, str]) -> Iterator[Dict[str, Any]]:
        cdn_url = file_info["cdn_url"]
        resp = requests.get(cdn_url, timeout=30)
        resp.raise_for_status()
        buf = io.BytesIO(resp.content)
        table = pq.read_table(buf, columns=self.columns)
        for row in table.to_pylist():
            yield {
                "prompt": row.get("prompt") or "",
                "response": row.get("response") or "",
            }

    def __iter__(self) -> Iterator[Dict[str, Any]]:
        worker_info = torch.utils.data.get_worker_info()
        if worker_info is None:
            files = self.files
        else:
            # shard per worker to avoid duplication
            per_worker = len(self.files) // worker_info.num_workers
            start = worker_info.id * per_worker
            end = (worker_info.id + 1) * per_worker if worker_info.id + 1 < worker_info.num_workers else len(self.files)
            files = self.files[start:end]

        for file_info in files:
            yield from self._stream_file(file_info)
```

```python
        pin_memory=True,

    model = ...  # your surrogate model init
    optimizer = torch.optim.AdamW(model.parameters(), lr=1e-4)

    model, optimizer = fabric.setup(model, optimizer)
    loader = fabric.setup_dataloaders(loader)

    fabric.run(_train_loop, loader, model, optimizer, max_steps=args.max_steps)

def _train_loop(loader, model, optimizer, max_steps):
    step = 0
    for batch in loader:
        if step >= max_steps:
            break
        out = model(batch["prompt"], batch["response"])
        loss = out["loss"]
        fabric.backward(loss)
        optimizer.step()
        optimizer.zero_grad()
        step += 1

if __name__ == "__main__":
    main()
```

### 4) Lightning Studio reuse + idle handling (surrogate/training/launch.py)

```python
# surrogate/training/launch.py
from lightning import Teamspace, Studio, Machine
import time

def ensure_running_studio(name: str, machine_type: str = "L40S") -> Studio:
    teamspace = Teamspace()
    for s in teamspace.studios:
        if s.name == name:
            if s.status == "Running":
                print(f"Reusing running studio: {name}")
                return s
            else:
                print(f"Restarting stopped studio: {name}")
                s.start(machine=Machine(machine_type))
                while s.status != "Running":
                    time.sleep(10)
                    s.refresh()
                return s

    print(f"Creating new studio: {name}")
    return Studio(
        name=name,
        machine=Machine(machine_type),
        create_ok=True,
    )

if __name__ == "__main__":
    studio = ensure_running_studio("surrogate-train-l40s", "L40S")
    studio.run(
        "python surrogate/training/train.py --manifest file_manifest.json --max-steps 5000",
        cwd="/workspace/airship/surrogate",
    )
```
