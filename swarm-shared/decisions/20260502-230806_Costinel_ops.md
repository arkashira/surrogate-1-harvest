# Costinel / ops

## Final Implementation Plan — Costinel read-only top-hub signal (≤2 h)

**Goal:**  
Ship a single, production-ready, read-only endpoint that surfaces the strongest knowledge-graph hub as a cost-anomaly signal (Sense + Signal). Zero execution. Zero state change. Reuses existing knowledge-rag/graph patterns.

---

### 1) Core design decisions (resolved)

- **Date handling**: accept `?for_date=YYYY-MM-DD`; default to today (UTC). Validate strictly; return 422 on bad format.
- **Response shape**: include `hub_id`, `hub_label`, `score`, `summary`, `related_docs` (lightweight), `for_date`, `generated_at`. This balances completeness with minimal payload.
- **Error handling**:
  - 422 for client input errors (bad date).
  - 500 for internal failures with generic message; log details internally.
- **No breaking changes**: no new secrets, no migrations, no background jobs, no execution path.
- **Testing**: one focused unit test that mocks KnowledgeRAG and verifies shape + status.

#### A) Signal service (core logic)
`costinel/services/top_hub_signal.py`
```python
from __future__ import annotations

from datetime import datetime, timezone
from typing import Any, Dict, Optional

from pydantic import BaseModel, Field, field_validator

from costinel.libs.knowledge_rag import KnowledgeRAG  # existing graph client


class TopHubSignal(BaseModel):
    hub_id: str = Field(..., description="Strongest hub node id (e.g., MOC)")
    hub_label: str = Field(..., description="Human-readable label")
    score: float = Field(..., ge=0, le=1, description="Connection strength / centrality")
    summary: str = Field(..., description="Short insight summary")
    related_docs: list[Dict[str, Any]] = Field(
        default_factory=list,
        description="Top related docs for context",
    )
    for_date: str = Field(..., description="Date evaluated (YYYY-MM-DD)", pattern=r"^\d{4}-\d{2}-\d{2}$")
    generated_at: str = Field(..., description="ISO-8601 UTC timestamp")

    @field_validator("score")
    @classmethod
    def clamp_score(cls, v: float) -> float:
        return max(0.0, min(1.0, float(v)))


class TopHubSignalService:
    """
    Read-only signal provider.
    Uses existing knowledge-rag/graph to resolve the strongest hub.
    """

    def __init__(self, rag: KnowledgeRAG) -> None:
        self.rag = rag

    def get_signal(self, for_date: Optional[str] = None) -> TopHubSignal:
        # Normalize and validate date
        if for_date:
            try:
                datetime.strptime(for_date, "%Y-%m-%d")
            except ValueError as exc:
                raise ValueError("for_date must be YYYY-MM-DD") from exc
            target_date = for_date
        else:
            target_date = datetime.now(timezone.utc).strftime("%Y-%m-%d")

        # Resolve top hub via existing graph API
        top_hub = self.rag.top_hub(for_date=target_date)  # expected keys: id, label/name, score, summary
        related = self.rag.related_docs(node_id=top_hub["id"], limit=5)

        signal = TopHubSignal(
            hub_id=top_hub["id"],
            hub_label=top_hub.get("label", top_hub.get("name", top_hub["id"])),
            score=float(top_hub.get("score", 0.0)),
            summary=top_hub.get("summary", ""),
            related_docs=[
                {
                    "doc_id": d["id"],
                    "title": d.get("title", ""),
                    "snippet": d.get("snippet", ""),
                }
                for d in related
            ],
            for_date=target_date,
            generated_at=datetime.now(timezone.utc).isoformat(),
        )
        return signal
```

#### B) API route (FastAPI)
`costinel/api/v1/endpoints/cost_anomaly.py`
```python
from fastapi import APIRouter, Depends, HTTPException, Query
from typing import Optional

from costinel.services.top_hub_signal import TopHubSignalService, TopHubSignal
from costinel.libs.knowledge_rag import KnowledgeRAG

router = APIRouter()


def get_signal_service() -> TopHubSignalService:
    # Lightweight factory; reuses existing RAG instance
    return TopHubSignalService(rag=KnowledgeRAG())


@router.get(
    "/cost-anomaly/signal/top-hub",
    response_model=TopHubSignal,
    tags=["cost-anomaly"],
    summary="Read-only top-hub signal (Sense + Signal)",
)
def read_top_hub_signal(
    for_date: Optional[str] = Query(
        None,
        description="Date to evaluate (YYYY-MM-DD). Defaults to today (UTC).",
        regex=r"^\d{4}-\d{2}-\d{2}$",
        example="2026-05-02",
    ),
    service: TopHubSignalService = Depends(get_signal_service),
) -> TopHubSignal:
    """
    Returns the strongest knowledge-graph hub as a cost-anomaly signal.
    Read-only. No execution.
    """
    try:
        return service.get_signal(for_date=for_date)
    except ValueError as exc:
        raise HTTPException(status_code=422, detail=str(exc)) from exc
    except Exception as exc:
        # Log internally in real implementation
        raise HTTPException(status_code=500, detail="Unable to produce signal") from exc
```

#### D) Minimal unit test

```python
from fastapi.testclient import TestClient
from unittest.mock import MagicMock, patch

from costinel.main import app  # adjust import to your app entrypoint

client = TestClient(app)


def test_read_top_hub_signal():
    mock_hub = {
        "id": "MOC",
        "label": "MOC",
        "score": 0.92,
        "summary": "Cost spike linked to MOC services",
    }
    mock_docs = [{"id": "d1", "title": "MOC anomaly brief", "snippet": "Detected 3x baseline"}]

    with patch("costinel.api.v1.endpoints.cost_anomaly.KnowledgeRAG") as MockRAG:
        mock_rag = MagicMock()
        mock_rag.top_hub.return_value = mock_hub
        mock_rag.related_docs.return_value = mock_docs
        MockRAG.return_value = mock_rag

        resp = client.get("/api/v1/cost-anomaly/signal/top-hub?for_date=2026-05-02")

    assert resp.status_code == 200
    data = resp.json()
    assert data["hub_id"] == "MOC"
    assert data["hub_label"] == "MOC"
    assert data["score"] == 0.92
    assert data["summary"] == "Cost spike linked to MOC services"
    assert data["for_date"] == "2026-05-02"
    assert len(data["related_docs"]) == 1
    assert data["related_docs"][0]["doc_id"] == "d1"
```
