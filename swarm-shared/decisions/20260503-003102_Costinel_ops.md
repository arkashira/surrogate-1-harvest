# Costinel / ops

## Implementation Plan — Costinel “Top-Hub Signal” Card (read-only)

**Scope**: Frontend-only card that identifies the most-connected hub from the knowledge-rag graph and surfaces the top hub with 3 contextual signals.  
**Timebox**: <2h (frontend-only, mock-first; backend adapter optional).

---

### 1) Design (5 min)

- Card title: **Top Hub Signal**
- Subtitle: Most-connected hub from knowledge-rag graph
- Body:
  - Hub name + short description
  - 3 contextual signals (bulleted, concise)
  - Timestamp of last graph refresh
- States:
  - Loading
  - Loaded
  - Empty / Error

### 2) Data contract (frontend-first)

```ts
// src/types/knowledge-rag.ts
export interface HubSignal {
  hub: {

  signals: Array<{
    id: string;
    title: string;

    href?: string;

  }>;
  lastUpdated: string; // ISO
}
```

### 3) Component (React + Tailwind)

```tsx
// src/components/TopHubSignalCard.tsx
import { useEffect, useState } from "react";
import type { HubSignal } from "../types/knowledge-rag";
import { mockTopHubSignal } from "../mocks/topHubSignal";

async function fetchTopHubSignal(): Promise<HubSignal> {
  // TODO: replace with real endpoint when available
  // Example: return fetch("/api/knowledge-rag/top-hub").then((r) => r.json());
  return Promise.resolve(mockTopHubSignal);
}

export function TopHubSignalCard() {
  const [data, setData] = useState<HubSignal | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    let mounted = true;
    setLoading(true);
    fetchTopHubSignal()
      .then((res) => {
        if (mounted) setData(res);
      })
      .catch((err) => {
        if (mounted) setError(err.message || "Failed to load top hub signal");
      })
      .finally(() => {
        if (mounted) setLoading(false);
      });
    return () => {
      mounted = false;
    };
  }, []);

  if (loading) {
    return (
      <div className="rounded-lg border border-gray-200 bg-white p-5 shadow-sm">
        <div className="h-6 w-32 animate-pulse rounded bg-gray-200" />
        <div className="mt-3 h-4 w-24 animate-pulse rounded bg-gray-100" />
        <div className="mt-4 space-y-3">
          {[...Array(3)].map((_, i) => (
            <div key={i} className="h-10 animate-pulse rounded bg-gray-100" />
          ))}
        </div>
      </div>
    );
  }

  if (error || !data) {
    return (
      <div className="rounded-lg border border-gray-200 bg-white p-5 shadow-sm">
        <p className="text-sm text-red-600">{error || "No data available"}</p>
      </div>
    );
  }

  return (
    <div className="rounded-lg border border-gray-200 bg-white p-5 shadow-sm">
      <div className="flex items-start justify-between gap-2">
        <div>
          <h3 className="text-base font-semibold text-gray-900">Top Hub Signal</h3>
          <p className="text-sm text-gray-500">Most-connected hub from knowledge-rag graph</p>
        </div>
        <span className="inline-flex items-center rounded-full bg-blue-50 px-2.5 py-0.5 text-xs font-medium text-blue-700">
          {data.hub.id}
        </span>
      </div>

      <div className="mt-3">
        <p className="font-medium text-gray-900">{data.hub.name}</p>
        {data.hub.description && (
          <p className="mt-1 text-sm text-gray-600">{data.hub.description}</p>
        )}
      </div>

      <ul className="mt-4 space-y-3" aria-label="Contextual signals">
        {data.signals.map((s) => (
          <li key={s.id} className="flex gap-3 text-sm">
            <span className="mt-1 flex-shrink-0 text-blue-600" aria-hidden="true">
              •
            </span>
            <div className="min-w-0">
              <p className="font-medium text-gray-900">{s.title}</p>
              <p className="text-gray-600">{s.summary}</p>
              {s.tags && s.tags.length > 0 && (
                <p className="mt-1 flex flex-wrap gap-1">
                  {s.tags.map((tag) => (
                    <span
                      key={tag}
                      className="inline-flex items-center rounded-full bg-gray-100 px-2 py-0.5 text-xs text-gray-600"
                    >
                      {tag}
                    </span>
                  ))}
                </p>
              )}
            </div>
          </li>
        ))}
      </ul>

      <p className="mt-4 text-xs text-gray-400">
        Last updated: {new Date(data.lastUpdated).toLocaleString()}
      </p>
    </div>
  );
}
```
