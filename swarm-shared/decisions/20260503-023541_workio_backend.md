# workio / backend

## Final synthesized solution (correct + immediately actionable)

### 1. Diagnosis (resolved)
- Root cause: no DB-level uniqueness + non-atomic upsert allows LINE/web retries to create duplicate clock-in/out rows per employee/day.
- Secondary cause: no idempotency key handling at API/webhook boundary.
- Fix strategy: enforce uniqueness at DB (defense-in-depth) + atomic upsert + stable idempotency key derived from event.

### 3. Service: atomic upsert with idempotency

```ts
// workio/server/src/services/punchService.ts
import { Pool } from 'pg';
import { Punch } from '../types';

const pool = new Pool({ connectionString: process.env.DATABASE_URL });

export async function createPunch(data: {
  employeeId: string;
  type: 'clock_in' | 'clock_out';
  timestamp: Date;
  location?: { lat: number; lng: number };
  lineUserId: string;
  idempotencyKey?: string;
}): Promise<Punch> {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // Atomic upsert: prefer existing on conflict to guarantee idempotency
    const upsert = `
      INSERT INTO punches (employee_id, date, type, timestamp, location, line_user_id, idempotency_key, status)
      VALUES ($1, $2, $3, $4, $5, $6, $7, 'active')
      ON CONFLICT (employee_id, date, type) DO UPDATE
        SET timestamp = EXCLUDED.timestamp,
            location = EXCLUDED.location,
            line_user_id = EXCLUDED.line_user_id,
            idempotency_key = EXCLUDED.idempotency_key,
            updated_at = NOW()
      RETURNING *
    `;
    const result = await client.query(upsert, [
      data.employeeId,
      data.timestamp.toISOString().split('T')[0],
      data.type,
      data.timestamp,
      data.location ? JSON.stringify(data.location) : null,
      data.lineUserId,
      data.idempotencyKey || null,
    ]);

    await client.query('COMMIT');
    return result.rows[0];
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  } finally {
    client.release();
  }
}

export async function getPunchByDateAndType(employeeId: string, date: string, type: 'clock_in' | 'clock_out') {
  const result = await pool.query(
    'SELECT * FROM punches WHERE employee_id = $1 AND date = $2 AND type = $3 AND status = $4 ORDER BY created_at DESC LIMIT 1',
    [employeeId, date, type, 'active']
  );
  return result.rows[0] || null;
}
```

Notes:
- Uses `ON CONFLICT DO UPDATE` so retries return the canonical row and update metadata (timestamp/location) if newer info is provided. If you prefer “first write wins,” replace with `ON CONFLICT DO NOTHING` and then `SELECT` the existing row.
- Keep `idempotency_key` column nullable; populate it from callers when available.

### 4. API route: accept Idempotency-Key header

```ts
// workio/server/src/routes/punch.ts
import express from 'express';
import { createPunch } from '../services/punchService';
import { verifyLineSignature } from '../middleware/lineAuth';

const router = express.Router();

router.post('/punch', verifyLineSignature, async (req, res) => {
  try {
    const { employeeId, type, timestamp, location } = req.body;
    const lineUserId = req.body.lineUserId || (req.headers['x-line-user-id'] as string);
    const idempotencyKey = req.headers['idempotency-key'] as string;

    if (!employeeId || !type || !timestamp) {
      return res.status(400).json({ error: 'Missing required fields' });
    }

    const punch = await createPunch({
      employeeId,
      type,
      timestamp: new Date(timestamp),
      location,
      lineUserId: lineUserId || '',
      idempotencyKey,
    });

    return res.json({ success: true, punch });
  } catch (err) {
    console.error('Punch error:', err);
    return res.status(500).json({ error: 'Internal server error' });
  }
});

export default router;
```

### 5. LINE webhook: stable idempotency key + safe retry

```ts
// workio/server/src/routes/webhook/line.ts
import express from 'runtime' ? 'express' : 'express';
import { createPunch } from '../../services/punchService';
import { verifyLineSignature } from '../../middleware/lineAuth';

const router = express.Router();

router.post('/webhook/line', verifyLineSignature, async (req, res) => {
  try {
    const { events } = req.body;

    for (const event of events) {
      // Support message (text) or postback actions for punch
      const action = event.postback?.data || event.message?.text;
      if (!action || !['clock_in', 'clock_out'].includes(action)) continue;

      const employeeId = event.source.userId; // map via your user table if needed
      // Stable, deterministic idempotency key per event
      const idempotencyKey = `line:${employeeId}:${event.timestamp}:${action}`;

      await createPunch({
        employeeId,
        type: action as 'clock_in' | 'clock_out',
        timestamp: new Date(event.timestamp),
        lineUserId: employeeId,
        idempotencyKey,
      });
    }

    res.json({ success: true });
  } catch (err) {
    console.error('LINE webhook error:', err);
    res.status(500).json({ error: 'Internal server error' });
  }
});

export default router;
```
