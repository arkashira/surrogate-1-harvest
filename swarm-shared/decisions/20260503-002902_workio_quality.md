# workio / quality

## 1) Diagnosis

- No idempotency enforcement on `/punches` write path → repeated LINE webhook deliveries or frontend retries create duplicate punch records.
- Missing DB uniqueness constraint for “one open punch per user” → allows multiple concurrent clock-ins or overlapping open punches.
- No server-side de-duplication key (idempotency key) stored with punch records → retries cannot be detected reliably.
- Punch creation accepts raw client timestamps without server-side monotonicity checks → potential back-dating or invalid sequences.
- No transactional upsert or `ON CONFLICT` handling in punch write path → race conditions under concurrent webhook deliveries.

## 2) Proposed change

File: `/opt/axentx/workio/server/src/routes/punches.ts` (or equivalent route handler for punch creation)  
Scope: add idempotency key enforcement and DB uniqueness constraint for open punches; implement transactional upsert with server-side validation.

## 3) Implementation

-- Prevent multiple open punches per user (null clock_out_ts means open)
CREATE UNIQUE INDEX IF NOT EXISTS idx_one_open_punch_per_user
ON punches (user_id)
WHERE clock_out_ts IS NULL;

-- Optional: idempotency table for webhook-level dedupe
CREATE TABLE IF NOT EXISTS punch_idempotency (
  idempotency_key TEXT PRIMARY KEY,
  punch_id INTEGER NOT NULL,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

```typescript
// 2) /opt/axentx/workio/server/src/routes/punches.ts
import express from 'express';
import { db } from '../db';
import { v4 as uuidv4 } from 'uuid';

const router = express.Router();

/**
 * POST /punches
 * Body: { user_id: number, type: 'in' | 'out', idempotency_key?: string, ts?: ISOString }
 */
router.post('/', async (req, res) => {
  const { user_id, type, idempotency_key, ts } = req.body;
  const client = await db.connect();

  try {
    await client.query('BEGIN');

    // Idempotency check
    if (idempotency_key) {
      const existing = await client.query(
        'SELECT punch_id FROM punch_idempotency WHERE idempotency_key = $1',
        [idempotency_key]
      );
      if (existing.rows.length > 0) {
        const punch = await client.query('SELECT * FROM punches WHERE id = $1', [existing.rows[0].punch_id]);
        await client.query('ROLLBACK');
        return res.json({ ok: true, punch: punch.rows[0], idempotent: true });
      }
    }

    const now = ts ? new Date(ts) : new Date();

    if (type === 'in') {
      // Ensure no open punch exists (DB constraint will enforce; defensive check)
      const open = await client.query(
        'SELECT id FROM punches WHERE user_id = $1 AND clock_out_ts IS NULL',
        [user_id]
      );
      if (open.rows.length > 0) {
        await client.query('ROLLBACK');
        return res.status(409).json({ error: 'open_punch_exists', punch_id: open.rows[0].id });
      }

      const insert = await client.query(
        'INSERT INTO punches (user_id, clock_in_ts, clock_out_ts) VALUES ($1, $2, $3) RETURNING *',
        [user_id, now, null]
      );

      if (idempotency_key) {
        await client.query(
          'INSERT INTO punch_idempotency (idempotency_key, punch_id) VALUES ($1, $2)',
          [idempotency_key, insert.rows[0].id]
        );
      }

      await client.query('COMMIT');
      return res.json({ ok: true, punch: insert.rows[0] });
    }

    if (type === 'out') {
      // Close most recent open punch
      const open = await client.query(
        'SELECT id FROM punches WHERE user_id = $1 AND clock_out_ts IS NULL ORDER BY clock_in_ts DESC LIMIT 1',
        [user_id]
      );
      if (open.rows.length === 0) {
        await client.query('ROLLBACK');
        return res.status(404).json({ error: 'no_open_punch' });
      }

      const punchId = open.rows[0].id;
      const update = await client.query(
        'UPDATE punches SET clock_out_ts = $1 WHERE id = $2 RETURNING *',
        [now, punchId]
      );

      if (idempotency_key) {
        // upsert idempotency
        await client.query(
          `INSERT INTO punch_idempotency (idempotency_key, punch_id)
           VALUES ($1, $2)
           ON CONFLICT (idempotency_key) DO NOTHING`,
          [idempotency_key, punchId]
        );
      }

      await client.query('COMMIT');
      return res.json({ ok: true, punch: update.rows[0] });
    }

    await client.query('ROLLBACK');
    return res.status(400).json({ error: 'invalid_type' });
  } catch (err: any) {
    await client.query('ROLLBACK');
    // Unique violation from idx_one_open_punch_per_user or other constraint
    if (err.code === '23505') {
      return res.status(409).json({ error: 'duplicate_open_punch' });
    }
    console.error(err);
    return res.status(500).json({ error: 'server_error' });
  } finally {
    client.release();
  }
});

export default router;
```
