## Final Implementation Plan (≤2h)

**Highest-value improvement**: Replace fragile shell-based ingestion with a **manifest-driven, CDN-first Python worker** that eliminates HF API rate limits during training data loads and prevents mixed-schema `CastError`s.

### 1. Add `bin/worker.py` (single-file, manifest-driven)

- Accepts `SHARD_ID` and `TOTAL_SHARDS` from the matrix workflow.
- One HF API call (after rate-limit window) to list a **single date folder** via `list_repo_tree(recursive=False)`.
- Persists file list to `manifest-{date}.json` (committed or embedded).
- During training, Lightning Studio loads data via **CDN-only URLs** (`https://huggingface.co/datasets/.../resolve/main/...`) with zero API calls.
- Projects heterogeneous files to `{prompt, response}` only at parse time (avoids `pyarrow.CastError`).
- Deterministic shard assignment via `hash(slug) % TOTAL_SHARDS`.
- Central dedup via existing `lib/dedup.py` SQLite store.
- Outputs `batches/public-merged/{date}/shard{N}-{HHMMSS}.jsonl`.
- Uploads shard back to the dataset repo (one commit per shard per run).

```python
#!/usr/bin/env python3
"""
Manifest-driven, CDN-first ingestion worker for surrogate-1.
Usage (GH Actions):
  SHARD_ID=0 TOTAL_SHARDS=16 python bin/worker.py
"""
import os
import sys
import json
import hashlib
import datetime
import sqlite3
from pathlib import Path

import requests
from huggingface_hub import list_repo_tree, hf_hub_download, upload_file

HF_REPO = "datasets/axentx/surrogate-1-training-pairs"
HF_TOKEN = os.getenv("HF_TOKEN")
if not HF_TOKEN:
    print("HF_TOKEN required", file=sys.stderr)
    sys.exit(1)

TOTAL_SHARDS = int(os.getenv("TOTAL_SHARDS", "16"))
SHARD_ID = int(os.getenv("SHARD_ID", "0"))
DATE = os.getenv("DATE", datetime.datetime.utcnow().strftime("%Y-%m-%d"))
OUT_DIR = Path(f"batches/public-merged/{DATE}")
OUT_DIR.mkdir(parents=True, exist_ok=True)
TS = datetime.datetime.utcnow().strftime("%H%M%S")
OUT_FILE = OUT_DIR / f"shard{SHARD_ID}-{TS}.jsonl"

DEDUP_DB = Path(os.getenv("DEDUP_DB", "dedup.db"))

def get_db() -> sqlite3.Connection:
    conn = sqlite3.connect(DEDUP_DB)
    conn.execute("CREATE TABLE IF NOT EXISTS seen (md5 TEXT PRIMARY KEY)")
    conn.commit()
    return conn

def is_seen(conn: sqlite3.Connection, md5: str) -> bool:
    cur = conn.execute("SELECT 1 FROM seen WHERE md5=?", (md5,))
    return cur.fetchone() is not None

def mark_seen(conn: sqlite3.Connection, md5: str) -> None:
    conn.execute("INSERT OR IGNORE INTO seen (md5) VALUES (?)", (md5,))

def list_date_files() -> list[str]:
    """Single API call: list only the target date folder (non-recursive)."""
    manifest_path = Path(f"manifest-{DATE}.json")
    if manifest_path.exists():
        print(f"Using manifest {manifest_path}")
        return json.loads(manifest_path.read_text())

    print(f"Listing {HF_REPO}/{DATE} (non-recursive)...")
    tree = list_repo_tree(
        repo_id=HF_REPO,
        path=DATE,
        repo_type="dataset",
        token=HF_TOKEN,
        recursive=False,
    )
    files = [f.rfilename for f in tree if f.type == "file"]
    manifest_path.write_text(json.dumps(files, indent=2))
    return files

def cdn_url(path: str) -> str:
    """CDN URL — no Authorization header, bypasses API rate limits."""
    return f"https://huggingface.co/datasets/{HF_REPO}/resolve/main/{path}"

def parse_file_to_pairs(local_path: Path) -> list[dict[str, str]]:
    """
    Project heterogeneous files to {prompt, response} only.
    Implement per known schema; keep minimal and defensive.
    """
    import pyarrow.parquet as pq

    try:
        table = pq.read_table(local_path, columns=["prompt", "response"])
    except Exception:
        try:
            table = pq.read_table(local_path)
            cols = set(table.column_names)
            prompt_col = next(
                (c for c in ("prompt", "instruction", "input") if c in cols), None
            )
            response_col = next(
                (c for c in ("response", "output", "completion") if c in cols), None
            )
            if not prompt_col or not response_col:
                return []
            table = table.select([prompt_col, response_col]).rename_columns(
                ["prompt", "response"]
            )
        except Exception:
            return []

    df = table.to_pandas()
    pairs = []
    for _, row in df.iterrows():
        prompt = str(row.get("prompt") or "").strip()
        response = str(row.get("response") or "").strip()
        if prompt and response:
            pairs.append({"prompt": prompt, "response": response})
    return pairs

def worker_loop() -> None:
    conn = get_db()
    files = list_date_files()

    # Deterministic shard assignment by slug (filename without extension)
    assigned = []
    for f in files:
        slug = Path(f).stem
        h = int(hashlib.sha256(slug.encode()).hexdigest(), 16)
        if h % TOTAL_SHARDS == SHARD_ID:
            assigned.append(f)

    print(f"Shard {SHARD_ID}/{TOTAL_SHARDS}: processing {len(assigned)} files")
    written = 0
    with open(OUT_FILE, "w", encoding="utf-8") as out_f:
        for f in assigned:
            try:
                local = hf_hub_download(
                    repo_id=HF_REPO,
                    filename=f,
                    repo_type="dataset",
                    token=HF_TOKEN,
                )
                pairs = parse_file_to_pairs(Path(local))
                for p in pairs:
                    blob = f"{p['prompt']}\n{p['response']}".encode()
                    md5 = hashlib.md5(blob).hexdigest()
                    if is_seen(conn, md5):
                        continue
                    mark_seen(conn, md5)
                    out_f.write(json.dumps(p, ensure_ascii=False) + "\n")
                    written += 1
                conn.commit()
            except Exception as e:
                print(f"Error processing {f}: {e}", file=sys.stderr)

    print(f"Shard {SHARD_ID} wrote {written} pairs to {OUT_FILE}")

    try:
        upload_file(
            path_or_fileobj=str(OUT_FILE),
            path_in_repo=str(OUT_FILE),
            repo_id=HF_REPO,
            repo_type="dataset",
            token=HF_TOKEN,
            commit_message=f"shard {SHARD_ID} {DATE} {TS}",
        )
        print("Upload complete")
    except Exception as e:
        print(f"Upload failed (may be handled by CI): {e}", file=sys.stderr)

if __name__ == "__main__":
    worker_loop()
```
