```python
# /opt/axentx/vanguard/design/cdn_manifest.py
import json, hashlib, os, time
from pathlib import Path
from typing import List, Dict, Any, Optional
import requests

try:
    from huggingface_hub import HfApi
    HF_HUB_AVAILABLE = True
except Exception:
    HF_HUB_AVAILABLE = False

try:
    from lightning import LightningWork, LightningApp, Machine, Teamspace
    LIGHTNING_AVAILABLE = True
except Exception:
    LIGHTNING_AVAILABLE = False

HF_CDN_ROOT = "https://huggingface.co/datasets"
MANIFEST_ROOT = Path(__file__).parent.parent / "manifests"
MANIFEST_ROOT.mkdir(parents=True, exist_ok=True)


def _cdn_url(repo: str, path: str) -> str:
    return f"{HF_CDN_ROOT}/{repo}/resolve/main/{path}"


def _hash_slug(path: str) -> str:
    return hashlib.sha256(path.encode()).hexdigest()[:16]


def _content_hash(entries: List[Dict[str, str]]) -> str:
    """Content-addressed hash of the file list for snapshot pinning."""
    canonical = json.dumps(
        sorted(entries, key=lambda e: (e["repo"], e["path"])),
        separators=(",", ":"),
        sort_keys=True,
    )
    return hashlib.sha256(canonical.encode()).hexdigest()[:16]


def build_manifest(
    repo: str,
    date_folder: str,
    hf_token: Optional[str] = None,
    out_dir: Optional[Path] = None,
    recursive: bool = False,
) -> Dict[str, Any]:
    """
    Build a CDN-first manifest for one date folder.
    Requires one API call (list_repo_tree) run from a controlled machine
    during a clear rate-limit window.
    """
    if not HF_HUB_AVAILABLE:
        raise RuntimeError("huggingface_hub required to list repo tree.")

    api = HfApi(token=hf_token) if hf_token else None
    if api is None:
        raise RuntimeError("hf_token required to list repo tree (one-time operation).")

    items = api.list_repo_tree(repo=repo, path=date_folder, recursive=recursive)
    files = [it.rfilename for it in items if it.type == "file"]

    entries = []
    for f in files:
        url = _cdn_url(repo, f)
        entries.append({
            "repo": repo,
            "path": f,
            "url": url,
            "slug": _hash_slug(f),
        })

    manifest = {
        "repo": repo,
        "date_folder": date_folder,
        "generated_at": time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime()),
        "count": len(entries),
        "snapshot_hash": _content_hash(entries),
        "entries": entries,
    }

    out_dir = Path(out_dir or MANIFEST_ROOT)
    out_dir.mkdir(parents=True, exist_ok=True)
    out_file = out_dir / f"{date_folder.replace('/', '_')}.json"
    out_file.write_text(json.dumps(manifest, indent=2))
    return manifest


def load_manifest(
    repo: str,
    date_folder: str,
    manifest_dir: Optional[Path] = None,
) -> Dict[str, Any]:
    manifest_dir = Path(manifest_dir or MANIFEST_ROOT)
    candidates = list(manifest_dir.glob(f"*{date_folder.replace('/', '_')}.json"))
    if not candidates:
        raise FileNotFoundError(f"No manifest found for {repo}/{date_folder}")
    manifest_file = sorted(candidates)[0]
    return json.loads(manifest_file.read_text())


def stream_cdn_files(
    manifest: Dict[str, Any],
    max_concurrent: int = 8,
    timeout: int = 30,
) -> Dict[str, bytes]:
    """
    Fetch files from CDN URLs in the manifest.
    Returns {path: content_bytes}.
    """
    from concurrent.futures import ThreadPoolExecutor, as_completed

    def _fetch(entry: Dict[str, str]) -> tuple:
        resp = requests.get(entry["url"], timeout=timeout)
        resp.raise_for_status()
        return entry["path"], resp.content

    results = {}
    with ThreadPoolExecutor(max_workers=max_concurrent) as ex:
        futures = {ex.submit(_fetch, e): e for e in manifest["entries"]}
        for fut in as_completed(futures):
            path, data = fut.result()
            results[path] = data
    return results


# ---- Lightning Studio reuse + idle-stop resilience ----
if LIGHTNING_AVAILABLE:
    def get_or_create_studio(
        name: str,
        machine: str = "Lightning-AI/L40S-1x",
        cloud: str = "lightning-public-prod",
        timeout_min: int = 5,
    ) -> LightningWork:
        """
        Reuse a running studio if present; if stopped, restart it.
        Avoids recreating studios and burning quota.
        """
        teamspace = Teamspace()
        running = None
        for s in teamspace.studios:
            if s.name == name:
                running = s
                break

        if running is None:
            studio = LightningWork(
                cloud=cloud,
                machine=machine,
                name=name,
                enable_app_server=False,
            )
            studio.start()
            return studio

        if running.status == "stopped":
            running.start(machine=machine)
            for _ in range(timeout_min * 6):
                time.sleep(10)
                if running.status == "running":
                    break
            if running.status != "running":
                raise RuntimeError(f"Studio {name} failed to start (status={running.status})")
        elif running.status != "running":
            for _ in range(timeout_min * 6):
                time.sleep(10)
                if running.status == "running":
                    break

        return running
else:
    def get_or_create_studio(name: str, machine: str = "Lightning-AI/L40S-1x", **kwargs):
        raise RuntimeError("Lightning SDK not available; cannot manage studios.")
```

Example usage (orchestrator on Mac):
```python
# build once per date folder (run during off-peak)
manifest = build_manifest(
    repo="myorg/surrogate-1-ingest",
    date_folder="batches/mirror-merged/2026-04-29",
    hf_token=os.getenv("HF_TOKEN"),
)

# training script (Lightning) uses CDN-only
manifest = load_manifest("myorg/surrogate-1-ingest", "batches/mirror-merged/2026-04-29")
# stream from CDN without HF API calls
files = stream_cdn_files(manifest, max_concurrent=8)
```

2. Inspect manifest URLs: all must be `https://huggingface.co/datasets/.../resolve/main/...`.
3. In a Lightning training script, replace `load_dataset` with `stream_cdn_files` (or equivalent) and confirm no HF API calls (no `/api/` and no 429s).
4. Call `get_or_create_studio("vanguard-train")` twice; second call must return the same running studio (no new studio created).
5. Stop the studio manually, call `get_or_create_studio` again; it should restart and return a running studio within ~60s.
