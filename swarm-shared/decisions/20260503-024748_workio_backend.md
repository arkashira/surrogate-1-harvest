# workio / backend

## Final Synthesis — Correct, Actionable Implementation

Below is the single, consolidated solution that keeps the strongest, most correct parts of both proposals and removes contradictions. Priority: **correctness under concurrency**, **idempotency against LINE retries**, and **concrete, copy-pasteable changes**.

---

### 1. Diagnosis (consolidated)

- **LINE retries on non-fast 2xx / timeouts** → duplicate webhook deliveries.
- **Handler is not idempotent** → duplicate punches for same logical event.
- **Race condition on “last punch” check** (read-then-insert) → concurrent clock-ins create two active punches.
- **No DB-level uniqueness guard** → duplicates can be written even if app logic is fixed later.
- **No transactional atomic transition** → lost/phantom updates under load.
- **Missing short-term dedup layer** → bursts/retries within minutes are not collapsed.

---

### 2. DB Changes (schema.sql)

Add two complementary protections:

1. **Idempotency table** for LINE webhook events (fast rejection of duplicates).
2. **Unique constraint** on punches to enforce one logical punch per user per type per day (prevents silent corruption).

```sql
-- /opt/axentx/workio/server/src/db/schema.sql

-- Idempotency for LINE webhook events (prevents duplicate processing)
CREATE TABLE IF NOT EXISTS line_webhook_events (
  id              BIGSERIC PRIMARY KEY,
  line_event_id   TEXT NOT NULL,
  tenant_id       BIGINT NOT NULL,
  user_id         BIGINT NOT NULL,
  event_type      TEXT NOT NULL,
  payload         JSONB NOT NULL,
  created_at      TIMESTAMPTZ DEFAULT NOW()
);
CREATE UNIQUE INDEX IF NOT EXISTS uq_line_webhook_event_id
  ON line_webhook_events (line_event_id);
CREATE INDEX IF NOT EXISTS idx_line_webhook_events_created
  ON line_webhook_events (created_at);

-- Punches table uniqueness guard (DB-level last line of defense)
-- If table already exists, add constraint safely
DO $$ BEGIN
  IF NOT EXISTS (
    SELECT 1 FROM pg_constraint
    WHERE conname = 'uniq_user_type_date'
  ) THEN
    ALTER TABLE punches
      ADD CONSTRAINT uniq_user_type_date
      UNIQUE (user_id, type, DATE(timestamp));
  END IF;
END $$;
```

### 3. Service Implementation (transactional + idempotent)

Use **advisory lock per user** to serialize punch transitions and **idempotency table** to short-circuit duplicates immediately.

```ts
// /opt/axentx/workio/server/src/services/punchService.ts
import { pool } from '../db';
import { PunchType, PunchRow } from '../types';

const USER_PUNCH_LOCK_BASE = 10000; // advisory lock space for punches

export async function upsertPunch(
  userId: number,
  type: PunchType,
  timestamp: Date,
  lineEventId?: string,
  tenantId?: number
): Promise<PunchRow> {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // 1) Fast idempotency check for LINE events
    if (lineEventId) {
      const exists = await client.query(
        `SELECT 1 FROM line_webhook_events WHERE line_event_id = $1`,
        [lineEventId]
      );
      if (exists.rows.length > 0) {
        // Already processed: return most recent punch for this user/day/type
        const existing = await client.query(
          `SELECT * FROM punches
           WHERE user_id = $1 AND type = $2 AND DATE(timestamp) = DATE($3)
           ORDER BY timestamp DESC LIMIT 1`,
          [userId, type, timestamp]
        );
        await client.query('ROLLBACK');
        if (existing.rows.length > 0) return existing.rows[0];
        // If punch missing but event recorded, continue to create consistent state
      }
    }

    // 2) Advisory lock to serialize per-user punch transitions
    const lockId = USER_PUNCH_LOCK_BASE + userId;
    await client.query('SELECT pg_advisory_xact_lock($1)', [lockId]);

    // 3) Record LINE event idempotency (if provided)
    if (lineEventId && tenantId !== undefined) {
      await client.query(
        `INSERT INTO line_webhook_events (line_event_id, tenant_id, user_id, event_type, payload)
         VALUES ($1, $2, $3, $4, $5) ON CONFLICT DO NOTHING`,
        [lineEventId, tenantId, userId, type, JSON.stringify({ timestamp })]
      );
    }

    // 4) Determine next punch type (alternating rule)
    const last = await client.query(
      `SELECT * FROM punches
       WHERE user_id = $1
       ORDER BY timestamp DESC LIMIT 1`,
      [userId]
    );
    const lastType = last.rows[0]?.type;
    const nextType = lastType === 'clock_in' ? 'clock_out' : 'clock_in';
    const effectiveType = type || nextType;

    // 5) Atomic insert (unique constraint protects against races)
    const result = await client.query(
      `INSERT INTO punches (user_id, type, timestamp, created_at)
       VALUES ($1, $2, $3, NOW())
       RETURNING *`,
      [userId, effectiveType, timestamp]
    );

    await client.query('COMMIT');
    return result.rows[0];
  } catch (err: any) {
    await client.query('ROLLBACK');

    // Unique violation: return existing punch (safe fallback)
    if (err.code === '23505') {
      const existing = await pool.query(
        `SELECT * FROM punches
         WHERE user_id = $1 AND type = $2 AND DATE(timestamp) = DATE($3)
         ORDER BY timestamp DESC LIMIT 1`,
        [userId, type, timestamp]
      );
      if (existing.rows.length > 0) return existing.rows[0];
    }

    throw err;
  } finally {
    client.release();
  }
}
```

### 4. Webhook Handler (quick 200 to stop LINE retries)

Return 200 immediately after idempotency/insert to minimize LINE retry window.

```ts
// /opt/axentx/workio/server/src/routes/line/webhook.ts
import { Router, Request, Response } from 'express';
import { upsertPunch } from '../../services/punchService';

const router = Router();

router.post('/webhook/line', async (req: Request, res: Response) => {
  try {
    const { userId, type, timestamp, line_event_id, tenant_id } = req.body;

    // Fire-and-forget style safe handling: always respond quickly
    upsertPunch(userId, type, new Date(timestamp), line_event_id, tenant_id)
      .catch((err) => {
        console.error('Punch upsert failed (async):', err);
      });

    return res.status(200).json({ ok: true });
  } catch (err) {
    console.error(err);
    return res.status(500).json({ ok: false, error: String(err) });
  }
});

export default router;
```
