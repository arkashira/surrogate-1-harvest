### 1) Deterministic Manifest Build (15 min)  
- **Action:** Add `scripts/build_manifest.py`.  
- **Behavior:**  
  - Use `HfApi.list_repo_tree(repo_id, path=date_folder, recursive=False)` to list a single date folder without recursion (avoids large payloads/429 bursts).  
  - Emit `manifest.json` with sorted parquet entries and full CDN URLs.  
- **Why deterministic:** Sorting + explicit CDN paths removes API variability; `recursive=False` keeps the call lightweight and repeatable.  

```python
#!/usr/bin/env python3
"""
Build deterministic CDN manifest for a date folder.
Usage:
  python build_manifest.py --repo surrogate-1-mirror --date 2026-04-29 --out manifest.json
"""
import argparse
import json
from pathlib import Path

from huggingface_hub import HfApi

HF_API = HfApi()

def build_manifest(repo_id: str, date_folder: str, out_path: Path):
    tree = HF_API.list_repo_tree(repo_id=repo_id, path=date_folder, recursive=False)
    files = sorted(entry.path for entry in tree if entry.path.endswith(".parquet"))

    entries = [
        {"path": f, "cdn_url": f"https://huggingface.co/datasets/{repo_id}/resolve/main/{f}"}
        for f in files
    ]

    manifest = {"repo_id": repo_id, "date_folder": date_folder, "entries": entries}
    out_path.parent.mkdir(parents=True, exist_ok=True)
    with open(out_path, "w") as fp:
        json.dump(manifest, fp, indent=2)
    print(f"Wrote {len(entries)} files to {out_path}")

if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--repo", default="surrogate-1-mirror")
    parser.add_argument("--date", required=True, help="e.g. 2026-04-29")
    parser.add_argument("--out", default="manifest.json")
    args = parser.parse_args()
    build_manifest(args.repo, args.date, Path(args.out))
```

### 2) CDN-Only IterableDataset (30 min)  
- **Action:** Add `surrogate/train/data/cdn_dataset.py`.  
- **Behavior:**  
  - `IterableDataset` that streams parquet via CDN URLs (no HF API calls).  
  - Project to `{prompt, response}` at parse time; skip non-text rows.  
  - Use `requests` with timeout; no auth header to bypass API rate limits.  
- **Why this over `load_dataset(streaming=True)`:** Avoids HF Data Server 429s entirely and handles heterogeneous schemas by explicit projection.  

```python
import io
import json
from typing import Dict, Iterator, List, Optional

import pyarrow.parquet as pq
import requests
from torch.utils.data import IterableDataset

class CDNParquetDataset(IterableDataset):
    """
    Zero HF-API dataset: reads parquet files directly from CDN.
    Manifest format:
      {"entries": [{"cdn_url": "...", "path": "..."}, ...]}
    """

    def __init__(self, manifest_path: str, columns: Optional[List[str]] = None):
        with open(manifest_path) as f:
            manifest = json.load(f)
        self.entries = manifest["entries"]
        self.columns = columns or ["prompt", "response"]

    def _stream_parquet(self, url: str):
        resp = requests.get(url, timeout=60)
        resp.raise_for_status()
        return pq.read_table(io.BytesIO(resp.content), columns=self.columns)

    def __iter__(self) -> Iterator[Dict]:
        for entry in self.entries:
            table = self._stream_parquet(entry["cdn_url"])
            if "prompt" in table.column_names and "response" in table.column_names:
                df = table.select(["prompt", "response"]).to_pydict()
                for prompt, response in zip(df["prompt"], df["response"]):
                    if isinstance(prompt, str) and isinstance(response, str):
                        yield {"prompt": prompt, "response": response}
```

### 3) Lightning Studio Lifecycle Resilience (30 min)  
- **Action:** Add `scripts/lightning_launcher.py`.  
- **Behavior:**  
  - Reuse a running Studio; restart if stopped (idle-timeout resilience).  
  - Run training with manifest baked in; use FSDP on L40S for Surrogate-1 scale.  
- **Why restart-if-stopped:** Prevents silent failures from idle kills and keeps iteration tight.  

```python
#!/usr/bin/env python3
"""
Reuse or start Lightning Studio, restart if stopped, run training.
"""
import argparse
import sys
import time

from lightning.pytorch import Trainer
from lightning.pytorch.loggers import WandbLogger
from lightning.pytorch.strategies import FSDPStrategy

try:
    from lightning.pytorch.studio import Studio, Teamspace
except ImportError:
    print("lightning[studio] required")
    sys.exit(1)

def get_or_create_studio(name: str, machine: str = "lightningai/L40S"):
    teamspace = Teamspace()
    for s in teamspace.studios:
        if s.name == name and s.status == "Running":
            print(f"Reusing running studio: {name}")
            return s
    print(f"Creating studio: {name}")
    return Studio(
        name=name,
        machine=machine,
        cloud="lightning-public-prod",
        create_ok=True,
    )

def run_training(studio, manifest_path, epochs=1, output_dir="outputs"):
    if studio.status != "Running":
        print(f"Studio stopped ({studio.status}), restarting...")
        studio.start()
        for _ in range(30):
            if studio.status == "Running":
                break
            time.sleep(10)
        else:
            raise RuntimeError("Studio failed to start")

    cmd = [
        sys.executable,
        "surrogate/train/train.py",
        "--manifest",
        manifest_path,
        "--epochs",
        str(epochs),
        "--accelerator",
        "gpu",
        "--devices",
        "1",
        "--strategy",
        "fsdp",
        "--default_root_dir",
        output_dir,
    ]
    studio.run(cmd)

if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--studio-name", default="surrogate-1-train")
    parser.add_argument("--manifest", default="manifest.json")
    parser.add_argument("--epochs", type=int, default=1)
    parser.add_argument("--output-dir", default="outputs")
    args = parser.parse_args()

    studio = get_or_create_studio(args.studio_name)
    run_training(studio, args.manifest, args.epochs, args.output_dir)
```
