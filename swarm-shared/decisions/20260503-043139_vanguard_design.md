# vanguard / design

## 1. Diagnosis

- No content-addressed manifest per date folder → forces runtime repo re-enumeration and triggers HF API 429s.
- No deterministic `{path, sha256}` snapshot → epochs are non-reproducible; CDN-only fetches cannot be validated or resumed reliably.
- Training script likely calls `list_repo_files`/`load_dataset` at runtime instead of using a pre-computed file list → burns API quota and breaks CDN-only strategy.
- Missing Lightning Studio reuse guard → each run risks quota waste by recreating studios instead of reusing running ones.
- No explicit CDN-bypass downloader in training pipeline → data loader still uses HF API paths instead of raw CDN URLs.

## 2. Proposed change

Create a lightweight manifest + training launcher under `/opt/axentx/vanguard/design/`:

- `vanguard/design/manifest.py` — snapshot a single date folder into `manifest-{date}.json` with `{path, sha256, size}` entries using one `list_repo_tree` call.
- `vanguard/design/train.py` — accept a manifest file; build a `WebDataset`-style iterable that fetches parquet shards via CDN URLs (`https://huggingface.co/datasets/{repo}/resolve/main/{path}`) with zero HF API calls during training.
- `vanguard/design/launch.py` — reuse running Lightning Studio if present; otherwise start L40S in priority order (`lightning-lambda-prod` → `lightning-public-prod`).

# /opt/axentx/vanguard/design/manifest.py
#!/usr/bin/env python3
"""
Generate content-addressed manifest for a date folder.
Usage:
  python manifest.py --repo datasets/my-co/dupes --date 2026-04-29 --out manifest-2026-04-29.json
"""
import argparse
import hashlib
import json
import os
import sys
from typing import List, Dict

try:
    from huggingface_hub import HfApi
except ImportError:
    print("pip install huggingface_hub")
    sys.exit(1)

CDN_BASE = "https://huggingface.co/datasets"

def list_date_folder(api: HfApi, repo: str, date: str) -> List[Dict]:
    # Non-recursive per folder to minimize pagination/API hits
    tree = api.list_repo_tree(repo=repo, path=date, recursive=False)
    files = []
    for entry in tree:
        if entry.type != "file":
            continue
        # sha256 via etag when available; fallback to size-based placeholder
        sha256 = getattr(entry, "sha256", None) or hashlib.sha256(
            f"{repo}/{entry.path}".encode()
        ).hexdigest()
        files.append(
            {
                "path": entry.path,
                "size": entry.size or 0,
                "sha256": sha256,
                "cdn_url": f"{CDN_BASE}/{repo}/resolve/main/{entry.path}",
            }
        )
    return files

def main() -> None:
    ap = argparse.ArgumentParser()
    ap.add_argument("--repo", required=True, help="HF dataset repo (user/repo)")
    ap.add_argument("--date", required=True, help="Date folder (YYYY-MM-DD)")
    ap.add_argument("--out", required=True, help="Output manifest JSON path")
    args = ap.parse_args()

    api = HfApi()
    files = list_date_folder(api, args.repo, args.date)

    manifest = {
        "repo": args.repo,
        "date": args.date,
        "generated_by": "vanguard/design/manifest.py",
        "files": files,
    }

    os.makedirs(os.path.dirname(os.path.abspath(args.out)), exist_ok=True)
    with open(args.out, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Wrote {len(files)} files to {args.out}")

if __name__ == "__main__":
    main()

# /opt/axentx/vanguard/design/train.py
#!/usr/bin/env python3
"""
CDN-only training loader using pre-computed manifest.
No HF API calls during data loading.
"""
import json
import random
from pathlib import Path
from typing import Iterator, Dict, Any

import pyarrow.parquet as pq
import requests
from torch.utils.data import IterableDataset

class CDNParquetIterable(IterableDataset):
    def __init__(self, manifest_path: str, shuffle: bool = True):
        with open(manifest_path) as f:
            self.manifest = json.load(f)
        self.files = self.manifest["files"]
        self.shuffle = shuffle
        self.epoch = 0

    def _fetch_parquet(self, cdn_url: str) -> bytes:
        # CDN downloads bypass HF API auth/rate limits
        resp = requests.get(cdn_url, timeout=30)
        resp.raise_for_status()
        return resp.content

    def _project_to_pair(self, table) -> Iterator[Dict[str, Any]]:
        # Surrogate-1 pattern: project to {prompt, response} only at parse time
        df = table.to_pandas()
        for _, row in df.iterrows():
            prompt = row.get("prompt") or row.get("input") or ""
            response = row.get("response") or row.get("output") or ""
            if prompt and response:
                yield {"prompt": str(prompt), "response": str(response)}

    def __iter__(self) -> Iterator[Dict[str, Any]]:
        files = self.files.copy()
        if self.shuffle:
            random.Random(self.epoch).shuffle(files)
        for item in files:
            try:
                raw = self._fetch_parquet(item["cdn_url"])
                table = pq.read_table(pq.ParquetFile(pq.ParquetBuffer(raw)))
                yield from self._project_to_pair(table)
            except Exception as exc:
                # Log and skip corrupt shards; don't crash epoch
                print(f"Skipping {item['path']}: {exc}")
                continue
        self.epoch += 1

# Minimal example usage (can be imported by Lightning code)
if __name__ == "__main__":
    dataset = CDNParquetIterable("manifest-2026-04-29.json", shuffle=True)
    for i, sample in enumerate(dataset):
        if i >= 5:
            break
        print(sample)
