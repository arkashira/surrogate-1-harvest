## Final Consolidated Implementation (CDN-only Surrogate-1 loader)

**Core idea:** Replace fragile `load_dataset(streaming=True)` with a manifest-driven, CDN-only data path that eliminates HF API 429s during training and avoids `pyarrow.CastError` from heterogeneous schemas.

### 2. Mac orchestration — list once, avoid API during training
```python
# /opt/axentx/airship/surrogate/training/scripts/build_file_manifest.py
#!/usr/bin/env python3
"""
Run on Mac after rate-limit window clears.
Produces file_list.json for CDN-only training.
"""
import json, os, hashlib
from huggingface_hub import list_repo_tree

REPO = "datasets/surrogate-1"
DATE_FOLDER = "2026-05-03"            # parameterized per run
OUTPUT = "../../file_list.json"
CDN_BASE = f"https://huggingface.co/datasets/{REPO}/resolve/main"

def main():
    # recursive=False keeps pagination minimal; CDN downloads are unlimited
    tree = list_repo_tree(REPO, path=DATE_FOLDER, recursive=False)
    files = sorted(f.rfilename for f in tree if f.rfilename.endswith(".jsonl"))
    manifest = {
        "repo": REPO,
        "date": DATE_FOLDER,
        "files": files,
        "cdn_base": CDN_BASE
    }
    os.makedirs(os.path.dirname(OUTPUT), exist_ok=True)
    with open(OUTPUT, "w") as f:
        json.dump(manifest, f, indent=2)
    print(f"Wrote {len(files)} files to {OUTPUT}")

if __name__ == "__main__":
    main()
```

### 3. Utility: CDN streaming + schema projection (avoids pyarrow errors)
```python
# /opt/axentx/airship/surrogate/training/surrogate/data/cdn_stream.py
import json, requests, io
from typing import Iterator, Tuple

def project_to_prompt_response(raw_bytes: bytes) -> Tuple[str, str]:
    """
    Parse one JSONL line and project to (prompt, response).
    Ignores extra fields to avoid pyarrow schema issues.
    """
    line = json.loads(raw_bytes.decode("utf-8").strip())
    prompt = line.get("prompt") or line.get("input") or ""
    response = line.get("response") or line.get("output") or ""
    return prompt, response

def stream_cdn_lines(manifest_path: str) -> Iterator[Tuple[str, str]]:
    with open(manifest_path) as f:
        manifest = json.load(f)

    base = manifest["cdn_base"]
    for fname in manifest["files"]:
        url = f"{base}/{fname}"
        # CDN download: no Authorization header => bypasses API rate limits
        with requests.get(url, stream=True, timeout=60) as r:
            r.raise_for_status()
            buffer = io.BytesIO()
            for chunk in r.iter_content(chunk_size=8192):
                buffer.write(chunk)
            buffer.seek(0)
            for raw_line in buffer:
                if raw_line.strip():
                    yield project_to_prompt_response(raw_line)
```

"""
Lightning Studio entrypoint.
Uses CDN-only fetches; zero HF API calls during data load.

import os
import lightning as L
from torch.utils.data import IterableDataset, DataLoader
from surrogate.data.cdn_stream import stream_cdn_lines

MANIFEST = os.getenv("MANIFEST_PATH", "../../file_list.json")

class CDNPairDataset(IterableDataset):
    def __init__(self, manifest_path: str):
        super().__init__()
        self.manifest_path = manifest_path

### 7. Commit-cap sharding (when pushing enriched parquet)
```python
# When dataset-mirror writes enriched/ files:
#   repo = f"datasets/surrogate-1-shard-{deterministic_repo_index(slug)}"
#   path = f"batches/mirror-merged/{date}/{slug}.parquet"
#   (no source/ts cols; attribution in filename pattern)

def deterministic_repo_index(slug: str, n_siblings: int = 5) -> int:
    """Map slug -> sibling repo index to spread HF commit cap."""
    import hashlib
    digest = hashlib.sha256(slug.encode()).digest()
    return int.from_bytes(digest[:2], "little") % n_siblings
```

### Why this ships in <2h and unblocks Surrogate-1 training
- **Eliminates HF API 429 during training** by switching to CDN-only fetches.
- **Avoids `pyarrow.CastError`** by projecting heterogeneous files to `{prompt, response}` at parse time.
- **Minimal API usage** — single `list_repo_tree` on Mac; training is CDN-only.
- **Reuses Lightning Studio pattern** — saves quota and avoids idle-stop surprises.
