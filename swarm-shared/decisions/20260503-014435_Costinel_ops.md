**Final Implementation — Top-hub Signal Panel (Costinel Dashboard)**

**Scope & Constraints**  
- Frontend-only, read-only panel.  
- Surface the most-connected hub and its actionable proposals.  
- Resilient to missing backend: graceful fallback UI + local sample + localStorage cache.  
- Ship in <2h (no backend changes, no DB migrations, no new infra).  
- Prefer CDN-only fetches (HuggingFace `/resolve/main/...`) to bypass HF gateway limits; minimal API surface.

---

### 1) File changes (minimal, high-value)

- `src/components/TopHubSignalPanel.tsx` — new React + Tailwind component.  
- `src/lib/knowledgeGraph.ts` — lightweight client with CDN-bypass strategy + localStorage cache.  
- `src/pages/Dashboard.tsx` (or equivalent) — mount panel near cost summary/sidebar.  
- `public/mock/top-hub-sample.json` — local fallback payload (committed).

---

### 2) Data strategy (CDN-first, resilient)

- Primary: fetch from CDN path (e.g. `/resolve/main/datasets/.../top-hub-latest.json` or project-specific CDN route).  
- Fallback 1: lightweight local API (`/api/knowledge-graph/top-hub`) if available, short timeout (4–5s).  
- Fallback 2: local sample file (`/mock/top-hub-sample.json`).  
- Cache: localStorage with TTL (e.g. 10–15 minutes) to reduce repeated CDN fetches and tolerate transient failures.  
- Minimal calls: one request per eligible source (try CDN first, then API, then local).

### 3) Knowledge-graph client (CDN-bypass + cache)

```ts
// src/lib/knowledgeGraph.ts
import type { HubInsights } from "../components/TopHubSignalPanel";

const SAMPLE_URL = "/mock/top-hub-sample.json";
const CDN_URL = "/resolve/main/datasets/costinel/top-hub-latest.json"; // adjust path per project
const API_URL = "/api/knowledge-graph/top-hub";
const CACHE_KEY = "costinel:top-hub:insights";
const CACHE_TTL_MS = 10 * 60 * 1000; // 10 minutes

function isHubInsights(value: unknown): value is HubInsights {
  if (!value || typeof value !== "object") return false;
  const v = value as Record<string, unknown>;

  if (!v.hub || typeof v.hub !== "object") return false;
  const hub = v.hub as Record<string, unknown>;
  if (typeof hub.name !== "string" || typeof hub.connections !== "number") return false;

  if (!Array.isArray(v.proposals)) return false;
  for (const p of v.proposals) {
    if (!p || typeof p !== "object") return false;
    if (typeof p.id !== "string" || typeof p.title !== "string" || typeof p.summary !== "string") return false;
    if (!["high", "medium", "low"].includes(p.impact as string)) return false;
  }

  if (typeof v.generatedAt !== "string") return false;
  return true;
}

function getCached(): HubInsights | null {
  try {
    const raw = localStorage.getItem(CACHE_KEY);
    if (!raw) return null;
    const parsed = JSON.parse(raw) as { data: HubInsights; ts: number };
    if (!parsed || typeof parsed.ts !== "number" || !isHubInsights(parsed.data)) return null;
    if (Date.now() - parsed.ts > CACHE_TTL_MS) return null;
    return parsed.data;
  } catch {
    return null;
  }
}

function setCached(data: HubInsights) {
  try {
    localStorage.setItem(CACHE_KEY, JSON.stringify({ data, ts: Date.now() }));
  } catch {
    // ignore storage failures
  }
}

async function tryFetch(url: string, timeoutMs = 4500): Promise<Response | null> {
  try {
    const controller = new AbortController();
    const id = setTimeout(() => controller.abort(), timeoutMs);
    const res = await fetch(url, { headers: { Accept: "application/json" }, signal: controller.signal });
    clearTimeout(id);
    return res.ok ? res : null;
  } catch {
    return null;
  }
}

export async function getTopHubInsights(): Promise<HubInsights> {
  // 1) cached
  const cached = getCached();
  if (cached) return cached;

  // 2) CDN first (preferred)
  const cdnRes = await tryFetch(CDN_URL, 5000);
  if (cdnRes) {
    const json = await cdnRes.json();
    if (isHubInsights(json)) {
      setCached(json);
      return json;
    }
  }

  // 3) lightweight API (if exists)
  const apiRes = await tryFetch(API_URL, 4500);
  if (apiRes) {
    const json = await apiRes.json();
    if (isHubInsights(json)) {
      setCached(json);
      return json;
    }
  }

  // 4) local sample fallback
  const sampleRes = await tryFetch(SAMPLE_URL, 3000);
  if (sampleRes) {
    const json = await sampleRes.json();
    if (isHubInsights(json)) {
      // do not cache sample to avoid masking live updates when CDN/API returns
      return json;
    }
  }

  throw new Error("Unable to load hub insights from any source");
}
```

import { useEffect, useState } from "react";
import { getTopHubInsights } from "../lib/knowledgeGraph";

interface Proposal {
  id: string;
  title: string;
  summary: string;
  impact: "high" | "medium" | "low";
  action?: string;
  href?: string;
}

interface HubInsights {
  hub: {
    name: string;
    description?: string;
    connections: number;
  };
  proposals: Proposal[];
  generatedAt: string;
}

const IMPACT_COLORS = {
  high: "bg-red-50 border-red-200 text-red-800",
  medium: "bg-amber-50 border-amber-200 text-amber-800",
  low: "bg-green-50 border-green-200 text-green-800",
};

  const [loading, setLoading] = useState(true);

            <div key={i} className="h-16 animate-pulse rounded bg-gray-100" />
          ))}
        </div>
      </section>
    );
