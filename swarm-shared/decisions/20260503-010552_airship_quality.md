### 1. `scripts/build_file_list.py` (run from Mac)
```python
#!/usr/bin/env python3
"""
Single API call to list files in a date folder.
Save to file_list.json for CDN-only training.
"""
import json
import os
import sys
from datetime import datetime, timedelta

from huggingface_hub import HfApi

REPO_ID = os.getenv("HF_DATASET_REPO", "axentx/surrogate-mirror")
DATE_FOLDER = os.getenv("DATE_FOLDER", (datetime.utcnow() - timedelta(days=1)).strftime("%Y-%m-%d"))
OUTPUT = "file_list.json"

api = HfApi()
# non-recursive per folder to avoid pagination explosion
tree = api.list_repo_tree(repo_id=REPO_ID, path=DATE_FOLDER, recursive=False)

files = [
    f"{DATE_FOLDER}/{node.path.split('/')[-1]}"
    for node in tree
    if node.path.lower().endswith((".parquet", ".jsonl"))
]

with open(OUTPUT, "w") as f:
    json.dump({"repo_id": REPO_ID, "date": DATE_FOLDER, "files": files}, f, indent=2)

print(f"Wrote {len(files)} files to {OUTPUT}")
```

### 2. `surrogate/train.py` (Lightning training script)
```python
#!/usr/bin/env python3
import json
import os
import time
from pathlib import Path
from typing import Dict, Any

import pyarrow as pa
import pyarrow.parquet as pq
import requests
from lightning.pytorch import Trainer
from lightning.pytorch.loggers import CSVLogger

HF_CDN_TEMPLATE = "https://huggingface.co/datasets/{repo_id}/resolve/main/{path}"
SIBLING_REPOS = [f"axentx/surrogate-mirror-{i}" for i in range(5)]

def pick_repo(slug: str) -> str:
    """Deterministic repo selection to spread HF commit cap."""
    idx = hash(slug) % len(SIBLING_REPOS)
    return SIBLING_REPOS[idx]

def download_via_cdn(repo_id: str, cdn_path: str, local_path: Path) -> Path:
    url = HF_CDN_TEMPLATE.format(repo_id=repo_id, path=cdn_path)
    local_path.parent.mkdir(parents=True, exist_ok=True)
    resp = requests.get(url, timeout=60)
    resp.raise_for_status()
    local_path.write_bytes(resp.content)
    return local_path

def project_to_pair(local_file: Path) -> Dict[str, str]:
    """Strict projection to {prompt,response}; ignore mixed schemas."""
    try:
        pf = pq.ParquetFile(local_file)
        schema_names = {c.name for c in pf.schema}
        required = {"prompt", "response"}
        if not required.issubset(schema_names):
            # fallback: try common aliases
            col_map = {}
            for c in schema_names:
                low = c.lower()
                if "prompt" in low or "input" in low or "question" in low:
                    col_map["prompt"] = c
                if "response" in low or "output" in low or "answer" in low:
                    col_map["response"] = c
            if set(col_map.keys()) != required:
                raise ValueError(f"Missing required cols in {local_file}: {schema_names}")
        else:
            col_map = {"prompt": "prompt", "response": "response"}

        table = pq.read_table(local_file, columns=[col_map["prompt"], col_map["response"]])
        df = table.to_pandas()
        # clean nulls
        df = df.dropna(subset=[col_map["prompt"], col_map["response"]])
        return {"prompt": df.iloc[0][col_map["prompt"]], "response": df.iloc[0][col_map["response"]]}
    except Exception as exc:
        raise ValueError(f"Failed to project {local_file}: {exc}") from exc

def load_file_list(path: str = "file_list.json"):
    with open(path) as f:
        return json.load(f)

def train():
    meta = load_file_list()
    repo_id = meta["repo_id"]
    files = meta["files"]

    pairs = []
    for cdn_path in files:
        local = Path("cache") / Path(cdn_path).name
        if not local.exists():
            download_via_cdn(repo_id, cdn_path, local)
        try:
            pairs.append(project_to_pair(local))
        except Exception as exc:
            print(f"Skipping {cdn_path}: {exc}")
            continue

    print(f"Loaded {len(pairs)} valid pairs")
    # TODO: plug into SurrogateDataModule and Trainer
    # For now, minimal Lightning run to validate pipeline
    logger = CSVLogger("logs", name="surrogate-run")
    trainer = Trainer(
        max_epochs=1,
        logger=logger,
        accelerator="auto",
        devices=1,
        enable_checkpointing=False,
    )
    # dummy fit to verify environment
    # trainer.fit(model, datamodule)

if __name__ == "__main__":
    # HF 429 backoff
    for attempt in range(3):
        try:
            train()
            break
        except Exception as exc:
            if "429" in str(exc):
                wait = 360 * (attempt + 1)
                print(f"HF 429, waiting {wait}s")
                time.sleep(wait)
            else:
                raise
```
