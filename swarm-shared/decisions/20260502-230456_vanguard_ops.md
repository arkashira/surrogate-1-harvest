# vanguard / ops

## 1. Diagnosis

- Hash-router navigations do not restore scroll position on back/forward → disorienting jumps to top instead of restoring prior viewport.
- Dataset list view state (page, pageSize, sort, dir, filters, selection) is not encoded in URL → reloads and shared links lose context.
- Dataset detail view loses filter/pagination state when navigating back to list → forces user to re-find prior position.
- No loading/error UI during async dataset fetches → blank screens on slow networks.
- Missing scroll-to-top on new hash navigations (non-back/forward) → new pages can open scrolled to old position.

## 2. Proposed change

File: `/opt/axentx/vanguard/src/router.js` (or equivalent main router entry)  
Scope: add hash-scroll manager + URL-state serializer for dataset list (page, pageSize, sort, dir, filters) and detail (id). Target: <2h, minimal diff.

// /opt/axentx/vanguard/src/router.js
(function () {
  'use strict';

  // ---- Config ----
  const STATE_KEYS = {
    list: ['page', 'pageSize', 'sort', 'dir', 'filters', 'selection'],
    detail: ['id', 'listPage', 'listPageSize', 'listSort', 'listDir', 'listFilters']
  };

  // ---- Utilities ----
  function parseHash() {
    const hash = location.hash.replace(/^#/, '');
    if (!hash) return { route: 'list', params: {} };
    const [route, ...rest] = hash.split('/');
    const params = {};
    for (const part of rest) {
      const [k, v] = part.split('=');
      if (k && v !== undefined) params[decodeURIComponent(k)] = decodeURIComponent(v);
    }
    return { route, params };
  }

  function buildHash(route, params = {}) {
    const parts = Object.entries(params)
      .filter(([, v]) => v != null && v !== '')
      .map(([k, v]) => `${encodeURIComponent(k)}=${encodeURIComponent(v)}`);
    return `#${route}${parts.length ? '/' + parts.join('/') : ''}`;
  }

  function saveListStateToUrl(state) {
    const params = {};
    STATE_KEYS.list.forEach((k) => {
      if (state[k] !== undefined) params[k] = state[k];
    });
    // compact: store filters JSON if present
    if (state.filters && typeof state.filters === 'object') {
      try { params.filters = btoa(JSON.stringify(state.filters)); } catch (e) { /* ignore */ }
    }
    if (state.selection && Array.isArray(state.selection)) {
      try { params.selection = btoa(JSON.stringify(state.selection)); } catch (e) { /* ignore */ }
    }
    history.replaceState(null, '', buildHash('list', params));
  }

  function readListStateFromUrl() {
    const { route, params } = parseHash();
    if (route !== 'list') return null;
    const state = {};
    STATE_KEYS.list.forEach((k) => {
      if (params[k] !== undefined) {
        if (k === 'page' || k === 'pageSize') {
          const n = parseInt(params[k], 10);
          if (!Number.isNaN(n)) state[k] = n;
        } else if (k === 'filters' || k === 'selection') {
          try { state[k] = JSON.parse(atob(params[k])); } catch (e) { /* ignore */ }
        } else {
          state[k] = params[k];
        }
      }
    });
    return state;
  }

  function navigateToList(state, replace = false) {
    saveListStateToUrl(state);
    if (!replace) renderList(state);
  }

  function navigateToDetail(id, listState = {}) {
    const params = { id };
    // preserve list state for back navigation
    STATE_KEYS.list.forEach((k) => {
      if (listState[k] !== undefined) params[`list${k.charAt(0).toUpperCase() + k.slice(1)}`] = listState[k];
    });
    history.pushState(null, '', buildHash('detail', params));
    renderDetail(id);
  }

  // ---- Scroll manager ----
  const scrollPositions = new Map(); // route+key -> y

  function saveScrollPosition(key, y) {
    scrollPositions.set(key, y);
  }

  function restoreScrollPosition(key) {
    const y = scrollPositions.get(key) || 0;
    // Use requestAnimationFrame to allow DOM to settle
    requestAnimationFrame(() => {
      window.scrollTo(0, y);
    });
  }

  function scrollToTop() {
    window.scrollTo(0, 0);
  }

    // fetch datasets using state, then render

      restoreScrollPosition(key);
    }).catch((err) => {
      showError(err);
      showLoading(false);
    });
  }

  function renderDetail(id) {
    showLoading(true);
    fetchDataset(id).then((dataset) => {
      renderDatasetDetailUI(dataset);
      showLoading(false);
      scrollToTop();
    }).catch((err) => {
      showError(err);
      showLoading(false);
    });
  }

      el.innerHTML = '<div>Loading...</div>';
      document.body.appendChild(el);
    }
    el.style.display = show ? 'flex' : 'none';
  }
