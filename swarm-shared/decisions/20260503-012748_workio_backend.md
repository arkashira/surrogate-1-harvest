# workio / backend

## 1. Diagnosis

- **No idempotency handling on LINE webhook ingestion** — LINE retries on 5xx/timeouts; duplicate `events[*].webhookEventId` can create double punches/leave/OT entries.
- **Race on punch state transitions** — `find-then-insert` for open punch allows concurrent requests to create multiple open punches for same tenant/employee/day.
- **No request deduplication layer** — repeated client retries (network flaps) can queue duplicate jobs in background processors.
- **Missing tenant context enforcement on webhook path** — webhook receives tenant-agnostic payload; tenant must be resolved from channel secret, but no strict scoping before DB writes.
- **No atomic punch-close with versioning** — closing an open punch is not protected against concurrent close attempts (lost update).

## 2. Proposed change

File: `workio/server/src/routes/webhook/line.ts` (create if absent) and `workio/server/src/services/punchService.ts`  
Scope: add idempotency middleware + atomic punch transition with optimistic locking; minimal surface, ~150 LoC total.

### 3.2 Idempotency middleware

```ts
// server/src/middleware/idempotency.ts
import { Request, Response, NextFunction } from 'express';
import { pool } from '../db';
import crypto from 'crypto';

export function idempotency(tenantIdPath = 'tenantId') {
  return async (req: Request, res: Response, next: NextFunction) => {
    const tenantId = req.body[tenantIdPath] || req.query[tenantIdPath];
    const key = req.header('Idempotency-Key') || req.body.idempotencyKey;
    if (!tenantId || !key) return next();

    const payloadHash = crypto
      .createHash('sha256')
      .update(JSON.stringify(req.body))
      .digest('hex');

    const client = await pool.connect();
    try {
      await client.query('BEGIN');
      const { rows } = await client.query(
        `SELECT payload_hash FROM idempotency_keys WHERE tenant_id = $1 AND key = $2 FOR UPDATE`,
        [tenantId, key]
      );

      if (rows.length > 0) {
        // Same payload: replay previous success; different payload: reject conflict
        if (rows[0].payload_hash === payloadHash) {
          await client.query('ROLLBACK');
          return res.status(200).json({ ok: true, replayed: true });
        }
        await client.query('ROLLBACK');
        return res.status(409).json({ error: 'Idempotency key collision' });
      }

      await client.query(
        `INSERT INTO idempotency_keys (tenant_id, key, payload_hash) VALUES ($1,$2,$3)`,
        [tenantId, key, payloadHash]
      );
      await client.query('COMMIT');
      res.locals.idempotencyClient = client;
      next();
    } catch (err) {
      await client.query('ROLLBACK');
      client.release();
      next(err);
    }
  };
}
```

### 3.3 Punch service with atomic state transition

```ts
// server/src/services/punchService.ts
import { pool } from '../db';

export async function createOrClosePunch(params: {
  tenantId: number;
  employeeId: number;
  type: 'IN' | 'OUT';
  timestamp: Date;
  latitude?: number;
  longitude?: number;
  lineEventId?: string;
}) {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // Deduplicate by LINE event id if provided
    if (params.lineEventId) {
      const { rows } = await client.query(
        `SELECT id FROM punches WHERE tenant_id = $1 AND employee_id = $2 AND line_event_id = $3`,
        [params.tenantId, params.employeeId, params.lineEventId]
      );
      if (rows.length > 0) {
        await client.query('ROLLBACK');
        return { id: rows[0].id, replayed: true };
      }
    }

    if (params.type === 'IN') {
      // Ensure no open punch for this employee today (tenant-scoped)
      const todayStart = new Date(params.timestamp);
      todayStart.setHours(0, 0, 0, 0);
      const todayEnd = new Date(params.timestamp);
      todayEnd.setHours(23, 59, 59, 999);

      const { rows } = await client.query(
        `SELECT id FROM punches
         WHERE tenant_id = $1 AND employee_id = $2 AND type = 'IN' AND timestamp BETWEEN $3 AND $4 AND "out" IS NULL
         LIMIT 1 FOR UPDATE`,
        [params.tenantId, params.employeeId, todayStart, todayEnd]
      );

      if (rows.length > 0) {
        // Already clocked in — treat as no-op or optionally auto-close previous?
        await client.query('ROLLBACK');
        return { id: rows[0].id, duplicate: true };
      }

      const insert = await client.query(
        `INSERT INTO punches (tenant_id, employee_id, type, timestamp, latitude, longitude, line_event_id)
         VALUES ($1,$2,$3,$4,$5,$6,$7) RETURNING id`,
        [params.tenantId, params.employeeId, params.type, params.timestamp, params.latitude, params.longitude, params.lineEventId]
      );
      await client.query('COMMIT');
      return { id: insert.rows[0].id };
    }

    // type === 'OUT'
    const { rows } = await client.query(
      `SELECT id FROM punches
       WHERE tenant_id = $1 AND employee_id = $2 AND type = 'IN' AND "out" IS NULL
       ORDER BY timestamp DESC LIMIT 1 FOR UPDATE`,
      [params.tenantId, params.employeeId]
    );

    if (rows.length === 0) {
      await client.query('ROLLBACK');
      return { error: 'no_open_punch' };
    }

    const update = await client.query(
      `UPDATE punches SET "out" = $1, latitude_out = $2, longitude_out = $3, line_event_id_out = $4
       WHERE id = $5 RETURNING id`,
      [params.timestamp, params.latitude, params.longitude, params.lineEventId, rows[0].id]
    );
    await client.query('COMMIT');
    return { id: update.rows[0].id };
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  } finally {
    client.release();
  }
}
```
