# Costinel / design

## Final Synthesis — CDN-First Top-Hub Signal Panel (<2h)

**Chosen approach:**  
A non-blocking, CDN-first Top-Hub Signal Panel that surfaces the most-connected hub (e.g., "MOC") with **zero HuggingFace API calls at runtime**. Data is produced by a Mac orchestration script, baked into the build, and served as static JSON from the CDN. The dashboard fetches a small static payload only; no Authorization headers, no rate-limit risk, and no parquet parsing in the browser.

#### 2.2 Mac orchestration script (CDN-first, no runtime HF API)

Path: `scripts/sync-top-hub.sh`

```bash
#!/usr/bin/env bash
set -euo pipefail

# Sync top-hub latest.json from CDN (bypass HF API rate limits).
# Run this on Mac as orchestration before build or on cron.
#
# Usage:
#   bash scripts/sync-top-hub.sh

REPO="axentx/costinel-knowledge"
REMOTE_CDN="https://huggingface.co/datasets/${REPO}/resolve/main/top-hub/latest.json"
LOCAL_OUT="public/data/top-hub/latest.json"
TMP_OUT=$(mktemp)

mkdir -p "$(dirname "${LOCAL_OUT}")"

echo "Fetching top-hub from CDN: ${REMOTE_CDN}"
curl --fail --silent --show-error --max-time 30 "${REMOTE_CDN}" -o "${TMP_OUT}"

# Basic schema sanity check (must contain hub)
if ! jq -e '.hub' "${TMP_OUT}" > /dev/null 2>&1; then
  echo "ERROR: Invalid top-hub payload (missing .hub)"
  rm -f "${TMP_OUT}"
  exit 1
fi

mv "${TMP_OUT}" "${LOCAL_OUT}"
echo "Updated ${LOCAL_OUT}"
```

Make executable:

```bash
chmod +x scripts/sync-top-hub.sh
```

Cron-friendly (ensure `SHELL=/bin/bash` in crontab):

```bash
# m h  dom mon dow   command
0 5 * * * cd /opt/axentx/Costinel && bash scripts/sync-top-hub.sh >> logs/sync-top-hub.log 2>&1
```

#### 2.3 React Top-Hub Signal Panel (lazy + suspense + error boundary)

Path: `src/components/TopHubSignalPanel.tsx`

```tsx
import React, { useEffect, useState } from "react";

interface TopHubPayload {
  hub: string;
  score: number;
  trend: "up" | "down" | "flat";
  delta7d: string;
  connections: number;
  lastUpdated: string;
  source?: string;
}

const TopHubSignalPanel: React.FC = () => {
  const [data, setData] = useState<TopHubPayload | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    let mounted = true;
    setLoading(true);

    // CDN-first fetch — no Authorization header required.
    fetch("/data/top-hub/latest.json", { cache: "no-cache" })
      .then((res) => {
        if (!res.ok) throw new Error(`HTTP ${res.status}`);
        return res.json();
      })
      .then((json) => {
        if (!mounted) return;
        setData(json);
        setError(null);
      })
      .catch((err) => {
        if (!mounted) return;
        setError(err.message || "Failed to load top-hub signal");
      })
      .finally(() => {
        if (!mounted) return;
        setLoading(false);
      });

    return () => {
      mounted = false;
    };
  }, []);

  if (loading) {
    return (
      <div className="top-hub-panel card card-skeleton" aria-busy="true">
        <div className="skeleton-line" style={{ width: "60%" }} />
        <div className="skeleton-line" style={{ width: "30%" }} />
      </div>
    );
  }

  if (error || !data) {
    return (
      <div className="top-hub-panel card card-error" role="status">
        <small>Signal unavailable</small>
      </div>
    );
  }

  const trendIcon =
    data.trend === "up" ? "↑" : data.trend === "down" ? "↓" : "→";

  return (
    <div className="top-hub-panel card" aria-label={`Top hub: ${data.hub}`}>
      <div className="top-hub-header">
        <span className="top-hub-badge">Top Hub</span>
        <span className="top-hub-name">{data.hub}</span>
      </div>

      <div className="top-hub-score">
        {data.score.toFixed(1)} <small>/100</small>
      </div>

      <div className="top-hub-meta">
        <span className="trend">
          {trendIcon} {data.delta7d} (7d)
        </span>
        <span className="connections">{data.connections.toLocaleString()} links</span>
      </div>

      <div className="top-hub-footer">
        <small className="muted">
          Updated {new Date(data.lastUpdated).toLocaleDateString()}
        </small>
      </div>
    </div>
  );
};

// Lazy wrapper for non-blocking insertion
export const LazyTopHubSignalPanel: React.FC = () => (
  <React.Suspense fallback={<div className="card card-placeholder">Loading signal…</div>}>
    <TopHubSignalPanel />
  </React.Suspense>
);

export default TopHubSignalPanel;
```
