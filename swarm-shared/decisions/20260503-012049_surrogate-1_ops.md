### bin/list_folder_files.py
```python
#!/usr/bin/env python3
"""
List files in a single date folder (non-recursive) for surrogate-1-training-pairs.
Outputs JSON with CDN URLs to avoid authenticated API calls during training/ingest.
"""
import argparse
import json
import os
import sys
import time
from pathlib import Path

from huggingface_hub import HfApi, Repository

HF_REPO = os.getenv("HF_DATASET_REPO", "axentx/surrogate-1-training-pairs")

def list_folder_files(date_folder: str, out_json: str):
    api = HfApi()
    # Non-recursive: list only immediate children in this date folder
    entries = api.list_repo_tree(
        repo_id=HF_REPO,
        path=date_folder,
        repo_type="dataset",
        recursive=False,
    )

    files = []
    for e in entries:
        if e.type != "file":
            continue
        # CDN URL (no auth) — bypasses /api/ rate limits
        cdn_url = f"https://huggingface.co/datasets/{HF_REPO}/resolve/main/{e.path}"
        files.append({
            "path": e.path,
            "cdn_url": cdn_url,
            "size": getattr(e, "size", None),
        })

    payload = {
        "date_folder": date_folder,
        "repo": HF_REPO,
        "generated_at": int(time.time()),
        "files": files,
    }

    Path(out_json).write_text(json.dumps(payload, indent=2))
    print(json.dumps(payload, indent=2))

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="List files in a date folder (non-recursive).")
    parser.add_argument("date_folder", help="Folder path in dataset repo, e.g. batches/public-merged/2026-05-03")
    parser.add_argument("--out", default="file-list.json", help="Output JSON path")
    args = parser.parse_args()
    try:
        list_folder_files(args.date_folder, args.out)
    except Exception as e:
        print(f"Error: {e}", file=sys.stderr)
        sys.exit(1)
```

retry_429() {
  local cmd=("$@")

    if "${cmd[@]}"; then
      return 0
    fi
    local rc=$?
