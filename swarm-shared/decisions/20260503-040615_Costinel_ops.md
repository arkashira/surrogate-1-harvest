- Avoids HF API rate limits by using CDN URLs (`/resolve/main/...`) and precomputed file list.
- Non-blocking UI addition (panel) that signals context without changing core flows.
- Minimal infra: one JSON file + one React component + one build script.

### 3) React hook (data layer)

`src/hooks/useTopHub.ts`

```ts
import { useEffect, useState, useCallback } from 'react';

export interface TopHubSignal {
  hub: string;
  connections: number;
  date: string;
  source: string;
  generatedAt: string;
}

const DEFAULT_SIGNAL: TopHubSignal = {
  hub: 'MOC',
  connections: 0,
  date: '—',
  source: 'knowledge-rag-graph',
  generatedAt: new Date().toISOString(),
};

export function useTopHub(pollingMs = 0) {
  const [signal, setSignal] = useState<TopHubSignal>(DEFAULT_SIGNAL);
  const [loading, setLoading] = useState(true);

  const fetchSignal = useCallback(async () => {
    try {
      const res = await fetch('/data/top-hub.json', { cache: 'no-store' });
      if (!res.ok) throw new Error('No signal payload');
      const data = await res.json();
      setSignal({
        hub: data.hub || DEFAULT_SIGNAL.hub,
        connections: Number(data.connections) || 0,
        date: data.date || DEFAULT_SIGNAL.date,
        source: data.source || DEFAULT_SIGNAL.source,
        generatedAt: data.generatedAt || DEFAULT_SIGNAL.generatedAt,
      });
    } catch {
      // Keep defaults; do not block UI
      console.warn('Top-hub signal unavailable, using defaults');
    } finally {
      setLoading(false);
    }
  }, []);

  useEffect(() => {
    fetchSignal();
    if (pollingMs > 0) {
      const id = setInterval(fetchSignal, pollingMs);
      return () => clearInterval(id);
    }
  }, [fetchSignal, pollingMs]);

  return { signal, loading, refetch: fetchSignal };
}
```

### 4) React panel component

`src/components/TopHubSignalPanel.tsx`

```tsx
import React from 'react';
import { useTopHub } from '../hooks/useTopHub';
import './TopHubSignalPanel.css';

export const TopHubSignalPanel: React.FC = () => {
  const { signal, loading } = useTopHub();

  return (
    <div className="top-hub-signal-panel" title={`Source: ${signal.source} • Generated: ${signal.generatedAt}`}>
      <div className="panel-icon" aria-hidden="true">📈</div>
      <div className="panel-content">
        <div className="panel-title">Top-Hub Insight</div>
        <div className="panel-hub">{loading ? 'Loading…' : signal.hub}</div>
        <div className="panel-meta">
          {loading ? '' : `${signal.connections} connections · ${signal.date}`}
        </div>
      </div>
    </div>
  );
};
```
