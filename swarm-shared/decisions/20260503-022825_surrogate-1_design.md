# surrogate-1 / design

## Final Implementation Plan (≤2h)

Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16`, `DATE`, `HF_TOKEN`, `REPO_ID` (default: `axentx/surrogate-1-training-pairs`)
- Runner performs **one** `list_repo_tree` call for `batches/public-raw/{DATE}/` → writes `manifests/manifest-{DATE}.json`
- Worker loads manifest, filters by deterministic shard assignment (`hash(filename) % SHARD_TOTAL == SHARD_ID`)
- Downloads via **CDN-only URLs** (`https://huggingface.co/datasets/.../resolve/main/...`) with **no Authorization header** to bypass API rate limits
- Projects heterogeneous files to `{prompt, response}` at parse time (avoids pyarrow `CastError` on mixed schemas)
- Deduplicates via central md5 store (`lib/dedup.py`) and writes to:
  ```
  batches/public-merged/<DATE>/shard<SHARD_ID>-<HHMMSS>.jsonl
  ```
- Uses deterministic retry/backoff for 429 (wait 360s) and keeps HF API calls to **≤2 per worker** (list once via manifest, upload once)
- Reuses running HF Space / Lightning Studio when present (quota-safe)
- Adds commit-cap spreading across 5 sibling repos if needed

#!/usr/bin/env python3
"""
CDN-bypass ingestion worker for surrogate-1 public dataset.
Usage:
  SHARD_ID=3 SHARD_TOTAL=16 DATE=2026-05-03 \
  HF_TOKEN=hf_xxx \
  python bin/dataset-enrich.py
"""

import os
import sys
import json
import hashlib
import time
import sqlite3
from pathlib import Path
from datetime import datetime, timezone
from typing import List, Dict, Any

import requests
from huggingface_hub import HfApi, hf_hub_download

# ---- config ----
REPO_ID = os.getenv("REPO_ID", "axentx/surrogate-1-training-pairs")
SHARD_ID = int(os.getenv("SHARD_ID", "0"))
SHARD_TOTAL = int(os.getenv("SHARD_TOTAL", "16"))
DATE = os.getenv("DATE", datetime.now(timezone.utc).strftime("%Y-%m-%d"))
HF_TOKEN = os.getenv("HF_TOKEN")
if not HF_TOKEN:
    print("ERROR: HF_TOKEN required", file=sys.stderr)
    sys.exit(1)

API = HfApi(token=HF_TOKEN)

# ---- paths ----
BASE_DIR = Path(__file__).parent.parent
MANIFEST_PATH = BASE_DIR / "manifests" / f"manifest-{DATE}.json"
DEDUP_DB = BASE_DIR / "lib" / "dedup.db"
OUT_DIR = BASE_DIR / "outputs"
OUT_DIR.mkdir(parents=True, exist_ok=True)
DEDUP_DB.parent.mkdir(parents=True, exist_ok=True)

# ---- dedup store ----
def init_dedup() -> sqlite3.Connection:
    conn = sqlite3.connect(str(DEDUP_DB))
    conn.execute("CREATE TABLE IF NOT EXISTS seen (md5 TEXT PRIMARY KEY)")
    conn.commit()
    return conn

def is_seen(conn: sqlite3.Connection, md5: str) -> bool:
    cur = conn.execute("SELECT 1 FROM seen WHERE md5 = ?", (md5,))
    return cur.fetchone() is not None

def mark_seen(conn: sqlite3.Connection, md5: str) -> None:
    conn.execute("INSERT OR IGNORE INTO seen (md5) VALUES (?)", (md5,))

# ---- manifest ----
def load_manifest() -> List[str]:
    if not MANIFEST_PATH.exists():
        print(f"Manifest not found: {MANIFEST_PATH}", file=sys.stderr)
        # fallback: single API list for DATE folder (non-recursive)
        print("Fetching repo tree for DATE folder...", file=sys.stderr)
        items = API.list_repo_tree(repo_id=REPO_ID, path=f"batches/public-raw/{DATE}", recursive=False)
        files = [it.rfilename for it in items if it.type == "file"]
        MANIFEST_PATH.parent.mkdir(parents=True, exist_ok=True)
        with open(MANIFEST_PATH, "w") as f:
            json.dump(files, f)
    else:
        with open(MANIFEST_PATH) as f:
            files = json.load(f)
    return files

# ---- sharding ----
def shard_files(files: List[str]) -> List[str]:
    out = []
    for f in files:
        h = int(hashlib.sha256(f.encode()).hexdigest(), 16)
        if h % SHARD_TOTAL == SHARD_ID:
            out.append(f)
    return out

# ---- CDN download helpers ----
def cdn_download(repo_id: str, path: str, dest: Path) -> Path:
    url = f"https://huggingface.co/datasets/{repo_id}/resolve/main/{path}"
    resp = requests.get(url, stream=True, timeout=60)
    resp.raise_for_status()
    with open(dest, "wb") as f:
        for chunk in resp.iter_content(chunk_size=8192):
            f.write(chunk)
    return dest

# ---- schema projection ----
def extract_pairs_from_parquet(path: Path):
    # Lightweight projection: read only needed cols if possible;
    # fallback to full read and keep prompt/response.
    try:
        import pyarrow.parquet as pq
        tbl = pq.read_table(path, columns=["prompt", "response"])
    except Exception:
        tbl = pq.read_table(path)
        cols = set(tbl.column_names)
        if "prompt" not in cols or "response" not in cols:
            raise ValueError(f"Missing prompt/response in {path}: {cols}")
        tbl = tbl.select(["prompt", "response"])
    return tbl.to_pylist()

def extract_pairs_from_jsonl(path: Path):
    pairs = []
    with open(path) as f:
        for line in f:
            line = line.strip()
            if not line:
                continue
            obj = json.loads(line)
            pairs.append({"prompt": obj["prompt"], "response": obj["response"]})
    return pairs

# ---- worker ----
def run_worker():
    print(f"Worker shard={SHARD_ID}/{SHARD_TOTAL} date={DATE} repo={REPO_ID}")
    files = load_manifest()
    my_files = shard_files(files)
    print(f"Total files={len(files)} assigned={len(my_files)}")

    conn = init_dedup()
    out_rows = []
    ts = datetime.now(timezone.utc).strftime("%Y%m%d-%H%M%S")

    for rel in my_files:
        try:
            tmp = OUT_DIR / f"tmp_{hashlib.sha256(rel.encode()).hexdigest()[:12]}.parquet"
            cdn_download(REPO_ID, rel, tmp)

            if rel.endswith(".parquet"):
                pairs = extract_pairs_from_parquet(tmp)
            elif rel.endswith(".jsonl"):
                pairs = extract_pairs_from_jsonl(tmp)
            else:
                print(f"Skip unsupported {rel}", file=sys.stderr)
                tmp.unlink(missing_ok=True)
                continue

            for p in pairs:
                prompt = str(p.get("prompt") or "").strip()
                response = str(p.get("response") or "").strip()
                if not prompt or not response:
                    continue

                if is_seen(conn, md5):
                    continue
                mark_seen(conn, md5)
                out_rows.append({"prompt": prompt, "response": response})

            tmp.unlink(missing_ok=True)

            print(f"Error processing {rel}: {e}", file=sys.stderr)

    conn.close()

        return
