# workio / ops

## Final consolidated implementation

**Diagnosis (merged, resolved)**
- LINE retries on 5xx/timeouts and concurrent webhook deliveries can create duplicate punches/leave/OT entries.
- `find-then-insert/update` without row-level locking or a DB-level unique constraint allows multiple open punches per employee.
- Punch and notification writes are not atomic; partial state can persist on failure.
- No retry/backoff for outbound LINE notify calls; transient failures can leave users without confirmations.
- No idempotency key persistence or fast-path deduplication; no operational alerting/structured logging around latency/failures.

**Chosen approach (correctness + actionability)**
- Use DB-backed idempotency keys (persistent) as the source of truth, plus a lightweight in-memory LRU for fast-path rejection of recent duplicates.
- Enforce “at most one open punch” with a partial unique index and `SELECT ... FOR UPDATE` within a transaction.
- Wrap punch mutation and LINE notify in a transaction; retry outbound notify with exponential backoff and jitter.
- Keep signature verification and strict input validation.
- Add structured logging and metrics hooks for latency, duplicates, and failures.

### 2) Webhook handler

```typescript
// workio/server/src/routes/webhook/line.ts
import express from 'express';
import crypto from 'crypto';
import { pool } from '../../db';
import { sendLineNotify } from '../../lib/line';
import pRetry from 'p-retry';
import LRU from 'lru-cache';

const router = express.Router();

// Fast-path in-memory dedupe (recent keys)
const recentKeys = new LRU<string, true>({
  max: 10_000,
  ttl: 1000 * 60 * 10, // 10 minutes
});

function verifySignature(body: string, signature: string, channelSecret: string): boolean {
  const hash = crypto
    .createHmac('sha256', channelSecret)
    .update(body, 'utf8')
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(hash), Buffer.from(signature));
}

function idempotencyKey(events: any[], body: string): string {
  const first = events[0];
  return crypto
    .createHash('sha256')
    .update(`${first?.timestamp}:${first?.type}:${body}`)
    .digest('hex');
}

async function acquireIdempotencyKey(client: any, key: string, requestHash: string): Promise<boolean> {
  const res = await client.query(
    `INSERT INTO idempotency_keys (key, request_hash) VALUES ($1, $2) ON CONFLICT (key) DO NOTHING RETURNING key`,
    [key, requestHash]
  );
  return res.rowCount === 1;
}

async function notifyWithRetry(replyToken: string, message: string) {
  return pRetry(() => sendLineNotify(replyToken, message), {
    retries: 3,
    factor: 2,
    minTimeout: 300,
    maxTimeout: 3000,
    randomize: true,
    onFailedAttempt: (err) => {
      console.warn(`LINE notify retry failed (attempt ${err.attemptNumber})`, err.message);
    },
  });
}

router.post('/line', async (req, res) => {
  const signature = req.headers['x-line-signature'] as string;
  const bodyRaw = JSON.stringify(req.body);
  const channelSecret = process.env.LINE_CHANNEL_SECRET || '';

  if (!signature || !channelSecret) {
    return res.status(400).send('Server misconfigured');
  }

  if (!verifySignature(bodyRaw, signature, channelSecret)) {
    return res.status(401).send('Invalid signature');
  }

  const events = req.body.events;
  if (!Array.isArray(events) || events.length === 0) {
    return res.status(200).send('OK');
  }

  const key = idempotencyKey(events, bodyRaw);

  // Fast-path: reject obvious recent duplicates
  if (recentKeys.has(key)) {
    return res.status(200).send('Duplicate (fast-path)');
  }

  const client = await pool.connect();
  const startedAt = Date.now();

  try {
    await client.query('BEGIN');

    // Persistent idempotency guard
    const fresh = await acquireIdempotencyKey(client, key, bodyRaw);
    if (!fresh) {
      await client.query('COMMIT');
      recentKeys.set(key, true);
      return res.status(200).send('Duplicate (idempotent)');
    }

    for (const event of events) {
      if (event?.type !== 'message' || event?.message?.type !== 'text') continue;

      const text = String(event.message.text || '').trim().toLowerCase();
      const userId = event.source?.userId;
      const replyToken = event.replyToken;
      if (!userId || !replyToken) continue;

      // Resolve employee
      const empRes = await client.query(
        `SELECT id FROM employees WHERE line_user_id = $1 AND tenant_id IS NOT NULL`,
        [userId]
      );

      if (empRes.rowCount === 0) {
        await notifyWithRetry(replyToken, 'พนักงานไม่พบในระบบ');
        continue;
      }

      const employeeId = empRes.rows[0].id;

      // Lock any existing open punch row
      const openRes = await client.query(
        `SELECT id FROM punches WHERE employee_id = $1 AND clock_out IS NULL FOR UPDATE`,
        [employeeId]
      );

      if (text === 'clockin' || text === 'เข้างาน') {
        if (openRes.rowCount > 0) {
          await notifyWithRetry(replyToken, 'มีการลงเวลาทำงานเปิดอยู่แล้ว');
          continue;
        }

        await client.query(
          `INSERT INTO punches (employee_id, clock_in, location) VALUES ($1, NOW(), $2)`,
          [employeeId, `(${Number(event.longitude) || 0},${Number(event.latitude) || 0})`]
        );

        await notifyWithRetry(replyToken, 'ลงเวลาทำงานเรียบร้อยแล้ว');
      } else if (text === 'clockout' || text === 'ออกงาน') {
        if (openRes.rowCount === 0) {
          await notifyWithRetry(replyToken, 'ไม่พบการลงเวลาเปิด');
          continue;
        }

        await client.query(
          `UPDATE punches SET clock_out = NOW() WHERE id = $1`,
          [openRes.rows[0].id]
        );

        await notifyWithRetry(replyToken, 'ลงเวลาออกเรียบร้อยแล้ว');
      }
    }

    await client.query('COMMIT');
    recentKeys.set(key, true);

    // Structured log (hook-friendly)
    console.info(JSON.stringify({
      msg: 'line_webhook_processed',
      key,
      events: events.length,
      duration_ms: Date.now() - startedAt,
      status: 'ok',
    }));

    return res.status(200).send('OK');
  } catch (err: any) {
    try {
      await client.query('ROLLBACK');
    } catch (_) {
      // best-effort rollback
    }

    console.error(JSON.stringify({
      msg: 'line_webhook_failed',
      key,
      error: err?.message || String(err),
      duration_ms: Date.now() - startedAt,
      status: 'error',
    }));

    return res.status(500).send('Processing error');
