# workio / quality

## 1. Diagnosis

- **No idempotency key storage** — LINE webhook `X-Line-Signature` is verified but not persisted; retries with same signature still produce duplicate punch rows.
- **Non-atomic upsert path** — current flow does `findOne → if exists update else insert`; race conditions under retry or parallel webhooks corrupt active-punch state.
- **Missing active-punch uniqueness constraint** — database allows multiple concurrent “IN” rows per employee; downstream reports and clock logic become inconsistent.
- **No replay-safe audit trail** — duplicate webhook deliveries cannot be detected after the fact; operational debugging and compliance are brittle.
- **Frontend punch UX races with retries** — optimistic UI can diverge from server truth when duplicates or reorder occur; no client-side idempotency coordination.

## 2. Proposed change

- **Scope**: `workio/server/src/services/punchService.ts` (or equivalent) + `workio/server/src/db/schema.sql` + `workio/src/hooks/usePunchQueue.ts`
- **Goal**: Add database-level idempotency table + unique constraint on active punches + atomic upsert; expose small hook to queue/acknowledge punches client-side with idempotency key.

## 3. Implementation

### 3.1 Database schema (add idempotency + constraint)

```sql
-- workio/server/src/db/schema.sql
-- Add idempotency table for LINE webhook deliveries
CREATE TABLE IF NOT EXISTS webhook_idempotency (
  idempotency_key VARCHAR(255) PRIMARY KEY,
  payload_hash VARCHAR(64) NOT NULL,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Ensure only one active punch per employee (no concurrent IN without OUT)
-- Assumes table name is punches and has employee_id, type, created_at, deleted_at/soft-delete
CREATE UNIQUE INDEX IF NOT EXISTS idx_one_active_punch_per_employee
ON punches (employee_id)
WHERE type = 'IN' AND (deleted_at IS NULL OR deleted_at > NOW());
```

### 3.2 Atomic punch service with idempotency

```ts
// workio/server/src/services/punchService.ts
import { Pool } from 'pg';
import crypto from 'crypto';

const pool = new Pool({ connectionString: process.env.DATABASE_URL });

export type PunchInput = {
  employeeId: string;
  type: 'IN' | 'OUT';
  latitude?: number;
  longitude?: number;
  lineSignature?: string;
  lineRawBody?: string;
  idempotencyKey?: string; // e.g., X-Line-Signature or client-generated uuid
};

export async function handlePunch(input: PunchInput) {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // Idempotency check (if key provided)
    if (input.idempotencyKey) {
      const hash = input.lineRawBody
        ? crypto.createHash('sha256').update(input.lineRawBody).digest('hex')
        : '';
      const idempotent = await client.query(
        `SELECT 1 FROM webhook_idempotency WHERE idempotency_key = $1 AND payload_hash = $2`,
        [input.idempotencyKey, hash]
      );
      if (idempotent.rows.length > 0) {
        // Fetch existing punch created by this delivery (best-effort)
        const existing = await client.query(
          `SELECT * FROM punches WHERE external_ref = $1 LIMIT 1`,
          [input.idempotencyKey]
        );
        await client.query('COMMIT');
        return existing.rows[0] || null;
      }
    }

    // Atomic upsert for active punch:
    // If latest punch for employee is IN and this punch is OUT -> close it.
    // Otherwise create new punch.
    const now = new Date();
    const latest = await client.query(
      `SELECT * FROM punches
       WHERE employee_id = $1 AND (deleted_at IS NULL OR deleted_at > $2)
       ORDER BY created_at DESC LIMIT 1`,
      [input.employeeId, now]
    );

    let result;
    if (latest.rows.length > 0 && latest.rows[0].type === 'IN' && input.type === 'OUT') {
      // Close the open punch
      result = await client.query(
        `UPDATE punches
         SET type = 'OUT', clock_out_at = $1, updated_at = NOW()
         WHERE id = $2 RETURNING *`,
        [now, latest.rows[0].id]
      );
    } else if (input.type === 'IN') {
      // Create new IN punch (unique constraint protects concurrent IN)
      result = await client.query(
        `INSERT INTO punches (employee_id, type, clock_in_at, latitude, longitude, external_ref, created_at, updated_at)
         VALUES ($1, $2, $3, $4, $5, $6, $7, $7) RETURNING *`,
        [
          input.employeeId,
          'IN',
          now,
          input.latitude ?? null,
          input.longitude ?? null,
          input.idempotencyKey || null,
          now,
        ]
      );
    } else {
      // Invalid transition (e.g., double OUT) — still record but keep semantics safe
      result = await client.query(
        `INSERT INTO punches (employee_id, type, clock_in_at, clock_out_at, latitude, longitude, external_ref, created_at, updated_at)
         VALUES ($1, $2, $1, $1, $3, $4, $5, $1, $1) RETURNING *`,
        [
          input.employeeId,
          input.type,
          now,
          input.latitude ?? null,
          input.longitude ?? null,
          input.idempotencyKey || null,
        ]
      );
    }

    // Store idempotency record (if key provided)
    if (input.idempotencyKey) {
      const hash = input.lineRawBody
        ? crypto.createHash('sha256').update(input.lineRawBody).digest('hex')
        : crypto.createHash('sha256').update(JSON.stringify(input)).digest('hex');
      await client.query(
        `INSERT INTO webhook_idempotency (idempotency_key, payload_hash, created_at)
         VALUES ($1, $2, NOW())`,
        [input.idempotencyKey, hash]
      );
    }

    await client.query('COMMIT');
    return result.rows[0];
  } catch (err) {
    await client.query('ROLLBACK');
    // Unique violation on idx_one_active_punch_per_employee is expected for duplicate IN
    // Return existing active punch in that case
    if ((err as any).code === '23505') {
      const existing = await client.query(
        `SELECT * FROM punches
         WHERE employee_id = $1 AND type = 'IN' AND (deleted_at IS NULL OR deleted_at > NOW())
         LIMIT 1`,
        [input.employeeId]
      );
      return existing.rows[0] || null;
    }
    throw err;
  } finally {
    client.release();
  }
}
```
