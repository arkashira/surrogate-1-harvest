# workio / frontend

## Final Synthesized Solution

### Diagnosis (merged)
- **No optimistic UI**: users wait on network round-trip and can double-tap, causing duplicate requests/race punches.
- **No local pending-state or error recovery**: if LINE webhook succeeds but client times out, employee sees failure and may retry, creating duplicates.
- **Stale reads after optimistic actions**: punch list/UI does not reflect local state immediately and can jump when server data refetches.
- **No client-side debounce/throttle**: rapid taps trigger multiple in-flight requests.
- **No visual feedback**: no distinction between “pending”, “committed”, and “error” states on the clock card.

---

### Implementation Plan (merged + concrete)

**Primary location**: `workio/src/components/ClockCard.tsx`  
**Optional shared layer**: `workio/src/hooks/usePunch.ts` (recommended for multi-component usage)

#### 1) Shared mutation hook (`usePunch`) — recommended
```ts
// workio/src/hooks/usePunch.ts
import { useMutation, useQueryClient } from '@tanstack/react-query';

export function usePunch() {
  const queryClient = useQueryClient();

  return useMutation({
    mutationFn: () =>
      fetch('/api/punches', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ type: 'IN_OUT' }),
        // signal handled by caller/component for per-request abort
      }).then((r) => {
        if (!r.ok) throw new Error(r.statusText);
        return r.json();
      }),
    onMutate: async () => {
      await queryClient.cancelQueries({ queryKey: ['punches'] });
      // optimistic update could be applied here if you want local cache preview
    },
    onError: (err, _, context) => {
      // rollback optimistic update if you snapshot in onMutate
      queryClient.invalidateQueries({ queryKey: ['punches'] });
    },
    onSettled: () => {
      queryClient.invalidateQueries({ queryKey: ['punches'] });
      queryClient.invalidateQueries({ queryKey: ['today-summary'] });
    },
  });
}
```

#### 2) ClockCard with optimistic UI + dedupe + rollback
```tsx
// workio/src/components/ClockCard.tsx
import { useState, useCallback, useRef } from 'react';
import { useQueryClient } from '@tanstack/react-query';
import { Clock, Check, AlertCircle } from 'lucide-react';
import { usePunch } from './hooks/usePunch';

type PunchStatus = 'idle' | 'pending' | 'committed' | 'error';

export function ClockCard() {
  const queryClient = useQueryClient();
  const { mutate: punchMutate, isPending: isMutating } = usePunch();

  const [status, setStatus] = useState<PunchStatus>('idle');
  const [lastError, setLastError] = useState<string | null>(null);
  const inflightRef = useRef<AbortController | null>(null);

  const punch = useCallback(async () => {
    // dedupe: cancel any previous in-flight request
    if (inflightRef.current) {
      inflightRef.current.abort();
    }
    const ac = new AbortController();
    inflightRef.current = ac;

    setLastError(null);
    setStatus('pending');

    try {
      // optimistic update: immediately update local cache to reflect pending punch
      const previous = queryClient.getQueryData(['punches']);
      const optimisticPunch = {
        id: `temp-${Date.now()}`,
        type: 'IN_OUT',
        timestamp: new Date().toISOString(),
        _optimistic: true,
      };
      queryClient.setQueryData(['punches'], (old: any[] = []) => [
        optimisticPunch,
        ...old,
      ]);

      const res = await fetch('/api/punches', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ type: 'IN_OUT' }),
        signal: ac.signal,
      });

      if (!res.ok) throw new Error(`Server responded ${res.status}`);
      await res.json();

      // commit: invalidate so server truth wins and removes optimistic entry
      queryClient.invalidateQueries({ queryKey: ['punches'] });
      queryClient.invalidateQueries({ queryKey: ['today-summary'] });
      setStatus('committed');
    } catch (err: any) {
      if (err.name === 'AbortError') return;
      // rollback: restore previous data
      queryClient.setQueryData(['punches'], (previous: any) => previous);
      setStatus('error');
      setLastError(err.message || 'Punch failed');
      queryClient.invalidateQueries({ queryKey: ['punches'] });
    } finally {
      if (inflightRef.current === ac) {
        inflightRef.current = null;
      }
      if (status !== 'error') {
        setTimeout(() => setStatus('idle'), 1200);
      }
    }
  }, [queryClient, status]);

  const handlePunch = useCallback(() => {
    if (status === 'pending') return;
    punch();
  }, [status, punch]);

  // derive UI state (prefer local status for responsiveness)
  const isPending = status === 'pending' || isMutating;

  return (
    <div className="rounded-xl bg-white p-6 shadow-sm ring-1 ring-gray-100">
      <div className="flex flex-col items-center gap-4">
        <div className="relative">
          <button
            onClick={handlePunch}
            disabled={isPending}
            className={`
              flex h-20 w-20 items-center justify-center rounded-full text-white transition-all
              ${
                status === 'pending'
                  ? 'cursor-wait bg-blue-400'
                  : status === 'committed'
                  ? 'cursor-default bg-green-500'
                  : status === 'error'
                  ? 'cursor-pointer bg-red-500'
                  : 'bg-blue-600 hover:bg-blue-700 active:scale-95'
              }
            `}
            aria-label={
              status === 'pending'
                ? 'กำลังบันทึก...'
                : status === 'committed'
                ? 'บันทึกแล้ว'
                : status === 'error'
                ? 'เกิดข้อผิดพลาด'
                : 'ลงเวลาทำงาน'
            }
          >
            {status === 'pending' && (
              <div className="h-8 w-8 animate-spin rounded-full border-2 border-white/40 border-t-white" />
            )}
            {status === 'committed' && <Check className="h-8 w-8" />}
            {status === 'error' && <AlertCircle className="h-8 w-8" />}
            {status === 'idle' && <Clock className="h-8 w-8" />}
          </button>
        </div>

        <div className="text-center">
          {status === 'pending' && (
            <p className="text-sm font-medium text-gray-600">กำลังบันทึก...</p>
          )}
          {status === 'committed' && (
            <p className="text-sm font-medium text-green-600">บันทึกเรียบร้อย</p>
          )}
          {status === 'error' && (
            <p className="text-sm font-medium text-red-600">
              {lastError || 'เกิดข้อผิดพลาด กรุณาลองใหม่'}
            </p>
          )}
          {status === 'idle' && (
            <p className="text-sm font-medium text-gray-600">กดเพื่อลงเวลา</p>
          )}
        </div>
      </div>
    </div>
  );
}
```

   - On success, invalidation replaces it with server truth; on failure, rollback restores previous state.
