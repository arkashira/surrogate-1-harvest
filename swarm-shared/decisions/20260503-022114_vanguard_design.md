# vanguard / design

## Final synthesized solution (correct + actionable)

**Core problem**: training burns HF API quota and risks 429s because it discovers files at runtime and loads via authenticated APIs instead of CDN; schema heterogeneity can cause pyarrow errors; no reuse or resilience.

**Goal**: eliminate authenticated API calls during training, guarantee schema-safe reads, minimize quota waste, and enable deterministic restarts.

### 1. One-time manifest generation (run from any machine with HF token)

File: `vanguard/data/manifest.py`

```python
#!/usr/bin/env python3
"""
Generate a (repo, date_folder) -> file-list manifest once.
Keeps training API-free.
"""
import json
import os
import sys
from datetime import datetime, timezone
from pathlib import Path

from huggingface_hub import list_repo_tree

HF_REPO = os.getenv("HF_DATASET_REPO", "datasets/axentx/surrogate-1")
MANIFEST_DIR = Path(__file__).parent.parent / "manifests"
MANIFEST_DIR.mkdir(exist_ok=True, parents=True)

def build_manifest(date_folder: str, repo: str = HF_REPO) -> dict:
    prefix = f"{date_folder.rstrip('/')}/"
    items = list_repo_tree(repo=repo, path=prefix, recursive=False)
    files = sorted(
        f"{prefix}{item.path.removeprefix(prefix)}"
        for item in items
        if item.type == "file"
    )
    return {
        "repo": repo,
        "date_folder": date_folder.rstrip("/"),
        "created_at": datetime.now(timezone.utc).isoformat(),
        "files": files,
    }

def save_manifest(manifest: dict) -> Path:
    slug = manifest["date_folder"].replace("/", "_")
    out = MANIFEST_DIR / f"manifest_{slug}.json"
    out.write_text(json.dumps(manifest, indent=2))
    return out

if __name__ == "__main__":
    if len(sys.argv) < 2:
        print("Usage: manifest.py <date_folder>  # e.g. 2026-04-29")
        sys.exit(1)
    m = build_manifest(sys.argv[1].strip())
    p = save_manifest(m)
    print(f"Saved: {p}")
```

Run once:

```bash
cd /opt/axentx/vanguard
python data/manifest.py 2026-04-29
```

### 2. CDN-only, schema-resilient loader (zero API calls during training)

File: `vanguard/data/cdn_loader.py`

```python
#!/usr/bin/env python3
"""
IterableDataset that streams parquet files via CDN only.
- No HF API calls during training.
- Schema-safe: coerces columns to expected names/types; skips corrupt rows.
"""
import io
import json
from pathlib import Path
from typing import Dict, Iterator, Optional

import pyarrow as pa
import pyarrow.parquet as pq
import requests
import torch
from torch.utils.data import IterableDataset

CDN_ROOT = "https://huggingface.co/datasets/datasets/axentx/surrogate-1/resolve/main"

# Expected schema names (flexible fallback)
PROMPT_KEYS = ("prompt", "text", "input", "instruction")
RESPONSE_KEYS = ("response", "output", "completion", "answer")

def _normalize_str(val) -> str:
    if val is None:
        return ""
    if isinstance(val, (list, tuple)):
        val = val[0] if val else ""
    return str(val).strip()

class CDNParquetIterable(IterableDataset):
    def __init__(
        self,
        manifest_path: str,
        start_index: int = 0,
        max_retries: int = 3,
        timeout: int = 30,
    ) -> None:
        p = Path(manifest_path)
        if not p.exists():
            raise FileNotFoundError(f"Manifest not found: {manifest_path}")
        self.manifest = json.loads(p.read_text())
        self.files = self.manifest["files"]
        self.start_index = start_index
        self.max_retries = max_retries
        self.timeout = timeout

    def _fetch_parquet(self, url: str) -> Optional[pa.Table]:
        for attempt in range(self.max_retries):
            try:
                resp = requests.get(url, timeout=self.timeout)
                resp.raise_for_status()
                return pq.read_table(io.BytesIO(resp.content))
            except Exception as exc:
                if attempt == self.max_retries - 1:
                    print(f"Failed to fetch {url}: {exc}")
                    return None
        return None

    def _rows_from_table(self, table: pa.Table) -> Iterator[Dict[str, str]]:
        # Resolve columns flexibly
        prompt_col = next((c for c in PROMPT_KEYS if c in table.column_names), None)
        response_col = next((c for c in RESPONSE_KEYS if c in table.column_names), None)

        if not prompt_col or not response_col:
            # If expected columns missing, skip file
            return

        # Coerce to string safely
        prompts = table.column(prompt_col).to_pylist()
        responses = table.column(response_col).to_pylist()
        for p, r in zip(prompts, responses):
            p_s = _normalize_str(p)
            r_s = _normalize_str(r)
            if p_s and r_s:
                yield {"prompt": p_s, "response": r_s}

    def _stream_files(self) -> Iterator[Dict[str, str]]:
        for rel_path in self.files[self.start_index:]:
            url = f"{CDN_ROOT}/{rel_path}"
            table = self._fetch_parquet(url)
            if table is None:
                continue
            yield from self._rows_from_table(table)

    def __iter__(self) -> Iterator[Dict[str, str]]:
        return self._stream_files()
```
