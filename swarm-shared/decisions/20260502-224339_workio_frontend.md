# workio / frontend

## Final Synthesis — One Correct, Actionable Plan

**Core principle:**  
Adopt **optimistic UI + idempotent mutations + durable offline queue**, but replace the fragile localStorage queue with **IndexedDB** for reliability and size, and add **client-side validation + sync status UI** to reduce wasted requests and user confusion.

---

### 1. Architecture (single source of truth)

- **Optimistic-first mutations** via a reusable hook (`useWorkioMutation`) that:
  - Accepts `optimisticUpdate` / `rollback`.
  - Enforces **idempotency keys** (client-generated or provided).
  - Prevents duplicate in-flight requests.
- **Offline queue** in `src/lib/offlineQueue.ts` using **IndexedDB** (with localStorage fallback for critical metadata only).
- **Feature layer** for attendance in `src/features/attendance/` to encapsulate rules (GPS validation, status mapping, sync indicators).
- **UI components** consume the hook and feature helpers, not raw fetch or scattered state.

#### `src/lib/offlineQueue.ts`
```ts
// IndexedDB-backed queue; falls back to in-memory only if blocked.
// No localStorage for large/complex queue (avoid size/serialization issues).

type QueueItem = {
  id: string;
  endpoint: string;
  method: 'POST' | 'PUT' | 'PATCH';
  body: Record<string, unknown>;
  retries: number;
  createdAt: number;
};

const DB_NAME = 'workio_offline';
const STORE_NAME = 'queue';
const MAX_RETRIES = 3;

function idb() {
  return new Promise<IDBDatabase>((resolve, reject) => {
    const req = indexedDB.open(DB_NAME, 1);
    req.onupgradeneeded = () => {
      const db = req.result;
      if (!db.objectStoreNames.contains(STORE_NAME)) {
        db.createObjectStore(STORE_NAME, { keyPath: 'id' });
      }
    };
    req.onsuccess = () => resolve(req.result);
    req.onerror = () => reject(req.error);
  });
}

async function withStore(mode: IDBTransactionMode, cb: (s: IDBObjectStore) => void) {
  const db = await idb();
  const tx = db.transaction(STORE_NAME, mode);
  const store = tx.objectStore(STORE_NAME);
  cb(store);
  return tx.complete;
}

export async function enqueueMutation(opts: {
  endpoint: string;
  method?: 'POST' | 'PUT' | 'PATCH';
  body: Record<string, unknown>;
}): Promise<string> {
  const id = crypto.randomUUID();
  const item: QueueItem = {
    id,
    endpoint: opts.endpoint,
    method: opts.method || 'POST',
    body: opts.body,
    retries: 0,
    createdAt: Date.now(),
  };
  await withStore('readwrite', (s) => s.add(item));
  return id;
}

export async function removeFromQueue(id: string) {
  await withStore('readwrite', (s) => s.delete(id));
}

export async function getPendingQueue(): Promise<QueueItem[]> {
  const db = await idb();
  const tx = db.transaction(STORE_NAME, 'readonly');
  const store = tx.objectStore(STORE_NAME);
  return store.getAll();
}

export async function flushQueue(fetcher: (item: QueueItem) => Promise<void>) {
  const items = await getPendingQueue();
  const remaining: QueueItem[] = [];
  for (const item of items) {
    try {
      await fetcher(item);
      await removeFromQueue(item.id);
    } catch (err) {
      item.retries += 1;
      if (item.retries < MAX_RETRIES) {
        remaining.push(item);
      } else {
        await removeFromQueue(item.id); // drop after max retries
      }
    }
  }
}
```

#### `src/hooks/useWorkioMutation.ts`
```ts
import { useCallback, useRef } from 'react';
import { enqueueMutation, removeFromQueue, flushQueue } from '../lib/offlineQueue';

type MutationOptions<TData = unknown> = {
  onSuccess?: (data: TData) => void;
  onError?: (error: Error) => void;
  optimisticUpdate?: () => void;
  rollback?: () => void;
};

export function useWorkioMutation<TVariables = Record<string, unknown>, TData = unknown>() {
  const inflight = useRef<Set<string>>(new Set());

  const mutate = useCallback(
    async (
      endpoint: string,
      options?: {
        method?: 'POST' | 'PUT' | 'PATCH';
        body?: TVariables;
        idempotencyKey?: string;
        optimistic?: boolean;
      } & MutationOptions<TData>
    ) => {
      const method = options?.method || 'POST';
      const body = options?.body || ({} as TVariables);
      const optimistic = options?.optimistic !== false;
      const idempotencyKey = options?.idempotencyKey || crypto.randomUUID();

      if (inflight.current.has(idempotencyKey)) {
        return; // prevent duplicate in-flight
      }

      if (optimistic) {
        try {
          options?.optimisticUpdate?.();
        } catch {
          // ignore optimistic failures
        }
      }

      const doFetch = async (): Promise<TData | null> => {
        inflight.current.add(idempotencyKey);
        try {
          const res = await fetch(`/api${endpoint}`, {
            method,
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ ...body, _idempotencyKey: idempotencyKey }),
          });
          if (!res.ok) throw new Error(`HTTP ${res.status}`);
          return (await res.json()) as TData;
        } finally {
          inflight.current.delete(idempotencyKey);
        }
      };

      // Immediate attempt
      try {
        const data = await doFetch();
        options?.onSuccess?.(data as TData);
        return data;
      } catch (err) {
        // rollback optimistic on immediate failure
        if (optimistic) {
          try {
            options?.rollback?.();
          } catch {
            // ignore rollback failures
          }
        }
        // enqueue for background retry
        const qid = await enqueueMutation({ endpoint, method, body: body as Record<string, unknown> });
        options?.onError?.(err instanceof Error ? err : new Error(String(err)));
        return null;
      }
    },
    []
  );

  const flush = useCallback(async () => {
    await flushQueue(async (item) => {
      const res = await fetch(`/api${item.endpoint}`, {
        method: item.method,
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(item.body),
      });
      if (!res.ok) throw new Error(`HTTP ${res.status}`);
      await removeFromQueue(item.id);
    });
  }, []);

  return { mutate, flush };
}
```
