### Final Synthesis: Manifest-Driven CDN-Only Dataset Loader for Surrogate-1 Training  
**Estimated effort:** <2h  
**Value:** Eliminates HF API rate limits (429), fixes `pyarrow.CastError` on mixed schemas, enables 24/7 autonomous training.

---

### Correctness + Actionable Implementation Plan

1. **Generate a file manifest once per date folder**  
   - Use a single `list_repo_tree(repo_id, path=date_folder, recursive=True)` call to list all dataset files.  
   - Save to `dataset_manifest.json` (list of relative paths).  
   - *Do not* embed market analysis or RAG queries here — they add risk and time without improving loader correctness.

2. **Download via HF CDN (no API calls during training)**  
   - Use direct CDN URLs:  
     `https://huggingface.co/datasets/{repo_id}/resolve/main/{rel_path}`  
   - Fallback: use `hf_hub_download(repo_id, filename, repo_type="dataset")` if CDN URL fails (handles auth/token edge cases).  
   - Download files to a local cache directory keyed by hash of path to avoid redundant fetches.

3. **Schema-safe parsing: project to `{prompt, response}` at parse time**  
   - For each file (JSON/JSONL/Parquet), stream rows and select only `prompt` and `response` fields.  
   - Coerce types explicitly (str) to prevent `pyarrow.CastError`.  
   - Drop or log rows missing either field.

4. **Integrate into Surrogate-1 training script**  
   - Add a `load_dataset_cdn(manifest_path, cache_dir)` function that returns an iterable of `{prompt, response}` dicts or a PyTorch Dataset.  
   - Replace existing HF dataset loading with this function.  
   - Ensure training loop is resilient to empty shards.

# cdn_loader.py
import json
import os
import hashlib
import requests
from pathlib import Path
from typing import List, Dict, Iterator
from huggingface_hub import list_repo_tree, hf_hub_download


def create_manifest(repo_id: str, date_folder: str, out_path: str) -> List[str]:
    """List files in repo/date_folder and save manifest."""
    items = list_repo_tree(repo_id=repo_id, path=date_folder, recursive=True)
    files = [item.rfilename for item in items if item.type == "file"]
    Path(out_path).parent.mkdir(parents=True, exist_ok=True)
    with open(out_path, "w") as f:
        json.dump(files, f)
    return files


def _cache_path(url: str, cache_dir: str) -> str:
    key = hashlib.sha256(url.encode()).hexdigest()
    return os.path.join(cache_dir, key + os.path.splitext(url)[1])


def _download_cdn(url: str, cache_dir: str) -> str:
    cache_file = _cache_path(url, cache_dir)
    if os.path.exists(cache_file):
        return cache_file
    Path(cache_dir).mkdir(parents=True, exist_ok=True)
    r = requests.get(url, timeout=60)
    r.raise_for_status()
    with open(cache_file, "wb") as f:
        f.write(r.content)
    return cache_file


def _download_fallback(repo_id: str, rel_path: str, cache_dir: str) -> str:
    # hf_hub_download uses CDN under the hood but handles auth/token edge cases
    return hf_hub_download(repo_id=repo_id, filename=rel_path, repo_type="dataset", cache_dir=cache_dir)


def load_dataset_cdn(
    repo_id: str,
    manifest_path: str,
    cache_dir: str = ".cdn_cache",
    cdn_base: str = "https://huggingface.co/datasets",
) -> Iterator[Dict[str, str]]:
    """
    Yield {'prompt': str, 'response': str} from manifest using CDN.
    """
    with open(manifest_path) as f:
        files = json.load(f)

    for rel_path in files:
        # Try direct CDN first; fallback to hf_hub_download
        cdn_url = f"{cdn_base}/{repo_id}/resolve/main/{rel_path}"
        try:
            local_path = _download_cdn(cdn_url, cache_dir)
        except Exception:
            local_path = _download_fallback(repo_id, rel_path, cache_dir)

        # Parse and project to {prompt, response}
        suffix = Path(rel_path).suffix.lower()
        try:
            if suffix == ".jsonl":
                import jsonlines
                with jsonlines.open(local_path) as reader:
                    for obj in reader:
                        prompt = str(obj.get("prompt", ""))
                        response = str(obj.get("response", ""))
                        if prompt and response:
                            yield {"prompt": prompt, "response": response}
            elif suffix == ".json":
                with open(local_path) as fp:
                    data = json.load(fp)
                if isinstance(data, list):
                    items = data
                else:
                    items = [data]
                for obj in items:
                    prompt = str(obj.get("prompt", ""))
                    response = str(obj.get("response", ""))
                    if prompt and response:
                        yield {"prompt": prompt, "response": response}
            elif suffix in (".parquet", ".pq"):
                import pyarrow.parquet as pq
                table = pq.read_table(local_path, columns=["prompt", "response"])
                df = table.to_pandas()
                for _, row in df.iterrows():
                    prompt = str(row.get("prompt", ""))
                    response = str(row.get("response", ""))
                    if prompt and response:
                        yield {"prompt": prompt, "response": response}
            else:
                # Best-effort text line parsing if format unknown
                with open(local_path) as fp:
                    for line in fp:
                        line = line.strip()
                        if not line:
                            continue
                        try:
                            obj = json.loads(line)
                            prompt = str(obj.get("prompt", ""))
                            response = str(obj.get("response", ""))
                            if prompt and response:
                                yield {"prompt": prompt, "response": response}
                        except Exception:
                            continue
        except Exception as e:
            # Log and continue to avoid crashing training on bad shard
            print(f"Skipping {rel_path} due to parse error: {e}")
            continue

# train.py  (minimal integration)
from torch.utils.data import Dataset, DataLoader
from cdn_loader import create_manifest, load_dataset_cdn

class SurrogateDataset(Dataset):
    def __init__(self, samples):
        self.samples = list(samples)

    def __len__(self):
        return len(self.samples)

    def __getitem__(self, idx):
        return self.samples[idx]


def build_dataloader(repo_id, date_folder, batch_size=8, num_workers=2):
    manifest_path = f"manifests/{date_folder}.json"
    create_manifest(repo_id, date_folder, manifest_path)
    samples = list(load_dataset_cdn(repo_id, manifest_path, cache_dir=".cdn_cache"))
    dataset = SurrogateDataset(samples)
    return DataLoader(dataset, batch_size=batch_size, num_workers=num_workers, shuffle=True)
