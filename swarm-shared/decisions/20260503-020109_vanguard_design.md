# vanguard / design

# 1. Diagnosis

- No persisted `(repo, dateFolder) → file-list` manifest: every training/data-selection run triggers authenticated `list_repo_tree` against HF API, burning quota and risking 429s.
- Training/data loader likely uses `load_dataset(streaming=True)` or repeated per-file API calls on heterogeneous repos, causing `pyarrow.CastError` from mixed schemas.
- No CDN-only ingestion path: authenticated API calls are used for file fetches during training instead of public CDN URLs, wasting rate-limit budget.
- Schema pollution: raw files written to `enriched/` with extra metadata columns (`source`, `ts`) instead of strict `{prompt, response}` projection, increasing casting risk and storage bloat.
- No Lightning Studio reuse guard: training jobs may recreate studios instead of reusing running ones, burning 80h/mo quota and risking idle-stop training death.

# 2. Proposed change

Create `/opt/axentx/vanguard/pipeline/manifest.py` + update `/opt/axentx/vanguard/train/train.py` (or create if absent) to:

- Add `build_manifest(repo, date_folder, out_json)` that calls `list_repo_tree(path=date_folder, recursive=False)` once, saves `{"repo": repo, "date": date_folder, "files": [...], "cdn_root": "https://huggingface.co/datasets/.../resolve/main/..."}` to `manifests/`.
- Make training script accept `--manifest` and use CDN URLs with `requests`/`aiohttp` + `datasets.Dataset.from_generator` or `IterableDataset` to stream bytes without HF API auth.
- Project each file to `{prompt, response}` at parse time; drop other fields.
- Add studio reuse guard: list running studios and reuse if name matches before creating.

```python
#!/usr/bin/env python3
"""
Build and persist repo+date -> file-list manifest for CDN-only training.
"""
import json
import os
import sys
from datetime import datetime
from pathlib import Path
from typing import Dict, List

try:
    from huggingface_hub import HfApi
except ImportError:
    print("Install: pip install huggingface_hub")
    sys.exit(1)

API = HfApi()
MANIFEST_DIR = Path(__file__).parents[2] / "manifests"
MANIFEST_DIR.mkdir(exist_ok=True, parents=True)

def build_manifest(repo: str, date_folder: str, out_path: str | None = None) -> Dict:
    """
    repo: e.g. 'datasets/myorg/surrogate-1'
    date_folder: e.g. 'batches/mirror-merged/2026-05-03'
    """
    root = f"{repo}/tree/main/{date_folder}"
    items = API.list_repo_tree(repo=repo, path=date_folder, recursive=False)
    files = [
        {
            "path": f"{date_folder}/{it.path.split('/')[-1]}",
            "cdn_url": f"https://huggingface.co/datasets/{repo}/resolve/main/{date_folder}/{it.path.split('/')[-1]}",
            "size": getattr(it, "size", None),
        }
        for it in items
        if not it.is_directory
    ]

    manifest = {
        "repo": repo,
        "date": date_folder,
        "generated_utc": datetime.utcnow().isoformat() + "Z",
        "files": files,
    }

    out = Path(out_path or MANIFEST_DIR / f"{repo.replace('/', '_')}_{date_folder.replace('/', '_')}.json")
    out.write_text(json.dumps(manifest, indent=2))
    print(f"Manifest written: {out}")
    return manifest

if __name__ == "__main__":
    if len(sys.argv) < 3:
        print("Usage: manifest.py <repo> <date_folder> [out.json]")
        sys.exit(1)
    build_manifest(sys.argv[1], sys.argv[2], sys.argv[3] if len(sys.argv) > 3 else None)
```

def parse_file_cdn(url: str) -> Iterator[Dict[str, str]]:
    """
    Download single file via CDN and project to {prompt, response}.
    Supports newline-delimited JSON or simple prompt/response heuristics.
    """
    resp = requests.get(url, timeout=30)
    resp.raise_for_status()
    text = resp.text

    for line in text.strip().splitlines():
        line = line.strip()
        if not line:
            continue
        try:
            obj = json.loads(line)
            # Allow flexible keys; prefer prompt/response
            prompt = obj.get("prompt") or obj.get("input") or obj.get("question") or ""
            response = obj.get("response") or obj.get("output") or obj.get("answer") or ""
            if prompt and response:
                yield {"prompt": str(prompt), "response": str(response)}
        except json.JSONDecodeError:
            # fallback: split by separator if needed (customize per dataset)
            continue

def build_dataset_from_manifest(manifest_path: str) -> Dataset:
    manifest = json.loads(Path(manifest_path).read_text())
    gen = (
        item
        for f in manifest["files"]
        for item in parse_file_cdn(f["cdn_url"])
    )
    return Dataset.from_generator(lambda: gen, features={"prompt": str, "response": str})
