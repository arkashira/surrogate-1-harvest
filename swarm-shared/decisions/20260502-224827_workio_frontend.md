# workio / frontend

## Final consolidated solution

**Diagnosis (resolved)**  
- The system currently relies on fragile `localStorage` for offline clock-in/out, lacks idempotency, has no durable retry/backoff, and is vulnerable to cross-tab conflicts and storage pressure.  
- All candidates agree on root causes; differences are in scope granularity and implementation detail. We adopt the strongest, production-ready choices: IndexedDB durability, client-generated idempotency keys, explicit sync state, exponential backoff, and cross-tab coordination via a single worker-like queue process.

**Proposed change (minimal, actionable scope)**  
- Add `src/lib/queue/attendanceQueue.ts` (IndexedDB queue), `src/hooks/useAttendanceQueue.ts` (processing + lifecycle), integrate into `src/components/AttendanceButton.tsx` and `src/pages/Dashboard.tsx`.  
- Keep API tiny: `enqueueAttendance`, `processQueue`, `getPendingCount`, plus status helpers.

### Queue store (`src/lib/queue/attendanceQueue.ts`)
```ts
import { openDB, DBSchema, IDBPDatabase } from 'idb';

interface AttendanceQueueDB extends DBSchema {
  records: {
    key: string; // idempotencyKey
    value: {
      idempotencyKey: string;
      type: 'CLOCK_IN' | 'CLOCK_OUT';
      timestamp: number; // client timestamp (ms)
      latitude?: number;
      longitude?: number;
      status: 'pending' | 'syncing' | 'confirmed' | 'failed';
      attempts: number;
      lastError?: string;
      createdAt: number;
    };
  };
}

const DB_NAME = 'workio-attendance-queue';
const STORE_NAME = 'records';

async function getDB(): Promise<IDBPDatabase<AttendanceQueueDB>> {
  return openDB<AttendanceQueueDB>(DB_NAME, 1, {
    upgrade(db) {
      db.createObjectStore(STORE_NAME);
    },
  });
}

export async function enqueueAttendance(params: {
  type: 'CLOCK_IN' | 'CLOCK_OUT';
  timestamp: number;
  latitude?: number;
  longitude?: number;
}): Promise<string> {
  const idempotencyKey = crypto.randomUUID();
  const db = await getDB();
  await db.put(
    STORE_NAME,
    {
      idempotencyKey,
      type: params.type,
      timestamp: params.timestamp,
      latitude: params.latitude,
      longitude: params.longitude,
      status: 'pending',
      attempts: 0,
      createdAt: Date.now(),
    },
    idempotencyKey
  );
  return idempotencyKey;
}

export async function peekPending(
  limit = 10
): Promise<{ key: string; value: AttendanceQueueDB['records']['value'] }[]> {
  const db = await getDB();
  const tx = db.transaction(STORE_NAME, 'readonly');
  const allKeys = (await tx.store.getAllKeys()) as string[];
  const allValues = (await tx.store.getAll()) as AttendanceQueueDB['records']['value'][];

  const combined = allKeys.map((key, idx) => ({
    key: key as string,
    value: allValues[idx],
  }));

  return combined
    .filter((item) => item.value && item.value.status === 'pending')
    .sort((a, b) => (a.value?.createdAt ?? 0) - (b.value?.createdAt ?? 0))
    .slice(0, limit);
}

export async function updateRecordStatus(
  key: string,
  status: 'syncing' | 'confirmed' | 'failed',
  opts?: { attempts?: number; lastError?: string }
) {
  const db = await getDB();
  const val = await db.get(STORE_NAME, key);
  if (!val) return;
  await db.put(
    STORE_NAME,
    {
      ...val,
      status,
      attempts: opts?.attempts ?? val.attempts,
      lastError: opts?.lastError ?? val.lastError,
    },
    key
  );
}

export async function deleteRecord(key: string) {
  const db = await getDB();
  await db.delete(STORE_NAME, key);
}

export async function getPendingCount(): Promise<number> {
  const db = await getDB();
  const tx = db.transaction(STORE_NAME, 'readonly');
  const all = (await tx.store.getAll()) as AttendanceQueueDB['records']['value'][];
  return all.filter((v) => v.status === 'pending').length;
}

export async function getRecord(key: string): Promise<AttendanceQueueDB['records']['value'] | undefined> {
  const db = await getDB();
  return db.get(STORE_NAME, key);
}
```

### Processing hook (`src/hooks/useAttendanceQueue.ts`)
```ts
import { useEffect, useRef } from 'react';
import { peekPending, updateRecordStatus, deleteRecord } from '../lib/queue/attendanceQueue';

const API_BASE = import.meta.env.VITE_API_BASE || '/api';
const MAX_ATTEMPTS = 5;
const BASE_DELAY = 1000; // 1s

async function tryUpload(record: { key: string; value: any }): Promise<boolean> {
  const { key, value } = record;
  await updateRecordStatus(key, 'syncing');
  try {
    const res = await fetch(`${API_BASE}/attendance`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        idempotencyKey: value.idempotencyKey,
        type: value.type,
        timestamp: value.timestamp,
        latitude: value.latitude,
        longitude: value.longitude,
      }),
    });
    if (!res.ok) throw new Error(`HTTP ${res.status}`);
    await deleteRecord(key);
    return true;
  } catch (err: any) {
    const attempts = value.attempts + 1;
    if (attempts >= MAX_ATTEMPTS) {
      await updateRecordStatus(key, 'failed', { attempts, lastError: err.message });
    } else {
      await updateRecordStatus(key, 'pending', { attempts, lastError: err.message });
    }
    return false;
  }
}

function delay(ms: number) {
  return new Promise((r) => setTimeout(r, ms));
}

export function useAttendanceQueue() {
  const processingRef = useRef(false);

  const processQueue = async () => {
    if (processingRef.current) return;
    processingRef.current = true;

    try {
      while (true) {
        const pending = await peekPending(1);
        if (pending.length === 0) break;
        const record = pending[0];
        const attempts = record.value.attempts;
        if (attempts > 0) {
          const backoff = BASE_DELAY * Math.pow(2, Math.min(attempts, 6));
          await delay(backoff);
        }
        await tryUpload(record);
      }
    } finally {
      processingRef.current = false;
    }
  };

  // Run on mount + when online
  useEffect(() => {
    processQueue();
    const onOnline = () => processQueue();
    window.addEventListener('online', onOnline);
    return () => window.removeEventListener('online', onOnline);
  }, []);

  // Periodic retry while page open (every 30s)
  useEffect(() => {
    const id = setInterval(processQueue, 30_000);
    return () => clearInterval(id);
  }, []);

  return { processQueue };
}
```
