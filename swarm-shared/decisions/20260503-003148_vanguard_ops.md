# vanguard / ops

## 1. Diagnosis
- No persisted `(repo, dateFolder)` manifest exists → every training run re-enumerates via authenticated HF API → quota burn + 429 risk.
- Data loader likely uses recursive `list_repo_files` or `load_dataset` during training → amplifies rate-limit exposure and slows epoch start.
- Lightning Studio reuse is not enforced → idle-stop kills training and quota is wasted on repeated studio creation.
- No CDN-only fetch strategy in training scripts → authenticated API calls occur during data loading instead of using public CDN URLs.
- Missing wrapper script hygiene (shebang, executable bit, `SHELL=/bin/bash`) for cron-launched ops scripts → sporadic exec failures.

## 2. Proposed change
Create `/opt/axentx/vanguard/ops/persist_manifest.py` and update `/opt/axentx/vanguard/ops/launch_training.py` (or create if absent) so that:
- A single authenticated HF API call produces `manifests/{repo}/{dateFolder}.json` listing CDN URLs for every parquet file.
- Training script reads the manifest and uses only CDN URLs (`https://huggingface.co/datasets/.../resolve/main/...`) with zero HF API calls during data loading.
- Reuse a running Lightning Studio if present; otherwise start one with L40S (free-tier fallback) and respect idle-stop by checking status before `.run()`.

### persist_manifest.py
```python
#!/usr/bin/env python3
"""
Generate and persist a CDN-only manifest for (repo, dateFolder).
Usage:
  python persist_manifest.py <repo> <dateFolder> [--out-dir=manifests]
"""
import json
import os
import sys
from pathlib import Path

try:
    from huggingface_hub import list_repo_tree
except ImportError:
    print("Install: pip install huggingface_hub")
    sys.exit(1)

CDN_BASE = "https://huggingface.co/datasets"

def build_manifest(repo: str, date_folder: str, out_dir: str = "manifests") -> str:
    out_path = Path(out_dir) / repo.replace("/", "_") / f"{date_folder}.json"
    out_path.parent.mkdir(parents=True, exist_ok=True)

    # Single authenticated API call (non-recursive per folder to reduce pagination)
    items = list_repo_tree(repo=repo, path=date_folder, recursive=False)

    entries = []
    for item in items:
        if not item.path.lower().endswith(".parquet"):
            continue
        cdn_url = f"{CDN_BASE}/{repo}/resolve/main/{item.path}"
        entries.append({
            "path": item.path,
            "cdn_url": cdn_url,
            "size": getattr(item, "size", None)
        })

    manifest = {
        "repo": repo,
        "date_folder": date_folder,
        "generated_by": "persist_manifest.py",
        "entries": entries
    }

    out_path.write_text(json.dumps(manifest, indent=2))
    print(f"Manifest written: {out_path} ({len(entries)} parquet files)")
    return str(out_path)

if __name__ == "__main__":
    if len(sys.argv) < 3:
        print("Usage: persist_manifest.py <repo> <dateFolder> [--out-dir=manifests]")
        sys.exit(1)

    repo_arg = sys.argv[1]
    date_arg = sys.argv[2]
    out_dir_arg = "manifests"
    for a in sys.argv[3:]:
        if a.startswith("--out-dir="):
            out_dir_arg = a.split("=", 1)[1]

    build_manifest(repo_arg, date_arg, out_dir=out_dir_arg)
```

### launch_training.py (or update existing)
```python
#!/usr/bin/env python3
"""
Launch Lightning training with CDN-only data loading and studio reuse.
Usage:
  python launch_training.py <repo> <dateFolder> [--script=train.py]
"""
import json
import subprocess
import sys
from pathlib import Path

try:
    from lightning import Lightning, Studio, L40S, Teamspace
except ImportError:
    print("Install: pip install lightning")
    sys.exit(1)

MANIFEST_DIR = Path(__file__).parent / "manifests"

def find_running_studio(name: str):
    for s in Teamspace.studios:
        if s.name == name and s.status == "Running":
            return s
    return None

def ensure_manifest(repo: str, date_folder: str) -> Path:
    manifest_path = MANIFEST_DIR / repo.replace("/", "_") / f"{date_folder}.json"
    if not manifest_path.exists():
        # Run manifest generation once (uses HF API)
        subprocess.run([
            sys.executable,
            str(Path(__file__).parent / "persist_manifest.py"),
            repo,
            date_folder,
            "--out-dir=" + str(MANIFEST_DIR)
        ], check=True)
    return manifest_path

def launch(repo: str, date_folder: str, train_script: str = "train.py"):
    manifest_path = ensure_manifest(repo, date_folder)
    manifest = json.loads(manifest_path.read_text())

    studio_name = f"vanguard-{repo.replace('/', '-')}-{date_folder}"
    studio = find_running_studio(studio_name)

    if studio is None:
        print(f"Starting new Lightning Studio: {studio_name}")
        # Prefer free-tier compatible machine; fallback to L40S if available
        studio = Studio(
            name=studio_name,
            machine=L40S,
            create_ok=True
        )
    else:
        print(f"Reusing running studio: {studio_name}")

    # Prepare train command that uses manifest (CDN-only)
    # Example: python train.py --manifest manifests/... --epochs 1
    train_cmd = [
        sys.executable, train_script,
        "--manifest", str(manifest_path),
        "--epochs", "1"
    ]

    # If studio was stopped by idle timeout, restart it
    if studio.status != "Running":
        print("Studio not running; restarting...")
        studio.start(machine=L40S)

    # Run training in the studio
    run = studio.run(
        command=" ".join(train_cmd),
        environment="lightning_base",
        cluster_size=1
    )
    print(f"Run submitted: {run}")
    return run

if __name__ == "__main__":
    if len(sys.argv) < 3:
        print("Usage: launch_training.py <repo> <dateFolder> [--script=train.py]")
        sys.exit(1)

    repo_arg = sys.argv[1]
    date_arg = sys.argv[2]
    script_arg = "train.py"
    for a in sys.argv[3:]:
        if a.startswith("--script="):
            script_arg = a.split("=", 1)[1]

    launch(repo_arg, date_arg, train_script=script_arg)
```
