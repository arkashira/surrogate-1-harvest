# vanguard / ops

## 1. Diagnosis
- Runtime `load_dataset`/recursive file enumeration triggers HF API 429s and non-reproducible epochs.
- No content-addressed manifest (file list + SHA256) → CDN-only fetching impossible; training still hits API during data loading.
- Mixed-schema parquet ingestion writes extra columns (`source`, `ts`) that break downstream surrogate-1 schema expectations.
- No deterministic repo-to-sibling mapping for HF commit cap (128/hr/repo) → ingestion bursts hit rate limits.
- No studio reuse guard in Lightning launcher → quota wasted on repeated studio creation and idle-stop training deaths.

## 2. Proposed change
Create `/opt/axentx/vanguard/ops/manifest.py` + update `/opt/axentx/vanguard/train.py` (or create if absent) to:
- Add `build_manifest(repo, date_folder)` that calls `list_repo_tree(recursive=False)` once, saves `manifest-{date}.json` with `{path, sha256, size}`.
- Add `pick_sibling(slug)` that hashes slug → sibling repo index (0…4) for HF commit cap spreading.
- Add `get_running_studio(name)` that reuses running studios before creating new ones.
- Update data loader to use CDN-only URLs from manifest (zero API calls during training).

# /opt/axentx/vanguard/ops/manifest.py
#!/usr/bin/env python3
"""
Content-addressed manifest + HF helpers for vanguard training pipeline.
- Builds manifest via single list_repo_tree call (non-recursive, per folder).
- CDN-only file fetches during training (bypasses API rate limits).
- Deterministic sibling repo selection to spread HF commit cap.
- Studio reuse to save Lightning quota.
"""
import hashlib
import json
import os
from pathlib import Path
from typing import Dict, List, Optional

from huggingface_hub import HfApi, list_repo_tree, hf_hub_download
from lightning import Studio, Teamspace

HF_CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def sha256_file(path: str) -> str:
    h = hashlib.sha256()
    with open(path, "rb") as f:
        for chunk in iter(lambda: f.read(8192), b""):
            h.update(chunk)
    return h.hexdigest()

def build_manifest(repo: str, folder: str, out_path: str) -> List[Dict]:
    """
    Single API call: list_repo_tree non-recursive for folder.
    Returns list[dict] with {path, sha256, size} and writes JSON manifest.
    """
    api = HfApi()
    tree = list_repo_tree(repo=repo, path=folder, recursive=False)
    entries = []
    for node in tree:
        if node.type != "file":
            continue
        entry = {
            "path": node.path,
            "size": node.size or 0,
            # sha256 optional on HF tree; fill later if local file exists or skip
            "sha256": getattr(node, "sha256", None),
            "cdn_url": HF_CDN_TEMPLATE.format(repo=repo, path=node.path.replace("/", "%2F")),
        }
        entries.append(entry)

    manifest = {
        "repo": repo,
        "folder": folder,
        "entries": entries,
    }
    os.makedirs(os.path.dirname(os.path.abspath(out_path)), exist_ok=True)
    with open(out_path, "w") as f:
        json.dump(manifest, f, indent=2)
    return entries

def pick_sibling(slug: str, n_siblings: int = 5) -> int:
    """Deterministic sibling index 0..n_siblings-1 to spread HF commit cap."""
    digest = hashlib.sha256(slug.encode()).digest()
    return int.from_bytes(digest, "big") % n_siblings

def sibling_repo_name(base_repo: str, idx: int) -> str:
    """Convert base repo 'org/dataset' -> 'org/dataset-sibling-idx'."""
    if idx == 0:
        return base_repo
    org, name = base_repo.rsplit("/", 1)
    return f"{org}/{name}-sibling-{idx}"

def get_running_studio(name: str, machine: Optional[str] = None):
    """Reuse running studio; avoids recreating and burning quota."""
    teamspace = Teamspace()
    for s in teamspace.studios:
        if s.name == name and s.status == "Running":
            return s
    # fallback: create if not running (caller should pass machine)
    return Studio(name=name, create_ok=True, machine=machine)

def cdn_only_dataloader(manifest_path: str):
    """Yield local file paths or stream from CDN URLs without HF API auth calls."""
    with open(manifest_path) as f:
        manifest = json.load(f)
    for entry in manifest["entries"]:
        yield entry["cdn_url"]
