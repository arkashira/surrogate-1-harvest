# workio / design

## 1. Diagnosis
- **No idempotent LINE webhook handling**: retries on 5xx/timeouts replay clock/leave/OT events and corrupt state (duplicate punches, double approvals).
- **Missing `X-Line-Signature` verification**: webhook requests can be spoofed and processed without cryptographic proof.
- **No real-time state propagation**: frontend relies on local React state/polling; LINE webhook updates are invisible until hard refresh → silent data drift and perceived latency across tabs/devices.
- **No optimistic UI + rollback**: clock/leave/OT actions block on network with no instant feedback; failures leave UI stuck.
- **Backend/frontend drift risk**: schema and event semantics are implicit; no shared contract between server webhook and client state machine.

## 2. Proposed change
- **Scope**: `workio/server/src/routes/webhook/line.ts` (create if absent) and `workio/src/lib/realtime.ts` + `workio/src/features/clock/ClockButton.tsx`
- **Goal**: add idempotent, signed LINE webhook handler + Server-Sent Events (SSE) broadcast + optimistic UI with rollback in <2h.

## 3. Implementation

### 3.1 Backend: idempotent, signed LINE webhook + SSE broadcast
File: `workio/server/src/routes/webhook/line.ts`

```ts
import { Router, Request, Response } from 'express';
import crypto from 'crypto';
import { db } from '../../db';
import { publishEvent } from '../../lib/sse';

const router = Router();
const LINE_CHANNEL_SECRET = process.env.LINE_CHANNEL_SECRET!;

function verifySignature(body: string, signature: string): boolean {
  const expected = crypto
    .createHmac('sha256', LINE_CHANNEL_SECRET)
    .update(body)
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected));
}

function idempotencyKey(req: Request): string {
  // LINE deliveries include a unique id in headers when retry occurs; use it.
  return req.get('X-Line-Retry-Id') || req.get('X-Line-Delivery') || req.get('Idempotency-Key') || req.body.events?.[0]?.mode + req.body.events?.[0]?.timestamp || String(Date.now());
}

router.post('/line', async (req: Request, res: Response) => {
  const signature = req.get('X-Line-Signature') || '';
  const rawBody = JSON.stringify(req.body);
  if (!verifySignature(rawBody, signature)) {
    return res.status(401).send('Invalid signature');
  }

  const key = idempotencyKey(req);
  // Fast in-memory guard + persistent guard (best-effort for this scope)
  if ((global as any)._lineDedupe?.has(key)) {
    return res.status(200).send('OK');
  }
  if (!((global as any)._lineDedupe)) (global as any)._lineDedupe = new Set();
  (global as any)._lineDedupe.add(key);
  setTimeout(() => (global as any)._lineDedupe?.delete(key), 60_000);

  try {
    const events = req.body.events || [];
    for (const ev of events) {
      if (ev.type === 'message' && ev.message.type === 'text') {
        // Simplified: parse clock in/out commands or rely on rich menu postback
        // For production, prefer postback events with explicit payloads
        const tenantId = ev.source.userId; // map via LINE userId->tenant/user in real impl
        const userId = ev.source.userId;
        const action = ev.message.text.toLowerCase().includes('ออก') ? 'clock_out' : 'clock_in';
        const ts = new Date(ev.timestamp);

        // Idempotent upsert by externalId (use LINE delivery id or event.id)
        const externalId = ev.mode + ':' + ev.timestamp + ':' + ev.source.userId + ':' + ev.type;
        const existing = await db.clockPunch.findFirst({ where: { externalId } });
        if (existing) continue;

        const punch = await db.clockPunch.create({
          data: {
            userId,
            tenantId,
            action,
            ts,
            externalId,
            // location: ev.location (if available)
          },
        });

        // Broadcast to tenant & user channels via SSE
        publishEvent(`tenant:${tenantId}`, { type: 'clock_update', payload: punch });
        publishEvent(`user:${userId}`, { type: 'clock_update', payload: punch });
      }

      if (ev.type === 'postback') {
        // structured payload preferred: data=action=clock_in&...
        const data = Object.fromEntries(new URLSearchParams(ev.postback.data).entries());
        const tenantId = ev.source.userId;
        const userId = ev.source.userId;
        const externalId = ev.deliveryId || ev.postback.data || ev.timestamp;

        const existing = await db.clockPunch.findFirst({ where: { externalId } });
        if (existing) continue;

        const punch = await db.clockPunch.create({
          data: {
            userId,
            tenantId,
            action: data.action || 'clock_in',
            ts: new Date(ev.timestamp),
            externalId,
          },
        });

        publishEvent(`tenant:${tenantId}`, { type: 'clock_update', payload: punch });
        publishEvent(`user:${userId}`, { type: 'clock_update', payload: punch });
      }
    }

    res.status(200).send('OK');
  } catch (err) {
    console.error('LINE webhook error', err);
    res.status(500).send('Error');
  }
});

export default router;
```

```ts
import type { Server } from 'http';
import { EventEmitter } from 'events';

const emitter = new EventEmitter();

export function publishEvent(channel: string, data: any) {
  emitter.emit(channel, JSON.stringify(data));
}

export function attachSSE(server: Server) {
  // Mount a lightweight SSE endpoint; clients connect to /sse?channel=user:xxx
  // Minimal implementation for immediate value
  const express = require('express');
  const app = express();

  app.get('/sse', (req: any, res: any) => {
    const channel = req.query.channel;
    if (!channel) return res.status(400).send('channel required');

    res.writeHead(200, {
      'Content-Type': 'text/event-stream',
      'Cache-Control': 'no-cache',
      Connection: 'keep-alive',
    });
    res.write('\n');

    const onEvent = (msg: string) => {
      res.write(`data: ${msg}\n\n`);
    };
    emitter.on(String(channel), onEvent);

    req.on('close', () => {
      emitter.off(String(channel), onEvent);
      res.end();
    });
  });

  // Mount onto existing server (simplified)
  // In your main server file: attachSSE(server) after app creation
  return app;
}
```
